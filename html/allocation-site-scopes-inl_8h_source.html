15void AllocationSiteContext::InitializeTraversal(Handle<AllocationSite> site) {
16 top_ = site;
17 // {current_} is updated in place to not create unnecessary Handles, hence
18 // we initially need a separate handle.
19 current_ = Handle<AllocationSite>::New(*top_, isolate());
20}

22Handle<AllocationSite> AllocationSiteUsageContext::EnterNewScope() {
23 if (top().is_null()) {
24 InitializeTraversal(top_site_);
25 } else {
26 // Advance current site
27 Tagged<Object> nested_site = current()->nested_site();
28 // Something is wrong if we advance to the end of the list here.
29 update_current_site(Cast<AllocationSite>(nested_site));
30 }
31 return Handle<AllocationSite>(*current(), isolate());
32}

34void AllocationSiteUsageContext::ExitScope(
35 DirectHandle<AllocationSite> scope_site, DirectHandle<JSObject> object) {
36 // This assert ensures that we are pointing at the right sub-object in a
37 // recursive walk of a nested literal.
38 DCHECK(object.is_null() || *object == scope_site->boilerplate());
39}

41bool AllocationSiteUsageContext::ShouldCreateMemento(
42 DirectHandle<JSObject> object) {
43 if (activated_ && AllocationSite::CanTrack(object->map()->instance_type())) {
44 if (v8_flags.allocation_site_pretenuring ||
45 AllocationSite::ShouldTrack(object->GetElementsKind())) {
46 if (v8_flags.trace_creation_allocation_sites) {
47 PrintF("*** Creating Memento for %s %p\n",
48 IsJSArray(*object) ? "JSArray" : "JSObject",
49 reinterpret_cast<void*>(object->ptr()));
50 }
51 return true;
52 }
53 }
54 return false;
55}
