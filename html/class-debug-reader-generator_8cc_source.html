27// An iterator for use in ValueTypeFieldsRange.
28class ValueTypeFieldIterator {
29 public:
30 ValueTypeFieldIterator(const Type* type, size_t index)
31 : type_(type), index_(index) {}
32 struct Result {
33 NameAndType name_and_type;
34 SourcePosition pos;
35 size_t offset_bytes;
36 int num_bits;
37 int shift_bits;
38 };
39 const Result operator*() const {
40 if (auto struct_type = type_->StructSupertype()) {
41 const auto& field = (*struct_type)->fields()[index_];
42 return {field.name_and_type, field.pos, *field.offset, 0, 0};
43 }
44 const Type* type = type_;
45 int bitfield_start_offset = 0;
46 if (const auto type_wrapped_in_smi =
47 Type::MatchUnaryGeneric(type_, TypeOracle::GetSmiTaggedGeneric())) {
48 type = *type_wrapped_in_smi;
49 bitfield_start_offset = TargetArchitecture::SmiTagAndShiftSize();
50 }
51 if (const BitFieldStructType* bit_field_struct_type =
52 BitFieldStructType::DynamicCast(type)) {
53 const auto& field = bit_field_struct_type->fields()[index_];
54 return {field.name_and_type, field.pos, 0, field.num_bits,
55 field.offset + bitfield_start_offset};
56 }
57 UNREACHABLE();
58 }
59 ValueTypeFieldIterator& operator++() {
60 ++index_;
61 return *this;
62 }
63 bool operator==(const ValueTypeFieldIterator& other) const {
64 return type_ == other.type_ && index_ == other.index_;
65 }
66 bool operator!=(const ValueTypeFieldIterator& other) const {
67 return !(*this == other);
68 }
69
70 private:
71 const Type* type_;
72 size_t index_;
73};

75// A way to iterate over the fields of structs or bitfield structs. For other
76// types, the iterators returned from begin() and end() are immediately equal.
77class ValueTypeFieldsRange {
78 public:
79 explicit ValueTypeFieldsRange(const Type* type) : type_(type) {}
80 ValueTypeFieldIterator begin() { return {type_, 0}; }
81 ValueTypeFieldIterator end() {
82 size_t index = 0;
83 std::optional<const StructType*> struct_type = type_->StructSupertype();
84 if (struct_type && *struct_type != TypeOracle::GetFloat64OrHoleType()) {
85 index = (*struct_type)->fields().size();
86 }
87 const Type* type = type_;
88 if (const auto type_wrapped_in_smi =
89 Type::MatchUnaryGeneric(type_, TypeOracle::GetSmiTaggedGeneric())) {
90 type = *type_wrapped_in_smi;
91 }
92 if (const BitFieldStructType* bit_field_struct_type =
93 BitFieldStructType::DynamicCast(type)) {
94 index = bit_field_struct_type->fields().size();
95 }
96 return {type_, index};
97 }
98
99 private:
100 const Type* type_;
101};

103// A convenient way to keep track of several different ways that we might need
104// to represent a field's type in the generated C++.
105class DebugFieldType {
106 public:
107 explicit DebugFieldType(const Field& field)
108 : name_and_type_(field.name_and_type), pos_(field.pos) {}
109 DebugFieldType(const NameAndType& name_and_type, const SourcePosition& pos)
110 : name_and_type_(name_and_type), pos_(pos) {}
111
112 bool IsTagged() const {
113 return name_and_type_.type->IsSubtypeOf(TypeOracle::GetTaggedType());
114 }

116 // Returns the type that should be used for this field's value within code
117 // that is compiled as part of the debug helper library. In particular, this
118 // simplifies any tagged type to a plain uintptr_t because the debug helper
119 // compiles without most of the V8 runtime code.
120 std::string GetValueType(TypeStorage storage) const {
121 if (IsTagged()) {
122 return storage == kAsStoredInHeap ? "i::Tagged_t" : "uintptr_t";
123 }
124
125 // We can't emit a useful error at this point if the constexpr type name is
126 // wrong, but we can include a comment that might be helpful.
127 return GetOriginalType(storage) +
128 " /*Failing? Ensure constexpr type name is correct, and the "
129 "necessary #include is in any .tq file*/";
130 }
131
132 // Returns the type that should be used to represent a field's type to
133 // debugging tools that have full V8 symbols. The types returned from this
134 // method are resolveable in the v8::internal namespace and may refer to
135 // object types that are not included in the compilation of the debug helper
136 // library.
137 std::string GetOriginalType(TypeStorage storage) const {
138 if (name_and_type_.type->StructSupertype()) {
139 // There's no meaningful type we could use here, because the V8 symbols
140 // don't have any definition of a C++ struct matching this struct type.
141 return "";
142 }
143 if (IsTagged()) {
144 std::optional<const ClassType*> field_class_type =
145 name_and_type_.type->ClassSupertype();
146 std::string result =
147 "v8::internal::" +
148 (field_class_type.has_value()
149 ? (*field_class_type)->GetGeneratedTNodeTypeName()
150 : "Object");
151 if (storage == kAsStoredInHeap) {
152 result = "v8::internal::TaggedMember<" + result + ">";
153 }
154 return result;
155 }
156 return name_and_type_.type->GetConstexprGeneratedTypeName();
157 }
158
159 // Returns a C++ expression that evaluates to a string (type `const char*`)
160 // containing the name of the field's type. The types returned from this
161 // method are resolveable in the v8::internal namespace and may refer to
162 // object types that are not included in the compilation of the debug helper
163 // library.
164 std::string GetTypeString(TypeStorage storage) const {
165 if (IsTagged() || name_and_type_.type->IsStructType()) {
166 // Wrap up the original type in a string literal.
167 return "\"" + GetOriginalType(storage) + "\"";
168 }
169
170 // We require constexpr type names to be resolvable in the v8::internal
171 // namespace, according to the contract in debug-helper.h. In order to
172 // verify at compile time that constexpr type names are resolvable, we use
173 // the type name as a dummy template parameter to a function that just
174 // returns its parameter.
175 return "CheckTypeName<" + GetValueType(storage) + ">(\"" +
176 GetOriginalType(storage) + "\")";
177 }
178
179 // Returns the field's size in bytes.
180 size_t GetSize() const {
181 auto opt_size = SizeOf(name_and_type_.type);
182 if (!opt_size.has_value()) {
183 Error("Size required for type ", name_and_type_.type->ToString())
184 .Position(pos_);
185 return 0;
186 }
187 return std::get<0>(*opt_size);
188 }
189
190 // Returns the name of the function for getting this field's address.
191 std::string GetAddressGetter() {
192 return "Get" + CamelifyString(name_and_type_.name) + "Address";
193 }
194
195 private:
196 NameAndType name_and_type_;
197 SourcePosition pos_;
198};

203//

223// Emits a function to get the value of a field, or the value from an indexed
224// position within an array field, based on the member variable {address_},
225// which is a tagged pointer, and the parameter {accessor}, a function pointer
226// that allows for fetching memory from the debuggee. The returned result
227// includes both a "validity", indicating whether the memory could be fetched,
228// and the fetched value. If the field contains tagged data, then these
229// functions call EnsureDecompressed to expand compressed data. Example:
230//
231// Value<uintptr_t> TqMap::GetPrototypeValue(d::MemoryAccessor accessor) const {
232// i::Tagged_t value{};
233// d::MemoryAccessResult validity = accessor(
234// GetPrototypeAddress(),
235// reinterpret_cast<uint8_t*>(&value),
236// sizeof(value));
237// return {validity, EnsureDecompressed(value, address_)};
238// }
239//
240// For array fields, an offset parameter is included. Example:
241//
242// Value<uintptr_t> TqFixedArray::GetObjectsValue(d::MemoryAccessor accessor,
243// size_t offset) const {
244// i::Tagged_t value{};
245// d::MemoryAccessResult validity = accessor(
246// GetObjectsAddress() + offset * sizeof(value),
247// reinterpret_cast<uint8_t*>(&value),
248// sizeof(value));
249// return {validity, EnsureDecompressed(value, address_)};
250// }
251void GenerateFieldValueAccessor(const Field& field,
252 const std::string& class_name,
253 std::ostream& h_contents,
254 std::ostream& cc_contents) {
255 // Currently not implemented for struct fields.
256 if (field.name_and_type.type->StructSupertype()) return;
257
258 DebugFieldType debug_field_type(field);
259
260 const std::string address_getter = debug_field_type.GetAddressGetter();
261 const std::string field_getter =
262 "Get" + CamelifyString(field.name_and_type.name) + "Value";
263
264 std::string index_param;
265 std::string index_offset;
266 if (field.index) {
267 index_param = ", size_t offset";
268 index_offset = " + offset * sizeof(value)";
269 }
270
271 std::string field_value_type = debug_field_type.GetValueType(kUncompressed);
272 h_contents << " Value<" << field_value_type << "> " << field_getter
273 << "(d::MemoryAccessor accessor " << index_param << ") const;\n";
274 cc_contents << "\nValue<" << field_value_type << "> Tq" << class_name
275 << "::" << field_getter << "(d::MemoryAccessor accessor"
276 << index_param << ") const {\n";
277 cc_contents << " " << debug_field_type.GetValueType(kAsStoredInHeap)
278 << " value{};\n";
279 cc_contents << " d::MemoryAccessResult validity = accessor("
280 << address_getter << "()" << index_offset
281 << ", reinterpret_cast<uint8_t*>(&value), sizeof(value));\n";
282#ifdef V8_MAP_PACKING
283 if (field_getter == "GetMapValue") {
284 cc_contents << " value = i::MapWord::Unpack(value);\n";
285 }
286#endif
287 cc_contents << " return {validity, "
288 << (debug_field_type.IsTagged()
289 ? "EnsureDecompressed(value, address_)"
290 : "value")
291 << "};\n";
292 cc_contents << "}\n";
293}

295// Emits a portion of the member function GetProperties that is responsible for
296// adding data about the current field to a result vector called "result".
297// Example output:
298//
299// std::vector<std::unique_ptr<StructProperty>> prototype_struct_field_list;
300// result.push_back(std::make_unique<ObjectProperty>(
301// "prototype", // Field name
302// "v8::internal::HeapObject", // Field type
303// "v8::internal::HeapObject", // Decompressed type
304// GetPrototypeAddress(), // Field address
305// 1, // Number of values
306// 8, // Size of value
307// std::move(prototype_struct_field_list), // Struct fields
308// d::PropertyKind::kSingle)); // Field kind
309//
310// In builds with pointer compression enabled, the field type for tagged values
311// is "v8::internal::TaggedValue" (a four-byte class) and the decompressed type
312// is a normal Object subclass that describes the expanded eight-byte type.
313//
314// If the field is an array, then its length is fetched from the debuggee. This
315// could fail if the debuggee has incomplete memory, so the "validity" from that
316// fetch is used to determine the result PropertyKind, which will say whether
317// the array's length is known.
318//
319// If the field's type is a struct, then a local variable is created and filled
320// with descriptions of each of the struct's fields. The type and decompressed
321// type in the ObjectProperty are set to the empty string, to indicate to the
322// caller that the struct fields vector should be used instead.
323//
324// The following example is an array of structs, so it uses both of the optional
325// components described above:
326//
327// std::vector<std::unique_ptr<StructProperty>> descriptors_struct_field_list;
328// descriptors_struct_field_list.push_back(std::make_unique<StructProperty>(
329// "key", // Struct field name
330// "v8::internal::PrimitiveHeapObject", // Struct field type
331// "v8::internal::PrimitiveHeapObject", // Struct field decompressed type
332// 0, // Byte offset within struct data
333// 0, // Bitfield size (0=not a bitfield)
334// 0)); // Bitfield shift
335// // The line above is repeated for other struct fields. Omitted here.
336// // Fetch the slice.
337// auto indexed_field_slice_descriptors =
338// TqDebugFieldSliceDescriptorArrayDescriptors(accessor, address_);
339// if (indexed_field_slice_descriptors.validity == d::MemoryAccessResult::kOk) {
340// result.push_back(std::make_unique<ObjectProperty>(
341// "descriptors", // Field name
342// "", // Field type
343// "", // Decompressed type
344// address_ - i::kHeapObjectTag +
345// std::get<1>(indexed_field_slice_descriptors.value), // Field address
346// std::get<2>(indexed_field_slice_descriptors.value), // Number of values
347// 12, // Size of value
348// std::move(descriptors_struct_field_list), // Struct fields
349// GetArrayKind(indexed_field_slice_descriptors.validity))); // Field kind
350// }
351void GenerateGetPropsChunkForField(const Field& field,
352 std::ostream& get_props_impl,
353 std::string class_name) {
354 DebugFieldType debug_field_type(field);
355
356 // If the current field is a struct or bitfield struct, create a vector
357 // describing its fields. Otherwise this vector will be empty.
358 std::string struct_field_list =
359 field.name_and_type.name + "_struct_field_list";
360 get_props_impl << " std::vector<std::unique_ptr<StructProperty>> "
361 << struct_field_list << ";\n";
362 for (const auto& struct_field :
363 ValueTypeFieldsRange(field.name_and_type.type)) {
364 DebugFieldType struct_field_type(struct_field.name_and_type,
365 struct_field.pos);
366 get_props_impl << " " << struct_field_list
367 << ".push_back(std::make_unique<StructProperty>(\""
368 << struct_field.name_and_type.name << "\", "
369 << struct_field_type.GetTypeString(kAsStoredInHeap) << ", "
370 << struct_field.offset_bytes << ", " << struct_field.num_bits
371 << ", " << struct_field.shift_bits << "));\n";
372 }
373 struct_field_list = "std::move(" + struct_field_list + ")";
374
375 // The number of values and property kind for non-indexed properties:
376 std::string count_value = "1";
377 std::string property_kind = "d::PropertyKind::kSingle";
378
379 // If the field is indexed, emit a fetch of the array length, and change
380 // count_value and property_kind to be the correct values for an array.
381 if (field.index) {
382 std::string indexed_field_slice =
383 "indexed_field_slice_" + field.name_and_type.name;
384 get_props_impl << " auto " << indexed_field_slice << " = "
385 << "TqDebugFieldSlice" << class_name
386 << CamelifyString(field.name_and_type.name)
387 << "(accessor, address_);\n";
388 std::string validity = indexed_field_slice + ".validity";
389 std::string value = indexed_field_slice + ".value";
390 property_kind = "GetArrayKind(" + validity + ")";
391
392 get_props_impl << " if (" << validity
393 << " == d::MemoryAccessResult::kOk) {\n"
394 << " result.push_back(std::make_unique<ObjectProperty>(\""
395 << field.name_and_type.name << "\", "
396 << debug_field_type.GetTypeString(kAsStoredInHeap) << ", "
397 << "address_ - i::kHeapObjectTag + std::get<1>(" << value
398 << "), "
399 << "std::get<2>(" << value << ")"
400 << ", " << debug_field_type.GetSize() << ", "
401 << struct_field_list << ", " << property_kind << "));\n"
402 << " }\n";
403 return;
404 }
405 get_props_impl << " result.push_back(std::make_unique<ObjectProperty>(\""
406 << field.name_and_type.name << "\", "
407 << debug_field_type.GetTypeString(kAsStoredInHeap) << ", "
408 << debug_field_type.GetAddressGetter() << "(), " << count_value
409 << ", " << debug_field_type.GetSize() << ", "
410 << struct_field_list << ", " << property_kind << "));\n";
411}

413// For any Torque-defined class Foo, this function generates a class TqFoo which
414// allows for convenient inspection of objects of type Foo in a crash dump or
415// time travel session (where we can't just run the object printer). The
416// generated class looks something like this:
417//
418// class TqFoo : public TqParentOfFoo {
419// public:
420// // {address} is an uncompressed tagged pointer.
421// inline TqFoo(uintptr_t address) : TqParentOfFoo(address) {}
422//
423// // Creates and returns a list of this object's properties.
424// std::vector<std::unique_ptr<ObjectProperty>> GetProperties(
425// d::MemoryAccessor accessor) const override;
426//
427// // Returns the name of this class, "v8::internal::Foo".
428// const char* GetName() const override;
429//
430// // Visitor pattern; implementation just calls visitor->VisitFoo(this).
431// void Visit(TqObjectVisitor* visitor) const override;
432//
433// // Returns whether Foo is a superclass of the other object's type.
434// bool IsSuperclassOf(const TqObject* other) const override;
435//
436// // Field accessors omitted here (see other comments above).
437// };
438//
439// Four output streams are written:
440//
441// h_contents: A header file which gets the class definition above.
442// cc_contents: A cc file which gets implementations of that class's members.
443// visitor: A stream that is accumulating the definition of the class
444// TqObjectVisitor. Each class Foo gets its own virtual method
445// VisitFoo in TqObjectVisitor.
446void GenerateClassDebugReader(const ClassType& type, std::ostream& h_contents,
447 std::ostream& cc_contents, std::ostream& visitor,
448 std::unordered_set<const ClassType*>* done) {
449 // Make sure each class only gets generated once.
450 if (!done->insert(&type).second) return;
451 const ClassType* super_type = type.GetSuperClass();
452
453 // We must emit the classes in dependency order. If the super class hasn't
454 // been emitted yet, go handle it first.
455 if (super_type != nullptr) {
456 GenerateClassDebugReader(*super_type, h_contents, cc_contents, visitor,
457 done);
458 }
459
460 // Classes with undefined layout don't grant any particular value here and may
461 // not correspond with actual C++ classes, so skip them.
462 if (type.HasUndefinedLayout()) return;
463
464 const std::string name = type.name();
465 const std::string super_name =
466 super_type == nullptr ? "Object" : super_type->name();
467 h_contents << "\nclass Tq" << name << " : public Tq" << super_name << " {\n";
468 h_contents << " public:\n";
469 h_contents << " inline Tq" << name << "(uintptr_t address) : Tq"
470 << super_name << "(address) {}\n";
471 h_contents << kTqObjectOverrideDecls;
472
473 cc_contents << "\nconst char* Tq" << name << "::GetName() const {\n";
474 cc_contents << " return \"v8::internal::" << name << "\";\n";
475 cc_contents << "}\n";
476
477 cc_contents << "\nvoid Tq" << name
478 << "::Visit(TqObjectVisitor* visitor) const {\n";
479 cc_contents << " visitor->Visit" << name << "(this);\n";
480 cc_contents << "}\n";
481
482 cc_contents << "\nbool Tq" << name
483 << "::IsSuperclassOf(const TqObject* other) const {\n";
484 cc_contents
485 << " return GetName() != other->GetName() && dynamic_cast<const Tq"
486 << name << "*>(other) != nullptr;\n";
487 cc_contents << "}\n";
488
489 // By default, the visitor method for this class just calls the visitor method
490 // for this class's parent. This allows custom visitors to only override a few
491 // classes they care about without needing to know about the entire hierarchy.
492 visitor << " virtual void Visit" << name << "(const Tq" << name
493 << "* object) {\n";
494 visitor << " Visit" << super_name << "(object);\n";
495 visitor << " }\n";
496
497 std::stringstream get_props_impl;
498
499 for (const Field& field : type.fields()) {
500 if (field.name_and_type.type == TypeOracle::GetVoidType()) continue;
501 if (field.offset.has_value()) {
502 GenerateFieldAddressAccessor(field, name, h_contents, cc_contents);
503 GenerateFieldValueAccessor(field, name, h_contents, cc_contents);
504 }
505 GenerateGetPropsChunkForField(field, get_props_impl, name);
506 }
507
508 h_contents << "};\n";
509
510 cc_contents << "\nstd::vector<std::unique_ptr<ObjectProperty>> Tq" << name
511 << "::GetProperties(d::MemoryAccessor accessor) const {\n";
512 // Start by getting the fields from the parent class.
513 cc_contents << " std::vector<std::unique_ptr<ObjectProperty>> result = Tq"
514 << super_name << "::GetProperties(accessor);\n";
515 // Then add the fields from this class.
516 cc_contents << get_props_impl.str();
517 cc_contents << " return result;\n";
518 cc_contents << "}\n";
519}

522void ImplementationVisitor::GenerateClassDebugReaders(
523 const std::string& output_directory) {
524 const std::string file_name = "class-debug-readers";
525 std::stringstream h_contents;
526 std::stringstream cc_contents;
527 h_contents << "// Provides the ability to read object properties in\n";
528 h_contents << "// postmortem or remote scenarios, where the debuggee's\n";
529 h_contents << "// memory is not part of the current process's address\n";
530 h_contents << "// space and must be read using a callback function.\n\n";
531 {
532 IncludeGuardScope include_guard(h_contents, file_name + ".h");
533
534 h_contents << "#include <cstdint>\n";
535 h_contents << "#include <vector>\n\n";
536
537 for (const std::string& include_path : GlobalContext::CppIncludes()) {
538 h_contents << "#include " << StringLiteralQuote(include_path) << "\n";
539 }
540
541 h_contents
542 << "\n#include \"tools/debug_helper/debug-helper-internal.h\"\n\n";
543
544 const char* kWingdiWorkaround =
545 "// Unset a wingdi.h macro that causes conflicts.\n"
546 "#ifdef GetBValue\n"
547 "#undef GetBValue\n"
548 "#endif\n\n";
549
550 h_contents << kWingdiWorkaround;
551
552 cc_contents << "#include \"torque-generated/" << file_name << ".h\"\n\n";
553 cc_contents << "#include \"src/objects/all-objects-inl.h\"\n";
554 cc_contents << "#include \"torque-generated/debug-macros.h\"\n\n";
555 cc_contents << kWingdiWorkaround;
556 cc_contents << "namespace i = v8::internal;\n\n";
557
558 NamespaceScope h_namespaces(h_contents,
559 {"v8", "internal", "debug_helper_internal"});
560 NamespaceScope cc_namespaces(cc_contents,
561 {"v8", "internal", "debug_helper_internal"});
562
563 std::stringstream visitor;
564 visitor << "\nclass TqObjectVisitor {\n";
565 visitor << " public:\n";
566 visitor << " virtual void VisitObject(const TqObject* object) {}\n";
567
568 std::unordered_set<const ClassType*> done;
569 for (const ClassType* type : TypeOracle::GetClasses()) {
570 GenerateClassDebugReader(*type, h_contents, cc_contents, visitor, &done);
571 }
572
573 visitor << "};\n";
574 h_contents << visitor.str();
575 }
576 WriteFile(output_directory + "/" + file_name + ".h", h_contents.str());
577 WriteFile(output_directory + "/" + file_name + ".cc", cc_contents.str());
578}
