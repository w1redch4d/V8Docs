55WaiterQueueNode* JSSynchronizationPrimitive::DestructivelyGetWaiterQueueHead(
56 Isolate* requester) {
57 if (V8_UNLIKELY(DEBUG_BOOL)) {
58 StateT state = AtomicStatePtr()->load(std::memory_order_relaxed);
59 DCHECK(IsWaiterQueueLockedField::decode(state));
60 USE(state);
61 }
62#if V8_COMPRESS_POINTERS
63 ExternalPointerHandle handle =
64 base::AsAtomic32::Relaxed_Load(waiter_queue_head_handle_location());
65 if (handle == kNullExternalPointerHandle) return nullptr;
66 // Clear external pointer after decoding as a safeguard, no other thread
67 // should be trying to access though the same non-null handle.
68 WaiterQueueNode* waiter_head = reinterpret_cast<WaiterQueueNode*>(
69 requester->shared_external_pointer_table().Exchange(handle, kNullAddress,
70 kWaiterQueueNodeTag));
71 return waiter_head;
72#else
73 return base::AsAtomicPointer::Relaxed_Load(waiter_queue_head_location());
74#endif // V8_COMPRESS_POINTERS
75}

77JSSynchronizationPrimitive::StateT
78JSSynchronizationPrimitive::SetWaiterQueueHead(Isolate* requester,
79 WaiterQueueNode* waiter_head,
80 StateT new_state) {
81 if (V8_UNLIKELY(DEBUG_BOOL)) {
82 StateT state = AtomicStatePtr()->load(std::memory_order_relaxed);
83 DCHECK(IsWaiterQueueLockedField::decode(state));
84 USE(state);
85 }
86#if V8_COMPRESS_POINTERS
87 ExternalPointerHandle handle =
88 base::AsAtomic32::Relaxed_Load(waiter_queue_head_handle_location());
89 if (waiter_head) {
90 new_state = HasWaitersField::update(new_state, true);
91 ExternalPointerTable& table = requester->shared_external_pointer_table();
92 if (handle == kNullExternalPointerHandle) {
93 handle = table.AllocateAndInitializeEntry(
94 requester->shared_external_pointer_space(),
95 reinterpret_cast<Address>(waiter_head), kWaiterQueueNodeTag);
96 // Use a Release_Store to ensure that the store of the pointer into the
97 // table is not reordered after the store of the handle. Otherwise, other
98 // threads may access an uninitialized table entry and crash.
99 base::AsAtomic32::Release_Store(waiter_queue_head_handle_location(),
100 handle);
101 EXTERNAL_POINTER_WRITE_BARRIER(*this, kWaiterQueueHeadOffset,
102 kWaiterQueueNodeTag);
103 return new_state;
104 }
105 if (DEBUG_BOOL) {
106 Address old = requester->shared_external_pointer_table().Exchange(
107 handle, reinterpret_cast<Address>(waiter_head), kWaiterQueueNodeTag);
108 DCHECK_EQ(kNullAddress, old);
109 USE(old);
110 } else {
111 requester->shared_external_pointer_table().Set(
112 handle, reinterpret_cast<Address>(waiter_head), kWaiterQueueNodeTag);
113 }
114 } else {
115 new_state = HasWaitersField::update(new_state, false);
116 if (handle) {
117 requester->shared_external_pointer_table().Set(handle, kNullAddress,
118 kWaiterQueueNodeTag);
119 }
120 }
121#else
122 new_state = HasWaitersField::update(new_state, waiter_head);
123 base::AsAtomicPointer::Relaxed_Store(waiter_queue_head_location(),
124 waiter_head);
125#endif // V8_COMPRESS_POINTERS
126 return new_state;
127}

150// static
151template <typename LockSlowPathWrapper, typename>
152bool JSAtomicsMutex::LockImpl(Isolate* requester,
153 DirectHandle<JSAtomicsMutex> mutex,
154 std::optional<base::TimeDelta> timeout,
155 LockSlowPathWrapper slow_path_wrapper) {
156 DisallowGarbageCollection no_gc;
157 // First try to lock an uncontended mutex, which should be the common case. If
158 // this fails, then go to the slow path to possibly put the current thread to
159 // sleep.
160 //
161 // The fast path is done using a weak CAS which may fail spuriously on
162 // architectures with load-link/store-conditional instructions.
163 std::atomic<StateT>* state = mutex->AtomicStatePtr();
164 StateT expected = kUnlockedUncontended;
165 bool locked;
166 if (V8_LIKELY(state->compare_exchange_weak(expected, kLockedUncontended,
167 std::memory_order_acquire,
168 std::memory_order_relaxed))) {
169 locked = true;
170 } else {
171 locked = slow_path_wrapper(state);
172 }
173 if (V8_LIKELY(locked)) {
174 mutex->SetCurrentThreadAsOwner();
175 }
176 return locked;
177}

179// static
180bool JSAtomicsMutex::Lock(Isolate* requester,
181 DirectHandle<JSAtomicsMutex> mutex,
182 std::optional<base::TimeDelta> timeout) {
183 return LockImpl(requester, mutex, timeout, [=](std::atomic<StateT>* state) {
184 return LockSlowPath(requester, mutex, state, timeout);
185 });
186}

188bool JSAtomicsMutex::TryLock() {
189 DisallowGarbageCollection no_gc;
190 StateT expected = kUnlockedUncontended;
191 if (V8_LIKELY(AtomicStatePtr()->compare_exchange_strong(
192 expected, kLockedUncontended, std::memory_order_acquire,
193 std::memory_order_relaxed))) {
194 SetCurrentThreadAsOwner();
195 return true;
196 }
197 return false;
198}

200void JSAtomicsMutex::Unlock(Isolate* requester) {
201 DisallowGarbageCollection no_gc;
202 // First try to unlock an uncontended mutex, which should be the common
203 // case. If this fails, then go to the slow path to wake a waiting thread.
204 //
205 // In contrast to Lock, the fast path is done using a strong CAS which does
206 // not fail spuriously. This simplifies the slow path by guaranteeing that
207 // there is at least one waiter to be notified.
208 DCHECK(IsCurrentThreadOwner());
209 ClearOwnerThread();
210 std::atomic<StateT>* state = AtomicStatePtr();
211 StateT expected = kLockedUncontended;
212 if (V8_LIKELY(state->compare_exchange_strong(expected, kUnlockedUncontended,
213 std::memory_order_release,
214 std::memory_order_relaxed))) {
215 return;
216 }
217 UnlockSlowPath(requester, state);
218}
