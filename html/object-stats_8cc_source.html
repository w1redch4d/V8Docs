55 void RecordStats(Tagged<HeapObject> host) {
56 size_t old_pointer_fields_count = *tagged_fields_count_;
57 VisitObject(heap_->isolate(), host, this);
58 size_t tagged_fields_count_in_object =
59 *tagged_fields_count_ - old_pointer_fields_count;
60
61 int object_size_in_words = host->Size(cage_base()) / kTaggedSize;
62 DCHECK_LE(tagged_fields_count_in_object, object_size_in_words);
63 size_t raw_fields_count_in_object =
64 object_size_in_words - tagged_fields_count_in_object;
65
66 if (IsJSObject(host, cage_base())) {
67 JSObjectFieldStats field_stats = GetInobjectFieldStats(host->map());
68 // Embedder fields are already included into pointer words.
69 DCHECK_LE(field_stats.embedded_fields_count_,
70 tagged_fields_count_in_object);
71 tagged_fields_count_in_object -= field_stats.embedded_fields_count_;
72 *tagged_fields_count_ -= field_stats.embedded_fields_count_;
73 *embedder_fields_count_ += field_stats.embedded_fields_count_;
74
75 // Smi fields are also included into pointer words.
76 tagged_fields_count_in_object -= field_stats.smi_fields_count_;
77 *tagged_fields_count_ -= field_stats.smi_fields_count_;
78 *inobject_smi_fields_count_ += field_stats.smi_fields_count_;
79 } else if (IsHeapNumber(host, cage_base())) {
80 DCHECK_LE(kDoubleSize / kTaggedSize, raw_fields_count_in_object);
81 raw_fields_count_in_object -= kDoubleSize / kTaggedSize;
82 *boxed_double_fields_count_ += 1;
83 } else if (IsSeqString(host, cage_base())) {
84 int string_data =
85 Cast<SeqString>(host)->length(kAcquireLoad) *
86 (Cast<String>(host)->IsOneByteRepresentation() ? 1 : 2) / kTaggedSize;
87 DCHECK_LE(string_data, raw_fields_count_in_object);
88 raw_fields_count_in_object -= string_data;
89 *string_data_count_ += string_data;
90 }
91 *raw_fields_count_ += raw_fields_count_in_object;
92 }

144FieldStatsCollector::JSObjectFieldStats
145FieldStatsCollector::GetInobjectFieldStats(Tagged<Map> map) {
146 auto iter = field_stats_cache_.find(map);
147 if (iter != field_stats_cache_.end()) {
148 return iter->second;
149 }
150 // Iterate descriptor array and calculate stats.
151 JSObjectFieldStats stats;
152 stats.embedded_fields_count_ = JSObject::GetEmbedderFieldCount(map);
153 if (!map->is_dictionary_map()) {
154 Tagged<DescriptorArray> descriptors = map->instance_descriptors();
155 for (InternalIndex descriptor : map->IterateOwnDescriptors()) {
156 PropertyDetails details = descriptors->GetDetails(descriptor);
157 if (details.location() == PropertyLocation::kField) {
158 FieldIndex index = FieldIndex::ForDetails(map, details);
159 // Stop on first out-of-object field.
160 if (!index.is_inobject()) break;
161 if (details.representation().IsSmi()) {
162 ++stats.smi_fields_count_;
163 }
164 }
165 }
166 }
167 field_stats_cache_.insert(std::make_pair(map, stats));
168 return stats;
169}

538 if (!site->PointsToLiteral()) return;
539 Tagged<JSObject> boilerplate = site->boilerplate();
540 if (IsJSArray(boilerplate)) {

550 // single instance, this is negligible.

552 RecordSimpleVirtualObjectStats(

556 RecordSimpleVirtualObjectStats(
557 site, properties, ObjectStats::BOILERPLATE_PROPERTY_DICTIONARY_TYPE);

595 !Cast<JSFunction>(object)->is_compiled(isolate())) {
596 RecordSimpleVirtualObjectStats(HeapObject(), object,
597 ObjectStats::JS_UNCOMPILED_FUNCTION_TYPE);
598 }
599
600 // Properties.
601 if (object->HasFastProperties()) {
602 Tagged<PropertyArray> properties = object->property_array();
603 if (properties != ReadOnlyRoots(heap_).empty_property_array()) {
604 size_t over_allocated =
605 object->map()->UnusedPropertyFields() * kTaggedSize;
606 RecordVirtualObjectStats(object, properties,
607 object->map()->is_prototype_map()
608 ? ObjectStats::PROTOTYPE_PROPERTY_ARRAY_TYPE
609 : ObjectStats::OBJECT_PROPERTY_ARRAY_TYPE,
610 properties->Size(), over_allocated);
611 }
612 } else {
613 Tagged<NameDictionary> properties = object->property_dictionary();
614 RecordHashTableVirtualObjectStats(
615 object, properties,
616 object->map()->is_prototype_map()
617 ? ObjectStats::PROTOTYPE_PROPERTY_DICTIONARY_TYPE
618 : ObjectStats::OBJECT_PROPERTY_DICTIONARY_TYPE);
619 }
620
621 // Elements.
622 Tagged<FixedArrayBase> elements = object->elements();
623 if (object->HasDictionaryElements()) {
624 RecordHashTableVirtualObjectStats(
625 object, Cast<NumberDictionary>(elements),
626 IsJSArray(object) ? ObjectStats::ARRAY_DICTIONARY_ELEMENTS_TYPE
627 : ObjectStats::OBJECT_DICTIONARY_ELEMENTS_TYPE);
628 } else if (IsJSArray(object)) {
629 if (elements != ReadOnlyRoots(heap_).empty_fixed_array()) {
630 size_t element_size =
631 (elements->Size() - FixedArrayBase::kHeaderSize) / elements->length();
632 uint32_t length = Object::NumberValue(Cast<JSArray>(object)->length());
633 size_t over_allocated = (elements->length() - length) * element_size;
634 RecordVirtualObjectStats(object, elements,
635 ObjectStats::ARRAY_ELEMENTS_TYPE,
636 elements->Size(), over_allocated);
637 }
638 } else {
639 RecordSimpleVirtualObjectStats(object, elements,
640 ObjectStats::OBJECT_ELEMENTS_TYPE);
641 }
642
643 // JSCollections.
644 if (IsJSCollection(object)) {
645 Tagged<Object> maybe_table = Cast<JSCollection>(object)->table();
646 if (!IsUndefined(maybe_table, isolate())) {
647 DCHECK(IsFixedArray(maybe_table, isolate()));
648 // TODO(bmeurer): Properly compute over-allocation here.
649 RecordSimpleVirtualObjectStats(object, Cast<HeapObject>(maybe_table),
650 ObjectStats::JS_COLLECTION_TABLE_TYPE);
651 }
652 }

655static ObjectStats::VirtualInstanceType GetFeedbackSlotType(
656 Tagged<MaybeObject> maybe_obj, FeedbackSlotKind kind, Isolate* isolate) {
657 if (maybe_obj.IsCleared())
658 return ObjectStats::FEEDBACK_VECTOR_SLOT_OTHER_TYPE;
659 Tagged<Object> obj = maybe_obj.GetHeapObjectOrSmi();
660 switch (kind) {
661 case FeedbackSlotKind::kCall:
662 if (obj == *isolate->factory()->uninitialized_symbol()) {
663 return ObjectStats::FEEDBACK_VECTOR_SLOT_CALL_UNUSED_TYPE;
664 }
665 return ObjectStats::FEEDBACK_VECTOR_SLOT_CALL_TYPE;
666
667 case FeedbackSlotKind::kLoadProperty:
668 case FeedbackSlotKind::kLoadGlobalInsideTypeof:
669 case FeedbackSlotKind::kLoadGlobalNotInsideTypeof:
670 case FeedbackSlotKind::kLoadKeyed:
671 case FeedbackSlotKind::kHasKeyed:
672 if (obj == *isolate->factory()->uninitialized_symbol()) {
673 return ObjectStats::FEEDBACK_VECTOR_SLOT_LOAD_UNUSED_TYPE;

675 return ObjectStats::FEEDBACK_VECTOR_SLOT_LOAD_TYPE;
676
677 case FeedbackSlotKind::kSetNamedSloppy:
678 case FeedbackSlotKind::kSetNamedStrict:
679 case FeedbackSlotKind::kDefineNamedOwn:
680 case FeedbackSlotKind::kStoreGlobalSloppy:
681 case FeedbackSlotKind::kStoreGlobalStrict:
682 case FeedbackSlotKind::kSetKeyedSloppy:
683 case FeedbackSlotKind::kSetKeyedStrict:
684 if (obj == *isolate->factory()->uninitialized_symbol()) {
685 return ObjectStats::FEEDBACK_VECTOR_SLOT_STORE_UNUSED_TYPE;
686 }
687 return ObjectStats::FEEDBACK_VECTOR_SLOT_STORE_TYPE;
688
689 case FeedbackSlotKind::kBinaryOp:
690 case FeedbackSlotKind::kCompareOp:
691 return ObjectStats::FEEDBACK_VECTOR_SLOT_ENUM_TYPE;
692
693 default:
694 return ObjectStats::FEEDBACK_VECTOR_SLOT_OTHER_TYPE;

698void ObjectStatsCollectorImpl::RecordVirtualFeedbackVectorDetails(
699 Tagged<FeedbackVector> vector) {
700 if (virtual_objects_.find(vector) != virtual_objects_.end()) return;
701 // Manually insert the feedback vector into the virtual object list, since
702 // we're logging its component parts separately.
703 virtual_objects_.insert(vector);
704
705 size_t calculated_size = 0;
706
707 // Log the feedback vector's header (fixed fields).
708 size_t header_size = vector->slots_start().address() - vector.address();
709 stats_->RecordVirtualObjectStats(ObjectStats::FEEDBACK_VECTOR_HEADER_TYPE,
710 header_size, ObjectStats::kNoOverAllocation);
711 calculated_size += header_size;
712
713 // Iterate over the feedback slots and log each one.
714 if (!vector->shared_function_info()->HasFeedbackMetadata()) return;
715
716 FeedbackMetadataIterator it(vector->metadata());
717 while (it.HasNext()) {
718 FeedbackSlot slot = it.Next();
719 // Log the entry (or entries) taken up by this slot.
720 size_t slot_size = it.entry_size() * kTaggedSize;
721 stats_->RecordVirtualObjectStats(
722 GetFeedbackSlotType(vector->Get(slot), it.kind(), heap_->isolate()),
723 slot_size, ObjectStats::kNoOverAllocation);
724 calculated_size += slot_size;
725
726 // Log the monomorphic/polymorphic helper objects that this slot owns.
727 for (int i = 0; i < it.entry_size(); i++) {
728 Tagged<MaybeObject> raw_object = vector->Get(slot.WithOffset(i));
729 Tagged<HeapObject> object;
730 if (raw_object.GetHeapObject(&object)) {
731 if (IsCell(object, cage_base()) ||
732 IsWeakFixedArray(object, cage_base())) {
733 RecordSimpleVirtualObjectStats(
734 vector, object, ObjectStats::FEEDBACK_VECTOR_ENTRY_TYPE);
735 }
736 }
737 }
738 }
739
740 CHECK_EQ(calculated_size, vector->Size());
741}

752void ObjectStatsCollectorImpl::CollectStatistics(
753 Tagged<HeapObject> obj, Phase phase,
754 CollectFieldStats collect_field_stats) {
755 DisallowGarbageCollection no_gc;
756 Tagged<Map> map = obj->map(cage_base());
757 InstanceType instance_type = map->instance_type();
758 switch (phase) {
759 case kPhase1:
760 if (InstanceTypeChecker::IsFeedbackVector(instance_type)) {
761 RecordVirtualFeedbackVectorDetails(Cast<FeedbackVector>(obj));
762 } else if (InstanceTypeChecker::IsMap(instance_type)) {
763 RecordVirtualMapDetails(Cast<Map>(obj));
764 } else if (InstanceTypeChecker::IsBytecodeArray(instance_type)) {
765 RecordVirtualBytecodeArrayDetails(Cast<BytecodeArray>(obj));
766 } else if (InstanceTypeChecker::IsInstructionStream(instance_type)) {
767 RecordVirtualCodeDetails(Cast<InstructionStream>(obj));
768 } else if (InstanceTypeChecker::IsFunctionTemplateInfo(instance_type)) {
769 RecordVirtualFunctionTemplateInfoDetails(
770 Cast<FunctionTemplateInfo>(obj));
771 } else if (InstanceTypeChecker::IsJSGlobalObject(instance_type)) {
772 RecordVirtualJSGlobalObjectDetails(Cast<JSGlobalObject>(obj));
773 } else if (InstanceTypeChecker::IsJSObject(instance_type)) {
774 // This phase needs to come after RecordVirtualAllocationSiteDetails
775 // to properly split among boilerplates.
776 RecordVirtualJSObjectDetails(Cast<JSObject>(obj));
777 } else if (InstanceTypeChecker::IsSharedFunctionInfo(instance_type)) {
778 RecordVirtualSharedFunctionInfoDetails(Cast<SharedFunctionInfo>(obj));
779 } else if (InstanceTypeChecker::IsContext(instance_type)) {
780 RecordVirtualContext(Cast<Context>(obj));
781 } else if (InstanceTypeChecker::IsScript(instance_type)) {
782 RecordVirtualScriptDetails(Cast<Script>(obj));
783 } else if (InstanceTypeChecker::IsArrayBoilerplateDescription(
784 instance_type)) {
785 RecordVirtualArrayBoilerplateDescription(
786 Cast<ArrayBoilerplateDescription>(obj));
787 } else if (InstanceTypeChecker::IsFixedArrayExact(instance_type)) {
788 // Has to go last as it triggers too eagerly.
789 RecordVirtualFixedArrayDetails(Cast<FixedArray>(obj));
790 }
791 break;
792 case kPhase2:
793 if (InstanceTypeChecker::IsExternalString(instance_type)) {
794 // This has to be in Phase2 to avoid conflicting with recording Script
795 // sources. We still want to run RecordObjectStats after though.
796 RecordVirtualExternalStringDetails(Cast<ExternalString>(obj));
797 }
798 size_t over_allocated = ObjectStats::kNoOverAllocation;
799 if (InstanceTypeChecker::IsJSObject(instance_type)) {
800 over_allocated = map->instance_size() - map->UsedInstanceSize();
801 }
802 RecordObjectStats(obj, instance_type, obj->Size(cage_base()),
803 over_allocated);
804 if (collect_field_stats == CollectFieldStats::kYes) {
805 field_stats_collector_.RecordStats(obj);
806 }
807 break;
808 }

869void ObjectStatsCollectorImpl::RecordVirtualMapDetails(Tagged<Map> map) {
870 // TODO(mlippautz): map->dependent_code(): DEPENDENT_CODE_TYPE.
871
872 // For Map we want to distinguish between various different states
873 // to get a better picture of what's going on in MapSpace. This
874 // method computes the virtual instance type to use for a given map,
875 // using MAP_TYPE for regular maps that aren't special in any way.
876 if (map->is_prototype_map()) {
877 if (map->is_dictionary_map()) {
878 RecordSimpleVirtualObjectStats(
879 HeapObject(), map, ObjectStats::MAP_PROTOTYPE_DICTIONARY_TYPE);
880 } else if (map->is_abandoned_prototype_map()) {
881 RecordSimpleVirtualObjectStats(HeapObject(), map,
882 ObjectStats::MAP_ABANDONED_PROTOTYPE_TYPE);
883 } else {
884 RecordSimpleVirtualObjectStats(HeapObject(), map,
885 ObjectStats::MAP_PROTOTYPE_TYPE);
886 }
887 } else if (map->is_deprecated()) {
888 RecordSimpleVirtualObjectStats(HeapObject(), map,
889 ObjectStats::MAP_DEPRECATED_TYPE);
890 } else if (map->is_dictionary_map()) {
891 RecordSimpleVirtualObjectStats(HeapObject(), map,
892 ObjectStats::MAP_DICTIONARY_TYPE);
893 } else if (map->is_stable()) {
894 RecordSimpleVirtualObjectStats(HeapObject(), map,
895 ObjectStats::MAP_STABLE_TYPE);
896 } else {
897 // This will be logged as MAP_TYPE in Phase2.
898 }
899
900 Tagged<DescriptorArray> array = map->instance_descriptors(cage_base());
901 if (map->owns_descriptors() &&
902 array != ReadOnlyRoots(heap_).empty_descriptor_array()) {
903 // Generally DescriptorArrays have their own instance type already
904 // (DESCRIPTOR_ARRAY_TYPE), but we'd like to be able to tell which
905 // of those are for (abandoned) prototypes, and which of those are
906 // owned by deprecated maps.
907 if (map->is_prototype_map()) {
908 RecordSimpleVirtualObjectStats(
909 map, array, ObjectStats::PROTOTYPE_DESCRIPTOR_ARRAY_TYPE);
910 } else if (map->is_deprecated()) {
911 RecordSimpleVirtualObjectStats(
912 map, array, ObjectStats::DEPRECATED_DESCRIPTOR_ARRAY_TYPE);
913 }
914
915 Tagged<EnumCache> enum_cache = array->enum_cache();
916 RecordSimpleVirtualObjectStats(array, enum_cache->keys(),
917 ObjectStats::ENUM_KEYS_CACHE_TYPE);
918 RecordSimpleVirtualObjectStats(array, enum_cache->indices(),
919 ObjectStats::ENUM_INDICES_CACHE_TYPE);
920 }
921
922 if (map->is_prototype_map()) {
923 Tagged<PrototypeInfo> prototype_info;
924 if (map->TryGetPrototypeInfo(&prototype_info)) {
925 Tagged<Object> users = prototype_info->prototype_users();
926 if (IsWeakFixedArray(users, cage_base())) {
927 RecordSimpleVirtualObjectStats(map, Cast<WeakArrayList>(users),
928 ObjectStats::PROTOTYPE_USERS_TYPE);
929 }
930 }
931 }
