102 template <int N>
103 V8_INLINE void AppendCStringLiteral(const char (&literal)[N]) {
104 // Note that the literal contains the zero char.
105 const int length = N - 1;
106 static_assert(length > 0);
107 if (length == 1) return AppendCharacter(literal[0]);
108 if (encoding_ == String::ONE_BYTE_ENCODING && CurrentPartCanFit(N)) {
109 const uint8_t* chars = reinterpret_cast<const uint8_t*>(literal);
110 CopyChars<uint8_t, uint8_t>(one_byte_ptr_ + current_index_, chars,
111 length);
112 current_index_ += length;

114 DCHECK(HasValidCurrentIndex());

116 }

138 V8_INLINE bool CurrentPartCanFit(uint32_t length) {
139 return part_length_ - current_index_ > length;

142 // We make a rough estimate to find out if the current string can be
143 // serialized without allocating a new string part. The worst case length of
144 // an escaped character is 6. Shifting the remaining string length right by 3
145 // is a more pessimistic estimate, but faster to calculate.
146 V8_INLINE bool EscapedLengthIfCurrentPartFits(uint32_t length) {
147 if (length > kMaxPartLength) return false;
148 static_assert(kMaxPartLength <= (String::kMaxLength >> 3));
149 // This shift will not overflow because length is already less than the
150 // maximum part length.
151 return CurrentPartCanFit(length << 3);
152 }

154 void AppendStringByCopy(Tagged<String> string, uint32_t length,
155 const DisallowGarbageCollection& no_gc) {
156 DCHECK_EQ(length, string->length());
157 DCHECK(encoding_ == String::TWO_BYTE_ENCODING ||
158 (string->IsFlat() && string->IsOneByteRepresentation()));
159 DCHECK(CurrentPartCanFit(length + 1));
160 String::FlatContent flat = string->GetFlatContent(no_gc);
161 if (encoding_ == String::ONE_BYTE_ENCODING) {
162 if (flat.IsOneByte()) {
163 CopyChars<uint8_t, uint8_t>(one_byte_ptr_ + current_index_,
164 flat.ToOneByteVector().begin(), length);
165 } else {
166 ChangeEncoding();
167 CopyChars<uint16_t, uint16_t>(two_byte_ptr_ + current_index_,
168 flat.ToUC16Vector().begin(), length);
169 }
170 } else {
171 if (flat.IsOneByte()) {
172 CopyChars<uint8_t, uint16_t>(two_byte_ptr_ + current_index_,
173 flat.ToOneByteVector().begin(), length);
174 } else {
175 CopyChars<uint16_t, uint16_t>(two_byte_ptr_ + current_index_,
176 flat.ToUC16Vector().begin(), length);
177 }
178 }
179 current_index_ += length;
180 DCHECK(current_index_ <= part_length_);
181 }

313 // A cache of recently seen property keys which were simple. Simple means:
314 //
315 // - Internalized, sequential, one-byte string
316 // - Contains no characters which need escaping
317 //
318 // This can be helpful because it's common for JSON to have lists of similar
319 // objects. Since property keys are internalized, we will see identical key
320 // pointers again and again, and we can use a fast path to copy those keys to
321 // the output. However, strings can be externalized any time JS runs, so the
322 // caller is responsible for checking whether a string is still the expected
323 // type. This cache is cleared on GC, since the GC could move those strings.
324 // Using Handles for the cache has been tried, but is too expensive to set up
325 // when JSON.stringify is called for tiny inputs.
326 class SimplePropertyKeyCache {
327 public:
328 explicit SimplePropertyKeyCache(Isolate* isolate) : isolate_(isolate) {
329 Clear();
330 isolate->main_thread_local_heap()->AddGCEpilogueCallback(
331 UpdatePointersCallback, this);
332 }
333
334 ~SimplePropertyKeyCache() {
335 isolate_->main_thread_local_heap()->RemoveGCEpilogueCallback(
336 UpdatePointersCallback, this);
337 }
338
339 void TryInsert(Tagged<String> string) {
340 ReadOnlyRoots roots(isolate_);
341 if (string->map() == roots.internalized_one_byte_string_map()) {
342 keys_[GetIndex(string)] = MaybeCompress(string);
343 }
344 }
345
346 bool Contains(Tagged<String> string) {
347 return keys_[GetIndex(string)] == MaybeCompress(string);
348 }
349
350 private:
351 size_t GetIndex(Tagged<String> string) {
352 // Short strings are 16 bytes long in pointer-compression builds, so the
353 // lower four bits of the pointer may not provide much entropy.
354 return (string.ptr() >> 4) & kIndexMask;
355 }
356
357 Tagged_t MaybeCompress(Tagged<String> string) {
358 return COMPRESS_POINTERS_BOOL
359 ? V8HeapCompressionScheme::CompressObject(string.ptr())
360 : static_cast<Tagged_t>(string.ptr());
361 }
362
363 void Clear() { MemsetTagged(keys_, Smi::zero(), kSize); }
364
365 static void UpdatePointersCallback(void* cache) {
366 reinterpret_cast<SimplePropertyKeyCache*>(cache)->Clear();
367 }
368
369 static constexpr size_t kSizeBits = 6;
370 static constexpr size_t kSize = 1 << kSizeBits;
371 static constexpr size_t kIndexMask = kSize - 1;
372
373 Isolate* isolate_;
374 Tagged_t keys_[kSize];
375 };

547MaybeHandle<Object> JsonStringifier::Stringify(Handle<JSAny> object,
548 Handle<JSAny> replacer,
549 Handle<Object> gap) {
550 if (!InitializeReplacer(replacer)) {
551 CHECK(isolate_->has_exception());
552 return MaybeHandle<Object>();
553 }
554 if (!IsUndefined(*gap, isolate_) && !InitializeGap(gap)) {
555 CHECK(isolate_->has_exception());
556 return MaybeHandle<Object>();
557 }
558 Result result = SerializeObject(object);
559 if (result == NEED_STACK) {
560 indent_ = 0;
561 current_index_ = 0;
562 result = SerializeObject(object);
563 }
564 if (result == UNCHANGED) return factory()->undefined_value();
565 if (result == SUCCESS) {
566 if (overflowed_ || current_index_ > String::kMaxLength) {
567 THROW_NEW_ERROR(isolate_, NewInvalidStringLengthError());
568 }
569 if (encoding_ == String::ONE_BYTE_ENCODING) {
570 return isolate_->factory()
571 ->NewStringFromOneByte(base::OneByteVector(
572 reinterpret_cast<char*>(one_byte_ptr_), current_index_))
573 .ToHandleChecked();
574 } else {
575 return isolate_->factory()->NewStringFromTwoByte(
576 base::Vector<const base::uc16>(two_byte_ptr_, current_index_));
577 }
578 }
579 DCHECK(result == EXCEPTION);
580 CHECK(isolate_->has_exception());
581 return MaybeHandle<Object>();
582}

584bool JsonStringifier::InitializeReplacer(Handle<JSAny> replacer) {
585 DCHECK(property_list_.is_null());
586 DCHECK(replacer_function_.is_null());
587 Maybe<bool> is_array = Object::IsArray(replacer);
588 if (is_array.IsNothing()) return false;
589 if (is_array.FromJust()) {
590 HandleScope handle_scope(isolate_);
591 Handle<OrderedHashSet> set = factory()->NewOrderedHashSet();
592 DirectHandle<Object> length_obj;
593 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
594 isolate_, length_obj,
595 Object::GetLengthFromArrayLike(isolate_, Cast<JSReceiver>(replacer)),
596 false);
597 uint32_t length;
598 if (!Object::ToUint32(*length_obj, &length)) length = kMaxUInt32;
599 for (uint32_t i = 0; i < length; i++) {
600 Handle<Object> element;
601 Handle<String> key;
602 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
603 isolate_, element, Object::GetElement(isolate_, replacer, i), false);
604 if (IsNumber(*element) || IsString(*element)) {
605 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
606 isolate_, key, Object::ToString(isolate_, element), false);
607 } else if (IsJSPrimitiveWrapper(*element)) {
608 DirectHandle<Object> value(Cast<JSPrimitiveWrapper>(element)->value(),
609 isolate_);
610 if (IsNumber(*value) || IsString(*value)) {
611 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
612 isolate_, key, Object::ToString(isolate_, element), false);
613 }
614 }
615 if (key.is_null()) continue;
616 // Object keys are internalized, so do it here.
617 key = factory()->InternalizeString(key);
618 MaybeHandle<OrderedHashSet> set_candidate =
619 OrderedHashSet::Add(isolate_, set, key);
620 if (!set_candidate.ToHandle(&set)) {
621 CHECK(isolate_->has_exception());
622 return false;
623 }
624 }
625 property_list_ = OrderedHashSet::ConvertToKeysArray(
626 isolate_, set, GetKeysConversion::kKeepNumbers);
627 property_list_ = handle_scope.CloseAndEscape(property_list_);
628 } else if (IsCallable(*replacer)) {
629 replacer_function_ = Cast<JSReceiver>(replacer);
630 }
631 return true;
632}

634bool JsonStringifier::InitializeGap(Handle<Object> gap) {
635 DCHECK_NULL(gap_);
636 HandleScope scope(isolate_);
637 if (IsJSPrimitiveWrapper(*gap)) {
638 DirectHandle<Object> value(Cast<JSPrimitiveWrapper>(gap)->value(),
639 isolate_);
640 if (IsString(*value)) {
641 ASSIGN_RETURN_ON_EXCEPTION_VALUE(isolate_, gap,
642 Object::ToString(isolate_, gap), false);
643 } else if (IsNumber(*value)) {
644 ASSIGN_RETURN_ON_EXCEPTION_VALUE(isolate_, gap,
645 Object::ToNumber(isolate_, gap), false);
646 }
647 }
648
649 if (IsString(*gap)) {
650 auto gap_string = Cast<String>(gap);
651 if (gap_string->length() > 0) {
652 uint32_t gap_length = std::min(gap_string->length(), 10u);
653 gap_ = NewArray<base::uc16>(gap_length + 1);
654 String::WriteToFlat(*gap_string, gap_, 0, gap_length);
655 for (uint32_t i = 0; i < gap_length; i++) {
656 if (gap_[i] > String::kMaxOneByteCharCode) {
657 ChangeEncoding();
658 break;
659 }
660 }
661 gap_[gap_length] = '\0';
662 }
663 } else if (IsNumber(*gap)) {
664 double value = std::min(Object::NumberValue(*gap), 10.0);
665 if (value > 0) {
666 uint32_t gap_length = DoubleToUint32(value);
667 gap_ = NewArray<base::uc16>(gap_length + 1);
668 for (uint32_t i = 0; i < gap_length; i++) gap_[i] = ' ';
669 gap_[gap_length] = '\0';
670 }
671 }
672 return true;
673}

675MaybeHandle<JSAny> JsonStringifier::ApplyToJsonFunction(
676 Handle<JSAny> object, DirectHandle<Object> key) {
677 HandleScope scope(isolate_);
678
679 // Retrieve toJSON function. The LookupIterator automatically handles
680 // the ToObject() equivalent ("GetRoot") if {object} is a BigInt.
681 DirectHandle<Object> fun;
682 LookupIterator it(isolate_, object, factory()->toJSON_string(),
683 LookupIterator::PROTOTYPE_CHAIN_SKIP_INTERCEPTOR);
684 ASSIGN_RETURN_ON_EXCEPTION(isolate_, fun, Object::GetProperty(&it));
685 if (!IsCallable(*fun)) return object;
686
687 // Call toJSON function.
688 if (IsSmi(*key)) key = factory()->NumberToString(key);
689 DirectHandle<Object> args[] = {key};
690 ASSIGN_RETURN_ON_EXCEPTION(isolate_, object,
691 Cast<JSAny>(Execution::Call(
692 isolate_, fun, object, base::VectorOf(args))));
693 return scope.CloseAndEscape(object);
694}

696MaybeHandle<JSAny> JsonStringifier::ApplyReplacerFunction(
697 Handle<JSAny> value, DirectHandle<Object> key,
698 DirectHandle<Object> initial_holder) {
699 HandleScope scope(isolate_);
700 if (IsSmi(*key)) key = factory()->NumberToString(key);
701 DirectHandle<Object> args[] = {key, value};
702 DirectHandle<JSReceiver> holder = CurrentHolder(value, initial_holder);
703 ASSIGN_RETURN_ON_EXCEPTION(

705 Cast<JSAny>(Execution::Call(isolate_, replacer_function_, holder,

707 return scope.CloseAndEscape(value);

712 if (stack_.empty()) {

724JsonStringifier::Result JsonStringifier::StackPush(Handle<Object> object,
725 Handle<Object> key) {
726 if (!need_stack_) {
727 ++stack_nesting_level_;
728 if V8_UNLIKELY (stack_nesting_level_ > 10) {
729 need_stack_ = true;
730 return NEED_STACK;
731 }
732 return SUCCESS;
733 }
734 StackLimitCheck check(isolate_);
735 if (check.HasOverflowed()) {
736 isolate_->StackOverflow();
737 return EXCEPTION;
738 }
739
740 {
741 DisallowGarbageCollection no_gc;
742 Tagged<Object> raw_obj = *object;
743 size_t size = stack_.size();
744 for (size_t i = 0; i < size; ++i) {
745 if (*stack_[i].second == raw_obj) {
746 AllowGarbageCollection allow_to_return_error;
747 Handle<String> circle_description =
748 ConstructCircularStructureErrorMessage(key, i);
749 DirectHandle<Object> error = factory()->NewTypeError(
750 MessageTemplate::kCircularStructure, circle_description);
751 isolate_->Throw(*error);
752 return EXCEPTION;
753 }
754 }
755 }
756 stack_.emplace_back(key, object);
757 return SUCCESS;
758}

760void JsonStringifier::StackPop() {
761 if V8_LIKELY (!need_stack_) {
762 --stack_nesting_level_;
763 return;
764 }
765 stack_.pop_back();
766}

768class CircularStructureMessageBuilder {
769 public:
770 explicit CircularStructureMessageBuilder(Isolate* isolate)
771 : builder_(isolate) {}
772
773 void AppendStartLine(DirectHandle<Object> start_object) {
774 builder_.AppendCString(kStartPrefix);
775 builder_.AppendCStringLiteral("starting at object with constructor ");
776 AppendConstructorName(start_object);
777 }
778
779 void AppendNormalLine(DirectHandle<Object> key, DirectHandle<Object> object) {
780 builder_.AppendCString(kLinePrefix);
781 AppendKey(key);
782 builder_.AppendCStringLiteral(" -> object with constructor ");
783 AppendConstructorName(object);
784 }
785
786 void AppendClosingLine(DirectHandle<Object> closing_key) {
787 builder_.AppendCString(kEndPrefix);
788 AppendKey(closing_key);
789 builder_.AppendCStringLiteral(" closes the circle");
790 }
791
792 void AppendEllipsis() {
793 builder_.AppendCString(kLinePrefix);
794 builder_.AppendCStringLiteral("...");
795 }
796
797 MaybeDirectHandle<String> Finish() { return builder_.Finish(); }
798
799 private:
800 void AppendConstructorName(DirectHandle<Object> object) {
801 builder_.AppendCharacter('\'');
802 DirectHandle<String> constructor_name = JSReceiver::GetConstructorName(
803 builder_.isolate(), Cast<JSReceiver>(object));
804 builder_.AppendString(constructor_name);
805 builder_.AppendCharacter('\'');
806 }
807
808 // A key can either be a string, the empty string or a Smi.
809 void AppendKey(DirectHandle<Object> key) {
810 if (IsSmi(*key)) {
811 builder_.AppendCStringLiteral("index ");
812 AppendSmi(Cast<Smi>(*key));
813 return;
814 }
815
816 CHECK(IsString(*key));
817 DirectHandle<String> key_as_string = Cast<String>(key);
818 if (key_as_string->length() == 0) {
819 builder_.AppendCStringLiteral("<anonymous>");
820 } else {
821 builder_.AppendCStringLiteral("property '");
822 builder_.AppendString(key_as_string);
823 builder_.AppendCharacter('\'');
824 }
825 }
826
827 void AppendSmi(Tagged<Smi> smi) {
828 static_assert(Smi::kMaxValue <= 2147483647);
829 static_assert(Smi::kMinValue >= -2147483648);
830 // sizeof(string) includes \0.
831 static const int kBufferSize = sizeof("-2147483648") - 1;
832 char chars[kBufferSize];
833 base::Vector<char> buffer(chars, kBufferSize);
834 builder_.AppendString(IntToStringView(smi.value(), buffer));
835 }
836
837 IncrementalStringBuilder builder_;
838 static constexpr const char* kStartPrefix = "\n --> ";
839 static constexpr const char* kEndPrefix = "\n --- ";
840 static constexpr const char* kLinePrefix = "\n | ";
841};

843Handle<String> JsonStringifier::ConstructCircularStructureErrorMessage(
844 DirectHandle<Object> last_key, size_t start_index) {
845 DCHECK(start_index < stack_.size());
846 CircularStructureMessageBuilder builder(isolate_);
847
848 // We track the index to be printed next for better readability.
849 size_t index = start_index;
850 const size_t stack_size = stack_.size();
851
852 builder.AppendStartLine(stack_[index++].second);
853
854 // Append a maximum of kCircularErrorMessagePrefixCount normal lines.
855 const size_t prefix_end =
856 std::min(stack_size, index + kCircularErrorMessagePrefixCount);
857 for (; index < prefix_end; ++index) {
858 builder.AppendNormalLine(stack_[index].first, stack_[index].second);
859 }
860
861 // If the circle consists of too many objects, we skip them and just
862 // print an ellipsis.
863 if (stack_size > index + kCircularErrorMessagePostfixCount) {
864 builder.AppendEllipsis();
865 }
866
867 // Since we calculate the postfix lines from the back of the stack,
868 // we have to ensure that lines are not printed twice.
869 index = std::max(index, stack_size - kCircularErrorMessagePostfixCount);
870 for (; index < stack_size; ++index) {
871 builder.AppendNormalLine(stack_[index].first, stack_[index].second);
872 }
873
874 builder.AppendClosingLine(last_key);
875
876 Handle<String> result;
877 ASSIGN_RETURN_ON_EXCEPTION_VALUE(isolate_, result,
878 indirect_handle(builder.Finish(), isolate_),
879 factory()->empty_string());
880 return result;
881}

893template <bool deferred_string_key>
894JsonStringifier::Result JsonStringifier::Serialize_(Handle<JSAny> object,
895 bool comma,
896 Handle<Object> key) {
897 StackLimitCheck interrupt_check(isolate_);
898 if (interrupt_check.InterruptRequested() &&
899 IsException(isolate_->stack_guard()->HandleInterrupts(), isolate_)) {
900 return EXCEPTION;
901 }
902
903 DirectHandle<JSAny> initial_value = object;
904 PtrComprCageBase cage_base(isolate_);
905 if (!IsSmi(*object)) {
906 InstanceType instance_type =
907 Cast<HeapObject>(*object)->map(cage_base)->instance_type();
908 if ((InstanceTypeChecker::IsJSReceiver(instance_type) &&
909 MayHaveInterestingProperties(isolate_, Cast<JSReceiver>(*object))) ||
910 InstanceTypeChecker::IsBigInt(instance_type)) {
911 if (!need_stack_ && stack_nesting_level_ > 0) {
912 need_stack_ = true;
913 return NEED_STACK;
914 }
915 need_stack_ = true;
916 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
917 isolate_, object, ApplyToJsonFunction(object, key), EXCEPTION);
918 }
919 }
920 if (!replacer_function_.is_null()) {
921 need_stack_ = true;
922 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
923 isolate_, object, ApplyReplacerFunction(object, key, initial_value),
924 EXCEPTION);
925 }
926
927 if (IsSmi(*object)) {
928 if (deferred_string_key) SerializeDeferredKey(comma, key);
929 return SerializeSmi(Cast<Smi>(*object));
930 }
931
932 InstanceType instance_type =
933 Cast<HeapObject>(*object)->map(cage_base)->instance_type();
934 switch (instance_type) {
935 case HEAP_NUMBER_TYPE:
936 if (deferred_string_key) SerializeDeferredKey(comma, key);
937 return SerializeHeapNumber(Cast<HeapNumber>(object));
938 case BIGINT_TYPE:
939 isolate_->Throw(
940 *factory()->NewTypeError(MessageTemplate::kBigIntSerializeJSON));
941 return EXCEPTION;
942 case ODDBALL_TYPE:
943 switch (Cast<Oddball>(*object)->kind()) {
944 case Oddball::kFalse:
945 if (deferred_string_key) SerializeDeferredKey(comma, key);
946 AppendCStringLiteral("false");
947 return SUCCESS;
948 case Oddball::kTrue:
949 if (deferred_string_key) SerializeDeferredKey(comma, key);
950 AppendCStringLiteral("true");
951 return SUCCESS;
952 case Oddball::kNull:
953 if (deferred_string_key) SerializeDeferredKey(comma, key);
954 AppendCStringLiteral("null");
955 return SUCCESS;
956 default:
957 return UNCHANGED;
958 }
959 case JS_ARRAY_TYPE:
960 if (deferred_string_key) SerializeDeferredKey(comma, key);
961 return SerializeJSArray(Cast<JSArray>(object), key);
962 case JS_PRIMITIVE_WRAPPER_TYPE:
963 if (!need_stack_) {
964 need_stack_ = true;
965 return NEED_STACK;
966 }
967 if (deferred_string_key) SerializeDeferredKey(comma, key);
968 return SerializeJSPrimitiveWrapper(Cast<JSPrimitiveWrapper>(object), key);
969 case SYMBOL_TYPE:
970 return UNCHANGED;
971 case JS_RAW_JSON_TYPE:
972 if (deferred_string_key) SerializeDeferredKey(comma, key);
973 {
974 DirectHandle<JSRawJson> raw_json_obj = Cast<JSRawJson>(object);
975 Handle<String> raw_json;
976 if (raw_json_obj->HasInitialLayout(isolate_)) {
977 // Fast path: the object returned by JSON.rawJSON has its initial map
978 // intact.
979 raw_json = Cast<String>(handle(
980 raw_json_obj->InObjectPropertyAt(JSRawJson::kRawJsonInitialIndex),
981 isolate_));
982 } else {
983 // Slow path: perform a property get for "rawJSON". Because raw JSON
984 // objects are created frozen, it is still guaranteed that there will
985 // be a property named "rawJSON" that is a String. Their initial maps
986 // only change due to VM-internal operations like being optimized for
987 // being used as a prototype.
988 raw_json = Cast<String>(
989 JSObject::GetProperty(isolate_, raw_json_obj,
990 isolate_->factory()->raw_json_string())
991 .ToHandleChecked());
992 }
993 AppendString(raw_json);
994 }
995 return SUCCESS;
996 case HOLE_TYPE:
997 UNREACHABLE();
998#if V8_ENABLE_WEBASSEMBLY
999 case WASM_STRUCT_TYPE:
1000 case WASM_ARRAY_TYPE:
1001 return UNCHANGED;
1002#endif
1003 default:
1004 if (InstanceTypeChecker::IsString(instance_type)) {
1005 if (deferred_string_key) SerializeDeferredKey(comma, key);
1006 SerializeString<false>(Cast<String>(object));
1007 return SUCCESS;
1008 } else {
1009 // Make sure that we have a JSReceiver before we cast it to one.
1010 // If we ever leak an internal object that is not a JSReceiver it could
1011 // end up here and lead to a type confusion.
1012 CHECK(IsJSReceiver(*object));
1013 if (IsCallable(Cast<HeapObject>(*object), cage_base)) return UNCHANGED;
1014 // Go to slow path for global proxy and objects requiring access checks.
1015 if (deferred_string_key) SerializeDeferredKey(comma, key);
1016 if (InstanceTypeChecker::IsJSProxy(instance_type)) {
1017 return SerializeJSProxy(Cast<JSProxy>(object), key);
1018 }
1019 // WASM_{STRUCT,ARRAY}_TYPE are handled in `case:` blocks above.
1020 DCHECK(IsJSObject(*object));
1021 return SerializeJSObject(Cast<JSObject>(object), key);
1022 }
1023 }
1024
1025 UNREACHABLE();
1026}

1028JsonStringifier::Result JsonStringifier::SerializeJSPrimitiveWrapper(
1029 Handle<JSPrimitiveWrapper> object, Handle<Object> key) {
1030 Tagged<Object> raw = object->value();
1031 if (IsString(raw)) {
1032 Handle<Object> value;
1033 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
1034 isolate_, value, Object::ToString(isolate_, object), EXCEPTION);
1035 SerializeString<false>(Cast<String>(value));
1036 } else if (IsNumber(raw)) {

1038 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
1039 isolate_, value, Object::ToNumber(isolate_, object), EXCEPTION);

1048 AppendCStringLiteral("true");
1049 } else {
1050 AppendCStringLiteral("false");
1051 }
1052 } else {
1053 // ES6 24.3.2.1 step 10.c, serialize as an ordinary JSObject.
1054 return SerializeJSObject(object, key);
1055 }
1056 return SUCCESS;

1083JsonStringifier::Result JsonStringifier::SerializeJSArray(
1084 Handle<JSArray> object, Handle<Object> key) {
1085 uint32_t length = 0;
1086 CHECK(Object::ToArrayLength(object->length(), &length));
1087 DCHECK(!IsAccessCheckNeeded(*object));
1088 if (length == 0) {
1089 AppendCStringLiteral("[]");
1090 return SUCCESS;
1091 }
1092
1093 Result stack_push = StackPush(object, key);
1094 if (stack_push != SUCCESS) return stack_push;
1095
1096 AppendCharacter('[');
1097 Indent();
1098 uint32_t slow_path_index = 0;
1099 Result result = UNCHANGED;
1100 if (replacer_function_.is_null()) {
1101#define CASE_WITH_INTERRUPT(kind) \
1102 case kind: \
1103 result = SerializeFixedArrayWithInterruptCheck<kind>(object, length, \
1104 &slow_path_index); \
1105 break;
1106#define CASE_WITH_TRANSITION(kind) \
1107 case kind: \
1108 result = SerializeFixedArrayWithPossibleTransitions<kind>( \
1109 object, length, &slow_path_index); \
1110 break;
1111
1112 switch (object->GetElementsKind()) {
1113 CASE_WITH_INTERRUPT(PACKED_SMI_ELEMENTS)
1114 CASE_WITH_INTERRUPT(HOLEY_SMI_ELEMENTS)
1115 CASE_WITH_TRANSITION(PACKED_ELEMENTS)
1116 CASE_WITH_TRANSITION(HOLEY_ELEMENTS)
1117 CASE_WITH_INTERRUPT(PACKED_DOUBLE_ELEMENTS)
1118 CASE_WITH_INTERRUPT(HOLEY_DOUBLE_ELEMENTS)
1119 default:
1120 break;
1121 }
1122
1123#undef CASE_WITH_TRANSITION
1124#undef CASE_WITH_INTERRUPT
1125 }
1126 if (result == UNCHANGED) {
1127 // Slow path for non-fast elements and fall-back in edge cases.
1128 result = SerializeArrayLikeSlow(object, slow_path_index, length);
1129 }
1130 if (result != SUCCESS) return result;
1131 Unindent();
1132 NewLine();
1133 AppendCharacter(']');
1134 StackPop();
1135 return SUCCESS;
1136}

1138template <ElementsKind kind>
1139JsonStringifier::Result JsonStringifier::SerializeFixedArrayWithInterruptCheck(
1140 DirectHandle<JSArray> array, uint32_t length, uint32_t* slow_path_index) {
1141 static_assert(IsSmiElementsKind(kind) || IsDoubleElementsKind(kind));
1142 using ArrayT = std::conditional_t<IsDoubleElementsKind(kind),
1143 FixedDoubleArray, FixedArray>;
1144
1145 StackLimitCheck interrupt_check(isolate_);
1146 constexpr uint32_t kInterruptLength = 4000;
1147 uint32_t limit = std::min(length, kInterruptLength);
1148 constexpr uint32_t kMaxAllowedFastPackedLength =
1149 std::numeric_limits<uint32_t>::max() - kInterruptLength;
1150 static_assert(FixedArray::kMaxLength < kMaxAllowedFastPackedLength);
1151
1152 constexpr bool is_holey = IsHoleyElementsKind(kind);
1153 bool bailout_on_hole =
1154 is_holey ? !CanFastSerializeJSArray(isolate_, *array) : true;
1155
1156 uint32_t i = 0;
1157 while (true) {
1158 for (; i < limit; i++) {
1159 Result result = SerializeFixedArrayElement<kind>(
1160 Cast<ArrayT>(array->elements()), i, *array, bailout_on_hole);
1161 if constexpr (is_holey) {
1162 if (result != SUCCESS) {
1163 *slow_path_index = i;
1164 return result;
1165 }
1166 } else {
1167 USE(result);
1168 DCHECK_EQ(result, SUCCESS);
1169 }
1170 }
1171 if (i >= length) return SUCCESS;
1172 DCHECK_LT(limit, kMaxAllowedFastPackedLength);
1173 limit = std::min(length, limit + kInterruptLength);
1174 if (interrupt_check.InterruptRequested() &&
1175 IsException(isolate_->stack_guard()->HandleInterrupts(), isolate_)) {
1176 return EXCEPTION;
1177 }
1178 }
1179 return SUCCESS;
1180}

1182template <ElementsKind kind>
1183JsonStringifier::Result
1184JsonStringifier::SerializeFixedArrayWithPossibleTransitions(
1185 DirectHandle<JSArray> array, uint32_t length, uint32_t* slow_path_index) {
1186 static_assert(IsObjectElementsKind(kind));
1187
1188 HandleScope handle_scope(isolate_);
1189 DirectHandle<Object> old_length(array->length(), isolate_);
1190 constexpr bool is_holey = IsHoleyElementsKind(kind);
1191 bool should_check_treat_hole_as_undefined = true;
1192 for (uint32_t i = 0; i < length; i++) {
1193 if (array->length() != *old_length || kind != array->GetElementsKind()) {
1194 // Array was modified during SerializeElement.
1195 *slow_path_index = i;
1196 return UNCHANGED;
1197 }
1198 Tagged<Object> current_element =
1199 Cast<FixedArray>(array->elements())->get(i);
1200 if (is_holey && IsTheHole(current_element)) {
1201 if (should_check_treat_hole_as_undefined) {
1202 if (!CanFastSerializeJSArray(isolate_, *array)) {
1203 *slow_path_index = i;
1204 return UNCHANGED;
1205 }
1206 should_check_treat_hole_as_undefined = false;
1207 }
1208 Separator(i == 0);
1209 AppendCStringLiteral("null");
1210 } else {
1211 Separator(i == 0);
1212 Result result = SerializeElement(
1213 isolate_, handle(Cast<JSAny>(current_element), isolate_), i);
1214 if (result == UNCHANGED) {
1215 AppendCStringLiteral("null");
1216 } else if (result != SUCCESS) {
1217 return result;
1218 }
1219 if constexpr (is_holey) {
1220 should_check_treat_hole_as_undefined = true;
1221 }
1222 }
1223 }
1224 return SUCCESS;
1225}

1227template <ElementsKind kind, typename T>
1228JsonStringifier::Result JsonStringifier::SerializeFixedArrayElement(
1229 Tagged<T> elements, uint32_t i, Tagged<JSArray> array,
1230 bool bailout_on_hole) {
1231 if constexpr (IsHoleyElementsKind(kind)) {
1232 if (elements->is_the_hole(isolate_, i)) {
1233 if (bailout_on_hole) return UNCHANGED;
1234 Separator(i == 0);
1235 AppendCStringLiteral("null");
1236 return SUCCESS;
1237 }
1238 }
1239 DCHECK(!elements->is_the_hole(isolate_, i));
1240 Separator(i == 0);
1241 if constexpr (IsSmiElementsKind(kind)) {
1242 SerializeSmi(Cast<Smi>(elements->get(i)));
1243 } else if constexpr (IsDoubleElementsKind(kind)) {
1244 SerializeDouble(elements->get_scalar(i));
1245 } else {
1246 UNREACHABLE();
1247 }
1248 return SUCCESS;
1249}

1251JsonStringifier::Result JsonStringifier::SerializeArrayLikeSlow(
1252 DirectHandle<JSReceiver> object, uint32_t start, uint32_t length) {
1253 if (!need_stack_) {
1254 need_stack_ = true;
1255 return NEED_STACK;
1256 }
1257 // We need to write out at least two characters per array element.
1258 static const int kMaxSerializableArrayLength = String::kMaxLength / 2;
1259 if (length > kMaxSerializableArrayLength) {
1260 isolate_->Throw(*isolate_->factory()->NewInvalidStringLengthError());
1261 return EXCEPTION;
1262 }
1263 HandleScope handle_scope(isolate_);
1264 for (uint32_t i = start; i < length; i++) {
1265 Separator(i == 0);
1266 Handle<Object> element;
1267 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
1268 isolate_, element, JSReceiver::GetElement(isolate_, object, i),
1269 EXCEPTION);
1270 Result result = SerializeElement(isolate_, Cast<JSAny>(element), i);
1271 if (result == SUCCESS) continue;
1272 if (result == UNCHANGED) {
1273 // Detect overflow sooner for large sparse arrays.
1274 if (overflowed_) {
1275 isolate_->Throw(*isolate_->factory()->NewInvalidStringLengthError());
1276 return EXCEPTION;
1277 }
1278 AppendCStringLiteral("null");
1279 } else {
1280 return result;
1281 }
1282 }
1283 return SUCCESS;
1284}

1286JsonStringifier::Result JsonStringifier::SerializeJSObject(
1287 Handle<JSObject> object, Handle<Object> key) {
1288 PtrComprCageBase cage_base(isolate_);
1289 HandleScope handle_scope(isolate_);
1290
1291 if (!property_list_.is_null() ||
1292 !CanFastSerializeJSObject(*object, isolate_)) {
1293 if (!need_stack_) {
1294 need_stack_ = true;
1295 return NEED_STACK;
1296 }
1297 Result stack_push = StackPush(object, key);
1298 if (stack_push != SUCCESS) return stack_push;
1299 Result result = SerializeJSReceiverSlow(object);
1300 if (result != SUCCESS) return result;
1301 StackPop();
1302 return SUCCESS;
1303 }
1304
1305 DCHECK(!IsJSGlobalProxy(*object));
1306 DCHECK(!object->HasIndexedInterceptor());
1307 DCHECK(!object->HasNamedInterceptor());
1308
1309 DirectHandle<Map> map(object->map(cage_base), isolate_);
1310 if (map->NumberOfOwnDescriptors() == 0) {
1311 AppendCStringLiteral("{}");
1312 return SUCCESS;
1313 }
1314
1315 Result stack_push = StackPush(object, key);
1316 if (stack_push != SUCCESS) return stack_push;
1317 AppendCharacter('{');
1318 Indent();
1319 bool comma = false;
1320 for (InternalIndex i : map->IterateOwnDescriptors()) {
1321 Handle<String> key_name;
1322 PropertyDetails details = PropertyDetails::Empty();
1323 {
1324 DisallowGarbageCollection no_gc;
1325 Tagged<DescriptorArray> descriptors =
1326 map->instance_descriptors(cage_base);
1327 Tagged<Name> name = descriptors->GetKey(i);
1328 // TODO(rossberg): Should this throw?
1329 if (!IsString(name, cage_base)) continue;
1330 key_name = handle(Cast<String>(name), isolate_);
1331 details = descriptors->GetDetails(i);
1332 }
1333 if (details.IsDontEnum()) continue;
1334 Handle<JSAny> property;
1335 if (details.location() == PropertyLocation::kField &&
1336 *map == object->map(cage_base)) {
1337 DCHECK_EQ(PropertyKind::kData, details.kind());
1338 FieldIndex field_index = FieldIndex::ForDetails(*map, details);
1339 if (replacer_function_.is_null()) {
1340 // If there's no replacer function, read the raw property to avoid
1341 // reboxing doubles in mutable boxes.
1342 property = handle(object->RawFastPropertyAt(field_index), isolate_);
1343 } else {
1344 // Rebox the value if there is a replacer function since it could change
1345 // the value in the box.
1346 property = JSObject::FastPropertyAt(
1347 isolate_, object, details.representation(), field_index);
1348 }
1349 } else {
1350 if (!need_stack_) {
1351 need_stack_ = true;
1352 return NEED_STACK;
1353 }
1354 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
1355 isolate_, property,
1356 Cast<JSAny>(Object::GetPropertyOrElement(isolate_, object, key_name)),
1357 EXCEPTION);
1358 }
1359 Result result = SerializeProperty(property, comma, key_name);
1360 if (!comma && result == SUCCESS) comma = true;
1361 if (result == EXCEPTION || result == NEED_STACK) return result;
1362 }
1363 Unindent();
1364 if (comma) NewLine();
1365 AppendCharacter('}');
1366 StackPop();
1367 return SUCCESS;
1368}

1436template <typename SrcChar, typename DestChar, bool raw_json>
1437bool JsonStringifier::SerializeStringUnchecked_(
1438 base::Vector<const SrcChar> src, NoExtendBuilder<DestChar>* dest) {
1439 // Assert that base::uc16 character is not truncated down to 8 bit.
1440 // The <base::uc16, char> version of this method must not be called.
1441 DCHECK(sizeof(DestChar) >= sizeof(SrcChar));
1442 bool required_escaping = false;
1443 int prev_escaped_offset = -1;
1444 for (int i = 0; i < src.length(); i++) {
1445 SrcChar c = src[i];
1446 if (raw_json || DoNotEscape(c)) {
1447 continue;
1448 } else if (sizeof(SrcChar) != 1 &&
1449 base::IsInRange(c, static_cast<SrcChar>(0xD800),
1450 static_cast<SrcChar>(0xDFFF))) {
1451 // The current character is a surrogate.
1452 required_escaping = true;
1453 dest->AppendSubstring(src.data(), prev_escaped_offset + 1, i);
1454
1455 char double_to_radix_chars[kDoubleToRadixMaxChars];
1456 base::Vector<char> double_to_radix_buffer =
1457 base::ArrayVector(double_to_radix_chars);
1458 if (c <= 0xDBFF) {
1459 // The current character is a leading surrogate.
1460 if (i + 1 < src.length()) {
1461 // There is a next character.
1462 SrcChar next = src[i + 1];
1463 if (base::IsInRange(next, static_cast<SrcChar>(0xDC00),
1464 static_cast<SrcChar>(0xDFFF))) {
1465 // The next character is a trailing surrogate, meaning this is a
1466 // surrogate pair.
1467 dest->Append(c);
1468 dest->Append(next);
1469 i++;
1470 } else {
1471 // The next character is not a trailing surrogate. Thus, the
1472 // current character is a lone leading surrogate.
1473 dest->AppendCString("\\u");
1474 std::string_view hex =
1475 DoubleToRadixStringView(c, 16, double_to_radix_buffer);
1476 dest->AppendString(hex);
1477 }
1478 } else {
1479 // There is no next character. Thus, the current character is a lone
1480 // leading surrogate.
1481 dest->AppendCString("\\u");
1482 std::string_view hex =
1483 DoubleToRadixStringView(c, 16, double_to_radix_buffer);
1484 dest->AppendString(hex);
1485 }
1486 } else {
1487 // The current character is a lone trailing surrogate. (If it had been
1488 // preceded by a leading surrogate, we would've ended up in the other
1489 // branch earlier on, and the current character would've been handled
1490 // as part of the surrogate pair already.)
1491 dest->AppendCString("\\u");
1492 std::string_view hex =
1493 DoubleToRadixStringView(c, 16, double_to_radix_buffer);
1494 dest->AppendString(hex);
1495 }
1496 prev_escaped_offset = i;
1497 } else {
1498 required_escaping = true;
1499 dest->AppendSubstring(src.data(), prev_escaped_offset + 1, i);
1500 DCHECK_LT(c, 0x60);
1501 dest->AppendCString(&JsonEscapeTable[c * kJsonEscapeTableEntrySize]);
1502 prev_escaped_offset = i;
1503 }
1504 }
1505 dest->AppendSubstring(src.data(), prev_escaped_offset + 1, src.length());
1506 return required_escaping;
1507}

1594template <typename DestChar>
1595bool JsonStringifier::TrySerializeSimplePropertyKey(
1596 Tagged<String> key, const DisallowGarbageCollection& no_gc) {
1597 ReadOnlyRoots roots(isolate_);
1598 if (key->map() != roots.internalized_one_byte_string_map()) {
1599 return false;
1600 }
1601 if (!key_cache_.Contains(key)) {
1602 return false;
1603 }
1604 int length = key->length();
1605 int copy_length = length;
1606 if constexpr (sizeof(DestChar) == 1) {
1607 // CopyChars has fast paths for small integer lengths, and is generally a
1608 // little faster if we round the length up to the nearest 4. This is still
1609 // within the bounds of the object on the heap, because object alignment is
1610 // never less than 4 for any build configuration.
1611 constexpr int kRounding = 4;
1612 static_assert(kRounding <= kObjectAlignment);
1613 copy_length = RoundUp(length, kRounding);
1614 }
1615 // Add three for the quote marks and colon, to determine how much output space
1616 // is needed. We might actually require a little less output space than this,
1617 // depending on how much rounding happened above, but it's more important to
1618 // compute the requirement quickly than to be precise.
1619 int required_length = copy_length + 3;
1620 if (!CurrentPartCanFit(required_length)) {
1621 return false;
1622 }
1623 NoExtendBuilder<DestChar> no_extend(
1624 reinterpret_cast<DestChar*>(part_ptr_) + current_index_, &current_index_);
1625 no_extend.Append('"');
1626 base::Vector<const uint8_t> chars(
1627 Cast<SeqOneByteString>(key)->GetChars(no_gc), copy_length);
1628 DCHECK_LE(reinterpret_cast<Address>(chars.end()),
1629 key.address() + key->Size());
1630#if DEBUG
1631 for (int i = 0; i < length; ++i) {
1632 DCHECK(DoNotEscape(chars[i]));
1633 }
1634#endif // DEBUG
1635 no_extend.AppendChars(chars, length);
1636 no_extend.Append('"');
1637 no_extend.Append(':');
1638 return true;
1639}

1700void JsonStringifier::Extend() {
1701 if (part_length_ >= String::kMaxLength) {
1702 // Set the flag and carry on. Delay throwing the exception till the end.
1703 current_index_ = 0;
1704 overflowed_ = true;
1705 return;
1706 }
1707 part_length_ *= kPartLengthGrowthFactor;
1708 if (encoding_ == String::ONE_BYTE_ENCODING) {
1709 uint8_t* tmp_ptr = new uint8_t[part_length_];
1710 memcpy(tmp_ptr, one_byte_ptr_, current_index_);
1711 if (one_byte_ptr_ != one_byte_array_) delete[] one_byte_ptr_;
1712 one_byte_ptr_ = tmp_ptr;
1713 part_ptr_ = one_byte_ptr_;
1714 } else {
1715 base::uc16* tmp_ptr = new base::uc16[part_length_];
1716 for (uint32_t i = 0; i < current_index_; i++) {
1717 tmp_ptr[i] = two_byte_ptr_[i];
1718 }
1719 delete[] two_byte_ptr_;
1720 two_byte_ptr_ = tmp_ptr;
1721 part_ptr_ = two_byte_ptr_;
1722 }
1723}

1725void JsonStringifier::ChangeEncoding() {
1726 encoding_ = String::TWO_BYTE_ENCODING;
1727 two_byte_ptr_ = new base::uc16[part_length_];
1728 for (uint32_t i = 0; i < current_index_; i++) {
1729 two_byte_ptr_[i] = one_byte_ptr_[i];
1730 }
1731 part_ptr_ = two_byte_ptr_;
1732 if (one_byte_ptr_ != one_byte_array_) delete[] one_byte_ptr_;
1733 one_byte_ptr_ = nullptr;
1734}

1736template <typename Char>
1737class OutBuffer {
1738 public:
1739 explicit OutBuffer(Zone* zone) : zone_(zone), segments_(1, zone_) {
1740 Extend();
1741 }
1742 template <typename SrcChar>
1743 requires(sizeof(Char) >= sizeof(SrcChar))
1744 V8_INLINE void AppendCharacter(SrcChar c) {
1745 DCHECK_GT(SegmentFreeChars(), 0);
1746 *cur_++ = c;
1747 if (cur_ == segment_end_) {
1748 Extend();
1749 }
1750 }
1751 template <typename SrcChar>
1752 requires(sizeof(Char) >= sizeof(SrcChar))
1753 void Append(const SrcChar* chars, int length) {
1754 DCHECK_GT(SegmentFreeChars(), 0);
1755 if (length <= SegmentFreeChars()) {
1756 CopyChars(cur_, chars, length);
1757 cur_ += length;
1758 } else {
1759 int cur_length = SegmentFreeChars();
1760 CopyChars(cur_, chars, cur_length);
1761 chars += cur_length;
1762 int remaining_length = length - cur_length;
1763 while (remaining_length > 0) {
1764 Extend();
1765 int copy_length = std::min(remaining_length, kSegmentLength);
1766 CopyChars(cur_, chars, copy_length);
1767 chars += copy_length;
1768 cur_ += copy_length;
1769 remaining_length -= copy_length;
1770 }
1771 }
1772 if (SegmentFreeChars() <= 0) Extend();
1773 }
1774 int length() const {
1775 return (segments_.length() - 1) * kSegmentLength + CurSegmentLength();
1776 }
1777 template <typename Dst>
1778 void CopyTo(Dst* dst) {
1779 // Copy full segments.
1780 for (int i = 0; i < segments_.length() - 1; i++) {
1781 CopyChars(dst, segments_[i].begin(), kSegmentLength);
1782 dst += kSegmentLength;
1783 }
1784 // Copy last (partially filled) segment.
1785 base::Vector<Char> segment = segments_.last();
1786 CopyChars(dst, segment.begin(), CurSegmentLength());
1787 }
1788
1789 private:
1790 static constexpr int kSegmentLength = 2048;
1791 void Extend() {
1792 segments_.Add(zone_->AllocateVector<Char>(kSegmentLength), zone_);
1793 cur_ = segments_.last().begin();
1794 segment_end_ = segments_.last().end();
1795 }
1796 V8_INLINE int SegmentFreeChars() const {
1797 return static_cast<int>(segment_end_ - cur_);
1798 }
1799 V8_INLINE int CurSegmentLength() const {
1800 return static_cast<int>(cur_ - segments_.last().begin());
1801 }
1802 Zone* zone_;
1803 ZoneList<base::Vector<Char>> segments_;
1804 Char* cur_;
1805 Char* segment_end_;
1806};

1947V8_INLINE bool CanFastSerializeJSArrayFastPath(Tagged<JSArray> object,
1948 Isolate* isolate) {
1949 // Check if the prototype is the initial array prototype without interesting
1950 // properties (toJSON).
1951 Tagged<Map> map = object->map();
1952 if (map->may_have_interesting_properties()) return false;
1953 Tagged<NativeContext> native_context = map->map()->native_context();
1954 Tagged<HeapObject> proto = map->prototype();
1955 if (native_context->get(Context::INITIAL_ARRAY_PROTOTYPE_INDEX) != proto) {
1956 return false;
1957 }
1958 Tagged<Map> proto_map = proto->map();
1959 if (proto_map->may_have_interesting_properties()) {
1960 return false;
1961 }
1962 // Additionally check if the Array prototype is the initial object prototype
1963 // without interesting properties (toJSON).
1964 proto = proto_map->prototype();
1965 if (native_context->get(Context::INITIAL_OBJECT_PROTOTYPE_INDEX) != proto) {
1966 return false;
1967 }
1968 return !proto->map()->may_have_interesting_properties();
1969}

1972 Tagged<Map> map,

2468template <typename Char>
2469template <typename OldChar>
2470 requires(sizeof(OldChar) < sizeof(Char))
2471FastJsonStringifierResult FastJsonStringifier<Char>::ResumeFrom(
2472 FastJsonStringifier<OldChar>& old_stringifier,
2473 const DisallowGarbageCollection& no_gc) {
2474 DCHECK_EQ(ResultLength(), 0);
2475 DCHECK(stack_.empty());
2476 DCHECK(!old_stringifier.stack_.empty());
2477
2478 stack_ = old_stringifier.stack_;
2479 ContinuationRecord cont = stack_.back();
2480 stack_.pop_back();
2481 DCHECK(cont.type == ContinuationRecord::kResumeFromOther ||
2482 cont.type == ContinuationRecord::kObjectKey);
2483 Tagged<JSAny> object = cont.object;
2484 FastJsonStringifierResult result;
2485 DCHECK(IsString(object));
2486 if (cont.type == ContinuationRecord::kObjectKey) {
2487 // Serializing an object key caused an encoding change.
2488 result = SerializeObjectKey(Cast<String>(object), cont.index, no_gc);
2489 DCHECK_EQ(result, SUCCESS);
2490 // Resuming due to encoding change of an object key guarantees that there
2491 // are at least two other objects on the stack (the value for that key, and
2492 // the continuation record for the object the key is a member of).
2493 DCHECK_GE(stack_.size(), 2);
2494 cont = stack_.back();
2495 stack_.pop_back();
2496 // Check that we have the object's continuation record.
2497 DCHECK_EQ(stack_.back().type, ContinuationRecord::kObject);
2498 }
2499 if (cont.type == ContinuationRecord::kResumeFromOther) {
2500 // Mutliple scenarios lead here:
2501 // 1) The object to serialize was a string on the top-level, which triggered
2502 // an encoding change.
2503 // 2) A string value in an array or object triggered an encoding change.
2504 // 3) Fall-through from the object key case above.
2505 object = cont.object;
2506 result = TrySerializeSimpleObject<false>(object);
2507 DCHECK_EQ(result, SUCCESS);
2508
2509 if (stack_.empty()) return result;
2510
2511 cont = stack_.back();
2512 DCHECK(cont.type == ContinuationRecord::kObject ||
2513 cont.type == ContinuationRecord::kArray);
2514 stack_.pop_back();
2515 uint32_t obj_cont_idx =
2516 cont.type == ContinuationRecord::kObject ? cont.index : 0;
2517 uint32_t array_cont_idx =
2518 cont.type == ContinuationRecord::kArray ? cont.index : 0;
2519 return SerializeObject(cont.object, obj_cont_idx, array_cont_idx);
2520 } else {
2521 // Object can be JSArray or JSObject. Right now this is only possible after
2522 // encoding change of an object's key, in which case we push a continuation
2523 // of kObject even for JSArrays (as index is 0 anyways).
2524 DCHECK_EQ(cont.type, ContinuationRecord::kObject);
2525 DCHECK_EQ(cont.index, 0);
2526 return SerializeObject(cont.object, 0, 0);

2549template <typename Char>
2550FastJsonStringifierResult FastJsonStringifier<Char>::SerializeObject(
2551 Tagged<JSAny> object, uint32_t obj_cont_idx, uint32_t array_cont_idx) {
2552 DisallowGarbageCollection no_gc;
2553 // GCMole doesn't handle kNoGC interrupts correctly.
2554 DisableGCMole no_gc_mole;
2555
2556 uint32_t interrupt_budget = kGlobalInterruptBudget;
2557
2558 FastJsonStringifierResult result;
2559 while (true) {
2560 if (--interrupt_budget == 0) {
2561 result = HandleInterruptAndCheckCycle();
2562 if (result != SUCCESS) return result;
2563 interrupt_budget = kGlobalInterruptBudget;
2564 }
2565 Tagged<HeapObject> obj = Cast<HeapObject>(object);
2566 Tagged<Map> map = obj->map();
2567 InstanceType instance_type = map->instance_type();
2568 switch (instance_type) {
2569 case JS_OBJECT_TYPE: {
2570 result = SerializeJSObject(Cast<JSObject>(obj), obj_cont_idx, no_gc);
2571 break;
2572 }
2573 case JS_ARRAY_TYPE: {
2574 result = SerializeJSArray(Cast<JSArray>(obj), array_cont_idx);
2575 break;
2576 }
2577 default:
2578 return SLOW_PATH;
2579 }
2580 static_assert(SUCCESS == 0);
2581 static_assert(UNCHANGED == 1);
2582 if (V8_UNLIKELY(result > UNCHANGED)) return result;
2583 if (stack_.empty()) return SUCCESS;
2584 ContinuationRecord cont = stack_.back();
2585 object = cont.object;
2586 obj_cont_idx = cont.type == ContinuationRecord::kObject ? cont.index : 0;
2587 array_cont_idx = cont.type == ContinuationRecord::kArray ? cont.index : 0;
2588 stack_.pop_back();
2589 }

2643template <typename Char>
2644template <typename SrcChar>
2645FastJsonStringifierResult FastJsonStringifier<Char>::AppendStringChecked(
2646 Tagged<String> string, const SrcChar* chars, uint32_t length,
2647 const DisallowGarbageCollection& no_gc) {
2648 int escape_char_idx = -1;
2649 // TODO(pthier): Add SIMD version.
2650 for (uint32_t i = 0; i < length; i++) {
2651 SrcChar c = chars[i];
2652 if (V8_LIKELY(DoNotEscape(c))) continue;
2653 if constexpr (!is_one_byte) {
2654 if (sizeof(SrcChar) != 1 &&
2655 base::IsInRange(c, static_cast<SrcChar>(0xD800),
2656 static_cast<SrcChar>(0xDFFF))) {
2657 // The current character is a surrogate.
2658 buffer_.Append(chars + escape_char_idx + 1, i - (escape_char_idx + 1));
2659
2660 char double_to_radix_chars[kDoubleToRadixMaxChars];
2661 base::Vector<char> double_to_radix_buffer =
2662 base::ArrayVector(double_to_radix_chars);
2663 if (c <= 0xDBFF) {
2664 // The current character is a leading surrogate.
2665 if (i + 1 < length) {
2666 // There is a next character.
2667 SrcChar next = chars[i + 1];
2668 if (base::IsInRange(next, static_cast<SrcChar>(0xDC00),
2669 static_cast<SrcChar>(0xDFFF))) {
2670 // The next character is a trailing surrogate, meaning this is a
2671 // surrogate pair.
2672 AppendCharacter(c);
2673 AppendCharacter(next);
2674 i++;
2675 } else {
2676 // The next character is not a trailing surrogate. Thus, the
2677 // current character is a lone leading surrogate.
2678 AppendCStringLiteral("\\u");
2679 std::string_view hex =
2680 DoubleToRadixStringView(c, 16, double_to_radix_buffer);
2681 AppendString(hex);
2682 }
2683 } else {
2684 // There is no next character. Thus, the current character is a lone
2685 // leading surrogate.
2686 AppendCStringLiteral("\\u");
2687 std::string_view hex =
2688 DoubleToRadixStringView(c, 16, double_to_radix_buffer);
2689 AppendString(hex);
2690 }
2691 } else {
2692 // The current character is a lone trailing surrogate. (If it had been
2693 // preceded by a leading surrogate, we would've ended up in the other
2694 // branch earlier on, and the current character would've been handled
2695 // as part of the surrogate pair already.)
2696 AppendCStringLiteral("\\u");
2697 std::string_view hex =
2698 DoubleToRadixStringView(c, 16, double_to_radix_buffer);
2699 AppendString(hex);
2700 }
2701 escape_char_idx = i;
2702 } else {
2703 buffer_.Append(chars + escape_char_idx + 1, i - (escape_char_idx + 1));
2704 DCHECK_LT(c, 0x60);
2705 AppendCString(&JsonEscapeTable[c * kJsonEscapeTableEntrySize]);
2706 escape_char_idx = i;
2707 }
2708 } else {
2709 buffer_.Append(chars + escape_char_idx + 1, i - (escape_char_idx + 1));
2710 AppendCString(&JsonEscapeTable[c * kJsonEscapeTableEntrySize]);
2711 escape_char_idx = i;
2712 }
2713 }
2714 if (static_cast<uint32_t>(escape_char_idx + 1) < length) {
2715 buffer_.Append(chars + escape_char_idx + 1, length - (escape_char_idx + 1));
2716 }
2717 return SUCCESS;
2718}

2729MaybeHandle<Object> FastJsonStringify(Isolate* isolate, Handle<JSAny> object) {
2730 DisallowGarbageCollection no_gc;
2731
2732 FastJsonStringifier<uint8_t> one_byte_stringifier(isolate);
2733 std::optional<FastJsonStringifier<base::uc16>> two_byte_stringifier;
2734 FastJsonStringifierResult result =
2735 one_byte_stringifier.SerializeObject(*object, no_gc);
2736 bool result_is_one_byte = true;
2737
2738 if (result == CHANGE_ENCODING) {
2739 two_byte_stringifier.emplace(isolate);
2740 result = two_byte_stringifier->ResumeFrom(one_byte_stringifier, no_gc);
2741 DCHECK_NE(result, CHANGE_ENCODING);
2742 result_is_one_byte = false;
2743 }
2744
2745 if (V8_LIKELY(result == SUCCESS)) {
2746 if (result_is_one_byte) {
2747 const int length = one_byte_stringifier.ResultLength();
2748 Handle<SeqOneByteString> ret;
2749 {
2750 AllowGarbageCollection allow_gc;
2751 ASSIGN_RETURN_ON_EXCEPTION(
2752 isolate, ret, isolate->factory()->NewRawOneByteString(length));
2753 }
2754 one_byte_stringifier.CopyResultTo(ret->GetChars(no_gc));
2755 return ret;
2756 } else {
2757 DCHECK(two_byte_stringifier.has_value());
2758 const int one_byte_length = one_byte_stringifier.ResultLength();
2759 const int two_byte_length = two_byte_stringifier->ResultLength();
2760 const int total_length = one_byte_length + two_byte_length;
2761 Handle<SeqTwoByteString> ret;
2762 {
2763 AllowGarbageCollection allow_gc;
2764 ASSIGN_RETURN_ON_EXCEPTION(
2765 isolate, ret,
2766 isolate->factory()->NewRawTwoByteString(total_length));
2767 }
2768 base::uc16* chars = ret->GetChars(no_gc);
2769 if (one_byte_length > 0) {
2770 one_byte_stringifier.CopyResultTo(chars);
2771 }
2772 DCHECK_GT(two_byte_length, 0);
2773 two_byte_stringifier->CopyResultTo(chars + one_byte_length);
2774 return ret;
2775 }
2776 } else if (result == UNDEFINED) {
2777 return isolate->factory()->undefined_value();
2778 } else if (result == SLOW_PATH) {
2779 // TODO(pthier): Resume instead of restarting.
2780 AllowGarbageCollection allow_gc;
2781 JsonStringifier stringifier(isolate);
2782 Handle<JSAny> undefined = isolate->factory()->undefined_value();
2783 return stringifier.Stringify(object, undefined, undefined);
2784 }
2785 DCHECK(result == EXCEPTION);
2786 CHECK(isolate->has_exception());
2787 return MaybeHandle<Object>();
2788}
