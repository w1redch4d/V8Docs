25void FrameElider::MarkBlocks() {
26 for (InstructionBlock* block : instruction_blocks()) {
27 if (block->needs_frame()) continue;
28 for (int i = block->code_start(); i < block->code_end(); ++i) {
29 const Instruction* instr = InstructionAt(i);
30 if (instr->IsCall() || instr->IsDeoptimizeCall() ||
31 instr->arch_opcode() == ArchOpcode::kArchStackPointerGreaterThan ||
32 instr->arch_opcode() == ArchOpcode::kArchFramePointer) {
33 block->mark_needs_frame();
34 break;
35 }
36 if (instr->arch_opcode() == ArchOpcode::kArchStackSlot &&
37 ((instr->InputAt(0)->IsImmediate() &&
38 code_->GetImmediate(ImmediateOperand::cast(instr->InputAt(0)))
39 .ToInt32() > 0) ||
40 is_wasm_to_js_)) {
41 // We shouldn't allow accesses to the stack below the current stack
42 // pointer (indicated by positive slot indices).
43 // This is in particular because signal handlers (which could, of
44 // course, be triggered at any point in time) will overwrite this
45 // memory.
46 // Additionally wasm-to-JS code always requires a frame to address
47 // stack slots, because the stack pointer may switch to the central
48 // stack at the beginning of the code.
49 block->mark_needs_frame();
50 break;
51 }
52 }
53 }
54}

61void FrameElider::MarkDeConstruction() {
62 for (InstructionBlock* block : instruction_blocks()) {
63 if (block->needs_frame()) {
64 // Special case: The start block needs a frame.
65 if (block->predecessors().empty()) {
66 block->mark_must_construct_frame();
67 if (block->SuccessorCount() == 0) {
68 // We only have a single block, so the block also needs to be marked
69 // to deconstruct the frame.
70 const Instruction* last =
71 InstructionAt(block->last_instruction_index());
72 // The only cases when we need to deconstruct are ret and jump.
73 if (last->IsRet() || last->IsJump()) {
74 block->mark_must_deconstruct_frame();
75 }
76 }
77 }
78 // Find "frame -> no frame" transitions, inserting frame
79 // deconstructions.
80 for (RpoNumber& succ : block->successors()) {
81 if (!InstructionBlockAt(succ)->needs_frame()) {
82 DCHECK_EQ(1U, block->SuccessorCount());
83 const Instruction* last =
84 InstructionAt(block->last_instruction_index());
85 if (last->IsThrow() || last->IsTailCall() ||
86 last->IsDeoptimizeCall()) {
87 // We need to keep the frame if we exit the block through any
88 // of these.
89 continue;
90 }
91 // The only cases when we need to deconstruct are ret and jump.
92 DCHECK(last->IsRet() || last->IsJump());
93 block->mark_must_deconstruct_frame();
94 }
95 }
96 if (block->SuccessorCount() == 0) {
97 const Instruction* last =
98 InstructionAt(block->last_instruction_index());
99 // The only cases when we need to deconstruct are ret and jump.
100 if (last->IsRet() || last->IsJump()) {
101 block->mark_must_deconstruct_frame();
102 }
103 }
104 } else {
105 // Find "no frame -> frame" transitions, inserting frame constructions.
106 for (RpoNumber& succ : block->successors()) {
107 if (InstructionBlockAt(succ)->needs_frame()) {
108 DCHECK_NE(1U, block->SuccessorCount());
109 InstructionBlockAt(succ)->mark_must_construct_frame();
110 }
111 }
112 }
113 }
114}

132bool FrameElider::PropagateIntoBlock(InstructionBlock* block) {
133 // Already marked, nothing to do...
134 if (block->needs_frame()) return false;
135
136 // Turbofan does have an empty dummy end block, which we need to ignore here.
137 // However, Turboshaft does not have such a block.
138 if (has_dummy_end_block_) {
139 // Never mark the dummy end node, otherwise we might incorrectly decide to
140 // put frame deconstruction code there later,
141 if (block->successors().empty()) return false;
142 }
143
144 // Propagate towards the end ("downwards") if there is a predecessor needing
145 // a frame, but don't "bleed" from deferred code to non-deferred code.
146 for (RpoNumber& pred : block->predecessors()) {
147 if (InstructionBlockAt(pred)->needs_frame() &&
148 (!InstructionBlockAt(pred)->IsDeferred() || block->IsDeferred())) {
149 block->mark_needs_frame();
150 return true;
151 }
152 }
153
154 // Propagate towards start ("upwards")
155 bool need_frame_successors = false;
156 if (block->SuccessorCount() == 1) {
157 // For single successors, propagate the needs_frame information.
158 need_frame_successors =
159 InstructionBlockAt(block->successors()[0])->needs_frame();
160 } else {
161 // For multiple successors, each successor must only have a single
162 // predecessor (because the graph is in edge-split form), so each successor
163 // can independently create/dismantle a frame if needed. Given this
164 // independent control, only propagate needs_frame if all non-deferred
165 // blocks need a frame.
166 for (RpoNumber& succ : block->successors()) {
167 InstructionBlock* successor_block = InstructionBlockAt(succ);
168 DCHECK_EQ(1, successor_block->PredecessorCount());
169 if (!successor_block->IsDeferred()) {
170 if (successor_block->needs_frame()) {
171 need_frame_successors = true;
172 } else {
173 return false;
174 }
175 }
176 }
177 }
178 if (need_frame_successors) {
179 block->mark_needs_frame();
180 return true;
181 } else {
182 return false;
183 }
184}
