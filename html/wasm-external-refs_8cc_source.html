97void uint64_to_float32_wrapper(Address data) {
98 uint64_t input = ReadUnalignedValue<uint64_t>(data);
99#if defined(V8_OS_WIN)
100 // On Windows, the FP stack registers calculate with less precision, which
101 // leads to a uint64_t to float32 conversion which does not satisfy the
102 // WebAssembly specification. Therefore we do a different approach here:
103 //
104 // / leading 0 \/ 24 float data bits \/ for rounding \/ trailing 0 \
105 // 00000000000001XXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXXX100000000000000
106 //
107 // Float32 can only represent 24 data bit (1 implicit 1 bit + 23 mantissa
108 // bits). Starting from the most significant 1 bit, we can therefore extract
109 // 24 bits and do the conversion only on them. The other bits can affect the
110 // result only through rounding. Rounding works as follows:
111 // * If the most significant rounding bit is not set, then round down.
112 // * If the most significant rounding bit is set, and at least one of the
113 // other rounding bits is set, then round up.
114 // * If the most significant rounding bit is set, but all other rounding bits
115 // are not set, then round to even.
116 // We can aggregate 'all other rounding bits' in the second-most significant
117 // rounding bit.
118 // The resulting algorithm is therefore as follows:
119 // * Check if the distance between the most significant bit (MSB) and the
120 // least significant bit (LSB) is greater than 25 bits. If the distance is
121 // less or equal to 25 bits, the uint64 to float32 conversion is anyways
122 // exact, and we just use the C++ conversion.
123 // * Find the most significant bit (MSB).
124 // * Starting from the MSB, extract 25 bits (24 data bits + the first rounding
125 // bit).
126 // * The remaining rounding bits are guaranteed to contain at least one 1 bit,
127 // due to the check we did above.
128 // * Store the 25 bits + 1 aggregated bit in an uint32_t.
129 // * Convert this uint32_t to float. The conversion does the correct rounding
130 // now.
131 // * Shift the result back to the original magnitude.
132 uint32_t leading_zeros = base::bits::CountLeadingZeros(input);
133 uint32_t trailing_zeros = base::bits::CountTrailingZeros(input);
134 constexpr uint32_t num_extracted_bits = 25;
135 // Check if there are any rounding bits we have to aggregate.
136 if (leading_zeros + trailing_zeros + num_extracted_bits < 64) {
137 // Shift to extract the data bits.
138 uint32_t num_aggregation_bits = 64 - num_extracted_bits - leading_zeros;
139 // We extract the bits we want to convert. Note that we convert one bit more
140 // than necessary. This bit is a placeholder where we will store the
141 // aggregation bit.
142 int32_t extracted_bits =
143 static_cast<int32_t>(input >> (num_aggregation_bits - 1));
144 // Set the aggregation bit. We don't have to clear the slot first, because
145 // the bit there is also part of the aggregation.
146 extracted_bits |= 1;
147 float result = static_cast<float>(extracted_bits);
148 // We have to shift the result back. The shift amount is
149 // (num_aggregation_bits - 1), which is the shift amount we did originally,
150 // and (-2), which is for the two additional bits we kept originally for
151 // rounding.
152 int32_t shift_back = static_cast<int32_t>(num_aggregation_bits) - 1 - 2;
153 // Calculate the multiplier to shift the extracted bits back to the original
154 // magnitude. This multiplier is a power of two, so in the float32 bit
155 // representation we just have to construct the correct exponent and put it
156 // at the correct bit offset. The exponent consists of 8 bits, starting at
157 // the second MSB (a.k.a '<< 23'). The encoded exponent itself is
158 // ('actual exponent' - 127).
159 int32_t multiplier_bits = ((shift_back - 127) & 0xff) << 23;
160 result *= base::bit_cast<float>(multiplier_bits);
161 WriteUnalignedValue<float>(data, result);
162 return;
163 }
164#endif // defined(V8_OS_WIN)
165 WriteUnalignedValue<float>(data, static_cast<float>(input));
166}

173void uint64_to_float64_wrapper(Address data) {
174 uint64_t input = ReadUnalignedValue<uint64_t>(data);
175 double result = static_cast<double>(input);
176
177#if V8_CC_MSVC
178 // With MSVC we use static_cast<double>(uint32_t) instead of
179 // static_cast<double>(uint64_t) to achieve round-to-nearest-ties-even
180 // semantics. The idea is to calculate
181 // static_cast<double>(high_word) * 2^32 + static_cast<double>(low_word).
182 uint32_t low_word = static_cast<uint32_t>(input & 0xFFFFFFFF);
183 uint32_t high_word = static_cast<uint32_t>(input >> 32);
184
185 double shift = static_cast<double>(1ull << 32);
186
187 result = static_cast<double>(high_word);
188 result *= shift;
189 result += static_cast<double>(low_word);
190#endif
191
192 WriteUnalignedValue<double>(data, result);
193}

825void array_copy_wrapper(Address raw_dst_array, uint32_t dst_index,
826 Address raw_src_array, uint32_t src_index,
827 uint32_t length) {
828 DCHECK_GT(length, 0);
829 ThreadNotInWasmScope thread_not_in_wasm_scope;
830 DisallowGarbageCollection no_gc;
831 Tagged<WasmArray> dst_array = Cast<WasmArray>(Tagged<Object>(raw_dst_array));
832 Tagged<WasmArray> src_array = Cast<WasmArray>(Tagged<Object>(raw_src_array));
833
834 bool overlapping_ranges =
835 dst_array.ptr() == src_array.ptr() &&
836 (dst_index < src_index ? dst_index + length > src_index
837 : src_index + length > dst_index);
838 wasm::ValueType element_type = src_array->type()->element_type();
839 if (element_type.is_reference()) {
840 ObjectSlot dst_slot = dst_array->ElementSlot(dst_index);
841 ObjectSlot src_slot = src_array->ElementSlot(src_index);
842 Heap* heap = dst_array->GetIsolate()->heap();
843 if (overlapping_ranges) {
844 heap->MoveRange(dst_array, dst_slot, src_slot, length,
845 UPDATE_WRITE_BARRIER);
846 } else {
847 heap->CopyRange(dst_array, dst_slot, src_slot, length,
848 UPDATE_WRITE_BARRIER);
849 }
850 } else {
851 int element_size_bytes = element_type.value_kind_size();
852 void* dst = ArrayElementAddress(dst_array, dst_index, element_size_bytes);
853 void* src = ArrayElementAddress(src_array, src_index, element_size_bytes);
854 size_t copy_size = length * element_size_bytes;
855 if (overlapping_ranges) {
856 MemMove(dst, src, copy_size);
857 } else {
858 MemCopy(dst, src, copy_size);
859 }
860 }

863void array_fill_wrapper(Address raw_array, uint32_t index, uint32_t length,
864 uint32_t emit_write_barrier, uint32_t raw_type,
865 Address initial_value_addr) {
866 ThreadNotInWasmScope thread_not_in_wasm_scope;
867 DisallowGarbageCollection no_gc;
868 ValueType type = ValueType::FromRawBitField(raw_type);
869 int8_t* initial_element_address = reinterpret_cast<int8_t*>(
870 ArrayElementAddress(raw_array, index, type.value_kind_size()));
871 const int bytes_to_set = length * type.value_kind_size();
872
873 // We implement the general case by setting the first 8 bytes manually, then
874 // filling the rest by exponentially growing {memcpy}s.
875
876 CHECK_GE(static_cast<size_t>(bytes_to_set), sizeof(int64_t));
877
878 switch (type.kind()) {
879 case kI64:
880 case kF64: {
881 // Stack pointers are only aligned to 4 bytes.
882 int64_t initial_value =
883 base::ReadUnalignedValue<int64_t>(initial_value_addr);
884 if (initial_value == 0) {
885 std::memset(initial_element_address, 0, bytes_to_set);
886 return;
887 }
888 // Array elements are only aligned to 4 bytes, therefore
889 // `initial_element_address` may be misaligned as a 64-bit pointer.
890 base::WriteUnalignedValue<int64_t>(
891 reinterpret_cast<Address>(initial_element_address), initial_value);
892 break;
893 }
894 case kI32:
895 case kF32: {
896 int32_t initial_value = *reinterpret_cast<int32_t*>(initial_value_addr);
897 if (initial_value == 0) {
898 std::memset(initial_element_address, 0, bytes_to_set);
899 return;
900 }
901 int32_t* base = reinterpret_cast<int32_t*>(initial_element_address);
902 base[0] = base[1] = initial_value;
903 break;
904 }
905 case kF16:
906 case kI16: {
907 // The array.fill input is an i32!
908 int16_t initial_value = *reinterpret_cast<int32_t*>(initial_value_addr);
909 if (initial_value == 0) {
910 std::memset(initial_element_address, 0, bytes_to_set);
911 return;
912 }
913 int16_t* base = reinterpret_cast<int16_t*>(initial_element_address);
914 base[0] = base[1] = base[2] = base[3] = initial_value;
915 break;
916 }
917 case kI8: {
918 // The array.fill input is an i32!
919 int8_t initial_value = *reinterpret_cast<int32_t*>(initial_value_addr);
920 if (initial_value == 0) {
921 std::memset(initial_element_address, 0, bytes_to_set);
922 return;
923 }
924 int8_t* base = reinterpret_cast<int8_t*>(initial_element_address);
925 for (size_t i = 0; i < sizeof(int64_t); i++) {
926 base[i] = initial_value;
927 }
928 break;
929 }
930 case kRefNull:
931 case kRef: {
932 intptr_t uncompressed_pointer =
933 base::ReadUnalignedValue<intptr_t>(initial_value_addr);
934 if constexpr (kTaggedSize == 4) {
935 int32_t* base = reinterpret_cast<int32_t*>(initial_element_address);
936 base[0] = base[1] = static_cast<int32_t>(uncompressed_pointer);
937 } else {
938 base::WriteUnalignedValue(
939 reinterpret_cast<Address>(initial_element_address),
940 uncompressed_pointer);
941 }
942 break;
943 }
944 case kS128:
945 // S128 can only be filled with zeros.
946 DCHECK_EQ(base::ReadUnalignedValue<int64_t>(initial_value_addr), 0);
947 std::memset(initial_element_address, 0, bytes_to_set);
948 return;
949 case kRtt:
950 case kVoid:
951 case kTop:
952 case kBottom:
953 UNREACHABLE();
954 }
955
956 int bytes_already_set = sizeof(int64_t);
957
958 while (bytes_already_set * 2 <= bytes_to_set) {
959 std::memcpy(initial_element_address + bytes_already_set,
960 initial_element_address, bytes_already_set);
961 bytes_already_set *= 2;
962 }
963
964 if (bytes_already_set < bytes_to_set) {
965 std::memcpy(initial_element_address + bytes_already_set,
966 initial_element_address, bytes_to_set - bytes_already_set);
967 }
968
969 if (emit_write_barrier) {
970 DCHECK(type.is_reference());
971 Tagged<WasmArray> array = Cast<WasmArray>(Tagged<Object>(raw_array));
972 Isolate* isolate = array->GetIsolate();
973 ObjectSlot start(reinterpret_cast<Address>(initial_element_address));
974 ObjectSlot end(
975 reinterpret_cast<Address>(initial_element_address + bytes_to_set));
976 WriteBarrier::ForRange(isolate->heap(), array, start, end);
977 }
978}

1064Address grow_stack(Isolate* isolate, void* current_sp, size_t frame_size,
1065 size_t gap, Address current_fp) {
1066 // Check if this is a real stack overflow.
1067 StackLimitCheck check(isolate);
1068 if (check.WasmHasOverflowed(gap)) {
1069 Tagged<WasmContinuationObject> current_continuation =
1070 Cast<WasmContinuationObject>(
1071 isolate->root(RootIndex::kActiveContinuation));
1072 // If there is no parent, then the current stack is the main isolate stack.
1073 if (IsUndefined(current_continuation->parent())) {
1074 return 0;
1075 }
1076 auto stack =
1077 reinterpret_cast<wasm::StackMemory*>(current_continuation->stack());
1078 DCHECK(stack->IsActive());
1079 if (!stack->Grow(current_fp)) {
1080 return 0;
1081 }
1082
1083 Address new_sp = stack->base() - frame_size;
1084 // Here we assume stack values don't refer other moved stack slots.
1085 // A stack grow event happens right in the beginning of the function
1086 // call so moved slots contain only incoming params and frame header.
1087 // So, it is reasonable to assume no self references.
1088 std::memcpy(reinterpret_cast<void*>(new_sp), current_sp, frame_size);
1089
1090#if V8_TARGET_ARCH_ARM64
1091 Address new_fp =
1092 new_sp + (current_fp - reinterpret_cast<Address>(current_sp));
1093 Address old_pc_address = current_fp + CommonFrameConstants::kCallerPCOffset;
1094 Address new_pc_address = new_fp + CommonFrameConstants::kCallerPCOffset;
1095 Address old_signed_pc = base::Memory<Address>(old_pc_address);
1096 Address new_signed_pc = PointerAuthentication::MoveSignedPC(
1097 isolate, old_signed_pc, new_pc_address + kSystemPointerSize,
1098 old_pc_address + kSystemPointerSize);
1099 WriteUnalignedValue<Address>(new_pc_address, new_signed_pc);
1100#endif
1101
1102 isolate->stack_guard()->SetStackLimitForStackSwitching(
1103 reinterpret_cast<uintptr_t>(stack->jslimit()));
1104 return new_sp;
1105 }
1106
1107 return 0;
1108}
