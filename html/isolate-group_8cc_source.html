35#ifdef V8_COMPRESS_POINTERS
36struct PtrComprCageReservationParams
37 : public VirtualMemoryCage::ReservationParams {
38 PtrComprCageReservationParams() {
39 page_allocator = GetPlatformPageAllocator();
40
41 reservation_size = kPtrComprCageReservationSize;
42 base_alignment = kPtrComprCageBaseAlignment;
43
44 // Simplify BoundedPageAllocator's life by configuring it to use same page
45 // size as the Heap will use (MemoryChunk::kPageSize).
46 page_size =
47 RoundUp(size_t{1} << kPageSizeBits, page_allocator->AllocatePageSize());
48 requested_start_hint = RoundDown(
49 reinterpret_cast<Address>(page_allocator->GetRandomMmapAddr()),
50 base_alignment);
51
52#if V8_OS_FUCHSIA && !V8_EXTERNAL_CODE_SPACE
53 // If external code space is not enabled then executable pages (e.g. copied
54 // builtins, and JIT pages) will fall under the pointer compression range.
55 // Under Fuchsia that means the entire range must be allocated as JITtable.
56 permissions = PageAllocator::Permission::kNoAccessWillJitLater;
57#else
58 permissions = PageAllocator::Permission::kNoAccess;
59#endif
60 page_initialization_mode =
61 base::PageInitializationMode::kAllocatedPagesCanBeUninitialized;
62 page_freeing_mode = base::PageFreeingMode::kMakeInaccessible;
63 }
64};
65#endif // V8_COMPRESS_POINTERS

68IsolateGroup::~IsolateGroup() {
69 DCHECK_EQ(reference_count_.load(), 0);
70 DCHECK_EQ(isolate_count_.load(), 0);
71 // If pointer compression is enabled but the external code space is disabled,
72 // the pointer cage's page allocator is used for the CodeRange, whose
73 // destructor calls it via VirtualMemory::Free. Therefore we explicitly clear
74 // the code range here while we know the reservation still has a valid page
75 // allocator.

79#ifdef V8_ENABLE_SANDBOX
80void IsolateGroup::Initialize(bool process_wide, Sandbox* sandbox) {
81 DCHECK(!reservation_.IsReserved());
82 CHECK(sandbox->is_initialized());
83 process_wide_ = process_wide;
84 PtrComprCageReservationParams params;
85 Address base = sandbox->address_space()->AllocatePages(
86 sandbox->base(), params.reservation_size, params.base_alignment,
87 PagePermissions::kNoAccess);
88 CHECK_EQ(sandbox->base(), base);
89 base::AddressRegion existing_reservation(base, params.reservation_size);
90 params.page_allocator = sandbox->page_allocator();
91 if (!reservation_.InitReservation(params, existing_reservation)) {
92 V8::FatalProcessOutOfMemory(
93 nullptr,
94 "Failed to reserve virtual memory for process-wide V8 "
95 "pointer compression cage");
96 }
97 page_allocator_ = reservation_.page_allocator();
98 pointer_compression_cage_ = &reservation_;
99 trusted_pointer_compression_cage_ =
100 TrustedRange::EnsureProcessWideTrustedRange(kMaximalTrustedRangeSize);
101 sandbox_ = sandbox;
102
103 code_pointer_table()->Initialize();
104
105#ifdef V8_ENABLE_LEAPTIERING
106 js_dispatch_table()->Initialize();
107#endif // V8_ENABLE_LEAPTIERING
108}

137// static
138void IsolateGroup::InitializeOncePerProcess() {
139 static base::LeakyObject<IsolateGroup> default_isolate_group;
140 default_isolate_group_ = default_isolate_group.get();
141
142 IsolateGroup* group = GetDefault();
143
144 DCHECK_NULL(group->page_allocator_);
145#ifdef V8_ENABLE_SANDBOX
146 group->Initialize(true, Sandbox::GetDefault());
147#else
148 group->Initialize(true);
149#endif
150 CHECK_NOT_NULL(group->page_allocator_);
151
152#ifdef V8_COMPRESS_POINTERS
153 V8HeapCompressionScheme::InitBase(group->GetPtrComprCageBase());
154#endif // V8_COMPRESS_POINTERS
155#ifdef V8_EXTERNAL_CODE_SPACE
156 // Speculatively set the code cage base to the same value in case jitless
157 // mode will be used. Once the process-wide CodeRange instance is created
158 // the code cage base will be set accordingly.
159 ExternalCodeCompressionScheme::InitBase(V8HeapCompressionScheme::base());
160#endif // V8_EXTERNAL_CODE_SPACE
161#ifdef V8_COMPRESS_POINTERS_IN_MULTIPLE_CAGES
162 IsolateGroup::set_current(group);
163#endif
164}

166void IsolateGroup::Release() {
167 DCHECK_LT(0, reference_count_.load());
168#ifdef V8_ENABLE_SANDBOX
169 Sandbox* sandbox = sandbox_;
170#endif
171 if (--reference_count_ == 0) {
172 delete this;
173#ifdef V8_ENABLE_SANDBOX
174 sandbox->TearDown();
175#endif
176 }

180void InitCodeRangeOnce(std::unique_ptr<CodeRange>* code_range_member,
181 v8::PageAllocator* page_allocator, size_t requested_size,
182 bool immutable) {
183 CodeRange* code_range = new CodeRange();

222// static
223IsolateGroup* IsolateGroup::New() {
224 if (!CanCreateNewGroups()) {
225 FATAL(
226 "Creation of new isolate groups requires enabling "
227 "multiple pointer compression cages at build-time");
228 }
229
230 IsolateGroup* group = new IsolateGroup;
231#ifdef V8_ENABLE_SANDBOX
232 Sandbox* sandbox = Sandbox::New(GetPlatformVirtualAddressSpace());
233 group->Initialize(false, sandbox);
234#else
235 group->Initialize(false);
236#endif
237 CHECK_NOT_NULL(group->page_allocator_);
238 ExternalReferenceTable::InitializeOncePerIsolateGroup(
239 group->external_ref_table());
240 return group;
241}

243// static
244void IsolateGroup::ReleaseDefault() {
245 IsolateGroup* group = GetDefault();
246 CHECK_EQ(group->reference_count_.load(), 1);
247 CHECK(!group->has_shared_space_isolate());
248 group->page_allocator_ = nullptr;
249 group->code_range_.reset();
250 group->init_code_range_ = base::ONCE_STATE_UNINITIALIZED;
251#ifdef V8_COMPRESS_POINTERS
252 group->trusted_pointer_compression_cage_ = nullptr;
253 group->pointer_compression_cage_ = nullptr;
254 DCHECK(group->reservation_.IsReserved());
255 group->reservation_.Free();
256#endif // V8_COMPRESS_POINTERS
257
258#ifdef V8_ENABLE_LEAPTIERING
259 group->js_dispatch_table_.TearDown();
260#endif // V8_ENABLE_LEAPTIERING
261}
