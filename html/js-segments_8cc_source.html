28// ecma402 #sec-createsegmentsobject
29MaybeHandle<JSSegments> JSSegments::Create(Isolate* isolate,
30 DirectHandle<JSSegmenter> segmenter,
31 DirectHandle<String> string) {
32 std::shared_ptr<icu::BreakIterator> break_iterator{
33 segmenter->icu_break_iterator()->raw()->clone()};
34 DCHECK_NOT_NULL(break_iterator);
35
36 DirectHandle<Managed<icu::UnicodeString>> unicode_string =
37 Intl::SetTextToBreakIterator(isolate, string, break_iterator.get());
38 DirectHandle<Managed<icu::BreakIterator>> managed_break_iterator =
39 Managed<icu::BreakIterator>::From(isolate, 0, std::move(break_iterator));
40
41 // 1. Let internalSlotsList be « [[SegmentsSegmenter]], [[SegmentsString]] ».
42 // 2. Let segments be ! ObjectCreate(%Segments.prototype%, internalSlotsList).
43 DirectHandle<Map> map(isolate->native_context()->intl_segments_map(),
44 isolate);
45 Handle<JSObject> result = isolate->factory()->NewJSObjectFromMap(map);
46
47 Handle<JSSegments> segments = Cast<JSSegments>(result);
48 segments->set_flags(0);
49
50 // 3. Set segments.[[SegmentsSegmenter]] to segmenter.
51 segments->set_icu_break_iterator(*managed_break_iterator);
52 segments->set_granularity(segmenter->granularity());
53
54 // 4. Set segments.[[SegmentsString]] to string.
55 segments->set_raw_string(*string);
56 segments->set_unicode_string(*unicode_string);
57
58 // 5. Return segments.
59 return segments;
60}

62// ecma402 #sec-%segmentsprototype%.containing
63MaybeHandle<Object> JSSegments::Containing(Isolate* isolate,
64 DirectHandle<JSSegments> segments,
65 double n_double) {
66 // 5. Let len be the length of string.
67 int32_t len = segments->unicode_string()->raw()->length();
68
69 // 7. If n < 0 or n ≥ len, return undefined.
70 if (n_double < 0 || n_double >= len) {
71 return isolate->factory()->undefined_value();
72 }
73
74 int32_t n = static_cast<int32_t>(n_double);
75 // n may point to the surrogate tail- adjust it back to the lead.
76 n = segments->unicode_string()->raw()->getChar32Start(n);
77
78 icu::BreakIterator* break_iterator = segments->icu_break_iterator()->raw();
79 // 8. Let startIndex be ! FindBoundary(segmenter, string, n, before).
80 int32_t start_index =
81 break_iterator->isBoundary(n) ? n : break_iterator->preceding(n);
82
83 // 9. Let endIndex be ! FindBoundary(segmenter, string, n, after).
84 int32_t end_index = break_iterator->following(n);
85
86 // 10. Return ! CreateSegmentDataObject(segmenter, string, startIndex,
87 // endIndex).
88 return CreateSegmentDataObject(
89 isolate, segments->granularity(), break_iterator,
90 direct_handle(segments->raw_string(), isolate),
91 *(segments->unicode_string()->raw()), start_index, end_index);
92}

96bool CurrentSegmentIsWordLike(icu::BreakIterator* break_iterator) {
97 int32_t rule_status = break_iterator->getRuleStatus();
98 return (rule_status >= UBRK_WORD_NUMBER &&
99 rule_status < UBRK_WORD_NUMBER_LIMIT) ||
100 (rule_status >= UBRK_WORD_LETTER &&
101 rule_status < UBRK_WORD_LETTER_LIMIT) ||
102 (rule_status >= UBRK_WORD_KANA &&
103 rule_status < UBRK_WORD_KANA_LIMIT) ||
104 (rule_status >= UBRK_WORD_IDEO && rule_status < UBRK_WORD_IDEO_LIMIT);
105}

109// ecma402 #sec-createsegmentdataobject
110MaybeHandle<JSSegmentDataObject> JSSegments::CreateSegmentDataObject(
111 Isolate* isolate, JSSegmenter::Granularity granularity,
112 icu::BreakIterator* break_iterator, DirectHandle<String> input_string,
113 const icu::UnicodeString& unicode_string, int32_t start_index,
114 int32_t end_index) {
115 Factory* factory = isolate->factory();
116
117 // 1. Let len be the length of string.
118 // 2. Assert: startIndex ≥ 0.
119 DCHECK_GE(start_index, 0);
120 // 3. Assert: endIndex ≤ len.
121 DCHECK_LE(end_index, unicode_string.length());
122 // 4. Assert: startIndex < endIndex.
123 DCHECK_LT(start_index, end_index);
124
125 // 5. Let result be ! ObjectCreate(%ObjectPrototype%).
126 DirectHandle<Map> map(
127 granularity == JSSegmenter::Granularity::WORD
128 ? isolate->native_context()->intl_segment_data_object_wordlike_map()
129 : isolate->native_context()->intl_segment_data_object_map(),
130 isolate);
131 Handle<JSSegmentDataObject> result =
132 Cast<JSSegmentDataObject>(factory->NewJSObjectFromMap(map));
133
134 // 6. Let segment be the String value equal to the substring of string
135 // consisting of the code units at indices startIndex (inclusive) through
136 // endIndex (exclusive).
137 DirectHandle<String> segment;
138 ASSIGN_RETURN_ON_EXCEPTION(
139 isolate, segment,
140 Intl::ToString(isolate, unicode_string, start_index, end_index));
141 DirectHandle<Number> index = factory->NewNumberFromInt(start_index);
142
143 // 7. Perform ! CreateDataPropertyOrThrow(result, "segment", segment).
144 DisallowGarbageCollection no_gc;
145 Tagged<JSSegmentDataObject> raw = Cast<JSSegmentDataObject>(*result);
146 raw->set_segment(*segment);
147 // 8. Perform ! CreateDataPropertyOrThrow(result, "index", startIndex).
148 raw->set_index(*index);
149 // 9. Perform ! CreateDataPropertyOrThrow(result, "input", string).
150 raw->set_input(*input_string);
151
152 // 10. Let granularity be segmenter.[[SegmenterGranularity]].
153 // 11. If granularity is "word", then
154 if (granularity == JSSegmenter::Granularity::WORD) {
155 // a. Let isWordLike be a Boolean value indicating whether the segment in
156 // string is "word-like" according to locale segmenter.[[Locale]].
157 DirectHandle<Boolean> is_word_like =
158 factory->ToBoolean(CurrentSegmentIsWordLike(break_iterator));
159 // b. Perform ! CreateDataPropertyOrThrow(result, "isWordLike", isWordLike).
160 Cast<JSSegmentDataObjectWithIsWordLike>(raw)->set_is_word_like(
161 *is_word_like);
162 }
163 return result;
164}
