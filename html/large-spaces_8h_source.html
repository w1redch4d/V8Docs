33class V8_EXPORT_PRIVATE LargeObjectSpace : public Space {
34 public:
35 using iterator = LargePageIterator;
36 using const_iterator = ConstLargePageIterator;
37
38 ~LargeObjectSpace() override { TearDown(); }

45

54 void ShrinkPageToObjectSize(LargePageMetadata* page,

57 // Checks whether a heap object is in this space; O(1).
58 bool Contains(Tagged<HeapObject> obj) const;
59 // Checks whether an address is in the object area in this space. Iterates all
60 // objects in the space. May be slow.
61 bool ContainsSlow(Address addr) const;
62
63 // Checks whether the space is empty.
64 bool IsEmpty() const { return first_page() == nullptr; }
65
66 virtual void AddPage(LargePageMetadata* page, size_t object_size);
67 virtual void RemovePage(LargePageMetadata* page);
68
69 LargePageMetadata* first_page() override {
70 return reinterpret_cast<LargePageMetadata*>(memory_chunk_list_.front());

77 iterator begin() { return iterator(first_page()); }

85 void AddAllocationObserver(AllocationObserver* observer);

91
92#ifdef DEBUG
93 void Print() override;
94#endif

96 // The last allocated object that is not guaranteed to be initialized when the
97 // concurrent marker visits it.
98 Address pending_object() const {
99 return pending_object_.load(std::memory_order_acquire);
100 }
101
102 void ResetPendingObject() {
103 pending_object_.store(0, std::memory_order_release);
104 }
105
106 base::SharedMutex* pending_allocation_mutex() {
107 return &pending_allocation_mutex_;
108 }
109
110 void set_objects_size(size_t objects_size) { objects_size_ = objects_size; }
111
112 protected:
113 LargeObjectSpace(Heap* heap, AllocationSpace id);
114
115 void AdvanceAndInvokeAllocationObservers(Address soon_object, size_t size);
116
117 LargePageMetadata* AllocateLargePage(int object_size,
118 Executability executable);
119
120 void UpdatePendingObject(Tagged<HeapObject> object);
121
122 std::atomic<size_t> size_; // allocated bytes
123 int page_count_; // number of chunks
124 std::atomic<size_t> objects_size_; // size of objects
125 // The mutex has to be recursive because profiler tick might happen while
126 // holding this lock, then the profiler will try to iterate the call stack
127 // which might end up calling CodeLargeObjectSpace::FindPage() and thus
128 // trying to lock the mutex for a second time.
129 base::RecursiveMutex allocation_mutex_;
130
131 // Current potentially uninitialized object. Protected by
132 // pending_allocation_mutex_.
133 std::atomic<Address> pending_object_;
134
135 // Used to protect pending_object_.
136 base::SharedMutex pending_allocation_mutex_;
137
138 AllocationCounter allocation_counter_;
139
140 private:
141 friend class LargeObjectSpaceObjectIterator;
142};
