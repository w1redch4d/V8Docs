461void Map::AccountAddedPropertyField() {
462 // Update used instance size and unused property fields number.
463 static_assert(JSObject::kFieldsAdded == JSObject::kHeaderSize / kTaggedSize);
464#ifdef DEBUG
465 int new_unused = UnusedPropertyFields() - 1;
466 if (new_unused < 0) new_unused += JSObject::kFieldsAdded;
467#endif
468 int value = used_or_unused_instance_size_in_words();
469 if (value >= JSObject::kFieldsAdded) {
470 if (value == instance_size_in_words()) {
471 AccountAddedOutOfObjectPropertyField(0);
472 } else {
473 // The property is added in-object, so simply increment the counter.
474 set_used_or_unused_instance_size_in_words(value + 1);
475 }
476 } else {
477 AccountAddedOutOfObjectPropertyField(value);
478 }
479 DCHECK_EQ(new_unused, UnusedPropertyFields());
480}

755 Tagged<DescriptorArray> descriptors = instance_descriptors(isolate);
756 int number_of_own_descriptors = NumberOfOwnDescriptors();

763#ifndef V8_DISABLE_WRITE_BARRIERS

774 DCHECK_GT(UnusedPropertyFields(), 0);

779// it should never try to (otherwise, layout descriptor must be updated too).
780#ifdef DEBUG

782 !details.representation().IsDouble());
783#endif
