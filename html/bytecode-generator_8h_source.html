111 // An assignment has to evaluate its LHS before its RHS, but has to assign to
112 // the LHS after both evaluations are done. This class stores the data
113 // computed in the LHS evaulation that has to live across the RHS evaluation,
114 // and is used in the actual LHS assignment.
115 class AssignmentLhsData {
116 public:
117 static AssignmentLhsData NonProperty(Expression* expr);
118 static AssignmentLhsData NamedProperty(Expression* object_expr,
119 Register object,
120 const AstRawString* name);
121 static AssignmentLhsData KeyedProperty(Register object, Register key);
122 static AssignmentLhsData PrivateMethodOrAccessor(AssignType type,
123 Property* property,
124 Register object,
125 Register key);
126 static AssignmentLhsData PrivateDebugEvaluate(AssignType type,
127 Property* property,
128 Register object);
129 static AssignmentLhsData NamedSuperProperty(
130 RegisterList super_property_args);
131 static AssignmentLhsData KeyedSuperProperty(
132 RegisterList super_property_args);
133
134 AssignType assign_type() const { return assign_type_; }
135 bool is_private_assign_type() const {
136 return assign_type_ == PRIVATE_METHOD ||
137 assign_type_ == PRIVATE_GETTER_ONLY ||
138 assign_type_ == PRIVATE_SETTER_ONLY ||
139 assign_type_ == PRIVATE_GETTER_AND_SETTER ||
140 assign_type_ == PRIVATE_DEBUG_DYNAMIC;
141 }
142 Expression* expr() const {
143 DCHECK(assign_type_ == NON_PROPERTY || is_private_assign_type());
144 return expr_;
145 }
146 Expression* object_expr() const {
147 DCHECK_EQ(assign_type_, NAMED_PROPERTY);
148 return object_expr_;
149 }
150 Register object() const {
151 DCHECK(assign_type_ == NAMED_PROPERTY || assign_type_ == KEYED_PROPERTY ||
152 is_private_assign_type());
153 return object_;
154 }
155 Register key() const {
156 DCHECK((assign_type_ == KEYED_PROPERTY || is_private_assign_type()) &&
157 assign_type_ != PRIVATE_DEBUG_DYNAMIC);
158 return key_;
159 }
160 const AstRawString* name() const {
161 DCHECK(assign_type_ == NAMED_PROPERTY);
162 return name_;
163 }
164 RegisterList super_property_args() const {
165 DCHECK(assign_type_ == NAMED_SUPER_PROPERTY ||
166 assign_type_ == KEYED_SUPER_PROPERTY);
167 return super_property_args_;
168 }
169
170 private:
171 AssignmentLhsData(AssignType assign_type, Expression* expr,
172 RegisterList super_property_args, Register object,
173 Register key, Expression* object_expr,
174 const AstRawString* name)
175 : assign_type_(assign_type),
176 expr_(expr),
177 super_property_args_(super_property_args),
178 object_(object),
179 key_(key),
180 object_expr_(object_expr),
181 name_(name) {}
182
183 AssignType assign_type_;
184
185 // Different assignment types use different fields:
186 //
187 // NON_PROPERTY: expr
188 // NAMED_PROPERTY: object_expr, object, name
189 // KEYED_PROPERTY, PRIVATE_METHOD: object, key
190 // NAMED_SUPER_PROPERTY: super_property_args
191 // KEYED_SUPER_PROPERT: super_property_args
192 Expression* expr_;
193 RegisterList super_property_args_;
194 Register object_;
195 Register key_;
196 Expression* object_expr_;
197 const AstRawString* name_;
198 };
