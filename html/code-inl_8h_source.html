69bool GcSafeCode::CanDeoptAt(Isolate* isolate, Address pc) const {
70 if (!UnsafeCastToCode()->uses_deoptimization_data()) return false;
71 Tagged<DeoptimizationData> deopt_data = UncheckedCast<DeoptimizationData>(
72 UnsafeCastToCode()->unchecked_deoptimization_data());
73 Address code_start_address = instruction_start();
74 for (int i = 0; i < deopt_data->DeoptCount(); i++) {
75 if (deopt_data->Pc(i).value() == -1) continue;
76 Address address = code_start_address + deopt_data->Pc(i).value();
77 if (address == pc && deopt_data->GetBytecodeOffsetOrBuiltinContinuationId(
78 i) != BytecodeOffset::None()) {
79 return true;
80 }
81 }
82 return false;
83}

345uintptr_t Code::GetBaselinePCForBytecodeOffset(
346 int bytecode_offset, BytecodeToPCPosition position,
347 Tagged<BytecodeArray> bytecodes) {
348 DisallowGarbageCollection no_gc;
349 CHECK_EQ(kind(), CodeKind::BASELINE);
350 // The following check ties together the bytecode being executed in
351 // Generate_BaselineOrInterpreterEntry with the bytecode that was used to
352 // compile this baseline code. Together, this ensures that we don't OSR into a
353 // wrong code object.
354 auto maybe_bytecodes = bytecode_or_interpreter_data();
355 if (IsBytecodeArray(maybe_bytecodes)) {
356 SBXCHECK_EQ(maybe_bytecodes, bytecodes);
357 } else {
358 CHECK(IsInterpreterData(maybe_bytecodes));
359 SBXCHECK_EQ(Cast<InterpreterData>(maybe_bytecodes)->bytecode_array(),
360 bytecodes);
361 }
362 baseline::BytecodeOffsetIterator offset_iterator(
363 Cast<TrustedByteArray>(bytecode_offset_table()), bytecodes);
364 offset_iterator.AdvanceToBytecodeOffset(bytecode_offset);
365 uintptr_t pc = 0;
366 if (position == kPcAtStartOfBytecode) {
367 pc = offset_iterator.current_pc_start_offset();
368 } else {
369 DCHECK_EQ(position, kPcAtEndOfBytecode);
370 pc = offset_iterator.current_pc_end_offset();
371 }
372 return pc;
373}

388 int bytecode_offset, Tagged<BytecodeArray> bytecodes) {
389 DisallowGarbageCollection no_gc;
390 CHECK_EQ(kind(), CodeKind::BASELINE);
391 baseline::BytecodeOffsetIterator offset_iterator(
392 Cast<TrustedByteArray>(bytecode_offset_table()), bytecodes);
393 Handle<BytecodeArray> bytecodes_handle(
394 reinterpret_cast<Address*>(&bytecodes));
395 interpreter::BytecodeArrayIterator bytecode_iterator(bytecodes_handle,
396 bytecode_offset);
397 interpreter::Bytecode bytecode = bytecode_iterator.current_bytecode();
398 if (bytecode == interpreter::Bytecode::kJumpLoop) {

400 bytecode_iterator.GetJumpTargetOffset(), bytecodes);

402 DCHECK(!interpreter::Bytecodes::IsJump(bytecode));
403 DCHECK(!interpreter::Bytecodes::IsSwitch(bytecode));
404 DCHECK(!interpreter::Bytecodes::Returns(bytecode));
405 return GetBaselineEndPCForBytecodeOffset(bytecode_offset, bytecodes);

511inline void Code::SetMarkedForDeoptimization(Isolate* isolate,
512 LazyDeoptimizeReason reason) {
513 set_marked_for_deoptimization(true);
514 // Eager deopts are already logged by the deoptimizer.
515 if (reason != LazyDeoptimizeReason::kEagerDeopt &&
516 V8_UNLIKELY(v8_flags.trace_deopt || v8_flags.log_deopt)) {
517 TraceMarkForDeoptimization(isolate, reason);
518 }
519#ifdef V8_ENABLE_LEAPTIERING
520 JSDispatchHandle handle = js_dispatch_handle();
521 if (handle != kNullJSDispatchHandle) {
522 JSDispatchTable* jdt = IsolateGroup::current()->js_dispatch_table();
523 Tagged<Code> cur = jdt->GetCode(handle);
524 if (SafeEquals(cur)) {
525 if (v8_flags.reopt_after_lazy_deopts &&
526 isolate->concurrent_recompilation_enabled()) {
527 jdt->SetCodeNoWriteBarrier(
528 handle, *BUILTIN_CODE(isolate, InterpreterEntryTrampoline));
529 // Somewhat arbitrary list of lazy deopt reasons which we expect to be
530 // stable enough to warrant either immediate re-optimization, or
531 // re-optimization after one invocation (to detect potential follow-up
532 // IC changes).
533 // TODO(olivf): We should also work on reducing the number of
534 // dependencies we create in the compilers to require less of these
535 // quick re-compilations.
536 switch (reason) {
537 case LazyDeoptimizeReason::kAllocationSiteTenuringChange:
538 case LazyDeoptimizeReason::kAllocationSiteTransitionChange:
539 case LazyDeoptimizeReason::kEmptyContextExtensionChange:
540 case LazyDeoptimizeReason::kExceptionCaught:
541 case LazyDeoptimizeReason::kFieldTypeConstChange:
542 case LazyDeoptimizeReason::kFrameValueMaterialized:
543 case LazyDeoptimizeReason::kPropertyCellChange:
544 case LazyDeoptimizeReason::kPrototypeChange:
545 case LazyDeoptimizeReason::kScriptContextSlotPropertyChange:
546 jdt->SetTieringRequest(
547 handle,
548 v8_flags.maglev ? TieringBuiltin::kStartMaglevOptimizeJob
549 : TieringBuiltin::kStartTurbofanOptimizeJob,
550 isolate);
551 break;
552 case LazyDeoptimizeReason::kFieldRepresentationChange:
553 case LazyDeoptimizeReason::kFieldTypeChange:
554 case LazyDeoptimizeReason::kInitialMapChange:
555 case LazyDeoptimizeReason::kMapDeprecated:
556 jdt->SetTieringRequest(
557 handle, TieringBuiltin::kMarkReoptimizeLazyDeoptimized,
558 isolate);
559 break;
560 default:
561 // TODO(olivf): This trampoline is just used to reset the budget. If
562 // we knew the feedback cell and the bytecode size here, we could
563 // directly reset the budget.
564 jdt->SetTieringRequest(handle, TieringBuiltin::kMarkLazyDeoptimized,
565 isolate);
566 break;
567 }
568 } else {
569 jdt->SetCodeNoWriteBarrier(handle, *BUILTIN_CODE(isolate, CompileLazy));
570 }
571 }
572 // Ensure we don't try to patch the entry multiple times.
573 set_js_dispatch_handle(kNullJSDispatchHandle);
574 }
575#endif
