112inline bool IsCompileTimeImport(WellKnownImport wki) {
113 using T = std::underlying_type_t<WellKnownImport>;
114 T num = static_cast<T>(wki);
115 constexpr T kFirst = static_cast<T>(WellKnownImport::kFirstCompileTimeImport);
116 constexpr T kLast = static_cast<T>(WellKnownImport::kLastCompileTimeImport);
117 return kFirst <= num && num <= kLast;
118}

126 // Regular initialization. Allocates size-dependent internal data.
127 void Initialize(int size) {
128#if DEBUG
129 DCHECK_EQ(-1, size_);
130 size_ = size;
131#endif
132 static_assert(static_cast<int>(WellKnownImport::kUninstantiated) == 0);
133 statuses_ = std::make_unique<std::atomic<WellKnownImport>[]>(size);
134#if !defined(__cpp_lib_atomic_value_initialization) || \
135 __cpp_lib_atomic_value_initialization < 201911L
136 for (int i = 0; i < size; i++) {
137 std::atomic_init(&statuses_.get()[i], WellKnownImport::kUninstantiated);
138 }
139#endif
140 }
141
142 // Intended for deserialization. Does not check consistency with code.
143 void Initialize(base::Vector<const WellKnownImport> entries);
144
145 WellKnownImport get(int index) const {
146 DCHECK_LT(index, size_);
147 return statuses_[index].load(std::memory_order_relaxed);
148 }
149
150 // Note: you probably want to be holding the associated NativeModule's
151 // {allocation_lock_} when calling this method.
152 V8_WARN_UNUSED_RESULT UpdateResult
153 Update(base::Vector<WellKnownImport> entries);
154
155 private:
156 // Operations that need to ensure that they see a consistent view of
157 // {statuses_} for some period of time should use the associated
158 // NativeModule's {allocation_lock_} for that purpose (which they will
159 // likely need anyway, due to WellKnownImport statuses and published
160 // code objects needing to stay in sync).
161 std::unique_ptr<std::atomic<WellKnownImport>[]> statuses_;
162
163#if DEBUG
164 int size_{-1};
165#endif
166};
