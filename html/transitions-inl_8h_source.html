66 SideStepTransition::Kind kind) {

223bool TransitionArray::GetTargetIfExists(int transition_number, Isolate* isolate,
224 Tagged<Map>* target) {
225 Tagged<MaybeObject> raw = GetRawTarget(transition_number);
226 Tagged<HeapObject> heap_object;
227 // If the raw target is a Smi, then this TransitionArray is in the process of
228 // being deserialized, and doesn't yet have an initialized entry for this
229 // transition.
230 if (raw.IsSmi()) {
231 DCHECK(isolate->has_active_deserializer());
232 DCHECK_EQ(raw.ToSmi(), Smi::uninitialized_deserialization_value());
233 return false;
234 }
235 if (raw.GetHeapObjectIfStrong(&heap_object) &&
236 IsUndefined(heap_object, isolate)) {
237 return false;
238 }
239 *target = TransitionsAccessor::GetTargetFromRaw(raw);
240 return true;
241}

259int TransitionArray::SearchName(Tagged<Name> name, bool concurrent_search,
260 int* out_insertion_index) {
261 DCHECK(IsUniqueName(name));
262 SLOW_DCHECK_IMPLIES(!concurrent_search, IsSortedNoDuplicates());
263
264 if (number_of_transitions() == 0) {
265 if (out_insertion_index != nullptr) {
266 *out_insertion_index = 0;
267 }
268 return kNotFound;
269 }
270
271 // Do linear search for small arrays, and for searches in the background
272 // thread.
273 const int kMaxElementsForLinearSearch = 8;
274 if (number_of_transitions() <= kMaxElementsForLinearSearch ||
275 concurrent_search) {
276 return LinearSearchName(name, out_insertion_index);
277 }
278
279 return BinarySearchName(name, out_insertion_index);
280}

282int TransitionArray::BinarySearchName(Tagged<Name> name,
283 int* out_insertion_index) {
284 int end = number_of_transitions();
285 uint32_t hash = name->hash();
286
287 // Find the first index whose key's hash is greater-than-or-equal-to the
288 // search hash.
289 int i = *std::ranges::lower_bound(std::views::iota(0, end), hash,
290 std::less<>(), [&](int i) {
291 Tagged<Name> entry = GetKey(i);
292 return entry->hash();
293 });
294
295 // There may have been hash collisions, so search for the name from the first
296 // index until the first non-matching hash.
297 for (; i < end; ++i) {
298 Tagged<Name> entry = GetKey(i);
299 if (entry == name) {
300 return i;
301 }
302 uint32_t entry_hash = entry->hash();
303 if (entry_hash != hash) {
304 if (out_insertion_index != nullptr) {
305 *out_insertion_index = i + (entry_hash > hash ? 0 : 1);
306 }
307 return kNotFound;
308 }
309 }
310
311 if (out_insertion_index != nullptr) {
312 *out_insertion_index = end;
313 }
314 return kNotFound;
315}
316
317int TransitionArray::LinearSearchName(Tagged<Name> name,
318 int* out_insertion_index) {
319 int len = number_of_transitions();
320 if (out_insertion_index != nullptr) {
321 uint32_t hash = name->hash();
322 for (int i = 0; i < len; i++) {
323 Tagged<Name> entry = GetKey(i);
324 if (entry == name) return i;
325 if (entry->hash() > hash) {
326 *out_insertion_index = i;
327 return kNotFound;
328 }
329 }
330 *out_insertion_index = len;
331 return kNotFound;
332 } else {
333 for (int i = 0; i < len; i++) {
334 if (GetKey(i) == name) return i;
335 }
336 return kNotFound;
337 }
338}

353TransitionsAccessor::Encoding TransitionsAccessor::GetEncoding(
354 Isolate* isolate, Tagged<MaybeObject> raw_transitions) {
355 Tagged<HeapObject> heap_object;
356 if (raw_transitions.IsSmi() || raw_transitions.IsCleared()) {
357 return kUninitialized;
358 } else if (raw_transitions.IsWeak()) {
359 return kWeakRef;
360 } else if (raw_transitions.GetHeapObjectIfStrong(isolate, &heap_object)) {
361 if (IsTransitionArray(heap_object)) {
362 return kFullTransitionArray;
363 } else if (IsPrototypeInfo(heap_object)) {
364 return kPrototypeInfo;
365 } else {
366 DCHECK(IsMap(heap_object));
367 return kMigrationTarget;
368 }
369 } else {
370 UNREACHABLE();
371 }

426int TransitionArray::CompareNames(Tagged<Name> key1, uint32_t hash1,
427 Tagged<Name> key2, uint32_t hash2) {
428 if (key1 != key2) {
429 // In case of hash collisions key1 is always "less" than key2.
430 return hash1 <= hash2 ? -1 : 1;
431 }
432
433 return 0;
434}

436int TransitionArray::CompareDetails(PropertyKind kind1,
437 PropertyAttributes attributes1,
438 PropertyKind kind2,
439 PropertyAttributes attributes2) {
440 if (kind1 != kind2) {
441 return static_cast<int>(kind1) < static_cast<int>(kind2) ? -1 : 1;
442 }
443
444 if (attributes1 != attributes2) {
445 return static_cast<int>(attributes1) < static_cast<int>(attributes2) ? -1
446 : 1;
447 }
448
449 return 0;
450}

469template <typename Char>
470bool TransitionsAccessor::IsExpectedTransition(
471 Tagged<Name> transition_name, Tagged<Map> transition_target,
472 base::Vector<const Char> key_chars) {
473 if (transition_target->NumberOfOwnDescriptors() == 0) return false;
474 PropertyDetails details = GetSimpleTargetDetails(transition_target);
475 if (details.location() != PropertyLocation::kField) return false;
476 DCHECK_EQ(PropertyKind::kData, details.kind());
477 if (details.attributes() != NONE) return false;
478 if (!IsString(transition_name)) return false;
479 if (!Cast<String>(transition_name)->IsEqualTo(key_chars)) return false;
480 return true;
481}

483template <typename Char>
484std::pair<Handle<String>, Handle<Map>> TransitionsAccessor::ExpectedTransition(
485 base::Vector<const Char> key_chars) {
486 DisallowGarbageCollection no_gc;
487 switch (encoding()) {
488 case kPrototypeInfo:
489 case kUninitialized:
490 case kMigrationTarget:
491 return {Handle<String>::null(), Handle<Map>::null()};
492 case kWeakRef: {
493 Tagged<Map> target =
494 Cast<Map>(raw_transitions_.GetHeapObjectAssumeWeak());
495 Tagged<Name> name = GetSimpleTransitionKey(target);
496 if (IsExpectedTransition(name, target, key_chars)) {
497 return {handle(Cast<String>(name), isolate_), handle(target, isolate_)};
498 }
499 return {Handle<String>::null(), Handle<Map>::null()};
500 }
501 case kFullTransitionArray: {
502 Tagged<TransitionArray> array =
503 Cast<TransitionArray>(raw_transitions_.GetHeapObjectAssumeStrong());
504 int entries = array->number_of_transitions();
505 // Do linear search for small entries.
506 const int kMaxEntriesForLinearSearch = 8;
507 if (entries > kMaxEntriesForLinearSearch)
508 return {Handle<String>::null(), Handle<Map>::null()};
509 for (int i = entries - 1; i >= 0; i--) {
510 Tagged<Name> name = array->GetKey(i);
511 Tagged<Map> target = array->GetTarget(i);
512 if (IsExpectedTransition(name, target, key_chars)) {
513 return {handle(Cast<String>(name), isolate_),
514 handle(GetTarget(i), isolate_)};
515 }
516 }
517 return {Handle<String>::null(), Handle<Map>::null()};
518 }
519 }
520 UNREACHABLE();
521}

523template <typename Callback, typename ProtoCallback, typename SideStepCallback,
524 bool with_key>
525void TransitionsAccessor::ForEachTransitionWithKey(
526 DisallowGarbageCollection* no_gc, Callback callback,
527 ProtoCallback proto_transition_callback,
528 SideStepCallback side_step_transition_callback) {
529 switch (encoding()) {
530 case kPrototypeInfo:
531 case kUninitialized:
532 case kMigrationTarget:
533 return;
534 case kWeakRef: {
535 Tagged<Map> target =
536 Cast<Map>(raw_transitions_.GetHeapObjectAssumeWeak());
537 if constexpr (with_key) {
538 callback(GetSimpleTransitionKey(target), target);
539 } else {
540 callback(target);
541 }
542 return;
543 }
544 case kFullTransitionArray: {
545 base::SharedMutexGuardIf<base::kShared> scope(
546 isolate_->full_transition_array_access(), concurrent_access_);
547 Tagged<TransitionArray> transition_array = transitions();
548 int num_transitions = transition_array->number_of_transitions();
549 ReadOnlyRoots roots(isolate_);
550 for (int i = 0; i < num_transitions; ++i) {
551 if constexpr (with_key) {
552 Tagged<Name> key = transition_array->GetKey(i);
553 callback(key, GetTarget(i));
554 } else {
555 callback(GetTarget(i));
556 }
557 }
558 if constexpr (!std::is_same<ProtoCallback, std::nullptr_t>::value) {
559 if (transitions()->HasPrototypeTransitions()) {
560 Tagged<WeakFixedArray> cache =
561 transitions()->GetPrototypeTransitions();
562 int length = TransitionArray::NumberOfPrototypeTransitions(cache);
563 for (int i = 0; i < length; i++) {
564 Tagged<MaybeObject> target =
565 cache->get(TransitionArray::kProtoTransitionHeaderSize + i);
566 Tagged<HeapObject> heap_object;
567 if (target.GetHeapObjectIfWeak(&heap_object)) {
568 proto_transition_callback(Cast<Map>(heap_object));
569 }
570 }
571 }
572 }
573 if constexpr (!std::is_same<SideStepCallback, std::nullptr_t>::value) {
574 if (transitions()->HasSideStepTransitions()) {
575 Tagged<WeakFixedArray> cache =
576 transitions()->GetSideStepTransitions();
577 for (uint32_t i = SideStepTransition::kFirstMapIdx;
578 i <= SideStepTransition::kLastMapIdx; i++) {
579 Tagged<MaybeObject> target = cache->get(i);
580 if (target.IsWeak() || target == SideStepTransition::Unreachable) {
581 if constexpr (with_key) {
582 side_step_transition_callback(
583 static_cast<SideStepTransition::Kind>(i),
584 target.GetHeapObjectOrSmi());
585 } else {
586 side_step_transition_callback(target.GetHeapObjectOrSmi());
587 }
588 }
589 }
590 }
591 }
592
593 return;
594 }
595 }
596 UNREACHABLE();
597}
