16// Incremental marking schedule that assumes a fixed time window for scheduling
17// incremental marking steps.
18//
19// Usage:
20// 1. NotifyIncrementalMarkingStart()
21// 2. Any combination of:
22// -> UpdateMutatorThreadMarkedBytes(mutator_marked_bytes)
23// -> AddConcurrentlyMarkedBytes(concurrently_marked_bytes_delta)
24// -> MarkSynchronously(GetNextIncrementalStepDuration(estimated_live_size))
25class V8_EXPORT_PRIVATE IncrementalMarkingSchedule final {
26 public:
27 struct StepInfo final {
28 size_t mutator_marked_bytes = 0;
29 size_t concurrent_marked_bytes = 0;
30 size_t estimated_live_bytes = 0;
31 size_t expected_marked_bytes = 0;
32 v8::base::TimeDelta elapsed_time;
33
34 size_t marked_bytes() const {
35 return mutator_marked_bytes + concurrent_marked_bytes;
36 }
37 // Returns the schedule delta in bytes. Positive and negative delta values
38 // indicate that the marked bytes are ahead and behind the expected
39 // schedule, respectively.
40 int64_t scheduled_delta_bytes() const {
41 return static_cast<int64_t>(marked_bytes()) - expected_marked_bytes;
42 }
43
44 // Returns whether the schedule is behind the expectation.
45 bool is_behind_expectation() const { return scheduled_delta_bytes() < 0; }
46 };
47
48 // Estimated walltime duration of incremental marking per GC cycle. This value
49 // determines how the mutator thread will try to catch up on incremental
50 // marking steps.
51 static constexpr v8::base::TimeDelta kEstimatedMarkingTime =
52 v8::base::TimeDelta::FromMilliseconds(500);
53
54 // Minimum number of bytes that should be marked during an incremental
55 // marking step.
56 static constexpr size_t kDefaultMinimumMarkedBytesPerIncrementalStep =
57 64 * 1024;
58
59 // Step size used when no progress is being made. This step size should allow
60 // for finalizing marking.
61 static constexpr size_t kStepSizeWhenNotMakingProgress = 64 * 1024;
62
63 static std::unique_ptr<IncrementalMarkingSchedule>
64 CreateWithDefaultMinimumMarkedBytesPerStep(bool predictable_schedule = false);
65 static std::unique_ptr<IncrementalMarkingSchedule>
66 CreateWithZeroMinimumMarkedBytesPerStep(bool predictable_schedule = false);
67
68 IncrementalMarkingSchedule(const IncrementalMarkingSchedule&) = delete;
69 IncrementalMarkingSchedule& operator=(const IncrementalMarkingSchedule&) =
70 delete;
71
72 // Notifies the schedule that incremental marking has been started.
73 void NotifyIncrementalMarkingStart();
74
75 // Updates the mutator marked bytes. Must be called from the thread owning the
76 // schedule. The schedule supports marked bytes being adjusted downwards,
77 // i.e., going backwards in the schedule.
78 void UpdateMutatorThreadMarkedBytes(size_t);
79
80 // Adds concurrently marked bytes. May be called from any thread. Not required
81 // to be complete, i.e., it is okay to not report bytes already marked for the
82 // schedule.
83 void AddConcurrentlyMarkedBytes(size_t);
84
85 // Returns the reported overall marked bytes including those marked by the
86 // mutator and concurrently.
87 size_t GetOverallMarkedBytes() const;
88
89 // Returns the reported concurrently marked bytes. Only as accurate as
90 // `AddConcurrentlyMarkedBytes()` is.
91 size_t GetConcurrentlyMarkedBytes() const;
92
93 // Computes the next step duration based on reported marked bytes and the
94 // current `estimated_live_bytes`.
95 size_t GetNextIncrementalStepDuration(size_t estimated_live_bytes);
96
97 // Returns the step info for the current step. This function is most useful
98 // after calling `GetNextIncrementalStepDuration()` to report scheduling
99 // details.
100 StepInfo GetCurrentStepInfo() const;
101
102 // Returns whether locally cached ephemerons should be flushed and made
103 // available globally. Will only return true once every
104 // `kEphemeronPairsFlushingRatioIncrements` percent of overall marked bytes.
105 bool ShouldFlushEphemeronPairs();
106
107 // The minimum marked bytes per step. This is a lower bound for all the step
108 // sizes returned from `GetNextIncrementalStepDuration()`.
109 size_t min_marked_bytes_per_step() const {
110 return min_marked_bytes_per_step_;
111 }
112
113 // Sets the elapsed time for testing purposes. Is reset after calling
114 // `GetNextIncrementalStepDuration()`.
115 void SetElapsedTimeForTesting(v8::base::TimeDelta);
116
117 private:
118 static constexpr double kEphemeronPairsFlushingRatioIncrements = 0.25;
119
120 IncrementalMarkingSchedule(size_t min_marked_bytes_per_step,
121 bool predictable_schedule);
122
123 v8::base::TimeDelta GetElapsedTime();
124
125 v8::base::TimeTicks incremental_marking_start_time_;
126 size_t mutator_thread_marked_bytes_ = 0;
127 std::atomic_size_t concurrently_marked_bytes_{0};
128 size_t last_estimated_live_bytes_ = 0;
129 double ephemeron_pairs_flushing_ratio_target_ = 0.25;
130 StepInfo current_step_;
131 const size_t min_marked_bytes_per_step_;
132 const bool predictable_schedule_ = false;
133 std::optional<v8::base::TimeDelta> elapsed_time_override_;
134};
