27Reduction JSContextSpecialization::Reduce(Node* node) {
28 switch (node->opcode()) {
29 case IrOpcode::kParameter:
30 return ReduceParameter(node);
31 case IrOpcode::kJSLoadContext:
32 return ReduceJSLoadContext(node);
33 case IrOpcode::kJSLoadScriptContext:
34 return ReduceJSLoadScriptContext(node);
35 case IrOpcode::kJSStoreContext:
36 return ReduceJSStoreContext(node);
37 case IrOpcode::kJSStoreScriptContext:
38 return ReduceJSStoreScriptContext(node);
39 case IrOpcode::kJSGetImportMeta:
40 return ReduceJSGetImportMeta(node);
41 default:
42 break;
43 }
44 return NoChange();
45}

47Reduction JSContextSpecialization::ReduceParameter(Node* node) {
48 DCHECK_EQ(IrOpcode::kParameter, node->opcode());
49 int const index = ParameterIndexOf(node->op());
50 if (index == Linkage::kJSCallClosureParamIndex) {
51 // Constant-fold the function parameter {node}.
52 Handle<JSFunction> function;
53 if (closure().ToHandle(&function)) {
54 Node* value =
55 jsgraph()->ConstantNoHole(MakeRef(broker_, function), broker());
56 return Replace(value);
57 }
58 }
59 return NoChange();
60}

62Reduction JSContextSpecialization::SimplifyJSLoadContext(Node* node,
63 Node* new_context,
64 size_t new_depth) {
65 DCHECK_EQ(IrOpcode::kJSLoadContext, node->opcode());
66 const ContextAccess& access = ContextAccessOf(node->op());
67 DCHECK_LE(new_depth, access.depth());
68
69 if (new_depth == access.depth() &&
70 new_context == NodeProperties::GetContextInput(node)) {
71 return NoChange();

74 const Operator* op = jsgraph_->javascript()->LoadContext(
75 new_depth, access.index(), access.immutable());
76 NodeProperties::ReplaceContextInput(node, new_context);
77 NodeProperties::ChangeOp(node, op);
78 return Changed(node);

138bool IsContextParameter(Node* node) {
139 DCHECK_EQ(IrOpcode::kParameter, node->opcode());
140 return ParameterIndexOf(node->op()) ==
141 StartNode{NodeProperties::GetValueInput(node, 0)}
142 .ContextParameterIndex_MaybeNonStandardLayout();
143}

145// Given a context {node} and the {distance} from that context to the target
146// context (which we want to read from or store to), try to return a
147// specialization context. If successful, update {distance} to whatever
148// distance remains from the specialization context.
149OptionalContextRef GetSpecializationContext(JSHeapBroker* broker, Node* node,
150 size_t* distance,
151 Maybe<OuterContext> maybe_outer) {
152 switch (node->opcode()) {
153 case IrOpcode::kHeapConstant: {
154 // TODO(jgruber,chromium:1209798): Using kAssumeMemoryFence works around
155 // the fact that the graph stores handles (and not refs). The assumption
156 // is that any handle inserted into the graph is safe to read; but we
157 // don't preserve the reason why it is safe to read. Thus we must
158 // over-approximate here and assume the existence of a memory fence. In
159 // the future, we should consider having the graph store ObjectRefs or
160 // ObjectData pointer instead, which would make new ref construction here
161 // unnecessary.
162 HeapObjectRef object =
163 MakeRefAssumeMemoryFence(broker, HeapConstantOf(node->op()));
164 if (object.IsContext()) return object.AsContext();
165 break;
166 }
167 case IrOpcode::kParameter: {
168 OuterContext outer;
169 if (maybe_outer.To(&outer) && IsContextParameter(node) &&
170 *distance >= outer.distance) {
171 *distance -= outer.distance;
172 return MakeRef(broker, outer.context);
173 }
174 break;
175 }
176 default:
177 break;
178 }
179 return OptionalContextRef();
180}

184Reduction JSContextSpecialization::ReduceJSLoadContext(Node* node) {
185 DCHECK_EQ(IrOpcode::kJSLoadContext, node->opcode());
186
187 const ContextAccess& access = ContextAccessOf(node->op());
188 size_t depth = access.depth();
189
190 // First walk up the context chain in the graph as far as possible.
191 Node* context = NodeProperties::GetOuterContext(node, &depth);
192
193 OptionalContextRef maybe_concrete =
194 GetSpecializationContext(broker(), context, &depth, outer());
195 if (!maybe_concrete.has_value()) {
196 // We do not have a concrete context object, so we can only partially reduce
197 // the load by folding-in the outer context node.
198 return SimplifyJSLoadContext(node, context, depth);
199 }
200
201 // Now walk up the concrete context chain for the remaining depth.
202 ContextRef concrete = maybe_concrete.value();
203 concrete = concrete.previous(broker(), &depth);
204 if (depth > 0) {
205 TRACE_BROKER_MISSING(broker(), "previous value for context " << concrete);
206 return SimplifyJSLoadContext(
207 node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
208 }
209
210 if (!access.immutable() &&
211 !broker()->dependencies()->DependOnScriptContextSlotProperty(
212 concrete, access.index(), ContextSidePropertyCell::kConst,
213 broker())) {
214 // We found the requested context object but since the context slot is
215 // mutable we can only partially reduce the load.
216 return SimplifyJSLoadContext(
217 node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
218 }
219
220 // This will hold the final value, if we can figure it out.
221 OptionalObjectRef maybe_value;
222 maybe_value = concrete.get(broker(), static_cast<int>(access.index()));
223
224 if (!maybe_value.has_value()) {
225 TRACE_BROKER_MISSING(broker(), "slot value " << access.index()
226 << " for context "
227 << concrete);
228 return SimplifyJSLoadContext(
229 node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
230 }
231
232 // Even though the context slot is immutable, the context might have escaped
233 // before the function to which it belongs has initialized the slot.
234 // We must be conservative and check if the value in the slot is currently
235 // the hole or undefined. Only if it is neither of these, can we be sure
236 // that it won't change anymore.
237 if (maybe_value->IsUndefined() || maybe_value->IsTheHole()) {
238 return SimplifyJSLoadContext(
239 node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
240 }
241
242 // Success. The context load can be replaced with the constant.
243 Node* constant = jsgraph_->ConstantNoHole(*maybe_value, broker());
244 ReplaceWithValue(node, constant);
245 return Replace(constant);
246}

248Reduction JSContextSpecialization::ReduceJSLoadScriptContext(Node* node) {
249 DCHECK_EQ(IrOpcode::kJSLoadScriptContext, node->opcode());
250
251 const ContextAccess& access = ContextAccessOf(node->op());
252 DCHECK(!access.immutable());
253 size_t depth = access.depth();
254
255 // First walk up the context chain in the graph as far as possible.
256 Node* effect = NodeProperties::GetEffectInput(node);
257 Node* control = NodeProperties::GetControlInput(node);
258 Node* context = NodeProperties::GetOuterContext(node, &depth);
259
260 OptionalContextRef maybe_concrete =
261 GetSpecializationContext(broker(), context, &depth, outer());
262 if (!maybe_concrete.has_value()) {
263 // We do not have a concrete context object, so we can only partially reduce
264 // the load by folding-in the outer context node.
265 return SimplifyJSLoadScriptContext(node, context, depth);
266 }
267
268 // Now walk up the concrete context chain for the remaining depth.
269 ContextRef concrete = maybe_concrete.value();
270 concrete = concrete.previous(broker(), &depth);
271 if (depth > 0) {
272 TRACE_BROKER_MISSING(broker(), "previous value for context " << concrete);
273 return SimplifyJSLoadScriptContext(
274 node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
275 }
276
277 DCHECK(concrete.object()->IsScriptContext());
278 auto maybe_property =
279 concrete.object()->GetScriptContextSideProperty(access.index());
280 auto property =
281 maybe_property ? maybe_property.value() : ContextSidePropertyCell::kOther;
282 switch (property) {
283 case ContextSidePropertyCell::kConst: {
284 OptionalObjectRef maybe_value =
285 concrete.get(broker(), static_cast<int>(access.index()));
286 if (!maybe_value.has_value()) {
287 TRACE_BROKER_MISSING(broker(), "slot value " << access.index()
288 << " for context "
289 << concrete);
290 return SimplifyJSLoadScriptContext(
291 node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
292 }
293 broker()->dependencies()->DependOnScriptContextSlotProperty(
294 concrete, access.index(), property, broker());
295 Node* constant = jsgraph_->ConstantNoHole(*maybe_value, broker());
296 ReplaceWithValue(node, constant, effect, control);
297 return Changed(node);
298 }
299 case ContextSidePropertyCell::kSmi: {
300 broker()->dependencies()->DependOnScriptContextSlotProperty(
301 concrete, access.index(), property, broker());
302 Node* load = effect = jsgraph_->graph()->NewNode(
303 jsgraph_->simplified()->LoadField(
304 AccessBuilder::ForContextSlotSmi(access.index())),
305 jsgraph_->ConstantNoHole(concrete, broker()), effect, control);
306 ReplaceWithValue(node, load, effect, control);
307 return Changed(node);
308 }
309 case ContextSidePropertyCell::kMutableInt32: {
310 Node* mutable_heap_number;
311 if (auto concrete_heap_number =
312 concrete.get(broker(), static_cast<int>(access.index()))) {
313 if (!concrete_heap_number->IsHeapNumber()) {
314 // TODO(victorgomes): In case the tag is out of date by now we could
315 // retry this reduction.
316 return NoChange();
317 }
318 mutable_heap_number = jsgraph_->ConstantMutableHeapNumber(
319 concrete_heap_number->AsHeapNumber(), broker());
320 } else {
321 mutable_heap_number = effect = jsgraph_->graph()->NewNode(
322 jsgraph_->simplified()->LoadField(
323 AccessBuilder::ForContextSlot(access.index())),
324 jsgraph_->ConstantNoHole(concrete, broker()), effect, control);
325 }
326 broker()->dependencies()->DependOnScriptContextSlotProperty(
327 concrete, access.index(), property, broker());
328 Node* int32_load = effect = jsgraph_->graph()->NewNode(
329 jsgraph_->simplified()->LoadField(AccessBuilder::ForHeapInt32Value()),
330 mutable_heap_number, effect, control);
331 ReplaceWithValue(node, int32_load, effect, control);
332 return Changed(node);
333 }
334 case ContextSidePropertyCell::kMutableHeapNumber: {
335 Node* mutable_heap_number;
336 if (auto concrete_heap_number =
337 concrete.get(broker(), static_cast<int>(access.index()))) {
338 if (!concrete_heap_number->IsHeapNumber()) {
339 // TODO(victorgomes): In case the tag is out of date by now we could
340 // retry this reduction.
341 return NoChange();
342 }
343 mutable_heap_number = jsgraph_->ConstantMutableHeapNumber(
344 concrete_heap_number->AsHeapNumber(), broker());
345 } else {
346 mutable_heap_number = effect = jsgraph_->graph()->NewNode(
347 jsgraph_->simplified()->LoadField(
348 AccessBuilder::ForContextSlot(access.index())),
349 jsgraph_->ConstantNoHole(concrete, broker()), effect, control);
350 }
351 broker()->dependencies()->DependOnScriptContextSlotProperty(
352 concrete, access.index(), property, broker());
353 Node* double_load = effect =
354 jsgraph_->graph()->NewNode(jsgraph_->simplified()->LoadField(
355 AccessBuilder::ForHeapNumberValue()),
356 mutable_heap_number, effect, control);
357 ReplaceWithValue(node, double_load, effect, control);
358 return Changed(node);
359 }
360 case ContextSidePropertyCell::kOther: {
361 // Do a normal context load.
362 Node* load = effect = jsgraph_->graph()->NewNode(
363 jsgraph_->simplified()->LoadField(
364 AccessBuilder::ForContextSlot(access.index())),
365 jsgraph_->ConstantNoHole(concrete, broker()), effect, control);
366 ReplaceWithValue(node, load, effect, control);
367 return Changed(node);
368 }
369 default:
370 UNREACHABLE();
371 }

375 DCHECK_EQ(IrOpcode::kJSStoreContext, node->opcode());

378 size_t depth = access.depth();
379

402 node, jsgraph()->ConstantNoHole(concrete, broker()), depth);

405Reduction JSContextSpecialization::ReduceJSStoreScriptContext(Node* node) {
406 DCHECK(v8_flags.const_tracking_let);
407 DCHECK_EQ(IrOpcode::kJSStoreScriptContext, node->opcode());
408
409 const ContextAccess& access = ContextAccessOf(node->op());
410 size_t depth = access.depth();
411
412 // First walk up the context chain in the graph until we reduce the depth to 0
413 // or hit a node that does not have a CreateXYZContext operator.
414 Node* context = NodeProperties::GetOuterContext(node, &depth);
415 Node* value = NodeProperties::GetValueInput(node, 0);
416 Effect effect{NodeProperties::GetEffectInput(node)};
417 Control control{NodeProperties::GetControlInput(node)};
418
419 OptionalContextRef maybe_concrete =
420 GetSpecializationContext(broker(), context, &depth, outer());
421 if (!maybe_concrete.has_value()) {
422 // We do not have a concrete context object, so we can only partially reduce
423 // the load by folding-in the outer context node.
424 return SimplifyJSStoreScriptContext(node, context, depth);
425 }
426
427 // Now walk up the concrete context chain for the remaining depth.
428 ContextRef concrete = maybe_concrete.value();
429 concrete = concrete.previous(broker(), &depth);
430 if (depth > 0) {
431 TRACE_BROKER_MISSING(broker(), "previous value for context " << concrete);
432 return SimplifyJSStoreScriptContext(
433 node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
434 }
435 DCHECK(concrete.object()->IsScriptContext());
436 auto maybe_property =
437 concrete.object()->GetScriptContextSideProperty(access.index());
438 if (!maybe_property) {
439 return SimplifyJSStoreScriptContext(
440 node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
441 }
442 auto property = maybe_property.value();
443 PropertyAccessBuilder access_builder(jsgraph(), broker());
444 if (property == ContextSidePropertyCell::kConst) {
445 compiler::OptionalObjectRef constant =
446 concrete.get(broker(), static_cast<int>(access.index()));
447 if (!constant.has_value() ||
448 (constant->IsString() && !constant->IsInternalizedString())) {
449 return SimplifyJSStoreScriptContext(
450 node, jsgraph()->ConstantNoHole(concrete, broker()), depth);
451 }
452 broker()->dependencies()->DependOnScriptContextSlotProperty(
453 concrete, access.index(), property, broker());
454 access_builder.BuildCheckValue(value, &effect, control, *constant);
455 ReplaceWithValue(node, effect, effect, control);
456 return Changed(node);
457 }
458
459 if (!v8_flags.script_context_mutable_heap_number) {
460 // Do a normal context store.
461 Node* store = jsgraph()->graph()->NewNode(
462 jsgraph()->simplified()->StoreField(
463 AccessBuilder::ForContextSlot(access.index())),
464 jsgraph()->ConstantNoHole(concrete, broker()), value, effect, control);
465 ReplaceWithValue(node, store, store, control);
466 return Changed(node);
467 }
468
469 switch (property) {
470 case ContextSidePropertyCell::kConst:
471 UNREACHABLE();
472 case ContextSidePropertyCell::kSmi: {
473 broker()->dependencies()->DependOnScriptContextSlotProperty(
474 concrete, access.index(), property, broker());
475 Node* smi_value = access_builder.BuildCheckSmi(value, &effect, control);
476 Node* smi_store = jsgraph()->graph()->NewNode(
477 jsgraph()->simplified()->StoreField(
478 AccessBuilder::ForContextSlotSmi(access.index())),
479 jsgraph()->ConstantNoHole(concrete, broker()), smi_value, effect,
480 control);
481 ReplaceWithValue(node, smi_store, smi_store, control);
482 return Changed(node);
483 }
484 case ContextSidePropertyCell::kMutableInt32: {
485 Node* mutable_heap_number;
486 if (auto concrete_heap_number =
487 concrete.get(broker(), static_cast<int>(access.index()))) {
488 if (!concrete_heap_number->IsHeapNumber()) {
489 // TODO(victorgomes): In case the tag is out of date by now we could
490 // retry this reduction.
491 return NoChange();
492 }
493 mutable_heap_number = jsgraph_->ConstantMutableHeapNumber(
494 concrete_heap_number->AsHeapNumber(), broker());
495 } else {
496 mutable_heap_number = effect = jsgraph_->graph()->NewNode(
497 jsgraph_->simplified()->LoadField(
498 AccessBuilder::ForContextSlot(access.index())),
499 jsgraph_->ConstantNoHole(concrete, broker()), effect, control);
500 }
501 broker()->dependencies()->DependOnScriptContextSlotProperty(
502 concrete, access.index(), property, broker());
503 Node* input_number =
504 access_builder.BuildCheckNumberFitsInt32(value, &effect, control);
505 Node* double_store = jsgraph()->graph()->NewNode(
506 jsgraph()->simplified()->StoreField(
507 AccessBuilder::ForHeapInt32Value()),
508 mutable_heap_number, input_number, effect, control);
509 ReplaceWithValue(node, double_store, double_store, control);
510 return Changed(node);
511 }
512 case ContextSidePropertyCell::kMutableHeapNumber: {
513 Node* mutable_heap_number;
514 if (auto concrete_heap_number =
515 concrete.get(broker(), static_cast<int>(access.index()))) {
516 if (!concrete_heap_number->IsHeapNumber()) {
517 // TODO(victorgomes): In case the tag is out of date by now we could
518 // retry this reduction.
519 return NoChange();
520 }
521 mutable_heap_number = jsgraph_->ConstantMutableHeapNumber(
522 concrete_heap_number->AsHeapNumber(), broker());
523 } else {
524 mutable_heap_number = effect = jsgraph_->graph()->NewNode(
525 jsgraph_->simplified()->LoadField(
526 AccessBuilder::ForContextSlot(access.index())),
527 jsgraph_->ConstantNoHole(concrete, broker()), effect, control);
528 }
529 broker()->dependencies()->DependOnScriptContextSlotProperty(
530 concrete, access.index(), property, broker());
531 Node* input_number =
532 access_builder.BuildCheckNumber(value, &effect, control);
533 Node* double_store = jsgraph()->graph()->NewNode(
534 jsgraph()->simplified()->StoreField(
535 AccessBuilder::ForHeapNumberValue()),

537 ReplaceWithValue(node, double_store, double_store, control);

539 }
540 case ContextSidePropertyCell::kOther: {
541 // Do a normal context store.
542 Node* store = jsgraph()->graph()->NewNode(
543 jsgraph()->simplified()->StoreField(

548 return Changed(node);
549 }

555OptionalContextRef GetModuleContext(JSHeapBroker* broker, Node* node,
556 Maybe<OuterContext> maybe_context) {
557 size_t depth = std::numeric_limits<size_t>::max();
558 Node* context = NodeProperties::GetOuterContext(node, &depth);
559
560 auto find_context = [broker](ContextRef c) {
561 while (c.map(broker).instance_type() != MODULE_CONTEXT_TYPE) {
562 size_t depth = 1;
563 c = c.previous(broker, &depth);
564 CHECK_EQ(depth, 0);
565 }
566 return c;
567 };
568
569 switch (context->opcode()) {
570 case IrOpcode::kHeapConstant: {
571 // TODO(jgruber,chromium:1209798): Using kAssumeMemoryFence works around
572 // the fact that the graph stores handles (and not refs). The assumption
573 // is that any handle inserted into the graph is safe to read; but we
574 // don't preserve the reason why it is safe to read. Thus we must
575 // over-approximate here and assume the existence of a memory fence. In
576 // the future, we should consider having the graph store ObjectRefs or
577 // ObjectData pointer instead, which would make new ref construction here
578 // unnecessary.
579 HeapObjectRef object =
580 MakeRefAssumeMemoryFence(broker, HeapConstantOf(context->op()));
581 if (object.IsContext()) {
582 return find_context(object.AsContext());
583 }
584 break;
585 }
586 case IrOpcode::kParameter: {
587 OuterContext outer;
588 if (maybe_context.To(&outer) && IsContextParameter(context)) {
589 return find_context(MakeRef(broker, outer.context));
590 }
591 break;
592 }
593 default:
594 break;
595 }
596
597 return OptionalContextRef();
598}

600Reduction JSContextSpecialization::ReduceJSGetImportMeta(Node* node) {
601 OptionalContextRef maybe_context = GetModuleContext(broker(), node, outer());
602 if (!maybe_context.has_value()) return NoChange();
603
604 ContextRef context = maybe_context.value();
605 OptionalObjectRef module = context.get(broker(), Context::EXTENSION_INDEX);
606 if (!module.has_value()) return NoChange();
607 OptionalObjectRef import_meta =
608 module->AsSourceTextModule().import_meta(broker());
609 if (!import_meta.has_value()) return NoChange();
610 if (!import_meta->IsJSObject()) {
611 DCHECK(import_meta->IsTheHole());
612 // The import.meta object has not yet been created. Let JSGenericLowering
613 // replace the operator with a runtime call.
614 return NoChange();
615 }
616
617 Node* import_meta_const = jsgraph()->ConstantNoHole(*import_meta, broker());
618 ReplaceWithValue(node, import_meta_const);
619 return Changed(import_meta_const);
620}
