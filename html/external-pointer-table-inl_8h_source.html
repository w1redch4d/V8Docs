17void ExternalPointerTableEntry::MakeExternalPointerEntry(Address value,
18 ExternalPointerTag tag,
19 bool mark_as_alive) {
20 // The 2nd most significant byte must be empty as we store the tag in int.
21 DCHECK_EQ(0, value & kExternalPointerTagAndMarkbitMask);
22 DCHECK_NE(tag, kExternalPointerFreeEntryTag);
23 DCHECK_NE(tag, kExternalPointerEvacuationEntryTag);
24
25 Payload new_payload(value, tag);
26 if (V8_UNLIKELY(mark_as_alive)) {
27 new_payload.SetMarkBit();
28 }
29 payload_.store(new_payload, std::memory_order_relaxed);
30 MaybeUpdateRawPointerForLSan(value);
31}

61Address ExternalPointerTableEntry::ExchangeExternalPointer(
62 Address value, ExternalPointerTag tag) {
63 // The 2nd most significant byte must be empty as we store the tag in int.
64 DCHECK_EQ(0, value & kExternalPointerTagAndMarkbitMask);
65
66 Payload new_payload(value, tag);
67 // Writing an entry currently also marks it as alive. In the future, we might
68 // want to drop this and instead use write barriers where necessary.
69 new_payload.SetMarkBit();
70 Payload old_payload =
71 payload_.exchange(new_payload, std::memory_order_relaxed);
72 DCHECK(old_payload.ContainsPointer());
73 MaybeUpdateRawPointerForLSan(value);
74 return old_payload.Untag(tag);
75}

109void ExternalPointerTableEntry::Mark() {
110 auto old_payload = payload_.load(std::memory_order_relaxed);
111 DCHECK(old_payload.ContainsPointer());
112
113 auto new_payload = old_payload;
114 new_payload.SetMarkBit();
115
116 // We don't need to perform the CAS in a loop: if the new value is not equal
117 // to the old value, then the mutator must've just written a new value into
118 // the entry. This in turn must've set the marking bit already (see e.g.
119 // SetExternalPointer), so we don't need to do it again.
120 bool success = payload_.compare_exchange_strong(old_payload, new_payload,
121 std::memory_order_relaxed);
122 DCHECK(success || old_payload.HasMarkBitSet());
123 USE(success);
124}

136void ExternalPointerTableEntry::Evacuate(ExternalPointerTableEntry& dest,
137 EvacuateMarkMode mode) {
138 auto payload = payload_.load(std::memory_order_relaxed);
139 // We expect to only evacuate entries containing external pointers.
140 DCHECK(payload.ContainsPointer());
141
142 switch (mode) {
143 case EvacuateMarkMode::kTransferMark:
144 break;
145 case EvacuateMarkMode::kLeaveUnmarked:
146 DCHECK(!payload.HasMarkBitSet());
147 break;
148 case EvacuateMarkMode::kClearMark:
149 DCHECK(payload.HasMarkBitSet());
150 payload.ClearMarkBit();
151 break;
152 }
153
154 dest.payload_.store(payload, std::memory_order_relaxed);
155#if defined(LEAK_SANITIZER)
156 dest.raw_pointer_for_lsan_ = raw_pointer_for_lsan_;
157#endif // LEAK_SANITIZER
158
159 // The destination entry takes ownership of the pointer.
160 MakeZappedEntry();
161}

171 ExternalPointerTag tag) {
172 DCHECK_NE(kNullExternalPointerHandle, handle);

174 // TODO(saelo): This works for now, but once we actually free the external
175 // object here, this will probably become awkward: it's likely not intuitive
176 // that a set_foo() call on some object causes another object to be freed.
177 // Probably at that point we should instead just forbid re-setting the
178 // external pointers if they are managed (via a DCHECK).

198void ExternalPointerTable::Zap(ExternalPointerHandle handle) {
199 // Zapping the null entry is a nop. This is useful as we reset the handle of
200 // managed resources to the kNullExternalPointerHandle when the entry is
201 // deleted. See SweepAndCompact.
202 if (handle == kNullExternalPointerHandle) return;
203 uint32_t index = HandleToIndex(handle);
204 at(index).MakeZappedEntry();
205}

218void ExternalPointerTable::Mark(Space* space, ExternalPointerHandle handle,
219 Address handle_location) {
220 DCHECK(space->BelongsTo(this));
221
222 // The handle_location must always contain the given handle. Except if the
223 // slot is lazily-initialized. In that case, the handle may transition from
224 // the null handle to a valid handle. However, in that case the
225 // newly-allocated entry will already have been marked as alive during
226 // allocation, and so we don't need to do anything here.
227#ifdef DEBUG
228 ExternalPointerHandle current_handle = base::AsAtomic32::Acquire_Load(
229 reinterpret_cast<ExternalPointerHandle*>(handle_location));
230 DCHECK(handle == kNullExternalPointerHandle || handle == current_handle);
231#endif
232
233 // If the handle is null, it doesn't have an EPT entry; no mark is needed.
234 if (handle == kNullExternalPointerHandle) return;
235
236 uint32_t index = HandleToIndex(handle);
237 DCHECK(space->Contains(index));
238
239 // If the table is being compacted and the entry is inside the evacuation
240 // area, then allocate and set up an evacuation entry for it.
241 MaybeCreateEvacuationEntry(space, index, handle_location);
242
243 // Even if the entry is marked for evacuation, it still needs to be marked as
244 // alive as it may be visited during sweeping before being evacuation.
245 at(index).Mark();
246}

248void ExternalPointerTable::Evacuate(Space* from_space, Space* to_space,
249 ExternalPointerHandle handle,
250 Address handle_location,
251 EvacuateMarkMode mode) {
252 DCHECK(from_space->BelongsTo(this));
253 DCHECK(to_space->BelongsTo(this));
254
255 CHECK(IsValidHandle(handle));
256
257 auto handle_ptr = reinterpret_cast<ExternalPointerHandle*>(handle_location);
258
259#ifdef DEBUG
260 // Unlike Mark(), we require that the mutator is stopped, so we can simply
261 // verify that the location stores the handle with a non-atomic load.
262 DCHECK_EQ(handle, *handle_ptr);
263#endif
264
265 // If the handle is null, it doesn't have an EPT entry; no evacuation is
266 // needed.
267 if (handle == kNullExternalPointerHandle) return;
268
269 uint32_t from_index = HandleToIndex(handle);
270 DCHECK(from_space->Contains(from_index));
271 uint32_t to_index = AllocateEntry(to_space);
272
273 at(from_index).Evacuate(at(to_index), mode);
274 ExternalPointerHandle new_handle = IndexToHandle(to_index);
275
276 if (Address addr = at(to_index).ExtractManagedResourceOrNull()) {
277 ManagedResource* resource = reinterpret_cast<ManagedResource*>(addr);
278 DCHECK_EQ(resource->ept_entry_, handle);
279 resource->ept_entry_ = new_handle;
280 }
281
282 // Update slot to point to new handle.
283 base::AsAtomic32::Relaxed_Store(handle_ptr, new_handle);
284}

287bool ExternalPointerTable::IsValidHandle(ExternalPointerHandle handle) {
288 uint32_t index = handle >> kExternalPointerIndexShift;
289 return handle == index << kExternalPointerIndexShift;
290}

292// static
293uint32_t ExternalPointerTable::HandleToIndex(ExternalPointerHandle handle) {
294 DCHECK(IsValidHandle(handle));
295 uint32_t index = handle >> kExternalPointerIndexShift;
296#if defined(LEAK_SANITIZER)
297 // When LSan is active, we use "fat" entries that also store the raw pointer
298 // to that LSan can find live references. However, we do this transparently:
299 // we simply multiply the handle by two so that `(handle >> index_shift) * 8`
300 // still produces the correct offset of the entry in the table. However, this
301 // is not secure as an attacker could reference the raw pointer instead of
302 // the encoded pointer in an entry, thereby bypassing the type checks. As
303 // such, this mode must only be used in testing environments. Alternatively,
304 // all places that access external pointer table entries must be made aware
305 // that the entries are 16 bytes large when LSan is active.
306 index /= 2;
307#endif // LEAK_SANITIZER
308 DCHECK_LE(index, kMaxExternalPointers);
309 return index;
310}
311
312// static
313ExternalPointerHandle ExternalPointerTable::IndexToHandle(uint32_t index) {
314 DCHECK_LE(index, kMaxExternalPointers);
315 ExternalPointerHandle handle = index << kExternalPointerIndexShift;
316#if defined(LEAK_SANITIZER)
317 handle *= 2;
318#endif // LEAK_SANITIZER
319 DCHECK_NE(handle, kNullExternalPointerHandle);
320 return handle;
321}

350void ExternalPointerTable::TakeOwnershipOfManagedResourceIfNecessary(
351 Address value, ExternalPointerHandle handle, ExternalPointerTag tag) {

353 ManagedResource* resource = reinterpret_cast<ManagedResource*>(value);

360void ExternalPointerTable::FreeManagedResourceIfPresent(uint32_t entry_index) {
361 // In the future, this would be where we actually delete the external
362 // resource. Currently, the deletion still happens elsewhere, and so here we
363 // instead set the resource's handle to the null handle so that the resource
364 // does not attempt to zap its entry when it is eventually destroyed.
365 if (Address addr = at(entry_index).ExtractManagedResourceOrNull()) {
366 ManagedResource* resource = reinterpret_cast<ManagedResource*>(addr);
367 DCHECK_EQ(resource->ept_entry_, IndexToHandle(entry_index));
368 resource->ept_entry_ = kNullExternalPointerHandle;
369 }
370}
