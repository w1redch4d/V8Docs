355// TNode<T> is an SSA value with the static type tag T, which is one of the
356// following:
357// - MaybeObject> represents the type of all tagged values, including weak
358// pointers.
359// - a subclass of internal::Object represents a non-weak tagged type.
360// - a subclass of internal::UntaggedT represents an untagged type
361// - ExternalReference
362// - PairT<T1, T2> for an operation returning two values, with types T1
363// and T2
364// - UnionOf<T1, T2, ...> represents a value of one of types T1, T2, etc.
365template <class T>
366class TNode {
367 public:
368 template <class U>
369 TNode(const TNode<U>& other) V8_NOEXCEPT
370 requires(is_subtype<U, T>::value)
371 : node_(other.node_) {
372 LazyTemplateChecks();
373 }
374
375 TNode(const TNode& other) V8_NOEXCEPT : node_(other.node_) {}
376 TNode() : node_(nullptr) {}
377
378 TNode operator=(TNode other) {
379 DCHECK_NOT_NULL(other.node_);
380 node_ = other.node_;
381 return *this;
382 }
383
384 operator compiler::Node*() const { return node_; }
385
386 explicit operator bool() const { return node_ != nullptr; }
387
388 static TNode UncheckedCast(compiler::Node* node) { return TNode(node); }
389
390 protected:
391 template <typename U>
392 friend class TNode;
393
394 explicit TNode(compiler::Node* node) : node_(node) { LazyTemplateChecks(); }
395 // These checks shouldn't be checked before TNode is actually used.
396 void LazyTemplateChecks() const {
397 static_assert(is_valid_type_tag<T>::value, "invalid type tag");
398 }
399
400 compiler::Node* node_;
401};
