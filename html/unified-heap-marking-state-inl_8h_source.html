31void UnifiedHeapMarkingState::MarkAndPush(
32 const TracedReferenceBase& reference) {
33 // The following code will crash with null pointer derefs when finding a
34 // non-empty `TracedReferenceBase` when `CppHeap` is in detached mode.
35 Address* traced_handle_location =
36 BasicTracedReferenceExtractor::GetObjectSlotForMarking(reference);
37 // We cannot assume that the reference is non-null as we may get here by
38 // tracing an ephemeron which doesn't have early bailouts, see
39 // `cppgc::Visitor::TraceEphemeron()` for non-Member values.
40 if (!traced_handle_location) {
41 return;
42 }
43 Tagged<Object> object =
44 TracedHandles::Mark(traced_handle_location, mark_mode_);
45 if (!IsHeapObject(object)) {
46 // The embedder is not aware of whether numbers are materialized as heap
47 // objects are just passed around as Smis.
48 return;
49 }
50 Tagged<HeapObject> heap_object = Cast<HeapObject>(object);
51 const auto worklist_target =
52 MarkingHelper::ShouldMarkObject(heap_, heap_object);
53 if (worklist_target) {
54 MarkingHelper::TryMarkAndPush(heap_, local_marking_worklist_,
55 marking_state_, worklist_target.value(),
56 heap_object);
57 }
58}
