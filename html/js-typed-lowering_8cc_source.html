142 // Check if a string addition will definitely result in creating a ConsString,
143 // i.e. if the combined length of the resulting string exceeds the ConsString
144 // minimum length.
145 bool ShouldCreateConsString() {
146 DCHECK_EQ(IrOpcode::kJSAdd, node_->opcode());
147 DCHECK(OneInputIs(Type::String()));
148 if (BothInputsAre(Type::String()) ||
149 GetBinaryOperationHint(node_) == BinaryOperationHint::kString) {
150 HeapObjectBinopMatcher m(node_);
151 JSHeapBroker* broker = lowering_->broker();
152 if (m.right().HasResolvedValue() && m.right().Ref(broker).IsString()) {
153 StringRef right_string = m.right().Ref(broker).AsString();
154 if (right_string.length() >= ConsString::kMinLength) return true;
155 }
156 if (m.left().HasResolvedValue() && m.left().Ref(broker).IsString()) {
157 StringRef left_string = m.left().Ref(broker).AsString();
158 if (left_string.length() >= ConsString::kMinLength) {
159 // The invariant for ConsString requires the left hand side to be
160 // a sequential or external string if the right hand side is the
161 // empty string. Since we don't know anything about the right hand
162 // side here, we must ensure that the left hand side satisfy the
163 // constraints independent of the right hand side.
164 return left_string.IsSeqString() || left_string.IsExternalString();
165 }
166 }
167 }
168 return false;
169 }

321 // Remove all effect and control inputs and outputs to this node and change
322 // to the pure operator {op}.
323 Reduction ChangeToPureOperator(const Operator* op, Type type = Type::Any()) {
324 DCHECK_EQ(0, op->EffectInputCount());
325 DCHECK_EQ(false, OperatorProperties::HasContextInput(op));
326 DCHECK_EQ(0, op->ControlInputCount());
327 DCHECK_EQ(2, op->ValueInputCount());
328
329 // Remove the effects from the node, and update its effect/control usages.
330 if (node_->op()->EffectInputCount() > 0) {
331 lowering_->RelaxEffectsAndControls(node_);
332 }
333 // Remove the inputs corresponding to context, effect, and control.
334 NodeProperties::RemoveNonValueInputs(node_);
335 // Remove the feedback vector input, if applicable.
336 if (JSOperator::IsBinaryWithFeedback(node_->opcode())) {
337 node_->RemoveInput(JSBinaryOpNode::FeedbackVectorIndex());
338 }
339 // Finally, update the operator to the new one.
340 NodeProperties::ChangeOp(node_, op);
341
342 // TODO(jarin): Replace the explicit typing hack with a call to some method
343 // that encapsulates changing the operator and re-typing.
344 Type node_type = NodeProperties::GetType(node_);
345 NodeProperties::SetType(node_, Type::Intersect(node_type, type, zone()));
346
347 return lowering_->Changed(node_);
348 }

350 Reduction ChangeToSpeculativeOperator(const Operator* op, Type upper_bound) {
351 DCHECK_EQ(1, op->EffectInputCount());
352 DCHECK_EQ(1, op->EffectOutputCount());
353 DCHECK_EQ(false, OperatorProperties::HasContextInput(op));
354 DCHECK_EQ(1, op->ControlInputCount());
355 DCHECK_EQ(0, op->ControlOutputCount());
356 DCHECK_EQ(0, OperatorProperties::GetFrameStateInputCount(op));
357 DCHECK_EQ(2, op->ValueInputCount());
358
359 DCHECK_EQ(1, node_->op()->EffectInputCount());
360 DCHECK_EQ(1, node_->op()->EffectOutputCount());
361 DCHECK_EQ(1, node_->op()->ControlInputCount());
362
363 // Reconnect the control output to bypass the IfSuccess node and
364 // possibly disconnect from the IfException node.
365 lowering_->RelaxControls(node_);
366
367 // Remove the frame state and the context.
368 if (OperatorProperties::HasFrameStateInput(node_->op())) {
369 node_->RemoveInput(NodeProperties::FirstFrameStateIndex(node_));
370 }
371 node_->RemoveInput(NodeProperties::FirstContextIndex(node_));
372
373 // Remove the feedback vector input, if applicable.
374 if (JSOperator::IsBinaryWithFeedback(node_->opcode())) {
375 node_->RemoveInput(JSBinaryOpNode::FeedbackVectorIndex());
376 }
377 // Finally, update the operator to the new one.
378 NodeProperties::ChangeOp(node_, op);
379
380 // Update the type to number.
381 Type node_type = NodeProperties::GetType(node_);
382 NodeProperties::SetType(node_,
383 Type::Intersect(node_type, upper_bound, zone()));
384
385 return lowering_->Changed(node_);
386 }

586Reduction JSTypedLowering::GenerateStringAddition(
587 Node* node, Node* left, Node* right, Node* context, Node* frame_state,
588 Node** effect, Node** control, bool should_create_cons_string) {
589 // Compute the resulting length.
590 Node* left_length = graph()->NewNode(simplified()->StringLength(), left);
591 Node* right_length = graph()->NewNode(simplified()->StringLength(), right);
592 Node* length =
593 graph()->NewNode(simplified()->NumberAdd(), left_length, right_length);
594
595 PropertyCellRef string_length_protector =
596 MakeRef(broker(), factory()->string_length_protector());
597 string_length_protector.CacheAsProtector(broker());
598
599 if (string_length_protector.value(broker()).AsSmi() ==
600 Protectors::kProtectorValid) {
601 // We can just deoptimize if the {length} is out-of-bounds. Besides
602 // generating a shorter code sequence than the version below, this
603 // has the additional benefit of not holding on to the lazy {frame_state}
604 // and thus potentially reduces the number of live ranges and allows for
605 // more truncations.
606 length = *effect = graph()->NewNode(
607 simplified()->CheckBounds(FeedbackSource()), length,
608 jsgraph()->ConstantNoHole(String::kMaxLength + 1), *effect, *control);
609 } else {
610 // Check if we would overflow the allowed maximum string length.
611 Node* check =
612 graph()->NewNode(simplified()->NumberLessThanOrEqual(), length,
613 jsgraph()->ConstantNoHole(String::kMaxLength));
614 Node* branch =
615 graph()->NewNode(common()->Branch(BranchHint::kTrue), check, *control);
616 Node* if_false = graph()->NewNode(common()->IfFalse(), branch);
617 Node* efalse = *effect;
618 {
619 // Throw a RangeError in case of overflow.
620 Node* vfalse = efalse = if_false = graph()->NewNode(
621 javascript()->CallRuntime(Runtime::kThrowInvalidStringLength),
622 context, frame_state, efalse, if_false);
623
624 // Update potential {IfException} uses of {node} to point to the
625 // %ThrowInvalidStringLength runtime call node instead.
626 Node* on_exception = nullptr;
627 if (NodeProperties::IsExceptionalCall(node, &on_exception)) {
628 NodeProperties::ReplaceControlInput(on_exception, vfalse);
629 NodeProperties::ReplaceEffectInput(on_exception, efalse);
630 if_false = graph()->NewNode(common()->IfSuccess(), vfalse);
631 Revisit(on_exception);
632 }
633
634 // The above %ThrowInvalidStringLength runtime call is an unconditional
635 // throw, making it impossible to return a successful completion in this
636 // case. We simply connect the successful completion to the graph end.
637 if_false = graph()->NewNode(common()->Throw(), efalse, if_false);
638 MergeControlToEnd(graph(), common(), if_false);
639 }
640 *control = graph()->NewNode(common()->IfTrue(), branch);
641 length = *effect =
642 graph()->NewNode(common()->TypeGuard(type_cache_->kStringLengthType),
643 length, *effect, *control);
644 }
645 // TODO(bmeurer): Ideally this should always use StringConcat and decide to
646 // optimize to NewConsString later during SimplifiedLowering, but for that
647 // to work we need to know that it's safe to create a ConsString.
648 Operator const* const op = should_create_cons_string
649 ? simplified()->NewConsString()
650 : simplified()->StringConcat();
651 Node* value = graph()->NewNode(op, length, left, right);
652 ReplaceWithValue(node, value, *effect, *control);
653 return Replace(value);
654}

656Node* JSTypedLowering::UnwrapStringWrapper(Node* string_or_wrapper,
657 Node** effect, Node** control) {
658 Node* check =
659 graph()->NewNode(simplified()->ObjectIsString(), string_or_wrapper);
660 Node* branch = graph()->NewNode(common()->Branch(), check, *control);
661
662 Node* if_true = graph()->NewNode(common()->IfTrue(), branch);
663 Node* etrue = *effect;
664 Node* vtrue = string_or_wrapper;
665
666 // We just checked that the value is a string.
667 vtrue = etrue = graph()->NewNode(common()->TypeGuard(Type::String()), vtrue,

670 Node* if_false = graph()->NewNode(common()->IfFalse(), branch);
671 Node* efalse = *effect;

678 vfalse = efalse = graph()->NewNode(common()->TypeGuard(Type::String()),

682 *effect = graph()->NewNode(common()->EffectPhi(2), etrue, efalse, *control);
683

688Reduction JSTypedLowering::ReduceJSAdd(Node* node) {
689 JSBinopReduction r(this, node);
690 if (r.BothInputsAre(Type::Number())) {
691 // JSAdd(x:number, y:number) => NumberAdd(x, y)
692 return r.ChangeToPureOperator(simplified()->NumberAdd(), Type::Number());
693 }
694 if (r.BothInputsAre(Type::PlainPrimitive()) &&
695 r.NeitherInputCanBe(Type::StringOrReceiver())) {
696 // JSAdd(x:-string, y:-string) => NumberAdd(ToNumber(x), ToNumber(y))
697 r.ConvertInputsToNumber();
698 return r.ChangeToPureOperator(simplified()->NumberAdd(), Type::Number());
699 }
700
701 // Strength-reduce if one input is already known to be a string.
702 if (r.LeftInputIs(Type::String())) {
703 // JSAdd(x:string, y) => JSAdd(x, JSToString(y))
704 Reduction const reduction = ReduceJSToStringInput(r.right());
705 if (reduction.Changed()) {
706 NodeProperties::ReplaceValueInput(node, reduction.replacement(), 1);
707 }
708 } else if (r.RightInputIs(Type::String())) {
709 // JSAdd(x, y:string) => JSAdd(JSToString(x), y)
710 Reduction const reduction = ReduceJSToStringInput(r.left());
711 if (reduction.Changed()) {
712 NodeProperties::ReplaceValueInput(node, reduction.replacement(), 0);
713 }
714 }
715
716 PropertyCellRef to_primitive_protector =
717 MakeRef(broker(), factory()->string_wrapper_to_primitive_protector());
718 to_primitive_protector.CacheAsProtector(broker());
719 bool can_inline_string_wrapper_add = false;
720
721 // Always bake in String feedback into the graph.
722 if (r.GetBinaryOperationHint(node) == BinaryOperationHint::kString) {
723 r.CheckInputsToString();
724 } else if (r.GetBinaryOperationHint(node) ==
725 BinaryOperationHint::kStringOrStringWrapper) {
726 can_inline_string_wrapper_add =
727 dependencies()->DependOnProtector(to_primitive_protector);
728 if (can_inline_string_wrapper_add) {
729 r.CheckInputsToStringOrStringWrapper();
730 }
731 }
732
733 // Strength-reduce concatenation of empty strings if both sides are
734 // primitives, as in that case the ToPrimitive on the other side is
735 // definitely going to be a no-op.
736 if (r.BothInputsAre(Type::Primitive())) {
737 if (r.LeftInputIs(empty_string_type_)) {
738 // JSAdd("", x:primitive) => JSToString(x)
739 NodeProperties::ReplaceValueInputs(node, r.right());
740 NodeProperties::ChangeOp(node, javascript()->ToString());
741 NodeProperties::SetType(
742 node, Type::Intersect(r.type(), Type::String(), graph()->zone()));
743 return Changed(node).FollowedBy(ReduceJSToString(node));
744 } else if (r.RightInputIs(empty_string_type_)) {
745 // JSAdd(x:primitive, "") => JSToString(x)
746 NodeProperties::ReplaceValueInputs(node, r.left());
747 NodeProperties::ChangeOp(node, javascript()->ToString());
748 NodeProperties::SetType(
749 node, Type::Intersect(r.type(), Type::String(), graph()->zone()));
750 return Changed(node).FollowedBy(ReduceJSToString(node));
751 }
752 }
753
754 Node* context = NodeProperties::GetContextInput(node);
755 Node* frame_state = NodeProperties::GetFrameStateInput(node);
756 Node* effect = NodeProperties::GetEffectInput(node);
757 Node* control = NodeProperties::GetControlInput(node);
758
759 // Lower to string addition if both inputs are known to be strings.
760 if (r.BothInputsAre(Type::String())) {
761 return GenerateStringAddition(node, r.left(), r.right(), context,
762 frame_state, &effect, &control,
763 r.ShouldCreateConsString());
764 } else if (r.BothInputsAre(Type::StringOrStringWrapper()) &&
765 can_inline_string_wrapper_add) {
766 // If the left hand side is a string wrapper, unwrap it.
767 Node* left_string = UnwrapStringWrapper(r.left(), &effect, &control);
768
769 // If the right hand side is a string wrapper, unwrap it.
770 Node* right_string = UnwrapStringWrapper(r.right(), &effect, &control);
771
772 // Generate the string addition.
773 return GenerateStringAddition(node, left_string, right_string, context,
774 frame_state, &effect, &control, false);
775 }
776
777 // We never get here when we had String feedback.
778 DCHECK_NE(BinaryOperationHint::kString, r.GetBinaryOperationHint(node));
779 if (r.OneInputIs(Type::String())) {
780 StringAddFlags flags = STRING_ADD_CHECK_NONE;
781 if (!r.LeftInputIs(Type::String())) {
782 flags = STRING_ADD_CONVERT_LEFT;
783 } else if (!r.RightInputIs(Type::String())) {
784 flags = STRING_ADD_CONVERT_RIGHT;
785 }
786 Operator::Properties properties = node->op()->properties();
787 if (r.NeitherInputCanBe(Type::Receiver())) {
788 // Both sides are already strings, so we know that the
789 // string addition will not cause any observable side
790 // effects; it can still throw obviously.
791 properties = Operator::kNoWrite | Operator::kNoDeopt;
792 }
793
794 // JSAdd(x:string, y) => CallStub[StringAdd](x, y)
795 // JSAdd(x, y:string) => CallStub[StringAdd](x, y)
796 Callable const callable = CodeFactory::StringAdd(isolate(), flags);
797 auto call_descriptor = Linkage::GetStubCallDescriptor(
798 graph()->zone(), callable.descriptor(),
799 callable.descriptor().GetStackParameterCount(),
800 CallDescriptor::kNeedsFrameState, properties);
801 DCHECK_EQ(1, OperatorProperties::GetFrameStateInputCount(node->op()));
802 node->RemoveInput(JSAddNode::FeedbackVectorIndex());
803 node->InsertInput(graph()->zone(), 0,
804 jsgraph()->HeapConstantNoHole(callable.code()));
805 NodeProperties::ChangeOp(node, common()->Call(call_descriptor));
806 return Changed(node);
807 }
808 return NoChange();
809}

842Reduction JSTypedLowering::ReduceJSComparison(Node* node) {
843 JSBinopReduction r(this, node);
844 if (r.BothInputsAre(Type::String())) {
845 // If both inputs are definitely strings, perform a string comparison.
846 const Operator* stringOp;
847 switch (node->opcode()) {
848 case IrOpcode::kJSLessThan:
849 stringOp = simplified()->StringLessThan();
850 break;
851 case IrOpcode::kJSGreaterThan:
852 stringOp = simplified()->StringLessThan();
853 r.SwapInputs(); // a > b => b < a
854 break;
855 case IrOpcode::kJSLessThanOrEqual:
856 stringOp = simplified()->StringLessThanOrEqual();
857 break;
858 case IrOpcode::kJSGreaterThanOrEqual:
859 stringOp = simplified()->StringLessThanOrEqual();
860 r.SwapInputs(); // a >= b => b <= a
861 break;
862 default:
863 return NoChange();
864 }
865 r.ChangeToPureOperator(stringOp);
866 return Changed(node);
867 }
868
869 const Operator* less_than;
870 const Operator* less_than_or_equal;
871 if (r.BothInputsAre(Type::Signed32()) ||
872 r.BothInputsAre(Type::Unsigned32())) {
873 less_than = simplified()->NumberLessThan();
874 less_than_or_equal = simplified()->NumberLessThanOrEqual();
875 } else if (r.OneInputCannotBe(Type::StringOrReceiver()) &&
876 r.BothInputsAre(Type::PlainPrimitive())) {
877 r.ConvertInputsToNumber();
878 less_than = simplified()->NumberLessThan();
879 less_than_or_equal = simplified()->NumberLessThanOrEqual();
880 } else if (r.IsStringCompareOperation()) {
881 r.CheckInputsToString();
882 less_than = simplified()->StringLessThan();
883 less_than_or_equal = simplified()->StringLessThanOrEqual();
884 } else {
885 return NoChange();
886 }
887 const Operator* comparison;
888 switch (node->opcode()) {
889 case IrOpcode::kJSLessThan:
890 comparison = less_than;
891 break;
892 case IrOpcode::kJSGreaterThan:
893 comparison = less_than;
894 r.SwapInputs(); // a > b => b < a
895 break;
896 case IrOpcode::kJSLessThanOrEqual:
897 comparison = less_than_or_equal;
898 break;
899 case IrOpcode::kJSGreaterThanOrEqual:
900 comparison = less_than_or_equal;
901 r.SwapInputs(); // a >= b => b <= a
902 break;
903 default:
904 return NoChange();
905 }
906 return r.ChangeToPureOperator(comparison);
907}

909Reduction JSTypedLowering::ReduceJSEqual(Node* node) {
910 JSBinopReduction r(this, node);
911
912 if (r.BothInputsAre(Type::UniqueName())) {
913 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
914 }
915 if (r.IsInternalizedStringCompareOperation()) {
916 r.CheckInputsToInternalizedString();
917 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
918 }
919 if (r.BothInputsAre(Type::String())) {
920 return r.ChangeToPureOperator(simplified()->StringEqual());
921 }
922 if (r.BothInputsAre(Type::Boolean())) {
923 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
924 }
925 if (r.BothInputsAre(Type::Receiver())) {
926 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
927 }
928 if (r.OneInputIs(Type::NullOrUndefined())) {
929 RelaxEffectsAndControls(node);
930 node->RemoveInput(r.LeftInputIs(Type::NullOrUndefined()) ? 0 : 1);
931 node->TrimInputCount(1);
932 NodeProperties::ChangeOp(node, simplified()->ObjectIsUndetectable());
933 return Changed(node);
934 }
935
936 if (r.BothInputsAre(Type::Signed32()) ||
937 r.BothInputsAre(Type::Unsigned32())) {
938 return r.ChangeToPureOperator(simplified()->NumberEqual());
939 } else if (r.BothInputsAre(Type::Number())) {
940 return r.ChangeToPureOperator(simplified()->NumberEqual());
941 } else if (r.IsReceiverCompareOperation()) {
942 r.CheckInputsToReceiver();
943 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
944 } else if (r.IsReceiverOrNullOrUndefinedCompareOperation()) {
945 // Check that both inputs are Receiver, Null or Undefined.
946 r.CheckInputsToReceiverOrNullOrUndefined();
947
948 // If one side is known to be a detectable receiver now, we
949 // can simply perform reference equality here, since this
950 // known detectable receiver is going to only match itself.
951 if (r.OneInputIs(Type::DetectableReceiver())) {
952 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
953 }
954
955 // Known that both sides are Receiver, Null or Undefined, the
956 // abstract equality operation can be performed like this:
957 //
958 // if left == undefined || left == null
959 // then ObjectIsUndetectable(right)
960 // else if right == undefined || right == null
961 // then ObjectIsUndetectable(left)
962 // else ReferenceEqual(left, right)
963#define __ gasm.
964 JSGraphAssembler gasm(broker(), jsgraph(), jsgraph()->zone(),
965 BranchSemantics::kJS);
966 gasm.InitializeEffectControl(r.effect(), r.control());
967
968 auto lhs = TNode<Object>::UncheckedCast(r.left());
969 auto rhs = TNode<Object>::UncheckedCast(r.right());
970
971 auto done = __ MakeLabel(MachineRepresentation::kTagged);
972 auto check_undetectable = __ MakeLabel(MachineRepresentation::kTagged);
973
974 __ GotoIf(__ ReferenceEqual(lhs, __ UndefinedConstant()),
975 &check_undetectable, rhs);
976 __ GotoIf(__ ReferenceEqual(lhs, __ NullConstant()), &check_undetectable,
977 rhs);
978 __ GotoIf(__ ReferenceEqual(rhs, __ UndefinedConstant()),
979 &check_undetectable, lhs);
980 __ GotoIf(__ ReferenceEqual(rhs, __ NullConstant()), &check_undetectable,
981 lhs);
982 __ Goto(&done, __ ReferenceEqual(lhs, rhs));
983
984 __ Bind(&check_undetectable);
985 __ Goto(&done,
986 __ ObjectIsUndetectable(check_undetectable.PhiAt<Object>(0)));
987
988 __ Bind(&done);
989 Node* value = done.PhiAt(0);
990 ReplaceWithValue(node, value, gasm.effect(), gasm.control());
991 return Replace(value);
992#undef __
993 } else if (r.IsStringCompareOperation()) {
994 r.CheckInputsToString();
995 return r.ChangeToPureOperator(simplified()->StringEqual());
996 } else if (r.IsSymbolCompareOperation()) {
997 r.CheckInputsToSymbol();
998 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
999 }
1000 return NoChange();
1001}

1003Reduction JSTypedLowering::ReduceJSStrictEqual(Node* node) {
1004 JSBinopReduction r(this, node);
1005 if (r.type().IsSingleton()) {
1006 // Let ConstantFoldingReducer handle this.
1007 return NoChange();
1008 }
1009 if (r.left() == r.right()) {
1010 // x === x is always true if x != NaN
1011 Node* replacement = graph()->NewNode(
1012 simplified()->BooleanNot(),
1013 graph()->NewNode(simplified()->ObjectIsNaN(), r.left()));
1014 DCHECK(NodeProperties::GetType(replacement).Is(r.type()));
1015 ReplaceWithValue(node, replacement);
1016 return Replace(replacement);
1017 }
1018
1019 if (r.BothInputsAre(Type::Unique())) {
1020 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
1021 }
1022 if (r.OneInputIs(pointer_comparable_type_)) {
1023 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
1024 }
1025 if (r.IsInternalizedStringCompareOperation()) {
1026 r.CheckInputsToInternalizedString();
1027 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
1028 }
1029 if (r.BothInputsAre(Type::String())) {
1030 return r.ChangeToPureOperator(simplified()->StringEqual());
1031 }
1032
1033 NumberOperationHint hint;
1034 BigIntOperationHint hint_bigint;
1035 if (r.BothInputsAre(Type::Signed32()) ||
1036 r.BothInputsAre(Type::Unsigned32())) {
1037 return r.ChangeToPureOperator(simplified()->NumberEqual());
1038 } else if (r.GetCompareNumberOperationHint(&hint) &&
1039 hint != NumberOperationHint::kNumberOrOddball &&
1040 hint != NumberOperationHint::kNumberOrBoolean) {
1041 // SpeculativeNumberEqual performs implicit conversion of oddballs to
1042 // numbers, so me must not generate it for strict equality with respective
1043 // hint.
1044 DCHECK(hint == NumberOperationHint::kNumber ||
1045 hint == NumberOperationHint::kSignedSmall);
1046 return r.ChangeToSpeculativeOperator(
1047 simplified()->SpeculativeNumberEqual(hint), Type::Boolean());
1048 } else if (r.BothInputsAre(Type::Number())) {
1049 return r.ChangeToPureOperator(simplified()->NumberEqual());
1050 } else if (r.GetCompareBigIntOperationHint(&hint_bigint)) {
1051 DCHECK(hint_bigint == BigIntOperationHint::kBigInt ||
1052 hint_bigint == BigIntOperationHint::kBigInt64);
1053 return r.ChangeToSpeculativeOperator(
1054 simplified()->SpeculativeBigIntEqual(hint_bigint), Type::Boolean());
1055 } else if (r.IsReceiverCompareOperation()) {
1056 // For strict equality, it's enough to know that one input is a Receiver,
1057 // as a strict equality comparison with a Receiver can only yield true if
1058 // both sides refer to the same Receiver.
1059 r.CheckLeftInputToReceiver();
1060 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
1061 } else if (r.IsReceiverOrNullOrUndefinedCompareOperation()) {
1062 // For strict equality, it's enough to know that one input is a Receiver,
1063 // Null or Undefined, as a strict equality comparison with a Receiver,
1064 // Null or Undefined can only yield true if both sides refer to the same
1065 // instance.
1066 r.CheckLeftInputToReceiverOrNullOrUndefined();
1067 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
1068 } else if (r.IsStringCompareOperation()) {
1069 r.CheckInputsToString();
1070 return r.ChangeToPureOperator(simplified()->StringEqual());
1071 } else if (r.IsSymbolCompareOperation()) {
1072 // For strict equality, it's enough to know that one input is a Symbol,
1073 // as a strict equality comparison with a Symbol can only yield true if
1074 // both sides refer to the same Symbol.
1075 r.CheckLeftInputToSymbol();
1076 return r.ChangeToPureOperator(simplified()->ReferenceEqual());
1077 }
1078 return NoChange();
1079}

1222 if (input->opcode() == IrOpcode::kJSToString) {
1223 // Recursively try to reduce the input first.
1224 Reduction result = ReduceJSToString(input);
1225 if (result.Changed()) return result;
1226 return Changed(input); // JSToString(JSToString(x)) => JSToString(x)
1227 }
1228 Type input_type = NodeProperties::GetType(input);
1229 if (input_type.Is(Type::String())) {
1230 return Changed(input); // JSToString(x:string) => x
1231 }
1232 if (input_type.Is(Type::Boolean())) {
1233 return Replace(graph()->NewNode(
1234 common()->Select(MachineRepresentation::kTagged), input,
1235 jsgraph()->HeapConstantNoHole(factory()->true_string()),
1236 jsgraph()->HeapConstantNoHole(factory()->false_string())));
1237 }
1238 if (input_type.Is(Type::Undefined())) {
1239 return Replace(
1240 jsgraph()->HeapConstantNoHole(factory()->undefined_string()));
1241 }
1242 if (input_type.Is(Type::Null())) {
1243 return Replace(jsgraph()->HeapConstantNoHole(factory()->null_string()));
1244 }
1245 if (input_type.Is(Type::NaN())) {
1246 return Replace(jsgraph()->HeapConstantNoHole(factory()->NaN_string()));
1247 }
1248 if (input_type.Is(Type::Number())) {
1249 return Replace(graph()->NewNode(simplified()->NumberToString(), input));
1250 }
1251 return NoChange();
1252}

1266Reduction JSTypedLowering::ReduceJSToObject(Node* node) {
1267 DCHECK_EQ(IrOpcode::kJSToObject, node->opcode());
1268 Node* receiver = NodeProperties::GetValueInput(node, 0);
1269 Type receiver_type = NodeProperties::GetType(receiver);
1270 Node* context = NodeProperties::GetContextInput(node);
1271 Node* frame_state = NodeProperties::GetFrameStateInput(node);
1272 Node* effect = NodeProperties::GetEffectInput(node);
1273 Node* control = NodeProperties::GetControlInput(node);
1274 if (receiver_type.Is(Type::Receiver())) {
1275 ReplaceWithValue(node, receiver, effect, control);
1276 return Replace(receiver);
1277 }
1278
1279 // Check whether {receiver} is a spec object.
1280 Node* check = graph()->NewNode(simplified()->ObjectIsReceiver(), receiver);
1281 Node* branch =
1282 graph()->NewNode(common()->Branch(BranchHint::kTrue), check, control);
1283
1284 Node* if_true = graph()->NewNode(common()->IfTrue(), branch);
1285 Node* etrue = effect;
1286 Node* rtrue = receiver;
1287
1288 Node* if_false = graph()->NewNode(common()->IfFalse(), branch);
1289 Node* efalse = effect;
1290 Node* rfalse;
1291 {
1292 // Convert {receiver} using the ToObjectStub.
1293 Callable callable = Builtins::CallableFor(isolate(), Builtin::kToObject);
1294 auto call_descriptor = Linkage::GetStubCallDescriptor(
1295 graph()->zone(), callable.descriptor(),
1296 callable.descriptor().GetStackParameterCount(),
1297 CallDescriptor::kNeedsFrameState, node->op()->properties());
1298 Node* call = rfalse = efalse = if_false =
1299 graph()->NewNode(common()->Call(call_descriptor),
1300 jsgraph()->HeapConstantNoHole(callable.code()),
1301 receiver, context, frame_state, efalse, if_false);
1302
1303 // We preserve the type of {node}. This is generally useful (to enable
1304 // type-based optimizations), and is also required in order to help
1305 // verification of TypeGuards.
1306 NodeProperties::SetType(call, NodeProperties::GetType(node));
1307 }
1308
1309 // Update potential {IfException} uses of {node} to point to the above
1310 // ToObject stub call node instead. Note that the stub can only throw on
1311 // receivers that can be null or undefined.
1312 Node* on_exception = nullptr;
1313 if (receiver_type.Maybe(Type::NullOrUndefined()) &&
1314 NodeProperties::IsExceptionalCall(node, &on_exception)) {
1315 NodeProperties::ReplaceControlInput(on_exception, if_false);
1316 NodeProperties::ReplaceEffectInput(on_exception, efalse);
1317 if_false = graph()->NewNode(common()->IfSuccess(), if_false);
1318 Revisit(on_exception);
1319 }
1320
1321 control = graph()->NewNode(common()->Merge(2), if_true, if_false);
1322 effect = graph()->NewNode(common()->EffectPhi(2), etrue, efalse, control);
1323
1324 // Morph the {node} into an appropriate Phi.
1325 ReplaceWithValue(node, node, effect, control);
1326 node->ReplaceInput(0, rtrue);
1327 node->ReplaceInput(1, rfalse);
1328 node->ReplaceInput(2, control);
1329 node->TrimInputCount(3);
1330 NodeProperties::ChangeOp(node,
1331 common()->Phi(MachineRepresentation::kTagged, 2));
1332 return Changed(node);
1333}

1350Reduction JSTypedLowering::ReduceJSHasInPrototypeChain(Node* node) {
1351 DCHECK_EQ(IrOpcode::kJSHasInPrototypeChain, node->opcode());
1352 Node* value = NodeProperties::GetValueInput(node, 0);
1353 Type value_type = NodeProperties::GetType(value);
1354 Node* prototype = NodeProperties::GetValueInput(node, 1);
1355 Node* context = NodeProperties::GetContextInput(node);
1356 Node* frame_state = NodeProperties::GetFrameStateInput(node);
1357 Node* effect = NodeProperties::GetEffectInput(node);
1358 Node* control = NodeProperties::GetControlInput(node);
1359
1360 // If {value} cannot be a receiver, then it cannot have {prototype} in
1361 // it's prototype chain (all Primitive values have a null prototype).
1362 if (value_type.Is(Type::Primitive())) {
1363 value = jsgraph()->FalseConstant();
1364 ReplaceWithValue(node, value, effect, control);
1365 return Replace(value);
1366 }
1367
1368 Node* check0 = graph()->NewNode(simplified()->ObjectIsSmi(), value);
1369 Node* branch0 =
1370 graph()->NewNode(common()->Branch(BranchHint::kFalse), check0, control);
1371
1372 Node* if_true0 = graph()->NewNode(common()->IfTrue(), branch0);
1373 Node* etrue0 = effect;
1374 Node* vtrue0 = jsgraph()->FalseConstant();
1375
1376 control = graph()->NewNode(common()->IfFalse(), branch0);
1377
1378 // Loop through the {value}s prototype chain looking for the {prototype}.
1379 Node* loop = control = graph()->NewNode(common()->Loop(2), control, control);
1380 Node* eloop = effect =
1381 graph()->NewNode(common()->EffectPhi(2), effect, effect, loop);
1382 Node* terminate = graph()->NewNode(common()->Terminate(), eloop, loop);
1383 MergeControlToEnd(graph(), common(), terminate);
1384 Node* vloop = value = graph()->NewNode(
1385 common()->Phi(MachineRepresentation::kTagged, 2), value, value, loop);
1386 NodeProperties::SetType(vloop, Type::NonInternal());
1387
1388 // Load the {value} map and instance type.
1389 Node* value_map = effect = graph()->NewNode(
1390 simplified()->LoadField(AccessBuilder::ForMap()), value, effect, control);
1391 Node* value_instance_type = effect = graph()->NewNode(
1392 simplified()->LoadField(AccessBuilder::ForMapInstanceType()), value_map,
1393 effect, control);
1394
1395 // Check if the {value} is a special receiver, because for special
1396 // receivers, i.e. proxies or API values that need access checks,
1397 // we have to use the %HasInPrototypeChain runtime function instead.
1398 Node* check1 = graph()->NewNode(
1399 simplified()->NumberLessThanOrEqual(), value_instance_type,
1400 jsgraph()->ConstantNoHole(LAST_SPECIAL_RECEIVER_TYPE));
1401 Node* branch1 =
1402 graph()->NewNode(common()->Branch(BranchHint::kFalse), check1, control);
1403
1404 control = graph()->NewNode(common()->IfFalse(), branch1);
1405
1406 Node* if_true1 = graph()->NewNode(common()->IfTrue(), branch1);
1407 Node* etrue1 = effect;
1408 Node* vtrue1;
1409
1410 // Check if the {value} is not a receiver at all.
1411 Node* check10 =
1412 graph()->NewNode(simplified()->NumberLessThan(), value_instance_type,
1413 jsgraph()->ConstantNoHole(FIRST_JS_RECEIVER_TYPE));
1414 Node* branch10 =
1415 graph()->NewNode(common()->Branch(BranchHint::kTrue), check10, if_true1);
1416
1417 // A primitive value cannot match the {prototype} we're looking for.
1418 if_true1 = graph()->NewNode(common()->IfTrue(), branch10);
1419 vtrue1 = jsgraph()->FalseConstant();
1420
1421 Node* if_false1 = graph()->NewNode(common()->IfFalse(), branch10);
1422 Node* efalse1 = etrue1;
1423 Node* vfalse1;
1424 {
1425 // Slow path, need to call the %HasInPrototypeChain runtime function.
1426 vfalse1 = efalse1 = if_false1 = graph()->NewNode(
1427 javascript()->CallRuntime(Runtime::kHasInPrototypeChain), value,
1428 prototype, context, frame_state, efalse1, if_false1);
1429
1430 // Replace any potential {IfException} uses of {node} to catch
1431 // exceptions from this %HasInPrototypeChain runtime call instead.
1432 Node* on_exception = nullptr;
1433 if (NodeProperties::IsExceptionalCall(node, &on_exception)) {
1434 NodeProperties::ReplaceControlInput(on_exception, vfalse1);
1435 NodeProperties::ReplaceEffectInput(on_exception, efalse1);
1436 if_false1 = graph()->NewNode(common()->IfSuccess(), vfalse1);
1437 Revisit(on_exception);
1438 }
1439 }
1440
1441 // Load the {value} prototype.
1442 Node* value_prototype = effect = graph()->NewNode(
1443 simplified()->LoadField(AccessBuilder::ForMapPrototype()), value_map,
1444 effect, control);
1445
1446 // Check if we reached the end of {value}s prototype chain.
1447 Node* check2 = graph()->NewNode(simplified()->ReferenceEqual(),
1448 value_prototype, jsgraph()->NullConstant());
1449 Node* branch2 = graph()->NewNode(common()->Branch(), check2, control);
1450
1451 Node* if_true2 = graph()->NewNode(common()->IfTrue(), branch2);
1452 Node* etrue2 = effect;
1453 Node* vtrue2 = jsgraph()->FalseConstant();
1454
1455 control = graph()->NewNode(common()->IfFalse(), branch2);
1456
1457 // Check if we reached the {prototype}.
1458 Node* check3 = graph()->NewNode(simplified()->ReferenceEqual(),
1459 value_prototype, prototype);
1460 Node* branch3 = graph()->NewNode(common()->Branch(), check3, control);
1461
1462 Node* if_true3 = graph()->NewNode(common()->IfTrue(), branch3);
1463 Node* etrue3 = effect;
1464 Node* vtrue3 = jsgraph()->TrueConstant();
1465
1466 control = graph()->NewNode(common()->IfFalse(), branch3);
1467
1468 // Close the loop.
1469 vloop->ReplaceInput(1, value_prototype);
1470 eloop->ReplaceInput(1, effect);
1471 loop->ReplaceInput(1, control);
1472
1473 control = graph()->NewNode(common()->Merge(5), if_true0, if_true1, if_true2,
1474 if_true3, if_false1);
1475 effect = graph()->NewNode(common()->EffectPhi(5), etrue0, etrue1, etrue2,
1476 etrue3, efalse1, control);
1477
1478 // Morph the {node} into an appropriate Phi.
1479 ReplaceWithValue(node, node, effect, control);
1480 node->ReplaceInput(0, vtrue0);
1481 node->ReplaceInput(1, vtrue1);
1482 node->ReplaceInput(2, vtrue2);
1483 node->ReplaceInput(3, vtrue3);
1484 node->ReplaceInput(4, vfalse1);
1485 node->ReplaceInput(5, control);
1486 node->TrimInputCount(6);
1487 NodeProperties::ChangeOp(node,
1488 common()->Phi(MachineRepresentation::kTagged, 5));
1489 return Changed(node);
1490}

1515 }
1516
1517 return NoChange();

1781 data, TNode<Object>::UncheckedCast(
1782 gasm.SmiConstant(ContextSidePropertyCell::kOther)));
1783 gasm.If(is_other)
1784 .Then([&] {
1785 gasm.StoreField(AccessBuilder::ForContextSlot(access.index()),
1786 context, new_value);
1787 })
1788 .Else([&] {
1789 gasm.CheckIf(gasm.BooleanNot(gasm.IsUndefined(data)),
1790 DeoptimizeReason::kWrongValue);
1791 TNode<Object> property =
1792 gasm.SelectIf<Object>(gasm.ObjectIsSmi(data))
1793 .Then([&] { return data; })
1794 .Else([&] {
1795 return gasm.LoadField<Object>(
1796 AccessBuilder::ForContextSideProperty(),
1797 TNode<HeapObject>::UncheckedCast(data));
1798 })
1799 .Value();
1800 TNode<Boolean> is_const = gasm.ReferenceEqual(
1801 property, TNode<Object>::UncheckedCast(
1802 gasm.SmiConstant(ContextSidePropertyCell::kConst)));
1803 gasm.CheckIf(gasm.BooleanNot(is_const),
1804 DeoptimizeReason::kWrongValue);
1805 if (v8_flags.script_context_mutable_heap_number) {
1806 TNode<Boolean> is_smi_marker = gasm.ReferenceEqual(
1807 property, TNode<Object>::UncheckedCast(
1808 gasm.SmiConstant(ContextSidePropertyCell::kSmi)));
1809 gasm.If(is_smi_marker)
1810 .Then([&] {
1811 Node* smi_value = gasm.CheckSmi(new_value);
1812 gasm.StoreField(
1813 AccessBuilder::ForContextSlotSmi(access.index()), context,
1814 smi_value);
1815 })
1816 .Else([&] {
1817 if (v8_flags.script_context_mutable_heap_int32) {
1818 TNode<Boolean> is_mutable_int32 = gasm.ReferenceEqual(
1819 property, TNode<Object>::UncheckedCast(gasm.SmiConstant(
1820 ContextSidePropertyCell::kMutableInt32)));
1821 gasm.If(is_mutable_int32)
1822 .Then([&] {
1823 // It is a mutable int32 number.
1824 Node* number_value =
1825 gasm.CheckNumberFitsInt32(new_value);
1826 gasm.StoreField(AccessBuilder::ForHeapInt32Value(),
1827 old_value, number_value);
1828 })
1829 .Else([&] {
1830 // It must be a mutable heap number in this case.
1831 Node* number_value = gasm.CheckNumber(new_value);
1832 gasm.StoreField(AccessBuilder::ForHeapNumberValue(),
1833 old_value, number_value);
1834 });
1835
1836 } else {
1837 // It must be a mutable heap number in this case.
1838 Node* number_value = gasm.CheckNumber(new_value);
1839 gasm.StoreField(AccessBuilder::ForHeapNumberValue(),
1840 old_value, number_value);
1841 }
1842 });
1843
1844 } else {
1845 gasm.StoreField(AccessBuilder::ForContextSlot(access.index()),
1846 context, new_value);
1847 }
1848 })
1849 .ExpectTrue();
1850 });
1851 ReplaceWithValue(node, gasm.effect(), gasm.effect(), gasm.control());
1852 return Changed(node);
1853}

1929void ReduceBuiltin(JSGraph* jsgraph, Node* node, Builtin builtin, int arity,
1930 CallDescriptor::Flags flags) {
1931 // Patch {node} to a direct CEntry call.
1932 // ----------- A r g u m e n t s -----------
1933 // -- 0: CEntry
1934 // --- Stack args ---
1935 // -- 1: new_target
1936 // -- 2: target
1937 // -- 3: argc, including the receiver and implicit args (Smi)
1938 // -- 4: padding
1939 // -- 5: receiver
1940 // -- [6, 6 + n[: the n actual arguments passed to the builtin
1941 // --- Register args ---
1942 // -- 6 + n: the C entry point
1943 // -- 6 + n + 1: argc (Int32)
1944 // -----------------------------------
1945
1946 // These SBXCHECKs are a defense-in-depth measure to ensure that we always
1947 // generate valid calls here (with matching signatures).
1948 SBXCHECK(Builtins::IsCpp(builtin));
1949 SBXCHECK_GE(arity + kJSArgcReceiverSlots,
1950 Builtins::GetFormalParameterCount(builtin));
1951
1952 // The logic contained here is mirrored in Builtins::Generate_Adaptor.
1953 // Keep these in sync.
1954
1955 Node* target = node->InputAt(JSCallOrConstructNode::TargetIndex());
1956
1957 // Unify representations between construct and call nodes. For construct
1958 // nodes, the receiver is undefined. For call nodes, the new_target is
1959 // undefined.
1960 Node* new_target;
1961 Zone* zone = jsgraph->zone();
1962 if (node->opcode() == IrOpcode::kJSConstruct) {
1963 static_assert(JSCallNode::ReceiverIndex() ==
1964 JSConstructNode::NewTargetIndex());
1965 new_target = JSConstructNode{node}.new_target();
1966 node->ReplaceInput(JSConstructNode::NewTargetIndex(),
1967 jsgraph->UndefinedConstant());
1968 node->RemoveInput(JSConstructNode{node}.FeedbackVectorIndex());
1969 } else {
1970 new_target = jsgraph->UndefinedConstant();
1971 node->RemoveInput(JSCallNode{node}.FeedbackVectorIndex());
1972 }
1973
1974 // CPP builtins are implemented in C++, and we can inline it.
1975 // CPP builtins create a builtin exit frame.
1976 const bool has_builtin_exit_frame = true;
1977
1978 Node* stub =
1979 jsgraph->CEntryStubConstant(1, ArgvMode::kStack, has_builtin_exit_frame);
1980 node->ReplaceInput(0, stub);
1981
1982 const int argc = arity + BuiltinArguments::kNumExtraArgsWithReceiver;
1983 Node* argc_node = jsgraph->ConstantNoHole(argc);
1984
1985 static const int kStub = 1;
1986 static_assert(BuiltinArguments::kNewTargetIndex == 0);
1987 static_assert(BuiltinArguments::kTargetIndex == 1);
1988 static_assert(BuiltinArguments::kArgcIndex == 2);
1989 static_assert(BuiltinArguments::kPaddingIndex == 3);
1990 node->InsertInput(zone, 1, new_target);
1991 node->InsertInput(zone, 2, target);
1992 node->InsertInput(zone, 3, argc_node);
1993 node->InsertInput(zone, 4, jsgraph->PaddingConstant());
1994 int cursor = arity + kStub + BuiltinArguments::kNumExtraArgsWithReceiver;
1995
1996 Address entry = Builtins::CppEntryOf(builtin);
1997 ExternalReference entry_ref = ExternalReference::Create(entry);
1998 Node* entry_node = jsgraph->ExternalConstant(entry_ref);
1999
2000 node->InsertInput(zone, cursor++, entry_node);
2001 node->InsertInput(zone, cursor++, argc_node);
2002
2003 static const int kReturnCount = 1;
2004 const char* debug_name = Builtins::name(builtin);
2005 Operator::Properties properties = node->op()->properties();
2006 auto call_descriptor = Linkage::GetCEntryStubCallDescriptor(
2007 zone, kReturnCount, argc, debug_name, properties, flags,
2008 StackArgumentOrder::kJS);
2009
2010 NodeProperties::ChangeOp(node, jsgraph->common()->Call(call_descriptor));
2011}

2125 Type receiver_type = NodeProperties::GetType(receiver);

2130 if (receiver_type.Is(Type::NullOrUndefined())) {
2131 convert_mode = ConvertReceiverMode::kNullOrUndefined;
2132 } else if (!receiver_type.Maybe(Type::NullOrUndefined())) {
2133 convert_mode = ConvertReceiverMode::kNotNullOrUndefined;
2134 }

2137 OptionalJSFunctionRef function;

2140 if (target_type.IsHeapConstant() &&
2141 target_type.AsHeapConstant()->Ref().IsJSFunction()) {
2142 function = target_type.AsHeapConstant()->Ref().AsJSFunction();
2143 shared = function->shared(broker());
2144 } else if (target->opcode() == IrOpcode::kJSCreateClosure) {

2146 JSCreateClosureNode{target}.Parameters();

2254 node->RemoveInput(n.FeedbackVectorIndex());

2257 jsgraph()->ConstantNoHole(JSParameterCount(arity)));

2360 // Don't need filtering since expected map still matches that of the
2361 // {receiver}.
2362 etrue = effect;
2363 vtrue = key;
2364 }
2365
2366 Node* if_false = graph()->NewNode(common()->IfFalse(), branch);
2367 Node* efalse;
2368 Node* vfalse;
2369 {
2370 // Filter the {key} to check if it's still a valid property of the
2371 // {receiver} (does the ToName conversion implicitly).
2372 Callable const callable =
2373 Builtins::CallableFor(isolate(), Builtin::kForInFilter);
2374 auto call_descriptor = Linkage::GetStubCallDescriptor(
2375 graph()->zone(), callable.descriptor(),
2376 callable.descriptor().GetStackParameterCount(),
2377 CallDescriptor::kNeedsFrameState);
2378 vfalse = efalse = if_false = graph()->NewNode(
2379 common()->Call(call_descriptor),
2380 jsgraph()->HeapConstantNoHole(callable.code()), key, receiver,
2381 context, frame_state, effect, if_false);
2382 NodeProperties::SetType(
2383 vfalse,
2384 Type::Union(Type::String(), Type::Undefined(), graph()->zone()));
2385
2386 // Update potential {IfException} uses of {node} to point to the above
2387 // ForInFilter stub call node instead.
2388 Node* if_exception = nullptr;
2389 if (NodeProperties::IsExceptionalCall(node, &if_exception)) {
2390 if_false = graph()->NewNode(common()->IfSuccess(), vfalse);
2391 NodeProperties::ReplaceControlInput(if_exception, vfalse);
2392 NodeProperties::ReplaceEffectInput(if_exception, efalse);
2393 Revisit(if_exception);
2394 }
2395 }
2396
2397 control = graph()->NewNode(common()->Merge(2), if_true, if_false);
2398 effect = graph()->NewNode(common()->EffectPhi(2), etrue, efalse, control);
2399 ReplaceWithValue(node, node, effect, control);
2400
2401 // Morph the {node} into a Phi.
2402 node->ReplaceInput(0, vtrue);
2403 node->ReplaceInput(1, vfalse);
2404 node->ReplaceInput(2, control);
2405 node->TrimInputCount(3);
2406 NodeProperties::ChangeOp(
2407 node, common()->Phi(MachineRepresentation::kTagged, 2));
2408 }
2409 }
2410
2411 return Changed(node);
2412}

2414Reduction JSTypedLowering::ReduceJSForInPrepare(Node* node) {
2415 JSForInPrepareNode n(node);
2416 Node* enumerator = n.enumerator();
2417 Effect effect = n.effect();
2418 Control control = n.control();
2419 Node* cache_type = enumerator;
2420 Node* cache_array = nullptr;
2421 Node* cache_length = nullptr;
2422
2423 switch (n.Parameters().mode()) {
2424 case ForInMode::kUseEnumCacheKeys:
2425 case ForInMode::kUseEnumCacheKeysAndIndices: {
2426 // Check that the {enumerator} is a Map.
2427 // The direct IsMap check requires reading of an instance type, so we
2428 // compare its map against fixed_array_map instead (by definition,
2429 // the {enumerator} is either the receiver's Map or a FixedArray).
2430 Node* check_for_fixed_array = effect =
2431 graph()->NewNode(simplified()->CompareMaps(
2432 ZoneRefSet<Map>(broker()->fixed_array_map())),
2433 enumerator, effect, control);
2434 Node* check_for_not_fixed_array =
2435 graph()->NewNode(simplified()->BooleanNot(), check_for_fixed_array);
2436 effect =
2437 graph()->NewNode(simplified()->CheckIf(DeoptimizeReason::kWrongMap),
2438 check_for_not_fixed_array, effect, control);
2439
2440 // Load the enum cache from the {enumerator} map.
2441 Node* descriptor_array = effect = graph()->NewNode(
2442 simplified()->LoadField(AccessBuilder::ForMapDescriptors()),
2443 enumerator, effect, control);
2444 Node* enum_cache = effect = graph()->NewNode(
2445 simplified()->LoadField(AccessBuilder::ForDescriptorArrayEnumCache()),
2446 descriptor_array, effect, control);
2447 cache_array = effect = graph()->NewNode(
2448 simplified()->LoadField(AccessBuilder::ForEnumCacheKeys()),
2449 enum_cache, effect, control);
2450
2451 // Load the enum length of the {enumerator} map.
2452 Node* bit_field3 = effect = graph()->NewNode(
2453 simplified()->LoadField(AccessBuilder::ForMapBitField3()), enumerator,
2454 effect, control);
2455 static_assert(Map::Bits3::EnumLengthBits::kShift == 0);
2456 cache_length = graph()->NewNode(
2457 simplified()->NumberBitwiseAnd(), bit_field3,
2458 jsgraph()->ConstantNoHole(Map::Bits3::EnumLengthBits::kMask));
2459 break;
2460 }
2461 case ForInMode::kGeneric: {
2462 // Check if the {enumerator} is a Map or a FixedArray.
2463 // The direct IsMap check requires reading of an instance type, so we
2464 // compare against fixed array map instead (by definition,
2465 // the {enumerator} is either the receiver's Map or a FixedArray).
2466 Node* check = effect =
2467 graph()->NewNode(simplified()->CompareMaps(
2468 ZoneRefSet<Map>(broker()->fixed_array_map())),
2469 enumerator, effect, control);
2470 Node* branch = graph()->NewNode(common()->Branch(BranchHint::kFalse),
2471 check, control);
2472
2473 Node* if_map = graph()->NewNode(common()->IfFalse(), branch);
2474 Node* etrue = effect;
2475 Node* cache_array_true;
2476 Node* cache_length_true;
2477 {
2478 // Load the enum cache from the {enumerator} map.
2479 Node* descriptor_array = etrue = graph()->NewNode(
2480 simplified()->LoadField(AccessBuilder::ForMapDescriptors()),
2481 enumerator, etrue, if_map);
2482 Node* enum_cache = etrue =
2483 graph()->NewNode(simplified()->LoadField(
2484 AccessBuilder::ForDescriptorArrayEnumCache()),
2485 descriptor_array, etrue, if_map);
2486 cache_array_true = etrue = graph()->NewNode(
2487 simplified()->LoadField(AccessBuilder::ForEnumCacheKeys()),
2488 enum_cache, etrue, if_map);
2489
2490 // Load the enum length of the {enumerator} map.
2491 Node* bit_field3 = etrue = graph()->NewNode(
2492 simplified()->LoadField(AccessBuilder::ForMapBitField3()),
2493 enumerator, etrue, if_map);
2494 static_assert(Map::Bits3::EnumLengthBits::kShift == 0);
2495 cache_length_true = graph()->NewNode(
2496 simplified()->NumberBitwiseAnd(), bit_field3,
2497 jsgraph()->ConstantNoHole(Map::Bits3::EnumLengthBits::kMask));
2498 }
2499
2500 Node* if_fixed_array = graph()->NewNode(common()->IfTrue(), branch);
2501 Node* efalse = effect;
2502 Node* cache_array_false;
2503 Node* cache_length_false;
2504 {
2505 // The {enumerator} is the FixedArray with the keys to iterate.
2506 cache_array_false = enumerator;
2507 cache_length_false = efalse = graph()->NewNode(
2508 simplified()->LoadField(AccessBuilder::ForFixedArrayLength()),
2509 cache_array_false, efalse, if_fixed_array);
2510 }
2511
2512 // Rewrite the uses of the {node}.
2513 control = graph()->NewNode(common()->Merge(2), if_map, if_fixed_array);
2514 effect = graph()->NewNode(common()->EffectPhi(2), etrue, efalse, control);
2515 cache_array =
2516 graph()->NewNode(common()->Phi(MachineRepresentation::kTagged, 2),
2517 cache_array_true, cache_array_false, control);
2518 cache_length =
2519 graph()->NewNode(common()->Phi(MachineRepresentation::kTagged, 2),
2520 cache_length_true, cache_length_false, control);
2521 break;
2522 }
2523 }
2524
2525 // Update the uses of {node}.
2526 for (Edge edge : node->use_edges()) {
2527 Node* const user = edge.from();
2528 if (NodeProperties::IsEffectEdge(edge)) {
2529 edge.UpdateTo(effect);
2530 Revisit(user);
2531 } else if (NodeProperties::IsControlEdge(edge)) {
2532 edge.UpdateTo(control);
2533 Revisit(user);
2534 } else {
2535 DCHECK(NodeProperties::IsValueEdge(edge));
2536 switch (ProjectionIndexOf(user->op())) {
2537 case 0:
2538 Replace(user, cache_type);
2539 break;
2540 case 1:
2541 Replace(user, cache_array);
2542 break;
2543 case 2:
2544 Replace(user, cache_length);
2545 break;
2546 default:
2547 UNREACHABLE();
2548 }
2549 }
2550 }
2551 node->Kill();
2552 return Replace(effect);
2553}

2690Reduction JSTypedLowering::ReduceObjectIsArray(Node* node) {
2691 Node* value = NodeProperties::GetValueInput(node, 0);
2692 Type value_type = NodeProperties::GetType(value);
2693 Node* context = NodeProperties::GetContextInput(node);
2694 Node* frame_state = NodeProperties::GetFrameStateInput(node);
2695 Node* effect = NodeProperties::GetEffectInput(node);
2696 Node* control = NodeProperties::GetControlInput(node);
2697
2698 // Constant-fold based on {value} type.
2699 if (value_type.Is(Type::Array())) {
2700 value = jsgraph()->TrueConstant();
2701 ReplaceWithValue(node, value);
2702 return Replace(value);
2703 } else if (!value_type.Maybe(Type::ArrayOrProxy())) {
2704 value = jsgraph()->FalseConstant();
2705 ReplaceWithValue(node, value);
2706 return Replace(value);
2707 }
2708
2709 int count = 0;
2710 Node* values[5];
2711 Node* effects[5];
2712 Node* controls[4];
2713
2714 // Check if the {value} is a Smi.
2715 Node* check = graph()->NewNode(simplified()->ObjectIsSmi(), value);
2716 control =
2717 graph()->NewNode(common()->Branch(BranchHint::kFalse), check, control);
2718
2719 // The {value} is a Smi.
2720 controls[count] = graph()->NewNode(common()->IfTrue(), control);
2721 effects[count] = effect;
2722 values[count] = jsgraph()->FalseConstant();
2723 count++;
2724
2725 control = graph()->NewNode(common()->IfFalse(), control);
2726
2727 // Load the {value}s instance type.
2728 Node* value_map = effect = graph()->NewNode(
2729 simplified()->LoadField(AccessBuilder::ForMap()), value, effect, control);
2730 Node* value_instance_type = effect = graph()->NewNode(
2731 simplified()->LoadField(AccessBuilder::ForMapInstanceType()), value_map,
2732 effect, control);
2733
2734 // Check if the {value} is a JSArray.
2735 check = graph()->NewNode(simplified()->NumberEqual(), value_instance_type,
2736 jsgraph()->ConstantNoHole(JS_ARRAY_TYPE));
2737 control = graph()->NewNode(common()->Branch(), check, control);
2738
2739 // The {value} is a JSArray.
2740 controls[count] = graph()->NewNode(common()->IfTrue(), control);
2741 effects[count] = effect;
2742 values[count] = jsgraph()->TrueConstant();
2743 count++;
2744
2745 control = graph()->NewNode(common()->IfFalse(), control);
2746
2747 // Check if the {value} is a JSProxy.
2748 check = graph()->NewNode(simplified()->NumberEqual(), value_instance_type,
2749 jsgraph()->ConstantNoHole(JS_PROXY_TYPE));
2750 control =
2751 graph()->NewNode(common()->Branch(BranchHint::kFalse), check, control);
2752
2753 // The {value} is neither a JSArray nor a JSProxy.
2754 controls[count] = graph()->NewNode(common()->IfFalse(), control);
2755 effects[count] = effect;
2756 values[count] = jsgraph()->FalseConstant();
2757 count++;
2758
2759 control = graph()->NewNode(common()->IfTrue(), control);
2760
2761 // Let the %ArrayIsArray runtime function deal with the JSProxy {value}.
2762 value = effect = control =
2763 graph()->NewNode(javascript()->CallRuntime(Runtime::kArrayIsArray), value,
2764 context, frame_state, effect, control);
2765 NodeProperties::SetType(value, Type::Boolean());
2766
2767 // Update potential {IfException} uses of {node} to point to the above
2768 // %ArrayIsArray runtime call node instead.
2769 Node* on_exception = nullptr;
2770 if (NodeProperties::IsExceptionalCall(node, &on_exception)) {
2771 NodeProperties::ReplaceControlInput(on_exception, control);
2772 NodeProperties::ReplaceEffectInput(on_exception, effect);
2773 control = graph()->NewNode(common()->IfSuccess(), control);
2774 Revisit(on_exception);
2775 }
2776
2777 // The {value} is neither a JSArray nor a JSProxy.
2778 controls[count] = control;
2779 effects[count] = effect;
2780 values[count] = value;
2781 count++;
2782
2783 control = graph()->NewNode(common()->Merge(count), count, controls);
2784 effects[count] = control;
2785 values[count] = control;
2786 effect = graph()->NewNode(common()->EffectPhi(count), count + 1, effects);
2787 value = graph()->NewNode(common()->Phi(MachineRepresentation::kTagged, count),
2788 count + 1, values);
2789 ReplaceWithValue(node, value, effect, control);
2790 return Replace(value);
2791}
