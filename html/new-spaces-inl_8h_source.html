63 : current_(space->first_allocatable_address()) {}

65Tagged<HeapObject> SemiSpaceObjectIterator::Next() {
66 while (true) {
67 if (PageMetadata::IsAlignedToPageSize(current_)) {
68 PageMetadata* page = PageMetadata::FromAllocationAreaAddress(current_);
69 page = page->next_page();
70 if (page == nullptr) return Tagged<HeapObject>();
71 current_ = page->area_start();
72 }
73 Tagged<HeapObject> object = HeapObject::FromAddress(current_);
74 current_ += ALIGN_TO_ALLOCATION_ALIGNMENT(object->Size());
75 if (!IsFreeSpaceOrFiller(object)) return object;
76 }
77}
