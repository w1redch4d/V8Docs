28bool IsValidCodePoint(Isolate* isolate, DirectHandle<Object> value) {
29 if (!IsNumber(*value) && !Object::ToNumber(isolate, value).ToHandle(&value)) {
30 return false;
31 }
32
33 if (Object::NumberValue(
34 *Object::ToInteger(isolate, value).ToHandleChecked()) !=
35 Object::NumberValue(*value)) {
36 return false;
37 }
38
39 if (Object::NumberValue(*value) < 0 ||
40 Object::NumberValue(*value) > 0x10FFFF) {
41 return false;
42 }
43
44 return true;
45}

47static constexpr base::uc32 kInvalidCodePoint = static_cast<base::uc32>(-1);
48
49base::uc32 NextCodePoint(Isolate* isolate, BuiltinArguments args, int index) {
50 DirectHandle<Object> value = args.at(1 + index);
51 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
52 isolate, value, Object::ToNumber(isolate, value), kInvalidCodePoint);
53 if (!IsValidCodePoint(isolate, value)) {
54 isolate->Throw(*isolate->factory()->NewRangeError(
55 MessageTemplate::kInvalidCodePoint, value));
56 return kInvalidCodePoint;
57 }
58 return DoubleToUint32(Object::NumberValue(*value));
59}

63// ES6 section 21.1.2.2 String.fromCodePoint ( ...codePoints )
64BUILTIN(StringFromCodePoint) {
65 HandleScope scope(isolate);
66 int const length = args.length() - 1;
67 if (length == 0) return ReadOnlyRoots(isolate).empty_string();
68 DCHECK_LT(0, length);
69
70 // Optimistically assume that the resulting String contains only one byte
71 // characters.
72 std::vector<uint8_t> one_byte_buffer;
73 one_byte_buffer.reserve(length);
74 base::uc32 code = 0;
75 int index;
76 for (index = 0; index < length; index++) {
77 code = NextCodePoint(isolate, args, index);
78 if (code == kInvalidCodePoint) {
79 return ReadOnlyRoots(isolate).exception();
80 }
81 if (code > String::kMaxOneByteCharCode) {
82 break;
83 }
84 one_byte_buffer.push_back(code);
85 }
86
87 if (index == length) {
88 RETURN_RESULT_OR_FAILURE(
89 isolate, isolate->factory()->NewStringFromOneByte(base::Vector<uint8_t>(
90 one_byte_buffer.data(), one_byte_buffer.size())));
91 }
92
93 std::vector<base::uc16> two_byte_buffer;
94 two_byte_buffer.reserve(length - index);
95
96 while (true) {
97 if (code <=
98 static_cast<base::uc32>(unibrow::Utf16::kMaxNonSurrogateCharCode)) {
99 two_byte_buffer.push_back(code);
100 } else {
101 two_byte_buffer.push_back(unibrow::Utf16::LeadSurrogate(code));
102 two_byte_buffer.push_back(unibrow::Utf16::TrailSurrogate(code));
103 }
104
105 if (++index == length) {
106 break;
107 }
108 code = NextCodePoint(isolate, args, index);
109 if (code == kInvalidCodePoint) {
110 return ReadOnlyRoots(isolate).exception();
111 }
112 }
113
114 DirectHandle<SeqTwoByteString> result;
115 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
116 isolate, result,
117 isolate->factory()->NewRawTwoByteString(
118 static_cast<int>(one_byte_buffer.size() + two_byte_buffer.size())));
119
120 DisallowGarbageCollection no_gc;
121 CopyChars(result->GetChars(no_gc), one_byte_buffer.data(),
122 one_byte_buffer.size());
123 CopyChars(result->GetChars(no_gc) + one_byte_buffer.size(),
124 two_byte_buffer.data(), two_byte_buffer.size());
125
126 return *result;
127}

139// ES6 section 21.1.3.10 String.prototype.localeCompare ( that )
140//
141// For now, we do not do anything locale specific.
142// If internationalization is enabled, then intl.js will override this function
143// and provide the proper functionality, so this is just a fallback.
144BUILTIN(StringPrototypeLocaleCompare) {
145 HandleScope handle_scope(isolate);
146
147 isolate->CountUsage(v8::Isolate::UseCounterFeature::kStringLocaleCompare);
148 static const char* const kMethod = "String.prototype.localeCompare";
149
150 DCHECK_LE(2, args.length());
151
152 TO_THIS_STRING(str1, kMethod);
153 DirectHandle<String> str2;
154 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, str2,
155 Object::ToString(isolate, args.at(1)));
156
157 if (str1.is_identical_to(str2)) return Smi::zero(); // Equal.
158 int str1_length = str1->length();
159 int str2_length = str2->length();
160
161 // Decide trivial cases without flattening.
162 if (str1_length == 0) {
163 if (str2_length == 0) return Smi::zero(); // Equal.
164 return Smi::FromInt(-str2_length);
165 } else {
166 if (str2_length == 0) return Smi::FromInt(str1_length);
167 }
168
169 int end = str1_length < str2_length ? str1_length : str2_length;
170
171 // No need to flatten if we are going to find the answer on the first
172 // character. At this point we know there is at least one character
173 // in each string, due to the trivial case handling above.
174 int d = str1->Get(0) - str2->Get(0);
175 if (d != 0) return Smi::FromInt(d);
176
177 str1 = String::Flatten(isolate, str1);
178 str2 = String::Flatten(isolate, str2);
179
180 DisallowGarbageCollection no_gc;
181 String::FlatContent flat1 = str1->GetFlatContent(no_gc);
182 String::FlatContent flat2 = str2->GetFlatContent(no_gc);
183
184 for (int i = 0; i < end; i++) {
185 if (flat1.Get(i) != flat2.Get(i)) {
186 return Smi::FromInt(flat1.Get(i) - flat2.Get(i));
187 }
188 }
189
190 return Smi::FromInt(str1_length - str2_length);
191}

228inline bool ToUpperOverflows(base::uc32 character) {
229 // y with umlauts and the micro sign are the only characters that stop
230 // fitting into one-byte when converting to uppercase.
231 static const base::uc32 yuml_code = 0xFF;
232 static const base::uc32 micro_code = 0xB5;
233 return (character == yuml_code || character == micro_code);
234}

236template <class Converter>
237V8_WARN_UNUSED_RESULT static Tagged<Object> ConvertCaseHelper(
238 Isolate* isolate, Tagged<String> string, Tagged<SeqString> result,
239 uint32_t result_length, unibrow::Mapping<Converter, 128>* mapping) {
240 DisallowGarbageCollection no_gc;
241 // We try this twice, once with the assumption that the result is no longer
242 // than the input and, if that assumption breaks, again with the exact
243 // length. This may not be pretty, but it is nicer than what was here before
244 // and I hereby claim my vaffel-is.
245 //
246 // NOTE: This assumes that the upper/lower case of an ASCII
247 // character is also ASCII. This is currently the case, but it
248 // might break in the future if we implement more context and locale
249 // dependent upper/lower conversions.
250 bool has_changed_character = false;
251
252 // Convert all characters to upper case, assuming that they will fit
253 // in the buffer
254 StringCharacterStream stream(string);
255 unibrow::uchar chars[Converter::kMaxWidth];
256 // We can assume that the string is not empty
257 base::uc32 current = stream.GetNext();
258 bool ignore_overflow = Converter::kIsToLower || IsSeqTwoByteString(result);
259 for (uint32_t i = 0; i < result_length;) {
260 bool has_next = stream.HasMore();
261 base::uc32 next = has_next ? stream.GetNext() : 0;
262 uint32_t char_length = mapping->get(current, next, chars);
263 if (char_length == 0) {
264 // The case conversion of this character is the character itself.
265 result->Set(i, current);
266 i++;
267 } else if (char_length == 1 &&
268 (ignore_overflow || !ToUpperOverflows(current))) {
269 // Common case: converting the letter resulted in one character.
270 DCHECK(static_cast<base::uc32>(chars[0]) != current);
271 result->Set(i, chars[0]);
272 has_changed_character = true;
273 i++;
274 } else if (result_length == string->length()) {
275 bool overflows = ToUpperOverflows(current);
276 // We've assumed that the result would be as long as the
277 // input but here is a character that converts to several
278 // characters. No matter, we calculate the exact length
279 // of the result and try the whole thing again.
280 //
281 // Note that this leaves room for optimization. We could just
282 // memcpy what we already have to the result string. Also,
283 // the result string is the last object allocated we could
284 // "realloc" it and probably, in the vast majority of cases,
285 // extend the existing string to be able to hold the full
286 // result.
287 uint32_t next_length = 0;
288 if (has_next) {
289 next_length = mapping->get(next, 0, chars);
290 if (next_length == 0) next_length = 1;
291 }
292 uint32_t current_length = i + char_length + next_length;
293 while (stream.HasMore()) {
294 current = stream.GetNext();
295 overflows |= ToUpperOverflows(current);
296 // NOTE: we use 0 as the next character here because, while
297 // the next character may affect what a character converts to,
298 // it does not in any case affect the length of what it convert
299 // to.
300 int char_length = mapping->get(current, 0, chars);
301 if (char_length == 0) char_length = 1;
302 current_length += char_length;
303 if (current_length > String::kMaxLength) {
304 AllowGarbageCollection allocate_error_and_return;
305 THROW_NEW_ERROR_RETURN_FAILURE(isolate,
306 NewInvalidStringLengthError());
307 }
308 }
309 // Try again with the real length. Return signed if we need
310 // to allocate a two-byte string for to uppercase.
311 return (overflows && !ignore_overflow) ? Smi::FromInt(-current_length)
312 : Smi::FromInt(current_length);
313 } else {
314 for (uint32_t j = 0; j < char_length; j++) {
315 result->Set(i, chars[j]);
316 i++;
317 }
318 has_changed_character = true;
319 }
320 current = next;
321 }
322 if (has_changed_character) {
323 return result;
324 } else {
325 // If we didn't actually change anything in doing the conversion
326 // we simple return the result and let the converted string
327 // become garbage; there is no reason to keep two identical strings
328 // alive.
329 return string;
330 }
331}

333template <class Converter>
334V8_WARN_UNUSED_RESULT static Tagged<Object> ConvertCase(
335 DirectHandle<String> s, Isolate* isolate,
336 unibrow::Mapping<Converter, 128>* mapping) {
337 s = String::Flatten(isolate, s);
338 uint32_t length = s->length();
339 // Assume that the string is not empty; we need this assumption later
340 if (length == 0) return *s;
341
342 // Simpler handling of ASCII strings.
343 //
344 // NOTE: This assumes that the upper/lower case of an ASCII
345 // character is also ASCII. This is currently the case, but it
346 // might break in the future if we implement more context and locale
347 // dependent upper/lower conversions.
348 if (s->IsOneByteRepresentation()) {
349 // Same length as input.
350 DirectHandle<SeqOneByteString> result =
351 isolate->factory()->NewRawOneByteString(length).ToHandleChecked();
352 DisallowGarbageCollection no_gc;
353 String::FlatContent flat_content = s->GetFlatContent(no_gc);
354 DCHECK(flat_content.IsFlat());
355 bool has_changed_character = false;
356 uint32_t index_to_first_unprocessed =
357 FastAsciiConvert<Converter::kIsToLower>(
358 reinterpret_cast<char*>(result->GetChars(no_gc)),
359 reinterpret_cast<const char*>(
360 flat_content.ToOneByteVector().begin()),
361 length, &has_changed_character);
362 // If not ASCII, we discard the result and take the 2 byte path.
363 if (index_to_first_unprocessed == length)
364 return has_changed_character ? *result : *s;
365 }
366
367 DirectHandle<SeqString> result; // Same length as input.
368 if (s->IsOneByteRepresentation()) {
369 result = isolate->factory()->NewRawOneByteString(length).ToHandleChecked();
370 } else {
371 result = isolate->factory()->NewRawTwoByteString(length).ToHandleChecked();
372 }
373
374 Tagged<Object> answer =
375 ConvertCaseHelper(isolate, *s, *result, length, mapping);
376 if (IsException(answer, isolate) || IsString(answer)) return answer;
377
378 DCHECK(IsSmi(answer));
379 // In this case we need to retry with a new string of the given length.
380 // If the value is negative, the string must be a two-byte string.
381 int int_answer = Smi::ToInt(answer);
382 if (s->IsOneByteRepresentation() && int_answer > 0) {
383 length = int_answer;
384 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
385 isolate, result, isolate->factory()->NewRawOneByteString(length));
386 } else {
387 length = abs(int_answer);
388 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
389 isolate, result, isolate->factory()->NewRawTwoByteString(length));
390 }
391 return ConvertCaseHelper(isolate, *s, *result, length, mapping);
392}

425// ES6 #sec-string.prototype.raw
426BUILTIN(StringRaw) {
427 HandleScope scope(isolate);
428 DirectHandle<Object> templ = args.atOrUndefined(isolate, 1);
429 const uint32_t argc = args.length();
430 DirectHandle<String> raw_string =
431 isolate->factory()->NewStringFromAsciiChecked("raw");
432
433 DirectHandle<JSReceiver> cooked;
434 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, cooked,
435 Object::ToObject(isolate, templ));
436
437 DirectHandle<JSAny> raw;
438 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
439 isolate, raw,
440 Cast<JSAny>(Object::GetProperty(isolate, cooked, raw_string)));
441 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, raw,
442 Object::ToObject(isolate, raw));
443 DirectHandle<Object> raw_len;
444 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
445 isolate, raw_len,
446 Object::GetProperty(isolate, raw, isolate->factory()->length_string()));
447
448 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, raw_len,
449 Object::ToLength(isolate, raw_len));
450
451 IncrementalStringBuilder result_builder(isolate);
452 // Intentional spec violation: we ignore {length} values >= 2^32, because
453 // assuming non-empty chunks they would generate too-long strings anyway.
454 const double raw_len_number = Object::NumberValue(*raw_len);
455 const uint32_t length = raw_len_number > std::numeric_limits<uint32_t>::max()
456 ? std::numeric_limits<uint32_t>::max()
457 : static_cast<uint32_t>(raw_len_number);
458 if (length > 0) {
459 DirectHandle<Object> first_element;
460 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, first_element,
461 Object::GetElement(isolate, raw, 0));
462
463 DirectHandle<String> first_string;
464 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
465 isolate, first_string, Object::ToString(isolate, first_element));
466 result_builder.AppendString(first_string);
467
468 for (uint32_t i = 1, arg_i = 2; i < length; i++, arg_i++) {
469 if (arg_i < argc) {
470 DirectHandle<String> argument_string;
471 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
472 isolate, argument_string,
473 Object::ToString(isolate, args.at(arg_i)));
474 result_builder.AppendString(argument_string);
475 }
476
477 DirectHandle<Object> element;
478 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, element,
479 Object::GetElement(isolate, raw, i));
480
481 DirectHandle<String> element_string;
482 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, element_string,
483 Object::ToString(isolate, element));
484 result_builder.AppendString(element_string);
485 }
486 }
487
488 RETURN_RESULT_OR_FAILURE(isolate, result_builder.Finish());
489}
