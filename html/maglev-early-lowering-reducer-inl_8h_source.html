35 void CheckInstanceType(V<Object> input, V<FrameState> frame_state,
36 const FeedbackSource& feedback,
37 InstanceType first_instance_type,
38 InstanceType last_instance_type, bool check_smi) {
39 if (check_smi) {
40 __ DeoptimizeIf(__ IsSmi(input), frame_state,
41 DeoptimizeReason::kWrongInstanceType, feedback);
42 }
43
44 V<i::Map> map = __ LoadMapField(input);
45
46 if (first_instance_type == last_instance_type) {
47#if V8_STATIC_ROOTS_BOOL
48 if (InstanceTypeChecker::UniqueMapOfInstanceType(first_instance_type)) {
49 std::optional<RootIndex> expected_index =
50 InstanceTypeChecker::UniqueMapOfInstanceType(first_instance_type);
51 CHECK(expected_index.has_value());
52 Handle<HeapObject> expected_map =
53 Cast<HeapObject>(isolate_->root_handle(expected_index.value()));
54 __ DeoptimizeIfNot(__ TaggedEqual(map, __ HeapConstant(expected_map)),
55 frame_state, DeoptimizeReason::kWrongInstanceType,
56 feedback);
57 return;
58 }
59#endif // V8_STATIC_ROOTS_BOOL
60 V<Word32> instance_type = __ LoadInstanceTypeField(map);
61 __ DeoptimizeIfNot(__ Word32Equal(instance_type, first_instance_type),
62 frame_state, DeoptimizeReason::kWrongInstanceType,
63 feedback);
64 } else {
65 __ DeoptimizeIfNot(CheckInstanceTypeIsInRange(map, first_instance_type,
66 last_instance_type),
67 frame_state, DeoptimizeReason::kWrongInstanceType,
68 feedback);
69 }
70 }

72 V<InternalizedString> CheckedInternalizedString(
73 V<Object> object, V<FrameState> frame_state, bool check_smi,
74 const FeedbackSource& feedback) {
75 if (check_smi) {
76 __ DeoptimizeIf(__ IsSmi(object), frame_state, DeoptimizeReason::kSmi,
77 feedback);
78 }
79
80 Label<InternalizedString> done(this);
81 V<Map> map = __ LoadMapField(object);
82 V<Word32> instance_type = __ LoadInstanceTypeField(map);
83
84 // Go to the slow path if this is a non-string, or a non-internalised
85 // string.
86 static_assert((kStringTag | kInternalizedTag) == 0);
87 IF (UNLIKELY(__ Word32BitwiseAnd(
88 instance_type, kIsNotStringMask | kIsNotInternalizedMask))) {
89 // Deopt if this isn't a string.
90 __ DeoptimizeIf(__ Word32BitwiseAnd(instance_type, kIsNotStringMask),
91 frame_state, DeoptimizeReason::kWrongMap, feedback);
92 V<Word32> representation_bits =
93 __ Word32BitwiseAnd(instance_type, kStringRepresentationMask);
94 // Deopt if this isn't a thin string.
95 __ DeoptimizeIfNot(__ Word32Equal(representation_bits, kThinStringTag),
96 frame_state, DeoptimizeReason::kWrongMap, feedback);
97 // Load internalized string from thin string.
98 V<InternalizedString> intern_string =
99 __ template LoadField<InternalizedString>(
100 object, AccessBuilder::ForThinStringActual());
101 GOTO(done, intern_string);
102 } ELSE {
103 GOTO(done, V<InternalizedString>::Cast(object));
104 }
105
106 BIND(done, result);
107 return result;
108 }

125 V<Object> CheckConstructResult(V<Object> construct_result,
126 V<Object> implicit_receiver) {
127 // If the result is an object (in the ECMA sense), we should get rid
128 // of the receiver and use the result; see ECMA-262 version 5.1
129 // section 13.2.2-7 on page 74.
130 Label<Object> done(this);
131
132 GOTO_IF(
133 __ RootEqual(construct_result, RootIndex::kUndefinedValue, isolate_),
134 done, implicit_receiver);
135
136 // If the result is a smi, it is *not* an object in the ECMA sense.
137 GOTO_IF(__ IsSmi(construct_result), done, implicit_receiver);
138
139 // Check if the type of the result is not an object in the ECMA sense.
140 GOTO_IF(JSAnyIsNotPrimitive(V<HeapObject>::Cast(construct_result)), done,
141 construct_result);
142
143 // Throw away the result of the constructor invocation and use the
144 // implicit receiver as the result.
145 GOTO(done, implicit_receiver);
146
147 BIND(done, result);
148 return result;
149 }

197 void StoreScriptContextSlowPath(V<Context> script_context,
198 V<Object> old_value, V<Object> new_value,
199 V<Object> side_data,
200 V<FrameState> frame_state,
201 const FeedbackSource& feedback,
202 Label<>& done) {
203 // Check if Undefined.
204 __ DeoptimizeIf(
205 __ RootEqual(side_data, RootIndex::kUndefinedValue, isolate_),
206 frame_state, DeoptimizeReason::kWrongValue, feedback);
207 V<Object> property = __ LoadScriptContextPropertyFromSideData(side_data);
208 // Check for const case.
209 __ DeoptimizeIf(
210 __ TaggedEqual(property,
211 __ SmiConstant(ContextSidePropertyCell::Const())),
212 frame_state, DeoptimizeReason::kWrongValue, feedback);
213 if (v8_flags.script_context_mutable_heap_number) {
214 // Check for smi case.
215 IF (__ TaggedEqual(
216 property, __ SmiConstant(ContextSidePropertyCell::SmiMarker()))) {
217 __ DeoptimizeIfNot(__ IsSmi(new_value), frame_state,
218 DeoptimizeReason::kWrongValue, feedback);
219 } ELSE {
220 if (v8_flags.script_context_mutable_heap_int32) {
221 // Check for mutable heap int32 case.
222 IF (__ TaggedEqual(
223 property,
224 __ SmiConstant(ContextSidePropertyCell::MutableInt32()))) {
225 ScopedVar<Word32> number_value(this);
226 IF (__ IsSmi(new_value)) {
227 number_value = __ UntagSmi(V<Smi>::Cast(new_value));
228 } ELSE {
229 V<i::Map> map = __ LoadMapField(new_value);
230 __ DeoptimizeIfNot(
231 __ TaggedEqual(map,
232 __ HeapConstant(factory_->heap_number_map())),
233 frame_state, DeoptimizeReason::kWrongValue, feedback);
234 number_value = __ ChangeFloat64ToInt32OrDeopt(
235 __ LoadHeapNumberValue(V<HeapNumber>::Cast(new_value)),
236 frame_state, CheckForMinusZeroMode::kCheckForMinusZero,
237 feedback);
238 }
239 __ StoreField(old_value, AccessBuilder::ForHeapInt32Value(),
240 number_value);
241 GOTO(done);
242 }
243 }
244 // It must be a mutable heap number case.
245 ScopedVar<Float64> number_value(this);
246 IF (__ IsSmi(new_value)) {
247 number_value =
248 __ ChangeInt32ToFloat64(__ UntagSmi(V<Smi>::Cast(new_value)));
249 } ELSE {
250 V<i::Map> map = __ LoadMapField(new_value);
251 __ DeoptimizeIfNot(
252 __ TaggedEqual(map, __ HeapConstant(factory_->heap_number_map())),
253 frame_state, DeoptimizeReason::kWrongValue, feedback);
254 number_value = __ LoadHeapNumberValue(V<HeapNumber>::Cast(new_value));
255 }
256 __ StoreField(old_value, AccessBuilder::ForHeapNumberValue(),
257 number_value);
258 GOTO(done);
259 }
260 }

264 V<FrameState> frame_state,
265 V<NativeContext> native_context,

267 // The result of a derived construct should be an object (in the ECMA

285 V<Smi> UpdateJSArrayLength(V<Word32> length_raw, V<JSArray> object,
286 V<Word32> index) {
287 Label<Smi> done(this);
288 IF (__ Uint32LessThan(index, length_raw)) {
289 GOTO(done, __ TagSmi(length_raw));
290 } ELSE {
291 V<Word32> new_length_raw =
292 __ Word32Add(index, 1); // This cannot overflow.
293 V<Smi> new_length_tagged = __ TagSmi(new_length_raw);
294 __ Store(object, new_length_tagged, StoreOp::Kind::TaggedBase(),
295 MemoryRepresentation::TaggedSigned(),
296 WriteBarrierKind::kNoWriteBarrier, JSArray::kLengthOffset);
297 GOTO(done, new_length_tagged);
298 }
299
300 BIND(done, length_tagged);
301 return length_tagged;
302 }

317 void TransitionElementsKind(
318 V<Object> object, V<Map> map,
319 const ZoneVector<compiler::MapRef>& transition_sources,
320 const MapRef transition_target, Label<Map>& end) {
321 // Turboshaft's TransitionElementsKind operation loads the map everytime, so
322 // we don't call it to have a single map load (in practice,
323 // LateLoadElimination should probably eliminate the subsequent map loads,
324 // but let's not risk it).
325 V<Map> target_map = __ HeapConstant(transition_target.object());
326
327 for (const compiler::MapRef transition_source : transition_sources) {
328 bool is_simple = IsSimpleMapChangeTransition(
329 transition_source.elements_kind(), transition_target.elements_kind());
330 IF (__ TaggedEqual(map, __ HeapConstant(transition_source.object()))) {
331 if (is_simple) {
332 __ StoreField(object, AccessBuilder::ForMap(), target_map);
333 } else {
334 __ CallRuntime_TransitionElementsKind(
335 isolate_, __ NoContextConstant(), V<HeapObject>::Cast(object),
336 target_map);
337 }
338 GOTO(end, target_map);
339 }
340 }

343 V<Word32> JSAnyIsNotPrimitive(V<HeapObject> heap_object) {
344 V<Map> map = __ LoadMapField(heap_object);
345 if (V8_STATIC_ROOTS_BOOL) {
346 // All primitive object's maps are allocated at the start of the read only
347 // heap. Thus JS_RECEIVER's must have maps with larger (compressed)
348 // addresses.
349 return __ Uint32LessThanOrEqual(
350 InstanceTypeChecker::kNonJsReceiverMapLimit,
351 __ TruncateWordPtrToWord32(__ BitcastTaggedToWordPtr(map)));
352 } else {
353 static_assert(LAST_JS_RECEIVER_TYPE == LAST_TYPE);
354 return __ Uint32LessThanOrEqual(FIRST_JS_RECEIVER_TYPE,
355 __ LoadInstanceTypeField(map));
356 }
357 }

359 V<Boolean> HasInPrototypeChain(V<Object> object, HeapObjectRef prototype,
360 V<FrameState> frame_state,
361 V<NativeContext> native_context,
362 LazyDeoptOnThrow lazy_deopt_on_throw) {
363 Label<Boolean> done(this);
364
365 V<Boolean> true_bool = __ HeapConstant(factory_->true_value());
366 V<Boolean> false_bool = __ HeapConstant(factory_->false_value());
367 V<HeapObject> target_proto = __ HeapConstant(prototype.object());
368
369 GOTO_IF(__ IsSmi(object), done, false_bool);
370
371 LoopLabel<Map> loop(this);
372 GOTO(loop, __ LoadMapField(object));
373
374 BIND_LOOP(loop, map) {
375 Label<> object_is_direct(this);
376
377 IF (UNLIKELY(CheckInstanceTypeIsInRange(map, FIRST_TYPE,
378 LAST_SPECIAL_RECEIVER_TYPE))) {
379 Label<> call_runtime(this);
380 V<Word32> instance_type = __ LoadInstanceTypeField(map);
381
382 GOTO_IF(__ Word32Equal(instance_type, JS_PROXY_TYPE), call_runtime);
383
384 V<Word32> bitfield =
385 __ template LoadField<Word32>(map, AccessBuilder::ForMapBitField());
386 int mask = Map::Bits1::HasNamedInterceptorBit::kMask |
387 Map::Bits1::IsAccessCheckNeededBit::kMask;
388 GOTO_IF_NOT(__ Word32BitwiseAnd(bitfield, mask), object_is_direct);
389 GOTO(call_runtime);
390
391 BIND(call_runtime);
392 GOTO(done, __ CallRuntime_HasInPrototypeChain(
393 isolate_, frame_state, native_context,
394 lazy_deopt_on_throw, object, target_proto));
395 }
396 GOTO(object_is_direct);
397
398 BIND(object_is_direct);
399 V<HeapObject> proto = __ template LoadField<HeapObject>(
400 map, AccessBuilder::ForMapPrototype());
401 GOTO_IF(__ RootEqual(proto, RootIndex::kNullValue, isolate_), done,
402 false_bool);
403 GOTO_IF(__ TaggedEqual(proto, target_proto), done, true_bool);
404
405 GOTO(loop, __ LoadMapField(proto));
406 }
407
408 BIND(done, result);
409 return result;
410 }

432 V<PropertyArray> ExtendPropertiesBackingStore(
433 V<PropertyArray> old_property_array, V<JSObject> object, int old_length,
434 V<FrameState> frame_state, const FeedbackSource& feedback) {
435 // Allocate new PropertyArray.
436 int new_length = old_length + JSObject::kFieldsAdded;
437 Uninitialized<PropertyArray> new_property_array =
438 __ template Allocate<PropertyArray>(
439 __ IntPtrConstant(PropertyArray::SizeFor(new_length)),
440 AllocationType::kYoung);
441 __ InitializeField(new_property_array, AccessBuilder::ForMap(),
442 __ HeapConstant(factory_->property_array_map()));
443
444 // Copy existing properties over.
445 for (int i = 0; i < old_length; i++) {
446 V<Object> old_value = __ template LoadField<Object>(
447 old_property_array, AccessBuilder::ForPropertyArraySlot(i));
448 __ InitializeField(new_property_array,
449 AccessBuilder::ForPropertyArraySlot(i), old_value);
450 }
451
452 // Initialize new properties to undefined.
453 V<Undefined> undefined = __ HeapConstant(factory_->undefined_value());
454 for (int i = 0; i < JSObject::kFieldsAdded; ++i) {
455 __ InitializeField(new_property_array,
456 AccessBuilder::ForPropertyArraySlot(old_length + i),
457 undefined);
458 }
459
460 // Read the hash.
461 ScopedVar<Word32> hash(this);
462 if (old_length == 0) {
463 // The object might still have a hash, stored in properties_or_hash. If
464 // properties_or_hash is a SMI, then it's the hash. It can also be an
465 // empty PropertyArray.
466 V<Object> hash_obj = __ template LoadField<Object>(
467 object, AccessBuilder::ForJSObjectPropertiesOrHash());
468 IF (__ IsSmi(hash_obj)) {
469 hash = __ Word32ShiftLeft(__ UntagSmi(V<Smi>::Cast(hash_obj)),
470 PropertyArray::HashField::kShift);
471 } ELSE {
472 hash = __ Word32Constant(PropertyArray::kNoHashSentinel);
473 }
474 } else {
475 V<Smi> hash_smi = __ template LoadField<Smi>(
476 old_property_array, AccessBuilder::ForPropertyArrayLengthAndHash());
477 hash = __ Word32BitwiseAnd(__ UntagSmi(hash_smi),
478 PropertyArray::HashField::kMask);
479 }
480
481 // Add the new length and write the length-and-hash field.
482 static_assert(PropertyArray::LengthField::kShift == 0);
483 V<Word32> length_and_hash = __ Word32BitwiseOr(hash, new_length);
484 __ InitializeField(new_property_array,
485 AccessBuilder::ForPropertyArrayLengthAndHash(),
486 __ TagSmi(length_and_hash));
487
488 V<PropertyArray> initialized_new_property_array =
489 __ FinishInitialization(std::move(new_property_array));
490
491 // Replace the old property array in {object}.
492 __ StoreField(object, AccessBuilder::ForJSObjectPropertiesOrHash(),
493 initialized_new_property_array);
494
495 return initialized_new_property_array;
496 }

498 void GeneratorStore(V<Context> context, V<JSGeneratorObject> generator,
499 base::SmallVector<OpIndex, 32> parameters_and_registers,
500 int suspend_id, int bytecode_offset) {
501 V<FixedArray> array = __ template LoadTaggedField<FixedArray>(
502 generator, JSGeneratorObject::kParametersAndRegistersOffset);
503 for (int i = 0; static_cast<size_t>(i) < parameters_and_registers.size();

505 __ Store(array, parameters_and_registers[i], StoreOp::Kind::TaggedBase(),

507 WriteBarrierKind::kFullWriteBarrier,
508 FixedArray::OffsetOfElementAt(i));
509 }
510 __ Store(generator, __ SmiConstant(Smi::FromInt(suspend_id)),
511 StoreOp::Kind::TaggedBase(), MemoryRepresentation::TaggedSigned(),
512 WriteBarrierKind::kNoWriteBarrier,
513 JSGeneratorObject::kContinuationOffset);
514 __ Store(generator, __ SmiConstant(Smi::FromInt(bytecode_offset)),
515 StoreOp::Kind::TaggedBase(), MemoryRepresentation::TaggedSigned(),
516 WriteBarrierKind::kNoWriteBarrier,
517 JSGeneratorObject::kInputOrDebugPosOffset);
518
519 __ Store(generator, context, StoreOp::Kind::TaggedBase(),
520 MemoryRepresentation::AnyTagged(),
521 WriteBarrierKind::kFullWriteBarrier,
522 JSGeneratorObject::kContextOffset);
523 }

526 V<Word32> CheckInstanceTypeIsInRange(V<Map> map,
527 InstanceType first_instance_type,
528 InstanceType last_instance_type) {
529 V<Word32> instance_type = __ LoadInstanceTypeField(map);
530
531 if (first_instance_type == 0) {
532 return __ Uint32LessThanOrEqual(instance_type, last_instance_type);
533 } else {
534 return __ Uint32LessThanOrEqual(
535 __ Word32Sub(instance_type, first_instance_type),
536 last_instance_type - first_instance_type);
537 }
