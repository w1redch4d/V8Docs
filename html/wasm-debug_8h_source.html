39// Side table storing information used to inspect Liftoff frames at runtime.
40// This table is only created on demand for debugging, so it is not optimized
41// for memory size.
42class DebugSideTable {
43 public:
44 class Entry {
45 public:
46 enum Storage : int8_t { kConstant, kRegister, kStack };
47 struct Value {
48 int index;
49 ValueType type;
50 const WasmModule* module;
51 Storage storage;
52 union {
53 int32_t i32_const; // if kind == kConstant
54 int reg_code; // if kind == kRegister
55 int stack_offset; // if kind == kStack
56 };
57
58 bool operator==(const Value& other) const {
59 if (index != other.index) return false;
60 if (!EquivalentTypes(type, other.type, module, other.module)) {
61 return false;
62 }
63 if (storage != other.storage) return false;
64 switch (storage) {
65 case kConstant:
66 return i32_const == other.i32_const;
67 case kRegister:
68 return reg_code == other.reg_code;
69 case kStack:
70 return stack_offset == other.stack_offset;
71 }
72 }
73 bool operator!=(const Value& other) const { return !(*this == other); }
74
75 bool is_constant() const { return storage == kConstant; }
76 bool is_register() const { return storage == kRegister; }
77 };
78
79 Entry(int pc_offset, int stack_height, std::vector<Value> changed_values)
80 : pc_offset_(pc_offset),
81 stack_height_(stack_height),
82 changed_values_(std::move(changed_values)) {}
83
84 // Constructor for map lookups (only initializes the {pc_offset_}).
85 explicit Entry(int pc_offset) : pc_offset_(pc_offset) {}
86
87 int pc_offset() const { return pc_offset_; }
88
89 // Stack height, including locals.
90 int stack_height() const { return stack_height_; }
91
92 base::Vector<const Value> changed_values() const {
93 return base::VectorOf(changed_values_);
94 }
95
96 const Value* FindChangedValue(int stack_index) const {
97 DCHECK_GT(stack_height_, stack_index);
98 auto it = std::lower_bound(
99 changed_values_.begin(), changed_values_.end(), stack_index,
100 [](const Value& changed_value, int stack_index) {
101 return changed_value.index < stack_index;
102 });
103 return it != changed_values_.end() && it->index == stack_index ? &*it
104 : nullptr;
105 }
106
107 void Print(std::ostream&) const;
108
109 size_t EstimateCurrentMemoryConsumption() const;
110
111 private:
112 int pc_offset_;
113 int stack_height_;
114 // Only store differences from the last entry, to keep the table small.
115 std::vector<Value> changed_values_;
116 };
117
118 // Technically it would be fine to copy this class, but there should not be a
119 // reason to do so, hence mark it move only.
120 MOVE_ONLY_NO_DEFAULT_CONSTRUCTOR(DebugSideTable);
121
122 explicit DebugSideTable(int num_locals, std::vector<Entry> entries)
123 : num_locals_(num_locals), entries_(std::move(entries)) {
124 DCHECK(
125 std::is_sorted(entries_.begin(), entries_.end(), EntryPositionLess{}));
126 }
127
128 const Entry* GetEntry(int pc_offset) const {
129 auto it = std::lower_bound(entries_.begin(), entries_.end(),
130 Entry{pc_offset}, EntryPositionLess{});
131 if (it == entries_.end() || it->pc_offset() != pc_offset) return nullptr;
132 DCHECK_LE(num_locals_, it->stack_height());
133 return &*it;
134 }
135
136 const Entry::Value* FindValue(const Entry* entry, int stack_index) const {
137 while (true) {
138 if (auto* value = entry->FindChangedValue(stack_index)) {
139 // Check that the table was correctly minimized: If the previous stack
140 // also had an entry for {stack_index}, it must be different.
141 DCHECK(entry == &entries_.front() ||
142 (entry - 1)->stack_height() <= stack_index ||
143 *FindValue(entry - 1, stack_index) != *value);
144 return value;
145 }
146 DCHECK_NE(&entries_.front(), entry);
147 --entry;
148 }
149 }
150
151 auto entries() const {
152 return base::make_iterator_range(entries_.begin(), entries_.end());
153 }
154
155 int num_locals() const { return num_locals_; }
156
157 void Print(std::ostream&) const;
158
159 size_t EstimateCurrentMemoryConsumption() const;
160
161 private:
162 struct EntryPositionLess {
163 bool operator()(const Entry& a, const Entry& b) const {
164 return a.pc_offset() < b.pc_offset();
165 }
166 };
167
168 int num_locals_;
169 std::vector<Entry> entries_;
170};
