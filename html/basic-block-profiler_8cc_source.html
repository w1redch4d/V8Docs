84void BasicBlockProfilerData::CopyFromJSHeap(
85 Tagged<OnHeapBasicBlockProfilerData> js_heap_data) {
86 function_name_ = js_heap_data->name()->ToCString().get();
87 schedule_ = js_heap_data->schedule()->ToCString().get();
88 code_ = js_heap_data->code()->ToCString().get();
89 Tagged<FixedUInt32Array> counts =
90 Cast<FixedUInt32Array>(js_heap_data->counts());
91 for (int i = 0; i < counts->length() / kBlockCountSlotSize; ++i) {
92 counts_.push_back(counts->get(i));
93 }
94 Tagged<FixedInt32Array> block_ids(js_heap_data->block_ids());
95 for (int i = 0; i < block_ids->length() / kBlockIdSlotSize; ++i) {
96 block_ids_.push_back(block_ids->get(i));
97 }
98 Tagged<PodArray<std::pair<int32_t, int32_t>>> branches =
99 js_heap_data->branches();
100 for (int i = 0; i < branches->length(); ++i) {
101 branches_.push_back(branches->get(i));
102 }
103 CHECK_EQ(block_ids_.size(), counts_.size());
104 hash_ = js_heap_data->hash();
105}

107Handle<OnHeapBasicBlockProfilerData> BasicBlockProfilerData::CopyToJSHeap(
108 Isolate* isolate) {
109 int id_array_size_in_bytes = static_cast<int>(n_blocks() * kBlockIdSlotSize);
110 CHECK(id_array_size_in_bytes >= 0 &&
111 static_cast<size_t>(id_array_size_in_bytes) / kBlockIdSlotSize ==
112 n_blocks()); // Overflow
113 DirectHandle<FixedInt32Array> block_ids = FixedInt32Array::New(
114 isolate, id_array_size_in_bytes, AllocationType::kOld);
115 for (int i = 0; i < static_cast<int>(n_blocks()); ++i) {
116 block_ids->set(i, block_ids_[i]);
117 }
118
119 int counts_array_size_in_bytes =
120 static_cast<int>(n_blocks() * kBlockCountSlotSize);
121 CHECK(counts_array_size_in_bytes >= 0 &&
122 static_cast<size_t>(counts_array_size_in_bytes) / kBlockCountSlotSize ==
123 n_blocks()); // Overflow
124 DirectHandle<FixedUInt32Array> counts = FixedUInt32Array::New(
125 isolate, counts_array_size_in_bytes, AllocationType::kOld);
126 for (int i = 0; i < static_cast<int>(n_blocks()); ++i) {
127 counts->set(i, counts_[i]);
128 }
129
130 DirectHandle<PodArray<std::pair<int32_t, int32_t>>> branches =
131 PodArray<std::pair<int32_t, int32_t>>::New(
132 isolate, static_cast<int>(branches_.size()), AllocationType::kOld);
133 for (int i = 0; i < static_cast<int>(branches_.size()); ++i) {
134 branches->set(i, branches_[i]);
135 }
136 DirectHandle<String> name = CopyStringToJSHeap(function_name_, isolate);
137 DirectHandle<String> schedule = CopyStringToJSHeap(schedule_, isolate);
138 DirectHandle<String> code = CopyStringToJSHeap(code_, isolate);
139
140 return isolate->factory()->NewOnHeapBasicBlockProfilerData(
141 block_ids, counts, branches, name, schedule, code, hash_,
142 AllocationType::kOld);
143}

205std::vector<bool> BasicBlockProfiler::GetCoverageBitmap(Isolate* isolate) {
206 DisallowGarbageCollection no_gc;
207 Tagged<ArrayList> list(isolate->heap()->basic_block_profiling_data());
208 std::vector<bool> out;
209 int list_length = list->length();
210 for (int i = 0; i < list_length; ++i) {
211 BasicBlockProfilerData data(
212 Cast<OnHeapBasicBlockProfilerData>(list->get(i)));
213 for (size_t j = 0; j < data.n_blocks(); ++j) {
214 out.push_back(data.counts_[j] > 0);
215 }
216 }

220void BasicBlockProfilerData::Log(Isolate* isolate, std::ostream& os) {
221 bool any_nonzero_counter = false;
222 constexpr char kNext[] = "\t";
223 for (size_t i = 0; i < n_blocks(); ++i) {
224 if (counts_[i] > 0) {
225 any_nonzero_counter = true;
226 os << ProfileDataFromFileConstants::kBlockCounterMarker << kNext
227 << function_name_.c_str() << kNext << block_ids_[i] << kNext
228 << counts_[i] << '\n';

231 if (any_nonzero_counter) {
232 for (size_t i = 0; i < branches_.size(); ++i) {
233 os << ProfileDataFromFileConstants::kBlockHintMarker << kNext
234 << function_name_.c_str() << kNext << branches_[i].first << kNext
235 << branches_[i].second << '\n';
236 }
237 os << ProfileDataFromFileConstants::kBuiltinHashMarker << kNext
238 << function_name_.c_str() << kNext << hash_ << '\n';

242std::ostream& operator<<(std::ostream& os, const BasicBlockProfilerData& d) {
243 if (std::all_of(d.counts_.cbegin(), d.counts_.cend(),
244 [](uint32_t count) { return count == 0; })) {
245 // No data was collected for this function.
246 return os;
247 }
248 const char* name = "unknown function";
249 if (!d.function_name_.empty()) {
250 name = d.function_name_.c_str();
251 }
252 if (!d.schedule_.empty()) {
253 os << "schedule for " << name << " (B0 entered " << d.counts_[0]
254 << " times)" << '\n';
255 os << d.schedule_.c_str() << '\n';
256 }
257 os << "block counts for " << name << ":" << '\n';
258 std::vector<std::pair<size_t, uint32_t>> pairs;
259 pairs.reserve(d.n_blocks());
260 for (size_t i = 0; i < d.n_blocks(); ++i) {
261 pairs.push_back(std::make_pair(i, d.counts_[i]));
262 }
263 std::sort(
264 pairs.begin(), pairs.end(),
265 [=](std::pair<size_t, uint32_t> left, std::pair<size_t, uint32_t> right) {
266 if (right.second == left.second) return left.first < right.first;
267 return right.second < left.second;
268 });
269 for (auto it : pairs) {
270 if (it.second == 0) break;
271 os << "block B" << it.first << " : " << it.second << '\n';
272 }
273 os << '\n';
274 if (!d.code_.empty()) {
275 os << d.code_.c_str() << '\n';
276 }
277 return os;
278}

282void BuiltinsCallGraph::AddBuiltinCall(Builtin caller, Builtin callee,
283 int32_t block_id) {
284 if (builtin_call_map_.count(caller) == 0) {
285 builtin_call_map_.emplace(caller, BuiltinCallees());
286 }
287 BuiltinCallees& callees = builtin_call_map_.at(caller);
288 if (callees.count(block_id) == 0) {
289 callees.emplace(block_id, BlockCallees());
290 }
291 BlockCallees& block_callees = callees.at(block_id);
292 if (block_callees.count(callee) == 0) {
293 block_callees.emplace(callee);
294 }
295}
