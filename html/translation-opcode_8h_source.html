100inline bool IsTranslationFrameOpcode(TranslationOpcode o) {
101 static_assert(
102 0 == static_cast<int>(TranslationOpcode::INTERPRETED_FRAME_WITH_RETURN));
103 static_assert(kNumTranslationJsFrameOpcodes < kNumTranslationFrameOpcodes);
104 return static_cast<int>(o) < kNumTranslationFrameOpcodes;
105}

107inline bool IsTranslationJsFrameOpcode(TranslationOpcode o) {
108 static_assert(
109 kNumTranslationJsFrameOpcodes ==

111 return static_cast<int>(o) < kNumTranslationJsFrameOpcodes;
112}
