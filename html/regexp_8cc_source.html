173// Identifies the sort of regexps where the regexp engine is faster
174// than the code used for atom matches.
175bool HasFewDifferentCharacters(DirectHandle<String> pattern) {
176 uint32_t length = std::min(kMaxLookaheadForBoyerMoore, pattern->length());
177 if (length <= kPatternTooShortForBoyerMoore) return false;
178 const int kMod = 128;
179 bool character_found[kMod];
180 uint32_t different = 0;
181 memset(&character_found[0], 0, sizeof(character_found));
182 for (uint32_t i = 0; i < length; i++) {
183 int ch = (pattern->Get(i) & (kMod - 1));
184 if (!character_found[ch]) {
185 character_found[ch] = true;
186 different++;
187 // We declare a regexp low-alphabet if it has at least 3 times as many
188 // characters as it has different characters.
189 if (different * 3 > length) return false;
190 }
191 }
192 return true;
193}

200MaybeHandle<Object> RegExp::Compile(Isolate* isolate, Handle<JSRegExp> re,
201 Handle<String> pattern, RegExpFlags flags,
202 uint32_t backtrack_limit) {
203 DCHECK(pattern->IsFlat());
204
205 // Caching is based only on the pattern and flags, but code also differs when
206 // a backtrack limit is set. A present backtrack limit is very much *not* the
207 // common case, so just skip the cache for these.
208 const bool is_compilation_cache_enabled =
209 (backtrack_limit == JSRegExp::kNoBacktrackLimit);
210
211 Zone zone(isolate->allocator(), ZONE_NAME);
212 CompilationCache* compilation_cache = nullptr;
213 if (is_compilation_cache_enabled) {
214 compilation_cache = isolate->compilation_cache();
215 MaybeDirectHandle<RegExpData> maybe_cached =
216 compilation_cache->LookupRegExp(pattern,
217 JSRegExp::AsJSRegExpFlags(flags));
218 DirectHandle<RegExpData> cached;
219 if (maybe_cached.ToHandle(&cached)) {
220 re->set_data(*cached);
221 return re;
222 }
223 }
224
225 PostponeInterruptsScope postpone(isolate);
226 RegExpCompileData parse_result;
227 DCHECK(!isolate->has_exception());
228 if (!RegExpParser::ParseRegExpFromHeapString(isolate, &zone, pattern, flags,
229 &parse_result)) {
230 // Throw an exception if we fail to parse the pattern.
231 return RegExp::ThrowRegExpException(isolate, flags, pattern,
232 parse_result.error);
233 }
234
235 bool has_been_compiled = false;
236
237 if (v8_flags.default_to_experimental_regexp_engine &&
238 ExperimentalRegExp::CanBeHandled(parse_result.tree, pattern, flags,
239 parse_result.capture_count)) {
240 DCHECK(v8_flags.enable_experimental_regexp_engine);
241 ExperimentalRegExp::Initialize(isolate, re, pattern, flags,
242 parse_result.capture_count);
243 has_been_compiled = true;
244 } else if (flags & JSRegExp::kLinear) {
245 DCHECK(v8_flags.enable_experimental_regexp_engine);
246 if (!ExperimentalRegExp::CanBeHandled(parse_result.tree, pattern, flags,
247 parse_result.capture_count)) {
248 // TODO(mbid): The error could provide a reason for why the regexp can't
249 // be executed in linear time (e.g. due to back references).
250 return RegExp::ThrowRegExpException(isolate, flags, pattern,
251 RegExpError::kNotLinear);
252 }
253 ExperimentalRegExp::Initialize(isolate, re, pattern, flags,
254 parse_result.capture_count);
255 has_been_compiled = true;
256 } else if (parse_result.simple && !IsIgnoreCase(flags) && !IsSticky(flags) &&
257 !HasFewDifferentCharacters(pattern)) {
258 // Parse-tree is a single atom that is equal to the pattern.
259 RegExpImpl::AtomCompile(isolate, re, pattern, flags, pattern);
260 has_been_compiled = true;
261 } else if (parse_result.tree->IsAtom() && !IsSticky(flags) &&
262 parse_result.capture_count == 0) {
263 RegExpAtom* atom = parse_result.tree->AsAtom();
264 // The pattern source might (?) contain escape sequences, but they're
265 // resolved in atom_string.
266 base::Vector<const base::uc16> atom_pattern = atom->data();
267 DirectHandle<String> atom_string;
268 ASSIGN_RETURN_ON_EXCEPTION(
269 isolate, atom_string,
270 isolate->factory()->NewStringFromTwoByte(atom_pattern));
271 if (!IsIgnoreCase(flags) && !HasFewDifferentCharacters(atom_string)) {
272 RegExpImpl::AtomCompile(isolate, re, pattern, flags, atom_string);
273 has_been_compiled = true;
274 }
275 }
276 if (!has_been_compiled) {
277 RegExpImpl::IrregexpInitialize(isolate, re, pattern, flags,
278 parse_result.capture_count, backtrack_limit);
279 }
280 // Compilation succeeded so the data is set on the regexp
281 // and we can store it in the cache.
282 DirectHandle<RegExpData> data(re->data(isolate), isolate);
283 if (is_compilation_cache_enabled) {
284 compilation_cache->PutRegExp(pattern, JSRegExp::AsJSRegExpFlags(flags),
285 data);
286 }
287
288 return re;
289}

292bool RegExp::EnsureFullyCompiled(Isolate* isolate,
293 DirectHandle<RegExpData> re_data,
294 Handle<String> subject) {
295 switch (re_data->type_tag()) {
296 case RegExpData::Type::ATOM:
297 return true;
298 case RegExpData::Type::IRREGEXP:
299 if (RegExpImpl::IrregexpPrepare(isolate, Cast<IrRegExpData>(re_data),
300 subject) == -1) {
301 DCHECK(isolate->has_exception());
302 return false;
303 }
304 return true;
305 case RegExpData::Type::EXPERIMENTAL:
306 if (!ExperimentalRegExp::IsCompiled(Cast<IrRegExpData>(re_data),
307 isolate) &&
308 !ExperimentalRegExp::Compile(isolate, Cast<IrRegExpData>(re_data))) {
309 DCHECK(isolate->has_exception());
310 return false;
311 }
312 return true;
313 }
314 UNREACHABLE();
315}

392template <typename SChar, typename PChar>
393int AtomExecRawImpl(Isolate* isolate, base::Vector<const SChar> subject,
394 base::Vector<const PChar> pattern, int index,
395 int32_t* output, int output_size,
396 const DisallowGarbageCollection& no_gc) {
397 const int subject_length = subject.length();
398 const int pattern_length = pattern.length();
399 DCHECK_GT(pattern_length, 0);
400 const int max_index = subject_length - pattern_length;
401
402 StringSearch<PChar, SChar> search(isolate, pattern);
403 for (int i = 0; i < output_size; i += JSRegExp::kAtomRegisterCount) {
404 if (index > max_index) {
405 static_assert(RegExp::RE_FAILURE == 0);
406 return i / JSRegExp::kAtomRegisterCount; // Return number of matches.
407 }
408 index = search.Search(subject, index);
409 if (index == -1) {
410 static_assert(RegExp::RE_FAILURE == 0);
411 return i / JSRegExp::kAtomRegisterCount; // Return number of matches.
412 } else {
413 output[i] = index; // match start
414 index += pattern_length;
415 output[i + 1] = index; // match end
416 }
417 }
418
419 return output_size / JSRegExp::kAtomRegisterCount;
420}

506// Irregexp implementation.
507
508// Ensures that the regexp object contains a compiled version of the
509// source for either one-byte or two-byte subject strings.
510// If the compiled version doesn't already exist, it is compiled
511// from the source pattern.
512// If compilation fails, an exception is thrown and this function
513// returns false.
514bool RegExpImpl::EnsureCompiledIrregexp(Isolate* isolate,
515 DirectHandle<IrRegExpData> re_data,
516 Handle<String> sample_subject,
517 bool is_one_byte) {
518 bool has_bytecode = re_data->has_bytecode(is_one_byte);
519 bool needs_initial_compilation = !re_data->has_code(is_one_byte);
520 // Recompile is needed when we're dealing with the first execution of the
521 // regexp after the decision to tier up has been made. If the tiering up
522 // strategy is not in use, this value is always false.
523 bool needs_tier_up_compilation = re_data->MarkedForTierUp() && has_bytecode;
524
525 if (v8_flags.trace_regexp_tier_up && needs_tier_up_compilation) {
526 PrintF("JSRegExp object (data: %p) needs tier-up compilation\n",
527 reinterpret_cast<void*>(re_data->ptr()));
528 }
529
530 if (!needs_initial_compilation && !needs_tier_up_compilation) {
531 DCHECK(re_data->has_code(is_one_byte));
532 DCHECK_IMPLIES(v8_flags.regexp_interpret_all, has_bytecode);
533 return true;
534 }
535
536 DCHECK_IMPLIES(needs_tier_up_compilation, has_bytecode);
537
538 return CompileIrregexp(isolate, re_data, sample_subject, is_one_byte);
539}

570// static
571Handle<FixedArray> RegExp::CreateCaptureNameMap(
572 Isolate* isolate, ZoneVector<RegExpCapture*>* named_captures) {
573 if (named_captures == nullptr) return Handle<FixedArray>();
574
575 DCHECK(!named_captures->empty());
576
577 // Named captures are sorted by name (because the set is used to ensure
578 // name uniqueness). But the capture name map must to be sorted by index.
579
580 std::sort(named_captures->begin(), named_captures->end(),
581 RegExpCaptureIndexLess{});
582
583 int len = static_cast<int>(named_captures->size()) * 2;
584 Handle<FixedArray> array = isolate->factory()->NewFixedArray(len);
585
586 int i = 0;
587 for (const RegExpCapture* capture : *named_captures) {
588 base::Vector<const base::uc16> capture_name(capture->name()->data(),
589 capture->name()->size());
590 // CSA code in ConstructNewResultFromMatchInfo requires these strings to be
591 // internalized so they can be used as property names in the 'exec' results.
592 DirectHandle<String> name =
593 isolate->factory()->InternalizeString(capture_name);
594 array->set(i * 2, *name);
595 array->set(i * 2 + 1, Smi::FromInt(capture->index()));
596
597 i++;
598 }
599 DCHECK_EQ(i * 2, len);
600
601 return array;
602}

610 PostponeInterruptsScope postpone(isolate);
611
612 DCHECK(RegExpCodeIsValidForPreCompilation(isolate, re_data, is_one_byte));
613
614 RegExpFlags flags = JSRegExp::AsRegExpFlags(re_data->flags());
615
616 Handle<String> pattern(re_data->source(), isolate);
617 pattern = String::Flatten(isolate, pattern);
618 RegExpCompileData compile_data;
619 if (!RegExpParser::ParseRegExpFromHeapString(isolate, &zone, pattern, flags,
620 &compile_data)) {
621 // Throw an exception if we fail to parse the pattern.
622 // THIS SHOULD NOT HAPPEN. We already pre-parsed it successfully once.
623 USE(RegExp::ThrowRegExpException(isolate, flags, pattern,
624 compile_data.error));
625 return false;
626 }
627 // The compilation target is a kBytecode if we're interpreting all regexp
628 // objects, or if we're using the tier-up strategy but the tier-up hasn't
629 // happened yet. The compilation target is a kNative if we're using the
630 // tier-up strategy and we need to recompile to tier-up, or if we're producing
631 // native code for all regexp objects.
632 compile_data.compilation_target = re_data->ShouldProduceBytecode()
633 ? RegExpCompilationTarget::kBytecode
634 : RegExpCompilationTarget::kNative;
635 uint32_t backtrack_limit = re_data->backtrack_limit();
636 const bool compilation_succeeded =
637 Compile(isolate, &zone, &compile_data, flags, pattern, sample_subject,
638 is_one_byte, backtrack_limit);
639 if (!compilation_succeeded) {
640 DCHECK(compile_data.error != RegExpError::kNone);
641 RegExp::ThrowRegExpException(isolate, re_data, compile_data.error);
642 return false;
643 }
644
645 if (compile_data.compilation_target == RegExpCompilationTarget::kNative) {
646 re_data->set_code(is_one_byte, Cast<Code>(*compile_data.code));
647
648 // Reset bytecode to uninitialized. In case we use tier-up we know that
649 // tier-up has happened this way.
650 re_data->clear_bytecode(is_one_byte);
651 } else {
652 DCHECK_EQ(compile_data.compilation_target,
653 RegExpCompilationTarget::kBytecode);
654 // Store code generated by compiler in bytecode and trampoline to
655 // interpreter in code.
656 re_data->set_bytecode(is_one_byte,
657 Cast<TrustedByteArray>(*compile_data.code));
658 DirectHandle<Code> trampoline =
659 BUILTIN_CODE(isolate, RegExpInterpreterTrampoline);
660 re_data->set_code(is_one_byte, *trampoline);
661 }
662 DirectHandle<FixedArray> capture_name_map =
663 RegExp::CreateCaptureNameMap(isolate, compile_data.named_captures);
664 re_data->set_capture_name_map(capture_name_map);
665 int register_max = re_data->max_register_count();
666 if (compile_data.register_count > register_max) {
667 re_data->set_max_register_count(compile_data.register_count);
668 }
669 re_data->set_backtrack_limit(backtrack_limit);
670
671 if (v8_flags.trace_regexp_tier_up) {
672 PrintF("JSRegExp data object %p %s size: %d\n",
673 reinterpret_cast<void*>(re_data->ptr()),
674 re_data->ShouldProduceBytecode() ? "bytecode" : "native code",
675 re_data->ShouldProduceBytecode()
676 ? re_data->bytecode(is_one_byte)->AllocatedSize()
677 : re_data->code(isolate, is_one_byte)->Size());
678 }
679
680 return true;
681}

695 DirectHandle<IrRegExpData> re_data,
696 Handle<String> subject) {
697 DCHECK(subject->IsFlat());

702 is_one_byte)) {

711int RegExpImpl::IrregexpExecRaw(Isolate* isolate,
712 DirectHandle<IrRegExpData> regexp_data,
713 Handle<String> subject, int index,
714 int32_t* output, int output_size) {
715 DCHECK_LE(0, index);
716 DCHECK_LE(index, subject->length());
717 DCHECK(subject->IsFlat());
718 DCHECK_GE(output_size,
719 JSRegExp::RegistersForCaptureCount(regexp_data->capture_count()));
720
721 bool is_one_byte = subject->IsOneByteRepresentation();
722
723 if (!regexp_data->ShouldProduceBytecode()) {
724 do {
725 EnsureCompiledIrregexp(isolate, regexp_data, subject, is_one_byte);
726 // The stack is used to allocate registers for the compiled regexp code.
727 // This means that in case of failure, the output registers array is left
728 // untouched and contains the capture results from the previous successful
729 // match. We can use that to set the last match info lazily.
730 int res = NativeRegExpMacroAssembler::Match(regexp_data, subject, output,
731 output_size, index, isolate);
732 if (res != NativeRegExpMacroAssembler::RETRY) {
733 DCHECK(res != NativeRegExpMacroAssembler::EXCEPTION ||
734 isolate->has_exception());
735 static_assert(static_cast<int>(NativeRegExpMacroAssembler::SUCCESS) ==
736 RegExp::RE_SUCCESS);
737 static_assert(static_cast<int>(NativeRegExpMacroAssembler::FAILURE) ==
738 RegExp::RE_FAILURE);
739 static_assert(static_cast<int>(NativeRegExpMacroAssembler::EXCEPTION) ==
740 RegExp::RE_EXCEPTION);
741 return res;
742 }
743 // If result is RETRY, the string has changed representation, and we
744 // must restart from scratch.
745 // In this case, it means we must make sure we are prepared to handle
746 // the, potentially, different subject (the string can switch between
747 // being internal and external, and even between being Latin1 and
748 // UC16, but the characters are always the same).
749 is_one_byte = subject->IsOneByteRepresentation();
750 } while (true);
751 UNREACHABLE();
752 } else {
753 DCHECK(regexp_data->ShouldProduceBytecode());
754
755 do {
756 int result = IrregexpInterpreter::MatchForCallFromRuntime(
757 isolate, regexp_data, subject, output, output_size, index);
758 DCHECK_IMPLIES(result == IrregexpInterpreter::EXCEPTION,
759 isolate->has_exception());
760
761 static_assert(IrregexpInterpreter::FAILURE == 0);
762 static_assert(IrregexpInterpreter::SUCCESS == 1);
763 static_assert(IrregexpInterpreter::FALLBACK_TO_EXPERIMENTAL < 0);
764 static_assert(IrregexpInterpreter::EXCEPTION < 0);
765 static_assert(IrregexpInterpreter::RETRY < 0);
766 if (result >= IrregexpInterpreter::FAILURE) {
767 return result;
768 }
769
770 if (result == IrregexpInterpreter::RETRY) {
771 // The string has changed representation, and we must restart the
772 // match. We need to reset the tier up to start over with compilation.
773 if (v8_flags.regexp_tier_up) regexp_data->ResetLastTierUpTick();
774 is_one_byte = subject->IsOneByteRepresentation();
775 EnsureCompiledIrregexp(isolate, regexp_data, subject, is_one_byte);
776 } else {
777 DCHECK(result == IrregexpInterpreter::EXCEPTION ||
778 result == IrregexpInterpreter::FALLBACK_TO_EXPERIMENTAL);
779 return result;
780 }
781 } while (true);
782 UNREACHABLE();
783 }
784}

786std::optional<int> RegExpImpl::IrregexpExec(
787 Isolate* isolate, DirectHandle<IrRegExpData> regexp_data,
788 Handle<String> subject, int previous_index, int32_t* result_offsets_vector,
789 uint32_t result_offsets_vector_length) {
790 subject = String::Flatten(isolate, subject);
791
792#ifdef DEBUG
793 if (v8_flags.trace_regexp_bytecodes && regexp_data->ShouldProduceBytecode()) {
794 PrintF("\n\nRegexp match: /%s/\n\n",
795 regexp_data->source()->ToCString().get());
796 PrintF("\n\nSubject string: '%s'\n\n", subject->ToCString().get());
797 }
798#endif
799
800 const int original_register_count =
801 JSRegExp::RegistersForCaptureCount(regexp_data->capture_count());
802
803 // Maybe force early tier up:
804 if (v8_flags.regexp_tier_up) {
805 if (subject->length() >= JSRegExp::kTierUpForSubjectLengthValue) {
806 // For very long subject strings, the regexp interpreter is currently much
807 // slower than the jitted code execution. If the tier-up strategy is
808 // turned on, we want to avoid this performance penalty so we eagerly
809 // tier-up if the subject string length is equal or greater than the given
810 // heuristic value.
811 regexp_data->MarkTierUpForNextExec();
812 if (v8_flags.trace_regexp_tier_up) {
813 PrintF(
814 "Forcing tier-up for very long strings in "
815 "RegExpImpl::IrregexpExec\n");
816 }
817 } else if (static_cast<uint32_t>(original_register_count) <
818 result_offsets_vector_length) {
819 // Tier up because the interpreter doesn't do global execution.
820 Cast<IrRegExpData>(regexp_data)->MarkTierUpForNextExec();
821 if (v8_flags.trace_regexp_tier_up) {
822 PrintF(
823 "Forcing tier-up of RegExpData object %p for global irregexp "
824 "mode\n",
825 reinterpret_cast<void*>(regexp_data->ptr()));
826 }
827 }
828 }
829
830 int output_register_count =
831 RegExpImpl::IrregexpPrepare(isolate, regexp_data, subject);
832 if (output_register_count < 0) {
833 DCHECK(isolate->has_exception());
834 return {};
835 }
836
837 // TODO(jgruber): Consider changing these into DCHECKs once we're convinced
838 // the conditions hold.
839 CHECK_EQ(original_register_count, output_register_count);
840 CHECK_LE(static_cast<uint32_t>(output_register_count),
841 result_offsets_vector_length);
842
843 RegExpStackScope stack_scope(isolate);
844
845 int res = RegExpImpl::IrregexpExecRaw(isolate, regexp_data, subject,
846 previous_index, result_offsets_vector,
847 result_offsets_vector_length);
848
849 if (res >= RegExp::RE_SUCCESS) {
850 DCHECK_LE(res * output_register_count, result_offsets_vector_length);
851 return res;
852 } else if (res == RegExp::RE_FALLBACK_TO_EXPERIMENTAL) {
853 return ExperimentalRegExp::OneshotExec(
854 isolate, regexp_data, subject, previous_index, result_offsets_vector,
855 result_offsets_vector_length);
856 } else if (res == RegExp::RE_EXCEPTION) {
857 DCHECK(isolate->has_exception());
858 return {};
859 } else {
860 DCHECK(res == RegExp::RE_FAILURE);
861 return 0;
862 }
863}

902// Returns true if we've either generated too much irregex code within this
903// isolate, or the pattern string is too long.
904bool TooMuchRegExpCode(Isolate* isolate, DirectHandle<String> pattern) {
905 // Limit the space regexps take up on the heap. In order to limit this we
906 // would like to keep track of the amount of regexp code on the heap. This
907 // is not tracked, however. As a conservative approximation we track the
908 // total regexp code compiled including code that has subsequently been freed
909 // and the total executable memory at any point.
910 static constexpr size_t kRegExpExecutableMemoryLimit = 16 * MB;
911 static constexpr size_t kRegExpCompiledLimit = 1 * MB;
912
913 Heap* heap = isolate->heap();
914 if (pattern->length() > RegExp::kRegExpTooLargeToOptimize) return true;
915 return (isolate->total_regexp_code_generated() > kRegExpCompiledLimit &&
916 heap->CommittedMemoryExecutable() > kRegExpExecutableMemoryLimit);
917}

932bool RegExpImpl::Compile(Isolate* isolate, Zone* zone, RegExpCompileData* data,
933 RegExpFlags flags, Handle<String> pattern,
934 Handle<String> sample_subject, bool is_one_byte,
935 uint32_t& backtrack_limit) {
936 if (JSRegExp::RegistersForCaptureCount(data->capture_count) >
937 RegExpMacroAssembler::kMaxRegisterCount) {
938 data->error = RegExpError::kTooLarge;
939 return false;
940 }
941
942 RegExpCompiler compiler(isolate, zone, data->capture_count, flags,
943 is_one_byte);
944
945 if (compiler.optimize()) {
946 compiler.set_optimize(!TooMuchRegExpCode(isolate, pattern));
947 }
948
949 // Sample some characters from the middle of the string.
950 static const int kSampleSize = 128;
951
952 sample_subject = String::Flatten(isolate, sample_subject);
953 uint32_t start, end;
954 if (sample_subject->length() > kSampleSize) {
955 start = (sample_subject->length() - kSampleSize) / 2;
956 end = start + kSampleSize;
957 } else {
958 start = 0;
959 end = sample_subject->length();
960 }
961 for (uint32_t i = start; i < end; i++) {
962 compiler.frequency_collator()->CountCharacter(sample_subject->Get(i));
963 }
964
965 data->node = compiler.PreprocessRegExp(data, is_one_byte);
966 if (data->error != RegExpError::kNone) {
967 return false;
968 }
969 data->error = AnalyzeRegExp(isolate, is_one_byte, flags, data->node);
970 if (data->error != RegExpError::kNone) {
971 return false;
972 }
973
974 if (v8_flags.trace_regexp_graph) DotPrinter::DotPrint("Start", data->node);
975
976 // Create the correct assembler for the architecture.
977 std::unique_ptr<RegExpMacroAssembler> macro_assembler;
978 if (data->compilation_target == RegExpCompilationTarget::kNative) {
979 // Native regexp implementation.
980 DCHECK(!v8_flags.jitless);
981
982 NativeRegExpMacroAssembler::Mode mode =
983 is_one_byte ? NativeRegExpMacroAssembler::LATIN1
984 : NativeRegExpMacroAssembler::UC16;
985
986 const int output_register_count =
987 JSRegExp::RegistersForCaptureCount(data->capture_count);
988#if V8_TARGET_ARCH_IA32
989 macro_assembler.reset(new RegExpMacroAssemblerIA32(isolate, zone, mode,
990 output_register_count));
991#elif V8_TARGET_ARCH_X64
992 macro_assembler.reset(new RegExpMacroAssemblerX64(isolate, zone, mode,
993 output_register_count));
994#elif V8_TARGET_ARCH_ARM
995 macro_assembler.reset(new RegExpMacroAssemblerARM(isolate, zone, mode,
996 output_register_count));
997#elif V8_TARGET_ARCH_ARM64
998 macro_assembler.reset(new RegExpMacroAssemblerARM64(isolate, zone, mode,
999 output_register_count));
1000#elif V8_TARGET_ARCH_S390X
1001 macro_assembler.reset(new RegExpMacroAssemblerS390(isolate, zone, mode,
1002 output_register_count));
1003#elif V8_TARGET_ARCH_PPC64
1004 macro_assembler.reset(new RegExpMacroAssemblerPPC(isolate, zone, mode,
1005 output_register_count));
1006#elif V8_TARGET_ARCH_MIPS64
1007 macro_assembler.reset(new RegExpMacroAssemblerMIPS(isolate, zone, mode,
1008 output_register_count));
1009#elif V8_TARGET_ARCH_RISCV64
1010 macro_assembler.reset(new RegExpMacroAssemblerRISCV(isolate, zone, mode,
1011 output_register_count));
1012#elif V8_TARGET_ARCH_RISCV32
1013 macro_assembler.reset(new RegExpMacroAssemblerRISCV(isolate, zone, mode,
1014 output_register_count));
1015#elif V8_TARGET_ARCH_LOONG64
1016 macro_assembler.reset(new RegExpMacroAssemblerLOONG64(
1017 isolate, zone, mode, output_register_count));
1018#else
1019#error "Unsupported architecture"
1020#endif
1021 } else {
1022 DCHECK_EQ(data->compilation_target, RegExpCompilationTarget::kBytecode);
1023 // Interpreted regexp implementation.
1024 macro_assembler.reset(new RegExpBytecodeGenerator(isolate, zone));
1025 }
1026
1027 macro_assembler->set_slow_safe(TooMuchRegExpCode(isolate, pattern));
1028 if (v8_flags.enable_experimental_regexp_engine_on_excessive_backtracks &&
1029 ExperimentalRegExp::CanBeHandled(data->tree, pattern, flags,
1030 data->capture_count)) {
1031 if (backtrack_limit == JSRegExp::kNoBacktrackLimit) {
1032 backtrack_limit = v8_flags.regexp_backtracks_before_fallback;
1033 } else {
1034 backtrack_limit = std::min(
1035 backtrack_limit, v8_flags.regexp_backtracks_before_fallback.value());
1036 }
1037 macro_assembler->set_backtrack_limit(backtrack_limit);
1038 macro_assembler->set_can_fallback(true);
1039 } else {
1040 macro_assembler->set_backtrack_limit(backtrack_limit);
1041 macro_assembler->set_can_fallback(false);
1042 }
1043
1044 // Inserted here, instead of in Assembler, because it depends on information
1045 // in the AST that isn't replicated in the Node structure.
1046 bool is_end_anchored = data->tree->IsAnchoredAtEnd();
1047 bool is_start_anchored = data->tree->IsAnchoredAtStart();
1048 int max_length = data->tree->max_match();
1049 static const int kMaxBacksearchLimit = 1024;
1050 if (is_end_anchored && !is_start_anchored && !IsSticky(flags) &&
1051 max_length < kMaxBacksearchLimit) {
1052 macro_assembler->SetCurrentPositionFromEnd(max_length);
1053 }
1054
1055 if (IsGlobal(flags)) {
1056 RegExpMacroAssembler::GlobalMode mode = RegExpMacroAssembler::GLOBAL;
1057 if (data->tree->min_match() > 0) {
1058 mode = RegExpMacroAssembler::GLOBAL_NO_ZERO_LENGTH_CHECK;
1059 } else if (IsEitherUnicode(flags)) {
1060 mode = RegExpMacroAssembler::GLOBAL_UNICODE;
1061 }
1062 macro_assembler->set_global_mode(mode);
1063 }
1064
1065 RegExpMacroAssembler* macro_assembler_ptr = macro_assembler.get();
1066#ifdef DEBUG
1067 std::unique_ptr<RegExpMacroAssembler> tracer_macro_assembler;
1068 if (v8_flags.trace_regexp_assembler) {
1069 tracer_macro_assembler.reset(
1070 new RegExpMacroAssemblerTracer(isolate, macro_assembler_ptr));
1071 macro_assembler_ptr = tracer_macro_assembler.get();
1072 }
1073#endif
1074
1075 RegExpCompiler::CompilationResult result = compiler.Assemble(
1076 isolate, macro_assembler_ptr, data->node, data->capture_count, pattern);
1077
1078 // Code / bytecode printing.
1079 {
1080#ifdef ENABLE_DISASSEMBLER
1081 if (v8_flags.print_regexp_code &&
1082 data->compilation_target == RegExpCompilationTarget::kNative) {
1083 CodeTracer::Scope trace_scope(isolate->GetCodeTracer());
1084 OFStream os(trace_scope.file());
1085 auto code = Cast<Code>(result.code);
1086 std::unique_ptr<char[]> pattern_cstring = pattern->ToCString();
1087 code->Disassemble(pattern_cstring.get(), os, isolate);
1088 }
1089#endif
1090 if (v8_flags.print_regexp_bytecode &&
1091 data->compilation_target == RegExpCompilationTarget::kBytecode) {
1092 auto bytecode = Cast<TrustedByteArray>(result.code);
1093 std::unique_ptr<char[]> pattern_cstring = pattern->ToCString();
1094 RegExpBytecodeDisassemble(bytecode->begin(), bytecode->length(),
1095 pattern_cstring.get());
1096 }
1097 }
1098
1099 if (result.error != RegExpError::kNone) {
1100 if (v8_flags.correctness_fuzzer_suppressions &&
1101 result.error == RegExpError::kStackOverflow) {
1102 FATAL("Aborting on stack overflow");
1103 }
1104 data->error = result.error;
1105 }
1106
1107 data->code = result.code;
1108 data->register_count = result.num_registers;
1109
1110 return result.Succeeded();
1111}

1113RegExpGlobalExecRunner::RegExpGlobalExecRunner(Handle<RegExpData> regexp_data,
1114 Handle<String> subject,
1115 Isolate* isolate)
1116 : result_vector_scope_(isolate),
1117 regexp_data_(regexp_data),
1118 subject_(subject),
1119 isolate_(isolate) {
1120 DCHECK(IsGlobal(JSRegExp::AsRegExpFlags(regexp_data->flags())));
1121
1122 switch (regexp_data_->type_tag()) {
1123 case RegExpData::Type::ATOM: {
1124 registers_per_match_ = JSRegExp::kAtomRegisterCount;
1125 register_array_size_ = Isolate::kJSRegexpStaticOffsetsVectorSize;
1126 break;
1127 }
1128 case RegExpData::Type::IRREGEXP: {
1129 registers_per_match_ = RegExpImpl::IrregexpPrepare(
1130 isolate_, Cast<IrRegExpData>(regexp_data_), subject_);
1131 if (registers_per_match_ < 0) {
1132 num_matches_ = -1; // Signal exception.
1133 return;
1134 }
1135 if (Cast<IrRegExpData>(regexp_data_)->ShouldProduceBytecode()) {
1136 // Global loop in interpreted regexp is not implemented. We choose the
1137 // size of the offsets vector so that it can only store one match.
1138 register_array_size_ = registers_per_match_;
1139 } else {
1140 register_array_size_ = std::max(
1141 {registers_per_match_, Isolate::kJSRegexpStaticOffsetsVectorSize});
1142 }
1143 break;
1144 }
1145 case RegExpData::Type::EXPERIMENTAL: {
1146 if (!ExperimentalRegExp::IsCompiled(Cast<IrRegExpData>(regexp_data_),
1147 isolate_) &&
1148 !ExperimentalRegExp::Compile(isolate_,
1149 Cast<IrRegExpData>(regexp_data_))) {
1150 DCHECK(isolate->has_exception());
1151 num_matches_ = -1; // Signal exception.
1152 return;
1153 }
1154 registers_per_match_ = JSRegExp::RegistersForCaptureCount(
1155 Cast<IrRegExpData>(regexp_data_)->capture_count());
1156 register_array_size_ = std::max(
1157 {registers_per_match_, Isolate::kJSRegexpStaticOffsetsVectorSize});
1158 break;
1159 }
1160 }
1161
1162 // Cache the result vector location.
1163
1164 register_array_ = result_vector_scope_.Initialize(register_array_size_);
1165
1166 // Set state so that fetching the results the first time triggers a call
1167 // to the compiled regexp.
1168 current_match_index_ = max_matches() - 1;
1169 num_matches_ = max_matches();
1170 DCHECK_LE(2, registers_per_match_); // Each match has at least one capture.
1171 DCHECK_GE(register_array_size_, registers_per_match_);
1172 int32_t* last_match =
1173 &register_array_[current_match_index_ * registers_per_match_];
1174 last_match[0] = -1;
1175 last_match[1] = 0;
1176}

1178int RegExpGlobalExecRunner::AdvanceZeroLength(int last_index) const {
1179 if (IsEitherUnicode(JSRegExp::AsRegExpFlags(regexp_data_->flags())) &&
1180 static_cast<uint32_t>(last_index + 1) < subject_->length() &&
1181 unibrow::Utf16::IsLeadSurrogate(subject_->Get(last_index)) &&
1182 unibrow::Utf16::IsTrailSurrogate(subject_->Get(last_index + 1))) {
1183 // Advance over the surrogate pair.
1184 return last_index + 2;
1185 }

1189int32_t* RegExpGlobalExecRunner::FetchNext() {
1190 current_match_index_++;
1191
1192 if (current_match_index_ >= num_matches_) {
1193 // Current batch of results exhausted.
1194 // Fail if last batch was not even fully filled.
1195 if (num_matches_ < max_matches()) {
1196 num_matches_ = 0; // Signal failed match.
1197 return nullptr;
1198 }
1199
1200 int32_t* last_match =
1201 &register_array_[(current_match_index_ - 1) * registers_per_match_];
1202 int last_end_index = last_match[1];
1203
1204 switch (regexp_data_->type_tag()) {
1205 case RegExpData::Type::ATOM:
1206 num_matches_ = RegExpImpl::AtomExecRaw(
1207 isolate_, Cast<AtomRegExpData>(regexp_data_), subject_,
1208 last_end_index, register_array_, register_array_size_);
1209 break;
1210 case RegExpData::Type::EXPERIMENTAL: {
1211 DCHECK(ExperimentalRegExp::IsCompiled(Cast<IrRegExpData>(regexp_data_),
1212 isolate_));
1213 DisallowGarbageCollection no_gc;
1214 num_matches_ = ExperimentalRegExp::ExecRaw(
1215 isolate_, RegExp::kFromRuntime, *Cast<IrRegExpData>(regexp_data_),
1216 *subject_, register_array_, register_array_size_, last_end_index);
1217 break;
1218 }
1219 case RegExpData::Type::IRREGEXP: {
1220 int last_start_index = last_match[0];
1221 if (last_start_index == last_end_index) {
1222 // Zero-length match. Advance by one code point.
1223 last_end_index = AdvanceZeroLength(last_end_index);
1224 }
1225 if (static_cast<uint32_t>(last_end_index) > subject_->length()) {
1226 num_matches_ = 0; // Signal failed match.
1227 return nullptr;
1228 }
1229 num_matches_ = RegExpImpl::IrregexpExecRaw(
1230 isolate_, Cast<IrRegExpData>(regexp_data_), subject_,
1231 last_end_index, register_array_, register_array_size_);
1232 break;
1233 }
1234 }
1235
1236 // Fall back to experimental engine if needed and possible.
1237 if (num_matches_ == RegExp::kInternalRegExpFallbackToExperimental) {
1238 num_matches_ = ExperimentalRegExp::OneshotExecRaw(
1239 isolate_, Cast<IrRegExpData>(regexp_data_), subject_, register_array_,
1240 register_array_size_, last_end_index);
1241 }
1242
1243 if (num_matches_ <= 0) {
1244 return nullptr;
1245 }
1246
1247 // Number of matches can't exceed maximum matches.
1248 // This check is enough to prevent OOB accesses to register_array_ in the
1249 // else branch below, since current_match_index < num_matches_ in this
1250 // branch, it follows that current_match_index < max_matches(). And since
1251 // max_matches() = register_array_size_ / registers_per_match it follows
1252 // that current_match_index * registers_per_match_ < register_array_size_.
1253 SBXCHECK_LE(num_matches_, max_matches());
1254
1255 current_match_index_ = 0;
1256 return register_array_;
1257 } else {
1258 return &register_array_[current_match_index_ * registers_per_match_];
1259 }
1260}

1262int32_t* RegExpGlobalExecRunner::LastSuccessfulMatch() const {
1263 int index = current_match_index_ * registers_per_match_;
1264 if (num_matches_ == 0) {
1265 // After a failed match we shift back by one result.
1266 index -= registers_per_match_;
1267 }
1268 return &register_array_[index];
1269}

1418 *number_of_matches_out = Smi::ToInt(cache->get(kNumberOfMatchesIndex));
1419 *last_match_index_out = Smi::ToInt(cache->get(kLastMatchIndexIndex));
