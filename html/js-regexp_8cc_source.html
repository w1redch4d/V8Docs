18Handle<JSRegExpResultIndices> JSRegExpResultIndices::BuildIndices(
19 Isolate* isolate, DirectHandle<RegExpMatchInfo> match_info,
20 Handle<Object> maybe_names) {
21 Handle<JSRegExpResultIndices> indices(
22 Cast<JSRegExpResultIndices>(isolate->factory()->NewJSObjectFromMap(
23 isolate->regexp_result_indices_map())));
24
25 // Initialize indices length to avoid having a partially initialized object
26 // should GC be triggered by creating a NewFixedArray.
27 indices->set_length(Smi::zero());
28
29 // Build indices array from RegExpMatchInfo.
30 int num_indices = match_info->number_of_capture_registers();
31 int num_results = num_indices >> 1;
32 DirectHandle<FixedArray> indices_array =
33 isolate->factory()->NewFixedArray(num_results);
34 JSArray::SetContent(indices, indices_array);
35
36 for (int i = 0; i < num_results; i++) {
37 const int start_offset =
38 match_info->capture(RegExpMatchInfo::capture_start_index(i));
39 const int end_offset =
40 match_info->capture(RegExpMatchInfo::capture_end_index(i));
41
42 // Any unmatched captures are set to undefined, otherwise we set them to a
43 // subarray of the indices.
44 if (start_offset == -1) {
45 indices_array->set(i, ReadOnlyRoots(isolate).undefined_value());
46 } else {
47 DirectHandle<FixedArray> indices_sub_array(
48 isolate->factory()->NewFixedArray(2));
49 indices_sub_array->set(0, Smi::FromInt(start_offset));
50 indices_sub_array->set(1, Smi::FromInt(end_offset));
51 DirectHandle<JSArray> indices_sub_jsarray =
52 isolate->factory()->NewJSArrayWithElements(indices_sub_array,
53 PACKED_SMI_ELEMENTS, 2);
54 indices_array->set(i, *indices_sub_jsarray);
55 }
56 }
57
58 // If there are no capture groups, set the groups property to undefined.
59 FieldIndex groups_index = FieldIndex::ForDescriptor(
60 indices->map(), InternalIndex(kGroupsDescriptorIndex));
61 if (IsUndefined(*maybe_names, isolate)) {
62 indices->FastPropertyAtPut(groups_index,
63 ReadOnlyRoots(isolate).undefined_value());
64 return indices;
65 }
66
67 // Create a groups property which returns a dictionary of named captures to
68 // their corresponding capture indices.
69 auto names = Cast<FixedArray>(maybe_names);
70 int num_names = names->length() >> 1;
71 Handle<HeapObject> group_names;
72 if constexpr (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
73 group_names = isolate->factory()->NewSwissNameDictionary(num_names);
74 } else {
75 group_names = isolate->factory()->NewNameDictionary(num_names);
76 }
77 Handle<PropertyDictionary> group_names_dict =
78 Cast<PropertyDictionary>(group_names);
79 for (int i = 0; i < num_names; i++) {
80 int base_offset = i * 2;
81 int name_offset = base_offset;
82 int index_offset = base_offset + 1;
83 DirectHandle<String> name(Cast<String>(names->get(name_offset)), isolate);
84 Tagged<Smi> smi_index = Cast<Smi>(names->get(index_offset));
85 Handle<Object> capture_indices(indices_array->get(smi_index.value()),
86 isolate);
87 if (!IsUndefined(*capture_indices, isolate)) {
88 capture_indices = Cast<JSArray>(capture_indices);
89 }
90 InternalIndex group_entry = group_names_dict->FindEntry(isolate, name);
91 // Duplicate group entries are possible if the capture groups are in
92 // different alternatives, i.e. only one of them can actually match.
93 // Therefore when we find a duplicate entry, either the current entry is
94 // undefined (didn't match anything) or the indices for the current capture
95 // are undefined. In the latter case we don't do anything, in the former
96 // case we update the entry.
97 if (group_entry.is_found()) {
98 DCHECK(v8_flags.js_regexp_duplicate_named_groups);
99 if (!IsUndefined(*capture_indices, isolate)) {
100 DCHECK(IsUndefined(group_names_dict->ValueAt(group_entry), isolate));
101 group_names_dict->ValueAtPut(group_entry, *capture_indices);
102 }
103 } else {
104 group_names_dict =
105 PropertyDictionary::Add(isolate, group_names_dict, name,
106 capture_indices, PropertyDetails::Empty());
107 }
108 }
109
110 // Convert group_names to a JSObject and store at the groups property of the
111 // result indices.
112 DirectHandle<FixedArrayBase> elements =
113 isolate->factory()->empty_fixed_array();
114 DirectHandle<Null> null = isolate->factory()->null_value();
115 DirectHandle<JSObject> js_group_names =
116 isolate->factory()->NewSlowJSObjectWithPropertiesAndElements(
117 null, group_names, elements);
118 indices->FastPropertyAtPut(groups_index, *js_group_names);
119 return indices;
120}

123std::optional<JSRegExp::Flags> JSRegExp::FlagsFromString(Isolate* isolate,
124 Handle<String> flags) {
125 const int length = flags->length();
126
127 // A longer flags string cannot be valid.
128 if (length > JSRegExp::kFlagCount) return {};
129
130 RegExpFlags value;
131 FlatStringReader reader(isolate, String::Flatten(isolate, flags));
132
133 for (int i = 0; i < length; i++) {
134 std::optional<RegExpFlag> flag = JSRegExp::FlagFromChar(reader.Get(i));
135 if (!flag.has_value()) return {};
136 if (value & flag.value()) return {}; // Duplicate.
137 value |= flag.value();
138 }
139
140 return JSRegExp::AsJSRegExpFlags(value);
141}

182bool IsLineTerminator(int c) {
183 // Expected to return true for '\n', '\r', 0x2028, and 0x2029.
184 return unibrow::IsLineTerminator(static_cast<unibrow::uchar>(c));
185}

187// TODO(jgruber): Consider merging CountAdditionalEscapeChars and
188// WriteEscapedRegExpSource into a single function to deduplicate dispatch logic
189// and move related code closer to each other.
190template <typename Char>
191int CountAdditionalEscapeChars(DirectHandle<String> source,
192 bool* needs_escapes_out) {
193 DisallowGarbageCollection no_gc;
194 int escapes = 0;
195 bool needs_escapes = false;
196 bool in_character_class = false;
197 base::Vector<const Char> src = source->GetCharVector<Char>(no_gc);
198 for (int i = 0; i < src.length(); i++) {
199 const Char c = src[i];
200 if (c == '\\') {
201 if (i + 1 < src.length() && IsLineTerminator(src[i + 1])) {
202 // This '\' is ignored since the next character itself will be escaped.
203 escapes--;
204 } else {
205 // Escape. Skip next character, which will be copied verbatim;
206 i++;
207 }
208 } else if (c == '/' && !in_character_class) {
209 // Not escaped forward-slash needs escape.
210 needs_escapes = true;
211 escapes++;
212 } else if (c == '[') {
213 in_character_class = true;
214 } else if (c == ']') {
215 in_character_class = false;
216 } else if (c == '\n') {
217 needs_escapes = true;
218 escapes++;
219 } else if (c == '\r') {
220 needs_escapes = true;
221 escapes++;
222 } else if (static_cast<int>(c) == 0x2028) {
223 needs_escapes = true;
224 escapes += std::strlen("\\u2028") - 1;
225 } else if (static_cast<int>(c) == 0x2029) {
226 needs_escapes = true;
227 escapes += std::strlen("\\u2029") - 1;
228 } else {
229 DCHECK(!IsLineTerminator(c));
230 }
231 }
232 DCHECK(!in_character_class);
233 DCHECK_GE(escapes, 0);
234 DCHECK_IMPLIES(escapes != 0, needs_escapes);
235 *needs_escapes_out = needs_escapes;
236 return escapes;
237}

239template <typename Char>
240void WriteStringToCharVector(base::Vector<Char> v, int* d, const char* string) {
241 int s = 0;
242 while (string[s] != '\0') v[(*d)++] = string[s++];
243}

245template <typename Char, typename StringType>
246Handle<StringType> WriteEscapedRegExpSource(DirectHandle<String> source,
247 Handle<StringType> result) {
248 DisallowGarbageCollection no_gc;
249 base::Vector<const Char> src = source->GetCharVector<Char>(no_gc);
250 base::Vector<Char> dst(result->GetChars(no_gc), result->length());
251 int s = 0;
252 int d = 0;
253 bool in_character_class = false;
254 while (s < src.length()) {
255 const Char c = src[s];
256 if (c == '\\') {
257 if (s + 1 < src.length() && IsLineTerminator(src[s + 1])) {
258 // This '\' is ignored since the next character itself will be escaped.
259 s++;
260 continue;
261 } else {
262 // Escape. Copy this and next character.
263 dst[d++] = src[s++];
264 }
265 if (s == src.length()) break;
266 } else if (c == '/' && !in_character_class) {
267 // Not escaped forward-slash needs escape.
268 dst[d++] = '\\';
269 } else if (c == '[') {
270 in_character_class = true;
271 } else if (c == ']') {
272 in_character_class = false;
273 } else if (c == '\n') {
274 WriteStringToCharVector(dst, &d, "\\n");
275 s++;
276 continue;
277 } else if (c == '\r') {
278 WriteStringToCharVector(dst, &d, "\\r");
279 s++;
280 continue;
281 } else if (static_cast<int>(c) == 0x2028) {
282 WriteStringToCharVector(dst, &d, "\\u2028");
283 s++;
284 continue;
285 } else if (static_cast<int>(c) == 0x2029) {
286 WriteStringToCharVector(dst, &d, "\\u2029");
287 s++;
288 continue;
289 } else {
290 DCHECK(!IsLineTerminator(c));
291 }
292 dst[d++] = src[s++];
293 }
294 DCHECK_EQ(result->length(), d);
295 DCHECK(!in_character_class);
296 return result;
297}

299MaybeHandle<String> EscapeRegExpSource(Isolate* isolate,
300 Handle<String> source) {
301 DCHECK(source->IsFlat());
302 if (source->length() == 0) return isolate->factory()->query_colon_string();
303 bool one_byte = source->IsOneByteRepresentation();
304 bool needs_escapes = false;
305 int additional_escape_chars =
306 one_byte ? CountAdditionalEscapeChars<uint8_t>(source, &needs_escapes)
307 : CountAdditionalEscapeChars<base::uc16>(source, &needs_escapes);
308 if (!needs_escapes) return source;
309 int length = source->length() + additional_escape_chars;
310 if (one_byte) {
311 Handle<SeqOneByteString> result;
312 ASSIGN_RETURN_ON_EXCEPTION(isolate, result,
313 isolate->factory()->NewRawOneByteString(length));
314 return WriteEscapedRegExpSource<uint8_t>(source, result);
315 } else {
316 Handle<SeqTwoByteString> result;
317 ASSIGN_RETURN_ON_EXCEPTION(isolate, result,
318 isolate->factory()->NewRawTwoByteString(length));
319 return WriteEscapedRegExpSource<base::uc16>(source, result);
320 }
321}

325// static
326MaybeHandle<JSRegExp> JSRegExp::Initialize(Handle<JSRegExp> regexp,
327 Handle<String> source, Flags flags,
328 uint32_t backtrack_limit) {
329 Isolate* isolate = regexp->GetIsolate();
330 Factory* factory = isolate->factory();
331 // If source is the empty string we set it to "(?:)" instead as
332 // suggested by ECMA-262, 5th, section 15.10.4.1.
333 if (source->length() == 0) source = factory->query_colon_string();
334
335 source = String::Flatten(isolate, source);
336
337 RETURN_ON_EXCEPTION(isolate, RegExp::Compile(isolate, regexp, source,
338 JSRegExp::AsRegExpFlags(flags),
339 backtrack_limit));
340
341 DirectHandle<String> escaped_source;
342 ASSIGN_RETURN_ON_EXCEPTION(isolate, escaped_source,
343 EscapeRegExpSource(isolate, source));
344
345 regexp->set_source(*escaped_source);
346 regexp->set_flags(Smi::FromInt(flags));
347
348 Tagged<Map> map = regexp->map();
349 Tagged<Object> constructor = map->GetConstructor();
350 if (IsJSFunction(constructor) &&
351 Cast<JSFunction>(constructor)->initial_map() == map) {
352 // If we still have the original map, set in-object properties directly.
353 regexp->InObjectPropertyAtPut(JSRegExp::kLastIndexFieldIndex,
354 Smi::FromInt(kInitialLastIndexValue),
355 SKIP_WRITE_BARRIER);
356 } else {
357 // Map has changed, so use generic, but slower, method.
358 RETURN_ON_EXCEPTION(
359 isolate,
360 Object::SetProperty(
361 isolate, regexp, factory->lastIndex_string(),
362 DirectHandle<Smi>(Smi::FromInt(kInitialLastIndexValue), isolate)));
363 }
364
365 return regexp;
366}

374// Only irregexps are subject to tier-up.
375bool IrRegExpData::CanTierUp() {
376 return v8_flags.regexp_tier_up && type_tag() == Type::IRREGEXP;
377}
378
379// An irregexp is considered to be marked for tier up if the tier-up ticks
380// value reaches zero.
381bool IrRegExpData::MarkedForTierUp() {
382 if (!CanTierUp()) {
383 return false;
384 }
385
386 return ticks_until_tier_up() == 0;
387}

390 DCHECK(v8_flags.regexp_tier_up);
391 int tier_up_ticks = ticks_until_tier_up();
392 set_ticks_until_tier_up(tier_up_ticks + 1);

395void IrRegExpData::TierUpTick() {
396 int tier_up_ticks = ticks_until_tier_up();
397 if (tier_up_ticks == 0) {
398 return;

400
401 set_ticks_until_tier_up(tier_up_ticks - 1);
402}

405 DCHECK(v8_flags.regexp_tier_up);
406 set_ticks_until_tier_up(0);
