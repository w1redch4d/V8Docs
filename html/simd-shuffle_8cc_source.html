15// Take a lane-wise shuffle and expand to a 16 byte-wise shuffle.
16template <size_t N>
17constexpr const SimdShuffle::ShuffleArray expand(
18 const std::array<uint8_t, N> in) {
19 SimdShuffle::ShuffleArray res{};
20 constexpr size_t lane_bytes = 16 / N;
21 for (unsigned i = 0; i < N; ++i) {
22 for (unsigned j = 0; j < lane_bytes; ++j) {
23 res[i * lane_bytes + j] = lane_bytes * in[i] + j;
24 }
25 }
26 return res;
27}

96bool SimdShuffle::TryMatch32x4Rotate(const uint8_t* shuffle,
97 uint8_t* shuffle32x4, bool is_swizzle) {
98 uint8_t offset;
99 bool is_concat = TryMatchConcat(shuffle, &offset);
100 DCHECK_NE(offset, 0); // 0 is identity, it should not be matched.
101 // Since we already have a concat shuffle, we know that the indices goes from:
102 // [ offset, ..., 15, 0, ... ], it suffices to check that the offset points
103 // to the low byte of a 32x4 element.
104 if (!is_concat || !is_swizzle || offset % 4 != 0) {
105 return false;
106 }
107
108 uint8_t offset_32 = offset / 4;
109 for (int i = 0; i < 4; i++) {
110 shuffle32x4[i] = (offset_32 + i) % 4;
111 }
112 return true;
113}

120bool SimdShuffle::TryMatch32x4OneLaneSwizzle(const uint8_t* shuffle32x4,
121 uint8_t* from_lane,
122 uint8_t* to_lane) {
123 constexpr uint32_t patterns[12]{
124 0x30200000, // 0 -> 1
125 0x30000100, // 0 -> 2
126 0x00020100, // 0 -> 3
127 0x03020101, // 1 -> 0
128 0x03010100, // 1 -> 2
129 0x01020100, // 1 -> 3
130 0x03020102, // 2 -> 0
131 0x03020200, // 2 -> 1
132 0x02020100, // 2 -> 3
133 0x03020103, // 3 -> 0
134 0x03020300, // 3 -> 1
135 0x03030100 // 3 -> 2
136 };
137
138 unsigned pattern_idx = 0;
139 uint32_t shuffle = *reinterpret_cast<const uint32_t*>(shuffle32x4);
140#ifdef V8_TARGET_BIG_ENDIAN
141 shuffle = base::bits::ReverseBytes(shuffle);
142#endif
143 for (unsigned from = 0; from < 4; ++from) {
144 for (unsigned to = 0; to < 4; ++to) {
145 if (from == to) {
146 continue;
147 }
148 if (shuffle == patterns[pattern_idx]) {
149 *from_lane = from;
150 *to_lane = to;
151 return true;
152 }
153 ++pattern_idx;
154 }
155 }
156 return false;
157}

159bool SimdShuffle::TryMatch64x2Shuffle(const uint8_t* shuffle,
160 uint8_t* shuffle64x2) {
161 constexpr std::array<std::array<uint8_t, 8>, 4> element_patterns = {
162 {{0, 1, 2, 3, 4, 5, 6, 7},
163 {8, 9, 10, 11, 12, 13, 14, 15},
164 {16, 17, 18, 19, 20, 21, 22, 23},
165 {24, 25, 26, 27, 28, 29, 30, 31}}};
166
167 for (unsigned i = 0; i < 2; ++i) {
168 uint64_t element = *reinterpret_cast<const uint64_t*>(&shuffle[i * 8]);
169 for (unsigned j = 0; j < 4; ++j) {
170 uint64_t pattern =
171 *reinterpret_cast<const uint64_t*>(element_patterns[j].data());
172 if (pattern == element) {
173 shuffle64x2[i] = j;
174 break;
175 }
176 if (j == 3) {
177 return false;
178 }
179 }
180 }
181 return true;
182}

184bool SimdShuffle::TryMatch32x4Shuffle(const uint8_t* shuffle,
185 uint8_t* shuffle32x4) {
186 for (int i = 0; i < 4; ++i) {
187 if (shuffle[i * 4] % 4 != 0) return false;
188 for (int j = 1; j < 4; ++j) {
189 if (shuffle[i * 4 + j] - shuffle[i * 4 + j - 1] != 1) return false;
190 }
191 shuffle32x4[i] = shuffle[i * 4] / 4;
192 }
193 return true;
194}

220bool SimdShuffle::TryMatchConcat(const uint8_t* shuffle, uint8_t* offset) {
221 // Don't match the identity shuffle (e.g. [0 1 2 ... 15]).
222 uint8_t start = shuffle[0];
223 if (start == 0) return false;
224 DCHECK_GT(kSimd128Size, start); // The shuffle should be canonicalized.
225 // A concatenation is a series of consecutive indices, with at most one jump
226 // in the middle from the last lane to the first.
227 for (int i = 1; i < kSimd128Size; ++i) {
228 if ((shuffle[i]) != ((shuffle[i - 1] + 1))) {
229 if (shuffle[i - 1] != 15) return false;
230 if (shuffle[i] % kSimd128Size != 0) return false;
231 }
232 }
233 *offset = start;
234 return true;
235}

244bool SimdShuffle::TryMatchByteToDwordZeroExtend(const uint8_t* shuffle) {
245 for (int i = 0; i < 16; ++i) {
246 if ((i % 4 != 0) && (shuffle[i] < 16)) return false;
247 if ((i % 4 == 0) && (shuffle[i] > 15 || (shuffle[i] != shuffle[0] + i / 4)))
248 return false;
249 }
250 return true;

254// Try to match the first step in a 32x4 pairwise shuffle.
255bool TryMatch32x4Pairwise(const uint8_t* shuffle) {
256 // Pattern to select 32-bit element 1.
257 constexpr uint8_t low_pattern_arr[4] = {4, 5, 6, 7};
258 // And we'll check that element 1 is shuffled into element 0.
259 uint32_t low_shuffle = reinterpret_cast<const uint32_t*>(shuffle)[0];
260 // Pattern to select 32-bit element 3.
261 constexpr uint8_t high_pattern_arr[4] = {12, 13, 14, 15};
262 // And we'll check that element 3 is shuffled into element 2.
263 uint32_t high_shuffle = reinterpret_cast<const uint32_t*>(shuffle)[2];
264 uint32_t low_pattern = *reinterpret_cast<const uint32_t*>(low_pattern_arr);
265 uint32_t high_pattern = *reinterpret_cast<const uint32_t*>(high_pattern_arr);
266 return low_shuffle == low_pattern && high_shuffle == high_pattern;
267}

269// Try to match the final step in a 32x4, now 32x2, pairwise shuffle.
270bool TryMatch32x2Pairwise(const uint8_t* shuffle) {
271 // Pattern to select 32-bit element 2.
272 constexpr uint8_t pattern_arr[4] = {8, 9, 10, 11};
273 // And we'll check that element 2 is shuffled to element 0.
274 uint32_t low_shuffle = reinterpret_cast<const uint32_t*>(shuffle)[0];
275 uint32_t pattern = *reinterpret_cast<const uint32_t*>(pattern_arr);
276 return low_shuffle == pattern;
277}

279// Try to match the first step in a upper-to-lower half shuffle.
280bool TryMatchUpperToLowerFirst(const uint8_t* shuffle) {
281 // There's 16 'active' bytes, so the pattern to select the upper half starts
282 // at byte 8.
283 constexpr uint8_t low_pattern_arr[8] = {8, 9, 10, 11, 12, 13, 14, 15};
284 // And we'll check that the top half is shuffled into the lower.
285 uint64_t low_shuffle = reinterpret_cast<const uint64_t*>(shuffle)[0];
286 uint64_t low_pattern = *reinterpret_cast<const uint64_t*>(low_pattern_arr);
287 return low_shuffle == low_pattern;
288}

290// Try to match the second step in a upper-to-lower half shuffle.
291bool TryMatchUpperToLowerSecond(const uint8_t* shuffle) {
292 // There's 8 'active' bytes, so the pattern to select the upper half starts
293 // at byte 4.
294 constexpr uint8_t low_pattern_arr[4] = {4, 5, 6, 7};
295 // And we'll check that the top half is shuffled into the lower.
296 uint32_t low_shuffle = reinterpret_cast<const uint32_t*>(shuffle)[0];
297 uint32_t low_pattern = *reinterpret_cast<const uint32_t*>(low_pattern_arr);
298 return low_shuffle == low_pattern;
299}

301// Try to match the third step in a upper-to-lower half shuffle.
302bool TryMatchUpperToLowerThird(const uint8_t* shuffle) {
303 // The vector now has 4 'active' bytes, select the top two.
304 constexpr uint8_t low_pattern_arr[2] = {2, 3};
305 // And check they're shuffled to the lower half.
306 uint16_t low_shuffle = reinterpret_cast<const uint16_t*>(shuffle)[0];
307 uint16_t low_pattern = *reinterpret_cast<const uint16_t*>(low_pattern_arr);
308 return low_shuffle == low_pattern;
309}

312bool TryMatchUpperToLowerFourth(const uint8_t* shuffle) {
313 return shuffle[0] == 1;

412// static
413bool SimdShuffle::TryMatchShufps256(const uint8_t* shuffle32x8,
414 uint8_t* control) {
415 *control = 0;
416 for (int i = 0; i < 4; ++i) {
417 // low 128-bits and high 128-bits should have the same shuffle order.
418 if (shuffle32x8[i + 4] - shuffle32x8[i] == 4) {
419 // [63:0] bits select from SRC1,
420 // [127:64] bits select from SRC2
421 if ((i < 2 && shuffle32x8[i] < 4) ||
422 (i >= 2 && shuffle32x8[i] >= 8 && shuffle32x8[i] < 12)) {
423 *control |= (shuffle32x8[i] % 4) << (2 * i);
424 continue;
425 }
426 return false;
427 }
428 return false;
429 }
430 return true;
431}
