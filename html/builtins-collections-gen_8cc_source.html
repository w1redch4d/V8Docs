28 Variant variant, TNode<Context> context, TNode<Object> collection,
29 TNode<Object> add_function, TNode<Object> key_value,
30 Label* if_may_have_side_effects, Label* if_exception,
31 TVariable<Object>* var_exception) {
32 compiler::ScopedExceptionHandler handler(this, if_exception, var_exception);
33 CSA_DCHECK(this, Word32BinaryNot(IsHashTableHole(key_value)));
34 if (variant == kMap || variant == kWeakMap) {
35 TorqueStructKeyValuePair pair =
36 if_may_have_side_effects != nullptr
37 ? LoadKeyValuePairNoSideEffects(context, key_value,
38 if_may_have_side_effects)
39 : LoadKeyValuePair(context, key_value);
40 TNode<Object> key_n = pair.key;
41 TNode<Object> value_n = pair.value;
42 Call(context, add_function, collection, key_n, value_n);
43 } else {
44 DCHECK(variant == kSet || variant == kWeakSet);
45 Call(context, add_function, collection, key_value);
46 }

49void BaseCollectionsAssembler::AddConstructorEntries(
50 Variant variant, TNode<Context> context,
51 TNode<NativeContext> native_context, TNode<HeapObject> collection,
52 TNode<Object> initial_entries) {
53 CSA_DCHECK(this, Word32BinaryNot(IsNullOrUndefined(initial_entries)));
54
55 enum Mode { kSlow, kFastJSArray, kFastCollection };
56 TVARIABLE(IntPtrT, var_at_least_space_for, IntPtrConstant(0));
57 TVARIABLE(HeapObject, var_entries_table, UndefinedConstant());
58 TVARIABLE(Int32T, var_mode, Int32Constant(kSlow));
59 Label if_fast_js_array(this), allocate_table(this);
60
61 // The slow path is taken if the initial add function is modified. This check
62 // must precede the kSet fast path below, which has the side effect of
63 // exhausting {initial_entries} if it is a JSSetIterator.
64 GotoIfInitialAddFunctionModified(variant, native_context, collection,
65 &allocate_table);
66
67 GotoIf(IsFastJSArrayWithNoCustomIteration(context, initial_entries),
68 &if_fast_js_array);
69 if (variant == Variant::kSet) {
70 GetEntriesIfFastCollectionOrIterable(
71 variant, initial_entries, context, &var_entries_table,
72 &var_at_least_space_for, &allocate_table);
73 var_mode = Int32Constant(kFastCollection);
74 Goto(&allocate_table);
75 } else {
76 Goto(&allocate_table);
77 }
78 BIND(&if_fast_js_array);
79 {
80 var_mode = Int32Constant(kFastJSArray);
81 if (variant == kWeakSet || variant == kWeakMap) {
82 var_at_least_space_for =
83 PositiveSmiUntag(LoadFastJSArrayLength(CAST(initial_entries)));
84 } else {
85 // TODO(ishell): consider using array length for all collections
86 static_assert(OrderedHashSet::kInitialCapacity ==
87 OrderedHashMap::kInitialCapacity);
88 var_at_least_space_for = IntPtrConstant(OrderedHashSet::kInitialCapacity);
89 }
90 Goto(&allocate_table);
91 }
92 TVARIABLE(JSReceiver, var_iterator_object);
93 TVARIABLE(Object, var_exception);
94 Label exit(this), from_fast_jsarray(this), from_fast_collection(this),
95 slow_loop(this, Label::kDeferred), if_exception(this, Label::kDeferred);
96 BIND(&allocate_table);
97 {
98 TNode<HeapObject> table =
99 AllocateTable(variant, var_at_least_space_for.value());
100 StoreObjectField(collection, GetTableOffset(variant), table);
101 if (variant == Variant::kSet) {
102 GotoIf(Word32Equal(var_mode.value(), Int32Constant(kFastCollection)),
103 &from_fast_collection);
104 }
105 Branch(Word32Equal(var_mode.value(), Int32Constant(kFastJSArray)),
106 &from_fast_jsarray, &slow_loop);
107 }
108 BIND(&from_fast_jsarray);
109 {
110 Label if_exception_during_fast_iteration(this, Label::kDeferred);
111 TVARIABLE(IntPtrT, var_index, IntPtrConstant(0));
112 TNode<JSArray> initial_entries_jsarray =
113 UncheckedCast<JSArray>(initial_entries);
114#if DEBUG
115 CSA_DCHECK(this, IsFastJSArrayWithNoCustomIteration(
116 context, initial_entries_jsarray));
117 TNode<Map> original_initial_entries_map = LoadMap(initial_entries_jsarray);
118#endif
119
120 Label if_may_have_side_effects(this, Label::kDeferred);
121 {
122 compiler::ScopedExceptionHandler handler(
123 this, &if_exception_during_fast_iteration, &var_exception);
124 AddConstructorEntriesFromFastJSArray(
125 variant, context, native_context, collection, initial_entries_jsarray,
126 &if_may_have_side_effects, var_index);
127 }
128 Goto(&exit);
129
130 if (variant == kMap || variant == kWeakMap) {
131 BIND(&if_may_have_side_effects);
132#if DEBUG
133 {
134 // Check that add/set function has not been modified.
135 Label if_not_modified(this), if_modified(this);
136 GotoIfInitialAddFunctionModified(variant, native_context, collection,
137 &if_modified);
138 Goto(&if_not_modified);
139 BIND(&if_modified);
140 Unreachable();
141 BIND(&if_not_modified);
142 }
143 CSA_DCHECK(this, TaggedEqual(original_initial_entries_map,
144 LoadMap(initial_entries_jsarray)));
145#endif
146 var_mode = Int32Constant(kSlow);
147 Goto(&allocate_table);
148 }
149 BIND(&if_exception_during_fast_iteration);
150 {
151 // In case exception is thrown during collection population, materialize
152 // the iteator and execute iterator closing protocol. It might be
153 // non-trivial in case "return" callback is added somewhere in the
154 // iterator's prototype chain.
155 TNode<NativeContext> native_context = LoadNativeContext(context);
156 TNode<IntPtrT> next_index =
157 IntPtrAdd(var_index.value(), IntPtrConstant(1));
158 var_iterator_object = CreateArrayIterator(
159 native_context, UncheckedCast<JSArray>(initial_entries),
160 IterationKind::kValues, SmiTag(next_index));
161 Goto(&if_exception);
162 }
163 }
164 if (variant == Variant::kSet) {
165 BIND(&from_fast_collection);
166 {
167 AddConstructorEntriesFromFastCollection(variant, collection,
168 var_entries_table.value());
169 Goto(&exit);
170 }
171 }
172 BIND(&slow_loop);
173 {
174 AddConstructorEntriesFromIterable(
175 variant, context, native_context, collection, initial_entries,
176 &if_exception, &var_iterator_object, &var_exception);
177 Goto(&exit);
178 }
179 BIND(&if_exception);
180 {
181 TNode<HeapObject> message = GetPendingMessage();
182 SetPendingMessage(TheHoleConstant());
183 // iterator.next field is not used by IteratorCloseOnException.
184 TorqueStructIteratorRecord iterator = {var_iterator_object.value(), {}};
185 IteratorCloseOnException(context, iterator);
186 CallRuntime(Runtime::kReThrowWithMessage, context, var_exception.value(),
187 message);
188 Unreachable();
189 }
190 BIND(&exit);
191}

193void BaseCollectionsAssembler::AddConstructorEntriesFromFastJSArray(
194 Variant variant, TNode<Context> context, TNode<Context> native_context,
195 TNode<Object> collection, TNode<JSArray> fast_jsarray,
196 Label* if_may_have_side_effects, TVariable<IntPtrT>& var_current_index) {
197 TNode<FixedArrayBase> elements = LoadElements(fast_jsarray);
198 TNode<Int32T> elements_kind = LoadElementsKind(fast_jsarray);
199 TNode<JSFunction> add_func = GetInitialAddFunction(variant, native_context);
200 CSA_DCHECK(this,
201 TaggedEqual(GetAddFunction(variant, native_context, collection),
202 add_func));
203 CSA_DCHECK(this, IsFastJSArrayWithNoCustomIteration(context, fast_jsarray));
204 TNode<IntPtrT> length = PositiveSmiUntag(LoadFastJSArrayLength(fast_jsarray));
205 CSA_DCHECK(
206 this, HasInitialCollectionPrototype(variant, native_context, collection));
207
208#if DEBUG
209 TNode<Map> original_collection_map = LoadMap(CAST(collection));
210 TNode<Map> original_fast_js_array_map = LoadMap(fast_jsarray);
211#endif
212 Label exit(this), if_doubles(this), if_smiorobjects(this);
213 GotoIf(IntPtrEqual(length, IntPtrConstant(0)), &exit);
214 Branch(IsFastSmiOrTaggedElementsKind(elements_kind), &if_smiorobjects,
215 &if_doubles);
216 BIND(&if_smiorobjects);
217 {
218 auto set_entry = [&](TNode<IntPtrT> index) {
219 TNode<Object> element =
220 LoadAndNormalizeFixedArrayElement(CAST(elements), index);
221 AddConstructorEntry(variant, context, collection, add_func, element,
222 if_may_have_side_effects);
223 };
224
225 // Instead of using the slower iteration protocol to iterate over the
226 // elements, a fast loop is used. This assumes that adding an element
227 // to the collection does not call user code that could mutate the elements
228 // or collection.
229 BuildFastLoop<IntPtrT>(var_current_index, IntPtrConstant(0), length,
230 set_entry, 1, LoopUnrollingMode::kNo,
231 IndexAdvanceMode::kPost);
232 Goto(&exit);
233 }
234 BIND(&if_doubles);
235 {
236 // A Map constructor requires entries to be arrays (ex. [key, value]),
237 // so a FixedDoubleArray can never succeed.
238 if (variant == kMap || variant == kWeakMap) {
239 CSA_DCHECK(this, IntPtrGreaterThan(length, IntPtrConstant(0)));
240 TNode<Object> element =
241 LoadAndNormalizeFixedDoubleArrayElement(elements, IntPtrConstant(0));
242 ThrowTypeError(context, MessageTemplate::kIteratorValueNotAnObject,
243 element);
244 } else {
245 DCHECK(variant == kSet || variant == kWeakSet);
246 auto set_entry = [&](TNode<IntPtrT> index) {
247 TNode<Object> entry = LoadAndNormalizeFixedDoubleArrayElement(
248 elements, UncheckedCast<IntPtrT>(index));
249 AddConstructorEntry(variant, context, collection, add_func, entry);
250 };
251 BuildFastLoop<IntPtrT>(var_current_index, IntPtrConstant(0), length,
252 set_entry, 1, LoopUnrollingMode::kNo,
253 IndexAdvanceMode::kPost);
254 Goto(&exit);
255 }
256 }
257 BIND(&exit);
258#if DEBUG
259 CSA_DCHECK(this,
260 TaggedEqual(original_collection_map, LoadMap(CAST(collection))));
261 CSA_DCHECK(this,
262 TaggedEqual(original_fast_js_array_map, LoadMap(fast_jsarray)));
263#endif
264}

266void BaseCollectionsAssembler::AddConstructorEntriesFromIterable(
267 Variant variant, TNode<Context> context, TNode<Context> native_context,
268 TNode<Object> collection, TNode<Object> iterable, Label* if_exception,
269 TVariable<JSReceiver>* var_iterator_object,
270 TVariable<Object>* var_exception) {
271 Label exit(this), loop(this);
272 CSA_DCHECK(this, Word32BinaryNot(IsNullOrUndefined(iterable)));
273 TNode<Object> add_func = GetAddFunction(variant, context, collection);
274 IteratorBuiltinsAssembler iterator_assembler(this->state());
275 TorqueStructIteratorRecord iterator =
276 iterator_assembler.GetIterator(context, iterable);
277 *var_iterator_object = iterator.object;
278
279 CSA_DCHECK(this, Word32BinaryNot(IsUndefined(iterator.object)));
280
281 TNode<Map> fast_iterator_result_map = CAST(
282 LoadContextElement(native_context, Context::ITERATOR_RESULT_MAP_INDEX));
283
284 Goto(&loop);
285 BIND(&loop);
286 {
287 TNode<JSReceiver> next = iterator_assembler.IteratorStep(
288 context, iterator, &exit, fast_iterator_result_map);
289 TNode<Object> next_value = iterator_assembler.IteratorValue(
290 context, next, fast_iterator_result_map);
291 AddConstructorEntry(variant, context, collection, add_func, next_value,
292 nullptr, if_exception, var_exception);
293 Goto(&loop);
294 }
295 BIND(&exit);
296}

373void BaseCollectionsAssembler::GenerateConstructor(
374 Variant variant, Handle<String> constructor_function_name,
375 TNode<Object> new_target, TNode<IntPtrT> argc, TNode<Context> context) {
376 const int kIterableArg = 0;
377 CodeStubArguments args(this, argc);
378 TNode<Object> iterable = args.GetOptionalArgumentValue(kIterableArg);
379
380 Label if_undefined(this, Label::kDeferred);
381 GotoIf(IsUndefined(new_target), &if_undefined);
382
383 TNode<NativeContext> native_context = LoadNativeContext(context);
384 TNode<JSObject> collection = AllocateJSCollection(
385 context, GetConstructor(variant, native_context), CAST(new_target));
386
387 Label add_constructor_entries(this);
388
389 // The empty case.
390 //
391 // This is handled specially to simplify AddConstructorEntries, which is
392 // complex and contains multiple fast paths.
393 GotoIfNot(IsNullOrUndefined(iterable), &add_constructor_entries);
394 TNode<HeapObject> table = AllocateTable(variant, IntPtrConstant(0));
395 StoreObjectField(collection, GetTableOffset(variant), table);
396 Return(collection);
397
398 BIND(&add_constructor_entries);
399 AddConstructorEntries(variant, context, native_context, collection, iterable);
400 Return(collection);
401
402 BIND(&if_undefined);
403 ThrowTypeError(context, MessageTemplate::kConstructorNotFunction,
404 HeapConstantNoHole(constructor_function_name));
405}

560template <typename CollectionType>
561void CollectionsBuiltinsAssembler::FindOrderedHashTableEntry(
562 const TNode<CollectionType> table, const TNode<Uint32T> hash,
563 const std::function<void(TNode<Object>, Label*, Label*)>& key_compare,
564 TVariable<IntPtrT>* entry_start_position, Label* entry_found,
565 Label* not_found) {
566 // Get the index of the bucket.
567 const TNode<Uint32T> number_of_buckets =
568 PositiveSmiToUint32(CAST(UnsafeLoadFixedArrayElement(
569 table, CollectionType::NumberOfBucketsIndex())));
570 const TNode<Uint32T> bucket =
571 Word32And(hash, Uint32Sub(number_of_buckets, Uint32Constant(1)));
572 const TNode<IntPtrT> first_entry = SmiUntag(CAST(UnsafeLoadFixedArrayElement(
573 table, Signed(ChangeUint32ToWord(bucket)),
574 CollectionType::HashTableStartIndex() * kTaggedSize)));
575 const TNode<IntPtrT> number_of_buckets_intptr =
576 Signed(ChangeUint32ToWord(number_of_buckets));
577
578 // Walk the bucket chain.
579 TNode<IntPtrT> entry_start;
580 Label if_key_found(this);
581 {
582 TVARIABLE(IntPtrT, var_entry, first_entry);
583 Label loop(this, {&var_entry, entry_start_position}),
584 continue_next_entry(this);
585 Goto(&loop);
586 BIND(&loop);
587
588 // If the entry index is the not-found sentinel, we are done.
589 GotoIf(IntPtrEqual(var_entry.value(),
590 IntPtrConstant(CollectionType::kNotFound)),
591 not_found);
592
593 // Make sure the entry index is within range.
594 CSA_DCHECK(
595 this,
596 UintPtrLessThan(
597 var_entry.value(),
598 PositiveSmiUntag(SmiAdd(
599 CAST(UnsafeLoadFixedArrayElement(
600 table, CollectionType::NumberOfElementsIndex())),
601 CAST(UnsafeLoadFixedArrayElement(
602 table, CollectionType::NumberOfDeletedElementsIndex()))))));
603
604 // Compute the index of the entry relative to kHashTableStartIndex.
605 entry_start =
606 IntPtrAdd(IntPtrMul(var_entry.value(),
607 IntPtrConstant(CollectionType::kEntrySize)),
608 number_of_buckets_intptr);
609
610 // Load the key from the entry.
611 const TNode<Object> candidate_key =
612 UnsafeLoadKeyFromOrderedHashTableEntry(table, entry_start);
613
614 key_compare(candidate_key, &if_key_found, &continue_next_entry);
615
616 BIND(&continue_next_entry);
617 // Load the index of the next entry in the bucket chain.
618 var_entry = SmiUntag(CAST(UnsafeLoadFixedArrayElement(
619 table, entry_start,
620 (CollectionType::HashTableStartIndex() + CollectionType::kChainOffset) *
621 kTaggedSize)));
622
623 Goto(&loop);
624 }
625
626 BIND(&if_key_found);

655void CollectionsBuiltinsAssembler::AddConstructorEntriesFromSet(
656 TNode<JSSet> collection, TNode<OrderedHashSet> table) {
657 TNode<OrderedHashSet> entry_table = LoadObjectField<OrderedHashSet>(
658 collection, GetTableOffset(Variant::kSet));
659
660 TNode<IntPtrT> number_of_buckets =
661 PositiveSmiUntag(CAST(UnsafeLoadFixedArrayElement(
662 table, OrderedHashSet::NumberOfBucketsIndex())));
663 TNode<IntPtrT> number_of_elements = LoadAndUntagPositiveSmiObjectField(
664 table, OrderedHashSet::NumberOfElementsOffset());
665 TNode<IntPtrT> number_of_deleted_elements = PositiveSmiUntag(CAST(
666 LoadObjectField(table, OrderedHashSet::NumberOfDeletedElementsOffset())));
667 TNode<IntPtrT> used_capacity =
668 IntPtrAdd(number_of_elements, number_of_deleted_elements);
669 TNode<IntPtrT> loop_bound = IntPtrAdd(
670 IntPtrMul(used_capacity, IntPtrConstant(OrderedHashSet::kEntrySize)),
671 number_of_buckets);
672
673 TNode<IntPtrT> number_of_buckets_entry_table =
674 PositiveSmiUntag(CAST(UnsafeLoadFixedArrayElement(
675 entry_table, OrderedHashSet::NumberOfBucketsIndex())));
676
677 TVARIABLE(Object, entry_key);
678 TVARIABLE(IntPtrT, var_entry_table_occupancy, IntPtrConstant(0));
679 VariableList loop_vars({&var_entry_table_occupancy}, zone());
680 Label exit(this);
681
682 auto set_entry = [&](TNode<IntPtrT> index) {
683 entry_key = UnsafeLoadKeyFromOrderedHashTableEntry(table, index);
684 Label if_key_is_not_hole(this), continue_loop(this);
685 Branch(IsHashTableHole(entry_key.value()), &continue_loop,
686 &if_key_is_not_hole);
687 BIND(&if_key_is_not_hole);
688 {
689 AddNewToOrderedHashSet(entry_table, entry_key.value(),
690 number_of_buckets_entry_table,
691 var_entry_table_occupancy.value());
692 Increment(&var_entry_table_occupancy, 1);
693 Goto(&continue_loop);
694 }
695 BIND(&continue_loop);
696 return;
697 };
698
699 // Instead of using the slower iteration protocol to iterate over the
700 // elements, a fast loop is used. This assumes that adding an element
701 // to the collection does not call user code that could mutate the elements
702 // or collection. The iteration is based on the layout of the ordered hash
703 // table.
704 BuildFastLoop<IntPtrT>(loop_vars, number_of_buckets, loop_bound, set_entry,
705 OrderedHashSet::kEntrySize, LoopUnrollingMode::kNo,
706 IndexAdvanceMode::kPost);
707 Goto(&exit);
708 BIND(&exit);
709}

1652template <typename CollectionType>
1653TNode<CollectionType> CollectionsBuiltinsAssembler::AddToOrderedHashTable(
1654 const TNode<CollectionType> table, const TNode<Object> key,
1655 const GrowCollection<CollectionType>& grow,
1656 const StoreAtEntry<CollectionType>& store_at_new_entry,
1657 const StoreAtEntry<CollectionType>& store_at_existing_entry) {
1658 TVARIABLE(CollectionType, table_var, table);
1659 TVARIABLE(IntPtrT, entry_start_position_or_hash, IntPtrConstant(0));
1660 Label entry_found(this), not_found(this), done(this);
1661
1662 TryLookupOrderedHashTableIndex<CollectionType>(
1663 table, key, &entry_start_position_or_hash, &entry_found, &not_found);
1664
1665 BIND(&entry_found);
1666 {
1667 // If we found the entry, we just store the value there.
1668 store_at_existing_entry(table, entry_start_position_or_hash.value());
1669 Goto(&done);
1670 }
1671
1672 Label no_hash(this), add_entry(this), store_new_entry(this);
1673 BIND(&not_found);
1674 {
1675 // If we have a hash code, we can start adding the new entry.
1676 GotoIf(IntPtrGreaterThan(entry_start_position_or_hash.value(),
1677 IntPtrConstant(0)),
1678 &add_entry);
1679
1680 // Otherwise, go to runtime to compute the hash code.
1681 entry_start_position_or_hash = SmiUntag(CallGetOrCreateHashRaw(CAST(key)));
1682 Goto(&add_entry);
1683 }
1684
1685 BIND(&add_entry);
1686 TVARIABLE(IntPtrT, number_of_buckets);
1687 TVARIABLE(IntPtrT, occupancy);
1688 {
1689 // Check we have enough space for the entry.
1690 number_of_buckets = PositiveSmiUntag(CAST(UnsafeLoadFixedArrayElement(
1691 table, CollectionType::NumberOfBucketsIndex())));
1692
1693 static_assert(CollectionType::kLoadFactor == 2);
1694 const TNode<WordT> capacity = WordShl(number_of_buckets.value(), 1);
1695 const TNode<IntPtrT> number_of_elements =
1696 LoadAndUntagPositiveSmiObjectField(
1697 table, CollectionType::NumberOfElementsOffset());
1698 const TNode<IntPtrT> number_of_deleted =
1699 PositiveSmiUntag(CAST(LoadObjectField(
1700 table, CollectionType::NumberOfDeletedElementsOffset())));
1701 occupancy = IntPtrAdd(number_of_elements, number_of_deleted);
1702 GotoIf(IntPtrLessThan(occupancy.value(), capacity), &store_new_entry);
1703
1704 // We do not have enough space, grow the table and reload the relevant
1705 // fields.
1706 table_var = grow();
1707 number_of_buckets = PositiveSmiUntag(CAST(UnsafeLoadFixedArrayElement(
1708 table_var.value(), CollectionType::NumberOfBucketsIndex())));
1709 const TNode<IntPtrT> new_number_of_elements =
1710 LoadAndUntagPositiveSmiObjectField(
1711 table_var.value(), CollectionType::NumberOfElementsOffset());
1712 const TNode<IntPtrT> new_number_of_deleted = PositiveSmiUntag(
1713 CAST(LoadObjectField(table_var.value(),
1714 CollectionType::NumberOfDeletedElementsOffset())));
1715 occupancy = IntPtrAdd(new_number_of_elements, new_number_of_deleted);
1716 Goto(&store_new_entry);
1717 }
1718
1719 BIND(&store_new_entry);
1720 {
1721 StoreOrderedHashTableNewEntry(
1722 table_var.value(), entry_start_position_or_hash.value(),
1723 number_of_buckets.value(), occupancy.value(), store_at_new_entry);
1724 Goto(&done);
1725 }
1726
1727 BIND(&done);
1728 return table_var.value();
1729}

1768 const TNode<CollectionType> table, const TNode<IntPtrT> hash,
1769 const TNode<IntPtrT> number_of_buckets, const TNode<IntPtrT> occupancy,

1772 WordAnd(hash, IntPtrSub(number_of_buckets, IntPtrConstant(1)));
