368Expression* MakeCall(IdentifierExpression* callee,
369 std::optional<Expression*> target,
370 std::vector<Expression*> arguments,
371 const std::vector<Statement*>& otherwise) {
372 std::vector<Identifier*> labels;
373
374 // All IdentifierExpressions are treated as label names and can be directly
375 // used as labels identifiers. All other statements in a call's otherwise
376 // must create intermediate Labels for the otherwise's statement code.
377 size_t label_id_count = 0;
378 std::vector<TryHandler*> temp_labels;
379 for (auto* statement : otherwise) {
380 if (auto* e = ExpressionStatement::DynamicCast(statement)) {
381 if (auto* id = IdentifierExpression::DynamicCast(e->expression)) {
382 if (!id->generic_arguments.empty()) {
383 ReportError("An otherwise label cannot have generic parameters");
384 }
385 labels.push_back(id->name);
386 continue;
387 }
388 }
389 auto label_name = std::string("__label") + std::to_string(label_id_count++);
390 auto label_id = MakeNode<Identifier>(label_name);
391 label_id->pos = SourcePosition::Invalid();
392 labels.push_back(label_id);
393 auto* handler =
394 MakeNode<TryHandler>(TryHandler::HandlerKind::kLabel, label_id,
395 ParameterList::Empty(), statement);
396 temp_labels.push_back(handler);
397 }
398
399 // Create nested try-label expression for all of the temporary Labels that
400 // were created.
401 Expression* result = nullptr;
402 if (target) {
403 result = MakeNode<CallMethodExpression>(
404 *target, callee, std::move(arguments), std::move(labels));
405 } else {
406 result = MakeNode<CallExpression>(callee, std::move(arguments),
407 std::move(labels));
408 }
409
410 for (auto* label : temp_labels) {
411 result = MakeNode<TryLabelExpression>(result, label);
412 }
413 return result;
414}

507template <bool has_varargs, bool has_explicit_parameter_names>
508std::optional<ParseResult> MakeParameterList(
509 ParseResultIterator* child_results) {
510 auto implicit_params =
511 child_results->NextAs<std::optional<ImplicitParameters>>();
512 ParameterList result;
513 result.has_varargs = has_varargs;
514 result.implicit_count = 0;
515 result.implicit_kind = ImplicitKind::kNoImplicit;
516 if (implicit_params) {
517 result.implicit_count = implicit_params->parameters.size();
518 if (implicit_params->kind->value == "implicit") {
519 result.implicit_kind = ImplicitKind::kImplicit;
520 } else {
521 DCHECK_EQ(implicit_params->kind->value, "js-implicit");
522 result.implicit_kind = ImplicitKind::kJSImplicit;
523 }
524 result.implicit_kind_pos = implicit_params->kind->pos;
525 for (NameAndTypeExpression& implicit_param : implicit_params->parameters) {
526 AddParameter(&result, implicit_param);
527 }
528 }
529 if (has_explicit_parameter_names) {
530 auto explicit_params =
531 child_results->NextAs<std::vector<NameAndTypeExpression>>();
532 std::string arguments_variable = "";
533 if (has_varargs) {
534 arguments_variable = child_results->NextAs<std::string>();
535 }
536 for (NameAndTypeExpression& param : explicit_params) {
537 AddParameter(&result, param);
538 }
539 result.arguments_variable = arguments_variable;
540 } else {
541 auto explicit_types = child_results->NextAs<TypeList>();
542 for (auto* explicit_type : explicit_types) {
543 result.types.push_back(explicit_type);
544 }
545 }
546 return ParseResult{std::move(result)};
547}

549std::optional<ParseResult> MakeAssertStatement(
550 ParseResultIterator* child_results) {
551 auto kind_string = child_results->NextAs<Identifier*>()->value;
552 auto expr_with_source = child_results->NextAs<ExpressionWithSource>();
553 AssertStatement::AssertKind kind;
554 if (kind_string == "dcheck") {
555 kind = AssertStatement::AssertKind::kDcheck;
556 } else if (kind_string == "check") {
557 kind = AssertStatement::AssertKind::kCheck;
558 } else if (kind_string == "sbxcheck") {
559#ifdef V8_ENABLE_SANDBOX
560 kind = AssertStatement::AssertKind::kSbxCheck;
561#else
562 kind = AssertStatement::AssertKind::kDcheck;
563#endif // V8_ENABLE_SANDBOX
564 } else if (kind_string == "static_assert") {
565 kind = AssertStatement::AssertKind::kStaticAssert;
566 } else {
567 UNREACHABLE();
568 }
569 Statement* result = MakeNode<AssertStatement>(
570 kind, expr_with_source.expression, expr_with_source.source);
571 return ParseResult{result};
572}

593std::optional<ParseResult> MakeExternalMacro(
594 ParseResultIterator* child_results) {
595 auto transitioning = child_results->NextAs<bool>();
596 auto operator_name = child_results->NextAs<std::optional<std::string>>();
597 auto external_assembler_name =
598 child_results->NextAs<std::optional<std::string>>();
599 auto name = child_results->NextAs<Identifier*>();
600 auto generic_parameters = child_results->NextAs<GenericParameters>();
601 LintGenericParameters(generic_parameters);
602
603 auto args = child_results->NextAs<ParameterList>();
604 auto return_type = child_results->NextAs<TypeExpression*>();
605 auto labels = child_results->NextAs<LabelAndTypesVector>();
606
607 Declaration* result = MakeNode<ExternalMacroDeclaration>(
608 transitioning,
609 external_assembler_name ? *external_assembler_name : "CodeStubAssembler",
610 name, operator_name, args, return_type, std::move(labels));
611 if (!generic_parameters.empty()) {
612 Error("External builtins cannot be generic.");
613 }
614 return ParseResult{result};
615}

617std::optional<ParseResult> MakeIntrinsicDeclaration(
618 ParseResultIterator* child_results) {
619 auto name = child_results->NextAs<Identifier*>();
620 auto generic_parameters = child_results->NextAs<GenericParameters>();
621 LintGenericParameters(generic_parameters);
622
623 auto args = child_results->NextAs<ParameterList>();
624 auto return_type = child_results->NextAs<TypeExpression*>();
625 auto body = child_results->NextAs<std::optional<Statement*>>();
626 CallableDeclaration* declaration;
627 if (body) {
628 declaration = MakeNode<TorqueMacroDeclaration>(
629 false, name, std::optional<std::string>{}, args, return_type,
630 LabelAndTypesVector{}, false, body);
631 } else {
632 declaration = MakeNode<IntrinsicDeclaration>(name, args, return_type);
633 }
634 Declaration* result = declaration;
635 if (!generic_parameters.empty()) {
636 result = MakeNode<GenericCallableDeclaration>(std::move(generic_parameters),
637 declaration);
638 }
639 return ParseResult{result};

796class AnnotationSet {
797 public:
798 AnnotationSet(ParseResultIterator* iter,
799 const std::set<std::string>& allowed_without_param,
800 const std::set<std::string>& allowed_with_param) {
801 auto list = iter->NextAs<std::vector<Annotation>>();
802 for (const Annotation& a : list) {
803 if (a.param.has_value()) {
804 if (allowed_with_param.find(a.name->value) ==
805 allowed_with_param.end()) {
806 const char* error_message =
807 allowed_without_param.find(a.name->value) ==
808 allowed_without_param.end()
809 ? " is not allowed here"
810 : " cannot have parameter here";
811 Lint("Annotation ", a.name->value, error_message)
812 .Position(a.name->pos);
813 }
814 if (!map_.insert({a.name->value, {*a.param, a.name->pos}}).second) {
815 Lint("Duplicate annotation ", a.name->value).Position(a.name->pos);
816 }
817 } else {
818 if (allowed_without_param.find(a.name->value) ==
819 allowed_without_param.end()) {
820 const char* error_message =
821 allowed_with_param.find(a.name->value) == allowed_with_param.end()
822 ? " is not allowed here"
823 : " requires a parameter here";
824 Lint("Annotation ", a.name->value, error_message)
825 .Position(a.name->pos);
826 }
827 if (!set_.insert(a.name->value).second) {
828 Lint("Duplicate annotation ", a.name->value).Position(a.name->pos);
829 }
830 }
831 }
832 }

835 return set_.find(s) != set_.end();

837 std::optional<std::string> GetStringParam(const std::string& s) const {
838 auto it = map_.find(s);
839 if (it == map_.end()) {
840 return {};
841 }
842 if (it->second.first.is_int) {
843 Error("Annotation ", s, " requires a string parameter but has an int")
844 .Position(it->second.second);
845 }
846 return it->second.first.string_value;
847 }
848 std::optional<int32_t> GetIntParam(const std::string& s) const {
849 auto it = map_.find(s);
850 if (it == map_.end()) {
851 return {};
852 }
853 if (!it->second.first.is_int) {
854 Error("Annotation ", s, " requires an int parameter but has a string")
855 .Position(it->second.second);
856 }
857 return it->second.first.int_value;
858 }

860 private:
861 std::set<std::string> set_;
862 std::map<std::string, std::pair<AnnotationParameter, SourcePosition>> map_;
863};

1018std::optional<ParseResult> MakeClassDeclaration(
1019 ParseResultIterator* child_results) {
1020 AnnotationSet annotations(
1021 child_results,
1022 {ANNOTATION_ABSTRACT, ANNOTATION_HAS_SAME_INSTANCE_TYPE_AS_PARENT,
1023 ANNOTATION_DO_NOT_GENERATE_CPP_CLASS, ANNOTATION_CUSTOM_CPP_CLASS,
1024 ANNOTATION_CUSTOM_MAP, ANNOTATION_GENERATE_BODY_DESCRIPTOR,
1025 ANNOTATION_EXPORT, ANNOTATION_DO_NOT_GENERATE_CAST,
1026 ANNOTATION_GENERATE_UNIQUE_MAP, ANNOTATION_GENERATE_FACTORY_FUNCTION,
1027 ANNOTATION_HIGHEST_INSTANCE_TYPE_WITHIN_PARENT,
1028 ANNOTATION_LOWEST_INSTANCE_TYPE_WITHIN_PARENT,
1029 ANNOTATION_CPP_OBJECT_DEFINITION,
1030 ANNOTATION_CPP_OBJECT_LAYOUT_DEFINITION},
1031 {ANNOTATION_RESERVE_BITS_IN_INSTANCE_TYPE,
1032 ANNOTATION_INSTANCE_TYPE_VALUE});
1033 ClassFlags flags = ClassFlag::kNone;
1034 if (annotations.Contains(ANNOTATION_ABSTRACT)) {
1035 flags |= ClassFlag::kAbstract;
1036 }
1037 if (annotations.Contains(ANNOTATION_HAS_SAME_INSTANCE_TYPE_AS_PARENT)) {
1038 flags |= ClassFlag::kHasSameInstanceTypeAsParent;
1039 }
1040 bool do_not_generate_cpp_class =
1041 annotations.Contains(ANNOTATION_DO_NOT_GENERATE_CPP_CLASS);
1042 if (annotations.Contains(ANNOTATION_CUSTOM_CPP_CLASS)) {
1043 Error(
1044 "@customCppClass is deprecated. Use 'extern' instead. "
1045 "@generateBodyDescriptor, @generateUniqueMap, and "
1046 "@generateFactoryFunction accomplish most of what '@export "
1047 "@customCppClass' used to.");
1048 }
1049 if (annotations.Contains(ANNOTATION_CUSTOM_MAP)) {
1050 Error(
1051 "@customMap is deprecated. Generating a unique map is opt-in now using "
1052 "@generateUniqueMap.");
1053 }
1054 if (annotations.Contains(ANNOTATION_DO_NOT_GENERATE_CAST)) {
1055 flags |= ClassFlag::kDoNotGenerateCast;
1056 }
1057 if (annotations.Contains(ANNOTATION_GENERATE_BODY_DESCRIPTOR)) {
1058 flags |= ClassFlag::kGenerateBodyDescriptor;
1059 }
1060 if (annotations.Contains(ANNOTATION_GENERATE_UNIQUE_MAP)) {
1061 flags |= ClassFlag::kGenerateUniqueMap;
1062 }
1063 if (annotations.Contains(ANNOTATION_GENERATE_FACTORY_FUNCTION)) {
1064 flags |= ClassFlag::kGenerateFactoryFunction;
1065 }
1066 if (annotations.Contains(ANNOTATION_EXPORT)) {
1067 flags |= ClassFlag::kExport;
1068 }
1069 if (annotations.Contains(ANNOTATION_HIGHEST_INSTANCE_TYPE_WITHIN_PARENT)) {
1070 flags |= ClassFlag::kHighestInstanceTypeWithinParent;
1071 }
1072 if (annotations.Contains(ANNOTATION_LOWEST_INSTANCE_TYPE_WITHIN_PARENT)) {
1073 flags |= ClassFlag::kLowestInstanceTypeWithinParent;
1074 }
1075 if (annotations.Contains(ANNOTATION_CPP_OBJECT_DEFINITION)) {
1076 flags |= ClassFlag::kCppObjectDefinition;
1077 }
1078 if (annotations.Contains(ANNOTATION_CPP_OBJECT_LAYOUT_DEFINITION)) {
1079 flags |= ClassFlag::kCppObjectLayoutDefinition;
1080 }
1081
1082 auto is_extern = child_results->NextAs<bool>();
1083 if (is_extern) flags |= ClassFlag::kExtern;
1084 auto transient = child_results->NextAs<bool>();
1085 if (transient) flags |= ClassFlag::kTransient;
1086 std::string kind = child_results->NextAs<Identifier*>()->value;
1087 if (kind == "shape") {
1088 flags |= ClassFlag::kIsShape;
1089 flags |= ClassFlag::kTransient;
1090 flags |= ClassFlag::kHasSameInstanceTypeAsParent;
1091 flags |= ClassFlag::kDoNotGenerateCast;
1092 } else {
1093 DCHECK_EQ(kind, "class");
1094 }
1095 auto name = child_results->NextAs<Identifier*>();
1096 if (!IsValidTypeName(name->value)) {
1097 NamingConventionError("Type", name, "UpperCamelCase");
1098 }
1099 auto extends = child_results->NextAs<TypeExpression*>();
1100 if (!BasicTypeExpression::DynamicCast(extends)) {
1101 ReportError("Expected type name in extends clause.");
1102 }
1103 auto generates = child_results->NextAs<std::optional<std::string>>();
1104 auto body = child_results->NextAs<std::optional<ClassBody*>>();
1105 std::vector<Declaration*> methods;
1106 std::vector<ClassFieldExpression> fields_raw;
1107 if (body.has_value()) {
1108 methods = (*body)->methods;
1109 fields_raw = (*body)->fields;
1110 } else {
1111 flags |= ClassFlag::kUndefinedLayout;
1112 }
1113
1114 if (is_extern && body.has_value()) {
1115 if (!do_not_generate_cpp_class) {
1116 flags |= ClassFlag::kGenerateCppClassDefinitions;
1117 }
1118 } else if (do_not_generate_cpp_class) {
1119 Lint("Annotation @doNotGenerateCppClass has no effect");
1120 }
1121
1122 // Filter to only include fields that should be present based on decoration.
1123 std::vector<ClassFieldExpression> fields;
1124 std::copy_if(
1125 fields_raw.begin(), fields_raw.end(), std::back_inserter(fields),
1126 [](const ClassFieldExpression& exp) {
1127 for (const ConditionalAnnotation& condition : exp.conditions) {
1128 if (condition.type == ConditionalAnnotationType::kPositive
1129 ? !BuildFlags::GetFlag(condition.condition, ANNOTATION_IF)
1130 : BuildFlags::GetFlag(condition.condition,
1131 ANNOTATION_IFNOT)) {
1132 return false;
1133 }
1134 }
1135 return true;
1136 });
1137
1138 std::vector<Declaration*> result;
1139
1140 result.push_back(MakeNode<ClassDeclaration>(
1141 name, flags, extends, generates, std::move(methods), std::move(fields),
1142 MakeInstanceTypeConstraints(annotations)));
1143
1144 Identifier* constexpr_name =
1145 MakeNode<Identifier>(CONSTEXPR_TYPE_PREFIX + name->value);
1146 constexpr_name->pos = name->pos;
1147 TypeExpression* constexpr_extends = AddConstexpr(extends);
1148 AbstractTypeFlags abstract_type_flags(AbstractTypeFlag::kConstexpr);
1149 if (transient) abstract_type_flags |= AbstractTypeFlag::kTransient;
1150 TypeDeclaration* constexpr_decl = MakeNode<AbstractTypeDeclaration>(
1151 constexpr_name, abstract_type_flags, constexpr_extends,
1152 generates ? UnwrapTNodeTypeName(*generates) : name->value);
1153 constexpr_decl->pos = name->pos;
1154 result.push_back(constexpr_decl);
1155
1156 if ((flags & ClassFlag::kDoNotGenerateCast) == 0 &&
1157 (flags & ClassFlag::kIsShape) == 0) {
1158 ParameterList parameters;
1159 parameters.names.push_back(MakeNode<Identifier>("obj"));
1160 parameters.types.push_back(MakeNode<BasicTypeExpression>(
1161 std::vector<std::string>{}, MakeNode<Identifier>("HeapObject"),
1162 std::vector<TypeExpression*>{}));
1163 LabelAndTypesVector labels;
1164 labels.push_back(LabelAndTypes{MakeNode<Identifier>("CastError"),
1165 std::vector<TypeExpression*>{}});
1166
1167 TypeExpression* class_type = MakeNode<BasicTypeExpression>(
1168 std::vector<std::string>{}, name, std::vector<TypeExpression*>{});
1169
1170 std::vector<std::string> namespace_qualification{
1171 TORQUE_INTERNAL_NAMESPACE_STRING};
1172
1173 IdentifierExpression* internal_downcast_target =
1174 MakeNode<IdentifierExpression>(
1175 std::move(namespace_qualification),
1176 MakeNode<Identifier>("DownCastForTorqueClass"),
1177 std::vector<TypeExpression*>{class_type});
1178 IdentifierExpression* internal_downcast_otherwise =
1179 MakeNode<IdentifierExpression>(std::vector<std::string>{},
1180 MakeNode<Identifier>("CastError"));
1181
1182 Expression* argument = MakeNode<IdentifierExpression>(
1183 std::vector<std::string>{}, MakeNode<Identifier>("obj"));
1184
1185 auto value = MakeCall(internal_downcast_target, std::nullopt,
1186 std::vector<Expression*>{argument},
1187 std::vector<Statement*>{MakeNode<ExpressionStatement>(
1188 internal_downcast_otherwise)});
1189
1190 auto cast_body = MakeNode<ReturnStatement>(value);
1191
1192 std::vector<TypeExpression*> generic_parameters;
1193 generic_parameters.push_back(MakeNode<BasicTypeExpression>(
1194 std::vector<std::string>{}, name, std::vector<TypeExpression*>{}));
1195
1196 Declaration* specialization = MakeNode<SpecializationDeclaration>(
1197 false, MakeNode<Identifier>("Cast"), std::move(generic_parameters),
1198 std::move(parameters), class_type, std::move(labels), cast_body);
1199 result.push_back(specialization);
1200 }
1201
1202 return ParseResult{std::move(result)};
1203}

1418std::optional<ParseResult> MakeEnumDeclaration(
1419 ParseResultIterator* child_results) {
1420 const bool is_extern = child_results->NextAs<bool>();
1421 auto name_identifier = child_results->NextAs<Identifier*>();
1422 auto name = name_identifier->value;
1423 auto base_type_expression =
1424 child_results->NextAs<std::optional<TypeExpression*>>();
1425 auto constexpr_generates_opt =
1426 child_results->NextAs<std::optional<std::string>>();
1427 auto entries = child_results->NextAs<std::vector<EnumEntry>>();
1428 const bool is_open = child_results->NextAs<bool>();
1429 CurrentSourcePosition::Scope current_source_position(
1430 child_results->matched_input().pos);
1431
1432 if (!is_extern) {
1433 ReportError("non-extern enums are not supported yet");
1434 }
1435
1436 if (!IsValidTypeName(name)) {
1437 NamingConventionError("Type", name, "UpperCamelCase");
1438 }
1439
1440 if (constexpr_generates_opt && *constexpr_generates_opt == name) {
1441 Lint("Unnecessary 'constexpr' clause for enum ", name);
1442 }
1443 auto constexpr_generates =
1444 constexpr_generates_opt ? *constexpr_generates_opt : name;
1445 const bool generate_nonconstexpr = base_type_expression.has_value();
1446
1447 std::vector<Declaration*> result;
1448 // Build non-constexpr types.
1449 if (generate_nonconstexpr) {
1450 DCHECK(base_type_expression.has_value());
1451
1452 if (is_open) {
1453 // For open enumerations, we define an abstract type and inherit all
1454 // entries' types from that:
1455 // type Enum extends Base;
1456 // namespace Enum {
1457 // type kEntry0 extends Enum;
1458 // ...
1459 // type kEntryN extends Enum;
1460 // }
1461 auto type_decl = MakeNode<AbstractTypeDeclaration>(
1462 name_identifier, AbstractTypeFlag::kNone, base_type_expression,
1463 std::nullopt);
1464
1465 TypeExpression* name_type_expression =
1466 MakeNode<BasicTypeExpression>(name_identifier);
1467 name_type_expression->pos = name_identifier->pos;
1468
1469 std::vector<Declaration*> entry_decls;
1470 entry_decls.reserve(entries.size());
1471 for (const auto& entry : entries) {
1472 entry_decls.push_back(MakeNode<AbstractTypeDeclaration>(
1473 entry.name, AbstractTypeFlag::kNone,
1474 entry.type.value_or(name_type_expression), std::nullopt));
1475 }
1476
1477 result.push_back(type_decl);
1478 result.push_back(
1479 MakeNode<NamespaceDeclaration>(name, std::move(entry_decls)));
1480 } else {
1481 // For closed enumerations, we define abstract types for all entries and
1482 // define the enumeration as a union of those:
1483 // namespace Enum {
1484 // type kEntry0 extends Base;
1485 // ...
1486 // type kEntryN extends Base;
1487 // }
1488 // type Enum = Enum::kEntry0 | ... | Enum::kEntryN;
1489 TypeExpression* union_type = nullptr;
1490 std::vector<Declaration*> entry_decls;
1491 for (const auto& entry : entries) {
1492 entry_decls.push_back(MakeNode<AbstractTypeDeclaration>(
1493 entry.name, AbstractTypeFlag::kNone,
1494 entry.type.value_or(*base_type_expression), std::nullopt));
1495
1496 auto entry_type = MakeNode<BasicTypeExpression>(
1497 std::vector<std::string>{name}, entry.name,
1498 std::vector<TypeExpression*>{});
1499 if (union_type) {
1500 union_type = MakeNode<UnionTypeExpression>(union_type, entry_type);
1501 } else {
1502 union_type = entry_type;
1503 }
1504 }
1505
1506 result.push_back(
1507 MakeNode<NamespaceDeclaration>(name, std::move(entry_decls)));
1508 result.push_back(
1509 MakeNode<TypeAliasDeclaration>(name_identifier, union_type));
1510 }
1511 }
1512
1513 // Build constexpr types.
1514 {
1515 // The constexpr entries inherit from an abstract enumeration type:
1516 // type constexpr Enum extends constexpr Base;
1517 // namespace Enum {
1518 // type constexpr kEntry0 extends constexpr Enum;
1519 // ...
1520 // type constexpr kEntry1 extends constexpr Enum;
1521 // }
1522 Identifier* constexpr_type_identifier =
1523 MakeNode<Identifier>(std::string(CONSTEXPR_TYPE_PREFIX) + name);
1524 TypeExpression* constexpr_type_expression = MakeNode<BasicTypeExpression>(
1525 MakeNode<Identifier>(std::string(CONSTEXPR_TYPE_PREFIX) + name));
1526 std::optional<TypeExpression*> base_constexpr_type_expression =
1527 std::nullopt;
1528 if (base_type_expression) {
1529 base_constexpr_type_expression = AddConstexpr(*base_type_expression);
1530 }
1531 result.push_back(MakeNode<AbstractTypeDeclaration>(
1532 constexpr_type_identifier, AbstractTypeFlag::kConstexpr,
1533 base_constexpr_type_expression, constexpr_generates));
1534
1535 TypeExpression* type_expr = nullptr;
1536 Identifier* fromconstexpr_identifier = nullptr;
1537 Identifier* fromconstexpr_parameter_identifier = nullptr;
1538 Statement* fromconstexpr_body = nullptr;
1539 if (generate_nonconstexpr) {
1540 DCHECK(base_type_expression.has_value());
1541 type_expr = MakeNode<BasicTypeExpression>(std::vector<std::string>{},
1542 MakeNode<Identifier>(name),
1543 std::vector<TypeExpression*>{});
1544
1545 // return %RawDownCast<Enum>(%FromConstexpr<Base>(o)))
1546 fromconstexpr_identifier = MakeNode<Identifier>("FromConstexpr");
1547 fromconstexpr_parameter_identifier = MakeNode<Identifier>("o");
1548 fromconstexpr_body =
1549 MakeNode<ReturnStatement>(MakeNode<IntrinsicCallExpression>(
1550 MakeNode<Identifier>("%RawDownCast"),
1551 std::vector<TypeExpression*>{type_expr},
1552 std::vector<Expression*>{MakeNode<IntrinsicCallExpression>(
1553 MakeNode<Identifier>("%FromConstexpr"),
1554 std::vector<TypeExpression*>{*base_type_expression},
1555 std::vector<Expression*>{MakeNode<IdentifierExpression>(
1556 std::vector<std::string>{},
1557 fromconstexpr_parameter_identifier)})}));
1558 }
1559
1560 EnumDescription enum_description{CurrentSourcePosition::Get(), name,
1561 constexpr_generates, is_open};
1562 std::vector<Declaration*> entry_decls;
1563 for (const auto& entry : entries) {
1564 const std::string entry_name = entry.name->value;
1565 const std::string entry_constexpr_type =
1566 CONSTEXPR_TYPE_PREFIX + entry_name;
1567 std::string alias_entry;
1568 if (entry.alias_entry) {
1569 alias_entry = constexpr_generates + "::" + *entry.alias_entry;
1570 }
1571 enum_description.entries.emplace_back(
1572 constexpr_generates + "::" + entry_name, alias_entry);
1573
1574 entry_decls.push_back(MakeNode<AbstractTypeDeclaration>(
1575 MakeNode<Identifier>(entry_constexpr_type),
1576 AbstractTypeFlag::kConstexpr, constexpr_type_expression,
1577 constexpr_generates));
1578
1579 bool generate_typed_constant = entry.type.has_value();
1580 if (generate_typed_constant) {
1581 // namespace Enum {
1582 // const constexpr_constant_kEntry0: constexpr kEntry0 constexpr
1583 // 'Enum::kEntry0'; const kEntry0 = %RawDownCast<T,
1584 // Base>(FromConstexpr<Enum>(constexpr_constant_kEntry0));
1585 // }
1586 if (!generate_nonconstexpr) {
1587 Error(
1588 "Enum constants with custom types require an enum with an "
1589 "extends clause.")
1590 .Position((*entry.type)->pos);
1591 }
1592 Identifier* constexpr_constant_name =
1593 MakeNode<Identifier>("constexpr constant " + entry_name);
1594 entry_decls.push_back(MakeNode<ExternConstDeclaration>(
1595 constexpr_constant_name,
1596 MakeNode<BasicTypeExpression>(
1597 std::vector<std::string>{},
1598 MakeNode<Identifier>(entry_constexpr_type),
1599 std::vector<TypeExpression*>{}),
1600 constexpr_generates + "::" + entry_name));
1601 entry_decls.push_back(MakeNode<ConstDeclaration>(
1602 entry.name, *entry.type,
1603 MakeNode<IntrinsicCallExpression>(
1604 MakeNode<Identifier>("%RawDownCast"),
1605 std::vector<TypeExpression*>{*entry.type,
1606 *base_type_expression},
1607 std::vector<Expression*>{MakeCall(
1608 MakeNode<Identifier>("FromConstexpr"), {type_expr},
1609 {MakeNode<IdentifierExpression>(std::vector<std::string>{},
1610 constexpr_constant_name)},
1611 {})})));
1612 } else {
1613 // namespace Enum {
1614 // const kEntry0: constexpr kEntry0 constexpr 'Enum::kEntry0';
1615 // }
1616 entry_decls.push_back(MakeNode<ExternConstDeclaration>(
1617 entry.name,
1618 MakeNode<BasicTypeExpression>(
1619 std::vector<std::string>{},
1620 MakeNode<Identifier>(entry_constexpr_type),
1621 std::vector<TypeExpression*>{}),
1622 constexpr_generates + "::" + entry_name));
1623 }
1624
1625 // FromConstexpr<Enum, Enum::constexpr kEntry0>(
1626 // : Enum::constexpr kEntry0): Enum
1627 if (generate_nonconstexpr) {
1628 TypeExpression* entry_constexpr_type_expr =
1629 MakeNode<BasicTypeExpression>(
1630 std::vector<std::string>{name},
1631 MakeNode<Identifier>(entry_constexpr_type),
1632 std::vector<TypeExpression*>{});
1633
1634 ParameterList parameters;
1635 parameters.names.push_back(fromconstexpr_parameter_identifier);
1636 parameters.types.push_back(entry_constexpr_type_expr);
1637 result.push_back(MakeNode<SpecializationDeclaration>(
1638 false, fromconstexpr_identifier,
1639 std::vector<TypeExpression*>{type_expr, entry_constexpr_type_expr},
1640 std::move(parameters), type_expr, LabelAndTypesVector{},
1641 fromconstexpr_body));
1642 }
1643 }
1644
1645 result.push_back(
1646 MakeNode<NamespaceDeclaration>(name, std::move(entry_decls)));
1647 CurrentAst::Get().AddEnumDescription(std::move(enum_description));
1648 }
1649
1650 return ParseResult{std::move(result)};
1651}

1653std::optional<ParseResult> MakeTypeswitchStatement(
1654 ParseResultIterator* child_results) {
1655 auto expression = child_results->NextAs<Expression*>();
1656 auto cases = child_results->NextAs<std::vector<TypeswitchCase>>();
1657 CurrentSourcePosition::Scope matched_input_current_source_position(
1658 child_results->matched_input().pos);
1659
1660 // typeswitch (expression) case (x1 : T1) {
1661 // ...b1
1662 // } case (x2 : T2) {
1663 // ...b2
1664 // } case (x3 : T3) {
1665 // ...b3
1666 // }
1667 //
1668 // desugars to
1669 //
1670 // {
1671 // const _value = expression;
1672 // try {
1673 // const x1 : T1 = cast<T1>(_value) otherwise _NextCase;
1674 // ...b1
1675 // } label _NextCase {
1676 // try {
1677 // const x2 : T2 = cast<T2>(%assume_impossible<T1>(_value));
1678 // ...b2
1679 // } label _NextCase {
1680 // const x3 : T3 = %assume_impossible<T1|T2>(_value);
1681 // ...b3
1682 // }
1683 // }
1684 // }
1685
1686 BlockStatement* current_block = MakeNode<BlockStatement>();
1687 Statement* result = current_block;
1688 {
1689 CurrentSourcePosition::Scope current_source_position(expression->pos);
1690 current_block->statements.push_back(MakeNode<VarDeclarationStatement>(
1691 true, MakeNode<Identifier>("__value"), std::nullopt, expression));
1692 }
1693
1694 TypeExpression* accumulated_types;
1695 for (size_t i = 0; i < cases.size(); ++i) {
1696 CurrentSourcePosition::Scope current_source_position(cases[i].pos);
1697 Expression* value =
1698 MakeNode<IdentifierExpression>(MakeNode<Identifier>("__value"));
1699 if (i >= 1) {
1700 value =
1701 MakeNode<AssumeTypeImpossibleExpression>(accumulated_types, value);
1702 }
1703 BlockStatement* case_block;
1704 if (i < cases.size() - 1) {
1705 value = MakeCall(MakeNode<Identifier>("Cast"),
1706 std::vector<TypeExpression*>{cases[i].type},
1707 std::vector<Expression*>{value},
1708 std::vector<Statement*>{MakeNode<ExpressionStatement>(
1709 MakeNode<IdentifierExpression>(
1710 MakeNode<Identifier>(kNextCaseLabelName)))});
1711 case_block = MakeNode<BlockStatement>();
1712 } else {
1713 case_block = current_block;
1714 }
1715 Identifier* name =
1716 cases[i].name ? *cases[i].name : MakeNode<Identifier>("__case_value");
1717 if (cases[i].name) name = *cases[i].name;
1718 case_block->statements.push_back(
1719 MakeNode<VarDeclarationStatement>(true, name, cases[i].type, value));
1720 case_block->statements.push_back(cases[i].block);
1721 if (i < cases.size() - 1) {
1722 BlockStatement* next_block = MakeNode<BlockStatement>();
1723 current_block->statements.push_back(
1724 MakeNode<ExpressionStatement>(MakeNode<TryLabelExpression>(
1725 MakeNode<StatementExpression>(case_block),
1726 MakeNode<TryHandler>(TryHandler::HandlerKind::kLabel,
1727 MakeNode<Identifier>(kNextCaseLabelName),
1728 ParameterList::Empty(), next_block))));
1729 current_block = next_block;
1730 }
1731 accumulated_types =
1732 i > 0 ? MakeNode<UnionTypeExpression>(accumulated_types, cases[i].type)
1733 : cases[i].type;
1734 }
1735 return ParseResult{result};
1736}
