24// The result of an allocation attempt. Either represents a successful
25// allocation that can be turned into an object or a failed attempt.
26class AllocationResult final {
27 public:
28 static AllocationResult Failure() { return AllocationResult(); }
29
30 static AllocationResult FromObject(Tagged<HeapObject> heap_object) {
31 return AllocationResult(heap_object);
32 }
33
34 // Empty constructor creates a failed result. The callsite determines which
35 // GC to invoke based on the requested allocation.
36 AllocationResult() = default;
37
38 bool IsFailure() const { return object_.is_null(); }
39
40 template <typename T>
41 bool To(Tagged<T>* obj) const {
42 if (IsFailure()) return false;
43 *obj = Cast<T>(object_);
44 return true;
45 }
46
47 Tagged<HeapObject> ToObjectChecked() const {
48 CHECK(!IsFailure());
49 return Cast<HeapObject>(object_);
50 }
51
52 Tagged<HeapObject> ToObject() const {
53 DCHECK(!IsFailure());
54 return Cast<HeapObject>(object_);
55 }
56
57 Address ToAddress() const {
58 DCHECK(!IsFailure());
59 return Cast<HeapObject>(object_).address();
60 }
61
62 private:
63 explicit AllocationResult(Tagged<HeapObject> heap_object)
64 : object_(heap_object) {}
65
66 Tagged<HeapObject> object_;
67};
