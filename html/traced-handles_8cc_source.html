49TracedNodeBlock* TracedNodeBlock::Create(TracedHandles& traced_handles) {
50 static_assert(alignof(TracedNodeBlock) >= alignof(TracedNode));
51 static_assert(sizeof(TracedNodeBlock) % alignof(TracedNode) == 0,
52 "TracedNodeBlock size is used to auto-align node FAM storage.");
53 const size_t min_wanted_size =
54 sizeof(TracedNodeBlock) +
55 sizeof(TracedNode) * TracedNodeBlock::kMinCapacity;
56 const auto raw_result = v8::base::AllocateAtLeast<char>(min_wanted_size);
57 const size_t capacity = std::min(
58 (raw_result.count - sizeof(TracedNodeBlock)) / sizeof(TracedNode),
59 kMaxCapacity);
60 CHECK_LT(capacity, std::numeric_limits<TracedNode::IndexType>::max());
61 const auto result = std::make_pair(raw_result.ptr, capacity);
62 return new (result.first) TracedNodeBlock(
63 traced_handles, static_cast<TracedNode::IndexType>(result.second));
64}

104void TracedHandles::RefillUsableNodeBlocks() {
105 TracedNodeBlock* block;
106 if (empty_blocks_.empty()) {
107 block = TracedNodeBlock::Create(*this);
108 block_size_bytes_ += block->size_bytes();
109 } else {
110 block = empty_blocks_.back();
111 empty_blocks_.pop_back();
112 }
113 usable_blocks_.PushFront(block);
114 blocks_.PushFront(block);
115 num_blocks_++;
116 DCHECK(!block->InYoungList());
117 DCHECK(block->IsEmpty());
118 DCHECK_EQ(usable_blocks_.Front(), block);
119 DCHECK(!usable_blocks_.empty());
120}

122void TracedHandles::FreeNode(TracedNode* node, Address zap_value) {
123 auto& block = TracedNodeBlock::From(*node);
124 if (V8_UNLIKELY(block.IsFull())) {
125 DCHECK(!usable_blocks_.ContainsSlow(&block));
126 usable_blocks_.PushFront(&block);
127 }
128 block.FreeNode(node, zap_value);
129 if (block.IsEmpty()) {
130 usable_blocks_.Remove(&block);
131 blocks_.Remove(&block);
132 if (block.InYoungList()) {
133 young_blocks_.Remove(&block);
134 block.SetInYoungList(false);
135 }
136 num_blocks_--;
137 empty_blocks_.push_back(&block);
138 }
139 used_nodes_--;
140}

144TracedHandles::~TracedHandles() {
145 size_t block_size_bytes = 0;
146 while (!blocks_.empty()) {
147 auto* block = blocks_.Front();
148 blocks_.PopFront();
149 block_size_bytes += block->size_bytes();
150 TracedNodeBlock::Delete(block);
151 }
152 for (auto* block : empty_blocks_) {
153 block_size_bytes += block->size_bytes();
154 TracedNodeBlock::Delete(block);
155 }
156 USE(block_size_bytes);
157 DCHECK_EQ(block_size_bytes, block_size_bytes_);

160void TracedHandles::Destroy(TracedNodeBlock& node_block, TracedNode& node) {
161 DCHECK_IMPLIES(is_marking_, !is_sweeping_on_mutator_thread_);
162 DCHECK_IMPLIES(is_sweeping_on_mutator_thread_, !is_marking_);
163
164 // If sweeping on the mutator thread is running then the handle destruction
165 // may be a result of a Reset() call from a destructor. The node will be
166 // reclaimed on the next cycle.
167 //
168 // This allows v8::TracedReference::Reset() calls from destructors on
169 // objects that may be used from stack and heap.
170 if (is_sweeping_on_mutator_thread_) {
171 return;
172 }
173
174 if (is_marking_) {
175 // Incremental/concurrent marking is running. This also covers the scavenge
176 // case which prohibits eagerly reclaiming nodes when marking is on during a
177 // scavenge.
178 //
179 // On-heap traced nodes are released in the atomic pause in
180 // `ResetDeadNodes()` when they are discovered as not marked. Eagerly clear
181 // out the object here to avoid needlessly marking it from this point on.
182 // The node will be reclaimed on the next cycle.
183 node.set_raw_object<AccessMode::ATOMIC>(kNullAddress);
184 return;
185 }
186
187 // In case marking and sweeping are off, the handle may be freed immediately.
188 // Note that this includes also the case when invoking the first pass
189 // callbacks during the atomic pause which requires releasing a node fully.
190 FreeNode(&node, kTracedHandleEagerResetZapValue);
191}

207void TracedHandles::Move(TracedNode& from_node, Address** from, Address** to) {
208 DCHECK(from_node.is_in_use());
209
210 // Deal with old "to".
211 auto* to_node = TracedNode::FromLocation(*to);
212 DCHECK_IMPLIES(*to, to_node->is_in_use());
213 DCHECK_IMPLIES(*to, kGlobalHandleZapValue != to_node->raw_object());
214 DCHECK_NE(kGlobalHandleZapValue, from_node.raw_object());
215 if (*to) {
216 auto& to_node_block = TracedNodeBlock::From(*to_node);
217 Destroy(to_node_block, *to_node);
218 }
219
220 // Set "to" to "from".
221 SetSlotThreadSafe(to, *from);
222 to_node = &from_node;
223
224 // Deal with new "to"
225 DCHECK_NOT_NULL(*to);
226 DCHECK_EQ(*from, *to);
227 if (is_marking_) {
228 // Write barrier needs to cover node as well as object.
229 to_node->set_markbit();
230 WriteBarrier::MarkingFromTracedHandle(to_node->object());
231 } else if (auto* cpp_heap = GetCppHeapIfUnifiedYoungGC(isolate_)) {
232 const bool object_is_young_and_not_yet_recorded =
233 !from_node.has_old_host() &&
234 HeapLayout::InYoungGeneration(from_node.object());
235 if (object_is_young_and_not_yet_recorded &&
236 IsCppGCHostOld(*cpp_heap, reinterpret_cast<Address>(to))) {
237 DCHECK(from_node.is_in_young_list());
238 from_node.set_has_old_host(true);
239 }
240 }
241 SetSlotThreadSafe(from, nullptr);
242}

268void TracedHandles::UpdateListOfYoungNodes() {
269 const bool needs_to_mark_as_old =
270 static_cast<bool>(GetCppHeapIfUnifiedYoungGC(isolate_));
271
272 for (auto it = young_blocks_.begin(); it != young_blocks_.end();) {
273 bool contains_young_node = false;
274 TracedNodeBlock* const block = *it;
275 DCHECK(block->InYoungList());
276
277 for (auto* node : *block) {
278 if (!node->is_in_young_list()) continue;
279 DCHECK(node->is_in_use());
280 if (HeapLayout::InYoungGeneration(node->object())) {
281 contains_young_node = true;
282 // The node was discovered through a cppgc object, which will be
283 // immediately promoted. Remember the object.
284 if (needs_to_mark_as_old) node->set_has_old_host(true);
285 } else {
286 node->set_is_in_young_list(false);
287 node->set_has_old_host(false);
288 }
289 }
290 if (contains_young_node) {
291 ++it;
292 } else {
293 it = young_blocks_.RemoveAt(it);
294 block->SetInYoungList(false);
295 }
296 }
297}

299void TracedHandles::DeleteEmptyBlocks() {
300 // Keep one node block around for fast allocation/deallocation patterns.
301 if (empty_blocks_.size() <= 1) return;
302
303 for (size_t i = 1; i < empty_blocks_.size(); i++) {
304 auto* block = empty_blocks_[i];
305 DCHECK(block->IsEmpty());
306 DCHECK_GE(block_size_bytes_, block->size_bytes());
307 block_size_bytes_ -= block->size_bytes();
308 TracedNodeBlock::Delete(block);
309 }
310 empty_blocks_.resize(1);
311 empty_blocks_.shrink_to_fit();
312}

314void TracedHandles::ResetDeadNodes(
315 WeakSlotCallbackWithHeap should_reset_handle) {
316 // Manual iteration as the block may be deleted in `FreeNode()`.
317 for (auto it = blocks_.begin(); it != blocks_.end();) {
318 auto* block = *(it++);
319 for (auto* node : *block) {
320 if (!node->is_in_use()) continue;
321
322 // Detect unreachable nodes first.
323 if (!node->markbit()) {
324 FreeNode(node, kTracedHandleFullGCResetZapValue);
325 continue;
326 }
327
328 // Node was reachable. Clear the markbit for the next GC.
329 node->clear_markbit();
330 // TODO(v8:13141): Turn into a DCHECK after some time.
331 CHECK(!should_reset_handle(isolate_->heap(), node->location()));
332 }
333
334 if (block->InYoungList()) {
335 young_blocks_.Remove(block);
336 block->SetInYoungList(false);
337 }
338 }
339
340 CHECK(young_blocks_.empty());
341}

343void TracedHandles::ResetYoungDeadNodes(
344 WeakSlotCallbackWithHeap should_reset_handle) {
345 for (auto* block : young_blocks_) {
346 for (auto* node : *block) {
347 if (!node->is_in_young_list()) continue;
348 DCHECK(node->is_in_use());
349 DCHECK_IMPLIES(node->has_old_host(), node->markbit());
350
351 if (!node->markbit()) {
352 FreeNode(node, kTracedHandleMinorGCResetZapValue);
353 continue;
354 }
355
356 // Node was reachable. Clear the markbit for the next GC.
357 node->clear_markbit();
358 // TODO(v8:13141): Turn into a DCHECK after some time.
359 CHECK(!should_reset_handle(isolate_->heap(), node->location()));
360 }
361 }

364void TracedHandles::ComputeWeaknessForYoungObjects() {
365 if (!v8_flags.reclaim_unmodified_wrappers) return;
366
367 // Treat all objects as roots during incremental marking to avoid corrupting
368 // marking worklists.
369 DCHECK_IMPLIES(v8_flags.minor_ms, !is_marking_);
370 if (is_marking_) return;
371
372 auto* const handler = isolate_->heap()->GetEmbedderRootsHandler();
373 if (!handler) return;
374
375 for (auto* block : young_blocks_) {
376 DCHECK(block->InYoungList());
377 for (auto* node : *block) {
378 if (!node->is_in_young_list()) continue;
379 DCHECK(node->is_in_use());
380 DCHECK(!node->is_weak());
381 if (node->is_droppable() &&
382 JSObject::IsUnmodifiedApiObject(node->location())) {
383 node->set_weak(true);
384 }
385 }
386 }

389void TracedHandles::ProcessYoungObjects(
390 RootVisitor* visitor, WeakSlotCallbackWithHeap should_reset_handle) {
391 if (!v8_flags.reclaim_unmodified_wrappers) return;
392
393 auto* const handler = isolate_->heap()->GetEmbedderRootsHandler();
394 if (!handler) return;
395
396 // ResetRoot should not trigger allocations in CppGC.
397 if (auto* cpp_heap = CppHeap::From(isolate_->heap()->cpp_heap())) {
398 cpp_heap->EnterDisallowGCScope();
399 cpp_heap->EnterNoGCScope();
400 }
401
402 for (auto it = young_blocks_.begin(); it != young_blocks_.end();) {
403 TracedNodeBlock* block = *it;
404 DCHECK(block->InYoungList());
405
406 // Avoid iterator invalidation by incrementing iterator here before
407 // ResetRoot().
408 it++;
409
410 for (auto* node : *block) {
411 if (!node->is_in_young_list()) continue;
412 DCHECK(node->is_in_use());
413
414 bool should_reset =
415 should_reset_handle(isolate_->heap(), node->location());
416 if (should_reset) {
417 CHECK(node->is_weak());
418 CHECK(!is_marking_);
419 FullObjectSlot slot = node->location();
420 handler->ResetRoot(
421 *reinterpret_cast<v8::TracedReference<v8::Value>*>(&slot));
422 // Mark as cleared due to weak semantics.
423 node->set_raw_object(kTracedHandleMinorGCWeakResetZapValue);
424 CHECK(!node->is_in_use());
425 } else {
426 if (node->is_weak()) {
427 node->set_weak(false);
428 if (visitor) {
429 visitor->VisitRootPointer(Root::kGlobalHandles, nullptr,
430 node->location());
431 }
432 }
433 }
434 }
435 }
436
437 if (auto* cpp_heap = CppHeap::From(isolate_->heap()->cpp_heap())) {
438 cpp_heap->LeaveNoGCScope();
439 cpp_heap->LeaveDisallowGCScope();
440 }
441}

557Tagged<Object> MarkObject(Tagged<Object> obj, TracedNode& node,
558 TracedHandles::MarkMode mark_mode) {
559 if (mark_mode == TracedHandles::MarkMode::kOnlyYoung &&
560 !node.is_in_young_list())
561 return Smi::zero();
562 node.set_markbit();
563 // Being in the young list, the node may still point to an old object, in
564 // which case we want to keep the node marked, but not follow the reference.
565 if (mark_mode == TracedHandles::MarkMode::kOnlyYoung &&
566 !HeapLayout::InYoungGeneration(obj))
567 return Smi::zero();
568 return obj;

574 // The load synchronizes internal bitfields that are also read atomically

580 DCHECK(node->is_in_use());

585Tagged<Object> TracedHandles::MarkConservatively(
586 Address* inner_location, Address* traced_node_block_base,
587 MarkMode mark_mode) {
588 // Compute the `TracedNode` address based on its inner pointer.

591 const auto index = delta / sizeof(TracedNode);
592 TracedNode& node =

594 if (!node.is_in_use()) return Smi::zero();
595 return MarkObject(node.object(), node, mark_mode);
