106 template <typename Callback>
107 void AddLocalHeap(LocalHeap* local_heap, Callback callback) {
108 // Safepoint holds this lock in order to stop threads from starting or
109 // stopping.
110 base::RecursiveMutexGuard guard(&local_heaps_mutex_);
111
112 // Additional code protected from safepoint
113 callback();
114
115 // Add list to doubly-linked list
116 if (local_heaps_head_) local_heaps_head_->prev_ = local_heap;
117 local_heap->prev_ = nullptr;
118 local_heap->next_ = local_heaps_head_;
119 local_heaps_head_ = local_heap;
120 }
121
122 template <typename Callback>
123 void RemoveLocalHeap(LocalHeap* local_heap, Callback callback) {
124 base::RecursiveMutexGuard guard(&local_heaps_mutex_);
125
126 // Additional code protected from safepoint
127 callback();
128
129 // Remove list from doubly-linked list
130 if (local_heap->next_) local_heap->next_->prev_ = local_heap->prev_;
131 if (local_heap->prev_)
132 local_heap->prev_->next_ = local_heap->next_;
133 else
134 local_heaps_head_ = local_heap->next_;
135 }
