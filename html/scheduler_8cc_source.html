49Schedule* Scheduler::ComputeSchedule(Zone* zone, Graph* graph, Flags flags,
50 TickCounter* tick_counter,
51 const ProfileDataFromFile* profile_data) {
52 Zone* schedule_zone =
53 (flags & Scheduler::kTempSchedule) ? zone : graph->zone();
54
55 // Reserve 10% more space for nodes if node splitting is enabled to try to
56 // avoid resizing the vector since that would triple its zone memory usage.
57 float node_hint_multiplier = (flags & Scheduler::kSplitNodes) ? 1.1 : 1;
58 size_t node_count_hint = node_hint_multiplier * graph->NodeCount();
59
60 Schedule* schedule =
61 schedule_zone->New<Schedule>(schedule_zone, node_count_hint);
62 Scheduler scheduler(zone, graph, schedule, flags, node_count_hint,
63 tick_counter, profile_data);
64
65 scheduler.BuildCFG();
66 scheduler.ComputeSpecialRPONumbering();
67 scheduler.GenerateDominatorTree();
68
69 scheduler.PrepareUses();
70 scheduler.ScheduleEarly();
71 scheduler.ScheduleLate();
72
73 scheduler.SealFinalSchedule();
74
75 return schedule;
76}

88Scheduler::Placement Scheduler::InitializePlacement(Node* node) {
89 SchedulerData* data = GetData(node);
90 if (data->placement_ == kFixed) {
91 // Nothing to do for control nodes that have been already fixed in
92 // the schedule.
93 return data->placement_;
94 }
95 DCHECK_EQ(kUnknown, data->placement_);
96 switch (node->opcode()) {
97 case IrOpcode::kParameter:
98 case IrOpcode::kOsrValue:
99 // Parameters and OSR values are always fixed to the start block.
100 data->placement_ = kFixed;
101 break;
102 case IrOpcode::kPhi:
103 case IrOpcode::kEffectPhi: {
104 // Phis and effect phis are fixed if their control inputs are, whereas
105 // otherwise they are coupled to a floating control node.
106 Placement p = GetPlacement(NodeProperties::GetControlInput(node));
107 data->placement_ = (p == kFixed ? kFixed : kCoupled);
108 break;
109 }
110 default:
111 // Control nodes that were not control-reachable from end may float.
112 data->placement_ = kSchedulable;
113 break;
114 }
115 return data->placement_;
116}

124void Scheduler::UpdatePlacement(Node* node, Placement placement) {
125 SchedulerData* data = GetData(node);
126 if (data->placement_ == kUnknown) {
127 // We only update control nodes from {kUnknown} to {kFixed}. Ideally, we
128 // should check that {node} is a control node (including exceptional calls),
129 // but that is expensive.
130 DCHECK_EQ(Scheduler::kFixed, placement);
131 data->placement_ = placement;
132 return;
133 }
134
135 switch (node->opcode()) {
136 case IrOpcode::kParameter:
137 // Parameters are fixed once and for all.
138 UNREACHABLE();
139 case IrOpcode::kPhi:
140 case IrOpcode::kEffectPhi: {
141 // Phis and effect phis are coupled to their respective blocks.
142 DCHECK_EQ(Scheduler::kCoupled, data->placement_);
143 DCHECK_EQ(Scheduler::kFixed, placement);
144 Node* control = NodeProperties::GetControlInput(node);
145 BasicBlock* block = schedule_->block(control);
146 schedule_->AddNode(block, node);
147 break;
148 }
149#define DEFINE_CONTROL_CASE(V) case IrOpcode::k##V:
150 CONTROL_OP_LIST(DEFINE_CONTROL_CASE)
151#undef DEFINE_CONTROL_CASE
152 {
153 // Control nodes force coupled uses to be placed.
154 for (auto use : node->uses()) {
155 if (GetPlacement(use) == Scheduler::kCoupled) {
156 DCHECK_EQ(node, NodeProperties::GetControlInput(use));
157 UpdatePlacement(use, placement);
158 }
159 }
160 break;
161 }
162 default:
163 DCHECK_EQ(Scheduler::kSchedulable, data->placement_);
164 DCHECK_EQ(Scheduler::kScheduled, placement);
165 break;
166 }
167 // Reduce the use count of the node's inputs to potentially make them
168 // schedulable. If all the uses of a node have been scheduled, then the node
169 // itself can be scheduled.
170 std::optional<int> coupled_control_edge = GetCoupledControlEdge(node);
171 for (Edge const edge : node->input_edges()) {
172 DCHECK_EQ(node, edge.from());
173 if (edge.index() != coupled_control_edge) {
174 DecrementUnscheduledUseCount(edge.to(), node);
175 }
176 }
177 data->placement_ = placement;
178}

187void Scheduler::IncrementUnscheduledUseCount(Node* node, Node* from) {
188 // Tracking use counts for fixed nodes is useless.
189 if (GetPlacement(node) == kFixed) return;
190
191 // Use count for coupled nodes is summed up on their control.
192 if (GetPlacement(node) == kCoupled) {
193 node = NodeProperties::GetControlInput(node);
194 DCHECK_NE(GetPlacement(node), Placement::kFixed);
195 DCHECK_NE(GetPlacement(node), Placement::kCoupled);
196 }
197
198 ++(GetData(node)->unscheduled_count_);
199 if (v8_flags.trace_turbo_scheduler) {
200 TRACE(" Use count of #%d:%s (used by #%d:%s)++ = %d\n", node->id(),
201 node->op()->mnemonic(), from->id(), from->op()->mnemonic(),
202 GetData(node)->unscheduled_count_);
203 }

206void Scheduler::DecrementUnscheduledUseCount(Node* node, Node* from) {
207 // Tracking use counts for fixed nodes is useless.
208 if (GetPlacement(node) == kFixed) return;
209
210 // Use count for coupled nodes is summed up on their control.
211 if (GetPlacement(node) == kCoupled) {
212 node = NodeProperties::GetControlInput(node);
213 DCHECK_NE(GetPlacement(node), Placement::kFixed);
214 DCHECK_NE(GetPlacement(node), Placement::kCoupled);
215 }
216
217 DCHECK_LT(0, GetData(node)->unscheduled_count_);
218 --(GetData(node)->unscheduled_count_);
219 if (v8_flags.trace_turbo_scheduler) {
220 TRACE(" Use count of #%d:%s (used by #%d:%s)-- = %d\n", node->id(),
221 node->op()->mnemonic(), from->id(), from->op()->mnemonic(),
222 GetData(node)->unscheduled_count_);
223 }
224 if (GetData(node)->unscheduled_count_ == 0) {
225 TRACE(" newly eligible #%d:%s\n", node->id(), node->op()->mnemonic());
226 schedule_queue_.push(node);
227 }

234// Internal class to build a control flow graph (i.e the basic blocks and edges
235// between them within a Schedule) from the node graph. Visits control edges of
236// the graph backwards from an end node in order to find the connected control
237// subgraph, needed for scheduling.
238class CFGBuilder : public ZoneObject {
239 public:
240 CFGBuilder(Zone* zone, Scheduler* scheduler)
241 : zone_(zone),
242 scheduler_(scheduler),
243 schedule_(scheduler->schedule_),
244 queued_(scheduler->graph_, 2),
245 queue_(zone),
246 control_(zone),
247 component_entry_(nullptr),
248 component_start_(nullptr),
249 component_end_(nullptr) {}
250
251 // Run the control flow graph construction algorithm by walking the graph
252 // backwards from end through control edges, building and connecting the
253 // basic blocks for control nodes.
254 void Run() {
255 ResetDataStructures();
256 Queue(scheduler_->graph_->end());
257
258 while (!queue_.empty()) { // Breadth-first backwards traversal.
259 scheduler_->tick_counter_->TickAndMaybeEnterSafepoint();
260 Node* node = queue_.front();
261 queue_.pop();
262 int max = NodeProperties::PastControlIndex(node);
263 for (int i = NodeProperties::FirstControlIndex(node); i < max; i++) {
264 Queue(node->InputAt(i));
265 }
266 }
267
268 for (NodeVector::iterator i = control_.begin(); i != control_.end(); ++i) {
269 ConnectBlocks(*i); // Connect block to its predecessor/successors.
270 }
271 }
272
273 // Run the control flow graph construction for a minimal control-connected
274 // component ending in {exit} and merge that component into an existing
275 // control flow graph at the bottom of {block}.
276 void Run(BasicBlock* block, Node* exit) {
277 ResetDataStructures();
278 Queue(exit);
279
280 component_entry_ = nullptr;
281 component_start_ = block;
282 component_end_ = schedule_->block(exit);
283 scheduler_->equivalence_->Run(exit);
284 while (!queue_.empty()) { // Breadth-first backwards traversal.
285 scheduler_->tick_counter_->TickAndMaybeEnterSafepoint();
286 Node* node = queue_.front();
287 queue_.pop();
288
289 // Use control dependence equivalence to find a canonical single-entry
290 // single-exit region that makes up a minimal component to be scheduled.
291 if (IsSingleEntrySingleExitRegion(node, exit)) {
292 TRACE("Found SESE at #%d:%s\n", node->id(), node->op()->mnemonic());
293 DCHECK(!component_entry_);
294 component_entry_ = node;
295 continue;
296 }
297
298 int max = NodeProperties::PastControlIndex(node);
299 for (int i = NodeProperties::FirstControlIndex(node); i < max; i++) {
300 Queue(node->InputAt(i));
301 }
302 }
303 DCHECK(component_entry_);
304
305 for (NodeVector::iterator i = control_.begin(); i != control_.end(); ++i) {
306 ConnectBlocks(*i); // Connect block to its predecessor/successors.
307 }
308 }

319 void Queue(Node* node) {
320 // Mark the connected control nodes as they are queued.
321 if (!queued_.Get(node)) {
322 BuildBlocks(node);
323 queue_.push(node);

329 void BuildBlocks(Node* node) {
330 switch (node->opcode()) {
331 case IrOpcode::kEnd:
332 FixNode(schedule_->end(), node);
333 break;
334 case IrOpcode::kStart:
335 FixNode(schedule_->start(), node);
336 break;
337 case IrOpcode::kLoop:
338 case IrOpcode::kMerge:
339 BuildBlockForNode(node);
340 break;
341 case IrOpcode::kTerminate: {
342 // Put Terminate in the loop to which it refers.
343 Node* loop = NodeProperties::GetControlInput(node);
344 BasicBlock* block = BuildBlockForNode(loop);
345 FixNode(block, node);
346 break;
347 }
348 case IrOpcode::kBranch:
349 case IrOpcode::kSwitch:
350 BuildBlocksForSuccessors(node);
351 break;
352#define BUILD_BLOCK_JS_CASE(Name, ...) case IrOpcode::k##Name:
353 JS_OP_LIST(BUILD_BLOCK_JS_CASE)
354// JS opcodes are just like calls => fall through.
355#undef BUILD_BLOCK_JS_CASE
356 case IrOpcode::kCall:
357 case IrOpcode::kFastApiCall:
358 if (NodeProperties::IsExceptionalCall(node)) {
359 BuildBlocksForSuccessors(node);
360 }
361 break;
362 default:
363 break;
364 }
365 }
366
367 void ConnectBlocks(Node* node) {
368 switch (node->opcode()) {
369 case IrOpcode::kLoop:
370 case IrOpcode::kMerge:
371 ConnectMerge(node);
372 break;
373 case IrOpcode::kBranch:
374 scheduler_->UpdatePlacement(node, Scheduler::kFixed);
375 ConnectBranch(node);
376 break;
377 case IrOpcode::kSwitch:
378 scheduler_->UpdatePlacement(node, Scheduler::kFixed);
379 ConnectSwitch(node);
380 break;
381 case IrOpcode::kDeoptimize:
382 scheduler_->UpdatePlacement(node, Scheduler::kFixed);
383 ConnectDeoptimize(node);
384 break;
385 case IrOpcode::kTailCall:
386 scheduler_->UpdatePlacement(node, Scheduler::kFixed);
387 ConnectTailCall(node);
388 break;
389 case IrOpcode::kReturn:
390 scheduler_->UpdatePlacement(node, Scheduler::kFixed);
391 ConnectReturn(node);
392 break;
393 case IrOpcode::kThrow:
394 scheduler_->UpdatePlacement(node, Scheduler::kFixed);
395 ConnectThrow(node);
396 break;
397#define CONNECT_BLOCK_JS_CASE(Name, ...) case IrOpcode::k##Name:
398 JS_OP_LIST(CONNECT_BLOCK_JS_CASE)
399// JS opcodes are just like calls => fall through.
400#undef CONNECT_BLOCK_JS_CASE
401 case IrOpcode::kCall:
402 case IrOpcode::kFastApiCall:
403 if (NodeProperties::IsExceptionalCall(node)) {
404 scheduler_->UpdatePlacement(node, Scheduler::kFixed);
405 ConnectCall(node);
406 }
407 break;
408 default:
409 break;
410 }
411 }

424 void BuildBlocksForSuccessors(Node* node) {
425 size_t const successor_cnt = node->op()->ControlOutputCount();
426 Node** successors = zone_->AllocateArray<Node*>(successor_cnt);
427 NodeProperties::CollectControlProjections(node, successors, successor_cnt);
428 for (size_t index = 0; index < successor_cnt; ++index) {
429 BuildBlockForNode(successors[index]);
430 }

433 void CollectSuccessorBlocks(Node* node, BasicBlock** successor_blocks,
434 size_t successor_cnt) {
435 Node** successors = reinterpret_cast<Node**>(successor_blocks);
436 NodeProperties::CollectControlProjections(node, successors, successor_cnt);
437 for (size_t index = 0; index < successor_cnt; ++index) {
438 successor_blocks[index] = schedule_->block(successors[index]);
439 }
440 }

442 BasicBlock* FindPredecessorBlock(Node* node) {
443 BasicBlock* predecessor_block = nullptr;
444 while (true) {
445 predecessor_block = schedule_->block(node);
446 if (predecessor_block != nullptr) break;
447 node = NodeProperties::GetControlInput(node);
448 }
449 return predecessor_block;
450 }

452 void ConnectCall(Node* call) {
453 BasicBlock* successor_blocks[2];
454 CollectSuccessorBlocks(call, successor_blocks, arraysize(successor_blocks));
455
456 // Consider the exception continuation to be deferred.
457 successor_blocks[1]->set_deferred(true);
458
459 Node* call_control = NodeProperties::GetControlInput(call);
460 BasicBlock* call_block = FindPredecessorBlock(call_control);
461 TraceConnect(call, call_block, successor_blocks[0]);
462 TraceConnect(call, call_block, successor_blocks[1]);
463 schedule_->AddCall(call_block, call, successor_blocks[0],
464 successor_blocks[1]);
465 }
466
467 void ConnectBranch(Node* branch) {
468 BasicBlock* successor_blocks[2];
469 CollectSuccessorBlocks(branch, successor_blocks,
470 arraysize(successor_blocks));
471
472 BranchHint hint_from_profile = BranchHint::kNone;
473 if (const ProfileDataFromFile* profile_data = scheduler_->profile_data()) {
474 hint_from_profile =
475 profile_data->GetHint(successor_blocks[0]->id().ToSize(),
476 successor_blocks[1]->id().ToSize());
477 }
478
479 // Consider branch hints.
480 switch (hint_from_profile) {
481 case BranchHint::kNone:
482 switch (BranchHintOf(branch->op())) {
483 case BranchHint::kNone:
484 break;
485 case BranchHint::kTrue:
486 successor_blocks[1]->set_deferred(true);
487 break;
488 case BranchHint::kFalse:
489 successor_blocks[0]->set_deferred(true);
490 break;
491 }
492 break;
493 case BranchHint::kTrue:
494 successor_blocks[1]->set_deferred(true);
495 break;
496 case BranchHint::kFalse:
497 successor_blocks[0]->set_deferred(true);
498 break;
499 }
500
501 if (branch == component_entry_) {
502 TraceConnect(branch, component_start_, successor_blocks[0]);
503 TraceConnect(branch, component_start_, successor_blocks[1]);
504 schedule_->InsertBranch(component_start_, component_end_, branch,
505 successor_blocks[0], successor_blocks[1]);
506 } else {
507 Node* branch_control = NodeProperties::GetControlInput(branch);
508 BasicBlock* branch_block = FindPredecessorBlock(branch_control);
509 TraceConnect(branch, branch_block, successor_blocks[0]);
510 TraceConnect(branch, branch_block, successor_blocks[1]);
511 schedule_->AddBranch(branch_block, branch, successor_blocks[0],
512 successor_blocks[1]);
513 }
514 }
515
516 void ConnectSwitch(Node* sw) {
517 size_t const successor_count = sw->op()->ControlOutputCount();
518 BasicBlock** successor_blocks =
519 zone_->AllocateArray<BasicBlock*>(successor_count);
520 CollectSuccessorBlocks(sw, successor_blocks, successor_count);
521
522 if (sw == component_entry_) {
523 for (size_t index = 0; index < successor_count; ++index) {
524 TraceConnect(sw, component_start_, successor_blocks[index]);
525 }
526 schedule_->InsertSwitch(component_start_, component_end_, sw,
527 successor_blocks, successor_count);
528 } else {
529 Node* switch_control = NodeProperties::GetControlInput(sw);
530 BasicBlock* switch_block = FindPredecessorBlock(switch_control);
531 for (size_t index = 0; index < successor_count; ++index) {
532 TraceConnect(sw, switch_block, successor_blocks[index]);
533 }
534 schedule_->AddSwitch(switch_block, sw, successor_blocks, successor_count);
535 }
536 for (size_t index = 0; index < successor_count; ++index) {
537 if (BranchHintOf(successor_blocks[index]->front()->op()) ==
538 BranchHint::kFalse) {
539 successor_blocks[index]->set_deferred(true);
540 }
541 }
542 }

544 void ConnectMerge(Node* merge) {
545 // Don't connect the special merge at the end to its predecessors.
546 if (IsFinalMerge(merge)) return;
547
548 BasicBlock* block = schedule_->block(merge);
549 DCHECK_NOT_NULL(block);
550 // For all of the merge's control inputs, add a goto at the end to the
551 // merge's basic block.
552 for (Node* const input : merge->inputs()) {
553 BasicBlock* predecessor_block = FindPredecessorBlock(input);
554 TraceConnect(merge, predecessor_block, block);
555 schedule_->AddGoto(predecessor_block, block);
556 }
557 }

559 void ConnectTailCall(Node* call) {
560 Node* call_control = NodeProperties::GetControlInput(call);
561 BasicBlock* call_block = FindPredecessorBlock(call_control);
562 TraceConnect(call, call_block, nullptr);
563 schedule_->AddTailCall(call_block, call);
564 }

567 Node* return_control = NodeProperties::GetControlInput(ret);
568 BasicBlock* return_block = FindPredecessorBlock(return_control);
569 TraceConnect(ret, return_block, nullptr);
570 schedule_->AddReturn(return_block, ret);
571 }

575 BasicBlock* deoptimize_block = FindPredecessorBlock(deoptimize_control);
576 TraceConnect(deopt, deoptimize_block, nullptr);
577 schedule_->AddDeoptimize(deoptimize_block, deopt);

649// Compute the special reverse-post-order block ordering, which is essentially
650// a RPO of the graph where loop bodies are contiguous. Properties:
651// 1. If block A is a predecessor of B, then A appears before B in the order,
652// unless B is a loop header and A is in the loop headed at B
653// (i.e. A -> B is a backedge).
654// => If block A dominates block B, then A appears before B in the order.
655// => If block A is a loop header, A appears before all blocks in the loop
656// headed at A.
657// 2. All loops are contiguous in the order (i.e. no intervening blocks that
658// do not belong to the loop.)
659// Note a simple RPO traversal satisfies (1) but not (2).
660class SpecialRPONumberer : public ZoneObject {
661 public:
662 SpecialRPONumberer(Zone* zone, Schedule* schedule)
663 : zone_(zone),
664 schedule_(schedule),
665 order_(nullptr),
666 beyond_end_(nullptr),
667 loops_(zone),
668 backedges_(zone),
669 stack_(zone),
670 previous_block_count_(0),
671 empty_(0, zone) {}

673 // Computes the special reverse-post-order for the main control flow graph,
674 // that is for the graph spanned between the schedule's start and end blocks.
675 void ComputeSpecialRPO() {
676 DCHECK_EQ(0, schedule_->end()->SuccessorCount());
677 DCHECK(!order_); // Main order does not exist yet.
678 ComputeAndInsertSpecialRPO(schedule_->start(), schedule_->end());
679 }
680
681 // Computes the special reverse-post-order for a partial control flow graph,
682 // that is for the graph spanned between the given {entry} and {end} blocks,
683 // then updates the existing ordering with this new information.
684 void UpdateSpecialRPO(BasicBlock* entry, BasicBlock* end) {
685 DCHECK(order_); // Main order to be updated is present.
686 ComputeAndInsertSpecialRPO(entry, end);

782 // Compute special RPO for the control flow graph between {entry} and {end},
783 // mutating any existing order so that the result is still valid.
784 void ComputeAndInsertSpecialRPO(BasicBlock* entry, BasicBlock* end) {
785 // RPO should not have been serialized for this schedule yet.
786 CHECK_EQ(kBlockUnvisited1, schedule_->start()->loop_number());
787 CHECK_EQ(kBlockUnvisited1, schedule_->start()->rpo_number());
788 CHECK_EQ(0, static_cast<int>(schedule_->rpo_order()->size()));
789
790 // Find correct insertion point within existing order.
791 BasicBlock* insertion_point = entry->rpo_next();
792 BasicBlock* order = insertion_point;
793
794 // Perform an iterative RPO traversal using an explicit stack,
795 // recording backedges that form cycles. O(|B|).
796 DCHECK_LT(previous_block_count_, schedule_->BasicBlockCount());
797 stack_.resize(schedule_->BasicBlockCount() - previous_block_count_);
798 previous_block_count_ = schedule_->BasicBlockCount();
799 int stack_depth = Push(0, entry, kBlockUnvisited1);
800 int num_loops = static_cast<int>(loops_.size());
801
802 while (stack_depth > 0) {
803 int current = stack_depth - 1;
804 SpecialRPOStackFrame* frame = &stack_[current];
805
806 if (frame->block != end &&
807 frame->index < frame->block->SuccessorCount()) {
808 // Process the next successor.
809 BasicBlock* succ = frame->block->SuccessorAt(frame->index++);
810 if (succ->rpo_number() == kBlockVisited1) continue;
811 if (succ->rpo_number() == kBlockOnStack) {
812 // The successor is on the stack, so this is a backedge (cycle).
813 backedges_.push_back(Backedge(frame->block, frame->index - 1));
814 if (!HasLoopNumber(succ)) {
815 // Assign a new loop number to the header if it doesn't have one.
816 SetLoopNumber(succ, num_loops++);
817 }
818 } else {
819 // Push the successor onto the stack.
820 DCHECK_EQ(kBlockUnvisited1, succ->rpo_number());
821 stack_depth = Push(stack_depth, succ, kBlockUnvisited1);
822 }
823 } else {
824 // Finished with all successors; pop the stack and add the block.
825 order = PushFront(order, frame->block);
826 frame->block->set_rpo_number(kBlockVisited1);
827 stack_depth--;
828 }
829 }
830
831 // If no loops were encountered, then the order we computed was correct.
832 if (num_loops > static_cast<int>(loops_.size())) {
833 // Otherwise, compute the loop information from the backedges in order
834 // to perform a traversal that groups loop bodies together.
835 ComputeLoopInfo(&stack_, num_loops, &backedges_);
836
837 // Initialize the "loop stack". Note the entry could be a loop header.
838 LoopInfo* loop =
839 HasLoopNumber(entry) ? &loops_[GetLoopNumber(entry)] : nullptr;
840 order = insertion_point;
841
842 // Perform an iterative post-order traversal, visiting loop bodies before
843 // edges that lead out of loops. Visits each block once, but linking loop
844 // sections together is linear in the loop size, so overall is
845 // O(|B| + max(loop_depth) * max(|loop|))
846 stack_depth = Push(0, entry, kBlockUnvisited2);
847 while (stack_depth > 0) {
848 SpecialRPOStackFrame* frame = &stack_[stack_depth - 1];
849 BasicBlock* block = frame->block;
850 BasicBlock* succ = nullptr;
851
852 if (block != end && frame->index < block->SuccessorCount()) {
853 // Process the next normal successor.
854 succ = block->SuccessorAt(frame->index++);
855 } else if (HasLoopNumber(block)) {
856 // Process additional outgoing edges from the loop header.
857 if (block->rpo_number() == kBlockOnStack) {
858 // Finish the loop body the first time the header is left on the
859 // stack.
860 DCHECK(loop != nullptr && loop->header == block);
861 loop->start = PushFront(order, block);
862 order = loop->end;
863 block->set_rpo_number(kBlockVisited2);
864 // Pop the loop stack and continue visiting outgoing edges within
865 // the context of the outer loop, if any.
866 loop = loop->prev;
867 // We leave the loop header on the stack; the rest of this iteration
868 // and later iterations will go through its outgoing edges list.
869 }
870
871 // Use the next outgoing edge if there are any.
872 size_t outgoing_index = frame->index - block->SuccessorCount();
873 LoopInfo* info = &loops_[GetLoopNumber(block)];
874 DCHECK(loop != info);
875 if (block != entry && info->outgoing != nullptr &&
876 outgoing_index < info->outgoing->size()) {
877 succ = info->outgoing->at(outgoing_index);
878 frame->index++;
879 }
880 }
881
882 if (succ != nullptr) {
883 // Process the next successor.
884 if (succ->rpo_number() == kBlockOnStack) continue;
885 if (succ->rpo_number() == kBlockVisited2) continue;
886 DCHECK_EQ(kBlockUnvisited2, succ->rpo_number());
887 if (loop != nullptr && !loop->members->Contains(succ->id().ToInt())) {
888 // The successor is not in the current loop or any nested loop.
889 // Add it to the outgoing edges of this loop and visit it later.
890 loop->AddOutgoing(zone_, succ);
891 } else {
892 // Push the successor onto the stack.
893 stack_depth = Push(stack_depth, succ, kBlockUnvisited2);
894 if (HasLoopNumber(succ)) {
895 // Push the inner loop onto the loop stack.
896 DCHECK(GetLoopNumber(succ) < num_loops);
897 LoopInfo* next = &loops_[GetLoopNumber(succ)];
898 next->end = order;
899 next->prev = loop;
900 loop = next;
901 }
902 }
903 } else {
904 // Finished with all successors of the current block.
905 if (HasLoopNumber(block)) {
906 // If we are going to pop a loop header, then add its entire body.
907 LoopInfo* info = &loops_[GetLoopNumber(block)];
908 for (BasicBlock* b = info->start; true; b = b->rpo_next()) {
909 if (b->rpo_next() == info->end) {
910 b->set_rpo_next(order);
911 info->end = order;
912 break;
913 }
914 }
915 order = info->start;
916 } else {
917 // Pop a single node off the stack and add it to the order.
918 order = PushFront(order, block);
919 block->set_rpo_number(kBlockVisited2);
920 }
921 stack_depth--;
922 }
923 }
924 }
925
926 // Publish new order the first time.
927 if (order_ == nullptr) order_ = order;
928
929 // Compute the correct loop headers and set the correct loop ends.
930 LoopInfo* current_loop = nullptr;
931 BasicBlock* current_header = entry->loop_header();
932 int32_t loop_depth = entry->loop_depth();
933 if (entry->IsLoopHeader()) --loop_depth; // Entry might be a loop header.
934 for (BasicBlock* b = order; b != insertion_point; b = b->rpo_next()) {
935 BasicBlock* current = b;
936
937 // Reset BasicBlock::rpo_number again.
938 current->set_rpo_number(kBlockUnvisited1);
939
940 // Finish the previous loop(s) if we just exited them.
941 while (current_header != nullptr &&
942 current == current_header->loop_end()) {
943 DCHECK(current_header->IsLoopHeader());
944 DCHECK_NOT_NULL(current_loop);
945 current_loop = current_loop->prev;
946 current_header =
947 current_loop == nullptr ? nullptr : current_loop->header;
948 --loop_depth;
949 }
950 current->set_loop_header(current_header);
951
952 // Push a new loop onto the stack if this loop is a loop header.
953 if (HasLoopNumber(current)) {
954 ++loop_depth;
955 current_loop = &loops_[GetLoopNumber(current)];
956 BasicBlock* loop_end = current_loop->end;
957 current->set_loop_end(loop_end == nullptr ? BeyondEndSentinel()
958 : loop_end);
959 current_header = current_loop->header;
960 TRACE("id:%d is a loop header, increment loop depth to %d\n",
961 current->id().ToInt(), loop_depth);
962 }
963
964 current->set_loop_depth(loop_depth);
965
966 if (current->loop_header() == nullptr) {
967 TRACE("id:%d is not in a loop (depth == %d)\n", current->id().ToInt(),
968 current->loop_depth());
969 } else {
970 TRACE("id:%d has loop header id:%d, (depth == %d)\n",
971 current->id().ToInt(), current->loop_header()->id().ToInt(),
972 current->loop_depth());
973 }
974 }
975 }

977 // Computes loop membership from the backedges of the control flow graph.
978 void ComputeLoopInfo(ZoneVector<SpecialRPOStackFrame>* queue,
979 size_t num_loops, ZoneVector<Backedge>* backedges) {
980 // Extend existing loop membership vectors.
981 for (LoopInfo& loop : loops_) {
982 loop.members->Resize(static_cast<int>(schedule_->BasicBlockCount()),
983 zone_);
984 }
985
986 // Extend loop information vector.
987 loops_.resize(num_loops, LoopInfo());
988
989 // Compute loop membership starting from backedges.
990 // O(max(loop_depth) * max(|loop|)
991 for (size_t i = 0; i < backedges->size(); i++) {
992 BasicBlock* member = backedges->at(i).first;
993 BasicBlock* header = member->SuccessorAt(backedges->at(i).second);
994 size_t loop_num = GetLoopNumber(header);
995 if (loops_[loop_num].header == nullptr) {
996 loops_[loop_num].header = header;
997 loops_[loop_num].members = zone_->New<BitVector>(
998 static_cast<int>(schedule_->BasicBlockCount()), zone_);
999 }
1000
1001 int queue_length = 0;
1002 if (member != header) {
1003 // As long as the header doesn't have a backedge to itself,
1004 // Push the member onto the queue and process its predecessors.
1005 if (!loops_[loop_num].members->Contains(member->id().ToInt())) {
1006 loops_[loop_num].members->Add(member->id().ToInt());
1007 }
1008 (*queue)[queue_length++].block = member;
1009 }
1010
1011 // Propagate loop membership backwards. All predecessors of M up to the
1012 // loop header H are members of the loop too. O(|blocks between M and H|).
1013 while (queue_length > 0) {
1014 BasicBlock* block = (*queue)[--queue_length].block;
1015 for (size_t j = 0; j < block->PredecessorCount(); j++) {
1016 BasicBlock* pred = block->PredecessorAt(j);
1017 if (pred != header) {
1018 if (!loops_[loop_num].members->Contains(pred->id().ToInt())) {
1019 loops_[loop_num].members->Add(pred->id().ToInt());
1020 (*queue)[queue_length++].block = pred;
1021 }
1022 }
1023 }
1024 }
1025 }
1026 }

1066 void VerifySpecialRPO() {
1067 BasicBlockVector* order = schedule_->rpo_order();
1068 DCHECK_LT(0, order->size());
1069 DCHECK_EQ(0, (*order)[0]->id().ToInt()); // entry should be first.
1070
1071 for (size_t i = 0; i < loops_.size(); i++) {
1072 LoopInfo* loop = &loops_[i];
1073 BasicBlock* header = loop->header;
1074 BasicBlock* end = header->loop_end();
1075
1076 DCHECK_NOT_NULL(header);
1077 DCHECK_LE(0, header->rpo_number());
1078 DCHECK_LT(header->rpo_number(), order->size());
1079 DCHECK_NOT_NULL(end);
1080 DCHECK_LE(end->rpo_number(), order->size());
1081 DCHECK_GT(end->rpo_number(), header->rpo_number());
1082 DCHECK_NE(header->loop_header(), header);
1083
1084 // Verify the start ... end list relationship.
1085 int links = 0;
1086 BasicBlock* block = loop->start;
1087 DCHECK_EQ(header, block);
1088 bool end_found;
1089 while (true) {
1090 if (block == nullptr || block == loop->end) {
1091 end_found = (loop->end == block);
1092 break;
1093 }
1094 // The list should be in same order as the final result.
1095 DCHECK(block->rpo_number() == links + header->rpo_number());
1096 links++;
1097 block = block->rpo_next();
1098 DCHECK_LT(links, static_cast<int>(2 * order->size())); // cycle?
1099 }
1100 DCHECK_LT(0, links);
1101 DCHECK_EQ(links, end->rpo_number() - header->rpo_number());
1102 DCHECK(end_found);
1103
1104 // Check loop depth of the header.
1105 int loop_depth = 0;
1106 for (LoopInfo* outer = loop; outer != nullptr; outer = outer->prev) {
1107 loop_depth++;
1108 }
1109 DCHECK_EQ(loop_depth, header->loop_depth());
1110
1111 // Check the contiguousness of loops.
1112 int count = 0;
1113 for (int j = 0; j < static_cast<int>(order->size()); j++) {
1114 block = order->at(j);
1115 DCHECK_EQ(block->rpo_number(), j);
1116 if (j < header->rpo_number() || j >= end->rpo_number()) {
1117 DCHECK(!header->LoopContains(block));
1118 } else {
1119 DCHECK(header->LoopContains(block));
1120 DCHECK_GE(block->loop_depth(), loop_depth);
1121 count++;
1122 }
1123 }
1124 DCHECK_EQ(links, count);
1125 }
1126 }
1127#endif // DEBUG

1167BasicBlock* Scheduler::GetCommonDominator(BasicBlock* b1, BasicBlock* b2) {
1168 // A very common fast case:
1169 if (b1 == b2) return b1;
1170 // Try to find the common dominator by walking, if there is a chance of
1171 // finding it quickly.
1172 constexpr int kCacheGranularity = 63;
1173 static_assert((kCacheGranularity & (kCacheGranularity + 1)) == 0);
1174 int depth_difference = b1->dominator_depth() - b2->dominator_depth();
1175 if (depth_difference > -kCacheGranularity &&
1176 depth_difference < kCacheGranularity) {
1177 for (int i = 0; i < kCacheGranularity; i++) {
1178 if (b1->dominator_depth() < b2->dominator_depth()) {
1179 b2 = b2->dominator();
1180 } else {
1181 b1 = b1->dominator();
1182 }
1183 if (b1 == b2) return b1;
1184 }
1185 // We might fall out of the loop here if the dominator tree has several
1186 // deep "parallel" subtrees.
1187 }
1188 // If it'd be a long walk, take the bus instead (i.e. use the cache).
1189 // To keep memory consumption low, there'll be a bus stop every 64 blocks.
1190 // First, walk to the nearest bus stop.
1191 if (b1->dominator_depth() < b2->dominator_depth()) std::swap(b1, b2);
1192 while ((b1->dominator_depth() & kCacheGranularity) != 0) {
1193 if (V8_LIKELY(b1->dominator_depth() > b2->dominator_depth())) {
1194 b1 = b1->dominator();
1195 } else {
1196 b2 = b2->dominator();
1197 }
1198 if (b1 == b2) return b1;
1199 }
1200 // Then, walk from bus stop to bus stop until we either find a bus (i.e. an
1201 // existing cache entry) or the result. Make a list of any empty bus stops
1202 // we'd like to populate for next time.
1203 constexpr int kMaxNewCacheEntries = 2 * 50; // Must be even.
1204 // This array stores a flattened list of pairs, e.g. if after finding the
1205 // {result}, we want to cache [(B11, B12) -> result, (B21, B22) -> result],
1206 // then we store [11, 12, 21, 22] here.
1207 int new_cache_entries[kMaxNewCacheEntries];
1208 // Next free slot in {new_cache_entries}.
1209 int new_cache_entries_cursor = 0;
1210 while (b1 != b2) {
1211 if ((b1->dominator_depth() & kCacheGranularity) == 0) {
1212 BasicBlock* maybe_cache_hit = GetCommonDominatorIfCached(b1, b2);
1213 if (maybe_cache_hit != nullptr) {
1214 b1 = b2 = maybe_cache_hit;
1215 break;
1216 } else if (new_cache_entries_cursor < kMaxNewCacheEntries) {
1217 new_cache_entries[new_cache_entries_cursor++] = b1->id().ToInt();
1218 new_cache_entries[new_cache_entries_cursor++] = b2->id().ToInt();
1219 }
1220 }
1221 if (V8_LIKELY(b1->dominator_depth() > b2->dominator_depth())) {
1222 b1 = b1->dominator();
1223 } else {
1224 b2 = b2->dominator();
1225 }
1226 }
1227 // Lastly, create new cache entries we noted down earlier.
1228 BasicBlock* result = b1;
1229 for (int i = 0; i < new_cache_entries_cursor;) {
1230 int id1 = new_cache_entries[i++];
1231 int id2 = new_cache_entries[i++];
1232 ZoneMap<int, BasicBlock*>* mapping;
1233 auto entry = common_dominator_cache_.find(id1);
1234 if (entry == common_dominator_cache_.end()) {
1235 mapping = zone_->New<ZoneMap<int, BasicBlock*>>(zone_);
1236 common_dominator_cache_[id1] = mapping;
1237 } else {
1238 mapping = entry->second;
1239 }
1240 // If there was an existing entry, we would have found it earlier.
1241 DCHECK_EQ(mapping->find(id2), mapping->end());
1242 mapping->insert({id2, result});
1243 }
1244 return result;
1245}

1247void Scheduler::PropagateImmediateDominators(BasicBlock* block) {
1248 for (/*nop*/; block != nullptr; block = block->rpo_next()) {
1249 auto pred = block->predecessors().begin();
1250 auto end = block->predecessors().end();
1251 DCHECK(pred != end); // All blocks except start have predecessors.
1252 BasicBlock* dominator = *pred;
1253 bool deferred = dominator->deferred();
1254 // For multiple predecessors, walk up the dominator tree until a common
1255 // dominator is found. Visitation order guarantees that all predecessors
1256 // except for backwards edges have been visited.
1257 // We use a one-element cache for previously-seen dominators. This gets
1258 // hit a lot for functions that have long chains of diamonds, and in
1259 // those cases turns quadratic into linear complexity.
1260 BasicBlock* cache = nullptr;
1261 for (++pred; pred != end; ++pred) {
1262 // Don't examine backwards edges.
1263 if ((*pred)->dominator_depth() < 0) continue;
1264 if ((*pred)->dominator_depth() > 3 &&
1265 ((*pred)->dominator()->dominator() == cache ||
1266 (*pred)->dominator()->dominator()->dominator() == cache)) {
1267 // Nothing to do, the last iteration covered this case.
1268 DCHECK_EQ(dominator, BasicBlock::GetCommonDominator(dominator, *pred));
1269 } else {
1270 dominator = BasicBlock::GetCommonDominator(dominator, *pred);
1271 }
1272 cache = (*pred)->dominator();
1273 deferred = deferred & (*pred)->deferred();
1274 }
1275 block->set_dominator(dominator);
1276 block->set_dominator_depth(dominator->dominator_depth() + 1);
1277 block->set_deferred(deferred | block->deferred());
1278 TRACE("Block id:%d's idom is id:%d, depth = %d\n", block->id().ToInt(),
1279 dominator->id().ToInt(), block->dominator_depth());
1280 }
1281}

1303 : scheduler_(scheduler),
1304 schedule_(scheduler->schedule_),
1305 graph_(graph),
1306 visited_(static_cast<int>(graph_->NodeCount()), zone),
1307 stack_(zone) {}
1308
1309 void Run() {
1310 InitializePlacement(graph_->end());
1311 while (!stack_.empty()) {
1312 Node* node = stack_.top();
1313 stack_.pop();
1314 VisitInputs(node);
1315 }
1316 }
1317
1318 private:
1319 void InitializePlacement(Node* node) {
1320 TRACE("Pre #%d:%s\n", node->id(), node->op()->mnemonic());
1321 DCHECK(!Visited(node));
1322 if (scheduler_->InitializePlacement(node) == Scheduler::kFixed) {
1323 // Fixed nodes are always roots for schedule late.
1324 scheduler_->schedule_root_nodes_.push_back(node);
1325 if (!schedule_->IsScheduled(node)) {
1326 // Make sure root nodes are scheduled in their respective blocks.
1327 TRACE("Scheduling fixed position node #%d:%s\n", node->id(),
1328 node->op()->mnemonic());
1329 IrOpcode::Value opcode = node->opcode();
1330 BasicBlock* block =
1331 opcode == IrOpcode::kParameter
1332 ? schedule_->start()
1333 : schedule_->block(NodeProperties::GetControlInput(node));
1334 DCHECK_NOT_NULL(block);
1335 schedule_->AddNode(block, node);
1336 }
1337 }
1338 stack_.push(node);
1339 visited_.Add(node->id());
1340 }
1341
1342 void VisitInputs(Node* node) {
1343 DCHECK_NE(scheduler_->GetPlacement(node), Scheduler::kUnknown);
1344 bool is_scheduled = schedule_->IsScheduled(node);
1345 std::optional<int> coupled_control_edge =
1346 scheduler_->GetCoupledControlEdge(node);
1347 for (auto edge : node->input_edges()) {
1348 Node* to = edge.to();
1349 DCHECK_EQ(node, edge.from());
1350 if (!Visited(to)) {
1351 InitializePlacement(to);
1352 }
1353 TRACE("PostEdge #%d:%s->#%d:%s\n", node->id(), node->op()->mnemonic(),
1354 to->id(), to->op()->mnemonic());
1355 DCHECK_NE(scheduler_->GetPlacement(to), Scheduler::kUnknown);
1356 if (!is_scheduled && edge.index() != coupled_control_edge) {
1357 scheduler_->IncrementUnscheduledUseCount(to, node);
1358 }
1359 }
1360 }
1361
1362 bool Visited(Node* node) { return visited_.Contains(node->id()); }
1363
1364 Scheduler* scheduler_;
1365 Schedule* schedule_;
1366 Graph* graph_;
1367 BitVector visited_;
1368 ZoneStack<Node*> stack_;
1369};

1383// Phase 4: Schedule nodes early.
1384
1385
1386class ScheduleEarlyNodeVisitor {
1387 public:
1388 ScheduleEarlyNodeVisitor(Zone* zone, Scheduler* scheduler)
1389 : scheduler_(scheduler), schedule_(scheduler->schedule_), queue_(zone) {}
1390
1391 // Run the schedule early algorithm on a set of fixed root nodes.
1392 void Run(NodeVector* roots) {
1393 for (Node* const root : *roots) {
1394 queue_.push(root);
1395 }
1396
1397 while (!queue_.empty()) {
1398 scheduler_->tick_counter_->TickAndMaybeEnterSafepoint();
1399 VisitNode(queue_.front());
1400 queue_.pop();
1401 }
1402 }
1403
1404 private:
1405 // Visits one node from the queue and propagates its current schedule early
1406 // position to all uses. This in turn might push more nodes onto the queue.
1407 void VisitNode(Node* node) {
1408 Scheduler::SchedulerData* data = scheduler_->GetData(node);
1409
1410 // Fixed nodes already know their schedule early position.
1411 if (scheduler_->GetPlacement(node) == Scheduler::kFixed) {
1412 data->minimum_block_ = schedule_->block(node);
1413 TRACE("Fixing #%d:%s minimum_block = id:%d, dominator_depth = %d\n",
1414 node->id(), node->op()->mnemonic(),
1415 data->minimum_block_->id().ToInt(),
1416 data->minimum_block_->dominator_depth());
1417 }
1418
1419 // No need to propagate unconstrained schedule early positions.
1420 if (data->minimum_block_ == schedule_->start()) return;
1421
1422 // Propagate schedule early position.
1423 DCHECK_NOT_NULL(data->minimum_block_);
1424 for (auto use : node->uses()) {
1425 if (scheduler_->IsLive(use)) {
1426 PropagateMinimumPositionToNode(data->minimum_block_, use);
1427 }
1428 }
1429 }
1430
1431 // Propagates {block} as another minimum position into the given {node}. This
1432 // has the net effect of computing the minimum dominator block of {node} that
1433 // still post-dominates all inputs to {node} when the queue is processed.
1434 void PropagateMinimumPositionToNode(BasicBlock* block, Node* node) {
1435 Scheduler::SchedulerData* data = scheduler_->GetData(node);
1436
1437 // No need to propagate to fixed node, it's guaranteed to be a root.
1438 if (scheduler_->GetPlacement(node) == Scheduler::kFixed) return;
1439
1440 // Coupled nodes influence schedule early position of their control.
1441 if (scheduler_->GetPlacement(node) == Scheduler::kCoupled) {
1442 Node* control = NodeProperties::GetControlInput(node);
1443 PropagateMinimumPositionToNode(block, control);
1444 }
1445
1446 // Propagate new position if it is deeper down the dominator tree than the
1447 // current. Note that all inputs need to have minimum block position inside
1448 // the dominator chain of {node}'s minimum block position.
1449 DCHECK(InsideSameDominatorChain(block, data->minimum_block_));
1450 if (block->dominator_depth() > data->minimum_block_->dominator_depth()) {
1451 data->minimum_block_ = block;
1452 queue_.push(node);
1453 TRACE("Propagating #%d:%s minimum_block = id:%d, dominator_depth = %d\n",
1454 node->id(), node->op()->mnemonic(),
1455 data->minimum_block_->id().ToInt(),
1456 data->minimum_block_->dominator_depth());
1457 }
1458 }
1459
1460#if DEBUG
1461 bool InsideSameDominatorChain(BasicBlock* b1, BasicBlock* b2) {
1462 BasicBlock* dominator = BasicBlock::GetCommonDominator(b1, b2);
1463 return dominator == b1 || dominator == b2;
1464 }
1465#endif
1466
1467 Scheduler* scheduler_;
1468 Schedule* schedule_;
1469 ZoneQueue<Node*> queue_;
1470};

1478
1479 TRACE("--- SCHEDULE EARLY -----------------------------------------\n");

1496// Phase 5: Schedule nodes late.
1497
1498
1499class ScheduleLateNodeVisitor {
1500 public:
1501 ScheduleLateNodeVisitor(Zone* zone, Scheduler* scheduler)
1502 : zone_(zone),
1503 scheduler_(scheduler),
1504 schedule_(scheduler_->schedule_),
1505 marking_queue_(scheduler->zone_) {}
1506
1507 // Run the schedule late algorithm on a set of fixed root nodes.
1508 void Run(NodeVector* roots) {
1509 for (Node* const root : *roots) {
1510 ProcessQueue(root);
1511 }
1512 }
1513
1514 private:
1515 void ProcessQueue(Node* root) {
1516 ZoneQueue<Node*>* queue = &(scheduler_->schedule_queue_);
1517 for (Node* node : root->inputs()) {
1518 // Don't schedule coupled nodes on their own.
1519 if (scheduler_->GetPlacement(node) == Scheduler::kCoupled) {
1520 node = NodeProperties::GetControlInput(node);
1521 }
1522
1523 // Test schedulability condition by looking at unscheduled use count.
1524 if (scheduler_->GetData(node)->unscheduled_count_ != 0) continue;
1525
1526 queue->push(node);
1527 do {
1528 scheduler_->tick_counter_->TickAndMaybeEnterSafepoint();
1529 Node* const n = queue->front();
1530 queue->pop();
1531 VisitNode(n);
1532 } while (!queue->empty());
1533 }
1534 }
1535
1536 // Visits one node from the queue of schedulable nodes and determines its
1537 // schedule late position. Also hoists nodes out of loops to find a more
1538 // optimal scheduling position.
1539 void VisitNode(Node* node) {
1540 DCHECK_EQ(0, scheduler_->GetData(node)->unscheduled_count_);
1541
1542 // Don't schedule nodes that are already scheduled.
1543 if (schedule_->IsScheduled(node)) return;
1544 DCHECK_EQ(Scheduler::kSchedulable, scheduler_->GetPlacement(node));
1545
1546 // Determine the dominating block for all of the uses of this node. It is
1547 // the latest block that this node can be scheduled in.
1548 TRACE("Scheduling #%d:%s\n", node->id(), node->op()->mnemonic());
1549 BasicBlock* block = GetCommonDominatorOfUses(node);
1550 DCHECK_NOT_NULL(block);
1551
1552 // The schedule early block dominates the schedule late block.
1553 BasicBlock* min_block = scheduler_->GetData(node)->minimum_block_;
1554 DCHECK_EQ(min_block, BasicBlock::GetCommonDominator(block, min_block));
1555
1556 TRACE(
1557 "Schedule late of #%d:%s is id:%d at loop depth %d, minimum = id:%d\n",
1558 node->id(), node->op()->mnemonic(), block->id().ToInt(),
1559 block->loop_depth(), min_block->id().ToInt());
1560
1561 // Hoist nodes out of loops if possible. Nodes can be hoisted iteratively
1562 // into enclosing loop pre-headers until they would precede their schedule
1563 // early position.
1564 BasicBlock* hoist_block = GetHoistBlock(block);
1565 if (hoist_block &&
1566 hoist_block->dominator_depth() >= min_block->dominator_depth()) {
1567 DCHECK(scheduler_->special_rpo_->HasLoopBlocks());
1568 do {
1569 TRACE(" hoisting #%d:%s to block id:%d\n", node->id(),
1570 node->op()->mnemonic(), hoist_block->id().ToInt());
1571 DCHECK_LT(hoist_block->loop_depth(), block->loop_depth());
1572 block = hoist_block;
1573 hoist_block = GetHoistBlock(hoist_block);
1574 } while (hoist_block &&
1575 hoist_block->dominator_depth() >= min_block->dominator_depth());
1576 } else if (scheduler_->flags_ & Scheduler::kSplitNodes) {
1577 // Split the {node} if beneficial and return the new {block} for it.
1578 block = SplitNode(block, node);
1579 }
1580
1581 // Schedule the node or a floating control structure.
1582 if (IrOpcode::IsMergeOpcode(node->opcode())) {
1583 ScheduleFloatingControl(block, node);
1584 } else if (node->opcode() == IrOpcode::kFinishRegion) {
1585 ScheduleRegion(block, node);
1586 } else {
1587 ScheduleNode(block, node);
1588 }
1589 }
1590
1591 bool IsMarked(BasicBlock* block) const {
1592 return marked_.Contains(block->id().ToInt());
1593 }
1594
1595 void Mark(BasicBlock* block) { marked_.Add(block->id().ToInt()); }
1596
1597 // Mark {block} and push its non-marked predecessor on the marking queue.
1598 void MarkBlock(BasicBlock* block) {
1599 Mark(block);
1600 for (BasicBlock* pred_block : block->predecessors()) {
1601 if (IsMarked(pred_block)) continue;
1602 marking_queue_.push_back(pred_block);
1603 }
1604 }
1605
1606 BasicBlock* SplitNode(BasicBlock* block, Node* node) {
1607 // For now, we limit splitting to pure nodes.
1608 if (!node->op()->HasProperty(Operator::kPure)) return block;
1609 // TODO(titzer): fix the special case of splitting of projections.
1610 if (node->opcode() == IrOpcode::kProjection) return block;
1611
1612 // The {block} is common dominator of all uses of {node}, so we cannot
1613 // split anything unless the {block} has at least two successors.
1614 DCHECK_EQ(block, GetCommonDominatorOfUses(node));
1615 if (block->SuccessorCount() < 2) return block;
1616
1617 // Clear marking bits.
1618 DCHECK(marking_queue_.empty());
1619 marked_.Clear();
1620 int new_size = static_cast<int>(schedule_->BasicBlockCount() + 1);
1621 if (marked_.length() < new_size) {
1622 marked_.Resize(new_size, scheduler_->zone_);
1623 }
1624
1625 // Check if the {node} has uses in {block}.
1626 for (Edge edge : node->use_edges()) {
1627 if (!scheduler_->IsLive(edge.from())) continue;
1628 BasicBlock* use_block = GetBlockForUse(edge);
1629 if (use_block == nullptr || IsMarked(use_block)) continue;
1630 if (use_block == block) {
1631 TRACE(" not splitting #%d:%s, it is used in id:%d\n", node->id(),
1632 node->op()->mnemonic(), block->id().ToInt());
1633 marking_queue_.clear();
1634 return block;
1635 }
1636 MarkBlock(use_block);
1637 }
1638
1639 // Compute transitive marking closure; a block is marked if all its
1640 // successors are marked.
1641 do {
1642 BasicBlock* top_block = marking_queue_.front();
1643 marking_queue_.pop_front();
1644 if (IsMarked(top_block)) continue;
1645 bool marked = true;
1646 if (top_block->loop_depth() == block->loop_depth()) {
1647 for (BasicBlock* successor : top_block->successors()) {
1648 if (!IsMarked(successor)) {
1649 marked = false;
1650 break;
1651 }
1652 }
1653 }
1654 if (marked) MarkBlock(top_block);
1655 } while (!marking_queue_.empty());
1656
1657 // If the (common dominator) {block} is marked, we know that all paths from
1658 // {block} to the end contain at least one use of {node}, and hence there's
1659 // no point in splitting the {node} in this case.
1660 if (IsMarked(block)) {
1661 TRACE(" not splitting #%d:%s, its common dominator id:%d is perfect\n",
1662 node->id(), node->op()->mnemonic(), block->id().ToInt());
1663 return block;
1664 }
1665
1666 // Split {node} for uses according to the previously computed marking
1667 // closure. Every marking partition has a unique dominator, which get's a
1668 // copy of the {node} with the exception of the first partition, which get's
1669 // the {node} itself.
1670 ZoneMap<BasicBlock*, Node*> dominators(scheduler_->zone_);
1671 for (Edge edge : node->use_edges()) {
1672 if (!scheduler_->IsLive(edge.from())) continue;
1673 BasicBlock* use_block = GetBlockForUse(edge);
1674 if (use_block == nullptr) continue;
1675 while (IsMarked(use_block->dominator())) {
1676 use_block = use_block->dominator();
1677 }
1678 auto& use_node = dominators[use_block];
1679 if (use_node == nullptr) {
1680 if (dominators.size() == 1u) {
1681 // Place the {node} at {use_block}.
1682 block = use_block;
1683 use_node = node;
1684 TRACE(" pushing #%d:%s down to id:%d\n", node->id(),
1685 node->op()->mnemonic(), block->id().ToInt());
1686 } else {
1687 // Place a copy of {node} at {use_block}.
1688 use_node = CloneNode(node);
1689 TRACE(" cloning #%d:%s for id:%d\n", use_node->id(),
1690 use_node->op()->mnemonic(), use_block->id().ToInt());
1691 scheduler_->schedule_queue_.push(use_node);
1692 }
1693 }
1694 edge.UpdateTo(use_node);
1695 }
1696 return block;
1697 }
1698
1699 BasicBlock* GetHoistBlock(BasicBlock* block) {
1700 if (!scheduler_->special_rpo_->HasLoopBlocks()) return nullptr;
1701 if (block->IsLoopHeader()) return block->dominator();
1702 // We have to check to make sure that the {block} dominates all
1703 // of the outgoing blocks. If it doesn't, then there is a path
1704 // out of the loop which does not execute this {block}, so we
1705 // can't hoist operations from this {block} out of the loop, as
1706 // that would introduce additional computations.
1707 if (BasicBlock* header_block = block->loop_header()) {
1708 for (BasicBlock* outgoing_block :
1709 scheduler_->special_rpo_->GetOutgoingBlocks(header_block)) {
1710 if (scheduler_->GetCommonDominator(block, outgoing_block) != block) {
1711 return nullptr;
1712 }
1713 }
1714 return header_block->dominator();
1715 }
1716 return nullptr;
1717 }
1718
1719 BasicBlock* GetCommonDominatorOfUses(Node* node) {
1720 BasicBlock* block = nullptr;
1721 for (Edge edge : node->use_edges()) {
1722 if (!scheduler_->IsLive(edge.from())) continue;
1723 BasicBlock* use_block = GetBlockForUse(edge);
1724 block = block == nullptr
1725 ? use_block
1726 : use_block == nullptr
1727 ? block
1728 : scheduler_->GetCommonDominator(block, use_block);
1729 }
1730 return block;
1731 }
1732
1733 BasicBlock* FindPredecessorBlock(Node* node) {
1734 return scheduler_->control_flow_builder_->FindPredecessorBlock(node);
1735 }
1736
1737 BasicBlock* GetBlockForUse(Edge edge) {
1738 // TODO(titzer): ignore uses from dead nodes (not visited in PrepareUses()).
1739 // Dead uses only occur if the graph is not trimmed before scheduling.
1740 Node* use = edge.from();
1741 if (IrOpcode::IsPhiOpcode(use->opcode())) {
1742 // If the use is from a coupled (i.e. floating) phi, compute the common
1743 // dominator of its uses. This will not recurse more than one level.
1744 if (scheduler_->GetPlacement(use) == Scheduler::kCoupled) {
1745 TRACE(" inspecting uses of coupled #%d:%s\n", use->id(),
1746 use->op()->mnemonic());
1747 // TODO(titzer): reenable once above TODO is addressed.
1748 // DCHECK_EQ(edge.to(), NodeProperties::GetControlInput(use));
1749 return GetCommonDominatorOfUses(use);
1750 }
1751 // If the use is from a fixed (i.e. non-floating) phi, we use the
1752 // predecessor block of the corresponding control input to the merge.
1753 if (scheduler_->GetPlacement(use) == Scheduler::kFixed) {
1754 TRACE(" input@%d into a fixed phi #%d:%s\n", edge.index(), use->id(),
1755 use->op()->mnemonic());
1756 Node* merge = NodeProperties::GetControlInput(use, 0);
1757 DCHECK(IrOpcode::IsMergeOpcode(merge->opcode()));
1758 Node* input = NodeProperties::GetControlInput(merge, edge.index());
1759 return FindPredecessorBlock(input);
1760 }
1761 } else if (IrOpcode::IsMergeOpcode(use->opcode())) {
1762 // If the use is from a fixed (i.e. non-floating) merge, we use the
1763 // predecessor block of the current input to the merge.
1764 if (scheduler_->GetPlacement(use) == Scheduler::kFixed) {
1765 TRACE(" input@%d into a fixed merge #%d:%s\n", edge.index(), use->id(),
1766 use->op()->mnemonic());
1767 return FindPredecessorBlock(edge.to());
1768 }
1769 }
1770 BasicBlock* result = schedule_->block(use);
1771 if (result == nullptr) return nullptr;
1772 TRACE(" must dominate use #%d:%s in id:%d\n", use->id(),
1773 use->op()->mnemonic(), result->id().ToInt());
1774 return result;
1775 }

1777 void ScheduleFloatingControl(BasicBlock* block, Node* node) {
1778 scheduler_->FuseFloatingControl(block, node);
1779 }
1780
1781 void ScheduleRegion(BasicBlock* block, Node* region_end) {
1782 // We only allow regions of instructions connected into a linear
1783 // effect chain. The only value allowed to be produced by a node
1784 // in the chain must be the value consumed by the FinishRegion node.
1785
1786 // We schedule back to front; we first schedule FinishRegion.
1787 CHECK_EQ(IrOpcode::kFinishRegion, region_end->opcode());
1788 ScheduleNode(block, region_end);
1789
1790 // Schedule the chain.
1791 Node* node = NodeProperties::GetEffectInput(region_end);
1792 while (node->opcode() != IrOpcode::kBeginRegion) {
1793 DCHECK_EQ(0, scheduler_->GetData(node)->unscheduled_count_);
1794 DCHECK_EQ(1, node->op()->EffectInputCount());
1795 DCHECK_EQ(1, node->op()->EffectOutputCount());
1796 DCHECK_EQ(0, node->op()->ControlOutputCount());
1797 // The value output (if there is any) must be consumed
1798 // by the EndRegion node.
1799 DCHECK(node->op()->ValueOutputCount() == 0 ||
1800 node == region_end->InputAt(0));
1801 ScheduleNode(block, node);
1802 node = NodeProperties::GetEffectInput(node);
1803 }
1804 // Schedule the BeginRegion node.
1805 DCHECK_EQ(0, scheduler_->GetData(node)->unscheduled_count_);
1806 ScheduleNode(block, node);
1807 }
1808
1809 void ScheduleNode(BasicBlock* block, Node* node) {
1810 schedule_->PlanNode(block, node);
1811 size_t block_id = block->id().ToSize();
1812 if (!scheduler_->scheduled_nodes_[block_id]) {
1813 scheduler_->scheduled_nodes_[block_id] = zone_->New<NodeVector>(zone_);
1814 }
1815 scheduler_->scheduled_nodes_[block_id]->push_back(node);
1816 scheduler_->UpdatePlacement(node, Scheduler::kScheduled);
1817 }
1818
1819 Node* CloneNode(Node* node) {
1820 int const input_count = node->InputCount();
1821 std::optional<int> coupled_control_edge =
1822 scheduler_->GetCoupledControlEdge(node);
1823 for (int index = 0; index < input_count; ++index) {
1824 if (index != coupled_control_edge) {
1825 Node* const input = node->InputAt(index);
1826 scheduler_->IncrementUnscheduledUseCount(input, node);
1827 }
1828 }
1829 Node* const copy = scheduler_->graph_->CloneNode(node);
1830 TRACE(("clone #%d:%s -> #%d\n"), node->id(), node->op()->mnemonic(),
1831 copy->id());
1832 scheduler_->node_data_.resize(copy->id() + 1,
1833 scheduler_->DefaultSchedulerData());
1834 scheduler_->node_data_[copy->id()] = scheduler_->node_data_[node->id()];
1835 return copy;
1836 }

1899void Scheduler::FuseFloatingControl(BasicBlock* block, Node* node) {
1900 TRACE("--- FUSE FLOATING CONTROL ----------------------------------\n");
1901 if (v8_flags.trace_turbo_scheduler) {
1902 StdoutStream{} << "Schedule before control flow fusion:\n" << *schedule_;
1903 }
1904
1905 // Iterate on phase 1: Build control-flow graph.
1906 control_flow_builder_->Run(block, node);
1907
1908 // Iterate on phase 2: Compute special RPO and dominator tree.
1909 special_rpo_->UpdateSpecialRPO(block, schedule_->block(node));
1910 // TODO(turbofan): Currently "iterate on" means "re-run". Fix that.
1911 for (BasicBlock* b = block->rpo_next(); b != nullptr; b = b->rpo_next()) {
1912 b->set_dominator_depth(-1);
1913 b->set_dominator(nullptr);

1920 // soon as the approach settled for all floating loops.

1938

1940 // TODO(turbofan): Improve that by supporting bulk moves.
1941 scheduled_nodes_.resize(schedule_->BasicBlockCount());
1942 MovePlannedNodes(block, schedule_->block(node));
1943
1944 if (v8_flags.trace_turbo_scheduler) {
