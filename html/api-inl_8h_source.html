56template <class From, class To>
57inline Local<To> Utils::Convert(v8::internal::DirectHandle<From> obj) {
58 DCHECK(obj.is_null() || IsSmi(*obj) || !IsTheHole(*obj));
59#ifdef V8_ENABLE_DIRECT_HANDLE
60 if (obj.is_null()) return Local<To>();
61 return Local<To>::FromAddress(obj.address());
62#else
63 // This simply uses the location of the indirect handle wrapped inside a
64 // "fake" direct handle.
65 return Local<To>::FromSlot(indirect_handle(obj).location());
66#endif
67}

166template <bool do_callback>
167class V8_NODISCARD CallDepthScope {
168 public:
169 CallDepthScope(i::Isolate* isolate, Local<Context> context)
170 : isolate_(isolate), saved_context_(isolate->context(), isolate_) {
171 isolate_->thread_local_top()->IncrementCallDepth<do_callback>(this);
172 i::Tagged<i::NativeContext> env = *Utils::OpenDirectHandle(*context);
173 isolate->set_context(env);
174
175 if (do_callback) isolate_->FireBeforeCallEnteredCallback();
176 }
177 ~CallDepthScope() {
178 i::MicrotaskQueue* microtask_queue =
179 i::Cast<i::NativeContext>(isolate_->context())
180 ->microtask_queue(isolate_);
181
182 isolate_->thread_local_top()->DecrementCallDepth(this);
183 // Clear the exception when exiting V8 to avoid memory leaks.
184 // Also clear termination exceptions iff there's no TryCatch handler.
185 // TODO(verwaest): Drop this once we propagate exceptions to external
186 // TryCatch on Throw. This should be debug-only.
187 if (isolate_->thread_local_top()->CallDepthIsZero() &&
188 (isolate_->thread_local_top()->try_catch_handler_ == nullptr ||
189 !isolate_->is_execution_terminating())) {
190 isolate_->clear_internal_exception();
191 }
192 if (do_callback) isolate_->FireCallCompletedCallback(microtask_queue);
193#ifdef DEBUG
194 if (do_callback) {
195 if (microtask_queue && microtask_queue->microtasks_policy() ==
196 v8::MicrotasksPolicy::kScoped) {
197 DCHECK(microtask_queue->GetMicrotasksScopeDepth() ||
198 !microtask_queue->DebugMicrotasksScopeDepthIsZero());
199 }
200 }
201 DCHECK(CheckKeptObjectsClearedAfterMicrotaskCheckpoint(microtask_queue));
202#endif
203
204 isolate_->set_context(*saved_context_);
205 }
206
207 CallDepthScope(const CallDepthScope&) = delete;
208 CallDepthScope& operator=(const CallDepthScope&) = delete;
209
210 private:
211#ifdef DEBUG
212 bool CheckKeptObjectsClearedAfterMicrotaskCheckpoint(
213 i::MicrotaskQueue* microtask_queue) {
214 bool did_perform_microtask_checkpoint =
215 isolate_->thread_local_top()->CallDepthIsZero() && do_callback &&
216 microtask_queue &&
217 microtask_queue->microtasks_policy() == MicrotasksPolicy::kAuto &&
218 !isolate_->is_execution_terminating();
219 return !did_perform_microtask_checkpoint ||
220 IsUndefined(isolate_->heap()->weak_refs_keep_during_job(), isolate_);
221 }
222#endif
223
224 i::Isolate* const isolate_;
225 i::Handle<i::Context> saved_context_;
226
227 i::Address previous_stack_height_;
228
229 friend class i::ThreadLocalTop;
230
231 DISALLOW_NEW_AND_DELETE()
232};

261template <typename T>
262void CopySmiElementsToTypedBuffer(T* dst, uint32_t length,
263 i::Tagged<i::FixedArray> elements) {
264 for (uint32_t i = 0; i < length; ++i) {
265 double value = i::Object::NumberValue(
266 i::Cast<i::Smi>(elements->get(static_cast<int>(i))));
267 // TODO(mslekova): Avoid converting back-and-forth when possible, e.g
268 // avoid int->double->int conversions to boost performance.
269 dst[i] = i::ConvertDouble<T>(value);
270 }
271}

273template <typename T>
274void CopyDoubleElementsToTypedBuffer(T* dst, uint32_t length,
275 i::Tagged<i::FixedDoubleArray> elements) {
276 for (uint32_t i = 0; i < length; ++i) {
277 double value = elements->get_scalar(static_cast<int>(i));
278 // TODO(mslekova): There are certain cases, e.g. double->double, in which
279 // we could do a memcpy directly.
280 dst[i] = i::ConvertDouble<T>(value);
281 }
282}

284template <CTypeInfo::Identifier type_info_id, typename T>
285bool CopyAndConvertArrayToCppBuffer(Local<Array> src, T* dst,
286 uint32_t max_length) {
287 static_assert(
288 std::is_same<T, typename i::CTypeInfoTraits<
289 CTypeInfo(type_info_id).GetType()>::ctype>::value,
290 "Type mismatch between the expected CTypeInfo::Type and the destination "
291 "array");
292
293 uint32_t length = src->Length();
294 if (length > max_length) {
295 return false;
296 }
297
298 i::DisallowGarbageCollection no_gc;
299 i::Tagged<i::JSArray> obj = *Utils::OpenDirectHandle(*src);
300 if (i::Object::IterationHasObservableEffects(obj)) {
301 // The array has a custom iterator.
302 return false;
303 }
304
305 i::Tagged<i::FixedArrayBase> elements = obj->elements();
306 switch (obj->GetElementsKind()) {
307 case i::PACKED_SMI_ELEMENTS:
308 CopySmiElementsToTypedBuffer(dst, length,
309 i::Cast<i::FixedArray>(elements));
310 return true;
311 case i::PACKED_DOUBLE_ELEMENTS:
312 CopyDoubleElementsToTypedBuffer(dst, length,
313 i::Cast<i::FixedDoubleArray>(elements));
314 return true;
315 default:
316 return false;
317 }
318}
