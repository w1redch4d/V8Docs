97void PropertyAccessBuilder::BuildCheckMaps(
98 Node* object, Effect* effect, Control control,
99 ZoneVector<MapRef> const& maps,
100 bool has_deprecated_map_without_migration_target) {
101 HeapObjectMatcher m(object);
102 if (m.HasResolvedValue()) {
103 MapRef object_map = m.Ref(broker()).map(broker());
104 if (object_map.is_stable()) {
105 for (MapRef map : maps) {
106 if (map.equals(object_map)) {
107 dependencies()->DependOnStableMap(object_map);
108 return;
109 }
110 }
111 }
112 }
113 ZoneRefSet<Map> map_set;
114 bool has_migration_target = false;
115 for (MapRef map : maps) {
116 map_set.insert(map, graph()->zone());
117 if (map.is_migration_target()) {
118 has_migration_target = true;
119 }
120 }
121 CheckMapsFlags flags = CheckMapsFlag::kNone;
122 if (has_migration_target) {
123 flags = CheckMapsFlag::kTryMigrateInstance;
124 } else if (has_deprecated_map_without_migration_target) {
125 flags = CheckMapsFlag::kTryMigrateInstanceAndDeopt;
126 }
127 *effect = graph()->NewNode(simplified()->CheckMaps(flags, map_set), object,
128 *effect, control);
129}

131Node* PropertyAccessBuilder::BuildCheckValue(Node* receiver, Effect* effect,
132 Control control, ObjectRef value) {
133 if (value.IsHeapObject()) {
134 HeapObjectMatcher m(receiver);
135 if (m.Is(value.AsHeapObject().object())) return receiver;
136 }
137 Node* expected = jsgraph()->ConstantNoHole(value, broker());
138 Node* check =
139 graph()->NewNode(simplified()->ReferenceEqual(), receiver, expected);
140 *effect =
141 graph()->NewNode(simplified()->CheckIf(DeoptimizeReason::kWrongValue),
142 check, *effect, control);
143 return expected;
144}

196std::optional<Node*> PropertyAccessBuilder::FoldLoadDictPrototypeConstant(
197 PropertyAccessInfo const& access_info) {
198 DCHECK(V8_DICT_PROPERTY_CONST_TRACKING_BOOL);
199 DCHECK(access_info.IsDictionaryProtoDataConstant());
200
201 InternalIndex index = access_info.dictionary_index();
202 OptionalObjectRef value = access_info.holder()->GetOwnDictionaryProperty(
203 broker(), index, dependencies());
204 if (!value) return {};
205
206 for (MapRef map : access_info.lookup_start_object_maps()) {
207 DirectHandle<Map> map_handle = map.object();
208 // Non-JSReceivers that passed AccessInfoFactory::ComputePropertyAccessInfo
209 // must have different lookup start map.
210 if (!IsJSReceiverMap(*map_handle)) {
211 // Perform the implicit ToObject for primitives here.
212 // Implemented according to ES6 section 7.3.2 GetV (V, P).
213 Tagged<JSFunction> constructor =
214 Map::GetConstructorFunction(
215 *map_handle, *broker()->target_native_context().object())
216 .value();
217 // {constructor.initial_map()} is loaded/stored with acquire-release
218 // semantics for constructors.
219 map = MakeRefAssumeMemoryFence(broker(), constructor->initial_map());
220 DCHECK(IsJSObjectMap(*map.object()));
221 }
222 dependencies()->DependOnConstantInDictionaryPrototypeChain(
223 map, access_info.name(), value.value(), PropertyKind::kData);
224 }
225
226 return jsgraph()->ConstantNoHole(value.value(), broker());
227}

229Node* PropertyAccessBuilder::TryFoldLoadConstantDataField(
230 NameRef name, PropertyAccessInfo const& access_info,
231 Node* lookup_start_object) {
232 if (!access_info.IsFastDataConstant()) return nullptr;
233
234 // First, determine if we have a constant holder to load from.
235 OptionalJSObjectRef holder = access_info.holder();
236
237 // If {access_info} has a holder, just use it.
238 if (!holder.has_value()) {
239 // Otherwise, try to match the {lookup_start_object} as a constant.
240 if (lookup_start_object->opcode() == IrOpcode::kCheckString ||
241 lookup_start_object->opcode() ==
242 IrOpcode::kCheckStringOrStringWrapper) {
243 // Bypassing Check inputs in order to allow constant folding.
244 lookup_start_object = lookup_start_object->InputAt(0);
245 }
246 HeapObjectMatcher m(lookup_start_object);
247 if (!m.HasResolvedValue() || !m.Ref(broker()).IsJSObject()) return nullptr;
248
249 // Let us make sure the actual map of the constant lookup_start_object is
250 // among the maps in {access_info}.
251 MapRef lookup_start_object_map = m.Ref(broker()).map(broker());
252 if (std::find_if(access_info.lookup_start_object_maps().begin(),
253 access_info.lookup_start_object_maps().end(),
254 [&](MapRef map) {
255 return map.equals(lookup_start_object_map);
256 }) == access_info.lookup_start_object_maps().end()) {
257 // The map of the lookup_start_object is not in the feedback, let us bail
258 // out.
259 return nullptr;
260 }
261 holder = m.Ref(broker()).AsJSObject();
262 }
263
264 if (access_info.field_representation().IsDouble()) {
265 std::optional<Float64> value = holder->GetOwnFastConstantDoubleProperty(
266 broker(), access_info.field_index(), dependencies());
267 return value.has_value() ? jsgraph()->ConstantNoHole(value->get_scalar())
268 : nullptr;
269 }
270 OptionalObjectRef value = holder->GetOwnFastConstantDataProperty(
271 broker(), access_info.field_representation(), access_info.field_index(),
272 dependencies());
273 return value.has_value() ? jsgraph()->ConstantNoHole(*value, broker())
274 : nullptr;
275}

277Node* PropertyAccessBuilder::BuildLoadDataField(NameRef name, Node* holder,
278 FieldAccess&& field_access,
279 bool is_inobject, Node** effect,
280 Node** control) {
281 Node* storage = holder;
282 if (!is_inobject) {
283 storage = *effect = graph()->NewNode(
284 simplified()->LoadField(
285 AccessBuilder::ForJSObjectPropertiesOrHashKnownPointer()),
286 storage, *effect, *control);
287 }
288 if (field_access.machine_type.representation() ==
289 MachineRepresentation::kFloat64) {
290 if (dependencies() == nullptr) {
291 FieldAccess const storage_access = {kTaggedBase,
292 field_access.offset,
293 name.object(),
294 OptionalMapRef(),
295 Type::Any(),
296 MachineType::AnyTagged(),
297 kPointerWriteBarrier,
298 "BuildLoadDataField",
299 field_access.const_field_info};
300 storage = *effect = graph()->NewNode(
301 simplified()->LoadField(storage_access), storage, *effect, *control);
302 // We expect the loaded value to be a heap number here. With
303 // in-place field representation changes it is possible this is a
304 // no longer a heap number without map transitions. If we haven't taken
305 // a dependency on field representation, we should verify the loaded
306 // value is a heap number.
307 storage = *effect = graph()->NewNode(simplified()->CheckHeapObject(),
308 storage, *effect, *control);
309 Node* map = *effect =
310 graph()->NewNode(simplified()->LoadField(AccessBuilder::ForMap()),
311 storage, *effect, *control);
312 Node* is_heap_number =
313 graph()->NewNode(simplified()->ReferenceEqual(), map,
314 jsgraph()->HeapNumberMapConstant());
315 *effect = graph()->NewNode(
316 simplified()->CheckIf(DeoptimizeReason::kNotAHeapNumber),
317 is_heap_number, *effect, *control);
318 } else {
319 FieldAccess const storage_access = {kTaggedBase,
320 field_access.offset,
321 name.object(),
322 OptionalMapRef(),
323 Type::OtherInternal(),
324 MachineType::TaggedPointer(),
325 kPointerWriteBarrier,
326 "BuildLoadDataField",
327 field_access.const_field_info};
328 storage = *effect = graph()->NewNode(
329 simplified()->LoadField(storage_access), storage, *effect, *control);
330 }
331 FieldAccess value_field_access = AccessBuilder::ForHeapNumberValue();
332 value_field_access.const_field_info = field_access.const_field_info;
333 field_access = value_field_access;
334 }
335 Node* value = *effect = graph()->NewNode(
336 simplified()->LoadField(field_access), storage, *effect, *control);
337 return value;
338}

340Node* PropertyAccessBuilder::BuildLoadDataField(
341 NameRef name, PropertyAccessInfo const& access_info,
342 Node* lookup_start_object, Node** effect, Node** control) {
343 DCHECK(access_info.IsDataField() || access_info.IsFastDataConstant());
344
345 if (Node* value = TryFoldLoadConstantDataField(name, access_info,
346 lookup_start_object)) {
347 return value;
348 }
349
350 MachineRepresentation const field_representation =
351 ConvertRepresentation(access_info.field_representation());
352 Node* storage = ResolveHolder(access_info, lookup_start_object);
353
354 FieldAccess field_access = {
355 kTaggedBase,
356 access_info.field_index().offset(),
357 name.object(),
358 OptionalMapRef(),
359 access_info.field_type(),
360 MachineType::TypeForRepresentation(field_representation),
361 kFullWriteBarrier,
362 "BuildLoadDataField",
363 access_info.GetConstFieldInfo()};
364 if (field_representation == MachineRepresentation::kTaggedPointer ||
365 field_representation == MachineRepresentation::kCompressedPointer) {
366 // Remember the map of the field value, if its map is stable. This is
367 // used by the LoadElimination to eliminate map checks on the result.
368 OptionalMapRef field_map = access_info.field_map();
369 if (field_map.has_value()) {
370 if (field_map->is_stable()) {
371 dependencies()->DependOnStableMap(field_map.value());
372 field_access.map = field_map;
373 }
374 }
375 }
376 return BuildLoadDataField(name, storage, std::move(field_access),
377 access_info.field_index().is_inobject(), effect,
378 control);
379}
