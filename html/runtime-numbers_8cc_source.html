21RUNTIME_FUNCTION(Runtime_StringParseInt) {
22 HandleScope handle_scope(isolate);
23 DCHECK_EQ(2, args.length());
24 Handle<Object> string = args.at(0);
25 Handle<Object> radix = args.at(1);
26
27 // Convert {string} to a String first, and flatten it.
28 Handle<String> subject;
29 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, subject,
30 Object::ToString(isolate, string));
31 subject = String::Flatten(isolate, subject);
32
33 // Convert {radix} to Int32.
34 if (!IsNumber(*radix)) {
35 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, radix,
36 Object::ToNumber(isolate, radix));
37 }
38 int radix32 = DoubleToInt32(Object::NumberValue(*radix));
39 if (radix32 != 0 && (radix32 < 2 || radix32 > 36)) {
40 return ReadOnlyRoots(isolate).nan_value();
41 }
42
43 double result = StringToInt(isolate, subject, radix32);
44 return *isolate->factory()->NewNumber(result);
45}
