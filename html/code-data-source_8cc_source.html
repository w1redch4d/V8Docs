33InternedV8JsScript::Type GetJsScriptType(Tagged<Script> script) {
34 if (script->compilation_type() == Script::CompilationType::kEval) {
35 return InternedV8JsScript::TYPE_EVAL;
36 }
37
38 // TODO(carlscab): Camillo to extend the Script::Type enum. compilation_type
39 // will no longer be needed.
40
41 switch (script->type()) {
42 case Script::Type::kNative:
43 return InternedV8JsScript::TYPE_NATIVE;
44 case Script::Type::kExtension:
45 return InternedV8JsScript::TYPE_EXTENSION;
46 case Script::Type::kNormal:
47 return InternedV8JsScript::TYPE_NORMAL;
48#if V8_ENABLE_WEBASSEMBLY
49 case Script::Type::kWasm:
50 UNREACHABLE();
51#endif // V8_ENABLE_WEBASSEMBLY
52 case Script::Type::kInspector:
53 return InternedV8JsScript::TYPE_INSPECTOR;
54 }
55}

138uint64_t CodeDataSourceIncrementalState::InternIsolate(Isolate& isolate) {
139 auto [it, was_inserted] = isolates_.emplace(isolate.id(), next_isolate_iid());
140 uint64_t iid = it->second;
141 if (!was_inserted) {
142 return iid;
143 }
144
145 auto* isolate_proto = serialized_interned_data_->add_v8_isolate();
146 isolate_proto->set_iid(iid);
147 isolate_proto->set_isolate_id(isolate.id());
148 isolate_proto->set_pid(base::OS::GetCurrentProcessId());
149 isolate_proto->set_embedded_blob_code_start_address(
150 reinterpret_cast<uint64_t>(isolate.embedded_blob_code()));
151 isolate_proto->set_embedded_blob_code_size(isolate.embedded_blob_code_size());
152 if (auto* code_range = isolate.heap()->code_range(); code_range != nullptr) {
153 auto* v8_code_range = isolate_proto->set_code_range();
154 v8_code_range->set_base_address(code_range->base());
155 v8_code_range->set_size(code_range->size());
156 if (code_range == IsolateGroup::current()->GetCodeRange()) {
157 // FIXME(42204573): Belongs to isolate group, not process.
158 v8_code_range->set_is_process_wide(true);
159 }
160 if (auto* embedded_builtins_start = code_range->embedded_blob_code_copy();
161 embedded_builtins_start != nullptr) {
162 v8_code_range->set_embedded_blob_code_copy_start_address(
163 reinterpret_cast<uint64_t>(embedded_builtins_start));
164 }
165 }
166
167 return iid;
168}

170uint64_t CodeDataSourceIncrementalState::InternJsScript(Isolate& isolate,
171 Tagged<Script> script) {
172 auto [it, was_inserted] = scripts_.emplace(
173 CodeDataSourceIncrementalState::ScriptUniqueId{isolate.id(),
174 script->id()},
175 next_script_iid());
176 uint64_t iid = it->second;
177 if (!was_inserted) {
178 return iid;
179 }
180
181 auto* proto = serialized_interned_data_->add_v8_js_script();
182 proto->set_iid(iid);
183 proto->set_script_id(script->id());
184 proto->set_type(GetJsScriptType(script));
185 if (IsString(script->name())) {
186 PerfettoV8String(Cast<String>(script->name()))
187 .WriteToProto(*proto->set_name());
188 }
189 if (log_script_sources() && IsString(script->source())) {
190 PerfettoV8String(Cast<String>(script->source()))
191 .WriteToProto(*proto->set_source());
192 }
193
194 return iid;
195}

197uint64_t CodeDataSourceIncrementalState::InternJsFunction(
198 Isolate& isolate, Handle<SharedFunctionInfo> info,
199 uint64_t v8_js_script_iid, int line_num, int column_num) {
200 Handle<String> function_name = SharedFunctionInfo::DebugName(&isolate, info);
201 uint64_t v8_js_function_name_iid = InternJsFunctionName(*function_name);
202
203 auto [it, was_inserted] = functions_.emplace(
204 CodeDataSourceIncrementalState::Function{
205 v8_js_script_iid, info->is_toplevel(), info->StartPosition()},
206 next_function_iid());
207 const uint64_t iid = it->second;
208 if (!was_inserted) {
209 return iid;
210 }
211
212 auto* function_proto = serialized_interned_data_->add_v8_js_function();
213 function_proto->set_iid(iid);
214 function_proto->set_v8_js_function_name_iid(v8_js_function_name_iid);
215 function_proto->set_v8_js_script_iid(v8_js_script_iid);
216 function_proto->set_kind(GetJsFunctionKind(info->kind()));
217 int32_t start_position = info->StartPosition();
218 if (start_position >= 0) {
219 function_proto->set_byte_offset(static_cast<uint32_t>(start_position));
220 }
221
222 return iid;
223}
