322// Helper function that can be used to ensure that certain objects are not
323// located inside the sandbox. Typically used for trusted objects.
324// Will always return false when the sandbox is disabled or partially reserved.
325V8_INLINE bool InsideSandbox(uintptr_t address) {
326#ifdef V8_ENABLE_SANDBOX
327 Sandbox* sandbox = Sandbox::current();
328 // Use ReservationContains (instead of just Contains) to correctly handle the
329 // case of partially-reserved sandboxes.
330 return sandbox->ReservationContains(address);
331#else
332 return false;
333#endif
334}
