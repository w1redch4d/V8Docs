19// static
20void PretenuringHandler::UpdateAllocationSite(
21 Heap* heap, Tagged<Map> map, Tagged<HeapObject> object, int object_size,
22 PretenuringFeedbackMap* pretenuring_feedback) {
23 DCHECK_NE(pretenuring_feedback,
24 &heap->pretenuring_handler()->global_pretenuring_feedback_);
25#ifdef DEBUG
26 MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
27 // MemoryChunk::IsToPage() is not available with sticky mark-bits.
28 DCHECK_IMPLIES(v8_flags.sticky_mark_bits || chunk->IsToPage(),
29 v8_flags.minor_ms);
30 DCHECK_IMPLIES(!v8_flags.minor_ms && !HeapLayout::InYoungGeneration(object),
31 chunk->IsFlagSet(MemoryChunk::PAGE_NEW_OLD_PROMOTION));
32#endif
33 if (V8_UNLIKELY(!v8_flags.allocation_site_pretenuring) ||
34 !AllocationSite::CanTrack(map->instance_type())) {
35 return;
36 }
37 Tagged<AllocationMemento> memento_candidate =
38 FindAllocationMemento<kForGC>(heap, map, object, object_size);
39 if (memento_candidate.is_null()) {
40 return;
41 }
42 DCHECK(IsJSObjectMap(map));
43
44 // Entering cached feedback is used in the parallel case. We are not allowed
45 // to dereference the allocation site and rather have to postpone all checks
46 // till actually merging the data.
47 Address key = memento_candidate->GetAllocationSiteUnchecked();
48 (*pretenuring_feedback)[UncheckedCast<AllocationSite>(Tagged<Object>(key))]++;
49}

59// static
60template <PretenuringHandler::FindMementoMode mode>
61Tagged<AllocationMemento> PretenuringHandler::FindAllocationMemento(
62 Heap* heap, Tagged<Map> map, Tagged<HeapObject> object, int object_size) {
63 // For uses from within the GC, the size here may actually change when e.g.
64 // updating mementos during marking in the young generation collector. This is
65 // not an issue with Scavenger that stops the mutator.
66 DCHECK_IMPLIES(mode != FindMementoMode::kForGC || !v8_flags.minor_ms,
67 object_size == object->SizeFromMap(map));
68 // For configurations where object size changes, we can check that it only
69 // shinks in case the sizes are not matching.
70 DCHECK_IMPLIES(mode == FindMementoMode::kForGC && v8_flags.minor_ms,
71 object_size >= object->SizeFromMap(map));
72 Address object_address = object.address();
73 Address memento_address =
74 object_address + ALIGN_TO_ALLOCATION_ALIGNMENT(object_size);
75 Address last_memento_word_address = memento_address + kTaggedSize;
76 // If the memento would be on another page, bail out immediately.
77 if (!PageMetadata::OnSamePage(object_address, last_memento_word_address)) {
78 return AllocationMemento();
79 }
80
81 // If the page is being swept, treat it as if the memento was already swept
82 // and bail out.
83 if constexpr (mode != FindMementoMode::kForGC) {
84 MemoryChunk* object_chunk = MemoryChunk::FromAddress(object_address);
85 PageMetadata* object_page = PageMetadata::cast(object_chunk->Metadata());
86 if (!object_page->SweepingDone()) {
87 return AllocationMemento();
88 }
89 }
90
91 Tagged<HeapObject> candidate = HeapObject::FromAddress(memento_address);
92 ObjectSlot candidate_map_slot = candidate->map_slot();
93 // This fast check may peek at an uninitialized word. However, the slow check
94 // below (memento_address == top) ensures that this is safe. Mark the word as
95 // initialized to silence MemorySanitizer warnings.
96 MSAN_MEMORY_IS_INITIALIZED(candidate_map_slot.address(), kTaggedSize);
97 if (!candidate_map_slot.Relaxed_ContainsMapValue(
98 ReadOnlyRoots(heap).allocation_memento_map().ptr())) {
99 return AllocationMemento();
100 }
101
102 // Bail out if the memento is below the age mark, which can happen when
103 // mementos survived because a page got moved within new space.
104 MemoryChunk* object_chunk = MemoryChunk::FromAddress(object_address);
105 if (object_chunk->IsFlagSet(MemoryChunk::NEW_SPACE_BELOW_AGE_MARK)) {
106 PageMetadata* object_page = PageMetadata::cast(object_chunk->Metadata());
107 Address age_mark =
108 reinterpret_cast<SemiSpace*>(object_page->owner())->age_mark();
109 if (!object_page->Contains(age_mark)) {
110 return AllocationMemento();
111 }
112 // Do an exact check in the case where the age mark is on the same page.
113 if (object_address < age_mark) {
114 return AllocationMemento();
115 }
116 }
117
118 Tagged<AllocationMemento> memento_candidate =
119 Cast<AllocationMemento>(candidate);
120
121 // Depending on what the memento is used for, we might need to perform
122 // additional checks.
123 Address top;
124 switch (mode) {
125 case kForGC:
126 return memento_candidate;
127 case kForRuntime:
128 if (memento_candidate.is_null()) return AllocationMemento();
129 // Either the object is the last object in the new space, or there is
130 // another object of at least word size (the header map word) following
131 // it, so suffices to compare ptr and top here.
132 top = heap->NewSpaceTop();
133 DCHECK(memento_address >= heap->NewSpaceLimit() ||
134 memento_address +
135 ALIGN_TO_ALLOCATION_ALIGNMENT(AllocationMemento::kSize) <=
136 top);
137 if ((memento_address != top) && memento_candidate->IsValid()) {
138 return memento_candidate;
139 }
140 return AllocationMemento();
141 default:
142 UNREACHABLE();
143 }
144 UNREACHABLE();
145}
