31// base::make_array: Create an array of fixed length, initialized by a function.
32// The content of the array is created by calling the function with 0 .. Size-1.
33// Example usage to create the array {0, 2, 4}:
34// std::array<int, 3> arr = base::make_array<3>(
35// [](std::size_t i) { return static_cast<int>(2 * i); });
36// The resulting array will be constexpr if the passed function is constexpr.
37template <std::size_t Size, class Function>
38constexpr auto make_array(Function f) {
39 return detail::make_array_helper(f, std::make_index_sequence<Size>{});
40}

42// Helper to determine how to pass values: Pass scalars and arrays by value,
43// others by const reference (even if it was a non-const ref before; this is
44// disallowed by the style guide anyway).
45// The default is to also remove array extends (int[5] -> int*), but this can be
46// disabled by setting {remove_array_extend} to false.
47template <typename T, bool remove_array_extend = true>
48struct pass_value_or_ref {
49 using noref_t = typename std::remove_reference<T>::type;
50 using decay_t = typename std::conditional<
51 std::is_array<noref_t>::value && !remove_array_extend, noref_t,
52 typename std::decay<noref_t>::type>::type;
53 using type = typename std::conditional<std::is_scalar<decay_t>::value ||
54 std::is_array<decay_t>::value,
55 decay_t, const decay_t&>::type;
56};

111template <size_t I, typename T, typename Tuple, typename Function>
112constexpr auto tuple_fold_impl(T&& initial, Tuple&& tpl, Function&& function) {
113 if constexpr (I == 0) {
114 return function(std::forward<T>(initial), std::get<0>(tpl));
115 } else {
116 return function(tuple_fold_impl<I - 1>(std::forward<T>(initial),
117 std::forward<Tuple>(tpl), function),
118 std::get<I>(tpl));

124// Get the first N elements from a tuple.
125template <size_t N, typename Tuple>
126constexpr auto tuple_head(Tuple&& tpl) {
127 constexpr size_t total_size = std::tuple_size_v<std::decay_t<Tuple>>;
128 static_assert(N <= total_size);
129 return detail::tuple_slice_impl<0>(std::forward<Tuple>(tpl),
130 std::make_index_sequence<N>());
131}

133// Drop the first N elements from a tuple.
134template <size_t N, typename Tuple>
135// If the user accidentally passes in an N that is larger than the tuple
136// size, the unsigned subtraction will create a giant index sequence and
137// crash the compiler. To avoid this and fail early, disable this function
138// for invalid N.
139 requires(detail::NIsNotGreaterThanTupleSize<N, Tuple>)
140constexpr auto tuple_drop(Tuple&& tpl) {
141 constexpr size_t total_size = std::tuple_size_v<std::decay_t<Tuple>>;
142 static_assert(N <= total_size);
143 return detail::tuple_slice_impl<N>(
144 std::forward<Tuple>(tpl), std::make_index_sequence<total_size - N>());
145}

174// Calls `function(v, u)` for pairs `v<I>, u<I>` in the
175// tuples and returns a new tuple with all the results.

178 constexpr size_t S = std::tuple_size_v<std::decay_t<TupleV>>;

218// Find SearchT in Ts. SearchT must be present at most once in Ts, and returns
219// sizeof...(Ts) if not found.
220template <typename SearchT, typename... Ts>
221struct index_of_type;
222
223template <typename SearchT, typename... Ts>
224constexpr size_t index_of_type_v = index_of_type<SearchT, Ts...>::value;
225template <typename SearchT, typename... Ts>
226constexpr bool has_type_v =
227 index_of_type<SearchT, Ts...>::value < sizeof...(Ts);
228
229// Not found / empty list.
230template <typename SearchT>
231struct index_of_type<SearchT> : public std::integral_constant<size_t, 0> {};
232
233// SearchT found at head of list.
234template <typename SearchT, typename... Ts>
235struct index_of_type<SearchT, SearchT, Ts...>
236 : public std::integral_constant<size_t, 0> {
237 // SearchT is not allowed to be anywhere else in the list.
238 static_assert(!has_type_v<SearchT, Ts...>);
239};
240
241// Recursion, SearchT not found at head of list.
242template <typename SearchT, typename T, typename... Ts>
243struct index_of_type<SearchT, T, Ts...>
244 : public std::integral_constant<size_t,
245 1 + index_of_type<SearchT, Ts...>::value> {
246};
