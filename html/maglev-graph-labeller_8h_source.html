55 void PrintNodeLabel(std::ostream& os, const NodeBase* node) {
56 if (node != nullptr && node->Is<VirtualObject>()) {
57 // VirtualObjects are unregisted nodes, since they are not attached to
58 // the graph, but its inlined allocation is.
59 const VirtualObject* vo = node->Cast<VirtualObject>();
60 os << "VO{" << vo->id() << "}:";
61 node = vo->allocation();
62 }
63 auto node_id_it = nodes_.find(node);
64
65 if (node_id_it == nodes_.end()) {
66 os << "<unregistered node " << node << ">";
67 return;
68 }
69
70 if (node->has_id()) {
71 os << "v" << node->id() << "/";
72 }
73 os << "n" << node_id_it->second.label;
74 }
75
76 void PrintInput(std::ostream& os, const Input& input) {
77 PrintNodeLabel(os, input.node());
78 os << ":" << input.operand();
79 }
80
81 private:
82 std::map<const BasicBlock*, int> block_ids_;
83 std::map<const NodeBase*, NodeInfo> nodes_;
84 int next_block_label_ = 1;
85 int next_node_label_ = 1;
86};
