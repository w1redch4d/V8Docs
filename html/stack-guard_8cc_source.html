34void StackGuard::update_interrupt_requests_and_stack_limits(
35 const ExecutionAccess& lock) {
36 DCHECK_NOT_NULL(isolate_);
37 if (has_pending_interrupts(lock)) {
38 thread_local_.set_jslimit(kInterruptLimit);
39#ifdef USE_SIMULATOR
40 thread_local_.set_climit(kInterruptLimit);
41#endif
42 } else {
43 thread_local_.set_jslimit(thread_local_.real_jslimit_);
44#ifdef USE_SIMULATOR
45 thread_local_.set_climit(thread_local_.real_climit_);
46#endif
47 }
48 for (InterruptLevel level :
49 std::array{InterruptLevel::kNoGC, InterruptLevel::kNoHeapWrites,
50 InterruptLevel::kAnyEffect}) {
51 thread_local_.set_interrupt_requested(
52 level, InterruptLevelMask(level) & thread_local_.interrupt_flags_);
53 }
54}

78void StackGuard::SetStackLimitForStackSwitching(uintptr_t limit) {
79 // Try to compare and swap the new jslimit without the ExecutionAccess lock.
80 uintptr_t old_jslimit = base::Relaxed_CompareAndSwap(
81 &thread_local_.jslimit_, thread_local_.real_jslimit_, limit);
82 USE(old_jslimit);
83 DCHECK_IMPLIES(old_jslimit != thread_local_.real_jslimit_,
84 old_jslimit == kInterruptLimit);
85 // Either way, set the real limit. This does not require synchronization.
86 thread_local_.real_jslimit_ = limit;
87}

102void StackGuard::PushInterruptsScope(InterruptsScope* scope) {
103 ExecutionAccess access(isolate_);
104 DCHECK_NE(scope->mode_, InterruptsScope::kNoop);
105 if (scope->mode_ == InterruptsScope::kPostponeInterrupts) {
106 // Intercept already requested interrupts.
107 uint32_t intercepted =
108 thread_local_.interrupt_flags_ & scope->intercept_mask_;
109 scope->intercepted_flags_ = intercepted;
110 thread_local_.interrupt_flags_ &= ~intercepted;
111 } else {
112 DCHECK_EQ(scope->mode_, InterruptsScope::kRunInterrupts);
113 // Restore postponed interrupts.
114 uint32_t restored_flags = 0;
115 for (InterruptsScope* current = thread_local_.interrupt_scopes_;
116 current != nullptr; current = current->prev_) {
117 restored_flags |= (current->intercepted_flags_ & scope->intercept_mask_);
118 current->intercepted_flags_ &= ~scope->intercept_mask_;
119 }
120 thread_local_.interrupt_flags_ |= restored_flags;
121 }
122 update_interrupt_requests_and_stack_limits(access);
123 // Add scope to the chain.
124 scope->prev_ = thread_local_.interrupt_scopes_;
125 thread_local_.interrupt_scopes_ = scope;
126}
127
128void StackGuard::PopInterruptsScope() {
129 ExecutionAccess access(isolate_);
130 InterruptsScope* top = thread_local_.interrupt_scopes_;
131 DCHECK_NE(top->mode_, InterruptsScope::kNoop);
132 if (top->mode_ == InterruptsScope::kPostponeInterrupts) {
133 // Make intercepted interrupts active.
134 DCHECK_EQ(thread_local_.interrupt_flags_ & top->intercept_mask_, 0);
135 thread_local_.interrupt_flags_ |= top->intercepted_flags_;
136 } else {
137 DCHECK_EQ(top->mode_, InterruptsScope::kRunInterrupts);
138 // Postpone existing interupts if needed.
139 if (top->prev_) {
140 for (uint32_t interrupt = 1; interrupt < ALL_INTERRUPTS;
141 interrupt = interrupt << 1) {
142 InterruptFlag flag = static_cast<InterruptFlag>(interrupt);
143 if ((thread_local_.interrupt_flags_ & flag) &&
144 top->prev_->Intercept(flag)) {
145 thread_local_.interrupt_flags_ &= ~flag;
146 }
147 }
148 }
149 }
150 update_interrupt_requests_and_stack_limits(access);
151 // Remove scope from chain.
152 thread_local_.interrupt_scopes_ = top->prev_;
153}

160void StackGuard::RequestInterrupt(InterruptFlag flag) {
161 ExecutionAccess access(isolate_);
162 // Check the chain of InterruptsScope for interception.
163 if (thread_local_.interrupt_scopes_ &&
164 thread_local_.interrupt_scopes_->Intercept(flag)) {
165 return;
166 }
167
168 // Not intercepted. Set as active interrupt flag.
169 thread_local_.interrupt_flags_ |= flag;
170 update_interrupt_requests_and_stack_limits(access);
171
172 // If this isolate is waiting in a futex, notify it to wake up.
173 isolate_->futex_wait_list_node()->NotifyWake();
174}

176void StackGuard::ClearInterrupt(InterruptFlag flag) {
177 ExecutionAccess access(isolate_);
178 // Clear the interrupt flag from the chain of InterruptsScope.
179 for (InterruptsScope* current = thread_local_.interrupt_scopes_;
180 current != nullptr; current = current->prev_) {
181 current->intercepted_flags_ &= ~flag;
182 }
183
184 // Clear the interrupt flag from the active interrupt flags.
185 thread_local_.interrupt_flags_ &= ~flag;
186 update_interrupt_requests_and_stack_limits(access);
187}

189bool StackGuard::HasTerminationRequest() {
190 if (!thread_local_.has_interrupt_requested(InterruptLevel::kNoGC)) {
191 return false;
192 }
193 ExecutionAccess access(isolate_);
194 if ((thread_local_.interrupt_flags_ & TERMINATE_EXECUTION) != 0) {
195 thread_local_.interrupt_flags_ &= ~TERMINATE_EXECUTION;
196 update_interrupt_requests_and_stack_limits(access);
197 return true;
198 }
199 return false;
200}

202int StackGuard::FetchAndClearInterrupts(InterruptLevel level) {
203 ExecutionAccess access(isolate_);
204 InterruptFlag mask = InterruptLevelMask(level);
205 if ((thread_local_.interrupt_flags_ & TERMINATE_EXECUTION) != 0) {
206 // The TERMINATE_EXECUTION interrupt is special, since it terminates
207 // execution but should leave V8 in a resumable state. If it exists, we only
208 // fetch and clear that bit. On resume, V8 can continue processing other
209 // interrupts.
210 mask = TERMINATE_EXECUTION;
211 }
212
213 int result = static_cast<int>(thread_local_.interrupt_flags_ & mask);
214 thread_local_.interrupt_flags_ &= ~mask;
215 update_interrupt_requests_and_stack_limits(access);
216 return result;
217}

238void StackGuard::ThreadLocal::Initialize(Isolate* isolate,
239 const ExecutionAccess& lock) {
240 const uintptr_t kLimitSize = v8_flags.stack_size * KB;
241 DCHECK_GT(GetCurrentStackPosition(), kLimitSize);
242 uintptr_t limit = GetCurrentStackPosition() - kLimitSize;
243 real_jslimit_ = SimulatorStack::JsLimitFromCLimit(isolate, limit);
244 set_jslimit(SimulatorStack::JsLimitFromCLimit(isolate, limit));
245#ifdef USE_SIMULATOR
246 real_climit_ = limit;
247 set_climit(limit);
248#endif
249 interrupt_scopes_ = nullptr;
250 interrupt_flags_ = 0;
251}

253void StackGuard::InitThread(const ExecutionAccess& lock) {
254 thread_local_.Initialize(isolate_, lock);
255 Isolate::PerIsolateThreadData* per_thread =
256 isolate_->FindOrAllocatePerThreadDataForThisThread();
257 uintptr_t stored_limit = per_thread->stack_limit();
258 // You should hold the ExecutionAccess lock when you call this.
259 if (stored_limit != 0) {
260 SetStackLimit(stored_limit);
261 }

289Tagged<Object> StackGuard::HandleInterrupts(InterruptLevel level) {
290 TRACE_EVENT0("v8.execute", "V8.HandleInterrupts");
291
292#if DEBUG
293 isolate_->heap()->VerifyNewSpaceTop();
294#endif
295
296 if (v8_flags.verify_predictable) {
297 // Advance synthetic time by making a time request.
298 isolate_->heap()->MonotonicallyIncreasingTimeInMs();
299 }
300
301 // Fetch and clear interrupt bits in one go. See comments inside the method
302 // for special handling of TERMINATE_EXECUTION.
303 int interrupt_flags = FetchAndClearInterrupts(level);
304
305 // All interrupts should be fully processed when returning from this method.
306 ShouldBeZeroOnReturnScope should_be_zero_on_return(&interrupt_flags);
307
308 if (TestAndClear(&interrupt_flags, TERMINATE_EXECUTION)) {
309 TRACE_EVENT0("v8.execute", "V8.TerminateExecution");
310 return isolate_->TerminateExecution();
311 }
312
313 if (TestAndClear(&interrupt_flags, GC_REQUEST)) {
314 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.gc"), "V8.GCHandleGCRequest");
315 isolate_->heap()->HandleGCRequest();
316 }
317
318 if (TestAndClear(&interrupt_flags, START_INCREMENTAL_MARKING)) {
319 isolate_->heap()->StartIncrementalMarkingOnInterrupt();
320 }
321
322 if (TestAndClear(&interrupt_flags, GLOBAL_SAFEPOINT)) {
323 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.gc"), "V8.GlobalSafepoint");
324 isolate_->main_thread_local_heap()->Safepoint();
325 }
326
327#if V8_ENABLE_WEBASSEMBLY
328 if (TestAndClear(&interrupt_flags, GROW_SHARED_MEMORY)) {
329 TRACE_EVENT0("v8.wasm", "V8.WasmGrowSharedMemory");
330 BackingStore::UpdateSharedWasmMemoryObjects(isolate_);
331 }
332
333 if (TestAndClear(&interrupt_flags, LOG_WASM_CODE)) {
334 TRACE_EVENT0("v8.wasm", "V8.LogCode");
335 wasm::GetWasmEngine()->LogOutstandingCodesForIsolate(isolate_);
336 }
337
338 if (TestAndClear(&interrupt_flags, WASM_CODE_GC)) {
339 TRACE_EVENT0("v8.wasm", "V8.WasmCodeGC");
340 wasm::GetWasmEngine()->ReportLiveCodeFromStackForGC(isolate_);
341 }
342#endif // V8_ENABLE_WEBASSEMBLY
343
344 if (TestAndClear(&interrupt_flags, DEOPT_MARKED_ALLOCATION_SITES)) {
345 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.gc"),
346 "V8.GCDeoptMarkedAllocationSites");
347 isolate_->heap()->DeoptMarkedAllocationSites();
348 }
349
350 if (TestAndClear(&interrupt_flags, INSTALL_CODE)) {
351 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
352 "V8.InstallOptimizedFunctions");
353 DCHECK(isolate_->concurrent_recompilation_enabled());
354 isolate_->optimizing_compile_dispatcher()->InstallOptimizedFunctions();
355 }
356
357#ifdef V8_ENABLE_SPARKPLUG
358 if (TestAndClear(&interrupt_flags, INSTALL_BASELINE_CODE)) {
359 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
360 "V8.FinalizeBaselineConcurrentCompilation");
361 isolate_->baseline_batch_compiler()->InstallBatch();
362 }
363#endif // V8_ENABLE_SPARKPLUG
364
365#ifdef V8_ENABLE_MAGLEV
366 if (TestAndClear(&interrupt_flags, INSTALL_MAGLEV_CODE)) {
367 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
368 "V8.FinalizeMaglevConcurrentCompilation");
369 isolate_->maglev_concurrent_dispatcher()->FinalizeFinishedJobs();
370 }
371#endif // V8_ENABLE_MAGLEV
372
373 if (TestAndClear(&interrupt_flags, API_INTERRUPT)) {
374 TRACE_EVENT0("v8.execute", "V8.InvokeApiInterruptCallbacks");
375 // Callbacks must be invoked outside of ExecutionAccess lock.
376 isolate_->InvokeApiInterruptCallbacks();
377 }
378
379#ifdef V8_RUNTIME_CALL_STATS
380 // Runtime call stats can be enabled at any via Chrome tracing and since
381 // there's no global list of active Isolates this seems to be the only
382 // simple way to invalidate the protector.
383 if (TracingFlags::is_runtime_stats_enabled() &&
384 Protectors::IsNoProfilingIntact(isolate_)) {
385 Protectors::InvalidateNoProfiling(isolate_);
386 }
387#endif
388
389 isolate_->counters()->stack_interrupts()->Increment();
390
391 return ReadOnlyRoots(isolate_).undefined_value();
392}
