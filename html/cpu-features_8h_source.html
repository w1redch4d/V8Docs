98// CpuFeatures keeps track of which features are supported by the target CPU.
99// Supported features must be enabled by a CpuFeatureScope before use.
100// Example:
101// if (assembler->IsSupported(SSE3)) {
102// CpuFeatureScope fscope(assembler, SSE3);
103// // Generate code containing SSE3 instructions.
104// } else {
105// // Generate alternative code.
106// }
107class V8_EXPORT_PRIVATE CpuFeatures : public AllStatic {
108 public:
109 CpuFeatures(const CpuFeatures&) = delete;
110 CpuFeatures& operator=(const CpuFeatures&) = delete;
111
112 static void Probe(bool cross_compile) {
113 static_assert(NUMBER_OF_CPU_FEATURES <= kBitsPerInt);
114 if (initialized_) return;
115 initialized_ = true;
116 ProbeImpl(cross_compile);
117 }
118
119 static unsigned SupportedFeatures() {
120 Probe(false);
121 return supported_;
122 }
123
124 static bool IsSupported(CpuFeature f) {
125 return (supported_ & (1u << f)) != 0;
126 }
127
128 static void SetSupported(CpuFeature f) { supported_ |= 1u << f; }
129 static void SetUnsupported(CpuFeature f) { supported_ &= ~(1u << f); }
130
131 static bool SupportsWasmSimd128();
132
133 static inline bool SupportsOptimizer();
134
135 static inline unsigned icache_line_size() {
136 DCHECK_NE(icache_line_size_, 0);
137 return icache_line_size_;
138 }
139
140 static inline unsigned dcache_line_size() {
141 DCHECK_NE(dcache_line_size_, 0);
142 return dcache_line_size_;
143 }
144
145 static void PrintTarget();
146 static void PrintFeatures();
147
148 private:
149 friend void V8_EXPORT_PRIVATE FlushInstructionCache(void*, size_t);
150 friend class ExternalReference;
151 // Flush instruction cache.
152 static void FlushICache(void* start, size_t size);
153
154 // Platform-dependent implementation.
155 static void ProbeImpl(bool cross_compile);
156
157 static unsigned supported_;
158 static unsigned icache_line_size_;
159 static unsigned dcache_line_size_;
160 static bool initialized_;
161 // This variable is only used for certain archs to query SupportWasmSimd128()
162 // at runtime in builtins using an extern ref. Other callers should use
163 // CpuFeatures::SupportWasmSimd128().
164 static bool supports_wasm_simd_128_;
165 static bool supports_cetss_;
