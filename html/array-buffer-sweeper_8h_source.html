39 V8_EXPORT_PRIVATE bool ContainsSlow(ArrayBufferExtension* extension) const;

41 private:
42 ArrayBufferExtension* head_ = nullptr;
43 ArrayBufferExtension* tail_ = nullptr;
44 // Bytes are approximate as they may be subtracted eagerly, while the
45 // `ArrayBufferExtension` is still in the list. The extension will only be
