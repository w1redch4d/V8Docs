93// RISC-V can perform PC-relative jumps within a 32-bit range using the
94// following two instructions:
95// auipc t6, imm20 ; t0 = PC + imm20 * 2^12
96// jalr ra, t6, imm12; ra = PC + 4, PC = t0 + imm12,
97// Both imm20 and imm12 are treated as two's-complement signed values, usually
98// calculated as:
99// imm20 = (offset + 0x800) >> 12
100// imm12 = offset & 0xfff
101// offset is the signed offset from the auipc instruction. Adding 0x800 handles
102// the offset, but if the offset is >= 2^31 - 2^11, it will overflow. Therefore,
103// the true 32-bit range is:
104// [-2^31 - 2^11, 2^31 - 2^11)
105constexpr size_t kMaxPCRelativeCodeRangeInMB = 2047;

208// - Breaks between 0 and kMaxWatchpointCode are treated as simple watchpoints,

497inline Condition NegateFpuCondition(Condition cc) {
498 DCHECK(cc != cc_always);
499 switch (cc) {
500 case ult:
501 return ge;
502 case ugt:
503 return le;
504 case uge:
505 return lt;
506 case ule:
507 return gt;
508 case lt:
509 return uge;
510 case gt:
511 return ule;
512 case ge:

514 case le:
515 return ugt;
516 case eq:
517 return ne;
518 case ne:
519 return eq;
