190 // Helper class for encoding and decoding a value into and from a bytecode.
191 //
192 // The value is encoded by allocating an entire bytecode range, and encoding
193 // the value as an index in that range, starting at kMinValue; thus the range
194 // of values
195 // [kMinValue, kMinValue + 1, ... , kMaxValue]
196 // is encoded as
197 // [kBytecode, kBytecode + 1, ... , kBytecode + (N - 1)]
198 // where N is the number of values, i.e. kMaxValue - kMinValue + 1.
199 template <Bytecode kBytecode, int kMinValue, int kMaxValue,
200 typename TValue = int>
201 struct BytecodeValueEncoder {
202 static_assert((kBytecode + kMaxValue - kMinValue) <= kMaxUInt8);
203
204 static constexpr bool IsEncodable(TValue value) {
205 return base::IsInRange(static_cast<int>(value), kMinValue, kMaxValue);
206 }
207
208 static constexpr uint8_t Encode(TValue value) {
209 DCHECK(IsEncodable(value));
210 return static_cast<uint8_t>(kBytecode + static_cast<int>(value) -
211 kMinValue);
212 }
213
214 static constexpr TValue Decode(uint8_t bytecode) {
215 DCHECK(base::IsInRange(bytecode, Encode(static_cast<TValue>(kMinValue)),
216 Encode(static_cast<TValue>(kMaxValue))));
217 return static_cast<TValue>(bytecode - kBytecode + kMinValue);
218 }
219 };
