27// Once the Locker is initialized, the current thread will be guaranteed to have
28// the lock for a given isolate.
29void Locker::Initialize(v8::Isolate* isolate) {
30 DCHECK_NOT_NULL(isolate);
31 has_lock_ = false;
32 top_level_ = true;
33 isolate_ = reinterpret_cast<i::Isolate*>(isolate);
34
35 // Record that the Locker has been used at least once.
36 base::Relaxed_Store(&g_locker_was_ever_used_, 1);
37 isolate_->set_was_locker_ever_used();
38
39 // Get the big lock if necessary.
40 if (!isolate_->thread_manager()->IsLockedByCurrentThread()) {
41 isolate_->thread_manager()->Lock();
42 has_lock_ = true;
43
44 // This may be a locker within an unlocker in which case we have to
45 // get the saved state for this thread and restore it.
46 if (isolate_->thread_manager()->RestoreThread()) {
47 top_level_ = false;
48 }
49 }
50 DCHECK(isolate_->thread_manager()->IsLockedByCurrentThread());
51}

53bool Locker::IsLocked(v8::Isolate* isolate) {
54 DCHECK_NOT_NULL(isolate);

93bool ThreadManager::RestoreThread() {
94 DCHECK(IsLockedByCurrentThread());
95 // First check whether the current thread has been 'lazily archived', i.e.
96 // not archived at all. If that is the case we put the state storage we
97 // had prepared back in the free list, since we didn't need it after all.
98 if (lazily_archived_thread_ == ThreadId::Current()) {
99 lazily_archived_thread_ = ThreadId::Invalid();
100 Isolate::PerIsolateThreadData* per_thread =
101 isolate_->FindPerThreadDataForThisThread();
102 DCHECK_NOT_NULL(per_thread);
103 DCHECK(per_thread->thread_state() == lazily_archived_thread_state_);
104 lazily_archived_thread_state_->set_id(ThreadId::Invalid());
105 lazily_archived_thread_state_->LinkInto(ThreadState::FREE_LIST);
106 lazily_archived_thread_state_ = nullptr;
107 per_thread->set_thread_state(nullptr);
108 return true;
109 }
110
111 // Make sure that the preemption thread cannot modify the thread state while
112 // it is being archived or restored.
113 ExecutionAccess access(isolate_);
114
115 // If there is another thread that was lazily archived then we have to really
116 // archive it now.
117 if (lazily_archived_thread_.IsValid()) {
118 EagerlyArchiveThread();
119 }
120 Isolate::PerIsolateThreadData* per_thread =
121 isolate_->FindPerThreadDataForThisThread();
122 if (per_thread == nullptr || per_thread->thread_state() == nullptr) {
123 // This is a new thread.
124 InitThread(access);
125 return false;
126 }
127 // In case multi-cage pointer compression mode is enabled ensure that
128 // current thread's cage base values are properly initialized.
129 PtrComprCageAccessScope ptr_compr_cage_access_scope(isolate_);
130
131 ThreadState* state = per_thread->thread_state();
132 char* from = state->data();
133 from = isolate_->handle_scope_implementer()->RestoreThread(from);
134 from = isolate_->RestoreThread(from);
135 from = Relocatable::RestoreState(isolate_, from);
136 // Stack guard should be restored before Debug, etc. since Debug etc. might
137 // depend on a correct stack guard.
138 from = isolate_->stack_guard()->RestoreStackGuard(from);
139 from = isolate_->debug()->RestoreDebug(from);
140 from = isolate_->regexp_stack()->RestoreStack(from);
141 from = isolate_->bootstrapper()->RestoreState(from);
142 per_thread->set_thread_state(nullptr);
143 state->set_id(ThreadId::Invalid());
144 state->Unlink();
145 state->LinkInto(ThreadState::FREE_LIST);
146 return true;
147}

187void ThreadState::LinkInto(List list) {
188 ThreadState* flying_anchor = list == FREE_LIST
189 ? thread_manager_->free_anchor_
190 : thread_manager_->in_use_anchor_;
191 next_ = flying_anchor->next_;
192 previous_ = flying_anchor;
193 flying_anchor->next_ = this;
194 next_->previous_ = this;

236void ThreadManager::DeleteThreadStateList(ThreadState* anchor) {
237 // The list starts and ends with the anchor.
238 for (ThreadState* current = anchor->next_; current != anchor;) {
239 ThreadState* next = current->next_;
240 delete current;
241 current = next;
242 }
243 delete anchor;

246void ThreadManager::ArchiveThread() {
247 DCHECK_EQ(lazily_archived_thread_, ThreadId::Invalid());
248 DCHECK(!IsArchived());
249 DCHECK(IsLockedByCurrentThread());
250 ThreadState* state = GetFreeThreadState();
251 state->Unlink();
252 Isolate::PerIsolateThreadData* per_thread =
253 isolate_->FindOrAllocatePerThreadDataForThisThread();
254 per_thread->set_thread_state(state);
255 lazily_archived_thread_ = ThreadId::Current();
256 lazily_archived_thread_state_ = state;
257 DCHECK_EQ(state->id(), ThreadId::Invalid());
258 state->set_id(CurrentId());
259 DCHECK_NE(state->id(), ThreadId::Invalid());
260}

262void ThreadManager::EagerlyArchiveThread() {
263 DCHECK(IsLockedByCurrentThread());
264 ThreadState* state = lazily_archived_thread_state_;
265 state->LinkInto(ThreadState::IN_USE_LIST);
266 char* to = state->data();
267 // Ensure that data containing GC roots are archived first, and handle them
268 // in ThreadManager::Iterate(RootVisitor*).
269 to = isolate_->handle_scope_implementer()->ArchiveThread(to);
270 to = isolate_->ArchiveThread(to);
271 to = Relocatable::ArchiveState(isolate_, to);
272 to = isolate_->stack_guard()->ArchiveStackGuard(to);
273 to = isolate_->debug()->ArchiveDebug(to);
274 to = isolate_->regexp_stack()->ArchiveStack(to);
275 to = isolate_->bootstrapper()->ArchiveState(to);
276 lazily_archived_thread_ = ThreadId::Invalid();
277 lazily_archived_thread_state_ = nullptr;
278}

305void ThreadManager::Iterate(RootVisitor* v) {
306 // Expecting no threads during serialization/deserialization
307 for (ThreadState* state = FirstThreadStateInUse(); state != nullptr;
308 state = state->Next()) {
309 char* data = state->data();
310 data = HandleScopeImplementer::Iterate(v, data);
311 data = isolate_->Iterate(v, data);
312 data = Relocatable::Iterate(v, data);
313 data = StackGuard::Iterate(v, data);
314 data = Debug::Iterate(v, data);
315 }
316}

318void ThreadManager::IterateArchivedThreads(ThreadVisitor* v) {
319 for (ThreadState* state = FirstThreadStateInUse(); state != nullptr;
320 state = state->Next()) {
321 char* data = state->data();
322 data += HandleScopeImplementer::ArchiveSpacePerThread();
323 isolate_->IterateThread(v, data);
324 }
325}
