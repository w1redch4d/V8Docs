23bool SourceTextModuleDescriptor::ModuleRequestComparer::operator()(
24 const AstModuleRequest* lhs, const AstModuleRequest* rhs) const {
25 if (int specifier_comparison =
26 AstRawString::Compare(lhs->specifier(), rhs->specifier())) {
27 return specifier_comparison < 0;
28 }
29
30 auto lhsIt = lhs->import_attributes()->cbegin();
31 auto rhsIt = rhs->import_attributes()->cbegin();
32 for (; lhsIt != lhs->import_attributes()->cend() &&
33 rhsIt != rhs->import_attributes()->cend();
34 ++lhsIt, ++rhsIt) {
35 if (int assertion_key_comparison =
36 AstRawString::Compare(lhsIt->first, rhsIt->first)) {
37 return assertion_key_comparison < 0;
38 }
39
40 if (int assertion_value_comparison =
41 AstRawString::Compare(lhsIt->second.first, rhsIt->second.first)) {
42 return assertion_value_comparison < 0;
43 }
44 }
45
46 if (lhs->import_attributes()->size() != rhs->import_attributes()->size()) {
47 return (lhs->import_attributes()->size() <
48 rhs->import_attributes()->size());
49 }
50
51 return false;
52}

132Handle<ModuleRequest> SourceTextModuleDescriptor::AstModuleRequest::Serialize(
133 IsolateT* isolate) const {
134 // The import attributes will be stored in this array in the form:
135 // [key1, value1, location1, key2, value2, location2, ...]
136 Handle<FixedArray> import_attributes_array =
137 isolate->factory()->NewFixedArray(
138 static_cast<int>(import_attributes()->size() *
139 ModuleRequest::kAttributeEntrySize),
140 AllocationType::kOld);
141 {
142 DisallowGarbageCollection no_gc;
143 Tagged<FixedArray> raw_import_attributes = *import_attributes_array;
144 int i = 0;
145 for (auto iter = import_attributes()->cbegin();
146 iter != import_attributes()->cend();
147 ++iter, i += ModuleRequest::kAttributeEntrySize) {
148 raw_import_attributes->set(i, *iter->first->string());
149 raw_import_attributes->set(i + 1, *iter->second.first->string());
150 raw_import_attributes->set(i + 2,
151 Smi::FromInt(iter->second.second.beg_pos));
152 }
153 }
154 return v8::internal::ModuleRequest::New(isolate, specifier()->string(),
155 phase_, import_attributes_array,
156 position());

179template <typename IsolateT>
180Handle<FixedArray> SourceTextModuleDescriptor::SerializeRegularExports(
181 IsolateT* isolate, Zone* zone) const {
182 // We serialize regular exports in a way that lets us later iterate over their
183 // local names and for each local name immediately access all its export
184 // names. (Regular exports have neither import name nor module request.)
185
186 ZoneVector<IndirectHandle<Object>> data(
187 SourceTextModuleInfo::kRegularExportLength * regular_exports_.size(),
188 zone);
189 int index = 0;
190
191 for (auto it = regular_exports_.begin(); it != regular_exports_.end();) {
192 // Find out how many export names this local name has.
193 auto next = it;
194 int count = 0;
195 do {
196 DCHECK_EQ(it->second->local_name, next->second->local_name);
197 DCHECK_EQ(it->second->cell_index, next->second->cell_index);
198 ++next;
199 ++count;
200 } while (next != regular_exports_.end() && next->first == it->first);
201
202 Handle<FixedArray> export_names =
203 isolate->factory()->NewFixedArray(count, AllocationType::kOld);
204 data[index + SourceTextModuleInfo::kRegularExportLocalNameOffset] =
205 it->second->local_name->string();
206 data[index + SourceTextModuleInfo::kRegularExportCellIndexOffset] =
207 handle(Smi::FromInt(it->second->cell_index), isolate);
208 data[index + SourceTextModuleInfo::kRegularExportExportNamesOffset] =
209 export_names;
210 index += SourceTextModuleInfo::kRegularExportLength;
211
212 // Collect the export names.
213 int i = 0;
214 for (; it != next; ++it) {
215 export_names->set(i++, *it->second->export_name->string());
216 }
217 DCHECK_EQ(i, count);
218
219 // Continue with the next distinct key.
220 DCHECK(it == next);
221 }
222 DCHECK_LE(index, static_cast<int>(data.size()));
223 data.resize(index);
224
225 // We cannot create the FixedArray earlier because we only now know the
226 // precise size.
227 Handle<FixedArray> result =
228 isolate->factory()->NewFixedArray(index, AllocationType::kOld);
229 for (int i = 0; i < index; ++i) {
230 result->set(i, *data[i]);
231 }
232 return result;
233}

239void SourceTextModuleDescriptor::MakeIndirectExportsExplicit(Zone* zone) {
240 for (auto it = regular_exports_.begin(); it != regular_exports_.end();) {
241 Entry* entry = it->second;
242 DCHECK_NOT_NULL(entry->local_name);
243 auto import = regular_imports_.find(entry->local_name);
244 if (import != regular_imports_.end()) {
245 // Found an indirect export. Patch export entry and move it from regular
246 // to special.
247 DCHECK_NULL(entry->import_name);
248 DCHECK_LT(entry->module_request, 0);
249 DCHECK_NOT_NULL(import->second->import_name);
250 DCHECK_LE(0, import->second->module_request);
251 DCHECK_LT(import->second->module_request,
252 static_cast<int>(module_requests_.size()));
253 entry->import_name = import->second->import_name;
254 entry->module_request = import->second->module_request;
255 // Hack: When the indirect export cannot be resolved, we want the error
256 // message to point at the import statement, not at the export statement.
257 // Therefore we overwrite [entry]'s location here. Note that Validate()
258 // has already checked for duplicate exports, so it's guaranteed that we
259 // won't need to report any error pointing at the (now lost) export
260 // location.
261 entry->location = import->second->location;
262 entry->local_name = nullptr;
263 AddSpecialExport(entry, zone);
264 it = regular_exports_.erase(it);
265 } else {
266 it++;
267 }
268 }
269}

278void SourceTextModuleDescriptor::AssignCellIndices() {
279 int export_index = 1;
280 for (auto it = regular_exports_.begin(); it != regular_exports_.end();) {
281 auto current_key = it->first;
282 // This local name may be exported under multiple export names. Assign the
283 // same index to each such entry.
284 do {
285 Entry* entry = it->second;
286 DCHECK_NOT_NULL(entry->local_name);
287 DCHECK_NULL(entry->import_name);
288 DCHECK_LT(entry->module_request, 0);
289 DCHECK_EQ(entry->cell_index, 0);
290 entry->cell_index = export_index;
291 it++;
292 } while (it != regular_exports_.end() && it->first == current_key);
293 export_index++;
294 }
295
296 int import_index = -1;
297 for (const auto& elem : regular_imports_) {
298 Entry* entry = elem.second;
299 DCHECK_NOT_NULL(entry->local_name);
300 DCHECK_NOT_NULL(entry->import_name);
301 DCHECK_LE(0, entry->module_request);
302 DCHECK_EQ(entry->cell_index, 0);
303 entry->cell_index = import_index;
304 import_index--;
305 }
306}

310const SourceTextModuleDescriptor::Entry* BetterDuplicate(
311 const SourceTextModuleDescriptor::Entry* candidate,
312 ZoneMap<const AstRawString*, const SourceTextModuleDescriptor::Entry*>&
313 export_names,
314 const SourceTextModuleDescriptor::Entry* current_duplicate) {
315 DCHECK_NOT_NULL(candidate->export_name);
316 DCHECK(candidate->location.IsValid());
317 auto insert_result =
318 export_names.insert(std::make_pair(candidate->export_name, candidate));
319 if (insert_result.second) return current_duplicate;
320 if (current_duplicate == nullptr) {
321 current_duplicate = insert_result.first->second;
322 }
323 return (candidate->location.beg_pos > current_duplicate->location.beg_pos)
324 ? candidate
325 : current_duplicate;
326}

330const SourceTextModuleDescriptor::Entry*
331SourceTextModuleDescriptor::FindDuplicateExport(Zone* zone) const {
332 const SourceTextModuleDescriptor::Entry* duplicate = nullptr;
333 ZoneMap<const AstRawString*, const SourceTextModuleDescriptor::Entry*>
334 export_names(zone);
335 for (const auto& elem : regular_exports_) {
336 duplicate = BetterDuplicate(elem.second, export_names, duplicate);
337 }
338 for (auto entry : special_exports_) {
339 if (entry->export_name == nullptr) continue; // Star export.
340 duplicate = BetterDuplicate(entry, export_names, duplicate);
341 }
342 return duplicate;
343}

345bool SourceTextModuleDescriptor::Validate(
346 ModuleScope* module_scope, PendingCompilationErrorHandler* error_handler,
347 Zone* zone) {
348 DCHECK_EQ(this, module_scope->module());
349 DCHECK_NOT_NULL(error_handler);
350
351 // Report error iff there are duplicate exports.
352 {
353 const Entry* entry = FindDuplicateExport(zone);
354 if (entry != nullptr) {
355 error_handler->ReportMessageAt(
356 entry->location.beg_pos, entry->location.end_pos,
357 MessageTemplate::kDuplicateExport, entry->export_name);
358 return false;
359 }
360 }
361
362 // Report error iff there are exports of non-existent local names.
363 for (const auto& elem : regular_exports_) {
364 const Entry* entry = elem.second;
365 DCHECK_NOT_NULL(entry->local_name);
366 if (module_scope->LookupLocal(entry->local_name) == nullptr) {
367 error_handler->ReportMessageAt(
368 entry->location.beg_pos, entry->location.end_pos,
369 MessageTemplate::kModuleExportUndefined, entry->local_name);
370 return false;
371 }
372 }
373
374 MakeIndirectExportsExplicit(zone);
375 AssignCellIndices();
376 return true;
377}
