46template <typename ScopeInfoPtr>
47class ScopeInfo::LocalNamesRange {
48 public:
49 class Iterator {
50 public:
51 Iterator(const LocalNamesRange* range, InternalIndex index)
52 : range_(range), index_(index) {
53 DCHECK_NOT_NULL(range);
54 if (!range_->inlined()) advance_hashtable_index();
55 }
56
57 Iterator& operator++() {
58 DCHECK_LT(index_, range_->max_index());
59 ++index_;
60 if (range_->inlined()) return *this;
61 advance_hashtable_index();
62 return *this;
63 }
64
65 friend bool operator==(const Iterator& a, const Iterator& b) {
66 return a.range_ == b.range_ && a.index_ == b.index_;
67 }
68
69 friend bool operator!=(const Iterator& a, const Iterator& b) {
70 return !(a == b);
71 }
72
73 Tagged<String> name(PtrComprCageBase cage_base) const {
74 DCHECK_LT(index_, range_->max_index());
75 if (range_->inlined()) {
76 return scope_info()->ContextInlinedLocalName(cage_base,
77 index_.as_int());
78 }
79 return Cast<String>(table()->KeyAt(cage_base, index_));
80 }
81
82 Tagged<String> name() const {
83 PtrComprCageBase cage_base = GetPtrComprCageBase(*scope_info());
84 return name(cage_base);
85 }
86
87 const Iterator* operator*() const { return this; }
88
89 int index() const {
90 if (range_->inlined()) return index_.as_int();
91 return table()->IndexAt(index_);
92 }
93
94 private:
95 const LocalNamesRange* range_;
96 InternalIndex index_;
97
98 ScopeInfoPtr scope_info() const { return range_->scope_info_; }
99
100 Tagged<NameToIndexHashTable> table() const {
101 return scope_info()->context_local_names_hashtable();
102 }
103
104 void advance_hashtable_index() {
105 DisallowGarbageCollection no_gc;
106 ReadOnlyRoots roots = GetReadOnlyRoots();
107 InternalIndex max = range_->max_index();
108 // Increment until iterator points to a valid key or max.
109 while (index_ < max) {
110 Tagged<Object> key = table()->KeyAt(index_);
111 if (table()->IsKey(roots, key)) break;
112 ++index_;
113 }
114 }
115
116 friend class LocalNamesRange;
117 };
118
119 bool inlined() const { return scope_info_->HasInlinedLocalNames(); }
120
121 InternalIndex max_index() const {
122 int max = inlined()
123 ? scope_info_->ContextLocalCount()
124 : scope_info_->context_local_names_hashtable()->Capacity();
125 return InternalIndex(max);
126 }
127
128 explicit LocalNamesRange(ScopeInfoPtr scope_info) : scope_info_(scope_info) {}
129
130 inline Iterator begin() const { return Iterator(this, InternalIndex(0)); }
131
132 inline Iterator end() const { return Iterator(this, max_index()); }
133
134 private:
135 ScopeInfoPtr scope_info_;
136};
