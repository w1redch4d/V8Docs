17// A poor man's version of STL's bitset: A bit set of enums E (without explicit
18// values), fitting into an integral type T.
19template <class E, class T = int>
20class EnumSet {
21 static_assert(std::is_enum<E>::value, "EnumSet can only be used with enums");
22
23 public:
