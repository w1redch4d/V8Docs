182bool IsMoreGeneralElementsKindTransition(ElementsKind from_kind,
183 ElementsKind to_kind) {
184 if (!IsFastElementsKind(from_kind)) return false;
185 if (!IsFastTransitionTarget(to_kind)) return false;
186 DCHECK(!IsTypedArrayOrRabGsabTypedArrayElementsKind(from_kind));
187 DCHECK(!IsTypedArrayOrRabGsabTypedArrayElementsKind(to_kind));
188 switch (from_kind) {
189 case PACKED_SMI_ELEMENTS:
190 return to_kind != PACKED_SMI_ELEMENTS;
191 case HOLEY_SMI_ELEMENTS:
192 return to_kind != PACKED_SMI_ELEMENTS && to_kind != HOLEY_SMI_ELEMENTS;
193 case PACKED_DOUBLE_ELEMENTS:
194 return to_kind != PACKED_SMI_ELEMENTS && to_kind != HOLEY_SMI_ELEMENTS &&
195 to_kind != PACKED_DOUBLE_ELEMENTS;
196 case HOLEY_DOUBLE_ELEMENTS:
197 return to_kind == PACKED_ELEMENTS || to_kind == HOLEY_ELEMENTS;
198 case PACKED_ELEMENTS:
199 return to_kind == HOLEY_ELEMENTS;
200 case HOLEY_ELEMENTS:
201 return false;
202 default:
203 return false;
204 }
205}

207bool UnionElementsKindUptoSize(ElementsKind* a_out, ElementsKind b) {
208 // Assert that the union of two ElementKinds can be computed via std::max.
209 static_assert(PACKED_SMI_ELEMENTS < HOLEY_SMI_ELEMENTS,
210 "ElementsKind union not computable via std::max.");
211 static_assert(HOLEY_SMI_ELEMENTS < PACKED_ELEMENTS,
212 "ElementsKind union not computable via std::max.");
213 static_assert(PACKED_ELEMENTS < HOLEY_ELEMENTS,
214 "ElementsKind union not computable via std::max.");
215 static_assert(PACKED_DOUBLE_ELEMENTS < HOLEY_DOUBLE_ELEMENTS,
216 "ElementsKind union not computable via std::max.");
217 ElementsKind a = *a_out;
218 switch (a) {
219 case PACKED_SMI_ELEMENTS:
220 switch (b) {
221 case PACKED_SMI_ELEMENTS:
222 case HOLEY_SMI_ELEMENTS:
223 case PACKED_ELEMENTS:
224 case HOLEY_ELEMENTS:
225 *a_out = b;
226 return true;
227 default:
228 return false;
229 }
230 case HOLEY_SMI_ELEMENTS:
231 switch (b) {
232 case PACKED_SMI_ELEMENTS:
233 case HOLEY_SMI_ELEMENTS:
234 *a_out = HOLEY_SMI_ELEMENTS;
235 return true;
236 case PACKED_ELEMENTS:
237 case HOLEY_ELEMENTS:
238 *a_out = HOLEY_ELEMENTS;
239 return true;
240 default:
241 return false;
242 }
243 case PACKED_ELEMENTS:
244 switch (b) {
245 case PACKED_SMI_ELEMENTS:
246 case PACKED_ELEMENTS:
247 *a_out = PACKED_ELEMENTS;
248 return true;
249 case HOLEY_SMI_ELEMENTS:
250 case HOLEY_ELEMENTS:
251 *a_out = HOLEY_ELEMENTS;
252 return true;
253 default:
254 return false;
255 }
256 case HOLEY_ELEMENTS:
257 switch (b) {
258 case PACKED_SMI_ELEMENTS:
259 case HOLEY_SMI_ELEMENTS:
260 case PACKED_ELEMENTS:
261 case HOLEY_ELEMENTS:
262 *a_out = HOLEY_ELEMENTS;
263 return true;
264 default:
265 return false;
266 }
267 case PACKED_DOUBLE_ELEMENTS:
268 switch (b) {
269 case PACKED_DOUBLE_ELEMENTS:
270 case HOLEY_DOUBLE_ELEMENTS:
271 *a_out = b;
272 return true;
273 default:
274 return false;
275 }
276 case HOLEY_DOUBLE_ELEMENTS:
277 switch (b) {
278 case PACKED_DOUBLE_ELEMENTS:
279 case HOLEY_DOUBLE_ELEMENTS:
280 *a_out = HOLEY_DOUBLE_ELEMENTS;
281 return true;
282 default:
283 return false;
284 }
285 default:
286 break;
287 }
288 return false;
289}
