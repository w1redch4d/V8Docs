20template <typename TCallable, class... TArgs>
21TNode<Object> CodeStubAssembler::Call(TNode<Context> context,
22 TNode<TCallable> callable,
23 ConvertReceiverMode mode,
24 TNode<Object> receiver, TArgs... args) {
25 static_assert(std::is_same<Object, TCallable>::value ||
26 std::is_base_of<HeapObject, TCallable>::value);
27 static_assert(!std::is_base_of<JSFunction, TCallable>::value,
28 "Use CallFunction() when the callable is a JSFunction.");
29
30 if (IsUndefinedConstant(receiver) || IsNullConstant(receiver)) {
31 DCHECK_NE(mode, ConvertReceiverMode::kNotNullOrUndefined);
32 return CallJS(Builtins::Call(ConvertReceiverMode::kNullOrUndefined),
33 context, callable, receiver, args...);
34 }
35 DCheckReceiver(mode, receiver);
36 return CallJS(Builtins::Call(mode), context, callable, receiver, args...);
37}

90template <typename Function>
91TNode<Object> CodeStubAssembler::FastCloneJSObject(
92 TNode<HeapObject> object, TNode<Map> source_map, TNode<Map> target_map,
93 const Function& materialize_target, bool target_is_new) {
94 Label done_copy_properties(this), done_copy_elements(this);
95
96 // This macro only suport JSObjects.
97 CSA_DCHECK(this, InstanceTypeEqual(LoadInstanceType(object), JS_OBJECT_TYPE));
98 CSA_DCHECK(this, IsStrong(TNode<MaybeObject>(target_map)));
99 CSA_DCHECK(
100 this, InstanceTypeEqual(LoadMapInstanceType(target_map), JS_OBJECT_TYPE));
101 // We do not want to deal with slack-tracking here.
102 CSA_DCHECK(this, IsNotSetWord32<Map::Bits3::ConstructionCounterBits>(
103 LoadMapBitField3(source_map)));
104 CSA_DCHECK(this, IsNotSetWord32<Map::Bits3::ConstructionCounterBits>(
105 LoadMapBitField3(target_map)));
106
107 TVARIABLE(HeapObject, var_properties, EmptyFixedArrayConstant());
108 TVARIABLE(FixedArray, var_elements, EmptyFixedArrayConstant());
109
110 // Copy the PropertyArray backing store. The source PropertyArray
111 // must be either an Smi, or a PropertyArray.
112 Comment("FastCloneJSObject: cloning properties");
113 TNode<Object> source_properties =
114 LoadObjectField(object, JSObject::kPropertiesOrHashOffset);
115 {
116 GotoIf(TaggedIsSmi(source_properties), &done_copy_properties);
117 GotoIf(IsEmptyFixedArray(source_properties), &done_copy_properties);
118
119 // This fastcase requires that the source object has fast properties.
120 TNode<PropertyArray> source_property_array = CAST(source_properties);
121
122 TNode<IntPtrT> length = LoadPropertyArrayLength(source_property_array);
123 GotoIf(IntPtrEqual(length, IntPtrConstant(0)), &done_copy_properties);
124
125 TNode<PropertyArray> property_array = AllocatePropertyArray(length);
126 FillPropertyArrayWithUndefined(property_array, IntPtrConstant(0), length);
127 CopyPropertyArrayValues(source_property_array, property_array, length,
128 SKIP_WRITE_BARRIER, DestroySource::kNo);
129 var_properties = property_array;
130 }
131
132 Goto(&done_copy_properties);
133 BIND(&done_copy_properties);
134
135 Comment("FastCloneJSObject: cloning elements");
136 TNode<FixedArrayBase> source_elements = LoadElements(CAST(object));
137 GotoIf(TaggedEqual(source_elements, EmptyFixedArrayConstant()),
138 &done_copy_elements);
139 var_elements = CAST(CloneFixedArray(
140 source_elements, ExtractFixedArrayFlag::kAllFixedArraysDontCopyCOW));
141
142 Goto(&done_copy_elements);
143 BIND(&done_copy_elements);
144
145 Comment("FastCloneJSObject: initialize the target object");
146 TNode<JSReceiver> target = materialize_target(
147 target_map, var_properties.value(), var_elements.value());
148
149 // Lastly, clone any in-object properties.
150#ifdef DEBUG
151 {
152 TNode<IntPtrT> source_used_instance_size =
153 MapUsedInstanceSizeInWords(source_map);
154 TNode<IntPtrT> target_used_instance_size =
155 MapUsedInstanceSizeInWords(target_map);
156 TNode<IntPtrT> source_inobject_properties_start =
157 LoadMapInobjectPropertiesStartInWords(source_map);
158 TNode<IntPtrT> target_inobject_properties_start =
159 LoadMapInobjectPropertiesStartInWords(target_map);
160 CSA_DCHECK(this, IntPtrEqual(IntPtrSub(target_used_instance_size,
161 target_inobject_properties_start),
162 IntPtrSub(source_used_instance_size,
163 source_inobject_properties_start)));
164 }
165#endif // DEBUG
166
167 // 1) Initialize unused in-object properties.
168 Comment("FastCloneJSObject: initializing unused in-object properties");
169 TNode<IntPtrT> target_used_payload_end =
170 TimesTaggedSize(MapUsedInstanceSizeInWords(target_map));
171 TNode<IntPtrT> target_payload_end =
172 TimesTaggedSize(LoadMapInstanceSizeInWords(target_map));
173 InitializeFieldsWithRoot(target, target_used_payload_end, target_payload_end,
174 RootIndex::kUndefinedValue);
175
176 // 2) Copy all used in-object properties.
177 Comment("FastCloneJSObject: copying used in-object properties");
178 TNode<IntPtrT> source_payload_start =
179 TimesTaggedSize(LoadMapInobjectPropertiesStartInWords(source_map));
180 TNode<IntPtrT> target_payload_start =
181 TimesTaggedSize(LoadMapInobjectPropertiesStartInWords(target_map));
182 TNode<IntPtrT> field_offset_difference =
183 IntPtrSub(source_payload_start, target_payload_start);
184
185 Label done_copy_used(this);
186 auto EmitCopyLoop = [&](bool write_barrier) {
187 if (write_barrier) {
188 Comment(
189 "FastCloneJSObject: copying used in-object properties with write "
190 "barrier");
191 } else {
192 Comment(
193 "FastCloneJSObject: copying used in-object properties without write "
194 "barrier");
195 }
196 BuildFastLoop<IntPtrT>(
197 target_payload_start, target_used_payload_end,
198 [&](TNode<IntPtrT> result_offset) {
199 TNode<IntPtrT> source_offset =
200 IntPtrSub(result_offset, field_offset_difference);
201 if (write_barrier) {
202 TNode<Object> field = LoadObjectField(object, source_offset);
203 StoreObjectField(target, result_offset, field);
204 } else {
205 TNode<TaggedT> field =
206 LoadObjectField<TaggedT>(object, source_offset);
207 StoreObjectFieldNoWriteBarrier(target, result_offset, field);
208 }
209 },
210 kTaggedSize, LoopUnrollingMode::kYes, IndexAdvanceMode::kPost);
211 };
212
213 if (!target_is_new) {
214 Label if_no_write_barrier(this),
215 if_needs_write_barrier(this, Label::kDeferred);
216
217 TNode<BoolT> needs_write_barrier = IsPageFlagReset(
218 BitcastTaggedToWord(target), MemoryChunk::kIsInYoungGenerationMask);
219 Branch(needs_write_barrier, &if_needs_write_barrier, &if_no_write_barrier);
220
221 BIND(&if_needs_write_barrier);
222 EmitCopyLoop(true);
223
224 Goto(&done_copy_used);
225 BIND(&if_no_write_barrier);
226 }
227
228 EmitCopyLoop(false);
229 Goto(&done_copy_used);
230
231 BIND(&done_copy_used);
232
233 // 3) Duplicate heap number boxes if needed.
234 // We need to go through the {object} again here and properly clone
235 // them. We use a second loop here to ensure that the GC (and heap
236 // verifier) always sees properly initialized objects, i.e. never
237 // hits undefined values in double fields.
238 Comment("FastCloneJSObject: cloning heap numbers");
239 ConstructorBuiltinsAssembler(state()).CopyMutableHeapNumbersInObject(
240 target, target_payload_start, target_used_payload_end);
241
242 return target;
243}
