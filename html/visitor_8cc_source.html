35// Conservative scanning of objects is not compatible with ASAN as we may scan
36// over objects reading poisoned memory. One such example was added to libc++
37// (June 2024) in the form of container annotations for short std::string.
38DISABLE_ASAN
39void ConservativeTracingVisitor::TraceConservatively(
40 const HeapObjectHeader& header) {
41 const auto object_view = ObjectView<>(header);
42 uintptr_t* word = reinterpret_cast<uintptr_t*>(object_view.Start());
43 for (size_t i = 0; i < (object_view.Size() / sizeof(uintptr_t)); ++i) {
44 uintptr_t maybe_full_ptr = word[i];
45 // |object| may be uninitialized by design or just contain padding bytes.
46 // Copy into a local variable that is not poisoned for conservative marking.
47 // Copy into a temporary variable to maintain the original MSAN state.
48 MSAN_MEMORY_IS_INITIALIZED(&maybe_full_ptr, sizeof(maybe_full_ptr));
49 // First, check the full pointer.
50 if (maybe_full_ptr > SentinelPointer::kSentinelValue)
51 this->TraceConservativelyIfNeeded(
52 reinterpret_cast<Address>(maybe_full_ptr));
53#if defined(CPPGC_POINTER_COMPRESSION)
54 // Then, check for compressed pointers.
55 auto decompressed_low = reinterpret_cast<Address>(
56 CompressedPointer::Decompress(static_cast<uint32_t>(maybe_full_ptr)));
57 if (decompressed_low >
58 reinterpret_cast<void*>(SentinelPointer::kSentinelValue))
59 this->TraceConservativelyIfNeeded(decompressed_low);
60 auto decompressed_high = reinterpret_cast<Address>(
61 CompressedPointer::Decompress(static_cast<uint32_t>(
62 maybe_full_ptr >> (sizeof(uint32_t) * CHAR_BIT))));
63 if (decompressed_high >
64 reinterpret_cast<void*>(SentinelPointer::kSentinelValue))
65 this->TraceConservativelyIfNeeded(decompressed_high);
66#endif // !defined(CPPGC_POINTER_COMPRESSION)
67 }
68}

70void ConservativeTracingVisitor::TryTracePointerConservatively(
71 Address address) {
72#if defined(CPPGC_CAGED_HEAP)
73 // TODO(chromium:1056170): Add support for SIMD in stack scanning.
74 if (V8_LIKELY(!CagedHeapBase::IsWithinCage(address))) return;
75#endif // defined(CPPGC_CAGED_HEAP)
76
77 const BasePage* page = reinterpret_cast<const BasePage*>(
78 page_backend_.Lookup(const_cast<ConstAddress>(address)));
79
80 if (!page) return;
81
82 DCHECK_EQ(&heap_, &page->heap());
83
84 auto* header = page->TryObjectHeaderFromInnerAddress(address);
85
86 if (!header) return;
87
88 TraceConservativelyIfNeeded(*header);
89}

91void ConservativeTracingVisitor::TraceConservativelyIfNeeded(
92 const void* address) {
93 auto pointer = reinterpret_cast<Address>(const_cast<void*>(address));
94 TryTracePointerConservatively(pointer);
95#if defined(CPPGC_POINTER_COMPRESSION)
96 auto try_trace = [this](Address ptr) {
97 if (ptr > reinterpret_cast<Address>(SentinelPointer::kSentinelValue))
98 TryTracePointerConservatively(ptr);
99 };
100 // If pointer compression enabled, we may have random compressed pointers on
101 // stack (e.g. due to inlined collections). Extract, decompress and trace both
102 // halfwords.
103 auto decompressed_low = static_cast<Address>(CompressedPointer::Decompress(
104 static_cast<uint32_t>(reinterpret_cast<uintptr_t>(pointer))));
105 try_trace(decompressed_low);
106 auto decompressed_high = static_cast<Address>(CompressedPointer::Decompress(
107 static_cast<uint32_t>(reinterpret_cast<uintptr_t>(pointer) >>
108 (sizeof(uint32_t) * CHAR_BIT))));
109 try_trace(decompressed_high);
110 // In addition, check half-compressed halfwords, since the compiler is free to
111 // spill intermediate results of compression/decompression onto the stack.
112 const uintptr_t base = CagedHeapBase::GetBase();
113 DCHECK(base);
114 auto intermediate_decompressed_low = reinterpret_cast<Address>(
115 static_cast<uint32_t>(reinterpret_cast<uintptr_t>(pointer)) | base);
116 try_trace(intermediate_decompressed_low);
117 auto intermediate_decompressed_high = reinterpret_cast<Address>(
118 static_cast<uint32_t>(reinterpret_cast<uintptr_t>(pointer) >>
119 (sizeof(uint32_t) * CHAR_BIT)) |
120 base);
121 try_trace(intermediate_decompressed_high);
122#endif // defined(CPPGC_POINTER_COMPRESSION)
123}

125void ConservativeTracingVisitor::TraceConservativelyIfNeeded(
126 HeapObjectHeader& header) {
127 if (!header.IsInConstruction<AccessMode::kNonAtomic>()) {
128 VisitFullyConstructedConservatively(header);
129 } else {
130 VisitInConstructionConservatively(
131 header,
132 [](ConservativeTracingVisitor* v, const HeapObjectHeader& header) {
133 v->TraceConservatively(header);
134 });
135 }
136}

144}
