148// The size of the buffer that is needed to store all converted parameters in
149// the generic js-to-wasm wrapper. The buffer size is the number of bytes that
150// parameters may use on the stack + the number of bytes parameters may use in
151// parameter registers. In the worst case all parameters need 8 bytes. Bigger
152// parameters are not possible because SIMD parameters cannot be passed from
153// JavaScript to WebAssembly.
154// Note that a WebAssembly function that uses the maximum number of parameters
155// may only use a single type for all parameters, in which case some parameter
156// registers will remain empty. Thus the buffer size is the maximum number of
157// parameters + the parameter registers that may remain empty.
158// TODO(ahaas): Figure out a lower number if possible.
159constexpr size_t kGenericJSToWasmWrapperParamBufferMaxSize =
160 (wasm::kV8MaxWasmFunctionParams + arraysize(wasm::kGpParamRegisters) +
161 arraysize(wasm::kFpParamRegisters)) *
162 kDoubleSize;

164class LinkageAllocator {
165 public:
166 template <size_t kNumGpRegs, size_t kNumFpRegs>
167 constexpr LinkageAllocator(const Register (&gp)[kNumGpRegs],
168 const DoubleRegister (&fp)[kNumFpRegs])
169 : LinkageAllocator(gp, kNumGpRegs, fp, kNumFpRegs) {}
170
171 constexpr LinkageAllocator(const Register* gp, int gpc,
172 const DoubleRegister* fp, int fpc)
173 : gp_count_(gpc), gp_regs_(gp), fp_count_(fpc), fp_regs_(fp) {}
174
175 bool CanAllocateGP() const { return gp_offset_ < gp_count_; }
176 bool CanAllocateFP(MachineRepresentation rep) const {
177#if V8_TARGET_ARCH_ARM
178 switch (rep) {
179 case MachineRepresentation::kFloat32: {
180 // Get the next D-register (Liftoff only uses the even S-registers).
181 int next = fp_allocator_.NextSlot(2) / 2;
182 // Only the lower 16 D-registers alias S-registers.
183 return next < fp_count_ && fp_regs_[next].code() < 16;
184 }
185 case MachineRepresentation::kFloat64: {
186 int next = fp_allocator_.NextSlot(2) / 2;
187 return next < fp_count_;
188 }
189 case MachineRepresentation::kSimd128: {
190 int next = fp_allocator_.NextSlot(4) / 2;
191 return next < fp_count_ - 1; // 2 D-registers are required.
192 }
193 default:
194 UNREACHABLE();
195 return false;
196 }
197#else
198 return fp_offset_ < fp_count_;
199#endif
200 }
201
202 int NextGpReg() {
203 DCHECK_LT(gp_offset_, gp_count_);
204 return gp_regs_[gp_offset_++].code();
205 }
206
207 int NextFpReg(MachineRepresentation rep) {
208 DCHECK(CanAllocateFP(rep));
209#if V8_TARGET_ARCH_ARM
210 switch (rep) {
211 case MachineRepresentation::kFloat32: {
212 // Liftoff uses only even-numbered S-registers, and encodes them using
213 // the code of the corresponding D-register. This limits the calling
214 // interface to only using the even-numbered S-registers.
215 int d_reg_code = NextFpReg(MachineRepresentation::kFloat64);
216 DCHECK_GT(16, d_reg_code); // D16 - D31 don't alias S-registers.
217 return d_reg_code * 2;
218 }
219 case MachineRepresentation::kFloat64: {
220 int next = fp_allocator_.Allocate(2) / 2;
221 return fp_regs_[next].code();
222 }
223 case MachineRepresentation::kSimd128: {
224 int next = fp_allocator_.Allocate(4) / 2;
225 int d_reg_code = fp_regs_[next].code();
226 // Check that result and the next D-register pair.
227 DCHECK_EQ(0, d_reg_code % 2);
228 DCHECK_EQ(d_reg_code + 1, fp_regs_[next + 1].code());
229 return d_reg_code / 2;
230 }
231 default:
232 UNREACHABLE();
233 }
234#else
235 return fp_regs_[fp_offset_++].code();
236#endif
237 }
238
239 // Stackslots are counted upwards starting from 0 (or the offset set by
240 // {SetStackOffset}. If {type} needs more than one stack slot, the lowest
241 // used stack slot is returned.
242 int NextStackSlot(MachineRepresentation type) {
243 int num_slots =
244 AlignedSlotAllocator::NumSlotsForWidth(ElementSizeInBytes(type));
245 int slot = slot_allocator_.Allocate(num_slots);
246 return slot;
247 }
248
249 // Set an offset for the stack slots returned by {NextStackSlot} and
250 // {NumStackSlots}. Can only be called before any call to {NextStackSlot}.
251 void SetStackOffset(int offset) {
252 DCHECK_LE(0, offset);
253 DCHECK_EQ(0, slot_allocator_.Size());
254 slot_allocator_.AllocateUnaligned(offset);
255 }
256
257 int NumStackSlots() const { return slot_allocator_.Size(); }
258
259 void EndSlotArea() { slot_allocator_.AllocateUnaligned(0); }
260
261 private:
262 const int gp_count_;
263 int gp_offset_ = 0;
264 const Register* const gp_regs_;
265
266 const int fp_count_;
267#if V8_TARGET_ARCH_ARM
268 // Use an aligned slot allocator to model ARM FP register aliasing. The slots
269 // are 32 bits, so 2 slots are required for a D-register, 4 for a Q-register.
270 AlignedSlotAllocator fp_allocator_;
271#else
272 int fp_offset_ = 0;
273#endif
274 const DoubleRegister* const fp_regs_;
275
276 AlignedSlotAllocator slot_allocator_;
277};

279// Helper for allocating either an GP or FP reg, or the next stack slot.
280class LinkageLocationAllocator {
281 public:
282 template <size_t kNumGpRegs, size_t kNumFpRegs>
283 constexpr LinkageLocationAllocator(const Register (&gp)[kNumGpRegs],
284 const DoubleRegister (&fp)[kNumFpRegs],
285 int slot_offset)
286 : allocator_(LinkageAllocator(gp, fp)), slot_offset_(slot_offset) {}
287
288 LinkageLocation Next(MachineRepresentation rep) {
289 MachineType type = MachineType::TypeForRepresentation(rep);
290 if (IsFloatingPoint(rep)) {
291 if (allocator_.CanAllocateFP(rep)) {
292 int reg_code = allocator_.NextFpReg(rep);
293 return LinkageLocation::ForRegister(reg_code, type);
294 }
295 } else if (allocator_.CanAllocateGP()) {
296 int reg_code = allocator_.NextGpReg();
297 return LinkageLocation::ForRegister(reg_code, type);
298 }
299 // Cannot use register; use stack slot.
300 int index = -1 - (slot_offset_ + allocator_.NextStackSlot(rep));
301 return LinkageLocation::ForCallerFrameSlot(index, type);
302 }
303
304 int NumStackSlots() const { return allocator_.NumStackSlots(); }
305 void EndSlotArea() { allocator_.EndSlotArea(); }
306
307 private:
308 LinkageAllocator allocator_;
309 // Since params and returns are in different stack frames, we must allocate
310 // them separately. Parameter slots don't need an offset, but return slots
311 // must be offset to just before the param slots, using this |slot_offset_|.
312 int slot_offset_;
313};
