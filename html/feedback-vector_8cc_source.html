35static bool IsPropertyNameFeedback(Tagged<MaybeObject> feedback) {
36 Tagged<HeapObject> heap_object;

81template <typename IsolateT>
82Handle<FeedbackMetadata> FeedbackMetadata::New(IsolateT* isolate,
83 const FeedbackVectorSpec* spec) {
84 auto* factory = isolate->factory();
85
86 const int slot_count = spec->slot_count();
87 const int create_closure_slot_count = spec->create_closure_slot_count();
88 if (slot_count == 0 && create_closure_slot_count == 0) {
89 return factory->empty_feedback_metadata();
90 }
91#ifdef DEBUG
92 for (int i = 0; i < slot_count;) {
93 FeedbackSlotKind kind = spec->GetKind(FeedbackSlot(i));
94 int entry_size = FeedbackMetadata::GetSlotSize(kind);
95 for (int j = 1; j < entry_size; j++) {
96 kind = spec->GetKind(FeedbackSlot(i + j));
97 DCHECK_EQ(FeedbackSlotKind::kInvalid, kind);
98 }
99 i += entry_size;
100 }
101#endif
102
103 Handle<FeedbackMetadata> metadata =
104 factory->NewFeedbackMetadata(slot_count, create_closure_slot_count);
105
106 // Initialize the slots. The raw data section has already been pre-zeroed in
107 // NewFeedbackMetadata.
108 for (int i = 0; i < slot_count; i++) {
109 FeedbackSlot slot(i);
110 FeedbackSlotKind kind = spec->GetKind(slot);
111 metadata->SetKind(slot, kind);
112 }
113
114 for (int i = 0; i < create_closure_slot_count; i++) {
115 uint16_t parameter_count = spec->GetCreateClosureParameterCount(i);
116 metadata->SetCreateClosureParameterCount(i, parameter_count);
117 }
118
119 return metadata;
120}

127bool FeedbackMetadata::SpecDiffersFrom(
128 const FeedbackVectorSpec* other_spec) const {
129 if (other_spec->slot_count() != slot_count()) {
130 return true;
131 }
132
133 int slots = slot_count();
134 for (int i = 0; i < slots;) {
135 FeedbackSlot slot(i);
136 FeedbackSlotKind kind = GetKind(slot);
137 int entry_size = FeedbackMetadata::GetSlotSize(kind);
138
139 if (kind != other_spec->GetKind(slot)) {
140 return true;
141 }
142 i += entry_size;
143 }
144 return false;
145}

214Handle<ClosureFeedbackCellArray> ClosureFeedbackCellArray::New(
215 Isolate* isolate, DirectHandle<SharedFunctionInfo> shared,
216 AllocationType allocation) {
217 int length = shared->feedback_metadata()->create_closure_slot_count();
218 if (length == 0) {
219 return isolate->factory()->empty_closure_feedback_cell_array();
220 }
221
222 // Pre-allocate the cells s.t. we can initialize `result` without further
223 // allocation.
224 DirectHandleVector<FeedbackCell> cells(isolate);
225 cells.reserve(length);
226 for (int i = 0; i < length; i++) {
227 DirectHandle<FeedbackCell> cell = isolate->factory()->NewNoClosuresCell();
228#ifdef V8_ENABLE_LEAPTIERING
229 uint16_t parameter_count =
230 shared->feedback_metadata()->GetCreateClosureParameterCount(i);
231 Tagged<Code> initial_code = *BUILTIN_CODE(isolate, CompileLazy);
232 cell->allocate_dispatch_handle(isolate, parameter_count, initial_code);
233#endif
234 cells.push_back(cell);
235 }
236
237 std::optional<DisallowGarbageCollection> no_gc;
238 auto result = Allocate(isolate, length, &no_gc, allocation);
239 for (int i = 0; i < length; i++) {
240 result->set(i, *cells[i]);
241 }
242
243 return result;
244}

247Handle<FeedbackVector> FeedbackVector::New(
248 Isolate* isolate, DirectHandle<SharedFunctionInfo> shared,
249 DirectHandle<ClosureFeedbackCellArray> closure_feedback_cell_array,
250 DirectHandle<FeedbackCell> parent_feedback_cell,
251 IsCompiledScope* is_compiled_scope) {
252 DCHECK(is_compiled_scope->is_compiled());
253 Factory* factory = isolate->factory();
254
255 DirectHandle<FeedbackMetadata> feedback_metadata(shared->feedback_metadata(),
256 isolate);
257 const int slot_count = feedback_metadata->slot_count();
258
259 Handle<FeedbackVector> vector = factory->NewFeedbackVector(
260 shared, closure_feedback_cell_array, parent_feedback_cell);
261
262 DCHECK_EQ(vector->length(), slot_count);
263
264 DCHECK_EQ(vector->shared_function_info(), *shared);
265 DCHECK_EQ(vector->invocation_count(), 0);
266#ifndef V8_ENABLE_LEAPTIERING
267 DCHECK_EQ(vector->tiering_state(), TieringState::kNone);
268 DCHECK(!vector->maybe_has_maglev_code());
269 DCHECK(!vector->maybe_has_turbofan_code());
270 DCHECK(vector->maybe_optimized_code().IsCleared());
271#endif // !V8_ENABLE_LEAPTIERING
272
273 // Ensure we can skip the write barrier
274 DirectHandle<Symbol> uninitialized_sentinel = UninitializedSentinel(isolate);
275 DCHECK_EQ(ReadOnlyRoots(isolate).uninitialized_symbol(),
276 *uninitialized_sentinel);
277 for (int i = 0; i < slot_count;) {
278 FeedbackSlot slot(i);
279 FeedbackSlotKind kind = feedback_metadata->GetKind(slot);
280 int entry_size = FeedbackMetadata::GetSlotSize(kind);
281
282 Tagged<MaybeObject> extra_value = *uninitialized_sentinel;
283 switch (kind) {
284 case FeedbackSlotKind::kLoadGlobalInsideTypeof:
285 case FeedbackSlotKind::kLoadGlobalNotInsideTypeof:
286 case FeedbackSlotKind::kStoreGlobalSloppy:
287 case FeedbackSlotKind::kStoreGlobalStrict:
288 case FeedbackSlotKind::kJumpLoop:
289 vector->Set(slot, ClearedValue(isolate), SKIP_WRITE_BARRIER);
290 break;
291 case FeedbackSlotKind::kForIn:
292 case FeedbackSlotKind::kCompareOp:
293 case FeedbackSlotKind::kBinaryOp:
294 case FeedbackSlotKind::kTypeOf:
295 vector->Set(slot, Smi::zero(), SKIP_WRITE_BARRIER);
296 break;
297 case FeedbackSlotKind::kLiteral:
298 vector->Set(slot, Smi::zero(), SKIP_WRITE_BARRIER);
299 break;
300 case FeedbackSlotKind::kCall:
301 vector->Set(slot, *uninitialized_sentinel, SKIP_WRITE_BARRIER);
302 extra_value = Smi::zero();
303 break;
304 case FeedbackSlotKind::kCloneObject:
305 case FeedbackSlotKind::kLoadProperty:
306 case FeedbackSlotKind::kLoadKeyed:
307 case FeedbackSlotKind::kHasKeyed:
308 case FeedbackSlotKind::kSetNamedSloppy:
309 case FeedbackSlotKind::kSetNamedStrict:
310 case FeedbackSlotKind::kDefineNamedOwn:
311 case FeedbackSlotKind::kDefineKeyedOwn:
312 case FeedbackSlotKind::kSetKeyedSloppy:
313 case FeedbackSlotKind::kSetKeyedStrict:
314 case FeedbackSlotKind::kStoreInArrayLiteral:
315 case FeedbackSlotKind::kDefineKeyedOwnPropertyInLiteral:
316 case FeedbackSlotKind::kInstanceOf:
317 vector->Set(slot, *uninitialized_sentinel, SKIP_WRITE_BARRIER);
318 break;
319
320 case FeedbackSlotKind::kInvalid:
321 UNREACHABLE();
322 }
323 for (int j = 1; j < entry_size; j++) {
324 vector->Set(slot.WithOffset(j), extra_value, SKIP_WRITE_BARRIER);
325 }
326 i += entry_size;
327 }
328
329 if (!isolate->is_best_effort_code_coverage()) {
330 AddToVectorsForProfilingTools(isolate, vector);
331 }
332 parent_feedback_cell->set_value(*vector, kReleaseStore);
333 return vector;
334}

392void FeedbackVector::SetOptimizedCode(IsolateForSandbox isolate,
393 Tagged<Code> code) {
394 DCHECK(CodeKindIsOptimizedJSFunction(code->kind()));
395 int32_t state = flags();
396 // Skip setting optimized code if it would cause us to tier down.
397 if (!has_optimized_code()) {
398 state = MaybeHasTurbofanCodeBit::update(state, false);
399 } else if (!CodeKindCanTierUp(optimized_code(isolate)->kind()) ||
400 optimized_code(isolate)->kind() > code->kind()) {
401 if (!v8_flags.stress_concurrent_inlining_attach_code &&
402 !optimized_code(isolate)->marked_for_deoptimization()) {
403 return;
404 }
405 // If we fall through, we may be tiering down. This is fine since we only do
406 // that when the current code is marked for deoptimization, or because we're
407 // stress testing.
408 state = MaybeHasTurbofanCodeBit::update(state, false);
409 }
410 // TODO(mythria): We could see a CompileOptimized state here either from
411 // tests that use %OptimizeFunctionOnNextCall, --always-turbofan or because we
412 // re-mark the function for non-concurrent optimization after an OSR. We
413 // should avoid these cases and also check that marker isn't
414 // TieringState::kRequestTurbofan*.
415 set_maybe_optimized_code(MakeWeak(code->wrapper()));
416 // TODO(leszeks): Reconsider whether this could clear the tiering state vs.
417 // the callers doing so.
418 state = TieringStateBits::update(state, TieringState::kNone);
419 if (code->is_maglevved()) {
420 DCHECK(!MaybeHasTurbofanCodeBit::decode(state));
421 state = MaybeHasMaglevCodeBit::update(state, true);
422 } else {
423 DCHECK(code->is_turbofanned());
424 state = MaybeHasTurbofanCodeBit::update(state, true);
425 state = MaybeHasMaglevCodeBit::update(state, false);
426 }
427 set_flags(state);
428}

497bool FeedbackVector::ClearSlots(Isolate* isolate, ClearBehavior behavior) {
498 if (!shared_function_info()->HasFeedbackMetadata()) return false;
499 Tagged<MaybeObject> uninitialized_sentinel =
500 FeedbackVector::RawUninitializedSentinel(isolate);
501
502 bool feedback_updated = false;
503 FeedbackMetadataIterator iter(metadata());
504 while (iter.HasNext()) {
505 FeedbackSlot slot = iter.Next();
506
507 Tagged<MaybeObject> obj = Get(slot);
508 if (obj != uninitialized_sentinel) {
509 FeedbackNexus nexus(isolate, *this, slot);
510 feedback_updated |= nexus.Clear(behavior);
511 }
512 }
513 return feedback_updated;
514}

652 case FeedbackSlotKind::kCompareOp:
653 case FeedbackSlotKind::kForIn:
654 case FeedbackSlotKind::kBinaryOp:
655 case FeedbackSlotKind::kTypeOf:

675 case FeedbackSlotKind::kSetKeyedStrict:

680 case FeedbackSlotKind::kLoadKeyed:

746InlineCacheState FeedbackNexus::ic_state() const {
747 Tagged<MaybeObject> feedback, extra;
748 std::tie(feedback, extra) = GetFeedbackPair();
749
750 switch (kind()) {
751 case FeedbackSlotKind::kLiteral:
752 if (IsSmi(feedback)) return InlineCacheState::UNINITIALIZED;
753 return InlineCacheState::MONOMORPHIC;
754
755 case FeedbackSlotKind::kStoreGlobalSloppy:
756 case FeedbackSlotKind::kStoreGlobalStrict:
757 case FeedbackSlotKind::kLoadGlobalNotInsideTypeof:
758 case FeedbackSlotKind::kLoadGlobalInsideTypeof:
759 case FeedbackSlotKind::kJumpLoop: {
760 if (IsSmi(feedback)) return InlineCacheState::MONOMORPHIC;
761
762 DCHECK(feedback.IsWeakOrCleared());
763 if (!feedback.IsCleared() || extra != UninitializedSentinel()) {
764 return InlineCacheState::MONOMORPHIC;
765 }
766 return InlineCacheState::UNINITIALIZED;
767 }
768
769 case FeedbackSlotKind::kSetNamedSloppy:
770 case FeedbackSlotKind::kSetNamedStrict:
771 case FeedbackSlotKind::kSetKeyedSloppy:
772 case FeedbackSlotKind::kSetKeyedStrict:
773 case FeedbackSlotKind::kStoreInArrayLiteral:
774 case FeedbackSlotKind::kDefineNamedOwn:
775 case FeedbackSlotKind::kDefineKeyedOwn:
776 case FeedbackSlotKind::kLoadProperty:
777 case FeedbackSlotKind::kLoadKeyed:
778 case FeedbackSlotKind::kHasKeyed: {
779 if (feedback == UninitializedSentinel()) {
780 return InlineCacheState::UNINITIALIZED;
781 }
782 if (feedback == MegamorphicSentinel()) {
783 return InlineCacheState::MEGAMORPHIC;
784 }
785 if (feedback == MegaDOMSentinel()) {
786 DCHECK(IsLoadICKind(kind()));
787 return InlineCacheState::MEGADOM;
788 }
789 if (feedback.IsWeakOrCleared()) {
790 // Don't check if the map is cleared.
791 return InlineCacheState::MONOMORPHIC;
792 }
793 Tagged<HeapObject> heap_object;
794 if (feedback.GetHeapObjectIfStrong(&heap_object)) {
795 if (IsWeakFixedArray(heap_object)) {
796 // Determine state purely by our structure, don't check if the maps
797 // are cleared.
798 return InlineCacheState::POLYMORPHIC;
799 }
800 if (IsName(heap_object)) {
801 DCHECK(IsKeyedLoadICKind(kind()) || IsKeyedStoreICKind(kind()) ||
802 IsKeyedHasICKind(kind()) || IsDefineKeyedOwnICKind(kind()));
803 Tagged<Object> extra_object = extra.GetHeapObjectAssumeStrong();
804 Tagged<WeakFixedArray> extra_array =
805 Cast<WeakFixedArray>(extra_object);
806 return extra_array->length() > 2 ? InlineCacheState::POLYMORPHIC
807 : InlineCacheState::MONOMORPHIC;
808 }
809 }
810 // TODO(1393773): Remove once the issue is solved.
811 Address vector_ptr = vector().ptr();
812 config_.isolate()->PushParamsAndDie(
813 reinterpret_cast<void*>(feedback.ptr()),
814 reinterpret_cast<void*>(extra.ptr()),
815 reinterpret_cast<void*>(vector_ptr),
816 reinterpret_cast<void*>(static_cast<intptr_t>(slot_.ToInt())),
817 reinterpret_cast<void*>(static_cast<intptr_t>(kind())),
818 // Include part of the feedback vector containing the slot.
819 reinterpret_cast<void*>(
820 vector_ptr + FeedbackVector::OffsetOfElementAt(slot_.ToInt())));
821 UNREACHABLE();
822 }
823 case FeedbackSlotKind::kCall: {
824 Tagged<HeapObject> heap_object;
825 if (feedback == MegamorphicSentinel()) {
826 return InlineCacheState::GENERIC;
827 } else if (feedback.IsWeakOrCleared()) {
828 if (feedback.GetHeapObjectIfWeak(&heap_object)) {
829 if (IsFeedbackCell(heap_object)) {
830 return InlineCacheState::POLYMORPHIC;
831 }
832 CHECK(IsJSFunction(heap_object) || IsJSBoundFunction(heap_object));
833 }
834 return InlineCacheState::MONOMORPHIC;
835 } else if (feedback.GetHeapObjectIfStrong(&heap_object) &&
836 IsAllocationSite(heap_object)) {
837 return InlineCacheState::MONOMORPHIC;
838 }
839
840 CHECK_EQ(feedback, UninitializedSentinel());
841 return InlineCacheState::UNINITIALIZED;
842 }
843 case FeedbackSlotKind::kBinaryOp: {
844 BinaryOperationHint hint = GetBinaryOperationFeedback();
845 if (hint == BinaryOperationHint::kNone) {
846 return InlineCacheState::UNINITIALIZED;
847 } else if (hint == BinaryOperationHint::kAny) {
848 return InlineCacheState::GENERIC;
849 }
850
851 return InlineCacheState::MONOMORPHIC;
852 }
853 case FeedbackSlotKind::kCompareOp: {
854 CompareOperationHint hint = GetCompareOperationFeedback();
855 if (hint == CompareOperationHint::kNone) {
856 return InlineCacheState::UNINITIALIZED;
857 } else if (hint == CompareOperationHint::kAny) {
858 return InlineCacheState::GENERIC;
859 }
860
861 return InlineCacheState::MONOMORPHIC;
862 }
863 case FeedbackSlotKind::kForIn: {
864 ForInHint hint = GetForInFeedback();
865 if (hint == ForInHint::kNone) {
866 return InlineCacheState::UNINITIALIZED;
867 } else if (hint == ForInHint::kAny) {
868 return InlineCacheState::GENERIC;
869 }
870 return InlineCacheState::MONOMORPHIC;
871 }
872 case FeedbackSlotKind::kTypeOf: {
873 if (feedback == Smi::zero()) {
874 return InlineCacheState::UNINITIALIZED;
875 } else if (feedback == Smi::FromInt(TypeOfFeedback::kAny)) {
876 return InlineCacheState::MEGAMORPHIC;
877 } else if (base::bits::CountPopulation(
878 static_cast<uint32_t>(feedback.ToSmi().value())) == 1) {
879 return InlineCacheState::MONOMORPHIC;
880 } else {
881 return InlineCacheState::POLYMORPHIC;
882 }
883 }
884 case FeedbackSlotKind::kInstanceOf: {
885 if (feedback == UninitializedSentinel()) {
886 return InlineCacheState::UNINITIALIZED;
887 } else if (feedback == MegamorphicSentinel()) {
888 return InlineCacheState::MEGAMORPHIC;
889 }
890 return InlineCacheState::MONOMORPHIC;
891 }
892 case FeedbackSlotKind::kDefineKeyedOwnPropertyInLiteral: {
893 if (feedback == UninitializedSentinel()) {
894 return InlineCacheState::UNINITIALIZED;
895 } else if (feedback.IsWeakOrCleared()) {
896 // Don't check if the map is cleared.
897 return InlineCacheState::MONOMORPHIC;
898 }
899
900 return InlineCacheState::MEGAMORPHIC;
901 }
902
903 case FeedbackSlotKind::kCloneObject: {
904 if (feedback == UninitializedSentinel()) {
905 return InlineCacheState::UNINITIALIZED;
906 }
907 if (feedback == MegamorphicSentinel()) {
908 return InlineCacheState::MEGAMORPHIC;
909 }
910 if (feedback.IsWeakOrCleared()) {
911 return InlineCacheState::MONOMORPHIC;
912 }
913
914 DCHECK(IsWeakFixedArray(feedback.GetHeapObjectAssumeStrong()));
915 return InlineCacheState::POLYMORPHIC;
916 }
917
918 case FeedbackSlotKind::kInvalid:
919 UNREACHABLE();
920 }
921 return InlineCacheState::UNINITIALIZED;
922}

971void FeedbackNexus::ConfigureCloneObject(
972 DirectHandle<Map> source_map, const MaybeObjectHandle& handler_handle) {
973 // TODO(olivf): Introduce a CloneHandler to deal with all the logic of this
974 // state machine which is now spread between Runtime_CloneObjectIC_Miss and
975 // this method.
976 auto GetHandler = [=]() {
977 if (IsSmi(*handler_handle)) {
978 return *handler_handle;
979 }
980 return MakeWeak(*handler_handle);
981 };
982 DCHECK(config()->can_write());
983 Isolate* isolate = config()->isolate();
984 DirectHandle<HeapObject> feedback;
985 {
986 Tagged<MaybeObject> maybe_feedback = GetFeedback();
987 if (maybe_feedback.IsStrongOrWeak()) {
988 feedback = direct_handle(maybe_feedback.GetHeapObject(), isolate);
989 } else {
990 DCHECK(maybe_feedback.IsCleared());
991 }
992 }
993 switch (ic_state()) {
994 case InlineCacheState::UNINITIALIZED:
995 // Cache the first map seen which meets the fast case requirements.
996 SetFeedback(MakeWeak(*source_map), UPDATE_WRITE_BARRIER, GetHandler());
997 break;
998 case InlineCacheState::MONOMORPHIC:
999 if (feedback.is_null() || feedback.is_identical_to(source_map) ||
1000 Cast<Map>(*feedback)->is_deprecated()) {
1001 SetFeedback(MakeWeak(*source_map), UPDATE_WRITE_BARRIER, GetHandler());
1002 } else {
1003 // Transition to POLYMORPHIC.
1004 DirectHandle<WeakFixedArray> array =
1005 CreateArrayOfSize(2 * kCloneObjectPolymorphicEntrySize);
1006 DisallowGarbageCollection no_gc;
1007 Tagged<WeakFixedArray> raw_array = *array;
1008 raw_array->set(0, MakeWeak(*feedback));
1009 raw_array->set(1, GetFeedbackExtra());
1010 raw_array->set(2, MakeWeak(*source_map));
1011 raw_array->set(3, GetHandler());
1012 SetFeedback(raw_array, UPDATE_WRITE_BARRIER, ClearedValue(isolate));
1013 }
1014 break;
1015 case InlineCacheState::POLYMORPHIC: {
1016 const int kMaxElements = v8_flags.max_valid_polymorphic_map_count *
1017 kCloneObjectPolymorphicEntrySize;
1018 DirectHandle<WeakFixedArray> array = Cast<WeakFixedArray>(feedback);
1019 int i = 0;
1020 for (; i < array->length(); i += kCloneObjectPolymorphicEntrySize) {
1021 Tagged<MaybeObject> feedback_map = array->get(i);
1022 if (feedback_map.IsCleared()) break;
1023 DirectHandle<Map> cached_map(Cast<Map>(feedback_map.GetHeapObject()),
1024 isolate);
1025 if (cached_map.is_identical_to(source_map) ||
1026 cached_map->is_deprecated())
1027 break;
1028 }
1029
1030 if (i >= array->length()) {
1031 if (i == kMaxElements) {
1032 // Transition to MEGAMORPHIC.
1033 Tagged<MaybeObject> sentinel = MegamorphicSentinel();
1034 SetFeedback(sentinel, SKIP_WRITE_BARRIER, ClearedValue(isolate));
1035 break;
1036 }
1037
1038 // Grow polymorphic feedback array.
1039 DirectHandle<WeakFixedArray> new_array = CreateArrayOfSize(
1040 array->length() + kCloneObjectPolymorphicEntrySize);
1041 for (int j = 0; j < array->length(); ++j) {
1042 new_array->set(j, array->get(j));
1043 }
1044 SetFeedback(*new_array);
1045 array = new_array;
1046 }
1047
1048 array->set(i, MakeWeak(*source_map));
1049 array->set(i + 1, GetHandler());
1050 break;
1051 }
1052
1053 default:
1054 UNREACHABLE();
1055 }
1056}

1294KeyedAccessStoreMode FeedbackNexus::GetKeyedAccessStoreMode() const {
1295 DCHECK(IsKeyedStoreICKind(kind()) || IsStoreInArrayLiteralICKind(kind()) ||
1296 IsDefineKeyedOwnPropertyInLiteralKind(kind()) ||
1297 IsDefineKeyedOwnICKind(kind()));
1298 KeyedAccessStoreMode mode = KeyedAccessStoreMode::kInBounds;
1299
1300 if (GetKeyType() == IcCheckType::kProperty) return mode;
1301
1302 MapsAndHandlers maps_and_handlers;
1303 ExtractMapsAndHandlers(&maps_and_handlers);
1304 for (const MapAndHandler& map_and_handler : maps_and_handlers) {
1305 const MaybeObjectHandle maybe_code_handler = map_and_handler.second;
1306 // The first handler that isn't the slow handler will have the bits we need.
1307 Builtin builtin_handler = Builtin::kNoBuiltinId;
1308 if (IsStoreHandler(*maybe_code_handler.object())) {
1309 auto data_handler = Cast<StoreHandler>(maybe_code_handler.object());
1310
1311 if (IsSmi(data_handler->smi_handler())) {
1312 // Decode the KeyedAccessStoreMode information from the Handler.
1313 mode =
1314 StoreHandler::GetKeyedAccessStoreMode(data_handler->smi_handler());
1315 if (!StoreModeIsInBounds(mode)) return mode;
1316 continue;
1317 } else {
1318 Tagged<Code> code = Cast<Code>(data_handler->smi_handler());
1319 builtin_handler = code->builtin_id();
1320 }
1321
1322 } else if (IsSmi(*maybe_code_handler.object())) {
1323 // Skip for Proxy Handlers.
1324 if (*maybe_code_handler.object() == StoreHandler::StoreProxy()) {
1325 continue;
1326 }
1327 // Decode the KeyedAccessStoreMode information from the Handler.
1328 mode = StoreHandler::GetKeyedAccessStoreMode(*maybe_code_handler);
1329 if (!StoreModeIsInBounds(mode)) return mode;
1330 continue;
1331 } else if (IsDefineKeyedOwnICKind(kind())) {
1332 mode = StoreHandler::GetKeyedAccessStoreMode(*maybe_code_handler);
1333 if (!StoreModeIsInBounds(mode)) return mode;
1334 continue;
1335 } else {
1336 // Element store without prototype chain check.
1337 Tagged<Code> code = Cast<Code>(*maybe_code_handler.object());
1338 builtin_handler = code->builtin_id();
1339 }
1340
1341 if (Builtins::IsBuiltinId(builtin_handler)) {
1342 if (!BuiltinHasKeyedAccessStoreMode(builtin_handler)) continue;
1343
1344 mode = KeyedAccessStoreModeForBuiltin(builtin_handler);
1345 break;
1346 }
1347 }
