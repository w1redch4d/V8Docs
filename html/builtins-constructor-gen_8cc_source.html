209TF_BUILTIN(FastNewClosure, ConstructorBuiltinsAssembler) {
210 auto shared_function_info =
211 Parameter<SharedFunctionInfo>(Descriptor::kSharedFunctionInfo);
212 auto feedback_cell = Parameter<FeedbackCell>(Descriptor::kFeedbackCell);
213 auto context = Parameter<Context>(Descriptor::kContext);
214
215 // Bump the closure counter encoded in the {feedback_cell}s map.
216 {
217 const TNode<Map> feedback_cell_map = LoadMap(feedback_cell);
218 Label no_closures(this), one_closure(this), cell_done(this);
219
220 GotoIf(IsNoClosuresCellMap(feedback_cell_map), &no_closures);
221 GotoIf(IsOneClosureCellMap(feedback_cell_map), &one_closure);
222 CSA_DCHECK(this, IsManyClosuresCellMap(feedback_cell_map),
223 feedback_cell_map, feedback_cell);
224 Goto(&cell_done);
225
226 BIND(&no_closures);
227 StoreMapNoWriteBarrier(feedback_cell, RootIndex::kOneClosureCellMap);
228 Goto(&cell_done);
229
230 BIND(&one_closure);
231#ifdef V8_ENABLE_LEAPTIERING
232 // The transition from one to many closures under leap tiering requires
233 // making sure that the dispatch_handle's code isn't context specialized for
234 // the single closure. This is handled in the runtime.
235 //
236 // TODO(leszeks): We could fast path this for the case where the dispatch
237 // handle either doesn't contain any code, or that code isn't context
238 // specialized.
239 TailCallRuntime(Runtime::kNewClosure, context, shared_function_info,
240 feedback_cell);
241#else
242 StoreMapNoWriteBarrier(feedback_cell, RootIndex::kManyClosuresCellMap);
243 Goto(&cell_done);
244#endif // V8_ENABLE_LEAPTIERING
245
246 BIND(&cell_done);
247 }
248
249 // The calculation of |function_map_index| must be in sync with
250 // SharedFunctionInfo::function_map_index().
251 TNode<Uint32T> flags = LoadObjectField<Uint32T>(
252 shared_function_info, SharedFunctionInfo::kFlagsOffset);
253 const TNode<IntPtrT> function_map_index = Signed(IntPtrAdd(
254 DecodeWordFromWord32<SharedFunctionInfo::FunctionMapIndexBits>(flags),
255 IntPtrConstant(Context::FIRST_FUNCTION_MAP_INDEX)));
256 CSA_DCHECK(this, UintPtrLessThanOrEqual(
257 function_map_index,
258 IntPtrConstant(Context::LAST_FUNCTION_MAP_INDEX)));
259
260 // Get the function map in the current native context and set that
261 // as the map of the allocated object.
262 const TNode<NativeContext> native_context = LoadNativeContext(context);
263 const TNode<Map> function_map =
264 CAST(LoadContextElement(native_context, function_map_index));
265
266 // Create a new closure from the given function info in new space
267 TNode<IntPtrT> instance_size_in_bytes =
268 TimesTaggedSize(LoadMapInstanceSizeInWords(function_map));
269 TNode<HeapObject> result = Allocate(instance_size_in_bytes);
270 StoreMapNoWriteBarrier(result, function_map);
271 InitializeJSObjectBodyNoSlackTracking(result, function_map,
272 instance_size_in_bytes,
273 JSFunction::kSizeWithoutPrototype);
274
275 // Initialize the rest of the function.
276 StoreObjectFieldRoot(result, JSObject::kPropertiesOrHashOffset,
277 RootIndex::kEmptyFixedArray);
278 StoreObjectFieldRoot(result, JSObject::kElementsOffset,
279 RootIndex::kEmptyFixedArray);
280 {
281 // Set function prototype if necessary.
282 Label done(this), init_prototype(this);
283 Branch(IsFunctionWithPrototypeSlotMap(function_map), &init_prototype,
284 &done);
285
286 BIND(&init_prototype);
287 StoreObjectFieldRoot(result, JSFunction::kPrototypeOrInitialMapOffset,
288 RootIndex::kTheHoleValue);
289 Goto(&done);
290 BIND(&done);
291 }
292
293 static_assert(JSFunction::kSizeWithoutPrototype == 7 * kTaggedSize);
294 StoreObjectFieldNoWriteBarrier(result, JSFunction::kFeedbackCellOffset,
295 feedback_cell);
296 StoreObjectFieldNoWriteBarrier(result, JSFunction::kSharedFunctionInfoOffset,
297 shared_function_info);
298 StoreObjectFieldNoWriteBarrier(result, JSFunction::kContextOffset, context);
299#ifdef V8_ENABLE_LEAPTIERING
300 TNode<JSDispatchHandleT> dispatch_handle = LoadObjectField<JSDispatchHandleT>(
301 feedback_cell, FeedbackCell::kDispatchHandleOffset);
302 CSA_DCHECK(this,
303 Word32NotEqual(dispatch_handle,
304 Int32Constant(kNullJSDispatchHandle.value())));
305 StoreObjectFieldNoWriteBarrier(result, JSFunction::kDispatchHandleOffset,
306 dispatch_handle);
307#else
308 TNode<Code> lazy_builtin =
309 HeapConstantNoHole(BUILTIN_CODE(isolate(), CompileLazy));
310 StoreCodePointerField(result, JSFunction::kCodeOffset, lazy_builtin);
311#endif // V8_ENABLE_LEAPTIERING
312 Return(result);
313}

347TNode<JSObject> ConstructorBuiltinsAssembler::FastNewObject(
348 TNode<Context> context, TNode<JSFunction> target,
349 TNode<JSReceiver> new_target, Label* call_runtime) {
350 // Verify that the new target is a JSFunction.
351 Label end(this);
352 TNode<JSFunction> new_target_func =
353 HeapObjectToJSFunctionWithPrototypeSlot(new_target, call_runtime);
354 // Fast path.
355
356 // Load the initial map and verify that it's in fact a map.
357 TNode<Object> initial_map_or_proto =
358 LoadJSFunctionPrototypeOrInitialMap(new_target_func);
359 GotoIf(TaggedIsSmi(initial_map_or_proto), call_runtime);
360 GotoIf(DoesntHaveInstanceType(CAST(initial_map_or_proto), MAP_TYPE),
361 call_runtime);
362 TNode<Map> initial_map = CAST(initial_map_or_proto);
363
364 // Fall back to runtime if the target differs from the new target's
365 // initial map constructor.
366 TNode<Object> new_target_constructor = LoadObjectField(
367 initial_map, Map::kConstructorOrBackPointerOrNativeContextOffset);
368 GotoIf(TaggedNotEqual(target, new_target_constructor), call_runtime);
369
370 TVARIABLE(HeapObject, properties);
371
372 Label instantiate_map(this), allocate_properties(this);
373 GotoIf(IsDictionaryMap(initial_map), &allocate_properties);
374 {
375 properties = EmptyFixedArrayConstant();
376 Goto(&instantiate_map);
377 }
378 BIND(&allocate_properties);
379 {
380 properties =
381 AllocatePropertyDictionary(PropertyDictionary::kInitialCapacity);
382 Goto(&instantiate_map);
383 }
384
385 BIND(&instantiate_map);
386 return AllocateJSObjectFromMap(initial_map, properties.value(), std::nullopt,
387 AllocationFlag::kNone, kWithSlackTracking);
388}

390TNode<Context> ConstructorBuiltinsAssembler::FastNewFunctionContext(
391 TNode<ScopeInfo> scope_info, TNode<Uint32T> slots, TNode<Context> context,
392 ScopeType scope_type) {
393 TNode<IntPtrT> slots_intptr = Signed(ChangeUint32ToWord(slots));
394 TNode<IntPtrT> size = ElementOffsetFromIndex(slots_intptr, PACKED_ELEMENTS,
395 Context::kTodoHeaderSize);
396
397 // Create a new closure from the given function info in new space
398 TNode<Context> function_context =
399 UncheckedCast<Context>(AllocateInNewSpace(size));
400
401 TNode<NativeContext> native_context = LoadNativeContext(context);
402 Context::Field index;
403 switch (scope_type) {
404 case EVAL_SCOPE:
405 index = Context::EVAL_CONTEXT_MAP_INDEX;
406 break;
407 case FUNCTION_SCOPE:
408 index = Context::FUNCTION_CONTEXT_MAP_INDEX;
409 break;
410 default:
411 UNREACHABLE();
412 }
413 TNode<Map> map = CAST(LoadContextElement(native_context, index));
414 // Set up the header.
415 StoreMapNoWriteBarrier(function_context, map);
416 TNode<IntPtrT> min_context_slots = IntPtrConstant(Context::MIN_CONTEXT_SLOTS);
417 // TODO(ishell): for now, length also includes MIN_CONTEXT_SLOTS.
418 TNode<IntPtrT> length = IntPtrAdd(slots_intptr, min_context_slots);
419 StoreObjectFieldNoWriteBarrier(function_context, Context::kLengthOffset,
420 SmiTag(length));
421 StoreObjectFieldNoWriteBarrier(function_context, Context::kScopeInfoOffset,
422 scope_info);
423 StoreObjectFieldNoWriteBarrier(function_context, Context::kPreviousOffset,
424 context);
425
426 // Initialize the varrest of the slots to undefined.
427 TNode<Oddball> undefined = UndefinedConstant();
428 TNode<IntPtrT> start_offset = IntPtrConstant(Context::kTodoHeaderSize);
429 CodeStubAssembler::VariableList vars(0, zone());
430 BuildFastLoop<IntPtrT>(
431 vars, start_offset, size,
432 [=, this](TNode<IntPtrT> offset) {
433 StoreObjectFieldNoWriteBarrier(function_context, offset, undefined);
434 },
435 kTaggedSize, LoopUnrollingMode::kYes, IndexAdvanceMode::kPost);
436 return function_context;
437}

439TNode<JSRegExp> ConstructorBuiltinsAssembler::CreateRegExpLiteral(
440 TNode<HeapObject> maybe_feedback_vector, TNode<TaggedIndex> slot,
441 TNode<Object> pattern, TNode<Smi> flags, TNode<Context> context) {
442 Label call_runtime(this, Label::kDeferred), end(this);
443
444 GotoIf(IsUndefined(maybe_feedback_vector), &call_runtime);
445
446 TVARIABLE(JSRegExp, result);
447 TNode<FeedbackVector> feedback_vector = CAST(maybe_feedback_vector);
448 TNode<Object> literal_site =
449 CAST(LoadFeedbackVectorSlot(feedback_vector, slot));
450 GotoIfNot(HasBoilerplate(literal_site), &call_runtime);
451 {
452 static_assert(JSRegExp::kDataOffset == JSObject::kHeaderSize);
453 static_assert(JSRegExp::kSourceOffset ==
454 JSRegExp::kDataOffset + kTaggedSize);
455 static_assert(JSRegExp::kFlagsOffset ==
456 JSRegExp::kSourceOffset + kTaggedSize);
457 static_assert(JSRegExp::kHeaderSize ==
458 JSRegExp::kFlagsOffset + kTaggedSize);
459 static_assert(JSRegExp::kLastIndexOffset == JSRegExp::kHeaderSize);
460 DCHECK_EQ(JSRegExp::Size(), JSRegExp::kLastIndexOffset + kTaggedSize);
461
462 TNode<RegExpBoilerplateDescription> boilerplate = CAST(literal_site);
463 TNode<HeapObject> new_object = Allocate(JSRegExp::Size());
464
465 // Initialize Object fields.
466 TNode<JSFunction> regexp_function = CAST(LoadContextElement(
467 LoadNativeContext(context), Context::REGEXP_FUNCTION_INDEX));
468 TNode<Map> initial_map = CAST(LoadObjectField(
469 regexp_function, JSFunction::kPrototypeOrInitialMapOffset));
470 StoreMapNoWriteBarrier(new_object, initial_map);
471 // Initialize JSReceiver fields.
472 StoreObjectFieldRoot(new_object, JSReceiver::kPropertiesOrHashOffset,
473 RootIndex::kEmptyFixedArray);
474 // Initialize JSObject fields.
475 StoreObjectFieldRoot(new_object, JSObject::kElementsOffset,
476 RootIndex::kEmptyFixedArray);
477 // Initialize JSRegExp fields.
478 StoreTrustedPointerField(
479 new_object, JSRegExp::kDataOffset, kRegExpDataIndirectPointerTag,
480 CAST(LoadTrustedPointerFromObject(
481 boilerplate, RegExpBoilerplateDescription::kDataOffset,
482 kRegExpDataIndirectPointerTag)));
483 StoreObjectFieldNoWriteBarrier(
484 new_object, JSRegExp::kSourceOffset,
485 LoadObjectField(boilerplate,
486 RegExpBoilerplateDescription::kSourceOffset));
487 StoreObjectFieldNoWriteBarrier(
488 new_object, JSRegExp::kFlagsOffset,
489 LoadObjectField(boilerplate,
490 RegExpBoilerplateDescription::kFlagsOffset));
491 StoreObjectFieldNoWriteBarrier(
492 new_object, JSRegExp::kLastIndexOffset,
493 SmiConstant(JSRegExp::kInitialLastIndexValue));
494
495 result = CAST(new_object);
496 Goto(&end);
497 }
498
499 BIND(&call_runtime);
500 {
501 result = CAST(CallRuntime(Runtime::kCreateRegExpLiteral, context,
502 maybe_feedback_vector, slot, pattern, flags));
503 Goto(&end);
504 }
505
506 BIND(&end);
507 return result.value();
508}

533 // elements transitions.
534 TNode<Object> maybe_allocation_site =
535 CAST(LoadFeedbackVectorSlot(feedback_vector, slot));
536 TVARIABLE(AllocationSite, allocation_site);

589 TNode<AllocationSite> allocation_site, TNode<JSObject> boilerplate,
590 Label* call_runtime, bool bailout_if_dictionary_properties) {
591 TNode<Map> boilerplate_map = LoadMap(boilerplate);
592 CSA_DCHECK(this, IsJSObjectMap(boilerplate_map));
593
594 TVARIABLE(HeapObject, var_properties);
595 {
596 TNode<Uint32T> bit_field_3 = LoadMapBitField3(boilerplate_map);
597 GotoIf(IsSetWord32<Map::Bits3::IsDeprecatedBit>(bit_field_3), call_runtime);
598 // Directly copy over the property store for dict-mode boilerplates.
599 Label if_dictionary(this), if_fast(this), done(this);
600 Branch(IsSetWord32<Map::Bits3::IsDictionaryMapBit>(bit_field_3),
601 &if_dictionary, &if_fast);
602 BIND(&if_dictionary);
603 {
604 if (bailout_if_dictionary_properties) {
605 Goto(call_runtime);
606 } else {
607 Comment("Copy dictionary properties");
608 if (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
609 var_properties =
610 CopySwissNameDictionary(CAST(LoadSlowProperties(boilerplate)));
611 } else {
612 var_properties = CopyNameDictionary(
613 CAST(LoadSlowProperties(boilerplate)), call_runtime);
614 }
615 // Slow objects have no in-object properties.
616 Goto(&done);
617 }
618 }
619 BIND(&if_fast);
620 {
621 // TODO(cbruni): support copying out-of-object properties.
622 // (CreateObjectFromSlowBoilerplate needs to handle them, too.)
623 TNode<HeapObject> boilerplate_properties =
624 LoadFastProperties(boilerplate);
625 GotoIfNot(IsEmptyFixedArray(boilerplate_properties), call_runtime);
626 var_properties = EmptyFixedArrayConstant();
627 Goto(&done);
628 }
629 BIND(&done);
630 }
631
632 TVARIABLE(FixedArrayBase, var_elements);
633 {
634 // Copy the elements backing store, assuming that it's flat.
635 Label if_empty_fixed_array(this), if_copy_elements(this), done(this);
636 TNode<FixedArrayBase> boilerplate_elements = LoadElements(boilerplate);
637 Branch(IsEmptyFixedArray(boilerplate_elements), &if_empty_fixed_array,
638 &if_copy_elements);
639
640 BIND(&if_empty_fixed_array);
641 var_elements = boilerplate_elements;
642 Goto(&done);
643
644 BIND(&if_copy_elements);
645 CSA_DCHECK(this, Word32BinaryNot(
646 IsFixedCOWArrayMap(LoadMap(boilerplate_elements))));
647 auto flags = ExtractFixedArrayFlag::kAllFixedArrays;
648 var_elements = CloneFixedArray(boilerplate_elements, flags);
649 Goto(&done);
650 BIND(&done);
651 }
652
653 // Ensure new-space allocation for a fresh JSObject so we can skip write
654 // barriers when copying all object fields.
655 static_assert(JSObject::kMaxInstanceSize < kMaxRegularHeapObjectSize);
656 TNode<IntPtrT> instance_size =
657 TimesTaggedSize(LoadMapInstanceSizeInWords(boilerplate_map));
658 TNode<IntPtrT> aligned_instance_size =
659 AlignToAllocationAlignment(instance_size);
660 TNode<IntPtrT> allocation_size = instance_size;
661 bool needs_allocation_memento = v8_flags.allocation_site_pretenuring;
662 if (needs_allocation_memento) {
663 DCHECK(V8_ALLOCATION_SITE_TRACKING_BOOL);
664 // Prepare for inner-allocating the AllocationMemento.
665 allocation_size = IntPtrAdd(aligned_instance_size,
666 IntPtrConstant(ALIGN_TO_ALLOCATION_ALIGNMENT(
667 AllocationMemento::kSize)));
668 }
669
670 TNode<HeapObject> copy =
671 UncheckedCast<HeapObject>(AllocateInNewSpace(allocation_size));
672 {
673 Comment("Initialize Literal Copy");
674 // Initialize Object fields.
675 StoreMapNoWriteBarrier(copy, boilerplate_map);
676 StoreObjectFieldNoWriteBarrier(copy, JSObject::kPropertiesOrHashOffset,
677 var_properties.value());
678 StoreObjectFieldNoWriteBarrier(copy, JSObject::kElementsOffset,
679 var_elements.value());
680 }
681
682 // Initialize the AllocationMemento before potential GCs due to heap number
683 // allocation when copying the in-object properties.
684 if (needs_allocation_memento) {
685 InitializeAllocationMemento(copy, aligned_instance_size, allocation_site);
686 }
687
688 {
689 // Copy over in-object properties.
690 Label continue_with_write_barrier(this), done_init(this);
691 TVARIABLE(IntPtrT, offset, IntPtrConstant(JSObject::kHeaderSize));
692 {
693 Comment("Copy in-object properties fast");
694 Label continue_fast(this, &offset);
695 Branch(IntPtrEqual(offset.value(), instance_size), &done_init,
696 &continue_fast);
697 BIND(&continue_fast);
698 TNode<Object> field = LoadObjectField(boilerplate, offset.value());
699 Label store_field(this);
700 GotoIf(TaggedIsSmi(field), &store_field);
701 // TODO(leszeks): Read the field descriptor to decide if this heap
702 // number is mutable or not.
703 GotoIf(IsHeapNumber(CAST(field)), &continue_with_write_barrier);
704 Goto(&store_field);
705 BIND(&store_field);
706 StoreObjectFieldNoWriteBarrier(copy, offset.value(), field);
707 offset = IntPtrAdd(offset.value(), IntPtrConstant(kTaggedSize));
708 Branch(WordNotEqual(offset.value(), instance_size), &continue_fast,
709 &done_init);
710 }
711
712 // Continue initializing the literal after seeing the first sub-object
713 // potentially causing allocation. In this case we prepare the new literal
714 // by copying all pending fields over from the boilerplate and emit full
715 // write barriers from here on.
716 BIND(&continue_with_write_barrier);
717 {
718 Comment("Copy in-object properties slow");
719 BuildFastLoop<IntPtrT>(
720 offset.value(), instance_size,
721 [=, this](TNode<IntPtrT> offset) {
722 // TODO(ishell): value decompression is not necessary here.
723 TNode<Object> field = LoadObjectField(boilerplate, offset);
724 StoreObjectFieldNoWriteBarrier(copy, offset, field);
725 },
726 kTaggedSize, LoopUnrollingMode::kYes, IndexAdvanceMode::kPost);
727 CopyMutableHeapNumbersInObject(copy, offset.value(), instance_size);
728 Goto(&done_init);
729 }
730 BIND(&done_init);
731 }
732 return copy;
733}

750void ConstructorBuiltinsAssembler::CopyMutableHeapNumbersInObject(
751 TNode<HeapObject> copy, TNode<IntPtrT> start_offset,
752 TNode<IntPtrT> end_offset) {
753 // Iterate over all object properties of a freshly copied object and
754 // duplicate mutable heap numbers.
755 Comment("Copy mutable HeapNumber values");
756 BuildFastLoop<IntPtrT>(
757 start_offset, end_offset,
758 [=, this](TNode<IntPtrT> offset) {
759 TNode<Object> field = LoadObjectField(copy, offset);
760 Label copy_heap_number(this, Label::kDeferred), continue_loop(this);
761 // We only have to clone complex field values.
762 GotoIf(TaggedIsSmi(field), &continue_loop);
763 // TODO(leszeks): Read the field descriptor to decide if this heap
764 // number is mutable or not.
765 Branch(IsHeapNumber(CAST(field)), &copy_heap_number, &continue_loop);
766 BIND(&copy_heap_number);
767 {
768 TNode<Float64T> double_value = LoadHeapNumberValue(CAST(field));
769 TNode<HeapNumber> heap_number =
770 AllocateHeapNumberWithValue(double_value);
771 StoreObjectField(copy, offset, heap_number);
772 Goto(&continue_loop);
773 }
774 BIND(&continue_loop);
775 },
776 kTaggedSize, LoopUnrollingMode::kNo, IndexAdvanceMode::kPost);
777}
