94 struct Payload {
95 Payload(Address pointer, CppHeapPointerTag tag)
96 : encoded_word_(Tag(pointer, tag)) {}
97
98 Address Untag(CppHeapPointerTagRange tag_range) const {
99 Address content = encoded_word_;
100 if (V8_LIKELY(tag_range.CheckTagOf(content))) {
101 content >>= kCppHeapPointerPayloadShift;
102 } else {
103 // If the type check failed, we simply return nullptr here. That way:
104 // 1. The null handle always results in nullptr being returned here,
105 // which is a desired property. Otherwise, we may need an explicit
106 // check for the null handle in the caller, and therefore an
107 // additional branch. This works because the 0th entry of the table
108 // always contains nullptr tagged with the null tag (i.e. an
109 // all-zeros entry). As such, regardless of whether the type check
110 // succeeds, the result will always be nullptr.
111 // 2. The returned pointer is guaranteed to crash even on platforms
112 // with top byte ignore (TBI), such as Arm64. The alternative would
113 // be to simply return the original entry with the left-shifted
114 // payload. However, due to TBI, an access to that may not always
115 // result in a crash (specifically, if the second most significant
116 // byte happens to be zero). In addition, there shouldn't be a
117 // difference on Arm64 between returning nullptr or the original
118 // entry, since it will simply compile to a `csel x0, x8, xzr, lo`
119 // instead of a `csel x0, x10, x8, lo` instruction.
120 content = 0;
121 }
122 return content;
123 }
124
125 Address Untag(CppHeapPointerTag tag) const {
126 return Untag(CppHeapPointerTagRange(tag, tag));
127 }
128
129 static Address Tag(Address pointer, CppHeapPointerTag tag) {
130 return (pointer << kCppHeapPointerPayloadShift) |
131 (static_cast<uint16_t>(tag) << kCppHeapPointerTagShift);
132 }
133
134 bool IsTaggedWithTagIn(CppHeapPointerTagRange tag_range) const {
135 return tag_range.CheckTagOf(encoded_word_);

138 bool IsTaggedWith(CppHeapPointerTag tag) const {
139 return IsTaggedWithTagIn(CppHeapPointerTagRange(tag, tag));

173 bool operator==(Payload other) const {
174 return encoded_word_ == other.encoded_word_;
175 }
176
177 bool operator!=(Payload other) const {
178 return encoded_word_ != other.encoded_word_;
179 }

181 private:
182 Address encoded_word_;
183 };
