51bool RootsTable::IsRootHandleLocation(Address* handle_location,
52 RootIndex* index) const {
53 FullObjectSlot location(handle_location);
54 FullObjectSlot first_root(&roots_[0]);
55 FullObjectSlot last_root(&roots_[kEntriesCount]);
56 if (location >= last_root) return false;
57 if (location < first_root) return false;
58 *index = static_cast<RootIndex>(location - first_root);
59 return true;
60}

62template <typename T>
63bool RootsTable::IsRootHandle(IndirectHandle<T> handle,
64 RootIndex* index) const {
65 // This can't use handle.location() because it is called from places
66 // where handle dereferencing is disallowed. Comparing the handle's
67 // location against the root handle list is safe though.
68 Address* handle_location = reinterpret_cast<Address*>(handle.address());
69 return IsRootHandleLocation(handle_location, index);
70}

145Address ReadOnlyRoots::address_at(RootIndex root_index) const {
146#if V8_STATIC_ROOTS_BOOL
147 return V8HeapCompressionScheme::DecompressTagged(
148 V8HeapCompressionScheme::base(),
149 StaticReadOnlyRootsPointerTable[static_cast<int>(root_index)]);
150#else
151 size_t index = static_cast<size_t>(root_index);
152 DCHECK_LT(index, kEntriesCount);
153 // Filler objects must be created before the free space map is initialized.
154 // Bootstrapping is able to handle kNullAddress being returned here.
155 DCHECK_IMPLIES(read_only_roots_[index] == kNullAddress,
156 root_index == RootIndex::kFreeSpaceMap);
157 return read_only_roots_[index];
158#endif
159}

161bool ReadOnlyRoots::is_initialized(RootIndex root_index) const {
162 size_t index = static_cast<size_t>(root_index);
163 DCHECK_LT(index, kEntriesCount);
164 return read_only_roots_[index] != kNullAddress;
165}
