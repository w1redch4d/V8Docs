186// Each JSArrayBuffer (with a backing store) has a corresponding native-heap
187// allocated ArrayBufferExtension for GC purposes and storing the backing store.
188// When marking a JSArrayBuffer, the GC also marks the native
189// extension-object. The GC periodically iterates all extensions concurrently
190// and frees unmarked ones.
191// https://docs.google.com/document/d/1-ZrLdlFX1nXT3z-FAgLbKal1gI8Auiaya_My-a0UJ28/edit
192class ArrayBufferExtension final
193#ifdef V8_COMPRESS_POINTERS
194 : public ExternalPointerTable::ManagedResource {
195#else
196 : public Malloced {
197#endif // V8_COMPRESS_POINTERS
198 public:
199 enum class Age : uint8_t { kYoung = 0, kOld = 1 };
200
201 // Packs `accounting_length` and `age` into a single integer for consistent
202 // accounting, allowing resize while concurrently sweeping.
203 struct AccountingState final {
204 size_t accounting_length() const {
205 return AccountingLengthField::decode(value);
206 }
207 Age age() const { return static_cast<Age>(AgeField::decode(value)); }
208
209 uint64_t value;
210 };
211
212 ArrayBufferExtension(std::shared_ptr<BackingStore> backing_store,
213 ArrayBufferExtension::Age age)
214 : backing_store_(std::move(backing_store)),
215 accounting_state_(AccountingLengthField::encode(static_cast<size_t>(
216 backing_store_->PerIsolateAccountingLength())) |
217 AgeField::encode(static_cast<uint8_t>(age))) {}
218
219 void Mark() { marked_.store(true, std::memory_order_relaxed); }
220 void Unmark() { marked_.store(false, std::memory_order_relaxed); }
221 bool IsMarked() const { return marked_.load(std::memory_order_relaxed); }
222
223 void YoungMark() { set_young_gc_state(GcState::Copied); }
224 void YoungMarkPromoted() { set_young_gc_state(GcState::Promoted); }
225 void YoungUnmark() { set_young_gc_state(GcState::Dead); }
226 bool IsYoungMarked() const { return young_gc_state() != GcState::Dead; }
227 bool IsYoungPromoted() const { return young_gc_state() == GcState::Promoted; }
228
229 std::shared_ptr<BackingStore> backing_store() { return backing_store_; }
230 void set_backing_store(std::shared_ptr<BackingStore> backing_store) {
231 backing_store_ = std::move(backing_store);
232 }
233 std::shared_ptr<BackingStore> RemoveBackingStore() {
234 return std::move(backing_store_);
235 }
236
237 size_t accounting_length() const {
238 return AccountingState{accounting_state_.load(std::memory_order_relaxed)}
239 .accounting_length();
240 }
241 // Applies `delta` to `accounting_length` and returns the AccountingState
242 // before the update.
243 AccountingState UpdateAccountingLength(int64_t delta) {
244 if (delta >= 0) {
245 return {accounting_state_.fetch_add(
246 AccountingLengthField::encode(static_cast<size_t>(delta)),
247 std::memory_order_relaxed)};
248 }
249 return {accounting_state_.fetch_sub(
250 AccountingLengthField::encode(static_cast<size_t>(-delta)),
251 std::memory_order_relaxed)};
252 }
253 // Clears `accounting_length` and returns the AccountingState before the
254 // update.
255 AccountingState ClearAccountingLength() {
256 return {accounting_state_.fetch_and(AgeField::kMask,
257 std::memory_order_relaxed)};
258 }
259
260 ArrayBufferExtension* next() const { return next_; }
261 void set_next(ArrayBufferExtension* extension) { next_ = extension; }
262
263 Age age() const {
264 return AccountingState{accounting_state_.load(std::memory_order_relaxed)}
265 .age();
266 }
267 // Updates `age` and returns the AccountingState before the update.
268 AccountingState SetOld() {
269 return {
270 accounting_state_.fetch_or(AgeField::kMask, std::memory_order_relaxed)};
271 }
272 AccountingState SetYoung() {
273 return {accounting_state_.fetch_and(~AgeField::kMask,
274 std::memory_order_relaxed)};
275 }
276
277 private:
278 enum class GcState : uint8_t { Dead = 0, Copied, Promoted };
279
280 using AgeField = base::BitField<uint8_t, 0, 1, uint64_t>;
281 using AccountingLengthField = AgeField::Next<size_t, 63>;
282
283 GcState young_gc_state() const {
284 return young_gc_state_.load(std::memory_order_relaxed);
285 }
286
287 void set_young_gc_state(GcState value) {
288 young_gc_state_.store(value, std::memory_order_relaxed);
289 }
290
291 std::shared_ptr<BackingStore> backing_store_;
292 ArrayBufferExtension* next_ = nullptr;
293 std::atomic<uint64_t> accounting_state_;
294 std::atomic<bool> marked_{false};
295 std::atomic<GcState> young_gc_state_{GcState::Dead};
296};
