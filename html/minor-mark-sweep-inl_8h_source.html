36template <typename TSlot>
37void YoungGenerationRootMarkingVisitor::VisitPointersImpl(Root root,
38 TSlot start,
39 TSlot end) {
40 if (root == Root::kStackRoots) {
41 for (TSlot slot = start; slot < end; ++slot) {
42 main_marking_visitor_->VisitObjectViaSlot<
43 YoungGenerationMainMarkingVisitor::ObjectVisitationMode::
44 kPushToWorklist,
45 YoungGenerationMainMarkingVisitor::SlotTreatmentMode::kReadOnly>(
46 slot);
47 }
48 } else {
49 for (TSlot slot = start; slot < end; ++slot) {
50 main_marking_visitor_->VisitObjectViaSlot<
51 YoungGenerationMainMarkingVisitor::ObjectVisitationMode::
52 kPushToWorklist,
53 YoungGenerationMainMarkingVisitor::SlotTreatmentMode::kReadWrite>(
54 slot);
55 }
56 }

59template <typename Visitor>
60bool YoungGenerationRememberedSetsMarkingWorklist::ProcessNextItem(
61 Visitor* visitor, std::optional<size_t>& index) {
62 if (remaining_remembered_sets_marking_items_.load(
63 std::memory_order_relaxed) == 0) {
64 return false;
65 }
66 while (true) {
67 if (index && (index < remembered_sets_marking_items_.size())) {
68 auto& work_item = remembered_sets_marking_items_[*index];
69 if (work_item.TryAcquire()) {
70 remaining_remembered_sets_marking_items_.fetch_sub(
71 1, std::memory_order_relaxed);
72 work_item.Process(visitor);
73 (*index)++;
74 return true;
75 }
76 }
77 index = remembered_sets_marking_index_generator_.GetNext();
78 if (!index) return false;
79 }

121template <typename Visitor>
122void YoungGenerationRememberedSetsMarkingWorklist::MarkingItem::
123 MarkTypedPointers(Visitor* visitor) {
124 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.gc"),
125 "MarkingItem::MarkTypedPointers");
126 DCHECK_NULL(background_slot_set_);
127 DCHECK_NOT_NULL(typed_slot_set_);
128 const auto slot_count = RememberedSet<OLD_TO_NEW>::IterateTyped(
129 typed_slot_set_,
130 [this, visitor](SlotType slot_type, Address slot_address) {
131 Tagged<HeapObject> object = UpdateTypedSlotHelper::GetTargetObject(
132 heap(), slot_type, slot_address);
133 FullMaybeObjectSlot slot(&object);
134 return CheckAndMarkObject(visitor, slot);
135 });
136 if (slot_count == 0) {
137 delete typed_slot_set_;
138 typed_slot_set_ = nullptr;
139 }
140}

143V8_INLINE SlotCallbackResult
144YoungGenerationRememberedSetsMarkingWorklist::MarkingItem::CheckAndMarkObject(

146 static_assert(
147 std::is_same<TSlot, FullMaybeObjectSlot>::value ||
148 std::is_same<TSlot, MaybeObjectSlot>::value,
149 "Only FullMaybeObjectSlot and MaybeObjectSlot are expected here");

151 : REMOVE_SLOT;
