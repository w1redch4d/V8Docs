36void CodeDeoptEventRecord::UpdateCodeMap(
37 InstructionStreamMap* instruction_stream_map) {
38 CodeEntry* entry = instruction_stream_map->FindEntry(instruction_start);
39 if (entry != nullptr) {
40 std::vector<CpuProfileDeoptFrame> frames_vector(
41 deopt_frames, deopt_frames + deopt_frame_count);
42 entry->set_deopt_info(deopt_reason, deopt_id, std::move(frames_vector));
43 }
44 delete[] deopt_frames;
45}

47void ReportBuiltinEventRecord::UpdateCodeMap(
48 InstructionStreamMap* instruction_stream_map) {
49 CodeEntry* entry = instruction_stream_map->FindEntry(instruction_start);
50 if (entry) {
51 entry->SetBuiltinId(builtin);
52 return;
53 }
54#if V8_ENABLE_WEBASSEMBLY
55 if (builtin == Builtin::kJSToWasmWrapper) {
56 // Make sure to add the generic js-to-wasm wrapper builtin, because that
57 // one is supposed to show up in profiles.
58 entry = instruction_stream_map->code_entries().Create(
59 LogEventListener::CodeTag::kBuiltin, "js-to-wasm");
60 instruction_stream_map->AddCode(instruction_start, entry, instruction_size);
61 }
62 if (builtin == Builtin::kWasmToJsWrapperCSA) {
63 // Make sure to add the generic wasm-to-js wrapper builtin, because that
64 // one is supposed to show up in profiles.
65 entry = instruction_stream_map->code_entries().Create(
66 LogEventListener::CodeTag::kBuiltin, "wasm-to-js");
67 instruction_stream_map->AddCode(instruction_start, entry, instruction_size);
68 }
69#endif // V8_ENABLE_WEBASSEMBLY
70}
