18template <typename T>
19bool GCAwareObjectTypeCheck(Tagged<Object> object, const Heap* heap) {
20 Tagged<HeapObject> heap_object = UncheckedCast<HeapObject>(object);
21 // `heap_object` must be of type `T`. One of 4 cases can apply:
22 // 1) A scavenger is ongoing and `object` is a promoted large object.
23 if ((heap->gc_state() == Heap::SCAVENGE) &&
24 MemoryChunk::FromHeapObject(heap_object)->IsLargePage()) {
25 return true;
26 }
27 // 2) A conservative scavenge is ongoing and `object` was pinned by
28 // conservative stack scanning.
29 MapWord map_word = heap_object->map_word(
30 PtrComprCageBase(heap->isolate()->cage_base()), kRelaxedLoad);
31 if ((heap->gc_state() == Heap::SCAVENGE) &&
32 HeapLayout::InYoungGeneration(heap_object) &&
33 v8_flags.scavenger_pinning_objects && map_word.IsForwardingAddress() &&
34 HeapLayout::IsSelfForwarded(heap_object)) {
35 return true;
36 }
37 // 3) A GC is ongoing, `object` was evacuated, and the new instance is a
38 // `T`.
39 if (heap->IsInGC() && map_word.IsForwardingAddress() &&
40 Is<T>(map_word.ToForwardingAddress(heap_object))) {
41 return true;
42 }
43 // 4) If none of the above special cases apply, `heap_object` must be a `T`.
44 return Is<T>(object);
45}
46#endif // DEBUG
