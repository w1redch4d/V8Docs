22const char* Truncation::description() const {
23 switch (kind()) {
24 case TruncationKind::kNone:
25 return "no-value-use";
26 case TruncationKind::kBool:
27 return "truncate-to-bool";
28 case TruncationKind::kWord32:
29 return "truncate-to-word32";
30 case TruncationKind::kWord64:
31 return "truncate-to-word64";
32 case TruncationKind::kOddballAndBigIntToNumber:
33 switch (identify_zeros()) {
34 case kIdentifyZeros:
35 return "truncate-oddball&bigint-to-number (identify zeros)";
36 case kDistinguishZeros:
37 return "truncate-oddball&bigint-to-number (distinguish zeros)";
38 }
39 case TruncationKind::kAny:
40 switch (identify_zeros()) {
41 case kIdentifyZeros:
42 return "no-truncation (but identify zeros)";
43 case kDistinguishZeros:
44 return "no-truncation (but distinguish zeros)";
45 }
46 }
47 UNREACHABLE();
48}

50// Partial order for truncations:
51//
52// kAny <-------+
53// ^ |
54// | |
55// kOddballAndBigIntToNumber |
56// ^ |
57// / |
58// kWord64 |
59// ^ |
60// | |
61// kWord32 kBool
62// ^ ^
63// \ /
64// \ /
65// \ /
66// \ /
67// \ /
68// kNone
69//
70// TODO(jarin) We might consider making kBool < kOddballAndBigIntToNumber.
71
72// static
73Truncation::TruncationKind Truncation::Generalize(TruncationKind rep1,
74 TruncationKind rep2) {
75 if (LessGeneral(rep1, rep2)) return rep2;
76 if (LessGeneral(rep2, rep1)) return rep1;
77 // Handle the generalization of float64-representable values.
78 if (LessGeneral(rep1, TruncationKind::kOddballAndBigIntToNumber) &&
79 LessGeneral(rep2, TruncationKind::kOddballAndBigIntToNumber)) {
80 return TruncationKind::kOddballAndBigIntToNumber;
81 }
82 // Handle the generalization of any-representable values.
83 if (LessGeneral(rep1, TruncationKind::kAny) &&
84 LessGeneral(rep2, TruncationKind::kAny)) {
85 return TruncationKind::kAny;
86 }
87 // All other combinations are illegal.
88 FATAL("Tried to combine incompatible truncations");
89}
90
91// static
92IdentifyZeros Truncation::GeneralizeIdentifyZeros(IdentifyZeros i1,
93 IdentifyZeros i2) {
94 if (i1 == i2) {
95 return i1;
96 } else {
97 return kDistinguishZeros;
98 }
99}

101// static
102bool Truncation::LessGeneral(TruncationKind rep1, TruncationKind rep2) {
103 switch (rep1) {
104 case TruncationKind::kNone:
105 return true;
106 case TruncationKind::kBool:
107 return rep2 == TruncationKind::kBool || rep2 == TruncationKind::kAny;
108 case TruncationKind::kWord32:
109 return rep2 == TruncationKind::kWord32 ||
110 rep2 == TruncationKind::kWord64 ||
111 rep2 == TruncationKind::kOddballAndBigIntToNumber ||
112 rep2 == TruncationKind::kAny;
113 case TruncationKind::kWord64:
114 return rep2 == TruncationKind::kWord64 ||
115 rep2 == TruncationKind::kOddballAndBigIntToNumber ||
116 rep2 == TruncationKind::kAny;
117 case TruncationKind::kOddballAndBigIntToNumber:
118 return rep2 == TruncationKind::kOddballAndBigIntToNumber ||
119 rep2 == TruncationKind::kAny;
120 case TruncationKind::kAny:
121 return rep2 == TruncationKind::kAny;
122 }
123 UNREACHABLE();
124}

126// static
127bool Truncation::LessGeneralIdentifyZeros(IdentifyZeros i1, IdentifyZeros i2) {
128 return i1 == i2 || i1 == kIdentifyZeros;
129}

156// Changes representation from {output_rep} to {use_rep}. The {truncation}
157// parameter is only used for checking - if the changer cannot figure
158// out signedness for the word32->float64 conversion, then we check that the
159// uses truncate to word32 (so they do not care about signedness).
160Node* RepresentationChanger::GetRepresentationFor(
161 Node* node, MachineRepresentation output_rep, Type output_type,
162 Node* use_node, UseInfo use_info) {
163 // We are currently not inserting conversions in machine graphs.
164 // We might add that, though.
165 DCHECK_IMPLIES(!output_type.IsNone(), !output_type.Is(Type::Machine()));
166 if (output_rep == MachineRepresentation::kNone && !output_type.IsNone()) {
167 // The output representation should be set if the type is inhabited (i.e.,
168 // if the value is possible).
169 return TypeError(node, output_rep, output_type, use_info.representation());
170 }
171
172 // Rematerialize any truncated BigInt if user is not expecting a BigInt.
173 if (output_type.Is(Type::BigInt()) &&
174 output_rep == MachineRepresentation::kWord64 &&
175 !TypeCheckIsBigInt(use_info.type_check())) {
176 if (output_type.Is(Type::UnsignedBigInt64())) {
177 node = InsertConversion(node, simplified()->ChangeUint64ToBigInt(),
178 use_node);
179 } else {
180 node =
181 InsertConversion(node, simplified()->ChangeInt64ToBigInt(), use_node);
182 }
183 output_rep = MachineRepresentation::kTaggedPointer;
184 }
185
186 // Handle the no-op shortcuts when no checking is necessary.
187 if (use_info.type_check() == TypeCheckKind::kNone ||
188 // TODO(nicohartmann@, chromium:1077804): Ignoring {use_info.type_check()}
189 // in case the representation already matches is not correct. For now,
190 // this behavior is disabled only for TypeCheckKind::kBigInt, but should
191 // be fixed for all other type checks.
192 (output_rep != MachineRepresentation::kWord32 &&
193 !TypeCheckIsBigInt(use_info.type_check()))) {
194 if (use_info.representation() == output_rep) {
195 // Representations are the same. That's a no-op.
196 return node;
197 }
198 if (IsWord(use_info.representation()) && IsWord(output_rep)) {
199 // Both are words less than or equal to 32-bits.
200 // Since loads of integers from memory implicitly sign or zero extend the
201 // value to the full machine word size and stores implicitly truncate,
202 // no representation change is necessary.
203 return node;
204 }
205 }
206
207 switch (use_info.representation()) {
208 case MachineRepresentation::kTaggedSigned:
209 DCHECK(use_info.type_check() == TypeCheckKind::kNone ||
210 use_info.type_check() == TypeCheckKind::kSignedSmall);
211 return GetTaggedSignedRepresentationFor(node, output_rep, output_type,
212 use_node, use_info);
213 case MachineRepresentation::kTaggedPointer:
214 DCHECK(use_info.type_check() == TypeCheckKind::kNone ||
215 use_info.type_check() == TypeCheckKind::kHeapObject ||
216 use_info.type_check() == TypeCheckKind::kBigInt);
217 return GetTaggedPointerRepresentationFor(node, output_rep, output_type,
218 use_node, use_info);
219 case MachineRepresentation::kTagged:
220 DCHECK_EQ(TypeCheckKind::kNone, use_info.type_check());
221 return GetTaggedRepresentationFor(node, output_rep, output_type,
222 use_info.truncation());
223 case MachineRepresentation::kFloat16:
224 case MachineRepresentation::kFloat32:
225 DCHECK_EQ(TypeCheckKind::kNone, use_info.type_check());
226 return GetFloat32RepresentationFor(node, output_rep, output_type,
227 use_info.truncation());
228 case MachineRepresentation::kFloat64:
229 DCHECK(use_info.type_check() == TypeCheckKind::kNone ||
230 use_info.type_check() == TypeCheckKind::kNumber ||
231 use_info.type_check() == TypeCheckKind::kNumberOrBoolean ||
232 use_info.type_check() == TypeCheckKind::kNumberOrOddball);
233 return GetFloat64RepresentationFor(node, output_rep, output_type,
234 use_node, use_info);
235 case MachineRepresentation::kBit:
236 DCHECK_EQ(TypeCheckKind::kNone, use_info.type_check());
237 return GetBitRepresentationFor(node, output_rep, output_type);
238 case MachineRepresentation::kWord8:
239 case MachineRepresentation::kWord16:
240 case MachineRepresentation::kWord32:
241 return GetWord32RepresentationFor(node, output_rep, output_type, use_node,
242 use_info);
243 case MachineRepresentation::kWord64:
244 DCHECK(use_info.type_check() == TypeCheckKind::kNone ||
245 use_info.type_check() == TypeCheckKind::kSigned64 ||
246 TypeCheckIsBigInt(use_info.type_check()) ||
247 use_info.type_check() == TypeCheckKind::kArrayIndex);
248 return GetWord64RepresentationFor(node, output_rep, output_type, use_node,
249 use_info);
250 case MachineRepresentation::kSimd128:
251 case MachineRepresentation::kSimd256:
252 case MachineRepresentation::kNone:
253 return node;
254 case MachineRepresentation::kCompressed:
255 case MachineRepresentation::kCompressedPointer:
256 case MachineRepresentation::kSandboxedPointer:
257 case MachineRepresentation::kProtectedPointer:
258 case MachineRepresentation::kIndirectPointer:
259 case MachineRepresentation::kMapWord:
260 UNREACHABLE();
261 }
262 UNREACHABLE();
263}

265Node* RepresentationChanger::GetTaggedSignedRepresentationFor(
266 Node* node, MachineRepresentation output_rep, Type output_type,
267 Node* use_node, UseInfo use_info) {
268 // Eagerly fold representation changes for constants.
269 switch (node->opcode()) {
270 case IrOpcode::kNumberConstant:
271 if (output_type.Is(Type::SignedSmall())) {
272 return node;
273 }
274 break;
275 default:
276 break;
277 }
278 // Select the correct X -> Tagged operator.
279 const Operator* op;
280 if (output_type.Is(Type::None())) {
281 // This is an impossible value; it should not be used at runtime.
282 return jsgraph()->graph()->NewNode(
283 jsgraph()->common()->DeadValue(MachineRepresentation::kTaggedSigned),
284 node);
285 } else if (IsWord(output_rep)) {
286 if (output_type.Is(Type::Signed31())) {
287 op = simplified()->ChangeInt31ToTaggedSigned();
288 } else if (output_type.Is(Type::Signed32())) {
289 if (SmiValuesAre32Bits()) {
290 op = simplified()->ChangeInt32ToTagged();
291 } else if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
292 op = simplified()->CheckedInt32ToTaggedSigned(use_info.feedback());
293 } else {
294 return TypeError(node, output_rep, output_type,
295 MachineRepresentation::kTaggedSigned);
296 }
297 } else if (output_type.Is(Type::Unsigned32()) &&
298 use_info.type_check() == TypeCheckKind::kSignedSmall) {
299 op = simplified()->CheckedUint32ToTaggedSigned(use_info.feedback());
300 } else {
301 return TypeError(node, output_rep, output_type,
302 MachineRepresentation::kTaggedSigned);
303 }
304 } else if (output_rep == MachineRepresentation::kWord64) {
305 if (output_type.Is(Type::Signed31())) {
306 // int64 -> int32 -> tagged signed
307 node = InsertTruncateInt64ToInt32(node);
308 op = simplified()->ChangeInt31ToTaggedSigned();
309 } else if (output_type.Is(Type::Signed32()) && SmiValuesAre32Bits()) {
310 // int64 -> int32 -> tagged signed
311 node = InsertTruncateInt64ToInt32(node);
312 op = simplified()->ChangeInt32ToTagged();
313 } else if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
314 if (output_type.Is(cache_->kPositiveSafeInteger)) {
315 op = simplified()->CheckedUint64ToTaggedSigned(use_info.feedback());
316 } else if (output_type.Is(cache_->kSafeInteger)) {
317 op = simplified()->CheckedInt64ToTaggedSigned(use_info.feedback());
318 } else {
319 return TypeError(node, output_rep, output_type,
320 MachineRepresentation::kTaggedSigned);
321 }
322 } else {
323 return TypeError(node, output_rep, output_type,
324 MachineRepresentation::kTaggedSigned);
325 }
326 } else if (output_rep == MachineRepresentation::kFloat64) {
327 if (output_type.Is(Type::Signed31())) {
328 // float64 -> int32 -> tagged signed
329 node = InsertChangeFloat64ToInt32(node);
330 op = simplified()->ChangeInt31ToTaggedSigned();
331 } else if (output_type.Is(Type::Signed32())) {
332 // float64 -> int32 -> tagged signed
333 node = InsertChangeFloat64ToInt32(node);
334 if (SmiValuesAre32Bits()) {
335 op = simplified()->ChangeInt32ToTagged();
336 } else if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
337 op = simplified()->CheckedInt32ToTaggedSigned(use_info.feedback());
338 } else {
339 return TypeError(node, output_rep, output_type,
340 MachineRepresentation::kTaggedSigned);
341 }
342 } else if (output_type.Is(Type::Unsigned32()) &&
343 use_info.type_check() == TypeCheckKind::kSignedSmall) {
344 // float64 -> uint32 -> tagged signed
345 node = InsertChangeFloat64ToUint32(node);
346 op = simplified()->CheckedUint32ToTaggedSigned(use_info.feedback());
347 } else if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
348 node = InsertCheckedFloat64ToInt32(
349 node,
350 output_type.Maybe(Type::MinusZero())
351 ? CheckForMinusZeroMode::kCheckForMinusZero
352 : CheckForMinusZeroMode::kDontCheckForMinusZero,
353 use_info.feedback(), use_node);
354 if (SmiValuesAre32Bits()) {
355 op = simplified()->ChangeInt32ToTagged();
356 } else {
357 op = simplified()->CheckedInt32ToTaggedSigned(use_info.feedback());
358 }
359 } else {
360 return TypeError(node, output_rep, output_type,
361 MachineRepresentation::kTaggedSigned);
362 }
363 } else if (output_rep == MachineRepresentation::kFloat32) {
364 if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
365 node = InsertChangeFloat32ToFloat64(node);
366 node = InsertCheckedFloat64ToInt32(
367 node,
368 output_type.Maybe(Type::MinusZero())
369 ? CheckForMinusZeroMode::kCheckForMinusZero
370 : CheckForMinusZeroMode::kDontCheckForMinusZero,
371 use_info.feedback(), use_node);
372 if (SmiValuesAre32Bits()) {
373 op = simplified()->ChangeInt32ToTagged();
374 } else {
375 op = simplified()->CheckedInt32ToTaggedSigned(use_info.feedback());
376 }
377 } else {
378 return TypeError(node, output_rep, output_type,
379 MachineRepresentation::kTaggedSigned);
380 }
381 } else if (CanBeTaggedPointer(output_rep)) {
382 if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
383 op = simplified()->CheckedTaggedToTaggedSigned(use_info.feedback());
384 } else if (output_type.Is(Type::SignedSmall())) {
385 op = simplified()->ChangeTaggedToTaggedSigned();
386 } else {
387 return TypeError(node, output_rep, output_type,
388 MachineRepresentation::kTaggedSigned);
389 }
390 } else if (output_rep == MachineRepresentation::kBit) {
391 if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
392 // TODO(turbofan): Consider adding a Bailout operator that just deopts.
393 // Also use that for MachineRepresentation::kPointer case above.
394 node = InsertChangeBitToTagged(node);
395 op = simplified()->CheckedTaggedToTaggedSigned(use_info.feedback());
396 } else {
397 return TypeError(node, output_rep, output_type,
398 MachineRepresentation::kTaggedSigned);
399 }
400 } else {
401 return TypeError(node, output_rep, output_type,
402 MachineRepresentation::kTaggedSigned);
403 }
404 return InsertConversion(node, op, use_node);
405}

407Node* RepresentationChanger::GetTaggedPointerRepresentationFor(
408 Node* node, MachineRepresentation output_rep, Type output_type,
409 Node* use_node, UseInfo use_info) {
410 // Eagerly fold representation changes for constants.
411 switch (node->opcode()) {
412 case IrOpcode::kHeapConstant:
413 if (TypeCheckIsBigInt(use_info.type_check())) break;
414 return node; // No change necessary.
415 case IrOpcode::kInt32Constant:
416 case IrOpcode::kFloat64Constant:
417 case IrOpcode::kFloat32Constant:
418 UNREACHABLE();
419 default:
420 break;
421 }
422 // Select the correct X -> TaggedPointer operator.
423 Operator const* op;
424 if (output_type.Is(Type::None())) {
425 // This is an impossible value; it should not be used at runtime.
426 return jsgraph()->graph()->NewNode(
427 jsgraph()->common()->DeadValue(MachineRepresentation::kTaggedPointer),
428 node);
429 }
430
431 if (TypeCheckIsBigInt(use_info.type_check()) &&
432 !output_type.Is(Type::BigInt())) {
433 // BigInt checks can only be performed on tagged representations. Note that
434 // a corresponding check is inserted down below.
435 if (!CanBeTaggedPointer(output_rep)) {
436 Node* unreachable =
437 InsertUnconditionalDeopt(use_node, DeoptimizeReason::kNotABigInt);
438 return jsgraph()->graph()->NewNode(
439 jsgraph()->common()->DeadValue(MachineRepresentation::kTaggedPointer),
440 unreachable);
441 }
442 }
443
444 if (output_rep == MachineRepresentation::kBit) {
445 if (output_type.Is(Type::Boolean())) {
446 op = simplified()->ChangeBitToTagged();
447 } else {
448 return TypeError(node, output_rep, output_type,
449 MachineRepresentation::kTagged);
450 }
451 } else if (IsWord(output_rep)) {
452 if (output_type.Is(Type::Unsigned32())) {
453 // uint32 -> float64 -> tagged
454 node = InsertChangeUint32ToFloat64(node);
455 } else if (output_type.Is(Type::Signed32())) {
456 // int32 -> float64 -> tagged
457 node = InsertChangeInt32ToFloat64(node);
458 } else {
459 return TypeError(node, output_rep, output_type,
460 MachineRepresentation::kTaggedPointer);
461 }
462 op = simplified()->ChangeFloat64ToTaggedPointer();
463 } else if (output_rep == MachineRepresentation::kWord64) {
464 if (output_type.Is(cache_->kSafeInteger)) {
465 // int64 -> float64 -> tagged pointer
466 op = machine()->ChangeInt64ToFloat64();
467 node = jsgraph()->graph()->NewNode(op, node);
468 op = simplified()->ChangeFloat64ToTaggedPointer();
469 } else if (output_type.Is(Type::SignedBigInt64()) &&
470 use_info.type_check() == TypeCheckKind::kBigInt) {
471 op = simplified()->ChangeInt64ToBigInt();
472 } else if (output_type.Is(Type::UnsignedBigInt64()) &&
473 use_info.type_check() == TypeCheckKind::kBigInt) {
474 op = simplified()->ChangeUint64ToBigInt();
475 } else {
476 return TypeError(node, output_rep, output_type,
477 MachineRepresentation::kTaggedPointer);
478 }
479 } else if (output_rep == MachineRepresentation::kFloat32) {
480 if (output_type.Is(Type::Number())) {
481 // float32 -> float64 -> tagged
482 node = InsertChangeFloat32ToFloat64(node);
483 op = simplified()->ChangeFloat64ToTaggedPointer();
484 } else {
485 return TypeError(node, output_rep, output_type,
486 MachineRepresentation::kTaggedPointer);
487 }
488 } else if (output_rep == MachineRepresentation::kFloat64) {
489 if (output_type.Is(Type::Number())) {
490 // float64 -> tagged
491 op = simplified()->ChangeFloat64ToTaggedPointer();
492 } else {
493 return TypeError(node, output_rep, output_type,
494 MachineRepresentation::kTaggedPointer);
495 }
496 } else if (IsAnyTagged(output_rep)) {
497 if (use_info.type_check() == TypeCheckKind::kBigInt) {
498 if (output_type.Is(Type::BigInt())) {
499 DCHECK_NE(output_rep, MachineRepresentation::kTaggedSigned);
500 return node;
501 }
502 op = simplified()->CheckBigInt(use_info.feedback());
503 } else if (use_info.type_check() == TypeCheckKind::kBigInt64) {
504 if (output_type.Is(Type::SignedBigInt64())) {
505 DCHECK_NE(output_rep, MachineRepresentation::kTaggedSigned);
506 return node;
507 }
508 if (!output_type.Is(Type::BigInt())) {
509 node = InsertConversion(
510 node, simplified()->CheckBigInt(use_info.feedback()), use_node);
511 }
512 op = simplified()->CheckedBigIntToBigInt64(use_info.feedback());
513 } else if (output_rep == MachineRepresentation::kTaggedPointer ||
514 !output_type.Maybe(Type::SignedSmall())) {
515 DCHECK_NE(output_rep, MachineRepresentation::kTaggedSigned);
516 return node;
517 } else {
518 op = simplified()->CheckedTaggedToTaggedPointer(use_info.feedback());
519 }
520 } else {
521 return TypeError(node, output_rep, output_type,
522 MachineRepresentation::kTaggedPointer);
523 }
524 return InsertConversion(node, op, use_node);
525}

527Node* RepresentationChanger::GetTaggedRepresentationFor(
528 Node* node, MachineRepresentation output_rep, Type output_type,
529 Truncation truncation) {
530 // Eagerly fold representation changes for constants.
531 switch (node->opcode()) {
532 case IrOpcode::kNumberConstant:
533 case IrOpcode::kHeapConstant:
534 return node; // No change necessary.
535 case IrOpcode::kInt32Constant:
536 case IrOpcode::kFloat64Constant:
537 case IrOpcode::kFloat32Constant:
538 UNREACHABLE();
539 default:
540 break;
541 }
542 if (output_rep == MachineRepresentation::kTaggedSigned ||
543 output_rep == MachineRepresentation::kTaggedPointer ||
544 output_rep == MachineRepresentation::kMapWord) {
545 // this is a no-op.
546 return node;
547 }
548 // Select the correct X -> Tagged operator.
549 const Operator* op;
550 if (output_type.Is(Type::None())) {
551 // This is an impossible value; it should not be used at runtime.
552 return jsgraph()->graph()->NewNode(
553 jsgraph()->common()->DeadValue(MachineRepresentation::kTagged), node);
554 } else if (output_rep == MachineRepresentation::kBit) {
555 if (output_type.Is(Type::Boolean())) {
556 op = simplified()->ChangeBitToTagged();
557 } else {
558 return TypeError(node, output_rep, output_type,
559 MachineRepresentation::kTagged);
560 }
561 } else if (IsWord(output_rep)) {
562 if (output_type.Is(Type::Signed31())) {
563 op = simplified()->ChangeInt31ToTaggedSigned();
564 } else if (output_type.Is(Type::Signed32()) ||
565 (output_type.Is(Type::Signed32OrMinusZero()) &&
566 truncation.IdentifiesZeroAndMinusZero())) {
567 op = simplified()->ChangeInt32ToTagged();
568 } else if (output_type.Is(Type::Unsigned32()) ||
569 (output_type.Is(Type::Unsigned32OrMinusZero()) &&
570 truncation.IdentifiesZeroAndMinusZero()) ||
571 truncation.IsUsedAsWord32()) {
572 // Either the output is uint32 or the uses only care about the
573 // low 32 bits (so we can pick uint32 safely).
574 op = simplified()->ChangeUint32ToTagged();
575 } else {
576 return TypeError(node, output_rep, output_type,
577 MachineRepresentation::kTagged);
578 }
579 } else if (output_rep == MachineRepresentation::kWord64) {
580 if (output_type.Is(Type::Signed31())) {
581 // int64 -> int32 -> tagged signed
582 node = InsertTruncateInt64ToInt32(node);
583 op = simplified()->ChangeInt31ToTaggedSigned();
584 } else if (output_type.Is(Type::Signed32())) {
585 // int64 -> int32 -> tagged
586 node = InsertTruncateInt64ToInt32(node);
587 op = simplified()->ChangeInt32ToTagged();
588 } else if (output_type.Is(Type::Unsigned32())) {
589 // int64 -> uint32 -> tagged
590 node = InsertTruncateInt64ToInt32(node);
591 op = simplified()->ChangeUint32ToTagged();
592 } else if (output_type.Is(cache_->kPositiveSafeInteger)) {
593 // uint64 -> tagged
594 op = simplified()->ChangeUint64ToTagged();
595 } else if (output_type.Is(cache_->kSafeInteger)) {
596 // int64 -> tagged
597 op = simplified()->ChangeInt64ToTagged();
598 } else if (output_type.Is(Type::SignedBigInt64())) {
599 // int64 -> BigInt
600 op = simplified()->ChangeInt64ToBigInt();
601 } else if (output_type.Is(Type::UnsignedBigInt64())) {
602 // uint64 -> BigInt
603 op = simplified()->ChangeUint64ToBigInt();
604 } else {
605 return TypeError(node, output_rep, output_type,
606 MachineRepresentation::kTagged);
607 }
608 } else if (output_rep ==
609 MachineRepresentation::kFloat32) { // float32 -> float64 -> tagged
610 node = InsertChangeFloat32ToFloat64(node);
611 op = simplified()->ChangeFloat64ToTagged(
612 output_type.Maybe(Type::MinusZero())
613 ? CheckForMinusZeroMode::kCheckForMinusZero
614 : CheckForMinusZeroMode::kDontCheckForMinusZero);
615 } else if (output_rep == MachineRepresentation::kFloat64) {
616 if (output_type.Is(Type::Signed31())) { // float64 -> int32 -> tagged
617 node = InsertChangeFloat64ToInt32(node);
618 op = simplified()->ChangeInt31ToTaggedSigned();
619 } else if (output_type.Is(
620 Type::Signed32())) { // float64 -> int32 -> tagged
621 node = InsertChangeFloat64ToInt32(node);
622 op = simplified()->ChangeInt32ToTagged();
623 } else if (output_type.Is(
624 Type::Unsigned32())) { // float64 -> uint32 -> tagged
625 node = InsertChangeFloat64ToUint32(node);
626 op = simplified()->ChangeUint32ToTagged();
627 } else if (output_type.Is(Type::Number()) ||
628 (output_type.Is(Type::NumberOrOddball()) &&
629 truncation.TruncatesOddballAndBigIntToNumber())) {
630 op = simplified()->ChangeFloat64ToTagged(
631 output_type.Maybe(Type::MinusZero())
632 ? CheckForMinusZeroMode::kCheckForMinusZero
633 : CheckForMinusZeroMode::kDontCheckForMinusZero);
634 } else {
635 return TypeError(node, output_rep, output_type,
636 MachineRepresentation::kTagged);
637 }
638 } else {
639 return TypeError(node, output_rep, output_type,
640 MachineRepresentation::kTagged);
641 }
642 return jsgraph()->graph()->NewNode(op, node);
643}

645Node* RepresentationChanger::GetFloat32RepresentationFor(
646 Node* node, MachineRepresentation output_rep, Type output_type,
647 Truncation truncation) {
648 // Eagerly fold representation changes for constants.
649 switch (node->opcode()) {
650 case IrOpcode::kNumberConstant:
651 return jsgraph()->Float32Constant(
652 DoubleToFloat32(OpParameter<double>(node->op())));
653 case IrOpcode::kInt32Constant:
654 case IrOpcode::kFloat64Constant:
655 case IrOpcode::kFloat32Constant:
656 UNREACHABLE();
657 default:
658 break;
659 }
660 // Select the correct X -> Float32 operator.
661 const Operator* op = nullptr;
662 if (output_type.Is(Type::None())) {
663 // This is an impossible value; it should not be used at runtime.
664 return jsgraph()->graph()->NewNode(
665 jsgraph()->common()->DeadValue(MachineRepresentation::kFloat32), node);
666 } else if (IsWord(output_rep)) {
667 if (output_type.Is(Type::Signed32())) {
668 // int32 -> float64 -> float32
669 op = machine()->ChangeInt32ToFloat64();
670 node = jsgraph()->graph()->NewNode(op, node);
671 op = machine()->TruncateFloat64ToFloat32();
672 } else if (output_type.Is(Type::Unsigned32()) ||
673 truncation.IsUsedAsWord32()) {
674 // Either the output is uint32 or the uses only care about the
675 // low 32 bits (so we can pick uint32 safely).
676
677 // uint32 -> float64 -> float32
678 op = machine()->ChangeUint32ToFloat64();
679 node = jsgraph()->graph()->NewNode(op, node);
680 op = machine()->TruncateFloat64ToFloat32();
681 }
682 } else if (IsAnyTagged(output_rep)) {
683 if (output_type.Is(Type::NumberOrOddball())) {
684 // tagged -> float64 -> float32
685 if (output_type.Is(Type::Number())) {
686 op = simplified()->ChangeTaggedToFloat64();
687 } else {
688 op = simplified()->TruncateTaggedToFloat64();
689 }
690 node = jsgraph()->graph()->NewNode(op, node);
691 op = machine()->TruncateFloat64ToFloat32();
692 }
693 } else if (output_rep == MachineRepresentation::kFloat64) {
694 op = machine()->TruncateFloat64ToFloat32();
695 } else if (output_rep == MachineRepresentation::kWord64) {
696 if (output_type.Is(cache_->kSafeInteger)) {
697 // int64 -> float64 -> float32
698 op = machine()->ChangeInt64ToFloat64();
699 node = jsgraph()->graph()->NewNode(op, node);
700 op = machine()->TruncateFloat64ToFloat32();
701 }
702 }
703 if (op == nullptr) {
704 return TypeError(node, output_rep, output_type,
705 MachineRepresentation::kFloat32);
706 }
707 return jsgraph()->graph()->NewNode(op, node);

710Node* RepresentationChanger::GetFloat64RepresentationFor(
711 Node* node, MachineRepresentation output_rep, Type output_type,
712 Node* use_node, UseInfo use_info) {
713 NumberMatcher m(node);
714 if (m.HasResolvedValue()) {
715 // BigInts are not used as number constants.
716 DCHECK(!TypeCheckIsBigInt(use_info.type_check()));
717 switch (use_info.type_check()) {
718 case TypeCheckKind::kNone:
719 case TypeCheckKind::kNumber:
720 case TypeCheckKind::kNumberOrBoolean:
721 case TypeCheckKind::kNumberOrOddball:
722 return jsgraph()->Float64Constant(m.ResolvedValue());
723 case TypeCheckKind::kBigInt:
724 case TypeCheckKind::kBigInt64:
725 case TypeCheckKind::kHeapObject:
726 case TypeCheckKind::kSigned32:
727 case TypeCheckKind::kSigned64:
728 case TypeCheckKind::kSignedSmall:
729 case TypeCheckKind::kArrayIndex:
730 break;
731 }
732 }
733 // Select the correct X -> Float64 operator.
734 const Operator* op = nullptr;
735 if (output_type.Is(Type::None())) {
736 // This is an impossible value; it should not be used at runtime.
737 return jsgraph()->graph()->NewNode(
738 jsgraph()->common()->DeadValue(MachineRepresentation::kFloat64), node);
739 } else if (IsWord(output_rep)) {
740 if (output_type.Is(Type::Signed32()) ||
741 (output_type.Is(Type::Signed32OrMinusZero()) &&
742 use_info.truncation().IdentifiesZeroAndMinusZero())) {
743 op = machine()->ChangeInt32ToFloat64();
744 } else if (output_type.Is(Type::Unsigned32()) ||
745 (output_type.Is(Type::Unsigned32OrMinusZero()) &&
746 use_info.truncation().IdentifiesZeroAndMinusZero()) ||
747 use_info.truncation().IsUsedAsWord32()) {
748 // Either the output is uint32 or the uses only care about the
749 // low 32 bits (so we can pick uint32 safely).
750 op = machine()->ChangeUint32ToFloat64();
751 }
752 } else if (output_rep == MachineRepresentation::kBit) {
753 CHECK(output_type.Is(Type::Boolean()));
754 if (use_info.truncation().TruncatesOddballAndBigIntToNumber() ||
755 use_info.type_check() == TypeCheckKind::kNumberOrBoolean ||
756 use_info.type_check() == TypeCheckKind::kNumberOrOddball) {
757 op = machine()->ChangeUint32ToFloat64();
758 } else {
759 CHECK_NE(use_info.type_check(), TypeCheckKind::kNone);
760 Node* unreachable =
761 InsertUnconditionalDeopt(use_node, DeoptimizeReason::kNotAHeapNumber);
762 return jsgraph()->graph()->NewNode(
763 jsgraph()->common()->DeadValue(MachineRepresentation::kFloat64),
764 unreachable);
765 }
766 } else if (IsAnyTagged(output_rep)) {
767 if (output_type.Is(Type::Undefined())) {
768 if (use_info.type_check() == TypeCheckKind::kNumberOrOddball ||
769 (use_info.type_check() == TypeCheckKind::kNone &&
770 use_info.truncation().TruncatesOddballAndBigIntToNumber())) {
771 return jsgraph()->Float64Constant(
772 std::numeric_limits<double>::quiet_NaN());
773 } else {
774 DCHECK(use_info.type_check() == TypeCheckKind::kNone ||
775 use_info.type_check() == TypeCheckKind::kNumber ||
776 use_info.type_check() == TypeCheckKind::kNumberOrBoolean);
777 Node* unreachable = InsertUnconditionalDeopt(
778 use_node, use_info.type_check() == TypeCheckKind::kNumber
779 ? DeoptimizeReason::kNotANumber
780 : DeoptimizeReason::kNotANumberOrBoolean);
781 return jsgraph()->graph()->NewNode(
782 jsgraph()->common()->DeadValue(MachineRepresentation::kFloat64),
783 unreachable);
784 }
785 } else if (output_rep == MachineRepresentation::kTaggedSigned) {
786 node = InsertChangeTaggedSignedToInt32(node);
787 op = machine()->ChangeInt32ToFloat64();
788 } else if (output_type.Is(Type::Number())) {
789 op = simplified()->ChangeTaggedToFloat64();
790 } else if ((output_type.Is(Type::NumberOrOddball()) &&
791 use_info.truncation().TruncatesOddballAndBigIntToNumber()) ||
792 output_type.Is(Type::NumberOrHole())) {
793 // JavaScript 'null' is an Oddball that results in +0 when truncated to
794 // Number. In a context like -0 == null, which must evaluate to false,
795 // this truncation must not happen. For this reason we restrict this
796 // case to when either the user explicitly requested a float (and thus
797 // wants +0 if null is the input) or we know from the types that the
798 // input can only be Number | Hole. The latter is necessary to handle
799 // the operator CheckFloat64Hole. We did not put in the type (Number |
800 // Oddball \ Null) to discover more bugs related to this conversion via
801 // crashes.
802 op = simplified()->TruncateTaggedToFloat64();
803 } else if (use_info.type_check() == TypeCheckKind::kNumber ||
804 (use_info.type_check() == TypeCheckKind::kNumberOrOddball &&
805 !output_type.Maybe(Type::BooleanOrNullOrNumber()))) {
806 op = simplified()->CheckedTaggedToFloat64(CheckTaggedInputMode::kNumber,
807 use_info.feedback());
808 } else if (use_info.type_check() == TypeCheckKind::kNumberOrBoolean) {
809 op = simplified()->CheckedTaggedToFloat64(
810 CheckTaggedInputMode::kNumberOrBoolean, use_info.feedback());
811 } else if (use_info.type_check() == TypeCheckKind::kNumberOrOddball) {
812 op = simplified()->CheckedTaggedToFloat64(
813 CheckTaggedInputMode::kNumberOrOddball, use_info.feedback());
814 }
815 } else if (output_rep == MachineRepresentation::kFloat32) {
816 op = machine()->ChangeFloat32ToFloat64();
817 } else if (output_rep == MachineRepresentation::kWord64) {
818 if (output_type.Is(cache_->kSafeInteger)) {
819 op = machine()->ChangeInt64ToFloat64();
820 }
821 }
822 if (op == nullptr) {
823 return TypeError(node, output_rep, output_type,
824 MachineRepresentation::kFloat64);
825 }
826 return InsertConversion(node, op, use_node);
827}

834 Node* node, DeoptimizeReason reason, const FeedbackSource& feedback) {
835 Node* effect = NodeProperties::GetEffectInput(node);

846Node* RepresentationChanger::GetWord32RepresentationFor(
847 Node* node, MachineRepresentation output_rep, Type output_type,
848 Node* use_node, UseInfo use_info) {
849 // Eagerly fold representation changes for constants.
850 switch (node->opcode()) {
851 case IrOpcode::kInt32Constant:
852 case IrOpcode::kInt64Constant:
853 case IrOpcode::kFloat32Constant:
854 case IrOpcode::kFloat64Constant:
855 UNREACHABLE();
856 case IrOpcode::kNumberConstant: {
857 double const fv = OpParameter<double>(node->op());
858 if (use_info.type_check() == TypeCheckKind::kNone ||
859 ((use_info.type_check() == TypeCheckKind::kSignedSmall ||
860 use_info.type_check() == TypeCheckKind::kSigned32 ||
861 use_info.type_check() == TypeCheckKind::kNumber ||
862 use_info.type_check() == TypeCheckKind::kNumberOrOddball ||
863 use_info.type_check() == TypeCheckKind::kArrayIndex) &&
864 IsInt32Double(fv))) {
865 return InsertTypeOverrideForVerifier(NodeProperties::GetType(node),
866 MakeTruncatedInt32Constant(fv));
867 }
868 break;
869 }
870 default:
871 break;
872 }
873
874 // Select the correct X -> Word32 operator.
875 const Operator* op = nullptr;
876 if (output_type.Is(Type::None())) {
877 // This is an impossible value; it should not be used at runtime.
878 return jsgraph()->graph()->NewNode(
879 jsgraph()->common()->DeadValue(MachineRepresentation::kWord32), node);
880 } else if (output_rep == MachineRepresentation::kBit) {
881 CHECK(output_type.Is(Type::Boolean()));
882 if (use_info.truncation().IsUsedAsWord32()) {
883 return node;
884 } else {
885 CHECK(Truncation::Any(kIdentifyZeros)
886 .IsLessGeneralThan(use_info.truncation()));
887 CHECK_NE(use_info.type_check(), TypeCheckKind::kNone);
888 CHECK_NE(use_info.type_check(), TypeCheckKind::kNumberOrOddball);
889 Node* unreachable =
890 InsertUnconditionalDeopt(use_node, DeoptimizeReason::kNotASmi);
891 return jsgraph()->graph()->NewNode(
892 jsgraph()->common()->DeadValue(MachineRepresentation::kWord32),
893 unreachable);
894 }
895 } else if (output_rep == MachineRepresentation::kFloat64) {
896 if (output_type.Is(Type::Signed32())) {
897 op = machine()->ChangeFloat64ToInt32();
898 } else if (use_info.type_check() == TypeCheckKind::kSignedSmall ||
899 use_info.type_check() == TypeCheckKind::kSigned32 ||
900 use_info.type_check() == TypeCheckKind::kArrayIndex) {
901 op = simplified()->CheckedFloat64ToInt32(
902 output_type.Maybe(Type::MinusZero())
903 ? use_info.minus_zero_check()
904 : CheckForMinusZeroMode::kDontCheckForMinusZero,
905 use_info.feedback());
906 } else if (output_type.Is(Type::Unsigned32())) {
907 op = machine()->ChangeFloat64ToUint32();
908 } else if (use_info.truncation().IsUsedAsWord32()) {
909 op = machine()->TruncateFloat64ToWord32();
910 } else {
911 return TypeError(node, output_rep, output_type,
912 MachineRepresentation::kWord32);
913 }
914 } else if (output_rep == MachineRepresentation::kFloat32) {
915 node = InsertChangeFloat32ToFloat64(node); // float32 -> float64 -> int32
916 if (output_type.Is(Type::Signed32())) {
917 op = machine()->ChangeFloat64ToInt32();
918 } else if (use_info.type_check() == TypeCheckKind::kSignedSmall ||
919 use_info.type_check() == TypeCheckKind::kSigned32 ||
920 use_info.type_check() == TypeCheckKind::kArrayIndex) {
921 op = simplified()->CheckedFloat64ToInt32(
922 output_type.Maybe(Type::MinusZero())
923 ? use_info.minus_zero_check()
924 : CheckForMinusZeroMode::kDontCheckForMinusZero,
925 use_info.feedback());
926 } else if (output_type.Is(Type::Unsigned32())) {
927 op = machine()->ChangeFloat64ToUint32();
928 } else if (use_info.truncation().IsUsedAsWord32()) {
929 op = machine()->TruncateFloat64ToWord32();
930 } else {
931 return TypeError(node, output_rep, output_type,
932 MachineRepresentation::kWord32);
933 }
934 } else if (IsAnyTagged(output_rep)) {
935 if (output_rep == MachineRepresentation::kTaggedSigned &&
936 output_type.Is(Type::SignedSmall())) {
937 op = simplified()->ChangeTaggedSignedToInt32();
938 } else if (output_type.Is(Type::Signed32())) {
939 op = simplified()->ChangeTaggedToInt32();
940 } else if (use_info.type_check() == TypeCheckKind::kSignedSmall) {
941 op = simplified()->CheckedTaggedSignedToInt32(use_info.feedback());
942 } else if (use_info.type_check() == TypeCheckKind::kSigned32) {
943 op = simplified()->CheckedTaggedToInt32(
944 output_type.Maybe(Type::MinusZero())
945 ? use_info.minus_zero_check()
946 : CheckForMinusZeroMode::kDontCheckForMinusZero,
947 use_info.feedback());
948 } else if (use_info.type_check() == TypeCheckKind::kArrayIndex) {
949 op = simplified()->CheckedTaggedToArrayIndex(use_info.feedback());
950 } else if (output_type.Is(Type::Unsigned32())) {
951 op = simplified()->ChangeTaggedToUint32();
952 } else if (use_info.truncation().IsUsedAsWord32()) {
953 if (output_type.Is(Type::NumberOrOddballOrHole())) {
954 op = simplified()->TruncateTaggedToWord32();
955 } else if (use_info.type_check() == TypeCheckKind::kNumber) {
956 op = simplified()->CheckedTruncateTaggedToWord32(
957 CheckTaggedInputMode::kNumber, use_info.feedback());
958 } else if (use_info.type_check() == TypeCheckKind::kNumberOrOddball) {
959 op = simplified()->CheckedTruncateTaggedToWord32(
960 CheckTaggedInputMode::kNumberOrOddball, use_info.feedback());
961 } else {
962 return TypeError(node, output_rep, output_type,
963 MachineRepresentation::kWord32);
964 }
965 } else {
966 return TypeError(node, output_rep, output_type,
967 MachineRepresentation::kWord32);
968 }
969 } else if (output_rep == MachineRepresentation::kWord32) {
970 // Only the checked case should get here, the non-checked case is
971 // handled in GetRepresentationFor.
972 if (use_info.type_check() == TypeCheckKind::kSignedSmall ||
973 use_info.type_check() == TypeCheckKind::kSigned32 ||
974 use_info.type_check() == TypeCheckKind::kArrayIndex) {
975 bool identify_zeros = use_info.truncation().IdentifiesZeroAndMinusZero();
976 if (output_type.Is(Type::Signed32()) ||
977 (identify_zeros && output_type.Is(Type::Signed32OrMinusZero()))) {
978 return node;
979 } else if (output_type.Is(Type::Unsigned32()) ||
980 (identify_zeros &&
981 output_type.Is(Type::Unsigned32OrMinusZero()))) {
982 op = simplified()->CheckedUint32ToInt32(use_info.feedback());
983 } else {
984 return TypeError(node, output_rep, output_type,
985 MachineRepresentation::kWord32);
986 }
987 } else if (use_info.type_check() == TypeCheckKind::kNumber ||
988 use_info.type_check() == TypeCheckKind::kNumberOrOddball) {
989 return node;
990 }
991 } else if (output_rep == MachineRepresentation::kWord8 ||
992 output_rep == MachineRepresentation::kWord16) {
993 DCHECK_EQ(MachineRepresentation::kWord32, use_info.representation());
994 DCHECK(use_info.type_check() == TypeCheckKind::kSignedSmall ||
995 use_info.type_check() == TypeCheckKind::kSigned32);
996 return node;
997 } else if (output_rep == MachineRepresentation::kWord64) {
998 if (output_type.Is(Type::Signed32()) ||
999 (output_type.Is(Type::Unsigned32()) &&
1000 use_info.type_check() == TypeCheckKind::kNone) ||
1001 (output_type.Is(cache_->kSafeInteger) &&
1002 use_info.truncation().IsUsedAsWord32())) {
1003 op = machine()->TruncateInt64ToInt32();
1004 } else if (use_info.type_check() == TypeCheckKind::kSignedSmall ||
1005 use_info.type_check() == TypeCheckKind::kSigned32 ||
1006 use_info.type_check() == TypeCheckKind::kArrayIndex) {
1007 if (output_type.Is(cache_->kPositiveSafeInteger)) {
1008 op = simplified()->CheckedUint64ToInt32(use_info.feedback());
1009 } else if (output_type.Is(cache_->kSafeInteger)) {
1010 op = simplified()->CheckedInt64ToInt32(use_info.feedback());
1011 } else {
1012 return TypeError(node, output_rep, output_type,
1013 MachineRepresentation::kWord32);
1014 }
1015 } else {
1016 return TypeError(node, output_rep, output_type,
1017 MachineRepresentation::kWord32);
1018 }
1019 }
1020
1021 if (op == nullptr) {
1022 return TypeError(node, output_rep, output_type,
1023 MachineRepresentation::kWord32);
1024 }
1025 return InsertConversion(node, op, use_node);
1026}

1028Node* RepresentationChanger::InsertConversion(Node* node, const Operator* op,
1029 Node* use_node) {
1030 if (op->ControlInputCount() > 0) {
1031 // If the operator can deoptimize (which means it has control
1032 // input), we need to connect it to the effect and control chains.
1033 Node* effect = NodeProperties::GetEffectInput(use_node);
1034 Node* control = NodeProperties::GetControlInput(use_node);
1035 Node* conversion = jsgraph()->graph()->NewNode(op, node, effect, control);
1036 NodeProperties::ReplaceEffectInput(use_node, conversion);
1037 return conversion;
1038 }
1039 return jsgraph()->graph()->NewNode(op, node);
1040}

1042Node* RepresentationChanger::GetBitRepresentationFor(
1043 Node* node, MachineRepresentation output_rep, Type output_type) {
1044 // Eagerly fold representation changes for constants.
1045 switch (node->opcode()) {
1046 case IrOpcode::kHeapConstant: {
1047 HeapObjectMatcher m(node);
1048 if (m.Is(factory()->false_value())) {
1049 return InsertTypeOverrideForVerifier(
1050 Type::Constant(broker_, broker_->false_value(), jsgraph()->zone()),
1051 jsgraph()->Int32Constant(0));
1052 } else if (m.Is(factory()->true_value())) {
1053 return InsertTypeOverrideForVerifier(
1054 Type::Constant(broker_, broker_->true_value(), jsgraph()->zone()),
1055 jsgraph()->Int32Constant(1));
1056 }
1057 break;
1058 }
1059 default:
1060 break;
1061 }
1062 // Select the correct X -> Bit operator.
1063 const Operator* op;
1064 if (output_type.Is(Type::None())) {
1065 // This is an impossible value; it should not be used at runtime.
1066 return jsgraph()->graph()->NewNode(
1067 jsgraph()->common()->DeadValue(MachineRepresentation::kBit), node);
1068 } else if (output_rep == MachineRepresentation::kTagged ||
1069 output_rep == MachineRepresentation::kTaggedPointer) {
1070 if (output_type.Is(Type::BooleanOrNullOrUndefined())) {
1071 // true is the only trueish Oddball.
1072 op = simplified()->ChangeTaggedToBit();
1073 } else {
1074 if (output_rep == MachineRepresentation::kTagged &&
1075 output_type.Maybe(Type::SignedSmall())) {
1076 op = simplified()->TruncateTaggedToBit();
1077 } else {
1078 // The {output_type} either doesn't include the Smi range,
1079 // or the {output_rep} is known to be TaggedPointer.
1080 op = simplified()->TruncateTaggedPointerToBit();
1081 }
1082 }
1083 } else if (output_rep == MachineRepresentation::kTaggedSigned) {
1084 if (COMPRESS_POINTERS_BOOL) {
1085 node = jsgraph()->graph()->NewNode(machine()->Word32Equal(), node,
1086 jsgraph()->Int32Constant(0));
1087 } else {
1088 node = jsgraph()->graph()->NewNode(machine()->WordEqual(), node,
1089 jsgraph()->IntPtrConstant(0));
1090 }
1091 return jsgraph()->graph()->NewNode(machine()->Word32Equal(), node,
1092 jsgraph()->Int32Constant(0));
1093 } else if (IsWord(output_rep)) {
1094 node = jsgraph()->graph()->NewNode(machine()->Word32Equal(), node,
1095 jsgraph()->Int32Constant(0));
1096 return jsgraph()->graph()->NewNode(machine()->Word32Equal(), node,
1097 jsgraph()->Int32Constant(0));
1098 } else if (output_rep == MachineRepresentation::kWord64) {
1099 node = jsgraph()->graph()->NewNode(machine()->Word64Equal(), node,
1100 jsgraph()->Int64Constant(0));
1101 return jsgraph()->graph()->NewNode(machine()->Word32Equal(), node,
1102 jsgraph()->Int32Constant(0));
1103 } else if (output_rep == MachineRepresentation::kFloat32) {
1104 node = jsgraph()->graph()->NewNode(machine()->Float32Abs(), node);
1105 return jsgraph()->graph()->NewNode(machine()->Float32LessThan(),
1106 jsgraph()->Float32Constant(0.0), node);
1107 } else if (output_rep == MachineRepresentation::kFloat64) {
1108 node = jsgraph()->graph()->NewNode(machine()->Float64Abs(), node);
1109 return jsgraph()->graph()->NewNode(machine()->Float64LessThan(),
1110 jsgraph()->Float64Constant(0.0), node);
1111 } else {
1112 return TypeError(node, output_rep, output_type,
1113 MachineRepresentation::kBit);
1114 }
1115 return jsgraph()->graph()->NewNode(op, node);
1116}

1118Node* RepresentationChanger::GetWord64RepresentationFor(
1119 Node* node, MachineRepresentation output_rep, Type output_type,
1120 Node* use_node, UseInfo use_info) {
1121 // Eagerly fold representation changes for constants.
1122 switch (node->opcode()) {
1123 case IrOpcode::kInt32Constant:
1124 case IrOpcode::kInt64Constant:
1125 case IrOpcode::kFloat32Constant:
1126 case IrOpcode::kFloat64Constant:
1127 UNREACHABLE();
1128 case IrOpcode::kNumberConstant: {
1129 if (!TypeCheckIsBigInt(use_info.type_check())) {
1130 double const fv = OpParameter<double>(node->op());
1131 if (base::IsValueInRangeForNumericType<int64_t>(fv)) {
1132 int64_t const iv = static_cast<int64_t>(fv);
1133 if (static_cast<double>(iv) == fv) {
1134 return InsertTypeOverrideForVerifier(NodeProperties::GetType(node),
1135 jsgraph()->Int64Constant(iv));
1136 }
1137 }
1138 }
1139 break;
1140 }
1141 case IrOpcode::kHeapConstant: {
1142 HeapObjectMatcher m(node);
1143 if (m.HasResolvedValue() && m.Ref(broker_).IsBigInt() &&
1144 (Is64() && use_info.truncation().IsUsedAsWord64())) {
1145 BigIntRef bigint = m.Ref(broker_).AsBigInt();
1146 return InsertTypeOverrideForVerifier(
1147 NodeProperties::GetType(node),
1148 jsgraph()->Int64Constant(static_cast<int64_t>(bigint.AsUint64())));
1149 }
1150 break;
1151 }
1152 default:
1153 break;
1154 }
1155
1156 if (TypeCheckIsBigInt(use_info.type_check())) {
1157 // BigInts are only represented as tagged pointer and word64.
1158 if (!CanBeTaggedPointer(output_rep) &&
1159 output_rep != MachineRepresentation::kWord64) {
1160 DCHECK(!output_type.Equals(Type::BigInt()));
1161 Node* unreachable = InsertUnconditionalDeopt(
1162 use_node, DeoptimizeReason::kNotABigInt, use_info.feedback());
1163 return jsgraph()->graph()->NewNode(
1164 jsgraph()->common()->DeadValue(MachineRepresentation::kWord64),
1165 unreachable);
1166 }
1167 }
1168
1169 // Select the correct X -> Word64 operator.
1170 const Operator* op;
1171 if (output_type.Is(Type::None())) {
1172 // This is an impossible value; it should not be used at runtime.
1173 return jsgraph()->graph()->NewNode(
1174 jsgraph()->common()->DeadValue(MachineRepresentation::kWord64), node);
1175 } else if (output_rep == MachineRepresentation::kBit) {
1176 CHECK(output_type.Is(Type::Boolean()));
1177 CHECK_NE(use_info.type_check(), TypeCheckKind::kNone);
1178 CHECK_NE(use_info.type_check(), TypeCheckKind::kNumberOrOddball);
1179 CHECK_NE(use_info.type_check(), TypeCheckKind::kBigInt);
1180 CHECK_NE(use_info.type_check(), TypeCheckKind::kBigInt64);
1181 Node* unreachable =
1182 InsertUnconditionalDeopt(use_node, DeoptimizeReason::kNotASmi);
1183 return jsgraph()->graph()->NewNode(
1184 jsgraph()->common()->DeadValue(MachineRepresentation::kWord64),
1185 unreachable);
1186 } else if (IsWord(output_rep)) {
1187 if (output_type.Is(Type::Unsigned32OrMinusZero())) {
1188 // uint32 -> uint64
1189 CHECK_IMPLIES(output_type.Maybe(Type::MinusZero()),
1190 use_info.truncation().IdentifiesZeroAndMinusZero());
1191 op = machine()->ChangeUint32ToUint64();
1192 } else if (output_type.Is(Type::Signed32OrMinusZero())) {
1193 // int32 -> int64
1194 CHECK_IMPLIES(output_type.Maybe(Type::MinusZero()),
1195 use_info.truncation().IdentifiesZeroAndMinusZero());
1196 op = machine()->ChangeInt32ToInt64();
1197 } else {
1198 return TypeError(node, output_rep, output_type,
1199 MachineRepresentation::kWord64);
1200 }
1201 } else if (output_rep == MachineRepresentation::kFloat32) {
1202 if (output_type.Is(cache_->kDoubleRepresentableInt64) ||
1203 (output_type.Is(cache_->kDoubleRepresentableInt64OrMinusZero) &&
1204 use_info.truncation().IdentifiesZeroAndMinusZero())) {
1205 // float32 -> float64 -> int64
1206 node = InsertChangeFloat32ToFloat64(node);
1207 op = machine()->ChangeFloat64ToInt64();
1208 } else if (output_type.Is(cache_->kDoubleRepresentableUint64)) {
1209 // float32 -> float64 -> uint64
1210 node = InsertChangeFloat32ToFloat64(node);
1211 op = machine()->ChangeFloat64ToUint64();
1212 } else if (use_info.type_check() == TypeCheckKind::kSigned64 ||
1213 use_info.type_check() == TypeCheckKind::kArrayIndex) {
1214 // float32 -> float64 -> int64
1215 node = InsertChangeFloat32ToFloat64(node);
1216 op = simplified()->CheckedFloat64ToInt64(
1217 output_type.Maybe(Type::MinusZero())
1218 ? use_info.minus_zero_check()
1219 : CheckForMinusZeroMode::kDontCheckForMinusZero,
1220 use_info.feedback());
1221 } else {
1222 return TypeError(node, output_rep, output_type,
1223 MachineRepresentation::kWord64);
1224 }
1225 } else if (output_rep == MachineRepresentation::kFloat64) {
1226 if (output_type.Is(cache_->kDoubleRepresentableInt64) ||
1227 (output_type.Is(cache_->kDoubleRepresentableInt64OrMinusZero) &&
1228 use_info.truncation().IdentifiesZeroAndMinusZero())) {
1229 op = machine()->ChangeFloat64ToInt64();
1230 } else if (output_type.Is(cache_->kDoubleRepresentableUint64)) {
1231 op = machine()->ChangeFloat64ToUint64();
1232 } else if (use_info.type_check() == TypeCheckKind::kSigned64 ||
1233 use_info.type_check() == TypeCheckKind::kArrayIndex) {
1234 op = simplified()->CheckedFloat64ToInt64(
1235 output_type.Maybe(Type::MinusZero())
1236 ? use_info.minus_zero_check()
1237 : CheckForMinusZeroMode::kDontCheckForMinusZero,
1238 use_info.feedback());
1239 } else {
1240 return TypeError(node, output_rep, output_type,
1241 MachineRepresentation::kWord64);
1242 }
1243 } else if (output_rep == MachineRepresentation::kTaggedSigned) {
1244 if (output_type.Is(Type::SignedSmall())) {
1245 op = simplified()->ChangeTaggedSignedToInt64();
1246 } else {
1247 return TypeError(node, output_rep, output_type,
1248 MachineRepresentation::kWord64);
1249 }
1250 } else if (IsAnyTagged(output_rep) &&
1251 ((Is64() && use_info.truncation().IsUsedAsWord64() &&
1252 (use_info.type_check() == TypeCheckKind::kBigInt ||
1253 output_type.Is(Type::BigInt()))) ||
1254 use_info.type_check() == TypeCheckKind::kBigInt64)) {
1255 node = GetTaggedPointerRepresentationFor(node, output_rep, output_type,
1256 use_node, use_info);
1257 op = simplified()->TruncateBigIntToWord64();
1258 } else if (CanBeTaggedPointer(output_rep)) {
1259 if (output_type.Is(cache_->kDoubleRepresentableInt64) ||
1260 (output_type.Is(cache_->kDoubleRepresentableInt64OrMinusZero) &&
1261 use_info.truncation().IdentifiesZeroAndMinusZero())) {
1262 op = simplified()->ChangeTaggedToInt64();
1263 } else if (use_info.type_check() == TypeCheckKind::kSigned64) {
1264 op = simplified()->CheckedTaggedToInt64(
1265 output_type.Maybe(Type::MinusZero())
1266 ? use_info.minus_zero_check()
1267 : CheckForMinusZeroMode::kDontCheckForMinusZero,
1268 use_info.feedback());
1269 } else if (use_info.type_check() == TypeCheckKind::kArrayIndex) {
1270 op = simplified()->CheckedTaggedToArrayIndex(use_info.feedback());
1271 } else {
1272 return TypeError(node, output_rep, output_type,
1273 MachineRepresentation::kWord64);
1274 }
1275 } else if (output_rep == MachineRepresentation::kWord64) {
1276 DCHECK(TypeCheckIsBigInt(use_info.type_check()));
1277 if (output_type.Is(Type::UnsignedBigInt64()) &&
1278 use_info.type_check() == TypeCheckKind::kBigInt64) {
1279 op = simplified()->CheckedUint64ToInt64(use_info.feedback());
1280 } else if ((output_type.Is(Type::BigInt()) &&
1281 use_info.type_check() == TypeCheckKind::kBigInt) ||
1282 (output_type.Is(Type::SignedBigInt64()) &&
1283 use_info.type_check() == TypeCheckKind::kBigInt64)) {
1284 return node;
1285 } else {
1286 DCHECK(output_type != Type::BigInt() ||
1287 use_info.type_check() != TypeCheckKind::kBigInt64);
1288 Node* unreachable = InsertUnconditionalDeopt(
1289 use_node, DeoptimizeReason::kNotABigInt, use_info.feedback());
1290 return jsgraph()->graph()->NewNode(
1291 jsgraph()->common()->DeadValue(MachineRepresentation::kWord64),
1292 unreachable);
1293 }
1294 } else if (output_rep == MachineRepresentation::kSandboxedPointer) {
1295 if (output_type.Is(Type::SandboxedPointer())) {
1296 return node;
1297 } else {
1298 return TypeError(node, output_rep, output_type,
1299 MachineRepresentation::kWord64);
1300 }
1301 } else {
1302 return TypeError(node, output_rep, output_type,
1303 MachineRepresentation::kWord64);
1304 }
1305 return InsertConversion(node, op, use_node);
