20MaybeHandle<JSV8BreakIterator> JSV8BreakIterator::New(
21 Isolate* isolate, DirectHandle<Map> map, DirectHandle<Object> locales,
22 DirectHandle<Object> options_obj, const char* service) {
23 Factory* factory = isolate->factory();
24
25 // 1. Let requestedLocales be ? CanonicalizeLocaleList(locales).
26 Maybe<std::vector<std::string>> maybe_requested_locales =
27 Intl::CanonicalizeLocaleList(isolate, locales);
28 MAYBE_RETURN(maybe_requested_locales, MaybeHandle<JSV8BreakIterator>());
29 std::vector<std::string> requested_locales =
30 maybe_requested_locales.FromJust();
31
32 DirectHandle<JSReceiver> options;
33 if (IsUndefined(*options_obj, isolate)) {
34 options = factory->NewJSObjectWithNullProto();
35 } else {
36 ASSIGN_RETURN_ON_EXCEPTION(isolate, options,
37 Object::ToObject(isolate, options_obj, service));
38 }
39
40 // Extract locale string
41 Maybe<Intl::MatcherOption> maybe_locale_matcher =
42 Intl::GetLocaleMatcher(isolate, options, service);
43 MAYBE_RETURN(maybe_locale_matcher, MaybeHandle<JSV8BreakIterator>());
44 Intl::MatcherOption matcher = maybe_locale_matcher.FromJust();
45
46 Maybe<Intl::ResolvedLocale> maybe_resolve_locale =
47 Intl::ResolveLocale(isolate, JSV8BreakIterator::GetAvailableLocales(),
48 requested_locales, matcher, {});
49 if (maybe_resolve_locale.IsNothing()) {
50 THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
51 }
52 Intl::ResolvedLocale r = maybe_resolve_locale.FromJust();
53
54 // Extract type from options
55 enum class Type { CHARACTER, WORD, SENTENCE, LINE };
56 Maybe<Type> maybe_type = GetStringOption<Type>(
57 isolate, options, "type", service,
58 {"word", "character", "sentence", "line"},
59 {Type::WORD, Type::CHARACTER, Type::SENTENCE, Type::LINE}, Type::WORD);
60 MAYBE_RETURN(maybe_type, MaybeHandle<JSV8BreakIterator>());
61 Type type_enum = maybe_type.FromJust();
62
63 icu::Locale icu_locale = r.icu_locale;
64 DCHECK(!icu_locale.isBogus());
65
66 // Construct break_iterator using icu_locale and type
67 UErrorCode status = U_ZERO_ERROR;
68 std::unique_ptr<icu::BreakIterator> break_iterator = nullptr;
69 switch (type_enum) {
70 case Type::CHARACTER:
71 break_iterator.reset(
72 icu::BreakIterator::createCharacterInstance(icu_locale, status));
73 break;
74 case Type::SENTENCE:
75 break_iterator.reset(
76 icu::BreakIterator::createSentenceInstance(icu_locale, status));
77 break;
78 case Type::LINE:
79 isolate->CountUsage(
80 v8::Isolate::UseCounterFeature::kBreakIteratorTypeLine);
81 break_iterator.reset(
82 icu::BreakIterator::createLineInstance(icu_locale, status));
83 break;
84 default:
85 isolate->CountUsage(
86 v8::Isolate::UseCounterFeature::kBreakIteratorTypeWord);
87 break_iterator.reset(
88 icu::BreakIterator::createWordInstance(icu_locale, status));
89 break;
90 }
91
92 // Error handling for break_iterator
93 if (U_FAILURE(status) || break_iterator == nullptr) {
94 THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
95 }
96 isolate->CountUsage(v8::Isolate::UseCounterFeature::kBreakIterator);
97
98 // Construct managed objects from pointers
99 DirectHandle<Managed<icu::BreakIterator>> managed_break_iterator =
100 Managed<icu::BreakIterator>::From(isolate, 0, std::move(break_iterator));
101 DirectHandle<Managed<icu::UnicodeString>> managed_unicode_string =
102 Managed<icu::UnicodeString>::From(isolate, 0, nullptr);
103
104 DirectHandle<String> locale_str =
105 isolate->factory()->NewStringFromAsciiChecked(r.locale.c_str());
106
107 // Now all properties are ready, so we can allocate the result object.
108 Handle<JSV8BreakIterator> break_iterator_holder = Cast<JSV8BreakIterator>(
109 isolate->factory()->NewFastOrSlowJSObjectFromMap(map));
110 DisallowGarbageCollection no_gc;
111 break_iterator_holder->set_locale(*locale_str);
112 break_iterator_holder->set_break_iterator(*managed_break_iterator);
113 break_iterator_holder->set_unicode_string(*managed_unicode_string);
114
115 // Return break_iterator_holder
116 return break_iterator_holder;
117}

119Handle<JSObject> JSV8BreakIterator::ResolvedOptions(
120 Isolate* isolate, DirectHandle<JSV8BreakIterator> break_iterator) {
121 Factory* factory = isolate->factory();
122 const auto as_string = [isolate](icu::BreakIterator* break_iterator) {
123 // Since the developer calling the Intl.v8BreakIterator already know the
124 // type, we usually do not need to know the type unless the
125 // resolvedOptions() is called, we use the following trick to figure out the
126 // type instead of storing it with the JSV8BreakIterator object to save
127 // memory. This routine is not fast but should be seldomly used only.
128
129 // We need to clone a copy of break iteator because we need to setText to
130 // it.
131 std::unique_ptr<icu::BreakIterator> cloned_break_iterator(
132 break_iterator->clone());
133 // Use a magic string "He is." to call next().
134 // character type: will return 1 for "H"
135 // word type: will return 2 for "He"
136 // line type: will return 3 for "He "
137 // sentence type: will return 6 for "He is."
138 icu::UnicodeString data("He is.");
139 cloned_break_iterator->setText(data);
140 switch (cloned_break_iterator->next()) {
141 case 1: // After "H"
142 return isolate->factory()->character_string();
143 case 2: // After "He"
144 return isolate->factory()->word_string();
145 case 3: // After "He "
146 return isolate->factory()->line_string();
147 case 6: // After "He is."
148 return isolate->factory()->sentence_string();
149 default:
150 UNREACHABLE();
151 }
152 };
153
154 Handle<JSObject> result = factory->NewJSObject(isolate->object_function());
155 DirectHandle<String> locale(break_iterator->locale(), isolate);
156
157 JSObject::AddProperty(isolate, result, factory->locale_string(), locale,
158 NONE);
159 JSObject::AddProperty(isolate, result, factory->type_string(),
160 as_string(break_iterator->break_iterator()->raw()),
161 NONE);
162 return result;
163}

194Tagged<String> JSV8BreakIterator::BreakType(
195 Isolate* isolate, DirectHandle<JSV8BreakIterator> break_iterator) {
196 int32_t status = break_iterator->break_iterator()->raw()->getRuleStatus();
197 // Keep return values in sync with JavaScript BreakType enum.
198 if (status >= UBRK_WORD_NONE && status < UBRK_WORD_NONE_LIMIT) {
199 return ReadOnlyRoots(isolate).none_string();
200 }
201 if (status >= UBRK_WORD_NUMBER && status < UBRK_WORD_NUMBER_LIMIT) {
202 return ReadOnlyRoots(isolate).number_string();
203 }
204 if (status >= UBRK_WORD_LETTER && status < UBRK_WORD_LETTER_LIMIT) {
205 return ReadOnlyRoots(isolate).letter_string();
206 }
207 if (status >= UBRK_WORD_KANA && status < UBRK_WORD_KANA_LIMIT) {
208 return ReadOnlyRoots(isolate).kana_string();
209 }
210 if (status >= UBRK_WORD_IDEO && status < UBRK_WORD_IDEO_LIMIT) {
211 return ReadOnlyRoots(isolate).ideo_string();
212 }
213 return ReadOnlyRoots(isolate).unknown_string();
214}
