17// ES#sec-canonicalnumericindexstring
18// Returns true if the lookup_key represents a valid index string.
19bool CanonicalNumericIndexString(Isolate* isolate,
20 const PropertyKey& lookup_key,
21 bool* is_minus_zero) {
22 // 1. Assert: Type(argument) is String.
23 DCHECK(lookup_key.is_element() || IsString(*lookup_key.name()));
24 *is_minus_zero = false;
25 if (lookup_key.is_element()) return true;
26
27 DirectHandle<String> key = Cast<String>(lookup_key.name());
28
29 // 3. Let n be ! ToNumber(argument).
30 DirectHandle<Object> result = String::ToNumber(isolate, key);
31 if (IsMinusZero(*result)) {
32 // 2. If argument is "-0", return -0ùîΩ.
33 // We are not performing SaveValue check for -0 because it'll be rejected
34 // anyway.
35 *is_minus_zero = true;
36 } else {
37 // 4. If SameValue(! ToString(n), argument) is false, return undefined.
38 DirectHandle<String> str =
39 Object::ToString(isolate, result).ToHandleChecked();
40 // Avoid treating strings like "2E1" and "20" as the same key.
41 if (!Object::SameValue(*str, *key)) return false;
42 }
43 return true;
44}

47void JSArrayBuffer::Setup(SharedFlag shared, ResizableFlag resizable,
48 std::shared_ptr<BackingStore> backing_store,
49 Isolate* isolate) {
50 if (shared == SharedFlag::kShared) {
51 isolate->CountUsage(
52 v8::Isolate::UseCounterFeature::kSharedArrayBufferConstructed);
53 }
54 clear_padding();
55 init_extension();
56 set_detach_key(ReadOnlyRoots(isolate).undefined_value());
57 set_bit_field(0);
58 set_is_shared(shared == SharedFlag::kShared);
59 set_is_resizable_by_js(resizable == ResizableFlag::kResizable);
60 set_is_detachable(shared != SharedFlag::kShared);
61 SetupLazilyInitializedCppHeapPointerField(
62 JSAPIObjectWithEmbedderSlots::kCppHeapWrappableOffset);
63 for (int i = 0; i < v8::ArrayBuffer::kEmbedderFieldCount; i++) {
64 SetEmbedderField(i, Smi::zero());
65 }
66 if (!backing_store) {
67 set_backing_store(isolate, EmptyBackingStoreBuffer());
68 set_byte_length(0);
69 set_max_byte_length(0);
70 return;
71 }
72 // Rest of the code here deals with attaching an the BackingStore.
73 DCHECK_EQ(is_shared(), backing_store->is_shared());
74 DCHECK_EQ(is_resizable_by_js(), backing_store->is_resizable_by_js());
75 DCHECK_IMPLIES(
76 !backing_store->is_wasm_memory() && !backing_store->is_resizable_by_js(),
77 backing_store->byte_length() == backing_store->max_byte_length());
78
79 void* backing_store_buffer = backing_store->buffer_start();
80 // Wasm memory always needs a backing store; this is guaranteed by reserving
81 // at least one page for the BackingStore (so {IsEmpty()} is always false).
82 DCHECK_IMPLIES(backing_store->is_wasm_memory(), !backing_store->IsEmpty());
83 // Non-empty backing stores must start at a non-null pointer.
84 DCHECK_IMPLIES(backing_store_buffer == EmptyBackingStoreBuffer(),
85 backing_store->IsEmpty());
86 // Empty backing stores can be backed by an empty buffer pointer or by an
87 // externally provided pointer: Either is acceptable. However, the pointer
88 // must always point into the sandbox, so nullptr is not acceptable if the
89 // sandbox is enabled.
90 DCHECK_IMPLIES(V8_ENABLE_SANDBOX_BOOL, backing_store_buffer != nullptr);
91 set_backing_store(isolate, backing_store_buffer);
92
93 // GSABs need to read their byte_length from the BackingStore. Maintain the
94 // invariant that their byte_length field is always 0.
95 auto byte_len =
96 (is_shared() && is_resizable_by_js()) ? 0 : backing_store->byte_length();
97 CHECK_LE(backing_store->byte_length(), kMaxByteLength);
98 set_byte_length(byte_len);
99
100 // For Wasm memories, it is possible for the backing store maximum to be
101 // different from the JSArrayBuffer maximum. The maximum pages allowed on a
102 // Wasm memory are tracked on the Wasm memory object, and not the
103 // JSArrayBuffer associated with it.
104 auto max_byte_len = is_resizable_by_js() ? backing_store->max_byte_length()
105 : backing_store->byte_length();
106 set_max_byte_length(max_byte_len);
107
108 if (backing_store->is_wasm_memory()) {
109 set_is_detachable(false);
110 }
111
112 CreateExtension(isolate, std::move(backing_store));
113}

115Maybe<bool> JSArrayBuffer::Detach(DirectHandle<JSArrayBuffer> buffer,
116 bool force_for_wasm_memory,
117 DirectHandle<Object> maybe_key) {
118 Isolate* const isolate = buffer->GetIsolate();
119
120 DirectHandle<Object> detach_key(buffer->detach_key(), isolate);
121
122 bool key_mismatch = false;
123
124 if (!IsUndefined(*detach_key, isolate)) {
125 key_mismatch =
126 maybe_key.is_null() || !Object::StrictEquals(*maybe_key, *detach_key);
127 } else {
128 // Detach key is undefined; allow not passing maybe_key but disallow passing
129 // something else than undefined.
130 key_mismatch =
131 !maybe_key.is_null() && !Object::StrictEquals(*maybe_key, *detach_key);
132 }
133 if (key_mismatch) {
134 THROW_NEW_ERROR_RETURN_VALUE(
135 isolate,
136 NewTypeError(MessageTemplate::kArrayBufferDetachKeyDoesntMatch),
137 Nothing<bool>());
138 }
139
140 if (buffer->was_detached()) return Just(true);
141
142 if (force_for_wasm_memory) {
143 // Skip the is_detachable() check.
144 } else if (!buffer->is_detachable()) {
145 // Not detachable, do nothing.
146 return Just(true);
147 }
148
149 buffer->DetachInternal(force_for_wasm_memory, isolate);
150 return Just(true);
151}

153void JSArrayBuffer::DetachInternal(bool force_for_wasm_memory,
154 Isolate* isolate) {
155 ArrayBufferExtension* extension = this->extension();
156
157 if (extension) {
158 DisallowGarbageCollection disallow_gc;
159 isolate->heap()->DetachArrayBufferExtension(extension);
160 std::shared_ptr<BackingStore> backing_store = RemoveExtension();
161 CHECK_IMPLIES(force_for_wasm_memory, backing_store->is_wasm_memory());
162 }
163
164 if (Protectors::IsArrayBufferDetachingIntact(isolate)) {
165 Protectors::InvalidateArrayBufferDetaching(isolate);
166 }
167
168 DCHECK(!is_shared());
169 set_backing_store(isolate, EmptyBackingStoreBuffer());
170 set_byte_length(0);
171 set_was_detached(true);
172}

240ArrayBufferExtension* JSArrayBuffer::CreateExtension(
241 Isolate* isolate, std::shared_ptr<BackingStore> backing_store) {
242 // `Heap::InYoungGeneration` during full GC with sticky markbits is generally
243 // inaccurate. However, a full GC will sweep both lists and promote all to
244 // old, so it doesn't matter which list initially holds the extension in this
245 // case.
246 const auto age =
247 HeapLayout::InYoungGeneration(UncheckedCast<JSArrayBuffer>(*this))
248 ? ArrayBufferExtension::Age::kYoung
249 : ArrayBufferExtension::Age::kOld;
250 ArrayBufferExtension* extension =
251 new ArrayBufferExtension(std::move(backing_store), age);
252 set_extension(extension);
253 isolate->heap()->AppendArrayBufferExtension(extension);
254 return extension;
255}

257std::shared_ptr<BackingStore> JSArrayBuffer::RemoveExtension() {
258 ArrayBufferExtension* extension = this->extension();
259 DCHECK_NOT_NULL(extension);
260 auto result = extension->RemoveBackingStore();
261 // Remove pointer to extension such that the next GC will free it
262 // automatically.
263 set_extension(nullptr);
264 return result;
265}

289Handle<JSArrayBuffer> JSTypedArray::GetBuffer() {
290 Isolate* isolate = GetIsolate();
291 DirectHandle<JSTypedArray> self(*this, isolate);
292 DCHECK(IsTypedArrayOrRabGsabTypedArrayElementsKind(self->GetElementsKind()));
293 Handle<JSArrayBuffer> array_buffer(Cast<JSArrayBuffer>(self->buffer()),
294 isolate);
295 if (!is_on_heap()) {
296 // Already is off heap, so return the existing buffer.
297 return array_buffer;
298 }
299 DCHECK(!array_buffer->is_resizable_by_js());
300
301 // The existing array buffer should be empty.
302 DCHECK(array_buffer->IsEmpty());
303
304 // Allocate a new backing store and attach it to the existing array buffer.
305 size_t byte_length = self->byte_length();
306 auto backing_store =
307 BackingStore::Allocate(isolate, byte_length, SharedFlag::kNotShared,
308 InitializedFlag::kUninitialized);
309
310 if (!backing_store) {
311 isolate->heap()->FatalProcessOutOfMemory("JSTypedArray::GetBuffer");
312 }
313
314 // Copy the elements into the backing store of the array buffer.
315 if (byte_length > 0) {
316 memcpy(backing_store->buffer_start(), self->DataPtr(), byte_length);
317 }
318
319 // Attach the backing store to the array buffer.
320 array_buffer->Setup(SharedFlag::kNotShared, ResizableFlag::kNotResizable,
321 std::move(backing_store), isolate);
322
323 // Clear the elements of the typed array.
324 self->set_elements(ReadOnlyRoots(isolate).empty_byte_array());
325 self->SetOffHeapDataPtr(isolate, array_buffer->backing_store(), 0);
326 DCHECK(!self->is_on_heap());
327
328 return array_buffer;
329}

331// ES#sec-integer-indexed-exotic-objects-defineownproperty-p-desc
332// static
333Maybe<bool> JSTypedArray::DefineOwnProperty(Isolate* isolate,
334 DirectHandle<JSTypedArray> o,
335 DirectHandle<Object> key,
336 PropertyDescriptor* desc,
337 Maybe<ShouldThrow> should_throw) {
338 DCHECK(IsName(*key) || IsNumber(*key));
339 // 1. If Type(P) is String, then
340 PropertyKey lookup_key(isolate, key);
341 if (lookup_key.is_element() || IsSmi(*key) || IsString(*key)) {
342 // 1a. Let numericIndex be ! CanonicalNumericIndexString(P)
343 // 1b. If numericIndex is not undefined, then
344 bool is_minus_zero = false;
345 if (IsSmi(*key) || // Smi keys are definitely canonical
346 CanonicalNumericIndexString(isolate, lookup_key, &is_minus_zero)) {
347 // 1b i. If IsValidIntegerIndex(O, numericIndex) is false, return false.
348
349 // IsValidIntegerIndex:
350 size_t index = lookup_key.index();
351 bool out_of_bounds = false;
352 size_t length = o->GetLengthOrOutOfBounds(out_of_bounds);
353 if (o->WasDetached() || out_of_bounds || index >= length) {
354 RETURN_FAILURE(isolate, GetShouldThrow(isolate, should_throw),
355 NewTypeError(MessageTemplate::kInvalidTypedArrayIndex));
356 }
357 if (!lookup_key.is_element() || is_minus_zero) {
358 RETURN_FAILURE(isolate, GetShouldThrow(isolate, should_throw),
359 NewTypeError(MessageTemplate::kInvalidTypedArrayIndex));
360 }
361
362 // 1b ii. If Desc has a [[Configurable]] field and if
363 // Desc.[[Configurable]] is false, return false.
364 // 1b iii. If Desc has an [[Enumerable]] field and if Desc.[[Enumerable]]
365 // is false, return false.
366 // 1b iv. If IsAccessorDescriptor(Desc) is true, return false.
367 // 1b v. If Desc has a [[Writable]] field and if Desc.[[Writable]] is
368 // false, return false.
369
370 if (PropertyDescriptor::IsAccessorDescriptor(desc)) {
371 RETURN_FAILURE(isolate, GetShouldThrow(isolate, should_throw),
372 NewTypeError(MessageTemplate::kRedefineDisallowed, key));
373 }
374
375 if ((desc->has_configurable() && !desc->configurable()) ||
376 (desc->has_enumerable() && !desc->enumerable()) ||
377 (desc->has_writable() && !desc->writable())) {
378 RETURN_FAILURE(isolate, GetShouldThrow(isolate, should_throw),
379 NewTypeError(MessageTemplate::kRedefineDisallowed, key));
380 }
381
382 // 1b vi. If Desc has a [[Value]] field, perform
383 // ? IntegerIndexedElementSet(O, numericIndex, Desc.[[Value]]).
384 if (desc->has_value()) {
385 if (!desc->has_configurable()) desc->set_configurable(true);
386 if (!desc->has_enumerable()) desc->set_enumerable(true);
387 if (!desc->has_writable()) desc->set_writable(true);
388 DirectHandle<Object> value = desc->value();
389 LookupIterator it(isolate, o, index, LookupIterator::OWN);
390 RETURN_ON_EXCEPTION_VALUE(
391 isolate,
392 DefineOwnPropertyIgnoreAttributes(&it, value, desc->ToAttributes()),
393 Nothing<bool>());
394 }
395 // 1b vii. Return true.
396 return Just(true);
397 }
398 }
399 // 4. Return ! OrdinaryDefineOwnProperty(O, P, Desc).
400 return OrdinaryDefineOwnProperty(isolate, o, lookup_key, desc, should_throw);
401}

433size_t JSTypedArray::LengthTrackingGsabBackedTypedArrayLength(
434 Isolate* isolate, Address raw_array) {
435 // TODO(v8:11111): Cache the last seen length in JSArrayBuffer and use it
436 // in bounds checks to minimize the need for calling this function.
437 DisallowGarbageCollection no_gc;
438 DisallowJavascriptExecution no_js(isolate);
439 Tagged<JSTypedArray> array = Cast<JSTypedArray>(Tagged<Object>(raw_array));
440 CHECK(array->is_length_tracking());
441 Tagged<JSArrayBuffer> buffer = array->buffer();
442 CHECK(buffer->is_resizable_by_js());
443 CHECK(buffer->is_shared());
444 size_t backing_byte_length =
445 buffer->GetBackingStore()->byte_length(std::memory_order_seq_cst);
446 CHECK_GE(backing_byte_length, array->byte_offset());
447 auto element_byte_size = ElementsKindToByteSize(array->GetElementsKind());
448 return (backing_byte_length - array->byte_offset()) / element_byte_size;
449}

451size_t JSTypedArray::GetVariableLengthOrOutOfBounds(bool& out_of_bounds) const {
452 DCHECK(!WasDetached());
453 if (is_length_tracking()) {
454 if (is_backed_by_rab()) {
455 if (byte_offset() > buffer()->byte_length()) {
456 out_of_bounds = true;
457 return 0;
458 }
459 return (buffer()->byte_length() - byte_offset()) / element_size();
460 }
461 if (byte_offset() >
462 buffer()->GetBackingStore()->byte_length(std::memory_order_seq_cst)) {
463 out_of_bounds = true;
464 return 0;
465 }
466 return (buffer()->GetBackingStore()->byte_length(
467 std::memory_order_seq_cst) -
468 byte_offset()) /
469 element_size();
470 }
471 DCHECK(is_backed_by_rab());
472 size_t array_length = LengthUnchecked();
473 // The sum can't overflow, since we have managed to allocate the
474 // JSTypedArray.
475 if (byte_offset() + array_length * element_size() > buffer()->byte_length()) {
476 out_of_bounds = true;
477 return 0;
478 }
479 return array_length;
480}
