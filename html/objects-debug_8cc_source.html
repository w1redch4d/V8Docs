108// Heap Verification Overview
109// --------------------------
110// - Each InstanceType has a separate XXXVerify method which checks an object's
111// integrity in isolation.
112// - --verify-heap will iterate over all gc spaces and call ObjectVerify() on
113// every encountered tagged pointer.
114// - Verification should be pushed down to the specific instance type if its
115// integrity is independent of an outer object.
116// - In cases where the InstanceType is too generic (e.g. FixedArray) the
117// XXXVerify of the outer method has to do recursive verification.
118// - If the corresponding objects have inheritence the parent's Verify method
119// is called as well.
120// - For any field containing pointes VerifyPointer(...) should be called.
121//
122// Caveats
123// -------
124// - Assume that any of the verify methods is incomplete!
125// - Some integrity checks are only partially done due to objects being in
126// partially initialized states when a gc happens, for instance when outer
127// objects are allocted before inner ones.
128//

190void HeapObject::HeapObjectVerify(Isolate* isolate) {
191 CHECK(IsHeapObject(*this));
192 PtrComprCageBase cage_base(isolate);
193 Object::VerifyPointer(isolate, map(cage_base));
194 CHECK(IsMap(map(cage_base), cage_base));
195
196 CHECK(CheckRequiredAlignment(isolate));
197
198 // Only TrustedObjects live in trusted space. See also TrustedObjectVerify.
199 CHECK_IMPLIES(!IsTrustedObject(*this) && !IsFreeSpaceOrFiller(*this),
200 !HeapLayout::InTrustedSpace(*this));
201
202 switch (map(cage_base)->instance_type()) {
203#define STRING_TYPE_CASE(TYPE, size, name, CamelName) case TYPE:
204 STRING_TYPE_LIST(STRING_TYPE_CASE)
205#undef STRING_TYPE_CASE
206 if (IsConsString(*this, cage_base)) {
207 Cast<ConsString>(*this)->ConsStringVerify(isolate);
208 } else if (IsSlicedString(*this, cage_base)) {
209 Cast<SlicedString>(*this)->SlicedStringVerify(isolate);
210 } else if (IsThinString(*this, cage_base)) {
211 Cast<ThinString>(*this)->ThinStringVerify(isolate);
212 } else if (IsSeqString(*this, cage_base)) {
213 Cast<SeqString>(*this)->SeqStringVerify(isolate);
214 } else if (IsExternalString(*this, cage_base)) {
215 Cast<ExternalString>(*this)->ExternalStringVerify(isolate);
216 } else {
217 Cast<String>(*this)->StringVerify(isolate);
218 }
219 break;
220 // FixedArray types
221 case HASH_TABLE_TYPE:
222 case ORDERED_HASH_MAP_TYPE:
223 case ORDERED_HASH_SET_TYPE:
224 case ORDERED_NAME_DICTIONARY_TYPE:
225 case NAME_TO_INDEX_HASH_TABLE_TYPE:
226 case REGISTERED_SYMBOL_TABLE_TYPE:
227 case NAME_DICTIONARY_TYPE:
228 case GLOBAL_DICTIONARY_TYPE:
229 case NUMBER_DICTIONARY_TYPE:
230 case SIMPLE_NUMBER_DICTIONARY_TYPE:
231 case EPHEMERON_HASH_TABLE_TYPE:
232 Cast<FixedArray>(*this)->FixedArrayVerify(isolate);
233 break;
234 case AWAIT_CONTEXT_TYPE:
235 case BLOCK_CONTEXT_TYPE:
236 case CATCH_CONTEXT_TYPE:
237 case DEBUG_EVALUATE_CONTEXT_TYPE:
238 case EVAL_CONTEXT_TYPE:
239 case FUNCTION_CONTEXT_TYPE:
240 case MODULE_CONTEXT_TYPE:
241 case SCRIPT_CONTEXT_TYPE:
242 case WITH_CONTEXT_TYPE:
243 Cast<Context>(*this)->ContextVerify(isolate);
244 break;

247 break;
248 case FEEDBACK_METADATA_TYPE:

264 case JS_OBJECT_PROTOTYPE_TYPE:

266 case JS_REG_EXP_PROTOTYPE_TYPE:
267 case JS_SET_ITERATOR_PROTOTYPE_TYPE:
268 case JS_SET_PROTOTYPE_TYPE:
269 case JS_SPECIAL_API_OBJECT_TYPE:
270 case JS_STRING_ITERATOR_PROTOTYPE_TYPE:
271 case JS_TYPED_ARRAY_PROTOTYPE_TYPE:
272 Cast<JSObject>(*this)->JSObjectVerify(isolate);
273 break;
274#if V8_ENABLE_WEBASSEMBLY
275 case WASM_TRUSTED_INSTANCE_DATA_TYPE:
276 Cast<WasmTrustedInstanceData>(*this)->WasmTrustedInstanceDataVerify(

278 break;
279 case WASM_DISPATCH_TABLE_TYPE:

382void BytecodeArray::BytecodeArrayVerify(Isolate* isolate) {
383 ExposedTrustedObjectVerify(isolate);
384
385 {
386 CHECK(IsSmi(TaggedField<Object>::load(*this, kLengthOffset)));
387 CHECK_LE(0, length());
388 CHECK_LE(length(), kMaxLength);
389 }
390 {
391 auto o = constant_pool();
392 Object::VerifyPointer(isolate, o);
393 CHECK(IsTrustedFixedArray(o));
394 }
395 {
396 auto o = handler_table();
397 Object::VerifyPointer(isolate, o);
398 CHECK(IsTrustedByteArray(o));
399 }
400 {
401 auto o = wrapper();
402 Object::VerifyPointer(isolate, o);
403 CHECK(IsBytecodeWrapper(o));
404 // Our wrapper must point back to us.
405 CHECK_EQ(o->bytecode(isolate), *this);
406 }
407 {
408 // Use the raw accessor here as source positions may not be available.
409 auto o = raw_source_position_table(kAcquireLoad);
410 Object::VerifyPointer(isolate, o);
411 CHECK(o == Smi::zero() || IsTrustedByteArray(o));
412 }
413
414 // TODO(oth): Walk bytecodes and immediate values to validate sanity.
415 // - All bytecodes are known and well formed.
416 // - Jumps must go to new instructions starts.
417 // - No Illegal bytecodes.
418 // - No consecutive sequences of prefix Wide / ExtraWide.
419 // - String constants for loads should be internalized strings.
420}

423 if (!this->has_bytecode()) return;
424 auto bytecode = this->bytecode(isolate);
425 Object::VerifyPointer(isolate, bytecode);
426 CHECK_EQ(bytecode->wrapper(), *this);
427}

437void VerifyJSObjectElements(Isolate* isolate, Tagged<JSObject> object) {
438 // Only TypedArrays can have these specialized elements.
439 if (IsJSTypedArray(object)) {
440 // TODO(bmeurer,v8:4153): Fix CreateTypedArray to either not instantiate
441 // the object or propertly initialize it on errors during construction.
442 /* CHECK(object->HasTypedArrayOrRabGsabTypedArrayElements()); */
443 return;
444 }
445 CHECK(!IsByteArray(object->elements()));
446
447 if (object->HasDoubleElements()) {
448 if (object->elements()->length() > 0) {
449 CHECK(IsFixedDoubleArray(object->elements()));
450 }
451 return;
452 }
453
454 if (object->HasSloppyArgumentsElements()) {
455 CHECK(IsSloppyArgumentsElements(object->elements()));
456 return;
457 }
458
459 Tagged<FixedArray> elements = Cast<FixedArray>(object->elements());
460 if (object->HasSmiElements()) {
461 // We might have a partially initialized backing store, in which case we
462 // allow the hole + smi values.
463 for (int i = 0; i < elements->length(); i++) {
464 Tagged<Object> value = elements->get(i);
465 CHECK(IsSmi(value) || IsTheHole(value, isolate));
466 }
467 } else if (object->HasObjectElements()) {
468 for (int i = 0; i < elements->length(); i++) {
469 Tagged<Object> element = elements->get(i);
470 CHECK(!HasWeakHeapObjectTag(element));
471 }
472 }
473}

476void JSObject::JSObjectVerify(Isolate* isolate) {
477 TorqueGeneratedClassVerifiers::JSObjectVerify(*this, isolate);
478 VerifyHeapPointer(isolate, elements());
479
480 CHECK_IMPLIES(HasSloppyArgumentsElements(), IsJSArgumentsObject(*this));
481 if (HasFastProperties()) {
482 int actual_unused_property_fields = map()->GetInObjectProperties() +
483 property_array()->length() -
484 map()->NextFreePropertyIndex();
485 if (map()->UnusedPropertyFields() != actual_unused_property_fields) {
486 // There are two reasons why this can happen:
487 // - in the middle of StoreTransitionStub when the new extended backing
488 // store is already set into the object and the allocation of the
489 // HeapNumber triggers GC while the map isn't updated yet.
490 // - deletion of the last property can leave additional backing store
491 // capacity behind.
492 CHECK_GT(actual_unused_property_fields, map()->UnusedPropertyFields());
493 int delta = actual_unused_property_fields - map()->UnusedPropertyFields();
494 CHECK_EQ(0, delta % JSObject::kFieldsAdded);
495 }
496 Tagged<DescriptorArray> descriptors = map()->instance_descriptors(isolate);
497 bool is_transitionable_fast_elements_kind =
498 IsTransitionableFastElementsKind(map()->elements_kind());
499
500 for (InternalIndex i : map()->IterateOwnDescriptors()) {
501 PropertyDetails details = descriptors->GetDetails(i);
502 if (details.location() == PropertyLocation::kField) {
503 CHECK_EQ(PropertyKind::kData, details.kind());
504 Representation r = details.representation();
505 FieldIndex index = FieldIndex::ForDetails(map(), details);
506 if (COMPRESS_POINTERS_BOOL && index.is_inobject()) {
507 VerifyObjectField(isolate, index.offset());
508 }
509 Tagged<Object> value = RawFastPropertyAt(index);
510 CHECK_IMPLIES(r.IsDouble(), IsHeapNumber(value));
511 if (IsUninitialized(value, isolate)) continue;
512 CHECK_IMPLIES(r.IsSmi(), IsSmi(value));
513 CHECK_IMPLIES(r.IsHeapObject(), IsHeapObject(value));
514 Tagged<FieldType> field_type = descriptors->GetFieldType(i);
515 bool type_is_none = IsNone(field_type);
516 bool type_is_any = IsAny(field_type);
517 if (r.IsNone()) {
518 CHECK(type_is_none);
519 } else if (r.IsHeapObject()) {
520 CHECK(!type_is_none);
521 if (!type_is_any) {
522 CHECK_IMPLIES(FieldType::NowStable(field_type),
523 map()->is_deprecated() ||
524 FieldType::NowContains(field_type, value));
525 }
526 } else {
527 CHECK(type_is_any);
528 }
529 CHECK_IMPLIES(is_transitionable_fast_elements_kind,
530 Map::IsMostGeneralFieldType(r, field_type));
531 }
532 }
533
534 if (map()->EnumLength() != kInvalidEnumCacheSentinel) {
535 Tagged<EnumCache> enum_cache = descriptors->enum_cache();
536 Tagged<FixedArray> keys = enum_cache->keys();
537 Tagged<FixedArray> indices = enum_cache->indices();
538 CHECK_LE(map()->EnumLength(), keys->length());
539 CHECK_IMPLIES(indices != ReadOnlyRoots(isolate).empty_fixed_array(),
540 keys->length() == indices->length());
541 }
542 }
543
544 // If a GC was caused while constructing this object, the elements
545 // pointer may point to a one pointer filler map.
546 if (ElementsAreSafeToExamine(isolate)) {
547 CHECK_EQ((map()->has_fast_smi_or_object_elements() ||
548 map()->has_any_nonextensible_elements() ||
549 (elements() == GetReadOnlyRoots().empty_fixed_array()) ||
550 HasFastStringWrapperElements()),
551 (elements()->map() == GetReadOnlyRoots().fixed_array_map() ||
552 elements()->map() == GetReadOnlyRoots().fixed_cow_array_map()));
553 CHECK_EQ(map()->has_fast_object_elements(), HasObjectElements());
554 VerifyJSObjectElements(isolate, *this);
555 }
556}

558void Map::MapVerify(Isolate* isolate) {
559 TorqueGeneratedClassVerifiers::MapVerify(*this, isolate);
560 Heap* heap = isolate->heap();
561 CHECK(!HeapLayout::InYoungGeneration(Tagged<Map>(*this)));
562 CHECK(FIRST_TYPE <= instance_type() && instance_type() <= LAST_TYPE);
563 CHECK(instance_size() == kVariableSizeSentinel ||
564 (kTaggedSize <= instance_size() &&
565 static_cast<size_t>(instance_size()) < heap->Capacity()));
566 if (IsContextMap(*this)) {
567 // The map for the NativeContext is allocated before the NativeContext
568 // itself, so it may happen that during a GC the native_context() is still
569 // null.
570 CHECK(IsNull(native_context_or_null()) ||
571 IsNativeContext(native_context_or_null()));
572 // The context's meta map is tied to the same native context.
573 CHECK_EQ(native_context_or_null(), map()->native_context_or_null());
574 } else {
575 if (IsUndefined(GetBackPointer(), isolate)) {
576 // Root maps must not have descriptors in the descriptor array that do not
577 // belong to the map.
578 CHECK_EQ(NumberOfOwnDescriptors(),
579 instance_descriptors(isolate)->number_of_descriptors());
580 } else {
581 // If there is a parent map it must be non-stable.

1619 TrustedObjectVerify(isolate);
1620 Tagged<Code> code;
1621 if (!TryGetCode(&code, kAcquireLoad)) return;
1622 CHECK(
1623 IsAligned(code->instruction_size(),
1624 static_cast<unsigned>(InstructionStream::kMetadataAlignment)));
1625#if (!defined(_MSC_VER) || defined(__clang__)) && !defined(V8_OS_ZOS)

1632 IsAligned(instruction_start(), kCodeAlignment));

1639
1640 for (RelocIterator it(code); !it.done(); it.next()) {
1641 it.rinfo()->Verify(isolate);
1642 // Ensure that GC will not iterate twice over the same pointer.
1643 if (RelocInfo::IsGCRelocMode(it.rinfo()->rmode())) {
