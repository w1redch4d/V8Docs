28bool CreateICUPluralRules(Isolate* isolate, const icu::Locale& icu_locale,
29 JSPluralRules::Type type,
30 std::unique_ptr<icu::PluralRules>* pl) {
31 // Make formatter from options. Numbering system is added
32 // to the locale as Unicode extension (if it was specified at all).
33 UErrorCode status = U_ZERO_ERROR;
34
35 UPluralType icu_type = UPLURAL_TYPE_CARDINAL;
36 if (type == JSPluralRules::Type::ORDINAL) {
37 icu_type = UPLURAL_TYPE_ORDINAL;
38 } else {
39 DCHECK_EQ(JSPluralRules::Type::CARDINAL, type);
40 }
41
42 std::unique_ptr<icu::PluralRules> plural_rules(
43 icu::PluralRules::forLocale(icu_locale, icu_type, status));
44 if (U_FAILURE(status)) {
45 return false;
46 }
47 DCHECK_NOT_NULL(plural_rules.get());
48
49 *pl = std::move(plural_rules);
50 return true;
51}

66MaybeHandle<JSPluralRules> JSPluralRules::New(
67 Isolate* isolate, DirectHandle<Map> map, DirectHandle<Object> locales,
68 DirectHandle<Object> options_obj) {
69 // 1. Let requestedLocales be ? CanonicalizeLocaleList(locales).
70 Maybe<std::vector<std::string>> maybe_requested_locales =
71 Intl::CanonicalizeLocaleList(isolate, locales);
72 MAYBE_RETURN(maybe_requested_locales, Handle<JSPluralRules>());
73 std::vector<std::string> requested_locales =
74 maybe_requested_locales.FromJust();
75
76 // 2. Set options to ? CoerceOptionsToObject(options).
77 DirectHandle<JSReceiver> options;
78 const char* service = "Intl.PluralRules";
79 ASSIGN_RETURN_ON_EXCEPTION(
80 isolate, options, CoerceOptionsToObject(isolate, options_obj, service));
81
82 // 5. Let matcher be ? GetOption(options, "localeMatcher", "string",
83 // Â« "lookup", "best fit" Â», "best fit").
84 // 6. Set opt.[[localeMatcher]] to matcher.
85 Maybe<Intl::MatcherOption> maybe_locale_matcher =
86 Intl::GetLocaleMatcher(isolate, options, service);
87 MAYBE_RETURN(maybe_locale_matcher, MaybeHandle<JSPluralRules>());
88 Intl::MatcherOption matcher = maybe_locale_matcher.FromJust();
89
90 // 7. Let t be ? GetOption(options, "type", "string", Â« "cardinal",
91 // "ordinal" Â», "cardinal").
92 Maybe<Type> maybe_type = GetStringOption<Type>(
93 isolate, options, "type", service, {"cardinal", "ordinal"},
94 {Type::CARDINAL, Type::ORDINAL}, Type::CARDINAL);
95 MAYBE_RETURN(maybe_type, MaybeHandle<JSPluralRules>());
96 Type type = maybe_type.FromJust();
97
98 // Note: The spec says we should do ResolveLocale after performing
99 // SetNumberFormatDigitOptions but we need the locale to create all
100 // the ICU data structures.
101 //
102 // This isn't observable so we aren't violating the spec.
103
104 // 11. Let r be ResolveLocale(%PluralRules%.[[AvailableLocales]],
105 // requestedLocales, opt, %PluralRules%.[[RelevantExtensionKeys]],
106 // localeData).
107 Maybe<Intl::ResolvedLocale> maybe_resolve_locale =
108 Intl::ResolveLocale(isolate, JSPluralRules::GetAvailableLocales(),
109 requested_locales, matcher, {});
110 if (maybe_resolve_locale.IsNothing()) {
111 THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
112 }
113 Intl::ResolvedLocale r = maybe_resolve_locale.FromJust();
114 DirectHandle<String> locale_str =
115 isolate->factory()->NewStringFromAsciiChecked(r.locale.c_str());
116
117 icu::Locale icu_locale = r.icu_locale;
118 icu::number::UnlocalizedNumberFormatter settings =
119 icu::number::UnlocalizedNumberFormatter().roundingMode(UNUM_ROUND_HALFUP);
120
121 std::unique_ptr<icu::PluralRules> icu_plural_rules;
122 bool success =
123 CreateICUPluralRules(isolate, r.icu_locale, type, &icu_plural_rules);
124 if (!success || icu_plural_rules == nullptr) {
125 // Remove extensions and try again.
126 icu::Locale no_extension_locale(icu_locale.getBaseName());
127 success = CreateICUPluralRules(isolate, no_extension_locale, type,
128 &icu_plural_rules);
129 icu_locale = no_extension_locale;
130
131 if (!success || icu_plural_rules == nullptr) {
132 THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
133 }
134 }
135
136 // 9. Perform ? SetNumberFormatDigitOptions(pluralRules, options, 0, 3).
137 Maybe<Intl::NumberFormatDigitOptions> maybe_digit_options =
138 Intl::SetNumberFormatDigitOptions(isolate, options, 0, 3, false, service);
139 MAYBE_RETURN(maybe_digit_options, MaybeHandle<JSPluralRules>());
140 Intl::NumberFormatDigitOptions digit_options = maybe_digit_options.FromJust();
141 settings =
142 JSNumberFormat::SetDigitOptionsToFormatter(settings, digit_options);
143
144 icu::number::LocalizedNumberFormatter icu_number_formatter =
145 settings.locale(icu_locale);
146
147 DirectHandle<Managed<icu::PluralRules>> managed_plural_rules =
148 Managed<icu::PluralRules>::From(isolate, 0, std::move(icu_plural_rules));
149
150 DirectHandle<Managed<icu::number::LocalizedNumberFormatter>>
151 managed_number_formatter =
152 Managed<icu::number::LocalizedNumberFormatter>::From(
153 isolate, 0,
154 std::make_shared<icu::number::LocalizedNumberFormatter>(
155 icu_number_formatter));
156
157 // Now all properties are ready, so we can allocate the result object.
158 Handle<JSPluralRules> plural_rules = Cast<JSPluralRules>(
159 isolate->factory()->NewFastOrSlowJSObjectFromMap(map));
160 DisallowGarbageCollection no_gc;
161 plural_rules->set_flags(0);
162
163 // 8. Set pluralRules.[[Type]] to t.
164 plural_rules->set_type(type);
165
166 // 12. Set pluralRules.[[Locale]] to the value of r.[[locale]].
167 plural_rules->set_locale(*locale_str);
168
169 plural_rules->set_icu_plural_rules(*managed_plural_rules);
170 plural_rules->set_icu_number_formatter(*managed_number_formatter);
171
172 // 13. Return pluralRules.
173 return plural_rules;
174}

176MaybeHandle<String> JSPluralRules::ResolvePlural(
177 Isolate* isolate, DirectHandle<JSPluralRules> plural_rules, double number) {
178 icu::PluralRules* icu_plural_rules = plural_rules->icu_plural_rules()->raw();
179 DCHECK_NOT_NULL(icu_plural_rules);
180
181 icu::number::LocalizedNumberFormatter* fmt =
182 plural_rules->icu_number_formatter()->raw();
183 DCHECK_NOT_NULL(fmt);
184
185 UErrorCode status = U_ZERO_ERROR;
186 icu::number::FormattedNumber formatted_number =
187 fmt->formatDouble(number, status);
188 DCHECK(U_SUCCESS(status));
189
190 icu::UnicodeString result =
191 icu_plural_rules->select(formatted_number, status);
192 DCHECK(U_SUCCESS(status));
193
194 return Intl::ToString(isolate, result);
195}

201 DCHECK_NOT_NULL(icu_plural_rules);
202
203 Maybe<icu::number::LocalizedNumberRangeFormatter> maybe_range_formatter =
204 JSNumberFormat::GetRangeFormatter(
205 isolate, plural_rules->locale(),
206 *plural_rules->icu_number_formatter()->raw());
207 MAYBE_RETURN(maybe_range_formatter, MaybeHandle<String>());
208
209 icu::number::LocalizedNumberRangeFormatter nrfmt =
210 maybe_range_formatter.FromJust();
211
212 UErrorCode status = U_ZERO_ERROR;
213 icu::number::FormattedNumberRange formatted = nrfmt.formatFormattableRange(
214 icu::Formattable(x), icu::Formattable(y), status);
215
216 DCHECK(U_SUCCESS(status));
217 icu::UnicodeString result = icu_plural_rules->select(formatted, status);
218 DCHECK(U_SUCCESS(status));
219
220 return Intl::ToString(isolate, result);
221}

248Handle<JSObject> JSPluralRules::ResolvedOptions(
249 Isolate* isolate, DirectHandle<JSPluralRules> plural_rules) {
250 Handle<JSObject> options =
251 isolate->factory()->NewJSObject(isolate->object_function());
252
253 DirectHandle<String> locale_value(plural_rules->locale(), isolate);
254 CreateDataPropertyForOptions(isolate, options, locale_value, "locale");
255
256 CreateDataPropertyForOptions(isolate, options,
257 plural_rules->TypeAsString(isolate), "type");
258
259 UErrorCode status = U_ZERO_ERROR;
260 icu::number::LocalizedNumberFormatter* icu_number_formatter =
261 plural_rules->icu_number_formatter()->raw();
262 icu::UnicodeString skeleton = icu_number_formatter->toSkeleton(status);
263 DCHECK(U_SUCCESS(status));
264
265 CreateDataPropertyForOptions(
266 isolate, options,
267 JSNumberFormat::MinimumIntegerDigitsFromSkeleton(skeleton),
268 "minimumIntegerDigits");
269 int32_t min = 0, max = 0;
270
271 if (JSNumberFormat::SignificantDigitsFromSkeleton(skeleton, &min, &max)) {
272 CreateDataPropertyForOptions(isolate, options, min,
273 "minimumSignificantDigits");
274 CreateDataPropertyForOptions(isolate, options, max,
275 "maximumSignificantDigits");
276 } else {
277 JSNumberFormat::FractionDigitsFromSkeleton(skeleton, &min, &max);
278 CreateDataPropertyForOptions(isolate, options, min,
279 "minimumFractionDigits");
280 CreateDataPropertyForOptions(isolate, options, max,
281 "maximumFractionDigits");
282 }
283
284 // 6. Let pluralCategories be a List of Strings containing all possible
285 // results of PluralRuleSelect for the selected locale pr.[[Locale]], sorted
286 // according to the following order: "zero", "one", "two", "few", "many",
287 // "other".
288 icu::PluralRules* icu_plural_rules = plural_rules->icu_plural_rules()->raw();
289 DCHECK_NOT_NULL(icu_plural_rules);
290
291 std::unique_ptr<icu::StringEnumeration> categories(
292 icu_plural_rules->getKeywords(status));
293 DCHECK(U_SUCCESS(status));
294 int32_t count = categories->count(status);
295 DCHECK(U_SUCCESS(status));
296
297 Factory* factory = isolate->factory();
298 DirectHandle<FixedArray> plural_categories = factory->NewFixedArray(count);
299 const std::vector<const char*> kCategories = {"zero", "one", "two",
300 "few", "many", "other"};
301 int32_t index = 0;
302 std::for_each(kCategories.cbegin(), kCategories.cend(), [&](const char* val) {
303 categories->reset(status);
304 DCHECK(U_SUCCESS(status));
305 for (int32_t i = 0; i < count; i++) {
306 int32_t len;
307 const char* cat = categories->next(&len, status);
308 DCHECK(U_SUCCESS(status));
309 if (cat == nullptr) break;
310 if (std::strcmp(val, cat) == 0) {
311 DirectHandle<String> value_string =
312 factory->NewStringFromAsciiChecked(val);
313 plural_categories->set(index++, *value_string);
314 break;
315 }
316 }
317 });
318 DCHECK(count == index);
319
320 // 7. Perform ! CreateDataProperty(options, "pluralCategories",
321 // CreateArrayFromList(pluralCategories)).
322 DirectHandle<JSArray> plural_categories_value =
323 factory->NewJSArrayWithElements(plural_categories);
324 CreateDataPropertyForOptions(isolate, options, plural_categories_value,
325 "pluralCategories");
326
327 CHECK(JSReceiver::CreateDataProperty(
328 isolate, options, factory->roundingIncrement_string(),
329 JSNumberFormat::RoundingIncrement(isolate, skeleton),
330 Just(kDontThrow))
331 .FromJust());
332 CHECK(JSReceiver::CreateDataProperty(
333 isolate, options, factory->roundingMode_string(),
334 JSNumberFormat::RoundingModeString(isolate, skeleton),
335 Just(kDontThrow))
336 .FromJust());
337 CHECK(JSReceiver::CreateDataProperty(
338 isolate, options, factory->roundingPriority_string(),
339 JSNumberFormat::RoundingPriorityString(isolate, skeleton),
340 Just(kDontThrow))
341 .FromJust());
342 CHECK(JSReceiver::CreateDataProperty(
343 isolate, options, factory->trailingZeroDisplay_string(),
344 JSNumberFormat::TrailingZeroDisplayString(isolate, skeleton),
345 Just(kDontThrow))
346 .FromJust());
347
348 return options;
349}
