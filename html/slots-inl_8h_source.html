196void ExternalPointerSlot::init(IsolateForSandbox isolate,
197 Tagged<HeapObject> host, Address value,
198 ExternalPointerTag tag) {
199#ifdef V8_ENABLE_SANDBOX
200 ExternalPointerTable& table = isolate.GetExternalPointerTableFor(tag);
201 ExternalPointerHandle handle = table.AllocateAndInitializeEntry(
202 isolate.GetExternalPointerTableSpaceFor(tag, host.address()), value, tag);
203 // Use a Release_Store to ensure that the store of the pointer into the
204 // table is not reordered after the store of the handle. Otherwise, other
205 // threads may access an uninitialized table entry and crash.
206 Release_StoreHandle(handle);
207#else
208 store(isolate, value, tag);
209#endif // V8_ENABLE_SANDBOX
210}

405Tagged<Object> IndirectPointerSlot::ResolveHandle(
406 IndirectPointerHandle handle, IsolateForSandbox isolate) const {
407#ifdef V8_ENABLE_SANDBOX
408 // TODO(saelo) Maybe come up with a different entry encoding scheme that
409 // returns Smi::zero for kNullCodePointerHandle?
410 if (!handle) return Smi::zero();
411
412 // Resolve the handle. The tag implies the pointer table to use.
413 if (tag_ == kUnknownIndirectPointerTag) {
414 // In this case we have to rely on the handle marking to determine which
415 // pointer table to use.
416 if (handle & kCodePointerHandleMarker) {
417 return ResolveCodePointerHandle(handle);
418 } else {
419 return ResolveTrustedPointerHandle(handle, isolate);
420 }
421 } else if (tag_ == kCodeIndirectPointerTag) {
422 return ResolveCodePointerHandle(handle);
423 } else {
424 return ResolveTrustedPointerHandle(handle, isolate);
425 }
426#else
427 UNREACHABLE();
428#endif // V8_ENABLE_SANDBOX
429}
430
431#ifdef V8_ENABLE_SANDBOX
432Tagged<Object> IndirectPointerSlot::ResolveTrustedPointerHandle(
433 IndirectPointerHandle handle, IsolateForSandbox isolate) const {
434 DCHECK_NE(handle, kNullIndirectPointerHandle);
435 const TrustedPointerTable& table = isolate.GetTrustedPointerTableFor(tag_);
436 return Tagged<Object>(table.Get(handle, tag_));
437}

439Tagged<Object> IndirectPointerSlot::ResolveCodePointerHandle(
440 IndirectPointerHandle handle) const {
441 DCHECK_NE(handle, kNullIndirectPointerHandle);
442 Address addr =
443 IsolateGroup::current()->code_pointer_table()->GetCodeObject(handle);
444 return Tagged<Object>(addr);
445}

457// Copies tagged words from |src| to |dst|. The data spans must not overlap.
458// |src| and |dst| must be kTaggedSize-aligned.
459inline void CopyTagged(Address dst, const Address src, size_t num_tagged) {
460 static const size_t kBlockCopyLimit = 16;
461 CopyImpl<kBlockCopyLimit>(reinterpret_cast<Tagged_t*>(dst),
462 reinterpret_cast<const Tagged_t*>(src), num_tagged);
463}

465// Sets |counter| number of kTaggedSize-sized values starting at |start| slot.
466inline void MemsetTagged(Tagged_t* start, Tagged<MaybeObject> value,
467 size_t counter) {
468#ifdef V8_COMPRESS_POINTERS
469 // CompressAny since many callers pass values which are not valid objects.
470 Tagged_t raw_value = V8HeapCompressionScheme::CompressAny(value.ptr());
471 MemsetUint32(start, raw_value, counter);
472#else
473 Address raw_value = value.ptr();
474 MemsetPointer(start, raw_value, counter);
475#endif
476}
