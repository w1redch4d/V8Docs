14// A cached map that speeds up `operator[]` if used in LRU fashion.
15template <typename _Key, typename _Value, typename _Hash = v8::base::hash<_Key>>
16class CachedUnorderedMap final {
17 using MapT = std::unordered_map<_Key, _Value, _Hash>;
18
19 public:
20 using Key = typename MapT::key_type;
21 using Mapped = typename MapT::mapped_type;
22
23 Mapped& operator[](const Key& key) {
24 // nullptr value is used to indicate absence of a last key.
25 DCHECK_NOT_NULL(key);
26
27 if (key == last_key_) {
28 return *last_mapped_;
29 }
30
31 auto it = map_.find(key);
32 if (it == map_.end()) {
33 auto result = map_.emplace(key, Mapped());
34 DCHECK(result.second);
35 it = result.first;
36 }
37
38 last_key_ = key;
39 last_mapped_ = &it->second;
40
41 return it->second;
42 }
43
44 typename MapT::size_type erase(const Key& key) {
45 if (key == last_key_) {
46 last_key_ = nullptr;
47 last_mapped_ = nullptr;
48 }
49 return map_.erase(key);
50 }
51
52 // No iterator is cached in this class so an actual find() has to be executed
53 // everytime.
54 typename MapT::iterator find(const Key& key) { return map_.find(key); }
55
56 typename MapT::iterator begin() { return map_.begin(); }
57 typename MapT::iterator end() { return map_.end(); }
58 typename MapT::const_iterator begin() const { return map_.begin(); }
59 typename MapT::const_iterator end() const { return map_.begin(); }
60
61 void clear() {
62 last_key_ = nullptr;
63 last_mapped_ = nullptr;
64 map_.clear();
65 }
66
67 bool empty() const { return map_.empty(); }
68
69 MapT Take() {
70 last_key_ = nullptr;
71 last_mapped_ = nullptr;
72
73 MapT tmp(std::move(map_));
74 map_.clear();
75 return tmp;
76 }
77
78 private:
79 Key last_key_ = nullptr;
80 Mapped* last_mapped_ = nullptr;
81 MapT map_;
82};
