47void MaybeSetHandlerNow(Isolate* isolate) {
48 DCHECK(v8_flags.enable_etw_stack_walking ||
49 v8_flags.enable_etw_by_custom_filter_only);
50 ETWTRACEDBG << "MaybeSetHandlerNow called" << std::endl;
51 // Iterating read-only heap before sealed might not be safe.
52 if (has_active_etw_tracing_session_or_custom_filter &&
53 !EtwIsolateOperations::Instance()->HeapReadOnlySpaceWritable(isolate)) {
54 if (etw_filter_payload.Pointer()->empty()) {
55 DCHECK(v8_flags.enable_etw_stack_walking);
56 IsolateLoadScriptData::EnableLog(
57 isolate, 0, std::weak_ptr<EtwIsolateCaptureStateMonitor>(),
58 kJitCodeEventDefault);
59 } else {
60 IsolateLoadScriptData::EnableLogWithFilterData(
61 isolate, 0, *etw_filter_payload.Pointer(),
62 std::weak_ptr<EtwIsolateCaptureStateMonitor>(), kJitCodeEventDefault);
63 }
64 }
65}

73std::wstring GetScriptMethodNameFromEvent(const JitCodeEvent* event) {
74 int name_len = static_cast<int>(event->name.len);
75 // Note: event->name.str is not null terminated.
76 std::wstring method_name(name_len + 1, '\0');
77 MultiByteToWideChar(
78 CP_UTF8, 0, event->name.str, name_len,
79 // Const cast needed as building with C++14 (not const in >= C++17)
80 const_cast<LPWSTR>(method_name.data()),
81 static_cast<int>(method_name.size()));
82 return method_name;
83}

110// This callback is invoked by Windows every time the ETW tracing status is
111// changed for this application. As such, V8 needs to track its value for
112// knowing if the event requires us to emit JIT runtime events.
113void WINAPI V8_EXPORT_PRIVATE ETWEnableCallback(
114 LPCGUID /* source_id */, ULONG is_enabled, UCHAR level,
115 ULONGLONG match_any_keyword, ULONGLONG match_all_keyword,
116 PEVENT_FILTER_DESCRIPTOR filter_data, PVOID /* callback_context */) {
117 ETWTRACEDBG << "ETWEnableCallback called with is_enabled==" << is_enabled
118 << std::endl;
119
120 bool is_etw_enabled_now =
121 is_enabled && level >= kTraceLevel &&
122 (match_any_keyword & kJScriptRuntimeKeyword) &&
123 ((match_all_keyword & kJScriptRuntimeKeyword) == match_all_keyword);
124
125 uint32_t options = kJitCodeEventDefault;
126 if (is_enabled == kEtwControlCaptureState) {
127 options |= kJitCodeEventEnumExisting;
128 }
129
130 FilterDataType* etw_filter = etw_filter_payload.Pointer();
131
132 if (!is_etw_enabled_now) {
133 // Disable the current tracing session.
134 ETWTRACEDBG << "Disabling" << std::endl;
135 etw_filter->clear();
136 UpdateETWEnabled(false, options);
137 return;
138 }
139
140 DCHECK(is_etw_enabled_now);
141
142 // Ignore the callback if ETW tracing is not fully enabled and we are not
143 // passing a custom filter.
144 if (!v8_flags.enable_etw_stack_walking &&
145 (!filter_data || filter_data->Type != EVENT_FILTER_TYPE_SCHEMATIZED)) {
146 return;
147 }
148
149 if (v8_flags.enable_etw_stack_walking &&
150 (!filter_data || filter_data->Type != EVENT_FILTER_TYPE_SCHEMATIZED)) {
151 etw_filter->clear();
152 ETWTRACEDBG << "Enabling without filter" << std::endl;
153 UpdateETWEnabled(is_etw_enabled_now, options);
154 return;
155 }
156
157 // Ignore the callback if the --enable-etw-by-custom-filter-only flag is not
158 // enabled.
159 if (!v8_flags.enable_etw_by_custom_filter_only) return;
160
161 // Validate custom filter data configured in a WPR profile and passed to the
162 // callback.
163
164 if (filter_data->Size <= sizeof(EVENT_FILTER_DESCRIPTOR)) {
165 return; // Invalid data
166 }
167
168 EVENT_FILTER_HEADER* filter_event_header =
169 reinterpret_cast<EVENT_FILTER_HEADER*>(filter_data->Ptr);
170 if (filter_event_header->Size < sizeof(EVENT_FILTER_HEADER)) {
171 return; // Invalid data
172 }
173
174 const uint8_t* payload_start =
175 reinterpret_cast<uint8_t*>(filter_event_header) +
176 sizeof(EVENT_FILTER_HEADER);
177 const size_t payload_size =
178 filter_event_header->Size - sizeof(EVENT_FILTER_HEADER);
179 etw_filter->assign(payload_start, payload_start + payload_size);
180 has_active_etw_tracing_session_or_custom_filter = true;
181
182 ETWTRACEDBG << "Enabling with filter data" << std::endl;
183 IsolateLoadScriptData::EnableLogWithFilterDataOnAllIsolates(
184 reinterpret_cast<const uint8_t*>(etw_filter->data()), etw_filter->size(),
185 options);
186}

208void EventHandler(const JitCodeEvent* event) {
209 v8::Isolate* script_context = event->isolate;
210 Isolate* isolate = reinterpret_cast<Isolate*>(script_context);
211 if (!isolate->IsETWTracingEnabled()) return;
212
213 if (event->code_type != v8::JitCodeEvent::CodeType::JIT_CODE) return;
214 if (event->type != v8::JitCodeEvent::EventType::CODE_ADDED) return;
215
216 std::wstring method_name = GetScriptMethodName(event);
217
218 // No heap allocations after this point.
219 DisallowGarbageCollection no_gc;
220
221 int script_id = 0;
222 uint32_t script_line = -1;
223 uint32_t script_column = -1;
224
225 Tagged<SharedFunctionInfo> sfi = GetSharedFunctionInfo(event);
226 if (!sfi.is_null() && IsScript(sfi->script())) {
227 Tagged<Script> script = Cast<Script>(sfi->script());
228
229 // if the first time seeing this source file, log the SourceLoad event
230 script_id = script->id();
231 if (IsolateLoadScriptData::MaybeAddLoadedScript(isolate, script_id)) {
232 std::wstring wstr_name(0, L'\0');
233 Tagged<Object> script_name = script->GetNameOrSourceURL();
234 if (IsString(script_name)) {
235 Tagged<String> v8str_name = Cast<String>(script_name);
236 wstr_name.resize(v8str_name->length());
237 // On Windows wchar_t == uint16_t. const_Cast needed for C++14.
238 uint16_t* wstr_data = const_cast<uint16_t*>(
239 reinterpret_cast<const uint16_t*>(wstr_name.data()));
240 String::WriteToFlat(v8str_name, wstr_data, 0, v8str_name->length());
241 }
242
243 constexpr static auto source_load_event_meta =
244 EventMetadata(kSourceLoadEventID, kJScriptRuntimeKeyword);
245 constexpr static auto source_load_event_fields = EventFields(
246 "SourceLoad", Field("SourceID", TlgInUINT64),
247 Field("ScriptContextID", TlgInPOINTER),
248 Field("SourceFlags", TlgInUINT32), Field("Url", TlgInUNICODESTRING));
249 LogEventData(g_v8Provider, &source_load_event_meta,
250 &source_load_event_fields, (uint64_t)script_id,
251 script_context,
252 (uint32_t)0, // SourceFlags
253 wstr_name);
254 }
255
256 Script::PositionInfo info;
257 script->GetPositionInfo(sfi->StartPosition(), &info);
258 script_line = info.line + 1;
259 script_column = info.column + 1;
260 }
261
262 auto code =
263 EtwIsolateOperations::Instance()->HeapGcSafeTryFindCodeForInnerPointer(
264 isolate, Address(event->code_start));
265 if (code && code.value()->is_builtin()) {
266 bool skip_emitting_builtin = true;
267 // Skip logging functions with code kind BUILTIN as they are already present
268 // in the PDB.
269
270 // We should still emit builtin addresses if they are an interpreter
271 // trampoline.
272 if (code.value()->has_instruction_stream()) {
273 skip_emitting_builtin = false;
274
275 // The only builtin that might have instruction stream is the
276 // InterpreterEntryTrampoline builtin and only when the
277 // v8_flags.interpreted_frames_native_stack flag is enabled.
278 DCHECK_IMPLIES(
279 code.value()->is_builtin(),
280 code.value()->builtin_id() == Builtin::kInterpreterEntryTrampoline &&
281 isolate->interpreted_frames_native_stack());
282 } else {
283 DCHECK(code.value()->is_builtin());
284 }
285
286 // If the builtin has been relocated, we still need to emit the address
287 if (skip_emitting_builtin && V8_SHORT_BUILTIN_CALLS_BOOL &&
288 v8_flags.short_builtin_calls) {
289 CodeRange* code_range = isolate->isolate_group()->GetCodeRange();
290 if (code_range && code_range->embedded_blob_code_copy() != nullptr) {
291 skip_emitting_builtin = false;
292 }
293 }
294
295 if (skip_emitting_builtin) {
296 return;
297 }
298 }
299
300 constexpr static auto method_load_event_meta =
301 EventMetadata(kMethodLoadEventID, kJScriptRuntimeKeyword);
302 constexpr static auto method_load_event_fields = EventFields(
303 "MethodLoad", Field("ScriptContextID", TlgInPOINTER),
304 Field("MethodStartAddress", TlgInPOINTER),
305 Field("MethodSize", TlgInUINT64), Field("MethodID", TlgInUINT32),
306 Field("MethodFlags", TlgInUINT16),
307 Field("MethodAddressRangeID", TlgInUINT16),
308 Field("SourceID", TlgInUINT64), Field("Line", TlgInUINT32),
309 Field("Column", TlgInUINT32), Field("MethodName", TlgInUNICODESTRING));
310
311 LogEventData(g_v8Provider, &method_load_event_meta, &method_load_event_fields,
312 script_context, event->code_start, (uint64_t)event->code_len,
313 (uint32_t)0, // MethodId
314 (uint16_t)0, // MethodFlags
315 (uint16_t)0, // MethodAddressRangeId
316 (uint64_t)script_id, script_line, script_column, method_name);
317}
