38// LocalHeap is used by the GC to track all threads with heap access in order to
39// stop them before performing a collection. LocalHeaps can be either Parked or
40// Running and are in Parked mode when initialized.
41// Running: Thread is allowed to access the heap but needs to give the GC the
42// chance to run regularly by manually invoking Safepoint(). The
43// thread can be parked using ParkedScope.
44// Parked: Heap access is not allowed, so the GC will not stop this thread
45// for a collection. Useful when threads do not need heap access for
46// some time or for blocking operations like locking a mutex.

236 class ThreadState final {
237 public:
238 static constexpr ThreadState Parked() {
239 return ThreadState(ParkedBit::kMask);
240 }
241 static constexpr ThreadState Running() { return ThreadState(0); }
242
243 constexpr bool IsRunning() const { return !ParkedBit::decode(raw_state_); }
244
245 constexpr ThreadState SetRunning() const V8_WARN_UNUSED_RESULT {
246 return ThreadState(raw_state_ & ~ParkedBit::kMask);
247 }
248
249 constexpr bool IsParked() const { return ParkedBit::decode(raw_state_); }
250
251 constexpr ThreadState SetParked() const V8_WARN_UNUSED_RESULT {
252 return ThreadState(ParkedBit::kMask | raw_state_);
253 }
254
255 constexpr bool IsSafepointRequested() const {
256 return SafepointRequestedBit::decode(raw_state_);
257 }
258
259 constexpr bool IsCollectionRequested() const {
260 return CollectionRequestedBit::decode(raw_state_);
261 }
262
263 constexpr bool IsRunningWithSlowPathFlag() const {
264 return IsRunning() && (raw_state_ & (SafepointRequestedBit::kMask |
265 CollectionRequestedBit::kMask));
266 }

328 void Park() {
329 DCHECK(AllowSafepoints::IsAllowed());
330 DCHECK(IsSafeForConservativeStackScanning());

333 ParkSlowPath();
