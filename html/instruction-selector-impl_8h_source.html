20template <typename Adapter>
21struct CaseInfoT {
22 int32_t value; // The case value.
23 int32_t order; // The order for lowering to comparisons (less means earlier).
24 typename Adapter::block_t
25 branch; // The basic blocks corresponding to the case value.
26};

28template <typename Adapter>
29inline bool operator<(const CaseInfoT<Adapter>& l,
30 const CaseInfoT<Adapter>& r) {
31 return l.order < r.order;
32}

34// Helper struct containing data about a table or lookup switch.
35template <typename Adapter>
36class SwitchInfoT {
37 public:
38 using CaseInfo = CaseInfoT<Adapter>;
39 using block_t = typename Adapter::block_t;
40 SwitchInfoT(ZoneVector<CaseInfo> const& cases, int32_t min_value,
41 int32_t max_value, block_t default_branch)
42 : cases_(cases),
43 min_value_(min_value),
44 max_value_(max_value),
45 default_branch_(default_branch) {
46 if (cases.size() != 0) {
47 DCHECK_LE(min_value, max_value);
48 // Note that {value_range} can be 0 if {min_value} is -2^31 and
49 // {max_value} is 2^31-1, so don't assume that it's non-zero below.
50 value_range_ = 1u + base::bit_cast<uint32_t>(max_value) -
51 base::bit_cast<uint32_t>(min_value);
52 } else {
53 value_range_ = 0;
54 }
55 }
56
57 std::vector<CaseInfo> CasesSortedByValue() const {
58 std::vector<CaseInfo> result(cases_.begin(), cases_.end());
59 std::stable_sort(result.begin(), result.end(),
60 [](CaseInfo a, CaseInfo b) { return a.value < b.value; });
61 return result;
62 }
63 const ZoneVector<CaseInfo>& CasesUnsorted() const { return cases_; }
64 int32_t min_value() const { return min_value_; }
65 int32_t max_value() const { return max_value_; }
66 size_t value_range() const { return value_range_; }
67 size_t case_count() const { return cases_.size(); }
68 block_t default_branch() const { return default_branch_; }
69
70 private:
71 const ZoneVector<CaseInfo>& cases_;
72 int32_t min_value_; // minimum value of {cases_}
73 int32_t max_value_; // maximum value of {cases_}
74 size_t value_range_; // |max_value - min_value| + 1
75 block_t default_branch_;
76};

370 Constant ToConstant(node_t node) {
371 if constexpr (Adapter::IsTurboshaft) {
372 using Kind = turboshaft::ConstantOp::Kind;
373 if (const turboshaft::ConstantOp* constant =
374 this->turboshaft_graph()
375 ->Get(node)
376 .template TryCast<turboshaft::ConstantOp>()) {
377 switch (constant->kind) {
378 case Kind::kWord32:
379 return Constant(static_cast<int32_t>(constant->word32()));
380 case Kind::kWord64:
381 return Constant(static_cast<int64_t>(constant->word64()));
382 case Kind::kSmi:
383 if constexpr (Is64()) {
384 return Constant(static_cast<int64_t>(constant->smi().ptr()));
385 } else {
386 return Constant(static_cast<int32_t>(constant->smi().ptr()));
387 }
388 case Kind::kHeapObject:
389 case Kind::kCompressedHeapObject:
390 case Kind::kTrustedHeapObject:
391 return Constant(constant->handle(),
392 constant->kind == Kind::kCompressedHeapObject);
393 case Kind::kExternal:
394 return Constant(constant->external_reference());
395 case Kind::kNumber:
396 return Constant(constant->number());
397 case Kind::kFloat32:
398 return Constant(constant->float32());
399 case Kind::kFloat64:
400 return Constant(constant->float64());
401 case Kind::kTaggedIndex: {
402 // Unencoded index value.
403 intptr_t value = static_cast<intptr_t>(constant->tagged_index());
404 DCHECK(TaggedIndex::IsValid(value));
405 // Generate it as 32/64-bit constant in a tagged form.
406 Address tagged_index = TaggedIndex::FromIntptr(value).ptr();
407 if (kSystemPointerSize == kInt32Size) {
408 return Constant(static_cast<int32_t>(tagged_index));
409 } else {
410 return Constant(static_cast<int64_t>(tagged_index));
411 }
412 }
413 case Kind::kRelocatableWasmCall:
414 case Kind::kRelocatableWasmStubCall: {
415 uint64_t value = constant->integral();
416 auto mode = constant->kind == Kind::kRelocatableWasmCall
417 ? RelocInfo::WASM_CALL
418 : RelocInfo::WASM_STUB_CALL;
419 using constant_type = std::conditional_t<Is64(), int64_t, int32_t>;
420 return Constant(RelocatablePtrConstantInfo(
421 base::checked_cast<constant_type>(value), mode));
422 }
423 case Kind::kRelocatableWasmCanonicalSignatureId:
424 return Constant(RelocatablePtrConstantInfo(
425 base::checked_cast<int32_t>(constant->integral()),
426 RelocInfo::WASM_CANONICAL_SIG_ID));
427 case Kind::kRelocatableWasmIndirectCallTarget:
428 uint64_t value = constant->integral();
429 return Constant(RelocatablePtrConstantInfo(
430 base::checked_cast<int32_t>(value),
431 RelocInfo::WASM_CODE_POINTER_TABLE_ENTRY));
432 }
433 }
434 UNREACHABLE();
435 } else {
436 switch (node->opcode()) {
437 case IrOpcode::kInt32Constant:
438 return Constant(OpParameter<int32_t>(node->op()));
439 case IrOpcode::kInt64Constant:
440 return Constant(OpParameter<int64_t>(node->op()));
441 case IrOpcode::kTaggedIndexConstant: {
442 // Unencoded index value.
443 intptr_t value =
444 static_cast<intptr_t>(OpParameter<int32_t>(node->op()));
445 DCHECK(TaggedIndex::IsValid(value));
446 // Generate it as 32/64-bit constant in a tagged form.
447 Address tagged_index = TaggedIndex::FromIntptr(value).ptr();
448 if (kSystemPointerSize == kInt32Size) {
449 return Constant(static_cast<int32_t>(tagged_index));
450 } else {
451 return Constant(static_cast<int64_t>(tagged_index));
452 }
453 }
454 case IrOpcode::kFloat32Constant:
455 return Constant(OpParameter<float>(node->op()));
456 case IrOpcode::kRelocatableInt32Constant:
457 case IrOpcode::kRelocatableInt64Constant:
458 return Constant(OpParameter<RelocatablePtrConstantInfo>(node->op()));
459 case IrOpcode::kFloat64Constant:
460 case IrOpcode::kNumberConstant:
461 return Constant(OpParameter<double>(node->op()));
462 case IrOpcode::kExternalConstant:
463 return Constant(OpParameter<ExternalReference>(node->op()));
464 case IrOpcode::kComment: {
465 // We cannot use {intptr_t} here, since the Constant constructor would
466 // be ambiguous on some architectures.
467 using ptrsize_int_t =
468 std::conditional<kSystemPointerSize == 8, int64_t, int32_t>::type;
469 return Constant(reinterpret_cast<ptrsize_int_t>(
470 OpParameter<const char*>(node->op())));
471 }
472 case IrOpcode::kHeapConstant:
473 return Constant(HeapConstantOf(node->op()));
474 case IrOpcode::kCompressedHeapConstant:
475 return Constant(HeapConstantOf(node->op()), true);
476 case IrOpcode::kDeadValue: {
477 switch (DeadValueRepresentationOf(node->op())) {
478 case MachineRepresentation::kBit:
479 case MachineRepresentation::kWord32:
480 case MachineRepresentation::kTagged:
481 case MachineRepresentation::kTaggedSigned:
482 case MachineRepresentation::kTaggedPointer:
483 case MachineRepresentation::kCompressed:
484 case MachineRepresentation::kCompressedPointer:
485 return Constant(static_cast<int32_t>(0));
486 case MachineRepresentation::kWord64:
487 return Constant(static_cast<int64_t>(0));
488 case MachineRepresentation::kFloat64:
489 return Constant(static_cast<double>(0));
490 case MachineRepresentation::kFloat32:
491 return Constant(static_cast<float>(0));
492 default:
493 UNREACHABLE();
494 }
495 break;
496 }
497 default:
498 break;
499 }
500 }
501 UNREACHABLE();
502 }

537 UnallocatedOperand ToUnallocatedOperand(LinkageLocation location,
538 int virtual_register) {
539 if (location.IsAnyRegister() || location.IsNullRegister()) {
540 // any machine register.
541 return UnallocatedOperand(UnallocatedOperand::MUST_HAVE_REGISTER,
542 virtual_register);
543 }
544 if (location.IsCallerFrameSlot()) {
545 // a location on the caller frame.
546 return UnallocatedOperand(UnallocatedOperand::FIXED_SLOT,
547 location.AsCallerFrameSlot(), virtual_register);
548 }
549 if (location.IsCalleeFrameSlot()) {
550 // a spill location on this (callee) frame.
551 return UnallocatedOperand(UnallocatedOperand::FIXED_SLOT,
552 location.AsCalleeFrameSlot(), virtual_register);
553 }
554 // a fixed register.
555 if (IsFloatingPoint(location.GetType().representation())) {
556 return UnallocatedOperand(UnallocatedOperand::FIXED_FP_REGISTER,
557 location.AsRegister(), virtual_register);
558 }
559 return UnallocatedOperand(UnallocatedOperand::FIXED_REGISTER,
560 location.AsRegister(), virtual_register);
561 }
