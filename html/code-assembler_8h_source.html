435 // A scheduler to ensure deterministic builtin compilation regardless of
436 // v8_flags.concurrent_builtin_generation.
437 //
438 // Builtin jobs are compiled in three (3) stages: PrepareJob, ExecuteJob, and
439 // FinalizeJob.
440 //
441 // PrepareJob and FinalizeJob may allocate on the heap and must run on the
442 // main thread. ExecuteJob only allocates in the job's Zone and may run on a
443 // helper thread. To ensure deterministic builds, there must be a total order
444 // of all heap allocations across all spaces. In other words, there must be a
445 // single total order of PrepareJob and FinalizeJob calls.
446 //
447 // This order is enforced by batching according to zone size. For each batch,
448 //
449 // 1. In ascending order of job->FinalizeOrder(), call PrepareJob for each
450 // job.
451 // 2. Call ExecuteJob for each job in the batch in any order, possibly in
452 // parallel.
453 // 3. In ascending order of job->FinalizeOrder(), call FinalizeJob for each
454 // job.
455 //
456 // Example use:
457 //
458 // BuiltinCompilationScheduler scheduler;
459 // scheduler.CompileCode(job1);
460 // scheduler.CompileCode(job2);
461 // scheduler.AwaitAndFinalizeCurrentBatch();
462 class BuiltinCompilationScheduler {
463 public:
464 ~BuiltinCompilationScheduler();
465
466 int builtins_installed_count() const { return builtins_installed_count_; }
467
468 void CompileCode(Isolate* isolate,
469 std::unique_ptr<TurbofanCompilationJob> job);
470
471 void AwaitAndFinalizeCurrentBatch(Isolate* isolate);
472
473 private:
474 void QueueJob(Isolate* isolate,
475 std::unique_ptr<TurbofanCompilationJob> job);
476
477 void FinalizeJobOnMainThread(Isolate* isolate, TurbofanCompilationJob* job);
478
479 int builtins_installed_count_ = 0;
480
481 // The sum of the size of Zones of all queued jobs.
482 size_t current_batch_zone_size_ = 0;
483
484 // Only used when !v8_flags.concurrent_builtin_generation. Used to keep the
485 // allocation order identical between generating builtins concurrently and
486 // non-concurrently for reproducible builds.
487 std::deque<std::unique_ptr<TurbofanCompilationJob>>
488 main_thread_output_queue_;
489 };
