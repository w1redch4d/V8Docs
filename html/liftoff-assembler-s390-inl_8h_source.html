21// half
22// slot Frame
23// -----+--------------------+---------------------------
24// n+3 | parameter n |
25// ... | ... |
26// 4 | parameter 1 | or parameter 2
27// 3 | parameter 0 | or parameter 1
28// 2 | (result address) | or parameter 0
29// -----+--------------------+---------------------------
30// 1 | return addr (lr) |
31// 0 | previous frame (fp)|
32// -----+--------------------+ <-- frame ptr (fp)
33// -1 | StackFrame::WASM |
34// -2 | instance |
35// -3 | feedback vector |
36// -4 | tiering budget |
37// -----+--------------------+---------------------------
38// -5 | slot 0 (high) | ^
39// -6 | slot 0 (low) | |
40// -7 | slot 1 (high) | Frame slots
41// -8 | slot 1 (low) | |
42// | | v
43// -----+--------------------+ <-- stack ptr (sp)
44//

53inline void StoreToMemory(LiftoffAssembler* assm, MemOperand dst,
54 const LiftoffAssembler::VarState& src,
55 Register scratch) {
56 if (src.is_reg()) {
57 switch (src.kind()) {
58 case kI16:
59 assm->StoreU16(src.reg().gp(), dst);
60 break;
61 case kI32:
62 assm->StoreU32(src.reg().gp(), dst);
63 break;
64 case kI64:
65 assm->StoreU64(src.reg().gp(), dst);
66 break;
67 case kF32:
68 assm->StoreF32(src.reg().fp(), dst);
69 break;
70 case kF64:
71 assm->StoreF64(src.reg().fp(), dst);
72 break;
73 case kS128:
74 assm->StoreV128(src.reg().fp(), dst, scratch);
75 break;
76 default:
77 UNREACHABLE();
78 }
79 } else if (src.is_const()) {
80 if (src.kind() == kI32) {
81 assm->mov(scratch, Operand(src.i32_const()));
82 assm->StoreU32(scratch, dst);
83 } else {
84 assm->mov(scratch, Operand(static_cast<int64_t>(src.i32_const())));
85 assm->StoreU64(scratch, dst);
86 }
87 } else if (value_kind_size(src.kind()) == 4) {
88 assm->LoadU32(scratch, liftoff::GetStackSlot(src.offset()), scratch);
89 assm->StoreU32(scratch, dst);
90 } else {
91 DCHECK_EQ(8, value_kind_size(src.kind()));
92 assm->LoadU64(scratch, liftoff::GetStackSlot(src.offset()), scratch);
93 assm->StoreU64(scratch, dst);
94 }
95}

105void LiftoffAssembler::CallFrameSetupStub(int declared_function_index) {
106// The standard library used by gcc tryjobs does not consider `std::find` to be
107// `constexpr`, so wrap it in a `#ifdef __clang__` block.
108#ifdef __clang__
109 static_assert(std::find(std::begin(wasm::kGpParamRegisters),
110 std::end(wasm::kGpParamRegisters),
111 kLiftoffFrameSetupFunctionReg) ==
112 std::end(wasm::kGpParamRegisters));
113#endif
114
115 // On ARM, we must push at least {lr} before calling the stub, otherwise
116 // it would get clobbered with no possibility to recover it.
117 Register scratch = ip;
118 mov(scratch, Operand(StackFrame::TypeToMarker(StackFrame::WASM)));
119 PushCommonFrame(scratch);
120 LoadConstant(LiftoffRegister(kLiftoffFrameSetupFunctionReg),
121 WasmValue(declared_function_index));
122 CallBuiltin(Builtin::kWasmLiftoffFrameSetup);
123}

125void LiftoffAssembler::PrepareTailCall(int num_callee_stack_params,
126 int stack_param_delta) {
127 Register scratch = r1;
128 // Push the return address and frame pointer to complete the stack frame.
129 lay(sp, MemOperand(sp, -2 * kSystemPointerSize));
130 LoadU64(scratch, MemOperand(fp, kSystemPointerSize));
131 StoreU64(scratch, MemOperand(sp, kSystemPointerSize));
132 LoadU64(scratch, MemOperand(fp));
133 StoreU64(scratch, MemOperand(sp));
134
135 // Shift the whole frame upwards.
136 int slot_count = num_callee_stack_params + 2;
137 for (int i = slot_count - 1; i >= 0; --i) {
138 LoadU64(scratch, MemOperand(sp, i * kSystemPointerSize));
139 StoreU64(scratch,
140 MemOperand(fp, (i - stack_param_delta) * kSystemPointerSize));
141 }
142
143 // Set the new stack and frame pointer.
144 lay(sp, MemOperand(fp, -stack_param_delta * kSystemPointerSize));
145 Pop(r14, fp);
146}

150void LiftoffAssembler::PatchPrepareStackFrame(
151 int offset, SafepointTableBuilder* safepoint_table_builder,
152 bool feedback_vector_slot, size_t stack_param_slots) {
153 int frame_size = GetTotalFrameSize() - 2 * kSystemPointerSize;
154 // The frame setup builtin also pushes the feedback vector.
155 if (feedback_vector_slot) {
156 frame_size -= kSystemPointerSize;
157 }
158
159 constexpr int LayInstrSize = 6;
160
161 Assembler patching_assembler(
162 AssemblerOptions{},
163 ExternalAssemblerBuffer(buffer_start_ + offset, LayInstrSize + kGap));
164 if (V8_LIKELY(frame_size < 4 * KB)) {
165 patching_assembler.lay(sp, MemOperand(sp, -frame_size));
166 return;
167 }
168
169 // The frame size is bigger than 4KB, so we might overflow the available stack
170 // space if we first allocate the frame and then do the stack check (we will
171 // need some remaining stack space for throwing the exception). That's why we
172 // check the available stack space before we allocate the frame. To do this we
173 // replace the {__ sub(sp, sp, framesize)} with a jump to OOL code that does
174 // this "extended stack check".
175 //
176 // The OOL code can simply be generated here with the normal assembler,
177 // because all other code generation, including OOL code, has already finished
178 // when {PatchPrepareStackFrame} is called. The function prologue then jumps
179 // to the current {pc_offset()} to execute the OOL code for allocating the
180 // large frame.
181
182 // Emit the unconditional branch in the function prologue (from {offset} to
183 // {pc_offset()}).
184
185 int jump_offset = pc_offset() - offset;
186 patching_assembler.branchOnCond(al, jump_offset, true, true);
187
188 // If the frame is bigger than the stack, we throw the stack overflow
189 // exception unconditionally. Thereby we can avoid the integer overflow
190 // check in the condition code.
191 RecordComment("OOL: stack check for large frame");
192 Label continuation;
193 if (frame_size < v8_flags.stack_size * 1024) {
194 Register stack_limit = ip;
195 LoadStackLimit(stack_limit, StackLimitKind::kRealStackLimit);
196 AddU64(stack_limit, Operand(frame_size));
197 CmpU64(sp, stack_limit);
198 bge(&continuation);
199 }
200
201 Call(static_cast<Address>(Builtin::kWasmStackOverflow),
202 RelocInfo::WASM_STUB_CALL);
203 // The call will not return; just define an empty safepoint.
204 safepoint_table_builder->DefineSafepoint(this);
205 if (v8_flags.debug_code) stop();
206
207 bind(&continuation);
208
209 // Now allocate the stack space. Note that this might do more than just
210 // decrementing the SP; consult {MacroAssembler::AllocateStackSpace}.
211 lay(sp, MemOperand(sp, -frame_size));
212
213 // Jump back to the start of the function, from {pc_offset()} to
214 // right after the reserved space for the {__ sub(sp, sp, framesize)} (which
215 // is a branch now).
216 jump_offset = offset - pc_offset() + 6;
217 branchOnCond(al, jump_offset, true);
218}

243 Label* ool_label,
244 const FreezeCacheState& frozen) {
245 Register budget_array = ip;
246 Register instance_data = cache_state_.cached_instance_data;

263 blt(ool_label);

392void LiftoffAssembler::Load(LiftoffRegister dst, Register src_addr,
393 Register offset_reg, uintptr_t offset_imm,
394 LoadType type, uint32_t* protected_load_pc,
395 bool is_load_mem, bool i64_offset,
396 bool needs_shift) {
397 UseScratchRegisterScope temps(this);
398 if (offset_reg != no_reg && !i64_offset) {
399 // Clear the upper 32 bits of the 64 bit offset register.
400 llgfr(ip, offset_reg);
401 offset_reg = ip;
402 }
403 unsigned shift_amount = needs_shift ? type.size_log_2() : 0;
404 if (offset_reg != no_reg && shift_amount != 0) {
405 ShiftLeftU64(ip, offset_reg, Operand(shift_amount));
406 offset_reg = ip;
407 }
408 if (!is_int20(offset_imm)) {
409 if (offset_reg != no_reg) {
410 mov(r0, Operand(offset_imm));
411 AddS64(r0, offset_reg);
412 mov(ip, r0);
413 } else {
414 mov(ip, Operand(offset_imm));
415 }
416 offset_reg = ip;
417 offset_imm = 0;
418 }
419 MemOperand src_op =
420 MemOperand(src_addr, offset_reg == no_reg ? r0 : offset_reg, offset_imm);
421 if (protected_load_pc) *protected_load_pc = pc_offset();
422 switch (type.value()) {
423 case LoadType::kI32Load8U:
424 case LoadType::kI64Load8U:
425 LoadU8(dst.gp(), src_op);
426 break;
427 case LoadType::kI32Load8S:
428 case LoadType::kI64Load8S:
429 LoadS8(dst.gp(), src_op);
430 break;
431 case LoadType::kI32Load16U:
432 case LoadType::kI64Load16U:
433 if (is_load_mem) {
434 LoadU16LE(dst.gp(), src_op);
435 } else {
436 LoadU16(dst.gp(), src_op);
437 }
438 break;
439 case LoadType::kI32Load16S:
440 case LoadType::kI64Load16S:
441 if (is_load_mem) {
442 LoadS16LE(dst.gp(), src_op);
443 } else {
444 LoadS16(dst.gp(), src_op);
445 }
446 break;
447 case LoadType::kI64Load32U:
448 if (is_load_mem) {
449 LoadU32LE(dst.gp(), src_op);
450 } else {
451 LoadU32(dst.gp(), src_op);
452 }
453 break;
454 case LoadType::kI32Load:
455 case LoadType::kI64Load32S:
456 if (is_load_mem) {
457 LoadS32LE(dst.gp(), src_op);
458 } else {
459 LoadS32(dst.gp(), src_op);
460 }
461 break;
462 case LoadType::kI64Load:
463 if (is_load_mem) {
464 LoadU64LE(dst.gp(), src_op);
465 } else {
466 LoadU64(dst.gp(), src_op);
467 }
468 break;
469 case LoadType::kF32Load:
470 if (is_load_mem) {
471 LoadF32LE(dst.fp(), src_op, r0);
472 } else {
473 LoadF32(dst.fp(), src_op);
474 }
475 break;
476 case LoadType::kF64Load:
477 if (is_load_mem) {
478 LoadF64LE(dst.fp(), src_op, r0);
479 } else {
480 LoadF64(dst.fp(), src_op);
481 }
482 break;
483 case LoadType::kS128Load:
484 if (is_load_mem) {
485 LoadV128LE(dst.fp(), src_op, r1, r0);
486 } else {
487 LoadV128(dst.fp(), src_op, r1);
488 }
489 break;
490 default:
491 UNREACHABLE();
492 }
493}

583void LiftoffAssembler::AtomicStore(Register dst_addr, Register offset_reg,
584 uintptr_t offset_imm, LiftoffRegister src,
585 StoreType type, LiftoffRegList /* pinned */,
586 bool i64_offset) {
587 PREP_MEM_OPERAND(offset_reg, offset_imm, ip)
588 lay(ip,
589 MemOperand(dst_addr, offset_reg == no_reg ? r0 : offset_reg, offset_imm));
590
591 switch (type.value()) {
592 case StoreType::kI32Store8:
593 case StoreType::kI64Store8: {
594 AtomicExchangeU8(ip, src.gp(), r1, r0);
595 break;
596 }
597 case StoreType::kI32Store16:
598 case StoreType::kI64Store16: {
599#ifdef V8_TARGET_BIG_ENDIAN
600 lrvr(r1, src.gp());
601 ShiftRightU32(r1, r1, Operand(16));
602#else
603 LoadU16(r1, src.gp());
604#endif
605 Push(r2);
606 AtomicExchangeU16(ip, r1, r2, r0);
607 Pop(r2);
608 break;
609 }
610 case StoreType::kI32Store:
611 case StoreType::kI64Store32: {
612#ifdef V8_TARGET_BIG_ENDIAN
613 lrvr(r1, src.gp());
614#else
615 LoadU32(r1, src.gp());
616#endif
617 Label do_cs;
618 bind(&do_cs);
619 cs(r0, r1, MemOperand(ip));
620 bne(&do_cs, Label::kNear);
621 break;
622 }
623 case StoreType::kI64Store: {
624#ifdef V8_TARGET_BIG_ENDIAN
625 lrvgr(r1, src.gp());
626#else
627 mov(r1, src.gp());
628#endif
629 Label do_cs;
630 bind(&do_cs);
631 csg(r0, r1, MemOperand(ip));
632 bne(&do_cs, Label::kNear);
633 break;
634 }
635 default:
636 UNREACHABLE();
637 }

640void LiftoffAssembler::AtomicAdd(Register dst_addr, Register offset_reg,
641 uintptr_t offset_imm, LiftoffRegister value,
642 LiftoffRegister result, StoreType type,
643 bool i64_offset) {
644 LiftoffRegList pinned = LiftoffRegList{dst_addr, value, result};
645 if (offset_reg != no_reg) pinned.set(offset_reg);
646 Register tmp1 = GetUnusedRegister(kGpReg, pinned).gp();
647 pinned.set(tmp1);
648 Register tmp2 = GetUnusedRegister(kGpReg, pinned).gp();
649
650 PREP_MEM_OPERAND(offset_reg, offset_imm, ip)
651 lay(ip,
652 MemOperand(dst_addr, offset_reg == no_reg ? r0 : offset_reg, offset_imm));
653
654 switch (type.value()) {
655 case StoreType::kI32Store8:
656 case StoreType::kI64Store8: {
657 Label doadd;
658 bind(&doadd);
659 LoadU8(tmp1, MemOperand(ip));
660 AddS32(tmp2, tmp1, value.gp());
661 AtomicCmpExchangeU8(ip, result.gp(), tmp1, tmp2, r0, r1);

1385 case kI32:
1386 case kF32:
1387 length = 4;
1388 break;

1390 case kRefNull:
1391 case kRef:
1392 case kRtt:

1395 break;
1396 case kS128:
1397 length = 16;
1398 break;
1399 default:

1403 dst_offset += (length == 4 ? stack_bias : 0);
1404 src_offset += (length == 4 ? stack_bias : 0);
1405

1407 lay(ip, liftoff::GetStackSlot(dst_offset));
1408 } else {
1409 mov(ip, Operand(-dst_offset));
1410 lay(ip, MemOperand(fp, ip));

1414 lay(r1, liftoff::GetStackSlot(src_offset));
1415 } else {
1416 mov(r1, Operand(-src_offset));
1417 lay(r1, MemOperand(fp, r1));
1418 }
1419
1420 MoveChar(MemOperand(ip), MemOperand(r1), Operand(length));

1525void LiftoffAssembler::FillStackSlotsWithZero(int start, int size) {
1526 DCHECK_LT(0, size);
1527 DCHECK_EQ(0, size % 4);
1528 RecordUsedSpillOffset(start + size);
1529
1530 // We need a zero reg. Always use r0 for that, and push it before to restore
1531 // its value afterwards.
1532 push(r0);
1533 mov(r0, Operand(0));
1534
1535 if (size <= 5 * kStackSlotSize) {
1536 // Special straight-line code for up to five slots. Generates two
1537 // instructions per slot.
1538 uint32_t remainder = size;
1539 for (; remainder >= kStackSlotSize; remainder -= kStackSlotSize) {
1540 StoreU64(r0, liftoff::GetStackSlot(start + remainder));
1541 }
1542 DCHECK(remainder == 4 || remainder == 0);
1543 if (remainder) {
1544 StoreU32(r0, liftoff::GetStackSlot(start + remainder));
1545 }
1546 } else {
1547 // General case for bigger counts (9 instructions).
1548 // Use r3 for start address (inclusive), r4 for end address (exclusive).
1549 push(r3);
1550 push(r4);
1551
1552 lay(r3, MemOperand(fp, -start - size));
1553 lay(r4, MemOperand(fp, -start));
1554
1555 Label loop;
1556 bind(&loop);
1557 StoreU64(r0, MemOperand(r3));
1558 lay(r3, MemOperand(r3, kSystemPointerSize));
1559 CmpU64(r3, r4);
1560 bne(&loop);
1561 pop(r4);
1562 pop(r3);
1563 }
1564
1565 pop(r0);
1566}

2779void LiftoffAssembler::LoadTransform(LiftoffRegister dst, Register src_addr,
2780 Register offset_reg, uintptr_t offset_imm,
2781 LoadType type,
2782 LoadTransformationKind transform,
2783 uint32_t* protected_load_pc) {
2784 if (!is_int20(offset_imm)) {
2785 mov(ip, Operand(offset_imm));
2786 if (offset_reg != no_reg) {
2787 AddS64(ip, offset_reg);
2788 }
2789 offset_reg = ip;
2790 offset_imm = 0;
2791 }
2792 MemOperand src_op =
2793 MemOperand(src_addr, offset_reg == no_reg ? r0 : offset_reg, offset_imm);
2794 *protected_load_pc = pc_offset();
2795 MachineType memtype = type.mem_type();
2796 if (transform == LoadTransformationKind::kExtend) {
2797 if (memtype == MachineType::Int8()) {
2798 LoadAndExtend8x8SLE(dst.fp(), src_op, r1);
2799 } else if (memtype == MachineType::Uint8()) {
2800 LoadAndExtend8x8ULE(dst.fp(), src_op, r1);
2801 } else if (memtype == MachineType::Int16()) {
2802 LoadAndExtend16x4SLE(dst.fp(), src_op, r1);
2803 } else if (memtype == MachineType::Uint16()) {
2804 LoadAndExtend16x4ULE(dst.fp(), src_op, r1);
2805 } else if (memtype == MachineType::Int32()) {
2806 LoadAndExtend32x2SLE(dst.fp(), src_op, r1);
2807 } else if (memtype == MachineType::Uint32()) {
2808 LoadAndExtend32x2ULE(dst.fp(), src_op, r1);
2809 }
2810 } else if (transform == LoadTransformationKind::kZeroExtend) {
2811 if (memtype == MachineType::Int32()) {
2812 LoadV32ZeroLE(dst.fp(), src_op, r1);
2813 } else {
2814 DCHECK_EQ(MachineType::Int64(), memtype);
2815 LoadV64ZeroLE(dst.fp(), src_op, r1);
2816 }
2817 } else {
2818 DCHECK_EQ(LoadTransformationKind::kSplat, transform);
2819 if (memtype == MachineType::Int8()) {
2820 LoadAndSplat8x16LE(dst.fp(), src_op, r1);
2821 } else if (memtype == MachineType::Int16()) {
2822 LoadAndSplat16x8LE(dst.fp(), src_op, r1);
2823 } else if (memtype == MachineType::Int32()) {
2824 LoadAndSplat32x4LE(dst.fp(), src_op, r1);
2825 } else if (memtype == MachineType::Int64()) {
2826 LoadAndSplat64x2LE(dst.fp(), src_op, r1);
2827 }
2828 }
