46void ConstantExpressionInterface::UnOp(FullDecoder* decoder, WasmOpcode opcode,
47 const Value& input, Value* result) {
48 if (!generate_value()) return;
49 switch (opcode) {
50 case kExprExternConvertAny: {
51 result->runtime_value = WasmValue(
52 WasmToJSObject(isolate_, input.runtime_value.to_ref()),
53 ValueType::RefMaybeNull(HeapType::kExtern, input.type.nullability()),
54 decoder->module_);
55 break;
56 }
57 case kExprAnyConvertExtern: {
58 const char* error_message = nullptr;
59 result->runtime_value = WasmValue(
60 JSToWasmObject(isolate_, input.runtime_value.to_ref(),
61 kCanonicalAnyRef, &error_message)
62 .ToHandleChecked(),
63 ValueType::RefMaybeNull(HeapType::kAny, input.type.nullability()),
64 decoder->module_);
65 break;
66 }
67 default:
68 UNREACHABLE();
69 }
70}

72void ConstantExpressionInterface::BinOp(FullDecoder* decoder, WasmOpcode opcode,
73 const Value& lhs, const Value& rhs,
74 Value* result) {
75 if (!generate_value()) return;
76 switch (opcode) {
77 case kExprI32Add:
78 result->runtime_value = WasmValue(base::AddWithWraparound(
79 lhs.runtime_value.to_i32(), rhs.runtime_value.to_i32()));
80 break;
81 case kExprI32Sub:
82 result->runtime_value = WasmValue(base::SubWithWraparound(
83 lhs.runtime_value.to_i32(), rhs.runtime_value.to_i32()));
84 break;
85 case kExprI32Mul:
86 result->runtime_value = WasmValue(base::MulWithWraparound(
87 lhs.runtime_value.to_i32(), rhs.runtime_value.to_i32()));
88 break;
89 case kExprI64Add:
90 result->runtime_value = WasmValue(base::AddWithWraparound(
91 lhs.runtime_value.to_i64(), rhs.runtime_value.to_i64()));
92 break;
93 case kExprI64Sub:
94 result->runtime_value = WasmValue(base::SubWithWraparound(
95 lhs.runtime_value.to_i64(), rhs.runtime_value.to_i64()));
96 break;
97 case kExprI64Mul:
98 result->runtime_value = WasmValue(base::MulWithWraparound(
99 lhs.runtime_value.to_i64(), rhs.runtime_value.to_i64()));
100 break;
101 default:
102 UNREACHABLE();
103 }

170void ConstantExpressionInterface::StringConst(FullDecoder* decoder,
171 const StringConstImmediate& imm,
172 Value* result) {
173 if (!generate_value()) return;
174 static_assert(base::IsInRange(kV8MaxWasmStringLiterals, 0, Smi::kMaxValue));
175
176 DCHECK_LT(imm.index, module_->stringref_literals.size());
177
178 const wasm::WasmStringRefLiteral& literal =
179 module_->stringref_literals[imm.index];
180 const base::Vector<const uint8_t> module_bytes =
181 trusted_instance_data_->native_module()->wire_bytes();
182 const base::Vector<const uint8_t> string_bytes = module_bytes.SubVector(
183 literal.source.offset(), literal.source.end_offset());
184 Handle<String> string =
185 isolate_->factory()
186 ->NewStringFromUtf8(string_bytes, unibrow::Utf8Variant::kWtf8)
187 .ToHandleChecked();
188 result->runtime_value =
189 WasmValue(string, kWasmStringRef.AsNonNull(), decoder->module_);
190}

193WasmValue DefaultValueForType(ValueType type, Isolate* isolate,
194 const WasmModule* module) {
195 switch (type.kind()) {
196 case kI32:
197 case kI8:
198 case kI16:
199 return WasmValue(0);
200 case kI64:
201 return WasmValue(int64_t{0});
202 case kF16:
203 case kF32:
204 return WasmValue(0.0f);
205 case kF64:
206 return WasmValue(0.0);
207 case kS128:
208 return WasmValue(Simd128());

211 ? Cast<Object>(isolate->factory()->wasm_null())
212 : Cast<Object>(isolate->factory()->null_value()),
213 type, module);
214 case kVoid:
215 case kRtt:
216 case kRef:
217 case kTop:
218 case kBottom:
219 UNREACHABLE();

297void ConstantExpressionInterface::ArrayNewSegment(
298 FullDecoder* decoder, const ArrayIndexImmediate& array_imm,
299 const IndexImmediate& segment_imm, const Value& offset_value,
300 const Value& length_value, Value* result) {
301 if (!generate_value()) return;
302
303 DirectHandle<WasmTrustedInstanceData> data =
304 GetTrustedInstanceDataForTypeIndex(array_imm.index);
305
306 DirectHandle<Map> rtt{
307 Cast<Map>(data->managed_object_maps()->get(array_imm.index.index)),
308 isolate_};
309
310 uint32_t length = length_value.runtime_value.to_u32();
311 uint32_t offset = offset_value.runtime_value.to_u32();
312 if (length >
313 static_cast<uint32_t>(WasmArray::MaxLength(array_imm.array_type))) {
314 error_ = MessageTemplate::kWasmTrapArrayTooLarge;
315 return;
316 }
317 ValueType element_type = array_imm.array_type->element_type();
318 ValueType result_type = ValueType::Ref(HeapType(array_imm.index));
319 if (element_type.is_numeric()) {
320 const WasmDataSegment& data_segment =
321 module_->data_segments[segment_imm.index];
322 uint32_t length_in_bytes =
323 length * array_imm.array_type->element_type().value_kind_size();
324
325 if (!base::IsInBounds<uint32_t>(offset, length_in_bytes,
326 data_segment.source.length())) {
327 error_ = MessageTemplate::kWasmTrapDataSegmentOutOfBounds;
328 return;
329 }
330
331 Address source =
332 data->data_segment_starts()->get(segment_imm.index) + offset;
333 Handle<WasmArray> array_value =
334 isolate_->factory()->NewWasmArrayFromMemory(length, rtt, source);
335 result->runtime_value =
336 WasmValue(array_value, result_type, decoder->module_);
337 } else {
338 const wasm::WasmElemSegment* elem_segment =
339 &decoder->module_->elem_segments[segment_imm.index];
340 // A constant expression should not observe if a passive segment is dropped.
341 // However, it should consider active and declarative segments as empty.
342 if (!base::IsInBounds<size_t>(
343 offset, length,
344 elem_segment->status == WasmElemSegment::kStatusPassive
345 ? elem_segment->element_count
346 : 0)) {
347 error_ = MessageTemplate::kWasmTrapElementSegmentOutOfBounds;
348 return;
349 }
350
351 Handle<Object> array_object =
352 isolate_->factory()->NewWasmArrayFromElementSegment(
353 trusted_instance_data_, shared_trusted_instance_data_,
354 segment_imm.index, offset, length, rtt);
355 if (IsSmi(*array_object)) {
356 // A smi result stands for an error code.
357 error_ = static_cast<MessageTemplate>(Cast<Smi>(*array_object).value());
358 } else {
359 result->runtime_value =
360 WasmValue(array_object, result_type, decoder->module_);
361 }
362 }

365void ConstantExpressionInterface::RefI31(FullDecoder* decoder,
366 const Value& input, Value* result) {
367 if (!generate_value()) return;
368 Address raw = input.runtime_value.to_i32();
369 // We have to craft the Smi manually because we accept out-of-bounds inputs.
370 // For 32-bit Smi builds, set the topmost bit to sign-extend the second bit.
371 // This way, interpretation in JS (if this value escapes there) will be the
372 // same as i31.get_s.
373 static_assert((SmiValuesAre31Bits() ^ SmiValuesAre32Bits()) == 1);
374 intptr_t shifted;
375 if constexpr (SmiValuesAre31Bits()) {
376 shifted = raw << (kSmiTagSize + kSmiShiftSize);
377 } else {
378 shifted =
379 static_cast<intptr_t>(raw << (kSmiTagSize + kSmiShiftSize + 1)) >> 1;
380 }
381 result->runtime_value =
382 WasmValue(handle(Tagged<Smi>(shifted), isolate_),
383 wasm::kWasmI31Ref.AsNonNull(), decoder->module_);
384}
