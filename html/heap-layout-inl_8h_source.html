23bool HeapLayout::InYoungGeneration(const MemoryChunk* chunk,
24 Tagged<HeapObject> object) {
25 if constexpr (v8_flags.single_generation.value()) {
26 return false;
27 }
28 if constexpr (v8_flags.sticky_mark_bits.value()) {
29 return InYoungGenerationForStickyMarkbits(chunk, object);
30 }
31 const bool in_young_generation = chunk->InYoungGeneration();
32#ifdef DEBUG
33 if (in_young_generation) {
34 CheckYoungGenerationConsistency(chunk);
35 }
36#endif // DEBUG
37 return in_young_generation;
38}
