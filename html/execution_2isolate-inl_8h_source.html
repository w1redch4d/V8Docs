66Handle<NativeContext> Isolate::GetIncumbentContext() {
67 Tagged<Context> maybe_topmost_script_having_context =
68 topmost_script_having_context();
69 if (V8_LIKELY(!maybe_topmost_script_having_context.is_null())) {
70 // The topmost script-having context value is guaranteed to be valid only
71 // inside the Api callback however direct calls of Api callbacks from
72 // builtins or optimized code do not change the current VM state, so we
73 // allow JS VM state too.
74 DCHECK(current_vm_state() == EXTERNAL || // called from C++ code
75 current_vm_state() == JS); // called from JS code directly
76
77 Tagged<NativeContext> incumbent_context =
78 maybe_topmost_script_having_context->native_context();
79 DCHECK_EQ(incumbent_context, *GetIncumbentContextSlow());
80 return handle(incumbent_context, this);
81 }
82 return GetIncumbentContextSlow();
83}

134#ifdef DEBUG
135Tagged<Object> Isolate::VerifyBuiltinsResult(Tagged<Object> result) {
136 if (is_execution_terminating() && !v8_flags.strict_termination_checks) {
137 // We may be missing places where termination checks are handled properly.
138 // If that's the case, it's likely that we'll have one sitting around when
139 // we return from a builtin. If we're not looking to find such bugs
140 // (strict_termination_checks is false), simply return the exception marker.
141 return ReadOnlyRoots(this).exception();
142 }
143
144 // Here we use full pointer comparison as the result might be an object
145 // outside of the main pointer compression heap (e.g. in trusted space).
146 DCHECK_EQ(has_exception(),
147 result.SafeEquals(ReadOnlyRoots(this).exception()));
148
149#ifdef V8_COMPRESS_POINTERS
150 // Check that the returned pointer is actually part of the current isolate (or
151 // the shared isolate), because that's the assumption in generated code (which
152 // might call this builtin).
153 Isolate* isolate;
154 if (!IsSmi(result) &&
155 GetIsolateFromHeapObject(Cast<HeapObject>(result), &isolate)) {
156 DCHECK(isolate == this || isolate == shared_space_isolate());
157 }
158#endif
159
160 return result;
161}
