76Tagged<MaybeObject> PrototypeInfo::GetDerivedMap(DirectHandle<Map> from) {
77 if (IsUndefined(derived_maps())) {
78 return Tagged<MaybeObject>();
79 }
80 auto derived = Cast<WeakArrayList>(derived_maps());
81 // Index 0 is the map for object create
82 for (int i = 1; i < derived->length(); ++i) {
83 Tagged<MaybeObject> el = derived->Get(i);
84 Tagged<Map> map_obj;
85 if (el.GetHeapObjectIfWeak(&map_obj)) {
86 Tagged<Map> to = Cast<Map>(map_obj);
87 if (to->GetConstructor() == from->GetConstructor() &&
88 to->instance_type() == from->instance_type()) {
89 return el;
90 }
91 }
92 }
93 return Tagged<MaybeObject>();
94}

97void PrototypeInfo::AddDerivedMap(DirectHandle<PrototypeInfo> info,
98 DirectHandle<Map> to, Isolate* isolate) {
99 if (IsUndefined(info->derived_maps())) {
100 // Index 0 is the map for object create
101 Tagged<WeakArrayList> derived = *isolate->factory()->NewWeakArrayList(2);
102 // GetConstructMap assumes a weak pointer.
103 derived->Set(0, ClearedValue(isolate));
104 derived->Set(1, MakeWeak(*to));
105 derived->set_length(2);
106 info->set_derived_maps(derived, kReleaseStore);
107 return;
108 }
109 auto derived = handle(Cast<WeakArrayList>(info->derived_maps()), isolate);
110 // Index 0 is the map for object create
111 int i = 1;
112 for (; i < derived->length(); ++i) {
113 Tagged<MaybeObject> el = derived->Get(i);
114 if (el.IsCleared()) {
115 derived->Set(i, MakeWeak(*to));
116 return;
117 }
118 }
119
120 auto bigger = WeakArrayList::EnsureSpace(isolate, derived, i + 1);
121 bigger->Set(i, MakeWeak(*to));
122 bigger->set_length(i + 1);
123 if (*bigger != *derived) {
124 info->set_derived_maps(*bigger, kReleaseStore);
125 }
126}
