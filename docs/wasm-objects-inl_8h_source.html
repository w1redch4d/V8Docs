115wasm::ValueType WasmGlobalObject::type() const {
116 // Various consumers of ValueKind (e.g. ValueKind::name()) use the raw enum
117 // value as index into a global array. As such, if the index is corrupted
118 // (which must be assumed, as it comes from within the sandbox), this can
119 // lead to out-of-bounds reads outside the sandbox. While these are not
120 // technically sandbox violations, we should still try to avoid them to keep
121 // fuzzers happy. This SBXCHECK accomplishes that.
122 wasm::ValueType type = wasm::ValueType::FromRawBitField(raw_type());
123 SBXCHECK(is_valid(type.kind()));
124 return type;
125}

127 set_raw_type(static_cast<int>(value.raw_bit_field()));

638Handle<Object> WasmObject::ReadValueAt(Isolate* isolate,
639 DirectHandle<HeapObject> obj,
640 wasm::ValueType type, uint32_t offset) {
641 Address field_address = obj->GetFieldAddress(offset);
642 switch (type.kind()) {
643 case wasm::kI8: {
644 int8_t value = base::Memory<int8_t>(field_address);
645 return handle(Smi::FromInt(value), isolate);
646 }
647 case wasm::kI16: {
648 int16_t value = base::Memory<int16_t>(field_address);
649 return handle(Smi::FromInt(value), isolate);
650 }
651 case wasm::kI32: {
652 int32_t value = base::Memory<int32_t>(field_address);
653 return isolate->factory()->NewNumberFromInt(value);
654 }
655 case wasm::kI64: {
656 int64_t value = base::ReadUnalignedValue<int64_t>(field_address);
657 return BigInt::FromInt64(isolate, value);
658 }
659 case wasm::kF16: {
660 uint16_t value = base::Memory<uint16_t>(field_address);
661 return isolate->factory()->NewNumber(fp16_ieee_to_fp32_value(value));
662 }
663 case wasm::kF32: {
664 float value = base::Memory<float>(field_address);
665 return isolate->factory()->NewNumber(value);
666 }
667 case wasm::kF64: {
668 double value = base::ReadUnalignedValue<double>(field_address);
669 return isolate->factory()->NewNumber(value);
670 }
671 case wasm::kS128:
672 // TODO(v8:11804): implement
673 UNREACHABLE();
674
675 case wasm::kRef:
676 case wasm::kRefNull: {
677 ObjectSlot slot(field_address);
678 return handle(slot.load(isolate), isolate);
679 }
680
681 case wasm::kRtt:
682 // Rtt values are not supposed to be made available to JavaScript side.
683 UNREACHABLE();
684
685 case wasm::kVoid:
686 case wasm::kTop:
687 case wasm::kBottom:
688 UNREACHABLE();
689 }
690}

735void WasmStruct::EncodeInstanceSizeInMap(int instance_size, Tagged<Map> map) {
736 // WasmStructs can be bigger than the {map.instance_size_in_words} field
737 // can describe; yet we have to store the instance size somewhere on the
