v8
V8 is Google’s open source high-performance JavaScript and WebAssembly engine, written in C++. It is used in Chrome and in Node.js, among others. It implements ECMAScript and WebAssembly, and runs on Windows, macOS, and Linux systems that use x64, IA-32, or ARM processors. V8 can be embedded into any C++ application.

22TNode<JSProxy> ProxiesCodeStubAssembler::AllocateProxy(
23 TNode<Context> context, TNode<JSReceiver> target,
24 TNode<JSReceiver> handler) {
25 TVARIABLE(Map, map);
26
27 Label callable_target(this), constructor_target(this), none_target(this),
28 create_proxy(this);
29
30 TNode<NativeContext> nativeContext = LoadNativeContext(context);
31
32 Branch(IsCallable(target), &callable_target, &none_target);
33
34 BIND(&callable_target);
35 {
36 // Every object that is a constructor is implicitly callable
37 // so it's okay to nest this check here
38 GotoIf(IsConstructor(target), &constructor_target);
39 map = CAST(
40 LoadContextElement(nativeContext, Context::PROXY_CALLABLE_MAP_INDEX));
41 Goto(&create_proxy);
42 }
43 BIND(&constructor_target);
44 {
45 map = CAST(LoadContextElement(nativeContext,
46 Context::PROXY_CONSTRUCTOR_MAP_INDEX));
47 Goto(&create_proxy);
48 }
49 BIND(&none_target);
50 {
51 map = CAST(LoadContextElement(nativeContext, Context::PROXY_MAP_INDEX));
52 Goto(&create_proxy);
53 }
54
55 BIND(&create_proxy);
56 TNode<HeapObject> proxy = Allocate(JSProxy::kSize);
57 StoreMapNoWriteBarrier(proxy, map.value());
58 RootIndex empty_dict = V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL
59 ? RootIndex::kEmptySwissPropertyDictionary
60 : RootIndex::kEmptyPropertyDictionary;
61 StoreObjectFieldRoot(proxy, JSProxy::kPropertiesOrHashOffset, empty_dict);
62 StoreObjectFieldNoWriteBarrier(proxy, JSProxy::kTargetOffset, target);
63 StoreObjectFieldNoWriteBarrier(proxy, JSProxy::kHandlerOffset, handler);
64
65 return CAST(proxy);
66}

86TF_BUILTIN(CallProxy, ProxiesCodeStubAssembler) {
87 auto argc = UncheckedParameter<Int32T>(Descriptor::kActualArgumentsCount);
88 TNode<IntPtrT> argc_ptr = ChangeInt32ToIntPtr(argc);
89 auto proxy = Parameter<JSProxy>(Descriptor::kFunction);
90 auto context = Parameter<Context>(Descriptor::kContext);
91
92 CSA_DCHECK(this, IsCallable(proxy));
93
94 PerformStackCheck(context);
95
96 Label throw_proxy_handler_revoked(this, Label::kDeferred),
97 trap_undefined(this);
98
99 // 1. Let handler be the value of the [[ProxyHandler]] internal slot of O.
100 TNode<HeapObject> handler =
101 CAST(LoadObjectField(proxy, JSProxy::kHandlerOffset));
102
103 // 2. If handler is null, throw a TypeError exception.
104 CSA_DCHECK(this, IsNullOrJSReceiver(handler));
105 GotoIfNot(JSAnyIsNotPrimitive(handler), &throw_proxy_handler_revoked);
106
107 // 3. Assert: Type(handler) is Object.
108 CSA_DCHECK(this, IsJSReceiver(handler));
109
110 // 4. Let target be the value of the [[ProxyTarget]] internal slot of O.
111 TNode<Object> target = LoadObjectField(proxy, JSProxy::kTargetOffset);
112
113 // 5. Let trap be ? GetMethod(handler, "apply").
114 // 6. If trap is undefined, then
115 Handle<Name> trap_name = factory()->apply_string();
116 TNode<Object> trap = GetMethod(context, handler, trap_name, &trap_undefined);
117
118 CodeStubArguments args(this, argc_ptr);
119 TNode<Object> receiver = args.GetReceiver();
120
121 // 7. Let argArray be CreateArrayFromList(argumentsList).
122 TNode<JSArray> array = EmitFastNewAllArguments(
123 UncheckedCast<Context>(context),
124 UncheckedCast<RawPtrT>(LoadFramePointer()),
125 UncheckedCast<IntPtrT>(args.GetLengthWithoutReceiver()));
126
127 // 8. Return Call(trap, handler, «target, thisArgument, argArray»).
128 TNode<Object> result = Call(context, trap, handler, target, receiver, array);
129 args.PopAndReturn(result);
130
131 BIND(&trap_undefined);
132 {
133 // 6.a. Return Call(target, thisArgument, argumentsList).
134 TailCallBuiltin(Builtins::Call(), context, target, argc);
135 }
136
137 BIND(&throw_proxy_handler_revoked);
138 { ThrowTypeError(context, MessageTemplate::kProxyRevoked, "apply"); }
139}

211void ProxiesCodeStubAssembler::CheckGetSetTrapResult(
212 TNode<Context> context, TNode<JSReceiver> target, TNode<JSProxy> proxy,
213 TNode<Name> name, TNode<Object> trap_result,
214 JSProxy::AccessKind access_kind) {
215 // TODO(mslekova): Think of a better name for the trap_result param.
216 TNode<Map> map = LoadMap(target);
217 TVARIABLE(Object, var_value);
218 TVARIABLE(Uint32T, var_details);
219 TVARIABLE(Object, var_raw_value);
220
221 Label if_found_value(this), check_in_runtime(this, Label::kDeferred),
222 check_passed(this);
223
224 GotoIfNot(IsUniqueNameNoIndex(name), &check_in_runtime);
225 TNode<Uint16T> instance_type = LoadInstanceType(target);
226 TryGetOwnProperty(context, target, target, map, instance_type, name,
227 &if_found_value, &var_value, &var_details, &var_raw_value,
228 &check_passed, &check_in_runtime, kReturnAccessorPair);
229
230 BIND(&if_found_value);
231 {
232 Label throw_non_configurable_data(this, Label::kDeferred),
233 throw_non_configurable_accessor(this, Label::kDeferred),
234 check_accessor(this), check_data(this);
235
236 // If targetDesc is not undefined and targetDesc.[[Configurable]] is
237 // false, then:
238 GotoIfNot(IsSetWord32(var_details.value(),
239 PropertyDetails::kAttributesDontDeleteMask),
240 &check_passed);
241
242 // If IsDataDescriptor(targetDesc) is true and
243 // targetDesc.[[Writable]] is false, then:
244 BranchIfAccessorPair(var_raw_value.value(), &check_accessor, &check_data);
245
246 BIND(&check_data);
247 {
248 TNode<BoolT> read_only = IsSetWord32(
249 var_details.value(), PropertyDetails::kAttributesReadOnlyMask);
250 GotoIfNot(read_only, &check_passed);
251
252 // If SameValue(trapResult, targetDesc.[[Value]]) is false,
253 // throw a TypeError exception.
254 BranchIfSameValue(trap_result, var_value.value(), &check_passed,
255 &throw_non_configurable_data);
256 }
257
258 BIND(&check_accessor);
259 {
260 TNode<HeapObject> accessor_pair = CAST(var_raw_value.value());
261
262 if (access_kind == JSProxy::kGet) {
263 Label continue_check(this, Label::kDeferred);
264 // 10.b. If IsAccessorDescriptor(targetDesc) is true and
265 // targetDesc.[[Get]] is undefined, then:
266 TNode<Object> getter =
267 LoadObjectField(accessor_pair, AccessorPair::kGetterOffset);
268 // Here we check for null as well because if the getter was never
269 // defined it's set as null.
270 GotoIf(IsUndefined(getter), &continue_check);
271 GotoIf(IsNull(getter), &continue_check);
272 Goto(&check_passed);
273
274 // 10.b.i. If trapResult is not undefined, throw a TypeError exception.
275 BIND(&continue_check);
276 GotoIfNot(IsUndefined(trap_result), &throw_non_configurable_accessor);
277 } else {
278 // 11.b.i. If targetDesc.[[Set]] is undefined, throw a TypeError
279 // exception.
280 TNode<Object> setter =
281 LoadObjectField(accessor_pair, AccessorPair::kSetterOffset);
282 GotoIf(IsUndefined(setter), &throw_non_configurable_accessor);
283 GotoIf(IsNull(setter), &throw_non_configurable_accessor);
284 }
285 Goto(&check_passed);
286 }
287
288 BIND(&throw_non_configurable_data);
289 {
290 if (access_kind == JSProxy::kGet) {
291 ThrowTypeError(context, MessageTemplate::kProxyGetNonConfigurableData,
292 name, var_value.value(), trap_result);
293 } else {
294 ThrowTypeError(context, MessageTemplate::kProxySetFrozenData, name);
295 }
296 }
297
298 BIND(&throw_non_configurable_accessor);
299 {
300 if (access_kind == JSProxy::kGet) {
301 ThrowTypeError(context,
302 MessageTemplate::kProxyGetNonConfigurableAccessor, name,
303 trap_result);
304 } else {
305 ThrowTypeError(context, MessageTemplate::kProxySetFrozenAccessor, name);
306 }
307 }
308
309 BIND(&check_in_runtime);
310 {
311 CallRuntime(Runtime::kCheckProxyGetSetTrapResult, context, name, target,
312 trap_result, SmiConstant(access_kind));
313 Goto(&check_passed);
314 }
315
316 BIND(&check_passed);
317 }

320void ProxiesCodeStubAssembler::CheckHasTrapResult(TNode<Context> context,
321 TNode<JSReceiver> target,
322 TNode<JSProxy> proxy,
323 TNode<Name> name) {
324 TNode<Map> target_map = LoadMap(target);
325 TVARIABLE(Object, var_value);
326 TVARIABLE(Uint32T, var_details);
327 TVARIABLE(Object, var_raw_value);
328
329 Label if_found_value(this, Label::kDeferred),
330 throw_non_configurable(this, Label::kDeferred),
331 throw_non_extensible(this, Label::kDeferred), check_passed(this),
332 check_in_runtime(this, Label::kDeferred);
333
334 // 9.a. Let targetDesc be ? target.[[GetOwnProperty]](P).
335 GotoIfNot(IsUniqueNameNoIndex(name), &check_in_runtime);
336 TNode<Uint16T> instance_type = LoadInstanceType(target);
337 TryGetOwnProperty(context, target, target, target_map, instance_type, name,
338 &if_found_value, &var_value, &var_details, &var_raw_value,
339 &check_passed, &check_in_runtime, kReturnAccessorPair);
340
341 // 9.b. If targetDesc is not undefined, then (see 9.b.i. below).
342 BIND(&if_found_value);
343 {
344 // 9.b.i. If targetDesc.[[Configurable]] is false, throw a TypeError
345 // exception.
346 TNode<BoolT> non_configurable = IsSetWord32(
347 var_details.value(), PropertyDetails::kAttributesDontDeleteMask);
348 GotoIf(non_configurable, &throw_non_configurable);
349
350 // 9.b.ii. Let extensibleTarget be ? IsExtensible(target).
351 TNode<BoolT> target_extensible = IsExtensibleMap(target_map);
352
353 // 9.b.iii. If extensibleTarget is false, throw a TypeError exception.
354 GotoIfNot(target_extensible, &throw_non_extensible);
355 Goto(&check_passed);
356 }
357
358 BIND(&throw_non_configurable);
359 { ThrowTypeError(context, MessageTemplate::kProxyHasNonConfigurable, name); }
360
361 BIND(&throw_non_extensible);
362 { ThrowTypeError(context, MessageTemplate::kProxyHasNonExtensible, name); }
363
364 BIND(&check_in_runtime);
365 {
366 CallRuntime(Runtime::kCheckProxyHasTrapResult, context, name, target);
367 Goto(&check_passed);
368 }
369
370 BIND(&check_passed);
371}
