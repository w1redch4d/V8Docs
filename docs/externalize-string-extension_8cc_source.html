94void ExternalizeStringExtension::Externalize(
95 const v8::FunctionCallbackInfo<v8::Value>& info) {
96 DCHECK(ValidateCallbackInfo(info));
97 if (info.Length() < 1 || !info[0]->IsString()) {
98 info.GetIsolate()->ThrowError(
99 "First parameter to externalizeString() must be a string.");
100 return;
101 }
102 bool result = false;
103 DirectHandle<String> string =
104 Utils::OpenDirectHandle(*info[0].As<v8::String>());
105 const bool externalize_as_one_byte = string->IsOneByteRepresentation();
106 if (!string->SupportsExternalization(
107 externalize_as_one_byte ? v8::String::Encoding::ONE_BYTE_ENCODING
108 : v8::String::Encoding::TWO_BYTE_ENCODING)) {
109 // If the string is shared, testing with the combination of
110 // --shared-string-table and --isolate in d8 may result in races to
111 // externalize the same string. If GC is stressed in addition, this test
112 // might fail as the string was already externalized (marked for
113 // externalization by another thread and externalized during GC).
114 if (!StringShape(*string).IsExternal()) {
115 info.GetIsolate()->ThrowError("string does not support externalization.");
116 }
117 return;
118 }
119 if (externalize_as_one_byte) {
120 uint8_t* data = new uint8_t[string->length()];
121 String::WriteToFlat(*string, data, 0, string->length());
122 SimpleOneByteStringResource* resource = new SimpleOneByteStringResource(
123 reinterpret_cast<char*>(data), string->length());
124 result = Utils::ToLocal(string)->MakeExternal(info.GetIsolate(), resource);
125 if (!result) delete resource;
126 } else {
127 base::uc16* data = new base::uc16[string->length()];
128 String::WriteToFlat(*string, data, 0, string->length());
129 SimpleTwoByteStringResource* resource = new SimpleTwoByteStringResource(
130 data, string->length());
131 result = Utils::ToLocal(string)->MakeExternal(info.GetIsolate(), resource);
132 if (!result) delete resource;
133 }
134 // If the string is shared, testing with the combination of
135 // --shared-string-table and --isolate in d8 may result in races to
136 // externalize the same string. Those races manifest as externalization
137 // sometimes failing if another thread won and already forwarded the string to
138 // the external resource. Don't consider those races as failures.
139 if (!result && !HasExternalForwardingIndex(string)) {
140 info.GetIsolate()->ThrowError("externalizeString() failed.");
141 return;
142 }
143}

156void ExternalizeStringExtension::CreateExternalizableString(
157 const v8::FunctionCallbackInfo<v8::Value>& info) {
158 DCHECK(ValidateCallbackInfo(info));
159 if (info.Length() < 1 || !info[0]->IsString()) {
160 info.GetIsolate()->ThrowError(
161 "First parameter to createExternalizableString() must be a string.");
162 return;
163 }
164 DirectHandle<String> string =
165 Utils::OpenDirectHandle(*info[0].As<v8::String>());
166 Isolate* isolate = reinterpret_cast<Isolate*>(info.GetIsolate());
167 v8::String::Encoding encoding = string->IsOneByteRepresentation()
168 ? v8::String::Encoding::ONE_BYTE_ENCODING
169 : v8::String::Encoding::TWO_BYTE_ENCODING;
170 if (string->SupportsExternalization(encoding)) {
171 info.GetReturnValue().Set(Utils::ToLocal(string));
172 return;
173 }
174 // Return the string if it is already externalized.
175 if (StringShape(*string).IsExternal()) {
176 info.GetReturnValue().Set(Utils::ToLocal(string));
177 return;
178 }
179
180 // Read-only strings are never externalizable. Don't try to copy them as
181 // some parts of the code might rely on some strings being in RO space (i.e.
182 // empty string).
183 if (HeapLayout::InReadOnlySpace(*string)) {
184 info.GetIsolate()->ThrowError("Read-only strings cannot be externalized.");
185 return;
186 }
187#ifdef V8_COMPRESS_POINTERS
188 // Small strings may not be in-place externalizable.
189 if (string->Size() < static_cast<int>(sizeof(UncachedExternalString))) {
190 info.GetIsolate()->ThrowError("String is too short to be externalized.");
191 return;
192 }
193#endif
194
195 // Special handling for ConsStrings, as the ConsString -> ExternalString
196 // migration is special for GC (Tagged pointers to Untagged pointers).
197 // Skip if the ConsString is flat (second is empty), as we won't be guaranteed
198 // a string in old space in that case.
199 if (IsConsString(*string, isolate) && !string->IsFlat()) {
200 DirectHandle<String> result;
201 if (CopyConsStringToOld(isolate, Cast<ConsString>(string))
202 .ToHandle(&result)) {
203 DCHECK(result->SupportsExternalization(encoding));
204 info.GetReturnValue().Set(Utils::ToLocal(result));
205 return;
206 }
207 }
208 // All other strings can be implicitly flattened.
209 if (encoding == v8::String::ONE_BYTE_ENCODING) {
210 MaybeDirectHandle<SeqOneByteString> maybe_result =
211 isolate->factory()->NewRawOneByteString(string->length(),
212 AllocationType::kOld);
213 DirectHandle<SeqOneByteString> result;
214 if (maybe_result.ToHandle(&result)) {
215 DisallowGarbageCollection no_gc;
216 String::WriteToFlat(*string, result->GetChars(no_gc), 0,
217 string->length());
218 DCHECK(result->SupportsExternalization(encoding));
219 info.GetReturnValue().Set(Utils::ToLocal(Cast<String>(result)));
220 return;
221 }
222 } else {
223 MaybeDirectHandle<SeqTwoByteString> maybe_result =
224 isolate->factory()->NewRawTwoByteString(string->length(),
225 AllocationType::kOld);
226 DirectHandle<SeqTwoByteString> result;
227 if (maybe_result.ToHandle(&result)) {
228 DisallowGarbageCollection no_gc;
229 String::WriteToFlat(*string, result->GetChars(no_gc), 0,
230 string->length());
231 DCHECK(result->SupportsExternalization(encoding));
232 info.GetReturnValue().Set(Utils::ToLocal(Cast<String>(result)));
233 return;
234 }
235 }
236 info.GetIsolate()->ThrowError("Unable to create string");
237}
