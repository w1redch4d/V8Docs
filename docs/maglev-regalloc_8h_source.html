23// Represents the state of the register frame during register allocation,
24// including current register values, and the state of each register.
25//
26// Register state encodes two orthogonal concepts:
27//
28// 1. Used/free registers: Which registers currently hold a valid value,
29// 2. Blocked/unblocked registers: Which registers can be modified during the
30// current allocation.
31//
32// The combination of these encodes values in different states:
33//
34// Free + unblocked: Completely unused registers which can be used for
35// anything.
36// Used + unblocked: Live values that can be spilled if there is register
37// pressure.
38// Used + blocked: Values that are in a register and are used as an input in
39// the current allocation.
40// Free + blocked: Unused registers that are reserved as temporaries, or
41// inputs that will get clobbered during the execution of the
42// node being allocated.
43template <typename RegisterT>
44class RegisterFrameState {
45 public:
46 static constexpr bool kIsGeneralRegister =
47 std::is_same<Register, RegisterT>();
48 static constexpr bool kIsDoubleRegister =
49 std::is_same<DoubleRegister, RegisterT>();
50
51 static_assert(kIsGeneralRegister || kIsDoubleRegister,
52 "RegisterFrameState should be used only for Register and "
53 "DoubleRegister.");
54
55 using RegTList = RegListBase<RegisterT>;
56
57 static constexpr RegTList kAllocatableRegisters =
58 AllocatableRegisters<RegisterT>::kRegisters;
59 static constexpr RegTList kEmptyRegList = {};
60
61 RegTList empty() const { return kEmptyRegList; }
62 RegTList free() const { return free_; }
63 RegTList unblocked_free() const { return free_ - blocked_; }
64 RegTList used() const {
65 // Only allocatable registers should be free.
66 DCHECK_EQ(free_, free_ & kAllocatableRegisters);
67 return kAllocatableRegisters ^ free_;
68 }
69
70 bool UnblockedFreeIsEmpty() const { return unblocked_free().is_empty(); }
71
72 template <typename Function>
73 void ForEachUsedRegister(Function&& f) const {
74 for (RegisterT reg : used()) {
75 f(reg, GetValue(reg));
76 }
77 }
78
79 void RemoveFromFree(RegisterT reg) { free_.clear(reg); }
80 void AddToFree(RegisterT reg) { free_.set(reg); }
81 void AddToFree(RegTList list) { free_ |= list; }
82
83 void FreeRegistersUsedBy(ValueNode* node) {
84 RegTList list = node->ClearRegisters<RegisterT>();
85 DCHECK_EQ(free_ & list, kEmptyRegList);
86 free_ |= list;
87 }
88
89 void SetValue(RegisterT reg, ValueNode* node) {
90 DCHECK(!free_.has(reg));
91 DCHECK(!blocked_.has(reg));
92 values_[reg.code()] = node;
93 block(reg);
94 node->AddRegister(reg);
95 }
96 void SetValueWithoutBlocking(RegisterT reg, ValueNode* node) {
97 DCHECK(!free_.has(reg));
98 DCHECK(!blocked_.has(reg));
99 values_[reg.code()] = node;
100 node->AddRegister(reg);
101 }
102 ValueNode* GetValue(RegisterT reg) const {
103 DCHECK(!free_.has(reg));
104 ValueNode* node = values_[reg.code()];
105 DCHECK_NOT_NULL(node);
106 return node;
107 }
108#ifdef DEBUG
109 // Like GetValue, but allow reading freed registers as long as they were also
110 // blocked. This allows us to DCHECK expected register state against node
111 // state, even if that node is dead or clobbered by the end of the current
112 // allocation.
113 ValueNode* GetValueMaybeFreeButBlocked(RegisterT reg) const {
114 DCHECK(!free_.has(reg) || blocked_.has(reg));
115 ValueNode* node = values_[reg.code()];
116 DCHECK_NOT_NULL(node);
117 return node;
118 }
119#endif
120
121 RegTList blocked() const { return blocked_; }
122 void block(RegisterT reg) { blocked_.set(reg); }
123 void unblock(RegisterT reg) { blocked_.clear(reg); }
124 bool is_blocked(RegisterT reg) { return blocked_.has(reg); }
125 void clear_blocked() { blocked_ = kEmptyRegList; }
126
127 compiler::InstructionOperand TryChooseInputRegister(
128 ValueNode* node, const compiler::InstructionOperand& hint =
129 compiler::InstructionOperand());
130 compiler::InstructionOperand TryChooseUnblockedInputRegister(ValueNode* node);
131 compiler::AllocatedOperand AllocateRegister(
132 ValueNode* node, const compiler::InstructionOperand& hint =
133 compiler::InstructionOperand());
134
135 private:
136 ValueNode* values_[RegisterT::kNumRegisters];
137 RegTList free_ = kAllocatableRegisters;
138 RegTList blocked_ = kEmptyRegList;
139};
