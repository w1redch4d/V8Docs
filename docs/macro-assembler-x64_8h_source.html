409 void Move(Register dst, intptr_t x) {
410 if (x == 0) {
411 xorl(dst, dst);
412 // The following shorter sequence for uint8 causes performance
413 // regressions:
414 // xorl(dst, dst); movb(dst,
415 // Immediate(static_cast<uint32_t>(x)));

418 } else if (is_int32(x)) {
419 // "movq reg64, imm32" is sign extending.
420 movq(dst, Immediate(static_cast<int32_t>(x)));
421 } else {
422 movq(dst, Immediate64(x));
423 }
