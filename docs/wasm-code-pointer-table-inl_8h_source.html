95WasmCodePointerTable::FreelistHead WasmCodePointerTable::ReadFreelistHead() {
96 while (true) {
97 FreelistHead freelist = freelist_head_.load(std::memory_order_acquire);
98 if (IsRetryMarker(freelist)) {
99 // The retry marker will only be stored for a short amount of time. We can
100 // check for it in a busy loop.
101 continue;
102 }
103 return freelist;
104 }
105}

107WasmCodePointer WasmCodePointerTable::AllocateUninitializedEntry() {
108 DCHECK(is_initialized());
109
110 while (true) {
111 // Fast path, try to take an entry from the freelist.
112 uint32_t allocated_entry;
113 if (TryAllocateFromFreelist(&allocated_entry)) {
114 return WasmCodePointer{allocated_entry};
115 }
116
117 // This is essentially DCLP (see
118 // https://preshing.com/20130930/double-checked-locking-is-fixed-in-cpp11/)
119 // and so requires an acquire load as well as a release store in
120 // AllocateTableSegment() to prevent reordering of memory accesses, which
121 // could for example cause one thread to read a freelist entry before it
122 // has been properly initialized.
123
124 // The freelist is empty. We take a lock to avoid another thread from
125 // allocating a new segment in the meantime. However, the freelist can
126 // still grow if another thread frees an entry, so we'll merge the
127 // freelists atomically in the end.
128 base::SpinningMutexGuard guard(&segment_allocation_mutex_);
129
130 // Reload freelist head in case another thread already grew the table.
131 if (!freelist_head_.load(std::memory_order_relaxed).is_empty()) {
132 // Something changed, retry.
133 continue;
134 }
135
136 // Freelist is (still) empty so extend this space by another segment.
137 auto [segment, freelist] = AllocateAndInitializeSegment();
138
139 // Take out the first entry before we link it to the freelist_head.
140 allocated_entry = AllocateEntryFromFreelistNonAtomic(&freelist);
141
142 // Merge the new freelist entries into our freelist.
143 LinkFreelist(freelist, segment.last_entry());
144
145 return WasmCodePointer{allocated_entry};
146 }
147}

149bool WasmCodePointerTable::TryAllocateFromFreelist(uint32_t* index) {
150 while (true) {
151 FreelistHead current_freelist_head = ReadFreelistHead();
152 if (current_freelist_head.is_empty()) {
153 return false;
154 }
155
156 // Temporarily replace the freelist head with a marker to gain exclusive
157 // access to it. This avoids a race condition where another thread could
158 // unmap the memory while we're trying to read from it.
159 if (!freelist_head_.compare_exchange_strong(current_freelist_head,
160 kRetryMarker)) {
161 continue;
162 }
163
164 uint32_t next_freelist_entry =
165 at(current_freelist_head.next()).GetNextFreelistEntryIndex();
166 FreelistHead new_freelist_head(next_freelist_entry,
167 current_freelist_head.length() - 1);
168
169 // We are allowed to overwrite the freelist_head_ since we stored the
170 // kRetryMarker in there.
171 freelist_head_.store(new_freelist_head, std::memory_order_relaxed);
172
173 *index = current_freelist_head.next();
174
175 return true;
176 }
177}

179uint32_t WasmCodePointerTable::AllocateEntryFromFreelistNonAtomic(
180 FreelistHead* freelist_head) {
181 DCHECK(!freelist_head->is_empty());
182 uint32_t index = freelist_head->next();
183 uint32_t next_next = at(freelist_head->next()).GetNextFreelistEntryIndex();
184 *freelist_head = FreelistHead(next_next, freelist_head->length() - 1);
185 return index;
186}

188void WasmCodePointerTable::FreeEntry(WasmCodePointer entry) {
189 // TODO(sroettger): adding to the inline freelist requires a WriteScope. We
190 // could keep a second fixed size out-of-line freelist to avoid frequent
191 // permission changes here.
192 LinkFreelist(FreelistHead(entry.value(), 1), entry.value());
193}

195WasmCodePointerTable::FreelistHead WasmCodePointerTable::LinkFreelist(
196 FreelistHead freelist_to_link, uint32_t last_element) {
197 DCHECK(!freelist_to_link.is_empty());
198
199 FreelistHead current_head, new_head;
200 do {
201 current_head = ReadFreelistHead();
202 new_head = FreelistHead(freelist_to_link.next(),
203 freelist_to_link.length() + current_head.length());
204
205 WriteScope write_scope("write free list entry");
206 at(last_element).MakeFreelistEntry(current_head.next());
207 // This must be a release store since we previously wrote the freelist
208 // entries in AllocateTableSegment() and we need to prevent the writes from
209 // being reordered past this store. See AllocateEntry() for more details.
210 } while (!freelist_head_.compare_exchange_strong(current_head, new_head,
211 std::memory_order_release));
212
213 return new_head;
214}
