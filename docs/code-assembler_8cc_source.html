167void CodeAssembler::BuiltinCompilationScheduler::CompileCode(
168 Isolate* isolate, std::unique_ptr<TurbofanCompilationJob> job) {
169#ifdef V8_USE_ADDRESS_SANITIZER
170 constexpr size_t kInputZoneBatchSize = 128UL * MB;
171#else // !V8_USE_ADDRESS_SANITIZER
172 constexpr size_t kInputZoneBatchSize = 1536UL * MB;
173#endif // V8_USE_ADDRESS_SANITIZER
174
175 // This must be called from the main thread.
176 DCHECK_EQ(ThreadId::Current(), isolate->thread_id());
177
178 DCHECK(job->compilation_info()->code_kind() == CodeKind::BUILTIN ||
179 job->compilation_info()->code_kind() == CodeKind::BYTECODE_HANDLER);
180
181 CHECK_EQ(CompilationJob::SUCCEEDED, job->PrepareJob(isolate));
182
183 if (current_batch_zone_size_ >= kInputZoneBatchSize) {
184 AwaitAndFinalizeCurrentBatch(isolate);
185 }
186
187 QueueJob(isolate, std::move(job));
188}
189
190void CodeAssembler::BuiltinCompilationScheduler::QueueJob(
191 Isolate* isolate, std::unique_ptr<TurbofanCompilationJob> job) {
192 current_batch_zone_size_ +=
193 job->compilation_info()->zone()->allocation_size();
194 if (v8_flags.concurrent_builtin_generation) {
195 auto* dispatcher = isolate->optimizing_compile_dispatcher();
196 // Spin until we can queue the job.
197 while (!dispatcher->IsQueueAvailable()) {
198 std::this_thread::yield();
199 }
200 dispatcher->QueueForOptimization(job.release());
201 } else {
202 CHECK_EQ(CompilationJob::SUCCEEDED,
203 job->ExecuteJob(isolate->counters()->runtime_call_stats(),
204 isolate->main_thread_local_isolate()));
205 if (!v8_flags.turbo_profiling) {
206 main_thread_output_queue_.push_back(std::move(job));
207 } else {
208 // When profiling builtins for PGO, each builtin must be completely
209 // generated one at a time (i.e. PrepareJob, ExecuteJob, and FinalizeJob)
210 // instead of batched.
211 FinalizeJobOnMainThread(isolate, job.get());
212 }
213 }
214}

216void CodeAssembler::BuiltinCompilationScheduler::FinalizeJobOnMainThread(
217 Isolate* isolate, TurbofanCompilationJob* job) {
218 CHECK_EQ(CompilationJob::SUCCEEDED, job->FinalizeJob(isolate));
219 builtins_installed_count_++;
220}
221
222void CodeAssembler::BuiltinCompilationScheduler::AwaitAndFinalizeCurrentBatch(
223 Isolate* isolate) {
224 if (v8_flags.concurrent_builtin_generation) {
225 auto* dispatcher = isolate->optimizing_compile_dispatcher();
226 dispatcher->AwaitCompileTasks();
227 builtins_installed_count_ =
228 dispatcher->InstallGeneratedBuiltins(builtins_installed_count_);
229 } else {
230 DCHECK_IMPLIES(v8_flags.turbo_profiling, main_thread_output_queue_.empty());
231 while (!main_thread_output_queue_.empty()) {
232 FinalizeJobOnMainThread(isolate, main_thread_output_queue_.front().get());
233 main_thread_output_queue_.pop_front();
234 }
235 }
236 current_batch_zone_size_ = 0;
237}

346void CodeAssembler::CanonicalizeEmbeddedBuiltinsConstantIfNeeded(
347 Handle<HeapObject> object) {
348 // This must be called on the main thread so that the builtins constant
349 // indices are reproducible from run to run of mksnapshot.
350 DCHECK_EQ(ThreadId::Current(), isolate()->thread_id());
351 RootIndex dummy_root;
352 Builtin dummy_builtin;
353 if (isolate()->IsGeneratingEmbeddedBuiltins() &&
354 !isolate()->roots_table().IsRootHandle(object, &dummy_root) &&
355 !isolate()->builtins()->IsBuiltinHandle(object, &dummy_builtin) &&
356 !IsInstructionStream(*object)) {
357 isolate()->builtins_constants_table_builder()->AddObject(object);
358 }

420bool CodeAssembler::TryToInt32Constant(TNode<IntegralT> node,
421 int32_t* out_value) {
422 {
423 Int64Matcher m(node);
424 if (m.HasResolvedValue() &&
425 m.IsInRange(std::numeric_limits<int32_t>::min(),
426 std::numeric_limits<int32_t>::max())) {
427 *out_value = static_cast<int32_t>(m.ResolvedValue());
428 return true;
429 }
430 }
431
432 {
433 Int32Matcher m(node);
434 if (m.HasResolvedValue()) {
435 *out_value = m.ResolvedValue();
436 return true;
437 }
438 }
439
440 return false;
441}

459 Tagged<Smi>* out_value) {
460 IntPtrMatcher m(node);
461 if (m.HasResolvedValue()) {
462 intptr_t value = m.ResolvedValue();
463 // Make sure that the value is actually a smi

1176void CodeAssembler::HandleException(Node* node) {
1177 if (state_->exception_handler_labels_.empty()) return;
1178 CodeAssemblerExceptionHandlerLabel* label =
1179 state_->exception_handler_labels_.back();
1180
1181 if (node->op()->HasProperty(Operator::kNoThrow)) {
1182 return;
1183 }
1184
1185 Label success(this), exception(this, Label::kDeferred);
1186 success.MergeVariables();
1187 exception.MergeVariables();
1188
1189 raw_assembler()->Continuations(node, success.label_, exception.label_);
1190
1191 Bind(&exception);
1192 const Operator* op = raw_assembler()->common()->IfException();
1193 Node* exception_value = raw_assembler()->AddNode(op, node, node);
1194 label->AddInputs({UncheckedCast<Object>(exception_value)});
1195 Goto(label->plain_label());
1196
1197 Bind(&success);
1198 raw_assembler()->AddNode(raw_assembler()->common()->IfSuccess(), node);
1199}

1587void CodeAssembler::Branch(TNode<IntegralT> condition, Label* true_label,
1588 Label* false_label, BranchHint branch_hint) {
1589 int32_t constant;
1590 if (TryToInt32Constant(condition, &constant)) {
1591 if ((true_label->is_used() || true_label->is_bound()) &&
1592 (false_label->is_used() || false_label->is_bound())) {
1593 return Goto(constant ? true_label : false_label);
1594 }
1595 }
1596 true_label->MergeVariables();
1597 false_label->MergeVariables();
1598 return raw_assembler()->Branch(condition, true_label->label_,
1599 false_label->label_, branch_hint);

1821void CodeAssemblerLabel::MergeVariables() {
1822 ++merge_count_;
1823 for (CodeAssemblerVariable::Impl* var : state_->variables_) {
1824 size_t count = 0;
1825 Node* node = var->value_;
1826 if (node != nullptr) {
1827 auto i = variable_merges_.find(var);
1828 if (i != variable_merges_.end()) {
1829 i->second.push_back(node);
1830 count = i->second.size();
1831 } else {
1832 count = 1;
1833 variable_merges_[var] = std::vector<Node*>(1, node);
1834 }
1835 }
1836 // If the following asserts, then you've jumped to a label without a bound
1837 // variable along that path that expects to merge its value into a phi.
1838 // This can also occur if a label is bound that is never jumped to.
1839 DCHECK(variable_phis_.find(var) == variable_phis_.end() ||
1840 count == merge_count_);
1841 USE(count);
1842
1843 // If the label is already bound, we already know the set of variables to
1844 // merge and phi nodes have already been created.
1845 if (bound_) {
1846 auto phi = variable_phis_.find(var);
1847 if (phi != variable_phis_.end()) {
1848 DCHECK_NOT_NULL(phi->second);
1849 state_->raw_assembler_->AppendPhiInput(phi->second, node);
1850 } else {
1851 auto i = variable_merges_.find(var);
1852 if (i != variable_merges_.end()) {
1853 // If the following assert fires, then you've declared a variable that
1854 // has the same bound value along all paths up until the point you
1855 // bound this label, but then later merged a path with a new value for
1856 // the variable after the label bind (it's not possible to add phis to
1857 // the bound label after the fact, just make sure to list the variable
1858 // in the label's constructor's list of merged variables).
1859#if DEBUG
1860 if (find_if(i->second.begin(), i->second.end(),
1861 [node](Node* e) -> bool { return node != e; }) !=
1862 i->second.end()) {
1863 std::stringstream str;
1864 str << "Unmerged variable found when jumping to block. \n"
1865 << "# Variable: " << *var;
1866 if (bound_) {
1867 str << "\n# Target block: " << *label_->block();
1868 }
1869 str << "\n# Current Block: ";
1870 state_->PrintCurrentBlock(str);
1871 FATAL("%s", str.str().c_str());
1872 }
1873#endif // DEBUG
1874 }
1875 }
1876 }
1877 }

1904void CodeAssemblerLabel::UpdateVariablesAfterBind() {
1905 // Make sure that all variables that have changed along any path up to this
1906 // point are marked as merge variables.
1907 for (auto var : state_->variables_) {
1908 Node* shared_value = nullptr;
1909 auto i = variable_merges_.find(var);
1910 if (i != variable_merges_.end()) {
1911 for (auto value : i->second) {
1912 DCHECK_NOT_NULL(value);
1913 if (value != shared_value) {
1914 if (shared_value == nullptr) {
1915 shared_value = value;
1916 } else {
1917 variable_phis_[var] = nullptr;
1918 }
1919 }
1920 }
1921 }
1922 }
1923
1924 for (auto var : variable_phis_) {
1925 CodeAssemblerVariable::Impl* var_impl = var.first;
1926 auto i = variable_merges_.find(var_impl);
1927#if DEBUG
1928 bool not_found = i == variable_merges_.end();
1929 if (not_found || i->second.size() != merge_count_) {
1930 std::stringstream str;
1931 str << "A variable that has been marked as beeing merged at the label"
1932 << "\n# doesn't have a bound value along all of the paths that "
1933 << "\n# have been merged into the label up to this point."
1934 << "\n#"
1935 << "\n# This can happen in the following cases:"
1936 << "\n# - By explicitly marking it so in the label constructor"
1937 << "\n# - By having seen different bound values at branches"
1938 << "\n#"
1939 << "\n# Merge count: expected=" << merge_count_
1940 << " vs. found=" << (not_found ? 0 : i->second.size())
1941 << "\n# Variable: " << *var_impl
1942 << "\n# Current Block: " << *label_->block();
1943 FATAL("%s", str.str().c_str());
1944 }
1945#endif // DEBUG
1946 Node* phi = state_->raw_assembler_->Phi(
1947 var.first->rep_, static_cast<int>(merge_count_), &(i->second[0]));
1948 variable_phis_[var_impl] = phi;
1949 }
1950
1951 // Bind all variables to a merge phi, the common value along all paths or
1952 // null.
1953 for (auto var : state_->variables_) {
1954 auto i = variable_phis_.find(var);
1955 if (i != variable_phis_.end()) {
1956 var->value_ = i->second;
1957 } else {
1958 auto j = variable_merges_.find(var);
1959 if (j != variable_merges_.end() && j->second.size() == merge_count_) {
1960 var->value_ = j->second.back();
1961 } else {
1962 var->value_ = nullptr;
1963 }
1964 }
1965 }
1966
1967 bound_ = true;
1968}
