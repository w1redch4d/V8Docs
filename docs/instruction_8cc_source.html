92bool InstructionOperand::InterferesWith(const InstructionOperand& other) const {
93 const bool combine_fp_aliasing = kFPAliasing == AliasingKind::kCombine &&
94 this->IsFPLocationOperand() &&
95 other.IsFPLocationOperand();
96 const bool stack_slots = this->IsAnyStackSlot() && other.IsAnyStackSlot();
97 if (!combine_fp_aliasing && !stack_slots) {
98 return EqualsCanonicalized(other);
99 }
100 const LocationOperand& loc = *LocationOperand::cast(this);
101 const LocationOperand& other_loc = LocationOperand::cast(other);
102 MachineRepresentation rep = loc.representation();
103 MachineRepresentation other_rep = other_loc.representation();
104 LocationOperand::LocationKind kind = loc.location_kind();
105 LocationOperand::LocationKind other_kind = other_loc.location_kind();
106 if (kind != other_kind) return false;
107
108 if (combine_fp_aliasing && !stack_slots) {
109 if (rep == other_rep) return EqualsCanonicalized(other);
110 DCHECK_EQ(kind, LocationOperand::REGISTER);
111 // FP register-register interference.
112 return GetRegConfig()->AreAliases(rep, loc.register_code(), other_rep,
113 other_loc.register_code());
114 }
115
116 DCHECK(stack_slots);
117 int num_slots =
118 AlignedSlotAllocator::NumSlotsForWidth(ElementSizeInBytes(rep));
119 int num_slots_other =
120 AlignedSlotAllocator::NumSlotsForWidth(ElementSizeInBytes(other_rep));
121 const bool complex_stack_slot_interference =
122 (num_slots > 1 || num_slots_other > 1);
123 if (!complex_stack_slot_interference) {
124 return EqualsCanonicalized(other);
125 }
126
127 // Complex multi-slot operand interference:
128 // - slots of different FP reps can alias because the gap resolver may break a
129 // move into 2 or 4 equivalent smaller moves,
130 // - stack layout can be rearranged for tail calls
131 DCHECK_EQ(LocationOperand::STACK_SLOT, kind);
132 int index_hi = loc.index();
133 int index_lo =
134 index_hi -
135 AlignedSlotAllocator::NumSlotsForWidth(ElementSizeInBytes(rep)) + 1;
136 int other_index_hi = other_loc.index();
137 int other_index_lo =
138 other_index_hi -
139 AlignedSlotAllocator::NumSlotsForWidth(ElementSizeInBytes(other_rep)) + 1;
140 return other_index_hi >= index_lo && index_hi >= other_index_lo;
141}

143bool LocationOperand::IsCompatible(LocationOperand* op) {
144 if (IsRegister() || IsStackSlot()) {
145 return op->IsRegister() || op->IsStackSlot();
146 } else if (kFPAliasing != AliasingKind::kCombine) {
147 // A backend may choose to generate the same instruction sequence regardless
148 // of the FP representation. As a result, we can relax the compatibility and
149 // allow a Double to be moved in a Float for example. However, this is only
150 // allowed if registers do not overlap.
151 return (IsFPRegister() || IsFPStackSlot()) &&
152 (op->IsFPRegister() || op->IsFPStackSlot());
153 } else if (IsFloatRegister() || IsFloatStackSlot()) {
154 return op->IsFloatRegister() || op->IsFloatStackSlot();
155 } else if (IsDoubleRegister() || IsDoubleStackSlot()) {
156 return op->IsDoubleRegister() || op->IsDoubleStackSlot();
157 } else {
158 return (IsSimd128Register() || IsSimd128StackSlot()) &&
159 (op->IsSimd128Register() || op->IsSimd128StackSlot());
160 }
161}

334void ParallelMove::PrepareInsertAfter(
335 MoveOperands* move, ZoneVector<MoveOperands*>* to_eliminate) const {
336 bool no_aliasing = kFPAliasing != AliasingKind::kCombine ||
337 !move->destination().IsFPLocationOperand();
338 MoveOperands* replacement = nullptr;
339 MoveOperands* eliminated = nullptr;
340 for (MoveOperands* curr : *this) {
341 if (curr->IsEliminated()) continue;
342 if (curr->destination().EqualsCanonicalized(move->source())) {
343 // We must replace move's source with curr's destination in order to
344 // insert it into this ParallelMove.
345 DCHECK(!replacement);
346 replacement = curr;
347 if (no_aliasing && eliminated != nullptr) break;
348 } else if (curr->destination().InterferesWith(move->destination())) {
349 // We can eliminate curr, since move overwrites at least a part of its
350 // destination, implying its value is no longer live.
351 eliminated = curr;
352 to_eliminate->push_back(curr);
353 if (no_aliasing && replacement != nullptr) break;
354 }
355 }
356 if (replacement != nullptr) move->set_source(replacement->source());
357}

707static RpoNumber GetLoopEndRpo(const turboshaft::Block* block) {
708 if (!block->IsLoop()) return RpoNumber::Invalid();
709 // In Turbofan, the `block->loop_end()` refers to the first after (outside)
710 // the loop. In the relevant use cases, we retrieve the backedge block by
711 // subtracting one from the rpo_number, so for Turboshaft we "fake" this by
712 // adding 1 to the backedge block's rpo_number.
713 return RpoNumber::FromInt(GetRpo(block->LastPredecessor()).ToInt() + 1);
714}

716static InstructionBlock* InstructionBlockFor(Zone* zone,
717 const BasicBlock* block) {
718 bool is_handler =
719 !block->empty() && block->front()->opcode() == IrOpcode::kIfException;
720 InstructionBlock* instr_block = zone->New<InstructionBlock>(
721 zone, GetRpo(block), GetRpo(block->loop_header()), GetLoopEndRpo(block),
722 GetRpo(block->dominator()), block->deferred(), is_handler);
723 // Map successors and precessors
724 instr_block->successors().reserve(block->SuccessorCount());
725 for (BasicBlock* successor : block->successors()) {
726 instr_block->successors().push_back(GetRpo(successor));
727 }
728 instr_block->predecessors().reserve(block->PredecessorCount());
729 for (BasicBlock* predecessor : block->predecessors()) {
730 instr_block->predecessors().push_back(GetRpo(predecessor));
731 }
732 if (block->PredecessorCount() == 1 &&
733 block->predecessors()[0]->control() == BasicBlock::Control::kSwitch) {
734 instr_block->set_switch_target(true);
735 }
736 return instr_block;
737}
738
739static InstructionBlock* InstructionBlockFor(
740 Zone* zone, const turboshaft::Graph& graph, const turboshaft::Block* block,
741 const turboshaft::Block* loop_header) {
742 bool is_handler =
743 block->FirstOperation(graph).Is<turboshaft::CatchBlockBeginOp>();
744 bool deferred = block->get_custom_data(
745 turboshaft::Block::CustomDataKind::kDeferredInSchedule);
746 InstructionBlock* instr_block = zone->New<InstructionBlock>(
747 zone, GetRpo(block), GetRpo(loop_header), GetLoopEndRpo(block),
748 GetRpo(block->GetDominator()), deferred, is_handler);
749 if (block->PredecessorCount() == 1) {
750 const turboshaft::Block* predecessor = block->LastPredecessor();
751 if (V8_UNLIKELY(
752 predecessor->LastOperation(graph).Is<turboshaft::SwitchOp>())) {
753 instr_block->set_switch_target(true);
754 }
755 }
756 // Map successors and predecessors.
757 base::SmallVector<turboshaft::Block*, 4> succs =
758 turboshaft::SuccessorBlocks(block->LastOperation(graph));
759 instr_block->successors().reserve(succs.size());
760 for (const turboshaft::Block* successor : succs) {
761 instr_block->successors().push_back(GetRpo(successor));
762 }
763 instr_block->predecessors().reserve(block->PredecessorCount());
764 for (const turboshaft::Block* predecessor = block->LastPredecessor();
765 predecessor; predecessor = predecessor->NeighboringPredecessor()) {
766 instr_block->predecessors().push_back(GetRpo(predecessor));
767 }
768 std::reverse(instr_block->predecessors().begin(),
769 instr_block->predecessors().end());
770 return instr_block;
771}

859void InstructionSequence::ValidateEdgeSplitForm() const {
860 // Validate blocks are in edge-split form: no block with multiple successors
861 // has an edge to a block (== a successor) with more than one predecessors.
862 for (const InstructionBlock* block : instruction_blocks()) {
863 if (block->SuccessorCount() > 1) {
864 for (const RpoNumber& successor_id : block->successors()) {
865 const InstructionBlock* successor = InstructionBlockAt(successor_id);
866 // Expect precisely one predecessor: "block".
867 CHECK(successor->PredecessorCount() == 1 &&
868 successor->predecessors()[0] == block->rpo_number());
869 }
870 }
871 }
872}

874void InstructionSequence::ValidateDeferredBlockExitPaths() const {
875 // A deferred block with more than one successor must have all its successors
876 // deferred.
877 for (const InstructionBlock* block : instruction_blocks()) {
878 if (!block->IsDeferred() || block->SuccessorCount() <= 1) continue;
879 for (RpoNumber successor_id : block->successors()) {
880 CHECK(InstructionBlockAt(successor_id)->IsDeferred());
881 }
882 }

884
885void InstructionSequence::ValidateDeferredBlockEntryPaths() const {
886 // If a deferred block has multiple predecessors, they have to
887 // all be deferred. Otherwise, we can run into a situation where a range
888 // that spills only in deferred blocks inserts its spill in the block, but
889 // other ranges need moves inserted by ResolveControlFlow in the predecessors,
890 // which may clobber the register of this range.
891 for (const InstructionBlock* block : instruction_blocks()) {
892 if (!block->IsDeferred() || block->PredecessorCount() <= 1) continue;
893 for (RpoNumber predecessor_id : block->predecessors()) {
894 CHECK(InstructionBlockAt(predecessor_id)->IsDeferred());
895 }
896 }

914void InstructionSequence::ComputeAssemblyOrder() {
915 int ao = 0;
916 RpoNumber invalid = RpoNumber::Invalid();
917
918 ao_blocks_ = zone()->AllocateArray<InstructionBlocks>(1);
919 new (ao_blocks_) InstructionBlocks(zone());
920 ao_blocks_->reserve(instruction_blocks_->size());
921
922 // Place non-deferred blocks.
923 for (InstructionBlock* const block : *instruction_blocks_) {
924 DCHECK_NOT_NULL(block);
925 if (block->IsDeferred()) continue; // skip deferred blocks.
926 if (block->ao_number() != invalid) continue; // loop rotated.
927 if (block->IsLoopHeader()) {
928 bool header_align = true;
929 if (v8_flags.turbo_loop_rotation) {
930 // Perform loop rotation for non-deferred loops.
931 InstructionBlock* loop_end =
932 instruction_blocks_->at(block->loop_end().ToSize() - 1);
933 if (loop_end->SuccessorCount() == 1 && /* ends with goto */
934 loop_end != block /* not a degenerate infinite loop */) {
935 // If the last block has an unconditional jump back to the header,
936 // then move it to be in front of the header in the assembly order.
937 DCHECK_EQ(block->rpo_number(), loop_end->successors()[0]);
938 loop_end->set_ao_number(RpoNumber::FromInt(ao++));
939 ao_blocks_->push_back(loop_end);
940 // This block will be the new machine-level loop header, so align
941 // this block instead of the loop header block.
942 loop_end->set_loop_header_alignment(true);
943 header_align = false;
944 }
945 }
946 block->set_loop_header_alignment(header_align);
947 }
948 if (block->loop_header().IsValid() && block->IsSwitchTarget()) {
949 block->set_code_target_alignment(true);
950 }
951 block->set_ao_number(RpoNumber::FromInt(ao++));
952 ao_blocks_->push_back(block);
953 }
954 // Add all leftover (deferred) blocks.
955 for (InstructionBlock* const block : *instruction_blocks_) {
956 if (block->ao_number() == invalid) {
957 block->set_ao_number(RpoNumber::FromInt(ao++));
958 ao_blocks_->push_back(block);
959 }
960 }
961 DCHECK_EQ(instruction_blocks_->size(), ao);
962}

966 for (InstructionBlock* block : *instruction_blocks_) {
967 block->set_ao_number(invalid);
968 }
969 ComputeAssemblyOrder();
