17bool IsMachineLoad(Node* const node) {
18 const IrOpcode::Value opcode = node->opcode();
19 return opcode == IrOpcode::kLoad || opcode == IrOpcode::kProtectedLoad ||
20 opcode == IrOpcode::kLoadTrapOnNull ||
21 opcode == IrOpcode::kUnalignedLoad ||
22 opcode == IrOpcode::kLoadImmutable;
23}
24
25bool IsTaggedMachineLoad(Node* const node) {
26 return IsMachineLoad(node) &&
27 CanBeTaggedPointer(LoadRepresentationOf(node->op()).representation());
28}
29
30bool IsHeapConstant(Node* const node) {
31 return node->opcode() == IrOpcode::kHeapConstant;
32}
33
34bool IsIntConstant(Node* const node) {
35 return node->opcode() == IrOpcode::kInt32Constant ||
36 node->opcode() == IrOpcode::kInt64Constant;
37}
38
39bool IsTaggedPhi(Node* const node) {
40 if (node->opcode() == IrOpcode::kPhi) {
41 return CanBeTaggedPointer(PhiRepresentationOf(node->op()));
42 }
43 return false;
44}
45
46bool IsWord64BitwiseOp(Node* const node) {
47 return node->opcode() == IrOpcode::kWord64And ||
48 node->opcode() == IrOpcode::kWord64Or;
49}
50
51bool CanBeCompressed(Node* const node) {
52 return IsHeapConstant(node) || IsTaggedMachineLoad(node) ||
53 IsTaggedPhi(node) || IsWord64BitwiseOp(node);
54}

56void Replace(Node* const node, Node* const replacement) {
57 for (Edge edge : node->use_edges()) {
58 edge.UpdateTo(replacement);
59 }
60 node->Kill();

84void DecompressionOptimizer::MarkNodeInputs(Node* node) {
85 // Mark the value inputs.
86 switch (node->opcode()) {
87 // UNOPS.
88 case IrOpcode::kBitcastTaggedToWord:
89 case IrOpcode::kBitcastTaggedToWordForTagAndSmiBits:
90 case IrOpcode::kBitcastWordToTagged:
91 // Replicate the bitcast's state for its input.
92 DCHECK_EQ(node->op()->ValueInputCount(), 1);
93 MaybeMarkAndQueueForRevisit(node->InputAt(0),
94 states_.Get(node)); // value
95 break;
96 case IrOpcode::kTruncateInt64ToInt32:
97 DCHECK_EQ(node->op()->ValueInputCount(), 1);
98 MaybeMarkAndQueueForRevisit(node->InputAt(0),
99 State::kOnly32BitsObserved); // value
100 break;
101 // BINOPS.
102 case IrOpcode::kInt32LessThan:
103 case IrOpcode::kInt32LessThanOrEqual:
104 case IrOpcode::kUint32LessThan:
105 case IrOpcode::kUint32LessThanOrEqual:
106 case IrOpcode::kWord32Equal:
107#define Word32Op(Name) case IrOpcode::k##Name:
108 MACHINE_BINOP_32_LIST(Word32Op)
109#undef Word32Op
110 DCHECK_EQ(node->op()->ValueInputCount(), 2);
111 MaybeMarkAndQueueForRevisit(node->InputAt(0),
112 State::kOnly32BitsObserved); // value_0
113 MaybeMarkAndQueueForRevisit(node->InputAt(1),
114 State::kOnly32BitsObserved); // value_1
115 break;
116 // SPECIAL CASES.
117 // SPECIAL CASES - Load.
118 case IrOpcode::kLoad:
119 case IrOpcode::kProtectedLoad:
120 case IrOpcode::kLoadTrapOnNull:
121 case IrOpcode::kUnalignedLoad:
122 case IrOpcode::kLoadImmutable:
123 DCHECK_EQ(node->op()->ValueInputCount(), 2);
124 // Mark addressing base pointer in compressed form to allow pointer
125 // decompression via complex addressing mode.
126 if (DECOMPRESS_POINTER_BY_ADDRESSING_MODE &&
127 node->InputAt(0)->OwnedBy(node) && IsIntConstant(node->InputAt(1))) {
128 MarkAddressingBase(node->InputAt(0));
129 } else {
130 MaybeMarkAndQueueForRevisit(
131 node->InputAt(0),
132 State::kEverythingObserved); // base pointer
133 MaybeMarkAndQueueForRevisit(node->InputAt(1),
134 State::kEverythingObserved); // index
135 }
136 break;
137 // SPECIAL CASES - Store.
138 case IrOpcode::kStore:
139 case IrOpcode::kStorePair:
140 case IrOpcode::kProtectedStore:
141 case IrOpcode::kStoreTrapOnNull:
142 case IrOpcode::kUnalignedStore: {
143 DCHECK(node->op()->ValueInputCount() == 3 ||
144 (node->opcode() == IrOpcode::kStorePair &&
145 node->op()->ValueInputCount() == 4));
146 MaybeMarkAndQueueForRevisit(node->InputAt(0),
147 State::kEverythingObserved); // base pointer
148 MaybeMarkAndQueueForRevisit(node->InputAt(1),
149 State::kEverythingObserved); // index
150 // TODO(v8:7703): When the implementation is done, check if this ternary
151 // operator is too restrictive, since we only mark Tagged stores as 32
152 // bits.
153 MachineRepresentation representation;
154 if (node->opcode() == IrOpcode::kUnalignedStore) {
155 representation = UnalignedStoreRepresentationOf(node->op());
156 } else if (node->opcode() == IrOpcode::kStorePair) {
157 representation =
158 StorePairRepresentationOf(node->op()).first.representation();
159 } else {
160 representation = StoreRepresentationOf(node->op()).representation();
161 }
162 State observed = ElementSizeLog2Of(representation) <= 2
163 ? State::kOnly32BitsObserved
164 : State::kEverythingObserved;
165
166 // We should never see indirect pointer stores here since they need
167 // kStoreIndirect. For indirect pointer stores we always need all pointer
168 // bits since we'll also perform a load (of the 'self' indirect pointer)
169 // from the value being stored.
170 DCHECK_NE(representation, MachineRepresentation::kIndirectPointer);
171
172 MaybeMarkAndQueueForRevisit(node->InputAt(2), observed); // value
173 if (node->opcode() == IrOpcode::kStorePair) {
174 MaybeMarkAndQueueForRevisit(node->InputAt(3), observed); // value 2
175 }
176 } break;
177 // SPECIAL CASES - Variable inputs.
178 // The deopt code knows how to handle Compressed inputs, both
179 // MachineRepresentation kCompressed values and CompressedHeapConstants.
180 case IrOpcode::kFrameState: // Fall through.
181 case IrOpcode::kStateValues:
182 for (int i = 0; i < node->op()->ValueInputCount(); ++i) {
183 // TODO(chromium:1470602): We assume that kStateValues has only tagged
184 // inputs so it is safe to mark them as kOnly32BitsObserved.
185 DCHECK(!IsWord64BitwiseOp(node->InputAt(i)));
186 MaybeMarkAndQueueForRevisit(node->InputAt(i),
187 State::kOnly32BitsObserved);
188 }
189 break;
190 case IrOpcode::kTypedStateValues: {
191 const ZoneVector<MachineType>* machine_types = MachineTypesOf(node->op());
192 for (int i = 0; i < node->op()->ValueInputCount(); ++i) {
193 State observed = IsAnyTagged(machine_types->at(i).representation())
194 ? State::kOnly32BitsObserved
195 : State::kEverythingObserved;
196 MaybeMarkAndQueueForRevisit(node->InputAt(i), observed);
197 }
198 break;
199 }
200 case IrOpcode::kPhi: {
201 // Replicate the phi's state for its inputs.
202 State curr_state = states_.Get(node);
203 for (int i = 0; i < node->op()->ValueInputCount(); ++i) {
204 MaybeMarkAndQueueForRevisit(node->InputAt(i), curr_state);
205 }
206 break;
207 }
208 default:
209 // To be conservative, we assume that all value inputs need to be 64 bits
210 // unless noted otherwise.
211 for (int i = 0; i < node->op()->ValueInputCount(); ++i) {
212 MaybeMarkAndQueueForRevisit(node->InputAt(i),
213 State::kEverythingObserved);
214 }
215 break;
216 }
217
218 // We always mark the non-value input nodes as kOnly32BitsObserved so that
219 // they will be visited. If they need to be kEverythingObserved, they will be
220 // marked as such in a future pass.
221 for (int i = node->op()->ValueInputCount(); i < node->InputCount(); ++i) {
222 MaybeMarkAndQueueForRevisit(node->InputAt(i), State::kOnly32BitsObserved);
223 }
224}

226// We mark the addressing base pointer as kOnly32BitsObserved so it can be
227// optimized to compressed form. This allows us to move the decompression to
228// use-site on X64.
229void DecompressionOptimizer::MarkAddressingBase(Node* base) {
230 if (IsTaggedMachineLoad(base)) {
231 MaybeMarkAndQueueForRevisit(base,
232 State::kOnly32BitsObserved); // base pointer
233 } else if (IsTaggedPhi(base)) {
234 bool should_compress = true;
235 for (int i = 0; i < base->op()->ValueInputCount(); ++i) {
236 if (!IsTaggedMachineLoad(base->InputAt(i)) ||
237 !base->InputAt(i)->OwnedBy(base)) {
238 should_compress = false;
239 break;
240 }
241 }
242 MaybeMarkAndQueueForRevisit(
243 base,
244 should_compress ? State::kOnly32BitsObserved
245 : State::kEverythingObserved); // base pointer
246 } else {
247 MaybeMarkAndQueueForRevisit(base,
248 State::kEverythingObserved); // base pointer
249 }
250}

252void DecompressionOptimizer::MaybeMarkAndQueueForRevisit(Node* const node,
253 State state) {
254 DCHECK_NE(state, State::kUnvisited);
255 State previous_state = states_.Get(node);
256 // Only update the state if we have relevant new information.
257 if (previous_state == State::kUnvisited ||
258 (previous_state == State::kOnly32BitsObserved &&
259 state == State::kEverythingObserved)) {
260 states_.Set(node, state);
261 to_visit_.push_back(node);
262
263 if (state == State::kOnly32BitsObserved && CanBeCompressed(node)) {
264 compressed_candidate_nodes_.push_back(node);
265 }
266 }

275void DecompressionOptimizer::ChangePhi(Node* const node) {
276 DCHECK(IsTaggedPhi(node));
277
278 MachineRepresentation mach_rep = PhiRepresentationOf(node->op());
279 if (mach_rep == MachineRepresentation::kTagged) {
280 mach_rep = MachineRepresentation::kCompressed;
281 } else {
282 DCHECK_EQ(mach_rep, MachineRepresentation::kTaggedPointer);
283 mach_rep = MachineRepresentation::kCompressedPointer;
284 }
285
286 NodeProperties::ChangeOp(
287 node, common()->Phi(mach_rep, node->op()->ValueInputCount()));
288}

290void DecompressionOptimizer::ChangeLoad(Node* const node) {
291 DCHECK(IsMachineLoad(node));
292 // Change to a Compressed MachRep to avoid the full decompression.
293 LoadRepresentation load_rep = LoadRepresentationOf(node->op());
294 LoadRepresentation compressed_load_rep;
295 if (load_rep == MachineType::AnyTagged()) {
296 compressed_load_rep = MachineType::AnyCompressed();
297 } else {
298 DCHECK_EQ(load_rep, MachineType::TaggedPointer());
299 compressed_load_rep = MachineType::CompressedPointer();
300 }
301
302 // Change to the Operator with the Compressed MachineRepresentation.
303 switch (node->opcode()) {
304 case IrOpcode::kLoad:
305 NodeProperties::ChangeOp(node, machine()->Load(compressed_load_rep));
306 break;
307 case IrOpcode::kLoadImmutable:
308 NodeProperties::ChangeOp(node,
309 machine()->LoadImmutable(compressed_load_rep));
310 break;
311 case IrOpcode::kProtectedLoad:
312 NodeProperties::ChangeOp(node,
313 machine()->ProtectedLoad(compressed_load_rep));
314 break;
315 case IrOpcode::kLoadTrapOnNull:
316 NodeProperties::ChangeOp(node,
317 machine()->LoadTrapOnNull(compressed_load_rep));
318 break;
319 case IrOpcode::kUnalignedLoad:
320 NodeProperties::ChangeOp(node,
321 machine()->UnalignedLoad(compressed_load_rep));
322 break;
323 default:
324 UNREACHABLE();
325 }

328void DecompressionOptimizer::ChangeWord64BitwiseOp(Node* const node,
329 const Operator* new_op) {
330 Int64Matcher mleft(node->InputAt(0));
331 Int64Matcher mright(node->InputAt(1));
332
333 // Replace inputs.
334 if (mleft.IsChangeInt32ToInt64() || mleft.IsChangeUint32ToUint64()) {
335 node->ReplaceInput(0, mleft.node()->InputAt(0));
336 } else if (mleft.IsInt64Constant()) {
337 node->ReplaceInput(0, graph()->NewNode(common()->Int32Constant(
338 static_cast<int32_t>(mleft.ResolvedValue()))));
339 } else {
340 node->ReplaceInput(
341 0, graph()->NewNode(machine()->TruncateInt64ToInt32(), mleft.node()));
342 }
343 if (mright.IsChangeInt32ToInt64() || mright.IsChangeUint32ToUint64()) {
344 node->ReplaceInput(1, mright.node()->InputAt(0));
345 } else if (mright.IsInt64Constant()) {
346 node->ReplaceInput(1, graph()->NewNode(common()->Int32Constant(
347 static_cast<int32_t>(mright.ResolvedValue()))));
348 } else {
349 node->ReplaceInput(
350 1, graph()->NewNode(machine()->TruncateInt64ToInt32(), mright.node()));
351 }
352
353 // Replace uses.
354 Node* replacement = nullptr;
355 for (Edge edge : node->use_edges()) {
356 Node* user = edge.from();
357 if (user->opcode() == IrOpcode::kTruncateInt64ToInt32) {
358 Replace(user, node);
359 } else {
360 if (replacement == nullptr) {
361 replacement =
362 graph()->NewNode(machine()->BitcastWord32ToWord64(), node);
363 }
364 edge.UpdateTo(replacement);
365 }
366 }
367
368 // Change operator.
369 NodeProperties::ChangeOp(node, new_op);
370}

372void DecompressionOptimizer::ChangeNodes() {
373 for (Node* const node : compressed_candidate_nodes_) {
374 // compressed_candidate_nodes_ contains all the nodes that once had the
375 // State::kOnly32BitsObserved. If we later updated the state to be
376 // State::IsEverythingObserved, then we have to ignore them. This is less
377 // costly than removing them from the compressed_candidate_nodes_ NodeVector
378 // when we update them to State::IsEverythingObserved.
379 if (IsEverythingObserved(node)) continue;
380
381 switch (node->opcode()) {
382 case IrOpcode::kHeapConstant:
383 ChangeHeapConstant(node);
384 break;
385 case IrOpcode::kPhi:
386 ChangePhi(node);
387 break;
388 case IrOpcode::kWord64And:
389 ChangeWord64BitwiseOp(node, machine()->Word32And());
390 break;
391 case IrOpcode::kWord64Or:
392 ChangeWord64BitwiseOp(node, machine()->Word32Or());
393 break;
394 default:
395 ChangeLoad(node);
396 break;
397 }
398 }
