124MemOperand MacroAssembler::ExternalReferenceAsOperand(
125 ExternalReference reference, Register scratch) {
126 if (root_array_available()) {
127 if (reference.IsIsolateFieldId()) {
128 return MemOperand(kRootRegister, reference.offset_from_root_register());
129 }
130 if (options().enable_root_relative_access) {
131 intptr_t offset =
132 RootRegisterOffsetForExternalReference(isolate(), reference);
133 if (is_int32(offset)) {
134 return MemOperand(kRootRegister, static_cast<int32_t>(offset));
135 }
136 }
137 if (options().isolate_independent_code) {
138 if (IsAddressableThroughRootRegister(isolate(), reference)) {
139 // Some external references can be efficiently loaded as an offset from
140 // kRootRegister.
141 intptr_t offset =
142 RootRegisterOffsetForExternalReference(isolate(), reference);
143 CHECK(is_int32(offset));
144 return MemOperand(kRootRegister, static_cast<int32_t>(offset));
145 } else {
146 // Otherwise, do a memory load from the external reference table.
147 ldr(scratch,
148 MemOperand(kRootRegister,
149 RootRegisterOffsetForExternalReferenceTableEntry(
150 isolate(), reference)));
151 return MemOperand(scratch, 0);
152 }
153 }
154 }
155 Move(scratch, reference);
156 return MemOperand(scratch, 0);
157}

159void MacroAssembler::GetLabelAddress(Register dest, Label* target) {
160 // This should be just a
161 // add(dest, pc, branch_offset(target));
162 // but current implementation of Assembler::bind_to()/target_at_put() add
163 // (InstructionStream::kHeaderSize - kHeapObjectTag) to a position of a label
164 // in a "linked" state and thus making it usable only for mov_label_offset().
165 // TODO(ishell): fix branch_offset() and re-implement
166 // RegExpMacroAssemblerARM::PushBacktrack() without mov_label_offset().
167 mov_label_offset(dest, target);
168 // mov_label_offset computes offset of the |target| relative to the "current
169 // InstructionStream object pointer" which is essentally pc_offset() of the
170 // label added with (InstructionStream::kHeaderSize - kHeapObjectTag).
171 // Compute "current InstructionStream object pointer" and add it to the
172 // offset in |lr| register.
173 int current_instr_code_object_relative_offset =
174 pc_offset() + Instruction::kPcLoadDelta +
175 (InstructionStream::kHeaderSize - kHeapObjectTag);
176 add(dest, pc, dest);
177 sub(dest, dest, Operand(current_instr_code_object_relative_offset));
178}

222void MacroAssembler::Call(Address target, RelocInfo::Mode rmode, Condition cond,
223 TargetAddressStorageMode mode,
224 bool check_constant_pool) {
225 // Check if we have to emit the constant pool before we block it.
226 if (check_constant_pool) MaybeCheckConstPool();
227 // Block constant pool for the call instruction sequence.
228 BlockConstPoolScope block_const_pool(this);
229
230 bool old_predictable_code_size = predictable_code_size();
231 if (mode == NEVER_INLINE_TARGET_ADDRESS) {
232 set_predictable_code_size(true);
233 }
234
235 // Use ip directly instead of using UseScratchRegisterScope, as we do not
236 // preserve scratch registers across calls.
237
238 // Call sequence on V7 or later may be :
239 // movw ip, #... @ call address low 16
240 // movt ip, #... @ call address high 16
241 // blx ip
242 // @ return address
243 // Or for pre-V7 or values that may be back-patched
244 // to avoid ICache flushes:
245 // ldr ip, [pc, #...] @ call address
246 // blx ip
247 // @ return address
248
249 mov(ip, Operand(target, rmode));
250 blx(ip, cond);
251
252 if (mode == NEVER_INLINE_TARGET_ADDRESS) {
253 set_predictable_code_size(old_predictable_code_size);
254 }
255}

308void MacroAssembler::CallBuiltin(Builtin builtin, Condition cond) {
309 ASM_CODE_COMMENT_STRING(this, CommentForOffHeapTrampoline("call", builtin));
310 // Use ip directly instead of using UseScratchRegisterScope, as we do not
311 // preserve scratch registers across calls.
312 switch (options().builtin_call_jump_mode) {
313 case BuiltinCallJumpMode::kAbsolute: {
314 mov(ip, Operand(BuiltinEntry(builtin), RelocInfo::OFF_HEAP_TARGET));
315 Call(ip, cond);
316 break;
317 }
318 case BuiltinCallJumpMode::kPCRelative:
319 UNREACHABLE();
320 case BuiltinCallJumpMode::kIndirect:
321 ldr(ip, EntryFromBuiltinAsOperand(builtin));
322 Call(ip, cond);
323 break;
324 case BuiltinCallJumpMode::kForMksnapshot: {
325 if (options().use_pc_relative_calls_and_jumps_for_mksnapshot) {
326 Handle<Code> code = isolate()->builtins()->code_handle(builtin);
327 int32_t code_target_index = AddCodeTarget(code);
328 bl(code_target_index * kInstrSize, cond,
329 RelocInfo::RELATIVE_CODE_TARGET);
330 } else {
331 ldr(ip, EntryFromBuiltinAsOperand(builtin));
332 Call(ip, cond);
333 }
334 break;
335 }
336 }
337}

488void MacroAssembler::StoreReturnAddressAndCall(Register target) {
489 ASM_CODE_COMMENT(this);
490 // This generates the final instruction sequence for calls to C functions
491 // once an exit frame has been constructed.
492 //
493 // Note that this assumes the caller code (i.e. the InstructionStream object
494 // currently being generated) is immovable or that the callee function cannot
495 // trigger GC, since the callee function will return to it.
496
497 // Compute the return address in lr to return to after the jump below. The pc
498 // is already at '+ 8' from the current instruction; but return is after three
499 // instructions, so add another 4 to pc to get the return address.
500 Assembler::BlockConstPoolScope block_const_pool(this);
501 add(lr, pc, Operand(4));
502 str(lr, MemOperand(sp));
503 Call(target);
504}

558void MacroAssembler::PushArray(Register array, Register size, Register scratch,
559 PushArrayOrder order) {
560 ASM_CODE_COMMENT(this);
561 UseScratchRegisterScope temps(this);
562 Register counter = scratch;

564 DCHECK(!AreAliased(array, size, counter, tmp));
565 Label loop, entry;

567 mov(counter, Operand(0));

904void MacroAssembler::MoveObjectAndSlot(Register dst_object, Register dst_slot,
905 Register object, Operand offset) {
906 DCHECK_NE(dst_object, dst_slot);
907 DCHECK(offset.IsRegister() || offset.IsImmediate());
908 // If `offset` is a register, it cannot overlap with `object`.
909 DCHECK_IMPLIES(offset.IsRegister(), offset.rm() != object);
910
911 // If the slot register does not overlap with the object register, we can
912 // overwrite it.
913 if (dst_slot != object) {
914 add(dst_slot, object, offset);
915 Move(dst_object, object);
916 return;
917 }
918
919 DCHECK_EQ(dst_slot, object);
920
921 // If the destination object register does not overlap with the offset
922 // register, we can overwrite it.
923 if (!offset.IsRegister() || (offset.rm() != dst_object)) {
924 Move(dst_object, dst_slot);
925 add(dst_slot, dst_slot, offset);
926 return;
927 }
928
929 DCHECK_EQ(dst_object, offset.rm());
930
931 // We only have `dst_slot` and `dst_object` left as distinct registers so we
932 // have to swap them. We write this as a add+sub sequence to avoid using a
933 // scratch register.
934 add(dst_slot, dst_slot, dst_object);
935 sub(dst_object, dst_slot, dst_object);
936}

938// The register 'object' contains a heap object pointer. The heap object tag is
939// shifted away. A scratch register also needs to be available.
940void MacroAssembler::RecordWrite(Register object, Operand offset,
941 Register value, LinkRegisterStatus lr_status,
942 SaveFPRegsMode fp_mode, SmiCheck smi_check) {
943 DCHECK(!AreAliased(object, value));
944 if (v8_flags.slow_debug_code) {
945 ASM_CODE_COMMENT_STRING(this, "Verify slot_address");
946 UseScratchRegisterScope temps(this);
947 Register scratch = temps.Acquire();
948 DCHECK(!AreAliased(object, value, scratch));
949 add(scratch, object, offset);
950 ldr(scratch, MemOperand(scratch));
951 cmp(scratch, value);
952 Check(eq, AbortReason::kWrongAddressOrValuePassedToRecordWrite);
953 }
954
955 if (v8_flags.disable_write_barriers) {
956 return;
957 }
958
959 // First, check if a write barrier is even needed. The tests below
960 // catch stores of smis and stores into the young generation.
961 Label done;
962
963 if (smi_check == SmiCheck::kInline) {
964 JumpIfSmi(value, &done);
965 }
966
967 CheckPageFlag(value, MemoryChunk::kPointersToHereAreInterestingMask, eq,
968 &done);
969 CheckPageFlag(object, MemoryChunk::kPointersFromHereAreInterestingMask, eq,
970 &done);
971
972 // Record the actual write.
973 if (lr_status == kLRHasNotBeenSaved) {
974 push(lr);
975 }
976
977 Register slot_address = WriteBarrierDescriptor::SlotAddressRegister();
978 DCHECK(!AreAliased(object, value, slot_address));
979 DCHECK(!offset.IsRegister());
980 add(slot_address, object, offset);
981 CallRecordWriteStub(object, slot_address, fp_mode);
982 if (lr_status == kLRHasNotBeenSaved) {
983 pop(lr);
984 }
985
986 if (v8_flags.slow_debug_code) Move(slot_address, Operand(kZapValue));
987
988 bind(&done);
989}

1147void MacroAssembler::VmovExtended(int dst_code, int src_code) {
1148 if (src_code == dst_code) return;
1149
1150 if (src_code < SwVfpRegister::kNumRegisters &&
1151 dst_code < SwVfpRegister::kNumRegisters) {
1152 // src and dst are both s-registers.
1153 vmov(SwVfpRegister::from_code(dst_code),
1154 SwVfpRegister::from_code(src_code));
1155 return;
1156 }
1157 DwVfpRegister dst_d_reg = DwVfpRegister::from_code(dst_code / 2);
1158 DwVfpRegister src_d_reg = DwVfpRegister::from_code(src_code / 2);
1159 int dst_offset = dst_code & 1;
1160 int src_offset = src_code & 1;
1161 if (CpuFeatures::IsSupported(NEON)) {
1162 UseScratchRegisterScope temps(this);
1163 DwVfpRegister scratch = temps.AcquireD();
1164 // On Neon we can shift and insert from d-registers.
1165 if (src_offset == dst_offset) {
1166 // Offsets are the same, use vdup to copy the source to the opposite lane.
1167 vdup(Neon32, scratch, src_d_reg, src_offset);
1168 // Here we are extending the lifetime of scratch.
1169 src_d_reg = scratch;
1170 src_offset = dst_offset ^ 1;
1171 }
1172 if (dst_offset) {
1173 if (dst_d_reg == src_d_reg) {
1174 vdup(Neon32, dst_d_reg, src_d_reg, 0);
1175 } else {
1176 vsli(Neon64, dst_d_reg, src_d_reg, 32);
1177 }
1178 } else {
1179 if (dst_d_reg == src_d_reg) {
1180 vdup(Neon32, dst_d_reg, src_d_reg, 1);
1181 } else {
1182 vsri(Neon64, dst_d_reg, src_d_reg, 32);
1183 }
1184 }
1185 return;
1186 }
1187
1188 // Without Neon, use the scratch registers to move src and/or dst into
1189 // s-registers.
1190 UseScratchRegisterScope temps(this);
1191 LowDwVfpRegister d_scratch = temps.AcquireLowD();
1192 LowDwVfpRegister d_scratch2 = temps.AcquireLowD();
1193 int s_scratch_code = d_scratch.low().code();
1194 int s_scratch_code2 = d_scratch2.low().code();
1195 if (src_code < SwVfpRegister::kNumRegisters) {
1196 // src is an s-register, dst is not.
1197 vmov(d_scratch, dst_d_reg);
1198 vmov(SwVfpRegister::from_code(s_scratch_code + dst_offset),
1199 SwVfpRegister::from_code(src_code));
1200 vmov(dst_d_reg, d_scratch);
1201 } else if (dst_code < SwVfpRegister::kNumRegisters) {
1202 // dst is an s-register, src is not.
1203 vmov(d_scratch, src_d_reg);
1204 vmov(SwVfpRegister::from_code(dst_code),
1205 SwVfpRegister::from_code(s_scratch_code + src_offset));
1206 } else {
1207 // Neither src or dst are s-registers. Both scratch double registers are
1208 // available when there are 32 VFP registers.
1209 vmov(d_scratch, src_d_reg);
1210 vmov(d_scratch2, dst_d_reg);
1211 vmov(SwVfpRegister::from_code(s_scratch_code + dst_offset),
1212 SwVfpRegister::from_code(s_scratch_code2 + src_offset));
1213 vmov(dst_d_reg, d_scratch2);
1214 }
1215}

1520#ifdef V8_OS_WIN
1521void MacroAssembler::AllocateStackSpace(Register bytes_scratch) {
1522 // "Functions that allocate 4 KB or more on the stack must ensure that each
1523 // page prior to the final page is touched in order." Source:
1524 // https://docs.microsoft.com/en-us/cpp/build/overview-of-arm-abi-conventions?view=vs-2019#stack
1525 ASM_CODE_COMMENT(this);
1526 UseScratchRegisterScope temps(this);
1527 DwVfpRegister scratch = temps.AcquireD();
1528 Label check_offset;
1529 Label touch_next_page;
1530 jmp(&check_offset);
1531 bind(&touch_next_page);
1532 sub(sp, sp, Operand(kStackPageSize));
1533 // Just to touch the page, before we increment further.
1534 vldr(scratch, MemOperand(sp));
1535 sub(bytes_scratch, bytes_scratch, Operand(kStackPageSize));
1536
1537 bind(&check_offset);
1538 cmp(bytes_scratch, Operand(kStackPageSize));
1539 b(gt, &touch_next_page);
1540
1541 sub(sp, sp, bytes_scratch);
1542}

1544void MacroAssembler::AllocateStackSpace(int bytes) {
1545 ASM_CODE_COMMENT(this);
1546 DCHECK_GE(bytes, 0);
1547 UseScratchRegisterScope temps(this);
1548 DwVfpRegister scratch = no_dreg;
1549 while (bytes > kStackPageSize) {
1550 if (scratch == no_dreg) {
1551 scratch = temps.AcquireD();
1552 }
1553 sub(sp, sp, Operand(kStackPageSize));
1554 vldr(scratch, MemOperand(sp));
1555 bytes -= kStackPageSize;
1556 }
1557 if (bytes == 0) return;
1558 sub(sp, sp, Operand(bytes));
1559}

1562void MacroAssembler::EnterExitFrame(Register scratch, int stack_space,
1563 StackFrame::Type frame_type) {
1564 ASM_CODE_COMMENT(this);
1565 DCHECK(frame_type == StackFrame::EXIT ||
1566 frame_type == StackFrame::BUILTIN_EXIT ||
1567 frame_type == StackFrame::API_ACCESSOR_EXIT ||
1568 frame_type == StackFrame::API_CALLBACK_EXIT);
1569
1570 using ER = ExternalReference;
1571
1572 // Set up the frame structure on the stack.
1573 DCHECK_EQ(2 * kPointerSize, ExitFrameConstants::kCallerSPDisplacement);
1574 DCHECK_EQ(1 * kPointerSize, ExitFrameConstants::kCallerPCOffset);
1575 DCHECK_EQ(0 * kPointerSize, ExitFrameConstants::kCallerFPOffset);
1576 mov(scratch, Operand(StackFrame::TypeToMarker(frame_type)));
1577 PushCommonFrame(scratch);
1578 // Reserve room for saved entry sp.
1579 sub(sp, fp, Operand(ExitFrameConstants::kFixedFrameSizeFromFp));
1580 if (v8_flags.debug_code) {
1581 mov(scratch, Operand::Zero());
1582 str(scratch, MemOperand(fp, ExitFrameConstants::kSPOffset));
1583 }
1584
1585 // Save the frame pointer and the context in top.
1586 ER c_entry_fp_address =
1587 ER::Create(IsolateAddressId::kCEntryFPAddress, isolate());
1588 str(fp, ExternalReferenceAsOperand(c_entry_fp_address, no_reg));
1589
1590 ER context_address = ER::Create(IsolateAddressId::kContextAddress, isolate());
1591 str(cp, ExternalReferenceAsOperand(context_address, no_reg));
1592
1593 // Reserve place for the return address and stack space and align the frame
1594 // preparing for calling the runtime function.
1595 AllocateStackSpace((stack_space + 1) * kPointerSize);
1596 EnforceStackAlignment();
1597
1598 // Set the exit frame sp value to point just before the return address
1599 // location.
1600 add(scratch, sp, Operand(kPointerSize));
1601 str(scratch, MemOperand(fp, ExitFrameConstants::kSPOffset));
1602}

1668void MacroAssembler::StackOverflowCheck(Register num_args, Register scratch,
1669 Label* stack_overflow) {
1670 ASM_CODE_COMMENT(this);

1683void MacroAssembler::InvokePrologue(Register expected_parameter_count,
1684 Register actual_parameter_count,
1685 InvokeType type) {
1686 ASM_CODE_COMMENT(this);
1687 Label regular_invoke;
1688 // r0: actual arguments count
1689 // r1: function (passed through to callee)
1690 // r2: expected arguments count
1691 DCHECK_EQ(actual_parameter_count, r0);
1692 DCHECK_EQ(expected_parameter_count, r2);
1693
1694 // If overapplication or if the actual argument count is equal to the
1695 // formal parameter count, no need to push extra undefined values.
1696 sub(expected_parameter_count, expected_parameter_count,
1697 actual_parameter_count, SetCC);
1698 b(le, &regular_invoke);
1699
1700 Label stack_overflow;
1701 Register scratch = r4;
1702 StackOverflowCheck(expected_parameter_count, scratch, &stack_overflow);
1703
1704 // Underapplication. Move the arguments already in the stack, including the
1705 // receiver and the return address.
1706 {
1707 Label copy, check;
1708 Register num = r5, src = r6, dest = r9; // r7 and r8 are context and root.
1709 mov(src, sp);
1710 // Update stack pointer.
1711 lsl(scratch, expected_parameter_count, Operand(kSystemPointerSizeLog2));
1712 AllocateStackSpace(scratch);
1713 mov(dest, sp);
1714 mov(num, actual_parameter_count);
1715 b(&check);
1716 bind(&copy);
1717 ldr(scratch, MemOperand(src, kSystemPointerSize, PostIndex));
1718 str(scratch, MemOperand(dest, kSystemPointerSize, PostIndex));
1719 sub(num, num, Operand(1), SetCC);
1720 bind(&check);
1721 b(gt, &copy);
1722 }
1723
1724 // Fill remaining expected arguments with undefined values.
1725 LoadRoot(scratch, RootIndex::kUndefinedValue);
1726 {
1727 Label loop;
1728 bind(&loop);
1729 str(scratch, MemOperand(r9, kSystemPointerSize, PostIndex));
1730 sub(expected_parameter_count, expected_parameter_count, Operand(1), SetCC);
1731 b(gt, &loop);
1732 }
1733 b(&regular_invoke);
1734
1735 bind(&stack_overflow);
1736 {
1737 FrameScope frame(
1738 this, has_frame() ? StackFrame::NO_FRAME_TYPE : StackFrame::INTERNAL);
1739 CallRuntime(Runtime::kThrowStackOverflow);
1740 bkpt(0);
1741 }
1742
1743 bind(&regular_invoke);
1744}

1780void MacroAssembler::InvokeFunctionCode(Register function, Register new_target,
1781 Register expected_parameter_count,
1782 Register actual_parameter_count,
1783 InvokeType type) {
1784 ASM_CODE_COMMENT(this);
1785 // You can't call a function without a valid frame.
1786 DCHECK_IMPLIES(type == InvokeType::kCall, has_frame());
1787 DCHECK_EQ(function, r1);
1788 DCHECK_IMPLIES(new_target.is_valid(), new_target == r3);
1789
1790 // On function call, call into the debugger if necessary.
1791 Label debug_hook, continue_after_hook;
1792 {
1793 ExternalReference debug_hook_active =
1794 ExternalReference::debug_hook_on_function_call_address(isolate());
1795 Move(r4, debug_hook_active);
1796 ldrsb(r4, MemOperand(r4));
1797 cmp(r4, Operand(0));
1798 b(ne, &debug_hook);
1799 }
1800 bind(&continue_after_hook);
1801
1802 // Clear the new.target register if not given.
1803 if (!new_target.is_valid()) {
1804 LoadRoot(r3, RootIndex::kUndefinedValue);
1805 }
1806
1807 InvokePrologue(expected_parameter_count, actual_parameter_count, type);
1808 // We call indirectly through the code field in the function to
1809 // allow recompilation to take effect without changing any of the
1810 // call sites.
1811 constexpr int unused_argument_count = 0;
1812 switch (type) {
1813 case InvokeType::kCall:
1814 CallJSFunction(function, unused_argument_count);
1815 break;
1816 case InvokeType::kJump:
1817 JumpJSFunction(function);
1818 break;
1819 }
1820 Label done;
1821 b(&done);
1822
1823 // Deferred debug hook.
1824 bind(&debug_hook);
1825 CallDebugOnFunctionCall(function, new_target, expected_parameter_count,
1826 actual_parameter_count);
1827 b(&continue_after_hook);
1828
1829 // Continue here if InvokePrologue does handle the invocation due to
1830 // mismatched parameter counts.
1831 bind(&done);
1832}

2447void MacroAssembler::Abort(AbortReason reason) {
2448 ASM_CODE_COMMENT(this);
2449 Label abort_start;
2450 bind(&abort_start);
2451 if (v8_flags.code_comments) {
2452 const char* msg = GetAbortReason(reason);
2453 RecordComment("Abort message: ");
2454 RecordComment(msg);
2455 }
2456
2457 // Avoid emitting call to builtin if requested.
2458 if (trap_on_abort()) {
2459 stop();
2460 return;
2461 }
2462
2463 if (should_abort_hard()) {
2464 // We don't care if we constructed a frame. Just pretend we did.
2465 FrameScope assume_frame(this, StackFrame::NO_FRAME_TYPE);
2466 Move32BitImmediate(r0, Operand(static_cast<int>(reason)));
2467 PrepareCallCFunction(1, 0, r1);
2468 Move(r1, ExternalReference::abort_with_reason());
2469 // Use Call directly to avoid any unneeded overhead. The function won't
2470 // return anyway.
2471 Call(r1);
2472 return;
2473 }
2474
2475 Move(r1, Smi::FromInt(static_cast<int>(reason)));
2476
2477 {
2478 // We don't actually want to generate a pile of code for this, so just
2479 // claim there is a stack frame, without generating one.
2480 FrameScope scope(this, StackFrame::NO_FRAME_TYPE);
2481 if (root_array_available()) {
2482 // Generate an indirect call via builtins entry table here in order to
2483 // ensure that the interpreter_entry_return_pc_offset is the same for
2484 // InterpreterEntryTrampoline and InterpreterEntryTrampolineForProfiling
2485 // when v8_flags.debug_code is enabled.
2486 LoadEntryFromBuiltin(Builtin::kAbort, ip);
2487 Call(ip);
2488 } else {
2489 CallBuiltin(Builtin::kAbort);
2490 }
2491 }
2492 // will not return here
2493}

2615template <typename T>
2616void MacroAssembler::FloatMaxHelper(T result, T left, T right,
2617 Label* out_of_line) {
2618 // This trivial case is caught sooner, so that the out-of-line code can be
2619 // completely avoided.
2620 DCHECK(left != right);
2621
2622 if (CpuFeatures::IsSupported(ARMv8)) {
2623 CpuFeatureScope scope(this, ARMv8);
2624 VFPCompareAndSetFlags(left, right);
2625 b(vs, out_of_line);
2626 vmaxnm(result, left, right);
2627 } else {
2628 Label done;
2629 VFPCompareAndSetFlags(left, right);
2630 b(vs, out_of_line);
2631 // Avoid a conditional instruction if the result register is unique.
2632 bool aliased_result_reg = result == left || result == right;
2633 Move(result, right, aliased_result_reg ? mi : al);
2634 Move(result, left, gt);
2635 b(ne, &done);
2636 // Left and right are equal, but check for +/-0.
2637 VFPCompareAndSetFlags(left, 0.0);
2638 b(eq, out_of_line);
2639 // The arguments are equal and not zero, so it doesn't matter which input we
2640 // pick. We have already moved one input into the result (if it didn't
2641 // already alias) so there's nothing more to do.
2642 bind(&done);
2643 }
2644}

2646template <typename T>
2647void MacroAssembler::FloatMaxOutOfLineHelper(T result, T left, T right) {
2648 DCHECK(left != right);
2649
2650 // ARMv8: At least one of left and right is a NaN.
2651 // Anything else: At least one of left and right is a NaN, or both left and
2652 // right are zeroes with unknown sign.
2653
2654 // If left and right are +/-0, select the one with the most positive sign.
2655 // If left or right are NaN, vadd propagates the appropriate one.
2656 vadd(result, left, right);
2657}

2659template <typename T>
2660void MacroAssembler::FloatMinHelper(T result, T left, T right,
2661 Label* out_of_line) {
2662 // This trivial case is caught sooner, so that the out-of-line code can be
2663 // completely avoided.
2664 DCHECK(left != right);
2665
2666 if (CpuFeatures::IsSupported(ARMv8)) {
2667 CpuFeatureScope scope(this, ARMv8);
2668 VFPCompareAndSetFlags(left, right);
2669 b(vs, out_of_line);
2670 vminnm(result, left, right);
2671 } else {
2672 Label done;
2673 VFPCompareAndSetFlags(left, right);
2674 b(vs, out_of_line);
2675 // Avoid a conditional instruction if the result register is unique.
2676 bool aliased_result_reg = result == left || result == right;
2677 Move(result, left, aliased_result_reg ? mi : al);
2678 Move(result, right, gt);
2679 b(ne, &done);
2680 // Left and right are equal, but check for +/-0.
2681 VFPCompareAndSetFlags(left, 0.0);
2682 // If the arguments are equal and not zero, it doesn't matter which input we
2683 // pick. We have already moved one input into the result (if it didn't
2684 // already alias) so there's nothing more to do.
2685 b(ne, &done);
2686 // At this point, both left and right are either 0 or -0.
2687 // We could use a single 'vorr' instruction here if we had NEON support.
2688 // The algorithm used is -((-L) + (-R)), which is most efficiently expressed
2689 // as -((-L) - R).
2690 if (left == result) {
2691 DCHECK(right != result);
2692 vneg(result, left);
2693 vsub(result, result, right);
2694 vneg(result, result);
2695 } else {
2696 DCHECK(left != result);
2697 vneg(result, right);
2698 vsub(result, result, left);
2699 vneg(result, result);
2700 }
2701 bind(&done);
2702 }
2703}

2754int MacroAssembler::CalculateStackPassedWords(int num_reg_arguments,
2755 int num_double_arguments) {
2756 int stack_passed_words = 0;
2757 if (use_eabi_hardfloat()) {
2758 // In the hard floating point calling convention, we can use the first 8
2759 // registers to pass doubles.
2760 if (num_double_arguments > kDoubleRegisterPassedArguments) {
2761 stack_passed_words +=
2762 2 * (num_double_arguments - kDoubleRegisterPassedArguments);
2763 }
2764 } else {
2765 // In the soft floating point calling convention, every double
2766 // argument is passed using two registers.
2767 num_reg_arguments += 2 * num_double_arguments;
2768 }
2769 // Up to four simple arguments are passed in registers r0..r3.
2770 if (num_reg_arguments > kRegisterPassedArguments) {
2771 stack_passed_words += num_reg_arguments - kRegisterPassedArguments;
2772 }
2773 return stack_passed_words;
2774}

2831int MacroAssembler::CallCFunction(Register function, int num_reg_arguments,
2832 int num_double_arguments,
2833 SetIsolateDataSlots set_isolate_data_slots,
2834 Label* return_label) {
2835 ASM_CODE_COMMENT(this);
2836 DCHECK_LE(num_reg_arguments + num_double_arguments, kMaxCParameters);
2837 DCHECK(has_frame());
2838 // Make sure that the stack is aligned before calling a C function unless
2839 // running in the simulator. The simulator has its own alignment check which
2840 // provides more information.
2841#if V8_HOST_ARCH_ARM
2842 if (v8_flags.debug_code) {
2843 int frame_alignment = base::OS::ActivationFrameAlignment();
2844 int frame_alignment_mask = frame_alignment - 1;
2845 if (frame_alignment > kPointerSize) {
2846 ASM_CODE_COMMENT_STRING(this, "Check stack alignment");
2847 DCHECK(base::bits::IsPowerOfTwo(frame_alignment));
2848 Label alignment_as_expected;
2849 tst(sp, Operand(frame_alignment_mask));
2850 b(eq, &alignment_as_expected);
2851 // Don't use Check here, as it will call Runtime_Abort possibly
2852 // re-entering here.
2853 stop();
2854 bind(&alignment_as_expected);
2855 }
2856 }
2857#endif
2858
2859 Label get_pc;
2860
2861 if (set_isolate_data_slots == SetIsolateDataSlots::kYes) {
2862 Register pc_scratch = r5;
2863 Push(pc_scratch);
2864 GetLabelAddress(pc_scratch, &get_pc);
2865
2866 // Save the frame pointer and PC so that the stack layout remains iterable,
2867 // even without an ExitFrame which normally exists between JS and C frames.
2868 CHECK(root_array_available());
2869 str(pc_scratch,
2870 ExternalReferenceAsOperand(IsolateFieldId::kFastCCallCallerPC));
2871 str(fp, ExternalReferenceAsOperand(IsolateFieldId::kFastCCallCallerFP));
2872
2873 Pop(pc_scratch);
2874 }
2875
2876 // Just call directly. The function called cannot cause a GC, or
2877 // allow preemption, so the return address in the link register
2878 // stays correct.
2879 Call(function);
2880 int call_pc_offset = pc_offset();
2881 bind(&get_pc);
2882 if (return_label) bind(return_label);
2883
2884 if (set_isolate_data_slots == SetIsolateDataSlots::kYes) {
2885 // We don't unset the PC; the FP is the source of truth.
2886 Register zero_scratch = r5;
2887 Push(zero_scratch);
2888 mov(zero_scratch, Operand::Zero());
2889
2890 str(zero_scratch,
2891 ExternalReferenceAsOperand(IsolateFieldId::kFastCCallCallerFP));
2892
2893 Pop(zero_scratch);
2894 }
2895
2896 int stack_passed_arguments =
2897 CalculateStackPassedWords(num_reg_arguments, num_double_arguments);
2898 if (ActivationFrameAlignment() > kPointerSize) {
2899 ldr(sp, MemOperand(sp, stack_passed_arguments * kPointerSize));
2900 } else {
2901 add(sp, sp, Operand(stack_passed_arguments * kPointerSize));
2902 }
2903
2904 return call_pc_offset;
2905}

3053 ASM_CODE_COMMENT(this);
3054 UseScratchRegisterScope temps(this);
3055 QwNeonRegister tmp = temps.AcquireQ();
3056 // src = | a | b | c | d |
3057 // tmp = | max(a,b) | max(c,d) | ...
3058 vpmax(NeonU32, tmp.low(), src.low(), src.high());
3059 // tmp = | max(a,b) == 0 | max(c,d) == 0 | ...
3060 vceq(Neon32, tmp, tmp, 0);
3061 // tmp = | max(a,b) == 0 or max(c,d) == 0 | ...
3062 vpmax(NeonU32, tmp.low(), tmp.low(), tmp.low());
3063 // dst = (max(a,b) == 0 || max(c,d) == 0)
3064 // dst will either be -1 or 0.
3065 vmov(NeonS32, dst, tmp.low(), 0);
3066 // dst = !dst (-1 -> 0, 0 -> 1)
3067 add(dst, dst, Operand(1));
3068 // This works because:
3069 // !dst
3070 // = !(max(a,b) == 0 || max(c,d) == 0)
