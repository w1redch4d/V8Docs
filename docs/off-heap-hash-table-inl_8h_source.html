14template <typename Derived>
15OffHeapHashTableBase<Derived>::OffHeapHashTableBase(int capacity)
16 : number_of_elements_(0),
17 number_of_deleted_elements_(0),
18 capacity_(capacity) {
19 MemsetTagged(slot(InternalIndex(0)), empty_element(),
20 capacity * Derived::kEntrySize);
21}

23template <typename Derived>
24void OffHeapHashTableBase<Derived>::RehashInto(PtrComprCageBase cage_base,
25 Derived* new_table) {
26 DCHECK_LT(number_of_elements(), new_table->capacity());
27 DCHECK(new_table->HasSufficientCapacityToAdd(number_of_elements()));
28
29 Derived* derived_this = static_cast<Derived*>(this);
30 // Rehash the elements and copy them into new_table.
31 for (InternalIndex i : InternalIndex::Range(capacity())) {
32 Tagged<Object> key = derived_this->GetKey(cage_base, i);
33 if (!IsKey(key)) continue;
34 uint32_t hash = Derived::Hash(cage_base, key);
35 InternalIndex insertion_index =
36 new_table->FindInsertionEntry(cage_base, hash);
37 new_table->SetKey(insertion_index, key);
38 derived_this->CopyEntryExcludingKeyInto(cage_base, i, new_table,
39 insertion_index);
40 }
41 new_table->number_of_elements_ = number_of_elements();
42}

44template <typename Derived>
45inline bool OffHeapHashTableBase<Derived>::ShouldResizeToAdd(
46 int additional_elements, int* new_capacity) {
47 DCHECK_NOT_NULL(new_capacity);
48 // Grow or shrink table if needed. We first try to shrink the table, if it
49 // is sufficiently empty; otherwise we make sure to grow it so that it has
50 // enough space.
51 int capacity_after_shrinking = ComputeCapacityWithShrink(
52 capacity_, number_of_elements_ + additional_elements);
53
54 if (capacity_after_shrinking < capacity_) {
55 DCHECK(HasSufficientCapacityToAdd(
56 capacity_after_shrinking, number_of_elements_, 0, additional_elements));
57 *new_capacity = capacity_after_shrinking;
58 return true;
59 } else if (!HasSufficientCapacityToAdd(additional_elements)) {
60 *new_capacity = ComputeCapacity(number_of_elements_ + additional_elements);
61 return true;
62 } else {
63 *new_capacity = -1;
64 return false;
65 }
66}

68// static
69template <typename Derived>
70bool OffHeapHashTableBase<Derived>::HasSufficientCapacityToAdd(
71 int capacity, int number_of_elements, int number_of_deleted_elements,
72 int number_of_additional_elements) {
73 int nof = number_of_elements + number_of_additional_elements;
74 // Return true if:
75 // 50% is still free after adding number_of_additional_elements elements and
76 // at most 50% of the free elements are deleted elements.
77 if ((nof < capacity) &&
78 ((number_of_deleted_elements <= (capacity - nof) / 2))) {
79 int needed_free = nof / 2;
80 if (nof + needed_free <= capacity) return true;
81 }
82 return false;
83}

85// static
86template <typename Derived>
87int OffHeapHashTableBase<Derived>::ComputeCapacity(int at_least_space_for) {
88 // Add 50% slack to make slot collisions sufficiently unlikely.
89 // See matching computation in HasSufficientCapacityToAdd().
90 int raw_capacity = at_least_space_for + (at_least_space_for >> 1);
91 int capacity = base::bits::RoundUpToPowerOfTwo32(raw_capacity);
92 return std::max(capacity, Derived::kMinCapacity);
93}

95// static
96template <typename Derived>
97int OffHeapHashTableBase<Derived>::ComputeCapacityWithShrink(
98 int current_capacity, int at_least_space_for) {
99 // Only shrink if the table is very empty to avoid performance penalty.
100 DCHECK_GE(current_capacity, Derived::kMinCapacity);
101 if (at_least_space_for > (current_capacity / Derived::kMaxEmptyFactor)) {
102 return current_capacity;
103 }
104
105 // Recalculate the smaller capacity actually needed.
106 int new_capacity = ComputeCapacity(at_least_space_for);
107 DCHECK_GE(new_capacity, at_least_space_for);
108 // Don't go lower than room for {kMinCapacity} elements.
109 if (new_capacity < Derived::kMinCapacity) return current_capacity;
110 return new_capacity;
111}

121template <typename Derived>
122template <typename IsolateT, typename FindKey>
123InternalIndex OffHeapHashTableBase<Derived>::FindEntry(IsolateT* isolate,
124 FindKey key,
125 uint32_t hash) const {
126 const Derived* derived_this = static_cast<const Derived*>(this);
127 uint32_t count = 1;
128 for (InternalIndex entry = FirstProbe(hash, capacity_);;
129 entry = NextProbe(entry, count++, capacity_)) {
130 // TODO(leszeks): Consider delaying the decompression until after the
131 // comparisons against empty/deleted.
132 Tagged<Object> element = derived_this->GetKey(isolate, entry);
133 if (element == empty_element()) return InternalIndex::NotFound();
134 if (element == deleted_element()) continue;
135 if (Derived::KeyIsMatch(isolate, key, element)) return entry;
136 }
137}

139template <typename Derived>
140InternalIndex OffHeapHashTableBase<Derived>::FindInsertionEntry(
141 PtrComprCageBase cage_base, uint32_t hash) const {
142 // The derived class must guarantee the hash table is never full.
143 DCHECK(HasSufficientCapacityToAdd(1));
144 const Derived* derived_this = static_cast<const Derived*>(this);
145 uint32_t count = 1;
146 for (InternalIndex entry = FirstProbe(hash, capacity_);;
147 entry = NextProbe(entry, count++, capacity_)) {
148 // TODO(leszeks): Consider delaying the decompression until after the
149 // comparisons against empty/deleted.
150 Tagged<Object> element = derived_this->GetKey(cage_base, entry);
151 if (!IsKey(element)) return entry;
152 }
153}

155template <typename Derived>
156template <typename IsolateT, typename FindKey>
157InternalIndex OffHeapHashTableBase<Derived>::FindEntryOrInsertionEntry(
158 IsolateT* isolate, FindKey key, uint32_t hash) const {
159 // The derived class must guarantee the hash table is never full.
160 DCHECK(HasSufficientCapacityToAdd(1));
161 const Derived* derived_this = static_cast<const Derived*>(this);
162 InternalIndex insertion_entry = InternalIndex::NotFound();
163 uint32_t count = 1;
164 for (InternalIndex entry = FirstProbe(hash, capacity_);;
165 entry = NextProbe(entry, count++, capacity_)) {
166 // TODO(leszeks): Consider delaying the decompression until after the
167 // comparisons against empty/deleted.
168 Tagged<Object> element = derived_this->GetKey(isolate, entry);
169 if (element == empty_element()) {
170 // Empty entry, it's our insertion entry if there was no previous Hole.
171 if (insertion_entry.is_not_found()) return entry;
172 return insertion_entry;
173 }
174
175 if (element == deleted_element()) {
176 // Holes are potential insertion candidates, but we continue the search
177 // in case we find the actual matching entry.
178 if (insertion_entry.is_not_found()) insertion_entry = entry;
179 continue;
180 }
181
182 if (Derived::KeyIsMatch(isolate, key, element)) return entry;
183 }
184}

187template <typename Derived>
188template <typename Container, size_t OffsetOfElementsInContainer>
189void* OffHeapHashTableBase<Derived>::Allocate(int capacity) {
190 // Make sure that the elements_ array is at the end of Container, with no
191 // padding, so that subsequent elements can be accessed as offsets from
192 // elements_.
193 static_assert(OffsetOfElementsInContainer ==
194 sizeof(Container) - sizeof(Tagged_t));
195 // Make sure that elements_ is aligned when Container is aligned.
196 static_assert(OffsetOfElementsInContainer % kTaggedSize == 0);
197
198 return AlignedAllocWithRetry(
199 sizeof(Container) + GetSizeExcludingHeader(capacity),
200 std::max(alignof(Container), alignof(void*)));
201}
