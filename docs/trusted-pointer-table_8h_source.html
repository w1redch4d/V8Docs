30struct TrustedPointerTableEntry {
31 // Make this entry a "regular" entry, containing an absolute pointer to a
32 // TrustedObject.
33 inline void MakeTrustedPointerEntry(Address pointer, IndirectPointerTag tag,
34 bool mark_as_alive);
35
36 // Make this entry a freelist entry, containing the index of the next entry
37 // on the freelist.
38 inline void MakeFreelistEntry(uint32_t next_entry_index);
39
40 // Make this entry a zapped entry. Zapped entries contain invalid pointers.
41 inline void MakeZappedEntry();
42
43 // Retrieve the pointer stored in this entry. This entry must be tagged with
44 // the given tag, otherwise an inaccessible pointer will be returned.
45 // This entry must not be a freelist entry.
46 inline Address GetPointer(IndirectPointerTag tag) const;
47
48 // Store the given pointer in this entry while preserving the marking state.
49 // This entry must not be a freelist entry.

58 // Get the index of the next entry on the freelist. This method may be

61 // for efficient entry allocation, see TryAllocateEntryFromFreelist.
62 inline uint32_t GetNextFreelistEntryIndex() const;

64 // Mark this entry as alive during garbage collection.
65 inline void Mark();

78 // TrustedPointerTable entries consist of a single pointer-sized word
79 // containing a tag and marking bit together with the actual pointer.
80 struct TrustedPointerTaggingScheme {
81 using TagType = IndirectPointerTag;
82 static constexpr uint64_t kMarkBit = kTrustedPointerTableMarkBit;
83 static constexpr uint64_t kTagMask = kIndirectPointerTagMask;
84 static constexpr TagType kFreeEntryTag = kFreeTrustedPointerTableEntryTag;
85 static constexpr bool kSupportsEvacuation = false;
86 static constexpr bool kSupportsZapping = false;
87 };
88
89 struct Payload : TaggedPayload<TrustedPointerTaggingScheme> {
90 static Payload ForTrustedPointerEntry(Address pointer,
91 IndirectPointerTag tag) {
92 // We expect to only store references to (trusted) HeapObjects in the
93 // TrustedPointerTable, so the HeapObject tag bit must be set.
94 DCHECK_EQ(pointer & kHeapObjectTag, kHeapObjectTag);
95 DCHECK_EQ(pointer & kTrustedPointerTableMarkBit, 0);
96 DCHECK_EQ(pointer & kIndirectPointerTagMask, 0);
97 return Payload(pointer, tag);
98 }
99
100 static Payload ForFreelistEntry(uint32_t next_entry) {
101 return Payload(next_entry, kFreeTrustedPointerTableEntryTag);
102 }
103
104 static Payload ForZappedEntry() {
105 return Payload(0, kIndirectPointerNullTag);
106 }
107
108 private:
109 Payload(Address pointer, IndirectPointerTag tag)
110 : TaggedPayload(pointer, tag) {}
111 };
112
113 std::atomic<Payload> payload_;
114};

188 //
189 // The callback function will be invoked once for every entry that is
