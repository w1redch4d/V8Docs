33 void SerializeTypeFeedback(ZoneBuffer& buffer) {
34 const std::unordered_map<uint32_t, FunctionTypeFeedback>&
35 feedback_for_function = module_->type_feedback.feedback_for_function;
36
37 // Get an ordered list of function indexes, so we generate deterministic
38 // data.
39 std::vector<uint32_t> ordered_function_indexes;
40 ordered_function_indexes.reserve(feedback_for_function.size());
41 for (const auto& entry : feedback_for_function) {
42 // Skip functions for which we have no feedback.
43 if (entry.second.feedback_vector.empty()) continue;
44 ordered_function_indexes.push_back(entry.first);
45 }
46 std::sort(ordered_function_indexes.begin(), ordered_function_indexes.end());
47
48 buffer.write_u32v(static_cast<uint32_t>(ordered_function_indexes.size()));
49 for (const uint32_t func_index : ordered_function_indexes) {
50 buffer.write_u32v(func_index);
51 // Serialize {feedback_vector}.
52 const FunctionTypeFeedback& feedback =
53 feedback_for_function.at(func_index);
54 buffer.write_u32v(static_cast<uint32_t>(feedback.feedback_vector.size()));
55 for (const CallSiteFeedback& call_site_feedback :
56 feedback.feedback_vector) {
57 int cases = call_site_feedback.num_cases();
58 buffer.write_i32v(cases);
59 for (int i = 0; i < cases; ++i) {
60 buffer.write_i32v(call_site_feedback.function_index(i));
61 buffer.write_i32v(call_site_feedback.call_count(i));
62 }
63 }
64 // Serialize {call_targets}.
65 buffer.write_u32v(static_cast<uint32_t>(feedback.call_targets.size()));
66 for (uint32_t call_target : feedback.call_targets) {
67 buffer.write_u32v(call_target);
68 }
69 }
70 }

72 void SerializeTieringInfo(ZoneBuffer& buffer) {
73 const std::unordered_map<uint32_t, FunctionTypeFeedback>&
74 feedback_for_function = module_->type_feedback.feedback_for_function;
75 const uint32_t initial_budget = v8_flags.wasm_tiering_budget;
76 for (uint32_t declared_index = 0;
77 declared_index < module_->num_declared_functions; ++declared_index) {
78 uint32_t func_index = declared_index + module_->num_imported_functions;
79 auto feedback_it = feedback_for_function.find(func_index);
80 int prio = feedback_it == feedback_for_function.end()
81 ? 0
82 : feedback_it->second.tierup_priority;
83 DCHECK_LE(0, prio);
84 uint32_t remaining_budget =
85 tiering_budget_array_[declared_index].load(std::memory_order_relaxed);
86 DCHECK_GE(initial_budget, remaining_budget);
87
88 bool was_tiered_up = prio > 0;
89 bool was_executed = was_tiered_up || remaining_budget != initial_budget;
90
91 // TODO(13209): Make this less V8-specific for productionization.
92 buffer.write_u8((was_executed ? kFunctionExecutedBit : 0) |
93 (was_tiered_up ? kFunctionTieredUpBit : 0));
94 }
95 }

105void DeserializeTypeFeedback(Decoder& decoder, const WasmModule* module) {
106 base::SharedMutexGuard<base::kShared> type_feedback_guard{
107 &module->type_feedback.mutex};
108 std::unordered_map<uint32_t, FunctionTypeFeedback>& feedback_for_function =
109 module->type_feedback.feedback_for_function;
110 uint32_t num_entries = decoder.consume_u32v("num function entries");
111 CHECK_LE(num_entries, module->num_declared_functions);
112 for (uint32_t missing_entries = num_entries; missing_entries > 0;
113 --missing_entries) {
114 FunctionTypeFeedback function_feedback;
115 uint32_t function_index = decoder.consume_u32v("function index");
116 // Deserialize {feedback_vector}.
117 uint32_t feedback_vector_size =
118 decoder.consume_u32v("feedback vector size");
119 function_feedback.feedback_vector.resize(feedback_vector_size);
120 for (CallSiteFeedback& feedback : function_feedback.feedback_vector) {
121 int num_cases = decoder.consume_i32v("num cases");
122 if (num_cases == 0) continue; // no feedback
123 if (num_cases == 1) { // monomorphic
124 int called_function_index = decoder.consume_i32v("function index");
125 int call_count = decoder.consume_i32v("call count");
126 feedback = CallSiteFeedback{called_function_index, call_count};
127 } else { // polymorphic
128 auto* polymorphic = new CallSiteFeedback::PolymorphicCase[num_cases];
129 for (int i = 0; i < num_cases; ++i) {
130 polymorphic[i].function_index =
131 decoder.consume_i32v("function index");
132 polymorphic[i].absolute_call_frequency =
133 decoder.consume_i32v("call count");
134 }
135 feedback = CallSiteFeedback{polymorphic, num_cases};
136 }
137 }
138 // Deserialize {call_targets}.
139 uint32_t num_call_targets = decoder.consume_u32v("num call targets");
140 function_feedback.call_targets =
141 base::OwnedVector<uint32_t>::NewForOverwrite(num_call_targets);
142 for (uint32_t& call_target : function_feedback.call_targets) {
143 call_target = decoder.consume_u32v("call target");
144 }
145
146 // Finally, insert the new feedback into the map. Overwrite existing
147 // feedback, but check for consistency.
148 auto [feedback_it, is_new] = feedback_for_function.emplace(
149 function_index, std::move(function_feedback));
150 if (!is_new) {
151 FunctionTypeFeedback& old_feedback = feedback_it->second;
152 CHECK(old_feedback.feedback_vector.empty() ||
153 old_feedback.feedback_vector.size() == feedback_vector_size);
154 CHECK_EQ(old_feedback.call_targets.as_vector(),
155 function_feedback.call_targets.as_vector());
156 std::swap(old_feedback.feedback_vector,
157 function_feedback.feedback_vector);
158 }
159 }
160}

162std::unique_ptr<ProfileInformation> DeserializeTieringInformation(
163 Decoder& decoder, const WasmModule* module) {
164 std::vector<uint32_t> executed_functions;
165 std::vector<uint32_t> tiered_up_functions;
166 uint32_t start = module->num_imported_functions;
167 uint32_t end = start + module->num_declared_functions;
168 for (uint32_t func_index = start; func_index < end; ++func_index) {
169 uint8_t tiering_info = decoder.consume_u8("tiering info");
170 CHECK_EQ(0, tiering_info & ~3);
171 bool was_executed = tiering_info & kFunctionExecutedBit;
172 bool was_tiered_up = tiering_info & kFunctionTieredUpBit;
173 if (was_tiered_up) tiered_up_functions.push_back(func_index);
174 if (was_executed) executed_functions.push_back(func_index);
175 }
176
177 return std::make_unique<ProfileInformation>(std::move(executed_functions),
178 std::move(tiered_up_functions));
179}

181std::unique_ptr<ProfileInformation> RestoreProfileData(
182 const WasmModule* module, base::Vector<uint8_t> profile_data) {
183 Decoder decoder{profile_data.begin(), profile_data.end()};
184
185 DeserializeTypeFeedback(decoder, module);
186 std::unique_ptr<ProfileInformation> pgo_info =
187 DeserializeTieringInformation(decoder, module);
188
189 CHECK(decoder.ok());
190 CHECK_EQ(decoder.pc(), decoder.end());
191
192 return pgo_info;
193}

195void DumpProfileToFile(const WasmModule* module,
196 base::Vector<const uint8_t> wire_bytes,
197 std::atomic<uint32_t>* tiering_budget_array) {
198 CHECK(!wire_bytes.empty());
199 // File are named `profile-wasm-<hash>`.
200 // We use the same hash as for reported scripts, to make it easier to
201 // correlate files to wasm modules (see {CreateWasmScript}).
202 uint32_t hash = static_cast<uint32_t>(GetWireBytesHash(wire_bytes));
203 base::EmbeddedVector<char, 32> filename;
204 SNPrintF(filename, "profile-wasm-%08x", hash);
205
206 ProfileGenerator profile_generator{module, tiering_budget_array};
207 base::OwnedVector<uint8_t> profile_data = profile_generator.GetProfileData();
208
209 PrintF(
210 "Dumping Wasm PGO data to file '%s' (module size %zu, %u declared "
211 "functions, %zu bytes PGO data)\n",
212 filename.begin(), wire_bytes.size(), module->num_declared_functions,
213 profile_data.size());
214 if (FILE* file = base::OS::FOpen(filename.begin(), "wb")) {
215 size_t written = fwrite(profile_data.begin(), 1, profile_data.size(), file);
216 CHECK_EQ(profile_data.size(), written);
217 base::Fclose(file);
218 }
219}

221std::unique_ptr<ProfileInformation> LoadProfileFromFile(
222 const WasmModule* module, base::Vector<const uint8_t> wire_bytes) {
223 CHECK(!wire_bytes.empty());
224 // File are named `profile-wasm-<hash>`.
225 // We use the same hash as for reported scripts, to make it easier to
226 // correlate files to wasm modules (see {CreateWasmScript}).
227 uint32_t hash = static_cast<uint32_t>(GetWireBytesHash(wire_bytes));
228 base::EmbeddedVector<char, 32> filename;
229 SNPrintF(filename, "profile-wasm-%08x", hash);
230
231 FILE* file = base::OS::FOpen(filename.begin(), "rb");
232 if (!file) {
233 PrintF("No Wasm PGO data found: Cannot open file '%s'\n", filename.begin());
234 return {};
235 }
236
237 fseek(file, 0, SEEK_END);
238 size_t size = ftell(file);
239 rewind(file);
240
241 PrintF("Loading Wasm PGO data from file '%s' (%zu bytes)\n", filename.begin(),
242 size);
243 base::OwnedVector<uint8_t> profile_data =
244 base::OwnedVector<uint8_t>::NewForOverwrite(size);
245 for (size_t read = 0; read < size;) {
246 read += fread(profile_data.begin() + read, 1, size - read, file);
247 CHECK(!ferror(file));
248 }
249
250 base::Fclose(file);
251
252 return RestoreProfileData(module, profile_data.as_vector());
253}
