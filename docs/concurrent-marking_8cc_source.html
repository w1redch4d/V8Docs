86 // Implements ephemeron semantics: Marks value if key is already reachable.
87 // Returns true if value was actually marked.
88 bool ProcessEphemeron(Tagged<HeapObject> key, Tagged<HeapObject> value) {
89 if (marking_state()->IsMarked(key)) {
90 const auto target_worklist =
91 MarkingHelper::ShouldMarkObject(heap_, value);
92 DCHECK(target_worklist.has_value());
93 if (MarkObject(key, value, target_worklist.value())) {
94 return true;
95 }
96 } else if (marking_state()->IsUnmarked(value)) {
97 local_weak_objects_->next_ephemerons_local.Push(Ephemeron{key, value});
98 }
99 return false;
100 }

265void ConcurrentMarking::RunMajor(JobDelegate* delegate,
266 base::EnumSet<CodeFlushMode> code_flush_mode,
267 unsigned mark_compact_epoch,
268 bool should_keep_ages_unchanged) {
269 size_t kBytesUntilInterruptCheck = 64 * KB;
270 int kObjectsUntilInterruptCheck = 1000;
271 uint8_t task_id = delegate->GetTaskId() + 1;
272 TaskState* task_state = task_state_[task_id].get();
273 auto* cpp_heap = CppHeap::From(heap_->cpp_heap());
274 MarkingWorklists::Local local_marking_worklists(
275 marking_worklists_, cpp_heap
276 ? cpp_heap->CreateCppMarkingState()
277 : MarkingWorklists::Local::kNoCppMarkingState);
278 WeakObjects::Local local_weak_objects(weak_objects_);
279 ConcurrentMarkingVisitor visitor(
280 &local_marking_worklists, &local_weak_objects, heap_, mark_compact_epoch,
281 code_flush_mode, should_keep_ages_unchanged,
282 heap_->tracer()->CodeFlushingIncrease(), &task_state->memory_chunk_data);
283 NativeContextInferrer native_context_inferrer;
284 NativeContextStats& native_context_stats = task_state->native_context_stats;
285 double time_ms;
286 size_t marked_bytes = 0;
287 Isolate* isolate = heap_->isolate();
288 if (v8_flags.trace_concurrent_marking) {
289 isolate->PrintWithTimestamp("Starting major concurrent marking task %d\n",
290 task_id);
291 }
292 bool another_ephemeron_iteration = false;
293 MainAllocator* const new_space_allocator =
294 heap_->use_new_space() ? heap_->allocator()->new_space_allocator()
295 : nullptr;
296
297 {
298 TimedScope scope(&time_ms);
299
300 {
301 Ephemeron ephemeron;
302 while (local_weak_objects.current_ephemerons_local.Pop(&ephemeron)) {
303 if (visitor.ProcessEphemeron(ephemeron.key, ephemeron.value)) {
304 another_ephemeron_iteration = true;
305 }
306 }
307 }
308 PtrComprCageBase cage_base(isolate);
309 bool is_per_context_mode = local_marking_worklists.IsPerContextMode();
310 bool done = false;
311 while (!done) {
312 size_t current_marked_bytes = 0;
313 int objects_processed = 0;
314 while (current_marked_bytes < kBytesUntilInterruptCheck &&
315 objects_processed < kObjectsUntilInterruptCheck) {
316 Tagged<HeapObject> object;
317 if (!local_marking_worklists.Pop(&object)) {
318 done = true;
319 break;
320 }
321 DCHECK(!HeapLayout::InReadOnlySpace(object));
322 DCHECK_EQ(HeapUtils::GetOwnerHeap(object), heap_);
323 objects_processed++;
324
325 Address new_space_top = kNullAddress;
326 Address new_space_limit = kNullAddress;
327 Address new_large_object = kNullAddress;
328
329 if (new_space_allocator) {
330 // The order of the two loads is important.
331 new_space_top = new_space_allocator->original_top_acquire();
332 new_space_limit = new_space_allocator->original_limit_relaxed();
333 }
334
335 if (heap_->new_lo_space()) {
336 new_large_object = heap_->new_lo_space()->pending_object();
337 }
338
339 Address addr = object.address();
340
341 if ((new_space_top <= addr && addr < new_space_limit) ||
342 addr == new_large_object) {
343 local_marking_worklists.PushOnHold(object);
344 } else {
345 Tagged<Map> map = object->map(cage_base, kAcquireLoad);
346 // The marking worklist should never contain filler objects.
347 CHECK(!IsFreeSpaceOrFillerMap(map));
348 if (is_per_context_mode) {
349 Address context;
350 if (native_context_inferrer.Infer(cage_base, map, object,
351 &context)) {
352 local_marking_worklists.SwitchToContext(context);
353 }
354 }
355 const auto visited_size = visitor.Visit(map, object);
356 visitor.IncrementLiveBytesCached(
357 MutablePageMetadata::cast(
358 MemoryChunkMetadata::FromHeapObject(object)),
359 ALIGN_TO_ALLOCATION_ALIGNMENT(visited_size));
360 if (is_per_context_mode) {
361 native_context_stats.IncrementSize(
362 local_marking_worklists.Context(), map, object, visited_size);
363 }
364 current_marked_bytes += visited_size;
365 }
366 }
367 if (objects_processed > 0) another_ephemeron_iteration = true;
368 marked_bytes += current_marked_bytes;
369 base::AsAtomicWord::Relaxed_Store<size_t>(&task_state->marked_bytes,
370 marked_bytes);
371 if (delegate->ShouldYield()) {
372 TRACE_GC_NOTE("ConcurrentMarking::RunMajor Preempted");
373 break;
374 }
375 }
376
377 if (done) {
378 Ephemeron ephemeron;
379 while (local_weak_objects.discovered_ephemerons_local.Pop(&ephemeron)) {
380 if (visitor.ProcessEphemeron(ephemeron.key, ephemeron.value)) {
381 another_ephemeron_iteration = true;
382 }
383 }
384 }
385
386 local_marking_worklists.Publish();
387 local_weak_objects.Publish();
388 base::AsAtomicWord::Relaxed_Store<size_t>(&task_state->marked_bytes, 0);
389 total_marked_bytes_ += marked_bytes;
390
391 if (another_ephemeron_iteration) {
392 set_another_ephemeron_iteration(true);
393 }
394 }
395 if (v8_flags.trace_concurrent_marking) {
396 heap_->isolate()->PrintWithTimestamp(
397 "Major task %d concurrently marked %dKB in %.2fms\n", task_id,
398 static_cast<int>(marked_bytes / KB), time_ms);
399 }
400
401 DCHECK(task_state->local_pretenuring_feedback.empty());
402}

423V8_INLINE bool IsYoungObjectInLab(MainAllocator* new_space_allocator,
424 NewLargeObjectSpace* new_lo_space,
425 Tagged<HeapObject> heap_object) {
426 // The order of the two loads is important.
427 Address new_space_top = new_space_allocator->original_top_acquire();
428 Address new_space_limit = new_space_allocator->original_limit_relaxed();
429 Address new_large_object = new_lo_space->pending_object();
430
431 Address addr = heap_object.address();
432
433 return (new_space_top <= addr && addr < new_space_limit) ||
434 addr == new_large_object;
435}

439template <YoungGenerationMarkingVisitationMode marking_mode>
440V8_INLINE size_t ConcurrentMarking::RunMinorImpl(JobDelegate* delegate,
441 TaskState* task_state) {
442 static constexpr size_t kBytesUntilInterruptCheck = 64 * KB;
443 static constexpr int kObjectsUntilInterruptCheck = 1000;
444 size_t marked_bytes = 0;
445 size_t current_marked_bytes = 0;
446 int objects_processed = 0;
447 YoungGenerationMarkingVisitor<marking_mode> visitor(
448 heap_, &task_state->local_pretenuring_feedback);
449 YoungGenerationRememberedSetsMarkingWorklist::Local remembered_sets(
450 heap_->minor_mark_sweep_collector()->remembered_sets_marking_handler());
451 auto& marking_worklists_local = visitor.marking_worklists_local();
452 Isolate* isolate = heap_->isolate();
453 minor_marking_state_->MarkerStarted();
454 MainAllocator* const new_space_allocator =
455 heap_->allocator()->new_space_allocator();
456 NewLargeObjectSpace* const new_lo_space = heap_->new_lo_space();
457
458 do {
459 if (delegate->IsJoiningThread()) {
460 marking_worklists_local.MergeOnHold();
461 }
462 Tagged<HeapObject> heap_object;
463 TRACE_GC_EPOCH(heap_->tracer(),
464 GCTracer::Scope::MINOR_MS_BACKGROUND_MARKING_CLOSURE,
465 ThreadKind::kBackground);
466 while (marking_worklists_local.Pop(&heap_object)) {
467 if (IsYoungObjectInLab(new_space_allocator, new_lo_space, heap_object)) {
468 visitor.marking_worklists_local().PushOnHold(heap_object);
469 } else {
470 Tagged<Map> map = heap_object->map(isolate);
471 const auto visited_size = visitor.Visit(map, heap_object);
472 if (visited_size) {
473 current_marked_bytes += visited_size;
474 visitor.IncrementLiveBytesCached(
475 MutablePageMetadata::FromHeapObject(heap_object),
476 ALIGN_TO_ALLOCATION_ALIGNMENT(visited_size));
477 }
478 }
479
480 if (current_marked_bytes >= kBytesUntilInterruptCheck ||
481 ++objects_processed >= kObjectsUntilInterruptCheck) {
482 marked_bytes += current_marked_bytes;
483 if (delegate->ShouldYield()) {
484 TRACE_GC_NOTE("ConcurrentMarking::RunMinor Preempted");
485 minor_marking_state_->MarkerDone();
486 return marked_bytes;
487 }
488 objects_processed = 0;
489 current_marked_bytes = 0;
490 }
491 }
492 } while (remembered_sets.ProcessNextItem(&visitor));
493 if (minor_marking_state_->MarkerDone()) {
494 // This is the last active marker and it ran out of work. Request GC
495 // finalization.
496 heap_->minor_mark_sweep_collector()->RequestGC();
497 }
498 return marked_bytes + current_marked_bytes;
499}

501void ConcurrentMarking::RunMinor(JobDelegate* delegate) {
502 DCHECK(heap_->use_new_space());
503 DCHECK_NOT_NULL(heap_->new_lo_space());
504 uint8_t task_id = delegate->GetTaskId() + 1;
505 DCHECK_LT(task_id, task_state_.size());
506 TaskState* task_state = task_state_[task_id].get();
507 double time_ms;
508 size_t marked_bytes = 0;
509 Isolate* isolate = heap_->isolate();
510 if (v8_flags.trace_concurrent_marking) {
511 isolate->PrintWithTimestamp("Starting minor concurrent marking task %d\n",
512 task_id);
513 }
514
515 {
516 TimedScope scope(&time_ms);
517 if (heap_->minor_mark_sweep_collector()->is_in_atomic_pause()) {
518 // This gets a lower bound for estimated concurrency as we may have marked
519 // most of the graph concurrently already and may not be using parallism
520 // as much.
521 estimate_concurrency_.fetch_add(1, std::memory_order_relaxed);
522 marked_bytes =
523 RunMinorImpl<YoungGenerationMarkingVisitationMode::kParallel>(
524 delegate, task_state);
525 } else {
526 marked_bytes =
527 RunMinorImpl<YoungGenerationMarkingVisitationMode::kConcurrent>(
528 delegate, task_state);
529 }
530 }
531
532 if (v8_flags.trace_concurrent_marking) {
533 heap_->isolate()->PrintWithTimestamp(
534 "Minor task %d concurrently marked %dKB in %.2fms\n", task_id,
535 static_cast<int>(marked_bytes / KB), time_ms);
536 }
537
538 DCHECK(task_state->memory_chunk_data.empty());
539 DCHECK(task_state->native_context_stats.Empty());
540 DCHECK_EQ(0, task_state->marked_bytes);
541}

543size_t ConcurrentMarking::GetMajorMaxConcurrency(size_t worker_count) {
544 size_t marking_items = marking_worklists_->shared()->Size();
545 marking_items += marking_worklists_->other()->Size();
546 for (auto& worklist : marking_worklists_->context_worklists()) {
547 marking_items += worklist.worklist->Size();
548 }
549 const size_t work = std::max<size_t>(
550 {marking_items, weak_objects_->discovered_ephemerons.Size(),
551 weak_objects_->current_ephemerons.Size()});
552 size_t jobs = worker_count + work;
553 jobs = std::min<size_t>(task_state_.size() - 1, jobs);
554 if (heap_->ShouldOptimizeForBattery()) {
555 return std::min<size_t>(jobs, 1);
556 }
557 return jobs;
558}

560size_t ConcurrentMarking::GetMinorMaxConcurrency(size_t worker_count) {
561 const size_t marking_items = marking_worklists_->shared()->Size() +
562 heap_->minor_mark_sweep_collector()
563 ->remembered_sets_marking_handler()
564 ->RemainingRememberedSetsMarkingIteams();
565 DCHECK(marking_worklists_->other()->IsEmpty());
566 DCHECK(!marking_worklists_->IsUsingContextWorklists());
567 size_t jobs = worker_count + marking_items;
568 jobs = std::min<size_t>(task_state_.size() - 1, jobs);
569 if (heap_->ShouldOptimizeForBattery()) {
570 return std::min<size_t>(jobs, 1);
571 }
572 return jobs;
573}

575void ConcurrentMarking::TryScheduleJob(GarbageCollector garbage_collector,
576 TaskPriority priority) {
577 DCHECK(v8_flags.parallel_marking || v8_flags.concurrent_marking ||
578 v8_flags.concurrent_minor_ms_marking);
579 DCHECK(!heap_->IsTearingDown());
580 DCHECK(IsStopped());
581
582 DCHECK_NE(garbage_collector, GarbageCollector::SCAVENGER);
583 if (garbage_collector == GarbageCollector::MARK_COMPACTOR &&
584 !heap_->mark_compact_collector()->UseBackgroundThreadsInCycle()) {
585 return;
586 }
587 if (garbage_collector == GarbageCollector::MINOR_MARK_SWEEPER &&
588 !heap_->minor_mark_sweep_collector()->UseBackgroundThreadsInCycle()) {
589 return;
590 }
591
592 if (v8_flags.concurrent_marking_high_priority_threads) {
593 priority = TaskPriority::kUserBlocking;
594 }
595
596 // Marking state can only be alive if the concurrent marker was previously
597 // stopped.
598 DCHECK_IMPLIES(
599 minor_marking_state_,
600 garbage_collector_.has_value() &&
601 (*garbage_collector_ == garbage_collector) &&
602 (garbage_collector == GarbageCollector::MINOR_MARK_SWEEPER));
603 DCHECK_IMPLIES(
604 !garbage_collector_.has_value() ||
605 *garbage_collector_ == GarbageCollector::MARK_COMPACTOR,
606 std::all_of(task_state_.begin(), task_state_.end(), [](auto& task_state) {
607 return task_state->local_pretenuring_feedback.empty();
608 }));
609 garbage_collector_ = garbage_collector;
610 if (garbage_collector == GarbageCollector::MARK_COMPACTOR) {
611 heap_->mark_compact_collector()->local_marking_worklists()->Publish();
612 marking_worklists_ = heap_->mark_compact_collector()->marking_worklists();
613 auto job = std::make_unique<JobTaskMajor>(
614 this, heap_->mark_compact_collector()->epoch(),
615 heap_->mark_compact_collector()->code_flush_mode(),
616 heap_->ShouldCurrentGCKeepAgesUnchanged());
617 current_job_trace_id_.emplace(job->trace_id());
618 TRACE_GC_NOTE_WITH_FLOW("Major concurrent marking started", job->trace_id(),
619 TRACE_EVENT_FLAG_FLOW_OUT);
620 job_handle_ = V8::GetCurrentPlatform()->PostJob(priority, std::move(job));
621 } else {
622 DCHECK(garbage_collector == GarbageCollector::MINOR_MARK_SWEEPER);
623 minor_marking_state_ = std::make_unique<MinorMarkingState>();
624 heap_->minor_mark_sweep_collector()->local_marking_worklists()->Publish();
625 marking_worklists_ =
626 heap_->minor_mark_sweep_collector()->marking_worklists();
627 auto job = std::make_unique<JobTaskMinor>(this);
628 current_job_trace_id_.emplace(job->trace_id());
629 TRACE_GC_NOTE_WITH_FLOW("Minor concurrent marking started", job->trace_id(),
630 TRACE_EVENT_FLAG_FLOW_OUT);
631 job_handle_ = V8::GetCurrentPlatform()->PostJob(priority, std::move(job));
632 }
633 DCHECK(job_handle_->IsValid());
634}

636bool ConcurrentMarking::IsWorkLeft() const {
637 DCHECK(garbage_collector_.has_value());
638 if (garbage_collector_ == GarbageCollector::MARK_COMPACTOR) {
639 return !marking_worklists_->shared()->IsEmpty() ||
640 !weak_objects_->current_ephemerons.IsEmpty() ||
641 !weak_objects_->discovered_ephemerons.IsEmpty();
642 }
643 DCHECK_EQ(GarbageCollector::MINOR_MARK_SWEEPER, garbage_collector_);
644 return !marking_worklists_->shared()->IsEmpty() ||
645 (heap_->minor_mark_sweep_collector()
646 ->remembered_sets_marking_handler()
647 ->RemainingRememberedSetsMarkingIteams() > 0);
648}

650void ConcurrentMarking::RescheduleJobIfNeeded(
651 GarbageCollector garbage_collector, TaskPriority priority) {
652 DCHECK(v8_flags.parallel_marking || v8_flags.concurrent_marking ||
653 v8_flags.concurrent_minor_ms_marking);
654
655 if (garbage_collector == GarbageCollector::MARK_COMPACTOR &&
656 !heap_->mark_compact_collector()->UseBackgroundThreadsInCycle()) {
657 return;
658 }
659
660 if (garbage_collector == GarbageCollector::MINOR_MARK_SWEEPER &&
661 !heap_->minor_mark_sweep_collector()->UseBackgroundThreadsInCycle()) {
662 return;
663 }
664
665 if (heap_->IsTearingDown()) return;
666
667 if (IsStopped()) {
668 // This DCHECK is for the case that concurrent marking was paused.
669 DCHECK_IMPLIES(garbage_collector_.has_value(),
670 garbage_collector == garbage_collector_);
671 TryScheduleJob(garbage_collector, priority);
672 } else {
673 DCHECK(garbage_collector_.has_value());
674 DCHECK_EQ(garbage_collector, garbage_collector_.value());
675 if (garbage_collector == GarbageCollector::MARK_COMPACTOR) {
676 heap_->mark_compact_collector()->local_marking_worklists()->Publish();
677 } else {
678 heap_->minor_mark_sweep_collector()->local_marking_worklists()->Publish();
679 }
680 if (!IsWorkLeft()) return;
681 if (priority != TaskPriority::kUserVisible)
682 job_handle_->UpdatePriority(priority);
683 DCHECK(current_job_trace_id_.has_value());
684 TRACE_GC_NOTE_WITH_FLOW(
685 garbage_collector_ == GarbageCollector::MARK_COMPACTOR
686 ? "Major concurrent marking rescheduled"
687 : "Minor concurrent marking rescheduled",
688 current_job_trace_id_.value(),
689 TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
690 job_handle_->NotifyConcurrencyIncrease();
691 }
692}

694void ConcurrentMarking::FlushPretenuringFeedback() {

696 for (auto& task_state : task_state_) {
697 pretenuring_handler->MergeAllocationSitePretenuringFeedback(
698 task_state->local_pretenuring_feedback);
699 task_state->local_pretenuring_feedback.clear();

703void ConcurrentMarking::Join() {
704 DCHECK(v8_flags.parallel_marking || v8_flags.concurrent_marking ||
705 v8_flags.concurrent_minor_ms_marking);
706 DCHECK_IMPLIES(
707 garbage_collector_ == GarbageCollector::MARK_COMPACTOR,
708 heap_->mark_compact_collector()->UseBackgroundThreadsInCycle());
709 if (!job_handle_ || !job_handle_->IsValid()) return;
710 job_handle_->Join();
711 current_job_trace_id_.reset();
712 garbage_collector_.reset();
713 minor_marking_state_.reset();
714}

716bool ConcurrentMarking::Pause() {
717 DCHECK(v8_flags.parallel_marking || v8_flags.concurrent_marking);
718 if (!job_handle_ || !job_handle_->IsValid()) return false;
719
720 job_handle_->Cancel();
721 DCHECK(current_job_trace_id_.has_value());
722 TRACE_GC_NOTE_WITH_FLOW(garbage_collector_ == GarbageCollector::MARK_COMPACTOR
723 ? "Major concurrent marking paused"
724 : "Minor concurrent marking paused",
725 current_job_trace_id_.value(),
726 TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
727 return true;
728}
729
730bool ConcurrentMarking::IsStopped() {
731 if (!v8_flags.concurrent_marking && !v8_flags.parallel_marking) return true;
732
733 return !job_handle_ || !job_handle_->IsValid();
734}

736void ConcurrentMarking::Resume() {
737 DCHECK(garbage_collector_.has_value());
738 DCHECK(current_job_trace_id_.has_value());
739 TRACE_GC_NOTE_WITH_FLOW(garbage_collector_ == GarbageCollector::MARK_COMPACTOR
740 ? "Major concurrent marking resumed"
741 : "Minor concurrent marking resumed",
742 current_job_trace_id_.value(),
743 TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
744 RescheduleJobIfNeeded(garbage_collector_.value());
745}

755void ConcurrentMarking::FlushMemoryChunkData() {
756 DCHECK(!job_handle_ || !job_handle_->IsValid());
757 for (size_t i = 1; i < task_state_.size(); i++) {
758 MemoryChunkDataMap& memory_chunk_data = task_state_[i]->memory_chunk_data;
759 for (auto& pair : memory_chunk_data) {
760 // ClearLiveness sets the live bytes to zero.
761 // Pages with zero live bytes might be already unmapped.
762 MutablePageMetadata* memory_chunk = pair.first;
763 MemoryChunkData& data = pair.second;
764 if (data.live_bytes) {
765 memory_chunk->IncrementLiveBytesAtomically(data.live_bytes);
766 }
767 if (data.typed_slots) {
768 RememberedSet<OLD_TO_OLD>::MergeTyped(memory_chunk,
769 std::move(data.typed_slots));
770 }
771 }
772 memory_chunk_data.clear();
773 task_state_[i]->marked_bytes = 0;
774 }
775 total_marked_bytes_ = 0;
776}

778void ConcurrentMarking::ClearMemoryChunkData(MutablePageMetadata* chunk) {
779 DCHECK(!job_handle_ || !job_handle_->IsValid());
780 for (size_t i = 1; i < task_state_.size(); i++) {
781 task_state_[i]->memory_chunk_data.erase(chunk);
782 }
783}

785size_t ConcurrentMarking::TotalMarkedBytes() {
786 size_t result = 0;
787 for (size_t i = 1; i < task_state_.size(); i++) {
788 result +=
789 base::AsAtomicWord::Relaxed_Load<size_t>(&task_state_[i]->marked_bytes);
790 }
791 result += total_marked_bytes_;
792 return result;
793}
