101bool Type::IsSubtypeOf(const Type* supertype) const {
102 if (supertype->IsTopType()) return true;
103 if (IsNever()) return true;
104 if (const UnionType* union_type = UnionType::DynamicCast(supertype)) {
105 return union_type->IsSupertypeOf(this);
106 }
107 const Type* subtype = this;
108 while (subtype != nullptr) {
109 if (subtype == supertype) return true;
110 subtype = subtype->parent();
111 }
112 return false;

152const Type* Type::CommonSupertype(const Type* a, const Type* b) {
153 int diff = a->Depth() - b->Depth();
154 const Type* a_supertype = a;
155 const Type* b_supertype = b;
156 for (; diff > 0; --diff) a_supertype = a_supertype->parent();
157 for (; diff < 0; ++diff) b_supertype = b_supertype->parent();
158 while (a_supertype && b_supertype) {
159 if (a_supertype == b_supertype) return a_supertype;
160 a_supertype = a_supertype->parent();
161 b_supertype = b_supertype->parent();
162 }
163 ReportError("types " + a->ToString() + " and " + b->ToString() +
164 " have no common supertype");
165}

322 // For non-tagged unions, use the superclass GetConstexprGeneratedTypeName.
323 for (const Type* t : types_) {
324 if (!t->IsSubtypeOf(TypeOracle::GetTaggedType())) {
325 return this->Type::GetConstexprGeneratedTypeName();

333 if (simple_name == "Numeric") return simple_name;

401void AggregateType::CheckForDuplicateFields() const {
402 // Check the aggregate hierarchy and currently defined class for duplicate
403 // field declarations.
404 auto hierarchy = GetHierarchy();
405 std::map<std::string, const AggregateType*> field_names;
406 for (const AggregateType* aggregate_type : hierarchy) {
407 for (const Field& field : aggregate_type->fields()) {
408 const std::string& field_name = field.name_and_type.name;
409 auto i = field_names.find(field_name);
410 if (i != field_names.end()) {
411 CurrentSourcePosition::Scope current_source_position(field.pos);
412 std::string aggregate_type_name =
413 aggregate_type->IsClassType() ? "class" : "struct";
414 if (i->second == this) {
415 ReportError(aggregate_type_name, " '", name(),
416 "' declares a field with the name '", field_name,
417 "' more than once");
418 } else {
419 ReportError(aggregate_type_name, " '", name(),
420 "' declares a field with the name '", field_name,
421 "' that masks an inherited field from class '",
422 i->second->name(), "'");
423 }
424 }
425 field_names[field_name] = aggregate_type;
426 }
427 }
428}

430std::vector<const AggregateType*> AggregateType::GetHierarchy() const {
431 if (!is_finalized_) Finalize();
432 std::vector<const AggregateType*> hierarchy;
433 const AggregateType* current_container_type = this;
434 while (current_container_type != nullptr) {
435 hierarchy.push_back(current_container_type);
436 current_container_type =
437 current_container_type->IsClassType()
438 ? ClassType::cast(current_container_type)->GetSuperClass()
439 : nullptr;
440 }
441 std::reverse(hierarchy.begin(), hierarchy.end());
442 return hierarchy;
443}

675void ComputeSlotKindsHelper(std::vector<ObjectSlotKind>* slots,
676 size_t start_offset,
677 const std::vector<Field>& fields) {
678 size_t offset = start_offset;
679 for (const Field& field : fields) {
680 size_t field_size = std::get<0>(field.GetFieldSizeInformation());
681 // Support optional padding fields.
682 if (field_size == 0) continue;
683 size_t slot_index = offset / TargetArchitecture::TaggedSize();
684 // Rounding-up division to find the number of slots occupied by all the
685 // fields up to and including the current one.
686 size_t used_slots =
687 (offset + field_size + TargetArchitecture::TaggedSize() - 1) /
688 TargetArchitecture::TaggedSize();
689 while (used_slots > slots->size()) {
690 slots->push_back(ObjectSlotKind::kNoPointer);
691 }
692 const Type* type = field.name_and_type.type;
693 if (auto struct_type = type->StructSupertype()) {
694 ComputeSlotKindsHelper(slots, offset, (*struct_type)->fields());
695 } else {
696 ObjectSlotKind kind;
697 if (type->IsSubtypeOf(TypeOracle::GetObjectType())) {
698 if (field.custom_weak_marking) {
699 kind = ObjectSlotKind::kCustomWeakPointer;
700 } else {
701 kind = ObjectSlotKind::kStrongPointer;
702 }
703 } else if (type->IsSubtypeOf(TypeOracle::GetTaggedType())) {
704 DCHECK(!field.custom_weak_marking);
705 kind = ObjectSlotKind::kMaybeObjectPointer;
706 } else {
707 kind = ObjectSlotKind::kNoPointer;
708 }
709 DCHECK(slots->at(slot_index) == ObjectSlotKind::kNoPointer);
710 slots->at(slot_index) = kind;
711 }
712
713 offset += field_size;
714 }
715}

718std::vector<ObjectSlotKind> ClassType::ComputeHeaderSlotKinds() const {
719 std::vector<ObjectSlotKind> result;
720 std::vector<Field> header_fields = ComputeHeaderFields();

789void ClassType::GenerateAccessors() {
790 bool at_or_after_indexed_field = false;
791 if (const ClassType* parent = GetSuperClass()) {
792 at_or_after_indexed_field = parent->HasIndexedFieldsIncludingInParents();
793 }
794 // For each field, construct AST snippets that implement a CSA accessor
795 // function. The implementation iterator will turn the snippets into code.
796 for (size_t field_index = 0; field_index < fields_.size(); ++field_index) {
797 Field& field = fields_[field_index];
798 if (field.name_and_type.type == TypeOracle::GetVoidType()) {
799 continue;
800 }
801 at_or_after_indexed_field =
802 at_or_after_indexed_field || field.index.has_value();
803 CurrentSourcePosition::Scope position_activator(field.pos);
804
805 IdentifierExpression* parameter = MakeIdentifierExpression("o");
806 IdentifierExpression* index = MakeIdentifierExpression("i");
807
808 std::string camel_field_name = CamelifyString(field.name_and_type.name);
809
810 if (at_or_after_indexed_field) {
811 if (!field.index.has_value()) {
812 // There's no fundamental reason we couldn't generate functions to get
813 // references instead of slices, but it's not yet implemented.
814 ReportError(
815 "Torque doesn't yet support non-indexed fields after indexed "
816 "fields");
817 }
818
819 GenerateSliceAccessor(field_index);
820 }
821
822 // For now, only generate indexed accessors for simple types
823 if (field.index.has_value() && field.name_and_type.type->IsStructType()) {
824 continue;
825 }
826
827 // An explicit index is only used for indexed fields not marked as optional.
828 // Optional fields implicitly load or store item zero.
829 bool use_index = field.index && !field.index->optional;
830
831 // Load accessor
832 std::string load_macro_name = "Load" + this->name() + camel_field_name;
833 Signature load_signature;
834 load_signature.parameter_names.push_back(MakeNode<Identifier>("o"));
835 load_signature.parameter_types.types.push_back(this);
836 if (use_index) {
837 load_signature.parameter_names.push_back(MakeNode<Identifier>("i"));
838 load_signature.parameter_types.types.push_back(
839 TypeOracle::GetIntPtrType());
840 }
841 load_signature.parameter_types.var_args = false;
842 load_signature.return_type = field.name_and_type.type;
843
844 Expression* load_expression =
845 MakeFieldAccessExpression(parameter, field.name_and_type.name);
846 if (use_index) {
847 load_expression =
848 MakeNode<ElementAccessExpression>(load_expression, index);
849 }
850 Statement* load_body = MakeNode<ReturnStatement>(load_expression);
851 Declarations::DeclareMacro(load_macro_name, true, std::nullopt,
852 load_signature, load_body, std::nullopt);
853
854 // Store accessor
855 if (!field.const_qualified) {
856 IdentifierExpression* value = MakeIdentifierExpression("v");
857 std::string store_macro_name = "Store" + this->name() + camel_field_name;
858 Signature store_signature;
859 store_signature.parameter_names.push_back(MakeNode<Identifier>("o"));
860 store_signature.parameter_types.types.push_back(this);
861 if (use_index) {
862 store_signature.parameter_names.push_back(MakeNode<Identifier>("i"));
863 store_signature.parameter_types.types.push_back(
864 TypeOracle::GetIntPtrType());
865 }
866 store_signature.parameter_names.push_back(MakeNode<Identifier>("v"));
867 store_signature.parameter_types.types.push_back(field.name_and_type.type);
868 store_signature.parameter_types.var_args = false;
869 // TODO(danno): Store macros probably should return their value argument
870 store_signature.return_type = TypeOracle::GetVoidType();
871 Expression* store_expression =
872 MakeFieldAccessExpression(parameter, field.name_and_type.name);
873 if (use_index) {
874 store_expression =
875 MakeNode<ElementAccessExpression>(store_expression, index);
876 }
877 Statement* store_body = MakeNode<ExpressionStatement>(
878 MakeNode<AssignmentExpression>(store_expression, value));
879 Declarations::DeclareMacro(store_macro_name, true, std::nullopt,
880 store_signature, store_body, std::nullopt,
881 false);
882 }
883 }

886void ClassType::GenerateSliceAccessor(size_t field_index) {
887 // Generate a Torque macro for getting a Slice to this field. This macro can
888 // be called by the dot operator for this field. In Torque, this function for
889 // class "ClassName" and field "field_name" and field type "FieldType" would
890 // be written as one of the following:
891 //
892 // If the field has a known offset (in this example, 16):
893 // FieldSliceClassNameFieldName(o: ClassName) {
894 // return torque_internal::unsafe::New{Const,Mutable}Slice<FieldType>(
895 // /*object:*/ o,
896 // /*offset:*/ 16,
897 // /*length:*/ torque_internal::%IndexedFieldLength<ClassName>(
898 // o, "field_name")
899 // );
900 // }
901 //
902 // If the field has an unknown offset, and the previous field is named p, is
903 // not const, and is of type PType with size 4:
904 // FieldSliceClassNameFieldName(o: ClassName) {
905 // const previous = %FieldSlice<ClassName, MutableSlice<PType>>(o, "p");
906 // return torque_internal::unsafe::New{Const,Mutable}Slice<FieldType>(
907 // /*object:*/ o,
908 // /*offset:*/ previous.offset + 4 * previous.length,
909 // /*length:*/ torque_internal::%IndexedFieldLength<ClassName>(
910 // o, "field_name")
911 // );
912 // }
913 const Field& field = fields_[field_index];
914 std::string macro_name = GetSliceMacroName(field);
915 Signature signature;
916 Identifier* parameter_identifier = MakeNode<Identifier>("o");
917 signature.parameter_names.push_back(parameter_identifier);
918 signature.parameter_types.types.push_back(this);
919 signature.parameter_types.var_args = false;
920 signature.return_type =
921 field.const_qualified
922 ? TypeOracle::GetConstSliceType(field.name_and_type.type)
923 : TypeOracle::GetMutableSliceType(field.name_and_type.type);
924
925 std::vector<Statement*> statements;
926 Expression* offset_expression = nullptr;
927 IdentifierExpression* parameter =
928 MakeNode<IdentifierExpression>(parameter_identifier);
929
930 if (field.offset.has_value()) {
931 offset_expression =
932 MakeNode<IntegerLiteralExpression>(IntegerLiteral(*field.offset));
933 } else {
934 const Field* previous = GetFieldPreceding(field_index);
935 DCHECK_NOT_NULL(previous);
936
937 const Type* previous_slice_type =
938 previous->const_qualified
939 ? TypeOracle::GetConstSliceType(previous->name_and_type.type)
940 : TypeOracle::GetMutableSliceType(previous->name_and_type.type);
941
942 // %FieldSlice<ClassName, MutableSlice<PType>>(o, "p")
943 Expression* previous_expression = MakeCallExpression(
944 MakeIdentifierExpression(
945 {"torque_internal"}, "%FieldSlice",
946 {MakeNode<PrecomputedTypeExpression>(this),
947 MakeNode<PrecomputedTypeExpression>(previous_slice_type)}),
948 {parameter, MakeNode<StringLiteralExpression>(
949 StringLiteralQuote(previous->name_and_type.name))});
950
951 // const previous = %FieldSlice<ClassName, MutableSlice<PType>>(o, "p");
952 Statement* define_previous =
953 MakeConstDeclarationStatement("previous", previous_expression);
954 statements.push_back(define_previous);
955
956 // 4
957 size_t previous_element_size;
958 std::tie(previous_element_size, std::ignore) =
959 *SizeOf(previous->name_and_type.type);
960 Expression* previous_element_size_expression =
961 MakeNode<IntegerLiteralExpression>(
962 IntegerLiteral(previous_element_size));
963
964 // previous.length
965 Expression* previous_length_expression = MakeFieldAccessExpression(
966 MakeIdentifierExpression("previous"), "length");
967
968 // previous.offset
969 Expression* previous_offset_expression = MakeFieldAccessExpression(
970 MakeIdentifierExpression("previous"), "offset");
971
972 // 4 * previous.length
973 // In contrast to the code used for allocation, we don't need overflow
974 // checks here because we already know all the offsets fit into memory.
975 offset_expression = MakeCallExpression(
976 "*", {previous_element_size_expression, previous_length_expression});
977
978 // previous.offset + 4 * previous.length
979 offset_expression = MakeCallExpression(
980 "+", {previous_offset_expression, offset_expression});
981 }
982
983 // torque_internal::%IndexedFieldLength<ClassName>(o, "field_name")
984 Expression* length_expression = MakeCallExpression(
985 MakeIdentifierExpression({"torque_internal"}, "%IndexedFieldLength",
986 {MakeNode<PrecomputedTypeExpression>(this)}),
987 {parameter, MakeNode<StringLiteralExpression>(
988 StringLiteralQuote(field.name_and_type.name))});
989
990 // torque_internal::unsafe::New{Const,Mutable}Slice<FieldType>(
991 // /*object:*/ o,
992 // /*offset:*/ <<offset_expression>>,
993 // /*length:*/ torque_internal::%IndexedFieldLength<ClassName>(
994 // o, "field_name")
995 // )
996 IdentifierExpression* new_struct = MakeIdentifierExpression(
997 {"torque_internal", "unsafe"},
998 field.const_qualified ? "NewConstSlice" : "NewMutableSlice",
999 {MakeNode<PrecomputedTypeExpression>(field.name_and_type.type)});
1000 Expression* slice_expression = MakeCallExpression(
1001 new_struct, {parameter, offset_expression, length_expression});
1002
1003 statements.push_back(MakeNode<ReturnStatement>(slice_expression));
1004 Statement* block =
1005 MakeNode<BlockStatement>(/*deferred=*/false, std::move(statements));
1006
1007 Macro* macro = Declarations::DeclareMacro(macro_name, true, std::nullopt,
1008 signature, block, std::nullopt);
1009 if (this->ShouldGenerateCppObjectLayoutDefinitionAsserts()) {
1010 GlobalContext::EnsureInCCDebugOutputList(TorqueMacro::cast(macro),
1011 macro->Position().source);
1012 } else {
1013 GlobalContext::EnsureInCCOutputList(TorqueMacro::cast(macro),
1014 macro->Position().source);
1015 }
1016}

1308 std::string size_string;
1309 size_t size;
1310 if (type->IsSubtypeOf(TypeOracle::GetTaggedType())) {
1311 size = TargetArchitecture::TaggedSize();
1312 size_string = "kTaggedSize";
1313 } else if (type->IsSubtypeOf(TypeOracle::GetRawPtrType())) {
1314 size = TargetArchitecture::RawPtrSize();
1315 size_string = "kSystemPointerSize";
1316 } else if (type->IsSubtypeOf(TypeOracle::GetExternalPointerType())) {
1317 size = TargetArchitecture::ExternalPointerSize();
1318 size_string = "kExternalPointerSlotSize";
1319 } else if (type->IsSubtypeOf(TypeOracle::GetCppHeapPointerType())) {
1320 size = TargetArchitecture::CppHeapPointerSize();
1321 size_string = "kCppHeapPointerSlotSize";
1322 } else if (type->IsSubtypeOf(TypeOracle::GetTrustedPointerType())) {
1323 size = TargetArchitecture::TrustedPointerSize();
1324 size_string = "kTrustedPointerSize";
1325 } else if (type->IsSubtypeOf(TypeOracle::GetProtectedPointerType())) {
1326 size = TargetArchitecture::ProtectedPointerSize();
1327 size_string = "kTaggedSize";
1328 } else if (type->IsSubtypeOf(TypeOracle::GetVoidType())) {
1329 size = 0;
1330 size_string = "0";
1331 } else if (type->IsSubtypeOf(TypeOracle::GetInt8Type())) {
1332 size = kUInt8Size;
1333 size_string = "kUInt8Size";
1334 } else if (type->IsSubtypeOf(TypeOracle::GetUint8Type())) {
1335 size = kUInt8Size;
1336 size_string = "kUInt8Size";
1337 } else if (type->IsSubtypeOf(TypeOracle::GetInt16Type())) {
