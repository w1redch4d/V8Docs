25MaybeHandle<Object> Runtime::GetObjectProperty(
26 Isolate* isolate, DirectHandle<JSAny> lookup_start_object,
27 DirectHandle<Object> key, DirectHandle<JSAny> receiver, bool* is_found) {
28 if (receiver.is_null()) {
29 receiver = lookup_start_object;
30 }
31 if (IsNullOrUndefined(*lookup_start_object, isolate)) {
32 ErrorUtils::ThrowLoadFromNullOrUndefined(isolate, lookup_start_object, key);
33 return MaybeHandle<Object>();
34 }
35
36 bool success = false;
37 PropertyKey lookup_key(isolate, key, &success);
38 if (!success) return MaybeHandle<Object>();
39 LookupIterator it =
40 LookupIterator(isolate, receiver, lookup_key, lookup_start_object);
41
42 MaybeHandle<Object> result = Object::GetProperty(&it);
43 if (result.is_null()) {
44 return result;
45 }
46 if (is_found) {
47 *is_found = it.state() != LookupIterator::TYPED_ARRAY_INDEX_NOT_FOUND &&
48 it.state() != LookupIterator::NOT_FOUND;
49 }
50
51 return result;
52}

54MaybeHandle<Object> Runtime::HasProperty(Isolate* isolate,
55 DirectHandle<Object> object,
56 DirectHandle<Object> key) {
57 // Check that {object} is actually a receiver.

59 THROW_NEW_ERROR(

61 NewTypeError(MessageTemplate::kInvalidInOperatorUse, key, object));
62 }
63 DirectHandle<JSReceiver> receiver = Cast<JSReceiver>(object);
64
65 // Convert the {key} to a name.
66 DirectHandle<Name> name;
67 ASSIGN_RETURN_ON_EXCEPTION(isolate, name, Object::ToName(isolate, key));
68
69 // Lookup the {name} on {receiver}.
70 Maybe<bool> maybe = JSReceiver::HasProperty(isolate, receiver, name);
71 if (maybe.IsNothing()) return MaybeHandle<Object>();
72 return isolate->factory()->ToBoolean(maybe.FromJust());
73}

159// ES6 19.1.3.2
160RUNTIME_FUNCTION(Runtime_ObjectHasOwnProperty) {
161 HandleScope scope(isolate);
162 DirectHandle<Object> property = args.at(1);
163
164 // TODO(ishell): To improve performance, consider performing the to-string
165 // conversion of {property} before calling into the runtime.
166 bool success;
167 PropertyKey key(isolate, property, &success);
168 if (!success) return ReadOnlyRoots(isolate).exception();
169
170 DirectHandle<JSAny> object = args.at<JSAny>(0);
171
172 if (IsJSModuleNamespace(*object)) {
173 LookupIterator it(isolate, object, key, LookupIterator::OWN);
174 PropertyDescriptor desc;
175 Maybe<bool> result = JSReceiver::GetOwnPropertyDescriptor(&it, &desc);
176 if (!result.IsJust()) return ReadOnlyRoots(isolate).exception();
177 return isolate->heap()->ToBoolean(result.FromJust());
178
179 } else if (IsJSObject(*object)) {
180 DirectHandle<JSObject> js_obj = Cast<JSObject>(object);
181 // Fast case: either the key is a real named property or it is not
182 // an array index and there are no interceptors or hidden
183 // prototypes.
184 // TODO(jkummerow): Make JSReceiver::HasOwnProperty fast enough to
185 // handle all cases directly (without this custom fast path).
186 {
187 LookupIterator::Configuration c = LookupIterator::OWN_SKIP_INTERCEPTOR;
188 LookupIterator it(isolate, js_obj, key, js_obj, c);
189 Maybe<bool> maybe = JSReceiver::HasProperty(&it);
190 if (maybe.IsNothing()) return ReadOnlyRoots(isolate).exception();
191 DCHECK(!isolate->has_exception());
192 if (maybe.FromJust()) return ReadOnlyRoots(isolate).true_value();
193 }
194
195 Tagged<Map> map = js_obj->map();
196 if (!IsJSGlobalProxyMap(map) &&
197 (key.is_element() && key.index() <= JSObject::kMaxElementIndex
198 ? !map->has_indexed_interceptor()
199 : !map->has_named_interceptor())) {
200 return ReadOnlyRoots(isolate).false_value();
201 }
202
203 // Slow case.
204 LookupIterator it(isolate, js_obj, key, js_obj, LookupIterator::OWN);
205 Maybe<bool> maybe = JSReceiver::HasProperty(&it);
206 if (maybe.IsNothing()) return ReadOnlyRoots(isolate).exception();
207 DCHECK(!isolate->has_exception());
208 return isolate->heap()->ToBoolean(maybe.FromJust());
209
210 } else if (IsJSProxy(*object)) {
211 LookupIterator it(isolate, object, key, Cast<JSProxy>(object),
212 LookupIterator::OWN);
213 Maybe<PropertyAttributes> attributes =
214 JSReceiver::GetPropertyAttributes(&it);
215 if (attributes.IsNothing()) return ReadOnlyRoots(isolate).exception();
216 return isolate->heap()->ToBoolean(attributes.FromJust() != ABSENT);
217
218 } else if (IsString(*object)) {
219 return isolate->heap()->ToBoolean(
220 key.is_element()
221 ? key.index() < static_cast<size_t>(Cast<String>(*object)->length())
222 : key.name()->Equals(ReadOnlyRoots(isolate).length_string()));
223 } else if (IsNullOrUndefined(*object, isolate)) {
224 THROW_NEW_ERROR_RETURN_FAILURE(
225 isolate, NewTypeError(MessageTemplate::kUndefinedOrNullToObject));
226 }
227
228 return ReadOnlyRoots(isolate).false_value();
229}

296RUNTIME_FUNCTION(Runtime_AddPrivateBrand) {
297 HandleScope scope(isolate);
298 DCHECK_EQ(args.length(), 4);
299 DirectHandle<JSReceiver> receiver = args.at<JSReceiver>(0);
300 DirectHandle<Symbol> brand = args.at<Symbol>(1);
301 DirectHandle<Context> context = args.at<Context>(2);
302 int depth = args.smi_value_at(3);
303 DCHECK(brand->is_private_name());
304
305 LookupIterator it(isolate, receiver, brand, LookupIterator::OWN);
306
307 if (it.IsFound()) {
308 THROW_NEW_ERROR_RETURN_FAILURE(
309 isolate,
310 NewTypeError(MessageTemplate::kInvalidPrivateBrandReinitialization,
311 brand));
312 }
313
314 PropertyAttributes attributes =
315 static_cast<PropertyAttributes>(DONT_ENUM | DONT_DELETE | READ_ONLY);
316
317 // Look for the context in |depth| in the context chain to store it
318 // in the instance with the brand variable as key, which is needed by
319 // the debugger for retrieving names of private methods.
320 DCHECK_GE(depth, 0);
321 for (; depth > 0; depth--) {
322 context = direct_handle(
323 Cast<Context>(context->get(Context::PREVIOUS_INDEX)), isolate);
324 }
325 DCHECK_EQ(context->scope_info()->scope_type(), ScopeType::CLASS_SCOPE);
326 Maybe<bool> added_brand = Object::AddDataProperty(
327 &it, context, attributes, Just(kThrowOnError), StoreOrigin::kMaybeKeyed);
328 // Objects in shared space are fixed shape, so private symbols cannot be
329 // added.
330 if (V8_UNLIKELY(IsAlwaysSharedSpaceJSObject(*receiver))) {
331 CHECK(added_brand.IsNothing());
332 return ReadOnlyRoots(isolate).exception();
333 }
334 CHECK(added_brand.IsJust());
335 return *receiver;
336}

619RUNTIME_FUNCTION(Runtime_GetProperty) {
620 HandleScope scope(isolate);
621 DCHECK(args.length() == 3 || args.length() == 2);
622 DirectHandle<JSAny> lookup_start_obj = args.at<JSAny>(0);
623 DirectHandle<Object> key_obj = args.at(1);
624 DirectHandle<JSAny> receiver_obj = lookup_start_obj;
625 if (args.length() == 3) {
626 receiver_obj = args.at<JSAny>(2);
627 }
628
629 // Fast cases for getting named properties of the lookup_start_obj JSObject
630 // itself.
631 //
632 // The global proxy objects has to be excluded since LookupOwn on
633 // the global proxy object can return a valid result even though the
634 // global proxy object never has properties. This is the case
635 // because the global proxy object forwards everything to its hidden
636 // prototype including own lookups.
637 //
638 // Additionally, we need to make sure that we do not cache results
639 // for objects that require access checks.
640
641 // Convert string-index keys to their number variant to avoid internalization
642 // below; and speed up subsequent conversion to index.
643 uint32_t index;
644 if (IsString(*key_obj) && Cast<String>(*key_obj)->AsArrayIndex(&index)) {
645 key_obj = isolate->factory()->NewNumberFromUint(index);
646 }
647 if (IsJSObject(*lookup_start_obj)) {
648 DirectHandle<JSObject> lookup_start_object =
649 Cast<JSObject>(lookup_start_obj);
650 if (!IsJSGlobalProxy(*lookup_start_object) &&
651 !IsAccessCheckNeeded(*lookup_start_object) && IsName(*key_obj)) {
652 DirectHandle<Name> key = Cast<Name>(key_obj);
653 key_obj = key = isolate->factory()->InternalizeName(key);
654
655 DisallowGarbageCollection no_gc;
656 if (IsJSGlobalObject(*lookup_start_object)) {
657 // Attempt dictionary lookup.
658 Tagged<GlobalDictionary> dictionary =
659 Cast<JSGlobalObject>(*lookup_start_object)
660 ->global_dictionary(kAcquireLoad);
661 InternalIndex entry = dictionary->FindEntry(isolate, key);
662 if (entry.is_found()) {
663 Tagged<PropertyCell> cell = dictionary->CellAt(entry);
664 if (cell->property_details().kind() == PropertyKind::kData) {
665 Tagged<Object> value = cell->value();
666 if (!IsPropertyCellHole(value, isolate)) return value;
667 // If value is the hole (meaning, absent) do the general lookup.
668 }
669 }
670 } else if (!lookup_start_object->HasFastProperties()) {
671 // Attempt dictionary lookup.
672 if (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
673 Tagged<SwissNameDictionary> dictionary =
674 lookup_start_object->property_dictionary_swiss();
675 InternalIndex entry = dictionary->FindEntry(isolate, *key);
676 if (entry.is_found() &&
677 (dictionary->DetailsAt(entry).kind() == PropertyKind::kData)) {
678 return dictionary->ValueAt(entry);
679 }
680 } else {
681 Tagged<NameDictionary> dictionary =
682 lookup_start_object->property_dictionary();
683 InternalIndex entry = dictionary->FindEntry(isolate, key);
684 if ((entry.is_found()) &&
685 (dictionary->DetailsAt(entry).kind() == PropertyKind::kData)) {
686 return dictionary->ValueAt(entry);
687 }
688 }
689 }
690 } else if (IsSmi(*key_obj)) {
691 // JSObject without a name key. If the key is a Smi, check for a
692 // definite out-of-bounds access to elements, which is a strong indicator
693 // that subsequent accesses will also call the runtime. Proactively
694 // transition elements to FAST_*_ELEMENTS to avoid excessive boxing of
695 // doubles for those future calls in the case that the elements would
696 // become PACKED_DOUBLE_ELEMENTS.
697 ElementsKind elements_kind = lookup_start_object->GetElementsKind();
698 if (IsDoubleElementsKind(elements_kind)) {
699 if (Smi::ToInt(*key_obj) >= lookup_start_object->elements()->length()) {
700 elements_kind = IsHoleyElementsKind(elements_kind) ? HOLEY_ELEMENTS
701 : PACKED_ELEMENTS;
702 JSObject::TransitionElementsKind(lookup_start_object, elements_kind);
703 }
704 } else {
705 DCHECK(IsSmiOrObjectElementsKind(elements_kind) ||
706 !IsFastElementsKind(elements_kind));
707 }
708 }
709 } else if (IsString(*lookup_start_obj) && IsSmi(*key_obj)) {
710 // Fast case for string indexing using [] with a smi index.
711 DirectHandle<String> str = Cast<String>(lookup_start_obj);
712 uint32_t smi_index = Cast<Smi>(*key_obj).value();
713 if (smi_index < str->length()) {
714 Factory* factory = isolate->factory();
715 return *factory->LookupSingleCharacterStringFromCode(
716 String::Flatten(isolate, str)->Get(smi_index));
717 }
718 }
719
720 // Fall back to GetObjectProperty.
721 RETURN_RESULT_OR_FAILURE(
722 isolate, Runtime::GetObjectProperty(isolate, lookup_start_obj, key_obj,
723 receiver_obj));
724}

1137// Check that the excluded properties are within the stack range of the top of
1138// the stack, and the start of the JS frame.
1139void CheckExcludedPropertiesAreOnCallerStack(Isolate* isolate, Address base,
1140 int count) {
1141#ifdef DEBUG
1142 StackFrameIterator it(isolate);
1143
1144 // Don't need to check when there's no excluded properties.
1145 if (count == 0) return;
1146
1147 DCHECK(!it.done());
1148
1149 // Properties are pass in order on the stack, which means that their addresses
1150 // are in reverse order in memory (because stacks grow backwards). So, we
1151 // need to check if the _last_ property address is before the stack end...
1152 Address last_property = base - (count - 1) * kSystemPointerSize;
1153 DCHECK_GE(last_property, it.frame()->sp());
1154
1155 // ... and for the first JS frame, make sure the _first_ property address is
1156 // after that stack frame's start.
1157 for (; !it.done(); it.Advance()) {
1158 if (it.frame()->is_javascript()) {
1159 DCHECK_LT(base, it.frame()->fp());
1160 return;
1161 }
1162 }
1163
1164 // We should always find a JS frame.
1165 UNREACHABLE();
1166#endif
1167}

1171RUNTIME_FUNCTION(Runtime_CopyDataPropertiesWithExcludedPropertiesOnStack) {
1172 HandleScope scope(isolate);

1174 DirectHandle<Object> source = args.at(0);
1175 int excluded_property_count = args.smi_value_at(1);
1176 // The excluded_property_base is passed as a raw stack pointer. This is safe
1177 // because the stack pointer is aligned, so it looks like a Smi to the GC.
1178 Address* excluded_property_base = reinterpret_cast<Address*>(args[2].ptr());
1179 DCHECK(HAS_SMI_TAG(reinterpret_cast<intptr_t>(excluded_property_base)));
1180 // Also make sure that the given base pointer points to to on-stack values.
1181 CheckExcludedPropertiesAreOnCallerStack(
1182 isolate, reinterpret_cast<Address>(excluded_property_base),
1183 excluded_property_count);
1184
1185 // If source is undefined or null, throw a non-coercible error.
1186 if (IsNullOrUndefined(*source, isolate)) {
1187 return ErrorUtils::ThrowLoadFromNullOrUndefined(
1188 isolate, source, MaybeDirectHandle<Object>());
1189 }
1190
1191 DirectHandleVector<Object> excluded_properties(isolate,
1192 excluded_property_count);
1193 for (int i = 0; i < excluded_property_count; i++) {
1194 // Because the excluded properties on stack is from high address
1195 // to low address, so we need to use sub
1196 IndirectHandle<Object> property(excluded_property_base - i);
1197 uint32_t property_num;

1199 // properties resolving to numbers, which would've been strings
1200 // instead because of our call to %ToName() in the desugaring for
1201 // computed properties.
1202 if (IsString(*property) &&
1203 Cast<String>(*property)->AsArrayIndex(&property_num)) {
1204 property = isolate->factory()->NewNumberFromUint(property_num);
1205 }
1206
1207 excluded_properties[i] = property;
1208 }
1209

1211 isolate->factory()->NewJSObject(isolate->object_function());
1212 MAYBE_RETURN(
1213 JSReceiver::SetOrCopyDataProperties(
1214 isolate, target, source,
1215 PropertiesEnumerationMode::kPropertyAdditionOrder,

1368void CollectPrivateMethodsAndAccessorsFromContext(
1369 Isolate* isolate, DirectHandle<Context> context, DirectHandle<String> desc,
1370 Handle<Object> brand, IsStaticFlag is_static_flag,
1371 std::vector<PrivateMember>* results) {
1372 DirectHandle<ScopeInfo> scope_info(context->scope_info(), isolate);
1373 VariableLookupResult lookup_result;
1374 int context_index = scope_info->ContextSlotIndex(desc, &lookup_result);
1375 if (context_index == -1 ||
1376 !IsPrivateMethodOrAccessorVariableMode(lookup_result.mode) ||
1377 lookup_result.is_static_flag != is_static_flag) {
1378 return;
1379 }
1380
1381 Handle<Object> slot_value(context->get(context_index), isolate);
1382 DCHECK_IMPLIES(lookup_result.mode == VariableMode::kPrivateMethod,
1383 IsJSFunction(*slot_value));
1384 DCHECK_IMPLIES(lookup_result.mode != VariableMode::kPrivateMethod,
1385 IsAccessorPair(*slot_value));
1386 results->push_back({
1387 lookup_result.mode == VariableMode::kPrivateMethod
1388 ? PrivateMemberType::kPrivateMethod
1389 : PrivateMemberType::kPrivateAccessor,
1390 brand,
1391 slot_value,
1392 });

1395Maybe<bool> CollectPrivateMembersFromReceiver(
1396 Isolate* isolate, DirectHandle<JSReceiver> receiver,
1397 DirectHandle<String> desc, std::vector<PrivateMember>* results) {
1398 PropertyFilter key_filter =
1399 static_cast<PropertyFilter>(PropertyFilter::PRIVATE_NAMES_ONLY);
1400 DirectHandle<FixedArray> keys;
1401 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
1402 isolate, keys,
1403 KeyAccumulator::GetKeys(isolate, receiver, KeyCollectionMode::kOwnOnly,
1404 key_filter, GetKeysConversion::kConvertToString),
1405 Nothing<bool>());
1406
1407 if (IsJSFunction(*receiver)) {
1408 Handle<JSFunction> func(Cast<JSFunction>(*receiver), isolate);
1409 DirectHandle<SharedFunctionInfo> shared(func->shared(), isolate);
1410 if (shared->is_class_constructor() &&
1411 shared->has_static_private_methods_or_accessors()) {
1412 DirectHandle<Context> receiver_context(
1413 Cast<JSFunction>(*receiver)->context(), isolate);
1414 CollectPrivateMethodsAndAccessorsFromContext(
1415 isolate, receiver_context, desc, func, IsStaticFlag::kStatic,
1416 results);
1417 }
1418 }
1419
1420 for (int i = 0; i < keys->length(); ++i) {
1421 DirectHandle<Object> obj_key(keys->get(i), isolate);
1422 Handle<Symbol> symbol(Cast<Symbol>(*obj_key), isolate);
1423 CHECK(symbol->is_private_name());
1424 Handle<Object> value;
1425 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
1426 isolate, value, Object::GetProperty(isolate, receiver, symbol),
1427 Nothing<bool>());
1428
1429 if (symbol->is_private_brand()) {
1430 DirectHandle<Context> value_context(Cast<Context>(*value), isolate);
1431 CollectPrivateMethodsAndAccessorsFromContext(
1432 isolate, value_context, desc, symbol, IsStaticFlag::kNotStatic,
1433 results);
1434 } else {
1435 DirectHandle<String> symbol_desc(Cast<String>(symbol->description()),
1436 isolate);
1437 if (symbol_desc->Equals(*desc)) {
1438 results->push_back({
1439 PrivateMemberType::kPrivateField,
1440 symbol,
1441 value,
1442 });
1443 }
1444 }
1445 }
1446
1447 return Just(true);
1448}
