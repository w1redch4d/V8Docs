51// Find the frame pointer of the interpreter frame on the stack.
52Address FindInterpreterEntryFramePointer(Isolate* isolate) {
53 StackFrameIterator it(isolate, isolate->thread_local_top());
54 // On top: C entry stub.
55 DCHECK_EQ(StackFrame::EXIT, it.frame()->type());
56 it.Advance();
57 // Next: the wasm interpreter entry.
58 DCHECK_EQ(StackFrame::WASM_INTERPRETER_ENTRY, it.frame()->type());
59 return it.frame()->fp();
60}

64RUNTIME_FUNCTION(Runtime_WasmRunInterpreter) {
65 DCHECK_EQ(3, args.length());
66 HandleScope scope(isolate);
67 Handle<WasmInstanceObject> instance = args.at<WasmInstanceObject>(0);
68 Handle<WasmTrustedInstanceData> trusted_data(instance->trusted_data(isolate),
69 isolate);
70 int32_t func_index = NumberToInt32(args[1]);
71 Handle<Object> arg_buffer_obj = args.at(2);
72
73 // The arg buffer is the raw pointer to the caller's stack. It looks like a
74 // Smi (lowest bit not set, as checked by IsSmi), but is no valid Smi. We just
75 // cast it back to the raw pointer.
76 CHECK(!IsHeapObject(*arg_buffer_obj));
77 CHECK(IsSmi(*arg_buffer_obj));
78 Address arg_buffer = (*arg_buffer_obj).ptr();
79
80 // Reserve buffers for argument and return values.
81 DCHECK_GT(trusted_data->module()->functions.size(), func_index);
82 const wasm::FunctionSig* sig =
83 trusted_data->module()->functions[func_index].sig;
84 DCHECK_GE(kMaxInt, sig->parameter_count());
85 int num_params = static_cast<int>(sig->parameter_count());
86 std::vector<wasm::WasmValue> wasm_args(num_params);
87 DCHECK_GE(kMaxInt, sig->return_count());
88 int num_returns = static_cast<int>(sig->return_count());
89 std::vector<wasm::WasmValue> wasm_rets(num_returns);
90
91 // Set the current isolate's context.
92 isolate->set_context(trusted_data->native_context());
93
94 // Make sure the WasmInterpreterObject and InterpreterHandle for this instance
95 // exist.
96 Handle<Tuple2> interpreter_object =
97 WasmTrustedInstanceData::GetOrCreateInterpreterObject(instance);
98 wasm::InterpreterHandle* interpreter_handle =
99 wasm::GetOrCreateInterpreterHandle(isolate, interpreter_object);
100
101 if (wasm::WasmBytecode::ContainsSimd(sig)) {
102 wasm::ClearThreadInWasmScope clear_wasm_flag(isolate);
103
104 interpreter_handle->SetTrapFunctionIndex(func_index);
105 isolate->Throw(*isolate->factory()->NewTypeError(
106 MessageTemplate::kWasmTrapJSTypeError));
107 return ReadOnlyRoots(isolate).exception();
108 }
109
110 Address frame_pointer = FindInterpreterEntryFramePointer(isolate);
111
112 // If there are Ref arguments or return values, we store their pointers into
113 // an array of bytes so we need to disable GC until they are unpacked by the
114 // callee.
115 {
116 DisallowHeapAllocation no_gc;
117
118 // Copy the arguments for the {arg_buffer} into a vector of {WasmValue}.
119 // This also boxes reference types into handles, which needs to happen
120 // before any methods that could trigger a GC are being called.
121 Address arg_buf_ptr = arg_buffer;
122 for (int i = 0; i < num_params; ++i) {
123#define CASE_ARG_TYPE(type, ctype) \
124 case wasm::type: \
125 DCHECK_EQ(wasm::ValueTypes::ElementSizeInBytes(sig->GetParam(i)), \
126 sizeof(ctype)); \
127 wasm_args[i] = \
128 wasm::WasmValue(base::ReadUnalignedValue<ctype>(arg_buf_ptr)); \
129 arg_buf_ptr += sizeof(ctype); \
130 break;
131
132 wasm::ValueType value_type = sig->GetParam(i);
133 wasm::ValueKind kind = value_type.kind();
134 switch (kind) {
135 CASE_ARG_TYPE(kWasmI32.kind(), uint32_t)
136 CASE_ARG_TYPE(kWasmI64.kind(), uint64_t)
137 CASE_ARG_TYPE(kWasmF32.kind(), float)
138 CASE_ARG_TYPE(kWasmF64.kind(), double)
139#undef CASE_ARG_TYPE
140 case wasm::kWasmRefString.kind():
141 case wasm::kWasmAnyRef.kind(): {
142 const bool anyref = (kind == wasm::kWasmAnyRef.kind());
143 DCHECK_EQ(wasm::ValueTypes::ElementSizeInBytes(sig->GetParam(i)),
144 kSystemPointerSize);
145 // MarkCompactCollector::RootMarkingVisitor requires ref slots to be
146 // 64-bit aligned.
147 arg_buf_ptr += (arg_buf_ptr & 0x04);
148
149 Handle<Object> ref(
150 base::ReadUnalignedValue<Tagged<Object>>(arg_buf_ptr), isolate);
151
152 const wasm::WasmInterpreterRuntime* wasm_runtime =
153 interpreter_handle->interpreter()->GetWasmRuntime();
154 ref = wasm_runtime->JSToWasmObject(ref, value_type);
155 if (isolate->has_exception()) {
156 interpreter_handle->SetTrapFunctionIndex(func_index);
157 return ReadOnlyRoots(isolate).exception();
158 }
159
160 if ((value_type != wasm::kWasmExternRef &&
161 value_type != wasm::kWasmNullExternRef) &&
162 IsNull(*ref, isolate)) {
163 ref = isolate->factory()->wasm_null();
164 }
165
166 wasm_args[i] = wasm::WasmValue(
167 ref, anyref ? wasm::kWasmAnyRef : wasm::kWasmRefString,
168 trusted_data->module());
169 arg_buf_ptr += kSystemPointerSize;
170 break;
171 }
172 case wasm::kWasmS128.kind():
173 default:
174 UNREACHABLE();
175 }
176 }
177
178 // Run the function in the interpreter. Note that neither the
179 // {WasmInterpreterObject} nor the {InterpreterHandle} have to exist,
180 // because interpretation might have been triggered by another Isolate
181 // sharing the same WasmEngine.
182 bool success = WasmInterpreterObject::RunInterpreter(
183 isolate, frame_pointer, instance, func_index, wasm_args, wasm_rets);
184
185 // Early return on failure.
186 if (!success) {
187 DCHECK(isolate->has_exception());
188 return ReadOnlyRoots(isolate).exception();
189 }
190
191 // Copy return values from the vector of {WasmValue} into {arg_buffer}. This
192 // also un-boxes reference types from handles into raw pointers.
193 arg_buf_ptr = arg_buffer;
194
195 for (int i = 0; i < num_returns; ++i) {
196#define CASE_RET_TYPE(type, ctype) \
197 case wasm::type: \
198 DCHECK_EQ(wasm::ValueTypes::ElementSizeInBytes(sig->GetReturn(i)), \
199 sizeof(ctype)); \
200 base::WriteUnalignedValue<ctype>(arg_buf_ptr, wasm_rets[i].to<ctype>()); \
201 arg_buf_ptr += sizeof(ctype); \
202 break;
203
204 switch (sig->GetReturn(i).kind()) {
205 CASE_RET_TYPE(kWasmI32.kind(), uint32_t)
206 CASE_RET_TYPE(kWasmI64.kind(), uint64_t)
207 CASE_RET_TYPE(kWasmF32.kind(), float)
208 CASE_RET_TYPE(kWasmF64.kind(), double)
209#undef CASE_RET_TYPE
210 case wasm::kWasmRefString.kind():
211 case wasm::kWasmAnyRef.kind(): {
212 DCHECK_EQ(wasm::ValueTypes::ElementSizeInBytes(sig->GetReturn(i)),
213 kSystemPointerSize);
214 Handle<Object> ref = wasm_rets[i].to_ref();
215 // Note: WasmToJSObject(ref) already called in ContinueExecution or
216 // CallExternalJSFunction.
217
218 // Make sure ref slots are 64-bit aligned.
219 arg_buf_ptr += (arg_buf_ptr & 0x04);
220 base::WriteUnalignedValue<Tagged<Object>>(arg_buf_ptr, *ref);
221 arg_buf_ptr += kSystemPointerSize;
222 break;
223 }
224 case wasm::kWasmS128.kind():
225 default:
226 UNREACHABLE();
227 }
228 }
229
230 return ReadOnlyRoots(isolate).undefined_value();
231 }

245V8_EXPORT_PRIVATE InterpreterHandle* GetOrCreateInterpreterHandle(
246 Isolate* isolate, Handle<Tuple2> interpreter_object) {
247 Handle<Object> handle(
248 WasmInterpreterObject::get_interpreter_handle(*interpreter_object),
249 isolate);
250 if (IsUndefined(*handle, isolate)) {
251 // Use the maximum stack size to estimate the maximum size of the
252 // interpreter. The interpreter keeps its own stack internally, and the size
253 // of the stack should dominate the overall size of the interpreter. We
254 // multiply by '2' to account for the growing strategy for the backing store
255 // of the stack.
256 size_t interpreter_size = v8_flags.stack_size * KB * 2;
257 handle = Managed<InterpreterHandle>::From(
258 isolate, interpreter_size,
259 std::make_shared<InterpreterHandle>(isolate, interpreter_object));
260 WasmInterpreterObject::set_interpreter_handle(*interpreter_object, *handle);
261 }
262
263 return Cast<Managed<InterpreterHandle>>(handle)->raw();
264}

266// A helper for an entry in an indirect function table (IFT).
267// The underlying storage in the instance is used by generated code to
268// call functions indirectly at runtime.
269// Each entry has the following fields:
270// - implicit_arg = A WasmTrustedInstanceData or a WasmImportData.
271// - sig_id = signature id of function.
272// - target = entrypoint to Wasm code or import wrapper code.
273// - function_index = function index, if a Wasm function, or
274// WasmDispatchTable::kInvalidFunctionIndex otherwise.
275class IndirectFunctionTableEntry {
276 public:
277 inline IndirectFunctionTableEntry(Handle<WasmInstanceObject>, int table_index,
278 int entry_index);
279
280 inline Tagged<Object> implicit_arg() const {
281 return table_->implicit_arg(index_);
282 }
283 inline wasm::CanonicalTypeIndex sig_id() const { return table_->sig(index_); }
284 inline WasmCodePointer target() const { return table_->target(index_); }
285 inline uint32_t function_index() const {
286 return table_->function_index(index_);
287 }
288
289 private:
290 Handle<WasmDispatchTable> const table_;
291 int const index_;
292};

573// Unpack the values encoded in the given exception. The exception values are
574// pushed onto the operand stack.
575void WasmInterpreterRuntime::UnpackException(
576 uint32_t* sp, const WasmTag& tag, Handle<Object> exception_object,
577 uint32_t first_param_slot_index, uint32_t first_param_ref_stack_index) {
578 Handle<FixedArray> encoded_values =
579 Cast<FixedArray>(WasmExceptionPackage::GetExceptionValues(
580 isolate_, Cast<WasmExceptionPackage>(exception_object)));
581 // Decode the exception values from the given exception package and push
582 // them onto the operand stack. This encoding has to be in sync with other
583 // backends so that exceptions can be passed between them.
584 const WasmTagSig* sig = tag.sig;
585 uint32_t encoded_index = 0;
586 uint32_t* p = sp + first_param_slot_index;
587 for (size_t i = 0; i < sig->parameter_count(); ++i) {
588#ifdef V8_ENABLE_DRUMBRAKE_TRACING
589 if (v8_flags.trace_drumbrake_execution) {
590 TracePush(sig->GetParam(i).kind(), static_cast<uint32_t>(p - sp));
591 }
592#endif // V8_ENABLE_DRUMBRAKE_TRACING
593
594 WasmValue value;
595 switch (sig->GetParam(i).kind()) {
596 case kI32: {
597 uint32_t u32 = 0;
598 DecodeI32ExceptionValue(encoded_values, &encoded_index, &u32);
599 base::WriteUnalignedValue<uint32_t>(reinterpret_cast<Address>(p), u32);
600 p += sizeof(uint32_t) / kSlotSize;
601 break;
602 }
603 case kF32: {
604 uint32_t f32_bits = 0;
605 DecodeI32ExceptionValue(encoded_values, &encoded_index, &f32_bits);
606 float f32 = Float32::FromBits(f32_bits).get_scalar();
607 base::WriteUnalignedValue<float>(reinterpret_cast<Address>(p), f32);
608 p += sizeof(float) / kSlotSize;
609 break;
610 }
611 case kI64: {
612 uint64_t u64 = 0;
613 DecodeI64ExceptionValue(encoded_values, &encoded_index, &u64);
614 base::WriteUnalignedValue<uint64_t>(reinterpret_cast<Address>(p), u64);
615 p += sizeof(uint64_t) / kSlotSize;
616 break;
617 }
618 case kF64: {
619 uint64_t f64_bits = 0;
620 DecodeI64ExceptionValue(encoded_values, &encoded_index, &f64_bits);
621 float f64 = Float64::FromBits(f64_bits).get_scalar();
622 base::WriteUnalignedValue<double>(reinterpret_cast<Address>(p), f64);
623 p += sizeof(double) / kSlotSize;
624 break;
625 }
626 case kS128: {
627 int32x4 s128 = {0, 0, 0, 0};
628 uint32_t* vals = reinterpret_cast<uint32_t*>(s128.val);
629 DecodeI32ExceptionValue(encoded_values, &encoded_index, &vals[0]);
630 DecodeI32ExceptionValue(encoded_values, &encoded_index, &vals[1]);
631 DecodeI32ExceptionValue(encoded_values, &encoded_index, &vals[2]);
632 DecodeI32ExceptionValue(encoded_values, &encoded_index, &vals[3]);
633 base::WriteUnalignedValue<Simd128>(reinterpret_cast<Address>(p),
634 Simd128(s128));
635 p += sizeof(Simd128) / kSlotSize;
636 break;
637 }
638 case kRef:
639 case kRefNull: {
640 Handle<Object> ref(encoded_values->get(encoded_index++), isolate_);
641 if (sig->GetParam(i).value_type_code() == wasm::kFuncRefCode &&
642 i::IsNull(*ref, isolate_)) {
643 ref = isolate_->factory()->wasm_null();
644 }
645 StoreWasmRef(first_param_ref_stack_index++, ref);
646 base::WriteUnalignedValue<WasmRef>(reinterpret_cast<Address>(p), ref);
647 p += sizeof(WasmRef) / kSlotSize;
648 break;
649 }
650 default:
651 UNREACHABLE();
652 }
653 }
654 DCHECK_EQ(WasmExceptionPackage::GetEncodedSize(&tag), encoded_index);
655}

709// Handle a thrown exception. Returns whether the exception was handled inside
710// of wasm. Unwinds the interpreted stack accordingly.
711WasmInterpreterThread::ExceptionHandlingResult
712WasmInterpreterRuntime::HandleException(uint32_t* sp,
713 const uint8_t*& current_code) {
714 DCHECK_IMPLIES(current_code, current_frame_.current_function_);
715 DCHECK_IMPLIES(!current_code, !current_frame_.current_function_);
716 DCHECK(isolate_->has_exception());
717
718 bool catchable = current_frame_.current_function_ &&
719 isolate_->is_catchable_by_wasm(isolate_->exception());
720 if (catchable) {
721 HandleScope scope(isolate_);
722 Handle<Object> exception = handle(isolate_->exception(), isolate_);
723 Tagged<WasmTrustedInstanceData> trusted_data =
724 *wasm_trusted_instance_data();
725
726 // We might need to allocate a new FixedArray<Object> to store the caught
727 // exception.
728 DCHECK(AllowHeapAllocation::IsAllowed());
729
730 size_t current_code_offset =
731 current_code - current_frame_.current_function_->GetCode();
732 const WasmEHData::TryBlock* try_block =
733 current_frame_.current_function_->GetTryBlock(current_code_offset);
734 while (try_block) {
735 for (const auto& catch_handler : try_block->catch_handlers) {
736 if (catch_handler.tag_index < 0) {
737 // Catch all.
738 current_code = current_frame_.current_function_->GetCode() +
739 catch_handler.code_offset;
740 current_frame_.SetCaughtException(
741 isolate_, catch_handler.catch_block_index, exception);
742 isolate_->clear_exception();
743 return WasmInterpreterThread::HANDLED;
744 } else if (IsWasmExceptionPackage(*exception, isolate_)) {
745 // The exception was thrown by Wasm code and it's wrapped in a
746 // WasmExceptionPackage.
747 Handle<Object> caught_tag = WasmExceptionPackage::GetExceptionTag(
748 isolate_, Cast<WasmExceptionPackage>(exception));
749 Handle<Object> expected_tag =
750 handle(trusted_data->tags_table()->get(catch_handler.tag_index),
751 isolate_);
752 DCHECK(IsWasmExceptionTag(*expected_tag));
753 // Determines whether the given exception has a tag matching the
754 // expected tag for the given index within the exception table of the
755 // current instance.
756 if (expected_tag.is_identical_to(caught_tag)) {
757 current_code = current_frame_.current_function_->GetCode() +
758 catch_handler.code_offset;
759 DCHECK_LT(catch_handler.tag_index, module_->tags.size());
760 const WasmTag& tag = module_->tags[catch_handler.tag_index];
761 auto exception_payload_slot_offsets =
762 current_frame_.current_function_
763 ->GetExceptionPayloadStartSlotOffsets(
764 catch_handler.catch_block_index);
765 UnpackException(
766 sp, tag, exception,
767 exception_payload_slot_offsets.first_param_slot_offset,
768 exception_payload_slot_offsets.first_param_ref_stack_index);
769 current_frame_.SetCaughtException(
770 isolate_, catch_handler.catch_block_index, exception);
771 isolate_->clear_exception();
772 return WasmInterpreterThread::HANDLED;
773 }
774 } else {
775 // Check for the special case where the tag is WebAssembly.JSTag and
776 // the exception is not a WebAssembly.Exception. In this case the
777 // exception is caught and pushed on the operand stack.
778 // Only perform this check if the tag signature is the same as
779 // the JSTag signature, i.e. a single externref, otherwise we know
780 // statically that it cannot be the JSTag.
781 DCHECK_LT(catch_handler.tag_index, module_->tags.size());
782 const WasmTagSig* sig = module_->tags[catch_handler.tag_index].sig;
783 if (sig->return_count() != 0 || sig->parameter_count() != 1 ||
784 (sig->GetParam(0).kind() != kRefNull &&
785 sig->GetParam(0).kind() != kRef)) {
786 continue;
787 }
788
789 Handle<JSObject> js_tag_object =
790 handle(isolate_->native_context()->wasm_js_tag(), isolate_);
791 Handle<WasmTagObject> wasm_tag_object(
792 Cast<WasmTagObject>(*js_tag_object), isolate_);
793 Handle<Object> caught_tag = handle(wasm_tag_object->tag(), isolate_);
794 Handle<Object> expected_tag =
795 handle(trusted_data->tags_table()->get(catch_handler.tag_index),
796 isolate_);
797 if (!expected_tag.is_identical_to(caught_tag)) {
798 continue;
799 }
800
801 current_code = current_frame_.current_function_->GetCode() +
802 catch_handler.code_offset;
803 // Push exception on the operand stack.
804 auto exception_payload_slot_offsets =
805 current_frame_.current_function_
806 ->GetExceptionPayloadStartSlotOffsets(
807 catch_handler.catch_block_index);
808 StoreWasmRef(
809 exception_payload_slot_offsets.first_param_ref_stack_index,
810 exception);
811 base::WriteUnalignedValue<WasmRef>(
812 reinterpret_cast<Address>(
813 sp + exception_payload_slot_offsets.first_param_slot_offset),
814 exception);
815
816 current_frame_.SetCaughtException(
817 isolate_, catch_handler.catch_block_index, exception);
818 isolate_->clear_exception();
819 return WasmInterpreterThread::HANDLED;
820 }
821 }
822 try_block =
823 current_frame_.current_function_->GetParentTryBlock(try_block);
824 }
825 }
826
827 DCHECK_NOT_NULL(current_thread_);
828 current_thread_->Unwinding();
829 return WasmInterpreterThread::UNWOUND;
830}

890void WasmInterpreterRuntime::BeginExecution(
891 WasmInterpreterThread* thread, uint32_t func_index, Address frame_pointer,
892 uint8_t* interpreter_fp, uint32_t ref_stack_offset,
893 const std::vector<WasmValue>* argument_values) {
894 current_thread_ = thread;
895 start_function_index_ = func_index;
896
897 thread->StartActivation(this, frame_pointer, interpreter_fp, current_frame_);
898
899 current_frame_.current_function_ = nullptr;
900 current_frame_.previous_frame_ = nullptr;
901 current_frame_.current_bytecode_ = nullptr;
902 current_frame_.current_sp_ = interpreter_fp;
903 current_frame_.ref_array_current_sp_ = ref_stack_offset;
904 current_frame_.thread_ = thread;
905#ifdef V8_ENABLE_DRUMBRAKE_TRACING
906 current_frame_.current_stack_start_args_ = thread->CurrentStackFrameStart();
907#endif // V8_ENABLE_DRUMBRAKE_TRACING
908
909 const FunctionSig* sig = module_->functions[func_index].sig;
910 size_t args_count = 0;
911 uint32_t rets_slots_size = 0;
912 uint32_t ref_rets_count = 0;
913 uint32_t ref_args_count = 0;
914 WasmBytecode* target_function = GetFunctionBytecode(func_index);
915 if (target_function) {
916 args_count = target_function->args_count();
917 rets_slots_size = target_function->rets_slots_size();
918 ref_rets_count = target_function->ref_rets_count();
919 ref_args_count = target_function->ref_args_count();
920 } else {
921 // We begin execution by calling an imported function.
922 args_count = sig->parameter_count();
923 rets_slots_size = WasmBytecode::RetsSizeInSlots(sig);
924 ref_rets_count = WasmBytecode::RefRetsCount(sig);
925 ref_args_count = WasmBytecode::RefArgsCount(sig);
926 }
927
928 // Here GC is disabled, we cannot "resize" the reference_stack_ FixedArray
929 // before having created Handles for the Ref arguments passed in
930 // argument_values.
931 HandleScope handle_scope(isolate_); // Avoid leaking handles.
932
933 std::vector<Handle<Object>> ref_args;
934 if (ref_args_count > 0) {
935 ref_args.reserve(ref_args_count);
936 }
937
938 uint8_t* p = interpreter_fp + rets_slots_size * kSlotSize;
939
940 // Check stack overflow.
941 const uint8_t* stack_limit = thread->StackLimitAddress();
942 if (V8_UNLIKELY(p + (ref_rets_count + ref_args_count) * sizeof(WasmRef) >=
943 stack_limit)) {
944 size_t additional_required_size =
945 p + (ref_rets_count + ref_args_count) * sizeof(WasmRef) - stack_limit;
946 if (!thread->ExpandStack(additional_required_size)) {
947 // TODO(paolosev@microsoft.com) - Calculate initial function offset.
948 ClearThreadInWasmScope clear_wasm_flag(isolate_);
949 SealHandleScope shs(isolate_);
950 isolate_->StackOverflow();
951 const pc_t trap_pc = 0;
952 SetTrap(TrapReason::kTrapUnreachable, trap_pc);
953 thread->FinishActivation();
954 return;
955 }
956 }
957
958 if (argument_values) {
959 // We are being called from JS, arguments are passed in the
960 // {argument_values} vector.
961 for (size_t i = 0; i < argument_values->size(); i++) {
962 const WasmValue& value = (*argument_values)[i];
963 switch (value.type().kind()) {
964 case kI32:
965 base::WriteUnalignedValue<int32_t>(reinterpret_cast<Address>(p),
966 value.to<int32_t>());
967 p += sizeof(int32_t);
968 break;
969 case kI64:
970 base::WriteUnalignedValue<int64_t>(reinterpret_cast<Address>(p),
971 value.to<int64_t>());
972 p += sizeof(int64_t);
973 break;
974 case kF32:
975 base::WriteUnalignedValue<float>(reinterpret_cast<Address>(p),
976 value.to<float>());
977 p += sizeof(float);
978 break;
979 case kF64:
980 base::WriteUnalignedValue<double>(reinterpret_cast<Address>(p),
981 value.to<double>());
982 p += sizeof(double);
983 break;
984 case kRef:
985 case kRefNull: {
986 Handle<Object> ref = value.to_ref();
987 if (IsJSFunction(*ref, isolate_)) {
988 Tagged<SharedFunctionInfo> sfi = Cast<JSFunction>(ref)->shared();
989 if (sfi->HasWasmExportedFunctionData()) {
990 Tagged<WasmExportedFunctionData> wasm_exported_function_data =
991 sfi->wasm_exported_function_data();
992 ref = handle(
993 wasm_exported_function_data->func_ref()->internal(isolate_),
994 isolate_);
995 }
996 }
997 ref_args.push_back(ref);
998 base::WriteUnalignedValue<WasmRef>(reinterpret_cast<Address>(p),
999 WasmRef(nullptr));
1000 p += sizeof(WasmRef);
1001 break;
1002 }
1003 case kS128:
1004 default:
1005 UNREACHABLE();
1006 }
1007 }
1008 } else {
1009 // We are being called from Wasm, arguments are already in the stack.
1010 for (size_t i = 0; i < args_count; i++) {
1011 switch (sig->GetParam(i).kind()) {
1012 case kI32:
1013 p += sizeof(int32_t);
1014 break;
1015 case kI64:
1016 p += sizeof(int64_t);
1017 break;
1018 case kF32:
1019 p += sizeof(float);
1020 break;
1021 case kF64:
1022 p += sizeof(double);
1023 break;
1024 case kS128:
1025 p += sizeof(Simd128);
1026 break;
1027 case kRef:
1028 case kRefNull: {
1029 Handle<Object> ref = base::ReadUnalignedValue<Handle<Object>>(
1030 reinterpret_cast<Address>(p));
1031 ref_args.push_back(ref);
1032 p += sizeof(WasmRef);
1033 break;
1034 }
1035 default:
1036 UNREACHABLE();
1037 }
1038 }
1039 }
1040
1041 {
1042 // Once we have read ref argument passed on the stack and we have stored
1043 // them into the ref_args vector of Handles, we can re-enable the GC.
1044 AllowHeapAllocation allow_gc;
1045
1046 if (ref_rets_count + ref_args_count > 0) {
1047 // Reserve space for reference args and return values in the
1048 // reference_stack_.
1049 current_thread_->EnsureRefStackSpace(
1050 current_frame_.ref_array_current_sp_ + ref_rets_count +
1051 ref_args_count);
1052
1053 uint32_t ref_stack_arg_index = ref_rets_count;
1054 for (uint32_t ref_arg_index = 0; ref_arg_index < ref_args_count;
1055 ref_arg_index++) {
1056 StoreWasmRef(ref_stack_arg_index++, ref_args[ref_arg_index]);
1057 }
1058 }
1059 }
1060}

1062void WasmInterpreterRuntime::ContinueExecution(WasmInterpreterThread* thread,
1063 bool called_from_js) {
1064 DCHECK_NE(start_function_index_, UINT_MAX);
1065
1066 uint32_t start_function_index = start_function_index_;
1067 FrameState current_frame = current_frame_;
1068
1069 const uint8_t* code = nullptr;
1070 const FunctionSig* sig = nullptr;
1071 uint32_t return_count = 0;
1072 WasmBytecode* target_function = GetFunctionBytecode(start_function_index_);
1073 if (target_function) {
1074 sig = target_function->GetFunctionSignature();
1075 return_count = target_function->return_count();
1076 ExecuteFunction(code, start_function_index_, target_function->args_count(),
1077 0, 0, 0);
1078 } else {
1079 sig = module_->functions[start_function_index_].sig;
1080 return_count = static_cast<uint32_t>(sig->return_count());
1081 ExecuteImportedFunction(code, start_function_index_,
1082 static_cast<uint32_t>(sig->parameter_count()), 0, 0,
1083 0);
1084 }
1085
1086 // If there are Ref types in the set of result types defined in the function
1087 // signature, they are located from the first ref_stack_ slot of the current
1088 // Activation.
1089 uint32_t ref_result_slot_index = 0;
1090
1091 if (state() == WasmInterpreterThread::State::RUNNING) {
1092 if (return_count > 0) {
1093 uint32_t* dst = reinterpret_cast<uint32_t*>(current_frame_.current_sp_);
1094
1095 if (called_from_js) {
1096 // We are returning the results to a JS caller, we need to store them
1097 // into the {function_result_} vector and they will be retrieved via
1098 // {GetReturnValue}.
1099 function_result_.resize(return_count);
1100 for (size_t index = 0; index < return_count; index++) {
1101 switch (sig->GetReturn(index).kind()) {
1102 case kI32:
1103 function_result_[index] =
1104 WasmValue(base::ReadUnalignedValue<int32_t>(
1105 reinterpret_cast<Address>(dst)));
1106 dst += sizeof(uint32_t) / kSlotSize;
1107 break;
1108 case kI64:
1109 function_result_[index] =
1110 WasmValue(base::ReadUnalignedValue<int64_t>(
1111 reinterpret_cast<Address>(dst)));
1112 dst += sizeof(uint64_t) / kSlotSize;
1113 break;
1114 case kF32:
1115 function_result_[index] =
1116 WasmValue(base::ReadUnalignedValue<float>(
1117 reinterpret_cast<Address>(dst)));
1118 dst += sizeof(float) / kSlotSize;
1119 break;
1120 case kF64:
1121 function_result_[index] =
1122 WasmValue(base::ReadUnalignedValue<double>(
1123 reinterpret_cast<Address>(dst)));
1124 dst += sizeof(double) / kSlotSize;
1125 break;
1126 case kRef:
1127 case kRefNull: {
1128 Handle<Object> ref = ExtractWasmRef(ref_result_slot_index++);
1129 ref = WasmToJSObject(ref);
1130 function_result_[index] = WasmValue(
1131 ref,
1132 sig->GetReturn(index).kind() == kRef ? kWasmRefString
1133 : kWasmAnyRef,
1134 module_);
1135 dst += sizeof(WasmRef) / kSlotSize;
1136 break;
1137 }
1138 case kS128:
1139 default:
1140 UNREACHABLE();
1141 }
1142 }
1143 } else {
1144 // We are returning the results on the stack
1145 for (size_t index = 0; index < return_count; index++) {
1146 switch (sig->GetReturn(index).kind()) {
1147 case kI32:
1148 dst += sizeof(uint32_t) / kSlotSize;
1149 break;
1150 case kI64:
1151 dst += sizeof(uint64_t) / kSlotSize;
1152 break;
1153 case kF32:
1154 dst += sizeof(float) / kSlotSize;
1155 break;
1156 case kF64:
1157 dst += sizeof(double) / kSlotSize;
1158 break;
1159 case kS128:
1160 dst += sizeof(Simd128) / kSlotSize;
1161 break;
1162 case kRef:
1163 case kRefNull: {
1164 // Make sure the ref result is termporarily stored in a stack
1165 // slot, to be retrieved by the caller.
1166 Handle<Object> ref = ExtractWasmRef(ref_result_slot_index++);
1167 base::WriteUnalignedValue<WasmRef>(reinterpret_cast<Address>(dst),
1168 ref);
1169 dst += sizeof(WasmRef) / kSlotSize;
1170 break;
1171 }
1172 default:
1173 UNREACHABLE();
1174 }
1175 }
1176 }
1177 }
1178
1179 if (ref_result_slot_index > 0) {
1180 current_thread_->ClearRefStackValues(current_frame_.ref_array_current_sp_,
1181 ref_result_slot_index);
1182 }
1183
1184 DCHECK(current_frame_.caught_exceptions_.is_null());
1185
1186 start_function_index_ = start_function_index;
1187 current_frame_ = current_frame;
1188 } else if (state() == WasmInterpreterThread::State::TRAPPED) {
1189 MessageTemplate message_id =
1190 WasmOpcodes::TrapReasonToMessageId(thread->GetTrapReason());
1191 thread->RaiseException(isolate_, message_id);
1192 } else if (state() == WasmInterpreterThread::State::EH_UNWINDING) {
1193 // Uncaught exception.
1194 thread->Stop();
1195 } else {
1196 DCHECK_EQ(state(), WasmInterpreterThread::State::STOPPED);
1197 }
1198
1199 thread->FinishActivation();
1200 const FrameState* frame_state = thread->GetCurrentActivationFor(this);
1201 current_frame_ = frame_state ? *frame_state : FrameState();
1202}

1222// A tail call should not add an additional stack frame to the interpreter
1223// stack. This is implemented by unwinding the current stack frame just before
1224// the tail call.
1225void WasmInterpreterRuntime::UnwindCurrentStackFrame(
1226 uint32_t* sp, uint32_t slot_offset, uint32_t rets_size, uint32_t args_size,
1227 uint32_t rets_refs, uint32_t args_refs, uint32_t ref_stack_fp_offset) {
1228 // At the moment of the call the interpreter stack is as in the diagram below.
1229 // A new interpreter frame for the callee function has been initialized, with
1230 // `R` slots to contain the R return values, followed by {args_size} slots to
1231 // contain the callee arguments.
1232 //
1233 // In order to unwind an interpreter stack frame we just copy the content of
1234 // the slots that contain the callee arguments into the caller stack frame,
1235 // just after the slots of the return values. Note that the return call is
1236 // invalid if the number and types of the return values of the callee function
1237 // do not exactly match the number and types of the return values of the
1238 // caller function. Instead, the number of types of the caller and callee
1239 // functions arguments can differ.
1240 //
1241 // The other slots in the caller frame, for const values and locals, will be
1242 // initialized later in ExecuteFunction().
1243 //
1244 // +----------------------+
1245 // | argA-1 | ^ ^
1246 // | ... | | | ->-----+
1247 // | ... | | | |
1248 // | arg0 | callee v |
1249 // | retR-1 | frame |
1250 // | ... | | |
1251 // | ret0 | v | copy
1252 // +----------------------+ (slot_offset) |
1253 // | ... | ^ V
1254 // | <stack slots> | | |
1255 // | <locals slots> | | |
1256 // | <const slots> | | ^ |
1257 // | argN-1 | caller | <------+
1258 // | ... | frame |
1259 // | arg0 | | v
1260 // | retR-1 | |
1261 // | ... | |
1262 // | ret0 | v
1263 // +----------------------+ (0)
1264
1265 uint8_t* next_sp = reinterpret_cast<uint8_t*>(sp);
1266 uint8_t* prev_sp = next_sp + slot_offset;
1267 // Here {args_size} is the number of arguments expected by the function we are
1268 // calling, which can be different from the number of args of the caller
1269 // function.
1270 ::memmove(next_sp + rets_size, prev_sp, args_size);
1271
1272 // If some of the argument-slots contain Ref values, we need to move them
1273 // accordingly, in the thread {reference_stack_}.
1274 if (rets_refs) {
1275 current_thread_->ClearRefStackValues(current_frame_.ref_array_current_sp_,
1276 rets_refs);
1277 }
1278 // Here {args_refs} is the number of reference args expected by the function
1279 // we are calling, which can be different from the number of reference args of
1280 // the caller function.
1281 for (uint32_t i = 0; i < args_refs; i++) {
1282 StoreWasmRef(rets_refs + i, ExtractWasmRef(ref_stack_fp_offset + i));
1283 }
1284 if (ref_stack_fp_offset > rets_refs + args_refs) {
1285 current_thread_->ClearRefStackValues(
1286 current_frame_.ref_array_current_sp_ + rets_refs + args_refs,
1287 ref_stack_fp_offset - rets_refs - args_refs);
1288 }
1289}

1291void WasmInterpreterRuntime::StoreRefArgsIntoStackSlots(
1292 uint8_t* sp, uint32_t ref_stack_fp_offset, const FunctionSig* sig) {
1293 // Argument values of type Ref, if present, are already stored in the thread's
1294 // reference_stack_ starting at index ref_stack_fp_offset + RefRetsCount(sig).
1295 // We want to temporarily copy the pointers to these object also in the stack
1296 // slots, because functions WasmInterpreter::RunInterpreter() and
1297 // WasmInterpreter::CallExternalJSFunction gets all arguments from the stack.
1298
1299 // TODO(paolosev@microsoft.com) - Too slow?
1300 ref_stack_fp_offset += WasmBytecode::RefRetsCount(sig);
1301
1302 size_t args_count = sig->parameter_count();
1303 sp += WasmBytecode::RetsSizeInSlots(sig) * kSlotSize;
1304 for (size_t i = 0; i < args_count; i++) {
1305 switch (sig->GetParam(i).kind()) {
1306 case kI32:
1307 case kF32:
1308 sp += sizeof(int32_t);
1309 break;
1310 case kI64:
1311 case kF64:
1312 sp += sizeof(int64_t);
1313 break;
1314 case kS128:
1315 sp += sizeof(Simd128);
1316 break;
1317 case kRef:
1318 case kRefNull: {
1319 WasmRef ref = ExtractWasmRef(ref_stack_fp_offset++);
1320 base::WriteUnalignedValue<WasmRef>(reinterpret_cast<Address>(sp), ref);
1321 sp += sizeof(WasmRef);
1322 break;
1323 }
1324 default:
1325 UNREACHABLE();
1326 }
1327 }
1328}

1410// Sets up the current interpreter stack frame to start executing a new function
1411// with a tail call. Do not move the stack pointer for the interpreter stack,
1412// and avoids calling WasmInterpreterRuntime::ExecuteFunction(), which would add
1413// a new C++ stack frame.
1414void WasmInterpreterRuntime::PrepareTailCall(const uint8_t*& code,
1415 uint32_t func_index,
1416 uint32_t current_stack_size,
1417 uint32_t return_slot_offset) {
1418 // TODO(paolosev@microsoft.com): avoid to duplicate code from ExecuteFunction?
1419
1420 WasmBytecode* target_function = GetFunctionBytecode(func_index);
1421 DCHECK_NOT_NULL(target_function);
1422
1423 // Update the current (caller) frame setting current_bytecode_ to the initial
1424 // instruction of the callee.
1425 current_frame_.current_function_ = target_function;
1426
1427#ifdef V8_ENABLE_DRUMBRAKE_TRACING
1428 current_frame_.current_stack_start_locals_ =
1429 current_frame_.current_stack_start_args_ + target_function->args_count();
1430 current_frame_.current_stack_start_stack_ =
1431 current_frame_.current_stack_start_locals_ +
1432 target_function->locals_count();
1433
1434 if (v8_flags.trace_drumbrake_execution) {
1435 Trace("\nTailCallFunction: %d\n", func_index);
1436 Trace("= > PushFrame #%d(#%d @%d)\n", current_frame_.current_stack_height_,
1437 func_index, 0);
1438 }
1439#endif // V8_ENABLE_DRUMBRAKE_TRACING
1440
1441 if (!WasmStackCheck(target_function->GetCode(), code)) {
1442 return;
1443 }
1444
1445 const uint8_t* stack_limit = current_frame_.thread_->StackLimitAddress();
1446 if (V8_UNLIKELY(stack_limit <= current_frame_.current_sp_ ||
1447 !target_function->InitializeSlots(
1448 current_frame_.current_sp_,
1449 stack_limit - current_frame_.current_sp_))) {
1450 // Try to resize the stack.
1451 size_t additional_required_space =
1452 target_function->frame_size() -
1453 (stack_limit - current_frame_.current_sp_);
1454 // Try again.
1455 if (!current_frame_.thread_->ExpandStack(additional_required_space) ||
1456 !target_function->InitializeSlots(
1457 current_frame_.current_sp_,
1458 (stack_limit = current_frame_.thread_->StackLimitAddress()) -
1459 current_frame_.current_sp_)) {
1460 ClearThreadInWasmScope clear_wasm_flag(isolate_);
1461 SealHandleScope shs(isolate_);
1462 SetTrap(TrapReason::kTrapUnreachable, code);
1463 isolate_->StackOverflow();
1464 return;
1465 }
1466 }
1467
1468 uint32_t ref_slots_count = target_function->ref_slots_count();
1469 if (V8_UNLIKELY(ref_slots_count > 0)) {
1470 uint32_t ref_array_length =
1471 current_frame_.ref_array_current_sp_ + ref_slots_count;
1472 current_thread_->EnsureRefStackSpace(ref_array_length);
1473
1474 // Initialize locals of ref types.
1475 if (V8_UNLIKELY(target_function->ref_locals_count() > 0)) {
1476 uint32_t ref_stack_index =
1477 target_function->ref_rets_count() + target_function->ref_args_count();
1478 for (uint32_t i = 0; i < target_function->ref_locals_count(); i++) {
1479 StoreWasmRef(ref_stack_index++,
1480 WasmRef(isolate_->factory()->null_value()));
1481 }
1482 }
1483 }
1484
1485#ifdef V8_ENABLE_DRUMBRAKE_TRACING
1486 uint32_t shadow_stack_offset = 0;
1487 if (v8_flags.trace_drumbrake_execution) {
1488 shadow_stack_offset = target_function->rets_slots_size() * kSlotSize;
1489 for (uint32_t i = 0; i < target_function->args_count(); i++) {
1490 shadow_stack_offset +=
1491 TracePush(target_function->arg_type(i).kind(), shadow_stack_offset);
1492 }
1493
1494 // Make room for locals in shadow stack
1495 shadow_stack_offset += target_function->const_slots_size_in_bytes();
1496 for (size_t i = 0; i < target_function->locals_count(); i++) {
1497 shadow_stack_offset +=
1498 TracePush(target_function->local_type(i).kind(), shadow_stack_offset);
1499 }
1500 }
1501#endif // V8_ENABLE_DRUMBRAKE_TRACING
1502
1503 code = target_function->GetCode();
1504}

1506void WasmInterpreterRuntime::ExecuteFunction(const uint8_t*& code,
1507 uint32_t func_index,
1508 uint32_t current_stack_size,
1509 uint32_t ref_stack_fp_offset,
1510 uint32_t slot_offset,
1511 uint32_t return_slot_offset) {
1512 // Execute an internal call.
1513 WasmBytecode* target_function = GetFunctionBytecode(func_index);
1514 DCHECK_NOT_NULL(target_function);
1515
1516#ifdef V8_ENABLE_DRUMBRAKE_TRACING
1517 ShadowStack* prev_shadow_stack = shadow_stack_;
1518 ShadowStack shadow_stack;
1519 if (v8_flags.trace_drumbrake_execution) {
1520 shadow_stack_ = &shadow_stack;
1521 }
1522#endif // V8_ENABLE_DRUMBRAKE_TRACING
1523
1524 // This HandleScope is used for all handles created in instruction handlers.
1525 // We reset it every time we get to a backward jump in a loop.
1526 HandleScope handle_scope(GetIsolate());
1527
1528 // Update the current (caller) frame setting current_bytecode_ to the current
1529 // call instruction.
1530 current_frame_.current_bytecode_ = code;
1531
1532 FrameState prev_frame_state = current_frame_;
1533 current_frame_.current_sp_ += slot_offset;
1534 current_frame_.handle_scope_ = &handle_scope;
1535
1536#ifdef V8_ENABLE_DRUMBRAKE_TRACING
1537 current_frame_.current_stack_start_args_ +=
1538 (current_stack_size - target_function->args_count());
1539#endif // V8_ENABLE_DRUMBRAKE_TRACING
1540
1541 current_frame_.current_function_ = target_function;
1542 current_frame_.previous_frame_ = &prev_frame_state;
1543 current_frame_.caught_exceptions_ = Handle<FixedArray>::null();
1544
1545#ifdef V8_ENABLE_DRUMBRAKE_TRACING
1546 current_frame_.current_stack_height_++;
1547 current_frame_.current_stack_start_locals_ =
1548 current_frame_.current_stack_start_args_ + target_function->args_count();
1549 current_frame_.current_stack_start_stack_ =
1550 current_frame_.current_stack_start_locals_ +
1551 target_function->locals_count();
1552
1553 if (v8_flags.trace_drumbrake_execution) {
1554 Trace("\nCallFunction: %d\n", func_index);
1555 Trace("= > PushFrame #%d(#%d @%d)\n", current_frame_.current_stack_height_,
1556 func_index, 0);
1557 }
1558#endif // V8_ENABLE_DRUMBRAKE_TRACING
1559
1560 if (!WasmStackCheck(target_function->GetCode(), code)) {
1561 return;
1562 }
1563
1564 const uint8_t* stack_limit = current_frame_.thread_->StackLimitAddress();
1565 if (V8_UNLIKELY(stack_limit <= current_frame_.current_sp_ ||
1566 !target_function->InitializeSlots(
1567 current_frame_.current_sp_,
1568 stack_limit - current_frame_.current_sp_))) {
1569 // Try to resize the stack.
1570 size_t additional_required_space =
1571 target_function->frame_size() -
1572 (stack_limit - current_frame_.current_sp_);
1573 // Try again.
1574 if (!current_frame_.thread_->ExpandStack(additional_required_space) ||
1575 !target_function->InitializeSlots(
1576 current_frame_.current_sp_,
1577 (stack_limit = current_frame_.thread_->StackLimitAddress()) -
1578 current_frame_.current_sp_)) {
1579 ClearThreadInWasmScope clear_wasm_flag(isolate_);
1580 SealHandleScope shs(isolate_);
1581 SetTrap(TrapReason::kTrapUnreachable, code);
1582 isolate_->StackOverflow();
1583 return;
1584 }
1585 }
1586
1587 uint32_t ref_slots_count = target_function->ref_slots_count();
1588 current_frame_.ref_array_current_sp_ += ref_stack_fp_offset;
1589 if (V8_UNLIKELY(ref_slots_count > 0)) {
1590 uint32_t ref_array_length =
1591 current_frame_.ref_array_current_sp_ + ref_slots_count;
1592 current_thread_->EnsureRefStackSpace(ref_array_length);
1593
1594 // Initialize locals of ref types.
1595 if (V8_UNLIKELY(target_function->ref_locals_count() > 0)) {
1596 uint32_t ref_stack_index =
1597 target_function->ref_rets_count() + target_function->ref_args_count();
1598 for (uint32_t i = 0; i < target_function->locals_count(); i++) {
1599 ValueType local_type = target_function->local_type(i);
1600 if (local_type == kWasmExternRef || local_type == kWasmNullExternRef) {
1601 StoreWasmRef(ref_stack_index++,
1602 WasmRef(isolate_->factory()->null_value()));
1603 } else if (local_type.is_reference()) {
1604 StoreWasmRef(ref_stack_index++,
1605 WasmRef(isolate_->factory()->wasm_null()));
1606 }
1607 }
1608 }
1609 }
1610
1611#ifdef V8_ENABLE_DRUMBRAKE_TRACING
1612 uint32_t shadow_stack_offset = 0;
1613 if (v8_flags.trace_drumbrake_execution) {
1614 shadow_stack_offset = target_function->rets_slots_size() * kSlotSize;
1615 for (uint32_t i = 0; i < target_function->args_count(); i++) {
1616 shadow_stack_offset +=
1617 TracePush(target_function->arg_type(i).kind(), shadow_stack_offset);
1618 }
1619
1620 // Make room for locals in shadow stack
1621 shadow_stack_offset += target_function->const_slots_size_in_bytes();
1622 for (size_t i = 0; i < target_function->locals_count(); i++) {
1623 shadow_stack_offset +=
1624 TracePush(target_function->local_type(i).kind(), shadow_stack_offset);
1625 }
1626 }
1627#endif // V8_ENABLE_DRUMBRAKE_TRACING
1628
1629 const uint8_t* callee_code = target_function->GetCode();
1630 int64_t r0 = 0;
1631 double fp0 = .0;
1632
1633 // Execute function
1634 CallThroughDispatchTable(
1635 callee_code, reinterpret_cast<uint32_t*>(current_frame_.current_sp_),
1636 this, r0, fp0);
1637
1638 uint32_t ref_slots_to_clear =
1639 ref_slots_count - target_function->ref_rets_count();
1640 if (V8_UNLIKELY(ref_slots_to_clear > 0)) {
1641 current_thread_->ClearRefStackValues(current_frame_.ref_array_current_sp_ +
1642 target_function->ref_rets_count(),
1643 ref_slots_to_clear);
1644 }
1645
1646#ifdef V8_ENABLE_DRUMBRAKE_TRACING
1647 shadow_stack_ = prev_shadow_stack;
1648
1649 if (v8_flags.trace_drumbrake_execution && shadow_stack_ != nullptr &&
1650 prev_frame_state.current_function_) {
1651 for (size_t i = 0; i < target_function->args_count(); i++) {
1652 TracePop();
1653 }
1654
1655 for (size_t i = 0; i < target_function->return_count(); i++) {
1656 return_slot_offset +=
1657 TracePush(target_function->return_type(i).kind(), return_slot_offset);
1658 }
1659 }
1660#endif // V8_ENABLE_DRUMBRAKE_TRACING
1661
1662 current_frame_.handle_scope_ = nullptr;
1663 current_frame_.DisposeCaughtExceptionsArray(isolate_);
1664 current_frame_ = prev_frame_state;
1665
1666 // Check state.
1667 WasmInterpreterThread::State current_state = state();
1668 if (V8_UNLIKELY(current_state != WasmInterpreterThread::State::RUNNING)) {
1669 switch (current_state) {
1670 case WasmInterpreterThread::State::EH_UNWINDING:
1671 DCHECK(isolate_->has_exception());
1672 if (!current_frame_.current_function_) {
1673 // We unwound the whole call stack without finding a catch handler.
1674 current_frame_.thread_->Stop();
1675 RedirectCodeToUnwindHandler(code);
1676 } else if (HandleException(
1677 reinterpret_cast<uint32_t*>(current_frame_.current_sp_),
1678 code) == WasmInterpreterThread::HANDLED) {
1679 current_frame_.thread_->Run();
1680 } else {
1681 // UNWOUND
1682 DCHECK(!trap_handler::IsThreadInWasm());
1683 RedirectCodeToUnwindHandler(code);
1684 }
1685 break;
1686
1687 case WasmInterpreterThread::State::TRAPPED:
1688 case WasmInterpreterThread::State::STOPPED:
1689 RedirectCodeToUnwindHandler(code);
1690 break;
1691
1692 default:
1693 UNREACHABLE();
1694 }
1695 }
1696 // TODO(paolosev@microsoft.com): StackCheck.
1697}

1743bool WasmInterpreterRuntime::CheckIndirectCallSignature(
1744 uint32_t table_index, uint32_t entry_index, uint32_t sig_index) const {
1745 const WasmTable& table = module_->tables[table_index];
1746 bool needs_type_check = !EquivalentTypes(
1747 table.type.AsNonNull(), ValueType::Ref(ModuleTypeIndex({sig_index})),
1748 module_, module_);
1749 bool needs_null_check = table.type.is_nullable();
1750
1751 // Copied from Liftoff.
1752 // We do both the type check and the null check by checking the signature,
1753 // so this shares most code. For the null check we then only check if the
1754 // stored signature is != -1.
1755 if (needs_type_check || needs_null_check) {
1756 const IndirectCallTable& dispatch_table =
1757 indirect_call_tables_[table_index];
1758 const wasm::CanonicalTypeIndex real_sig_id =
1759 dispatch_table[entry_index].sig_index;
1760 wasm::CanonicalTypeIndex canonical_sig_id =
1761 module_->canonical_sig_id(ModuleTypeIndex({sig_index}));
1762 if (!needs_type_check) {
1763 // Only check for -1 (nulled table entry).
1764 if (!real_sig_id.valid()) return false;
1765 } else if (!module_->types[sig_index].is_final) {
1766 if (real_sig_id == canonical_sig_id) return true;
1767 if (needs_null_check && !real_sig_id.valid()) return false;
1768
1769 Tagged<Map> rtt = Tagged<Map>::cast(isolate_->heap()
1770 ->wasm_canonical_rtts()
1771 ->get(real_sig_id.index)
1772 .GetHeapObjectAssumeWeak());
1773 Handle<Map> formal_rtt = RttCanon(sig_index);
1774 return SubtypeCheck(rtt, *formal_rtt, sig_index);
1775 } else {
1776 if (real_sig_id != canonical_sig_id) return false;
1777 }
1778 }
1779
1780 return true;
1781}

1783void WasmInterpreterRuntime::ExecuteIndirectCall(
1784 const uint8_t*& current_code, uint32_t table_index, uint32_t sig_index,
1785 uint32_t entry_index, uint32_t stack_pos, uint32_t* sp,
1786 uint32_t ref_stack_fp_offset, uint32_t slot_offset,
1787 uint32_t return_slot_offset, bool is_tail_call) {
1788 DCHECK_LT(table_index, indirect_call_tables_.size());
1789
1790 IndirectCallTable& table = indirect_call_tables_[table_index];
1791
1792 // Bounds check against table size.
1793 DCHECK_GE(
1794 table.size(),
1795 Tagged<WasmDispatchTable>::cast(
1796 wasm_trusted_instance_data()->dispatch_tables()->get(table_index))
1797 ->length());
1798 if (entry_index >= table.size()) {
1799 SetTrap(TrapReason::kTrapTableOutOfBounds, current_code);
1800 return;
1801 }
1802
1803 if (!table[entry_index]) {
1804 HandleScope handle_scope(isolate_); // Avoid leaking handles.
1805
1806 IndirectFunctionTableEntry entry(instance_object_, table_index,
1807 entry_index);
1808 const FunctionSig* signature = module_->signature({sig_index});
1809
1810 Handle<Object> object_implicit_arg = handle(entry.implicit_arg(), isolate_);
1811 if (IsWasmTrustedInstanceData(*object_implicit_arg)) {
1812 Tagged<WasmTrustedInstanceData> trusted_instance_object =
1813 Cast<WasmTrustedInstanceData>(*object_implicit_arg);
1814 Handle<WasmInstanceObject> instance_object = handle(
1815 Cast<WasmInstanceObject>(trusted_instance_object->instance_object()),
1816 isolate_);
1817 if (instance_object_.is_identical_to(instance_object)) {
1818 // Call to an import.
1819 uint32_t func_index = entry.function_index();
1820 table[entry_index] = IndirectCallValue(func_index, entry.sig_id());
1821 } else {
1822 // Cross-instance call.
1823 table[entry_index] = IndirectCallValue(signature, entry.sig_id());
1824 }
1825 } else {
1826 // Call JS function.
1827 table[entry_index] = IndirectCallValue(signature, entry.sig_id());
1828 }
1829 }
1830
1831 if (!CheckIndirectCallSignature(table_index, entry_index, sig_index)) {
1832 SetTrap(TrapReason::kTrapFuncSigMismatch, current_code);
1833 return;
1834 }
1835
1836 IndirectCallValue indirect_call = table[entry_index];
1837 DCHECK(indirect_call);
1838
1839 if (indirect_call.mode == IndirectCallValue::Mode::kInternalCall) {
1840 if (is_tail_call) {
1841 PrepareTailCall(current_code, indirect_call.func_index, stack_pos,
1842 return_slot_offset);
1843 } else {
1844 ExecuteFunction(current_code, indirect_call.func_index, stack_pos,
1845 ref_stack_fp_offset, slot_offset, return_slot_offset);
1846 if (state() == WasmInterpreterThread::State::TRAPPED ||
1847 state() == WasmInterpreterThread::State::STOPPED ||
1848 state() == WasmInterpreterThread::State::EH_UNWINDING) {
1849 RedirectCodeToUnwindHandler(current_code);
1850 }
1851 }
1852 } else {
1853 // ExternalCall
1854 HandleScope handle_scope(isolate_); // Avoid leaking handles.
1855
1856 DCHECK_NOT_NULL(indirect_call.signature);
1857
1858 // Store a pointer to the current FrameState before leaving the current
1859 // Activation.
1860 WasmInterpreterThread* thread = this->thread();
1861 current_frame_.current_bytecode_ = current_code;
1862 thread->SetCurrentFrame(current_frame_);
1863 thread->SetCurrentActivationFrame(
1864 sp, slot_offset, stack_pos,
1865 current_frame_.ref_array_current_sp_ + ref_stack_fp_offset,
1866 ref_stack_fp_offset);
1867
1868 // TODO(paolosev@microsoft.com): Optimize this code.
1869 IndirectFunctionTableEntry entry(instance_object_, table_index,
1870 entry_index);
1871 Handle<Object> object_implicit_arg = handle(entry.implicit_arg(), isolate_);
1872
1873 if (IsWasmTrustedInstanceData(*object_implicit_arg)) {
1874 // Call Wasm function in a different instance.
1875
1876 // Note that tail calls across WebAssembly module boundaries should
1877 // guarantee tail behavior, so this implementation does not conform to the
1878 // spec for a tail call. But it is really difficult to implement
1879 // cross-instance calls in the interpreter without recursively adding C++
1880 // stack frames.
1881 Handle<WasmInstanceObject> target_instance =
1882 handle(Cast<WasmInstanceObject>(
1883 Cast<WasmTrustedInstanceData>(*object_implicit_arg)
1884 ->instance_object()),
1885 isolate_);
1886
1887 // Make sure the target WasmInterpreterObject and InterpreterHandle exist.
1888 Handle<Tuple2> interpreter_object =
1889 WasmTrustedInstanceData::GetOrCreateInterpreterObject(
1890 target_instance);
1891 GetOrCreateInterpreterHandle(isolate_, interpreter_object);
1892
1893 Address frame_pointer = FindInterpreterEntryFramePointer(isolate_);
1894
1895 {
1896 // We should not allocate anything in the heap and avoid GCs after we
1897 // store ref arguments into stack slots.
1898 DisallowHeapAllocation no_gc;
1899
1900 uint8_t* fp = reinterpret_cast<uint8_t*>(sp) + slot_offset;
1901 StoreRefArgsIntoStackSlots(
1902 fp, current_frame_.ref_array_current_sp_ + ref_stack_fp_offset,
1903 indirect_call.signature);
1904 bool success = WasmInterpreterObject::RunInterpreter(
1905 isolate_, frame_pointer, target_instance, entry.function_index(),
1906 fp);
1907 if (success) {
1908 StoreRefResultsIntoRefStack(
1909 fp, current_frame_.ref_array_current_sp_ + ref_stack_fp_offset,
1910 indirect_call.signature);
1911
1912#ifdef V8_ENABLE_DRUMBRAKE_TRACING
1913 // Update shadow stack
1914 if (v8_flags.trace_drumbrake_execution && shadow_stack_ != nullptr) {
1915 for (size_t i = 0; i < indirect_call.signature->parameter_count();
1916 i++) {
1917 TracePop();
1918 }
1919
1920 for (size_t i = 0; i < indirect_call.signature->return_count();
1921 i++) {
1922 return_slot_offset +=
1923 TracePush(indirect_call.signature->GetReturn(i).kind(),
1924 return_slot_offset);
1925 }
1926 }
1927#endif // V8_ENABLE_DRUMBRAKE_TRACING
1928 } else {
1929 thread->Stop();
1930 RedirectCodeToUnwindHandler(current_code);
1931 }
1932 }
1933 } else {
1934 // We should not allocate anything in the heap and avoid GCs after we
1935 // store ref arguments into stack slots.
1936 DisallowHeapAllocation no_gc;
1937
1938 // Note that tail calls to host functions do not have to guarantee tail
1939 // behaviour, so it is ok to recursively allocate C++ stack frames here.
1940 uint8_t* fp = reinterpret_cast<uint8_t*>(sp) + slot_offset;
1941 StoreRefArgsIntoStackSlots(
1942 fp, current_frame_.ref_array_current_sp_ + ref_stack_fp_offset,
1943 indirect_call.signature);
1944 ExternalCallResult result = CallExternalJSFunction(
1945 current_code, module_, object_implicit_arg, indirect_call.signature,
1946 sp + slot_offset / kSlotSize, slot_offset);
1947 if (result == ExternalCallResult::EXTERNAL_RETURNED) {
1948 StoreRefResultsIntoRefStack(
1949 fp, current_frame_.ref_array_current_sp_ + ref_stack_fp_offset,
1950 indirect_call.signature);
1951 } else { // ExternalCallResult::EXTERNAL_EXCEPTION
1952 AllowHeapAllocation allow_gc;
1953
1954 if (HandleException(sp, current_code) ==
1955 WasmInterpreterThread::ExceptionHandlingResult::UNWOUND) {
1956 DCHECK(!trap_handler::IsThreadInWasm());
1957 thread->Stop();
1958 RedirectCodeToUnwindHandler(current_code);
1959 } else {
1960 thread->Run();
1961 }
1962 }
1963 }

2227 const uint8_t*& current_code, const WasmModule* module,
2228 Handle<Object> object_ref, const FunctionSig* sig, uint32_t* sp,
2229 uint32_t current_stack_slot) {
2230 // TODO(paolosev@microsoft.com) Cache IsJSCompatibleSignature result?
2231 if (!IsJSCompatibleSignature(
2232 reinterpret_cast<const wasm::CanonicalSig*>(sig))) {
2233 AllowHeapAllocation allow_gc;
2234 ClearThreadInWasmScope clear_wasm_flag(isolate_);

2237 MessageTemplate::kWasmTrapJSTypeError));

2240
2241#ifdef V8_ENABLE_DRUMBRAKE_TRACING

2255 p += sizeof(int32_t) / kSlotSize;

2276 ref = WasmToJSObject(ref);

2322 sp += sizeof(uint64_t) / kSlotSize;
2323 break;

2325 base::WriteUnalignedValue<float>(reinterpret_cast<Address>(sp),

2355#endif // V8_COMPRESS_POINTERS

2357 case kS128:
2358 default:
2359 UNREACHABLE();
2360 }
2361 }
2362 }

2368 TracePop();
2369 }

2372 return_slot_offset +=
2373 TracePush(sig->GetReturn(i).kind(), return_slot_offset);
2374 }
2375 }
2376#endif // V8_ENABLE_DRUMBRAKE_TRACING

2579 Tagged<Map> formal_rtt,

2582 // supertypes list.
2583 // Step 1: load the WasmTypeInfo.

2600// Implementation similar to Liftoff's SubtypeCheck in
2601// src\wasm\baseline\liftoff-compiler.cc.
2602bool WasmInterpreterRuntime::SubtypeCheck(const WasmRef obj,
2603 const ValueType obj_type,
2604 const Handle<Map> rtt,
2605 const ValueType rtt_type,
2606 bool null_succeeds) const {
2607 bool is_cast_from_any = obj_type.is_reference_to(HeapType::kAny);
2608
2609 // Skip the null check if casting from any and not {null_succeeds}.
2610 // In that case the instance type check will identify null as not being a
2611 // wasm object and fail.
2612 if (obj_type.is_nullable() && (!is_cast_from_any || null_succeeds)) {
2613 if (obj_type == kWasmExternRef || obj_type == kWasmNullExternRef) {
2614 if (i::IsNull(*obj, isolate_)) return null_succeeds;
2615 } else {
2616 if (i::IsWasmNull(*obj, isolate_)) return null_succeeds;
2617 }
2618 }
2619
2620 // Add Smi check if the source type may store a Smi (i31ref or JS Smi).
2621 ValueType i31ref = ValueType::Ref(HeapType::kI31);
2622 // Ref.extern can also contain Smis, however there isn't any type that
2623 // could downcast to ref.extern.
2624 DCHECK(!rtt_type.is_reference_to(HeapType::kExtern));
2625 // Ref.i31 check has its own implementation.
2626 DCHECK(!rtt_type.is_reference_to(HeapType::kI31));
2627 if (IsSmi(*obj)) {
2628 return IsSubtypeOf(i31ref, rtt_type, module_);
2629 }
2630
2631 if (!IsHeapObject(*obj)) return false;
2632 Tagged<Map> obj_map = Cast<HeapObject>(obj)->map();
2633
2634 if (module_->types[rtt_type.ref_index().index].is_final) {
2635 // In this case, simply check for map equality.
2636 if (*obj_map != *rtt) {
2637 return false;
2638 }
2639 } else {
2640 // Check for rtt equality, and if not, check if the rtt is a struct/array
2641 // rtt.
2642 if (*obj_map == *rtt) {
2643 return true;
2644 }
2645
2646 if (is_cast_from_any) {
2647 // Check for map being a map for a wasm object (struct, array, func).
2648 InstanceType obj_type = obj_map->instance_type();
2649 if (obj_type < FIRST_WASM_OBJECT_TYPE ||
2650 obj_type > LAST_WASM_OBJECT_TYPE) {
2651 return false;
2652 }
2653 }
2654
2655 return SubtypeCheck(obj_map, *rtt, rtt_type.ref_index().index);
2656 }
2657
2658 return true;
2659}

2862inline WasmInterpreterThread::State InterpreterHandle::RunExecutionLoop(
2863 WasmInterpreterThread* thread, bool called_from_js) {
2864 // If there were Ref values passed as arguments they have already been read
2865 // in BeginExecution(), so we can re-enable GC.
2866 AllowHeapAllocation allow_gc;
2867
2868 bool finished = false;
2869 WasmInterpreterThread::State state = thread->state();
2870 if (state != WasmInterpreterThread::State::RUNNING) {
2871 return state;
2872 }
2873
2874 while (!finished) {
2875 state = ContinueExecution(thread, called_from_js);
2876 switch (state) {
2877 case WasmInterpreterThread::State::FINISHED:
2878 case WasmInterpreterThread::State::RUNNING:
2879 // Perfect, just break the switch and exit the loop.
2880 finished = true;
2881 break;
2882 case WasmInterpreterThread::State::TRAPPED: {
2883 if (!isolate_->has_exception()) {
2884 // An exception handler was found, keep running the loop.
2885 if (!trap_handler::IsThreadInWasm()) {
2886 trap_handler::SetThreadInWasm();
2887 }
2888 break;
2889 }
2890 thread->Stop();
2891 [[fallthrough]];
2892 }
2893 case WasmInterpreterThread::State::STOPPED:
2894 // An exception happened, and the current activation was unwound
2895 // without hitting a local exception handler. All that remains to be
2896 // done is finish the activation and let the exception propagate.
2897 DCHECK(isolate_->has_exception());
2898 return state; // Either STOPPED or TRAPPED.
2899 case WasmInterpreterThread::State::EH_UNWINDING: {
2900 thread->Stop();
2901 return WasmInterpreterThread::State::STOPPED;
2902 }
2903 }
2904 }
2905 return state;
2906}

2909 WasmInterpreterThread* thread, Address frame_pointer, uint32_t func_index,
2910 const std::vector<WasmValue>& argument_values,
2911 std::vector<WasmValue>& return_values) {
2912 DCHECK_GT(module()->functions.size(), func_index);
2913 const FunctionSig* sig = module()->functions[func_index].sig;
2914 DCHECK_EQ(sig->parameter_count(), argument_values.size());
2915 DCHECK_EQ(sig->return_count(), return_values.size());
2916
2917 thread->StartExecutionTimer();
2918 interpreter_.BeginExecution(thread, func_index, frame_pointer,
2919 thread->NextFrameAddress(),
2920 thread->NextRefStackOffset(), argument_values);
2921
2922 WasmInterpreterThread::State state = RunExecutionLoop(thread, true);
2923 thread->StopExecutionTimer();
2924
2925 switch (state) {
2926 case WasmInterpreterThread::RUNNING:
2927 case WasmInterpreterThread::FINISHED:
2928 for (unsigned i = 0; i < sig->return_count(); ++i) {
2929 return_values[i] = interpreter_.GetReturnValue(i);

2932

2934 for (unsigned i = 0; i < sig->return_count(); ++i) {
2935 return_values[i] = WasmValue(0xDEADBEEF);
2936 }
2937 return false;
2938
2939 case WasmInterpreterThread::STOPPED:
2940 return false;
2941
2942 case WasmInterpreterThread::EH_UNWINDING:
