62 DirectHandle<Object> input_options) {
63 // 3. Let requestedLocales be ? CanonicalizeLocaleList(locales).
64 Maybe<std::vector<std::string>> maybe_requested_locales =
65 Intl::CanonicalizeLocaleList(isolate, locales);

72 // 4. Let options be GetOptionsObject(_options_).
73 ASSIGN_RETURN_ON_EXCEPTION(isolate, options,
74 GetOptionsObject(isolate, input_options, service));

76 // Note: No need to create a record. It's not observable.
77 // 6. Let opt be a new Record.

82 Intl::GetLocaleMatcher(isolate, options, service);
83 MAYBE_RETURN(maybe_locale_matcher, MaybeHandle<JSListFormat>());
84
85 // 8. Set opt.[[localeMatcher]] to matcher.
86 Intl::MatcherOption matcher = maybe_locale_matcher.FromJust();
87
88 // 10. Let r be ResolveLocale(%ListFormat%.[[AvailableLocales]],
89 // requestedLocales, opt, undefined, localeData).
90 Maybe<Intl::ResolvedLocale> maybe_resolve_locale =
91 Intl::ResolveLocale(isolate, JSListFormat::GetAvailableLocales(),
92 requested_locales, matcher, {});
93 if (maybe_resolve_locale.IsNothing()) {
94 THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
95 }
96 Intl::ResolvedLocale r = maybe_resolve_locale.FromJust();
97 DirectHandle<String> locale_str =
98 isolate->factory()->NewStringFromAsciiChecked(r.locale.c_str());
99
100 // 12. Let t be GetOption(options, "type", "string", Â«"conjunction",
101 // "disjunction", "unit"Â», "conjunction").
102 Maybe<Type> maybe_type = GetStringOption<Type>(
103 isolate, options, "type", service, {"conjunction", "disjunction", "unit"},
104 {Type::CONJUNCTION, Type::DISJUNCTION, Type::UNIT}, Type::CONJUNCTION);
105 MAYBE_RETURN(maybe_type, MaybeHandle<JSListFormat>());
106 Type type_enum = maybe_type.FromJust();
107
108 // 14. Let s be ? GetOption(options, "style", "string",
109 // Â«"long", "short", "narrow"Â», "long").

113 MAYBE_RETURN(maybe_style, MaybeHandle<JSListFormat>());
114 Style style_enum = maybe_style.FromJust();
115
116 icu::Locale icu_locale = r.icu_locale;
117 UErrorCode status = U_ZERO_ERROR;
118 std::shared_ptr<icu::ListFormatter> formatter{
119 icu::ListFormatter::createInstance(icu_locale, GetIcuType(type_enum),

126 Managed<icu::ListFormatter>::From(isolate, 0, std::move(formatter));
127
128 // Now all properties are ready, so we can allocate the result object.
129 Handle<JSListFormat> list_format =
130 Cast<JSListFormat>(isolate->factory()->NewFastOrSlowJSObjectFromMap(map));
131 DisallowGarbageCollection no_gc;
132 list_format->set_flags(0);
133 list_format->set_icu_formatter(*managed_formatter);
134
135 // 11. Set listFormat.[[Locale]] to r.[[Locale]].
136 list_format->set_locale(*locale_str);
137
138 // 13. Set listFormat.[[Type]] to t.
139 list_format->set_type(type_enum);
140
141 // 15. Set listFormat.[[Style]] to s.
142 list_format->set_style(style_enum);
143
144 return list_format;

197// Extract String from FixedArray into array of UnicodeString
198Maybe<std::vector<icu::UnicodeString>> ToUnicodeStringArray(
199 Isolate* isolate, DirectHandle<FixedArray> array) {
200 int length = array->length();
201 std::vector<icu::UnicodeString> result;
202 for (int i = 0; i < length; i++) {
203 Handle<Object> item(array->get(i), isolate);
204 DCHECK(IsString(*item));
205 Handle<String> item_str = Cast<String>(item);
206 if (!item_str->IsFlat()) item_str = String::Flatten(isolate, item_str);
207 result.push_back(Intl::ToICUUnicodeString(isolate, item_str));
208 }
209 return Just(result);
210}

212template <typename T>
213MaybeHandle<T> FormatListCommon(
214 Isolate* isolate, DirectHandle<JSListFormat> format,
215 DirectHandle<FixedArray> list,
216 const std::function<MaybeHandle<T>(Isolate*, const icu::FormattedValue&)>&
217 formatToResult) {
218 DCHECK(!IsUndefined(*list));
219 Maybe<std::vector<icu::UnicodeString>> maybe_array =
220 ToUnicodeStringArray(isolate, list);
221 MAYBE_RETURN(maybe_array, Handle<T>());
222 std::vector<icu::UnicodeString> array = maybe_array.FromJust();
223
224 icu::ListFormatter* formatter = format->icu_formatter()->raw();
225 DCHECK_NOT_NULL(formatter);
226
227 UErrorCode status = U_ZERO_ERROR;
228 icu::FormattedList formatted = formatter->formatStringsToValue(
229 array.data(), static_cast<int32_t>(array.size()), status);
230 if (U_FAILURE(status)) {
231 THROW_NEW_ERROR(isolate, NewTypeError(MessageTemplate::kIcuError));
232 }
233 return formatToResult(isolate, formatted);
234}

247// A helper function to convert the FormattedList to a
248// MaybeHandle<JSArray> for the implementation of formatToParts.
249MaybeHandle<JSArray> FormattedListToJSArray(
250 Isolate* isolate, const icu::FormattedValue& formatted) {
251 Handle<JSArray> array = isolate->factory()->NewJSArray(0);
252 icu::ConstrainedFieldPosition cfpos;
253 cfpos.constrainCategory(UFIELD_CATEGORY_LIST);
254 int index = 0;
255 UErrorCode status = U_ZERO_ERROR;
256 icu::UnicodeString string = formatted.toString(status);
257 DirectHandle<String> substring;
258 while (formatted.nextPosition(cfpos, status) && U_SUCCESS(status)) {
259 ASSIGN_RETURN_ON_EXCEPTION(
260 isolate, substring,
261 Intl::ToString(isolate, string, cfpos.getStart(), cfpos.getLimit()));
262 Intl::AddElement(isolate, array, index++,
263 IcuFieldIdToType(isolate, cfpos.getField()), substring);
264 }
265 if (U_FAILURE(status)) {
266 THROW_NEW_ERROR(isolate, NewTypeError(MessageTemplate::kIcuError));
267 }
268 JSObject::ValidateElements(*array);
269 return array;
270}
