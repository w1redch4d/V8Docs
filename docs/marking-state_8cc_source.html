61void MutatorMarkingState::FlushNotFullyConstructedObjects() {
62 std::unordered_set<HeapObjectHeader*> objects =
63 not_fully_constructed_worklist_.Extract<AccessMode::kAtomic>();
64 for (HeapObjectHeader* object : objects) {
65 if (MarkNoPush(*object))
66 previously_not_fully_constructed_worklist_.Push(object);
67 }
