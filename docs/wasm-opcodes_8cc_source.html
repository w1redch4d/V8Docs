18std::ostream& operator<<(std::ostream& os, const FunctionSig& sig) {
19 if (sig.return_count() == 0) os << "v";
20 for (auto ret : sig.returns()) {
21 os << ret.short_name();
22 }
23 os << "_";
24 if (sig.parameter_count() == 0) os << "v";
25 for (auto param : sig.parameters()) {
26 os << param.short_name();
27 }
28 return os;
29}

31bool IsJSCompatibleSignature(const CanonicalSig* sig) {
32 for (auto type : sig->all()) {
33 // Rtts are internal-only. They should never be part of a signature.
34 DCHECK(!type.is_rtt());
35 if (type == kCanonicalS128) return false;
36 if (type.is_object_reference()) {
37 switch (type.heap_representation_non_shared()) {
38 case HeapType::kStringViewWtf8:
39 case HeapType::kStringViewWtf16:
40 case HeapType::kStringViewIter:
41 case HeapType::kExn:
42 case HeapType::kNoExn:
43 return false;
44 default:
45 break;
46 }
47 }
48 }
49 return true;
50}
