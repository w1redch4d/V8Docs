31void* BoundedPageAllocator::AllocatePages(void* hint, size_t size,
32 size_t alignment,
33 PageAllocator::Permission access) {
34 MutexGuard guard(&mutex_);
35 DCHECK(IsAligned(alignment, region_allocator_.page_size()));
36 DCHECK(IsAligned(alignment, allocate_page_size_));
37
38 Address address = RegionAllocator::kAllocationFailure;
39
40 Address hint_address = reinterpret_cast<Address>(hint);
41 if (hint_address && IsAligned(hint_address, alignment) &&
42 region_allocator_.contains(hint_address, size)) {
43 if (region_allocator_.AllocateRegionAt(hint_address, size)) {
44 address = hint_address;
45 }
46 }
47
48 if (address == RegionAllocator::kAllocationFailure) {
49 if (alignment <= allocate_page_size_) {
50 // TODO(ishell): Consider using randomized version here.
51 address = region_allocator_.AllocateRegion(size);
52 } else {
53 address = region_allocator_.AllocateAlignedRegion(size, alignment);
54 }
55 }
56
57 if (address == RegionAllocator::kAllocationFailure) {
58 allocation_status_ = AllocationStatus::kRanOutOfReservation;
59 return nullptr;
60 }
61
62 void* ptr = reinterpret_cast<void*>(address);
63 // It's assumed that free regions are in kNoAccess/kNoAccessWillJitLater
64 // state.
65 if (access == PageAllocator::kNoAccess ||
66 access == PageAllocator::kNoAccessWillJitLater) {
67 allocation_status_ = AllocationStatus::kSuccess;
68 return ptr;
69 }
70
71 if (page_initialization_mode_ == PageInitializationMode::kRecommitOnly) {
72 if (page_allocator_->RecommitPages(ptr, size, access)) {
73 allocation_status_ = AllocationStatus::kSuccess;
74 return ptr;
75 }
76 } else {
77 if (page_allocator_->SetPermissions(ptr, size, access)) {
78 allocation_status_ = AllocationStatus::kSuccess;
79 return ptr;
80 }
81 }
82
83 // This most likely means that we ran out of memory.
84 CHECK_EQ(region_allocator_.FreeRegion(address), size);
85 allocation_status_ = AllocationStatus::kFailedToCommit;
86 return nullptr;
87}

89bool BoundedPageAllocator::AllocatePagesAt(Address address, size_t size,
90 PageAllocator::Permission access) {
91 MutexGuard guard(&mutex_);
92
93 DCHECK(IsAligned(address, allocate_page_size_));
94 DCHECK(IsAligned(size, allocate_page_size_));
95
96 DCHECK(region_allocator_.contains(address, size));
97
98 if (!region_allocator_.AllocateRegionAt(address, size)) {
99 allocation_status_ = AllocationStatus::kHintedAddressTakenOrNotFound;
100 return false;
101 }
102
103 void* ptr = reinterpret_cast<void*>(address);
104 if (!page_allocator_->SetPermissions(ptr, size, access)) {
105 // This most likely means that we ran out of memory.
106 CHECK_EQ(region_allocator_.FreeRegion(address), size);
107 allocation_status_ = AllocationStatus::kFailedToCommit;
108 return false;
109 }
110
111 allocation_status_ = AllocationStatus::kSuccess;
112 return true;
113}

115bool BoundedPageAllocator::ReserveForSharedMemoryMapping(void* ptr,
116 size_t size) {
117 MutexGuard guard(&mutex_);
118
119 Address address = reinterpret_cast<Address>(ptr);
120 DCHECK(IsAligned(address, allocate_page_size_));
121 DCHECK(IsAligned(size, commit_page_size_));
122
123 DCHECK(region_allocator_.contains(address, size));
124
125 // Region allocator requires page size rather than commit size so just over-
126 // allocate there since any extra space couldn't be used anyway.
127 size_t region_size = RoundUp(size, allocate_page_size_);
128 if (!region_allocator_.AllocateRegionAt(
129 address, region_size, RegionAllocator::RegionState::kExcluded)) {
130 allocation_status_ = AllocationStatus::kHintedAddressTakenOrNotFound;
131 return false;
132 }
133
134 const bool success = page_allocator_->SetPermissions(
135 ptr, size, PageAllocator::Permission::kNoAccess);
136 if (success) {
137 allocation_status_ = AllocationStatus::kSuccess;
138 } else {
139 allocation_status_ = AllocationStatus::kFailedToCommit;
140 }
141 return success;
142}

144bool BoundedPageAllocator::FreePages(void* raw_address, size_t size) {
145 MutexGuard guard(&mutex_);
146
147 Address address = reinterpret_cast<Address>(raw_address);
148 CHECK_EQ(size, region_allocator_.FreeRegion(address));
149 if (page_initialization_mode_ ==
150 PageInitializationMode::kAllocatedPagesMustBeZeroInitialized) {
151 DCHECK_NE(page_freeing_mode_, PageFreeingMode::kDiscard);
152 // When we are required to return zero-initialized pages, we decommit the
153 // pages here, which will cause any wired pages to be removed by the OS.
154 return page_allocator_->DecommitPages(raw_address, size);
155 }
156 if (page_freeing_mode_ == PageFreeingMode::kMakeInaccessible) {
157 DCHECK_EQ(page_initialization_mode_,
158 PageInitializationMode::kAllocatedPagesCanBeUninitialized);
159 return page_allocator_->SetPermissions(raw_address, size,
160 PageAllocator::kNoAccess);
161 }
162 CHECK_EQ(page_freeing_mode_, PageFreeingMode::kDiscard);
163 return page_allocator_->DiscardSystemPages(raw_address, size);
164}

166bool BoundedPageAllocator::ReleasePages(void* raw_address, size_t size,
167 size_t new_size) {
168 Address address = reinterpret_cast<Address>(raw_address);
169 DCHECK(IsAligned(address, allocate_page_size_));
170
171 DCHECK_LT(new_size, size);
172 DCHECK(IsAligned(size - new_size, commit_page_size_));
173
174 // This must be held until the page permissions are updated.
175 MutexGuard guard(&mutex_);
176
177 // Check if we freed any allocatable pages by this release.
178 size_t allocated_size = RoundUp(size, allocate_page_size_);
179 size_t new_allocated_size = RoundUp(new_size, allocate_page_size_);
180
181#ifdef DEBUG
182 {
183 // There must be an allocated region at given |address| of a size not
184 // smaller than |size|.
185 DCHECK_EQ(allocated_size, region_allocator_.CheckRegion(address));
186 }
187#endif
188
189 if (new_allocated_size < allocated_size) {
190 region_allocator_.TrimRegion(address, new_allocated_size);
191 }
192
193 // Keep the region in "used" state just uncommit some pages.
194 void* free_address = reinterpret_cast<void*>(address + new_size);
195 size_t free_size = size - new_size;
196 if (page_initialization_mode_ ==
197 PageInitializationMode::kAllocatedPagesMustBeZeroInitialized) {
198 DCHECK_NE(page_freeing_mode_, PageFreeingMode::kDiscard);
199 // See comment in FreePages().
200 return (page_allocator_->DecommitPages(free_address, free_size));
201 }
202 if (page_freeing_mode_ == PageFreeingMode::kMakeInaccessible) {
203 DCHECK_EQ(page_initialization_mode_,
204 PageInitializationMode::kAllocatedPagesCanBeUninitialized);
205 return page_allocator_->SetPermissions(free_address, free_size,
206 PageAllocator::kNoAccess);
207 }
208 CHECK_EQ(page_freeing_mode_, PageFreeingMode::kDiscard);
209 return page_allocator_->DiscardSystemPages(free_address, free_size);
210}

213 PageAllocator::Permission access) {
214 DCHECK(IsAligned(reinterpret_cast<Address>(address), commit_page_size_));
