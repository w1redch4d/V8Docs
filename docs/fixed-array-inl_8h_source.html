218template <class D, class S, class P>
219void TaggedArrayBase<D, S, P>::MoveElements(Isolate* isolate, Tagged<D> dst,
220 int dst_index, Tagged<D> src,
221 int src_index, int len,
222 WriteBarrierMode mode) {
223 if (len == 0) return;
224
225 DCHECK_GE(len, 0);
226 DCHECK(dst->IsInBounds(dst_index));
227 DCHECK_LE(dst_index + len, dst->length());
228 DCHECK(src->IsInBounds(src_index));
229 DCHECK_LE(src_index + len, src->length());
230
231 DisallowGarbageCollection no_gc;
232 SlotType dst_slot(&dst->objects()[dst_index]);
233 SlotType src_slot(&src->objects()[src_index]);
234 isolate->heap()->MoveRange(dst, dst_slot, src_slot, len, mode);
235}

256template <class D, class S, class P>
257void TaggedArrayBase<D, S, P>::RightTrim(Isolate* isolate, int new_capacity) {
258 int old_capacity = this->capacity();
259 CHECK_GT(new_capacity, 0); // Due to possible canonicalization.
260 CHECK_LE(new_capacity, old_capacity);
261 if (new_capacity == old_capacity) return;
262 isolate->heap()->RightTrimArray(Cast<D>(this), new_capacity, old_capacity);
263}

286template <class IsolateT>
287Handle<FixedArray> FixedArray::New(IsolateT* isolate, int capacity,
288 AllocationType allocation) {
289 if (V8_UNLIKELY(static_cast<unsigned>(capacity) >
290 FixedArrayBase::kMaxLength)) {
291 FATAL("Fatal JavaScript invalid size error %d (see crbug.com/1201626)",
292 capacity);
293 } else if (V8_UNLIKELY(capacity == 0)) {
294 return isolate->factory()->empty_fixed_array();
295 }
296
297 std::optional<DisallowGarbageCollection> no_gc;
298 Handle<FixedArray> result =
299 Cast<FixedArray>(Allocate(isolate, capacity, &no_gc, allocation));
300 ReadOnlyRoots roots{isolate};
301 MemsetTagged((*result)->RawFieldOfFirstElement(), roots.undefined_value(),
302 capacity);
303 return result;
304}

349template <class D, class S, class P>
350template <class IsolateT>
351Handle<D> TaggedArrayBase<D, S, P>::Allocate(
352 IsolateT* isolate, int capacity,
353 std::optional<DisallowGarbageCollection>* no_gc_out,
354 AllocationType allocation) {
355 // Note 0-capacity is explicitly allowed since not all subtypes can be
356 // assumed to have canonical 0-capacity instances.
357 DCHECK_GE(capacity, 0);
358 DCHECK_LE(capacity, kMaxCapacity);
359 DCHECK(!no_gc_out->has_value());
360
361 Tagged<D> xs = UncheckedCast<D>(
362 isolate->factory()->AllocateRawArray(SizeFor(capacity), allocation));
363
364 ReadOnlyRoots roots{isolate};
365 if (DEBUG_BOOL) no_gc_out->emplace();
366 Tagged<Map> map = Cast<Map>(roots.object_at(S::kMapRootIndex));
367 DCHECK(ReadOnlyHeap::Contains(map));
368
369 xs->set_map_after_allocation(isolate, map, SKIP_WRITE_BARRIER);
370 xs->set_capacity(capacity);
371
372 return handle(xs, isolate);
373}

376template <class D, class S, class P>
377constexpr int TaggedArrayBase<D, S, P>::NewCapacityForIndex(int index,
378 int old_capacity) {
379 DCHECK_GE(index, old_capacity);
380 // Note this is currently based on JSObject::NewElementsCapacity.
381 int capacity = old_capacity;
382 do {
383 capacity = capacity + (capacity >> 1) + 16;
384 } while (capacity <= index);
385 return capacity;
386}
