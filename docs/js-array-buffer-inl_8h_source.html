63size_t JSArrayBuffer::GetByteLength() const {
64 if (V8_UNLIKELY(is_shared() && is_resizable_by_js())) {
65 // Invariant: byte_length for GSAB is 0 (it needs to be read from the
66 // BackingStore).
67 DCHECK_EQ(0, byte_length());
68
69 // If the byte length is read after the JSArrayBuffer object is allocated
70 // but before it's attached to the backing store, GetBackingStore returns
71 // nullptr. This is rare, but can happen e.g., when memory measurements
72 // are enabled (via performance.measureMemory()).
73 auto backing_store = GetBackingStore();
74 if (!backing_store) {
75 return 0;
76 }
77
78 return backing_store->byte_length(std::memory_order_seq_cst);
79 }
80 return byte_length();
81}

101ArrayBufferExtension* JSArrayBuffer::extension() const {
102#if V8_COMPRESS_POINTERS
103 // We need Acquire semantics here when loading the entry, see below.
104 // Consider adding respective external pointer accessors if non-relaxed
105 // ordering semantics are ever needed in other places as well.
106 Isolate* isolate = GetIsolateFromWritableObject(*this);
107 ExternalPointerHandle handle =
108 base::AsAtomic32::Acquire_Load(extension_handle_location());
109 return reinterpret_cast<ArrayBufferExtension*>(
110 isolate->external_pointer_table().Get(handle, kArrayBufferExtensionTag));
111#else
112 return base::AsAtomicPointer::Acquire_Load(extension_location());
113#endif // V8_COMPRESS_POINTERS
114}

116void JSArrayBuffer::set_extension(ArrayBufferExtension* extension) {
117#if V8_COMPRESS_POINTERS
118 // TODO(saelo): if we ever use the external pointer table for all external
119 // pointer fields in the no-sandbox-ptr-compression config, replace this code
120 // here and above with the respective external pointer accessors.
121 IsolateForPointerCompression isolate = GetIsolateFromWritableObject(*this);
122 const ExternalPointerTag tag = kArrayBufferExtensionTag;
123 Address value = reinterpret_cast<Address>(extension);
124 ExternalPointerTable& table = isolate.GetExternalPointerTableFor(tag);
125
126 ExternalPointerHandle current_handle =
127 base::AsAtomic32::Relaxed_Load(extension_handle_location());
128 if (current_handle == kNullExternalPointerHandle) {
129 // We need Release semantics here, see above.
130 ExternalPointerHandle handle = table.AllocateAndInitializeEntry(
131 isolate.GetExternalPointerTableSpaceFor(tag, address()), value, tag);
132 base::AsAtomic32::Release_Store(extension_handle_location(), handle);
133 EXTERNAL_POINTER_WRITE_BARRIER(*this, kExtensionOffset, tag);
134 } else {
135 table.Set(current_handle, value, tag);
136 }
137#else
138 base::AsAtomicPointer::Release_Store(extension_location(), extension);
139#endif // V8_COMPRESS_POINTERS
140 WriteBarrier::ForArrayBufferExtension(*this, extension);
141}

245bool JSTypedArray::IsDetachedOrOutOfBounds() const {
246 if (WasDetached()) {
247 return true;
248 }
249 if (!is_backed_by_rab()) {
250 // TypedArrays backed by GSABs or regular AB/SABs are never out of bounds.
251 // This shortcut is load-bearing; this enables determining
252 // IsDetachedOrOutOfBounds without consulting the BackingStore.
253 return false;
254 }
255 return IsOutOfBounds();

332void* JSTypedArray::DataPtr() {
333 // Zero-extend Tagged_t to Address according to current compression scheme
334 // so that the addition with |external_pointer| (which already contains
335 // compensated offset value) will decompress the tagged value.
336 // See JSTypedArray::ExternalPointerCompensationForOnHeapArray() for details.
337 static_assert(kOffHeapDataPtrEqualsExternalPointer);
338 return reinterpret_cast<void*>(external_pointer() +
339 static_cast<Tagged_t>(base_pointer().ptr()));
340}
341
342void JSTypedArray::SetOffHeapDataPtr(Isolate* isolate, void* base,
343 Address offset) {
344 Address address = reinterpret_cast<Address>(base) + offset;
345 set_external_pointer(isolate, address);
346 // This is the only spot in which the `base_pointer` field can be mutated
347 // after object initialization. Note this can happen at most once, when
348 // `JSTypedArray::GetBuffer` transitions from an on- to off-heap
349 // representation.
350 // To play well with Turbofan concurrency requirements, `base_pointer` is set
351 // with a release store, after external_pointer has been set.
352 set_base_pointer(Smi::zero(), kReleaseStore, SKIP_WRITE_BARRIER);
353 DCHECK_EQ(address, reinterpret_cast<Address>(DataPtr()));
354}

370// static
371MaybeHandle<JSTypedArray> JSTypedArray::Validate(Isolate* isolate,
372 Handle<Object> receiver,
373 const char* method_name) {
374 if (V8_UNLIKELY(!IsJSTypedArray(*receiver))) {
375 const MessageTemplate message = MessageTemplate::kNotTypedArray;
376 THROW_NEW_ERROR(isolate, NewTypeError(message));
377 }
378
379 Handle<JSTypedArray> array = Cast<JSTypedArray>(receiver);
380 if (V8_UNLIKELY(array->WasDetached())) {
381 const MessageTemplate message = MessageTemplate::kDetachedOperation;
382 DirectHandle<String> operation =
383 isolate->factory()->NewStringFromAsciiChecked(method_name);
384 THROW_NEW_ERROR(isolate, NewTypeError(message, operation));
385 }
386
387 if (V8_UNLIKELY(array->IsVariableLength() && array->IsOutOfBounds())) {
388 const MessageTemplate message = MessageTemplate::kDetachedOperation;
389 DirectHandle<String> operation =
390 isolate->factory()->NewStringFromAsciiChecked(method_name);
391 THROW_NEW_ERROR(isolate, NewTypeError(message, operation));
392 }
393
394 // spec describes to return `buffer`, but it may disrupt current
395 // implementations, and it's much useful to return array for now.
396 return array;
397}

410size_t JSRabGsabDataView::GetByteLength() const {
411 if (IsOutOfBounds()) {
412 return 0;
413 }
414 if (is_length_tracking()) {
415 // Invariant: byte_length of length tracking DataViews is 0.
416 DCHECK_EQ(0, byte_length());
417 return buffer()->GetByteLength() - byte_offset();
418 }
419 return byte_length();
420}
421
422bool JSRabGsabDataView::IsOutOfBounds() const {
423 if (!is_backed_by_rab()) {
424 return false;
425 }
426 if (is_length_tracking()) {
427 return byte_offset() > buffer()->GetByteLength();
428 }
429 return byte_offset() + byte_length() > buffer()->GetByteLength();
430}
