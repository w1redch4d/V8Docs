18// Support for emulating futexes, a low-level synchronization primitive. They
19// are natively supported by Linux, but must be emulated for other platforms.
20// This library emulates them on all platforms using mutexes and condition
21// variables for consistency.
22//
23// This is used by the Futex API defined in the SharedArrayBuffer draft spec,
24// found here: https://github.com/tc39/ecmascript_sharedmem

80 // Async wait requires substantially more information than synchronous wait.
81 // Hence store that additional information in a heap-allocated struct to make
82 // it more obvious that this will only be needed for the async case.
83 struct AsyncState {
84 AsyncState(Isolate* isolate, std::shared_ptr<TaskRunner> task_runner,
85 std::weak_ptr<BackingStore> backing_store,
86 v8::Global<v8::Promise> promise,
87 v8::Global<v8::Context> native_context)
88 : isolate_for_async_waiters(isolate),
89 task_runner(std::move(task_runner)),
90 backing_store(std::move(backing_store)),
91 promise(std::move(promise)),
92 native_context(std::move(native_context)) {
93 DCHECK(this->promise.IsWeak());
94 DCHECK(this->native_context.IsWeak());
95 }
96
97 ~AsyncState() {
98 // Assert that the timeout task was cancelled.
99 DCHECK_EQ(CancelableTaskManager::kInvalidTaskId, timeout_task_id);
100 }
101
102 Isolate* const isolate_for_async_waiters;
103 std::shared_ptr<TaskRunner> const task_runner;
104
105 // The backing store on which we are waiting might die in an async wait.
106 // We keep a weak_ptr to verify during a wake operation that the original
107 // backing store is still mapped to that address.
108 std::weak_ptr<BackingStore> const backing_store;
109
110 // Weak Global handle. Must not be synchronously resolved by a non-owner
111 // Isolate.
112 v8::Global<v8::Promise> const promise;
113
114 // Weak Global handle.
115 v8::Global<v8::Context> const native_context;
116
117 // If timeout_time_ is base::TimeTicks(), this async waiter doesn't have a
118 // timeout or has already been notified. Values other than base::TimeTicks()
119 // are used for async waiters with an active timeout.
120 base::TimeTicks timeout_time;
121
122 // The task ID of the timeout task.
123 CancelableTaskManager::Id timeout_task_id =
124 CancelableTaskManager::kInvalidTaskId;
125 };
126
127 base::ConditionVariable cond_;
128 // prev_ and next_ are protected by FutexEmulationGlobalState::mutex.
129 FutexWaitListNode* prev_ = nullptr;
130 FutexWaitListNode* next_ = nullptr;
131
132 // The memory location the FutexWaitListNode is waiting on. Equals
133 // backing_store_->buffer_start() + wait_addr at FutexWaitListNode creation
134 // time. This address is used find the node in the per-location list, or to
135 // remove it.
136 // Note that during an async wait the BackingStore might get deleted while
137 // this node is alive.
138 void* wait_location_ = nullptr;
139
140 // waiting_ and interrupted_ are protected by FutexEmulationGlobalState::mutex
141 // if this node is currently contained in FutexEmulationGlobalState::wait_list
142 // or an AtomicsWaitWakeHandle has access to it.
143 bool waiting_ = false;
144 bool interrupted_ = false;
145
146 // State used for an async wait; nullptr on sync waits.
147 const std::unique_ptr<AsyncState> async_state_;

155 // Pass to Wake() to wake all waiters.

158 // Check that array_buffer[addr] == value, and return "not-equal" if not. If
159 // they are equal, block execution on |isolate|'s thread until woken via
160 // |Wake|, or when the time given in |rel_timeout_ms| elapses. Note that
161 // |rel_timeout_ms| can be Infinity.
162 // If woken, return "ok", otherwise return "timed-out". The initial check and
163 // the decision to wait happen atomically.
164 static Tagged<Object> WaitJs32(Isolate* isolate, WaitMode mode,
165 Handle<JSArrayBuffer> array_buffer,
166 size_t addr, int32_t value,
167 double rel_timeout_ms);
168
169 // An version of WaitJs32 for int64_t values.
170 static Tagged<Object> WaitJs64(Isolate* isolate, WaitMode mode,
171 Handle<JSArrayBuffer> array_buffer,
172 size_t addr, int64_t value,
173 double rel_timeout_ms);
174
175 // Same as WaitJs above except it returns 0 (ok), 1 (not equal) and 2 (timed
176 // out) as expected by Wasm.
177 V8_EXPORT_PRIVATE static Tagged<Object> WaitWasm32(
178 Isolate* isolate, Handle<JSArrayBuffer> array_buffer, size_t addr,
179 int32_t value, int64_t rel_timeout_ns);
180
181 // Same as Wait32 above except it checks for an int64_t value in the
182 // array_buffer.
183 V8_EXPORT_PRIVATE static Tagged<Object> WaitWasm64(
184 Isolate* isolate, Handle<JSArrayBuffer> array_buffer, size_t addr,
185 int64_t value, int64_t rel_timeout_ns);
186
187 // Wake |num_waiters_to_wake| threads that are waiting on the given |addr|.
188 // |num_waiters_to_wake| can be kWakeAll, in which case all waiters are
189 // woken. The rest of the waiters will continue to wait. The return value is
190 // the number of woken waiters.
191 // Variant 1: Compute the wait address from the |array_buffer| and |addr|.
192 V8_EXPORT_PRIVATE static int Wake(Tagged<JSArrayBuffer> array_buffer,
193 size_t addr, uint32_t num_waiters_to_wake);
194 // Variant 2: Pass raw |addr| (used for WebAssembly atomic.notify).
195 static int Wake(void* addr, uint32_t num_waiters_to_wake);
196
197 // Called before |isolate| dies. Removes async waiters owned by |isolate|.
198 static void IsolateDeinit(Isolate* isolate);
