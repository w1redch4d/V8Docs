137uint32_t CodeGenerator::GetStackCheckOffset() {
138 if (!frame_access_state()->has_frame()) {
139 DCHECK_EQ(max_unoptimized_frame_height_, 0);
140 DCHECK_EQ(max_pushed_argument_count_, 0);
141 return 0;
142 }
143
144 size_t incoming_parameter_count =
145 linkage_->GetIncomingDescriptor()->ParameterSlotCount();
146 DCHECK(is_int32(incoming_parameter_count));
147 int32_t optimized_frame_height =
148 static_cast<int32_t>(incoming_parameter_count) * kSystemPointerSize +
149 frame()->GetTotalFrameSlotCount() * kSystemPointerSize;
150 DCHECK(is_int32(max_unoptimized_frame_height_));
151 int32_t signed_max_unoptimized_frame_height =
152 static_cast<int32_t>(max_unoptimized_frame_height_);
153
154 // The offset is either the delta between the optimized frames and the
155 // interpreted frame, or the maximal number of bytes pushed to the stack
156 // while preparing for function calls, whichever is bigger.
157 uint32_t frame_height_delta = static_cast<uint32_t>(std::max(
158 signed_max_unoptimized_frame_height - optimized_frame_height, 0));
159 uint32_t max_pushed_argument_bytes =
160 static_cast<uint32_t>(max_pushed_argument_count_ * kSystemPointerSize);
161 return std::max(frame_height_delta, max_pushed_argument_bytes);
162}

164CodeGenerator::CodeGenResult CodeGenerator::AssembleDeoptimizerCall(
165 DeoptimizationExit* exit) {
166 int deoptimization_id = exit->deoptimization_id();
167 if (deoptimization_id > Deoptimizer::kMaxNumberOfEntries) {
168 return kTooManyDeoptimizationBailouts;
169 }
170
171 DeoptimizeKind deopt_kind = exit->kind();
172 DeoptimizeReason deoptimization_reason = exit->reason();
173 Label* jump_deoptimization_entry_label =
174 &jump_deoptimization_entry_labels_[static_cast<int>(deopt_kind)];
175 if (info()->source_positions()) {
176 masm()->RecordDeoptReason(deoptimization_reason, exit->node_id(),
177 exit->pos(), deoptimization_id);
178 }
179
180 if (deopt_kind == DeoptimizeKind::kLazy) {
181 ++lazy_deopt_count_;
182 masm()->BindExceptionHandler(exit->label());
183 } else {
184 ++eager_deopt_count_;
185 masm()->bind(exit->label());
186 }
187 Builtin target = Deoptimizer::GetDeoptimizationEntry(deopt_kind);
188 masm()->CallForDeoptimization(target, deoptimization_id, exit->label(),
189 deopt_kind, exit->continue_label(),
190 jump_deoptimization_entry_label);
191
192 exit->set_emitted();
193
194 return kSuccess;
195}

201void CodeGenerator::AssembleCode() {
202 OptimizedCompilationInfo* info = this->info();
203 auto call_descriptor = linkage()->GetIncomingDescriptor();
204
205 // Compute incoming parameter count for code using JS linkage. This will
206 // ultimately set the parameter count on the resulting Code object.
207 if (call_descriptor->IsJSFunctionCall()) {
208 parameter_count_ = call_descriptor->ParameterSlotCount();
209 if (Builtins::IsBuiltinId(info->builtin())) {
210 CHECK_EQ(parameter_count_,
211 Builtins::GetStackParameterCount(info->builtin()));
212 } else if (info->has_bytecode_array()) {
213 CHECK_EQ(parameter_count_, info->bytecode_array()->parameter_count());
214 }
215 }
216
217 // Open a frame scope to indicate that there is a frame on the stack. The
218 // MANUAL indicates that the scope shouldn't actually generate code to set up
219 // the frame (that is done in AssemblePrologue).
220 FrameScope frame_scope(masm(), StackFrame::MANUAL);
221
222 if (info->source_positions()) {
223 AssembleSourcePosition(start_source_position());
224 }
225 offsets_info_.code_start_register_check = masm()->pc_offset();
226
227 masm()->CodeEntry();
228
229 // Check that {kJavaScriptCallCodeStartRegister} has been set correctly.
230 if (v8_flags.debug_code && info->called_with_code_start_register()) {
231 masm()->RecordComment("-- Prologue: check code start register --");
232 AssembleCodeStartRegisterCheck();
233 }
234
235#ifdef V8_ENABLE_LEAPTIERING
236 // Check that {kJavaScriptCallDispatchHandleRegister} has been set correctly.
237 if (v8_flags.debug_code && call_descriptor->IsJSFunctionCall()) {
238 masm()->RecordComment("-- Prologue: check dispatch handle register --");
239 AssembleDispatchHandleRegisterCheck();
240 }
241#endif
242
243#if V8_ENABLE_WEBASSEMBLY
244 if (info->code_kind() == CodeKind::WASM_TO_JS_FUNCTION ||
245 info->builtin() == Builtin::kWasmToJsWrapperCSA ||
246 wasm::BuiltinLookup::IsWasmBuiltinId(info->builtin())) {
247 // By default the code generator can convert slot IDs to SP-relative memory
248 // operands depending on the offset if the encoding is more efficient.
249 // However the SP may switch to the central stack for wasm-to-js wrappers
250 // and wasm builtins, so disable this optimization there.
251 // TODO(thibaudm): Disable this more selectively, only wasm builtins that
252 // call JS builtins can switch, and only around the call site.
253 frame_access_state()->SetFPRelativeOnly(true);
254 }
255#endif
256
257 offsets_info_.deopt_check = masm()->pc_offset();
258 // We want to bailout only from JS functions, which are the only ones
259 // that are optimized.
260 if (info->IsOptimizing()) {
261 DCHECK(call_descriptor->IsJSFunctionCall());
262 masm()->RecordComment("-- Prologue: check for deoptimization --");
263 BailoutIfDeoptimized();
264 }
265
266 // Define deoptimization literals for all inlined functions.
267 DCHECK_EQ(0u, deoptimization_literals_.size());
268 for (OptimizedCompilationInfo::InlinedFunctionHolder& inlined :
269 info->inlined_functions()) {
270 if (!inlined.shared_info.equals(info->shared_info())) {
271 int index = DefineDeoptimizationLiteral(
272 DeoptimizationLiteral(inlined.shared_info));
273 inlined.RegisterInlinedFunctionId(index);
274 }
275 }
276 inlined_function_count_ = deoptimization_literals_.size();
277
278 unwinding_info_writer_.SetNumberOfInstructionBlocks(
279 instructions()->InstructionBlockCount());
280
281 if (info->trace_turbo_json()) {
282 block_starts_.assign(instructions()->instruction_blocks().size(), -1);
283 instr_starts_.assign(instructions()->instructions().size(), {});
284 }
285 // Assemble instructions in assembly order.
286 offsets_info_.blocks_start = masm()->pc_offset();
287 for (const InstructionBlock* block : instructions()->ao_blocks()) {
288 // Align loop headers on vendor recommended boundaries.
289 if (block->ShouldAlignLoopHeader()) {
290 masm()->LoopHeaderAlign();
291 } else if (block->ShouldAlignCodeTarget()) {
292 masm()->CodeTargetAlign();
293 }
294
295 if (info->trace_turbo_json()) {
296 block_starts_[block->rpo_number().ToInt()] = masm()->pc_offset();
297 }
298 // Bind a label for a block.
299 current_block_ = block->rpo_number();
300 unwinding_info_writer_.BeginInstructionBlock(masm()->pc_offset(), block);
301 if (v8_flags.code_comments && !block->omitted_by_jump_threading()) {
302 std::ostringstream buffer;
303 buffer << "-- B" << block->rpo_number().ToInt() << " start";
304 if (block->IsDeferred()) buffer << " (deferred)";
305 if (!block->needs_frame()) buffer << " (no frame)";
306 if (block->must_construct_frame()) buffer << " (construct frame)";
307 if (block->must_deconstruct_frame()) buffer << " (deconstruct frame)";
308
309 if (block->IsLoopHeader()) {
310 buffer << " (loop up to " << block->loop_end().ToInt() << ")";
311 }
312 if (block->loop_header().IsValid()) {
313 buffer << " (in loop " << block->loop_header().ToInt() << ")";
314 }
315 buffer << " --";
316 masm()->RecordComment(buffer.str().c_str(), SourceLocation());
317 }
318
319 frame_access_state()->MarkHasFrame(block->needs_frame());
320
321 masm()->bind(GetLabel(current_block_));
322
323#ifdef V8_ENABLE_CONTROL_FLOW_INTEGRITY
324 if (block->IsSwitchTarget()) {
325 masm()->JumpTarget();
326 }
327#endif
328
329 if (block->must_construct_frame()) {
330 AssembleConstructFrame();
331 // We need to setup the root register after we assemble the prologue, to
332 // avoid clobbering callee saved registers in case of C linkage and
333 // using the roots.
334 // TODO(mtrofin): investigate how we can avoid doing this repeatedly.
335 if (call_descriptor->InitializeRootRegister()) {
336 masm()->InitializeRootRegister();
337 }
338 }
339#ifdef CAN_USE_RVV_INSTRUCTIONS
340 // RVV uses VectorUnit to emit vset{i}vl{i}, reducing the static and dynamic
341 // overhead of the vset{i}vl{i} instruction. However there are some jumps
342 // back between blocks. the Rvv instruction may get an incorrect vtype. so
343 // here VectorUnit needs to be cleared to ensure that the vtype is correct
344 // within the block.
345 masm()->VU.clear();
346#endif
347 if (V8_EMBEDDED_CONSTANT_POOL_BOOL && !block->needs_frame()) {
348 ConstantPoolUnavailableScope constant_pool_unavailable(masm());
349 result_ = AssembleBlock(block);
350 } else {
351 result_ = AssembleBlock(block);
352 }
353 if (result_ != kSuccess) return;
354 unwinding_info_writer_.EndInstructionBlock(block);
355 }
356
357 // Assemble all out-of-line code.
358 offsets_info_.out_of_line_code = masm()->pc_offset();
359 if (ools_) {
360 masm()->RecordComment("-- Out of line code --");
361 for (OutOfLineCode* ool = ools_; ool; ool = ool->next()) {
362 masm()->bind(ool->entry());
363 ool->Generate();
364 if (ool->exit()->is_bound()) masm()->jmp(ool->exit());
365 }
366 }
367
368 // This nop operation is needed to ensure that the trampoline is not
369 // confused with the pc of the call before deoptimization.
370 // The test regress/regress-259 is an example of where we need it.
371 masm()->nop();
372
373 // For some targets, we must make sure that constant and veneer pools are
374 // emitted before emitting the deoptimization exits.
375 PrepareForDeoptimizationExits(&deoptimization_exits_);
376
377 deopt_exit_start_offset_ = masm()->pc_offset();
378
379 // Assemble deoptimization exits.
380 offsets_info_.deoptimization_exits = masm()->pc_offset();
381 int last_updated = 0;
382 // We sort the deoptimization exits here so that the lazy ones will be visited
383 // last. We need this as lazy deopts might need additional instructions.
384 auto cmp = [](const DeoptimizationExit* a, const DeoptimizationExit* b) {
385 // The deoptimization exits are sorted so that lazy deopt exits appear after
386 // eager deopts.
387 static_assert(static_cast<int>(DeoptimizeKind::kLazy) ==
388 static_cast<int>(kLastDeoptimizeKind),
389 "lazy deopts are expected to be emitted last");
390 if (a->kind() != b->kind()) {
391 return a->kind() < b->kind();
392 }
393 return a->pc_offset() < b->pc_offset();
394 };
395 std::sort(deoptimization_exits_.begin(), deoptimization_exits_.end(), cmp);
396
397 {
398#ifdef V8_TARGET_ARCH_PPC64
399 v8::internal::Assembler::BlockTrampolinePoolScope block_trampoline_pool(
400 masm());
401#endif
402 for (DeoptimizationExit* exit : deoptimization_exits_) {
403 if (exit->emitted()) continue;
404 exit->set_deoptimization_id(next_deoptimization_id_++);
405 result_ = AssembleDeoptimizerCall(exit);
406 if (result_ != kSuccess) return;
407
408 // UpdateDeoptimizationInfo expects lazy deopts to be visited in pc_offset
409 // order, which is always the case since they are added to
410 // deoptimization_exits_ in that order, and the optional sort operation
411 // above preserves that order.
412 if (exit->kind() == DeoptimizeKind::kLazy) {
413 int trampoline_pc = exit->label()->pos();
414 last_updated = safepoints()->UpdateDeoptimizationInfo(
415 exit->pc_offset(), trampoline_pc, last_updated,
416 exit->deoptimization_id());
417 }
418 }
419 }
420
421 offsets_info_.pools = masm()->pc_offset();
422 // TODO(jgruber): Move all inlined metadata generation into a new,
423 // architecture-independent version of FinishCode. Currently, this includes
424 // the safepoint table, handler table, constant pool, and code comments, in
425 // that order.
426 FinishCode();
427
428 offsets_info_.jump_tables = masm()->pc_offset();
429 // Emit the jump tables.
430 if (jump_tables_) {
431 masm()->Align(kSystemPointerSize);
432 for (JumpTable* table = jump_tables_; table; table = table->next()) {
433 masm()->bind(table->label());
434 AssembleJumpTable(table->targets());
435 }
436 }
437
438 // The LinuxPerfJitLogger logs code up until here, excluding the safepoint
439 // table. Resolve the unwinding info now so it is aware of the same code
440 // size as reported by perf.
441 unwinding_info_writer_.Finish(masm()->pc_offset());
442
443 // Final alignment before starting on the metadata section.
444 masm()->Align(InstructionStream::kMetadataAlignment);
445
446 safepoints()->Emit(masm(), frame()->GetTotalFrameSlotCount());
447
448 // Emit the exception handler table.
449 if (!handlers_.empty()) {
450 handler_table_offset_ = HandlerTable::EmitReturnTableStart(masm());
451 for (size_t i = 0; i < handlers_.size(); ++i) {
452 int pos = handlers_[i].handler != nullptr ? handlers_[i].handler->pos()
453 : HandlerTable::kLazyDeopt;
454 HandlerTable::EmitReturnEntry(masm(), handlers_[i].pc_offset, pos);
455 }
456 }
457
458 masm()->MaybeEmitOutOfLineConstantPool();
459 masm()->FinalizeJumpOptimizationInfo();
460
461 result_ = kSuccess;
462}

465void CodeGenerator::AssembleArchBinarySearchSwitchRange(
466 Register input, RpoNumber def_block, std::pair<int32_t, Label*>* begin,
467 std::pair<int32_t, Label*>* end) {
468 if (end - begin < kBinarySearchSwitchMinimalCases) {

470 masm()->JumpIfEqual(input, begin->first, begin->second);
471 ++begin;
472 }
473 AssembleArchJumpRegardlessOfAssemblyOrder(def_block);
474 return;

477 Label less_label;
478 masm()->JumpIfLessThan(input, middle->first, &less_label);
479 AssembleArchBinarySearchSwitchRange(input, def_block, middle, end);
480 masm()->bind(&less_label);
481 AssembleArchBinarySearchSwitchRange(input, def_block, begin, middle);
482}

503MaybeHandle<Code> CodeGenerator::FinalizeCode() {
504 if (result_ != kSuccess) {
505 masm()->AbortedCodeGeneration();
506 return {};
507 }
508
509 // Allocate the source position table.
510 Handle<TrustedByteArray> source_positions =
511 source_position_table_builder_.ToSourcePositionTable(isolate());
512
513 // Allocate and install the code.
514 CodeDesc desc;
515 masm()->GetCode(isolate()->main_thread_local_isolate(), &desc, safepoints(),
516 handler_table_offset_);
517
518#if defined(V8_OS_WIN64)
519 if (Builtins::IsBuiltinId(info_->builtin())) {
520 isolate_->SetBuiltinUnwindData(info_->builtin(), masm()->GetUnwindInfo());
521 }
522#endif // V8_OS_WIN64
523
524 if (unwinding_info_writer_.eh_frame_writer()) {
525 unwinding_info_writer_.eh_frame_writer()->GetEhFrame(&desc);
526 }
527
528 Factory::CodeBuilder builder(isolate(), desc, info()->code_kind());
529 builder.set_builtin(info()->builtin())
530 .set_inlined_bytecode_size(info()->inlined_bytecode_size())
531 .set_parameter_count(parameter_count_)
532 .set_source_position_table(source_positions)
533 .set_is_turbofanned()
534 .set_stack_slots(frame()->GetTotalFrameSlotCount())
535 .set_profiler_data(info()->profiler_data())
536 .set_osr_offset(info()->osr_offset());
537
538 if (info()->function_context_specializing()) {
539 builder.set_is_context_specialized();
540 }
541
542 if (CodeKindUsesDeoptimizationData(info()->code_kind())) {
543 builder.set_deoptimization_data(GenerateDeoptimizationData());
544 DCHECK(info()->has_bytecode_array() ||
545 info()->code_kind() == CodeKind::WASM_FUNCTION);
546 }
547
548 MaybeHandle<Code> maybe_code = builder.TryBuild();
549
550 Handle<Code> code;
551 if (!maybe_code.ToHandle(&code)) {
552 masm()->AbortedCodeGeneration();
553 return {};
554 }
555
556 LOG_CODE_EVENT(isolate(), CodeLinePosInfoRecordEvent(
557 code->instruction_start(), *source_positions,
558 JitCodeEvent::JIT_CODE));
559
560 return code;
561}

570void CodeGenerator::RecordSafepoint(ReferenceMap* references, int pc_offset) {
571 auto safepoint = safepoints()->DefineSafepoint(masm(), pc_offset);
572
573 for (int tagged : frame()->tagged_slots()) {
574 safepoint.DefineTaggedStackSlot(tagged);
575 }
576
577 int frame_header_offset = frame()->GetFixedSlotCount();
578 for (const InstructionOperand& operand : references->reference_operands()) {
579 if (operand.IsStackSlot()) {
580 int index = LocationOperand::cast(operand).index();
581 DCHECK_LE(0, index);
582 // We might index values in the fixed part of the frame (i.e. the
583 // closure pointer or the context pointer); these are not spill slots
584 // and therefore don't work with the SafepointTable currently, but
585 // we also don't need to worry about them, since the GC has special
586 // knowledge about those fields anyway.
587 if (index < frame_header_offset) continue;
588 safepoint.DefineTaggedStackSlot(index);
589 }
590 }
591}

635 ZoneVector<MoveOperands*>* pushes) {
636 static constexpr int first_push_compatible_index =
637 kReturnAddressStackSlotCount;
638 pushes->clear();
639 for (int i = Instruction::FIRST_GAP_POSITION;
640 i <= Instruction::LAST_GAP_POSITION; ++i) {
641 Instruction::GapPosition inner_pos =
642 static_cast<Instruction::GapPosition>(i);
643 ParallelMove* parallel_move = instr->GetParallelMove(inner_pos);
644 if (parallel_move != nullptr) {
645 for (auto move : *parallel_move) {
646 InstructionOperand source = move->source();
647 InstructionOperand destination = move->destination();
648 // If there are any moves from slots that will be overridden by pushes,
649 // then the full gap resolver must be used since optimization with
650 // pushes don't participate in the parallel move and might clobber
651 // values needed for the gap resolve.
652 if (source.IsAnyStackSlot() && LocationOperand::cast(source).index() >=
653 first_push_compatible_index) {
654 pushes->clear();
655 return;
656 }
657 // TODO(danno): Right now, only consider moves from the FIRST gap for
658 // pushes. Theoretically, we could extract pushes for both gaps (there
659 // are cases where this happens), but the logic for that would also have
660 // to check to make sure that non-memory inputs to the pushes from the
661 // LAST gap don't get clobbered in the FIRST gap.
662 if (i == Instruction::FIRST_GAP_POSITION) {
663 if (destination.IsStackSlot() &&
664 LocationOperand::cast(destination).index() >=
665 first_push_compatible_index) {
666 int index = LocationOperand::cast(destination).index();
667 if (IsValidPush(source, push_type)) {
668 if (index >= static_cast<int>(pushes->size())) {
669 pushes->resize(index + 1);
670 }
671 (*pushes)[index] = move;
672 }
673 }
674 }
675 }
676 }
677 }
678
679 // For now, only support a set of continuous pushes at the end of the list.
680 size_t push_count_upper_bound = pushes->size();
681 size_t push_begin = push_count_upper_bound;
682 for (auto move : base::Reversed(*pushes)) {
683 if (move == nullptr) break;
684 push_begin--;
685 }
686 size_t push_count = pushes->size() - push_begin;
687 std::copy(pushes->begin() + push_begin,
688 pushes->begin() + push_begin + push_count, pushes->begin());
689 pushes->resize(push_count);
690}

740RpoNumber CodeGenerator::ComputeBranchInfo(BranchInfo* branch,
741 FlagsCondition condition,
742 Instruction* instr) {
743 // Assemble a branch after this instruction.
744 InstructionOperandConverter i(this, instr);
745 RpoNumber true_rpo =
746 i.InputRpo(instr->InputCount() - kBranchEndOffsetOfTrueBlock);
747 RpoNumber false_rpo =
748 i.InputRpo(instr->InputCount() - kBranchEndOffsetOfFalseBlock);
749
750 if (true_rpo == false_rpo) {
751 return true_rpo;
752 }
753 if (IsNextInAssemblyOrder(true_rpo) || instructions()
754 ->InstructionBlockAt(false_rpo)
755 ->IsLoopHeaderInAssemblyOrder()) {
756 // true block is next, can fall through if condition negated.
757 // false block is loop header, can save one jump if condition negated.
758 std::swap(true_rpo, false_rpo);
759 condition = NegateFlagsCondition(condition);
760 }
761 branch->condition = condition;
762 branch->true_label = GetLabel(true_rpo);
763 branch->false_label = GetLabel(false_rpo);
764 branch->fallthru = IsNextInAssemblyOrder(false_rpo);
765 return RpoNumber::Invalid();
766}

768CodeGenerator::CodeGenResult CodeGenerator::AssembleInstruction(
769 int instruction_index, const InstructionBlock* block) {
770 Instruction* instr = instructions()->InstructionAt(instruction_index);
771 if (info()->trace_turbo_json()) {
772 instr_starts_[instruction_index].gap_pc_offset = masm()->pc_offset();
773 }
774 int first_unused_stack_slot;
775 FlagsMode mode = FlagsModeField::decode(instr->opcode());
776 if (mode != kFlags_trap) {
777 AssembleSourcePosition(instr);
778 }
779 bool adjust_stack =
780 GetSlotAboveSPBeforeTailCall(instr, &first_unused_stack_slot);
781 if (adjust_stack) AssembleTailCallBeforeGap(instr, first_unused_stack_slot);
782 if (instr->opcode() == kArchNop && block->successors().empty() &&
783 block->code_end() - block->code_start() == 1) {
784 // When the frame-less dummy end block in Turbofan contains a Phi node,
785 // don't attempt to access spill slots.
786 // TODO(dmercadier): When the switch to Turboshaft is complete, this
787 // will no longer be required.
788 } else {
789 AssembleGaps(instr);
790 }
791 if (adjust_stack) AssembleTailCallAfterGap(instr, first_unused_stack_slot);
792 DCHECK_IMPLIES(
793 block->must_deconstruct_frame(),
794 instr != instructions()->InstructionAt(block->last_instruction_index()) ||
795 instr->IsRet() || instr->IsJump());
796 if (instr->IsJump() && block->must_deconstruct_frame()) {
797 AssembleDeconstructFrame();
798 }
799 if (info()->trace_turbo_json()) {
800 instr_starts_[instruction_index].arch_instr_pc_offset = masm()->pc_offset();
801 }
802 // Assemble architecture-specific code for the instruction.
803 CodeGenResult result = AssembleArchInstruction(instr);
804 if (result != kSuccess) return result;
805
806 if (info()->trace_turbo_json()) {
807 instr_starts_[instruction_index].condition_pc_offset = masm()->pc_offset();
808 }
809
810 FlagsCondition condition = FlagsConditionField::decode(instr->opcode());
811 switch (mode) {
812 case kFlags_branch:
813 case kFlags_conditional_branch: {
814 if (mode == kFlags_conditional_branch) {
815 InstructionOperandConverter i(this, instr);
816 condition = static_cast<FlagsCondition>(
817 i.ToConstant(instr->InputAt(instr->InputCount() -
818 kConditionalBranchEndOffsetOfCondition))
819 .ToInt64());
820 }
821 BranchInfo branch;
822 RpoNumber target = ComputeBranchInfo(&branch, condition, instr);
823 if (target.IsValid()) {
824 // redundant branch.
825 if (!IsNextInAssemblyOrder(target)) {
826 AssembleArchJump(target);
827 }
828 return kSuccess;
829 }
830 if (mode == kFlags_branch) {
831 // Assemble architecture-specific branch.
832 AssembleArchBranch(instr, &branch);
833 } else {
834 AssembleArchConditionalBranch(instr, &branch);
835 }
836 break;
837 }
838 case kFlags_deoptimize: {
839 // Assemble a conditional eager deoptimization after this instruction.
840 InstructionOperandConverter i(this, instr);
841 size_t frame_state_offset =
842 DeoptFrameStateOffsetField::decode(instr->opcode());
843 size_t immediate_args_count =
844 DeoptImmedArgsCountField::decode(instr->opcode());
845 DeoptimizationExit* const exit = AddDeoptimizationExit(
846 instr, frame_state_offset, immediate_args_count);
847 BranchInfo branch;
848 branch.condition = condition;
849 branch.true_label = exit->label();
850 branch.false_label = exit->continue_label();
851 branch.fallthru = true;
852 AssembleArchDeoptBranch(instr, &branch);
853 masm()->bind(exit->continue_label());
854 break;
855 }
856 case kFlags_set: {
857 // Assemble a boolean materialization after this instruction.
858 AssembleArchBoolean(instr, condition);
859 break;
860 }
861 case kFlags_conditional_set: {
862 // Assemble a conditional boolean materialization after this instruction.
863 AssembleArchConditionalBoolean(instr);
864 break;
865 }
866 case kFlags_select: {
867 AssembleArchSelect(instr, condition);
868 break;
869 }
870 case kFlags_trap: {
871#if V8_ENABLE_WEBASSEMBLY
872 AssembleArchTrap(instr, condition);
873 break;
874#else
875 UNREACHABLE();
876#endif // V8_ENABLE_WEBASSEMBLY
877 }
878 case kFlags_none: {
879 break;
880 }
881 }
882
883 return kSuccess;
884}

896 if (!source_position.IsKnown()) return;
897 source_position_table_builder_.AddPosition(masm()->pc_offset(),

981 if (deopt_count == 0 && !info->is_osr()) {

1045 DeoptimizationExit* deoptimization_exit = deoptimization_exits_[i];
1046 CHECK_NOT_NULL(deoptimization_exit);

1050 i, Smi::FromInt(deoptimization_exit->translation_id()));
1051 data->SetPc(i, Smi::FromInt(deoptimization_exit->pc_offset()));
1052#ifdef DEBUG
1053 data->SetNodeId(i, Smi::FromInt(deoptimization_exit->node_id()));
1054#endif // DEBUG
1055 }
1056
1057#ifdef DEBUG
1058 data->Verify(info->bytecode_array());
1059#endif // DEBUG
1060 return data;

1255#endif
1256 ) {
1257 return; // Stub with no SharedFunctionInfo.
1258 }
1259 shared_info = info()->shared_info();
1260 }
1261
1262 const BytecodeOffset bailout_id = descriptor->bailout_id();
1263
1264 const int shared_info_id =
1265#if V8_ENABLE_WEBASSEMBLY
1266 shared_info.is_null()
1267 ? DefineDeoptimizationLiteral(DeoptimizationLiteral(uint64_t{0}))
1268 : DefineDeoptimizationLiteral(DeoptimizationLiteral(shared_info));
1269 CHECK_IMPLIES(shared_info.is_null(), v8_flags.wasm_deopt);
1270#else
1271 DefineDeoptimizationLiteral(DeoptimizationLiteral(shared_info));
1272#endif
1273
1274 const unsigned int height =
1275 static_cast<unsigned int>(descriptor->GetHeight());
1276
1277 switch (descriptor->type()) {
1278 case FrameStateType::kUnoptimizedFunction: {
1279 int bytecode_array_id = DefineProtectedDeoptimizationLiteral(
1280 descriptor->bytecode_array().ToHandleChecked());
1281 int return_offset = 0;
1282 int return_count = 0;
1283 if (!state_combine.IsOutputIgnored()) {
1284 return_offset = static_cast<int>(state_combine.GetOffsetToPokeAt());
1285 return_count = static_cast<int>(iter->instruction()->OutputCount());
1286 }
1287 translations_.BeginInterpretedFrame(bailout_id, shared_info_id,
1288 bytecode_array_id, height,
1289 return_offset, return_count);
1290 break;
1291 }
1292 case FrameStateType::kInlinedExtraArguments:
1293 translations_.BeginInlinedExtraArguments(
1294 shared_info_id, height,
1295 descriptor->bytecode_array().ToHandleChecked()->parameter_count());
1296 break;
1297 case FrameStateType::kConstructCreateStub:
1298 translations_.BeginConstructCreateStubFrame(shared_info_id, height);
1299 break;
1300 case FrameStateType::kConstructInvokeStub:
1301 translations_.BeginConstructInvokeStubFrame(shared_info_id);
1302 break;
1303 case FrameStateType::kBuiltinContinuation: {
1304 translations_.BeginBuiltinContinuationFrame(bailout_id, shared_info_id,
1305 height);
1306 break;
1307 }
1308#if V8_ENABLE_WEBASSEMBLY
1309 case FrameStateType::kWasmInlinedIntoJS:
1310 translations_.BeginWasmInlinedIntoJSFrame(bailout_id, shared_info_id,
1311 height);
1312 break;
1313 case FrameStateType::kJSToWasmBuiltinContinuation: {
1314 const JSToWasmFrameStateDescriptor* js_to_wasm_descriptor =
1315 static_cast<const JSToWasmFrameStateDescriptor*>(descriptor);
1316 translations_.BeginJSToWasmBuiltinContinuationFrame(
1317 bailout_id, shared_info_id, height,
1318 js_to_wasm_descriptor->return_kind());
1319 break;
1320 }
1321 case FrameStateType::kLiftoffFunction:
1322 translations_.BeginLiftoffFrame(bailout_id, height,
1323 descriptor->GetWasmFunctionIndex());
1324 break;
1325#endif // V8_ENABLE_WEBASSEMBLY
1326 case FrameStateType::kJavaScriptBuiltinContinuation: {
1327 translations_.BeginJavaScriptBuiltinContinuationFrame(
1328 bailout_id, shared_info_id, height);
1329 break;
1330 }
1331 case FrameStateType::kJavaScriptBuiltinContinuationWithCatch: {
1332 translations_.BeginJavaScriptBuiltinContinuationWithCatchFrame(
1333 bailout_id, shared_info_id, height);
1334 break;
1335 }
1336 }
1337
1338 TranslateFrameStateDescriptorOperands(descriptor, iter);
1339}

1341DeoptimizationExit* CodeGenerator::BuildTranslation(
1342 Instruction* instr, int pc_offset, size_t frame_state_offset,
1343 size_t immediate_args_count, OutputFrameStateCombine state_combine) {
1344 DeoptimizationEntry const& entry =
1345 GetDeoptimizationEntry(instr, frame_state_offset);
1346 FrameStateDescriptor* const descriptor = entry.descriptor();
1347 frame_state_offset++;
1348
1349 const int translation_index = translations_.BeginTranslation(
1350 static_cast<int>(descriptor->GetFrameCount()),
1351 static_cast<int>(descriptor->GetJSFrameCount()),
1352 entry.feedback().IsValid());
1353 if (entry.feedback().IsValid()) {
1354 DeoptimizationLiteral literal =
1355 DeoptimizationLiteral(entry.feedback().vector);
1356 int literal_id = DefineDeoptimizationLiteral(literal);
1357 translations_.AddUpdateFeedback(literal_id, entry.feedback().slot.ToInt());
1358 }
1359 InstructionOperandIterator iter(instr, frame_state_offset);
1360 BuildTranslationForFrameStateDescriptor(descriptor, &iter, state_combine);
1361
1362 DeoptimizationExit* const exit = zone()->New<DeoptimizationExit>(
1363 current_source_position_, descriptor->bailout_id(), translation_index,
1364 pc_offset, entry.kind(), entry.reason(),
1365#ifdef DEBUG
1366 entry.node_id());
1367#else // DEBUG
1368 0);
1369#endif // DEBUG
1370 if (immediate_args_count != 0) {
1371 auto immediate_args = zone()->New<ZoneVector<ImmediateOperand*>>(zone());
1372 InstructionOperandIterator imm_iter(
1373 instr, frame_state_offset - immediate_args_count - 1);
1374 for (size_t i = 0; i < immediate_args_count; i++) {
1375 immediate_args->emplace_back(ImmediateOperand::cast(imm_iter.Advance()));
1376 }
1377 exit->set_immediate_args(immediate_args);
1378 }
1379
1380 deoptimization_exits_.push_back(exit);
1381 return exit;
1382}

1386 MachineType type) {

1388 if (type.representation() == MachineRepresentation::kBit) {
1389 translations_.StoreBoolStackSlot(LocationOperand::cast(op)->index());
1390 } else if (type == MachineType::Int8() || type == MachineType::Int16() ||
1391 type == MachineType::Int32()) {
1392 translations_.StoreInt32StackSlot(LocationOperand::cast(op)->index());
1393 } else if (type == MachineType::Uint8() || type == MachineType::Uint16() ||
1394 type == MachineType::Uint32()) {
1395 translations_.StoreUint32StackSlot(LocationOperand::cast(op)->index());
1396 } else if (type == MachineType::Int64()) {
1397 translations_.StoreInt64StackSlot(LocationOperand::cast(op)->index());
1398 } else if (type == MachineType::SignedBigInt64()) {
1399 translations_.StoreSignedBigInt64StackSlot(
1400 LocationOperand::cast(op)->index());
1401 } else if (type == MachineType::UnsignedBigInt64()) {
1402 translations_.StoreUnsignedBigInt64StackSlot(
1403 LocationOperand::cast(op)->index());
1404 } else {
1405#if defined(V8_COMPRESS_POINTERS)
1406 CHECK(MachineRepresentation::kTagged == type.representation() ||
1407 MachineRepresentation::kCompressed == type.representation());
1408#else
1409 CHECK(MachineRepresentation::kTagged == type.representation());
1410#endif
1411 translations_.StoreStackSlot(LocationOperand::cast(op)->index());
1412 }
1413 } else if (op->IsFPStackSlot()) {
1414 switch (type.representation()) {
1415 case MachineRepresentation::kFloat32:
1416 translations_.StoreFloatStackSlot(LocationOperand::cast(op)->index());
1417 break;
1418 case MachineRepresentation::kFloat64:
1419 if (type.semantic() == MachineSemantic::kHoleyFloat64) {
1420 translations_.StoreHoleyDoubleStackSlot(
1421 LocationOperand::cast(op)->index());
1422 } else {
1423 translations_.StoreDoubleStackSlot(
1424 LocationOperand::cast(op)->index());
1425 }
1426 break;
1427 case MachineRepresentation::kSimd128:
1428 translations_.StoreSimd128StackSlot(LocationOperand::cast(op)->index());
1429 break;
1430 default:
1431 UNREACHABLE();
1432 }
1433 } else if (op->IsRegister()) {
1434 InstructionOperandConverter converter(this, instr);
1435 if (type.representation() == MachineRepresentation::kBit) {
1436 translations_.StoreBoolRegister(converter.ToRegister(op));
1437 } else if (type == MachineType::Int8() || type == MachineType::Int16() ||
1438 type == MachineType::Int32()) {
1439 translations_.StoreInt32Register(converter.ToRegister(op));
1440 } else if (type == MachineType::Uint8() || type == MachineType::Uint16() ||
1441 type == MachineType::Uint32()) {
1442 translations_.StoreUint32Register(converter.ToRegister(op));
1443 } else if (type == MachineType::Int64()) {
1444 translations_.StoreInt64Register(converter.ToRegister(op));
1445 } else if (type == MachineType::SignedBigInt64()) {
1446 translations_.StoreSignedBigInt64Register(converter.ToRegister(op));
1447 } else if (type == MachineType::UnsignedBigInt64()) {
1448 translations_.StoreUnsignedBigInt64Register(converter.ToRegister(op));
1449 } else {
1450#if defined(V8_COMPRESS_POINTERS)
1451 CHECK(MachineRepresentation::kTagged == type.representation() ||
1452 MachineRepresentation::kCompressed == type.representation());
1453#else
1454 CHECK(MachineRepresentation::kTagged == type.representation());
1455#endif
1456 translations_.StoreRegister(converter.ToRegister(op));
1457 }
1458 } else if (op->IsFPRegister()) {
1459 InstructionOperandConverter converter(this, instr);
1460 switch (type.representation()) {
1461 case MachineRepresentation::kFloat32:
1462 translations_.StoreFloatRegister(converter.ToFloatRegister(op));
1463 break;
1464 case MachineRepresentation::kFloat64:
1465 if (type.semantic() == MachineSemantic::kHoleyFloat64) {
1466 translations_.StoreHoleyDoubleRegister(
1467 converter.ToDoubleRegister(op));
1468 } else {
1469 translations_.StoreDoubleRegister(converter.ToDoubleRegister(op));
1470 }
1471 break;
1472 case MachineRepresentation::kSimd128:
1473 translations_.StoreSimd128Register(converter.ToSimd128Register(op));
1474 break;
1475 default:
1476 UNREACHABLE();
1477 }
1478 } else {
1479 CHECK(op->IsImmediate());
1480 InstructionOperandConverter converter(this, instr);
1481 Constant constant = converter.ToConstant(op);
1482 DeoptimizationLiteral literal;
1483
1484#if V8_ENABLE_WEBASSEMBLY
1485 if (info_->IsWasm() && v8_flags.wasm_deopt) {
1486 switch (type.representation()) {
1487 case MachineRepresentation::kWord32:
1488 literal = DeoptimizationLiteral(constant.ToInt32());
1489 break;
1490 case MachineRepresentation::kWord64:
1491 literal = DeoptimizationLiteral(constant.ToInt64());
1492 break;
1493 case MachineRepresentation::kFloat32:
1494 literal = DeoptimizationLiteral(constant.ToFloat32Safe());
1495 break;
1496 case MachineRepresentation::kFloat64:
1497 literal = DeoptimizationLiteral(Float64(constant.ToFloat64()));
1498 break;
1499 case MachineRepresentation::kTagged: {
1500 DCHECK(!PointerCompressionIsEnabled() ||
1501 base::IsInRange(constant.ToInt64(), 0u, UINT32_MAX));
1502 Tagged<Smi> smi(static_cast<Address>(constant.ToInt64()));
1503 DCHECK(IsSmi(smi));
1504 literal = DeoptimizationLiteral(smi);
1505 break;
1506 }
1507 default:
1508 UNIMPLEMENTED();
1509 }
1510 int literal_id = DefineDeoptimizationLiteral(literal);
1511 translations_.StoreLiteral(literal_id);
1512 return;
1513 }
1514#endif
1515
1516 switch (constant.type()) {
1517 case Constant::kInt32:
1518 if (type.representation() == MachineRepresentation::kTagged) {
1519 // When pointers are 4 bytes, we can use int32 constants to represent
1520 // Smis.
1521 DCHECK_EQ(4, kSystemPointerSize);
1522 Tagged<Smi> smi(static_cast<Address>(constant.ToInt32()));
1523 DCHECK(IsSmi(smi));
1524 literal = DeoptimizationLiteral(static_cast<double>(smi.value()));
1525 } else if (type.representation() == MachineRepresentation::kBit) {
1526 if (constant.ToInt32() == 0) {
1527 literal =
1528 DeoptimizationLiteral(isolate()->factory()->false_value());
1529 } else {
1530 DCHECK_EQ(1, constant.ToInt32());
1531 literal = DeoptimizationLiteral(isolate()->factory()->true_value());
1532 }
1533 } else {
1534 DCHECK(type == MachineType::Int32() ||
1535 type == MachineType::Uint32() ||
1536 type.representation() == MachineRepresentation::kWord32 ||
1537 type.representation() == MachineRepresentation::kNone);
1538 DCHECK(type.representation() != MachineRepresentation::kNone ||
1539 constant.ToInt32() == FrameStateDescriptor::kImpossibleValue);
1540 if (type == MachineType::Uint32()) {
1541 literal = DeoptimizationLiteral(
1542 static_cast<double>(static_cast<uint32_t>(constant.ToInt32())));
1543 } else {
1544 literal =
1545 DeoptimizationLiteral(static_cast<double>(constant.ToInt32()));
1546 }
1547 }
1548 break;
1549 case Constant::kInt64:
1550 DCHECK_EQ(8, kSystemPointerSize);
1551 if (type == MachineType::SignedBigInt64()) {
1552 literal = DeoptimizationLiteral(constant.ToInt64());
1553 } else if (type == MachineType::UnsignedBigInt64()) {
1554 literal =
1555 DeoptimizationLiteral(static_cast<uint64_t>(constant.ToInt64()));
1556 } else if (type.representation() == MachineRepresentation::kWord64) {
1557 // TODO(nicohartmann@, chromium:41497374): Disabling this CHECK
1558 // because we can see cases where this is violated in unreachable
1559 // code. We should re-enable once we have an idea on how to prevent
1560 // this from happening.
1561 // CHECK_EQ(
1562 // constant.ToInt64(),
1563 // static_cast<int64_t>(static_cast<double>(constant.ToInt64())));
1564 literal =
1565 DeoptimizationLiteral(static_cast<double>(constant.ToInt64()));
1566 } else {
1567 // When pointers are 8 bytes, we can use int64 constants to represent
1568 // Smis.
1569 DCHECK_EQ(MachineRepresentation::kTagged, type.representation());
1570 Tagged<Smi> smi(static_cast<Address>(constant.ToInt64()));
1571 DCHECK(IsSmi(smi));
1572 literal = DeoptimizationLiteral(static_cast<double>(smi.value()));
1573 }
1574 break;
1575 case Constant::kFloat32:
1576 DCHECK(type.representation() == MachineRepresentation::kFloat32 ||
1577 type.representation() == MachineRepresentation::kTagged);
1578 literal =
1579 DeoptimizationLiteral(static_cast<double>(constant.ToFloat32()));
1580 break;
1581 case Constant::kFloat64:
1582 DCHECK(type.representation() == MachineRepresentation::kFloat64 ||
1583 type.representation() == MachineRepresentation::kTagged);
1584 if (type == MachineType::HoleyFloat64() &&
1585 constant.ToFloat64().AsUint64() == kHoleNanInt64) {
1586 literal = DeoptimizationLiteral::HoleNaN();
1587 } else {
1588 literal = DeoptimizationLiteral(constant.ToFloat64().value());
1589 }
1590 break;
1591 case Constant::kHeapObject:
1592 DCHECK_EQ(MachineRepresentation::kTagged, type.representation());
1593 literal = DeoptimizationLiteral(constant.ToHeapObject());
1594 break;
1595 case Constant::kCompressedHeapObject:
1596 DCHECK_EQ(MachineType::AnyTagged(), type);
1597 literal = DeoptimizationLiteral(constant.ToHeapObject());
1598 break;
1599 default:
1600 UNREACHABLE();
1601 }
1602 if (literal.object().equals(info()->closure()) &&
1603 info()->function_context_specializing()) {
1604 translations_.StoreJSFrameFunction();
1605 } else {
1606 int literal_id = DefineDeoptimizationLiteral(literal);
1607 translations_.StoreLiteral(literal_id);
1608 }
1609 }
1610}
