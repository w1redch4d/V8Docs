22void HeapAllocator::Setup(LinearAllocationArea* new_allocation_info,
23 LinearAllocationArea* old_allocation_info) {
24 for (int i = FIRST_SPACE; i <= LAST_SPACE; ++i) {
25 spaces_[i] = heap_->space(i);
26 }
27
28 if ((heap_->new_space() || v8_flags.sticky_mark_bits) &&
29 local_heap_->is_main_thread()) {
30 new_space_allocator_.emplace(
31 local_heap_,
32 v8_flags.sticky_mark_bits
33 ? static_cast<SpaceWithLinearArea*>(heap_->sticky_space())
34 : static_cast<SpaceWithLinearArea*>(heap_->new_space()),
35 MainAllocator::IsNewGeneration::kYes, new_allocation_info);
36 }
37
38 old_space_allocator_.emplace(local_heap_, heap_->old_space(),
39 MainAllocator::IsNewGeneration::kNo,
40 old_allocation_info);
41
42 trusted_space_allocator_.emplace(local_heap_, heap_->trusted_space(),
43 MainAllocator::IsNewGeneration::kNo);
44 code_space_allocator_.emplace(local_heap_, heap_->code_space(),
45 MainAllocator::IsNewGeneration::kNo);
46
47 if (heap_->isolate()->has_shared_space()) {
48 shared_space_allocator_.emplace(local_heap_,
49 heap_->shared_allocation_space(),
50 MainAllocator::IsNewGeneration::kNo);
51 shared_lo_space_ = heap_->shared_lo_allocation_space();
52
53 shared_trusted_space_allocator_.emplace(
54 local_heap_, heap_->shared_trusted_allocation_space(),
55 MainAllocator::IsNewGeneration::kNo);
56 shared_trusted_lo_space_ = heap_->shared_trusted_lo_allocation_space();
57 }
58}

64AllocationResult HeapAllocator::AllocateRawLargeInternal(
65 int size_in_bytes, AllocationType allocation, AllocationOrigin origin,
66 AllocationAlignment alignment) {
67 DCHECK_GT(size_in_bytes, heap_->MaxRegularHeapObjectSize(allocation));
68 switch (allocation) {
69 case AllocationType::kYoung:
70 return new_lo_space()->AllocateRaw(local_heap_, size_in_bytes);
71 case AllocationType::kOld:
72 return lo_space()->AllocateRaw(local_heap_, size_in_bytes);
73 case AllocationType::kCode:
74 return code_lo_space()->AllocateRaw(local_heap_, size_in_bytes);
75 case AllocationType::kSharedOld:
76 return shared_lo_space()->AllocateRaw(local_heap_, size_in_bytes);
77 case AllocationType::kTrusted:
78 return trusted_lo_space()->AllocateRaw(local_heap_, size_in_bytes);
79 case AllocationType::kSharedTrusted:
80 return shared_trusted_lo_space()->AllocateRaw(local_heap_, size_in_bytes);
81 case AllocationType::kMap:
82 case AllocationType::kReadOnly:
83 case AllocationType::kSharedMap:
84 UNREACHABLE();
85 }
86}

90constexpr AllocationSpace AllocationTypeToGCSpace(AllocationType type) {
91 switch (type) {
92 case AllocationType::kYoung:
93 return NEW_SPACE;
94 case AllocationType::kOld:
95 case AllocationType::kCode:
96 case AllocationType::kMap:
97 case AllocationType::kTrusted:
98 // OLD_SPACE indicates full GC.
99 return OLD_SPACE;
100 case AllocationType::kReadOnly:
101 case AllocationType::kSharedMap:
102 case AllocationType::kSharedOld:
103 case AllocationType::kSharedTrusted:
104 UNREACHABLE();
105 }
106}

110AllocationResult HeapAllocator::AllocateRawWithLightRetrySlowPath(
111 int size, AllocationType allocation, AllocationOrigin origin,
112 AllocationAlignment alignment) {
113 AllocationResult result = AllocateRaw(size, allocation, origin, alignment);
114 if (!result.IsFailure()) {
115 return result;
116 }
117
118 // Two GCs before returning failure.
119 for (int i = 0; i < 2; i++) {
120 CollectGarbage(allocation);
121 result = RetryAllocateRaw(size, allocation, origin, alignment);
122 if (!result.IsFailure()) {
123 return result;
124 }
125 }
126 return result;
127}

129void HeapAllocator::CollectGarbage(AllocationType allocation) {
130 if (IsSharedAllocationType(allocation)) {
131 heap_->CollectGarbageShared(local_heap_,
132 GarbageCollectionReason::kAllocationFailure);
133 } else if (local_heap_->is_main_thread()) {
134 // On the main thread we can directly start the GC.
135 AllocationSpace space_to_gc = AllocationTypeToGCSpace(allocation);
136 heap_->CollectGarbage(space_to_gc,
137 GarbageCollectionReason::kAllocationFailure);
138 } else {
139 // Request GC from main thread.
140 heap_->CollectGarbageFromAnyThread(local_heap_);
141 }
142}

144AllocationResult HeapAllocator::AllocateRawWithRetryOrFailSlowPath(
145 int size, AllocationType allocation, AllocationOrigin origin,
146 AllocationAlignment alignment) {
147 AllocationResult result =
148 AllocateRawWithLightRetrySlowPath(size, allocation, origin, alignment);
149 if (!result.IsFailure()) return result;
150
151 CollectAllAvailableGarbage(allocation);
152 result = RetryAllocateRaw(size, allocation, origin, alignment);
153
154 if (!result.IsFailure()) {
155 return result;
156 }
157
158 V8::FatalProcessOutOfMemory(heap_->isolate(), "CALL_AND_RETRY_LAST",
159 V8::kHeapOOM);
160}

162void HeapAllocator::CollectAllAvailableGarbage(AllocationType allocation) {
163 if (IsSharedAllocationType(allocation)) {
164 heap_->CollectGarbageShared(heap_->main_thread_local_heap(),
165 GarbageCollectionReason::kLastResort);
166 } else if (local_heap_->is_main_thread()) {
167 // On the main thread we can directly start the GC.
168 heap_->CollectAllAvailableGarbage(GarbageCollectionReason::kLastResort);
169 } else {
170 // Request GC from main thread.
171 heap_->CollectGarbageFromAnyThread(local_heap_);
172 }
173}

175AllocationResult HeapAllocator::RetryAllocateRaw(
176 int size_in_bytes, AllocationType allocation, AllocationOrigin origin,
177 AllocationAlignment alignment) {
178 // Initially flags on the LocalHeap are always disabled. They are only
179 // active while this method is running.
180 DCHECK(!local_heap_->IsRetryOfFailedAllocation());
181 local_heap_->SetRetryOfFailedAllocation(true);
182 AllocationResult result =
183 AllocateRaw(size_in_bytes, allocation, origin, alignment);
184 local_heap_->SetRetryOfFailedAllocation(false);
185 return result;
186}

389void HeapAllocator::UpdateAllocationTimeout() {
390 if (v8_flags.random_gc_interval > 0) {
391 const int new_timeout = heap_->isolate()->fuzzer_rng()->NextInt(
392 v8_flags.random_gc_interval + 1);
393 // Reset the allocation timeout, but make sure to allow at least a few
394 // allocations after a collection. The reason for this is that we have a lot
395 // of allocation sequences and we assume that a garbage collection will
396 // allow the subsequent allocation attempts to go through.
397 constexpr int kFewAllocationsHeadroom = 6;
398 int timeout = std::max(kFewAllocationsHeadroom, new_timeout);
399 SetAllocationTimeout(timeout);
400 DCHECK(allocation_timeout_.has_value());
401 return;
402 }
403
404 int timeout = allocation_gc_interval_.load(std::memory_order_relaxed);
405 SetAllocationTimeout(timeout);
406}

408bool HeapAllocator::ReachedAllocationTimeout() {
409 DCHECK(allocation_timeout_.has_value());
410
411 if (heap_->always_allocate() || local_heap_->IsRetryOfFailedAllocation()) {
412 return false;
413 }
414
415 allocation_timeout_ = std::max(0, allocation_timeout_.value() - 1);
416 return allocation_timeout_.value() <= 0;
417}
