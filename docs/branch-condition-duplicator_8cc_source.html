19bool CanDuplicate(Node* node) {
20 // We only allow duplication of comparisons and "cheap" binary operations
21 // (cheap = not multiplication or division). The idea is that those
22 // instructions set the ZF flag, and thus do not require a "== 0" to be added
23 // before the branch. Duplicating other nodes, on the other hand, makes little
24 // sense, because a "== 0" would need to be inserted in branches anyways.
25 switch (node->opcode()) {
26#define BRANCH_CASE(op) \
27 case IrOpcode::k##op: \
28 break;
29 MACHINE_COMPARE_BINOP_LIST(BRANCH_CASE)
30 case IrOpcode::kInt32Add:
31 case IrOpcode::kInt32Sub:
32 case IrOpcode::kWord32And:
33 case IrOpcode::kWord32Or:
34 case IrOpcode::kInt64Add:
35 case IrOpcode::kInt64Sub:
36 case IrOpcode::kWord64And:
37 case IrOpcode::kWord64Or:
38 case IrOpcode::kWord32Shl:
39 case IrOpcode::kWord32Shr:
40 case IrOpcode::kWord64Shl:
41 case IrOpcode::kWord64Shr:
42 break;
43 default:
44 return false;
45 }
46
47 // We do not duplicate nodes if all their inputs are used a single time,
48 // because this would keep those inputs alive, thus increasing register
49 // pressure.
50 int all_inputs_have_only_a_single_use = true;
51 for (Node* input : node->inputs()) {
52 if (input->UseCount() > 1) {
53 all_inputs_have_only_a_single_use = false;
54 }
55 }
56 if (all_inputs_have_only_a_single_use) {
57 return false;
58 }
59
60 return true;
61}

69void BranchConditionDuplicator::DuplicateConditionIfNeeded(Node* node) {
70 if (!IsBranch(node)) return;
71
72 Node* condNode = node->InputAt(0);
73 if (condNode->BranchUseCount() > 1 && CanDuplicate(condNode)) {
74 node->ReplaceInput(0, DuplicateNode(condNode));
75 }
76}

78void BranchConditionDuplicator::Enqueue(Node* node) {
79 if (seen_.Get(node)) return;
80 seen_.Set(node, true);
81 to_visit_.push(node);

84void BranchConditionDuplicator::VisitNode(Node* node) {
85 DuplicateConditionIfNeeded(node);
86
87 for (int i = 0; i < node->op()->ControlInputCount(); i++) {
88 Enqueue(NodeProperties::GetControlInput(node, i));
89 }
90}

92void BranchConditionDuplicator::ProcessGraph() {
93 Enqueue(graph_->end());
94 while (!to_visit_.empty()) {
95 Node* node = to_visit_.front();
96 to_visit_.pop();
97 VisitNode(node);
98 }
