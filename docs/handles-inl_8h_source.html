204HandleScope& HandleScope::operator=(HandleScope&& other) V8_NOEXCEPT {
205 if (isolate_ == nullptr) {

209#ifdef V8_ENABLE_CHECKS
210 CHECK_EQ(scope_level_, isolate_->handle_scope_data()->level);

212 CloseScope(isolate_, prev_next_, prev_limit_);
213 }
214 prev_next_ = other.prev_next_;
215 prev_limit_ = other.prev_limit_;
216 other.isolate_ = nullptr;
217#ifdef V8_ENABLE_CHECKS
218 scope_level_ = other.scope_level_;
219#endif
220 return *this;
221}

223void HandleScope::CloseScope(Isolate* isolate, Address* prev_next,
224 Address* prev_limit) {
225#ifdef DEBUG
226 int before = v8_flags.check_handle_count ? NumberOfHandles(isolate) : 0;
227#endif
228 DCHECK_NOT_NULL(isolate);
229 HandleScopeData* current = isolate->handle_scope_data();
230
231 std::swap(current->next, prev_next);
232 current->level--;
233 Address* limit = prev_next;
234 if (V8_UNLIKELY(current->limit != prev_limit)) {
235 current->limit = prev_limit;
236 limit = prev_limit;
237 DeleteExtensions(isolate);
238 }
239#ifdef ENABLE_LOCAL_HANDLE_ZAPPING
240 ZapRange(current->next, limit);
241#endif
242 MSAN_ALLOCATED_UNINITIALIZED_MEMORY(
243 current->next,
244 static_cast<size_t>(reinterpret_cast<Address>(limit) -
245 reinterpret_cast<Address>(current->next)));
246#ifdef DEBUG
247 int after = v8_flags.check_handle_count ? NumberOfHandles(isolate) : 0;
248 DCHECK_LT(after - before, kCheckHandleThreshold);
249 DCHECK_LT(before, kCheckHandleThreshold);
250#endif
251}

253template <typename T, template <typename> typename HandleType>
254 requires(std::is_convertible_v<HandleType<T>, DirectHandle<T>>)
255HandleType<T> HandleScope::CloseAndEscape(HandleType<T> handle_value) {
256 HandleScopeData* current = isolate_->handle_scope_data();
257 Tagged<T> value = *handle_value;
258#ifdef V8_ENABLE_CHECKS
259 CHECK_EQ(scope_level_, isolate_->handle_scope_data()->level);
260#endif
261 // Throw away all handles in the current scope.
262 CloseScope(isolate_, prev_next_, prev_limit_);
263 // Allocate one handle in the parent scope.
264 DCHECK(current->level > current->sealed_level);
265 HandleType<T> result(value, isolate_);
266 // Reinitialize the current scope (so that it's ready
267 // to be used or closed again).
268 prev_next_ = current->next;
269 prev_limit_ = current->limit;
270 current->level++;
271 return result;
272}

274Address* HandleScope::CreateHandle(Isolate* isolate, Address value) {
275 DCHECK(AllowHandleAllocation::IsAllowed());
276#ifdef DEBUG
277 if (!AllowHandleUsageOnAllThreads::IsAllowed()) {
278 DCHECK(isolate->main_thread_local_heap()->IsRunning());
279 DCHECK_WITH_MSG(
280 isolate->thread_id() == ThreadId::Current(),
281 "main-thread handle can only be created on the main thread.");
282 }
283#endif
284 HandleScopeData* data = isolate->handle_scope_data();
285 Address* result = data->next;
286 if (V8_UNLIKELY(result == data->limit)) {
287 result = Extend(isolate);
288 }
289 // Update the current next field, set the value in the created handle,
290 // and return the result.
291 DCHECK_LT(reinterpret_cast<Address>(result),
292 reinterpret_cast<Address>(data->limit));
293 data->next = reinterpret_cast<Address*>(reinterpret_cast<Address>(result) +
294 sizeof(Address));
295 *result = value;
296 return result;
297}

299#ifdef DEBUG
300inline SealHandleScope::SealHandleScope(Isolate* isolate) : isolate_(isolate) {
301 // Make sure the current thread is allowed to create handles to begin with.
302 DCHECK(AllowHandleAllocation::IsAllowed());
303 HandleScopeData* current = isolate_->handle_scope_data();
304 // Shrink the current handle scope to make it impossible to do
305 // handle allocations without an explicit handle scope.
306 prev_limit_ = current->limit;
307 current->limit = current->next;
308 prev_sealed_level_ = current->sealed_level;
309 current->sealed_level = current->level;
310}
311
312inline SealHandleScope::~SealHandleScope() {
313 // Restore state in current handle scope to re-enable handle
314 // allocations.
315 HandleScopeData* current = isolate_->handle_scope_data();
316 DCHECK_EQ(current->next, current->limit);
317 current->limit = prev_limit_;
318 DCHECK_EQ(current->level, current->sealed_level);
319 current->sealed_level = prev_sealed_level_;
320}
