176// static
177void NodeProperties::ReplaceUses(Node* node, Node* value, Node* effect,
178 Node* success, Node* exception) {
179 // Requires distinguishing between value, effect and control edges.
180 for (Edge edge : node->use_edges()) {
181 if (IsControlEdge(edge)) {
182 if (edge.from()->opcode() == IrOpcode::kIfSuccess) {
183 DCHECK_NOT_NULL(success);
184 edge.UpdateTo(success);
185 } else if (edge.from()->opcode() == IrOpcode::kIfException) {
186 DCHECK_NOT_NULL(exception);
187 edge.UpdateTo(exception);
188 } else {
189 DCHECK_NOT_NULL(success);
190 edge.UpdateTo(success);
191 }
192 } else if (IsEffectEdge(edge)) {
193 DCHECK_NOT_NULL(effect);
194 edge.UpdateTo(effect);
195 } else {
196 DCHECK_NOT_NULL(value);
197 edge.UpdateTo(value);
198 }
199 }
200}

216 Node* unreachable_sentinel) {
217 Node* effect = NodeProperties::GetEffectInput(node);
218 while (effect->opcode() != IrOpcode::kCheckpoint) {

260// static
261void NodeProperties::CollectControlProjections(Node* node, Node** projections,
262 size_t projection_count) {
263#ifdef DEBUG
264 DCHECK_LE(static_cast<int>(projection_count), node->UseCount());
265 std::memset(projections, 0, sizeof(*projections) * projection_count);
266#endif
267 size_t if_value_index = 0;
268 for (Edge const edge : node->use_edges()) {
269 if (!IsControlEdge(edge)) continue;
270 Node* use = edge.from();
271 size_t index;
272 switch (use->opcode()) {
273 case IrOpcode::kIfTrue:
274 DCHECK_EQ(IrOpcode::kBranch, node->opcode());
275 index = 0;
276 break;
277 case IrOpcode::kIfFalse:
278 DCHECK_EQ(IrOpcode::kBranch, node->opcode());
279 index = 1;
280 break;
281 case IrOpcode::kIfSuccess:
282 DCHECK(!node->op()->HasProperty(Operator::kNoThrow));
283 index = 0;
284 break;
285 case IrOpcode::kIfException:
286 DCHECK(!node->op()->HasProperty(Operator::kNoThrow));
287 index = 1;
288 break;
289 case IrOpcode::kIfValue:
290 DCHECK_EQ(IrOpcode::kSwitch, node->opcode());
291 index = if_value_index++;
292 break;
293 case IrOpcode::kIfDefault:
294 DCHECK_EQ(IrOpcode::kSwitch, node->opcode());
295 index = projection_count - 1;
296 break;
297 default:
298 continue;
299 }
300 DCHECK_LT(if_value_index, projection_count);
301 DCHECK_LT(index, projection_count);
302 DCHECK_NULL(projections[index]);
303 projections[index] = use;
304 }
305#ifdef DEBUG
306 for (size_t index = 0; index < projection_count; ++index) {
307 DCHECK_NOT_NULL(projections[index]);
308 }
309#endif
310}

312// static
313MachineRepresentation NodeProperties::GetProjectionType(
314 Node const* projection) {
315 size_t index = ProjectionIndexOf(projection->op());
316 Node* input = projection->InputAt(0);
317 switch (input->opcode()) {
318 case IrOpcode::kInt32AddWithOverflow:
319 case IrOpcode::kInt32SubWithOverflow:
320 case IrOpcode::kInt32MulWithOverflow:
321 case IrOpcode::kInt32AbsWithOverflow:
322 CHECK_LE(index, static_cast<size_t>(1));
323 return index == 0 ? MachineRepresentation::kWord32
324 : MachineRepresentation::kBit;
325 case IrOpcode::kInt64AddWithOverflow:
326 case IrOpcode::kInt64SubWithOverflow:
327 case IrOpcode::kInt64MulWithOverflow:
328 case IrOpcode::kInt64AbsWithOverflow:
329 CHECK_LE(index, static_cast<size_t>(1));
330 return index == 0 ? MachineRepresentation::kWord64
331 : MachineRepresentation::kBit;
332 case IrOpcode::kTryTruncateFloat64ToInt32:
333 case IrOpcode::kTryTruncateFloat64ToUint32:
334 CHECK_LE(index, static_cast<size_t>(1));
335 return index == 0 ? MachineRepresentation::kWord32
336 : MachineRepresentation::kBit;
337 case IrOpcode::kTryTruncateFloat32ToInt64:
338 case IrOpcode::kTryTruncateFloat64ToInt64:
339 case IrOpcode::kTryTruncateFloat64ToUint64:
340 case IrOpcode::kTryTruncateFloat32ToUint64:
341 CHECK_LE(index, static_cast<size_t>(1));
342 return index == 0 ? MachineRepresentation::kWord64
343 : MachineRepresentation::kBit;
344 case IrOpcode::kCall: {
345 auto call_descriptor = CallDescriptorOf(input->op());
346 return call_descriptor->GetReturnType(index).representation();
347 }
348 case IrOpcode::kInt32PairAdd:
349 case IrOpcode::kInt32PairSub:
350 case IrOpcode::kWord32AtomicPairLoad:
351 case IrOpcode::kWord32AtomicPairAdd:
352 case IrOpcode::kWord32AtomicPairSub:
353 case IrOpcode::kWord32AtomicPairAnd:
354 case IrOpcode::kWord32AtomicPairOr:
355 case IrOpcode::kWord32AtomicPairXor:
356 case IrOpcode::kWord32AtomicPairExchange:
357 case IrOpcode::kWord32AtomicPairCompareExchange:
358 CHECK_LE(index, static_cast<size_t>(1));
359 return MachineRepresentation::kWord32;
360 default:
361 return MachineRepresentation::kNone;
362 }
363}

380// static
381OptionalMapRef NodeProperties::GetJSCreateMap(JSHeapBroker* broker,
382 Node* receiver) {
383 DCHECK(receiver->opcode() == IrOpcode::kJSCreate ||
384 receiver->opcode() == IrOpcode::kJSCreateArray);
385 HeapObjectMatcher mtarget(GetValueInput(receiver, 0));
386 HeapObjectMatcher mnewtarget(GetValueInput(receiver, 1));
387 if (mtarget.HasResolvedValue() && mnewtarget.HasResolvedValue() &&
388 mnewtarget.Ref(broker).IsJSFunction()) {
389 ObjectRef target = mtarget.Ref(broker);
390 JSFunctionRef newtarget = mnewtarget.Ref(broker).AsJSFunction();
391 if (newtarget.map(broker).has_prototype_slot() &&
392 newtarget.has_initial_map(broker)) {
393 MapRef initial_map = newtarget.initial_map(broker);
394 if (initial_map.GetConstructor(broker).equals(target)) {
395 DCHECK(target.AsJSFunction().map(broker).is_constructor());
396 DCHECK(newtarget.map(broker).is_constructor());
397 return initial_map;
398 }
399 }
400 }
401 return std::nullopt;
402}

404// static
405NodeProperties::InferMapsResult NodeProperties::InferMapsUnsafe(
406 JSHeapBroker* broker, Node* receiver, Effect effect,
407 ZoneRefSet<Map>* maps_out) {
408 HeapObjectMatcher m(receiver);
409 if (m.HasResolvedValue()) {
410 HeapObjectRef ref = m.Ref(broker);
411 // We don't use ICs for the Array.prototype and the Object.prototype
412 // because the runtime has to be able to intercept them properly, so
413 // we better make sure that TurboFan doesn't outsmart the system here
414 // by storing to elements of either prototype directly.
415 //
416 // TODO(bmeurer): This can be removed once the Array.prototype and
417 // Object.prototype have NO_ELEMENTS elements kind.
418 if (!ref.IsJSObject() ||
419 !broker->IsArrayOrObjectPrototype(ref.AsJSObject())) {
420 if (ref.map(broker).is_stable()) {
421 // The {receiver_map} is only reliable when we install a stability
422 // code dependency.
423 *maps_out = ZoneRefSet<Map>{ref.map(broker)};
424 return kUnreliableMaps;
425 }
426 }
427 }
428 InferMapsResult result = kReliableMaps;
429 while (true) {
430 switch (effect->opcode()) {
431 case IrOpcode::kMapGuard: {
432 Node* const object = GetValueInput(effect, 0);
433 if (IsSame(receiver, object)) {
434 *maps_out = MapGuardMapsOf(effect->op());
435 return result;
436 }
437 break;
438 }
439 case IrOpcode::kCheckMaps: {
440 Node* const object = GetValueInput(effect, 0);
441 if (IsSame(receiver, object)) {
442 *maps_out = CheckMapsParametersOf(effect->op()).maps();
443 return result;
444 }
445 break;
446 }
447 case IrOpcode::kTransitionElementsKindOrCheckMap: {
448 Node* const object = GetValueInput(effect, 0);
449 if (IsSame(receiver, object)) {
450 *maps_out = ZoneRefSet<Map>{
451 ElementsTransitionWithMultipleSourcesOf(effect->op()).target()};
452 return result;
453 }
454 break;
455 }
456 case IrOpcode::kJSCreate: {
457 if (IsSame(receiver, effect)) {
458 OptionalMapRef initial_map = GetJSCreateMap(broker, receiver);
459 if (initial_map.has_value()) {
460 *maps_out = ZoneRefSet<Map>{initial_map.value()};
461 return result;
462 }
463 // We reached the allocation of the {receiver}.
464 return kNoMaps;
465 }
466 result = kUnreliableMaps; // JSCreate can have side-effect.
467 break;
468 }
469 case IrOpcode::kJSCreatePromise: {
470 if (IsSame(receiver, effect)) {
471 *maps_out = ZoneRefSet<Map>{broker->target_native_context()
472 .promise_function(broker)
473 .initial_map(broker)};
474 return result;
475 }
476 break;
477 }
478 case IrOpcode::kStoreField: {
479 // We only care about StoreField of maps.
480 Node* const object = GetValueInput(effect, 0);
481 FieldAccess const& access = FieldAccessOf(effect->op());
482 if (access.base_is_tagged == kTaggedBase &&
483 access.offset == HeapObject::kMapOffset) {
484 if (IsSame(receiver, object)) {
485 Node* const value = GetValueInput(effect, 1);
486 HeapObjectMatcher m2(value);
487 if (m2.HasResolvedValue()) {
488 *maps_out = ZoneRefSet<Map>{m2.Ref(broker).AsMap()};
489 return result;
490 }
491 }
492 // Without alias analysis we cannot tell whether this
493 // StoreField[map] affects {receiver} or not.
494 result = kUnreliableMaps;
495 }
496 break;
497 }
498 case IrOpcode::kJSStoreMessage:
499 case IrOpcode::kJSStoreModule:
500 case IrOpcode::kStoreElement:
501 case IrOpcode::kStoreTypedElement: {
502 // These never change the map of objects.
503 break;
504 }
505 case IrOpcode::kFinishRegion: {
506 // FinishRegion renames the output of allocations, so we need
507 // to update the {receiver} that we are looking for, if the
508 // {receiver} matches the current {effect}.
509 if (IsSame(receiver, effect)) receiver = GetValueInput(effect, 0);
510 break;
511 }
512 case IrOpcode::kEffectPhi: {
513 Node* control = GetControlInput(effect);
514 if (control->opcode() != IrOpcode::kLoop) {
515 DCHECK(control->opcode() == IrOpcode::kDead ||
516 control->opcode() == IrOpcode::kMerge);
517 return kNoMaps;
518 }
519
520 // Continue search for receiver map outside the loop. Since operations
521 // inside the loop may change the map, the result is unreliable.
522 effect = GetEffectInput(effect, 0);
523 result = kUnreliableMaps;
524 continue;
525 }
526 default: {
527 DCHECK_EQ(1, effect->op()->EffectOutputCount());
528 if (effect->op()->EffectInputCount() != 1) {
529 // Didn't find any appropriate CheckMaps node.
530 return kNoMaps;
531 }
532 if (!effect->op()->HasProperty(Operator::kNoWrite)) {
533 // Without alias/escape analysis we cannot tell whether this
534 // {effect} affects {receiver} or not.
535 result = kUnreliableMaps;
536 }
537 break;
538 }
539 }
540
541 // Stop walking the effect chain once we hit the definition of
542 // the {receiver} along the {effect}s.
543 if (IsSame(receiver, effect)) return kNoMaps;
544
545 // Continue with the next {effect}.
546 DCHECK_EQ(1, effect->op()->EffectInputCount());
547 effect = NodeProperties::GetEffectInput(effect);
548 }
549}

565// static
566bool NodeProperties::CanBePrimitive(JSHeapBroker* broker, Node* receiver,
567 Effect effect) {
568 switch (receiver->opcode()) {
569#define CASE(Opcode) case IrOpcode::k##Opcode:
570 JS_CONSTRUCT_OP_LIST(CASE)
571 JS_CREATE_OP_LIST(CASE)
572#undef CASE
573 case IrOpcode::kCheckReceiver:
574 case IrOpcode::kConvertReceiver:
575 case IrOpcode::kJSGetSuperConstructor:
576 case IrOpcode::kJSToObject:
577 return false;
578 case IrOpcode::kHeapConstant: {
579 HeapObjectRef value = HeapObjectMatcher(receiver).Ref(broker);
580 return value.map(broker).IsPrimitiveMap();
581 }
582 default: {
583 MapInference inference(broker, receiver, effect);
584 return !inference.HaveMaps() ||
585 !inference.AllOfInstanceTypesAreJSReceiver();
586 }
587 }
588}

662// static
663bool NodeProperties::Equals(Node* a, Node* b) {
664 DCHECK_NOT_NULL(a);
665 DCHECK_NOT_NULL(b);
666 DCHECK_NOT_NULL(a->op());
667 DCHECK_NOT_NULL(b->op());
668 if (!a->op()->Equals(b->op())) return false;
669 if (a->InputCount() != b->InputCount()) return false;
670 Node::Inputs aInputs = a->inputs();
671 Node::Inputs bInputs = b->inputs();
672
673 auto aIt = aInputs.begin();
674 auto bIt = bInputs.begin();
675 auto aEnd = aInputs.end();
676
677 for (; aIt != aEnd; ++aIt, ++bIt) {
678 DCHECK_NOT_NULL(*aIt);
679 DCHECK_NOT_NULL(*bIt);
680 if ((*aIt)->id() != (*bIt)->id()) return false;
681 }
682 return true;
683}
