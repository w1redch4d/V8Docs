110 // Mask constant for checking if a name has a computed hash code and the type
111 // of information stored in the hash field. The least significant bit
112 // indicates whether the value can be used as a hash (i.e. different values
113 // imply different strings).
114 enum class HashFieldType : uint32_t {
115 kHash = 0b10,
116 kIntegerIndex = 0b00,
117 kForwardingIndex = 0b01,
118 kEmpty = 0b11
119 };
120
121 using HashFieldTypeBits = base::BitField<HashFieldType, 0, 2>;
122 using HashBits =
123 HashFieldTypeBits::Next<uint32_t, kBitsPerInt - HashFieldTypeBits::kSize>;
124
125 static constexpr int kHashNotComputedMask = 1;
126 // Value of empty hash field indicating that the hash is not computed.
127 static constexpr int kEmptyHashField =
128 HashFieldTypeBits::encode(HashFieldType::kEmpty);
129
130 // Empty hash and forwarding indices can not be used as hash.
131 static_assert((kEmptyHashField & kHashNotComputedMask) != 0);
132 static_assert((HashFieldTypeBits::encode(HashFieldType::kForwardingIndex) &
133 kHashNotComputedMask) != 0);
134
135 using IsInternalizedForwardingIndexBit = HashFieldTypeBits::Next<bool, 1>;
136 using IsExternalForwardingIndexBit =
137 IsInternalizedForwardingIndexBit::Next<bool, 1>;
138 using ForwardingIndexValueBits = IsExternalForwardingIndexBit::Next<
139 unsigned int, kBitsPerInt - HashFieldTypeBits::kSize -
140 IsInternalizedForwardingIndexBit::kSize -
141 IsExternalForwardingIndexBit::kSize>;
142
143 // Array index strings this short can keep their index in the hash field.
144 static const int kMaxCachedArrayIndexLength = 7;
145
146 // Maximum number of characters to consider when trying to convert a string
147 // value into an array index.
148 static const int kMaxArrayIndexSize = 10;
149 // Maximum number of characters in a string that can possibly be an
150 // "integer index" in the spec sense, i.e. a canonical representation of a
151 // number in the range up to MAX_SAFE_INTEGER. We parse these into a size_t,
152 // so the size of that type also factors in as a limit: 10 characters per
153 // 32 bits of size_t width.
154 static const int kMaxIntegerIndexSize =
155 std::min(16, int{10 * (sizeof(size_t) / 4)});
156
157 // For strings which are array indexes the hash value has the string length
158 // mixed into the hash, mainly to avoid a hash value of zero which would be
159 // the case for the string '0'. 24 bits are used for the array index value.
160 static const int kArrayIndexValueBits = 24;
161 static const int kArrayIndexLengthBits =
162 kBitsPerInt - kArrayIndexValueBits - HashFieldTypeBits::kSize;
163
164 static_assert(kArrayIndexLengthBits > 0);
165 static_assert(kMaxArrayIndexSize < (1 << kArrayIndexLengthBits));
166
167 using ArrayIndexValueBits =
168 HashFieldTypeBits::Next<unsigned int, kArrayIndexValueBits>;
169 using ArrayIndexLengthBits =
170 ArrayIndexValueBits::Next<unsigned int, kArrayIndexLengthBits>;
171
172 // Check that kMaxCachedArrayIndexLength + 1 is a power of two so we
173 // could use a mask to test if the length of string is less than or equal to
174 // kMaxCachedArrayIndexLength.
175 static_assert(base::bits::IsPowerOfTwo(kMaxCachedArrayIndexLength + 1),
176 "(kMaxCachedArrayIndexLength + 1) must be power of two");
177
178 // When any of these bits is set then the hash field does not contain a cached
179 // array index.
180 static_assert(HashFieldTypeBits::encode(HashFieldType::kIntegerIndex) == 0);
181 static const unsigned int kDoesNotContainCachedArrayIndexMask =
182 (~static_cast<unsigned>(kMaxCachedArrayIndexLength)
183 << ArrayIndexLengthBits::kShift) |
184 HashFieldTypeBits::kMask;
185
186 // When any of these bits is set then the hash field does not contain an
187 // integer or forwarding index.
188 static const unsigned int kDoesNotContainIntegerOrForwardingIndexMask = 0b10;
189 static_assert((HashFieldTypeBits::encode(HashFieldType::kIntegerIndex) &
190 kDoesNotContainIntegerOrForwardingIndexMask) == 0);
191 static_assert((HashFieldTypeBits::encode(HashFieldType::kForwardingIndex) &
192 kDoesNotContainIntegerOrForwardingIndexMask) == 0);
