33void MarkingBarrier::MarkValue(Tagged<HeapObject> host,
34 Tagged<HeapObject> value) {
35 if (HeapLayout::InReadOnlySpace(value)) return;
36
37 DCHECK(IsCurrentMarkingBarrier(host));
38 DCHECK(is_activated_ || shared_heap_worklists_.has_value());
39
40 // When shared heap isn't enabled all objects are local, we can just run the
41 // local marking barrier. Also from the point-of-view of the shared space
42 // isolate (= main isolate) also shared objects are considered local.
43 if (V8_UNLIKELY(uses_shared_heap_) && !is_shared_space_isolate_) {
44 // Check whether incremental marking is enabled for that object's space.
45 if (!MemoryChunk::FromHeapObject(host)->IsMarking()) {
46 return;
47 }
48
49 if (v8_flags.black_allocated_pages &&
50 HeapLayout::InBlackAllocatedPage(value)) {
51 return;
52 }
53
54 if (HeapLayout::InWritableSharedSpace(host)) {
55 // Invoking shared marking barrier when storing into shared objects.
56 MarkValueShared(value);
57 return;
58 } else if (HeapLayout::InWritableSharedSpace(value)) {
59 // No marking needed when storing shared objects in local objects.
60 return;
61 }
62 }
63
64 DCHECK_IMPLIES(HeapLayout::InWritableSharedSpace(host),
65 is_shared_space_isolate_);
66 DCHECK_IMPLIES(HeapLayout::InWritableSharedSpace(value),
67 is_shared_space_isolate_);
68
69 DCHECK(is_activated_);
70 MarkValueLocal(value);
71}

73void MarkingBarrier::MarkValueShared(Tagged<HeapObject> value) {
74 // Value is either in read-only space or shared heap.
75 DCHECK(HeapLayout::InAnySharedSpace(value));
76
77 // We should only reach this on client isolates (= worker isolates).
78 DCHECK(!is_shared_space_isolate_);
79 DCHECK(shared_heap_worklists_.has_value());
80
81 // Mark shared object and push it onto shared heap worklist.
82 if (marking_state_.TryMark(value)) {
83 shared_heap_worklists_->Push(value);
84 }
85}

87void MarkingBarrier::MarkValueLocal(Tagged<HeapObject> value) {
88 DCHECK(!HeapLayout::InReadOnlySpace(value));
89 if (is_minor()) {
90 // We do not need to insert into RememberedSet<OLD_TO_NEW> here because the
91 // C++ marking barrier already does this for us.
92 // TODO(v8:13012): Consider updating C++ barriers to respect
93 // POINTERS_TO_HERE_ARE_INTERESTING and POINTERS_FROM_HERE_ARE_INTERESTING
94 // page flags and make the following branch a DCHECK.
95 if (HeapLayout::InYoungGeneration(value)) {
96 MarkingHelper::TryMarkAndPush(
97 heap_, current_worklists_.get(), &marking_state_,
98 MarkingHelper::WorklistTarget::kRegular, value);
99 }
100 } else {
101 // At this point `ShouldMarkObject()` should always succeed here because
102 // value has gone through all the necessary filters. However, we do want to
103 // push to the right target worklist immediately.
104 const auto target_worklist = MarkingHelper::ShouldMarkObject(heap_, value);
105 if (!target_worklist) return;
106 MarkingHelper::TryMarkAndPush(heap_, current_worklists_.get(),
107 &marking_state_, target_worklist.value(),
108 value);
109 }
110}

112template <typename TSlot>
113inline void MarkingBarrier::MarkRange(Tagged<HeapObject> host, TSlot start,
114 TSlot end) {
115 auto* isolate = heap_->isolate();
116 const bool record_slots =
117 IsCompacting(host) &&
118 !MemoryChunk::FromHeapObject(host)->ShouldSkipEvacuationSlotRecording();
119 for (TSlot slot = start; slot < end; ++slot) {
120 typename TSlot::TObject object = slot.Relaxed_Load();
121 Tagged<HeapObject> heap_object;
122 // Mark both, weak and strong edges.
123 if (object.GetHeapObject(isolate, &heap_object)) {
124 MarkValue(host, heap_object);
125 if (record_slots) {
126 major_collector_->RecordSlot(host, HeapObjectSlot(slot), heap_object);
127 }
128 }
129 }
130}

132bool MarkingBarrier::IsCompacting(Tagged<HeapObject> object) const {
133 if (is_compacting_) {
134 DCHECK(is_major());
135 return true;
136 }
137
138 return shared_heap_worklists_.has_value() &&
139 HeapLayout::InWritableSharedSpace(object);
140}
