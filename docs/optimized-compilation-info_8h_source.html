196 template <typename T>
197 IndirectHandle<T> CanonicalHandle(Tagged<T> object, Isolate* isolate) {
198 DCHECK_NOT_NULL(canonical_handles_);
199 DCHECK(PersistentHandlesScope::IsActive(isolate));
200 auto find_result = canonical_handles_->FindOrInsert(object);
201 if (!find_result.already_exists) {
202 *find_result.entry = IndirectHandle<T>(object, isolate).location();
203 }
204 return IndirectHandle<T>(*find_result.entry);
205 }
