45// A code range is a virtual memory cage that may contain executable code. It
46// has the following layout.
47//
48// +---------+-----+----------------- ~~~ -+
49// | RW | ... | ... |
50// +---------+-----+------------------ ~~~ -+
51// ^ ^
52// base allocatable base
53//
54// <--------> <------------------------->
55// reserved allocatable region
56// <----------------------------------------->
57// CodeRange
58//
59// The start of the reservation may include reserved page with read-write access
60// as required by some platforms (Win64) followed by an unmapped region which
61// make allocatable base MemoryChunk::kAlignment-aligned. The cage's page
62// allocator explicitly marks the optional reserved page as occupied, so it's
63// excluded from further allocations.
64//
65// The following conditions hold:
66// 1) |reservation()->region()| == [base(), base() + size()[,
67// 2) if optional RW pages are not necessary, then |base| == |allocatable base|,
68// 3) both |base| and |allocatable base| are MemoryChunk::kAlignment-aligned.
69class CodeRange final : public VirtualMemoryCage {
70 public:
71 V8_EXPORT_PRIVATE ~CodeRange() override;
72
73 // Returns the size of the initial area of a code range, which is marked
74 // writable and reserved to contain unwind information.
75 static size_t GetWritableReservedAreaSize();

77 uint8_t* embedded_blob_code_copy() const {
78 // remap_embedded_builtins_mutex_ is designed to protect write contention to
79 // embedded_blob_code_copy_. It is safe to be read without taking the
80 // mutex. It is read to check if short builtins ought to be enabled because
81 // a shared CodeRange has already remapped builtins and to find where the
82 // instruction stream for a builtin is.
83 //
84 // For the first, this racing with an Isolate calling RemapEmbeddedBuiltins
85 // may result in disabling short builtins, which is not a correctness issue.
86 //
87 // For the second, this racing with an Isolate calling RemapEmbeddedBuiltins
88 // may result in an already running Isolate that did not have short builtins
89 // enabled (due to max old generation size) to switch over to using remapped
90 // builtins, which is also not a correctness issue as the remapped builtins
91 // are byte-equivalent.
92 //
93 // Both these scenarios should be rare. The initial Isolate is usually
94 // created by itself, i.e. without contention. Additionally, the first
95 // Isolate usually remaps builtins on machines with enough memory, not
96 // subsequent Isolates in the same process.
97 return embedded_blob_code_copy_.load(std::memory_order_acquire);
98 }
99
100 // Initialize the address space reservation for the code range. The immutable
101 // flag specifies if the reservation will live until the end of the process
102 // and can be sealed.
103 bool InitReservation(v8::PageAllocator* page_allocator, size_t requested,
104 bool immutable);
105
106 V8_EXPORT_PRIVATE void Free();
107
108 // Remap and copy the embedded builtins into this CodeRange. This method is
109 // idempotent and only performs the copy once. This property is so that this
110 // method can be used uniformly regardless of whether there is a single global
111 // pointer address space or multiple pointer cages. Returns the address of
112 // the copy.
113 //
114 // The builtins code region will be freed with the code range at tear down.
115 //
116 // When ENABLE_SLOW_DCHECKS is on, the contents of the embedded_blob_code are
117 // compared against the already copied version.
118 uint8_t* RemapEmbeddedBuiltins(Isolate* isolate,
119 const uint8_t* embedded_blob_code,
120 size_t embedded_blob_code_size);
121
122 private:
123 static base::AddressRegion GetPreferredRegion(size_t radius_in_megabytes,
124 size_t allocate_page_size);
125
126 // Used when short builtin calls are enabled, where embedded builtins are
127 // copied into the CodeRange so calls can be nearer.
128 std::atomic<uint8_t*> embedded_blob_code_copy_{nullptr};

132 base::SpinningMutex remap_embedded_builtins_mutex_;
133
134#if !defined(V8_OS_WIN) && defined(DEBUG)
135 bool immutable_ = false;
136#endif
137};
