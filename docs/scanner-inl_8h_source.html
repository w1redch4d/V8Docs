191// Character flags for the fast path of scanning a keyword or identifier token.
192enum class ScanFlags : uint8_t {
193 kTerminatesLiteral = 1 << 0,
194 // "Cannot" rather than "can" so that this flag can be ORed together across
195 // multiple characters.
196 kCannotBeKeyword = 1 << 1,
197 kCannotBeKeywordStart = 1 << 2,
198 kStringTerminator = 1 << 3,
199 kIdentifierNeedsSlowPath = 1 << 4,
200 kMultilineCommentCharacterNeedsSlowPath = 1 << 5,
201};
202constexpr uint8_t GetScanFlags(char c) {
203 return
204 // Keywords are all lowercase and only contain letters.
205 // Note that non-identifier characters do not set this flag, so
206 // that it plays well with kTerminatesLiteral.
207 (IsAsciiIdentifier(c) && !CanBeKeywordCharacter(c)
208 ? static_cast<uint8_t>(ScanFlags::kCannotBeKeyword)
209 : 0) |
210 (IsKeywordStart(c)
211 ? 0
212 : static_cast<uint8_t>(ScanFlags::kCannotBeKeywordStart)) |
213 // Anything that isn't an identifier character will terminate the
214 // literal, or at least terminates the literal fast path processing
215 // (like an escape).
216 (!IsAsciiIdentifier(c)
217 ? static_cast<uint8_t>(ScanFlags::kTerminatesLiteral)
218 : 0) |
219 // Possible string termination characters.
220 ((c == '\'' || c == '"' || c == '\n' || c == '\r' || c == '\\')
221 ? static_cast<uint8_t>(ScanFlags::kStringTerminator)
222 : 0) |
223 // Escapes are processed on the slow path.
224 (c == '\\' ? static_cast<uint8_t>(ScanFlags::kIdentifierNeedsSlowPath)
225 : 0) |
226 // Newlines and * are interesting characters for multiline comment
227 // scanning.
228 (c == '\n' || c == '\r' || c == '*'
229 ? static_cast<uint8_t>(
230 ScanFlags::kMultilineCommentCharacterNeedsSlowPath)
231 : 0);
232}
233inline bool TerminatesLiteral(uint8_t scan_flags) {
234 return (scan_flags & static_cast<uint8_t>(ScanFlags::kTerminatesLiteral));
235}
236inline bool CanBeKeyword(uint8_t scan_flags) {
237 return !(scan_flags & static_cast<uint8_t>(ScanFlags::kCannotBeKeyword));
238}
239inline bool IdentifierNeedsSlowPath(uint8_t scan_flags) {
240 return (scan_flags &
241 static_cast<uint8_t>(ScanFlags::kIdentifierNeedsSlowPath));
242}
243inline bool MultilineCommentCharacterNeedsSlowPath(uint8_t scan_flags) {
244 return (scan_flags & static_cast<uint8_t>(
245 ScanFlags::kMultilineCommentCharacterNeedsSlowPath));
246}
247inline bool MayTerminateString(uint8_t scan_flags) {
248 return (scan_flags & static_cast<uint8_t>(ScanFlags::kStringTerminator));
249}
250// Table of precomputed scan flags for the 128 ASCII characters, for branchless
251// flag calculation during the scan.
252static constexpr const uint8_t character_scan_flags[128] = {
253#define CALL_GET_SCAN_FLAGS(N) GetScanFlags(N),
254 INT_0_TO_127_LIST(CALL_GET_SCAN_FLAGS)
255#undef CALL_GET_SCAN_FLAGS
256};
257
258inline bool CharCanBeKeyword(base::uc32 c) {
259 return static_cast<uint32_t>(c) < arraysize(character_scan_flags) &&
260 CanBeKeyword(character_scan_flags[c]);
261}

263V8_INLINE Token::Value Scanner::ScanIdentifierOrKeywordInner() {
264 DCHECK(IsIdentifierStart(c0_));
265 bool escaped = false;
266 bool can_be_keyword = true;
267
268 static_assert(arraysize(character_scan_flags) == kMaxAscii + 1);
269 if (V8_LIKELY(static_cast<uint32_t>(c0_) <= kMaxAscii)) {
270 if (V8_LIKELY(c0_ != '\\')) {
271 uint8_t scan_flags = character_scan_flags[c0_];
272 DCHECK(!TerminatesLiteral(scan_flags));
273 static_assert(static_cast<uint8_t>(ScanFlags::kCannotBeKeywordStart) ==
274 static_cast<uint8_t>(ScanFlags::kCannotBeKeyword) << 1);
275 scan_flags >>= 1;
276 // Make sure the shifting above doesn't set IdentifierNeedsSlowPath.
277 // Otherwise we'll fall into the slow path after scanning the identifier.
278 DCHECK(!IdentifierNeedsSlowPath(scan_flags));
279 AddLiteralChar(static_cast<char>(c0_));
280 AdvanceUntil([this, &scan_flags](base::uc32 c0) {
281 if (V8_UNLIKELY(static_cast<uint32_t>(c0) > kMaxAscii)) {
282 // A non-ascii character means we need to drop through to the slow
283 // path.
284 // TODO(leszeks): This would be most efficient as a goto to the slow
285 // path, check codegen and maybe use a bool instead.
286 scan_flags |=
287 static_cast<uint8_t>(ScanFlags::kIdentifierNeedsSlowPath);
288 return true;
289 }
290 uint8_t char_flags = character_scan_flags[c0];
291 scan_flags |= char_flags;
292 if (TerminatesLiteral(char_flags)) {
293 return true;
294 } else {
295 AddLiteralChar(static_cast<char>(c0));
296 return false;
297 }
298 });
299
300 if (V8_LIKELY(!IdentifierNeedsSlowPath(scan_flags))) {
301 if (!CanBeKeyword(scan_flags)) return Token::kIdentifier;
302 // Could be a keyword or identifier.
303 base::Vector<const uint8_t> chars =
304 next().literal_chars.one_byte_literal();
305 return KeywordOrIdentifierToken(chars.begin(), chars.length());
306 }
307
308 can_be_keyword = CanBeKeyword(scan_flags);
309 } else {
310 // Special case for escapes at the start of an identifier.
311 escaped = true;
312 base::uc32 c = ScanIdentifierUnicodeEscape();
313 DCHECK(!IsIdentifierStart(Invalid()));
314 if (c == '\\' || !IsIdentifierStart(c)) {
315 return Token::kIllegal;
316 }
317 AddLiteralChar(c);
318 can_be_keyword = CharCanBeKeyword(c);
319 }
320 }
321
322 return ScanIdentifierOrKeywordInnerSlow(escaped, can_be_keyword);
323}

325V8_INLINE Token::Value Scanner::SkipWhiteSpace() {
326 if (!IsWhiteSpaceOrLineTerminator(c0_)) return Token::kIllegal;
327
328 if (!next().after_line_terminator && unibrow::IsLineTerminator(c0_)) {
329 next().after_line_terminator = true;
330 }
331
332 // Advance as long as character is a WhiteSpace or LineTerminator.
333 base::uc32 hint = ' ';
334 AdvanceUntil([this, &hint](base::uc32 c0) {
335 if (V8_LIKELY(c0 == hint)) return false;
336 if (IsWhiteSpaceOrLineTerminator(c0)) {
337 if (!next().after_line_terminator && unibrow::IsLineTerminator(c0)) {
338 next().after_line_terminator = true;
339 }
340 hint = c0;
341 return false;
342 }
343 return true;
344 });
345
346 return Token::kWhitespace;
347}

349V8_INLINE Token::Value Scanner::ScanSingleToken() {
350 Token::Value token;
351 do {
352 next().location.beg_pos = source_pos();
353
354 if (V8_LIKELY(static_cast<unsigned>(c0_) <= kMaxAscii)) {
355 token = one_char_tokens[c0_];
356
357 switch (token) {
358 case Token::kLeftParen:
359 case Token::kRightParen:
360 case Token::kLeftBrace:
361 case Token::kRightBrace:
362 case Token::kLeftBracket:
363 case Token::kRightBracket:
364 case Token::kColon:
365 case Token::kSemicolon:
366 case Token::kComma:
367 case Token::kBitNot:
368 case Token::kIllegal:
369 // One character tokens.
370 return Select(token);
371
372 case Token::kConditional:
373 // ? ?. ?? ??=
374 Advance();
375 if (c0_ == '.') {
376 Advance();
377 if (!IsDecimalDigit(c0_)) return Token::kQuestionPeriod;
378 PushBack('.');
379 } else if (c0_ == '?') {
380 return Select('=', Token::kAssignNullish, Token::kNullish);
381 }
382 return Token::kConditional;
383
384 case Token::kString:
385 return ScanString();
386
387 case Token::kLessThan:
388 // < <= << <<= <!--
389 Advance();
390 if (c0_ == '=') return Select(Token::kLessThanEq);
391 if (c0_ == '<') return Select('=', Token::kAssignShl, Token::kShl);
392 if (c0_ == '!') {
393 token = ScanHtmlComment();
394 continue;
395 }
396 return Token::kLessThan;
397
398 case Token::kGreaterThan:
399 // > >= >> >>= >>> >>>=
400 Advance();
401 if (c0_ == '=') return Select(Token::kGreaterThanEq);
402 if (c0_ == '>') {
403 // >> >>= >>> >>>=
404 Advance();
405 if (c0_ == '=') return Select(Token::kAssignSar);
406 if (c0_ == '>') return Select('=', Token::kAssignShr, Token::kShr);
407 return Token::kSar;
408 }
409 return Token::kGreaterThan;
410
411 case Token::kAssign:
412 // = == === =>
413 Advance();
414 if (c0_ == '=') return Select('=', Token::kEqStrict, Token::kEq);
415 if (c0_ == '>') return Select(Token::kArrow);
416 return Token::kAssign;
417
418 case Token::kNot:
419 // ! != !==
420 Advance();
421 if (c0_ == '=')
422 return Select('=', Token::kNotEqStrict, Token::kNotEq);
423 return Token::kNot;
424
425 case Token::kAdd:
426 // + ++ +=
427 Advance();
428 if (c0_ == '+') return Select(Token::kInc);
429 if (c0_ == '=') return Select(Token::kAssignAdd);
430 return Token::kAdd;
431
432 case Token::kSub:
433 // - -- --> -=
434 Advance();
435 if (c0_ == '-') {
436 Advance();
437 if (c0_ == '>' && next().after_line_terminator) {
438 // For compatibility with SpiderMonkey, we skip lines that
439 // start with an HTML comment end '-->'.
440 token = SkipSingleHTMLComment();
441 continue;
442 }
443 return Token::kDec;
444 }
445 if (c0_ == '=') return Select(Token::kAssignSub);
446 return Token::kSub;
447
448 case Token::kMul:
449 // * *=
450 Advance();
451 if (c0_ == '*') return Select('=', Token::kAssignExp, Token::kExp);
452 if (c0_ == '=') return Select(Token::kAssignMul);
453 return Token::kMul;
454
455 case Token::kMod:
456 // % %=
457 return Select('=', Token::kAssignMod, Token::kMod);
458
459 case Token::kDiv:
460 // / // /* /=
461 Advance();
462 if (c0_ == '/') {
463 base::uc32 c = Peek();
464 if (c == '#' || c == '@') {
465 Advance();
466 Advance();
467 token = SkipMagicComment(c);
468 continue;
469 }
470 token = SkipSingleLineComment();
471 continue;
472 }
473 if (c0_ == '*') {
474 token = SkipMultiLineComment();
475 continue;
476 }
477 if (c0_ == '=') return Select(Token::kAssignDiv);
478 return Token::kDiv;
479
480 case Token::kBitAnd:
481 // & && &= &&=
482 Advance();
483 if (c0_ == '&') return Select('=', Token::kAssignAnd, Token::kAnd);
484 if (c0_ == '=') return Select(Token::kAssignBitAnd);
485 return Token::kBitAnd;
486
487 case Token::kBitOr:
488 // | || |= ||=
489 Advance();
490 if (c0_ == '|') return Select('=', Token::kAssignOr, Token::kOr);
491 if (c0_ == '=') return Select(Token::kAssignBitOr);
492 return Token::kBitOr;
493
494 case Token::kBitXor:
495 // ^ ^=
496 return Select('=', Token::kAssignBitXor, Token::kBitXor);
497
498 case Token::kPeriod:
499 // . Number
500 Advance();
501 if (IsDecimalDigit(c0_)) return ScanNumber(true);
502 if (c0_ == '.') {
503 if (Peek() == '.') {
504 Advance();
505 Advance();
506 return Token::kEllipsis;
507 }
508 }
509 return Token::kPeriod;
510
511 case Token::kTemplateSpan:
512 Advance();
513 return ScanTemplateSpan();
514
515 case Token::kPrivateName:
516 if (source_pos() == 0 && Peek() == '!') {
517 token = SkipSingleLineComment();
518 continue;
519 }
520 return ScanPrivateName();
521
522 case Token::kWhitespace:
523 token = SkipWhiteSpace();
524 continue;
525
526 case Token::kNumber:
527 return ScanNumber(false);
528
529 case Token::kIdentifier:
530 return ScanIdentifierOrKeyword();
531
532 default:
533 UNREACHABLE();
534 }
535 }
536
537 if (IsIdentifierStart(c0_) ||
538 (CombineSurrogatePair() && IsIdentifierStart(c0_))) {
539 return ScanIdentifierOrKeyword();
540 }
541 if (c0_ == kEndOfInput) {
542 return source_->has_parser_error() ? Token::kIllegal : Token::kEos;
543 }
544 token = SkipWhiteSpace();
545
546 // Continue scanning for tokens as long as we're just skipping whitespace.
547 } while (token == Token::kWhitespace);
548
549 return token;
550}
