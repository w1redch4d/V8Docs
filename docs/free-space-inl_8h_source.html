35Tagged<FreeSpace> FreeSpace::next() const {
36 DCHECK(IsValid());
37#ifdef V8_EXTERNAL_CODE_SPACE
38 intptr_t diff_to_next =
39 static_cast<intptr_t>(TaggedField<Smi, kNextOffset>::load(*this).value());
40 if (diff_to_next == 0) {
41 return FreeSpace();
42 }
43 Address next_ptr = ptr() + diff_to_next * kObjectAlignment;
44 return UncheckedCast<FreeSpace>(Tagged<Object>(next_ptr));
45#else
46 return UncheckedCast<FreeSpace>(
47 TaggedField<Object, kNextOffset>::load(*this));
48#endif // V8_EXTERNAL_CODE_SPACE
49}

51void FreeSpace::SetNext(const WritableFreeSpace& writable_free_space,
52 Tagged<FreeSpace> next) {
53 DCHECK(IsValid());
54
55#ifdef V8_EXTERNAL_CODE_SPACE
56 if (next.is_null()) {
57 writable_free_space.WriteHeaderSlot<Smi, kNextOffset>(Smi::zero(),
58 kRelaxedStore);
59 return;
60 }
61 intptr_t diff_to_next = next.ptr() - ptr();
62 DCHECK(IsAligned(diff_to_next, kObjectAlignment));
63 writable_free_space.WriteHeaderSlot<Smi, kNextOffset>(
64 Smi::FromIntptr(diff_to_next / kObjectAlignment), kRelaxedStore);
65#else
66 writable_free_space.WriteHeaderSlot<Object, kNextOffset>(next, kRelaxedStore);
67#endif // V8_EXTERNAL_CODE_SPACE
68}
