51// WARNING: This serialization format MUST remain backward compatible!
52//
53// This format is used by APIs to persist values to disk, e.g. IndexedDB.
54//
55// Backward compatibility means that when the format changes, deserializing
56// valid values in the older format must behave identically as before the
57// change. To maintain compatibility, either a format change does not affect the
58// deserializing behavior of valid values in the older format, or the
59// kLatestVersion constant is bumped.
60//
61// Adding a new tag is backwards compatible because no valid serialized value in
62// older formats would contain the new object tag.
63//
64// On the other hand, changing the format of a particular tag is backwards
65// incompatible and the version must be bumped. For example, a JSArrayBufferView
66// tag prior to version 14 was followed by the sub-tag, the byte offset, and the
67// byte length. Starting with version 14, a JSArrayBufferView tag is followed by
68// the sub-tag, the byte offset, the byte length, and flags. Due the addition of
69// flags, older valid serialized values for JSArrayBufferViews would be
70// misinterpreted by newer deserializers. This requires the version to be bumped
71// and the deserializer to handle both the old and new formats depending on the
72// version.
73
74// Version 9: (imported from Blink)
75// Version 10: one-byte (Latin-1) strings
76// Version 11: properly separate undefined from the hole in arrays
77// Version 12: regexp and string objects share normal string encoding
78// Version 13: host objects have an explicit tag (rather than handling all
79// unknown tags)
80// Version 14: flags for JSArrayBufferViews
81// Version 15: support for shared objects with an explicit tag
82//
83// WARNING: Increasing this value is a change which cannot safely be rolled
84// back without breaking compatibility with data stored on disk. It is
85// strongly recommended that you do not make such changes near a release
86// milestone branch point.
87//
88// Recent changes are routinely reverted in preparation for branch, and this
89// has been the cause of at least one bug in the past.
90static const uint32_t kLatestVersion = 15;
91static_assert(kLatestVersion == v8::CurrentValueSerializerFormatVersion(),
92 "Exported format version must match latest version.");

105template <typename T>
106static size_t BytesNeededForVarint(T value) {
107 static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
108 "Only unsigned integer types can be written as varints.");
109 size_t result = 0;
110 do {
111 result++;
112 value >>= 7;
113 } while (value);
114 return result;
115}

327template <typename T>
328void ValueSerializer::WriteVarint(T value) {
329 // Writes an unsigned integer as a base-128 varint.
330 // The number is written, 7 bits at a time, from the least significant to the
331 // most significant 7 bits. Each byte, except the last, has the MSB set.
332 // See also https://developers.google.com/protocol-buffers/docs/encoding
333 static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
334 "Only unsigned integer types can be written as varints.");
335 uint8_t stack_buffer[sizeof(T) * 8 / 7 + 1];
336 uint8_t* next_byte = &stack_buffer[0];
337 do {
338 *next_byte = (value & 0x7F) | 0x80;
339 next_byte++;
340 value >>= 7;
341 } while (value);
342 *(next_byte - 1) &= 0x7F;
343 WriteRawBytes(stack_buffer, next_byte - stack_buffer);
344}

346template <typename T>
347void ValueSerializer::WriteZigZag(T value) {
348 // Writes a signed integer as a varint using ZigZag encoding (i.e. 0 is
349 // encoded as 0, -1 as 1, 1 as 2, -2 as 3, and so on).
350 // See also https://developers.google.com/protocol-buffers/docs/encoding
351 // Note that this implementation relies on the right shift being arithmetic.
352 static_assert(std::is_integral_v<T> && std::is_signed_v<T>,
353 "Only signed integer types can be written as zigzag.");
354 using UnsignedT = std::make_unsigned_t<T>;
355 WriteVarint((static_cast<UnsignedT>(value) << 1) ^
356 (value >> (8 * sizeof(T) - 1)));
357}

395Maybe<uint8_t*> ValueSerializer::ReserveRawBytes(size_t bytes) {
396 size_t old_size = buffer_size_;
397 size_t new_size = old_size + bytes;
398 if (V8_UNLIKELY(new_size > buffer_capacity_)) {
399 bool ok;
400 if (!ExpandBuffer(new_size).To(&ok)) {
401 return Nothing<uint8_t*>();
402 }
403 }
404 buffer_size_ = new_size;
405 return Just(&buffer_[old_size]);
406}

408Maybe<bool> ValueSerializer::ExpandBuffer(size_t required_capacity) {
409 DCHECK_GT(required_capacity, buffer_capacity_);
410 size_t requested_capacity =
411 std::max(required_capacity, buffer_capacity_ * 2) + 64;
412 size_t provided_capacity = 0;
413 void* new_buffer = nullptr;
414 if (delegate_) {
415 new_buffer = delegate_->ReallocateBufferMemory(buffer_, requested_capacity,
416 &provided_capacity);
417 } else {
418 new_buffer = base::Realloc(buffer_, requested_capacity);
419 provided_capacity = requested_capacity;
420 }
421 if (new_buffer) {
422 DCHECK(provided_capacity >= requested_capacity);
423 buffer_ = reinterpret_cast<uint8_t*>(new_buffer);
424 buffer_capacity_ = provided_capacity;
425 return Just(true);
426 } else {
427 out_of_memory_ = true;
428 return Nothing<bool>();
429 }
430}

462Maybe<bool> ValueSerializer::WriteObject(Handle<Object> object) {
463 // There is no sense in trying to proceed if we've previously run out of
464 // memory. Bail immediately, as this likely implies that some write has
465 // previously failed and so the buffer is corrupt.
466 if (V8_UNLIKELY(out_of_memory_)) return ThrowIfOutOfMemory();
467
468 if (IsSmi(*object)) {
469 WriteSmi(Cast<Smi>(*object));
470 return ThrowIfOutOfMemory();
471 }
472
473 DCHECK(IsHeapObject(*object));
474 InstanceType instance_type =
475 Cast<HeapObject>(*object)->map(isolate_)->instance_type();
476 switch (instance_type) {
477 case ODDBALL_TYPE:
478 WriteOddball(Cast<Oddball>(*object));
479 return ThrowIfOutOfMemory();
480 case HEAP_NUMBER_TYPE:
481 WriteHeapNumber(Cast<HeapNumber>(*object));
482 return ThrowIfOutOfMemory();
483 case BIGINT_TYPE:
484 WriteBigInt(Cast<BigInt>(*object));
485 return ThrowIfOutOfMemory();
486 case JS_TYPED_ARRAY_TYPE:
487 case JS_DATA_VIEW_TYPE:
488 case JS_RAB_GSAB_DATA_VIEW_TYPE: {
489 // Despite being JSReceivers, these have their wrapped buffer serialized
490 // first. That makes this logic a little quirky, because it needs to
491 // happen before we assign object IDs.
492 // TODO(jbroman): It may be possible to avoid materializing a typed
493 // array's buffer here.
494 Handle<JSArrayBufferView> view = Cast<JSArrayBufferView>(object);
495 if (!id_map_.Find(view) && !treat_array_buffer_views_as_host_objects_) {
496 Handle<JSArrayBuffer> buffer(
497 InstanceTypeChecker::IsJSTypedArray(instance_type)
498 ? Cast<JSTypedArray>(view)->GetBuffer()
499 : handle(Cast<JSArrayBuffer>(view->buffer()), isolate_));
500 if (!WriteJSReceiver(buffer).FromMaybe(false)) return Nothing<bool>();
501 }
502 return WriteJSReceiver(view);
503 }
504 default:
505 if (InstanceTypeChecker::IsString(instance_type)) {
506 WriteString(Cast<String>(object));
507 return ThrowIfOutOfMemory();
508 } else if (InstanceTypeChecker::IsJSReceiver(instance_type)) {
509 return WriteJSReceiver(Cast<JSReceiver>(object));
510 } else {
511 return ThrowDataCloneError(MessageTemplate::kDataCloneError, object);
512 }
513 }
514}

575Maybe<bool> ValueSerializer::WriteJSReceiver(Handle<JSReceiver> receiver) {
576 // If the object has already been serialized, just write its ID.
577 auto find_result = id_map_.FindOrInsert(receiver);
578 if (find_result.already_exists) {
579 WriteTag(SerializationTag::kObjectReference);
580 WriteVarint(*find_result.entry - 1);
581 return ThrowIfOutOfMemory();
582 }
583
584 // Otherwise, allocate an ID for it.
585 uint32_t id = next_id_++;
586 *find_result.entry = id + 1;
587
588 // Eliminate callable and exotic objects, which should not be serialized.
589 InstanceType instance_type = receiver->map()->instance_type();
590 if (IsCallable(*receiver) || (IsSpecialReceiverInstanceType(instance_type) &&
591 instance_type != JS_SPECIAL_API_OBJECT_TYPE)) {
592 return ThrowDataCloneError(MessageTemplate::kDataCloneError, receiver);
593 }
594
595 // If we are at the end of the stack, abort. This function may recurse.
596 STACK_CHECK(isolate_, Nothing<bool>());
597
598 HandleScope scope(isolate_);
599 switch (instance_type) {
600 case JS_ARRAY_TYPE:
601 return WriteJSArray(Cast<JSArray>(receiver));
602 case JS_ARRAY_ITERATOR_PROTOTYPE_TYPE:
603 case JS_ITERATOR_PROTOTYPE_TYPE:
604 case JS_MAP_ITERATOR_PROTOTYPE_TYPE:
605 case JS_OBJECT_PROTOTYPE_TYPE:
606 case JS_OBJECT_TYPE:
607 case JS_PROMISE_PROTOTYPE_TYPE:
608 case JS_REG_EXP_PROTOTYPE_TYPE:
609 case JS_SET_ITERATOR_PROTOTYPE_TYPE:
610 case JS_SET_PROTOTYPE_TYPE:
611 case JS_STRING_ITERATOR_PROTOTYPE_TYPE:
612 case JS_TYPED_ARRAY_PROTOTYPE_TYPE:
613 case JS_API_OBJECT_TYPE: {
614 Handle<JSObject> js_object = Cast<JSObject>(receiver);
615 Maybe<bool> is_host_object = IsHostObject(js_object);
616 if (is_host_object.IsNothing()) {
617 return is_host_object;
618 }
619 if (is_host_object.FromJust()) {
620 return WriteHostObject(js_object);
621 } else {
622 return WriteJSObject(js_object);
623 }
624 }
625 case JS_SPECIAL_API_OBJECT_TYPE:
626 return WriteHostObject(Cast<JSObject>(receiver));
627 case JS_DATE_TYPE:
628 WriteJSDate(Cast<JSDate>(*receiver));
629 return ThrowIfOutOfMemory();
630 case JS_PRIMITIVE_WRAPPER_TYPE:
631 return WriteJSPrimitiveWrapper(Cast<JSPrimitiveWrapper>(receiver));
632 case JS_REG_EXP_TYPE:
633 WriteJSRegExp(Cast<JSRegExp>(receiver));
634 return ThrowIfOutOfMemory();
635 case JS_MAP_TYPE:
636 return WriteJSMap(Cast<JSMap>(receiver));
637 case JS_SET_TYPE:
638 return WriteJSSet(Cast<JSSet>(receiver));
639 case JS_ARRAY_BUFFER_TYPE:
640 return WriteJSArrayBuffer(Cast<JSArrayBuffer>(receiver));
641 case JS_TYPED_ARRAY_TYPE:
642 case JS_DATA_VIEW_TYPE:
643 case JS_RAB_GSAB_DATA_VIEW_TYPE:
644 return WriteJSArrayBufferView(Cast<JSArrayBufferView>(*receiver));
645 case JS_ERROR_TYPE:
646 return WriteJSError(Cast<JSObject>(receiver));
647 case JS_SHARED_ARRAY_TYPE:
648 return WriteJSSharedArray(Cast<JSSharedArray>(receiver));
649 case JS_SHARED_STRUCT_TYPE:
650 return WriteJSSharedStruct(Cast<JSSharedStruct>(receiver));
651 case JS_ATOMICS_MUTEX_TYPE:
652 case JS_ATOMICS_CONDITION_TYPE:
653 return WriteSharedObject(receiver);
654#if V8_ENABLE_WEBASSEMBLY
655 case WASM_MODULE_OBJECT_TYPE:
656 return WriteWasmModule(Cast<WasmModuleObject>(receiver));
657 case WASM_MEMORY_OBJECT_TYPE:
658 return WriteWasmMemory(Cast<WasmMemoryObject>(receiver));
659#endif // V8_ENABLE_WEBASSEMBLY
660 default:
661 break;
662 }
663
664 return ThrowDataCloneError(MessageTemplate::kDataCloneError, receiver);
665}

667Maybe<bool> ValueSerializer::WriteJSObject(DirectHandle<JSObject> object) {
668 DCHECK(!IsCustomElementsReceiverMap(object->map()));
669 const bool can_serialize_fast =
670 object->HasFastProperties(isolate_) && object->elements()->length() == 0;
671 if (!can_serialize_fast) return WriteJSObjectSlow(object);
672
673 DirectHandle<Map> map(object->map(), isolate_);
674 WriteTag(SerializationTag::kBeginJSObject);
675
676 // Write out fast properties as long as they are only data properties and the
677 // map doesn't change.
678 uint32_t properties_written = 0;
679 bool map_changed = false;
680 for (InternalIndex i : map->IterateOwnDescriptors()) {
681 Handle<Name> key(map->instance_descriptors(isolate_)->GetKey(i), isolate_);
682 if (!IsString(*key, isolate_)) continue;
683 PropertyDetails details =
684 map->instance_descriptors(isolate_)->GetDetails(i);
685 if (details.IsDontEnum()) continue;
686
687 Handle<Object> value;
688 if (V8_LIKELY(!map_changed)) map_changed = *map != object->map();
689 if (V8_LIKELY(!map_changed &&
690 details.location() == PropertyLocation::kField)) {
691 DCHECK_EQ(PropertyKind::kData, details.kind());
692 FieldIndex field_index = FieldIndex::ForDetails(*map, details);
693 value = handle(object->RawFastPropertyAt(field_index), isolate_);
694 } else {
695 // This logic should essentially match WriteJSObjectPropertiesSlow.
696 // If the property is no longer found, do not serialize it.
697 // This could happen if a getter deleted the property.
698 LookupIterator it(isolate_, object, key, LookupIterator::OWN);
699 if (!it.IsFound()) continue;
700 if (!Object::GetProperty(&it).ToHandle(&value)) return Nothing<bool>();
701 }
702
703 if (!WriteObject(key).FromMaybe(false) ||
704 !WriteObject(value).FromMaybe(false)) {
705 return Nothing<bool>();
706 }
707 properties_written++;
708 }
709
710 WriteTag(SerializationTag::kEndJSObject);
711 WriteVarint<uint32_t>(properties_written);
712 return ThrowIfOutOfMemory();
713}

715Maybe<bool> ValueSerializer::WriteJSObjectSlow(DirectHandle<JSObject> object) {
716 WriteTag(SerializationTag::kBeginJSObject);
717 DirectHandle<FixedArray> keys;
718 uint32_t properties_written = 0;
719 if (!KeyAccumulator::GetKeys(isolate_, object, KeyCollectionMode::kOwnOnly,
720 ENUMERABLE_STRINGS)
721 .ToHandle(&keys) ||
722 !WriteJSObjectPropertiesSlow(object, keys).To(&properties_written)) {
723 return Nothing<bool>();
724 }
725 WriteTag(SerializationTag::kEndJSObject);
726 WriteVarint<uint32_t>(properties_written);
727 return ThrowIfOutOfMemory();
728}

730Maybe<bool> ValueSerializer::WriteJSArray(DirectHandle<JSArray> array) {
731 PtrComprCageBase cage_base(isolate_);
732 uint32_t length = 0;
733 bool valid_length = Object::ToArrayLength(array->length(), &length);
734 DCHECK(valid_length);
735 USE(valid_length);
736
737 // To keep things simple, for now we decide between dense and sparse
738 // serialization based on elements kind. A more principled heuristic could
739 // count the elements, but would need to take care to note which indices
740 // existed (as only indices which were enumerable own properties at this point
741 // should be serialized).
742 const bool should_serialize_densely =
743 array->HasFastElements(cage_base) && !array->HasHoleyElements(cage_base);
744
745 if (should_serialize_densely) {
746 DCHECK_LE(length, static_cast<uint32_t>(FixedArray::kMaxLength));
747 WriteTag(SerializationTag::kBeginDenseJSArray);
748 WriteVarint<uint32_t>(length);
749 uint32_t i = 0;
750
751 // Fast paths. Note that PACKED_ELEMENTS in particular can bail due to the
752 // structure of the elements changing.
753 switch (array->GetElementsKind(cage_base)) {
754 case PACKED_SMI_ELEMENTS: {
755 DisallowGarbageCollection no_gc;
756 Tagged<FixedArray> elements = Cast<FixedArray>(array->elements());
757 for (i = 0; i < length; i++) {
758 WriteSmi(Cast<Smi>(elements->get(i)));
759 }
760 break;
761 }
762 case PACKED_DOUBLE_ELEMENTS: {
763 // Elements are empty_fixed_array, not a FixedDoubleArray, if the array
764 // is empty. No elements to encode in this case anyhow.
765 if (length == 0) break;
766 DisallowGarbageCollection no_gc;
767 Tagged<FixedDoubleArray> elements =
768 Cast<FixedDoubleArray>(array->elements());
769 for (i = 0; i < length; i++) {
770 WriteTag(SerializationTag::kDouble);
771 WriteDouble(elements->get_scalar(i));
772 }
773 break;
774 }
775 case PACKED_ELEMENTS: {
776 DirectHandle<Object> old_length(array->length(cage_base), isolate_);
777 for (; i < length; i++) {
778 if (array->length(cage_base) != *old_length ||
779 array->GetElementsKind(cage_base) != PACKED_ELEMENTS) {
780 // Fall back to slow path.
781 break;
782 }
783 Handle<Object> element(Cast<FixedArray>(array->elements())->get(i),
784 isolate_);
785 if (!WriteObject(element).FromMaybe(false)) return Nothing<bool>();
786 }
787 break;
788 }
789 default:
790 break;
791 }
792
793 // If there are elements remaining, serialize them slowly.
794 for (; i < length; i++) {
795 // Serializing the array's elements can have arbitrary side effects, so we
796 // cannot rely on still having fast elements, even if it did to begin
797 // with.
798 Handle<Object> element;
799 LookupIterator it(isolate_, array, i, array, LookupIterator::OWN);
800 if (!it.IsFound()) {
801 // This can happen in the case where an array that was originally dense
802 // became sparse during serialization. It's too late to switch to the
803 // sparse format, but we can mark the elements as absent.
804 WriteTag(SerializationTag::kTheHole);
805 continue;
806 }
807 if (!Object::GetProperty(&it).ToHandle(&element) ||
808 !WriteObject(element).FromMaybe(false)) {
809 return Nothing<bool>();
810 }
811 }
812
813 DirectHandle<FixedArray> keys;
814 if (!KeyAccumulator::GetKeys(isolate_, array, KeyCollectionMode::kOwnOnly,
815 ENUMERABLE_STRINGS,
816 GetKeysConversion::kKeepNumbers, false, true)
817 .ToHandle(&keys)) {
818 return Nothing<bool>();
819 }
820
821 uint32_t properties_written;
822 if (!WriteJSObjectPropertiesSlow(array, keys).To(&properties_written)) {
823 return Nothing<bool>();
824 }
825 WriteTag(SerializationTag::kEndDenseJSArray);
826 WriteVarint<uint32_t>(properties_written);
827 WriteVarint<uint32_t>(length);
828 } else {
829 WriteTag(SerializationTag::kBeginSparseJSArray);
830 WriteVarint<uint32_t>(length);
831 DirectHandle<FixedArray> keys;
832 uint32_t properties_written = 0;
833 if (!KeyAccumulator::GetKeys(isolate_, array, KeyCollectionMode::kOwnOnly,
834 ENUMERABLE_STRINGS)
835 .ToHandle(&keys) ||
836 !WriteJSObjectPropertiesSlow(array, keys).To(&properties_written)) {
837 return Nothing<bool>();
838 }
839 WriteTag(SerializationTag::kEndSparseJSArray);
840 WriteVarint<uint32_t>(properties_written);
841 WriteVarint<uint32_t>(length);
842 }
843 return ThrowIfOutOfMemory();
844}

886 // First copy the key-value pairs, since getters could mutate them.
887 DirectHandle<OrderedHashMap> table(Cast<OrderedHashMap>(js_map->table()),
888 isolate_);
889 int length = table->NumberOfElements() * 2;
890 DirectHandle<FixedArray> entries = isolate_->factory()->NewFixedArray(length);

892 DisallowGarbageCollection no_gc;
893 Tagged<OrderedHashMap> raw_table = *table;

908 WriteTag(SerializationTag::kBeginJSMap);
909 for (int i = 0; i < length; i++) {
910 if (!WriteObject(handle(entries->get(i), isolate_)).FromMaybe(false)) {
911 return Nothing<bool>();
912 }
913 }
914 WriteTag(SerializationTag::kEndJSMap);
915 WriteVarint<uint32_t>(length);
916 return ThrowIfOutOfMemory();

1043Maybe<bool> ValueSerializer::WriteJSError(DirectHandle<JSObject> error) {
1044 Handle<Object> stack;
1045 PropertyDescriptor message_desc;
1046 Maybe<bool> message_found = JSReceiver::GetOwnPropertyDescriptor(
1047 isolate_, error, isolate_->factory()->message_string(), &message_desc);
1048 MAYBE_RETURN(message_found, Nothing<bool>());
1049 PropertyDescriptor cause_desc;
1050 Maybe<bool> cause_found = JSReceiver::GetOwnPropertyDescriptor(
1051 isolate_, error, isolate_->factory()->cause_string(), &cause_desc);
1052
1053 WriteTag(SerializationTag::kError);
1054
1055 Handle<Object> name_object;
1056 if (!JSObject::GetProperty(isolate_, error, "name").ToHandle(&name_object)) {
1057 return Nothing<bool>();
1058 }
1059 DirectHandle<String> name;
1060 if (!Object::ToString(isolate_, name_object).ToHandle(&name)) {
1061 return Nothing<bool>();
1062 }
1063
1064 if (name->IsOneByteEqualTo(base::CStrVector("EvalError"))) {
1065 WriteVarint(static_cast<uint8_t>(ErrorTag::kEvalErrorPrototype));
1066 } else if (name->IsOneByteEqualTo(base::CStrVector("RangeError"))) {
1067 WriteVarint(static_cast<uint8_t>(ErrorTag::kRangeErrorPrototype));
1068 } else if (name->IsOneByteEqualTo(base::CStrVector("ReferenceError"))) {
1069 WriteVarint(static_cast<uint8_t>(ErrorTag::kReferenceErrorPrototype));
1070 } else if (name->IsOneByteEqualTo(base::CStrVector("SyntaxError"))) {
1071 WriteVarint(static_cast<uint8_t>(ErrorTag::kSyntaxErrorPrototype));
1072 } else if (name->IsOneByteEqualTo(base::CStrVector("TypeError"))) {
1073 WriteVarint(static_cast<uint8_t>(ErrorTag::kTypeErrorPrototype));
1074 } else if (name->IsOneByteEqualTo(base::CStrVector("URIError"))) {
1075 WriteVarint(static_cast<uint8_t>(ErrorTag::kUriErrorPrototype));
1076 } else {
1077 // The default prototype in the deserialization side is Error.prototype, so
1078 // we don't have to do anything here.
1079 }
1080
1081 if (message_found.FromJust() &&
1082 PropertyDescriptor::IsDataDescriptor(&message_desc)) {
1083 Handle<String> message;
1084 if (!Object::ToString(isolate_, message_desc.value()).ToHandle(&message)) {
1085 return Nothing<bool>();
1086 }
1087 WriteVarint(static_cast<uint8_t>(ErrorTag::kMessage));
1088 WriteString(message);
1089 }
1090
1091 if (!Object::GetProperty(isolate_, error, isolate_->factory()->stack_string())
1092 .ToHandle(&stack)) {
1093 return Nothing<bool>();
1094 }
1095 if (IsString(*stack)) {
1096 WriteVarint(static_cast<uint8_t>(ErrorTag::kStack));
1097 WriteString(Cast<String>(stack));
1098 }
1099
1100 // The {cause} can self-reference the error. We add at the end, so that we can
1101 // create the Error first when deserializing.
1102 if (cause_found.FromJust() &&
1103 PropertyDescriptor::IsDataDescriptor(&cause_desc)) {
1104 Handle<Object> cause = cause_desc.value();
1105 WriteVarint(static_cast<uint8_t>(ErrorTag::kCause));
1106 if (!WriteObject(cause).FromMaybe(false)) {
1107 return Nothing<bool>();
1108 }
1109 }
1110
1111 WriteVarint(static_cast<uint8_t>(ErrorTag::kEnd));
1112 return ThrowIfOutOfMemory();
1113}

1347template <typename T>
1348Maybe<T> ValueDeserializer::ReadVarint() {
1349 // Reads an unsigned integer as a base-128 varint.
1350 // The number is written, 7 bits at a time, from the least significant to the
1351 // most significant 7 bits. Each byte, except the last, has the MSB set.
1352 // If the varint is larger than T, any more significant bits are discarded.
1353 // See also https://developers.google.com/protocol-buffers/docs/encoding
1354 static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
1355 "Only unsigned integer types can be read as varints.");
1356 if (sizeof(T) > 4) return ReadVarintLoop<T>();
1357 auto max_read_position = position_ + sizeof(T) + 1;
1358 if (V8_UNLIKELY(max_read_position >= end_)) return ReadVarintLoop<T>();
1359#ifdef DEBUG
1360 // DCHECK code to make sure the manually unrolled loop yields the exact
1361 // same end state and result.
1362 auto previous_position = position_;
1363 Maybe<T> maybe_expected_value = ReadVarintLoop<T>();
1364 // ReadVarintLoop can't return Nothing here; all such conditions have been
1365 // checked above.
1366 T expected_value = maybe_expected_value.ToChecked();
1367 auto expected_position = position_;
1368 position_ = previous_position;
1369#endif // DEBUG
1370#define EXIT_DCHECK() \
1371 DCHECK_LE(position_, end_); \
1372 DCHECK_EQ(position_, expected_position); \
1373 DCHECK_EQ(value, expected_value)
1374
1375 T value = 0;
1376#define ITERATION_SHIFTED(shift) \
1377 if (shift < sizeof(T) * 8) { \
1378 uint8_t byte = *position_; \
1379 position_++; \
1380 if (byte < 0x80) { \
1381 value |= static_cast<T>(byte) << shift; \
1382 EXIT_DCHECK(); \
1383 return Just(value); \
1384 } else { \
1385 value |= static_cast<T>(byte & 0x7F) << shift; \
1386 } \
1387 }
1388 // Manually unroll the loop to achieve the best measured peformance.
1389 // This is ~15% faster than ReadVarintLoop.
1390 ITERATION_SHIFTED(0);
1391 ITERATION_SHIFTED(7);
1392 ITERATION_SHIFTED(14);
1393 ITERATION_SHIFTED(21);
1394 ITERATION_SHIFTED(28);
1395
1396 EXIT_DCHECK();
1397 return Just(value);
1398#undef ITERATION_SHIFTED
1399#undef EXIT_DCHECK
1400}
1401
1402template <typename T>
1403Maybe<T> ValueDeserializer::ReadVarintLoop() {
1404 static_assert(std::is_integral_v<T> && std::is_unsigned_v<T>,
1405 "Only unsigned integer types can be read as varints.");
1406 T value = 0;
1407 unsigned shift = 0;
1408 bool has_another_byte;
1409 do {
1410 if (position_ >= end_) return Nothing<T>();
1411 uint8_t byte = *position_;
1412 has_another_byte = byte & 0x80;
1413 if (V8_LIKELY(shift < sizeof(T) * 8)) {
1414 value |= static_cast<T>(byte & 0x7F) << shift;
1415 shift += 7;
1416 } else {
1417 // For consistency with the fast unrolled loop in ReadVarint we return
1418 // after we have read size(T) + 1 bytes.
1419#ifdef V8_VALUE_DESERIALIZER_HARD_FAIL
1420 CHECK(!has_another_byte);
1421#endif // V8_VALUE_DESERIALIZER_HARD_FAIL
1422 return Just(value);
1423 }
1424 position_++;
1425 } while (has_another_byte);
1426 return Just(value);
1427}

1429template <typename T>
1430Maybe<T> ValueDeserializer::ReadZigZag() {
1431 // Writes a signed integer as a varint using ZigZag encoding (i.e. 0 is
1432 // encoded as 0, -1 as 1, 1 as 2, -2 as 3, and so on).
1433 // See also https://developers.google.com/protocol-buffers/docs/encoding
1434 static_assert(std::is_integral_v<T> && std::is_signed_v<T>,
1435 "Only signed integer types can be read as zigzag.");
1436 using UnsignedT = std::make_unsigned_t<T>;
1437 UnsignedT unsigned_value;
1438 if (!ReadVarint<UnsignedT>().To(&unsigned_value)) return Nothing<T>();
1439 return Just(static_cast<T>((unsigned_value >> 1) ^
1440 -static_cast<T>(unsigned_value & 1)));
1441}

1523MaybeHandle<Object> ValueDeserializer::ReadObjectWrapper() {
1524 // We had a bug which produced invalid version 13 data (see
1525 // crbug.com/1284506). This compatibility mode tries to first read the data
1526 // normally, and if it fails, and the version is 13, tries to read the broken
1527 // format.
1528 const uint8_t* original_position = position_;
1529 suppress_deserialization_errors_ = true;
1530 MaybeHandle<Object> result = ReadObject();
1531
1532 // The deserialization code doesn't throw errors for invalid data. It throws
1533 // errors for stack overflows, though, and in that case we won't retry.
1534 if (result.is_null() && version_ == 13 && !isolate_->has_exception()) {
1535 version_13_broken_data_mode_ = true;
1536 position_ = original_position;
1537 result = ReadObject();
1538 }
1539
1540 if (result.is_null() && !isolate_->has_exception()) {
1541 isolate_->Throw(*isolate_->factory()->NewError(
1542 MessageTemplate::kDataCloneDeserializationError));
1543 }
1544
1545 return result;
1546}

1548MaybeHandle<Object> ValueDeserializer::ReadObject() {
1549 DisallowJavascriptExecution no_js(isolate_);
1550 // If we are at the end of the stack, abort. This function may recurse.
1551 STACK_CHECK(isolate_, MaybeHandle<Object>());
1552
1553 MaybeHandle<Object> result = ReadObjectInternal();
1554
1555 // ArrayBufferView is special in that it consumes the value before it, even
1556 // after format version 0.
1557 DirectHandle<Object> object;
1558 SerializationTag tag;
1559 if (result.ToHandle(&object) && V8_UNLIKELY(IsJSArrayBuffer(*object)) &&
1560 PeekTag().To(&tag) && tag == SerializationTag::kArrayBufferView) {
1561 ConsumeTag(SerializationTag::kArrayBufferView);
1562 result = ReadJSArrayBufferView(Cast<JSArrayBuffer>(object));
1563 }
1564
1565 if (result.is_null() && !suppress_deserialization_errors_ &&
1566 !isolate_->has_exception()) {
1567 isolate_->Throw(*isolate_->factory()->NewError(
1568 MessageTemplate::kDataCloneDeserializationError));
1569 }
1570#if defined(DEBUG) && defined(VERIFY_HEAP)
1571 if (!result.is_null() && v8_flags.enable_slow_asserts &&
1572 v8_flags.verify_heap) {
1573 Object::ObjectVerify(*object, isolate_);
1574 }
1575#endif
1576
1577 return result;
1578}

1580MaybeHandle<Object> ValueDeserializer::ReadObjectInternal() {
1581 SerializationTag tag;
1582 if (!ReadTag().To(&tag)) return MaybeHandle<Object>();
1583 switch (tag) {
1584 case SerializationTag::kVerifyObjectCount:
1585 // Read the count and ignore it.
1586 if (ReadVarint<uint32_t>().IsNothing()) return MaybeHandle<Object>();
1587 return ReadObject();
1588 case SerializationTag::kUndefined:
1589 return isolate_->factory()->undefined_value();
1590 case SerializationTag::kNull:
1591 return isolate_->factory()->null_value();
1592 case SerializationTag::kTrue:
1593 return isolate_->factory()->true_value();
1594 case SerializationTag::kFalse:
1595 return isolate_->factory()->false_value();
1596 case SerializationTag::kInt32: {
1597 Maybe<int32_t> number = ReadZigZag<int32_t>();
1598 if (number.IsNothing()) return MaybeHandle<Object>();
1599 return isolate_->factory()->NewNumberFromInt(number.FromJust());
1600 }
1601 case SerializationTag::kUint32: {
1602 Maybe<uint32_t> number = ReadVarint<uint32_t>();
1603 if (number.IsNothing()) return MaybeHandle<Object>();
1604 return isolate_->factory()->NewNumberFromUint(number.FromJust());
1605 }
1606 case SerializationTag::kDouble: {
1607 Maybe<double> number = ReadDouble();
1608 if (number.IsNothing()) return MaybeHandle<Object>();
1609 return isolate_->factory()->NewNumber(number.FromJust());
1610 }
1611 case SerializationTag::kBigInt:
1612 return ReadBigInt();
1613 case SerializationTag::kUtf8String:
1614 return ReadUtf8String();
1615 case SerializationTag::kOneByteString:
1616 return ReadOneByteString();
1617 case SerializationTag::kTwoByteString:
1618 return ReadTwoByteString();
1619 case SerializationTag::kObjectReference: {
1620 uint32_t id;
1621 if (!ReadVarint<uint32_t>().To(&id)) return MaybeHandle<Object>();
1622 return GetObjectWithID(id);
1623 }
1624 case SerializationTag::kBeginJSObject:
1625 return ReadJSObject();
1626 case SerializationTag::kBeginSparseJSArray:
1627 return ReadSparseJSArray();
1628 case SerializationTag::kBeginDenseJSArray:
1629 return ReadDenseJSArray();
1630 case SerializationTag::kDate:
1631 return ReadJSDate();
1632 case SerializationTag::kTrueObject:
1633 case SerializationTag::kFalseObject:
1634 case SerializationTag::kNumberObject:
1635 case SerializationTag::kBigIntObject:
1636 case SerializationTag::kStringObject:
1637 return ReadJSPrimitiveWrapper(tag);
1638 case SerializationTag::kRegExp:
1639 return ReadJSRegExp();
1640 case SerializationTag::kBeginJSMap:
1641 return ReadJSMap();
1642 case SerializationTag::kBeginJSSet:
1643 return ReadJSSet();
1644 case SerializationTag::kArrayBuffer: {
1645 constexpr bool is_shared = false;
1646 constexpr bool is_resizable = false;
1647 return ReadJSArrayBuffer(is_shared, is_resizable);
1648 }
1649 case SerializationTag::kResizableArrayBuffer: {
1650 constexpr bool is_shared = false;
1651 constexpr bool is_resizable = true;
1652 return ReadJSArrayBuffer(is_shared, is_resizable);
1653 }
1654 case SerializationTag::kArrayBufferTransfer: {
1655 return ReadTransferredJSArrayBuffer();
1656 }
1657 case SerializationTag::kSharedArrayBuffer: {
1658 constexpr bool is_shared = true;
1659 constexpr bool is_resizable = false;
1660 return ReadJSArrayBuffer(is_shared, is_resizable);
1661 }
1662 case SerializationTag::kError:
1663 return ReadJSError();
1664#if V8_ENABLE_WEBASSEMBLY
1665 case SerializationTag::kWasmModuleTransfer:
1666 return ReadWasmModuleTransfer();
1667 case SerializationTag::kWasmMemoryTransfer:
1668 return ReadWasmMemory();
1669#endif // V8_ENABLE_WEBASSEMBLY
1670 case SerializationTag::kHostObject:
1671 return ReadHostObject();
1672 case SerializationTag::kSharedObject:
1673 if (version_ >= 15) return ReadSharedObject();
1674 // If the data doesn't support shared values because it is from an older
1675 // version, treat the tag as unknown.
1676 [[fallthrough]];
1677 default:
1678 // Before there was an explicit tag for host objects, all unknown tags
1679 // were delegated to the host.
1680 if (version_ < 13) {
1681 position_--;
1682 return ReadHostObject();
1683 }
1684 return MaybeHandle<Object>();
1685 }
1686}

1809MaybeHandle<JSArray> ValueDeserializer::ReadDenseJSArray() {
1810 // If we are at the end of the stack, abort. This function may recurse.
1811 STACK_CHECK(isolate_, MaybeHandle<JSArray>());
1812
1813 // We shouldn't permit an array larger than the biggest we can request from
1814 // V8. As an additional sanity check, since each entry will take at least one
1815 // byte to encode, if there are fewer bytes than that we can also fail fast.
1816 uint32_t length;
1817 if (!ReadVarint<uint32_t>().To(&length) ||
1818 length > static_cast<uint32_t>(FixedArray::kMaxLength) ||
1819 length > static_cast<size_t>(end_ - position_)) {
1820 return MaybeHandle<JSArray>();
1821 }
1822
1823 uint32_t id = next_id_++;
1824 HandleScope scope(isolate_);
1825 Handle<JSArray> array = isolate_->factory()->NewJSArray(
1826 HOLEY_ELEMENTS, length, length,
1827 ArrayStorageAllocationMode::INITIALIZE_ARRAY_ELEMENTS_WITH_HOLE);
1828 AddObjectWithID(id, array);
1829
1830 DirectHandle<FixedArray> elements(Cast<FixedArray>(array->elements()),
1831 isolate_);
1832 auto elements_length = static_cast<uint32_t>(elements->length());
1833 for (uint32_t i = 0; i < length; i++) {
1834 SerializationTag tag;
1835 if (PeekTag().To(&tag) && tag == SerializationTag::kTheHole) {
1836 ConsumeTag(SerializationTag::kTheHole);
1837 continue;
1838 }
1839
1840 DirectHandle<Object> element;
1841 if (!ReadObject().ToHandle(&element)) return MaybeHandle<JSArray>();
1842
1843 // Serialization versions less than 11 encode the hole the same as
1844 // undefined. For consistency with previous behavior, store these as the
1845 // hole. Past version 11, undefined means undefined.

2440Maybe<uint32_t> ValueDeserializer::ReadJSObjectProperties(
2441 DirectHandle<JSObject> object, SerializationTag end_tag,
2442 bool can_use_transitions) {
2443 uint32_t num_properties = 0;
2444
2445 // Fast path (following map transitions).
2446 if (can_use_transitions) {
2447 bool transitioning = true;
2448 Handle<Map> map(object->map(), isolate_);
2449 DCHECK(!map->is_dictionary_map());
2450 DCHECK_EQ(0, map->instance_descriptors(isolate_)->number_of_descriptors());
2451 std::vector<Handle<Object>> properties;
2452 properties.reserve(8);
2453
2454 while (transitioning) {
2455 // If there are no more properties, finish.
2456 SerializationTag tag;
2457 if (!PeekTag().To(&tag)) return Nothing<uint32_t>();
2458 if (tag == end_tag) {
2459 ConsumeTag(end_tag);
2460 CommitProperties(object, map, properties);
2461 CHECK_LT(properties.size(), std::numeric_limits<uint32_t>::max());
2462 return Just(static_cast<uint32_t>(properties.size()));
2463 }
2464
2465 // Determine the key to be used and the target map to transition to, if
2466 // possible. Transitioning may abort if the key is not a string, or if no
2467 // transition was found.
2468 Handle<Object> key;
2469 Handle<Map> target;
2470 bool transition_was_found = false;
2471 const uint8_t* start_position = position_;
2472 uint32_t byte_length;
2473 if (!ReadTag().To(&tag) || !ReadVarint<uint32_t>().To(&byte_length)) {
2474 return Nothing<uint32_t>();
2475 }
2476 // Length is also checked in ReadRawBytes.

2483 TransitionsAccessor transitions(isolate_, *map);
2484 if (tag == SerializationTag::kOneByteString) {
2485 base::Vector<const uint8_t> key_chars;
2486 if (ReadRawBytes(byte_length).To(&key_chars)) {
2487 expected_transition = transitions.ExpectedTransition(key_chars);
2488 }
2489 } else if (tag == SerializationTag::kTwoByteString) {
2490 base::Vector<const base::uc16> key_chars;
2491 if (ReadRawTwoBytes(byte_length).To(&key_chars)) {

2504 target = expected_transition.second;

2524 Handle<Object> value;

2528 // (though generalization may be required), store the property value so
2529 // that we can copy them all at once. Otherwise, stop transitioning.

2532 // ensure we are working with the up-to-date version.

2537 target->instance_descriptors(isolate_)->GetDetails(descriptor);

2541 !FieldType::NowContains(
2542 target->instance_descriptors(isolate_)->GetFieldType(
2543 descriptor),
2544 value)) {

2546 *value, isolate_, expected_representation);
2547 MapUpdater::GeneralizeField(isolate_, target, descriptor,
2548 details.constness(),
2549 expected_representation, value_type);
2550 }
2551 DCHECK(FieldType::NowContains(
2552 target->instance_descriptors(isolate_)->GetFieldType(
2553 descriptor),
2554 value));
2555 properties.push_back(value);
2556 map = target;
2557 continue;
2558 }
2559 }
2560 transitioning = false;
2561 }
2562
2563 // Fell out of transitioning fast path. Commit the properties gathered so
2564 // far, and then start setting properties slowly instead.
2565 DCHECK(!transitioning);
2566 CHECK_LT(properties.size(), std::numeric_limits<uint32_t>::max());

2568 CommitProperties(object, map, properties);
2569 num_properties = static_cast<uint32_t>(properties.size());
2570
2571 // We checked earlier that IsValidObjectKey(key).
2572 PropertyKey lookup_key(isolate_, key);
2573 LookupIterator it(isolate_, object, lookup_key, LookupIterator::OWN);
2574 if (it.state() != LookupIterator::NOT_FOUND ||
2575 JSObject::DefineOwnPropertyIgnoreAttributes(&it, value, NONE)
2576 .is_null()) {
2577 return Nothing<uint32_t>();
2578 }
2579 num_properties++;
2580 }

2582 // At this point, transitioning should be done, but at least one property
2583 // should have been written (in the zero-property case, there is an early
2584 // return).
2585 DCHECK(!transitioning);
2586 DCHECK_GE(num_properties, 1u);
2587 }
2588
2589 // Slow path.
2590 for (;; num_properties++) {
2591 SerializationTag tag;
2592 if (!PeekTag().To(&tag)) return Nothing<uint32_t>();
2593 if (tag == end_tag) {
2594 ConsumeTag(end_tag);
2595 return Just(num_properties);
2596 }
2597
2598 Handle<Object> key;
2599 if (!ReadObject().ToHandle(&key) || !IsValidObjectKey(*key, isolate_)) {
2600 return Nothing<uint32_t>();
2601 }
2602 Handle<Object> value;
2603 if (!ReadObject().ToHandle(&value)) return Nothing<uint32_t>();
2604
2605 // We checked earlier that IsValidObjectKey(key).
2606 PropertyKey lookup_key(isolate_, key);
2607 LookupIterator it(isolate_, object, lookup_key, LookupIterator::OWN);
2608 if (it.state() != LookupIterator::NOT_FOUND ||
2609 JSObject::DefineOwnPropertyIgnoreAttributes(&it, value, NONE)
2610 .is_null()) {

2664ValueDeserializer::ReadObjectUsingEntireBufferForLegacyFormat() {
2665 DCHECK_EQ(version_, 0u);
2666 HandleScope scope(isolate_);

2668 while (position_ < end_) {
2669 SerializationTag tag;
2670 if (!PeekTag().To(&tag)) break;
2671
2672 Handle<Object> new_object;
2673 switch (tag) {
2674 case SerializationTag::kEndJSObject: {
2675 ConsumeTag(SerializationTag::kEndJSObject);
2676
2677 // JS Object: Read the last 2*n values from the stack and use them as
2678 // key-value pairs.
2679 uint32_t num_properties;
2680 if (!ReadVarint<uint32_t>().To(&num_properties) ||
2681 stack.size() / 2 < num_properties) {
2682 isolate_->Throw(*isolate_->factory()->NewError(
2683 MessageTemplate::kDataCloneDeserializationError));
2684 return MaybeHandle<Object>();
2685 }
2686
2687 size_t begin_properties =
2688 stack.size() - 2 * static_cast<size_t>(num_properties);
2689 Handle<JSObject> js_object =
2690 isolate_->factory()->NewJSObject(isolate_->object_function());
2691 if (num_properties &&
2692 !SetPropertiesFromKeyValuePairs(
2693 isolate_, js_object, &stack[begin_properties], num_properties)
2694 .FromMaybe(false)) {
2695 ThrowDeserializationExceptionIfNonePending(isolate_);
2696 return MaybeHandle<Object>();
2697 }
2698
2699 stack.resize(begin_properties);
2700 new_object = js_object;
2701 break;
2702 }
2703 case SerializationTag::kEndSparseJSArray: {
2704 ConsumeTag(SerializationTag::kEndSparseJSArray);
2705
2706 // Sparse JS Array: Read the last 2*|num_properties| from the stack.
2707 uint32_t num_properties;
2708 uint32_t length;
2709 if (!ReadVarint<uint32_t>().To(&num_properties) ||
2710 !ReadVarint<uint32_t>().To(&length) ||
2711 stack.size() / 2 < num_properties) {
2712 isolate_->Throw(*isolate_->factory()->NewError(
2713 MessageTemplate::kDataCloneDeserializationError));
2714 return MaybeHandle<Object>();
2715 }
2716
2717 Handle<JSArray> js_array =
2718 isolate_->factory()->NewJSArray(0, TERMINAL_FAST_ELEMENTS_KIND);
2719 MAYBE_RETURN_NULL(JSArray::SetLength(js_array, length));
2720 size_t begin_properties =
2721 stack.size() - 2 * static_cast<size_t>(num_properties);
2722 if (num_properties &&
2723 !SetPropertiesFromKeyValuePairs(
2724 isolate_, js_array, &stack[begin_properties], num_properties)
2725 .FromMaybe(false)) {
2726 ThrowDeserializationExceptionIfNonePending(isolate_);
2727 return MaybeHandle<Object>();
2728 }
2729
2730 stack.resize(begin_properties);
2731 new_object = js_array;
2732 break;
2733 }
2734 case SerializationTag::kEndDenseJSArray: {
2735 // This was already broken in Chromium, and apparently wasn't missed.
2736 isolate_->Throw(*isolate_->factory()->NewError(
2737 MessageTemplate::kDataCloneDeserializationError));
2738 return MaybeHandle<Object>();
2739 }
2740 default:
2741 if (!ReadObject().ToHandle(&new_object)) return MaybeHandle<Object>();
2742 break;
2743 }
2744 stack.push_back(new_object);
2745 }
2746
2747// Nothing remains but padding.
2748#ifdef DEBUG
2749 while (position_ < end_) {
2750 DCHECK(*position_++ == static_cast<uint8_t>(SerializationTag::kPadding));
2751 }
2752#endif
2753 position_ = end_;
2754
2755 if (stack.size() != 1) {
2756 isolate_->Throw(*isolate_->factory()->NewError(
2757 MessageTemplate::kDataCloneDeserializationError));
2758 return MaybeHandle<Object>();
2759 }
2760 return scope.CloseAndEscape(stack[0]);
2761}
