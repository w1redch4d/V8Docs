27double GetPretenuringRatioThreshold(size_t new_space_capacity) {
28 static constexpr double kScavengerPretenureRatio = 0.85;
29 // MinorMS allows for a much larger new space, thus we require a lower
30 // survival rate for pretenuring.
31 static constexpr double kMinorMSPretenureMaxRatio = 0.8;
32 static constexpr double kMinorMSMinCapacity = 16 * MB;
33 if (!v8_flags.minor_ms) return kScavengerPretenureRatio;
34 if (new_space_capacity <= kMinorMSMinCapacity)
35 return kMinorMSPretenureMaxRatio;
36 // When capacity is 64MB, the pretenuring ratio would be 0.2.
37 return kMinorMSPretenureMaxRatio * kMinorMSMinCapacity / new_space_capacity;
38}

40inline bool MakePretenureDecision(
41 Tagged<AllocationSite> site,
42 AllocationSite::PretenureDecision current_decision, double ratio,
43 bool new_space_capacity_was_above_pretenuring_threshold,
44 size_t new_space_capacity) {
45 // Here we just allow state transitions from undecided or maybe tenure
46 // to don't tenure, maybe tenure, or tenure.
47 if ((current_decision == AllocationSite::kUndecided ||
48 current_decision == AllocationSite::kMaybeTenure)) {
49 if (ratio >= GetPretenuringRatioThreshold(new_space_capacity)) {
50 // We just transition into tenure state when the semi-space was at
51 // maximum capacity.
52 if (new_space_capacity_was_above_pretenuring_threshold) {
53 site->set_deopt_dependent_code(true);
54 site->set_pretenure_decision(AllocationSite::kTenure);
55 // Currently we just need to deopt when we make a state transition to
56 // tenure.
57 return true;
58 }
59 site->set_pretenure_decision(AllocationSite::kMaybeTenure);
60 } else {
61 site->set_pretenure_decision(AllocationSite::kDontTenure);
62 }
63 }
64 return false;
65}

73inline bool DigestPretenuringFeedback(
74 Isolate* isolate, Tagged<AllocationSite> site,
75 bool new_space_capacity_was_above_pretenuring_threshold,
76 size_t new_space_capacity) {
77 bool deopt = false;
78 int create_count = site->memento_create_count();
79 int found_count = site->memento_found_count();
80 bool minimum_mementos_created = create_count >= kMinMementoCount;
81 double ratio =
82 minimum_mementos_created || v8_flags.trace_pretenuring_statistics
83 ? static_cast<double>(found_count) / create_count
84 : 0.0;
85 AllocationSite::PretenureDecision current_decision =
86 site->pretenure_decision();
87
88 if (minimum_mementos_created) {
89 deopt = MakePretenureDecision(
90 site, current_decision, ratio,
91 new_space_capacity_was_above_pretenuring_threshold, new_space_capacity);
92 }
93
94 if (V8_UNLIKELY(v8_flags.trace_pretenuring_statistics)) {
95 PrintIsolate(isolate,
96 "pretenuring: AllocationSite(%p): (created, found, ratio) "
97 "(%d, %d, %f) %s => %s\n",
98 reinterpret_cast<void*>(site.ptr()), create_count, found_count,
99 ratio, site->PretenureDecisionName(current_decision),
100 site->PretenureDecisionName(site->pretenure_decision()));
101 }
102
103 ResetPretenuringFeedback(site);
104 return deopt;
105}

107bool PretenureAllocationSiteManually(Isolate* isolate,
108 Tagged<AllocationSite> site) {
109 AllocationSite::PretenureDecision current_decision =
110 site->pretenure_decision();
111 bool deopt = true;
112 if (current_decision == AllocationSite::kUndecided ||
113 current_decision == AllocationSite::kMaybeTenure) {
114 site->set_deopt_dependent_code(true);
115 site->set_pretenure_decision(AllocationSite::kTenure);
116 } else {
117 deopt = false;
118 }
119 if (v8_flags.trace_pretenuring_statistics) {
120 PrintIsolate(isolate,
121 "pretenuring manually requested: AllocationSite(%p): "
122 "%s => %s\n",
123 reinterpret_cast<void*>(site.ptr()),
124 site->PretenureDecisionName(current_decision),
125 site->PretenureDecisionName(site->pretenure_decision()));
126 }
127
128 ResetPretenuringFeedback(site);
129 return deopt;
130}

139void PretenuringHandler::MergeAllocationSitePretenuringFeedback(
140 const PretenuringFeedbackMap& local_pretenuring_feedback) {
141 PtrComprCageBase cage_base(heap_->isolate());
142 Tagged<AllocationSite> site;
143 for (auto& site_and_count : local_pretenuring_feedback) {
144 site = site_and_count.first;
145 MapWord map_word = site->map_word(cage_base, kRelaxedLoad);
146 if (map_word.IsForwardingAddress()) {
147 site = Cast<AllocationSite>(map_word.ToForwardingAddress(site));
148 }
149
150 // We have not validated the allocation site yet, since we have not
151 // dereferenced the site during collecting information.
152 // This is an inlined check of AllocationMemento::IsValid.
153 if (!IsAllocationSite(site) || site->IsZombie()) continue;
154
155 const int value = static_cast<int>(site_and_count.second);
156 DCHECK_LT(0, value);
157 if (site->IncrementMementoFoundCount(value) >= kMinMementoCount) {
158 // For sites in the global map the count is accessed through the site.
159 global_pretenuring_feedback_.insert(std::make_pair(site, 0));
160 }
161 }
162}

169void PretenuringHandler::ProcessPretenuringFeedback(
170 size_t new_space_capacity_target_capacity) {
171 // The minimum new space capacity from which allocation sites can be
172 // pretenured. A too small capacity means frequent GCs. Objects thus don't get
173 // a chance to die before being promoted, which may lead to wrong pretenuring
174 // decisions.
175 static constexpr size_t kDefaultMinNewSpaceCapacityForPretenuring =
176 8192 * KB * Heap::kPointerMultiplier;
177
178 DCHECK(heap_->tracer()->IsInAtomicPause());
179
180 if (!v8_flags.allocation_site_pretenuring) return;
181
182 // TODO(333906585): Adjust capacity for sticky bits.
183 const size_t max_capacity = v8_flags.sticky_mark_bits
184 ? heap_->sticky_space()->Capacity()
185 : heap_->new_space()->MaximumCapacity();
186 const size_t min_new_space_capacity_for_pretenuring =
187 std::min(max_capacity, kDefaultMinNewSpaceCapacityForPretenuring);
188
189 bool trigger_deoptimization = false;
190 int tenure_decisions = 0;
191 int dont_tenure_decisions = 0;
192 int allocation_mementos_found = 0;
193 int allocation_sites = 0;
194 int active_allocation_sites = 0;
195
196 Tagged<AllocationSite> site;
197
198 // Step 1: Digest feedback for recorded allocation sites.
199 // This is the pretenuring trigger for allocation sites that are in maybe
200 // tenure state. When we switched to a large enough new space size we
201 // deoptimize the code that belongs to the allocation site and derive the
202 // lifetime of the allocation site.
203 const bool new_space_was_above_pretenuring_threshold =
204 new_space_capacity_target_capacity >=
205 min_new_space_capacity_for_pretenuring;
206
207 for (auto& site_and_count : global_pretenuring_feedback_) {
208 allocation_sites++;
209 site = site_and_count.first;
210 // Count is always access through the site.
211 DCHECK_EQ(0, site_and_count.second);
212 int found_count = site->memento_found_count();
213 // An entry in the storage does not imply that the count is > 0 because
214 // allocation sites might have been reset due to too many objects dying
215 // in old space.
216 if (found_count > 0) {
217 DCHECK(IsAllocationSite(site));
218 active_allocation_sites++;
219 allocation_mementos_found += found_count;
220 if (DigestPretenuringFeedback(heap_->isolate(), site,
221 new_space_was_above_pretenuring_threshold,
222 new_space_capacity_target_capacity)) {
223 trigger_deoptimization = true;
224 }
225 if (site->GetAllocationType() == AllocationType::kOld) {
226 tenure_decisions++;
227 } else {
228 dont_tenure_decisions++;
229 }
230 }
231 }
232
233 // Step 2: Pretenure allocation sites for manual requests.
234 if (allocation_sites_to_pretenure_) {
235 while (!allocation_sites_to_pretenure_->empty()) {
236 auto pretenure_site = allocation_sites_to_pretenure_->Pop();
237 if (PretenureAllocationSiteManually(heap_->isolate(), pretenure_site)) {
238 trigger_deoptimization = true;
239 }
240 }
241 allocation_sites_to_pretenure_.reset();
242 }
243
244 // Step 3: Deopt maybe tenured allocation sites if necessary.
245 // New space capacity was too low for pretenuring but is now above the
246 // threshold. Maybe tenured allocation sites may be pretenured on the next GC.
247 bool deopt_maybe_tenured = (heap_->NewSpaceTargetCapacity() >=
248 min_new_space_capacity_for_pretenuring) &&
249 !new_space_was_above_pretenuring_threshold;
250 if (deopt_maybe_tenured) {
251 heap_->ForeachAllocationSite(heap_->allocation_sites_list(),
252 [&allocation_sites, &trigger_deoptimization](
253 Tagged<AllocationSite> site) {
254 DCHECK(IsAllocationSite(site));
255 allocation_sites++;
256 if (site->IsMaybeTenure()) {
257 site->set_deopt_dependent_code(true);
258 trigger_deoptimization = true;
259 }
260 });
261 }
262
263 if (trigger_deoptimization) {
264 heap_->isolate()->stack_guard()->RequestDeoptMarkedAllocationSites();
265 }
266
267 if (V8_UNLIKELY(v8_flags.trace_pretenuring_statistics) &&
268 (allocation_mementos_found > 0 || tenure_decisions > 0 ||
269 dont_tenure_decisions > 0)) {
270 PrintIsolate(
271 heap_->isolate(),
272 "pretenuring: threshold=%.2f deopt_maybe_tenured=%d visited_sites=%d "
273 "active_sites=%d "
274 "mementos=%d tenured=%d not_tenured=%d\n",
275 GetPretenuringRatioThreshold(new_space_capacity_target_capacity),
276 deopt_maybe_tenured ? 1 : 0, allocation_sites, active_allocation_sites,
277 allocation_mementos_found, tenure_decisions, dont_tenure_decisions);
278 }
279
280 global_pretenuring_feedback_.clear();
281 global_pretenuring_feedback_.reserve(kInitialFeedbackCapacity);
282}
