577 static inline uint32_t NonOneByteStart(const base::uc16* chars,
578 uint32_t length) {
579 DCHECK(IsAligned(reinterpret_cast<Address>(chars), sizeof(base::uc16)));
580 const uint16_t* start = chars;
581 const uint16_t* limit = chars + length;
582
583 if (static_cast<size_t>(length) >= kUIntptrSize) {
584 // Check unaligned chars.
585 while (!IsAligned(reinterpret_cast<Address>(chars), kUIntptrSize)) {
586 if (*chars > unibrow::Latin1::kMaxChar) {
587 return static_cast<uint32_t>(chars - start);
588 }
589 ++chars;
590 }
591
592 // Check aligned words.
593 static_assert(unibrow::Latin1::kMaxChar == 0xFF);
594#ifdef V8_TARGET_LITTLE_ENDIAN
595 const uintptr_t non_one_byte_mask = kUintptrAllBitsSet / 0xFFFF * 0xFF00;
596#else
597 const uintptr_t non_one_byte_mask = kUintptrAllBitsSet / 0xFFFF * 0x00FF;
598#endif
599 while (chars + sizeof(uintptr_t) <= limit) {
600 if (*reinterpret_cast<const uintptr_t*>(chars) & non_one_byte_mask) {
601 break;
602 }
603 chars += (sizeof(uintptr_t) / sizeof(base::uc16));
604 }
605 }
606
607 // Check remaining unaligned chars, or find non-one-byte char in word.
608 while (chars < limit) {
