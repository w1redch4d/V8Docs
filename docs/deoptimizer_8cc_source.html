51class DeoptimizableCodeIterator {
52 public:
53 explicit DeoptimizableCodeIterator(Isolate* isolate);
54 DeoptimizableCodeIterator(const DeoptimizableCodeIterator&) = delete;
55 DeoptimizableCodeIterator& operator=(const DeoptimizableCodeIterator&) =
56 delete;
57 Tagged<Code> Next();
58
59 private:
60 Isolate* const isolate_;
61 std::unique_ptr<SafepointScope> safepoint_scope_;
62 std::unique_ptr<ObjectIterator> object_iterator_;
63 enum { kIteratingCodeSpace, kIteratingCodeLOSpace, kDone } state_;
64
65 DISALLOW_GARBAGE_COLLECTION(no_gc)
66};
67
68DeoptimizableCodeIterator::DeoptimizableCodeIterator(Isolate* isolate)
69 : isolate_(isolate),
70 safepoint_scope_(std::make_unique<SafepointScope>(
71 isolate, isolate->is_shared_space_isolate()
72 ? SafepointKind::kGlobal
73 : SafepointKind::kIsolate)),
74 object_iterator_(
75 isolate->heap()->code_space()->GetObjectIterator(isolate->heap())),
76 state_(kIteratingCodeSpace) {}
77
78Tagged<Code> DeoptimizableCodeIterator::Next() {
79 while (true) {
80 Tagged<HeapObject> object = object_iterator_->Next();
81 if (object.is_null()) {
82 // No objects left in the current iterator, try to move to the next space
83 // based on the state.
84 switch (state_) {
85 case kIteratingCodeSpace: {
86 object_iterator_ =
87 isolate_->heap()->code_lo_space()->GetObjectIterator(
88 isolate_->heap());
89 state_ = kIteratingCodeLOSpace;
90 continue;
91 }
92 case kIteratingCodeLOSpace:
93 // No other spaces to iterate, so clean up and we're done. Keep the
94 // object iterator so that it keeps returning null on Next(), to avoid
95 // needing to branch on state_ before the while loop, but drop the
96 // safepoint scope since we no longer need to stop the heap from
97 // moving.
98 safepoint_scope_.reset();
99 state_ = kDone;
100 [[fallthrough]];
101 case kDone:
102 return Code();
103 }
104 }
105 Tagged<InstructionStream> istream = Cast<InstructionStream>(object);
106 Tagged<Code> code;
107 if (!istream->TryGetCode(&code, kAcquireLoad)) continue;
108 if (!CodeKindCanDeoptimize(code->kind())) continue;
109 return code;
110 }
111}

297DeoptimizedFrameInfo* Deoptimizer::DebuggerInspectableFrame(
298 JavaScriptFrame* frame, int jsframe_index, Isolate* isolate) {
299 CHECK(frame->is_optimized_js());
300
301 TranslatedState translated_values(frame);
302 translated_values.Prepare(frame->fp());
303
304 TranslatedState::iterator frame_it = translated_values.end();
305 int counter = jsframe_index;
306 for (auto it = translated_values.begin(); it != translated_values.end();
307 it++) {
308 if (it->kind() == TranslatedFrame::kUnoptimizedFunction ||
309 it->kind() == TranslatedFrame::kJavaScriptBuiltinContinuation ||
310 it->kind() ==
311 TranslatedFrame::kJavaScriptBuiltinContinuationWithCatch) {
312 if (counter == 0) {
313 frame_it = it;
314 break;
315 }
316 counter--;
317 }
318 }
319 CHECK(frame_it != translated_values.end());
320 // We only include kJavaScriptBuiltinContinuation frames above to get the
321 // counting right.
322 CHECK_EQ(frame_it->kind(), TranslatedFrame::kUnoptimizedFunction);
323
324 DeoptimizedFrameInfo* info =
325 new DeoptimizedFrameInfo(&translated_values, frame_it, isolate);
326
327 return info;
328}

331class ActivationsFinder : public ThreadVisitor {
332 public:
333 ActivationsFinder(Tagged<GcSafeCode> topmost_optimized_code,
334 bool safe_to_deopt_topmost_optimized_code) {
335#ifdef DEBUG
336 topmost_ = topmost_optimized_code;
337 safe_to_deopt_ = safe_to_deopt_topmost_optimized_code;
338#endif

341 // Find the frames with activations of codes marked for deoptimization, search
342 // for the trampoline to the deoptimizer call respective to each code, and use
343 // it to replace the current pc on the stack.
344 void VisitThread(Isolate* isolate, ThreadLocalTop* top) override {
345 for (StackFrameIterator it(isolate, top, StackFrameIterator::NoHandles{});
346 !it.done(); it.Advance()) {
347 if (it.frame()->is_optimized_js()) {
348 Tagged<GcSafeCode> code = it.frame()->GcSafeLookupCode();
349 if (CodeKindCanDeoptimize(code->kind()) &&
350 code->marked_for_deoptimization()) {
351 // Obtain the trampoline to the deoptimizer call.
352 int trampoline_pc;
353 if (code->is_maglevved()) {
354 MaglevSafepointEntry safepoint = MaglevSafepointTable::FindEntry(
355 isolate, code, it.frame()->pc());
356 trampoline_pc = safepoint.trampoline_pc();
357 } else {
358 SafepointEntry safepoint = SafepointTable::FindEntry(
359 isolate, code, it.frame()->maybe_unauthenticated_pc());
360 trampoline_pc = safepoint.trampoline_pc();
361 }
362 // TODO(saelo): currently we have to use full pointer comparison as
363 // builtin Code is still inside the sandbox while runtime-generated
364 // Code is in trusted space.
365 static_assert(!kAllCodeObjectsLiveInTrustedSpace);
366 DCHECK_IMPLIES(code.SafeEquals(topmost_), safe_to_deopt_);
367 static_assert(SafepointEntry::kNoTrampolinePC == -1);
368 CHECK_GE(trampoline_pc, 0);
369 if (!it.frame()->InFastCCall()) {
370 Address new_pc = code->instruction_start() + trampoline_pc;
371 if (v8_flags.cet_compatible) {
372 Address pc = *it.frame()->pc_address();
373 Deoptimizer::PatchToJump(pc, new_pc);
374 } else {
375 // Replace the current pc on the stack with the trampoline.
376 // TODO(v8:10026): avoid replacing a signed pointer.
377 Address* pc_addr = it.frame()->pc_address();
378 PointerAuthentication::ReplacePC(pc_addr, new_pc,
379 kSystemPointerSize);
380 }
381 }
382 }
383 }
384 }
385 }
386
387 private:
388#ifdef DEBUG
389 Tagged<GcSafeCode> topmost_;
390 bool safe_to_deopt_;
391#endif
392};

395// Replace pc on the stack for codes marked for deoptimization.
396// static
397void Deoptimizer::DeoptimizeMarkedCode(Isolate* isolate) {
398 DisallowGarbageCollection no_gc;
399
400 Tagged<GcSafeCode> topmost_optimized_code;
401 bool safe_to_deopt_topmost_optimized_code = false;
402#ifdef DEBUG
403 // Make sure all activations of optimized code can deopt at their current PC.
404 // The topmost optimized code has special handling because it cannot be
405 // deoptimized due to weak object dependency.
406 for (StackFrameIterator it(isolate, isolate->thread_local_top(),
407 StackFrameIterator::NoHandles{});
408 !it.done(); it.Advance()) {
409 if (it.frame()->is_optimized_js()) {
410 Tagged<GcSafeCode> code = it.frame()->GcSafeLookupCode();
411 Tagged<JSFunction> function =
412 static_cast<OptimizedJSFrame*>(it.frame())->function();
413 TraceFoundActivation(isolate, function);
414 bool safe_if_deopt_triggered;
415 if (code->is_maglevved()) {
416 MaglevSafepointEntry safepoint =
417 MaglevSafepointTable::FindEntry(isolate, code, it.frame()->pc());
418 safe_if_deopt_triggered = safepoint.has_deoptimization_index();
419 } else {
420 SafepointEntry safepoint = SafepointTable::FindEntry(
421 isolate, code, it.frame()->maybe_unauthenticated_pc());
422 safe_if_deopt_triggered = safepoint.has_deoptimization_index();
423 }
424
425 // Deopt is checked when we are patching addresses on stack.
426 bool is_builtin_code = code->kind() == CodeKind::BUILTIN;
427 DCHECK(topmost_optimized_code.is_null() || safe_if_deopt_triggered ||
428 is_builtin_code);
429 if (topmost_optimized_code.is_null()) {
430 topmost_optimized_code = code;
431 safe_to_deopt_topmost_optimized_code = safe_if_deopt_triggered;
432 }
433 }
434 }
435#endif
436
437 ActivationsFinder visitor(topmost_optimized_code,
438 safe_to_deopt_topmost_optimized_code);
439 // Iterate over the stack of this thread.
440 visitor.VisitThread(isolate, isolate->thread_local_top());
441 // In addition to iterate over the stack of this thread, we also
442 // need to consider all the other threads as they may also use
443 // the code currently beings deoptimized.
444 isolate->thread_manager()->IterateArchivedThreads(&visitor);
445}

448 RCS_SCOPE(isolate, RuntimeCallCounterId::kDeoptimizeCode);
449 TimerEventScope<TimerEventDeoptimizeCode> timer(isolate);

461 }
462
463 DeoptimizeMarkedCode(isolate);

467void Deoptimizer::DeoptimizeFunction(Tagged<JSFunction> function,
468 LazyDeoptimizeReason reason,
469 Tagged<Code> code) {
470 Isolate* isolate = function->GetIsolate();
471 RCS_SCOPE(isolate, RuntimeCallCounterId::kDeoptimizeCode);
472 TimerEventScope<TimerEventDeoptimizeCode> timer(isolate);
473 TRACE_EVENT0("v8", "V8.DeoptimizeCode");
474 function->ResetIfCodeFlushed(isolate);
475 if (code.is_null()) code = function->code(isolate);
476
477 if (CodeKindCanDeoptimize(code->kind())) {
478 // Mark the code for deoptimization and unlink any functions that also
479 // refer to that code. The code cannot be shared across native contexts,
480 // so we only need to search one.
481 code->SetMarkedForDeoptimization(isolate, reason);
482#ifndef V8_ENABLE_LEAPTIERING_BOOL
483 // The code in the function's optimized code feedback vector slot might
484 // be different from the code on the function - evict it if necessary.
485 function->feedback_vector()->EvictOptimizedCodeMarkedForDeoptimization(
486 isolate, function->shared(), "unlinking code marked for deopt");
487#endif // !V8_ENABLE_LEAPTIERING_BOOL
488
489 DeoptimizeMarkedCode(isolate);
490 }

495 Isolate* isolate, DirectHandle<SharedFunctionInfo> function) {
496 RCS_SCOPE(isolate, RuntimeCallCounterId::kDeoptimizeCode);

512 }
513 }
514 }

516 DeoptimizeMarkedCode(isolate);

638 compiled_optimized_wasm_code_ = code;
639 DCHECK_NOT_NULL(code);
640 CHECK_EQ(code->kind(), wasm::WasmCode::kWasmFunction);
641 wasm::WasmDeoptView deopt_view(code->deopt_data());
642 const wasm::WasmDeoptData& deopt_data = deopt_view.GetDeoptData();
643 DCHECK_NE(deopt_data.translation_array_size, 0);
644 CHECK_GE(from, deopt_data.deopt_exit_start_offset);
645 Address deopt_exit_offset = from - code->instruction_start();
646 // All eager deopt exits are calls "at the end" of the code to the builtin
647 // generated by Generate_DeoptimizationEntry_Eager. These calls have a fixed
648 // size kEagerDeoptExitsSize and the deopt data contains the offset of the
649 // first such call to the beginning of the code, so we can map any PC of
650 // such call to a unique index for this deopt point.
651 deopt_exit_index_ =
652 static_cast<uint32_t>(deopt_exit_offset -
653 deopt_data.deopt_exit_start_offset -
654 kEagerDeoptExitSize) /
655 kEagerDeoptExitSize;
656
657 // Note: The parameter stack slots are not really part of the frame.
658 // However, the deoptimizer needs access to the incoming parameter values
659 // and therefore they need to be included in the FrameDescription. Between
660 // the parameters and the actual frame there are 2 pointers (the caller's pc
661 // and saved stack pointer) that therefore also need to be included. Both
662 // pointers as well as the incoming parameter stack slots are going to be
663 // copied into the outgoing FrameDescription which will "push" them back
664 // onto the stack. (This is consistent with how JS handles this.)
665 const wasm::FunctionSig* sig =
666 code->native_module()->module()->functions[code->index()].sig;
667 int parameter_stack_slots, return_stack_slots;
668 GetWasmStackSlotsCounts(sig, &parameter_stack_slots, &return_stack_slots);
669
670 unsigned input_frame_size = fp_to_sp_delta +
671 parameter_stack_slots * kSystemPointerSize +
672 CommonFrameConstants::kFixedFrameSizeAboveFp;
673 input_ = FrameDescription::Create(input_frame_size, parameter_stack_slots,
674 isolate_);
675 return;
676 }
677#endif
678
679 compiled_code_ = isolate_->heap()->FindCodeForInnerPointer(from);
680 DCHECK(!compiled_code_.is_null());
681 DCHECK(IsCode(compiled_code_));
682
683 DCHECK(IsJSFunction(function));
684 CHECK(CodeKindCanDeoptimize(compiled_code_->kind()));
685 {
686 HandleScope scope(isolate_);
687 PROFILE(isolate_, CodeDeoptEvent(handle(compiled_code_, isolate_), kind,
688 from_, fp_to_sp_delta_));
689 }
690 unsigned size = ComputeInputFrameSize();
691 const int parameter_count = compiled_code_->parameter_count();
692 DCHECK_EQ(
693 parameter_count,
694 function->shared()->internal_formal_parameter_count_with_receiver());
695 input_ = FrameDescription::Create(size, parameter_count, isolate_);
696
697 DCHECK_EQ(deopt_exit_index_, kFixedExitSizeMarker);
698 // Calculate the deopt exit index from return address.
699 DCHECK_GT(kEagerDeoptExitSize, 0);
700 DCHECK_GT(kLazyDeoptExitSize, 0);
701 Tagged<DeoptimizationData> deopt_data =
702 Cast<DeoptimizationData>(compiled_code_->deoptimization_data());
703 Address deopt_start = compiled_code_->instruction_start() +
704 deopt_data->DeoptExitStart().value();
705 int eager_deopt_count = deopt_data->EagerDeoptCount().value();
706 Address lazy_deopt_start =
707 deopt_start + eager_deopt_count * kEagerDeoptExitSize;
708 // The deoptimization exits are sorted so that lazy deopt exits appear after
709 // eager deopts.
710 static_assert(static_cast<int>(DeoptimizeKind::kLazy) ==
711 static_cast<int>(kLastDeoptimizeKind),
712 "lazy deopts are expected to be emitted last");
713 // from_ is the value of the link register after the call to the
714 // deoptimizer, so for the last lazy deopt, from_ points to the first
715 // non-lazy deopt, so we use <=, similarly for the last non-lazy deopt and
716 // the first deopt with resume entry.
717 if (from_ <= lazy_deopt_start) {
718 DCHECK_EQ(kind, DeoptimizeKind::kEager);
719 int offset = static_cast<int>(from_ - kEagerDeoptExitSize - deopt_start);
720 DCHECK_EQ(0, offset % kEagerDeoptExitSize);
721 deopt_exit_index_ = offset / kEagerDeoptExitSize;
722 } else {
723 DCHECK_EQ(kind, DeoptimizeKind::kLazy);
724 int offset =
725 static_cast<int>(from_ - kLazyDeoptExitSize - lazy_deopt_start);
726 DCHECK_EQ(0, offset % kLazyDeoptExitSize);
727 deopt_exit_index_ = eager_deopt_count + (offset / kLazyDeoptExitSize);
728 }
729}

954
955FrameDescription* Deoptimizer::DoComputeWasmLiftoffFrame(
956 TranslatedFrame& frame, wasm::NativeModule* native_module,
957 Tagged<WasmTrustedInstanceData> wasm_trusted_instance, int frame_index,
958 std::stack<intptr_t>& shadow_stack) {
959 // Given inlined frames where function a calls b, b is considered the topmost
960 // because b is on top of the call stack! This is aligned with the names used
961 // by the JS deopt.
962 const bool is_bottommost = frame_index == 0;
963 const bool is_topmost = output_count_ - 1 == frame_index;
964 // Recompile the liftoff (unoptimized) wasm code for the input frame.
965 // TODO(mliedtke): This recompiles every single function even if it never got
966 // optimized and exists as a liftoff variant in the WasmCodeManager as we also
967 // need to compute the deopt information. Can we avoid some of the extra work
968 // here?
969 auto [wasm_code, liftoff_description] = CompileWithLiftoffAndGetDeoptInfo(
970 native_module, frame.wasm_function_index(), frame.bytecode_offset(),
971 is_topmost);
972
973 DCHECK(liftoff_description);
974
975 int parameter_stack_slots, return_stack_slots;
976 const wasm::FunctionSig* sig =
977 native_module->module()->functions[frame.wasm_function_index()].sig;
978 GetWasmStackSlotsCounts(sig, &parameter_stack_slots, &return_stack_slots);
979
980 // Allocate and populate the FrameDescription describing the output frame.
981 const uint32_t output_frame_size = liftoff_description->total_frame_size;
982 const uint32_t total_output_frame_size =
983 output_frame_size + parameter_stack_slots * kSystemPointerSize +
984 CommonFrameConstants::kFixedFrameSizeAboveFp;
985
986 if (verbose_tracing_enabled()) {
987 std::ostringstream outstream;
988 outstream << " Liftoff stack & register state for function index "
989 << frame.wasm_function_index() << ", frame size "
990 << output_frame_size << ", total frame size "
991 << total_output_frame_size << '\n';
992 size_t index = 0;
993 for (const wasm::LiftoffVarState& state : liftoff_description->var_state) {
994 outstream << " " << index++ << ": " << state << '\n';
995 }
996 FILE* file = trace_scope()->file();
997 PrintF(file, "%s", outstream.str().c_str());
998 }
999
1000 FrameDescription* output_frame = FrameDescription::Create(
1001 total_output_frame_size, parameter_stack_slots, isolate());
1002
1003 // Copy the parameter stack slots.
1004 static_assert(CommonFrameConstants::kFixedFrameSizeAboveFp ==
1005 2 * kSystemPointerSize);
1006 uint32_t output_offset = total_output_frame_size;
1007 // Zero out the incoming parameter slots. This will make sure that tagged
1008 // values are safely ignored by the gc.
1009 // Note that zero is clearly not the correct value. Still, liftoff copies
1010 // all parameters into "its own" stack slots at the beginning and always
1011 // uses these slots to restore parameters from the stack.
1012 for (int i = 0; i < parameter_stack_slots; ++i) {
1013 output_offset -= kSystemPointerSize;
1014 output_frame->SetFrameSlot(output_offset, 0);
1015 }
1016
1017 // Calculate top and update previous caller's pc.
1018 Address top = is_bottommost ? caller_frame_top_ - total_output_frame_size
1019 : output_[frame_index - 1]->GetTop() -
1020 total_output_frame_size;
1021 output_frame->SetTop(top);
1022 Address pc = wasm_code->instruction_start() + liftoff_description->pc_offset;
1023 // Sign the PC. Note that for the non-topmost frames the stack pointer at
1024 // which the PC is stored as the "caller pc" / return address depends on the
1025 // amount of parameter stack slots of the callee. To simplify the code, we
1026 // just sign it as if there weren't any parameter stack slots.
1027 // When building up the next frame we can check and "move" the caller PC by
1028 // signing it again with the correct stack pointer.
1029 Address signed_pc = PointerAuthentication::SignAndCheckPC(
1030 isolate(), pc, output_frame->GetTop());
1031 output_frame->SetPc(signed_pc);
1032#ifdef V8_ENABLE_CET_SHADOW_STACK
1033 if (v8_flags.cet_compatible) {
1034 if (is_topmost) {
1035 shadow_stack.push(pc);
1036 } else {
1037 shadow_stack.push(wasm_code->instruction_start() +
1038 liftoff_description->adapt_shadow_stack_pc_offset);
1039 }
1040 }
1041#endif // V8_ENABLE_CET_SHADOW_STACK
1042
1043 // Sign the previous frame's PC.
1044 if (is_bottommost) {
1045 Address old_context =
1046 caller_frame_top_ - input_->parameter_count() * kSystemPointerSize;
1047 Address new_context =
1048 caller_frame_top_ - parameter_stack_slots * kSystemPointerSize;
1049 caller_pc_ = PointerAuthentication::MoveSignedPC(isolate(), caller_pc_,
1050 new_context, old_context);
1051 } else if (parameter_stack_slots != 0) {
1052 // The previous frame's PC is stored at a different stack slot, so we need
1053 // to re-sign the PC for the new context (stack pointer).
1054 FrameDescription* previous_frame = output_[frame_index - 1];
1055 Address pc = previous_frame->GetPc();
1056 Address old_context = previous_frame->GetTop();
1057 Address new_context =
1058 old_context - parameter_stack_slots * kSystemPointerSize;
1059 Address signed_pc = PointerAuthentication::MoveSignedPC(
1060 isolate(), pc, new_context, old_context);
1061 previous_frame->SetPc(signed_pc);
1062 }
1063
1064 // Store the caller PC.
1065 output_offset -= kSystemPointerSize;
1066 output_frame->SetFrameSlot(
1067 output_offset,
1068 is_bottommost ? caller_pc_ : output_[frame_index - 1]->GetPc());
1069 // Store the caller frame pointer.
1070 output_offset -= kSystemPointerSize;
1071 output_frame->SetFrameSlot(
1072 output_offset,
1073 is_bottommost ? caller_fp_ : output_[frame_index - 1]->GetFp());
1074
1075 CHECK_EQ(output_frame_size, output_offset);
1076 int base_offset = output_frame_size;
1077
1078 // Set trusted instance data on output frame.
1079 output_frame->SetFrameSlot(
1080 base_offset - WasmLiftoffFrameConstants::kInstanceDataOffset,
1081 wasm_trusted_instance.ptr());
1082 if (liftoff_description->trusted_instance != no_reg) {
1083 output_frame->SetRegister(liftoff_description->trusted_instance.code(),
1084 wasm_trusted_instance.ptr());
1085 }
1086
1087 DCHECK_GE(translated_state_.frames().size(), 1);
1088 auto liftoff_iter = liftoff_description->var_state.begin();
1089 if constexpr (Is64()) {
1090 // On 32 bit platforms int64s are represented as 2 values on Turbofan.
1091 // Liftoff on the other hand treats them as 1 value (a register pair).
1092 CHECK_EQ(liftoff_description->var_state.size(), frame.GetValueCount());
1093 }
1094
1095 bool int64_lowering_is_low = true;
1096
1097 for (const TranslatedValue& value : frame) {
1098 bool skip_increase_liftoff_iter = false;
1099 switch (liftoff_iter->loc()) {
1100 case wasm::LiftoffVarState::kIntConst:
1101 if (!Is64() && liftoff_iter->kind() == wasm::ValueKind::kI64) {
1102 if (int64_lowering_is_low) skip_increase_liftoff_iter = true;
1103 int64_lowering_is_low = !int64_lowering_is_low;
1104 }
1105 break; // Nothing to be done for constants in liftoff frame.
1106 case wasm::LiftoffVarState::kRegister:
1107 if (liftoff_iter->is_gp_reg()) {
1108 intptr_t reg_value = kZapValue;
1109 switch (value.kind()) {
1110 case TranslatedValue::Kind::kInt32:
1111 // Ensure that the upper half is zeroed out.
1112 reg_value = static_cast<uint32_t>(value.int32_value());
1113 break;
1114 case TranslatedValue::Kind::kTagged:
1115 reg_value = value.raw_literal().ptr();
1116 break;
1117 case TranslatedValue::Kind::kInt64:
1118 reg_value = value.int64_value();
1119 break;
1120 default:
1121 UNIMPLEMENTED();
1122 }
1123 output_frame->SetRegister(liftoff_iter->reg().gp().code(), reg_value);
1124 } else if (liftoff_iter->is_fp_reg()) {
1125 switch (value.kind()) {
1126 case TranslatedValue::Kind::kDouble:
1127 output_frame->SetDoubleRegister(liftoff_iter->reg().fp().code(),
1128 value.double_value());
1129 break;
1130 case TranslatedValue::Kind::kFloat:
1131 // Liftoff doesn't have a concept of floating point registers.
1132 // This is an important distinction as e.g. on arm s1 and d1 are
1133 // two completely distinct registers.
1134 static_assert(std::is_same_v<decltype(liftoff_iter->reg().fp()),
1135 DoubleRegister>);
1136 output_frame->SetDoubleRegister(
1137 liftoff_iter->reg().fp().code(),
1138 Float64::FromBits(value.float_value().get_bits()));
1139 break;
1140 case TranslatedValue::Kind::kSimd128:
1141 output_frame->SetSimd128Register(liftoff_iter->reg().fp().code(),
1142 value.simd_value());
1143 break;
1144 default:
1145 UNIMPLEMENTED();
1146 }
1147 } else if (!Is64() && liftoff_iter->is_gp_reg_pair()) {
1148 intptr_t reg_value = kZapValue;
1149 switch (value.kind()) {
1150 case TranslatedValue::Kind::kInt32:
1151 // Ensure that the upper half is zeroed out.
1152 reg_value = static_cast<uint32_t>(value.int32_value());
1153 break;
1154 case TranslatedValue::Kind::kTagged:
1155 reg_value = value.raw_literal().ptr();
1156 break;
1157 default:
1158 UNREACHABLE();
1159 }
1160 int8_t reg = int64_lowering_is_low
1161 ? liftoff_iter->reg().low_gp().code()
1162 : liftoff_iter->reg().high_gp().code();
1163 output_frame->SetRegister(reg, reg_value);
1164 if (int64_lowering_is_low) skip_increase_liftoff_iter = true;
1165 int64_lowering_is_low = !int64_lowering_is_low;
1166 } else if (!Is64() && liftoff_iter->is_fp_reg_pair()) {
1167 CHECK_EQ(value.kind(), TranslatedValue::Kind::kSimd128);
1168 Simd128 simd_value = value.simd_value();
1169 Address val_ptr = reinterpret_cast<Address>(&simd_value);
1170 output_frame->SetDoubleRegister(
1171 liftoff_iter->reg().low_fp().code(),
1172 Float64::FromBits(base::ReadUnalignedValue<uint64_t>(val_ptr)));
1173 output_frame->SetDoubleRegister(
1174 liftoff_iter->reg().high_fp().code(),
1175 Float64::FromBits(base::ReadUnalignedValue<uint64_t>(
1176 val_ptr + sizeof(double))));
1177 } else {
1178 UNREACHABLE();
1179 }
1180 break;
1181 case wasm::LiftoffVarState::kStack:
1182#ifdef V8_TARGET_BIG_ENDIAN
1183 static constexpr int kLiftoffStackBias = 4;
1184#else
1185 static constexpr int kLiftoffStackBias = 0;
1186#endif
1187 switch (liftoff_iter->kind()) {
1188 case wasm::ValueKind::kI32:
1189 CHECK(value.kind() == TranslatedValue::Kind::kInt32 ||
1190 value.kind() == TranslatedValue::Kind::kUint32);
1191 output_frame->SetLiftoffFrameSlot32(
1192 base_offset - liftoff_iter->offset() + kLiftoffStackBias,
1193 value.int32_value_);
1194 break;
1195 case wasm::ValueKind::kF32:
1196 CHECK_EQ(value.kind(), TranslatedValue::Kind::kFloat);
1197 output_frame->SetLiftoffFrameSlot32(
1198 base_offset - liftoff_iter->offset() + kLiftoffStackBias,
1199 value.float_value().get_bits());
1200 break;
1201 case wasm::ValueKind::kI64:
1202 if constexpr (Is64()) {
1203 CHECK(value.kind() == TranslatedValue::Kind::kInt64 ||
1204 value.kind() == TranslatedValue::Kind::kUint64);
1205 output_frame->SetLiftoffFrameSlot64(
1206 base_offset - liftoff_iter->offset(), value.int64_value_);
1207 } else {
1208 CHECK(value.kind() == TranslatedValue::Kind::kInt32 ||
1209 value.kind() == TranslatedValue::Kind::kUint32);
1210 // TODO(bigendian): Either the offsets or the default for
1211 // int64_lowering_is_low might have to be swapped.
1212 if (int64_lowering_is_low) {
1213 skip_increase_liftoff_iter = true;
1214 output_frame->SetLiftoffFrameSlot32(
1215 base_offset - liftoff_iter->offset(), value.int32_value_);
1216 } else {
1217 output_frame->SetLiftoffFrameSlot32(
1218 base_offset - liftoff_iter->offset() + sizeof(int32_t),
1219 value.int32_value_);
1220 }
1221 int64_lowering_is_low = !int64_lowering_is_low;
1222 }
1223 break;
1224 case wasm::ValueKind::kS128: {
1225 int64x2 values = value.simd_value().to_i64x2();
1226 const int offset = base_offset - liftoff_iter->offset();
1227 output_frame->SetLiftoffFrameSlot64(offset, values.val[0]);
1228 output_frame->SetLiftoffFrameSlot64(offset + sizeof(int64_t),
1229 values.val[1]);
1230 break;
1231 }
1232 case wasm::ValueKind::kF64:
1233 CHECK_EQ(value.kind(), TranslatedValue::Kind::kDouble);
1234 output_frame->SetLiftoffFrameSlot64(
1235 base_offset - liftoff_iter->offset(),
1236 value.double_value().get_bits());
1237 break;
1238 case wasm::ValueKind::kRef:
1239 case wasm::ValueKind::kRefNull:
1240 CHECK_EQ(value.kind(), TranslatedValue::Kind::kTagged);
1241 output_frame->SetLiftoffFrameSlotPointer(
1242 base_offset - liftoff_iter->offset(), value.raw_literal_.ptr());
1243 break;
1244 default:
1245 UNIMPLEMENTED();
1246 }
1247 break;
1248 }
1249 DCHECK_IMPLIES(skip_increase_liftoff_iter, !Is64());
1250 if (!skip_increase_liftoff_iter) {
1251 ++liftoff_iter;
1252 }
1253 }
1254
1255 // Store frame kind.
1256 uint32_t frame_type_offset =
1257 base_offset + WasmLiftoffFrameConstants::kFrameTypeOffset;
1258 output_frame->SetFrameSlot(frame_type_offset,
1259 StackFrame::TypeToMarker(StackFrame::WASM));
1260 // Store feedback vector in stack slot.
1261 Tagged<FixedArray> module_feedback =
1262 wasm_trusted_instance->feedback_vectors();
1263 uint32_t feedback_offset =
1264 base_offset - WasmLiftoffFrameConstants::kFeedbackVectorOffset;
1265 uint32_t fct_feedback_index = wasm::declared_function_index(
1266 native_module->module(), frame.wasm_function_index());
1267 CHECK_LT(fct_feedback_index, module_feedback->length());
1268 Tagged<Object> feedback_vector = module_feedback->get(fct_feedback_index);
1269 if (IsSmi(feedback_vector)) {
1270 if (verbose_tracing_enabled()) {
1271 PrintF(trace_scope()->file(),
1272 "Deopt with uninitialized feedback vector for function %s [%d]\n",
1273 wasm_code->DebugName().c_str(), frame.wasm_function_index());
1274 }
1275 // Not having a feedback vector can happen with multiple instantiations of
1276 // the same module as the type feedback is separate per instance but the
1277 // code is shared (even cross-isolate).
1278 // Note that we cannot allocate the feedback vector here. Instead, store
1279 // the function index, so that the feedback vector can be populated by the
1280 // deopt finish builtin called from Liftoff.
1281 output_frame->SetFrameSlot(feedback_offset,
1282 Smi::FromInt(fct_feedback_index).ptr());
1283 } else {
1284 output_frame->SetFrameSlot(feedback_offset, feedback_vector.ptr());
1285 }
1286
1287 // Instead of a builtin continuation for wasm the deopt builtin will
1288 // call a c function to destroy the Deoptimizer object and then directly
1289 // return to the liftoff code.
1290 output_frame->SetContinuation(0);
1291
1292 const intptr_t fp_value = top + output_frame_size;
1293 output_frame->SetFp(fp_value);
1294 Register fp_reg = JavaScriptFrame::fp_register();
1295 output_frame->SetRegister(fp_reg.code(), fp_value);
1296 output_frame->SetRegister(kRootRegister.code(), isolate()->isolate_root());
1297#ifdef V8_COMPRESS_POINTERS
1298 output_frame->SetRegister(kPtrComprCageBaseRegister.code(),
1299 isolate()->cage_base());
1300#endif
1301
1302 return output_frame;
1303}

1305// Build up the output frames for a wasm deopt. This creates the
1306// FrameDescription objects representing the output frames to be "materialized"
1307// on the stack.
1308void Deoptimizer::DoComputeOutputFramesWasmImpl() {
1309 CHECK(v8_flags.wasm_deopt);
1310 base::ElapsedTimer timer;
1311 // Lookup the deopt info for the input frame.
1312 wasm::WasmCode* code = compiled_optimized_wasm_code_;
1313 DCHECK_NOT_NULL(code);
1314 DCHECK_EQ(code->kind(), wasm::WasmCode::kWasmFunction);
1315 wasm::WasmDeoptView deopt_view(code->deopt_data());
1316 wasm::WasmDeoptEntry deopt_entry =
1317 deopt_view.GetDeoptEntry(deopt_exit_index_);
1318
1319 if (tracing_enabled()) {
1320 timer.Start();
1321 FILE* file = trace_scope()->file();
1322 PrintF(file,
1323 "[bailout (kind: %s, reason: %s, type: Wasm): begin. deoptimizing "
1324 "%s, function index %d, bytecode offset %d, deopt exit %d, FP to SP "
1325 "delta %d, "
1326 "pc " V8PRIxPTR_FMT "]\n",
1327 MessageFor(deopt_kind_),
1328 DeoptimizeReasonToString(DeoptimizeReason::kWrongCallTarget),
1329 code->DebugName().c_str(), code->index(),
1330 deopt_entry.bytecode_offset.ToInt(), deopt_entry.translation_index,
1331 fp_to_sp_delta_, PointerAuthentication::StripPAC(from_));
1332 }
1333
1334 base::Vector<const uint8_t> off_heap_translations =
1335 deopt_view.GetTranslationsArray();
1336
1337 DeoptTranslationIterator state_iterator(off_heap_translations,
1338 deopt_entry.translation_index);
1339 wasm::NativeModule* native_module = code->native_module();
1340 int parameter_count = static_cast<int>(
1341 native_module->module()->functions[code->index()].sig->parameter_count());
1342 DeoptimizationLiteralProvider literals(
1343 deopt_view.BuildDeoptimizationLiteralArray());
1344
1345 Register fp_reg = JavaScriptFrame::fp_register();
1346 stack_fp_ = input_->GetRegister(fp_reg.code());
1347 Address fp_address = input_->GetFramePointerAddress();
1348 caller_fp_ = Memory<intptr_t>(fp_address);
1349 caller_pc_ =
1350 Memory<intptr_t>(fp_address + CommonFrameConstants::kCallerPCOffset);
1351 caller_frame_top_ = stack_fp_ + CommonFrameConstants::kFixedFrameSizeAboveFp +
1352 input_->parameter_count() * kSystemPointerSize;
1353
1354 FILE* trace_file =
1355 verbose_tracing_enabled() ? trace_scope()->file() : nullptr;
1356 translated_state_.Init(isolate_, input_->GetFramePointerAddress(), stack_fp_,
1357 &state_iterator, {}, literals,
1358 input_->GetRegisterValues(), trace_file,
1359 parameter_count, parameter_count);
1360
1361 const size_t output_frames = translated_state_.frames().size();
1362 CHECK_GT(output_frames, 0);
1363 output_count_ = static_cast<int>(output_frames);
1364 output_ = new FrameDescription* [output_frames] {};
1365
1366 // The top output function *should* be the same as the optimized function
1367 // with the deopt. However, this is not the case in case of inlined return
1368 // calls. The optimized function still needs to be invalidated.
1369 if (translated_state_.frames()[0].wasm_function_index() !=
1370 compiled_optimized_wasm_code_->index()) {
1371 CompileWithLiftoffAndGetDeoptInfo(native_module,
1372 compiled_optimized_wasm_code_->index(),
1373 deopt_entry.bytecode_offset, false);
1374 }
1375
1376 // Read the trusted instance data from the input frame.
1377 Tagged<WasmTrustedInstanceData> wasm_trusted_instance =
1378 Cast<WasmTrustedInstanceData>((Tagged<Object>(input_->GetFrameSlot(
1379 input_->GetFrameSize() -
1380 (2 + input_->parameter_count()) * kSystemPointerSize -
1381 WasmLiftoffFrameConstants::kInstanceDataOffset))));
1382
1383 std::stack<intptr_t> shadow_stack;
1384 for (int i = 0; i < output_count_; ++i) {
1385 TranslatedFrame& frame = translated_state_.frames()[i];
1386 output_[i] = DoComputeWasmLiftoffFrame(
1387 frame, native_module, wasm_trusted_instance, i, shadow_stack);
1388 }
1389
1390#ifdef V8_ENABLE_CET_SHADOW_STACK
1391 if (v8_flags.cet_compatible) {
1392 CHECK_EQ(shadow_stack_count_, 0);
1393 shadow_stack_ = new intptr_t[shadow_stack.size()];
1394 while (!shadow_stack.empty()) {
1395 shadow_stack_[shadow_stack_count_++] = shadow_stack.top();
1396 shadow_stack.pop();
1397 }
1398 CHECK_EQ(shadow_stack_count_, output_count_);
1399 }
1400#endif // V8_ENABLE_CET_SHADOW_STACK
1401
1402 {
1403 // Mark the cached feedback result produced by the
1404 // TransitiveTypeFeedbackProcessor as outdated.
1405 // This is required to prevent deopt loops as new feedback is ignored
1406 // otherwise.
1407 wasm::TypeFeedbackStorage& feedback =
1408 native_module->module()->type_feedback;
1409 base::SharedMutexGuard<base::kExclusive> mutex_guard(&feedback.mutex);
1410 for (const TranslatedFrame& frame : translated_state_) {
1411 int index = frame.wasm_function_index();
1412 auto iter = feedback.feedback_for_function.find(index);
1413 if (iter != feedback.feedback_for_function.end()) {
1414 iter->second.needs_reprocessing_after_deopt = true;
1415 }
1416 }
1417 // Reset tierup priority. This is important as the tierup trigger will only
1418 // be taken into account if the tierup_priority is a power of two (to
1419 // prevent a hot function being enqueued too many times into the compilation
1420 // queue.)
1421 feedback.feedback_for_function[code->index()].tierup_priority = 0;
1422 // Add sample for how many times this function was deopted.
1423 isolate()->counters()->wasm_deopts_per_function()->AddSample(
1424 ++feedback.deopt_count_for_function[code->index()]);
1425 }
1426
1427 // Reset tiering budget of the function that triggered the deopt.
1428 int declared_func_index =
1429 wasm::declared_function_index(native_module->module(), code->index());
1430 wasm_trusted_instance->tiering_budget_array()[declared_func_index].store(
1431 v8_flags.wasm_tiering_budget, std::memory_order_relaxed);
1432
1433 isolate()->counters()->wasm_deopts_executed()->AddSample(
1434 wasm::GetWasmEngine()->IncrementDeoptsExecutedCount());
1435
1436 if (verbose_tracing_enabled()) {
1437 TraceDeoptEnd(timer.Elapsed().InMillisecondsF());
1438 }
1439}

1489// We rely on this function not causing a GC. It is called from generated code
1490// without having a real stack frame in place.
1491void Deoptimizer::DoComputeOutputFrames() {
1492 // When we call this function, the return address of the previous frame has
1493 // been removed from the stack by the DeoptimizationEntry builtin, so the
1494 // stack is not iterable by the StackFrameIteratorForProfiler.
1495#if V8_TARGET_ARCH_STORES_RETURN_ADDRESS_ON_STACK
1496 DCHECK_EQ(0, isolate()->isolate_data()->stack_is_iterable());
1497#endif
1498 base::ElapsedTimer timer;
1499
1500#if V8_ENABLE_WEBASSEMBLY
1501 if (v8_flags.wasm_deopt && function_.is_null()) {
1502 trap_handler::ClearThreadInWasm();
1503 DoComputeOutputFramesWasmImpl();
1504 trap_handler::SetThreadInWasm();
1505 return;
1506 }
1507#endif
1508
1509 // Determine basic deoptimization information. The optimized frame is
1510 // described by the input data.
1511 Tagged<DeoptimizationData> input_data =
1512 Cast<DeoptimizationData>(compiled_code_->deoptimization_data());
1513
1514 {
1515 // Read caller's PC, caller's FP and caller's constant pool values
1516 // from input frame. Compute caller's frame top address.
1517
1518 Register fp_reg = JavaScriptFrame::fp_register();
1519 stack_fp_ = input_->GetRegister(fp_reg.code());
1520
1521 caller_frame_top_ = stack_fp_ + ComputeInputFrameAboveFpFixedSize();
1522
1523 Address fp_address = input_->GetFramePointerAddress();
1524 caller_fp_ = Memory<intptr_t>(fp_address);
1525 caller_pc_ =
1526 Memory<intptr_t>(fp_address + CommonFrameConstants::kCallerPCOffset);
1527 actual_argument_count_ = static_cast<int>(
1528 Memory<intptr_t>(fp_address + StandardFrameConstants::kArgCOffset));
1529
1530 if (V8_EMBEDDED_CONSTANT_POOL_BOOL) {
1531 caller_constant_pool_ = Memory<intptr_t>(
1532 fp_address + CommonFrameConstants::kConstantPoolOffset);
1533 }
1534 }
1535
1536 StackGuard* const stack_guard = isolate()->stack_guard();
1537 CHECK_GT(static_cast<uintptr_t>(caller_frame_top_),
1538 stack_guard->real_jslimit());
1539
1540 BytecodeOffset bytecode_offset =
1541 input_data->GetBytecodeOffsetOrBuiltinContinuationId(deopt_exit_index_);
1542 auto translations = input_data->FrameTranslation();
1543 unsigned translation_index =
1544 input_data->TranslationIndex(deopt_exit_index_).value();
1545
1546 if (tracing_enabled()) {
1547 timer.Start();
1548 TraceDeoptBegin(input_data->OptimizationId().value(), bytecode_offset);
1549 }
1550
1551 FILE* trace_file =
1552 verbose_tracing_enabled() ? trace_scope()->file() : nullptr;
1553 DeoptimizationFrameTranslation::Iterator state_iterator(translations,
1554 translation_index);
1555 DeoptimizationLiteralProvider literals(input_data->LiteralArray());
1556 translated_state_.Init(isolate_, input_->GetFramePointerAddress(), stack_fp_,
1557 &state_iterator, input_data->ProtectedLiteralArray(),
1558 literals, input_->GetRegisterValues(), trace_file,
1559 compiled_code_->parameter_count_without_receiver(),
1560 actual_argument_count_ - kJSArgcReceiverSlots);
1561
1562 bytecode_offset_in_outermost_frame_ =
1563 translated_state_.frames()[0].bytecode_offset();
1564
1565 // Do the input frame to output frame(s) translation.
1566 size_t count = translated_state_.frames().size();
1567 if (is_restart_frame()) {
1568 // If the debugger requested to restart a particular frame, only materialize
1569 // up to that frame.
1570 count = restart_frame_index_ + 1;
1571 } else if (deoptimizing_throw_) {
1572 // If we are supposed to go to the catch handler, find the catching frame
1573 // for the catch and make sure we only deoptimize up to that frame.
1574 size_t catch_handler_frame_index = count;
1575 for (size_t i = count; i-- > 0;) {
1576 catch_handler_pc_offset_ = LookupCatchHandler(
1577 isolate(), &(translated_state_.frames()[i]), &catch_handler_data_);
1578 if (catch_handler_pc_offset_ >= 0) {
1579 catch_handler_frame_index = i;
1580 break;
1581 }
1582 }
1583 CHECK_LT(catch_handler_frame_index, count);
1584 count = catch_handler_frame_index + 1;
1585 }
1586
1587 DCHECK_NULL(output_);
1588 output_ = new FrameDescription* [count] {};
1589 output_count_ = static_cast<int>(count);
1590
1591 // Translate each output frame.
1592 int frame_index = 0;
1593 size_t total_output_frame_size = 0;
1594 for (size_t i = 0; i < count; ++i, ++frame_index) {
1595 TranslatedFrame* translated_frame = &(translated_state_.frames()[i]);
1596 const bool handle_exception = deoptimizing_throw_ && i == count - 1;
1597 switch (translated_frame->kind()) {
1598 case TranslatedFrame::kUnoptimizedFunction:
1599 DoComputeUnoptimizedFrame(translated_frame, frame_index,
1600 handle_exception);
1601 break;
1602 case TranslatedFrame::kInlinedExtraArguments:
1603 DoComputeInlinedExtraArguments(translated_frame, frame_index);
1604 break;
1605 case TranslatedFrame::kConstructCreateStub:
1606 DoComputeConstructCreateStubFrame(translated_frame, frame_index);
1607 break;
1608 case TranslatedFrame::kConstructInvokeStub:
1609 DoComputeConstructInvokeStubFrame(translated_frame, frame_index);
1610 break;
1611 case TranslatedFrame::kBuiltinContinuation:
1612#if V8_ENABLE_WEBASSEMBLY
1613 case TranslatedFrame::kJSToWasmBuiltinContinuation:
1614#endif // V8_ENABLE_WEBASSEMBLY
1615 DoComputeBuiltinContinuation(translated_frame, frame_index,
1616 BuiltinContinuationMode::STUB);
1617 break;
1618 case TranslatedFrame::kJavaScriptBuiltinContinuation:
1619 DoComputeBuiltinContinuation(translated_frame, frame_index,
1620 BuiltinContinuationMode::JAVASCRIPT);
1621 break;
1622 case TranslatedFrame::kJavaScriptBuiltinContinuationWithCatch:
1623 DoComputeBuiltinContinuation(
1624 translated_frame, frame_index,
1625 handle_exception
1626 ? BuiltinContinuationMode::JAVASCRIPT_HANDLE_EXCEPTION
1627 : BuiltinContinuationMode::JAVASCRIPT_WITH_CATCH);
1628 break;
1629#if V8_ENABLE_WEBASSEMBLY
1630 case TranslatedFrame::kWasmInlinedIntoJS:
1631 FATAL("inlined wasm frames may not appear in JS deopts");
1632 case TranslatedFrame::kLiftoffFunction:
1633 FATAL("wasm liftoff frames may not appear in JS deopts");
1634#endif
1635 case TranslatedFrame::kInvalid:
1636 FATAL("invalid frame");
1637 }
1638 total_output_frame_size += output_[frame_index]->GetFrameSize();
1639 }
1640
1641 FrameDescription* topmost = output_[count - 1];
1642 topmost->GetRegisterValues()->SetRegister(kRootRegister.code(),
1643 isolate()->isolate_root());
1644#ifdef V8_COMPRESS_POINTERS
1645 topmost->GetRegisterValues()->SetRegister(kPtrComprCageBaseRegister.code(),
1646 isolate()->cage_base());
1647#endif
1648
1649#ifdef V8_ENABLE_CET_SHADOW_STACK
1650 if (v8_flags.cet_compatible) {
1651 CHECK_EQ(shadow_stack_count_, 0);
1652 shadow_stack_ = new intptr_t[count + 1];
1653
1654 // We should jump to the continuation through AdaptShadowStack to avoid
1655 // security exception.
1656 // Clear the continuation so that DeoptimizationEntry does not push the
1657 // address onto the stack, and push it to the shadow stack instead.
1658 if (output_[count - 1]->GetContinuation()) {
1659 shadow_stack_[shadow_stack_count_++] =
1660 output_[count - 1]->GetContinuation();
1661 output_[count - 1]->SetContinuation(0);
1662 }
1663
1664 // Add topmost frame's pc to the shadow stack.
1665 shadow_stack_[shadow_stack_count_++] =
1666 output_[count - 1]->GetPc() -
1667 Deoptimizer::kAdaptShadowStackOffsetToSubtract;
1668
1669 // Add return addresses to the shadow stack, except for the bottommost.
1670 // The bottommost frame's return address already exists in the shadow stack.
1671 for (int i = static_cast<int>(count) - 1; i > 0; i--) {
1672 if (!output_[i]->HasCallerPc()) continue;
1673 shadow_stack_[shadow_stack_count_++] =
1674 output_[i]->GetCallerPc() -
1675 Deoptimizer::kAdaptShadowStackOffsetToSubtract;
1676 }
1677 }
1678#endif // V8_ENABLE_CET_SHADOW_STACK
1679
1680 // Don't reset the tiering state for OSR code since we might reuse OSR code
1681 // after deopt, and we still want to tier up to non-OSR code even if OSR code
1682 // deoptimized.
1683 bool osr_early_exit = Deoptimizer::GetDeoptInfo().deopt_reason ==
1684 DeoptimizeReason::kOSREarlyExit;
1685 // TODO(saelo): We have to use full pointer comparisons here while not all
1686 // Code objects have been migrated into trusted space.
1687 static_assert(!kAllCodeObjectsLiveInTrustedSpace);
1688 if (IsJSFunction(function_) &&
1689 (compiled_code_->osr_offset().IsNone()
1690 ? function_->code(isolate()).SafeEquals(compiled_code_)
1691 : (!osr_early_exit &&
1692 DeoptExitIsInsideOsrLoop(isolate(), function_,
1693 bytecode_offset_in_outermost_frame_,
1694 compiled_code_->osr_offset())))) {
1695 if (v8_flags.profile_guided_optimization &&
1696 function_->shared()->cached_tiering_decision() !=
1697 CachedTieringDecision::kDelayMaglev) {
1698 if (DeoptimizedMaglevvedCodeEarly(isolate(), function_, compiled_code_)) {
1699 function_->shared()->set_cached_tiering_decision(
1700 CachedTieringDecision::kDelayMaglev);
1701 } else {
1702 function_->shared()->set_cached_tiering_decision(
1703 CachedTieringDecision::kNormal);
1704 }
1705 }
1706 function_->ResetTieringRequests(isolate_);
1707 // This allows us to quickly re-spawn a new compilation request even if
1708 // there is already one running. In particular it helps to squeeze in a
1709 // maglev compilation when there is a long running turbofan one that was
1710 // started right before the deopt.
1711 function_->SetTieringInProgress(false);
1712 function_->SetInterruptBudget(isolate_, BudgetModification::kReset,
1713 CodeKind::INTERPRETED_FUNCTION);
1714 function_->feedback_vector()->set_was_once_deoptimized();
1715 }
1716
1717 // Print some helpful diagnostic information.
1718 if (verbose_tracing_enabled()) {
1719 TraceDeoptEnd(timer.Elapsed().InMillisecondsF());
1720 }
1721
1722 // The following invariant is fairly tricky to guarantee, since the size of
1723 // an optimized frame and its deoptimized counterparts usually differs. We
1724 // thus need to consider the case in which deoptimized frames are larger than
1725 // the optimized frame in stack checks in optimized code. We do this by
1726 // applying an offset to stack checks (see kArchStackPointerGreaterThan in the
1727 // code generator).
1728 // Note that we explicitly allow deopts to exceed the limit by a certain
1729 // number of slack bytes.
1730 CHECK_GT(
1731 static_cast<uintptr_t>(caller_frame_top_) - total_output_frame_size,
1732 stack_guard->real_jslimit() - kStackLimitSlackForDeoptimizationInBytes);
1733}
1734
1735// static
1736bool Deoptimizer::DeoptExitIsInsideOsrLoop(Isolate* isolate,
1737 Tagged<JSFunction> function,
1738 BytecodeOffset deopt_exit_offset,
1739 BytecodeOffset osr_offset) {
1740 DisallowGarbageCollection no_gc;
1741 HandleScope scope(isolate);
1742 DCHECK(!deopt_exit_offset.IsNone());
1743 DCHECK(!osr_offset.IsNone());
1744
1745 Handle<BytecodeArray> bytecode_array(
1746 function->shared()->GetBytecodeArray(isolate), isolate);
1747 DCHECK(interpreter::BytecodeArrayIterator::IsValidOffset(
1748 bytecode_array, deopt_exit_offset.ToInt()));
1749
1750 interpreter::BytecodeArrayIterator it(bytecode_array, osr_offset.ToInt());
1751 CHECK(it.CurrentBytecodeIsValidOSREntry());
1752
1753 for (; !it.done(); it.Advance()) {
1754 const int current_offset = it.current_offset();
1755 // If we've reached the deopt exit, it's contained in the current loop
1756 // (this is covered by IsInRange below, but this check lets us avoid
1757 // useless iteration).
1758 if (current_offset == deopt_exit_offset.ToInt()) return true;
1759 // We're only interested in loop ranges.
1760 if (it.current_bytecode() != interpreter::Bytecode::kJumpLoop) continue;
1761 // Is the deopt exit contained in the current loop?
1762 if (base::IsInRange(deopt_exit_offset.ToInt(), it.GetJumpTargetOffset(),
1763 current_offset)) {
1764 return true;
1765 }
1766 // We've reached nesting level 0, i.e. the current JumpLoop concludes a
1767 // top-level loop.
1768 const int loop_nesting_level = it.GetImmediateOperand(1);
1769 if (loop_nesting_level == 0) return false;
1770 }
1771
1772 UNREACHABLE();

1786void Deoptimizer::DoComputeUnoptimizedFrame(TranslatedFrame* translated_frame,
1787 int frame_index,
1788 bool goto_catch_handler) {
1789 Tagged<BytecodeArray> bytecode_array = translated_frame->raw_bytecode_array();
1790 TranslatedFrame::iterator value_iterator = translated_frame->begin();
1791 const bool is_bottommost = (0 == frame_index);
1792 const bool is_topmost = (output_count_ - 1 == frame_index);
1793
1794 const int real_bytecode_offset = translated_frame->bytecode_offset().ToInt();
1795 const int bytecode_offset =
1796 goto_catch_handler ? catch_handler_pc_offset_ : real_bytecode_offset;
1797
1798 const int parameters_count = bytecode_array->parameter_count();
1799
1800 // If this is the bottom most frame or the previous frame was the inlined
1801 // extra arguments frame, then we already have extra arguments in the stack
1802 // (including any extra padding). Therefore we should not try to add any
1803 // padding.
1804 bool should_pad_arguments =
1805 !is_bottommost && (translated_state_.frames()[frame_index - 1]).kind() !=
1806 TranslatedFrame::kInlinedExtraArguments;
1807
1808 const int locals_count = translated_frame->height();
1809 UnoptimizedFrameInfo frame_info = UnoptimizedFrameInfo::Precise(
1810 parameters_count, locals_count, is_topmost, should_pad_arguments);
1811 const uint32_t output_frame_size = frame_info.frame_size_in_bytes();
1812
1813 TranslatedFrame::iterator function_iterator = value_iterator++;
1814
1815 std::optional<Tagged<DebugInfo>> debug_info =
1816 translated_frame->raw_shared_info()->TryGetDebugInfo(isolate());
1817 if (debug_info.has_value() && debug_info.value()->HasBreakInfo()) {
1818 // TODO(leszeks): Validate this bytecode.
1819 bytecode_array = debug_info.value()->DebugBytecodeArray(isolate());
1820 }
1821
1822 // Allocate and store the output frame description.
1823 FrameDescription* output_frame =
1824 FrameDescription::Create(output_frame_size, parameters_count, isolate());
1825 FrameWriter frame_writer(this, output_frame, verbose_trace_scope());
1826
1827 CHECK(frame_index >= 0 && frame_index < output_count_);
1828 CHECK_NULL(output_[frame_index]);
1829 output_[frame_index] = output_frame;
1830
1831 // Compute this frame's PC and state.
1832 // For interpreted frames, the PC will be a special builtin that
1833 // continues the bytecode dispatch. Note that non-topmost and lazy-style
1834 // bailout handlers also advance the bytecode offset before dispatch, hence
1835 // simulating what normal handlers do upon completion of the operation.
1836 // For baseline frames, the PC will be a builtin to convert the interpreter
1837 // frame to a baseline frame before continuing execution of baseline code.
1838 // We can't directly continue into baseline code, because of CFI.
1839 Builtins* builtins = isolate_->builtins();
1840 const bool advance_bc =
1841 (!is_topmost || (deopt_kind_ == DeoptimizeKind::kLazy)) &&
1842 !goto_catch_handler;
1843 const bool restart_frame = goto_catch_handler && is_restart_frame();
1844 Tagged<Code> dispatch_builtin =
1845 builtins->code(DispatchBuiltinFor(advance_bc, restart_frame));
1846
1847 if (verbose_tracing_enabled()) {
1848 PrintF(trace_scope()->file(), " translating interpreted frame ");
1849 std::unique_ptr<char[]> name =
1850 translated_frame->raw_shared_info()->DebugNameCStr();
1851 PrintF(trace_scope()->file(), "%s", name.get());
1852 PrintF(trace_scope()->file(), " => bytecode_offset=%d, ",
1853 real_bytecode_offset);
1854 PrintF(trace_scope()->file(), "variable_frame_size=%d, frame_size=%d%s\n",
1855 frame_info.frame_size_in_bytes_without_fixed(), output_frame_size,
1856 goto_catch_handler ? " (throw)" : "");
1857 }
1858
1859 // The top address of the frame is computed from the previous frame's top and
1860 // this frame's size.
1861 const intptr_t top_address =
1862 is_bottommost ? caller_frame_top_ - output_frame_size
1863 : output_[frame_index - 1]->GetTop() - output_frame_size;
1864 output_frame->SetTop(top_address);
1865
1866 // Compute the incoming parameter translation.
1867 ReadOnlyRoots roots(isolate());
1868 if (should_pad_arguments) {
1869 for (int i = 0; i < ArgumentPaddingSlots(parameters_count); ++i) {
1870 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
1871 }
1872 }
1873
1874 if (verbose_tracing_enabled() && is_bottommost &&
1875 actual_argument_count_ > parameters_count) {
1876 PrintF(trace_scope_->file(),
1877 " -- %d extra argument(s) already in the stack --\n",
1878 actual_argument_count_ - parameters_count);
1879 }
1880 frame_writer.PushStackJSArguments(value_iterator, parameters_count);
1881
1882 DCHECK_EQ(output_frame->GetLastArgumentSlotOffset(should_pad_arguments),
1883 frame_writer.top_offset());
1884 if (verbose_tracing_enabled()) {
1885 PrintF(trace_scope()->file(), " -------------------------\n");
1886 }
1887
1888 // There are no translation commands for the caller's pc and fp, the
1889 // context, the function and the bytecode offset. Synthesize
1890 // their values and set them up
1891 // explicitly.
1892 //
1893 // The caller's pc for the bottommost output frame is the same as in the
1894 // input frame. For all subsequent output frames, it can be read from the
1895 // previous one. This frame's pc can be computed from the non-optimized
1896 // function code and bytecode offset of the bailout.
1897 if (is_bottommost) {
1898 frame_writer.PushBottommostCallerPc(caller_pc_);
1899 } else {
1900 frame_writer.PushApprovedCallerPc(output_[frame_index - 1]->GetPc());
1901 }
1902
1903 // The caller's frame pointer for the bottommost output frame is the same
1904 // as in the input frame. For all subsequent output frames, it can be
1905 // read from the previous one. Also compute and set this frame's frame
1906 // pointer.
1907 const intptr_t caller_fp =
1908 is_bottommost ? caller_fp_ : output_[frame_index - 1]->GetFp();
1909 frame_writer.PushCallerFp(caller_fp);
1910
1911 const intptr_t fp_value = top_address + frame_writer.top_offset();
1912 output_frame->SetFp(fp_value);
1913 if (is_topmost) {
1914 Register fp_reg = UnoptimizedJSFrame::fp_register();
1915 output_frame->SetRegister(fp_reg.code(), fp_value);
1916 }
1917
1918 if (V8_EMBEDDED_CONSTANT_POOL_BOOL) {
1919 // For the bottommost output frame the constant pool pointer can be gotten
1920 // from the input frame. For subsequent output frames, it can be read from
1921 // the previous frame.
1922 const intptr_t caller_cp =
1923 is_bottommost ? caller_constant_pool_
1924 : output_[frame_index - 1]->GetConstantPool();
1925 frame_writer.PushCallerConstantPool(caller_cp);
1926 }
1927
1928 // For the bottommost output frame the context can be gotten from the input
1929 // frame. For all subsequent output frames it can be gotten from the function
1930 // so long as we don't inline functions that need local contexts.
1931
1932 // When deoptimizing into a catch block, we need to take the context
1933 // from a register that was specified in the handler table.
1934 TranslatedFrame::iterator context_pos = value_iterator++;
1935 if (goto_catch_handler) {
1936 // Skip to the translated value of the register specified
1937 // in the handler table.
1938 for (int i = 0; i < catch_handler_data_ + 1; ++i) {
1939 context_pos++;
1940 }
1941 }
1942 // Read the context from the translations.
1943 frame_writer.PushTranslatedValue(context_pos, "context");
1944
1945 // The function was mentioned explicitly in the BEGIN_FRAME.
1946 frame_writer.PushTranslatedValue(function_iterator, "function");
1947
1948 // Actual argument count.
1949 int argc;
1950 if (is_bottommost) {
1951 argc = actual_argument_count_;
1952 } else {
1953 TranslatedFrame::Kind previous_frame_kind =
1954 (translated_state_.frames()[frame_index - 1]).kind();
1955 argc = previous_frame_kind == TranslatedFrame::kInlinedExtraArguments
1956 ? output_[frame_index - 1]->parameter_count()
1957 : parameters_count;
1958 }
1959 frame_writer.PushRawValue(argc, "actual argument count\n");
1960
1961 // Set the bytecode array pointer.
1962 frame_writer.PushRawObject(bytecode_array, "bytecode array\n");
1963
1964 // The bytecode offset was mentioned explicitly in the BEGIN_FRAME.
1965 const int raw_bytecode_offset =
1966 BytecodeArray::kHeaderSize - kHeapObjectTag + bytecode_offset;
1967 Tagged<Smi> smi_bytecode_offset = Smi::FromInt(raw_bytecode_offset);
1968 frame_writer.PushRawObject(smi_bytecode_offset, "bytecode offset\n");
1969
1970 // We need to materialize the closure before getting the feedback vector.
1971 frame_writer.PushFeedbackVectorForMaterialization(function_iterator);
1972
1973 if (verbose_tracing_enabled()) {
1974 PrintF(trace_scope()->file(), " -------------------------\n");
1975 }
1976
1977 // Translate the rest of the interpreter registers in the frame.
1978 // The return_value_offset is counted from the top. Here, we compute the
1979 // register index (counted from the start).
1980 const int return_value_first_reg =
1981 locals_count - translated_frame->return_value_offset();
1982 const int return_value_count = translated_frame->return_value_count();
1983 for (int i = 0; i < locals_count; ++i, ++value_iterator) {
1984 // Ensure we write the return value if we have one and we are returning
1985 // normally to a lazy deopt point.
1986 if (is_topmost && !goto_catch_handler &&
1987 deopt_kind_ == DeoptimizeKind::kLazy && i >= return_value_first_reg &&
1988 i < return_value_first_reg + return_value_count) {
1989 const int return_index = i - return_value_first_reg;
1990 if (return_index == 0) {
1991 frame_writer.PushRawValue(input_->GetRegister(kReturnRegister0.code()),
1992 "return value 0\n");
1993 // We do not handle the situation when one return value should go into
1994 // the accumulator and another one into an ordinary register. Since
1995 // the interpreter should never create such situation, just assert
1996 // this does not happen.
1997 CHECK_LE(return_value_first_reg + return_value_count, locals_count);
1998 } else {
1999 CHECK_EQ(return_index, 1);
2000 frame_writer.PushRawValue(input_->GetRegister(kReturnRegister1.code()),
2001 "return value 1\n");
2002 }
2003 } else {
2004 // This is not return value, just write the value from the translations.
2005 frame_writer.PushTranslatedValue(value_iterator, "stack parameter");
2006 }
2007 }
2008
2009 uint32_t register_slots_written = static_cast<uint32_t>(locals_count);
2010 DCHECK_LE(register_slots_written, frame_info.register_stack_slot_count());
2011 // Some architectures must pad the stack frame with extra stack slots
2012 // to ensure the stack frame is aligned. Do this now.
2013 while (register_slots_written < frame_info.register_stack_slot_count()) {
2014 register_slots_written++;
2015 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
2016 }
2017
2018 // Translate the accumulator register (depending on frame position).
2019 if (is_topmost) {
2020 for (int i = 0; i < ArgumentPaddingSlots(1); ++i) {
2021 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
2022 }
2023 // For topmost frame, put the accumulator on the stack. The
2024 // {NotifyDeoptimized} builtin pops it off the topmost frame (possibly
2025 // after materialization).
2026 if (goto_catch_handler) {
2027 // If we are lazy deopting to a catch handler, we set the accumulator to
2028 // the exception (which lives in the result register).
2029 intptr_t accumulator_value =
2030 input_->GetRegister(kInterpreterAccumulatorRegister.code());
2031 frame_writer.PushRawObject(Tagged<Object>(accumulator_value),
2032 "accumulator\n");
2033 } else {
2034 // If we are lazily deoptimizing make sure we store the deopt
2035 // return value into the appropriate slot.
2036 if (deopt_kind_ == DeoptimizeKind::kLazy &&
2037 translated_frame->return_value_offset() == 0 &&
2038 translated_frame->return_value_count() > 0) {
2039 CHECK_EQ(translated_frame->return_value_count(), 1);
2040 frame_writer.PushRawValue(input_->GetRegister(kReturnRegister0.code()),
2041 "return value 0\n");
2042 } else {
2043 frame_writer.PushTranslatedValue(value_iterator, "accumulator");
2044 }
2045 }
2046 ++value_iterator; // Move over the accumulator.
2047 } else {
2048 // For non-topmost frames, skip the accumulator translation. For those
2049 // frames, the return value from the callee will become the accumulator.
2050 ++value_iterator;
2051 }
2052 CHECK_EQ(translated_frame->end(), value_iterator);
2053 CHECK_EQ(0u, frame_writer.top_offset());
2054
2055 const intptr_t pc =
2056 static_cast<intptr_t>(dispatch_builtin->instruction_start()) +
2057 isolate()->heap()->deopt_pc_offset_after_adapt_shadow_stack().value();
2058 if (is_topmost) {
2059 // Only the pc of the topmost frame needs to be signed since it is
2060 // authenticated at the end of the DeoptimizationEntry builtin.
2061 const intptr_t top_most_pc = PointerAuthentication::SignAndCheckPC(
2062 isolate(), pc, frame_writer.frame()->GetTop());
2063 output_frame->SetPc(top_most_pc);
2064 } else {
2065 output_frame->SetPc(pc);
2066 }
2067
2068 // Update constant pool.
2069 if (V8_EMBEDDED_CONSTANT_POOL_BOOL) {
2070 intptr_t constant_pool_value =
2071 static_cast<intptr_t>(dispatch_builtin->constant_pool());
2072 output_frame->SetConstantPool(constant_pool_value);
2073 if (is_topmost) {
2074 Register constant_pool_reg =
2075 UnoptimizedJSFrame::constant_pool_pointer_register();
2076 output_frame->SetRegister(constant_pool_reg.code(), constant_pool_value);
2077 }
2078 }
2079
2080 // Clear the context register. The context might be a de-materialized object
2081 // and will be materialized by {Runtime_NotifyDeoptimized}. For additional
2082 // safety we use Tagged<Smi>(0) instead of the potential {arguments_marker}
2083 // here.
2084 if (is_topmost) {
2085 intptr_t context_value = static_cast<intptr_t>(Smi::zero().ptr());
2086 Register context_reg = JavaScriptFrame::context_register();
2087 output_frame->SetRegister(context_reg.code(), context_value);
2088 // Set the continuation for the topmost frame.
2089 Tagged<Code> continuation = builtins->code(Builtin::kNotifyDeoptimized);
2090 output_frame->SetContinuation(
2091 static_cast<intptr_t>(continuation->instruction_start()));
2092 }
2093}

2095void Deoptimizer::DoComputeInlinedExtraArguments(
2096 TranslatedFrame* translated_frame, int frame_index) {
2097 // Inlined arguments frame can not be the topmost, nor the bottom most frame.
2098 CHECK(frame_index < output_count_ - 1);
2099 CHECK_GT(frame_index, 0);
2100 CHECK_NULL(output_[frame_index]);
2101
2102 // During deoptimization we need push the extra arguments of inlined functions
2103 // (arguments with index greater than the formal parameter count).
2104 // For more info, see the design document:
2105 // https://docs.google.com/document/d/150wGaUREaZI6YWqOQFD5l2mWQXaPbbZjcAIJLOFrzMs
2106
2107 TranslatedFrame::iterator value_iterator = translated_frame->begin();
2108 const int argument_count_without_receiver = translated_frame->height() - 1;
2109 const int formal_parameter_count_without_receiver =
2110 translated_frame->formal_parameter_count() - 1;
2111 SBXCHECK_GE(formal_parameter_count_without_receiver, 0);
2112 const int extra_argument_count =
2113 argument_count_without_receiver - formal_parameter_count_without_receiver;
2114 // The number of pushed arguments is the maximum of the actual argument count
2115 // and the formal parameter count + the receiver.
2116 const int padding =
2117 ArgumentPaddingSlots(std::max(argument_count_without_receiver,
2118 formal_parameter_count_without_receiver) +
2119 1);
2120 const int output_frame_size =
2121 (std::max(0, extra_argument_count) + padding) * kSystemPointerSize;
2122 if (verbose_tracing_enabled()) {
2123 PrintF(trace_scope_->file(),
2124 " translating inlined arguments frame => variable_size=%d\n",
2125 output_frame_size);
2126 }
2127
2128 // Allocate and store the output frame description.
2129 FrameDescription* output_frame = FrameDescription::Create(
2130 output_frame_size, JSParameterCount(argument_count_without_receiver),
2131 isolate());
2132 // The top address of the frame is computed from the previous frame's top and
2133 // this frame's size.
2134 const intptr_t top_address =
2135 output_[frame_index - 1]->GetTop() - output_frame_size;
2136 output_frame->SetTop(top_address);
2137 // This is not a real frame, we take PC and FP values from the parent frame.
2138 output_frame->SetPc(output_[frame_index - 1]->GetPc());
2139 output_frame->SetFp(output_[frame_index - 1]->GetFp());
2140 output_[frame_index] = output_frame;
2141
2142 FrameWriter frame_writer(this, output_frame, verbose_trace_scope());
2143
2144 ReadOnlyRoots roots(isolate());
2145 for (int i = 0; i < padding; ++i) {
2146 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
2147 }
2148
2149 if (extra_argument_count > 0) {
2150 // The receiver and arguments with index below the formal parameter
2151 // count are in the fake adaptor frame, because they are used to create the
2152 // arguments object. We should however not push them, since the interpreter
2153 // frame will do that.
2154 value_iterator++; // Skip function.
2155 value_iterator++; // Skip receiver.
2156 for (int i = 0; i < formal_parameter_count_without_receiver; i++)
2157 value_iterator++;
2158 frame_writer.PushStackJSArguments(value_iterator, extra_argument_count);
2159 }

2210 frame_writer.PushStackJSArguments(value_iterator, parameters_count);

2213 frame_writer.top_offset());

2217 frame_writer.PushApprovedCallerPc(caller_pc);

2222

2255
2256 if (is_topmost) {
2257 for (int i = 0; i < ArgumentPaddingSlots(1); ++i) {

2259 }
2260 // Ensure the result is restored back when we return to the stub.

2267 CHECK_EQ(0u, frame_writer.top_offset());

2271 isolate_->builtins()->code(Builtin::kJSConstructStubGeneric);
2272 Address start = construct_stub->instruction_start();
2273 const int pc_offset =
2274 isolate_->heap()->construct_stub_create_deopt_pc_offset().value();

2278 // authenticated at the end of the DeoptimizationEntry builtin.

2280 isolate(), pc_value, frame_writer.frame()->GetTop()));

2287 intptr_t constant_pool_value =
2288 static_cast<intptr_t>(construct_stub->constant_pool());
2289 output_frame->SetConstantPool(constant_pool_value);
2290 if (is_topmost) {
2291 Register constant_pool_reg =
2292 JavaScriptFrame::constant_pool_pointer_register();
2293 output_frame->SetRegister(constant_pool_reg.code(), constant_pool_value);
2294 }
2295 }
2296
2297 // Clear the context register. The context might be a de-materialized object
2298 // and will be materialized by {Runtime_NotifyDeoptimized}. For additional
2299 // safety we use Tagged<Smi>(0) instead of the potential {arguments_marker}
2300 // here.
2301 if (is_topmost) {
2302 intptr_t context_value = static_cast<intptr_t>(Smi::zero().ptr());
2303 Register context_reg = JavaScriptFrame::context_register();
2304 output_frame->SetRegister(context_reg.code(), context_value);
2305

2315void Deoptimizer::DoComputeConstructInvokeStubFrame(
2316 TranslatedFrame* translated_frame, int frame_index) {
2317 TranslatedFrame::iterator value_iterator = translated_frame->begin();
2318 const bool is_topmost = (output_count_ - 1 == frame_index);
2319 // The construct frame could become topmost only if we inlined a constructor
2320 // call which does a tail call (otherwise the tail callee's frame would be
2321 // the topmost one). So it could only be the DeoptimizeKind::kLazy case.
2322 CHECK(!is_topmost || deopt_kind_ == DeoptimizeKind::kLazy);
2323 DCHECK_EQ(translated_frame->kind(), TranslatedFrame::kConstructInvokeStub);
2324 DCHECK_EQ(translated_frame->height(), 0);
2325
2326 FastConstructStubFrameInfo frame_info =
2327 FastConstructStubFrameInfo::Precise(is_topmost);
2328 const uint32_t output_frame_size = frame_info.frame_size_in_bytes();
2329 if (verbose_tracing_enabled()) {
2330 PrintF(trace_scope()->file(),
2331 " translating construct invoke stub => variable_frame_size=%d, "
2332 "frame_size=%d\n",
2333 frame_info.frame_size_in_bytes_without_fixed(), output_frame_size);
2334 }
2335
2336 // Allocate and store the output frame description.
2337 FrameDescription* output_frame =
2338 FrameDescription::Create(output_frame_size, 0, isolate());
2339 FrameWriter frame_writer(this, output_frame, verbose_trace_scope());
2340 DCHECK(frame_index > 0 && frame_index < output_count_);
2341 DCHECK_NULL(output_[frame_index]);
2342 output_[frame_index] = output_frame;
2343
2344 // The top address of the frame is computed from the previous frame's top and
2345 // this frame's size.
2346 const intptr_t top_address =
2347 output_[frame_index - 1]->GetTop() - output_frame_size;
2348 output_frame->SetTop(top_address);
2349
2350 // The allocated receiver of a construct stub frame is passed as the
2351 // receiver parameter through the translation. It might be encoding
2352 // a captured object, so we need save it for later.
2353 TranslatedFrame::iterator receiver_iterator = value_iterator;
2354 value_iterator++;
2355
2356 // Read caller's PC from the previous frame.
2357 const intptr_t caller_pc = output_[frame_index - 1]->GetPc();
2358 frame_writer.PushApprovedCallerPc(caller_pc);
2359
2360 // Read caller's FP from the previous frame, and set this frame's FP.
2361 const intptr_t caller_fp = output_[frame_index - 1]->GetFp();
2362 frame_writer.PushCallerFp(caller_fp);
2363
2364 const intptr_t fp_value = top_address + frame_writer.top_offset();
2365 output_frame->SetFp(fp_value);
2366 if (is_topmost) {
2367 Register fp_reg = JavaScriptFrame::fp_register();
2368 output_frame->SetRegister(fp_reg.code(), fp_value);
2369 }
2370
2371 if (V8_EMBEDDED_CONSTANT_POOL_BOOL) {
2372 // Read the caller's constant pool from the previous frame.
2373 const intptr_t caller_cp = output_[frame_index - 1]->GetConstantPool();
2374 frame_writer.PushCallerConstantPool(caller_cp);
2375 }
2376 intptr_t marker = StackFrame::TypeToMarker(StackFrame::FAST_CONSTRUCT);
2377 frame_writer.PushRawValue(marker, "fast construct stub sentinel\n");
2378 frame_writer.PushTranslatedValue(value_iterator++, "context");
2379 frame_writer.PushTranslatedValue(receiver_iterator, "implicit receiver");
2380
2381 // The FastConstructFrame needs to be aligned in some architectures.
2382 ReadOnlyRoots roots(isolate());
2383 for (int i = 0; i < ArgumentPaddingSlots(1); ++i) {
2384 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
2385 }
2386
2387 if (is_topmost) {
2388 for (int i = 0; i < ArgumentPaddingSlots(1); ++i) {
2389 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
2390 }
2391 // Ensure the result is restored back when we return to the stub.
2392 Register result_reg = kReturnRegister0;
2393 intptr_t result = input_->GetRegister(result_reg.code());
2394 frame_writer.PushRawValue(result, "subcall result\n");
2395 }
2396
2397 CHECK_EQ(translated_frame->end(), value_iterator);
2398 CHECK_EQ(0u, frame_writer.top_offset());
2399
2400 // Compute this frame's PC.
2401 Tagged<Code> construct_stub = isolate_->builtins()->code(
2402 Builtin::kInterpreterPushArgsThenFastConstructFunction);
2403 Address start = construct_stub->instruction_start();
2404 const int pc_offset =
2405 isolate_->heap()->construct_stub_invoke_deopt_pc_offset().value();
2406 intptr_t pc_value = static_cast<intptr_t>(start + pc_offset);
2407 if (is_topmost) {
2408 // Only the pc of the topmost frame needs to be signed since it is
2409 // authenticated at the end of the DeoptimizationEntry builtin.
2410 output_frame->SetPc(PointerAuthentication::SignAndCheckPC(
2411 isolate(), pc_value, frame_writer.frame()->GetTop()));
2412 } else {
2413 output_frame->SetPc(pc_value);
2414 }
2415
2416 // Update constant pool.
2417 if (V8_EMBEDDED_CONSTANT_POOL_BOOL) {
2418 intptr_t constant_pool_value =
2419 static_cast<intptr_t>(construct_stub->constant_pool());
2420 output_frame->SetConstantPool(constant_pool_value);
2421 if (is_topmost) {
2422 Register constant_pool_reg =
2423 JavaScriptFrame::constant_pool_pointer_register();
2424 output_frame->SetRegister(constant_pool_reg.code(), constant_pool_value);
2425 }
2426 }
2427
2428 // Clear the context register. The context might be a de-materialized object
2429 // and will be materialized by {Runtime_NotifyDeoptimized}. For additional
2430 // safety we use Tagged<Smi>(0) instead of the potential {arguments_marker}
2431 // here.
2432 if (is_topmost) {
2433 intptr_t context_value = static_cast<intptr_t>(Smi::zero().ptr());
2434 Register context_reg = JavaScriptFrame::context_register();
2435 output_frame->SetRegister(context_reg.code(), context_value);
2436
2437 // Set the continuation for the topmost frame.
2438 DCHECK_EQ(DeoptimizeKind::kLazy, deopt_kind_);
2439 Tagged<Code> continuation =
2440 isolate_->builtins()->code(Builtin::kNotifyDeoptimized);
2441 output_frame->SetContinuation(
2442 static_cast<intptr_t>(continuation->instruction_start()));
2443 }

2530// BuiltinContinuationFrames capture the machine state that is expected as input
2531// to a builtin, including both input register values and stack parameters. When
2532// the frame is reactivated (i.e. the frame below it returns), a
2533// ContinueToBuiltin stub restores the register state from the frame and tail
2534// calls to the actual target builtin, making it appear that the stub had been
2535// directly called by the frame above it. The input values to populate the frame
2536// are taken from the deopt's FrameState.
2537//
2538// Frame translation happens in two modes, EAGER and LAZY. In EAGER mode, all of
2539// the parameters to the Builtin are explicitly specified in the TurboFan
2540// FrameState node. In LAZY mode, there is always one fewer parameters specified
2541// in the FrameState than expected by the Builtin. In that case, construction of
2542// BuiltinContinuationFrame adds the final missing parameter during
2543// deoptimization, and that parameter is always on the stack and contains the
2544// value returned from the callee of the call site triggering the LAZY deopt
2545// (e.g. rax on x64). This requires that continuation Builtins for LAZY deopts
2546// must have at least one stack parameter.
2547//
2548// TO
2549// | .... |
2550// +-------------------------+
2551// | arg padding (arch dept) |<- at most 1*kSystemPointerSize
2552// +-------------------------+
2553// | builtin param 0 |<- FrameState input value n becomes
2554// +-------------------------+
2555// | ... |
2556// +-------------------------+
2557// | builtin param m |<- FrameState input value n+m-1, or in
2558// +-----needs-alignment-----+ the LAZY case, return LAZY result value
2559// | ContinueToBuiltin entry |
2560// +-------------------------+
2561// | | saved frame (FP) |
2562// | +=====needs=alignment=====+<- fpreg
2563// | |constant pool (if ool_cp)|
2564// v +-------------------------+
2565// |BUILTIN_CONTINUATION mark|
2566// +-------------------------+
2567// | JSFunction (or zero) |<- only if JavaScript builtin
2568// +-------------------------+
2569// | frame height above FP |
2570// +-------------------------+
2571// | context |<- this non-standard context slot contains
2572// +-------------------------+ the context, even for non-JS builtins.
2573// | builtin index |
2574// +-------------------------+
2575// | builtin input GPR reg0 |<- populated from deopt FrameState using
2576// +-------------------------+ the builtin's CallInterfaceDescriptor
2577// | ... | to map a FrameState's 0..n-1 inputs to
2578// +-------------------------+ the builtin's n input register params.
2579// | builtin input GPR regn |
2580// +-------------------------+
2581// | reg padding (arch dept) |
2582// +-----needs--alignment----+
2583// | res padding (arch dept) |<- only if {is_topmost}; result is pop'd by
2584// +-------------------------+<- kNotifyDeopt ASM stub and moved to acc
2585// | result value |<- reg, as ContinueToBuiltin stub expects.
2586// +-----needs-alignment-----+<- spreg
2587//
2588void Deoptimizer::DoComputeBuiltinContinuation(
2589 TranslatedFrame* translated_frame, int frame_index,
2590 BuiltinContinuationMode mode) {
2591 TranslatedFrame::iterator result_iterator = translated_frame->end();
2592
2593 bool is_js_to_wasm_builtin_continuation = false;
2594#if V8_ENABLE_WEBASSEMBLY
2595 is_js_to_wasm_builtin_continuation =
2596 translated_frame->kind() == TranslatedFrame::kJSToWasmBuiltinContinuation;
2597 if (is_js_to_wasm_builtin_continuation) {
2598 // For JSToWasmBuiltinContinuations, add a TranslatedValue with the result
2599 // of the Wasm call, extracted from the input FrameDescription.
2600 // This TranslatedValue will be written in the output frame in place of the
2601 // hole and we'll use ContinueToCodeStubBuiltin in place of
2602 // ContinueToCodeStubBuiltinWithResult.
2603 TranslatedValue result = TranslatedValueForWasmReturnKind(
2604 translated_frame->wasm_call_return_kind());
2605 translated_frame->Add(result);
2606 }
2607#endif // V8_ENABLE_WEBASSEMBLY
2608
2609 TranslatedFrame::iterator value_iterator = translated_frame->begin();
2610
2611 const BytecodeOffset bytecode_offset = translated_frame->bytecode_offset();
2612 Builtin builtin = Builtins::GetBuiltinFromBytecodeOffset(bytecode_offset);
2613 CallInterfaceDescriptor continuation_descriptor =
2614 Builtins::CallInterfaceDescriptorFor(builtin);
2615
2616 const RegisterConfiguration* config = RegisterConfiguration::Default();
2617
2618 const bool is_bottommost = (0 == frame_index);
2619 const bool is_topmost = (output_count_ - 1 == frame_index);
2620
2621 const int parameters_count = translated_frame->height();
2622 BuiltinContinuationFrameInfo frame_info =
2623 BuiltinContinuationFrameInfo::Precise(parameters_count,
2624 continuation_descriptor, config,
2625 is_topmost, deopt_kind_, mode);
2626
2627 const unsigned output_frame_size = frame_info.frame_size_in_bytes();
2628 const unsigned output_frame_size_above_fp =
2629 frame_info.frame_size_in_bytes_above_fp();
2630
2631 // Validate types of parameters. They must all be tagged except for argc and
2632 // the dispatch handle for JS builtins.
2633 bool has_argc = false;
2634 const int register_parameter_count =
2635 continuation_descriptor.GetRegisterParameterCount();
2636 for (int i = 0; i < register_parameter_count; ++i) {
2637 MachineType type = continuation_descriptor.GetParameterType(i);
2638 int code = continuation_descriptor.GetRegisterParameter(i).code();
2639 // Only tagged and int32 arguments are supported, and int32 only for the
2640 // arguments count and dispatch handle on JavaScript builtins.
2641 if (type == MachineType::Int32()) {
2642 CHECK(code == kJavaScriptCallArgCountRegister.code() ||
2643 code == kJavaScriptCallDispatchHandleRegister.code());
2644 has_argc = true;
2645 } else {
2646 // Any other argument must be a tagged value.
2647 CHECK(IsAnyTagged(type.representation()));
2648 }
2649 }
2650 CHECK_EQ(BuiltinContinuationModeIsJavaScript(mode), has_argc);
2651
2652 if (verbose_tracing_enabled()) {
2653 PrintF(trace_scope()->file(),
2654 " translating BuiltinContinuation to %s,"
2655 " => register_param_count=%d,"
2656 " stack_param_count=%d, frame_size=%d\n",
2657 Builtins::name(builtin), register_parameter_count,
2658 frame_info.stack_parameter_count(), output_frame_size);
2659 }
2660
2661 FrameDescription* output_frame = FrameDescription::Create(
2662 output_frame_size, frame_info.stack_parameter_count(), isolate());
2663 output_[frame_index] = output_frame;
2664 FrameWriter frame_writer(this, output_frame, verbose_trace_scope());
2665
2666 // The top address of the frame is computed from the previous frame's top and
2667 // this frame's size.
2668 const intptr_t top_address =
2669 is_bottommost ? caller_frame_top_ - output_frame_size
2670 : output_[frame_index - 1]->GetTop() - output_frame_size;
2671 output_frame->SetTop(top_address);
2672
2673 // Get the possible JSFunction for the case that this is a
2674 // JavaScriptBuiltinContinuationFrame, which needs the JSFunction pointer
2675 // like a normal JavaScriptFrame.
2676 const intptr_t maybe_function = value_iterator->GetRawValue().ptr();
2677 ++value_iterator;
2678
2679 ReadOnlyRoots roots(isolate());
2680 const int padding = ArgumentPaddingSlots(frame_info.stack_parameter_count());
2681 for (int i = 0; i < padding; ++i) {
2682 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
2683 }
2684
2685 if (mode == BuiltinContinuationMode::STUB) {
2686 DCHECK_EQ(continuation_descriptor.GetStackArgumentOrder(),
2687 StackArgumentOrder::kDefault);
2688 for (uint32_t i = 0; i < frame_info.translated_stack_parameter_count();
2689 ++i, ++value_iterator) {
2690 frame_writer.PushTranslatedValue(value_iterator, "stack parameter");
2691 }
2692 if (frame_info.frame_has_result_stack_slot()) {
2693 if (is_js_to_wasm_builtin_continuation) {
2694 frame_writer.PushTranslatedValue(result_iterator,
2695 "return result on lazy deopt\n");
2696 } else {
2697 DCHECK_EQ(result_iterator, translated_frame->end());
2698 frame_writer.PushRawObject(
2699 roots.the_hole_value(),
2700 "placeholder for return result on lazy deopt\n");
2701 }
2702 }
2703 } else {
2704 // JavaScript builtin.
2705 if (frame_info.frame_has_result_stack_slot()) {
2706 frame_writer.PushRawObject(
2707 roots.the_hole_value(),
2708 "placeholder for return result on lazy deopt\n");
2709 }
2710 switch (mode) {
2711 case BuiltinContinuationMode::STUB:
2712 UNREACHABLE();
2713 case BuiltinContinuationMode::JAVASCRIPT:
2714 break;
2715 case BuiltinContinuationMode::JAVASCRIPT_WITH_CATCH: {
2716 frame_writer.PushRawObject(roots.the_hole_value(),
2717 "placeholder for exception on lazy deopt\n");
2718 } break;
2719 case BuiltinContinuationMode::JAVASCRIPT_HANDLE_EXCEPTION: {
2720 intptr_t accumulator_value =
2721 input_->GetRegister(kInterpreterAccumulatorRegister.code());
2722 frame_writer.PushRawObject(Tagged<Object>(accumulator_value),
2723 "exception (from accumulator)\n");
2724 } break;
2725 }
2726 frame_writer.PushStackJSArguments(
2727 value_iterator, frame_info.translated_stack_parameter_count());
2728 }
2729
2730 DCHECK_EQ(output_frame->GetLastArgumentSlotOffset(),
2731 frame_writer.top_offset());
2732
2733 std::vector<TranslatedFrame::iterator> register_values;
2734 int total_registers = config->num_general_registers();
2735 register_values.resize(total_registers, {value_iterator});
2736
2737 for (int i = 0; i < register_parameter_count; ++i, ++value_iterator) {
2738 int code = continuation_descriptor.GetRegisterParameter(i).code();
2739 register_values[code] = value_iterator;
2740 }
2741
2742 // The context register is always implicit in the CallInterfaceDescriptor but
2743 // its register must be explicitly set when continuing to the builtin. Make
2744 // sure that it's harvested from the translation and copied into the register
2745 // set (it was automatically added at the end of the FrameState by the
2746 // instruction selector).
2747 Tagged<Object> context = value_iterator->GetRawValue();
2748 const intptr_t value = context.ptr();
2749 TranslatedFrame::iterator context_register_value = value_iterator++;
2750 register_values[kContextRegister.code()] = context_register_value;
2751 output_frame->SetRegister(kContextRegister.code(), value);
2752
2753 // Set caller's PC (JSFunction continuation).
2754 if (is_bottommost) {
2755 frame_writer.PushBottommostCallerPc(caller_pc_);
2756 } else {
2757 frame_writer.PushApprovedCallerPc(output_[frame_index - 1]->GetPc());
2758 }
2759
2760 // Read caller's FP from the previous frame, and set this frame's FP.
2761 const intptr_t caller_fp =
2762 is_bottommost ? caller_fp_ : output_[frame_index - 1]->GetFp();
2763 frame_writer.PushCallerFp(caller_fp);
2764
2765 const intptr_t fp_value = top_address + frame_writer.top_offset();
2766 output_frame->SetFp(fp_value);
2767
2768 DCHECK_EQ(output_frame_size_above_fp, frame_writer.top_offset());
2769
2770 if (V8_EMBEDDED_CONSTANT_POOL_BOOL) {
2771 // Read the caller's constant pool from the previous frame.
2772 const intptr_t caller_cp =
2773 is_bottommost ? caller_constant_pool_
2774 : output_[frame_index - 1]->GetConstantPool();
2775 frame_writer.PushCallerConstantPool(caller_cp);
2776 }
2777
2778 // A marker value is used in place of the context.
2779 const intptr_t marker =
2780 StackFrame::TypeToMarker(BuiltinContinuationModeToFrameType(mode));
2781 frame_writer.PushRawValue(marker,
2782 "context (builtin continuation sentinel)\n");
2783
2784 if (BuiltinContinuationModeIsJavaScript(mode)) {
2785 frame_writer.PushRawValue(maybe_function, "JSFunction\n");
2786 } else {
2787 frame_writer.PushRawValue(0, "unused\n");
2788 }
2789
2790 // The delta from the SP to the FP; used to reconstruct SP in
2791 // Isolate::UnwindAndFindHandler.
2792 frame_writer.PushRawObject(Smi::FromInt(output_frame_size_above_fp),
2793 "frame height at deoptimization\n");
2794
2795 // The context even if this is a stub continuation frame. We can't use the
2796 // usual context slot, because we must store the frame marker there.
2797 frame_writer.PushTranslatedValue(context_register_value,
2798 "builtin JavaScript context\n");
2799
2800 // The builtin to continue to.
2801 frame_writer.PushRawObject(Smi::FromInt(static_cast<int>(builtin)),
2802 "builtin index\n");
2803
2804 const int allocatable_register_count =
2805 config->num_allocatable_general_registers();
2806 for (int i = 0; i < allocatable_register_count; ++i) {
2807 int code = config->GetAllocatableGeneralCode(i);
2808 base::ScopedVector<char> str(128);
2809 if (verbose_tracing_enabled()) {
2810 if (BuiltinContinuationModeIsJavaScript(mode) &&
2811 code == kJavaScriptCallArgCountRegister.code()) {
2812 SNPrintF(
2813 str,
2814 "tagged argument count %s (will be untagged by continuation)\n",
2815 RegisterName(Register::from_code(code)));
2816 } else {
2817 SNPrintF(str, "builtin register argument %s\n",
2818 RegisterName(Register::from_code(code)));
2819 }
2820 }
2821 frame_writer.PushTranslatedValue(
2822 register_values[code], verbose_tracing_enabled() ? str.begin() : "");
2823 }
2824
2825 // Some architectures must pad the stack frame with extra stack slots
2826 // to ensure the stack frame is aligned.
2827 const int padding_slot_count =
2828 BuiltinContinuationFrameConstants::PaddingSlotCount(
2829 allocatable_register_count);
2830 for (int i = 0; i < padding_slot_count; ++i) {
2831 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
2832 }
2833
2834 if (is_topmost) {
2835 for (int i = 0; i < ArgumentPaddingSlots(1); ++i) {
2836 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
2837 }
2838
2839 // Ensure the result is restored back when we return to the stub.
2840 // For JS-to-Wasm builtin continuations the returns are handled differently
2841 // and we can't push untagged return values onto the stack as they would be
2842 // visited during a GC and treated as tagged stack slots.
2843 if (frame_info.frame_has_result_stack_slot() &&
2844 !is_js_to_wasm_builtin_continuation) {
2845 Register result_reg = kReturnRegister0;
2846 frame_writer.PushRawValue(input_->GetRegister(result_reg.code()),
2847 "callback result\n");
2848 } else {
2849 frame_writer.PushRawObject(roots.undefined_value(), "callback result\n");
2850 }
2851 }
2852
2853 CHECK_EQ(result_iterator, value_iterator);
2854 CHECK_EQ(0u, frame_writer.top_offset());
2855
2856 // Clear the context register. The context might be a de-materialized object
2857 // and will be materialized by {Runtime_NotifyDeoptimized}. For additional
2858 // safety we use Tagged<Smi>(0) instead of the potential {arguments_marker}
2859 // here.
2860 if (is_topmost) {
2861 intptr_t context_value = static_cast<intptr_t>(Smi::zero().ptr());
2862 Register context_reg = JavaScriptFrame::context_register();
2863 output_frame->SetRegister(context_reg.code(), context_value);
2864 }
2865

2867 // will build its own frame once we continue to it.

2997unsigned Deoptimizer::ComputeInputFrameSize() const {
2998 // The fp-to-sp delta already takes the context, constant pool pointer and the
2999 // function into account so we have to avoid double counting them.
3000 unsigned fixed_size_above_fp = ComputeInputFrameAboveFpFixedSize();
3001 unsigned result = fixed_size_above_fp + fp_to_sp_delta_;
3002 DCHECK(CodeKindCanDeoptimize(compiled_code_->kind()));
3003 unsigned stack_slots = compiled_code_->stack_slots();
3004 if (compiled_code_->is_maglevved() && !deoptimizing_throw_) {
3005 // Maglev code can deopt in deferred code which has spilled registers across
3006 // the call. These will be included in the fp_to_sp_delta, but the expected
3007 // frame size won't include them, so we need to check for less-equal rather
3008 // than equal. For deoptimizing throws, these will have already been trimmed
3009 // off.
3010 CHECK_LE(fixed_size_above_fp + (stack_slots * kSystemPointerSize) -
3011 CommonFrameConstants::kFixedFrameSizeAboveFp,
3012 result);
3013 // With slow asserts we can check this exactly, by looking up the safepoint.
3014 if (v8_flags.enable_slow_asserts) {
3015 Address deopt_call_pc = GetDeoptCallPCFromReturnPC(from_, compiled_code_);
3016 MaglevSafepointTable table(isolate_, deopt_call_pc, compiled_code_);
3017 MaglevSafepointEntry safepoint = table.FindEntry(deopt_call_pc);
3018 unsigned extra_spills = safepoint.num_extra_spill_slots();
3019 CHECK_EQ(fixed_size_above_fp + (stack_slots * kSystemPointerSize) -
3020 CommonFrameConstants::kFixedFrameSizeAboveFp +
3021 extra_spills * kSystemPointerSize,
3022 result);
3023 }
3024 } else {
3025 unsigned outgoing_size = 0;
3026 CHECK_EQ(fixed_size_above_fp + (stack_slots * kSystemPointerSize) -
3027 CommonFrameConstants::kFixedFrameSizeAboveFp + outgoing_size,
3028 result);
3029 }
3030 return result;
3031}

3039Deoptimizer::DeoptInfo Deoptimizer::GetDeoptInfo(Tagged<Code> code,
3040 Address pc) {
3041 CHECK(code->instruction_start() <= pc && pc <= code->instruction_end());
3042 SourcePosition last_position = SourcePosition::Unknown();
3043 DeoptimizeReason last_reason = DeoptimizeReason::kUnknown;
3044 uint32_t last_node_id = 0;
3045 int last_deopt_id = kNoDeoptimizationId;
3046 int mask = RelocInfo::ModeMask(RelocInfo::DEOPT_REASON) |
3047 RelocInfo::ModeMask(RelocInfo::DEOPT_ID) |
3048 RelocInfo::ModeMask(RelocInfo::DEOPT_SCRIPT_OFFSET) |
3049 RelocInfo::ModeMask(RelocInfo::DEOPT_INLINING_ID) |
3050 RelocInfo::ModeMask(RelocInfo::DEOPT_NODE_ID);
3051 for (RelocIterator it(code, mask); !it.done(); it.next()) {
3052 RelocInfo* info = it.rinfo();
3053 if (info->pc() >= pc) break;
3054 if (info->rmode() == RelocInfo::DEOPT_SCRIPT_OFFSET) {
3055 int script_offset = static_cast<int>(info->data());
3056 it.next();
3057 DCHECK(it.rinfo()->rmode() == RelocInfo::DEOPT_INLINING_ID);
3058 int inlining_id = static_cast<int>(it.rinfo()->data());
3059 last_position = SourcePosition(script_offset, inlining_id);
3060 } else if (info->rmode() == RelocInfo::DEOPT_ID) {
3061 last_deopt_id = static_cast<int>(info->data());
3062 } else if (info->rmode() == RelocInfo::DEOPT_REASON) {
3063 last_reason = static_cast<DeoptimizeReason>(info->data());
3064 } else if (info->rmode() == RelocInfo::DEOPT_NODE_ID) {
3065 last_node_id = static_cast<uint32_t>(info->data());
3066 }
3067 }
3068 return DeoptInfo(last_position, last_reason, last_node_id, last_deopt_id);
3069}
