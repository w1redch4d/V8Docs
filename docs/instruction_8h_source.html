188class UnallocatedOperand final : public InstructionOperand {
189 public:
190 enum BasicPolicy { FIXED_SLOT, EXTENDED_POLICY };
191
192 enum ExtendedPolicy {
193 NONE,
194 REGISTER_OR_SLOT,
195 REGISTER_OR_SLOT_OR_CONSTANT,
196 FIXED_REGISTER,
197 FIXED_FP_REGISTER,
198 MUST_HAVE_REGISTER,
199 MUST_HAVE_SLOT,
200 SAME_AS_INPUT
201 };
202
203 // Lifetime of operand inside the instruction.
204 enum Lifetime {
205 // USED_AT_START operand is guaranteed to be live only at instruction start.
206 // The register allocator is free to assign the same register to some other
207 // operand used inside instruction (i.e. temporary or output).
208 USED_AT_START,
209
210 // USED_AT_END operand is treated as live until the end of instruction.
211 // This means that register allocator will not reuse its register for any
212 // other operand inside instruction.
213 USED_AT_END
214 };
215
216 UnallocatedOperand(ExtendedPolicy policy, int virtual_register)
217 : UnallocatedOperand(virtual_register) {
218 value_ |= BasicPolicyField::encode(EXTENDED_POLICY);
219 value_ |= ExtendedPolicyField::encode(policy);
220 value_ |= LifetimeField::encode(USED_AT_END);
221 }
222
223 UnallocatedOperand(int virtual_register, int input_index)
224 : UnallocatedOperand(virtual_register) {
225 value_ |= BasicPolicyField::encode(EXTENDED_POLICY);
226 value_ |= ExtendedPolicyField::encode(SAME_AS_INPUT);
227 value_ |= LifetimeField::encode(USED_AT_END);
228 value_ |= InputIndexField::encode(input_index);
229 }
230
231 UnallocatedOperand(BasicPolicy policy, int index, int virtual_register)
232 : UnallocatedOperand(virtual_register) {
233 DCHECK(policy == FIXED_SLOT);
234 value_ |= BasicPolicyField::encode(policy);
235 value_ |= static_cast<uint64_t>(static_cast<int64_t>(index))
236 << FixedSlotIndexField::kShift;
237 DCHECK(this->fixed_slot_index() == index);
238 }
239
240 UnallocatedOperand(ExtendedPolicy policy, int index, int virtual_register)
241 : UnallocatedOperand(virtual_register) {
242 DCHECK(policy == FIXED_REGISTER || policy == FIXED_FP_REGISTER);
243 value_ |= BasicPolicyField::encode(EXTENDED_POLICY);
244 value_ |= ExtendedPolicyField::encode(policy);
245 value_ |= LifetimeField::encode(USED_AT_END);
246 value_ |= FixedRegisterField::encode(index);
247 }
248
249 UnallocatedOperand(ExtendedPolicy policy, Lifetime lifetime,
250 int virtual_register)
251 : UnallocatedOperand(virtual_register) {
252 value_ |= BasicPolicyField::encode(EXTENDED_POLICY);
253 value_ |= ExtendedPolicyField::encode(policy);
254 value_ |= LifetimeField::encode(lifetime);
255 }
256
257 UnallocatedOperand(int reg_id, int slot_id, int virtual_register)
258 : UnallocatedOperand(FIXED_REGISTER, reg_id, virtual_register) {
259 value_ |= HasSecondaryStorageField::encode(true);
260 value_ |= SecondaryStorageField::encode(slot_id);
261 }
262
263 UnallocatedOperand(const UnallocatedOperand& other, int virtual_register) {
264 DCHECK_NE(kInvalidVirtualRegister, virtual_register);
265 value_ = VirtualRegisterField::update(
266 other.value_, static_cast<uint32_t>(virtual_register));
267 }

354 // The encoding used for UnallocatedOperand operands depends on the policy
355 // that is
356 // stored within the operand. The FIXED_SLOT policy uses a compact encoding
357 // because it accommodates a larger pay-load.
358 //
359 // For FIXED_SLOT policy:
360 // +------------------------------------------------+
361 // | slot_index | 0 | virtual_register | 001 |
362 // +------------------------------------------------+
363 //
364 // For all other (extended) policies:
365 // +-----------------------------------------------------+
366 // | reg_index | L | PPP | 1 | virtual_register | 001 |
367 // +-----------------------------------------------------+
368 // L ... Lifetime
369 // P ... Policy
370 //
371 // The slot index is a signed value which requires us to decode it manually
372 // instead of using the base::BitField utility class.
373
374 using VirtualRegisterField = KindField::Next<uint32_t, 32>;
375
376 // base::BitFields for all unallocated operands.
377 using BasicPolicyField = VirtualRegisterField::Next<BasicPolicy, 1>;
378
379 // BitFields specific to BasicPolicy::FIXED_SLOT.
380 using FixedSlotIndexField = BasicPolicyField::Next<int, 28>;
381 static_assert(FixedSlotIndexField::kLastUsedBit == 63);
382
383 // BitFields specific to BasicPolicy::EXTENDED_POLICY.
384 using ExtendedPolicyField = BasicPolicyField::Next<ExtendedPolicy, 3>;
385 using LifetimeField = ExtendedPolicyField::Next<Lifetime, 1>;
386 using HasSecondaryStorageField = LifetimeField::Next<bool, 1>;
387 using FixedRegisterField = HasSecondaryStorageField::Next<int, 6>;
388 using SecondaryStorageField = FixedRegisterField::Next<int, 3>;
389 using InputIndexField = SecondaryStorageField::Next<int, 3>;
390
391 private:
392 explicit UnallocatedOperand(int virtual_register)
393 : InstructionOperand(UNALLOCATED) {
394 value_ |=
395 VirtualRegisterField::encode(static_cast<uint32_t>(virtual_register));
396 }

459class PendingOperand : public InstructionOperand {
460 public:
461 PendingOperand() : InstructionOperand(PENDING) {}
462 explicit PendingOperand(PendingOperand* next_operand) : PendingOperand() {
463 set_next(next_operand);
464 }
465
466 void set_next(PendingOperand* next) {
467 DCHECK_NULL(this->next());
468 uintptr_t shifted_value =
469 reinterpret_cast<uintptr_t>(next) >> kPointerShift;
470 DCHECK_EQ(reinterpret_cast<uintptr_t>(next),
471 shifted_value << kPointerShift);
472 value_ |= NextOperandField::encode(static_cast<uint64_t>(shifted_value));
473 }
474
475 PendingOperand* next() const {
476 uintptr_t shifted_value =
477 static_cast<uint64_t>(NextOperandField::decode(value_));
478 return reinterpret_cast<PendingOperand*>(shifted_value << kPointerShift);
479 }
480
481 static PendingOperand* New(Zone* zone, PendingOperand* previous_operand) {
482 return InstructionOperand::New(zone, PendingOperand(previous_operand));
483 }
484
485 INSTRUCTION_OPERAND_CASTS(PendingOperand, PENDING)
486
487 private:
488 // Operands are uint64_t values and so are aligned to 8 byte boundaries,
489 // therefore we can shift off the bottom three zeros without losing data.
490 static const uint64_t kPointerShift = 3;
491 static_assert(alignof(InstructionOperand) >= (1 << kPointerShift));
492
493 using NextOperandField = KindField::Next<uint64_t, 61>;
494 static_assert(NextOperandField::kLastUsedBit == 63);
495};

732uint64_t InstructionOperand::GetCanonicalizedValue() const {
733 if (IsAnyLocationOperand()) {
734 MachineRepresentation canonical = MachineRepresentation::kNone;
735 if (IsFPRegister()) {
736 if (kFPAliasing == AliasingKind::kOverlap) {
737 // We treat all FP register operands the same for simple aliasing.
738 canonical = MachineRepresentation::kFloat64;
739 } else if (kFPAliasing == AliasingKind::kIndependent) {
740 if (IsSimd128Register()) {
741 canonical = MachineRepresentation::kSimd128;
742 } else {
743 canonical = MachineRepresentation::kFloat64;
744 }
745 } else {
746 // We need to distinguish FP register operands of different reps when
747 // aliasing is AliasingKind::kCombine (e.g. ARM).

749 canonical = LocationOperand::cast(this)->representation();
750 }
751 }
752 return InstructionOperand::KindField::update(
753 LocationOperand::RepresentationField::update(this->value_, canonical),
754 LocationOperand::ALLOCATED);
755 }
756 return this->value_;
757}
