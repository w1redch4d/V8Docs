18// The compilation cache consists of several sub-caches: one each for evals and
19// scripts, which use this class as a base class, and a separate generational
20// sub-cache for RegExps. Since the same source code string has different
21// compiled code for scripts and evals, we use separate sub-caches for different
22// compilation modes, to avoid retrieving the wrong result.

67// Sub-cache for eval scripts. Two caches for eval are used. One for eval calls
68// in native contexts and one for eval calls in other contexts. The cache
69// considers the following pieces of information when checking for matching
70// entries:
71// 1. The source string.
72// 2. The shared function info of the calling function.
73// 3. Whether the source should be compiled as strict code or as sloppy code.
74// Note: Currently there are clients of CompileEval that always compile
75// sloppy code even if the calling function is a strict mode function.
76// More specifically these are the CompileString, DebugEvaluate and
77// DebugEvaluateGlobal runtime functions.
78// 4. The start position of the calling scope.
79class CompilationCacheEval : public CompilationCacheEvalOrScript {
80 public:
81 explicit CompilationCacheEval(Isolate* isolate)
82 : CompilationCacheEvalOrScript(isolate) {}
83
84 InfoCellPair Lookup(Handle<String> source,

90 DirectHandle<SharedFunctionInfo> function_info,
91 DirectHandle<NativeContext> native_context,
92 DirectHandle<FeedbackCell> feedback_cell, int position);
93
94 void Age();
95
96 private:

136// The compilation cache keeps shared function infos for compiled
137// scripts and evals. The shared function infos are looked up using
138// the source string as the key. For regular expressions the
139// compilation data is cached.
140class V8_EXPORT_PRIVATE CompilationCache {
141 public:
142 CompilationCache(const CompilationCache&) = delete;
143 CompilationCache& operator=(const CompilationCache&) = delete;
144
145 // Finds the Script and root SharedFunctionInfo for a script source string.
146 // Returns empty handles if the cache doesn't contain a script for the given
147 // source string with the right origin.
148 CompilationCacheScript::LookupResult LookupScript(
149 Handle<String> source, const ScriptDetails& script_details,
150 LanguageMode language_mode);
151
152 // Finds the shared function info for a source string for eval in a
153 // given context. Returns an empty handle if the cache doesn't
154 // contain a script for the given source string.
155 InfoCellPair LookupEval(Handle<String> source,
156 Handle<SharedFunctionInfo> outer_info,
157 DirectHandle<Context> context,
158 LanguageMode language_mode, int position);
159
160 // Returns the regexp data associated with the given regexp if it
161 // is in cache, otherwise an empty handle.
162 MaybeHandle<RegExpData> LookupRegExp(Handle<String> source,
163 JSRegExp::Flags flags);
164
165 // Associate the (source, kind) pair to the shared function
166 // info. This may overwrite an existing mapping.
167 void PutScript(Handle<String> source, LanguageMode language_mode,
168 DirectHandle<SharedFunctionInfo> function_info);
169
170 // Associate the (source, context->closure()->shared(), kind) triple
171 // with the shared function info. This may overwrite an existing mapping.
172 void PutEval(Handle<String> source, Handle<SharedFunctionInfo> outer_info,
173 DirectHandle<Context> context,
174 DirectHandle<SharedFunctionInfo> function_info,
175 DirectHandle<FeedbackCell> feedback_cell, int position);
176
177 // Associate the (source, flags) pair to the given regexp data.
178 // This may overwrite an existing mapping.
179 void PutRegExp(Handle<String> source, JSRegExp::Flags flags,
180 DirectHandle<RegExpData> data);
181
182 // Clear the cache - also used to initialize the cache at startup.
183 void Clear();
184
185 // Remove given shared function info from all caches.
186 void Remove(DirectHandle<SharedFunctionInfo> function_info);
187
188 // GC support.
189 void Iterate(RootVisitor* v);
190
191 // Notify the cache that a mark-sweep garbage collection is about to
192 // take place. This is used to retire entries from the cache to
193 // avoid keeping them alive too long without using them.
194 void MarkCompactPrologue();
195
196 // Enable/disable compilation cache. Used by debugger to disable compilation
197 // cache during debugging so that eval and new scripts are always compiled.
198 // TODO(bmeurer, chromium:992277): The RegExp cache cannot be enabled and/or
199 // disabled, since it doesn't affect debugging. However ideally the other
200 // caches should also be always on, even in the presence of the debugger,
201 // but at this point there are too many unclear invariants, and so I decided
202 // to just fix the pressing performance problem for RegExp individually first.
203 void EnableScriptAndEval();
204 void DisableScriptAndEval();
205
206 private:
207 explicit CompilationCache(Isolate* isolate);
208 ~CompilationCache() = default;
209
210 base::HashMap* EagerOptimizingSet();
211
212 bool IsEnabledScriptAndEval() const {
213 return v8_flags.compilation_cache && enabled_script_and_eval_;
214 }
215 bool IsEnabledScript(LanguageMode language_mode) {
216 // Tests can change v8_flags.use_strict at runtime. The compilation cache
217 // only contains scripts which were compiled with the default language mode.
218 return IsEnabledScriptAndEval() && language_mode == LanguageMode::kSloppy;
219 }

231 bool enabled_script_and_eval_;
