35void Disassemble(base::Vector<const uint8_t> wire_bytes,
36 v8::debug::DisassemblyCollector* collector,
37 std::vector<int>* function_body_offsets) {
38 std::unique_ptr<OffsetsProvider> offsets = AllocateOffsetsProvider();
39 ModuleResult result =
40 DecodeWasmModuleForDisassembler(wire_bytes, offsets.get());
41 MultiLineStringBuilder out;
42 AccountingAllocator allocator;
43 if (result.failed()) {
44 WasmError error = result.error();
45 out << "Decoding error: " << error.message() << " at offset "
46 << error.offset();
47 out.ToDisassemblyCollector(collector);
48 return;
49 }
50 const WasmModule* module = result.value().get();
51 NamesProvider names(module, wire_bytes);
52 ModuleWireBytes module_bytes(wire_bytes);
53 ModuleDisassembler md(out, module, &names, module_bytes, &allocator,
54 std::move(offsets), function_body_offsets);
55 md.PrintModule({0, 2}, v8_flags.wasm_disassembly_max_mb);
56 out.ToDisassemblyCollector(collector);
57}

70 base::Vector<const uint8_t> function_body,
71 ModuleWireBytes module_bytes, NamesProvider* names,
72 std::ostream& os, std::vector<uint32_t>* offsets) {

82 module_bytes, names);
83 d.DecodeAsWat(sb, {0, 2}, FunctionBodyDisassembler::kPrintHeader);
84 const bool print_offsets = false;
85 sb.WriteTo(os, print_offsets, offsets);

120// Returns the log2 of the alignment, e.g. "4" means 2<<4 == 16 bytes.
121// This is the same format as used in .wasm binary modules.
122uint32_t GetDefaultAlignment(WasmOpcode opcode) {
123 switch (opcode) {
124 case kExprS128LoadMem:
125 case kExprS128StoreMem:
126 return 4;
127 case kExprS128Load8x8S:
128 case kExprS128Load8x8U:
129 case kExprS128Load16x4S:
130 case kExprS128Load16x4U:
131 case kExprS128Load32x2S:
132 case kExprS128Load32x2U:
133 case kExprS128Load64Splat:
134 case kExprS128Load64Zero:
135 case kExprS128Load64Lane:
136 case kExprS128Store64Lane:
137 return 3;
138 case kExprS128Load32Splat:
139 case kExprS128Load32Zero:
140 case kExprS128Load32Lane:
141 case kExprS128Store32Lane:
142 return 2;
143 case kExprS128Load16Splat:
144 case kExprS128Load16Lane:
145 case kExprS128Store16Lane:
146 return 1;
147 case kExprS128Load8Splat:
148 case kExprS128Load8Lane:
149 case kExprS128Store8Lane:
150 return 0;
151
152#define CASE(Opcode, ...) \
153 case kExpr##Opcode: \
154 return GetLoadType(kExpr##Opcode).size_log_2();
155 FOREACH_LOAD_MEM_OPCODE(CASE)
156#undef CASE
157#define CASE(Opcode, ...) \
158 case kExpr##Opcode: \
159 return GetStoreType(kExpr##Opcode).size_log_2();
160 FOREACH_STORE_MEM_OPCODE(CASE)
161#undef CASE
162
163#define CASE(Opcode, Type) \
164 case kExpr##Opcode: \
165 return ElementSizeLog2Of(MachineType::Type().representation());
166 ATOMIC_OP_LIST(CASE)
167 ATOMIC_STORE_OP_LIST(CASE)
168#undef CASE
169
170 default:
171 UNREACHABLE();
172 }
173}

190 out << " ";
191 names->PrintValueType(out, sig->GetParam(i));
192 }
193 out << ")";
194 }
195 for (size_t i = 0; i < sig->return_count(); i++) {
196 out << " (result ";
197 names->PrintValueType(out, sig->GetReturn(i));
198 out << ")";
199 }

217void FunctionBodyDisassembler::DecodeAsWat(MultiLineStringBuilder& out,
218 Indentation indentation,
219 FunctionHeader include_header,
220 uint32_t* first_instruction_offset) {
221 out_ = &out;
222 int base_indentation = indentation.current();
223 // Print header.
224 if (include_header == kPrintHeader) {
225 out << indentation << "(func ";
226 names_->PrintFunctionName(out, func_index_, NamesProvider::kDevTools);
227 PrintSignatureOneLine(out, sig_, func_index_, names_, true,
228 kIndicesAsComments);
229 out.NextLine(pc_offset());
230 } else {
231 out.set_current_line_bytecode_offset(pc_offset());
232 }
233 indentation.increase();
234
235 // Decode and print locals.
236 uint32_t locals_length = DecodeLocals(pc_);
237 if (failed()) {
238 // TODO(jkummerow): Improve error handling.
239 out << "Failed to decode locals\n";
240 return;
241 }
242 for (uint32_t i = static_cast<uint32_t>(sig_->parameter_count());
243 i < num_locals_; i++) {
244 out << indentation << "(local ";
245 names_->PrintLocalName(out, func_index_, i);
246 out << " ";
247 names_->PrintValueType(out, local_type(i));
248 out << ")";
249 out.NextLine(pc_offset());
250 }
251 consume_bytes(locals_length);
252 out.set_current_line_bytecode_offset(pc_offset());
253 if (first_instruction_offset) *first_instruction_offset = pc_offset();
254
255 // Main loop.
256 while (pc_ < end_ && ok()) {
257 WasmOpcode opcode = GetOpcode();
258 current_opcode_ = opcode; // Some immediates need to know this.
259
260 // Deal with indentation.
261 if (opcode == kExprEnd || opcode == kExprElse || opcode == kExprCatch ||
262 opcode == kExprCatchAll || opcode == kExprDelegate) {
263 if (indentation.current() >= base_indentation) {
264 indentation.decrease();
265 }
266 }
267 out << indentation;
268 if (opcode == kExprElse || opcode == kExprCatch ||
269 opcode == kExprCatchAll || opcode == kExprBlock || opcode == kExprIf ||
270 opcode == kExprLoop || opcode == kExprTry || opcode == kExprTryTable) {
271 indentation.increase();
272 }
273
274 // Print the opcode and its immediates.
275 if (opcode == kExprEnd) {
276 if (indentation.current() < base_indentation) {
277 out << ";; Unexpected end byte";
278 } else if (indentation.current() == base_indentation) {
279 out << ")"; // End of the function.
280 } else {
281 out << "end";
282 const LabelInfo& label = label_stack_.back();
283 if (label.start != nullptr) {
284 out << " ";
285 out.write(label.start, label.length);
286 }
287 label_stack_.pop_back();
288 }
289 } else {
290 out << WasmOpcodes::OpcodeName(opcode);
291 }
292 if (opcode == kExprBlock || opcode == kExprIf || opcode == kExprLoop ||
293 opcode == kExprTry || opcode == kExprTryTable) {
294 // Create the LabelInfo now to get the correct offset, but only push it
295 // after printing the immediates because the immediates don't see the new
296 // label yet.
297 LabelInfo label(out.line_number(), out.length(),
298 label_occurrence_index_++);
299 pc_ += PrintImmediatesAndGetLength(out);
300 label_stack_.push_back(label);
301 } else {
302 pc_ += PrintImmediatesAndGetLength(out);
303 }
304
305 out.NextLine(pc_offset());
306 }
307
308 if (pc_ != end_) {
309 // TODO(jkummerow): Improve error handling.
310 out << "Beyond end of code";
311 }
312}

314void FunctionBodyDisassembler::DecodeGlobalInitializer(StringBuilder& out) {
315 while (pc_ < end_) {
316 WasmOpcode opcode = GetOpcode();
317 current_opcode_ = opcode; // Some immediates need to know this.
318 // Don't print the final "end".
319 if (opcode == kExprEnd && pc_ + 1 == end_) break;
320 uint32_t length;
321 out << " (" << WasmOpcodes::OpcodeName(opcode);
322 length = PrintImmediatesAndGetLength(out);
323 out << ")";
324 pc_ += length;
325 }

334void FunctionBodyDisassembler::PrintHexNumber(StringBuilder& out,
335 uint64_t number) {
336 constexpr size_t kBufferSize = sizeof(number) * 2 + 2; // +2 for "0x".
337 char buffer[kBufferSize];
338 char* end = buffer + kBufferSize;
339 char* ptr = end;
340 do {
341 *(--ptr) = kHexChars[number & 0xF];
342 number >>= 4;
343 } while (number > 0);
344 *(--ptr) = 'x';
345 *(--ptr) = '0';
346 size_t length = static_cast<size_t>(end - ptr);
347 char* output = out.allocate(length);
348 memcpy(output, ptr, length);
349}

360 void PrintDepthAsLabel(int imm_depth) {
361 out_ << " ";
362 size_t label_start_position = out_.length();
363 int depth = imm_depth;
364 if (owner_->current_opcode_ == kExprDelegate) depth++;
365 // Be robust: if the module is invalid, print what we got.
366 if (depth < 0 || depth >= static_cast<int>(owner_->label_stack_.size())) {
367 out_ << imm_depth;
368 return;
369 }
370 // If the label's name has already been determined and backpatched, just
371 // copy it here.
372 LabelInfo& label_info = owner_->label_info(depth);
373 if (label_info.start != nullptr) {
374 out_.write(label_info.start, label_info.length);
375 return;
376 }
377 // Determine the label's name and backpatch the line that opened the block.
378 names()->PrintLabelName(out_, owner_->func_index_,
379 label_info.name_section_index,
380 owner_->label_generation_index_++);
381 label_info.length = out_.length() - label_start_position;
382 owner_->out_->PatchLabel(label_info, out_.start() + label_start_position);
383 }

754void ModuleDisassembler::PrintTypeDefinition(uint32_t type_index,
755 Indentation indentation,
756 IndexAsComment index_as_comment) {
757 uint32_t offset = offsets_->type_offset(type_index);
758 out_.NextLine(offset);
759 out_ << indentation << "(type ";
760 names_->PrintTypeName(out_, type_index, index_as_comment);
761 const TypeDefinition& type = module_->types[type_index];
762 bool has_super = type.supertype != kNoSuperType;
763 if (has_super) {
764 out_ << " (sub ";
765 if (type.is_final) out_ << "final ";
766 names_->PrintHeapType(out_, HeapType(type.supertype));
767 }

769 const ArrayType* atype = type.array_type;
770 out_ << " (array";
771 if (type.is_shared) out_ << " shared";
772 out_ << " (field ";
773 PrintMutableType(atype->mutability(), atype->element_type());
774 out_ << ")"; // Closes "(field ...".
775 } else if (type.kind == TypeDefinition::kStruct) {
776 const StructType* stype = type.struct_type;
777 out_ << " (struct";

781 LineBreakOrSpace(break_lines, indentation, offset);
782 out_ << "(field ";
783 names_->PrintFieldName(out_, type_index, i);
784 out_ << " ";
785 PrintMutableType(stype->mutability(i), stype->field(i));
786 out_ << ")";
787 }
788 } else if (type.kind == TypeDefinition::kFunction) {
789 const FunctionSig* sig = type.function_sig;
790 out_ << " (func";
791 if (type.is_shared) out_ << " shared";
792 bool break_lines = sig->parameter_count() + sig->return_count() > 2;
793 for (uint32_t i = 0; i < sig->parameter_count(); i++) {
794 LineBreakOrSpace(break_lines, indentation, offset);

796 names_->PrintLocalName(out_, type_index, i);
797 out_ << " ";
798 names_->PrintValueType(out_, sig->GetParam(i));
799 out_ << ")";
800 }
801 for (uint32_t i = 0; i < sig->return_count(); i++) {
802 LineBreakOrSpace(break_lines, indentation, offset);
803 out_ << "(result ";
804 names_->PrintValueType(out_, sig->GetReturn(i));
805 out_ << ")";
806 }
807 }
808 // Closes "(type", "(sub", and "(array" / "(struct" / "(func".
809 out_ << (has_super ? ")))" : "))");

812void ModuleDisassembler::PrintModule(Indentation indentation, size_t max_mb) {
813 // 0. General infrastructure.
814 // We don't store import/export information on {WasmTag} currently.
815 size_t num_tags = module_->tags.size();
816 std::vector<bool> exported_tags(num_tags, false);
817 for (const WasmExport& ex : module_->export_table) {
818 if (ex.kind == kExternalTag) exported_tags[ex.index] = true;
819 }
820
821 // I. Module name.
822 out_ << indentation << "(module";
823 if (module_->name.is_set()) {
824 out_ << " $";
825 const uint8_t* name_start = start_ + module_->name.offset();
826 out_.write(name_start, module_->name.length());
827 }
828 indentation.increase();
829
830 // II. Types
831 uint32_t recgroup_index = 0;
832 OffsetsProvider::RecGroup recgroup = offsets_->recgroup(recgroup_index++);
833 bool in_explicit_recgroup = false;
834 for (uint32_t i = 0; i < module_->types.size(); i++) {
835 // No need to check {recgroup.valid()}, as the comparison will simply
836 // never be true otherwise.
837 while (i == recgroup.start_type_index) {
838 out_.NextLine(recgroup.offset);
839 out_ << indentation << "(rec";
840 if V8_UNLIKELY (recgroup.end_type_index == i) {
841 // Empty recgroup.
842 out_ << ")";
843 DCHECK(!in_explicit_recgroup);
844 recgroup = offsets_->recgroup(recgroup_index++);
845 continue;
846 } else {
847 in_explicit_recgroup = true;
848 indentation.increase();
849 break;
850 }
851 }
852 if (kSkipFunctionTypesInTypeSection &&
853 module_->has_signature(ModuleTypeIndex{i}) && !in_explicit_recgroup) {
854 continue;
855 }
856 PrintTypeDefinition(i, indentation, kIndicesAsComments);
857 if (in_explicit_recgroup && i == recgroup.end_type_index - 1) {
858 in_explicit_recgroup = false;
859 indentation.decrease();
860 // The end of a recgroup is implicit in the wire bytes, so repeat the
861 // previous line's offset for it.
862 uint32_t offset = out_.current_line_bytecode_offset();
863 out_.NextLine(offset);
864 out_ << indentation << ")";
865 recgroup = offsets_->recgroup(recgroup_index++);
866 }
867 }
868 while (recgroup.valid()) {
869 // There could be empty recgroups at the end of the type section.
870 DCHECK_GE(recgroup.start_type_index, module_->types.size());
871 DCHECK_EQ(recgroup.start_type_index, recgroup.end_type_index);
872 out_.NextLine(recgroup.offset);
873 out_ << indentation << "(rec)";
874 recgroup = offsets_->recgroup(recgroup_index++);
875 }
876
877 // III. Imports
878 for (uint32_t i = 0; i < module_->import_table.size(); i++) {
879 const WasmImport& import = module_->import_table[i];
880 out_.NextLine(offsets_->import_offset(i));
881 out_ << indentation;
882 switch (import.kind) {
883 case kExternalTable: {
884 out_ << "(table ";
885 names_->PrintTableName(out_, import.index, kIndicesAsComments);
886 const WasmTable& table = module_->tables[import.index];
887 if (table.exported) PrintExportName(kExternalTable, import.index);
888 PrintImportName(import);
889 PrintTable(table);
890 break;
891 }
892 case kExternalFunction: {
893 out_ << "(func ";
894 names_->PrintFunctionName(out_, import.index, NamesProvider::kDevTools,
895 kIndicesAsComments);
896 const WasmFunction& func = module_->functions[import.index];
897 if (func.exported) PrintExportName(kExternalFunction, import.index);
898 PrintImportName(import);
899 PrintSignatureOneLine(out_, func.sig, import.index, names_, false);
900 break;
901 }
902 case kExternalGlobal: {
903 out_ << "(global ";
904 names_->PrintGlobalName(out_, import.index, kIndicesAsComments);
905 const WasmGlobal& global = module_->globals[import.index];
906 if (global.exported) PrintExportName(kExternalGlobal, import.index);
907 PrintImportName(import);
908 PrintGlobal(global);
909 break;
910 }
911 case kExternalMemory:
912 out_ << "(memory ";
913 names_->PrintMemoryName(out_, import.index, kIndicesAsComments);
914 if (module_->memories[import.index].exported) {
915 PrintExportName(kExternalMemory, 0);
916 }
917 PrintImportName(import);
918 PrintMemory(module_->memories[import.index]);
919 break;
920 case kExternalTag:
921 out_ << "(tag ";
922 names_->PrintTagName(out_, import.index, kIndicesAsComments);
923 PrintImportName(import);
924 if (exported_tags[import.index]) {
925 PrintExportName(kExternalTag, import.index);
926 }
927 PrintTagSignature(module_->tags[import.index].sig);
928 break;
929 }
930 out_ << ")";
931 }
932
933 // IV. Tables
934 for (uint32_t i = module_->num_imported_tables; i < module_->tables.size();
935 i++) {
936 const WasmTable& table = module_->tables[i];
937 DCHECK(!table.imported);
938 out_.NextLine(offsets_->table_offset(i));
939 out_ << indentation << "(table ";
940 names_->PrintTableName(out_, i, kIndicesAsComments);
941 if (table.exported) PrintExportName(kExternalTable, i);
942 PrintTable(table);
943 out_ << ")";
944 }
945
946 // V. Memories
947 uint32_t num_memories = static_cast<uint32_t>(module_->memories.size());
948 for (uint32_t memory_index = 0; memory_index < num_memories; ++memory_index) {
949 const WasmMemory& memory = module_->memories[memory_index];
950 if (memory.imported) continue;
951 out_.NextLine(offsets_->memory_offset());
952 out_ << indentation << "(memory ";
953 names_->PrintMemoryName(out_, memory_index, kIndicesAsComments);
954 if (memory.exported) PrintExportName(kExternalMemory, memory_index);
955 PrintMemory(memory);
956 out_ << ")";
957 }
958
959 // VI.Tags
960 for (uint32_t i = module_->num_imported_tags; i < module_->tags.size(); i++) {
961 const WasmTag& tag = module_->tags[i];
962 out_.NextLine(offsets_->tag_offset(i));
963 out_ << indentation << "(tag ";
964 names_->PrintTagName(out_, i, kIndicesAsComments);
965 if (exported_tags[i]) PrintExportName(kExternalTag, i);
966 PrintTagSignature(tag.sig);
967 out_ << ")";
968 }
969
970 // VII. String literals
971 size_t num_strings = module_->stringref_literals.size();
972 for (uint32_t i = 0; i < num_strings; i++) {
973 const WasmStringRefLiteral lit = module_->stringref_literals[i];
974 out_.NextLine(offsets_->string_offset(i));
975 out_ << indentation << "(string \"";
976 PrintString(lit.source);
977 out_ << '"';
978 if (kIndicesAsComments) out_ << " (;" << i << ";)";
979 out_ << ")";
980 }
981
982 // VIII. Globals
983 for (uint32_t i = module_->num_imported_globals; i < module_->globals.size();
984 i++) {
985 const WasmGlobal& global = module_->globals[i];
986 DCHECK(!global.imported);
987 out_.NextLine(offsets_->global_offset(i));
988 out_ << indentation << "(global ";
989 names_->PrintGlobalName(out_, i, kIndicesAsComments);
990 if (global.exported) PrintExportName(kExternalGlobal, i);
991 PrintGlobal(global);
992 PrintInitExpression(global.init, global.type);
993 out_ << ")";
994 }
995
996 // IX. Start
997 if (module_->start_function_index >= 0) {
998 out_.NextLine(offsets_->start_offset());
999 out_ << indentation << "(start ";
1000 names_->PrintFunctionName(out_, module_->start_function_index,
1001 NamesProvider::kDevTools);
1002 out_ << ")";
1003 }
1004
1005 // X. Elements
1006 for (uint32_t i = 0; i < module_->elem_segments.size(); i++) {
1007 const WasmElemSegment& elem = module_->elem_segments[i];
1008 out_.NextLine(offsets_->element_offset(i));
1009 out_ << indentation << "(elem ";
1010 names_->PrintElementSegmentName(out_, i, kIndicesAsComments);
1011 if (elem.status == WasmElemSegment::kStatusDeclarative) {
1012 out_ << " declare";
1013 } else if (elem.status == WasmElemSegment::kStatusActive) {
1014 if (elem.table_index != 0) {
1015 out_ << " (table ";
1016 names_->PrintTableName(out_, elem.table_index);
1017 out_ << ")";
1018 }
1019 PrintInitExpression(elem.offset, kWasmI32);
1020 }
1021 out_ << " ";
1022 if (elem.shared) out_ << "shared ";
1023 names_->PrintValueType(out_, elem.type);
1024
1025 WasmDetectedFeatures unused_detected_features;
1026 ModuleDecoderImpl decoder(
1027 WasmEnabledFeatures::All(), wire_bytes_.module_bytes(),
1028 ModuleOrigin::kWasmOrigin, &unused_detected_features);
1029 decoder.consume_bytes(elem.elements_wire_bytes_offset);
1030 for (size_t j = 0; j < elem.element_count; j++) {
1031 ConstantExpression entry = decoder.consume_element_segment_entry(
1032 const_cast<WasmModule*>(module_), elem);
1033 PrintInitExpression(entry, elem.type);
1034 }
1035 out_ << ")";
1036 }
1037
1038 // For the FunctionBodyDisassembler, we flip the convention: {NextLine} is
1039 // now called *after* printing something, instead of before.
1040 if (out_.length() != 0) out_.NextLine(0);
1041
1042 // XI. Code / function bodies.
1043 if (function_body_offsets_ != nullptr) {
1044 size_t num_defined_functions =
1045 module_->functions.size() - module_->num_imported_functions;
1046 function_body_offsets_->reserve(num_defined_functions * 2);
1047 }
1048 for (uint32_t i = module_->num_imported_functions;
1049 i < module_->functions.size(); i++) {
1050 const WasmFunction* func = &module_->functions[i];
1051 out_.set_current_line_bytecode_offset(func->code.offset());
1052 out_ << indentation << "(func ";
1053 names_->PrintFunctionName(out_, i, NamesProvider::kDevTools,
1054 kIndicesAsComments);
1055 if (func->exported) PrintExportName(kExternalFunction, i);
1056 PrintSignatureOneLine(out_, func->sig, i, names_, true, kIndicesAsComments);
1057 out_.NextLine(func->code.offset());
1058 bool shared = module_->type(func->sig_index).is_shared;
1059 WasmDetectedFeatures detected;
1060 base::Vector<const uint8_t> code = wire_bytes_.GetFunctionBytes(func);
1061 FunctionBodyDisassembler d(&zone_, module_, i, shared, &detected, func->sig,
1062 code.begin(), code.end(), func->code.offset(),
1063 wire_bytes_, names_);
1064 uint32_t first_instruction_offset;
1065 d.DecodeAsWat(out_, indentation, FunctionBodyDisassembler::kSkipHeader,
1066 &first_instruction_offset);
1067 if (function_body_offsets_ != nullptr) {
1068 function_body_offsets_->push_back(first_instruction_offset);
1069 function_body_offsets_->push_back(d.pc_offset());
1070 }
1071 if (out_.ApproximateSizeMB() > max_mb) {
1072 out_ << "<truncated...>";
1073 return;
1074 }
1075 }
1076
1077 // XII. Data
1078 for (uint32_t i = 0; i < module_->data_segments.size(); i++) {
1079 const WasmDataSegment& data = module_->data_segments[i];
1080 out_.set_current_line_bytecode_offset(offsets_->data_offset(i));
1081 out_ << indentation << "(data";
1082 if (!kSkipDataSegmentNames) {
1083 out_ << " ";
1084 names_->PrintDataSegmentName(out_, i, kIndicesAsComments);
1085 }
1086 if (data.shared) out_ << " shared";
1087 if (data.active) {
1088 ValueType type = module_->memories[data.memory_index].is_memory64()
1089 ? kWasmI64
1090 : kWasmI32;
1091 PrintInitExpression(data.dest_addr, type);
1092 }
1093 out_ << " \"";
1094 PrintString(data.source);
1095 out_ << "\")";
1096 out_.NextLine(0);
1097
1098 if (out_.ApproximateSizeMB() > max_mb) {
1099 out_ << "<truncated...>";
1100 return;
1101 }
1102 }
1103
1104 indentation.decrease();
1105 out_.set_current_line_bytecode_offset(
1106 static_cast<uint32_t>(wire_bytes_.length()));
1107 out_ << indentation << ")"; // End of the module.
1108 out_.NextLine(0);
1109}
