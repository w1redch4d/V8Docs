118template <bool capture_raw, bool unicode>
119base::uc32 Scanner::ScanHexNumber(int expected_length) {
120 DCHECK_LE(expected_length, 4); // prevent overflow
121
122 int begin = source_pos() - 2;
123 base::uc32 x = 0;
124 for (int i = 0; i < expected_length; i++) {
125 int d = base::HexValue(c0_);
126 if (d < 0) {
127 ReportScannerError(Location(begin, begin + expected_length + 2),
128 unicode
129 ? MessageTemplate::kInvalidUnicodeEscapeSequence
130 : MessageTemplate::kInvalidHexEscapeSequence);
131 return Invalid();
132 }
133 x = x * 16 + d;
134 Advance<capture_raw>();
135 }
136
137 return x;
138}

161Token::Value Scanner::Next() {
162 // Rotate through tokens.
163 TokenDesc* previous = current_;
164 current_ = next_;
165 // Either we already have the next token lined up, in which case next_next_
166 // simply becomes next_. In that case we use current_ as new next_next_ and
167 // clear its token to indicate that it wasn't scanned yet. Otherwise we use
168 // current_ as next_ and scan into it, leaving next_next_ uninitialized.
169 if (V8_LIKELY(next_next().token == Token::kUninitialized)) {
170 DCHECK(next_next_next().token == Token::kUninitialized);
171 next_ = previous;
172 // User 'previous' instead of 'next_' because for some reason the compiler
173 // thinks 'next_' could be modified before the entry into Scan.
174 previous->after_line_terminator = false;
175 Scan(previous);
176 } else {
177 next_ = next_next_;
178
179 if (V8_LIKELY(next_next_next().token == Token::kUninitialized)) {
180 next_next_ = previous;
181 } else {
182 next_next_ = next_next_next_;
183 next_next_next_ = previous;
184 }
185
186 previous->token = Token::kUninitialized;
187 DCHECK_NE(Token::kUninitialized, current().token);
188 }
189 return current().token;
190}

192Token::Value Scanner::PeekAhead() {
193 DCHECK(next().token != Token::kDiv);
194 DCHECK(next().token != Token::kAssignDiv);
195
196 if (next_next().token != Token::kUninitialized) {
197 return next_next().token;
198 }
199 TokenDesc* temp = next_;
200 next_ = next_next_;
201 next().after_line_terminator = false;
202 Scan();
203 next_next_ = next_;
204 next_ = temp;
205 return next_next().token;
206}

210 return next_next_next().token;

252void Scanner::TryToParseMagicComment(base::uc32 hash_or_at_sign) {
253 // Magic comments are of the form: //[#@]\s<name>=\s*<value>\s*.* and this
254 // function will just return if it cannot parse a magic comment.
255 DCHECK(!IsWhiteSpaceOrLineTerminator(kEndOfInput));
256 if (!IsWhiteSpace(c0_)) return;
257 Advance();
258 LiteralBuffer name;
259 name.Start();
260
261 while (c0_ != kEndOfInput && !IsWhiteSpaceOrLineTerminator(c0_) &&
262 c0_ != '=') {
263 name.AddChar(c0_);
264 Advance();
265 }
266 if (!name.is_one_byte()) return;
267 base::Vector<const uint8_t> name_literal = name.one_byte_literal();
268 LiteralBuffer* value;
269 LiteralBuffer compile_hints_value;
270 if (name_literal == base::StaticOneByteVector("sourceURL")) {
271 value = &source_url_;
272 } else if (name_literal == base::StaticOneByteVector("sourceMappingURL")) {
273 value = &source_mapping_url_;
274 DCHECK(hash_or_at_sign == '#' || hash_or_at_sign == '@');
275 saw_source_mapping_url_magic_comment_at_sign_ = hash_or_at_sign == '@';
276 } else if (name_literal == base::StaticOneByteVector("eagerCompilation")) {
277 value = &compile_hints_value;
278 } else {
279 return;
280 }
281 if (c0_ != '=')
282 return;
283 value->Start();
284 Advance();
285 while (IsWhiteSpace(c0_)) {
286 Advance();
287 }
288 while (c0_ != kEndOfInput && !unibrow::IsLineTerminator(c0_)) {
289 if (IsWhiteSpace(c0_)) {
290 break;
291 }
292 value->AddChar(c0_);
293 Advance();
294 }
295 // Allow whitespace at the end.
296 while (c0_ != kEndOfInput && !unibrow::IsLineTerminator(c0_)) {
297 if (!IsWhiteSpace(c0_)) {
298 value->Start();
299 break;
300 }
301 Advance();
302 }
303 if (value == &compile_hints_value && compile_hints_value.is_one_byte()) {
304 base::Vector<const uint8_t> value_literal =
305 compile_hints_value.one_byte_literal();
306 if (value_literal == base::StaticOneByteVector("all")) {
307 saw_magic_comment_compile_hints_all_ = true;
308 }
309 }
310}

312Token::Value Scanner::SkipMultiLineComment() {
313 DCHECK_EQ(c0_, '*');
314
315 // Until we see the first newline, check for * and newline characters.
316 if (!next().after_line_terminator) {
317 do {
318 AdvanceUntil([](base::uc32 c0) {
319 if (V8_UNLIKELY(static_cast<uint32_t>(c0) > kMaxAscii)) {
320 return unibrow::IsLineTerminator(c0);
321 }
322 uint8_t char_flags = character_scan_flags[c0];
323 return MultilineCommentCharacterNeedsSlowPath(char_flags);
324 });
325
326 while (c0_ == '*') {
327 Advance();
328 if (c0_ == '/') {
329 Advance();
330 return Token::kWhitespace;
331 }
332 }
333
334 if (unibrow::IsLineTerminator(c0_)) {
335 next().after_line_terminator = true;
336 break;
337 }
338 } while (c0_ != kEndOfInput);
339 }
340
341 // After we've seen newline, simply try to find '*/'.
342 while (c0_ != kEndOfInput) {
343 AdvanceUntil([](base::uc32 c0) { return c0 == '*'; });
344
345 while (c0_ == '*') {
346 Advance();
347 if (c0_ == '/') {
348 Advance();
349 return Token::kWhitespace;
350 }
351 }
352 }
353
354 return Token::kIllegal;
355}

390void Scanner::SeekForward(int pos) {
391 // After this call, we will have the token at the given position as
392 // the "next" token. The "current" token will be invalid.
393 if (pos == next().location.beg_pos) return;
394 int current_pos = source_pos();
395 DCHECK_EQ(next().location.end_pos, current_pos);
396 // Positions inside the lookahead token aren't supported.
397 DCHECK(pos >= current_pos);
398 if (pos != current_pos) {
399 source_->Seek(pos);
400 Advance();
401 // This function is only called to seek to the location
402 // of the end of a function (at the "}" token). It doesn't matter
403 // whether there was a line terminator in the part we skip.
404 next().after_line_terminator = false;
405 }
406 Scan();
407}

409template <bool capture_raw>
410bool Scanner::ScanEscape() {
411 base::uc32 c = c0_;
412 Advance<capture_raw>();
413
414 // Skip escaped newlines.
415 DCHECK(!unibrow::IsLineTerminator(kEndOfInput));
416 if (!capture_raw && unibrow::IsLineTerminator(c)) {
417 // Allow escaped CR+LF newlines in multiline string literals.
418 if (IsCarriageReturn(c) && IsLineFeed(c0_)) Advance();
419 return true;
420 }
421
422 switch (c) {
423 case 'b' : c = '\b'; break;
424 case 'f' : c = '\f'; break;
425 case 'n' : c = '\n'; break;
426 case 'r' : c = '\r'; break;
427 case 't' : c = '\t'; break;
428 case 'u' : {
429 c = ScanUnicodeEscape<capture_raw>();
430 if (IsInvalid(c)) return false;
431 break;
432 }
433 case 'v':
434 c = '\v';
435 break;
436 case 'x': {
437 c = ScanHexNumber<capture_raw>(2);
438 if (IsInvalid(c)) return false;
439 break;
440 }
441 case '0':
442 case '1':
443 case '2':
444 case '3':
445 case '4':
446 case '5':
447 case '6':
448 case '7':
449 c = ScanOctalEscape<capture_raw>(c, 2);
450 break;
451 case '8':
452 case '9':
453 // '\8' and '\9' are disallowed in strict mode.
454 // Re-use the octal error state to propagate the error.
455 octal_pos_ = Location(source_pos() - 2, source_pos() - 1);
456 octal_message_ = capture_raw ? MessageTemplate::kTemplate8Or9Escape
457 : MessageTemplate::kStrict8Or9Escape;
458 break;
459 }
460
461 // Other escaped characters are interpreted as their non-escaped version.
462 AddLiteralChar(c);
463 return true;
464}

466template <bool capture_raw>
467base::uc32 Scanner::ScanOctalEscape(base::uc32 c, int length) {
468 DCHECK('0' <= c && c <= '7');
469 base::uc32 x = c - '0';
470 int i = 0;
471 for (; i < length; i++) {
472 int d = c0_ - '0';
473 if (d < 0 || d > 7) break;
474 int nx = x * 8 + d;
475 if (nx >= 256) break;
476 x = nx;
477 Advance<capture_raw>();
478 }
479 // Anything except '\0' is an octal escape sequence, illegal in strict mode.
480 // Remember the position of octal escape sequences so that an error
481 // can be reported later (in strict mode).
482 // We don't report the error immediately, because the octal escape can
483 // occur before the "use strict" directive.
484 if (c != '0' || i > 0 || IsNonOctalDecimalDigit(c0_)) {
485 octal_pos_ = Location(source_pos() - i - 1, source_pos() - 1);
486 octal_message_ = capture_raw ? MessageTemplate::kTemplateOctalLiteral
487 : MessageTemplate::kStrictOctalEscape;
488 }
489 return x;
490}

492Token::Value Scanner::ScanString() {
493 base::uc32 quote = c0_;
494
495 next().literal_chars.Start();
496 while (true) {
497 AdvanceUntil([this](base::uc32 c0) {
498 if (V8_UNLIKELY(static_cast<uint32_t>(c0) > kMaxAscii)) {
499 if (V8_UNLIKELY(unibrow::IsStringLiteralLineTerminator(c0))) {
500 return true;
501 }
502 AddLiteralChar(c0);
503 return false;
504 }
505 uint8_t char_flags = character_scan_flags[c0];
506 if (MayTerminateString(char_flags)) return true;
507 AddLiteralChar(c0);
508 return false;
509 });
510
511 while (c0_ == '\\') {
512 Advance();
513 // TODO(verwaest): Check whether we can remove the additional check.
514 if (V8_UNLIKELY(c0_ == kEndOfInput || !ScanEscape<false>())) {
515 return Token::kIllegal;
516 }
517 }
518
519 if (c0_ == quote) {
520 Advance();
521 return Token::kString;
522 }
523
524 if (V8_UNLIKELY(c0_ == kEndOfInput ||
525 unibrow::IsStringLiteralLineTerminator(c0_))) {
526 return Token::kIllegal;
527 }
528
529 AddLiteralChar(c0_);
530 }
531}

550Token::Value Scanner::ScanTemplateSpan() {
551 // When scanning a TemplateSpan, we are looking for the following construct:
552 // kTemplateSpan ::
553 // ` LiteralChars* ${
554 // | } LiteralChars* ${
555 //
556 // kTemplateTail ::
557 // ` LiteralChars* `
558 // | } LiteralChar* `
559 //
560 // A kTemplateSpan should always be followed by an Expression, while a
561 // kTemplateTail terminates a TemplateLiteral and does not need to be
562 // followed by an Expression.
563
564 // These scoped helpers save and restore the original error state, so that we
565 // can specially treat invalid escape sequences in templates (which are
566 // handled by the parser).
567 ErrorState scanner_error_state(&scanner_error_, &scanner_error_location_);
568 ErrorState octal_error_state(&octal_message_, &octal_pos_);
569
570 Token::Value result = Token::kTemplateSpan;
571 next().literal_chars.Start();
572 next().raw_literal_chars.Start();
573 const bool capture_raw = true;
574 while (true) {
575 base::uc32 c = c0_;
576 if (c == '`') {
577 Advance(); // Consume '`'
578 result = Token::kTemplateTail;
579 break;
580 } else if (c == '$' && Peek() == '{') {
581 Advance(); // Consume '$'
582 Advance(); // Consume '{'
583 break;
584 } else if (c == '\\') {
585 Advance(); // Consume '\\'
586 DCHECK(!unibrow::IsLineTerminator(kEndOfInput));
587 if (capture_raw) AddRawLiteralChar('\\');
588 if (unibrow::IsLineTerminator(c0_)) {
589 // The TV of LineContinuation :: \ LineTerminatorSequence is the empty
590 // code unit sequence.
591 base::uc32 lastChar = c0_;
592 Advance();
593 if (lastChar == '\r') {
594 // Also skip \n.
595 if (c0_ == '\n') Advance();
596 lastChar = '\n';
597 }
598 if (capture_raw) AddRawLiteralChar(lastChar);
599 } else {
600 bool success = ScanEscape<capture_raw>();
601 USE(success);
602 DCHECK_EQ(!success, has_error());
603 // For templates, invalid escape sequence checking is handled in the
604 // parser.
605 scanner_error_state.MoveErrorTo(next_);
606 octal_error_state.MoveErrorTo(next_);
607 }
608 } else if (c == kEndOfInput) {
609 // Unterminated template literal
610 break;
611 } else {
612 Advance(); // Consume c.
613 // The TRV of LineTerminatorSequence :: <CR> is the CV 0x000A.
614 // The TRV of LineTerminatorSequence :: <CR><LF> is the sequence
615 // consisting of the CV 0x000A.
616 if (c == '\r') {
617 if (c0_ == '\n') Advance(); // Consume '\n'
618 c = '\n';
619 }
620 if (capture_raw) AddRawLiteralChar(c);
621 AddLiteralChar(c);
622 }
623 }
624 next().location.end_pos = source_pos();
625 next().token = result;
626
627 return result;
628}

654bool Scanner::ScanDigitsWithNumericSeparators(bool (*predicate)(base::uc32 ch),

657 if (is_check_first_digit && !predicate(c0_)) return false;

781Token::Value Scanner::ScanNumber(bool seen_period) {
782 DCHECK(IsDecimalDigit(c0_)); // the first digit of the number or the fraction
783
784 NumberKind kind = DECIMAL;
785
786 next().literal_chars.Start();
787 bool at_start = !seen_period;
788 int start_pos = source_pos(); // For reporting octal positions.
789 if (seen_period) {
790 // we have already seen a decimal point of the float
791 AddLiteralChar('.');
792 if (c0_ == '_') {
793 return Token::kIllegal;
794 }
795 // we know we have at least one digit
796 if (!ScanDecimalDigits(true)) return Token::kIllegal;
797 } else {
798 // if the first character is '0' we must check for octals and hex
799 if (c0_ == '0') {
800 AddLiteralCharAdvance();
801
802 // either 0, 0exxx, 0Exxx, 0.xxx, a hex number, a binary number or
803 // an octal number.
804 if (AsciiAlphaToLower(c0_) == 'x') {
805 AddLiteralCharAdvance();
806 kind = HEX;
807 if (!ScanHexDigits()) return Token::kIllegal;
808 } else if (AsciiAlphaToLower(c0_) == 'o') {
809 AddLiteralCharAdvance();
810 kind = OCTAL;
811 if (!ScanOctalDigits()) return Token::kIllegal;
812 } else if (AsciiAlphaToLower(c0_) == 'b') {
813 AddLiteralCharAdvance();
814 kind = BINARY;
815 if (!ScanBinaryDigits()) return Token::kIllegal;
816 } else if (IsOctalDigit(c0_)) {
817 kind = IMPLICIT_OCTAL;
818 if (!ScanImplicitOctalDigits(start_pos, &kind)) {
819 return Token::kIllegal;
820 }
821 if (kind == DECIMAL_WITH_LEADING_ZERO) {
822 at_start = false;
823 }
824 } else if (IsNonOctalDecimalDigit(c0_)) {
825 kind = DECIMAL_WITH_LEADING_ZERO;
826 } else if (c0_ == '_') {
827 ReportScannerError(Location(source_pos(), source_pos() + 1),
828 MessageTemplate::kZeroDigitNumericSeparator);
829 return Token::kIllegal;
830 }
831 }
832
833 // Parse decimal digits and allow trailing fractional part.
834 if (IsDecimalNumberKind(kind)) {
835 bool allow_numeric_separator = kind != DECIMAL_WITH_LEADING_ZERO;
836 // This is an optimization for parsing Decimal numbers as Smi's.
837 if (at_start) {
838 uint64_t value = 0;
839 // scan subsequent decimal digits
840 if (!ScanDecimalAsSmi(&value, allow_numeric_separator)) {
841 return Token::kIllegal;
842 }
843
844 if (next().literal_chars.one_byte_literal().length() <= 10 &&
845 value <= Smi::kMaxValue && c0_ != '.' && !IsIdentifierStart(c0_)) {
846 next().smi_value = static_cast<uint32_t>(value);
847
848 if (kind == DECIMAL_WITH_LEADING_ZERO) {
849 octal_pos_ = Location(start_pos, source_pos());
850 octal_message_ = MessageTemplate::kStrictDecimalWithLeadingZero;
851 }
852 return Token::kSmi;
853 }
854 }
855
856 if (!ScanDecimalDigits(allow_numeric_separator)) {
857 return Token::kIllegal;
858 }
859 if (c0_ == '.') {
860 seen_period = true;
861 AddLiteralCharAdvance();
862 if (c0_ == '_') {
863 return Token::kIllegal;
864 }
865 if (!ScanDecimalDigits(true)) return Token::kIllegal;
866 }
867 }
868 }
869
870 bool is_bigint = false;
871 if (c0_ == 'n' && !seen_period && IsValidBigIntKind(kind)) {
872 // Check that the literal is within our limits for BigInt length.
873 // For simplicity, use 4 bits per character to calculate the maximum
874 // allowed literal length.
875 static const int kMaxBigIntCharacters = BigInt::kMaxLengthBits / 4;
876 int length = source_pos() - start_pos - (kind != DECIMAL ? 2 : 0);
877 if (length > kMaxBigIntCharacters) {
878 ReportScannerError(Location(start_pos, source_pos()),
879 MessageTemplate::kBigIntTooBig);
880 return Token::kIllegal;
881 }
882
883 is_bigint = true;
884 Advance();
885 } else if (AsciiAlphaToLower(c0_) == 'e') {
886 // scan exponent, if any
887 DCHECK_NE(kind, HEX); // 'e'/'E' must be scanned as part of the hex number
888
889 if (!IsDecimalNumberKind(kind)) return Token::kIllegal;
890
891 // scan exponent
892 AddLiteralCharAdvance();
893
894 if (!ScanSignedInteger()) return Token::kIllegal;
895 }
896
897 // The source character immediately following a numeric literal must
898 // not be an identifier start or a decimal digit; see ECMA-262
899 // section 7.8.3, page 17 (note that we read only one decimal digit
900 // if the value is 0).
901 if (IsDecimalDigit(c0_) || IsIdentifierStart(c0_)) {
902 return Token::kIllegal;
903 }
904
905 if (kind == DECIMAL_WITH_LEADING_ZERO) {
906 octal_pos_ = Location(start_pos, source_pos());
907 octal_message_ = MessageTemplate::kStrictDecimalWithLeadingZero;
908 }
909
910 next().number_kind = kind;
911 return is_bigint ? Token::kBigInt : Token::kNumber;
912}

989bool Scanner::ScanRegExpPattern() {
990 DCHECK_EQ(Token::kUninitialized, next_next().token);
991 DCHECK(next().token == Token::kDiv || next().token == Token::kAssignDiv);
992
993 // Scan: ('/' | '/=') RegularExpressionBody '/' RegularExpressionFlags
994 bool in_character_class = false;
995
996 // Scan regular expression body: According to ECMA-262, 3rd, 7.8.5,
997 // the scanner should pass uninterpreted bodies to the RegExp
998 // constructor.
999 next().literal_chars.Start();
1000 if (next().token == Token::kAssignDiv) {
1001 AddLiteralChar('=');
1002 }
1003
1004 while (c0_ != '/' || in_character_class) {
1005 if (c0_ == kEndOfInput || unibrow::IsLineTerminator(c0_)) {
1006 return false;
1007 }
1008 if (c0_ == '\\') { // Escape sequence.
1009 AddLiteralCharAdvance();
1010 if (c0_ == kEndOfInput || unibrow::IsLineTerminator(c0_)) {
1011 return false;
1012 }
1013 AddLiteralCharAdvance();
1014 // If the escape allows more characters, i.e., \x??, \u????, or \c?,
1015 // only "safe" characters are allowed (letters, digits, underscore),
1016 // otherwise the escape isn't valid and the invalid character has
1017 // its normal meaning. I.e., we can just continue scanning without
1018 // worrying whether the following characters are part of the escape
1019 // or not, since any '/', '\\' or '[' is guaranteed to not be part
1020 // of the escape sequence.
1021 } else { // Unescaped character.
1022 if (c0_ == '[') in_character_class = true;
1023 if (c0_ == ']') in_character_class = false;
1024 AddLiteralCharAdvance();
1025 }
1026 }
1027 Advance(); // consume '/'
1028
1029 next().token = Token::kRegExpLiteral;
1030 return true;
1031}

1034 DCHECK_EQ(Token::kRegExpLiteral, next().token);
1035
1036 RegExpFlags flags;

1047 next().location.end_pos = source_pos();
