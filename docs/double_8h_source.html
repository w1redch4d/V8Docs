22// Helper functions for doubles.
23class Double {
24 public:
25 static constexpr uint64_t kSignMask = 0x8000'0000'0000'0000;
26 static constexpr uint64_t kExponentMask = 0x7FF0'0000'0000'0000;
27 static constexpr uint64_t kSignificandMask = 0x000F'FFFF'FFFF'FFFF;
28 static constexpr uint64_t kHiddenBit = 0x0010'0000'0000'0000;
29 static constexpr int kPhysicalSignificandSize =
30 52; // Excludes the hidden bit.
31 static constexpr int kSignificandSize = 53;
32
33 Double() : d64_(0) {}
34 explicit Double(double d) : d64_(double_to_uint64(d)) {}
35 explicit Double(uint64_t d64) : d64_(d64) {}
36 explicit Double(DiyFp diy_fp) : d64_(DiyFpToUint64(diy_fp)) {}
37
38 // The value encoded by this Double must be greater or equal to +0.0.
39 // It must not be special (infinity, or NaN).
40 DiyFp AsDiyFp() const {
41 DCHECK_GT(Sign(), 0);
42 DCHECK(!IsSpecial());
43 return DiyFp(Significand(), Exponent());
44 }
45
46 // The value encoded by this Double must be strictly greater than 0.
47 DiyFp AsNormalizedDiyFp() const {
48 DCHECK_GT(value(), 0.0);
49 uint64_t f = Significand();
50 int e = Exponent();
51
52 // The current double could be a denormal.
53 while ((f & kHiddenBit) == 0) {
54 f <<= 1;
55 e--;
56 }
57 // Do the final shifts in one go.
58 f <<= DiyFp::kSignificandSize - kSignificandSize;
59 e -= DiyFp::kSignificandSize - kSignificandSize;
60 return DiyFp(f, e);
61 }
62
63 // Returns the double's bit as uint64.
64 uint64_t AsUint64() const { return d64_; }
65
66 // Returns the next greater double. Returns +infinity on input +infinity.
67 double NextDouble() const {
68 if (d64_ == kInfinity) return Double(kInfinity).value();
69 if (Sign() < 0 && Significand() == 0) {
70 // -0.0
71 return 0.0;
72 }
73 if (Sign() < 0) {
74 return Double(d64_ - 1).value();
75 } else {
76 return Double(d64_ + 1).value();
77 }
78 }
79
80 int Exponent() const {
81 if (IsDenormal()) return kDenormalExponent;
82
83 uint64_t d64 = AsUint64();
84 int biased_e =
85 static_cast<int>((d64 & kExponentMask) >> kPhysicalSignificandSize);
86 return biased_e - kExponentBias;
87 }
88
89 uint64_t Significand() const {
90 uint64_t d64 = AsUint64();
91 uint64_t significand = d64 & kSignificandMask;
92 if (!IsDenormal()) {
93 return significand + kHiddenBit;
94 } else {
95 return significand;
96 }
97 }
98
99 // Returns true if the double is a denormal.
100 bool IsDenormal() const {
101 uint64_t d64 = AsUint64();
102 return (d64 & kExponentMask) == 0;
103 }
104
105 // We consider denormals not to be special.
106 // Hence only Infinity and NaN are special.
107 bool IsSpecial() const {
108 uint64_t d64 = AsUint64();
109 return (d64 & kExponentMask) == kExponentMask;
110 }
111
112 bool IsInfinite() const {
113 uint64_t d64 = AsUint64();
114 return ((d64 & kExponentMask) == kExponentMask) &&
115 ((d64 & kSignificandMask) == 0);
116 }
117
118 int Sign() const {
119 uint64_t d64 = AsUint64();
120 return (d64 & kSignMask) == 0 ? 1 : -1;
121 }
122
123 // Precondition: the value encoded by this Double must be greater or equal
124 // than +0.0.
125 DiyFp UpperBoundary() const {
126 DCHECK_GT(Sign(), 0);
127 return DiyFp(Significand() * 2 + 1, Exponent() - 1);
128 }
129
130 // Returns the two boundaries of this.
131 // The bigger boundary (m_plus) is normalized. The lower boundary has the same
132 // exponent as m_plus.
133 // Precondition: the value encoded by this Double must be greater than 0.
134 void NormalizedBoundaries(DiyFp* out_m_minus, DiyFp* out_m_plus) const {
135 DCHECK_GT(value(), 0.0);
136 DiyFp v = this->AsDiyFp();
137 DiyFp m_plus = DiyFp::Normalize(DiyFp((v.f() << 1) + 1, v.e() - 1));
138 DiyFp m_minus;
139 if ((AsUint64() & kSignificandMask) == 0 && v.e() != kDenormalExponent) {
140 // The boundary is closer. Think of v = 1000e10 and v- = 9999e9.
141 // Then the boundary (== (v - v-)/2) is not just at a distance of 1e9 but
142 // at a distance of 1e8.
143 // The only exception is for the smallest normal: the largest denormal is
144 // at the same distance as its successor.
145 // Note: denormals have the same exponent as the smallest normals.
146 m_minus = DiyFp((v.f() << 2) - 1, v.e() - 2);
147 } else {
148 m_minus = DiyFp((v.f() << 1) - 1, v.e() - 1);
149 }
150 m_minus.set_f(m_minus.f() << (m_minus.e() - m_plus.e()));
151 m_minus.set_e(m_plus.e());
152 *out_m_plus = m_plus;
153 *out_m_minus = m_minus;
154 }
155
156 double value() const { return uint64_to_double(d64_); }
157
158 // Returns the significand size for a given order of magnitude.
159 // If v = f*2^e with 2^p-1 <= f <= 2^p then p+e is v's order of magnitude.
160 // This function returns the number of significant binary digits v will have
161 // once its encoded into a double. In almost all cases this is equal to
162 // kSignificandSize. The only exception are denormals. They start with leading
163 // zeroes and their effective significand-size is hence smaller.
164 static int SignificandSizeForOrderOfMagnitude(int order) {
165 if (order >= (kDenormalExponent + kSignificandSize)) {
166 return kSignificandSize;
167 }
168 if (order <= kDenormalExponent) return 0;
169 return order - kDenormalExponent;
170 }
171
172 private:
173 static constexpr int kExponentBias = 0x3FF + kPhysicalSignificandSize;
174 static constexpr int kDenormalExponent = -kExponentBias + 1;
175 static constexpr int kMaxExponent = 0x7FF - kExponentBias;
176 static constexpr uint64_t kInfinity = 0x7FF0'0000'0000'0000;
177
178 // The field d64_ is not marked as const to permit the usage of the copy
179 // constructor.
180 uint64_t d64_;
181
182 static uint64_t DiyFpToUint64(DiyFp diy_fp) {
183 uint64_t significand = diy_fp.f();
184 int exponent = diy_fp.e();
185 while (significand > kHiddenBit + kSignificandMask) {
186 significand >>= 1;
187 exponent++;
188 }
189 if (exponent >= kMaxExponent) {
190 return kInfinity;
191 }
192 if (exponent < kDenormalExponent) {
193 return 0;
194 }
195 while (exponent > kDenormalExponent && (significand & kHiddenBit) == 0) {
196 significand <<= 1;
197 exponent--;
198 }
199 uint64_t biased_exponent;
200 if (exponent == kDenormalExponent && (significand & kHiddenBit) == 0) {
201 biased_exponent = 0;
202 } else {
203 biased_exponent = static_cast<uint64_t>(exponent + kExponentBias);
204 }
205 return (significand & kSignificandMask) |
206 (biased_exponent << kPhysicalSignificandSize);
207 }
208};
