90size_t IncrementalMarkingSchedule::GetNextIncrementalStepDuration(
91 size_t estimated_live_bytes) {
92 last_estimated_live_bytes_ = estimated_live_bytes;
93 DCHECK(!incremental_marking_start_time_.IsNull());
94 const auto elapsed_time = GetElapsedTime();
95 const size_t last_marked_bytes = current_step_.marked_bytes();
96 const size_t actual_marked_bytes = GetOverallMarkedBytes();
97 const size_t expected_marked_bytes =
98 std::ceil(estimated_live_bytes * elapsed_time.InMillisecondsF() /
99 kEstimatedMarkingTime.InMillisecondsF());
100 // Stash away the current data for others to access.
101 current_step_ = {mutator_thread_marked_bytes_, GetConcurrentlyMarkedBytes(),
102 estimated_live_bytes, expected_marked_bytes, elapsed_time};
103 if ((actual_marked_bytes >= last_marked_bytes) &&
104 (actual_marked_bytes - last_marked_bytes) <
105 kStepSizeWhenNotMakingProgress) {
106 return std::max(kStepSizeWhenNotMakingProgress, min_marked_bytes_per_step_);
107 }
108 if (expected_marked_bytes < actual_marked_bytes) {
109 // Marking is ahead of schedule, incremental marking should do the minimum.
110 return min_marked_bytes_per_step_;
111 }
112 // Assuming marking will take |kEstimatedMarkingTime|, overall there will
113 // be |estimated_live_bytes| live bytes to mark, and that marking speed is
114 // constant, after |elapsed_time| the number of marked_bytes should be
115 // |estimated_live_bytes| * (|elapsed_time| / |kEstimatedMarkingTime|),
116 // denoted as |expected_marked_bytes|. If |actual_marked_bytes| is less,
117 // i.e. marking is behind schedule, incremental marking should help "catch
118 // up" by marking (|expected_marked_bytes| - |actual_marked_bytes|).
119 return std::max(min_marked_bytes_per_step_,
120 expected_marked_bytes - actual_marked_bytes);
121}
