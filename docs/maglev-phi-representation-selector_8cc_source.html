30BlockProcessResult MaglevPhiRepresentationSelector::PreProcessBasicBlock(
31 BasicBlock* block) {
32 PreparePhiTaggings(current_block_, block);
33 current_block_ = block;
34
35 if (block->has_phi()) {
36 auto& phis = *block->phis();
37
38 auto first_retry = phis.begin();
39 auto end_retry = first_retry;
40 bool any_change = false;
41
42 for (auto it = phis.begin(); it != phis.end(); ++it) {
43 Phi* phi = *it;
44 switch (ProcessPhi(phi)) {
45 case ProcessPhiResult::kNone:
46 break;
47 case ProcessPhiResult::kChanged:
48 any_change = true;
49 break;
50 case ProcessPhiResult::kRetryOnChange:
51 if (end_retry == first_retry) {
52 first_retry = it;
53 }
54 end_retry = it;
55 ++end_retry;
56 break;
57 }
58 }
59 // Give it one more shot in case an earlier phi has a later one as input.
60 if (any_change) {
61 for (auto it = first_retry; it != end_retry; ++it) {
62 ProcessPhi(*it);
63 }
64 }
65 }
66
67 return BlockProcessResult::kContinue;
68}

70bool MaglevPhiRepresentationSelector::CanHoistUntaggingTo(BasicBlock* block) {
71 if (block->successors().size() != 1) return false;
72 BasicBlock* next = block->successors()[0];
73 // To be able to hoist above resumable loops we would have to be able to
74 // convert during resumption.
75 return !next->state()->is_resumable_loop();
76}

78MaglevPhiRepresentationSelector::ProcessPhiResult
79MaglevPhiRepresentationSelector::ProcessPhi(Phi* node) {
80 if (node->value_representation() != ValueRepresentation::kTagged) {
81 return ProcessPhiResult::kNone;
82 }
83
84 if (node->is_exception_phi()) {
85 // Exception phis have no inputs (or, at least, none accessible through
86 // `node->input(...)`), so we don't know if the inputs could be untagged or
87 // not, so we just keep those Phis tagged.
88 return ProcessPhiResult::kNone;
89 }
90
91 TRACE_UNTAGGING(
92 "Considering for untagging: " << PrintNodeLabel(graph_labeller(), node));
93
94 // {input_mask} represents the ValueRepresentation that {node} could have,
95 // based on the ValueRepresentation of its inputs.
96 ValueRepresentationSet input_reprs;
97 HoistTypeList hoist_untagging;
98 hoist_untagging.resize_and_init(node->input_count(), HoistType::kNone);
99
100 bool has_tagged_phi_input = false;
101 for (int i = 0; i < node->input_count(); i++) {
102 ValueNode* input = node->input(i).node();
103 if (input->Is<SmiConstant>()) {
104 // Could be any representation. We treat such inputs as Int32, since we
105 // later allow ourselves to promote Int32 to Float64 if needed (but we
106 // never downgrade Float64 to Int32, as it could cause deopt loops).
107 input_reprs.Add(ValueRepresentation::kInt32);
108 } else if (Constant* constant = input->TryCast<Constant>()) {
109 if (constant->object().IsHeapNumber()) {
110 input_reprs.Add(ValueRepresentation::kFloat64);
111 } else {
112 // Not a Constant that we can untag.
113 // TODO(leszeks): Consider treating 'undefined' as a potential
114 // HoleyFloat64.
115 input_reprs.RemoveAll();
116 break;
117 }
118 } else if (input->properties().is_conversion()) {
119 DCHECK_EQ(input->input_count(), 1);
120 // The graph builder tags all Phi inputs, so this conversion should
121 // produce a tagged value.
122 DCHECK_EQ(input->value_representation(), ValueRepresentation::kTagged);
123 // If we want to untag {node}, then we'll drop the conversion and use its
124 // input instead.
125 input_reprs.Add(
126 input->input(0).node()->properties().value_representation());
127 } else if (Phi* input_phi = input->TryCast<Phi>()) {
128 if (input_phi->value_representation() != ValueRepresentation::kTagged) {
129 input_reprs.Add(input_phi->value_representation());
130 } else {
131 // An untagged phi is an input of the current phi.
132 if (node->is_backedge_offset(i) &&
133 node->merge_state()->is_loop_with_peeled_iteration()) {
134 // This is the backedge of a loop that has a peeled iteration. We
135 // ignore it and speculatively assume that it will be the same as the
136 // 1st input.
137 DCHECK_EQ(node->input_count(), 2);
138 DCHECK_EQ(i, 1);
139 break;
140 }
141 has_tagged_phi_input = true;
142 input_reprs.RemoveAll();
143 break;
144 }
145 } else {
146 // This is the case where we don't have an existing conversion to attach
147 // the untagging to. In the general case we give up, however in the
148 // special case of the value originating from the loop entry branch, we
149 // can try to hoist untagging out of the loop.
150 if (builder_->graph()->is_osr() &&
151 v8_flags.maglev_hoist_osr_value_phi_untagging &&
152 input->Is<InitialValue>() &&
153 CanHoistUntaggingTo(*builder_->graph()->begin())) {
154 hoist_untagging[i] = HoistType::kPrologue;
155 continue;
156 }
157 if (node->is_loop_phi() && !node->is_backedge_offset(i)) {
158 BasicBlock* pred = node->merge_state()->predecessor_at(i);
159 if (CanHoistUntaggingTo(pred)) {
160 auto static_type = StaticTypeForNode(
161 builder_->broker(), builder_->local_isolate(), input);
162 if (NodeTypeIs(static_type, NodeType::kSmi)) {
163 input_reprs.Add(ValueRepresentation::kInt32);
164 hoist_untagging[i] = HoistType::kLoopEntryUnchecked;
165 continue;
166 }
167 if (NodeTypeIs(static_type, NodeType::kNumber)) {
168 input_reprs.Add(ValueRepresentation::kFloat64);
169 hoist_untagging[i] = HoistType::kLoopEntryUnchecked;
170 continue;
171 }
172
173 // TODO(olivf): Unless we untag OSR values, speculatively untagging
174 // could end us in deopt loops. To enable this by default we need to
175 // add some feedback to be able to back off. Or, ideally find the
176 // respective checked conversion from within the loop to wire up the
177 // feedback collection.
178 if (v8_flags.maglev_speculative_hoist_phi_untagging) {
179 // TODO(olivf): Currently there is no hard guarantee that the phi
180 // merge state has a checkpointed jump.
181 if (pred->control_node()->Is<CheckpointedJump>()) {
182 DCHECK(!node->merge_state()->is_resumable_loop());
183 hoist_untagging[i] = HoistType::kLoopEntry;
184 continue;
185 }
186 }
187 }
188 }
189
190 // This input is tagged, didn't require a tagging operation to be
191 // tagged and we decided not to hosit; we won't untag {node}.
192 // TODO(dmercadier): this is a bit suboptimal, because some nodes start
193 // tagged, and later become untagged (parameters for instance). Such nodes
194 // will have their untagged alternative passed to {node} without any
195 // explicit conversion, and we thus won't untag {node} even though we
196 // could have.
197 input_reprs.RemoveAll();
198 break;
199 }
200 }
201 ProcessPhiResult default_result = has_tagged_phi_input
202 ? ProcessPhiResult::kRetryOnChange
203 : ProcessPhiResult::kNone;
204
205 UseRepresentationSet use_reprs;
206 if (node->is_loop_phi() && !node->get_same_loop_uses_repr_hints().empty()) {
207 // {node} is a loop phi that has uses inside the loop; we will tag/untag
208 // based on those uses, ignoring uses after the loop.
209 use_reprs = node->get_same_loop_uses_repr_hints();
210 } else {
211 use_reprs = node->get_uses_repr_hints();
212 }
213
214 TRACE_UNTAGGING(" + use_reprs : " << use_reprs);
215 TRACE_UNTAGGING(" + input_reprs: " << input_reprs);
216
217 if (use_reprs.contains(UseRepresentation::kTagged) ||
218 use_reprs.contains(UseRepresentation::kUint32) || use_reprs.empty()) {
219 // We don't untag phis that are used as tagged (because we'd have to retag
220 // them later). We also ignore phis that are used as Uint32, because this is
221 // a fairly rare case and supporting it doesn't improve performance all that
222 // much but will increase code complexity.
223 // TODO(dmercadier): consider taking into account where those Tagged uses
224 // are: Tagged uses outside of a loop or for a Return could probably be
225 // ignored.
226 TRACE_UNTAGGING(" => Leaving tagged [incompatible uses]");
227 EnsurePhiInputsTagged(node);
228 return default_result;
229 }
230
231 if (input_reprs.contains(ValueRepresentation::kTagged) ||
232 input_reprs.contains(ValueRepresentation::kUint32) ||
233 input_reprs.empty()) {
234 TRACE_UNTAGGING(" => Leaving tagged [tagged or uint32 inputs]");
235 EnsurePhiInputsTagged(node);
236 return default_result;
237 }
238
239 // Only allowed to have Int32, Float64 and HoleyFloat64 inputs from here.
240 DCHECK_EQ(input_reprs -
241 ValueRepresentationSet({ValueRepresentation::kInt32,
242 ValueRepresentation::kFloat64,
243 ValueRepresentation::kHoleyFloat64}),
244 ValueRepresentationSet());
245
246 DCHECK_EQ(
247 use_reprs - UseRepresentationSet({UseRepresentation::kInt32,
248 UseRepresentation::kTruncatedInt32,
249 UseRepresentation::kFloat64,
250 UseRepresentation::kHoleyFloat64}),
251 UseRepresentationSet());
252
253 // The rules for untagging are that we can only widen input representations,
254 // i.e. promote Int32 -> Float64 -> HoleyFloat64.
255 //
256 // Inputs can always be used as more generic uses, and tighter uses always
257 // block more generic inputs. So, we can find the minimum generic use and
258 // maximum generic input, extend inputs upwards, uses downwards, and convert
259 // to the least generic use in the intersection.
260 //
261 // Of interest is the fact that we don't want to insert conversions which
262 // reduce genericity, e.g. Float64->Int32 conversions, since they could deopt
263 // and lead to deopt loops. The above logic ensures that if a Phi has Float64
264 // inputs and Int32 uses, we simply don't untag it.
265 //
266 // TODO(leszeks): The above logic could be implemented with bit magic if the
267 // representations were contiguous.
268
269 ValueRepresentationSet possible_inputs;
270 if (input_reprs.contains(ValueRepresentation::kHoleyFloat64)) {
271 possible_inputs = {ValueRepresentation::kHoleyFloat64};
272 } else if (input_reprs.contains(ValueRepresentation::kFloat64)) {
273 possible_inputs = {ValueRepresentation::kFloat64,
274 ValueRepresentation::kHoleyFloat64};
275 } else {
276 DCHECK(input_reprs.contains_only(ValueRepresentation::kInt32));
277 possible_inputs = {ValueRepresentation::kInt32,
278 ValueRepresentation::kFloat64,
279 ValueRepresentation::kHoleyFloat64};
280 }
281
282 ValueRepresentationSet allowed_inputs_for_uses;
283 if (use_reprs.contains(UseRepresentation::kInt32)) {
284 allowed_inputs_for_uses = {ValueRepresentation::kInt32};
285 } else if (use_reprs.contains(UseRepresentation::kFloat64)) {
286 allowed_inputs_for_uses = {ValueRepresentation::kInt32,
287 ValueRepresentation::kFloat64};
288 } else {
289 DCHECK(!use_reprs.empty() &&
290 use_reprs.is_subset_of({UseRepresentation::kHoleyFloat64,
291 UseRepresentation::kTruncatedInt32}));
292 allowed_inputs_for_uses = {ValueRepresentation::kInt32,
293 ValueRepresentation::kFloat64,
294 ValueRepresentation::kHoleyFloat64};
295 }
296
297 // When hoisting we must ensure that we don't turn a tagged flowing into
298 // CheckedSmiUntag into a float64. This would cause us to loose the smi check
299 // which in turn can invalidate assumptions on aliasing values.
300 if (hoist_untagging.size() && node->uses_require_31_bit_value()) {
301 allowed_inputs_for_uses.Remove(
302 {ValueRepresentation::kFloat64, ValueRepresentation::kHoleyFloat64});
303 }
304
305 auto intersection = possible_inputs & allowed_inputs_for_uses;
306
307 TRACE_UNTAGGING(" + intersection reprs: " << intersection);
308 if (intersection.contains(ValueRepresentation::kInt32)) {
309 TRACE_UNTAGGING(" => Untagging to Int32");
310 ConvertTaggedPhiTo(node, ValueRepresentation::kInt32, hoist_untagging);
311 return ProcessPhiResult::kChanged;
312 } else if (intersection.contains(ValueRepresentation::kFloat64)) {
313 TRACE_UNTAGGING(" => Untagging to kFloat64");
314 ConvertTaggedPhiTo(node, ValueRepresentation::kFloat64, hoist_untagging);
315 return ProcessPhiResult::kChanged;
316 } else if (intersection.contains(ValueRepresentation::kHoleyFloat64)) {
317 TRACE_UNTAGGING(" => Untagging to HoleyFloat64");
318 ConvertTaggedPhiTo(node, ValueRepresentation::kHoleyFloat64,
319 hoist_untagging);
320 return ProcessPhiResult::kChanged;
321 }
322
323 DCHECK(intersection.empty());
324 // We don't untag the Phi.
325 TRACE_UNTAGGING(" => Leaving tagged [incompatible inputs/uses]");
326 EnsurePhiInputsTagged(node);
327 return default_result;
328}

330void MaglevPhiRepresentationSelector::EnsurePhiInputsTagged(Phi* phi) {
331 // Since we are untagging some Phis, it's possible that one of the inputs of
332 // {phi} is an untagged Phi. However, if this function is called, then we've
333 // decided that {phi} is going to stay tagged, and thus, all of its inputs
334 // should be tagged. We'll thus insert tagging operation on the untagged phi
335 // inputs of {phi}.
336
337 const int skip_backedge = phi->is_loop_phi() ? 1 : 0;
338 for (int i = 0; i < phi->input_count() - skip_backedge; i++) {
339 ValueNode* input = phi->input(i).node();
340 if (Phi* phi_input = input->TryCast<Phi>()) {
341 phi->change_input(
342 i, EnsurePhiTagged(phi_input, phi->predecessor_at(i),
343 NewNodePosition::kEndOfBlock, nullptr, i));
344 } else {
345 // Inputs of Phis that aren't Phi should always be tagged (except for the
346 // phis untagged by this class, but {phi} isn't one of them).
347 DCHECK(input->is_tagged());
348 }
349 }

354Opcode GetOpcodeForConversion(ValueRepresentation from, ValueRepresentation to,
355 bool truncating) {
356 DCHECK_NE(from, ValueRepresentation::kTagged);
357 DCHECK_NE(to, ValueRepresentation::kTagged);
358
359 switch (from) {
360 case ValueRepresentation::kInt32:
361 switch (to) {
362 case ValueRepresentation::kUint32:
363 return Opcode::kCheckedInt32ToUint32;
364 case ValueRepresentation::kFloat64:
365 case ValueRepresentation::kHoleyFloat64:
366 return Opcode::kChangeInt32ToFloat64;
367
368 case ValueRepresentation::kInt32:
369 case ValueRepresentation::kTagged:
370 case ValueRepresentation::kIntPtr:
371 UNREACHABLE();
372 }
373 case ValueRepresentation::kUint32:
374 switch (to) {
375 case ValueRepresentation::kInt32:
376 return Opcode::kCheckedUint32ToInt32;
377
378 case ValueRepresentation::kFloat64:
379 case ValueRepresentation::kHoleyFloat64:
380 return Opcode::kChangeUint32ToFloat64;
381
382 case ValueRepresentation::kUint32:
383 case ValueRepresentation::kTagged:
384 case ValueRepresentation::kIntPtr:
385 UNREACHABLE();
386 }
387 case ValueRepresentation::kFloat64:
388 switch (to) {
389 case ValueRepresentation::kInt32:
390 if (truncating) {
391 return Opcode::kTruncateFloat64ToInt32;
392 }
393 return Opcode::kCheckedTruncateFloat64ToInt32;
394 case ValueRepresentation::kUint32:
395 // The graph builder never inserts Tagged->Uint32 conversions, so we
396 // don't have to handle this case.
397 UNREACHABLE();
398 case ValueRepresentation::kHoleyFloat64:
399 return Opcode::kIdentity;
400
401 case ValueRepresentation::kFloat64:
402 case ValueRepresentation::kTagged:
403 case ValueRepresentation::kIntPtr:
404 UNREACHABLE();
405 }
406 case ValueRepresentation::kHoleyFloat64:
407 switch (to) {
408 case ValueRepresentation::kInt32:
409 // Holes are NaNs, so we can truncate them to int32 same as real NaNs.
410 if (truncating) {
411 return Opcode::kTruncateFloat64ToInt32;
412 }
413 return Opcode::kCheckedTruncateFloat64ToInt32;
414 case ValueRepresentation::kUint32:
415 // The graph builder never inserts Tagged->Uint32 conversions, so we
416 // don't have to handle this case.
417 UNREACHABLE();
418 case ValueRepresentation::kFloat64:
419 return Opcode::kHoleyFloat64ToMaybeNanFloat64;
420
421 case ValueRepresentation::kHoleyFloat64:
422 case ValueRepresentation::kTagged:
423 case ValueRepresentation::kIntPtr:
424 UNREACHABLE();
425 }
426
427 case ValueRepresentation::kTagged:
428 case ValueRepresentation::kIntPtr:
429 UNREACHABLE();
430 }
431 UNREACHABLE();
432}

436void MaglevPhiRepresentationSelector::ConvertTaggedPhiTo(
437 Phi* phi, ValueRepresentation repr, const HoistTypeList& hoist_untagging) {
438 // We currently only support Int32, Float64, and HoleyFloat64 untagged phis.
439 DCHECK(repr == ValueRepresentation::kInt32 ||
440 repr == ValueRepresentation::kFloat64 ||
441 repr == ValueRepresentation::kHoleyFloat64);
442 phi->change_representation(repr);
443 // Re-initialise register data, since we might have changed from integer
444 // registers to floating registers.
445 phi->InitializeRegisterData();
446
447 for (int i = 0; i < phi->input_count(); i++) {
448 ValueNode* input = phi->input(i).node();
449#define TRACE_INPUT_LABEL \
450 " @ Input " << i << " (" << PrintNodeLabel(graph_labeller(), input) << ")"
451
452 if (input->Is<SmiConstant>()) {
453 switch (repr) {
454 case ValueRepresentation::kInt32:
455 TRACE_UNTAGGING(TRACE_INPUT_LABEL << ": Making Int32 instead of Smi");
456 phi->change_input(i,
457 builder_->GetInt32Constant(
458 input->Cast<SmiConstant>()->value().value()));
459 break;
460 case ValueRepresentation::kFloat64:
461 case ValueRepresentation::kHoleyFloat64:
462 TRACE_UNTAGGING(TRACE_INPUT_LABEL
463 << ": Making Float64 instead of Smi");
464 phi->change_input(i,
465 builder_->GetFloat64Constant(
466 input->Cast<SmiConstant>()->value().value()));
467 break;
468 case ValueRepresentation::kUint32:
469 UNIMPLEMENTED();
470 default:
471 UNREACHABLE();
472 }
473 } else if (Constant* constant = input->TryCast<Constant>()) {
474 TRACE_UNTAGGING(TRACE_INPUT_LABEL
475 << ": Making Float64 instead of Constant");
476 DCHECK(constant->object().IsHeapNumber());
477 DCHECK(repr == ValueRepresentation::kFloat64 ||
478 repr == ValueRepresentation::kHoleyFloat64);
479 phi->change_input(i, builder_->GetFloat64Constant(
480 constant->object().AsHeapNumber().value()));
481 } else if (input->properties().is_conversion()) {
482 // Unwrapping the conversion.
483 DCHECK_EQ(input->value_representation(), ValueRepresentation::kTagged);
484 // Needs to insert a new conversion.
485 ValueNode* bypassed_input = input->input(0).node();
486 ValueRepresentation from_repr = bypassed_input->value_representation();
487 ValueNode* new_input;
488 if (from_repr == repr) {
489 TRACE_UNTAGGING(TRACE_INPUT_LABEL << ": Bypassing conversion");
490 new_input = bypassed_input;
491 } else {
492 Opcode conv_opcode =
493 GetOpcodeForConversion(from_repr, repr, /*truncating*/ false);
494 switch (conv_opcode) {
495 case Opcode::kChangeInt32ToFloat64: {
496 TRACE_UNTAGGING(
497 TRACE_INPUT_LABEL
498 << ": Replacing old conversion with a ChangeInt32ToFloat64");
499 ValueNode* new_node = NodeBase::New<ChangeInt32ToFloat64>(
500 builder_->zone(), {input->input(0).node()});
501 new_input = AddNodeAtBlockEnd(new_node, phi->predecessor_at(i));
502 break;
503 }
504 case Opcode::kIdentity:
505 TRACE_UNTAGGING(TRACE_INPUT_LABEL << ": Bypassing conversion");
506 new_input = bypassed_input;
507 break;
508 default:
509 UNREACHABLE();
510 }
511 }
512 phi->change_input(i, new_input);
513 } else if (Phi* input_phi = input->TryCast<Phi>()) {
514 ValueRepresentation from_repr = input_phi->value_representation();
515 if (from_repr == ValueRepresentation::kTagged) {
516 // We allow speculative untagging of the backedge for loop phis from
517 // loops that have been peeled.
518 // This can lead to deopt loops (eg, if after the last iteration of a
519 // loop, a loop Phi has a specific representation that it never has in
520 // the loop), but this case should (hopefully) be rare.
521
522 // We know that we are on the backedge input of a peeled loop, because
523 // if it wasn't the case, then Process(Phi*) would not have decided to
524 // untag this Phi, and this function would not have been called (because
525 // except for backedges of peeled loops, tagged inputs prevent phi
526 // untagging).
527 DCHECK(phi->merge_state()->is_loop_with_peeled_iteration());
528 DCHECK(phi->is_backedge_offset(i));
529
530 DeoptFrame* deopt_frame = phi->merge_state()->backedge_deopt_frame();
531 switch (repr) {
532 case ValueRepresentation::kInt32: {
533 phi->change_input(
534 i, AddNodeAtBlockEnd(NodeBase::New<CheckedSmiUntag>(
535 builder_->zone(), {input_phi}),
536 phi->predecessor_at(i), deopt_frame));
537 break;
538 }
539 case ValueRepresentation::kFloat64: {
540 phi->change_input(
541 i, AddNodeAtBlockEnd(
542 NodeBase::New<CheckedNumberOrOddballToFloat64>(
543 builder_->zone(), {input_phi},
544 TaggedToFloat64ConversionType::kOnlyNumber),
545 phi->predecessor_at(i), deopt_frame));
546 break;
547 }
548 case ValueRepresentation::kHoleyFloat64: {
549 phi->change_input(
550 i, AddNodeAtBlockEnd(
551 NodeBase::New<CheckedNumberOrOddballToHoleyFloat64>(
552 builder_->zone(), {input_phi},
553 TaggedToFloat64ConversionType::kNumberOrOddball),
554 phi->predecessor_at(i), deopt_frame));
555 break;
556 }
557 case ValueRepresentation::kTagged:
558 case ValueRepresentation::kIntPtr:
559 case ValueRepresentation::kUint32:
560 UNREACHABLE();
561 }
562 TRACE_UNTAGGING(TRACE_INPUT_LABEL
563 << ": Eagerly untagging Phi on backedge");
564 } else if (from_repr != repr &&
565 from_repr == ValueRepresentation::kInt32) {
566 // We allow widening of Int32 inputs to Float64, which can lead to the
567 // current Phi having a Float64 representation but having some Int32
568 // inputs, which will require an Int32ToFloat64 conversion.
569 DCHECK(repr == ValueRepresentation::kFloat64 ||
570 repr == ValueRepresentation::kHoleyFloat64);
571 phi->change_input(i,
572 AddNodeAtBlockEnd(NodeBase::New<ChangeInt32ToFloat64>(
573 builder_->zone(), {input_phi}),
574 phi->predecessor_at(i)));
575 TRACE_UNTAGGING(
576 TRACE_INPUT_LABEL
577 << ": Converting phi input with a ChangeInt32ToFloat64");
578 } else {
579 // We allow Float64 to silently be used as HoleyFloat64.
580 DCHECK_IMPLIES(from_repr != repr,
581 from_repr == ValueRepresentation::kFloat64 &&
582 repr == ValueRepresentation::kHoleyFloat64);
583 TRACE_UNTAGGING(TRACE_INPUT_LABEL
584 << ": Keeping untagged Phi input as-is");
585 }
586 } else if (hoist_untagging[i] != HoistType::kNone) {
587 CHECK_EQ(input->value_representation(), ValueRepresentation::kTagged);
588 BasicBlock* block;
589 DeoptFrame* deopt_frame;
590 auto GetDeoptFrame = [](BasicBlock* block) {
591 return &block->control_node()
592 ->Cast<CheckpointedJump>()
593 ->eager_deopt_info()
594 ->top_frame();
595 };
596 switch (hoist_untagging[i]) {
597 case HoistType::kLoopEntryUnchecked:
598 block = phi->merge_state()->predecessor_at(i);
599 deopt_frame = nullptr;
600 break;
601 case HoistType::kLoopEntry:
602 block = phi->merge_state()->predecessor_at(i);
603 deopt_frame = GetDeoptFrame(block);
604 break;
605 case HoistType::kPrologue:
606 block = *builder_->graph()->begin();
607 deopt_frame = GetDeoptFrame(block);
608 break;

626

628 untagged = AddNodeAtBlockEnd(
629 NodeBase::New<CheckedNumberOrOddballToFloat64>(
630 builder_->zone(), {input},
631 TaggedToFloat64ConversionType::kOnlyNumber),
632 block, deopt_frame);
633 untagged =
634 AddNodeAtBlockEnd(NodeBase::New<CheckedTruncateFloat64ToInt32>(
635 builder_->zone(), {untagged}),
636 block, deopt_frame);
637 }
638 break;
639 case ValueRepresentation::kFloat64:
640 case ValueRepresentation::kHoleyFloat64:

642 DCHECK(
643 NodeTypeIs(StaticTypeForNode(builder_->broker(),
644 builder_->local_isolate(), input),
645 NodeType::kNumber));
646 untagged = AddNodeAtBlockEnd(
647 NodeBase::New<UncheckedNumberOrOddballToFloat64>(
648 builder_->zone(), {input},
649 TaggedToFloat64ConversionType::kOnlyNumber),
650 block);
651 } else {
652 DCHECK(!phi->uses_require_31_bit_value());
653 untagged = AddNodeAtBlockEnd(
654 NodeBase::New<CheckedNumberOrOddballToFloat64>(
655 builder_->zone(), {input},
656 TaggedToFloat64ConversionType::kOnlyNumber),
657 block, deopt_frame);
658 if (repr != ValueRepresentation::kHoleyFloat64) {
659 untagged =

661 builder_->zone(), {untagged}),

664 }
665 break;

672 } else {

696void MaglevPhiRepresentationSelector::UpdateUntaggingOfPhi(
697 Phi* phi, ValueNode* old_untagging) {
698 DCHECK_EQ(old_untagging->input_count(), 1);
699 DCHECK(old_untagging->input(0).node()->Is<Phi>());
700
701 ValueRepresentation from_repr =
702 old_untagging->input(0).node()->value_representation();
703 ValueRepresentation to_repr = old_untagging->value_representation();
704
705 // Since initially Phis are tagged, it would make not sense for
706 // {old_conversion} to convert a Phi to a Tagged value.
707 DCHECK_NE(to_repr, ValueRepresentation::kTagged);
708 // The graph builder never inserts Tagged->Uint32 conversions (and thus, we
709 // don't handle them in GetOpcodeForCheckedConversion).
710 DCHECK_NE(to_repr, ValueRepresentation::kUint32);
711
712 if (from_repr == ValueRepresentation::kTagged) {
713 // The Phi hasn't been untagged, so we leave the conversion as it is.
714 return;
715 }
716
717 if (from_repr == to_repr) {
718 if (from_repr == ValueRepresentation::kInt32) {
719 if (phi->uses_require_31_bit_value() &&
720 old_untagging->Is<CheckedSmiUntag>()) {
721 old_untagging->OverwriteWith<CheckedSmiSizedInt32>();
722 return;
723 }
724 }
725 old_untagging->OverwriteWith<Identity>();
726 return;
727 }
728
729 if (old_untagging->Is<UnsafeSmiUntag>()) {
730 // UnsafeSmiTag are only inserted when the node is a known Smi. If the
731 // current phi has a Float64/Uint32 representation, then we can safely
732 // truncate it to Int32, because we know that the Float64/Uint32 fits in a
733 // Smi, and therefore in an Int32.
734 if (from_repr == ValueRepresentation::kFloat64 ||
735 from_repr == ValueRepresentation::kHoleyFloat64) {
736 old_untagging->OverwriteWith<UnsafeTruncateFloat64ToInt32>();
737 } else if (from_repr == ValueRepresentation::kUint32) {
738 old_untagging->OverwriteWith<UnsafeTruncateUint32ToInt32>();
739 } else {
740 DCHECK_EQ(from_repr, ValueRepresentation::kInt32);
741 old_untagging->OverwriteWith<Identity>();
742 }
743 return;
744 }
745
746 // The graph builder inserts 3 kind of Tagged->Int32 conversions that can have
747 // heap number as input: CheckedTruncateNumberToInt32, which truncates its
748 // input (and deopts if it's not a HeapNumber), TruncateNumberToInt32, which
749 // truncates its input (assuming that it's indeed a HeapNumber) and
750 // CheckedSmiTag, which deopts on non-smi inputs. The first 2 cannot deopt if
751 // we have Float64 phi and will happily truncate it, but the 3rd one should
752 // deopt if it cannot be converted without loss of precision.
753 bool conversion_is_truncating_float64 =
754 old_untagging->Is<CheckedTruncateNumberOrOddballToInt32>() ||
755 old_untagging->Is<TruncateNumberOrOddballToInt32>();
756
757 Opcode needed_conversion = GetOpcodeForConversion(
758 from_repr, to_repr, conversion_is_truncating_float64);
759
760 if (CheckedNumberOrOddballToFloat64* number_untagging =
761 old_untagging->TryCast<CheckedNumberOrOddballToFloat64>()) {
762 if (from_repr == ValueRepresentation::kHoleyFloat64 &&
763 number_untagging->conversion_type() !=
764 TaggedToFloat64ConversionType::kNumberOrOddball) {
765 // {phi} is a HoleyFloat64 (and thus, it could be a hole), but the
766 // original untagging did not allow holes.
767 needed_conversion = Opcode::kCheckedHoleyFloat64ToFloat64;
768 }
769 }
770
771 if (needed_conversion != old_untagging->opcode()) {
772 old_untagging->OverwriteWith(needed_conversion);
773 }
774}

933ValueNode* MaglevPhiRepresentationSelector::EnsurePhiTagged(
934 Phi* phi, BasicBlock* block, NewNodePosition pos,
935 const ProcessingState* state, std::optional<int> predecessor_index) {
936 DCHECK_IMPLIES(state == nullptr, pos == NewNodePosition::kEndOfBlock);
937
938 if (phi->value_representation() == ValueRepresentation::kTagged) {
939 return phi;
940 }
941
942 // Try to find an existing Tagged conversion for {phi} in {phi_taggings_}.
943 if (phi->has_key()) {
944 if (predecessor_index.has_value()) {
945 if (ValueNode* tagging = phi_taggings_.GetPredecessorValue(
946 phi->key(), predecessor_index.value())) {
947 return tagging;
948 }
949 } else {
950 if (ValueNode* tagging = phi_taggings_.Get(phi->key())) {
951 return tagging;
952 }
953 }
954 }
955
956 // We didn't already Tag {phi} on the current path; creating this tagging now.
957 ValueNode* tagged = nullptr;
958 switch (phi->value_representation()) {
959 case ValueRepresentation::kFloat64:
960 // It's important to use kCanonicalizeSmi for Float64ToTagged, as
961 // otherwise, we could end up storing HeapNumbers in Smi fields.
962 tagged = AddNode(NodeBase::New<Float64ToTagged>(
963 builder_->zone(), {phi},
964 Float64ToTagged::ConversionMode::kCanonicalizeSmi),
965 block, pos, state);
966 break;
967 case ValueRepresentation::kHoleyFloat64:
968 // It's important to use kCanonicalizeSmi for HoleyFloat64ToTagged, as
969 // otherwise, we could end up storing HeapNumbers in Smi fields.
970 tagged =
971 AddNode(NodeBase::New<HoleyFloat64ToTagged>(
972 builder_->zone(), {phi},
973 HoleyFloat64ToTagged::ConversionMode::kCanonicalizeSmi),
974 block, pos, state);
975 break;
976 case ValueRepresentation::kInt32:
977 tagged = AddNode(NodeBase::New<Int32ToNumber>(builder_->zone(), {phi}),
978 block, pos, state);
979 break;
980 case ValueRepresentation::kUint32:
981 tagged = AddNode(NodeBase::New<Uint32ToNumber>(builder_->zone(), {phi}),
982 block, pos, state);
983 break;
984 case ValueRepresentation::kTagged:
985 // Already handled at the begining of this function.
986 case ValueRepresentation::kIntPtr:
987 UNREACHABLE();
988 }
989
990 if (predecessor_index.has_value()) {
991 // We inserted the new tagging node in a predecessor of the current block,
992 // so we shouldn't update the snapshot table for the current block (and we
993 // can't update it for the predecessor either since its snapshot is sealed).
994 DCHECK_IMPLIES(block == current_block_,
995 block->is_loop() && block->successors().size() == 1 &&
996 block->successors().at(0) == block);
997 return tagged;
998 }
999
1000 if (phi->has_key()) {
1001 // The Key already existed, but wasn't set on the current path.
1002 phi_taggings_.Set(phi->key(), tagged);
1003 } else {
1004 // The Key didn't already exist, so we create it now.
1005 auto key = phi_taggings_.NewKey();
1006 phi->set_key(key);
1007 phi_taggings_.Set(key, tagged);
1008 }
1009 return tagged;
1010}

1012void MaglevPhiRepresentationSelector::FixLoopPhisBackedge(BasicBlock* block) {
1013 // TODO(dmercadier): it would be interesting to compute a fix point for loop
1014 // phis, or at least to go over the loop header twice.
1015 if (!block->has_phi()) return;
1016 for (Phi* phi : *block->phis()) {
1017 int last_input_idx = phi->input_count() - 1;
1018 ValueNode* backedge = phi->input(last_input_idx).node();
1019 if (phi->value_representation() == ValueRepresentation::kTagged) {
1020 // If the backedge is a Phi that was untagged, but {phi} is tagged, then
1021 // we need to retag the backedge.
1022
1023 // Identity nodes are used to replace outdated untagging nodes after a phi
1024 // has been untagged. Here, since the backedge was initially tagged, it
1025 // couldn't have been such an untagging node, so it shouldn't be an
1026 // Identity node now.
1027 DCHECK(!backedge->Is<Identity>());
1028
1029 if (backedge->value_representation() != ValueRepresentation::kTagged) {
1030 // Since all Phi inputs are initially tagged, the fact that the backedge
1031 // is not tagged means that it's a Phi that we recently untagged.
1032 DCHECK(backedge->Is<Phi>());
1033 phi->change_input(
1034 last_input_idx,
1035 EnsurePhiTagged(backedge->Cast<Phi>(), current_block_,
1036 NewNodePosition::kEndOfBlock, /*state*/ nullptr));
1037 }
1038 } else {
1039 // If {phi} was untagged and its backedge became Identity, then we need to
1040 // unwrap it.
1041 DCHECK_NE(phi->value_representation(), ValueRepresentation::kTagged);
1042 if (backedge->Is<Identity>()) {
1043 // {backedge} should have the same representation as {phi}, although if
1044 // {phi} has HoleyFloat64 representation, the backedge is allowed to
1045 // have Float64 representation rather than HoleyFloat64.
1046 DCHECK((backedge->input(0).node()->value_representation() ==
1047 phi->value_representation()) ||
1048 (backedge->input(0).node()->value_representation() ==
1049 ValueRepresentation::kFloat64 &&
1050 phi->value_representation() ==
1051 ValueRepresentation::kHoleyFloat64));
1052 phi->change_input(last_input_idx, backedge->input(0).node());
1053 }
1054 }
1055 }

1111void MaglevPhiRepresentationSelector::PreparePhiTaggings(
1112 BasicBlock* old_block, const BasicBlock* new_block) {
1113 // Sealing and saving current snapshot
1114 if (phi_taggings_.IsSealed()) {
1115 phi_taggings_.StartNewSnapshot();
1116 return;
1117 }
1118 old_block->SetSnapshot(phi_taggings_.Seal());
1119
1120 // Setting up new snapshot
1121 predecessors_.clear();
1122
1123 if (!new_block->is_merge_block()) {
1124 BasicBlock* pred = new_block->predecessor();
1125 predecessors_.push_back(pred->snapshot());
1126 } else {
1127 int skip_backedge = new_block->is_loop();
1128 for (int i = 0; i < new_block->predecessor_count() - skip_backedge; i++) {
1129 BasicBlock* pred = new_block->predecessor_at(i);
1130 predecessors_.push_back(pred->snapshot());
1131 }
1132 }
1133
1134 auto merge_taggings =
1135 [&](Key key, base::Vector<ValueNode* const> predecessors) -> ValueNode* {
1136 for (ValueNode* node : predecessors) {
1137 if (node == nullptr) {
1138 // There is a predecessor that doesn't have this Tagging, so we'll
1139 // return nullptr, and if we need it in the future, we'll have to
1140 // recreate it. An alternative would be to eagerly insert this Tagging
1141 // in all of the other predecesors, but it's possible that it's not used
1142 // anymore or not on all future path, so this could also introduce
1143 // unnecessary tagging.
1144 return static_cast<Phi*>(nullptr);
1145 }
1146 }
1147
1148 // Only merge blocks should require Phis.
1149 DCHECK(new_block->is_merge_block());
1150
1151 // We create a Phi to merge all of the existing taggings.
1152 int predecessor_count = new_block->predecessor_count();
1153 Phi* phi = Node::New<Phi>(builder_->zone(), predecessor_count,
1154 new_block->state(), interpreter::Register());
1155 for (int i = 0; static_cast<size_t>(i) < predecessors.size(); i++) {
1156 phi->set_input(i, predecessors[i]);
1157 }
1158 if (predecessors.size() != static_cast<size_t>(predecessor_count)) {
1159 // The backedge is omitted from {predecessors}. With set the Phi as its
1160 // own backedge.
1161 DCHECK(new_block->is_loop());
1162 phi->set_input(predecessor_count - 1, phi);
1163 }
1164 RegisterNewNode(phi);
1165 new_block->AddPhi(phi);
1166
1167 return phi;
1168 };
1169
1170 phi_taggings_.StartNewSnapshot(base::VectorOf(predecessors_), merge_taggings);
1171}
