312template <typename IsolateT>
313SharedFunctionInfo::Inlineability SharedFunctionInfo::GetInlineability(
314 IsolateT* isolate) const {
315 if (!IsScript(script())) return kHasNoScript;
316
317 if (isolate->is_precise_binary_code_coverage() &&
318 !has_reported_binary_coverage()) {
319 // We may miss invocations if this function is inlined.
320 return kNeedsBinaryCoverage;
321 }
322
323 // Built-in functions are handled by the JSCallReducer.
324 if (HasBuiltinId()) return kIsBuiltin;
325
326 if (!IsUserJavaScript()) return kIsNotUserCode;
327
328 // If there is no bytecode array, it is either not compiled or it is compiled
329 // with WebAssembly for the asm.js pipeline. In either case we don't want to
330 // inline.
331 if (!HasBytecodeArray()) return kHasNoBytecode;
332
333 if (GetBytecodeArray(isolate)->length() >
334 v8_flags.max_inlined_bytecode_size) {
335 return kExceedsBytecodeLimit;
336 }
337
338 {
339 SharedMutexGuardIfOffThread<IsolateT, base::kShared> mutex_guard(
340 isolate->shared_function_info_access(), isolate);
341 if (HasBreakInfo(isolate->GetMainThreadIsolateUnsafe())) {
342 return kMayContainBreakPoints;
343 }
344 }
345
346 if (optimization_disabled()) return kHasOptimizationDisabled;
347
348 return kIsInlineable;
349}

555 WriteBarrierMode mode) {
556 // Move the existing name onto the ScopeInfo.
557 Tagged<NameOrScopeInfoT> name_or_scope_info =
558 this->name_or_scope_info(kAcquireLoad);
559 Tagged<UnionOf<Smi, String>> name;

567 // flushing bytecode, we'll actually reinstall read-only scopeinfos on

569 // should already be fully initialized though, and hence will already have the
570 // right FunctionName (and InferredName if relevant).
571 if (scope_info->FunctionName() != name) {
572 scope_info->SetFunctionName(name);
573 }
574 if (HasInferredName() && inferred_name()->length() != 0 &&
575 scope_info->InferredFunctionName() != inferred_name()) {
576 scope_info->SetInferredFunctionName(inferred_name());

994void SharedFunctionInfo::ClearPreparseData(IsolateForSandbox isolate) {
995 DCHECK(HasUncompiledDataWithPreparseData());
996 Tagged<UncompiledDataWithPreparseData> data =
997 uncompiled_data_with_preparse_data(isolate);
998
999 // Trim off the pre-parsed scope data from the uncompiled data by swapping the
1000 // map, leaving only an uncompiled data without pre-parsed scope.
1001 DisallowGarbageCollection no_gc;
1002 Heap* heap = GetHeapFromWritableObject(data);
1003
1004 // We are basically trimming that object to its supertype, so recorded slots
1005 // within the object don't need to be invalidated.
1006 heap->NotifyObjectLayoutChange(data, no_gc, InvalidateRecordedSlots::kNo,
1007 InvalidateExternalPointerSlots::kNo);
1008 static_assert(UncompiledDataWithoutPreparseData::kSize <
1009 UncompiledDataWithPreparseData::kSize);
1010 static_assert(UncompiledDataWithoutPreparseData::kSize ==
1011 UncompiledData::kHeaderSize);
1012
1013 // Fill the remaining space with filler and clear slots in the trimmed area.
1014 heap->NotifyObjectSizeChange(data, UncompiledDataWithPreparseData::kSize,
1015 UncompiledDataWithoutPreparseData::kSize,
1016 ClearRecordedSlots::kYes);
1017
1018 // Swap the map.
1019 data->set_map(heap->isolate(),
1020 GetReadOnlyRoots().uncompiled_data_without_preparse_data_map(),
1021 kReleaseStore);
1022
1023 // Ensure that the clear was successful.
1024 DCHECK(HasUncompiledDataWithoutPreparseData());
1025}
