158// static
159constexpr Builtin Builtins::CEntry(int result_size, ArgvMode argv_mode,
160 bool builtin_exit_frame,
161 bool switch_to_central_stack) {
162 // Aliases for readability below.
163 const int rs = result_size;
164 const ArgvMode am = argv_mode;
165 const bool be = builtin_exit_frame;
166
167 if (switch_to_central_stack) {
168 DCHECK_EQ(result_size, 1);
169 DCHECK_EQ(argv_mode, ArgvMode::kStack);
170 DCHECK_EQ(builtin_exit_frame, false);
171 return Builtin::kWasmCEntry;
172 }
173
174 if (rs == 1 && am == ArgvMode::kStack && !be) {
175 return Builtin::kCEntry_Return1_ArgvOnStack_NoBuiltinExit;
176 } else if (rs == 1 && am == ArgvMode::kStack && be) {
177 return Builtin::kCEntry_Return1_ArgvOnStack_BuiltinExit;
178 } else if (rs == 1 && am == ArgvMode::kRegister && !be) {
179 return Builtin::kCEntry_Return1_ArgvInRegister_NoBuiltinExit;
180 } else if (rs == 2 && am == ArgvMode::kStack && !be) {
181 return Builtin::kCEntry_Return2_ArgvOnStack_NoBuiltinExit;
182 } else if (rs == 2 && am == ArgvMode::kStack && be) {
183 return Builtin::kCEntry_Return2_ArgvOnStack_BuiltinExit;
184 } else if (rs == 2 && am == ArgvMode::kRegister && !be) {
185 return Builtin::kCEntry_Return2_ArgvInRegister_NoBuiltinExit;
186 }
187
188 UNREACHABLE();
189}

202// static
203constexpr Builtin Builtins::InterpreterPushArgsThenCall(
204 ConvertReceiverMode receiver_mode, InterpreterPushArgsMode mode) {
205 switch (mode) {
206 case InterpreterPushArgsMode::kArrayFunction:
207 // There is no special-case handling of calls to Array. They will all go
208 // through the kOther case below.
209 UNREACHABLE();
210 case InterpreterPushArgsMode::kWithFinalSpread:
211 return Builtin::kInterpreterPushArgsThenCallWithFinalSpread;
212 case InterpreterPushArgsMode::kOther:
213 switch (receiver_mode) {
214 case ConvertReceiverMode::kNullOrUndefined:
215 return Builtin::kInterpreterPushUndefinedAndArgsThenCall;
216 case ConvertReceiverMode::kNotNullOrUndefined:
217 case ConvertReceiverMode::kAny:
218 return Builtin::kInterpreterPushArgsThenCall;
219 }
220 }
221 UNREACHABLE();
222}

256// static
257int Builtins::GetFormalParameterCount(Builtin builtin) {
258 CHECK(HasJSLinkage(builtin));
259
260 // TODO(saelo): consider merging GetFormalParameterCount and
261 // GetStackParameterCount into a single function.
262 if (Builtins::KindOf(builtin) == TSJ || Builtins::KindOf(builtin) == TFJ) {
263 return Builtins::GetStackParameterCount(builtin);
264 } else if (Builtins::KindOf(builtin) == ASM ||
265 Builtins::KindOf(builtin) == TFC) {
266 // At the moment, all ASM builtins are varargs builtins. This is verified
267 // in CheckFormalParameterCount.
268 return kDontAdaptArgumentsSentinel;
269 } else if (Builtins::KindOf(builtin) == CPP) {
270#define CPP_BUILTIN(Name, Argc) \
271 case Builtin::k##Name: \
272 return Argc;
273
274 switch (builtin) {
275 BUILTIN_LIST_C(CPP_BUILTIN)
276 default:
277 UNREACHABLE();
278 }
279#undef CPP_BUILTIN
280 } else {
281 UNREACHABLE();
282 }
283}
