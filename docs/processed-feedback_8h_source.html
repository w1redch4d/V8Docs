220template <class T, ProcessedFeedback::Kind K>
221class SingleValueFeedback : public ProcessedFeedback {
222 public:
223 explicit SingleValueFeedback(T value, FeedbackSlotKind slot_kind)
224 : ProcessedFeedback(K, slot_kind), value_(value) {
225 DCHECK(
226 (K == kBinaryOperation && slot_kind == FeedbackSlotKind::kBinaryOp) ||
227 (K == kTypeOf && slot_kind == FeedbackSlotKind::kTypeOf) ||
228 (K == kCompareOperation && slot_kind == FeedbackSlotKind::kCompareOp) ||
229 (K == kForIn && slot_kind == FeedbackSlotKind::kForIn) ||
230 (K == kInstanceOf && slot_kind == FeedbackSlotKind::kInstanceOf) ||
231 ((K == kLiteral || K == kRegExpLiteral || K == kTemplateObject) &&
232 slot_kind == FeedbackSlotKind::kLiteral));
233 }
234
235 T value() const { return value_; }
236
237 private:
238 T const value_;
239};
