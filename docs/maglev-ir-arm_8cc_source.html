101 const ProcessingState& state) {
102 MaglevAssembler::TemporaryRegisterScope temps(masm);
103 Register scratch = temps.Acquire();
104 Register result_string = ToRegister(result());
105 if (Int32Constant* constant = code_input().node()->TryCast<Int32Constant>()) {
106 int32_t char_code = constant->value() & 0xFFFF;
107 if (0 <= char_code && char_code < String::kMaxOneByteCharCode) {
108 __ LoadSingleCharacterString(result_string, char_code);
109 } else {
110 DCHECK_NE(scratch, result_string);
111 __ AllocateTwoByteString(register_snapshot(), result_string, 1);
112 __ Move(scratch, char_code);
113 __ strh(scratch, FieldMemOperand(result_string,
114 OFFSET_OF_DATA_START(SeqTwoByteString)));
115 }
116 } else {
117 __ StringFromCharCode(register_snapshot(), nullptr, result_string,
118 ToRegister(code_input()), scratch,
119 MaglevAssembler::CharCodeMaskMode::kMustApplyMask);
120 }

207void Int32MultiplyWithOverflow::GenerateCode(MaglevAssembler* masm,
208 const ProcessingState& state) {
209 Register left = ToRegister(left_input());
210 Register right = ToRegister(right_input());
211 Register out = ToRegister(result());
212
213 // TODO(leszeks): peephole optimise multiplication by a constant.
214
215 MaglevAssembler::TemporaryRegisterScope temps(masm);
216 bool out_alias_input = out == left || out == right;
217 Register res_low = out;
218 if (out_alias_input) {
219 res_low = temps.AcquireScratch();
220 }
221 Register res_high = temps.AcquireScratch();
222 __ smull(res_low, res_high, left, right);
223
224 // ARM doesn't set the overflow flag for multiplication, so we need to
225 // test on kNotEqual.
226 __ cmp(res_high, Operand(res_low, ASR, 31));
227 __ EmitEagerDeoptIf(ne, DeoptimizeReason::kOverflow, this);
228
229 // If the result is zero, check if either lhs or rhs is negative.
230 Label end;
231 __ tst(res_low, res_low);
232 __ b(ne, &end);
233 Register temp = res_high;
234 __ orr(temp, left, right, SetCC);
235 // If one of them is negative, we must have a -0 result, which is non-int32,
236 // so deopt.
237 __ EmitEagerDeoptIf(mi, DeoptimizeReason::kOverflow, this);
238
239 __ bind(&end);
240 if (out_alias_input) {
241 __ Move(out, res_low);
242 }
243}

250void Int32DivideWithOverflow::GenerateCode(MaglevAssembler* masm,
251 const ProcessingState& state) {
252 Register left = ToRegister(left_input());
253 Register right = ToRegister(right_input());
254 Register out = ToRegister(result());
255
256 // TODO(leszeks): peephole optimise division by a constant.
257
258 // Pre-check for overflow, since idiv throws a division exception on overflow
259 // rather than setting the overflow flag. Logic copied from
260 // effect-control-linearizer.cc
261
262 // Check if {right} is positive (and not zero).
263 __ cmp(right, Operand(0));
264 ZoneLabelRef done(masm);
265 __ JumpToDeferredIf(
266 le,
267 [](MaglevAssembler* masm, ZoneLabelRef done, Register left,
268 Register right, Int32DivideWithOverflow* node) {
269 // {right} is negative or zero.
270
271 // TODO(leszeks): Using kNotInt32 here, but in same places
272 // kDivisionByZerokMinusZero/kMinusZero/kOverflow would be better. Right
273 // now all eager deopts in a node have to be the same -- we should allow
274 // a node to emit multiple eager deopts with different reasons.
275 Label* deopt = __ GetDeoptLabel(node, DeoptimizeReason::kNotInt32);
276
277 // Check if {right} is zero.
278 // We've already done the compare and flags won't be cleared yet.
279 __ JumpIf(eq, deopt);
280
281 // Check if {left} is zero, as that would produce minus zero.
282 __ tst(left, left);
283 __ JumpIf(eq, deopt);
284
285 // Check if {left} is kMinInt and {right} is -1, in which case we'd have
286 // to return -kMinInt, which is not representable as Int32.
287 __ cmp(left, Operand(kMinInt));
288 __ JumpIf(ne, *done);
289 __ cmp(right, Operand(-1));
290 __ JumpIf(ne, *done);
291 __ JumpToDeopt(deopt);
292 },
293 done, left, right, this);
294 __ bind(*done);
295
296 // Perform the actual integer division.
297 MaglevAssembler::TemporaryRegisterScope temps(masm);
298 bool out_alias_input = out == left || out == right;
299 Register res = out;
300 if (out_alias_input) {
301 res = temps.AcquireScratch();
302 }
303 if (CpuFeatures::IsSupported(SUDIV)) {
304 CpuFeatureScope scope(masm, SUDIV);
305 __ sdiv(res, left, right);
306 } else {
307 UseScratchRegisterScope temps(masm);
308 LowDwVfpRegister double_right = temps.AcquireLowD();
309 SwVfpRegister tmp = double_right.low();
310 DwVfpRegister double_left = temps.AcquireD();
311 DwVfpRegister double_res = double_left;
312 __ vmov(tmp, left);
313 __ vcvt_f64_s32(double_left, tmp);
314 __ vmov(tmp, right);
315 __ vcvt_f64_s32(double_right, tmp);
316 __ vdiv(double_res, double_left, double_right);
317 __ vcvt_s32_f64(tmp, double_res);
318 __ vmov(res, tmp);
319 }
320
321 // Check that the remainder is zero.
322 Register temp = temps.AcquireScratch();
323 __ mul(temp, res, right);
324 __ cmp(temp, left);
325 __ EmitEagerDeoptIf(ne, DeoptimizeReason::kNotInt32, this);
326
327 __ Move(out, res);
328}

331void Uint32Mod(MaglevAssembler* masm, Register out, Register left,
332 Register right) {
333 MaglevAssembler::TemporaryRegisterScope temps(masm);
334 Register res = temps.AcquireScratch();
335 if (CpuFeatures::IsSupported(SUDIV)) {
336 CpuFeatureScope scope(masm, SUDIV);
337 __ udiv(res, left, right);
338 } else {
339 UseScratchRegisterScope temps(masm);
340 LowDwVfpRegister double_right = temps.AcquireLowD();
341 SwVfpRegister tmp = double_right.low();
342 DwVfpRegister double_left = temps.AcquireD();
343 DwVfpRegister double_res = double_left;
344 __ vmov(tmp, left);
345 __ vcvt_f64_s32(double_left, tmp);
346 __ vmov(tmp, right);
347 __ vcvt_f64_s32(double_right, tmp);
348 __ vdiv(double_res, double_left, double_right);
349 __ vcvt_s32_f64(tmp, double_res);
350 __ vmov(res, tmp);
351 }
352 if (CpuFeatures::IsSupported(ARMv7)) {
353 __ mls(out, res, right, left);
354 } else {
355 __ mul(res, res, right);
356 __ sub(out, left, res);
357 }
358}

366void Int32ModulusWithOverflow::GenerateCode(MaglevAssembler* masm,
367 const ProcessingState& state) {
368 // If AreAliased(lhs, rhs):
369 // deopt if lhs < 0 // Minus zero.
370 // 0
371 //
372 // Using same algorithm as in EffectControlLinearizer:
373 // if rhs <= 0 then
374 // rhs = -rhs
375 // deopt if rhs == 0
376 // if lhs < 0 then
377 // let lhs_abs = -lsh in
378 // let res = lhs_abs % rhs in
379 // deopt if res == 0
380 // -res
381 // else
382 // let msk = rhs - 1 in
383 // if rhs & msk == 0 then
384 // lhs & msk
385 // else
386 // lhs % rhs
387
388 Register lhs = ToRegister(left_input());
389 Register rhs = ToRegister(right_input());
390 Register out = ToRegister(result());
391
392 static constexpr DeoptimizeReason deopt_reason =
393 DeoptimizeReason::kDivisionByZero;
394
395 if (lhs == rhs) {
396 // For the modulus algorithm described above, lhs and rhs must not alias
397 // each other.
398 __ tst(lhs, lhs);
399 // TODO(victorgomes): This ideally should be kMinusZero, but Maglev only
400 // allows one deopt reason per IR.
401 __ EmitEagerDeoptIf(mi, deopt_reason, this);
402 __ Move(ToRegister(result()), 0);
403 return;
404 }
405
406 DCHECK_NE(lhs, rhs);
407
408 ZoneLabelRef done(masm);
409 ZoneLabelRef rhs_checked(masm);
410 __ cmp(rhs, Operand(0));
411 __ JumpToDeferredIf(
412 le,
413 [](MaglevAssembler* masm, ZoneLabelRef rhs_checked, Register rhs,
414 Int32ModulusWithOverflow* node) {
415 __ rsb(rhs, rhs, Operand(0), SetCC);
416 __ b(ne, *rhs_checked);
417 __ EmitEagerDeopt(node, deopt_reason);
418 },
419 rhs_checked, rhs, this);
420 __ bind(*rhs_checked);
421
422 __ cmp(lhs, Operand(0));
423 __ JumpToDeferredIf(
424 lt,
425 [](MaglevAssembler* masm, ZoneLabelRef done, Register lhs, Register rhs,
426 Register out, Int32ModulusWithOverflow* node) {
427 __ rsb(lhs, lhs, Operand(0));
428 Uint32Mod(masm, out, lhs, rhs);
429 __ rsb(out, out, Operand(0), SetCC);
430 // TODO(victorgomes): This ideally should be kMinusZero, but Maglev
431 // only allows one deopt reason per IR.
432 __ b(ne, *done);
433 __ EmitEagerDeopt(node, deopt_reason);
434 },
435 done, lhs, rhs, out, this);
436
437 Label rhs_not_power_of_2;
438 {
439 MaglevAssembler::TemporaryRegisterScope temps(masm);
440 Register mask = temps.AcquireScratch();
441 __ add(mask, rhs, Operand(-1));
442 __ tst(mask, rhs);
443 __ JumpIf(ne, &rhs_not_power_of_2);
444
445 // {rhs} is power of 2.
446 __ and_(out, mask, lhs);
447 __ Jump(*done);
448 // {mask} can be reused from now on.
449 }
450
451 __ bind(&rhs_not_power_of_2);
452 Uint32Mod(masm, out, lhs, rhs);
453 __ bind(*done);
454}

616void Float64Round::GenerateCode(MaglevAssembler* masm,
617 const ProcessingState& state) {
618 DoubleRegister in = ToDoubleRegister(input());
619 DoubleRegister out = ToDoubleRegister(result());
620 CpuFeatureScope scope(masm, ARMv8);
621 if (kind_ == Kind::kNearest) {
622 MaglevAssembler::TemporaryRegisterScope temps(masm);
623 DoubleRegister temp = temps.AcquireDouble();
624 __ Move(temp, in);
625 // vrintn rounds to even on tie, while JS expects it to round towards
626 // +Infinity. Fix the difference by checking if we rounded down by exactly
627 // 0.5, and if so, round to the other side.
628 __ vrintn(out, in);
629 __ vsub(temp, temp, out);
630 DoubleRegister half_one = temps.AcquireScratchDouble();
631 __ Move(half_one, 0.5);
632 __ VFPCompareAndSetFlags(temp, half_one);
633 Label done;
634 __ JumpIf(ne, &done, Label::kNear);
635 // Fix wrong tie-to-even by adding 0.5 twice.
636 __ vadd(out, out, half_one);
637 __ vadd(out, out, half_one);
638 __ bind(&done);
639 } else if (kind_ == Kind::kCeil) {
640 __ vrintp(out, in);
641 } else if (kind_ == Kind::kFloor) {
642 __ vrintm(out, in);
643 }
644}

747void HandleInterruptsAndTiering(MaglevAssembler* masm, ZoneLabelRef done,
748 Node* node, ReduceInterruptBudgetType type,
749 Register scratch0) {
750 // For loops, first check for interrupts. Don't do this for returns, as we
751 // can't lazy deopt to the end of a return.
752 if (type == ReduceInterruptBudgetType::kLoop) {
753 Label next;
754 // Here, we only care about interrupts since we've already guarded against
755 // real stack overflows on function entry.
756 {
757 Register stack_limit = scratch0;
758 __ LoadStackLimit(stack_limit, StackLimitKind::kInterruptStackLimit);
759 __ cmp(sp, stack_limit);
760 __ b(hi, &next);
761 }
762
763 // An interrupt has been requested and we must call into runtime to handle
764 // it; since we already pay the call cost, combine with the TieringManager
765 // call.
766 {
767 SaveRegisterStateForCall save_register_state(masm,
768 node->register_snapshot());
769 Register function = scratch0;
770 __ ldr(function, MemOperand(fp, StandardFrameConstants::kFunctionOffset));
771 __ Push(function);
772 // Move into kContextRegister after the load into scratch0, just in case
773 // scratch0 happens to be kContextRegister.
774 __ Move(kContextRegister, masm->native_context().object());
775 __ CallRuntime(Runtime::kBytecodeBudgetInterruptWithStackCheck_Maglev, 1);
776 save_register_state.DefineSafepointWithLazyDeopt(node->lazy_deopt_info());
777 }
778 __ b(*done); // All done, continue.
779 __ bind(&next);
780 }
781
782 // No pending interrupts. Call into the TieringManager if needed.
783 {
784 SaveRegisterStateForCall save_register_state(masm,
785 node->register_snapshot());
786 Register function = scratch0;
787 __ ldr(function, MemOperand(fp, StandardFrameConstants::kFunctionOffset));
788 __ Push(function);
789 // Move into kContextRegister after the load into scratch0, just in case
790 // scratch0 happens to be kContextRegister.
791 __ Move(kContextRegister, masm->native_context().object());
792 // Note: must not cause a lazy deopt!
793 __ CallRuntime(Runtime::kBytecodeBudgetInterrupt_Maglev, 1);
794 save_register_state.DefineSafepoint();
795 }
796 __ b(*done);
797}

799void GenerateReduceInterruptBudget(MaglevAssembler* masm, Node* node,
800 ReduceInterruptBudgetType type, int amount) {
801 MaglevAssembler::TemporaryRegisterScope temps(masm);
802 Register scratch = temps.Acquire();
803 Register feedback_cell = scratch;
804 Register budget = temps.Acquire();
805 __ ldr(feedback_cell,
806 MemOperand(fp, StandardFrameConstants::kFunctionOffset));
807 __ LoadTaggedField(
808 feedback_cell,
809 FieldMemOperand(feedback_cell, JSFunction::kFeedbackCellOffset));
810 __ ldr(budget,
811 FieldMemOperand(feedback_cell, FeedbackCell::kInterruptBudgetOffset));
812 __ sub(budget, budget, Operand(amount), SetCC);
813 __ str(budget,
814 FieldMemOperand(feedback_cell, FeedbackCell::kInterruptBudgetOffset));
815 ZoneLabelRef done(masm);
816 __ JumpToDeferredIf(lt, HandleInterruptsAndTiering, done, node, type,
817 scratch);
818 __ bind(*done);
819}

849void Return::GenerateCode(MaglevAssembler* masm, const ProcessingState& state) {
850 DCHECK_EQ(ToRegister(value_input()), kReturnRegister0);
851
852 // Read the formal number of parameters from the top level compilation unit
853 // (i.e. the outermost, non inlined function).
854 int formal_params_size =
855 masm->compilation_info()->toplevel_compilation_unit()->parameter_count();
856
857 // We're not going to continue execution, so we can use an arbitrary register
858 // here instead of relying on temporaries from the register allocator.
859 Register actual_params_size = r4;
860 Register params_size = r8;
861
862 // Compute the size of the actual parameters + receiver (in bytes).
863 // TODO(leszeks): Consider making this an input into Return to re-use the
864 // incoming argc's register (if it's still valid).
865 __ ldr(actual_params_size,
866 MemOperand(fp, StandardFrameConstants::kArgCOffset));
867
868 // Leave the frame.
869 __ LeaveFrame(StackFrame::MAGLEV);
870
871 // If actual is bigger than formal, then we should use it to free up the stack
872 // arguments.
873 Label corrected_args_count;
874 __ Move(params_size, formal_params_size);
875 __ cmp(params_size, actual_params_size);
876 __ b(kGreaterThanEqual, &corrected_args_count);
877 __ Move(params_size, actual_params_size);
878 __ bind(&corrected_args_count);
879
880 // Drop receiver + arguments according to dynamic arguments size.
881 __ DropArguments(params_size);
882 __ Ret();
883}
