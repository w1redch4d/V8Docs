20template <typename Entry, size_t size>
21uint32_t CompactibleExternalEntityTable<Entry, size>::AllocateEntry(
22 Space* space) {
23 uint32_t index = Base::AllocateEntry(space);
24
25 // When we're compacting a space, we're trying to move all entries above a
26 // threshold index (the start of the evacuation area) into segments below
27 // that threshold. However, if the freelist becomes too short and we start
28 // allocating entries inside the area that is supposed to be evacuated, we
29 // need to abort compaction. This is not just an optimization but is also
30 // required for correctness: during sweeping we might otherwise assume that
31 // all entries inside the evacuation area have been moved and that these
32 // segments can therefore be deallocated. In particular, this check will also
33 // make sure that we abort compaction if we extend the space with a new
34 // segment and allocate at least one entry in it (if that segment is located
35 // after the threshold, otherwise it is unproblematic).
36 uint32_t start_of_evacuation_area =
37 space->start_of_evacuation_area_.load(std::memory_order_relaxed);
38 if (V8_UNLIKELY(index >= start_of_evacuation_area)) {
39 space->AbortCompacting(start_of_evacuation_area);
40 }
41
42 return index;
43}

45template <typename Entry, size_t size>
46typename CompactibleExternalEntityTable<Entry, size>::CompactionResult
47CompactibleExternalEntityTable<Entry, size>::FinishCompaction(
48 Space* space, Histogram* counter) {
49 DCHECK(space->BelongsTo(this));
50 DCHECK(!space->is_internal_read_only_space());
51
52 // When compacting, we can compute the number of unused segments at the end of
53 // the table and deallocate those after sweeping.
54 uint32_t start_of_evacuation_area =
55 space->start_of_evacuation_area_.load(std::memory_order_relaxed);
56 bool evacuation_was_successful = false;
57 if (space->IsCompacting()) {
58 auto outcome = ExternalEntityTableCompactionOutcome::kAborted;
59 if (space->CompactingWasAborted()) {
60 // Compaction was aborted during marking because the freelist grew to
61 // short. In this case, it is not guaranteed that any segments will now be
62 // completely free. Extract the original start_of_evacuation_area value.
63 start_of_evacuation_area &= ~Space::kCompactionAbortedMarker;
64 } else {
65 // Entry evacuation was successful so all segments inside the evacuation
66 // area are now guaranteed to be free and so can be deallocated.
67 evacuation_was_successful = true;
68 outcome = ExternalEntityTableCompactionOutcome::kSuccess;
69 }
70 DCHECK(IsAligned(start_of_evacuation_area,
71 ExternalEntityTable<Entry, size>::kEntriesPerSegment));
72
73 space->StopCompacting();
74 counter->AddSample(static_cast<int>(outcome));
75 }
76
77 return {start_of_evacuation_area, evacuation_was_successful};
78}

80template <typename Entry, size_t size>
81void CompactibleExternalEntityTable<Entry, size>::MaybeCreateEvacuationEntry(
82 Space* space, uint32_t index, Address handle_location) {
83 // Check if the entry should be evacuated for table compaction.
84 // The current value of the start of the evacuation area is cached in a local
85 // variable here as it otherwise may be changed by another marking thread
86 // while this method runs, causing non-optimal behaviour (for example, the
87 // allocation of an evacuation entry _after_ the entry that is evacuated).
88 uint32_t start_of_evacuation_area =
89 space->start_of_evacuation_area_.load(std::memory_order_relaxed);
90 if (index >= start_of_evacuation_area) {
91 DCHECK(space->IsCompacting());
92 uint32_t new_index =
93 Base::AllocateEntryBelow(space, start_of_evacuation_area);
94 if (new_index) {
95 DCHECK_LT(new_index, start_of_evacuation_area);
96 DCHECK(space->Contains(new_index));
97 // Even though the new entry will only be accessed during sweeping, this
98 // still needs to be an atomic write as another thread may attempt (and
99 // fail) to allocate the same table entry, thereby causing a read from
100 // this memory location. Without an atomic store here, TSan would then
101 // complain about a data race.
102 Base::at(new_index).MakeEvacuationEntry(handle_location);
103 } else {
104 // In this case, the application has allocated a sufficiently large
105 // number of entries from the freelist so that new entries would now be
106 // allocated inside the area that is being compacted. While it would be
107 // possible to shrink that area and continue compacting, we probably do
108 // not want to put more pressure on the freelist and so instead simply
109 // abort compaction here. Entries that have already been visited will
110 // still be compacted during Sweep, but there is no guarantee that any
111 // blocks at the end of the table will now be completely free.
112 space->AbortCompacting(start_of_evacuation_area);
113 }
114 }

132void CompactibleExternalEntityTable<Entry, size>::Space::AbortCompacting(

134 uint32_t compaction_aborted_marker =
135 start_of_evacuation_area | kCompactionAbortedMarker;
136 DCHECK_NE(compaction_aborted_marker, kNotCompactingMarker);
137 start_of_evacuation_area_.store(compaction_aborted_marker,

178template <typename Entry, size_t size>
179void CompactibleExternalEntityTable<Entry,
180 size>::Space::StartCompactingIfNeeded() {
181 // Take the lock so that we can be sure that no other thread modifies the
182 // segments set concurrently.
183 base::SpinningMutexGuard guard(&this->mutex_);
184
185 // This method may be executed while other threads allocate entries from the
186 // freelist. In that case, this method may use incorrect data to determine if
187 // table compaction is necessary. That's fine however since in the worst
188 // case, compaction will simply be aborted right away if the freelist became
189 // too small.
190 uint32_t num_free_entries = this->freelist_length();
191 uint32_t num_total_entries = this->capacity();
192
193 // Current (somewhat arbitrary) heuristic: need compacting if the space is
194 // more than 1MB in size, is at least 10% empty, and if at least one segment
195 // can be freed after successful compaction.
196 double free_ratio = static_cast<double>(num_free_entries) /
197 static_cast<double>(num_total_entries);
198 uint32_t num_segments_to_evacuate =
199 (num_free_entries / 2) / Base::kEntriesPerSegment;
200 uint32_t space_size = num_total_entries * Base::kEntrySize;
201 bool should_compact = (space_size >= 1 * MB) && (free_ratio >= 0.10) &&
202 (num_segments_to_evacuate >= 1);
203
204 // However, if --stress-compaction is enabled, we compact whenever possible:
205 // whenever we have at least two segments, one to evacuate entries into and
206 // the other to evacuate entries from.
207 if (v8_flags.stress_compaction) {
208 should_compact = this->num_segments() > 1;
209 num_segments_to_evacuate = std::max(1u, num_segments_to_evacuate);
210 }
211
212 if (should_compact) {
213 // If we're compacting, attempt to free up the last N segments so that they
214 // can be decommitted afterwards.
215 auto first_segment_to_evacuate =
216 *std::prev(this->segments_.end(), num_segments_to_evacuate);
217 uint32_t start_of_evacuation_area = first_segment_to_evacuate.first_entry();
218 StartCompacting(start_of_evacuation_area);
219 }
220}
