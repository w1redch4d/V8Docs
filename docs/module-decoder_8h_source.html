31inline bool IsValidSectionCode(uint8_t byte) {
32 // Allow everything within [kUnknownSectionCode, kLastKnownModuleSection].
33 static_assert(kUnknownSectionCode == 0);
34 return byte <= kLastKnownModuleSection;
35}
