41void PropertyCell::UpdatePropertyDetailsExceptCellType(
42 PropertyDetails details) {
43 DCHECK(CheckDataIsCompatible(details, value()));
44 PropertyDetails old_details = property_details();
45 CHECK_EQ(old_details.cell_type(), details.cell_type());
46 set_property_details_raw(details.AsSmi(), kReleaseStore);
47 // Deopt when making a writable property read-only. The reverse direction
48 // is uninteresting because Turbofan does not currently rely on read-only
49 // unless the property is also configurable, in which case it will stay
50 // read-only forever.
51 if (!old_details.IsReadOnly() && details.IsReadOnly()) {
52 // TODO(11527): pass Isolate as an argument.
53 Isolate* isolate = GetIsolateFromWritableObject(*this);
54 DependentCode::DeoptimizeDependencyGroups(
55 isolate, *this, DependentCode::kPropertyCellChangedGroup);
56 }
57}

59void PropertyCell::Transition(PropertyDetails new_details,
60 DirectHandle<Object> new_value) {
61 DCHECK(CanTransitionTo(new_details, *new_value));
62 // This code must be in sync with its counterpart in
63 // PropertyCellData::Serialize.
64 PropertyDetails transition_marker = new_details;
65 transition_marker.set_cell_type(PropertyCellType::kInTransition);
66 set_property_details_raw(transition_marker.AsSmi(), kReleaseStore);
67 set_value(*new_value, kReleaseStore);
68 set_property_details_raw(new_details.AsSmi(), kReleaseStore);
69}
