230Tagged<Code> FeedbackVector::optimized_code(IsolateForSandbox isolate) const {
231 Tagged<MaybeObject> slot = maybe_optimized_code();
232 DCHECK(slot.IsWeakOrCleared());
233 Tagged<HeapObject> heap_object;
234 Tagged<Code> code;
235 if (slot.GetHeapObject(&heap_object)) {
236 code = Cast<CodeWrapper>(heap_object)->code(isolate);
237 }
238 // It is possible that the maybe_optimized_code slot is cleared but the flags
239 // haven't been updated yet. We update them when we execute the function next
240 // time / when we create new closure.
241 DCHECK_IMPLIES(!code.is_null(),
242 maybe_has_maglev_code() || maybe_has_turbofan_code());
243 DCHECK_IMPLIES(!code.is_null() && code->is_maglevved(),
244 maybe_has_maglev_code());
245 DCHECK_IMPLIES(!code.is_null() && code->is_turbofanned(),
246 maybe_has_turbofan_code());
247 return code;
248}
249
250bool FeedbackVector::has_optimized_code() const {
251 bool is_cleared = maybe_optimized_code().IsCleared();
252 DCHECK_IMPLIES(!is_cleared,
253 maybe_has_maglev_code() || maybe_has_turbofan_code());
254 return !is_cleared;
255}

275std::optional<Tagged<Code>> FeedbackVector::GetOptimizedOsrCode(
276 Isolate* isolate, FeedbackSlot slot) {
277 Tagged<MaybeObject> maybe_code = Get(isolate, slot);
278 if (maybe_code.IsCleared()) return {};
279
280 Tagged<Code> code =
281 Cast<CodeWrapper>(maybe_code.GetHeapObject())->code(isolate);
282 if (code->marked_for_deoptimization()) {
283 // Clear the cached Code object if deoptimized.
284 // TODO(jgruber): Add tracing.
285 Set(slot, ClearedValue(isolate));
286 return {};
287 }
288
289 return code;
290}

399template <CompareOperationFeedback::Type Feedback>
400bool Is(int type_feedback) {

403
404CompareOperationHint CompareOperationHintFromFeedback(int type_feedback) {
405 if (Is<CompareOperationFeedback::kNone>(type_feedback)) {
406 return CompareOperationHint::kNone;
407 }
408
409 if (Is<CompareOperationFeedback::kSignedSmall>(type_feedback)) {
410 return CompareOperationHint::kSignedSmall;
411 } else if (Is<CompareOperationFeedback::kNumber>(type_feedback)) {
412 return CompareOperationHint::kNumber;
413 } else if (Is<CompareOperationFeedback::kNumberOrBoolean>(type_feedback)) {
414 return CompareOperationHint::kNumberOrBoolean;
415 }
416
417 if (Is<CompareOperationFeedback::kInternalizedString>(type_feedback)) {
418 return CompareOperationHint::kInternalizedString;
419 } else if (Is<CompareOperationFeedback::kString>(type_feedback)) {
420 return CompareOperationHint::kString;
421 }
422
423 if (Is<CompareOperationFeedback::kReceiver>(type_feedback)) {
424 return CompareOperationHint::kReceiver;
425 } else if (Is<CompareOperationFeedback::kReceiverOrNullOrUndefined>(
426 type_feedback)) {
427 return CompareOperationHint::kReceiverOrNullOrUndefined;
428 }
429
430 if (Is<CompareOperationFeedback::kBigInt64>(type_feedback)) {
431 return CompareOperationHint::kBigInt64;
432 } else if (Is<CompareOperationFeedback::kBigInt>(type_feedback)) {
433 return CompareOperationHint::kBigInt;
434 }
435
436 if (Is<CompareOperationFeedback::kSymbol>(type_feedback)) {
437 return CompareOperationHint::kSymbol;
438 }
439
440 DCHECK(Is<CompareOperationFeedback::kAny>(type_feedback));
441 return CompareOperationHint::kAny;
442}

543std::pair<Tagged<MaybeObject>, Tagged<MaybeObject>>
544FeedbackNexus::GetFeedbackPair() const {
545 if (config()->mode() == NexusConfig::BackgroundThread &&
546 feedback_cache_.has_value()) {
547 return std::make_pair(FromHandle(feedback_cache_->first),
548 FromHandle(feedback_cache_->second));
549 }
550 auto pair = FeedbackMetadata::GetSlotSize(kind()) == 2
551 ? config()->GetFeedbackPair(vector(), slot())
552 : std::make_pair(config()->GetFeedback(vector(), slot()),
553 Tagged<MaybeObject>());
554 if (config()->mode() == NexusConfig::BackgroundThread &&
555 !feedback_cache_.has_value()) {
556 feedback_cache_ =
557 std::make_pair(ToHandle(pair.first), ToHandle(pair.second));
558 }
559 return pair;
560}

577template <typename F>
578void FeedbackNexus::IterateMapsWithUnclearedHandler(F function) const {
579 // We don't need DisallowGarbageCollection here: accessing it.map() and
580 // it.handle() is safe between it.Advance() and a potential GC call in
581 // function(). The it itself is not invalidated, since it holds the
582 // polymorphic array by handle.
583 // TODO(370727490): Make the FeedbackIterator GC safe (e.g. look up
584 // map/handler in the feedback array on-demand).
585 for (FeedbackIterator it(this); !it.done(); it.Advance()) {
586 Handle<Map> map = config()->NewHandle(it.map());
587 if (!it.handler().IsCleared()) {
588 function(map);
589 }
590 }
591}
