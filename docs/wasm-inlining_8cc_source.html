49Reduction WasmInliner::ReduceCall(Node* call) {
50 DCHECK(call->opcode() == IrOpcode::kCall ||
51 call->opcode() == IrOpcode::kTailCall);
52
53 if (seen_.find(call) != seen_.end()) {
54 TRACE("[function %d: have already seen node %d, skipping]\n",
55 data_.func_index, call->id());
56 return NoChange();
57 }
58 seen_.insert(call);
59
60 Node* callee = NodeProperties::GetValueInput(call, 0);
61 IrOpcode::Value reloc_opcode = mcgraph_->machine()->Is32()
62 ? IrOpcode::kRelocatableInt32Constant
63 : IrOpcode::kRelocatableInt64Constant;
64 if (callee->opcode() != reloc_opcode) {
65 TRACE("[function %d: node %d: not a relocatable constant]\n",
66 data_.func_index, call->id());
67 return NoChange();
68 }
69 auto info = OpParameter<RelocatablePtrConstantInfo>(callee->op());
70 uint32_t inlinee_index = static_cast<uint32_t>(info.value());
71 if (info.rmode() != RelocInfo::WASM_CALL) {
72 Trace(call, inlinee_index, "not a wasm call");
73 return NoChange();
74 }
75 if (inlinee_index < module()->num_imported_functions) {
76 Trace(call, inlinee_index, "imported function");
77 return NoChange();
78 }
79
80 // We limit the times a function can be inlined to avoid repeatedly inlining
81 // recursive calls. Since we only check here (and not in {Finalize}), it is
82 // possible to exceed this limit if we find a large number of calls in a
83 // single pass.
84 constexpr int kMaximumInlinedCallsPerFunction = 3;
85 if (function_inlining_count_[inlinee_index] >=
86 kMaximumInlinedCallsPerFunction) {
87 Trace(call, inlinee_index,
88 "too many inlined calls to (recursive?) function");
89 return NoChange();
90 }
91
92 CHECK_LT(inlinee_index, module()->functions.size());
93 const wasm::WasmFunction* inlinee = &module()->functions[inlinee_index];
94 base::Vector<const uint8_t> function_bytes =
95 data_.wire_bytes_storage->GetCode(inlinee->code);
96
97 int call_count = GetCallCount(call);
98
99 int wire_byte_size = static_cast<int>(function_bytes.size());
100 int min_count_for_inlining =
101 v8_flags.wasm_inlining_ignore_call_counts ? 0 : wire_byte_size / 2;
102
103 // If liftoff ran and collected call counts, only inline calls that have been
104 // invoked often, except for truly tiny functions.
105 if (v8_flags.liftoff && v8_flags.wasm_inlining && wire_byte_size >= 12 &&
106 call_count < min_count_for_inlining) {
107 Trace(call, inlinee_index, "not called often enough");
108 return NoChange();
109 }
110
111 Trace(call, inlinee_index, "adding to inlining candidates");
112
113 CandidateInfo candidate{call, inlinee_index, call_count,
114 function_bytes.length()};
115
116 inlining_candidates_.push(candidate);
117 return NoChange();
118}

120bool SmallEnoughToInline(const wasm::WasmModule* module,
121 size_t current_graph_size, uint32_t candidate_size,
122 size_t initial_graph_size) {
123 if (candidate_size > v8_flags.wasm_inlining_max_size) {
124 return false;
125 }
126 if (WasmInliner::graph_size_allows_inlining(
127 module, current_graph_size + candidate_size, initial_graph_size)) {
128 return true;
129 }
130 // For truly tiny functions, let's be a bit more generous.
131 return candidate_size <= 12 &&
132 WasmInliner::graph_size_allows_inlining(
133 module, current_graph_size - 100, initial_graph_size);
134}

136bool WasmInliner::graph_size_allows_inlining(const wasm::WasmModule* module,
137 size_t graph_size,
138 size_t initial_graph_size) {
139 size_t budget =
140 std::max<size_t>(v8_flags.wasm_inlining_min_budget,
141 v8_flags.wasm_inlining_factor * initial_graph_size);
142 // For large-ish functions, the inlining budget is mainly defined by the
143 // wasm_inlining_budget.
144 size_t upper_budget = v8_flags.wasm_inlining_budget;
145 double small_function_percentage =
146 module->num_small_functions * 100.0 / module->num_declared_functions;
147 if (small_function_percentage < 50) {
148 // If there are few small functions, it indicates that the toolchain already
149 // performed significant inlining. Reduce the budget significantly as
150 // inlining has a diminishing ROI.
151
152 // We also apply a linear progression of the budget in the interval [25, 50]
153 // for the small_function_percentage. This progression is just added to
154 // prevent performance cliffs (e.g. when just performing a sharp cutoff at
155 // the 50% point) and not based on actual data.
156 double smallishness = std::max(25.0, small_function_percentage) - 25.0;
157 size_t lower_budget = upper_budget / 10;
158 double step = (upper_budget - lower_budget) / 25.0;
159 upper_budget = lower_budget + smallishness * step;
160 }
161 // Independent of the wasm_inlining_budget, for large functions we should
162 // still allow some inlining.
163 size_t full_budget = std::max<size_t>(upper_budget, initial_graph_size * 1.1);
164 budget = std::min<size_t>(full_budget, budget);
165 return graph_size < budget;
166}

177void WasmInliner::Finalize() {
178 TRACE("[function %d (%s): %s]\n", data_.func_index, debug_name_,
179 inlining_candidates_.empty() ? "no inlining candidates"
180 : "going through inlining candidates");
181 if (inlining_candidates_.empty()) return;
182 while (!inlining_candidates_.empty()) {
183 CandidateInfo candidate = inlining_candidates_.top();
184 inlining_candidates_.pop();
185 Node* call = candidate.node;
186 if (call->IsDead()) {
187 Trace(candidate, "dead node");
188 continue;
189 }
190 // We could build the candidate's graph first and consider its node count,
191 // but it turns out that wire byte size and node count are quite strongly
192 // correlated, at about 0.74 nodes per wire byte (measured for J2Wasm).
193 if (!SmallEnoughToInline(module(), current_graph_size_,
194 candidate.wire_byte_size, initial_graph_size_)) {
195 Trace(candidate, "not enough inlining budget");
196 continue;
197 }
198 const wasm::WasmFunction* inlinee =
199 &module()->functions[candidate.inlinee_index];
200
201 DCHECK_EQ(inlinee->sig->parameter_count(),
202 call->op()->ValueInputCount() - 2);
203#if DEBUG
204 // The two first parameters in the call are the function and instance, and
205 // then come the wasm function parameters.
206 for (uint32_t i = 0; i < inlinee->sig->parameter_count(); i++) {
207 if (!NodeProperties::IsTyped(call->InputAt(i + 2))) continue;
208 wasm::TypeInModule param_type =
209 NodeProperties::GetType(call->InputAt(i + 2)).AsWasm();
210 CHECK(IsSubtypeOf(param_type.type, inlinee->sig->GetParam(i),
211 param_type.module, module()));
212 }
213#endif
214
215 base::Vector<const uint8_t> function_bytes =
216 data_.wire_bytes_storage->GetCode(inlinee->code);
217
218 bool is_shared = module()->type(inlinee->sig_index).is_shared;
219
220 const wasm::FunctionBody inlinee_body{inlinee->sig, inlinee->code.offset(),
221 function_bytes.begin(),
222 function_bytes.end(), is_shared};
223
224 // If the inlinee was not validated before, do that now.
225 if (V8_UNLIKELY(
226 !module()->function_was_validated(candidate.inlinee_index))) {
227 if (ValidateFunctionBody(zone(), env_->enabled_features, module(),
228 detected_, inlinee_body)
229 .failed()) {
230 Trace(candidate, "function is invalid");
231 // At this point we cannot easily raise a compilation error any more.
232 // Since this situation is highly unlikely though, we just ignore this
233 // inlinee and move on. The same validation error will be triggered
234 // again when actually compiling the invalid function.
235 continue;
236 }
237 module()->set_function_validated(candidate.inlinee_index);
238 }
239
240 std::vector<WasmLoopInfo> inlinee_loop_infos;
241 wasm::DanglingExceptions dangling_exceptions;
242
243 size_t subgraph_min_node_id = graph()->NodeCount();
244 Node* inlinee_start;
245 Node* inlinee_end;
246 SourcePosition caller_pos =
247 data_.source_positions->GetSourcePosition(candidate.node);
248 inlining_positions_->push_back({static_cast<int>(candidate.inlinee_index),
249 call->opcode() == IrOpcode::kTailCall,
250 caller_pos});
251 int inlining_position_id =
252 static_cast<int>(inlining_positions_->size()) - 1;
253 WasmGraphBuilder builder(env_, zone(), mcgraph_, inlinee_body.sig,
254 data_.source_positions,
255 WasmGraphBuilder::kInstanceParameterMode,
256 nullptr /* isolate */, env_->enabled_features);
257 builder.set_inlining_id(inlining_position_id);
258 {
259 Graph::SubgraphScope scope(graph());
260 wasm::BuildTFGraph(zone()->allocator(), env_->enabled_features, module(),
261 &builder, detected_, inlinee_body, &inlinee_loop_infos,
262 &dangling_exceptions, data_.node_origins,
263 candidate.inlinee_index, data_.assumptions,
264 NodeProperties::IsExceptionalCall(call)
265 ? wasm::kInlinedHandledCall
266 : wasm::kInlinedNonHandledCall);
267 inlinee_start = graph()->start();
268 inlinee_end = graph()->end();
269 }
270
271 size_t additional_nodes = graph()->NodeCount() - subgraph_min_node_id;
272 Trace(candidate, "decided to inline");
273 current_graph_size_ += additional_nodes;
274 DCHECK_GE(function_inlining_count_[candidate.inlinee_index], 0);
275 function_inlining_count_[candidate.inlinee_index]++;
276
277 if (call->opcode() == IrOpcode::kCall) {
278 InlineCall(call, inlinee_start, inlinee_end, inlinee->sig, caller_pos,
279 &dangling_exceptions);
280 } else {
281 InlineTailCall(call, inlinee_start, inlinee_end);
282 }
283 call->Kill();
284 data_.loop_infos->insert(data_.loop_infos->end(),
285 inlinee_loop_infos.begin(),
286 inlinee_loop_infos.end());
287 // Returning after inlining, so that new calls in the inlined body are added
288 // to the candidates list and prioritized if they have a higher score.
289 return;
290 }

293/* Rewire callee formal parameters to the call-site real parameters. Rewire
294 * effect and control dependencies of callee's start node with the respective
295 * inputs of the call node.
296 */
297void WasmInliner::RewireFunctionEntry(Node* call, Node* callee_start) {
298 Node* control = NodeProperties::GetControlInput(call);
299 Node* effect = NodeProperties::GetEffectInput(call);
300
301 for (Edge edge : callee_start->use_edges()) {
302 Node* use = edge.from();
303 switch (use->opcode()) {
304 case IrOpcode::kParameter: {
305 // Index 0 is the callee node.
306 int index = 1 + ParameterIndexOf(use->op());
307 Replace(use, NodeProperties::GetValueInput(call, index));
308 break;
309 }
310 default:
311 if (NodeProperties::IsEffectEdge(edge)) {
312 edge.UpdateTo(effect);
313 } else if (NodeProperties::IsControlEdge(edge)) {
314 // Projections pointing to the inlinee start are floating control.
315 // They should point to the graph's start.
316 edge.UpdateTo(use->opcode() == IrOpcode::kProjection
317 ? graph()->start()
318 : control);
319 } else {
320 UNREACHABLE();
321 }
322 Revisit(edge.from());
323 break;
324 }
325 }
326}

328void WasmInliner::InlineTailCall(Node* call, Node* callee_start,
329 Node* callee_end) {
330 DCHECK_EQ(call->opcode(), IrOpcode::kTailCall);
331 // 1) Rewire function entry.
332 RewireFunctionEntry(call, callee_start);
333 // 2) For tail calls, all we have to do is rewire all terminators of the
334 // inlined graph to the end of the caller graph.
335 for (Node* const input : callee_end->inputs()) {
336 DCHECK(IrOpcode::IsGraphTerminator(input->opcode()));
337 MergeControlToEnd(graph(), common(), input);
338 }
339 for (Edge edge_to_end : call->use_edges()) {
340 DCHECK_EQ(edge_to_end.from(), graph()->end());
341 edge_to_end.UpdateTo(mcgraph()->Dead());
342 }
343 callee_end->Kill();
344 call->Kill();
345 Revisit(graph()->end());
346}

348void WasmInliner::InlineCall(Node* call, Node* callee_start, Node* callee_end,
349 const wasm::FunctionSig* inlinee_sig,
350 SourcePosition parent_pos,
351 wasm::DanglingExceptions* dangling_exceptions) {
352 DCHECK_EQ(call->opcode(), IrOpcode::kCall);
353
354 Node* handler = nullptr;
355 bool is_exceptional_call = NodeProperties::IsExceptionalCall(call, &handler);
356
357 // 1) Rewire function entry.
358 RewireFunctionEntry(call, callee_start);
359
360 // 2) Handle all graph terminators for the callee.
361 NodeVector return_nodes(zone());
362 for (Node* const input : callee_end->inputs()) {
363 DCHECK(IrOpcode::IsGraphTerminator(input->opcode()));
364 switch (input->opcode()) {
365 case IrOpcode::kReturn:
366 // Returns are collected to be rewired into the caller graph later.
367 return_nodes.push_back(input);
368 break;
369 case IrOpcode::kDeoptimize:
370 case IrOpcode::kTerminate:
371 case IrOpcode::kThrow:
372 MergeControlToEnd(graph(), common(), input);
373 break;
374 case IrOpcode::kTailCall: {
375 // A tail call in the callee inlined in a regular call in the caller has
376 // to be transformed into a regular call, and then returned from the
377 // inlinee. It will then be handled like any other return.
378 auto descriptor = CallDescriptorOf(input->op());
379 NodeProperties::ChangeOp(input, common()->Call(descriptor));
380 // Consider a function f which calls g which tail calls h. If h traps,
381 // we need the stack trace to include h and f (g's frame is gone due to
382 // the tail call). The way to achieve this is to set this call's
383 // position to the position of g's call in f.
384 data_.source_positions->SetSourcePosition(input, parent_pos);
385
386 DCHECK_GT(input->op()->EffectOutputCount(), 0);
387 DCHECK_GT(input->op()->ControlOutputCount(), 0);
388 Node* effect = input;
389 Node* control = input;
390 if (is_exceptional_call) {
391 // Remember dangling exception (will be connected later).
392 Node* if_exception = graph()->NewNode(
393 mcgraph()->common()->IfException(), input, control);
394 dangling_exceptions->Add(if_exception, if_exception, if_exception);
395 control = graph()->NewNode(mcgraph()->common()->IfSuccess(), input);
396 }
397
398 int return_arity = static_cast<int>(inlinee_sig->return_count());
399 NodeVector return_inputs(zone());
400 // The first input of a return node is always the 0 constant.
401 return_inputs.push_back(graph()->NewNode(common()->Int32Constant(0)));
402 if (return_arity == 1) {
403 // Tail calls are untyped; we have to type the node here.
404 // TODO(manoskouk): Try to compute a more precise type from the callee
405 // node.
406 NodeProperties::SetType(
407 input, Type::Wasm({inlinee_sig->GetReturn(0), module()},
408 graph()->zone()));
409 return_inputs.push_back(input);
410 } else if (return_arity > 1) {
411 for (int i = 0; i < return_arity; i++) {
412 Node* ith_projection =
413 graph()->NewNode(common()->Projection(i), input, control);
414 // Similarly here we have to type the call's projections.
415 NodeProperties::SetType(
416 ith_projection,
417 Type::Wasm({inlinee_sig->GetReturn(i), module()},
418 graph()->zone()));
419 return_inputs.push_back(ith_projection);
420 }
421 }
422
423 // Add effect and control inputs.
424 return_inputs.push_back(effect);
425 return_inputs.push_back(control);
426
427 Node* ret = graph()->NewNode(common()->Return(return_arity),
428 static_cast<int>(return_inputs.size()),
429 return_inputs.data());
430 return_nodes.push_back(ret);
431 break;
432 }
433 default:
434 UNREACHABLE();
435 }
436 }
437 callee_end->Kill();
438
439 // 3) Rewire unhandled calls to the handler.
440 if (is_exceptional_call) {
441 int handler_count = static_cast<int>(dangling_exceptions->Size());
442 if (handler_count > 0) {
443 Node* control_output =
444 graph()->NewNode(common()->Merge(handler_count), handler_count,
445 dangling_exceptions->controls.data());
446 std::vector<Node*>& effects(dangling_exceptions->effects);
447 std::vector<Node*>& values(dangling_exceptions->exception_values);
448
449 effects.push_back(control_output);
450 values.push_back(control_output);
451 Node* value_output = graph()->NewNode(
452 common()->Phi(MachineRepresentation::kTagged, handler_count),
453 handler_count + 1, values.data());
454 Node* effect_output = graph()->NewNode(common()->EffectPhi(handler_count),
455 handler_count + 1, effects.data());
456 ReplaceWithValue(handler, value_output, effect_output, control_output);
457 } else {
458 // Nothing in the inlined function can throw. Remove the handler.
459 ReplaceWithValue(handler, mcgraph()->Dead(), mcgraph()->Dead(),
460 mcgraph()->Dead());
461 }
462 }
463
464 if (!return_nodes.empty()) {
465 /* 4) Collect all return site value, effect, and control inputs into phis
466 * and merges. */
467 int const return_count = static_cast<int>(return_nodes.size());
468 NodeVector controls(zone());
469 NodeVector effects(zone());
470 for (Node* const return_node : return_nodes) {
471 controls.push_back(NodeProperties::GetControlInput(return_node));
472 effects.push_back(NodeProperties::GetEffectInput(return_node));
473 }
474 Node* control_output = graph()->NewNode(common()->Merge(return_count),
475 return_count, &controls.front());
476 effects.push_back(control_output);
477 Node* effect_output =
478 graph()->NewNode(common()->EffectPhi(return_count),
479 static_cast<int>(effects.size()), &effects.front());
480
481 // The first input of a return node is discarded. This is because Wasm
482 // functions always return an additional 0 constant as a first return value.
483 DCHECK(
484 Int32Matcher(NodeProperties::GetValueInput(return_nodes[0], 0)).Is(0));
485 int const return_arity = return_nodes[0]->op()->ValueInputCount() - 1;
486 NodeVector values(zone());
487#if DEBUG
488 for (Node* const return_node : return_nodes) {
489 // 3 = effect, control, first 0 return value.
490 CHECK_EQ(return_arity, return_node->InputCount() - 3);
491 }
492#endif
493 for (int i = 0; i < return_arity; i++) {
494 NodeVector ith_values(zone());
495 for (Node* const return_node : return_nodes) {
496 Node* value = NodeProperties::GetValueInput(return_node, i + 1);
497 ith_values.push_back(value);
498 }
499 ith_values.push_back(control_output);
500 // Find the correct machine representation for the return values from the
501 // inlinee signature.
502 MachineRepresentation repr =
503 inlinee_sig->GetReturn(i).machine_representation();
504 Node* ith_value_output = graph()->NewNode(
505 common()->Phi(repr, return_count),
506 static_cast<int>(ith_values.size()), &ith_values.front());
507 values.push_back(ith_value_output);
508 }
509 for (Node* return_node : return_nodes) return_node->Kill();
510
511 if (return_arity == 0) {
512 // Void function, no value uses.
513 ReplaceWithValue(call, mcgraph()->Dead(), effect_output, control_output);
514 } else if (return_arity == 1) {
515 // One return value. Just replace value uses of the call node with it.
516 // Note: This will automatically detect and replace the IfSuccess node
517 // correctly.
518 ReplaceWithValue(call, values[0], effect_output, control_output);
519 } else {
520 // Multiple returns. We have to find the projections of the call node and
521 // replace them with the returned values.
522 for (Edge use_edge : call->use_edges()) {
523 if (NodeProperties::IsValueEdge(use_edge)) {
524 Node* use = use_edge.from();
525 DCHECK_EQ(use->opcode(), IrOpcode::kProjection);
526 ReplaceWithValue(use, values[ProjectionIndexOf(use->op())]);
527 }
528 }
529 // All value inputs are replaced by the above loop, so it is ok to use
530 // Dead() as a dummy for value replacement.
531 ReplaceWithValue(call, mcgraph()->Dead(), effect_output, control_output);
532 }
533 } else {
534 // The callee can never return. The call node and all its uses are dead.
535 ReplaceWithValue(call, mcgraph()->Dead(), mcgraph()->Dead(),
536 mcgraph()->Dead());
537 }
538}
