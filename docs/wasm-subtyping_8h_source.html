25// Checks if type1, defined in module1, is equivalent with type2, defined in
26// module2.
27// Type equivalence (~) is described by the following rules:
28// - Two numeric types are equivalent iff they are equal.
29// - T(ht1) ~ T(ht2) iff ht1 ~ ht2 for T in {ref, ref null, rtt}.
30// Equivalence of heap types ht1 ~ ht2 is defined as follows:
31// - Two non-index heap types are equivalent iff they are equal.
32// - Two indexed heap types are equivalent iff they are iso-recursive
33// equivalent.

39// Checks if {subtype}, defined in {module1}, is a subtype of {supertype},
40// defined in {module2}.
41// Subtyping between value types is described by the following rules
42// (structural subtyping):
43// - numeric types are subtype-related iff they are equal.
44// - (ref null ht1) <: (ref null ht2) iff ht1 <: ht2.
45// - (ref ht1) <: (ref null? ht2) iff ht1 <: ht2.
46// - rtt1 <: rtt2 iff rtt1 ~ rtt2.
47// For heap types, the following subtyping rules hold:
48// - The abstract heap types form the following type hierarchies:
49//
50// any func extern
51// / \ | |
52// eq \ nofunc noextern
53// / | \ \
54// i31 array struct string
55// \___|______|_____/
56// |
57// none
58//
59// - All functions are subtypes of func.
60// - All structs are subtypes of struct.
61// - All arrays are subtypes of array.
62// - An indexed heap type h1 is a subtype of indexed heap type h2 if h2 is
63// transitively an explicit canonical supertype of h1.
64// Note that {any} includes references introduced by the host which belong to
65// none of any's subtypes (e.g. JS objects).
66V8_INLINE bool IsSubtypeOf(ValueType subtype, ValueType supertype,
67 const WasmModule* sub_module,
68 const WasmModule* super_module) {
69 if (subtype == supertype && sub_module == super_module) return true;
70 return IsSubtypeOfImpl(subtype, supertype, sub_module, super_module);
71}
72
73// Checks if {subtype} is a subtype of {supertype} (both defined in {module}).
74// TODO(369369573): Make sure this overload is not misused.

76 const WasmModule* module) {
77 // If the types are trivially identical, exit early.
78 if (V8_LIKELY(subtype == supertype)) return true;

111// Checks whether {subtype_index} is valid as a declared subtype of
112// {supertype_index}.
113// - Both type must be of the same kind (function, struct, or array).
114// - Both type must have the same {is_shared} flag.
115// - Structs: Subtype must have at least as many fields as supertype,
116// covariance for respective immutable fields, equivalence for respective
117// mutable fields.
118// - Arrays: subtyping of respective element types for immutable arrays,

121// respective parameter types, covariance for respective return types.

129struct TypeInModule {
130 ValueType type;
131 const WasmModule* module;
132
133 TypeInModule(ValueType type, const WasmModule* module)
134 : type(type), module(module) {}
135
136 TypeInModule() : TypeInModule(kWasmBottom, nullptr) {}
137
138 bool operator==(const TypeInModule& other) const {
139 return type == other.type && module == other.module;
140 }
141
142 bool operator!=(const TypeInModule& other) const {
143 return type != other.type || module != other.module;
144 }
145};
