14// A class that allows the allocation of contiguous temporary registers.
15class BytecodeRegisterAllocator final {
16 public:
17 // Enables observation of register allocation and free events.
18 class Observer {
19 public:
20 virtual ~Observer() = default;
21 virtual void RegisterAllocateEvent(Register reg) = 0;
22 virtual void RegisterListAllocateEvent(RegisterList reg_list) = 0;
23 virtual void RegisterListFreeEvent(RegisterList reg_list) = 0;
24 virtual void RegisterFreeEvent(Register reg_list) = 0;
25 };
26
27 explicit BytecodeRegisterAllocator(int start_index)
28 : next_register_index_(start_index),
29 max_register_count_(start_index),
30 observer_(nullptr) {}
31 ~BytecodeRegisterAllocator() = default;
32 BytecodeRegisterAllocator(const BytecodeRegisterAllocator&) = delete;
33 BytecodeRegisterAllocator& operator=(const BytecodeRegisterAllocator&) =
34 delete;

36 // Returns a new register.
37 Register NewRegister() {
38 Register reg(next_register_index_++);
39 max_register_count_ = std::max(next_register_index_, max_register_count_);
40 if (observer_) {
41 observer_->RegisterAllocateEvent(reg);
42 }
43 return reg;
44 }
45
46 // Returns a consecutive list of |count| new registers.
47 RegisterList NewRegisterList(int count) {
48 RegisterList reg_list(next_register_index_, count);
49 next_register_index_ += count;
50 max_register_count_ = std::max(next_register_index_, max_register_count_);
51 if (observer_) {
52 observer_->RegisterListAllocateEvent(reg_list);
53 }
54 return reg_list;
55 }
56
57 // Returns a growable register list.
58 RegisterList NewGrowableRegisterList() {
59 RegisterList reg_list(next_register_index_, 0);
60 return reg_list;
61 }
62
63 // Appends a new register to |reg_list| increasing it's count by one and
64 // returning the register added.
65 //
66 // Note: no other new registers must be currently allocated since the register
67 // list was originally allocated.
68 Register GrowRegisterList(RegisterList* reg_list) {
69 Register reg(NewRegister());
70 reg_list->IncrementRegisterCount();
71 // If the following CHECK fails then a register was allocated (and not
72 // freed) between the creation of the RegisterList and this call to add a
73 // Register.
74 CHECK_EQ(reg.index(), reg_list->last_register().index());
75 return reg;
76 }
77
78 // Release all registers above |register_index|.
79 void ReleaseRegisters(int register_index) {
80 int count = next_register_index_ - register_index;
81 next_register_index_ = register_index;
82 if (observer_) {
83 observer_->RegisterListFreeEvent(RegisterList(register_index, count));
84 }
85 }
86
87 // Release last allocated register
88 void ReleaseRegister(Register reg) {
89 DCHECK_EQ(next_register_index_ - 1, reg.index());
90 if (observer_) {
91 observer_->RegisterFreeEvent(reg);
92 }
93 next_register_index_--;
94 }
95
96 // Returns true if the register |reg| is a live register.
97 bool RegisterIsLive(Register reg) const {
98 return reg.index() < next_register_index_;
99 }
100
101 // Returns a register list for all currently live registers.
102 RegisterList AllLiveRegisters() const {
103 return RegisterList(0, next_register_index());
104 }
105
106 void set_observer(Observer* observer) { observer_ = observer; }
107
108 int next_register_index() const { return next_register_index_; }
109 int maximum_register_count() const { return max_register_count_; }
110
111 private:
112 int next_register_index_;
113 int max_register_count_;
114 Observer* observer_;
115};
