20class ReadOnlyHeapImageDeserializer final {
21 public:
22 static void Deserialize(Isolate* isolate, SnapshotByteSource* source) {
23 ReadOnlyHeapImageDeserializer{isolate, source}.DeserializeImpl();
24 }
25
26 private:
27 using Bytecode = ro::Bytecode;
28
29 ReadOnlyHeapImageDeserializer(Isolate* isolate, SnapshotByteSource* source)
30 : source_(source), isolate_(isolate) {}
31
32 void DeserializeImpl() {
33 while (true) {
34 int bytecode_as_int = source_->Get();
35 DCHECK_LT(bytecode_as_int, ro::kNumberOfBytecodes);
36 switch (static_cast<Bytecode>(bytecode_as_int)) {
37 case Bytecode::kAllocatePage:
38 AllocatePage(false);
39 break;
40 case Bytecode::kAllocatePageAt:
41 AllocatePage(true);
42 break;
43 case Bytecode::kSegment:
44 DeserializeSegment();
45 break;
46 case Bytecode::kRelocateSegment:
47 UNREACHABLE(); // Handled together with kSegment.
48 case Bytecode::kReadOnlyRootsTable:
49 DeserializeReadOnlyRootsTable();
50 break;
51 case Bytecode::kFinalizeReadOnlySpace:
52 ro_space()->FinalizeSpaceForDeserialization();
53 return;
54 }
55 }
56 }
57
58 void AllocatePage(bool fixed_offset) {
59 CHECK_EQ(V8_STATIC_ROOTS_BOOL, fixed_offset);
60 size_t expected_page_index = static_cast<size_t>(source_->GetUint30());
61 size_t actual_page_index = static_cast<size_t>(-1);
62 size_t area_size_in_bytes = static_cast<size_t>(source_->GetUint30());
63 if (fixed_offset) {
64#ifdef V8_COMPRESS_POINTERS
65 uint32_t compressed_page_addr = source_->GetUint32();
66 Address pos = isolate_->cage_base() + compressed_page_addr;
67 actual_page_index = ro_space()->AllocateNextPageAt(pos);
68#else
69 UNREACHABLE();
70#endif // V8_COMPRESS_POINTERS
71 } else {
72 actual_page_index = ro_space()->AllocateNextPage();
73 }
74 CHECK_EQ(actual_page_index, expected_page_index);
75 ro_space()->InitializePageForDeserialization(PageAt(actual_page_index),
76 area_size_in_bytes);
77 }
78
79 void DeserializeSegment() {
80 uint32_t page_index = source_->GetUint30();
81 ReadOnlyPageMetadata* page = PageAt(page_index);
82
83 // Copy over raw contents.
84 Address start = page->area_start() + source_->GetUint30();
85 int size_in_bytes = source_->GetUint30();
86 CHECK_LE(start + size_in_bytes, page->area_end());
87 source_->CopyRaw(reinterpret_cast<void*>(start), size_in_bytes);
88
89 if (!V8_STATIC_ROOTS_BOOL) {
90 uint8_t relocate_marker_bytecode = source_->Get();
91 CHECK_EQ(relocate_marker_bytecode, Bytecode::kRelocateSegment);
92 int tagged_slots_size_in_bits = size_in_bytes / kTaggedSize;
93 // The const_cast is unfortunate, but we promise not to mutate data.
94 uint8_t* data =
95 const_cast<uint8_t*>(source_->data() + source_->position());
96 ro::BitSet tagged_slots(data, tagged_slots_size_in_bits);
97 DecodeTaggedSlots(start, tagged_slots);
98 source_->Advance(static_cast<int>(tagged_slots.size_in_bytes()));
99 }
100 }
101
102 Address Decode(ro::EncodedTagged encoded) const {
103 ReadOnlyPageMetadata* page = PageAt(encoded.page_index);
104 return page->OffsetToAddress(encoded.offset * kTaggedSize);
105 }
106
107 void DecodeTaggedSlots(Address segment_start,
108 const ro::BitSet& tagged_slots) {
109 DCHECK(!V8_STATIC_ROOTS_BOOL);
110 for (size_t i = 0; i < tagged_slots.size_in_bits(); i++) {
111 // TODO(jgruber): Depending on sparseness, different iteration methods
112 // could be more efficient.
113 if (!tagged_slots.contains(static_cast<int>(i))) continue;
114 Address slot_addr = segment_start + i * kTaggedSize;
115 Address obj_addr = Decode(ro::EncodedTagged::FromAddress(slot_addr));
116 Address obj_ptr = obj_addr + kHeapObjectTag;
117
118 Tagged_t* dst = reinterpret_cast<Tagged_t*>(slot_addr);
119 *dst = COMPRESS_POINTERS_BOOL
120 ? V8HeapCompressionScheme::CompressObject(obj_ptr)
121 : static_cast<Tagged_t>(obj_ptr);
122 }
123 }
124
125 ReadOnlyPageMetadata* PageAt(size_t index) const {
126 DCHECK_LT(index, ro_space()->pages().size());
127 return ro_space()->pages()[index];
128 }
129
130 void DeserializeReadOnlyRootsTable() {
131 ReadOnlyRoots roots(isolate_);
132 if (V8_STATIC_ROOTS_BOOL) {
133 roots.InitFromStaticRootsTable(isolate_->cage_base());
134 } else {
135 for (size_t i = 0; i < ReadOnlyRoots::kEntriesCount; i++) {
136 uint32_t encoded_as_int = source_->GetUint32();
137 Address rudolf = Decode(ro::EncodedTagged::FromUint32(encoded_as_int));
138 roots.read_only_roots_[i] = rudolf + kHeapObjectTag;
139 }
140 }
141 }
142
143 ReadOnlySpace* ro_space() const {
144 return isolate_->read_only_heap()->read_only_space();
145 }

157void ReadOnlyDeserializer::DeserializeIntoIsolate() {
158 base::ElapsedTimer timer;
159 if (V8_UNLIKELY(v8_flags.profile_deserialization)) timer.Start();
160 NestedTimedHistogramScope histogram_timer(
161 isolate()->counters()->snapshot_deserialize_rospace());
162 HandleScope scope(isolate());
163
164 ReadOnlyHeapImageDeserializer::Deserialize(isolate(), source());
165 ReadOnlyHeap* ro_heap = isolate()->read_only_heap();
166 ro_heap->read_only_space()->RepairFreeSpacesAfterDeserialization();
167 PostProcessNewObjects();
168

171#ifdef DEBUG
172 roots.VerifyNameForProtectors();
173#endif
174

181 // ATTENTION: The Memory.json benchmark greps for this exact output. Do not

186 ms);

190void NoExternalReferencesCallback() {
191 // The following check will trigger if a function or object template with
192 // references to native functions have been deserialized from snapshot, but
193 // no actual external references were provided when the isolate was created.
194 FATAL("No external references provided via API");
195}

197class ObjectPostProcessor final {
198 public:
199 explicit ObjectPostProcessor(Isolate* isolate)
200 : isolate_(isolate), embedded_data_(EmbeddedData::FromBlob(isolate_)) {}
201
202 void Finalize() {
203#ifdef V8_ENABLE_SANDBOX
204 std::vector<ReadOnlyArtifacts::ExternalPointerRegistryEntry> registry;
205 registry.reserve(external_pointer_slots_.size());
206 for (auto& slot : external_pointer_slots_) {
207 registry.emplace_back(slot.Relaxed_LoadHandle(), slot.load(isolate_),
208 slot.exact_tag());
209 }
210
211 isolate_->read_only_artifacts()->set_external_pointer_registry(
212 std::move(registry));
213#endif // V8_ENABLE_SANDBOX
214 }
215#define POST_PROCESS_TYPE_LIST(V) \
216 V(AccessorInfo) \
217 V(FunctionTemplateInfo) \
218 V(Code) \
219 V(SharedFunctionInfo)
220
221 V8_INLINE void PostProcessIfNeeded(Tagged<HeapObject> o,
222 InstanceType instance_type) {
223 DCHECK_EQ(o->map(isolate_)->instance_type(), instance_type);
224#define V(TYPE) \
225 if (InstanceTypeChecker::Is##TYPE(instance_type)) { \
226 return PostProcess##TYPE(Cast<TYPE>(o)); \
227 }
228 POST_PROCESS_TYPE_LIST(V)
229#undef V
230 // If we reach here, no postprocessing is needed for this object.
231 }
232#undef POST_PROCESS_TYPE_LIST
233
234 private:
235 Address GetAnyExternalReferenceAt(int index, bool is_api_reference) const {
236 if (is_api_reference) {
237 const intptr_t* refs = isolate_->api_external_references();
238 Address address =
239 refs == nullptr
240 ? reinterpret_cast<Address>(NoExternalReferencesCallback)
241 : static_cast<Address>(refs[index]);
242 DCHECK_NE(address, kNullAddress);
243 return address;
244 }
245 // Note we allow `address` to be kNullAddress since some of our tests
246 // rely on this (e.g. when testing an incompletely initialized ER table).
247 return isolate_->external_reference_table_unsafe()->address(index);
248 }
249
250 void DecodeExternalPointerSlot(Tagged<HeapObject> host,
251 ExternalPointerSlot slot) {
252 // Constructing no_gc here is not the intended use pattern (instead we
253 // should pass it along the entire callchain); but there's little point of
254 // doing that here - all of the code in this file relies on GC being
255 // disabled, and that's guarded at entry points.
256 DisallowGarbageCollection no_gc;
257 auto encoded = ro::EncodedExternalReference::FromUint32(
258 slot.GetContentAsIndexAfterDeserialization(no_gc));
259 Address slot_value =
260 GetAnyExternalReferenceAt(encoded.index, encoded.is_api_reference);
261 DCHECK(slot.ExactTagIsKnown());
262 slot.init(isolate_, host, slot_value, slot.exact_tag());
263#ifdef V8_ENABLE_SANDBOX
264 // Register these slots during deserialization s.t. later isolates (which
265 // share the RO space we are currently deserializing) can properly
266 // initialize their external pointer table RO space. Note that slot values
267 // are only fully finalized at the end of deserialization, thus we only
268 // register the slot itself now and read the handle/value in Finalize.
269 external_pointer_slots_.emplace_back(slot);
270#endif // V8_ENABLE_SANDBOX
271 }
272 void PostProcessAccessorInfo(Tagged<AccessorInfo> o) {
273 DecodeExternalPointerSlot(
274 o, o->RawExternalPointerField(AccessorInfo::kSetterOffset,
275 kAccessorInfoSetterTag));
276 DecodeExternalPointerSlot(o, o->RawExternalPointerField(
277 AccessorInfo::kMaybeRedirectedGetterOffset,
278 kAccessorInfoGetterTag));
279 if (USE_SIMULATOR_BOOL) o->init_getter_redirection(isolate_);
280 }
281 void PostProcessFunctionTemplateInfo(Tagged<FunctionTemplateInfo> o) {
282 DecodeExternalPointerSlot(
283 o, o->RawExternalPointerField(
284 FunctionTemplateInfo::kMaybeRedirectedCallbackOffset,
285 kFunctionTemplateInfoCallbackTag));
286 if (USE_SIMULATOR_BOOL) o->init_callback_redirection(isolate_);
287 }
288 void PostProcessCode(Tagged<Code> o) {
289 o->init_self_indirect_pointer(isolate_);
290 o->wrapper()->set_code(o);
291 // RO space only contains builtin Code objects which don't have an
292 // attached InstructionStream.
293 DCHECK(o->is_builtin());
294 DCHECK(!o->has_instruction_stream());
295 o->SetInstructionStartForOffHeapBuiltin(
296 isolate_,
297 EmbeddedData::FromBlob(isolate_).InstructionStartOf(o->builtin_id()));
298 }
299 void PostProcessSharedFunctionInfo(Tagged<SharedFunctionInfo> o) {
300 // Reset the id to avoid collisions - it must be unique in this isolate.
301 o->set_unique_id(isolate_->GetAndIncNextUniqueSfiId());
302 }
303
304 Isolate* const isolate_;
305 const EmbeddedData embedded_data_;
306
307#ifdef V8_ENABLE_SANDBOX
308 std::vector<ExternalPointerSlot> external_pointer_slots_;
309#endif // V8_ENABLE_SANDBOX
310};

312void ReadOnlyDeserializer::PostProcessNewObjects() {
313 // Since we are not deserializing individual objects we need to scan the
314 // heap and search for objects that need post-processing.
315 //
316 // See also Deserializer<IsolateT>::PostProcessNewObject.
317 PtrComprCageBase cage_base(isolate());
318#ifdef V8_COMPRESS_POINTERS
319 ExternalPointerTable::UnsealReadOnlySegmentScope unseal_scope(
320 &isolate()->external_pointer_table());
321#endif // V8_COMPRESS_POINTERS
322 ObjectPostProcessor post_processor(isolate());
323 ReadOnlyHeapObjectIterator it(isolate()->read_only_heap());
324 for (Tagged<HeapObject> o = it.Next(); !o.is_null(); o = it.Next()) {
325 const InstanceType instance_type = o->map(cage_base)->instance_type();
326 if (should_rehash()) {
327 if (InstanceTypeChecker::IsString(instance_type)) {
328 Tagged<String> str = Cast<String>(o);
329 str->set_raw_hash_field(Name::kEmptyHashField);
330 PushObjectToRehash(direct_handle(str, isolate()));
331 } else if (o->NeedsRehashing(instance_type)) {
332 PushObjectToRehash(direct_handle(o, isolate()));
333 }
334 }
335
336 post_processor.PostProcessIfNeeded(o, instance_type);
337 }
338 post_processor.Finalize();
339}
