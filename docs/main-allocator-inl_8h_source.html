16AllocationResult MainAllocator::AllocateRaw(int size_in_bytes,
17 AllocationAlignment alignment,
18 AllocationOrigin origin) {
19 size_in_bytes = ALIGN_TO_ALLOCATION_ALIGNMENT(size_in_bytes);
20
21 DCHECK_EQ(in_gc(), origin == AllocationOrigin::kGC);
22 DCHECK_EQ(in_gc(), isolate_heap()->IsInGC());
23
24 // We are not supposed to allocate in fast c calls.
25 DCHECK_IMPLIES(is_main_thread(),
26 v8_flags.allow_allocation_in_fast_api_call ||
27 !isolate_heap()->isolate()->InFastCCall());
28
29 AllocationResult result;
30
31 if (USE_ALLOCATION_ALIGNMENT_BOOL && alignment != kTaggedAligned) {
32 result = AllocateFastAligned(size_in_bytes, nullptr, alignment, origin);
33 } else {
34 result = AllocateFastUnaligned(size_in_bytes, origin);
35 }
36
37 return result.IsFailure() ? AllocateRawSlow(size_in_bytes, alignment, origin)
38 : result;
39}

41AllocationResult MainAllocator::AllocateFastUnaligned(int size_in_bytes,
42 AllocationOrigin origin) {
43 size_in_bytes = ALIGN_TO_ALLOCATION_ALIGNMENT(size_in_bytes);
44 if (!allocation_info().CanIncrementTop(size_in_bytes)) {
45 return AllocationResult::Failure();
46 }
47 Tagged<HeapObject> obj =
48 HeapObject::FromAddress(allocation_info().IncrementTop(size_in_bytes));
49
50 MSAN_ALLOCATED_UNINITIALIZED_MEMORY(obj.address(), size_in_bytes);
51
52 DCHECK_IMPLIES(black_allocation_ == BlackAllocation::kAlwaysEnabled,
53 space_heap()->marking_state()->IsMarked(obj));
54
55 return AllocationResult::FromObject(obj);
56}

58AllocationResult MainAllocator::AllocateFastAligned(
59 int size_in_bytes, int* result_aligned_size_in_bytes,
60 AllocationAlignment alignment, AllocationOrigin origin) {
61 Address top = allocation_info().top();
62 int filler_size = Heap::GetFillToAlign(top, alignment);
63 int aligned_size_in_bytes = size_in_bytes + filler_size;
64
65 if (!allocation_info().CanIncrementTop(aligned_size_in_bytes)) {
66 return AllocationResult::Failure();
67 }
68 Tagged<HeapObject> obj = HeapObject::FromAddress(
69 allocation_info().IncrementTop(aligned_size_in_bytes));
70 if (result_aligned_size_in_bytes)
71 *result_aligned_size_in_bytes = aligned_size_in_bytes;
72
73 if (filler_size > 0) {
74 obj = space_heap()->PrecedeWithFiller(obj, filler_size);
75 }
76
77 MSAN_ALLOCATED_UNINITIALIZED_MEMORY(obj.address(), size_in_bytes);
78
79 DCHECK_IMPLIES(black_allocation_ == BlackAllocation::kAlwaysEnabled,
80 space_heap()->marking_state()->IsMarked(obj));
81
82 return AllocationResult::FromObject(obj);
83}

85bool MainAllocator::TryFreeLast(Address object_address, int object_size) {
86 if (top() != kNullAddress) {
87 return allocation_info().DecrementTopIfAdjacent(object_address,
88 object_size);
89 }
90 return false;
91}
