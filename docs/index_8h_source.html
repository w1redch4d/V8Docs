26// Operations are stored in possibly muliple sequential storage slots.
27using OperationStorageSlot = std::aligned_storage_t<8, 8>;
28// Operations occupy at least 2 slots, therefore we assign one id per two slots.
29constexpr size_t kSlotsPerId = 2;

34// `OpIndex` is an offset from the beginning of the operations buffer.
35// Compared to `Operation*`, it is more memory efficient (32bit) and stable when
36// the operations buffer is re-allocated.
37class OpIndex {
38 protected:
39 // We make this constructor protected so that integers are not easily
40 // convertible to OpIndex. FromOffset should be used instead to create an
41 // OpIndex from an offset.
42 explicit constexpr OpIndex(uint32_t offset) : offset_(offset) {
43 SLOW_DCHECK(CheckInvariants());
44 }
45 friend class OperationBuffer;
46
47 public:
48 static constexpr OpIndex FromOffset(uint32_t offset) {
49 return OpIndex(offset);
50 }
51 constexpr OpIndex() : offset_(std::numeric_limits<uint32_t>::max()) {}
52 template <typename T, typename C>
53 OpIndex(const ConstOrV<T, C>&) { // NOLINT(runtime/explicit)
54 static_assert(base::tmp::lazy_false<T>::value,
55 "Cannot initialize OpIndex from ConstOrV<>. Did you forget "
56 "to resolve() it in the assembler?");
57 }
58
59 constexpr uint32_t id() const {
60 // Operations are stored at an offset that's a multiple of
61 // `sizeof(OperationStorageSlot)`. In addition, an operation occupies at
62 // least `kSlotsPerId` many `OperationSlot`s. Therefore, we can assign id's
63 // by dividing by `kSlotsPerId`. A compact id space is important, because it
64 // makes side-tables smaller.
65 SLOW_DCHECK(CheckInvariants());
66 return offset_ / sizeof(OperationStorageSlot) / kSlotsPerId;
67 }
68 uint32_t hash() const {
69 // It can be useful to hash OpIndex::Invalid(), so we have this `hash`
70 // function, which returns the id, but without DCHECKing that Invalid is
71 // valid.
72 SLOW_DCHECK_IMPLIES(valid(), CheckInvariants());
73 return offset_ / sizeof(OperationStorageSlot) / kSlotsPerId;
74 }
75 uint32_t offset() const {
76 SLOW_DCHECK(CheckInvariants());
77#ifdef DEBUG
78 return offset_ & kUnmaskGenerationMask;
79#else
80 return offset_;
81#endif
82 }
83
84 constexpr bool valid() const { return *this != Invalid(); }
85
86 static constexpr OpIndex Invalid() { return OpIndex(); }
87
88 // Encode a sea-of-nodes node id in the `OpIndex` type.
89 // Only used for node origins that actually point to sea-of-nodes graph nodes.
90 static OpIndex EncodeTurbofanNodeId(uint32_t id) {
91 OpIndex result = OpIndex(id * sizeof(OperationStorageSlot));
92 result.offset_ += kTurbofanNodeIdFlag;
93 return result;
94 }
95 uint32_t DecodeTurbofanNodeId() const {
96 DCHECK(IsTurbofanNodeId());
97 return offset_ / sizeof(OperationStorageSlot);
98 }
99 bool IsTurbofanNodeId() const {
100 return offset_ % sizeof(OperationStorageSlot) == kTurbofanNodeIdFlag;
101 }
102
103 constexpr bool operator==(OpIndex other) const {
104 return offset_ == other.offset_;
105 }
106 constexpr bool operator!=(OpIndex other) const {
107 return offset_ != other.offset_;
108 }
109 constexpr bool operator<(OpIndex other) const {
110 return offset_ < other.offset_;
111 }
112 constexpr bool operator>(OpIndex other) const {
113 return offset_ > other.offset_;
114 }
115 constexpr bool operator<=(OpIndex other) const {
116 return offset_ <= other.offset_;
117 }
118 constexpr bool operator>=(OpIndex other) const {
119 return offset_ >= other.offset_;
120 }
121
122#ifdef DEBUG
123 int generation_mod2() const {
124 return (offset_ & kGenerationMask) >> kGenerationMaskShift;
125 }
126 void set_generation_mod2(int generation_mod2) {
127 DCHECK_LE(generation_mod2, 1);
128 offset_ |= generation_mod2 << kGenerationMaskShift;
129 }
130
131 constexpr bool CheckInvariants() const {
132 DCHECK(valid());
133 // The second lowest significant bit of the offset is used to store the
134 // graph generation modulo 2. The lowest and 3rd lowest bits should always
135 // be 0 (as long as sizeof(OperationStorageSlot) is 8).
136 static_assert(sizeof(OperationStorageSlot) == 8);
137 return (offset_ & 0b101) == 0;
138 }
139#endif
140
141 protected:
142 static constexpr uint32_t kGenerationMaskShift = 1;
143 static constexpr uint32_t kGenerationMask = 1 << kGenerationMaskShift;
144 static constexpr uint32_t kUnmaskGenerationMask = ~kGenerationMask;
145
146 // In DEBUG builds, the offset's second lowest bit contains the graph
147 // generation % 2, so one should keep this in mind when looking at the value
148 // of the offset.
149 uint32_t offset_;
150
151 static constexpr uint32_t kTurbofanNodeIdFlag = 1;
152
153 template <typename H>
154 friend H AbslHashValue(H h, const OpIndex& idx) {
155 return H::combine(std::move(h), idx.offset_);
156 }
157};

570template <typename T>
571constexpr bool IsValidTypeFor(RegisterRepresentation repr) {
572 if (std::is_same_v<T, Any>) return true;
573
574 switch (repr.value()) {
575 case RegisterRepresentation::Enum::kWord32:
576 return std::is_same_v<T, Word> || std::is_same_v<T, Word32> ||
577 std::is_same_v<T, Untagged>;
578 case RegisterRepresentation::Enum::kWord64:
579 return std::is_same_v<T, Word> || std::is_same_v<T, Word64> ||
580 std::is_same_v<T, Untagged>;
581 case RegisterRepresentation::Enum::kFloat32:
582 return std::is_same_v<T, Float> || std::is_same_v<T, Float32> ||
583 std::is_same_v<T, Untagged>;
584 case RegisterRepresentation::Enum::kFloat64:
585 return std::is_same_v<T, Float> || std::is_same_v<T, Float64> ||
586 std::is_same_v<T, Untagged>;
587 case RegisterRepresentation::Enum::kTagged:
588 return is_subtype_v<T, Object>;
589 case RegisterRepresentation::Enum::kCompressed:
590 return is_subtype_v<T, Object>;
591 case RegisterRepresentation::Enum::kSimd128:
592 return std::is_same_v<T, Simd128>;
593 case RegisterRepresentation::Enum::kSimd256:
594 return std::is_same_v<T, Simd256>;
595 }

598// V<> represents an SSA-value that is parameterized with the type of the value.
599// Types from the `Object` hierarchy can be provided as well as the abstract
600// representation classes (`Word32`, ...) defined above.
601// Prefer using V<> instead of a plain OpIndex where possible.
602template <typename T>
603class V : public OpIndex {
604 // V<T> is implicitly constructible from V<U> iff
605 // `v_traits<T>::implicitly_constructible_from<U>::value`. This is typically
606 // the case if T == U or T is a subclass of U. Different types may specify
607 // different conversion rules in the corresponding `v_traits` when necessary.
608 template <typename U>
609 constexpr static bool implicitly_constructible_from =
610 v_traits<T>::template implicitly_constructible_from<U>::value;
611
612 public:
613 using type = T;
614 static constexpr auto rep = v_traits<type>::rep;
615 constexpr V() : OpIndex() {}
616
617 // V<T> is implicitly constructible from V<U> iff
618 // `v_traits<T>::implicitly_constructible_from<U>::value`. This is typically
619 // the case if T == U or T is a subclass of U. Different types may specify
620 // different conversion rules in the corresponding `v_traits` when necessary.
621 template <typename U>
622 requires implicitly_constructible_from<U>
623 V(V<U> index) : OpIndex(index) {} // NOLINT(runtime/explicit)
624
625 static V Invalid() { return V<T>(OpIndex::Invalid()); }
626
627 template <typename U>
628 static V<T> Cast(V<U> index) {
629 return V<T>(OpIndex{index});
630 }
631 static V<T> Cast(OpIndex index) { return V<T>(index); }
632
633 static constexpr bool allows_representation(
634 RegisterRepresentation maybe_allowed_rep) {
635 return v_traits<T>::allows_representation(maybe_allowed_rep);
636 }
637
638#if !defined(TURBOSHAFT_ALLOW_IMPLICIT_OPINDEX_INITIALIZATION_FOR_V)
639
640 protected:
641#endif
642 // V<T> is implicitly constructible from plain OpIndex.

644 requires(std::is_same_v<U, OpIndex>)

703// ConstOrV<> is a generalization of V<> that allows constexpr values
704// (constants) to be passed implicitly. This allows reducers to write things
705// like
706//
707// __ Word32Add(value, 1)
708//
709// instead of having to write
710//
711// __ Word32Add(value, __ Word32Constant(1))
712//
713// which makes overall code more compact and easier to read. Functions need to
714// call `resolve` on the assembler in order to convert to V<> (which will then
715// construct the corresponding ConstantOp if the given ConstOrV<> holds a
716// constexpr value).
717// NOTICE: `ConstOrV<T>` can only be used if `v_traits<T>` provides a
718// `constexpr_type`.
719template <typename T, typename C = typename v_traits<T>::constexpr_type>
720class ConstOrV {
721 public:
722 using type = T;
723 using constant_type = C;
724
725 ConstOrV(constant_type value) // NOLINT(runtime/explicit)
726 : constant_value_(value), value_() {}
727
728 // ConstOrV<T> is implicitly constructible from V<U> iff V<T> is
729 // constructible from V<U>.
730 template <typename U>
731 ConstOrV(V<U> index) // NOLINT(runtime/explicit)
732 requires(std::is_constructible_v<V<T>, V<U>>)
733 : constant_value_(std::nullopt), value_(index) {}
734
735 bool is_constant() const { return constant_value_.has_value(); }
736 constant_type constant_value() const {
737 DCHECK(is_constant());
738 return *constant_value_;
739 }
740 V<type> value() const {
741 DCHECK(!is_constant());
742 return value_;
743 }
744
745#if !defined(TURBOSHAFT_ALLOW_IMPLICIT_OPINDEX_INITIALIZATION_FOR_V)
746
747 protected:
748#endif
749 // ConstOrV<T> is implicitly constructible from plain OpIndex.
750 template <typename U>
751 ConstOrV(U index) // NOLINT(runtime/explicit)
752 requires(std::is_same_v<U, OpIndex>)
753 : constant_value_(), value_(index) {}

755 private:
756 std::optional<constant_type> constant_value_;
