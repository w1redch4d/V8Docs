18void WasmGCTypeAnalyzer::Run() {
19 LoopFinder loop_finder(phase_zone_, &graph_);
20 AnalyzerIterator iterator(phase_zone_, graph_, loop_finder);
21 while (iterator.HasNext()) {
22 const Block& block = *iterator.Next();
23 ProcessBlock(block);
24
25 // Finish snapshot.
26 block_to_snapshot_[block.index()] = MaybeSnapshot(types_table_.Seal());
27
28 // Consider re-processing for loops.
29 if (const GotoOp* last = block.LastOperation(graph_).TryCast<GotoOp>()) {
30 if (IsReachable(block) && last->destination->IsLoop() &&
31 last->destination->LastPredecessor() == &block) {
32 TRACE("[b%u] Reprocessing loop header b%u at backedge #%u\n",
33 block.index().id(), last->destination->index().id(),
34 graph_.Index(block.LastOperation(graph_)).id());
35 const Block& loop_header = *last->destination;
36 // Create a merged snapshot state for the forward- and backedge and
37 // process all operations inside the loop header.
38 ProcessBlock(loop_header);
39 Snapshot old_snapshot = block_to_snapshot_[loop_header.index()].value();
40 Snapshot snapshot = types_table_.Seal();
41 // TODO(14108): The snapshot isn't needed at all, we only care about the
42 // information if two snapshots are equivalent. Unfortunately, currently
43 // this can only be answered by creating a merge snapshot.
44 bool needs_revisit =
45 CreateMergeSnapshot(base::VectorOf({old_snapshot, snapshot}),
46 base::VectorOf({true, true}));
47 types_table_.Seal(); // Discard the snapshot.
48
49 TRACE("[b%u] Loop header b%u reprocessed at backedge #%u: %s\n",
50 block.index().id(), last->destination->index().id(),
51 graph_.Index(block.LastOperation(graph_)).id(),
52 needs_revisit ? "Scheduling loop body revisitation"
53 : "No revisit of loop body needed");
54
55 // TODO(14108): This currently encodes a fixed point analysis where the
56 // analysis is finished once the backedge doesn't provide updated type
57 // information any more compared to the previous iteration. This could
58 // be stopped in cases where the backedge only refines types (i.e. only
59 // defines more precise types than the previous iteration).
60 if (needs_revisit) {
61 block_to_snapshot_[loop_header.index()] = MaybeSnapshot(snapshot);
62 if (block.index() != loop_header.index()) {
63 // This will push the successors of the loop header to the iterator
64 // stack, so the loop body will be visited in the next iteration.
65 iterator.MarkLoopForRevisitSkipHeader();
66 } else {
67 // A single-block loop doesn't have any successors which would be
68 // re-evaluated and which might trigger another re-evaluation of the
69 // loop header.
70 // TODO(mliedtke): This is not a great design: We don't just
71 // schedule revisiting the loop header but afterwards we revisit it
72 // once again to evaluate whether we need to revisit it more times,
73 // so for single block loops the revisitation count will always be a
74 // multiple of 2. While this is inefficient, single-block loops are
75 // rare and are either endless loops or need to trigger an exception
76 // (e.g. a wasm trap) to terminate.
77 iterator.MarkLoopForRevisit();
78 }
79 }
80 }
81 }
82 }
83}

86 DCHECK_NULL(current_block_);
87 current_block_ = &block;
88 StartNewSnapshotFor(block);
89 ProcessOperations(block);
90 current_block_ = nullptr;

93void WasmGCTypeAnalyzer::StartNewSnapshotFor(const Block& block) {
94 is_first_loop_header_evaluation_ = false;
95 // Reset reachability information. This can be outdated in case of loop
96 // revisits. Below the reachability is calculated again and potentially
97 // re-added.
98 bool block_was_previously_reachable = IsReachable(block);
99 if (!block_was_previously_reachable) {
100 TRACE("[b%u] Removing unreachable flag as block is re-evaluated\n",
101 block.index().id());
102 }
103 block_is_unreachable_.Remove(block.index().id());
104 // Start new snapshot based on predecessor information.
105 if (block.HasPredecessors() == 0) {
106 // The first block just starts with an empty snapshot.
107 DCHECK_EQ(block.index().id(), 0);
108 types_table_.StartNewSnapshot();
109 } else if (block.IsLoop()) {
110 const Block& forward_predecessor =
111 *block.LastPredecessor()->NeighboringPredecessor();
112 if (!IsReachable(forward_predecessor)) {
113 // If a loop isn't reachable through its forward edge, it can't possibly
114 // become reachable via the backedge.
115 TRACE(
116 "[b%uu] Loop unreachable as forward predecessor b%u is unreachable\n",
117 block.index().id(), forward_predecessor.index().id());
118 block_is_unreachable_.Add(block.index().id());
119 }
120 MaybeSnapshot back_edge_snap =
121 block_to_snapshot_[block.LastPredecessor()->index()];
122 if (back_edge_snap.has_value() && block_was_previously_reachable) {
123 // The loop was already visited at least once. In this case use the
124 // available information from the backedge.
125 // Note that we only do this if the loop wasn't marked as unreachable
126 // before. This solves an issue where a single block loop would think the
127 // backedge is reachable as we just removed the unreachable information
128 // above. Once the analyzer hits the backedge, it will re-evaluate if the
129 // backedge changes any analysis results and then potentially revisit
130 // this loop with forward edge and backedge.
131 CreateMergeSnapshot(block);
132 } else {
133 // The loop wasn't visited yet. There isn't any type information available
134 // for the backedge.
135 TRACE(
136 "[b%u%s] First loop header evaluation: Ignoring all backedges on "
137 "phis\n",
138 block.index().id(), !IsReachable(*current_block_) ? "u" : "");
139 is_first_loop_header_evaluation_ = true;
140 Snapshot forward_edge_snap =
141 block_to_snapshot_[forward_predecessor.index()].value();
142 types_table_.StartNewSnapshot(forward_edge_snap);
143 }
144 } else if (block.IsBranchTarget()) {
145 DCHECK_EQ(block.PredecessorCount(), 1);
146 const Block& predecessor = *block.LastPredecessor();
147 types_table_.StartNewSnapshot(
148 block_to_snapshot_[predecessor.index()].value());
149 if (IsReachable(predecessor)) {
150 const BranchOp* branch =
151 block.Predecessors()[0]->LastOperation(graph_).TryCast<BranchOp>();
152 if (branch != nullptr) {
153 ProcessBranchOnTarget(*branch, block);
154 }
155 } else {
156 TRACE("[b%uu] Block unreachable as sole predecessor b%u is unreachable\n",
157 block.index().id(), predecessor.index().id());
158 block_is_unreachable_.Add(block.index().id());
159 }
160 } else {
161 DCHECK_EQ(block.kind(), Block::Kind::kMerge);
162 CreateMergeSnapshot(block);
163 }
164}

166void WasmGCTypeAnalyzer::ProcessOperations(const Block& block) {
167 for (OpIndex op_idx : graph_.OperationIndices(block)) {
168 Operation& op = graph_.Get(op_idx);
169 switch (op.opcode) {
170 case Opcode::kWasmTypeCast:
171 ProcessTypeCast(op.Cast<WasmTypeCastOp>());
172 break;
173 case Opcode::kWasmTypeCheck:
174 ProcessTypeCheck(op.Cast<WasmTypeCheckOp>());
175 break;
176 case Opcode::kAssertNotNull:
177 ProcessAssertNotNull(op.Cast<AssertNotNullOp>());
178 break;
179 case Opcode::kNull:
180 ProcessNull(op.Cast<NullOp>());
181 break;
182 case Opcode::kIsNull:
183 ProcessIsNull(op.Cast<IsNullOp>());
184 break;
185 case Opcode::kParameter:
186 ProcessParameter(op.Cast<ParameterOp>());
187 break;
188 case Opcode::kStructGet:
189 ProcessStructGet(op.Cast<StructGetOp>());
190 break;
191 case Opcode::kStructSet:
192 ProcessStructSet(op.Cast<StructSetOp>());
193 break;
194 case Opcode::kArrayGet:
195 ProcessArrayGet(op.Cast<ArrayGetOp>());
196 break;
197 case Opcode::kArrayLength:
198 ProcessArrayLength(op.Cast<ArrayLengthOp>());
199 break;
200 case Opcode::kGlobalGet:
201 ProcessGlobalGet(op.Cast<GlobalGetOp>());
202 break;
203 case Opcode::kWasmRefFunc:
204 ProcessRefFunc(op.Cast<WasmRefFuncOp>());
205 break;
206 case Opcode::kWasmAllocateArray:
207 ProcessAllocateArray(op.Cast<WasmAllocateArrayOp>());
208 break;
209 case Opcode::kWasmAllocateStruct:
210 ProcessAllocateStruct(op.Cast<WasmAllocateStructOp>());
211 break;
212 case Opcode::kPhi:
213 ProcessPhi(op.Cast<PhiOp>());
214 break;
215 case Opcode::kWasmTypeAnnotation:
216 ProcessTypeAnnotation(op.Cast<WasmTypeAnnotationOp>());
217 break;
218 case Opcode::kBranch:
219 // Handling branch conditions implying special values is handled on the
220 // beginning of the successor block.
221 default:
222 break;
223 }
224 }
225}

323wasm::ValueType WasmGCTypeAnalyzer::GetTypeForPhiInput(const PhiOp& phi,
324 int input_index) {
325 OpIndex phi_id = graph_.Index(phi);
326 OpIndex input = ResolveAliases(phi.input(input_index));
327 // If the input of the phi is in the same block as the phi and appears
328 // before the phi, don't use the predecessor value.
329
330 if (current_block_->begin().id() <= input.id() && input.id() < phi_id.id()) {
331 // Phi instructions have to be at the beginning of the block, so this can
332 // only happen for inputs that are also phis. Furthermore, this is only
333 // possible in loop headers of loops and only for the backedge-input.
334 DCHECK(graph_.Get(input).Is<PhiOp>());
335 DCHECK(current_block_->IsLoop());
336 DCHECK_EQ(input_index, 1);
337 return types_table_.Get(input);
338 }
339 return types_table_.GetPredecessorValue(input, input_index);
340}

342void WasmGCTypeAnalyzer::ProcessPhi(const PhiOp& phi) {
343 // The result type of a phi is the union of all its input types.
344 // If any of the inputs is the default value ValueType(), there isn't any type
345 // knowledge inferrable.
346 DCHECK_GT(phi.input_count, 0);
347 if (is_first_loop_header_evaluation_) {
348 // We don't know anything about the backedge yet, so we only use the
349 // forward edge. We will revisit the loop header again once the block with
350 // the back edge is evaluated.
351 RefineTypeKnowledge(graph_.Index(phi), GetResolvedType((phi.input(0))),
352 phi);
353 return;
354 }
355 wasm::ValueType union_type = GetTypeForPhiInput(phi, 0);
356 if (union_type == wasm::ValueType()) return;
357 for (int i = 1; i < phi.input_count; ++i) {
358 wasm::ValueType input_type = GetTypeForPhiInput(phi, i);
359 if (input_type == wasm::ValueType()) return;
360 // <bottom> types have to be skipped as an unreachable predecessor doesn't
361 // change our type knowledge.
362 // TODO(mliedtke): Ideally, we'd skip unreachable predecessors here
363 // completely, as we might loosen the known type due to an unreachable
364 // predecessor.
365 if (input_type.is_uninhabited()) continue;
366 if (union_type.is_uninhabited()) {
367 union_type = input_type;
368 } else {
369 union_type = wasm::Union(union_type, input_type, module_, module_).type;
370 }
371 }
372 RefineTypeKnowledge(graph_.Index(phi), union_type, phi);
373 if (v8_flags.trace_wasm_typer) {
374 for (int i = 0; i < phi.input_count; ++i) {
375 OpIndex input = phi.input(i);
376 wasm::ValueType type = GetTypeForPhiInput(phi, i);
377 TRACE("- phi input %d: #%u(%s) -> %s\n", i, input.id(),
378 OpcodeName(graph_.Get(input).opcode), type.name().c_str());
379 }
380 }
381}

389void WasmGCTypeAnalyzer::ProcessBranchOnTarget(const BranchOp& branch,
390 const Block& target) {
391 DCHECK_EQ(current_block_, &target);
392 const Operation& condition = graph_.Get(branch.condition());
393 switch (condition.opcode) {
394 case Opcode::kWasmTypeCheck: {
395 const WasmTypeCheckOp& check = condition.Cast<WasmTypeCheckOp>();
396 if (branch.if_true == &target) {
397 // It is known from now on that the type is at least the checked one.
398 RefineTypeKnowledge(check.object(), check.config.to, branch);
399 } else {
400 DCHECK_EQ(branch.if_false, &target);
401 if (wasm::IsSubtypeOf(GetResolvedType(check.object()), check.config.to,
402 module_)) {
403 // The type check always succeeds, the target is impossible to be
404 // reached.
405 DCHECK_EQ(target.PredecessorCount(), 1);
406 block_is_unreachable_.Add(target.index().id());
407 TRACE(
408 "[b%uu] Block unreachable as #%u(%s) used in #%u(%s) is always "
409 "true\n",
410 target.index().id(), branch.condition().id(),
411 OpcodeName(condition.opcode), graph_.Index(branch).id(),
412 OpcodeName(branch.opcode));
413 }
414 }
415 } break;
416 case Opcode::kIsNull: {
417 const IsNullOp& is_null = condition.Cast<IsNullOp>();
418 if (branch.if_true == &target) {
419 if (GetResolvedType(is_null.object()).is_non_nullable()) {
420 // The target is impossible to be reached.
421 DCHECK_EQ(target.PredecessorCount(), 1);
422 block_is_unreachable_.Add(target.index().id());
423 TRACE(
424 "[b%uu] Block unreachable as #%u(%s) used in #%u(%s) is always "
425 "false\n",
426 target.index().id(), branch.condition().id(),
427 OpcodeName(condition.opcode), graph_.Index(branch).id(),
428 OpcodeName(branch.opcode));
429 return;
430 }
431 RefineTypeKnowledge(is_null.object(),
432 wasm::ToNullSentinel({is_null.type, module_}),
433 branch);
434 } else {
435 DCHECK_EQ(branch.if_false, &target);
436 RefineTypeKnowledge(is_null.object(), is_null.type.AsNonNull(), branch);
437 }
438 } break;
439 default:
440 break;
441 }
442}

449void WasmGCTypeAnalyzer::CreateMergeSnapshot(const Block& block) {
450 base::SmallVector<Snapshot, 8> snapshots;
451 // Unreachable predecessors should be ignored when merging but we can't remove
452 // them from the predecessors as that would mess up the phi inputs. Therefore
453 // the reachability of the predecessors is passed as a separate list.
454 base::SmallVector<bool, 8> reachable;
455 bool all_predecessors_unreachable = true;
456 for (const Block* predecessor : block.PredecessorsIterable()) {
457 snapshots.push_back(block_to_snapshot_[predecessor->index()].value());
458 bool predecessor_reachable = IsReachable(*predecessor);
459 reachable.push_back(predecessor_reachable);
460 all_predecessors_unreachable &= !predecessor_reachable;
461 }
462 if (all_predecessors_unreachable) {
463 TRACE("[b%u] Block unreachable as all predecessors are unreachable\n",
464 block.index().id());
465 block_is_unreachable_.Add(block.index().id());
466 } else if (v8_flags.trace_wasm_typer) {
467 std::stringstream str;
468 size_t i = 0;
469 for (const Block* predecessor : block.PredecessorsIterable()) {
470 if (i != 0) str << ", ";
471 str << 'b' << predecessor->index().id() << (reachable[i] ? "" : "u");
472 ++i;
473 }
474 TRACE("[b%u] Predecessors reachability: %s\n", block.index().id(),
475 str.str().c_str());
476 }
477 // The predecessor snapshots need to be reversed to restore the "original"
478 // order of predecessors. (This is used to map phi inputs to their
479 // corresponding predecessor.)
480 std::reverse(snapshots.begin(), snapshots.end());
481 std::reverse(reachable.begin(), reachable.end());
482 CreateMergeSnapshot(base::VectorOf(snapshots), base::VectorOf(reachable));
483}

485bool WasmGCTypeAnalyzer::CreateMergeSnapshot(
486 base::Vector<const Snapshot> predecessors,
487 base::Vector<const bool> reachable) {
488 DCHECK_EQ(predecessors.size(), reachable.size());
489 // The merging logic is also used to evaluate if two snapshots are
490 // "identical", i.e. the known types for all operations are the same.
491 bool types_are_equivalent = true;
492 types_table_.StartNewSnapshot(
493 predecessors, [this, &types_are_equivalent, reachable](
494 TypeSnapshotTable::Key,
495 base::Vector<const wasm::ValueType> predecessors) {
496 DCHECK_GT(predecessors.size(), 1);
497 size_t i = 0;
498 // Initialize the type based on the first reachable predecessor.
499 wasm::ValueType first = wasm::kWasmBottom;
500 for (; i < reachable.size(); ++i) {
501 // Uninhabitated types can only occur in unreachable code e.g. as a
502 // result of an always failing cast. Still reachability tracking might
503 // in some cases miss that a block becomes unreachable, so we still
504 // check for uninhabited in the if below.
505 DCHECK_IMPLIES(reachable[i], !predecessors[i].is_uninhabited());
506 if (reachable[i] && !predecessors[i].is_uninhabited()) {
507 first = predecessors[i];
508 ++i;
509 break;
510 }
511 }
512
513 wasm::ValueType res = first;
514 for (; i < reachable.size(); ++i) {
515 if (!reachable[i]) continue; // Skip unreachable predecessors.
516 wasm::ValueType type = predecessors[i];
517 // Uninhabitated types can only occur in unreachable code e.g. as a
518 // result of an always failing cast. Still reachability tracking might
519 // in some cases miss that a block becomes unreachable, so we still
520 // check for uninhabited in the if below.
521 DCHECK(!type.is_uninhabited());
522 if (type.is_uninhabited()) continue;
523 types_are_equivalent &= first == type;
524 if (res == wasm::ValueType() || type == wasm::ValueType()) {
525 res = wasm::ValueType();
526 } else {
527 res = wasm::Union(res, type, module_, module_).type;
528 }
529 }
530 return res;
531 });
532 return !types_are_equivalent;
533}

535wasm::ValueType WasmGCTypeAnalyzer::RefineTypeKnowledge(
536 OpIndex object, wasm::ValueType new_type, const Operation& op) {
537 DCHECK_NOT_NULL(current_block_);
538 object = ResolveAliases(object);
539 wasm::ValueType previous_value = types_table_.Get(object);
540 wasm::ValueType intersection_type =
541 previous_value == wasm::ValueType()
542 ? new_type
543 : wasm::Intersection(previous_value, new_type, module_, module_).type;
544 if (intersection_type == previous_value) return previous_value;
545
546 TRACE("[b%u%s] #%u(%s): Refine type for object #%u(%s) -> %s%s\n",
547 current_block_->index().id(), !IsReachable(*current_block_) ? "u" : "",
548 graph_.Index(op).id(), OpcodeName(op.opcode), object.id(),
549 OpcodeName(graph_.Get(object).opcode), intersection_type.name().c_str(),
550 intersection_type.is_uninhabited() ? " (unreachable!)" : "");
551
552 types_table_.Set(object, intersection_type);
553 if (intersection_type.is_uninhabited()) {
554 // After this instruction all other instructions in the current block are
555 // unreachable.
556 block_is_unreachable_.Add(current_block_->index().id());
557 // Return bottom to indicate that the operation `op` shall always trap.
558 return wasm::kWasmBottom;
559 }
560 return previous_value;
561}

563wasm::ValueType WasmGCTypeAnalyzer::RefineTypeKnowledgeNotNull(
564 OpIndex object, const Operation& op) {
565 object = ResolveAliases(object);
566 wasm::ValueType previous_value = types_table_.Get(object);
567 if (previous_value.is_non_nullable()) return previous_value;
568
569 wasm::ValueType not_null_type = previous_value.AsNonNull();
570 TRACE("[b%u%s] #%u(%s): Refine type for object #%u(%s) -> %s%s\n",
571 current_block_->index().id(), !IsReachable(*current_block_) ? "u" : "",
572 graph_.Index(op).id(), OpcodeName(op.opcode), object.id(),
573 OpcodeName(graph_.Get(object).opcode), not_null_type.name().c_str(),
574 not_null_type.is_uninhabited() ? " (unreachable!)" : "");
575
576 types_table_.Set(object, not_null_type);
577 if (not_null_type.is_uninhabited()) {
578 // After this instruction all other instructions in the current block are
579 // unreachable.
580 block_is_unreachable_.Add(current_block_->index().id());
581 // Return bottom to indicate that the operation `op` shall always trap.
582 return wasm::kWasmBottom;
583 }
584 return previous_value;
585}
586
587OpIndex WasmGCTypeAnalyzer::ResolveAliases(OpIndex object) const {
588 while (true) {
589 const Operation* op = &graph_.Get(object);
590 switch (op->opcode) {
591 case Opcode::kWasmTypeCast:
592 object = op->Cast<WasmTypeCastOp>().object();
593 break;
594 case Opcode::kAssertNotNull:
595 object = op->Cast<AssertNotNullOp>().object();
596 break;
597 case Opcode::kWasmTypeAnnotation:
598 object = op->Cast<WasmTypeAnnotationOp>().value();
599 break;
600 default:
601 return object;
602 }
603 }
604}
605
606bool WasmGCTypeAnalyzer::IsReachable(const Block& block) const {
607 return !block_is_unreachable_.Contains(block.index().id());
608}
609
610wasm::ValueType WasmGCTypeAnalyzer::GetResolvedType(OpIndex object) const {
611 return types_table_.Get(ResolveAliases(object));
612}
