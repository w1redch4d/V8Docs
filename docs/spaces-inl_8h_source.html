66MutablePageMetadata* OldGenerationMemoryChunkIterator::next() {
67 switch (state_) {
68 case kOldSpace: {
69 PageIterator& iterator = std::get<PageIterator>(iterator_);
70 if (iterator != heap_->old_space()->end()) return *(iterator++);
71 state_ = kCodeSpace;
72 iterator_ = heap_->code_space()->begin();
73 [[fallthrough]];
74 }
75 case kCodeSpace: {
76 PageIterator& iterator = std::get<PageIterator>(iterator_);
77 if (iterator != heap_->code_space()->end()) return *(iterator++);
78 state_ = kLargeObjectSpace;
79 iterator_ = heap_->lo_space()->begin();
80 [[fallthrough]];
81 }
82 case kLargeObjectSpace: {
83 LargePageIterator& iterator = std::get<LargePageIterator>(iterator_);
84 if (iterator != heap_->lo_space()->end()) return *(iterator++);
85 state_ = kCodeLargeObjectSpace;
86 iterator_ = heap_->code_lo_space()->begin();
87 [[fallthrough]];
88 }
89 case kCodeLargeObjectSpace: {
90 LargePageIterator& iterator = std::get<LargePageIterator>(iterator_);
91 if (iterator != heap_->code_lo_space()->end()) return *(iterator++);
92 state_ = kTrustedSpace;
93 iterator_ = heap_->trusted_space()->begin();
94 [[fallthrough]];
95 }
96 case kTrustedSpace: {
97 PageIterator& iterator = std::get<PageIterator>(iterator_);
98 if (iterator != heap_->trusted_space()->end()) return *(iterator++);
99 state_ = kTrustedLargeObjectSpace;
100 iterator_ = heap_->trusted_lo_space()->begin();
101 [[fallthrough]];
102 }
103 case kTrustedLargeObjectSpace: {
104 LargePageIterator& iterator = std::get<LargePageIterator>(iterator_);
105 if (iterator != heap_->trusted_lo_space()->end()) return *(iterator++);
106 state_ = kFinished;
107 [[fallthrough]];
