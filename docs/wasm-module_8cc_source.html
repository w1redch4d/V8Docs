33template <class Value>
34void AdaptiveMap<Value>::FinishInitialization() {
35 uint32_t count = 0;
36 uint32_t max = 0;
37 DCHECK_EQ(mode_, kInitializing);
38 for (const auto& entry : *map_) {
39 count++;
40 max = std::max(max, entry.first);
41 }
42 if (count >= (max + 1) / kLoadFactor) {
43 mode_ = kDense;
44 vector_.resize(max + 1);
45 for (auto& entry : *map_) {
46 vector_[entry.first] = std::move(entry.second);
47 }
48 map_.reset();
49 } else {
50 mode_ = kSparse;
51 }
52}

82// static
83int GetNearestWasmFunction(const WasmModule* module, uint32_t byte_offset) {
84 const std::vector<WasmFunction>& functions = module->functions;
85
86 // Binary search for a function containing the given position.
87 int left = 0; // inclusive
88 int right = static_cast<int>(functions.size()); // exclusive
89 if (right == 0) return -1;
90 while (right - left > 1) {
91 int mid = left + (right - left) / 2;
92 if (functions[mid].code.offset() <= byte_offset) {
93 left = mid;
94 } else {
95 right = mid;
96 }
97 }
98
99 return left;
100}

102// static
103int GetContainingWasmFunction(const WasmModule* module, uint32_t byte_offset) {
104 int func_index = GetNearestWasmFunction(module, byte_offset);
105
106 if (func_index >= 0) {
107 // If the found function does not contain the given position, return -1.
108 const WasmFunction& func = module->functions[func_index];
109 if (byte_offset < func.code.offset() ||
110 byte_offset >= func.code.end_offset()) {
111 return -1;
112 }
113 }
114 return func_index;
115}

136int AsmJsOffsetInformation::GetSourcePosition(int declared_func_index,
137 int byte_offset,
138 bool is_at_number_conversion) {
139 EnsureDecodedOffsets();
140
141 DCHECK_LE(0, declared_func_index);
142 DCHECK_GT(decoded_offsets_->functions.size(), declared_func_index);
143 std::vector<AsmJsOffsetEntry>& function_offsets =
144 decoded_offsets_->functions[declared_func_index].entries;
145
146 auto byte_offset_less = [](const AsmJsOffsetEntry& a,
147 const AsmJsOffsetEntry& b) {
148 return a.byte_offset < b.byte_offset;
149 };
150 SLOW_DCHECK(std::is_sorted(function_offsets.begin(), function_offsets.end(),
151 byte_offset_less));
152 // If there are no positions recorded, map offset 0 (for function entry) to
153 // position 0.
154 if (function_offsets.empty() && byte_offset == 0) return 0;
155 auto it =
156 std::lower_bound(function_offsets.begin(), function_offsets.end(),
157 AsmJsOffsetEntry{byte_offset, 0, 0}, byte_offset_less);
158 DCHECK_NE(function_offsets.end(), it);
159 DCHECK_EQ(byte_offset, it->byte_offset);
160 return is_at_number_conversion ? it->source_position_number_conversion
161 : it->source_position_call;
162}

261Handle<JSObject> GetTypeForFunction(Isolate* isolate, const FunctionSig* sig,
262 bool for_exception) {
263 Factory* factory = isolate->factory();
264
265 // Extract values for the {ValueType[]} arrays.
266 int param_index = 0;
267 int param_count = static_cast<int>(sig->parameter_count());
268 DirectHandle<FixedArray> param_values = factory->NewFixedArray(param_count);
269 for (ValueType type : sig->parameters()) {
270 DirectHandle<String> type_value = ToValueTypeString(isolate, type);
271 param_values->set(param_index++, *type_value);
272 }
273
274 // Create the resulting {FunctionType} object.
275 DirectHandle<JSFunction> object_function = isolate->object_function();
276 Handle<JSObject> object = factory->NewJSObject(object_function);
277 DirectHandle<JSArray> params = factory->NewJSArrayWithElements(param_values);
278 DirectHandle<String> params_string =
279 factory->InternalizeUtf8String("parameters");
280 DirectHandle<String> results_string =
281 factory->InternalizeUtf8String("results");
282 JSObject::AddProperty(isolate, object, params_string, params, NONE);
283
284 // Now add the result types if needed.
285 if (for_exception) {
286 DCHECK_EQ(sig->returns().size(), 0);
287 } else {
288 int result_index = 0;
289 int result_count = static_cast<int>(sig->return_count());
290 DirectHandle<FixedArray> result_values =
291 factory->NewFixedArray(result_count);
292 for (ValueType type : sig->returns()) {
293 DirectHandle<String> type_value = ToValueTypeString(isolate, type);
294 result_values->set(result_index++, *type_value);
295 }
296 DirectHandle<JSArray> results =
297 factory->NewJSArrayWithElements(result_values);
298 JSObject::AddProperty(isolate, object, results_string, results, NONE);
299 }
300
301 return object;
302}

395Handle<JSArray> GetImports(Isolate* isolate,
396 DirectHandle<WasmModuleObject> module_object) {
397 auto enabled_features = i::wasm::WasmEnabledFeatures::FromIsolate(isolate);
398 Factory* factory = isolate->factory();
399
400 DirectHandle<String> module_string = factory->InternalizeUtf8String("module");
401 DirectHandle<String> name_string = factory->name_string();
402 DirectHandle<String> kind_string = factory->InternalizeUtf8String("kind");
403 DirectHandle<String> type_string = factory->InternalizeUtf8String("type");
404
405 DirectHandle<String> function_string = factory->function_string();
406 DirectHandle<String> table_string = factory->InternalizeUtf8String("table");
407 DirectHandle<String> memory_string = factory->InternalizeUtf8String("memory");
408 DirectHandle<String> global_string = factory->global_string();
409 DirectHandle<String> tag_string = factory->InternalizeUtf8String("tag");
410
411 // Create the result array.
412 NativeModule* native_module = module_object->native_module();
413 const WasmModule* module = native_module->module();
414 int num_imports = static_cast<int>(module->import_table.size());
415 Handle<JSArray> array_object = factory->NewJSArray(PACKED_ELEMENTS, 0, 0);
416 DirectHandle<FixedArray> storage = factory->NewFixedArray(num_imports);
417 JSArray::SetContent(array_object, storage);
418
419 DirectHandle<JSFunction> object_function = DirectHandle<JSFunction>(
420 isolate->native_context()->object_function(), isolate);
421
422 // Populate the result array.
423 const WellKnownImportsList& well_known_imports =
424 module->type_feedback.well_known_imports;
425 const std::string& magic_string_constants =
426 native_module->compile_imports().constants_module();
427 const bool has_magic_string_constants =
428 native_module->compile_imports().contains(
429 CompileTimeImport::kStringConstants);
430
431 int cursor = 0;
432 for (int index = 0; index < num_imports; ++index) {
433 const WasmImport& import = module->import_table[index];
434
435 DirectHandle<JSObject> entry = factory->NewJSObject(object_function);
436
437 DirectHandle<String> import_kind;
438 DirectHandle<JSObject> type_value;
439 switch (import.kind) {
440 case kExternalFunction:
441 if (IsCompileTimeImport(well_known_imports.get(import.index))) {
442 continue;
443 }
444 if (enabled_features.has_type_reflection()) {
445 auto& func = module->functions[import.index];
446 type_value = GetTypeForFunction(isolate, func.sig);
447 }
448 import_kind = function_string;
449 break;
450 case kExternalTable:
451 if (enabled_features.has_type_reflection()) {
452 auto& table = module->tables[import.index];
453 std::optional<uint32_t> maximum_size;
454 if (table.has_maximum_size) maximum_size.emplace(table.maximum_size);
455 type_value = GetTypeForTable(isolate, table.type, table.initial_size,
456 maximum_size, table.address_type);
457 }
458 import_kind = table_string;
459 break;
460 case kExternalMemory:
461 if (enabled_features.has_type_reflection()) {
462 auto& memory = module->memories[import.index];
463 std::optional<uint32_t> maximum_size;
464 if (memory.has_maximum_pages) {
465 maximum_size.emplace(memory.maximum_pages);
466 }
467 type_value =
468 GetTypeForMemory(isolate, memory.initial_pages, maximum_size,
469 memory.is_shared, memory.address_type);
470 }
471 import_kind = memory_string;
472 break;
473 case kExternalGlobal:
474 if (has_magic_string_constants &&
475 import.module_name.length() == magic_string_constants.size() &&
476 std::equal(magic_string_constants.begin(),
477 magic_string_constants.end(),
478 module_object->native_module()->wire_bytes().begin() +
479 import.module_name.offset())) {
480 continue;
481 }
482 if (enabled_features.has_type_reflection()) {
483 auto& global = module->globals[import.index];
484 type_value =
485 GetTypeForGlobal(isolate, global.mutability, global.type);
486 }
487 import_kind = global_string;
488 break;
489 case kExternalTag:
490 import_kind = tag_string;
491 break;
492 }
493 DCHECK(!import_kind.is_null());
494
495 DirectHandle<String> import_module =
496 WasmModuleObject::ExtractUtf8StringFromModuleBytes(
497 isolate, module_object, import.module_name, kInternalize);
498
499 DirectHandle<String> import_name =
500 WasmModuleObject::ExtractUtf8StringFromModuleBytes(
501 isolate, module_object, import.field_name, kInternalize);
502
503 JSObject::AddProperty(isolate, entry, module_string, import_module, NONE);
504 JSObject::AddProperty(isolate, entry, name_string, import_name, NONE);
505 JSObject::AddProperty(isolate, entry, kind_string, import_kind, NONE);
506 if (!type_value.is_null()) {
507 JSObject::AddProperty(isolate, entry, type_string, type_value, NONE);
508 }
509
510 storage->set(cursor++, *entry);
511 }
512
513 array_object->set_length(Smi::FromInt(cursor));
514 return array_object;
515}

517Handle<JSArray> GetExports(Isolate* isolate,
518 DirectHandle<WasmModuleObject> module_object) {
519 auto enabled_features = i::wasm::WasmEnabledFeatures::FromIsolate(isolate);
520 Factory* factory = isolate->factory();
521
522 DirectHandle<String> name_string = factory->name_string();

548 DirectHandle<JSObject> type_value;
549 switch (exp.kind) {

560 std::optional<uint32_t> maximum_size;
561 if (table.has_maximum_size) maximum_size.emplace(table.maximum_size);
562 type_value = GetTypeForTable(isolate, table.type, table.initial_size,
563 maximum_size, table.address_type);
564 }
565 export_kind = table_string;
566 break;
567 case kExternalMemory:
568 if (enabled_features.has_type_reflection()) {
569 auto& memory = module->memories[exp.index];
570 std::optional<uint32_t> maximum_size;
571 if (memory.has_maximum_pages) {
572 maximum_size.emplace(memory.maximum_pages);
573 }
574 type_value =
575 GetTypeForMemory(isolate, memory.initial_pages, maximum_size,
576 memory.is_shared, memory.address_type);
577 }
578 export_kind = memory_string;
579 break;
580 case kExternalGlobal:
581 if (enabled_features.has_type_reflection()) {
582 auto& global = module->globals[exp.index];
583 type_value =
584 GetTypeForGlobal(isolate, global.mutability, global.type);

587 break;
588 case kExternalTag:

590 break;

593 }

595 DirectHandle<JSObject> entry = factory->NewJSObject(object_function);
596
597 DirectHandle<String> export_name =
598 WasmModuleObject::ExtractUtf8StringFromModuleBytes(
599 isolate, module_object, exp.name, kNoInternalize);
600
601 JSObject::AddProperty(isolate, entry, name_string, export_name, NONE);
602 JSObject::AddProperty(isolate, entry, kind_string, export_kind, NONE);
603 if (!type_value.is_null()) {
604 JSObject::AddProperty(isolate, entry, type_string, type_value, NONE);
605 }
606
607 storage->set(index, *entry);
608 }
609
610 return array_object;
611}

613Handle<JSArray> GetCustomSections(Isolate* isolate,
614 DirectHandle<WasmModuleObject> module_object,
615 DirectHandle<String> name,
616 ErrorThrower* thrower) {
617 Factory* factory = isolate->factory();
618
619 base::Vector<const uint8_t> wire_bytes =
620 module_object->native_module()->wire_bytes();
621 std::vector<CustomSectionOffset> custom_sections =
622 DecodeCustomSections(wire_bytes);
623
624 std::vector<Handle<Object>> matching_sections;
625
626 // Gather matching sections.
627 for (auto& section : custom_sections) {
628 DirectHandle<String> section_name =
629 WasmModuleObject::ExtractUtf8StringFromModuleBytes(
630 isolate, module_object, section.name, kNoInternalize);
631
632 if (!name->Equals(*section_name)) continue;
633
634 // Make a copy of the payload data in the section.
635 size_t size = section.payload.length();
636 MaybeHandle<JSArrayBuffer> result =
637 isolate->factory()->NewJSArrayBufferAndBackingStore(
638 size, InitializedFlag::kUninitialized);
639 Handle<JSArrayBuffer> array_buffer;
640 if (!result.ToHandle(&array_buffer)) {
641 thrower->RangeError("out of memory allocating custom section data");
642 return Handle<JSArray>();
643 }
644 memcpy(array_buffer->backing_store(),
645 wire_bytes.begin() + section.payload.offset(),
646 section.payload.length());
647
648 matching_sections.push_back(array_buffer);
649 }
650
651 int num_custom_sections = static_cast<int>(matching_sections.size());
652 Handle<JSArray> array_object = factory->NewJSArray(PACKED_ELEMENTS, 0, 0);
653 DirectHandle<FixedArray> storage =
654 factory->NewFixedArray(num_custom_sections);
655 JSArray::SetContent(array_object, storage);
656 array_object->set_length(Smi::FromInt(num_custom_sections));
657
658 for (int i = 0; i < num_custom_sections; i++) {
659 storage->set(i, *matching_sections[i]);
660 }
661
662 return array_object;
663}

665// Get the source position from a given function index and wire bytes offset
666// (relative to the function entry), for either asm.js or pure Wasm modules.
667int GetSourcePosition(const WasmModule* module, uint32_t func_index,
668 uint32_t byte_offset, bool is_at_number_conversion) {
669 DCHECK_EQ(is_asmjs_module(module),
670 module->asm_js_offset_information != nullptr);

672 // For non-asm.js modules, we just add the function's start offset

678 return module->asm_js_offset_information->GetSourcePosition(
679 declared_function_index(module, func_index), byte_offset,
680 is_at_number_conversion);
681}

784size_t PrintSignature(base::Vector<char> buffer, const CanonicalSig* sig,
785 char delimiter) {
786 if (buffer.empty()) return 0;
787 size_t old_size = buffer.size();
788 auto append_char = [&buffer](char c) {
789 if (buffer.size() == 1) return; // Keep last character for '\0'.
790 buffer[0] = c;
791 buffer += 1;
792 };
793 for (CanonicalValueType t : sig->parameters()) {
794 append_char(t.short_name());
795 }
796 append_char(delimiter);
797 for (CanonicalValueType t : sig->returns()) {
798 append_char(t.short_name());
799 }
800 buffer[0] = '\0';
801 return old_size - buffer.size();
802}
