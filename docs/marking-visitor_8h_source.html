28// The base class for all marking visitors (main and concurrent marking) but
29// also for e.g. the reference summarizer. It implements marking logic with
30// support for bytecode flushing, embedder tracing and weak references.
31//
32// Derived classes are expected to provide the following methods:
33// - CanUpdateValuesInHeap
34// - AddStrongReferenceForReferenceSummarizer
35// - AddWeakReferenceForReferenceSummarizer
36// - marking_state
37// - MarkPointerTableEntry
38// - RecordSlot
39// - RecordRelocSlot
40//
41// These methods capture the difference between the different visitor
42// implementations. For example, the concurrent visitor has to use the locking
43// for string types that can be transitioned to other types on the main thread
44// concurrently. On the other hand, the reference summarizer is not supposed to
45// write into heap objects.
46template <typename ConcreteVisitor>
47class MarkingVisitorBase : public ConcurrentHeapVisitor<ConcreteVisitor> {
48 public:
49 using Base = ConcurrentHeapVisitor<ConcreteVisitor>;
