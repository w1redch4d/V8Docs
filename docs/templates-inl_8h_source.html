174void FunctionTemplateInfo::SetInstanceType(int api_instance_type) {
175 // Translate |api_instance_type| value from range
176 // [Internals::kFirstEmbedderJSApiObjectType,
177 // Internals::kLastEmbedderJSApiObjectType] to range
178 // [Internals::kFirstJSApiObjectType, Internals::kLastJSApiObjectType].
179 DCHECK_LE(Internals::kFirstEmbedderJSApiObjectType, api_instance_type);
180 DCHECK_LE(api_instance_type, Internals::kLastEmbedderJSApiObjectType);
181 // kNoJSApiObjectType must correspond to JS_API_OBJECT_TYPE.
182 static_assert(kNoJSApiObjectType == 0);
183 static_assert(JS_API_OBJECT_TYPE == Internals::kFirstJSApiObjectType);
184 set_instance_type(static_cast<InstanceType>(
185 api_instance_type + Internals::kFirstJSApiObjectType));
186}

240 Tagged<Object> maybe_ctor = constructor();
241 if (IsUndefined(maybe_ctor, isolate)) return ObjectTemplateInfo();
242 Tagged<FunctionTemplateInfo> constructor =
243 Cast<FunctionTemplateInfo>(maybe_ctor);
244 while (true) {

246 if (constructor.is_null()) return ObjectTemplateInfo();
247 Tagged<Object> maybe_obj = constructor->GetInstanceTemplate();
248 if (!IsUndefined(maybe_obj, isolate)) {

300// static
301template <typename ReturnType>
302MaybeHandle<ReturnType> TemplateInfo::ProbeInstantiationsCache(
303 Isolate* isolate, DirectHandle<NativeContext> native_context,
304 int serial_number, CachingMode caching_mode) {
305 DCHECK_NE(serial_number, TemplateInfo::kDoNotCache);
306 if (serial_number == TemplateInfo::kUncached) {
307 return {};
308 }
309
310 if (serial_number < TemplateInfo::kFastTemplateInstantiationsCacheSize) {
311 Tagged<FixedArray> fast_cache =
312 native_context->fast_template_instantiations_cache();
313 Handle<Object> object{fast_cache->get(serial_number), isolate};
314 if (IsTheHole(*object, isolate)) {
315 return {};
316 }
317 return Cast<ReturnType>(object);
318 }
319 if (caching_mode == CachingMode::kUnlimited ||
320 (serial_number < TemplateInfo::kSlowTemplateInstantiationsCacheSize)) {
321 Tagged<SimpleNumberDictionary> slow_cache =
322 native_context->slow_template_instantiations_cache();
323 InternalIndex entry = slow_cache->FindEntry(isolate, serial_number);
324 if (entry.is_found()) {
325 return handle(Cast<ReturnType>(slow_cache->ValueAt(entry)), isolate);
326 }
327 }
328 return {};
329}

331// static
332template <typename InstantiationType, typename TemplateInfoType>
333void TemplateInfo::CacheTemplateInstantiation(
334 Isolate* isolate, DirectHandle<NativeContext> native_context,
335 DirectHandle<TemplateInfoType> data, CachingMode caching_mode,
336 Handle<InstantiationType> object) {
337 DCHECK_NE(TemplateInfo::kDoNotCache, data->serial_number());
338
339 int serial_number = data->serial_number();
340 if (serial_number == TemplateInfo::kUncached) {
341 serial_number = isolate->heap()->GetNextTemplateSerialNumber();
342 }
343
344 if (serial_number < TemplateInfo::kFastTemplateInstantiationsCacheSize) {
345 Handle<FixedArray> fast_cache =
346 handle(native_context->fast_template_instantiations_cache(), isolate);
347 DirectHandle<FixedArray> new_cache =
348 FixedArray::SetAndGrow(isolate, fast_cache, serial_number, object);
349 if (*new_cache != *fast_cache) {
350 native_context->set_fast_template_instantiations_cache(*new_cache);
351 }
352 data->set_serial_number(serial_number);
353 } else if (caching_mode == CachingMode::kUnlimited ||
354 (serial_number <
355 TemplateInfo::kSlowTemplateInstantiationsCacheSize)) {
356 Handle<SimpleNumberDictionary> cache =
357 handle(native_context->slow_template_instantiations_cache(), isolate);
358 auto new_cache =
359 SimpleNumberDictionary::Set(isolate, cache, serial_number, object);
360 if (*new_cache != *cache) {
361 native_context->set_slow_template_instantiations_cache(*new_cache);
362 }
363 data->set_serial_number(serial_number);
364 } else {
365 // we've overflowed the cache limit, no more caching
366 data->set_serial_number(TemplateInfo::kDoNotCache);
367 }
368}

370// static
371template <typename TemplateInfoType>
372void TemplateInfo::UncacheTemplateInstantiation(
373 Isolate* isolate, DirectHandle<NativeContext> native_context,
374 DirectHandle<TemplateInfoType> data, CachingMode caching_mode) {
375 int serial_number = data->serial_number();
376 if (serial_number < 0) return;
377
378 if (serial_number < TemplateInfo::kFastTemplateInstantiationsCacheSize) {
379 Tagged<FixedArray> fast_cache =
380 native_context->fast_template_instantiations_cache();
381 DCHECK(!IsUndefined(fast_cache->get(serial_number), isolate));
382 fast_cache->set(serial_number, ReadOnlyRoots{isolate}.the_hole_value(),
383 SKIP_WRITE_BARRIER);
384 data->set_serial_number(TemplateInfo::kUncached);
385 } else if (caching_mode == CachingMode::kUnlimited ||
386 (serial_number <
387 TemplateInfo::kSlowTemplateInstantiationsCacheSize)) {
388 Handle<SimpleNumberDictionary> cache =
389 handle(native_context->slow_template_instantiations_cache(), isolate);
390 InternalIndex entry = cache->FindEntry(isolate, serial_number);
391 DCHECK(entry.is_found());
392 cache = SimpleNumberDictionary::DeleteEntry(isolate, cache, entry);
393 native_context->set_slow_template_instantiations_cache(*cache);
394 data->set_serial_number(TemplateInfo::kUncached);
395 }
396}
