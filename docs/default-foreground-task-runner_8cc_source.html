29void DefaultForegroundTaskRunner::Terminate() {
30 // Drain the task queues.
31 // We make sure to delete tasks outside the TaskRunner lock, to avoid
32 // potential deadlocks.
33 std::deque<TaskQueueEntry> obsolete_tasks;
34 std::priority_queue<DelayedEntry, std::vector<DelayedEntry>,
35 DelayedEntryCompare>
36 obsolete_delayed_tasks;
37 std::queue<std::unique_ptr<IdleTask>> obsolete_idle_tasks;
38 {
39 base::MutexGuard guard(&mutex_);
40 terminated_ = true;
41 task_queue_.swap(obsolete_tasks);
42 delayed_task_queue_.swap(obsolete_delayed_tasks);
43 idle_task_queue_.swap(obsolete_idle_tasks);
44 }
45 while (!obsolete_tasks.empty()) obsolete_tasks.pop_front();
46 while (!obsolete_delayed_tasks.empty()) obsolete_delayed_tasks.pop();
47 while (!obsolete_idle_tasks.empty()) obsolete_idle_tasks.pop();
48}

52 mutex_.AssertHeld();
53 if (terminated_) return task;

116bool DefaultForegroundTaskRunner::HasPoppableTaskInQueue() const {
117 if (nesting_depth_ == 0) return !task_queue_.empty();
118 for (auto it = task_queue_.cbegin(); it != task_queue_.cend(); it++) {
119 if (it->first == kNestable) return true;
120 }
121 return false;

124std::vector<std::unique_ptr<Task>>
125DefaultForegroundTaskRunner::MoveExpiredDelayedTasksLocked() {
126 Nestability nestability;
127 std::vector<std::unique_ptr<Task>> expired_tasks_to_delete;
128 while (std::unique_ptr<Task> task =
129 PopTaskFromDelayedQueueLocked(&nestability)) {
130 auto to_delete = PostTaskLocked(std::move(task), nestability);
131 if (to_delete) expired_tasks_to_delete.emplace_back(std::move(to_delete));
132 }
133 return expired_tasks_to_delete;
134}

136std::unique_ptr<Task> DefaultForegroundTaskRunner::PopTaskFromQueue(
137 MessageLoopBehavior wait_for_work) {
138 std::vector<std::unique_ptr<Task>> tasks_to_delete;
139 base::MutexGuard guard(&mutex_);
140 tasks_to_delete = MoveExpiredDelayedTasksLocked();
141
142 while (!HasPoppableTaskInQueue()) {
143 if (wait_for_work == MessageLoopBehavior::kDoNotWait) return {};
144 WaitForTaskLocked();
145 auto new_tasks_to_delete = MoveExpiredDelayedTasksLocked();
146 tasks_to_delete.insert(tasks_to_delete.end(),
147 std::make_move_iterator(new_tasks_to_delete.begin()),
148 std::make_move_iterator(new_tasks_to_delete.end()));
149 }
150
151 auto it = task_queue_.begin();
152 for (; it != task_queue_.end(); it++) {
153 // When the task queue is nested (i.e. popping a task from the queue from
154 // within a task), only nestable tasks may run. Otherwise, any task may run.
155 if (nesting_depth_ == 0 || it->first == kNestable) break;
156 }
157 DCHECK(it != task_queue_.end());
158 std::unique_ptr<Task> task = std::move(it->second);
159 task_queue_.erase(it);
160
161 return task;
162}

164std::unique_ptr<Task>
165DefaultForegroundTaskRunner::PopTaskFromDelayedQueueLocked(
166 Nestability* nestability) {
167 mutex_.AssertHeld();
168 if (delayed_task_queue_.empty()) return {};
169
170 double now = MonotonicallyIncreasingTime();
171 const DelayedEntry& entry = delayed_task_queue_.top();
172 if (entry.timeout_time > now) return {};
173 // The const_cast here is necessary because there does not exist a clean way
174 // to get a unique_ptr out of the priority queue. We provide the priority
175 // queue with a custom comparison operator to make sure that the priority
176 // queue does not access the unique_ptr. Therefore it should be safe to reset
177 // the unique_ptr in the priority queue here. Note that the DelayedEntry is
178 // removed from the priority_queue immediately afterwards.
179 std::unique_ptr<Task> task = std::move(const_cast<DelayedEntry&>(entry).task);
180 *nestability = entry.nestability;
181 delayed_task_queue_.pop();
182 return task;
183}

185std::unique_ptr<IdleTask> DefaultForegroundTaskRunner::PopTaskFromIdleQueue() {
186 base::MutexGuard guard(&mutex_);
187 if (idle_task_queue_.empty()) return {};
188
189 std::unique_ptr<IdleTask> task = std::move(idle_task_queue_.front());
190 idle_task_queue_.pop();
191
192 return task;
193}

195void DefaultForegroundTaskRunner::WaitForTaskLocked() {
196 mutex_.AssertHeld();
197 if (!delayed_task_queue_.empty()) {
198 double now = MonotonicallyIncreasingTime();
199 const DelayedEntry& entry = delayed_task_queue_.top();
200 double time_until_task = entry.timeout_time - now;
201 if (time_until_task > 0) {
202 bool woken_up = event_loop_control_.WaitFor(
203 &mutex_,
204 base::TimeDelta::FromMicroseconds(
205 time_until_task * base::TimeConstants::kMicrosecondsPerSecond));
206 USE(woken_up);
207 }
208 } else {
209 event_loop_control_.Wait(&mutex_);
210 }
211}
