96// We use 7/8th as maximum load factor for non-special cases.
97// For 16-wide groups, that gives an average of two empty slots per group.
98// Similar to Abseil's CapacityToGrowth.
99// static
100constexpr int SwissNameDictionary::MaxUsableCapacity(int capacity) {
101 DCHECK(IsValidCapacity(capacity));
102
103 if (Group::kWidth == 8 && capacity == 4) {
104 // If the group size is 16 we can fully utilize capacity 4: There will be
105 // enough kEmpty entries in the ctrl table.
106 return 3;
107 }
108 return capacity - capacity / 8;
109}
110
111// Returns |at_least_space_for| * 8/7 for non-special cases. Similar to Abseil's
112// GrowthToLowerboundCapacity.
113// static
114int SwissNameDictionary::CapacityFor(int at_least_space_for) {
115 if (at_least_space_for <= 4) {
116 if (at_least_space_for == 0) {
117 return 0;
118 } else if (at_least_space_for < 4) {
119 return 4;
120 } else if (kGroupWidth == 16) {
121 DCHECK_EQ(4, at_least_space_for);
122 return 4;
123 } else if (kGroupWidth == 8) {
124 DCHECK_EQ(4, at_least_space_for);
125 return 8;
126 }
127 }
128
129 int non_normalized = at_least_space_for + at_least_space_for / 7;
130 return base::bits::RoundUpToPowerOfTwo32(non_normalized);
131}

149template <typename IsolateT>
150InternalIndex SwissNameDictionary::FindEntry(IsolateT* isolate,
151 Tagged<Object> key) {
152 Tagged<Name> name = Cast<Name>(key);
153 DCHECK(IsUniqueName(name));
154 uint32_t hash = name->hash();
155
156 // We probe the hash table in groups of |kGroupWidth| buckets. One bucket
157 // corresponds to a 1-byte entry in the control table.
158 // Each group can be uniquely identified by the index of its first bucket,
159 // which must be a value between 0 (inclusive) and Capacity() (exclusive).
160 // Note that logically, groups wrap around after index Capacity() - 1. This
161 // means that probing the group starting at, for example, index Capacity() - 1
162 // means probing CtrlTable()[Capacity() - 1] followed by CtrlTable()[0] to
163 // CtrlTable()[6], assuming a group width of 8. However, in memory, this is
164 // achieved by maintaining an additional |kGroupWidth| bytes after the first
165 // Capacity() entries of the control table. These contain a copy of the first
166 // max(Capacity(), kGroupWidth) entries of the control table. If Capacity() <
167 // |kGroupWidth|, then the remaining |kGroupWidth| - Capacity() control bytes
168 // are left as |kEmpty|.
169 // This means that actually, probing the group starting
170 // at index Capacity() - 1 is achieved by probing CtrlTable()[Capacity() - 1],
171 // followed by CtrlTable()[Capacity()] to CtrlTable()[Capacity() + 7].
172
173 ctrl_t* ctrl = CtrlTable();
174 auto seq = probe(hash, Capacity());
175 // At this point, seq.offset() denotes the index of the first bucket in the
176 // first group to probe. Note that this doesn't have to be divisible by
177 // |kGroupWidth|, but can have any value between 0 (inclusive) and Capacity()
178 // (exclusive).
179 while (true) {
180 Group g{ctrl + seq.offset()};
181 for (int i : g.Match(swiss_table::H2(hash))) {
182 int candidate_entry = seq.offset(i);
183 Tagged<Object> candidate_key = KeyAt(candidate_entry);
184 // This key matching is SwissNameDictionary specific!
185 if (candidate_key == key) return InternalIndex(candidate_entry);
186 }
187 if (g.MatchEmpty()) return InternalIndex::NotFound();
188
189 // The following selects the next group to probe. Note that seq.offset()
190 // always advances by a multiple of |kGroupWidth|, modulo Capacity(). This
191 // is done in a way such that we visit Capacity() / |kGroupWidth|
192 // non-overlapping (!) groups before we would visit the same group (or
193 // bucket) again.
194 seq.next();
195
196 // If the following DCHECK weren't true, we would have probed all Capacity()
197 // different buckets without finding one containing |kEmpty| (which would
198 // haved triggered the g.MatchEmpty() check above). This must not be the
199 // case because the maximum load factor of 7/8 guarantees that there must
200 // always remain empty buckets.
201 //
202 // The only exception from this rule are small tables, where 2 * Capacity()
203 // < |kGroupWidth|, in which case all Capacity() entries can be filled
204 // without leaving empty buckets. The layout of the control
205 // table guarantees that after the first Capacity() entries of the control
206 // table, the control table contains a copy of those first Capacity()
207 // entries, followed by kGroupWidth - 2 * Capacity() entries containing
208 // |kEmpty|. This guarantees that the g.MatchEmpty() check above will
209 // always trigger if the element wasn't found, correctly preventing us from
210 // probing more than one group in this special case.
211 DCHECK_LT(seq.index(), Capacity());
212 }
213}

344 IsolateT* isolate, HandleType<SwissNameDictionary> table) {
345 int capacity = table->Capacity();
346
347 if (table->UsedCapacity() < MaxUsableCapacity(capacity)) {
348 // We have room for at least one more entry, nothing to do.
349 return table;

351

362void SwissNameDictionary::SetCtrl(int entry, ctrl_t h) {
363 int capacity = Capacity();
364 DCHECK_LT(static_cast<unsigned>(entry), static_cast<unsigned>(capacity));
365
366 ctrl_t* ctrl = CtrlTable();
367 ctrl[entry] = h;
368
369 // The ctrl table contains a copy of the first group (i.e., the group starting
370 // at entry 0) after the first |capacity| entries of the ctrl table. This
371 // means that the ctrl table always has size |capacity| + |kGroupWidth|.
372 // However, note that we may have |capacity| < |kGroupWidth|. For example, if
373 // Capacity() == 8 and |kGroupWidth| == 16, then ctrl[0] is copied to ctrl[8],
374 // ctrl[1] to ctrl[9], etc. In this case, ctrl[16] to ctrl[23] remain unused,
375 // which means that their values are always Ctrl::kEmpty.
376 // We achieve the necessary copying without branching here using some bit
377 // magic: We set {copy_entry = entry} in those cases where we don't actually
378 // have to perform a copy (meaning that we just repeat the {ctrl[entry] = h}
379 // from above). If we do need to do some actual copying, we set {copy_entry =
380 // Capacity() + entry}.
381
382 int mask = capacity - 1;
383 int copy_entry =
384 ((entry - Group::kWidth) & mask) + 1 + ((Group::kWidth - 1) & mask);
385 DCHECK_IMPLIES(entry < static_cast<int>(Group::kWidth),
386 copy_entry == capacity + entry);
387 DCHECK_IMPLIES(entry >= static_cast<int>(Group::kWidth), copy_entry == entry);
388 ctrl[copy_entry] = h;
389}

391// static
392inline int SwissNameDictionary::FindFirstEmpty(uint32_t hash) {
393 // See SwissNameDictionary::FindEntry for description of probing algorithm.
394
395 auto seq = probe(hash, Capacity());
396 while (true) {
397 Group g{CtrlTable() + seq.offset()};
398 auto mask = g.MatchEmpty();
399 if (mask) {
400 // Note that picking the lowest bit set here means using the leftmost
401 // empty bucket in the group. Here, "left" means smaller entry/bucket
402 // index.
403 return seq.offset(mask.LowestBitSet());
404 }
405 seq.next();
406 DCHECK_LT(seq.index(), Capacity());
407 }
408}

510// static
511template <typename IsolateT, template <typename> typename HandleType>
512 requires(std::is_convertible_v<HandleType<SwissNameDictionary>,
513 DirectHandle<SwissNameDictionary>>)
514HandleType<SwissNameDictionary> SwissNameDictionary::Add(
515 IsolateT* isolate, HandleType<SwissNameDictionary> original_table,
516 DirectHandle<Name> key, DirectHandle<Object> value, PropertyDetails details,
517 InternalIndex* entry_out) {
518 DCHECK(original_table->FindEntry(isolate, *key).is_not_found());
519
520 HandleType<SwissNameDictionary> table =
521 EnsureGrowable(isolate, original_table);
522 DisallowGarbageCollection no_gc;
523 Tagged<SwissNameDictionary> raw_table = *table;
524 int nof = raw_table->NumberOfElements();
525 int nod = raw_table->NumberOfDeletedElements();
526 int new_enum_index = nof + nod;
527
528 int new_entry = raw_table->AddInternal(*key, *value, details);
529
530 raw_table->SetNumberOfElements(nof + 1);
531 raw_table->SetEntryForEnumerationIndex(new_enum_index, new_entry);
532
533 if (entry_out) {
534 *entry_out = InternalIndex(new_entry);
535 }
536
537 return table;
538}
539
540int SwissNameDictionary::AddInternal(Tagged<Name> key, Tagged<Object> value,
541 PropertyDetails details) {
542 DisallowHeapAllocation no_gc;
543
544 DCHECK(IsUniqueName(key));
545 DCHECK_LE(UsedCapacity(), MaxUsableCapacity(Capacity()));
546
547 uint32_t hash = key->hash();
548
549 // For now we don't re-use deleted buckets (due to enumeration table
550 // complications), which is why we only look for empty buckets here, not
551 // deleted ones.
552 int target = FindFirstEmpty(hash);
553
554 SetCtrl(target, swiss_table::H2(hash));
555 SetKey(target, key);
556 ValueAtPut(target, value);
557 DetailsAtPut(target, details);
558
559 // Note that we do not update the number of elements or the enumeration table
560 // in this function.
561
562 return target;
563}

645SwissNameDictionary::IndexIterable
646SwissNameDictionary::IterateEntriesOrdered() {
647 // If we are supposed to iterate the empty dictionary (which is non-writable),
648 // we have no simple way to get the isolate, which we would need to create a
649 // handle.
650 // TODO(emrich): Consider always using roots.empty_swiss_dictionary()
651 // in the condition once this function gets Isolate as a parameter in order to
652 // avoid empty dict checks.
653 if (Capacity() == 0) {
654 return IndexIterable(DirectHandle<SwissNameDictionary>::null());
655 }
