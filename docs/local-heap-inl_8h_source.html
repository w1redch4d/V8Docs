51template <typename Callback>
52V8_INLINE void LocalHeap::ParkAndExecuteCallback(Callback callback) {
53 // This method is given as a callback to the stack trampoline, when the stack
54 // marker has just been set.
55#if defined(V8_ENABLE_DIRECT_HANDLE) && defined(ENABLE_SLOW_DCHECKS)
56 // Reset the number of direct handles that are below the stack marker.
57 // It will be restored before the method returns.
58 DirectHandleBase::ResetNumberOfHandlesScope scope;
59#endif // V8_ENABLE_DIRECT_HANDLE && ENABLE_SLOW_DCHECKS
60 ParkedScope parked(this);
61 // Provide the parked scope as a witness, if the callback expects it.
62 if constexpr (std::is_invocable_v<Callback, const ParkedScope&>) {
63 callback(parked);
64 } else {
65 callback();
66 }
67}
