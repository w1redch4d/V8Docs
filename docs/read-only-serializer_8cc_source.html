20// Preprocess an object to prepare it for serialization.
21class ObjectPreProcessor final {
22 public:
23 explicit ObjectPreProcessor(Isolate* isolate)
24 : isolate_(isolate), extref_encoder_(isolate) {}
25
26#define PRE_PROCESS_TYPE_LIST(V) \
27 V(AccessorInfo) \
28 V(FunctionTemplateInfo) \
29 V(Code)
30
31 void PreProcessIfNeeded(Tagged<HeapObject> o) {
32 const InstanceType itype = o->map(isolate_)->instance_type();
33#define V(TYPE) \
34 if (InstanceTypeChecker::Is##TYPE(itype)) { \
35 return PreProcess##TYPE(Cast<TYPE>(o)); \
36 }
37 PRE_PROCESS_TYPE_LIST(V)
38#undef V
39 // If we reach here, no preprocessing is needed for this object.
40 }
41#undef PRE_PROCESS_TYPE_LIST
42
43 private:
44 void EncodeExternalPointerSlot(ExternalPointerSlot slot) {
45 Address value = slot.load(isolate_);
46 EncodeExternalPointerSlot(slot, value);
47 }
48
49 void EncodeExternalPointerSlot(ExternalPointerSlot slot, Address value) {
50 // Note it's possible that `value != slot.load(...)`, e.g. for
51 // AccessorInfo::remove_getter_indirection.
52 ExternalReferenceEncoder::Value encoder_value =
53 extref_encoder_.Encode(value);
54 DCHECK_LT(encoder_value.index(),
55 1UL << ro::EncodedExternalReference::kIndexBits);
56 ro::EncodedExternalReference encoded{encoder_value.is_from_api(),
57 encoder_value.index()};
58 // Constructing no_gc here is not the intended use pattern (instead we
59 // should pass it along the entire callchain); but there's little point of
60 // doing that here - all of the code in this file relies on GC being
61 // disabled, and that's guarded at entry points.
62 DisallowGarbageCollection no_gc;
63 slot.ReplaceContentWithIndexForSerialization(no_gc, encoded.ToUint32());
64 }
65 void PreProcessAccessorInfo(Tagged<AccessorInfo> o) {
66 EncodeExternalPointerSlot(
67 o->RawExternalPointerField(AccessorInfo::kMaybeRedirectedGetterOffset,
68 kAccessorInfoGetterTag),
69 o->getter(isolate_)); // Pass the non-redirected value.
70 EncodeExternalPointerSlot(o->RawExternalPointerField(
71 AccessorInfo::kSetterOffset, kAccessorInfoSetterTag));
72 }
73 void PreProcessFunctionTemplateInfo(Tagged<FunctionTemplateInfo> o) {
74 EncodeExternalPointerSlot(
75 o->RawExternalPointerField(
76 FunctionTemplateInfo::kMaybeRedirectedCallbackOffset,
77 kFunctionTemplateInfoCallbackTag),
78 o->callback(isolate_)); // Pass the non-redirected value.
79 }
80 void PreProcessCode(Tagged<Code> o) {
81 o->ClearInstructionStartForSerialization(isolate_);
82 CHECK(!o->has_source_position_table_or_bytecode_offset_table());
83 CHECK(!o->has_deoptimization_data_or_interpreter_data());
84#ifdef V8_ENABLE_LEAPTIERING
85 CHECK_EQ(o->js_dispatch_handle(), kNullJSDispatchHandle);
86#endif
87 }
88
89 Isolate* const isolate_;
90 ExternalReferenceEncoder extref_encoder_;
91};

93struct ReadOnlySegmentForSerialization {
94 ReadOnlySegmentForSerialization(Isolate* isolate,
95 const ReadOnlyPageMetadata* page,
96 Address segment_start, size_t segment_size,
97 ObjectPreProcessor* pre_processor)
98 : page(page),
99 segment_start(segment_start),
100 segment_size(segment_size),
101 segment_offset(segment_start - page->area_start()),
102 contents(new uint8_t[segment_size]),
103 tagged_slots(segment_size / kTaggedSize) {
104 // .. because tagged_slots records a bit for each slot:
105 DCHECK(IsAligned(segment_size, kTaggedSize));
106 // Ensure incoming pointers to this page are representable.
107 CHECK_LT(isolate->read_only_heap()->read_only_space()->IndexOf(page),
108 1UL << ro::EncodedTagged::kPageIndexBits);
109
110 MemCopy(contents.get(), reinterpret_cast<void*>(segment_start),
111 segment_size);
112 PreProcessSegment(pre_processor);
113 if (!V8_STATIC_ROOTS_BOOL) EncodeTaggedSlots(isolate);
114 }
115
116 void PreProcessSegment(ObjectPreProcessor* pre_processor) {
117 // Iterate the RO page and the contents copy in lockstep, preprocessing
118 // objects as we go along.
119 //
120 // See also ObjectSerializer::OutputRawData.
121 DCHECK_GE(segment_start, page->area_start());
122 const Address segment_end = segment_start + segment_size;
123 ReadOnlyPageObjectIterator it(page, segment_start);
124 for (Tagged<HeapObject> o = it.Next(); !o.is_null(); o = it.Next()) {
125 if (o.address() >= segment_end) break;
126 size_t o_offset = o.ptr() - segment_start;
127 Address o_dst = reinterpret_cast<Address>(contents.get()) + o_offset;
128 pre_processor->PreProcessIfNeeded(
129 Cast<HeapObject>(Tagged<Object>(o_dst)));
130 }
131 }
132
133 void EncodeTaggedSlots(Isolate* isolate);
134
135 const ReadOnlyPageMetadata* const page;
136 const Address segment_start;
137 const size_t segment_size;
138 const size_t segment_offset;
139 // The (mutated) off-heap copy of the on-heap segment.
140 std::unique_ptr<uint8_t[]> contents;
141 // The relocation table.
142 ro::BitSet tagged_slots;
143
144 friend class EncodeRelocationsVisitor;
145};

147ro::EncodedTagged Encode(Isolate* isolate, Tagged<HeapObject> o) {
148 Address o_address = o.address();
149 MemoryChunkMetadata* chunk = MemoryChunkMetadata::FromAddress(o_address);
150
151 ReadOnlySpace* ro_space = isolate->read_only_heap()->read_only_space();
152 int index = static_cast<int>(ro_space->IndexOf(chunk));
153 uint32_t offset = static_cast<int>(chunk->Offset(o_address));
154 DCHECK(IsAligned(offset, kTaggedSize));
155
156 return ro::EncodedTagged(index, offset / kTaggedSize);
157}

160// - encodes all tagged slots s.t. valid pointers can be reconstructed during
161// deserialization, and

163class EncodeRelocationsVisitor final : public ObjectVisitor {
164 public:
165 EncodeRelocationsVisitor(Isolate* isolate,
166 ReadOnlySegmentForSerialization* segment)
167 : isolate_(isolate), segment_(segment) {
168 DCHECK(!V8_STATIC_ROOTS_BOOL);
169 }
170
171 void VisitPointers(Tagged<HeapObject> host, ObjectSlot start,
172 ObjectSlot end) override {
173 VisitPointers(host, MaybeObjectSlot(start), MaybeObjectSlot(end));
174 }
175
176 void VisitPointers(Tagged<HeapObject> host, MaybeObjectSlot start,
177 MaybeObjectSlot end) override {
178 for (MaybeObjectSlot slot = start; slot < end; slot++) {
179 ProcessSlot(slot);
180 }
181 }
182
183 void VisitMapPointer(Tagged<HeapObject> host) override {
184 ProcessSlot(host->RawMaybeWeakField(HeapObject::kMapOffset));
185 }
186
187 // Sanity-checks:
188 void VisitInstructionStreamPointer(Tagged<Code> host,
189 InstructionStreamSlot slot) override {
190 // RO space contains only builtin Code objects.
191 DCHECK(!host->has_instruction_stream());
192 }
193 void VisitCodeTarget(Tagged<InstructionStream>, RelocInfo*) override {
194 UNREACHABLE();
195 }
196 void VisitEmbeddedPointer(Tagged<InstructionStream>, RelocInfo*) override {
197 UNREACHABLE();
198 }
199 void VisitExternalReference(Tagged<InstructionStream>, RelocInfo*) override {
200 UNREACHABLE();
201 }
202 void VisitInternalReference(Tagged<InstructionStream>, RelocInfo*) override {
203 UNREACHABLE();
204 }
205 void VisitOffHeapTarget(Tagged<InstructionStream>, RelocInfo*) override {
206 UNREACHABLE();
207 }
208 void VisitExternalPointer(Tagged<HeapObject>,
209 ExternalPointerSlot slot) override {
210 // This slot was encoded in a previous pass, see EncodeExternalPointerSlot.
211#ifdef DEBUG
212 ExternalPointerSlot slot_in_segment{
213 reinterpret_cast<Address>(segment_->contents.get() +
214 SegmentOffsetOf(slot)),
215 slot.exact_tag()};
216 // Constructing no_gc here is not the intended use pattern (instead we
217 // should pass it along the entire callchain); but there's little point of
218 // doing that here - all of the code in this file relies on GC being
219 // disabled, and that's guarded at entry points.
220 DisallowGarbageCollection no_gc;
221 auto encoded = ro::EncodedExternalReference::FromUint32(
222 slot_in_segment.GetContentAsIndexAfterDeserialization(no_gc));
223 if (encoded.is_api_reference) {
224 // Can't validate these since we don't know how many entries
225 // api_external_references contains.
226 } else {
227 CHECK_LT(encoded.index, ExternalReferenceTable::kSize);
228 }
229#endif // DEBUG
230 }
231
232 private:
233 void ProcessSlot(MaybeObjectSlot slot) {
234 Tagged<MaybeObject> o = *slot;
235 if (!o.IsStrongOrWeak()) return; // Smis don't need relocation.
236 DCHECK(o.IsStrong());
237
238 int slot_offset = SegmentOffsetOf(slot);
239 DCHECK(IsAligned(slot_offset, kTaggedSize));
240
241 // Encode:
242 ro::EncodedTagged encoded = Encode(isolate_, o.GetHeapObject());
243 memcpy(segment_->contents.get() + slot_offset, &encoded,
244 ro::EncodedTagged::kSize);
245
246 // Record:
247 segment_->tagged_slots.set(AsSlot(slot_offset));
248 }

250 template <class SlotT>
251 int SegmentOffsetOf(SlotT slot) const {
252 Address addr = slot.address();
253 DCHECK_GE(addr, segment_->segment_start);
254 DCHECK_LT(addr, segment_->segment_start + segment_->segment_size);
255 return static_cast<int>(addr - segment_->segment_start);
256 }
257
258 static constexpr int AsSlot(int byte_offset) {
259 return byte_offset / kTaggedSize;
260 }

263 ReadOnlySegmentForSerialization* const segment_;

281class ReadOnlyHeapImageSerializer {
282 public:
283 struct MemoryRegion {
284 Address start;
285 size_t size;
286 };
287
288 static void Serialize(Isolate* isolate, SnapshotByteSink* sink,
289 const std::vector<MemoryRegion>& unmapped_regions) {
290 ReadOnlyHeapImageSerializer{isolate, sink}.SerializeImpl(unmapped_regions);
291 }
292
293 private:
294 using Bytecode = ro::Bytecode;
295
296 ReadOnlyHeapImageSerializer(Isolate* isolate, SnapshotByteSink* sink)
297 : isolate_(isolate), sink_(sink), pre_processor_(isolate) {}
298
299 void SerializeImpl(const std::vector<MemoryRegion>& unmapped_regions) {
300 DCHECK_EQ(sink_->Position(), 0);
301
302 ReadOnlySpace* ro_space = isolate_->read_only_heap()->read_only_space();
303
304 // Allocate all pages first s.t. the deserializer can easily handle forward
305 // references (e.g.: an object on page i points at an object on page i+1).
306 for (const ReadOnlyPageMetadata* page : ro_space->pages()) {
307 EmitAllocatePage(page, unmapped_regions);
308 }
309
310 // Now write the page contents.
311 for (const ReadOnlyPageMetadata* page : ro_space->pages()) {
312 SerializePage(page, unmapped_regions);
313 }
314
315 EmitReadOnlyRootsTable();
316 sink_->Put(Bytecode::kFinalizeReadOnlySpace, "space end");
317 }
318
319 uint32_t IndexOf(const ReadOnlyPageMetadata* page) {
320 ReadOnlySpace* ro_space = isolate_->read_only_heap()->read_only_space();
321 return static_cast<uint32_t>(ro_space->IndexOf(page));
322 }
323
324 void EmitAllocatePage(const ReadOnlyPageMetadata* page,
325 const std::vector<MemoryRegion>& unmapped_regions) {
326 if (V8_STATIC_ROOTS_BOOL) {
327 sink_->Put(Bytecode::kAllocatePageAt, "fixed page begin");
328 } else {
329 sink_->Put(Bytecode::kAllocatePage, "page begin");
330 }
331 sink_->PutUint30(IndexOf(page), "page index");
332 sink_->PutUint30(
333 static_cast<uint32_t>(page->HighWaterMark() - page->area_start()),
334 "area size in bytes");
335 if (V8_STATIC_ROOTS_BOOL) {
336 auto page_addr = page->ChunkAddress();
337 sink_->PutUint32(V8HeapCompressionScheme::CompressAny(page_addr),
338 "page start offset");
339 }
340 }
341
342 void SerializePage(const ReadOnlyPageMetadata* page,
343 const std::vector<MemoryRegion>& unmapped_regions) {
344 Address pos = page->area_start();
345
346 // If this page contains unmapped regions split it into multiple segments.
347 for (auto r = unmapped_regions.begin(); r != unmapped_regions.end(); ++r) {
348 // Regions must be sorted and non-overlapping.
349 if (r + 1 != unmapped_regions.end()) {
350 CHECK(r->start < (r + 1)->start);
351 CHECK(r->start + r->size < (r + 1)->start);
352 }
353 if (base::IsInRange(r->start, pos, page->HighWaterMark())) {
354 size_t segment_size = r->start - pos;
355 ReadOnlySegmentForSerialization segment(isolate_, page, pos,
356 segment_size, &pre_processor_);
357 EmitSegment(&segment);
358 pos += segment_size + r->size;
359 }
360 }
361
362 // Pages are shrunk, but memory at the end of the area is still
363 // uninitialized and we do not want to include it in the snapshot.
364 size_t segment_size = page->HighWaterMark() - pos;
365 ReadOnlySegmentForSerialization segment(isolate_, page, pos, segment_size,
366 &pre_processor_);
367 EmitSegment(&segment);
368 }
369
370 void EmitSegment(const ReadOnlySegmentForSerialization* segment) {
371 sink_->Put(Bytecode::kSegment, "segment begin");
372 sink_->PutUint30(IndexOf(segment->page), "page index");
373 sink_->PutUint30(static_cast<uint32_t>(segment->segment_offset),
374 "segment start offset");
375 sink_->PutUint30(static_cast<uint32_t>(segment->segment_size),
376 "segment byte size");
377 sink_->PutRaw(segment->contents.get(),
378 static_cast<int>(segment->segment_size), "page");
379 if (!V8_STATIC_ROOTS_BOOL) {
380 sink_->Put(Bytecode::kRelocateSegment, "relocate segment");
381 sink_->PutRaw(segment->tagged_slots.data(),
382 static_cast<int>(segment->tagged_slots.size_in_bytes()),
383 "tagged_slots");
384 }
385 }
386
387 void EmitReadOnlyRootsTable() {
388 sink_->Put(Bytecode::kReadOnlyRootsTable, "read only roots table");
389 if (!V8_STATIC_ROOTS_BOOL) {
390 ReadOnlyRoots roots(isolate_);
391 for (size_t i = 0; i < ReadOnlyRoots::kEntriesCount; i++) {
392 RootIndex rudi = static_cast<RootIndex>(i);
393 Tagged<HeapObject> rudolf = Cast<HeapObject>(roots.object_at(rudi));
394 ro::EncodedTagged encoded = Encode(isolate_, rudolf);
395 sink_->PutUint32(encoded.ToUint32(), "read only roots entry");
396 }
397 }
398 }

400 Isolate* const isolate_;
401 SnapshotByteSink* const sink_;
402 ObjectPreProcessor pre_processor_;
403};

405std::vector<ReadOnlyHeapImageSerializer::MemoryRegion> GetUnmappedRegions(
406 Isolate* isolate) {
407#ifdef V8_STATIC_ROOTS
408 // WasmNull's payload is aligned to the OS page and consists of
409 // WasmNull::kPayloadSize bytes of unmapped memory. To avoid inflating the
410 // snapshot size and accessing uninitialized and/or unmapped memory, the
411 // serializer skips the padding bytes and the payload.
412 ReadOnlyRoots ro_roots(isolate);
413 Tagged<WasmNull> wasm_null = ro_roots.wasm_null();
414 Tagged<HeapObject> wasm_null_padding = ro_roots.wasm_null_padding();
415 CHECK(IsFreeSpace(wasm_null_padding));
416 Address wasm_null_padding_start =
417 wasm_null_padding.address() + FreeSpace::kHeaderSize;
418 std::vector<ReadOnlyHeapImageSerializer::MemoryRegion> unmapped;
419 if (wasm_null.address() > wasm_null_padding_start) {
420 unmapped.push_back({wasm_null_padding_start,
421 wasm_null.address() - wasm_null_padding_start});
422 }
423 unmapped.push_back({wasm_null->payload(), WasmNull::kPayloadSize});
424 return unmapped;
425#else
426 return {};
427#endif // V8_STATIC_ROOTS
428}
