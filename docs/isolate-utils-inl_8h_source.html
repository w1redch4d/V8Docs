16V8_INLINE Heap* GetHeapFromWritableObject(Tagged<HeapObject> object) {
17 MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
18 // Do not use this method on shared objects. This method would always return
19 // the shared space isolate for shared objects. However, on worker isolates
20 // this might be different from the current isolate. In such cases either
21 // require the current isolate as an additional argument from the caller or
22 // use Isolate::Current(). From there you can access the shared space isolate
23 // with `isolate->shared_space_isolate()` if needed.
24 DCHECK(!chunk->InWritableSharedSpace());
25 return chunk->GetHeap();
26}

41V8_INLINE bool GetIsolateFromHeapObject(Tagged<HeapObject> object,
42 Isolate** isolate) {
43 MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
44 if (chunk->InReadOnlySpace()) {
45 *isolate = nullptr;
46 return false;
47 }
48 *isolate = Isolate::FromHeap(chunk->GetHeap());
49 return true;
50}

52// Use this function instead of Internals::GetIsolateForSandbox for internal
53// code, as this function is fully inlinable.
54V8_INLINE static IsolateForSandbox GetIsolateForSandbox(
55 Tagged<HeapObject> object) {
56#ifdef V8_ENABLE_SANDBOX
57 // This method can be used on shared objects as opposed to
58 // GetHeapFromWritableObject because it only returns IsolateForSandbox instead
59 // of the Isolate. This is because shared objects will go to shared external
60 // pointer table which is the same for main and all worker isolates.
61 MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
62 return Isolate::FromHeap(chunk->GetHeap());
63#else
64 // Not used in non-sandbox mode.
65 return {};
66#endif
67}
