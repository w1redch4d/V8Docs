80bool AstRawString::AsArrayIndex(uint32_t* index) const {
81 // The StringHasher will set up the hash. Bail out early if we know it
82 // can't be convertible to an array index.
83 if (!IsIntegerIndex()) return false;
84 if (length() <= Name::kMaxCachedArrayIndexLength) {
85 *index = Name::ArrayIndexValueBits::decode(raw_hash_field_);
86 return true;
87 }
88 // Might be an index, but too big to cache it. Do the slow conversion. This
89 // might fail if the string is outside uint32_t (but within "safe integer")
90 // range.
91 OneByteStringStream stream(literal_bytes_);
92 return StringToIndex(&stream, index);
93}

99bool AstRawString::IsOneByteEqualTo(const char* data) const {
100 if (!is_one_byte()) return false;
101
102 size_t length = literal_bytes_.size();
103 if (length != strlen(data)) return false;
104
105 return 0 == strncmp(reinterpret_cast<const char*>(literal_bytes_.begin()),
106 data, length);
107}

115bool AstRawString::Equal(const AstRawString* lhs, const AstRawString* rhs) {
116 DCHECK_EQ(lhs->Hash(), rhs->Hash());
117
118 if (lhs->length() != rhs->length()) return false;
119 if (lhs->length() == 0) return true;
120 const unsigned char* l = lhs->raw_data();
121 const unsigned char* r = rhs->raw_data();
122 size_t length = rhs->length();
123 if (lhs->is_one_byte()) {
124 if (rhs->is_one_byte()) {
125 return CompareCharsEqualUnsigned(reinterpret_cast<const uint8_t*>(l),
126 reinterpret_cast<const uint8_t*>(r),
127 length);
128 } else {
129 return CompareCharsEqualUnsigned(reinterpret_cast<const uint8_t*>(l),
130 reinterpret_cast<const uint16_t*>(r),
131 length);
132 }
133 } else {
134 if (rhs->is_one_byte()) {
135 return CompareCharsEqualUnsigned(reinterpret_cast<const uint16_t*>(l),
136 reinterpret_cast<const uint8_t*>(r),
137 length);
138 } else {
139 return CompareCharsEqualUnsigned(reinterpret_cast<const uint16_t*>(l),
140 reinterpret_cast<const uint16_t*>(r),
141 length);
142 }
143 }

146int AstRawString::Compare(const AstRawString* lhs, const AstRawString* rhs) {
147 // Fast path for equal pointers.
148 if (lhs == rhs) return 0;
149
150 const unsigned char* lhs_data = lhs->raw_data();
151 const unsigned char* rhs_data = rhs->raw_data();
152 size_t length = std::min(lhs->length(), rhs->length());
153
154 // Code point order by contents.
155 if (lhs->is_one_byte()) {
156 if (rhs->is_one_byte()) {
157 if (int result = CompareCharsUnsigned(
158 reinterpret_cast<const uint8_t*>(lhs_data),
159 reinterpret_cast<const uint8_t*>(rhs_data), length))
160 return result;
161 } else {
162 if (int result = CompareCharsUnsigned(
163 reinterpret_cast<const uint8_t*>(lhs_data),
164 reinterpret_cast<const uint16_t*>(rhs_data), length))
165 return result;
166 }
167 } else {
168 if (rhs->is_one_byte()) {
169 if (int result = CompareCharsUnsigned(
170 reinterpret_cast<const uint16_t*>(lhs_data),
171 reinterpret_cast<const uint8_t*>(rhs_data), length))
172 return result;
173 } else {
174 if (int result = CompareCharsUnsigned(
175 reinterpret_cast<const uint16_t*>(lhs_data),
176 reinterpret_cast<const uint16_t*>(rhs_data), length))
177 return result;
178 }
179 }
180
181 return lhs->byte_length() - rhs->byte_length();
182}

188template <typename IsolateT>
189Handle<String> AstConsString::Allocate(IsolateT* isolate) const {
190 DCHECK(string_.is_null());
191
192 if (IsEmpty()) {
193 return isolate->factory()->empty_string();
194 }
195 // AstRawStrings are internalized before AstConsStrings are allocated, so
196 // AstRawString::string() will just work.
197 Handle<String> tmp = segment_.string->string();
198 for (AstConsString::Segment* current = segment_.next; current != nullptr;
199 current = current->next) {
200 tmp = isolate->factory()
201 ->NewConsString(current->string->string(), tmp,
202 AllocationType::kOld)
203 .ToHandleChecked();
204 }
205 return tmp;
206}

213template <typename IsolateT>
214Handle<String> AstConsString::AllocateFlat(IsolateT* isolate) const {
215 if (IsEmpty()) {
216 return isolate->factory()->empty_string();
217 }
218 if (!segment_.next) {
219 return segment_.string->string();
220 }
221
222 int result_length = 0;
223 bool is_one_byte = true;
224 for (const AstConsString::Segment* current = &segment_; current != nullptr;
225 current = current->next) {
226 result_length += current->string->length();
227 is_one_byte = is_one_byte && current->string->is_one_byte();
228 }
229
230 if (is_one_byte) {
231 Handle<SeqOneByteString> result =
232 isolate->factory()
233 ->NewRawOneByteString(result_length, AllocationType::kOld)
234 .ToHandleChecked();
235 DisallowGarbageCollection no_gc;
236 uint8_t* dest =
237 result->GetChars(no_gc, SharedStringAccessGuardIfNeeded::NotNeeded()) +
238 result_length;
239 for (const AstConsString::Segment* current = &segment_; current != nullptr;
240 current = current->next) {
241 int length = current->string->length();
242 dest -= length;
243 CopyChars(dest, current->string->raw_data(), length);
244 }
245 DCHECK_EQ(dest, result->GetChars(
246 no_gc, SharedStringAccessGuardIfNeeded::NotNeeded()));
247 return result;
248 }
249
250 Handle<SeqTwoByteString> result =
251 isolate->factory()
252 ->NewRawTwoByteString(result_length, AllocationType::kOld)
253 .ToHandleChecked();
254 DisallowGarbageCollection no_gc;
255 uint16_t* dest =
256 result->GetChars(no_gc, SharedStringAccessGuardIfNeeded::NotNeeded()) +
257 result_length;
258 for (const AstConsString::Segment* current = &segment_; current != nullptr;
259 current = current->next) {
260 int length = current->string->length();
261 dest -= length;
262 if (current->string->is_one_byte()) {
263 CopyChars(dest, current->string->raw_data(), length);
264 } else {
265 CopyChars(dest,
266 reinterpret_cast<const uint16_t*>(current->string->raw_data()),
267 length);
268 }
269 }
270 DCHECK_EQ(dest, result->GetChars(
271 no_gc, SharedStringAccessGuardIfNeeded::NotNeeded()));
272 return result;
273}

317const AstRawString* AstValueFactory::GetOneByteStringInternal(
318 base::Vector<const uint8_t> literal) {
319 if (literal.length() == 1 && literal[0] < kMaxOneCharStringValue) {
320 int key = literal[0];
321 if (V8_UNLIKELY(one_character_strings_[key] == nullptr)) {
322 uint32_t raw_hash_field = StringHasher::HashSequentialString<uint8_t>(
323 literal.begin(), literal.length(), hash_seed_);
324 one_character_strings_[key] = GetString(raw_hash_field, true, literal);
325 }
326 return one_character_strings_[key];
327 }
328 uint32_t raw_hash_field = StringHasher::HashSequentialString<uint8_t>(
329 literal.begin(), literal.length(), hash_seed_);
330 return GetString(raw_hash_field, true, literal);
331}

341const AstRawString* AstValueFactory::GetString(
342 Tagged<String> literal,
343 const SharedStringAccessGuardIfNeeded& access_guard) {
344 const AstRawString* result = nullptr;
345 DisallowGarbageCollection no_gc;
346 String::FlatContent content = literal->GetFlatContent(no_gc, access_guard);
347 if (content.IsOneByte()) {
348 result = GetOneByteStringInternal(content.ToOneByteVector());
349 } else {
350 DCHECK(content.IsTwoByte());
351 result = GetTwoByteStringInternal(content.ToUC16Vector());
352 }
353 return result;
354}

388const AstRawString* AstValueFactory::GetString(
389 uint32_t raw_hash_field, bool is_one_byte,
390 base::Vector<const uint8_t> literal_bytes) {
391 // literal_bytes here points to whatever the user passed, and this is OK
392 // because we use vector_compare (which checks the contents) to compare
393 // against the AstRawStrings which are in the string_table_. We should not
394 // return this AstRawString.
395 AstRawString key(is_one_byte, literal_bytes, raw_hash_field);
396 AstRawStringMap::Entry* entry = string_table_.LookupOrInsert(
397 &key, key.Hash(),
398 [&]() {
399 // Copy literal contents for later comparison.
400 int length = literal_bytes.length();
401 uint8_t* new_literal_bytes =
402 ast_raw_string_zone()->AllocateArray<uint8_t>(length);
403 memcpy(new_literal_bytes, literal_bytes.begin(), length);
404 AstRawString* new_string = ast_raw_string_zone()->New<AstRawString>(
405 is_one_byte, base::Vector<const uint8_t>(new_literal_bytes, length),
406 raw_hash_field);
407 CHECK_NOT_NULL(new_string);
408 AddString(new_string);
409 return new_string;
410 },
411 [&]() { return base::NoHashMapValue(); });
412 return entry->key;
413}
