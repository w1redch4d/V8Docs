24Reduction RedundancyElimination::Reduce(Node* node) {
25 if (node_checks_.Get(node)) return NoChange();
26 switch (node->opcode()) {
27 case IrOpcode::kCheckBigInt:
28 case IrOpcode::kCheckedBigIntToBigInt64:
29 case IrOpcode::kCheckBounds:
30 case IrOpcode::kCheckClosure:
31 case IrOpcode::kCheckEqualsInternalizedString:
32 case IrOpcode::kCheckEqualsSymbol:
33 case IrOpcode::kCheckFloat64Hole:
34 case IrOpcode::kCheckHeapObject:
35 case IrOpcode::kCheckIf:
36 case IrOpcode::kCheckInternalizedString:
37 case IrOpcode::kCheckNotTaggedHole:
38 case IrOpcode::kCheckNumber:
39 case IrOpcode::kCheckNumberFitsInt32:
40 case IrOpcode::kCheckReceiver:
41 case IrOpcode::kCheckReceiverOrNullOrUndefined:
42 case IrOpcode::kCheckSmi:
43 case IrOpcode::kCheckString:
44 case IrOpcode::kCheckStringOrStringWrapper:
45 case IrOpcode::kCheckSymbol:
46 // These are not really check nodes, but behave the same in that they can be
47 // folded together if repeated with identical inputs.
48 case IrOpcode::kStringCharCodeAt:
49 case IrOpcode::kStringCodePointAt:
50 case IrOpcode::kStringFromCodePointAt:
51 case IrOpcode::kStringSubstring:
52#define SIMPLIFIED_OP(Opcode) case IrOpcode::k##Opcode:
53 SIMPLIFIED_CHECKED_OP_LIST(SIMPLIFIED_OP)
54 SIMPLIFIED_BIGINT_BINOP_LIST(SIMPLIFIED_OP)
55#undef SIMPLIFIED_OP
56 return ReduceCheckNode(node);
57 case IrOpcode::kSpeculativeNumberEqual:
58 case IrOpcode::kSpeculativeNumberLessThan:
59 case IrOpcode::kSpeculativeNumberLessThanOrEqual:
60 return ReduceSpeculativeNumberComparison(node);
61 case IrOpcode::kSpeculativeNumberAdd:
62 case IrOpcode::kSpeculativeNumberSubtract:
63 case IrOpcode::kSpeculativeSafeIntegerAdd:
64 case IrOpcode::kSpeculativeSafeIntegerSubtract:
65 case IrOpcode::kSpeculativeToNumber:
66 return ReduceSpeculativeNumberOperation(node);
67 case IrOpcode::kEffectPhi:
68 return ReduceEffectPhi(node);
69 case IrOpcode::kDead:
70 break;
71 case IrOpcode::kStart:
72 return ReduceStart(node);
73 default:
74 return ReduceOtherNode(node);
75 }
76 return NoChange();
77}

92bool RedundancyElimination::EffectPathChecks::Equals(
93 EffectPathChecks const* that) const {
94 if (this->size_ != that->size_) return false;
95 Check* this_head = this->head_;
96 Check* that_head = that->head_;
97 while (this_head != that_head) {
98 if (this_head->node != that_head->node) return false;
99 this_head = this_head->next;
100 that_head = that_head->next;
101 }
102 return true;
103}

105void RedundancyElimination::EffectPathChecks::Merge(
106 EffectPathChecks const* that) {
107 // Change the current check list to a longest common tail of this check
108 // list and the other list.
109
110 // First, we throw away the prefix of the longer list, so that
111 // we have lists of the same length.
112 Check* that_head = that->head_;
113 size_t that_size = that->size_;
114 while (that_size > size_) {
115 that_head = that_head->next;
116 that_size--;
117 }
118 while (size_ > that_size) {
119 head_ = head_->next;
120 size_--;
121 }
122
123 // Then we go through both lists in lock-step until we find
124 // the common tail.
125 while (head_ != that_head) {
126 DCHECK_LT(0u, size_);
127 DCHECK_NOT_NULL(head_);
128 size_--;
129 head_ = head_->next;
130 that_head = that_head->next;
131 }
132}

143struct Subsumption {
144 enum class Kind {
145 kNone,
146 kImplicit,
147 kWithConversion,
148 };
149
150 static Subsumption None() { return Subsumption(Kind::kNone, nullptr); }
151 static Subsumption Implicit() {
152 return Subsumption(Kind::kImplicit, nullptr);
153 }
154 static Subsumption WithConversion(const Operator* conversion_op) {
155 return Subsumption(Kind::kWithConversion, conversion_op);
156 }
157
158 bool IsNone() const { return kind_ == Kind::kNone; }
159 bool IsImplicit() const { return kind_ == Kind::kImplicit; }
160 bool IsWithConversion() const { return kind_ == Kind::kWithConversion; }
161 const Operator* conversion_operator() const {
162 DCHECK(IsWithConversion());
163 return conversion_op_;
164 }
165
166 private:
167 Subsumption(Kind kind, const Operator* conversion_op)
168 : kind_(kind), conversion_op_(conversion_op) {
169 DCHECK_EQ(kind_ == Kind::kWithConversion, conversion_op_ != nullptr);
170 }
171
172 Kind kind_;
173 const Operator* conversion_op_;
174};

176// Does check {a} subsume check {b}?
177Subsumption CheckSubsumes(Node const* a, Node const* b,
178 MachineOperatorBuilder* machine) {
179 Subsumption subsumption = Subsumption::Implicit();
180 if (a->op() != b->op()) {
181 if (a->opcode() == IrOpcode::kCheckInternalizedString &&
182 b->opcode() == IrOpcode::kCheckString) {
183 // CheckInternalizedString(node) implies CheckString(node)
184 } else if (a->opcode() == IrOpcode::kCheckString &&
185 b->opcode() == IrOpcode::kCheckStringOrStringWrapper) {
186 // CheckString(node) implies CheckStringOrStringWrapper(node)
187 } else if (a->opcode() == IrOpcode::kCheckInternalizedString &&
188 b->opcode() == IrOpcode::kCheckStringOrStringWrapper) {
189 // CheckInteralizedString(node) implies CheckStringOrStringWrapper(node)
190 } else if (a->opcode() == IrOpcode::kCheckSmi &&
191 b->opcode() == IrOpcode::kCheckNumber) {
192 // CheckSmi(node) implies CheckNumber(node)
193 } else if (a->opcode() == IrOpcode::kCheckSmi &&
194 b->opcode() == IrOpcode::kCheckNumberFitsInt32) {
195 // CheckSmi(node) implies CheckNumberFitsInt32(node)
196 } else if (a->opcode() == IrOpcode::kCheckNumberFitsInt32 &&
197 b->opcode() == IrOpcode::kCheckNumber) {
198 // CheckNumberFitsInt32(node) implies CheckNumber(node)
199 } else if (a->opcode() == IrOpcode::kCheckedTaggedSignedToInt32 &&
200 b->opcode() == IrOpcode::kCheckedTaggedToInt32) {
201 // CheckedTaggedSignedToInt32(node) implies CheckedTaggedToInt32(node)
202 } else if (a->opcode() == IrOpcode::kCheckedTaggedSignedToInt32 &&
203 b->opcode() == IrOpcode::kCheckedTaggedToArrayIndex) {
204 // CheckedTaggedSignedToInt32(node) implies
205 // CheckedTaggedToArrayIndex(node)
206 if (machine->Is64()) {
207 // On 64 bit architectures, ArrayIndex is 64 bit.
208 subsumption =
209 Subsumption::WithConversion(machine->ChangeInt32ToInt64());
210 }
211 } else if (a->opcode() == IrOpcode::kCheckedTaggedToInt32 &&
212 b->opcode() == IrOpcode::kCheckedTaggedToArrayIndex) {
213 // CheckedTaggedToInt32(node) implies CheckedTaggedToArrayIndex(node)
214 if (machine->Is64()) {
215 // On 64 bit architectures, ArrayIndex is 64 bit.
216 subsumption =
217 Subsumption::WithConversion(machine->ChangeInt32ToInt64());
218 }
219 } else if (a->opcode() == IrOpcode::kCheckReceiver &&
220 b->opcode() == IrOpcode::kCheckReceiverOrNullOrUndefined) {
221 // CheckReceiver(node) implies CheckReceiverOrNullOrUndefined(node)
222 } else if (a->opcode() != b->opcode()) {
223 return Subsumption::None();
224 } else {
225 switch (a->opcode()) {
226 case IrOpcode::kCheckBounds:
227 case IrOpcode::kCheckSmi:
228 case IrOpcode::kCheckString:
229 case IrOpcode::kCheckStringOrStringWrapper:
230 case IrOpcode::kCheckNumber:
231 case IrOpcode::kCheckNumberFitsInt32:
232 case IrOpcode::kCheckBigInt:
233 case IrOpcode::kCheckedBigIntToBigInt64:
234 break;
235 case IrOpcode::kCheckedInt32ToTaggedSigned:
236 case IrOpcode::kCheckedInt64ToInt32:
237 case IrOpcode::kCheckedInt64ToTaggedSigned:
238 case IrOpcode::kCheckedTaggedSignedToInt32:
239 case IrOpcode::kCheckedTaggedToTaggedPointer:
240 case IrOpcode::kCheckedTaggedToTaggedSigned:
241 case IrOpcode::kCheckedTaggedToArrayIndex:
242 case IrOpcode::kCheckedUint32Bounds:
243 case IrOpcode::kCheckedUint32ToInt32:
244 case IrOpcode::kCheckedUint32ToTaggedSigned:
245 case IrOpcode::kCheckedUint64Bounds:
246 case IrOpcode::kCheckedUint64ToInt32:
247 case IrOpcode::kCheckedUint64ToTaggedSigned:
248 break;
249 case IrOpcode::kCheckedFloat64ToInt32:
250 case IrOpcode::kCheckedFloat64ToInt64:
251 case IrOpcode::kCheckedTaggedToInt32:
252 case IrOpcode::kCheckedTaggedToInt64: {
253 const CheckMinusZeroParameters& ap =
254 CheckMinusZeroParametersOf(a->op());
255 const CheckMinusZeroParameters& bp =
256 CheckMinusZeroParametersOf(b->op());
257 if (ap.mode() != bp.mode()) {
258 return Subsumption::None();
259 }
260 break;
261 }
262 case IrOpcode::kCheckedTaggedToFloat64:
263 case IrOpcode::kCheckedTruncateTaggedToWord32: {
264 CheckTaggedInputParameters const& ap =
265 CheckTaggedInputParametersOf(a->op());
266 CheckTaggedInputParameters const& bp =
267 CheckTaggedInputParametersOf(b->op());
268 // {a} subsumes {b} if the modes are either the same, or {a} checks
269 // for Number, in which case {b} will be subsumed no matter what.
270 if (ap.mode() != bp.mode() &&
271 ap.mode() != CheckTaggedInputMode::kNumber) {
272 return Subsumption::None();
273 }
274 break;
275 }
276 default:
277 DCHECK(!IsCheckedWithFeedback(a->op()));
278 return Subsumption::None();
279 }
280 }
281 }
282 for (int i = a->op()->ValueInputCount(); --i >= 0;) {
283 if (a->InputAt(i) != b->InputAt(i)) return Subsumption::None();
284 }
285 return subsumption;
286}

288bool TypeSubsumes(Node* node, Node* replacement) {
289 if (!NodeProperties::IsTyped(node) || !NodeProperties::IsTyped(replacement)) {
290 // If either node is untyped, we are running during an untyped optimization
291 // phase, and replacement is OK.
292 return true;
293 }
294 Type node_type = NodeProperties::GetType(node);
295 Type replacement_type = NodeProperties::GetType(replacement);
296 return replacement_type.Is(node_type);
297}

301Node* RedundancyElimination::EffectPathChecks::LookupCheck(
302 Node* node, JSGraph* jsgraph) const {
303 for (Check const* check = head_; check != nullptr; check = check->next) {
304 Subsumption subsumption =
305 CheckSubsumes(check->node, node, jsgraph->machine());
306 if (!subsumption.IsNone() && TypeSubsumes(node, check->node)) {
307 DCHECK(!check->node->IsDead());
308 Node* result = check->node;
309 if (subsumption.IsWithConversion()) {
310 result = jsgraph->graph()->NewNode(subsumption.conversion_operator(),
311 result);
312 }
313 return result;
314 }
315 }
316 return nullptr;
317}

319Node* RedundancyElimination::EffectPathChecks::LookupBoundsCheckFor(
320 Node* node) const {
321 for (Check const* check = head_; check != nullptr; check = check->next) {
322 if (check->node->opcode() == IrOpcode::kCheckBounds &&
323 check->node->InputAt(0) == node && TypeSubsumes(node, check->node) &&
324 !(CheckBoundsParametersOf(check->node->op()).flags() &
325 CheckBoundsFlag::kConvertStringAndMinusZero)) {
326 return check->node;
327 }
328 }
329 return nullptr;

346Reduction RedundancyElimination::ReduceCheckNode(Node* node) {
347 Node* const effect = NodeProperties::GetEffectInput(node);
348 EffectPathChecks const* checks = node_checks_.Get(effect);
349 // If we do not know anything about the predecessor, do not propagate just yet
350 // because we will have to recompute anyway once we compute the predecessor.
351 if (checks == nullptr) return NoChange();
352 // See if we have another check that dominates us.
353 if (Node* check = checks->LookupCheck(node, jsgraph_)) {
354 ReplaceWithValue(node, check);
355 return Replace(check);
356 }
357
358 // Learn from this check.
359 return UpdateChecks(node, checks->AddCheck(zone(), node));
360}

362Reduction RedundancyElimination::ReduceEffectPhi(Node* node) {
363 Node* const control = NodeProperties::GetControlInput(node);
364 if (control->opcode() == IrOpcode::kLoop) {
365 // Here we rely on having only reducible loops:
366 // The loop entry edge always dominates the header, so we can just use
367 // the information from the loop entry edge.
368 return TakeChecksFromFirstEffect(node);
369 }
370 DCHECK_EQ(IrOpcode::kMerge, control->opcode());
371
372 // Shortcut for the case when we do not know anything about some input.
373 int const input_count = node->op()->EffectInputCount();
374 for (int i = 0; i < input_count; ++i) {
375 Node* const effect = NodeProperties::GetEffectInput(node, i);
376 if (node_checks_.Get(effect) == nullptr) return NoChange();
377 }
378
379 // Make a copy of the first input's checks and merge with the checks
380 // from other inputs.
381 EffectPathChecks* checks = EffectPathChecks::Copy(
382 zone(), node_checks_.Get(NodeProperties::GetEffectInput(node, 0)));
383 for (int i = 1; i < input_count; ++i) {
384 Node* const input = NodeProperties::GetEffectInput(node, i);
385 checks->Merge(node_checks_.Get(input));
386 }
387 return UpdateChecks(node, checks);
388}

390Reduction RedundancyElimination::ReduceSpeculativeNumberComparison(Node* node) {
391 NumberOperationHint const hint = NumberOperationHintOf(node->op());
392 Node* const first = NodeProperties::GetValueInput(node, 0);
393 Type const first_type = NodeProperties::GetType(first);
394 Node* const second = NodeProperties::GetValueInput(node, 1);
395 Type const second_type = NodeProperties::GetType(second);
396 Node* const effect = NodeProperties::GetEffectInput(node);
397 EffectPathChecks const* checks = node_checks_.Get(effect);
398
399 // If we do not know anything about the predecessor, do not propagate just yet
400 // because we will have to recompute anyway once we compute the predecessor.
401 if (checks == nullptr) return NoChange();
402
403 // Avoid the potentially expensive lookups below if the {node}
404 // has seen non-Smi inputs in the past, which is a clear signal
405 // that the comparison is probably not performed on a value that
406 // already passed an array bounds check.
407 if (hint == NumberOperationHint::kSignedSmall) {
408 // Don't bother trying to find a CheckBounds for the {first} input
409 // if it's type is already in UnsignedSmall range, since the bounds
410 // check is only going to narrow that range further, but the result
411 // is not going to make the representation selection any better.
412 if (!first_type.Is(Type::UnsignedSmall())) {
413 if (Node* check = checks->LookupBoundsCheckFor(first)) {
414 if (!first_type.Is(NodeProperties::GetType(check))) {
415 // Replace the {first} input with the {check}. This is safe,
416 // despite the fact that {check} can truncate -0 to 0, because
417 // the regular Number comparisons in JavaScript also identify
418 // 0 and -0 (unlike special comparisons as Object.is).
419 NodeProperties::ReplaceValueInput(node, check, 0);
420 return Changed(node).FollowedBy(
421 ReduceSpeculativeNumberComparison(node));
422 }
423 }
424 }
425
426 // Don't bother trying to find a CheckBounds for the {second} input
427 // if it's type is already in UnsignedSmall range, since the bounds
428 // check is only going to narrow that range further, but the result
429 // is not going to make the representation selection any better.
430 if (!second_type.Is(Type::UnsignedSmall())) {
431 if (Node* check = checks->LookupBoundsCheckFor(second)) {
432 if (!second_type.Is(NodeProperties::GetType(check))) {
433 // Replace the {second} input with the {check}. This is safe,
434 // despite the fact that {check} can truncate -0 to 0, because
435 // the regular Number comparisons in JavaScript also identify
436 // 0 and -0 (unlike special comparisons as Object.is).
437 NodeProperties::ReplaceValueInput(node, check, 1);
438 return Changed(node).FollowedBy(
439 ReduceSpeculativeNumberComparison(node));
440 }
441 }
442 }
443 }
444
445 return UpdateChecks(node, checks);
446}

448Reduction RedundancyElimination::ReduceSpeculativeNumberOperation(Node* node) {
449 DCHECK(node->opcode() == IrOpcode::kSpeculativeNumberAdd ||
450 node->opcode() == IrOpcode::kSpeculativeNumberSubtract ||
451 node->opcode() == IrOpcode::kSpeculativeSafeIntegerAdd ||
452 node->opcode() == IrOpcode::kSpeculativeSafeIntegerSubtract ||
453 node->opcode() == IrOpcode::kSpeculativeToNumber);
454 DCHECK_EQ(1, node->op()->EffectInputCount());
455 DCHECK_EQ(1, node->op()->EffectOutputCount());
456
457 Node* const first = NodeProperties::GetValueInput(node, 0);
458 Node* const effect = NodeProperties::GetEffectInput(node);
459 EffectPathChecks const* checks = node_checks_.Get(effect);
460 // If we do not know anything about the predecessor, do not propagate just yet
461 // because we will have to recompute anyway once we compute the predecessor.
462 if (checks == nullptr) return NoChange();
463
464 // Check if there's a CheckBounds operation on {first}
465 // in the graph already, which we might be able to
466 // reuse here to improve the representation selection
467 // for the {node} later on.
468 if (Node* check = checks->LookupBoundsCheckFor(first)) {
469 // Only use the bounds {check} if its type is better
470 // than the type of the {first} node, otherwise we
471 // would end up replacing NumberConstant inputs with
472 // CheckBounds operations, which is kind of pointless.
473 if (!NodeProperties::GetType(first).Is(NodeProperties::GetType(check))) {
474 NodeProperties::ReplaceValueInput(node, check, 0);
475 }
476 }
477
478 return UpdateChecks(node, checks);
479}

485Reduction RedundancyElimination::ReduceOtherNode(Node* node) {
486 if (node->op()->EffectInputCount() == 1) {
487 if (node->op()->EffectOutputCount() == 1) {
488 return TakeChecksFromFirstEffect(node);
489 } else {
490 // Effect terminators should be handled specially.
491 return NoChange();
492 }
493 }
494 DCHECK_EQ(0, node->op()->EffectInputCount());
495 DCHECK_EQ(0, node->op()->EffectOutputCount());
496 return NoChange();
497}

499Reduction RedundancyElimination::TakeChecksFromFirstEffect(Node* node) {
500 DCHECK_EQ(1, node->op()->EffectOutputCount());
501 Node* const effect = NodeProperties::GetEffectInput(node);
502 EffectPathChecks const* checks = node_checks_.Get(effect);
503 // If we do not know anything about the predecessor, do not propagate just yet
504 // because we will have to recompute anyway once we compute the predecessor.
505 if (checks == nullptr) return NoChange();
506 // We just propagate the information from the effect input (ideally,
507 // we would only revisit effect uses if there is change).
508 return UpdateChecks(node, checks);
509}

511Reduction RedundancyElimination::UpdateChecks(Node* node,
512 EffectPathChecks const* checks) {
513 EffectPathChecks const* original = node_checks_.Get(node);
514 // Only signal that the {node} has Changed, if the information about {checks}
515 // has changed wrt. the {original}.
516 if (checks != original) {
517 if (original == nullptr || !checks->Equals(original)) {
518 node_checks_.Set(node, checks);
519 return Changed(node);
520 }
521 }
522 return NoChange();
523}
