35void WritableRelocInfo::apply(intptr_t delta) {
36 if (IsInternalReference(rmode_)) {
37 // Absolute code pointer inside code object moves with the code object.
38 intptr_t internal_ref = ReadUnalignedValue<intptr_t>(pc_);
39 internal_ref += delta; // Relocate entry.
40 jit_allocation_.WriteUnalignedValue<intptr_t>(pc_, internal_ref);
41 } else {
42 DCHECK(IsRelativeCodeTarget(rmode_) || IsNearBuiltinEntry(rmode_));
43 Assembler::RelocateRelativeReference(rmode_, pc_, delta, &jit_allocation_);
44 }
45}

53Address RelocInfo::target_address_address() {
54 DCHECK(HasTargetAddressAddress());
55 // Read the address of the word containing the target_address in an
56 // instruction stream.
57 // The only architecture-independent user of this function is the serializer.
58 // The serializer uses it to find out how many raw bytes of instruction to
59 // output before the next target.
60 // For an instruction like LUI/ORI where the target bits are mixed into the
61 // instruction bits, the size of the target will be zero, indicating that the
62 // serializer should not step forward in memory after a target is resolved
63 // and written. In this case the target_address_address function should
64 // return the end of the instructions to be patched, allowing the
65 // deserializer to deserialize the instructions as raw bytes and put them in
66 // place, ready to be patched with the target. After jump optimization,
67 // that is the address of the instruction that follows J/JAL/JR/JALR
68 // instruction.
69 return pc_ + Assembler::kInstructionsFor64BitConstant * kInstrSize;
70}

110Tagged<HeapObject> RelocInfo::target_object(PtrComprCageBase cage_base) {
111 DCHECK(IsCodeTarget(rmode_) || IsFullEmbeddedObject(rmode_));
112 if (IsCompressedEmbeddedObject(rmode_)) {
113 Tagged_t compressed =
114 Assembler::target_compressed_address_at(pc_, constant_pool_);
115 DCHECK(!HAS_SMI_TAG(compressed));
116 Tagged<Object> obj(
117 V8HeapCompressionScheme::DecompressTagged(cage_base, compressed));
118 return Cast<HeapObject>(obj);
119 } else {
120 return Cast<HeapObject>(
121 Tagged<Object>(Assembler::target_address_at(pc_, constant_pool_)));
122 }
123}

138void WritableRelocInfo::set_target_object(Tagged<HeapObject> target,
139 ICacheFlushMode icache_flush_mode) {
140 DCHECK(IsCodeTarget(rmode_) || IsEmbeddedObjectMode(rmode_));
141 if (IsCompressedEmbeddedObject(rmode_)) {
142 DCHECK(COMPRESS_POINTERS_BOOL);
143 // We must not compress pointers to objects outside of the main pointer
144 // compression cage as we wouldn't be able to decompress them with the
145 // correct cage base.
146 DCHECK_IMPLIES(V8_ENABLE_SANDBOX_BOOL, !HeapLayout::InTrustedSpace(target));
147 DCHECK_IMPLIES(V8_EXTERNAL_CODE_SPACE_BOOL,
148 !HeapLayout::InCodeSpace(target));
149 Assembler::set_target_compressed_address_at(
150 pc_, constant_pool_,
151 V8HeapCompressionScheme::CompressObject(target.ptr()), &jit_allocation_,
152 icache_flush_mode);
153 } else {
154 Assembler::set_target_address_at(pc_, constant_pool_, target.ptr(),
155 &jit_allocation_, icache_flush_mode);
156 }
157}

201Handle<Code> Assembler::relative_code_target_object_handle_at(
202 Address pc) const {
203 Instr instr = instr_at(pc);
204 int32_t code_target_index = instr & kImm26Mask;
205 code_target_index = ((code_target_index & 0x3ff) << 22 >> 6) |
206 ((code_target_index >> 10) & kImm16Mask);
207 return GetCodeTarget(code_target_index);
208}
