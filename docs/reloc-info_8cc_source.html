24uint32_t RelocInfoWriter::WriteLongPCJump(uint32_t pc_delta) {
25 // Return if the pc_delta can fit in kSmallPCDeltaBits bits.
26 // Otherwise write a variable length PC jump for the bits that do
27 // not fit in the kSmallPCDeltaBits bits.
28 if (is_uintn(pc_delta, kSmallPCDeltaBits)) return pc_delta;
29 WriteMode(RelocInfo::PC_JUMP);
30 uint32_t pc_jump = pc_delta >> kSmallPCDeltaBits;
31 DCHECK_GT(pc_jump, 0);
32 base::VLQEncodeUnsigned([this](uint8_t byte) { *--pos_ = byte; }, pc_jump);
33 // Return the remaining kSmallPCDeltaBits of the pc_delta.
34 return pc_delta & kSmallPCDeltaMask;
35}

37void RelocInfoWriter::WriteShortTaggedPC(uint32_t pc_delta, int tag) {
38 // Write a byte of tagged pc-delta, possibly preceded by an explicit pc-jump.
39 pc_delta = WriteLongPCJump(pc_delta);
40 *--pos_ = pc_delta << kTagBits | tag;

52void RelocInfoWriter::WriteModeAndPC(uint32_t pc_delta, RelocInfo::Mode rmode) {
53 // Write two-byte tagged pc-delta, possibly preceded by var. length pc-jump.
54 pc_delta = WriteLongPCJump(pc_delta);
55 WriteMode(rmode);
56 *--pos_ = pc_delta;

59void RelocInfoWriter::WriteIntData(int number) {
60 for (int i = 0; i < kIntSize; i++) {

63 number = number >> kBitsPerByte;

67void RelocInfoWriter::Write(const RelocInfo* rinfo) {
68 RelocInfo::Mode rmode = rinfo->rmode();
69#ifdef DEBUG
70 uint8_t* begin_pos = pos_;
71#endif
72 DCHECK(rinfo->rmode() < RelocInfo::NUMBER_OF_MODES);
73 DCHECK_GE(rinfo->pc() - reinterpret_cast<Address>(last_pc_), 0);
74 // Use unsigned delta-encoding for pc.
75 uint32_t pc_delta =
76 static_cast<uint32_t>(rinfo->pc() - reinterpret_cast<Address>(last_pc_));
77
78 // The two most common modes are given small tags, and usually fit in a byte.
79 if (rmode == RelocInfo::FULL_EMBEDDED_OBJECT) {
80 WriteShortTaggedPC(pc_delta, kEmbeddedObjectTag);
81 } else if (rmode == RelocInfo::CODE_TARGET) {
82 WriteShortTaggedPC(pc_delta, kCodeTargetTag);
83 DCHECK_LE(begin_pos - pos_, RelocInfo::kMaxCallSize);
84 } else if (rmode == RelocInfo::WASM_STUB_CALL) {
85 WriteShortTaggedPC(pc_delta, kWasmStubCallTag);
86 } else {
87 WriteModeAndPC(pc_delta, rmode);
88 if (RelocInfo::IsDeoptReason(rmode)) {
89 DCHECK_LT(rinfo->data(), 1 << kBitsPerByte);
90 WriteShortData(static_cast<uint8_t>(rinfo->data()));
91 } else if (RelocInfo::IsConstPool(rmode) ||
92 RelocInfo::IsVeneerPool(rmode) || RelocInfo::IsDeoptId(rmode) ||
93 RelocInfo::IsDeoptPosition(rmode) ||
94 RelocInfo::IsDeoptNodeId(rmode)) {
95 WriteIntData(static_cast<int>(rinfo->data()));
96 }
97 }
98 last_pc_ = reinterpret_cast<uint8_t*>(rinfo->pc());
99#ifdef DEBUG
100 DCHECK_LE(begin_pos - pos_, kMaxSize);
101#endif
102}

104template <typename RelocInfoT>
105void RelocIteratorBase<RelocInfoT>::AdvanceReadInt() {
106 int x = 0;
107 for (int i = 0; i < kIntSize; i++) {
108 x |= static_cast<int>(*--pos_) << i * kBitsPerByte;
109 }
110 rinfo_.data_ = x;
111}

113template <typename RelocInfoT>
114void RelocIteratorBase<RelocInfoT>::AdvanceReadLongPCJump() {
115 // Read the 32-kSmallPCDeltaBits most significant bits of the
116 // pc jump as a VLQ encoded integer.
117 uint32_t pc_jump = base::VLQDecodeUnsigned([this] { return *--pos_; });
118 // The least significant kSmallPCDeltaBits bits will be added
119 // later.
120 rinfo_.pc_ += pc_jump << kSmallPCDeltaBits;
121}

129template <typename RelocInfoT>
130void RelocIteratorBase<RelocInfoT>::next() {
131 DCHECK(!done());
132 // Basically, do the opposite of RelocInfoWriter::Write.
133 // Reading of data is as far as possible avoided for unwanted modes,
134 // but we must always update the pc.
135 //
136 // We exit this loop by returning when we find a mode we want.
137 while (pos_ > end_) {
138 int tag = AdvanceGetTag();
139 if (tag == kEmbeddedObjectTag) {
140 ReadShortTaggedPC();
141 if (SetMode(RelocInfo::FULL_EMBEDDED_OBJECT)) return;
142 } else if (tag == kCodeTargetTag) {
143 ReadShortTaggedPC();
144 if (SetMode(RelocInfo::CODE_TARGET)) return;
145 } else if (tag == kWasmStubCallTag) {
146 ReadShortTaggedPC();
147 if (SetMode(RelocInfo::WASM_STUB_CALL)) return;
148 } else {
149 DCHECK_EQ(tag, kDefaultTag);
150 RelocInfo::Mode rmode = GetMode();
151 if (rmode == RelocInfo::PC_JUMP) {
152 AdvanceReadLongPCJump();
153 } else {
154 AdvanceReadPC();
155 if (RelocInfo::IsDeoptReason(rmode)) {
156 Advance();
157 if (SetMode(rmode)) {
158 ReadShortData();
159 return;
160 }
161 } else if (RelocInfo::IsConstPool(rmode) ||
162 RelocInfo::IsVeneerPool(rmode) ||
163 RelocInfo::IsDeoptId(rmode) ||
164 RelocInfo::IsDeoptPosition(rmode) ||
165 RelocInfo::IsDeoptNodeId(rmode)) {
166 if (SetMode(rmode)) {
167 AdvanceReadInt();
168 return;
169 }
170 Advance(kIntSize);
171 } else if (SetMode(static_cast<RelocInfo::Mode>(rmode))) {
172 return;
173 }
174 }
175 }
176 }
177 done_ = true;
178}

313bool RelocInfo::HasTargetAddressAddress() const {
314 // TODO(jgruber): Investigate whether WASM_CALL is still appropriate on
315 // non-intel platforms now that wasm code is no longer on the heap.
316#if defined(V8_TARGET_ARCH_IA32) || defined(V8_TARGET_ARCH_X64)
317 static constexpr int kTargetAddressAddressModeMask =
318 ModeMask(CODE_TARGET) | ModeMask(FULL_EMBEDDED_OBJECT) |
319 ModeMask(COMPRESSED_EMBEDDED_OBJECT) | ModeMask(EXTERNAL_REFERENCE) |
320 ModeMask(OFF_HEAP_TARGET) | ModeMask(WASM_CALL) |
321 ModeMask(WASM_STUB_CALL);
322#else
323 static constexpr int kTargetAddressAddressModeMask =
324 ModeMask(CODE_TARGET) | ModeMask(RELATIVE_CODE_TARGET) |
325 ModeMask(FULL_EMBEDDED_OBJECT) | ModeMask(EXTERNAL_REFERENCE) |
326 ModeMask(OFF_HEAP_TARGET) | ModeMask(WASM_CALL);
327#endif
328 return (ModeMask(rmode_) & kTargetAddressAddressModeMask) != 0;
329}

385void RelocInfo::Print(Isolate* isolate, std::ostream& os) {
386 os << reinterpret_cast<const void*>(pc_) << " " << RelocModeName(rmode_);
387 if (rmode_ == DEOPT_SCRIPT_OFFSET || rmode_ == DEOPT_INLINING_ID) {
388 os << " (" << data() << ")";
389 } else if (rmode_ == DEOPT_REASON) {
390 os << " ("
391 << DeoptimizeReasonToString(static_cast<DeoptimizeReason>(data_)) << ")";
392 } else if (rmode_ == FULL_EMBEDDED_OBJECT) {
393 os << " (" << Brief(target_object(isolate)) << ")";
394 } else if (rmode_ == COMPRESSED_EMBEDDED_OBJECT) {
395 os << " (" << Brief(target_object(isolate)) << " compressed)";
396 } else if (rmode_ == EXTERNAL_REFERENCE) {
397 if (isolate) {
398 ExternalReferenceEncoder ref_encoder(isolate);
399 os << " ("
400 << ref_encoder.NameOfAddress(isolate, target_external_reference())
401 << ") ";
402 }
403 os << " (" << reinterpret_cast<const void*>(target_external_reference())
404 << ")";
405 } else if (rmode_ == JS_DISPATCH_HANDLE) {
406#ifdef V8_ENABLE_LEAPTIERING
407 Tagged<Code> target_code =
408 IsolateGroup::current()->js_dispatch_table()->GetCode(
409 js_dispatch_handle());
410 os << " (" << CodeKindToString(target_code->kind());
411 if (Builtins::IsBuiltin(target_code)) {
412 os << " " << Builtins::name(target_code->builtin_id());
413 }
414 os << ") (" << reinterpret_cast<const void*>(target_address()) << ")";
415#else
416 UNREACHABLE();
417#endif
418 } else if (IsCodeTargetMode(rmode_)) {
419 const Address code_target = target_address();
420 Tagged<Code> target_code = Code::FromTargetAddress(code_target);
421 os << " (" << CodeKindToString(target_code->kind());
422 if (Builtins::IsBuiltin(target_code)) {
423 os << " " << Builtins::name(target_code->builtin_id());
424 }
425 os << ") (" << reinterpret_cast<const void*>(target_address()) << ")";
426 } else if (IsConstPool(rmode_)) {
427 os << " (size " << static_cast<int>(data_) << ")";
428 } else if (IsWasmStubCall(rmode_)) {
429 os << " (";
430 Address addr = target_address();
431 if (isolate != nullptr) {
432 Builtin builtin = OffHeapInstructionStream::TryLookupCode(isolate, addr);
433 os << (Builtins::IsBuiltinId(builtin) ? Builtins::name(builtin)
434 : "<UNRECOGNIZED>")
435 << ") (";
436 }
437 os << reinterpret_cast<const void*>(addr) << ")";
438 }
439
440 os << "\n";
441}

444#ifdef VERIFY_HEAP
445void RelocInfo::Verify(Isolate* isolate) {
446 switch (rmode_) {
447 case COMPRESSED_EMBEDDED_OBJECT:
448 Object::VerifyPointer(isolate, target_object(isolate));
449 break;
450 case FULL_EMBEDDED_OBJECT:
451 Object::VerifyAnyTagged(isolate, target_object(isolate));
452 break;
453 case CODE_TARGET:
454 case RELATIVE_CODE_TARGET: {
455 // convert inline target address to code object
456 Address addr = target_address();
457 CHECK_NE(addr, kNullAddress);
458 // Check that we can find the right code object.
459 Tagged<InstructionStream> code =
460 InstructionStream::FromTargetAddress(addr);
461 Tagged<Code> lookup_result =
462 isolate->heap()->FindCodeForInnerPointer(addr);
463 CHECK_EQ(code.address(), lookup_result->instruction_stream().address());
464 break;
465 }
466 case INTERNAL_REFERENCE:
467 case INTERNAL_REFERENCE_ENCODED: {
468 Address target = target_internal_reference();
469 Address pc = target_internal_reference_address();
470 Tagged<Code> lookup_result = isolate->heap()->FindCodeForInnerPointer(pc);
471 CHECK_GE(target, lookup_result->instruction_start());
472 CHECK_LT(target, lookup_result->instruction_end());
473 break;
474 }
475 case JS_DISPATCH_HANDLE: {
476#ifdef V8_ENABLE_LEAPTIERING
477 JSDispatchTable::Space* space =
478 isolate->heap()->js_dispatch_table_space();
479 JSDispatchTable::Space* ro_space =
480 isolate->read_only_heap()->js_dispatch_table_space();
481 IsolateGroup::current()->js_dispatch_table()->VerifyEntry(
482 js_dispatch_handle(), space, ro_space);
483 break;
484#else
485 UNREACHABLE();
486#endif
487 }
488 case OFF_HEAP_TARGET: {
489 Address addr = target_off_heap_target();
490 CHECK_NE(addr, kNullAddress);
491 CHECK(Builtins::IsBuiltinId(
492 OffHeapInstructionStream::TryLookupCode(isolate, addr)));
493 break;
494 }
495 case WASM_STUB_CALL:
496 case NEAR_BUILTIN_ENTRY: {
497 Address addr = target_address();
498 CHECK_NE(addr, kNullAddress);
499 CHECK(Builtins::IsBuiltinId(
500 OffHeapInstructionStream::TryLookupCode(isolate, addr)));
501 break;
502 }
503 case EXTERNAL_REFERENCE:
504 case DEOPT_SCRIPT_OFFSET:
505 case DEOPT_INLINING_ID:
506 case DEOPT_REASON:
507 case DEOPT_ID:
508 case DEOPT_NODE_ID:
509 case CONST_POOL:
510 case VENEER_POOL:
511 case WASM_CALL:
512 case NO_INFO:
513 case WASM_CANONICAL_SIG_ID:
514 case WASM_CODE_POINTER_TABLE_ENTRY:
515 break;
516 case NUMBER_OF_MODES:
517 case PC_JUMP:
518 UNREACHABLE();
519 }
520}
