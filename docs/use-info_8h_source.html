15// Enum to specify if `+0` and `-0` should be treated as the same value.

45 }
46 static Truncation Any(IdentifyZeros identify_zeros = kDistinguishZeros) {

49
50 static Truncation Generalize(Truncation t1, Truncation t2) {
51 return Truncation(
52 Generalize(t1.kind(), t2.kind()),
53 GeneralizeIdentifyZeros(t1.identify_zeros(), t2.identify_zeros()));
54 }
55
56 // Queries.
57 bool IsUnused() const { return kind_ == TruncationKind::kNone; }
58 bool IsUsedAsBool() const {
59 return LessGeneral(kind_, TruncationKind::kBool);
60 }
61 bool IsUsedAsWord32() const {
62 return LessGeneral(kind_, TruncationKind::kWord32);
63 }
64 bool IsUsedAsWord64() const {
65 DCHECK(Is64());
66 return LessGeneral(kind_, TruncationKind::kWord64);
67 }
68 bool TruncatesOddballAndBigIntToNumber() const {
69 return LessGeneral(kind_, TruncationKind::kOddballAndBigIntToNumber);
70 }
71 bool IdentifiesUndefinedAndZero() {
72 return LessGeneral(kind_, TruncationKind::kWord32) ||
73 LessGeneral(kind_, TruncationKind::kBool);
74 }
75 bool IdentifiesZeroAndMinusZero() const {
76 return identify_zeros() == kIdentifyZeros;
77 }
78
79 // Operators.
80 bool operator==(Truncation other) const {
81 return kind() == other.kind() && identify_zeros() == other.identify_zeros();
82 }
83 bool operator!=(Truncation other) const { return !(*this == other); }
84
85 // Debug utilities.
86 const char* description() const;
87 bool IsLessGeneralThan(Truncation other) const {
88 return LessGeneral(kind(), other.kind()) &&
89 LessGeneralIdentifyZeros(identify_zeros(), other.identify_zeros());
90 }
91
92 IdentifyZeros identify_zeros() const { return identify_zeros_; }
93
94 private:

96 kNone,
97 kBool,
98 kWord32,
99 kWord64,
100 kOddballAndBigIntToNumber,
101 kAny
102 };

104 explicit Truncation(TruncationKind kind, IdentifyZeros identify_zeros)
105 : kind_(kind), identify_zeros_(identify_zeros) {}

107 TruncationKind kind() const { return kind_; }

172// The {UseInfo} class is used to describe a use of an input of a node.
173//
174// This information is used in two different ways, based on the phase:
175//
176// 1. During propagation, the use info is used to inform the input node
177// about what part of the input is used (we call this truncation) and what
178// is the preferred representation. For conversions that will require
179// checks, we also keep track of whether a minus zero check is needed.
180//
181// 2. During lowering, the use info is used to properly convert the input
182// to the preferred representation. The preferred representation might be
183// insufficient to do the conversion (e.g. word32->float64 conv), so we also
184// need the signedness information to produce the correct value.
185// Additionally, use info may contain {CheckParameters} which contains
186// information for the deoptimizer such as a CallIC on which speculation
187// should be disallowed if the check fails.
188class UseInfo {
189 public:
190 UseInfo(MachineRepresentation representation, Truncation truncation,
191 TypeCheckKind type_check = TypeCheckKind::kNone,
192 const FeedbackSource& feedback = FeedbackSource())
193 : representation_(representation),
194 truncation_(truncation),
195 type_check_(type_check),
196 feedback_(feedback) {}
197 static UseInfo TruncatingWord32() {
198 return UseInfo(MachineRepresentation::kWord32, Truncation::Word32());
199 }
200 static UseInfo TruncatingWord64() {
201 return UseInfo(MachineRepresentation::kWord64, Truncation::Word64());
202 }
203 static UseInfo CheckedBigIntTruncatingWord64(const FeedbackSource& feedback) {
204 DCHECK(Is64());
205 // Note that Trunction::Word64() can safely use kIdentifyZero, because
206 // TypeCheckKind::kBigInt will make sure we deopt for anything other than
207 // type BigInt anyway.
208 return UseInfo(MachineRepresentation::kWord64, Truncation::Word64(),
209 TypeCheckKind::kBigInt, feedback);
210 }
211 static UseInfo CheckedBigInt64AsWord64(const FeedbackSource& feedback) {
212 DCHECK(Is64());
213 return UseInfo(MachineRepresentation::kWord64, Truncation::Any(),
214 TypeCheckKind::kBigInt64, feedback);
215 }
216 static UseInfo Word64(IdentifyZeros identify_zeros = kDistinguishZeros) {
217 return UseInfo(MachineRepresentation::kWord64,
218 Truncation::Any(identify_zeros));
219 }
220 static UseInfo Word() {
221 return UseInfo(MachineType::PointerRepresentation(), Truncation::Any());
222 }
223 static UseInfo Bool() {
224 return UseInfo(MachineRepresentation::kBit, Truncation::Bool());
225 }
226 static UseInfo Float32() {
227 return UseInfo(MachineRepresentation::kFloat32, Truncation::Any());
228 }
229 static UseInfo Float64() {
230 return UseInfo(MachineRepresentation::kFloat64, Truncation::Any());
231 }
232 static UseInfo TruncatingFloat64(
233 IdentifyZeros identify_zeros = kDistinguishZeros) {
234 return UseInfo(MachineRepresentation::kFloat64,
235 Truncation::OddballAndBigIntToNumber(identify_zeros));
236 }
237 static UseInfo AnyTagged() {
238 return UseInfo(MachineRepresentation::kTagged, Truncation::Any());
239 }
240 static UseInfo TaggedSigned() {
241 return UseInfo(MachineRepresentation::kTaggedSigned, Truncation::Any());
242 }
243 static UseInfo TaggedPointer() {
244 return UseInfo(MachineRepresentation::kTaggedPointer, Truncation::Any());
245 }
246
247 // Possibly deoptimizing conversions.
248 static UseInfo CheckedTaggedAsArrayIndex(const FeedbackSource& feedback) {
249 return UseInfo(MachineType::PointerRepresentation(),
250 Truncation::Any(kIdentifyZeros), TypeCheckKind::kArrayIndex,
251 feedback);
252 }
253 static UseInfo CheckedHeapObjectAsTaggedPointer(
254 const FeedbackSource& feedback) {
255 return UseInfo(MachineRepresentation::kTaggedPointer, Truncation::Any(),
256 TypeCheckKind::kHeapObject, feedback);
257 }
258
259 static UseInfo CheckedBigIntAsTaggedPointer(const FeedbackSource& feedback) {
260 return UseInfo(MachineRepresentation::kTaggedPointer, Truncation::Any(),
261 TypeCheckKind::kBigInt, feedback);
262 }
263
264 static UseInfo CheckedSignedSmallAsTaggedSigned(
265 const FeedbackSource& feedback,
266 IdentifyZeros identify_zeros = kDistinguishZeros) {
267 return UseInfo(MachineRepresentation::kTaggedSigned,
268 Truncation::Any(identify_zeros), TypeCheckKind::kSignedSmall,
269 feedback);
270 }
271 static UseInfo CheckedSignedSmallAsWord32(IdentifyZeros identify_zeros,
272 const FeedbackSource& feedback) {
273 return UseInfo(MachineRepresentation::kWord32,
274 Truncation::Any(identify_zeros), TypeCheckKind::kSignedSmall,
275 feedback);
276 }
277 static UseInfo CheckedSigned32AsWord32(IdentifyZeros identify_zeros,
278 const FeedbackSource& feedback) {
279 return UseInfo(MachineRepresentation::kWord32,
280 Truncation::Any(identify_zeros), TypeCheckKind::kSigned32,
281 feedback);
282 }
283 static UseInfo CheckedSigned64AsWord64(IdentifyZeros identify_zeros,
284 const FeedbackSource& feedback) {

286 Truncation::Any(identify_zeros), TypeCheckKind::kSigned64,
287 feedback);
288 }
289 static UseInfo CheckedNumberAsFloat64(IdentifyZeros identify_zeros,
290 const FeedbackSource& feedback) {
291 return UseInfo(MachineRepresentation::kFloat64,
292 Truncation::Any(identify_zeros), TypeCheckKind::kNumber,
293 feedback);
294 }
295 static UseInfo CheckedNumberAsWord32(const FeedbackSource& feedback) {
296 return UseInfo(MachineRepresentation::kWord32, Truncation::Word32(),
297 TypeCheckKind::kNumber, feedback);
298 }
299 static UseInfo CheckedNumberOrBooleanAsFloat64(
300 IdentifyZeros identify_zeros, const FeedbackSource& feedback) {
301 return UseInfo(MachineRepresentation::kFloat64,
302 Truncation::Any(identify_zeros),
303 TypeCheckKind::kNumberOrBoolean, feedback);
304 }
305 static UseInfo CheckedNumberOrOddballAsFloat64(
306 IdentifyZeros identify_zeros, const FeedbackSource& feedback) {
307 return UseInfo(MachineRepresentation::kFloat64,
308 Truncation::Any(identify_zeros),
309 TypeCheckKind::kNumberOrOddball, feedback);
310 }
311 static UseInfo CheckedNumberOrOddballAsWord32(
312 const FeedbackSource& feedback) {
313 return UseInfo(MachineRepresentation::kWord32, Truncation::Word32(),
314 TypeCheckKind::kNumberOrOddball, feedback);
315 }
316
317 // Undetermined representation.
318 static UseInfo Any() {
319 return UseInfo(MachineRepresentation::kNone, Truncation::Any());
320 }
321 static UseInfo AnyTruncatingToBool() {
322 return UseInfo(MachineRepresentation::kNone, Truncation::Bool());
323 }
324
325 // Value not used.
326 static UseInfo None() {
327 return UseInfo(MachineRepresentation::kNone, Truncation::None());
328 }
329
330 MachineRepresentation representation() const { return representation_; }
331 Truncation truncation() const { return truncation_; }
332 TypeCheckKind type_check() const { return type_check_; }
333 CheckForMinusZeroMode minus_zero_check() const {
334 return truncation().IdentifiesZeroAndMinusZero()
335 ? CheckForMinusZeroMode::kDontCheckForMinusZero
336 : CheckForMinusZeroMode::kCheckForMinusZero;
337 }
338 const FeedbackSource& feedback() const { return feedback_; }
339
340 private:
341 MachineRepresentation representation_;
342 Truncation truncation_;
343 TypeCheckKind type_check_;
344 FeedbackSource feedback_;
345};

347inline bool operator==(const UseInfo& lhs, const UseInfo& rhs) {
348 return lhs.representation() == rhs.representation() &&
349 lhs.truncation() == rhs.truncation() &&
350 lhs.type_check() == rhs.type_check() &&
351 lhs.feedback() == rhs.feedback();
352}

356 use_info.type_check(), use_info.feedback());
