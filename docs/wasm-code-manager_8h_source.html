263 void IncRef() {
264 [[maybe_unused]] int old_val =
265 ref_count_.fetch_add(1, std::memory_order_acq_rel);
266 DCHECK_LE(1, old_val);
267 DCHECK_GT(kMaxInt, old_val);
268 }

270 // Decrement the ref count. Returns whether this code becomes dead and needs
271 // to be freed.
272 V8_WARN_UNUSED_RESULT bool DecRef() {
273 int old_count = ref_count_.load(std::memory_order_acquire);
274 while (true) {
275 DCHECK_LE(1, old_count);
276 if (V8_UNLIKELY(old_count == 1)) return DecRefOnPotentiallyDeadCode();
277 if (ref_count_.compare_exchange_weak(old_count, old_count - 1,
278 std::memory_order_acq_rel)) {
279 return false;
280 }
281 }
282 }

284 // Decrement the ref count on code that is known to be in use (i.e. the ref
285 // count cannot drop to zero here).
286 void DecRefOnLiveCode() {
287 [[maybe_unused]] int old_count =
288 ref_count_.fetch_sub(1, std::memory_order_acq_rel);
289 DCHECK_LE(2, old_count);
290 }

292 // Decrement the ref count on code that is known to be dead, even though there
293 // might still be C++ references. Returns whether this drops the last
294 // reference and the code needs to be freed.
295 V8_WARN_UNUSED_RESULT bool DecRefOnDeadCode() {
296 return ref_count_.fetch_sub(1, std::memory_order_acq_rel) == 1;
297 }

460 // it's being used. Once the ref count drops to zero (i.e. after being removed

840 // This function tries to set the fast API call target of function import
841 // `index`. If the call target has been set before with a different value,
842 // then this function returns false, and this import will be marked as not
843 // suitable for wellknown imports, i.e. all existing compiled code of the
844 // module gets flushed, and future calls to this import will not use fast API
845 // calls.
846 bool TrySetFastApiCallTarget(int func_index, Address target) {
847 Address old_val =
848 fast_api_targets_[func_index].load(std::memory_order_relaxed);
849 if (old_val == target) {
850 return true;
851 }
852 if (old_val != kNullAddress) {
853 // If already a different target is stored, then there are conflicting
854 // targets and fast api calls are not possible. In that case the import
855 // will be marked as not suitable for wellknown imports, and the
856 // `fast_api_target` of this import will never be used anymore in the
857 // future.
858 return false;
859 }
860 if (fast_api_targets_[func_index].compare_exchange_strong(
861 old_val, target, std::memory_order_relaxed)) {
862 return true;
863 }
864 // If a concurrent call to `TrySetFastAPICallTarget` set the call target to
865 // the same value as this call, we consider also this call successful.
866 return old_val == target;
867 }
