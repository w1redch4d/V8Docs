168 bool operator==(const WasmInitExpr& other) const {
169 if (kind() != other.kind()) return false;
170 switch (kind()) {
171 case kGlobalGet:
172 case kRefFuncConst:
173 case kStringConst:
174 return immediate().index == other.immediate().index;
175 case kI32Const:
176 return immediate().i32_const == other.immediate().i32_const;
177 case kI64Const:
178 return immediate().i64_const == other.immediate().i64_const;
179 case kF32Const:
180 return immediate().f32_const == other.immediate().f32_const;
181 case kF64Const:
182 return immediate().f64_const == other.immediate().f64_const;
183 case kI32Add:
184 case kI32Sub:
185 case kI32Mul:
186 case kI64Add:
187 case kI64Sub:
188 case kI64Mul:
189 return operands_[0] == other.operands_[0] &&
190 operands_[1] == other.operands_[1];
191 case kS128Const:
192 return immediate().s128_const == other.immediate().s128_const;
193 case kRefNullConst:
194 return immediate().heap_type == other.immediate().heap_type;
195 case kStructNew:
196 case kStructNewDefault:
197 case kArrayNew:
198 case kArrayNewDefault:
199 if (immediate().index != other.immediate().index) return false;
200 DCHECK_EQ(operands()->size(), other.operands()->size());
201 for (uint32_t i = 0; i < operands()->size(); i++) {
202 if (operands()[i] != other.operands()[i]) return false;
203 }
204 return true;
205 case kArrayNewFixed:
206 if (immediate().index != other.immediate().index) return false;
207 if (operands()->size() != other.operands()->size()) return false;
208 for (uint32_t i = 0; i < operands()->size(); i++) {
209 if (operands()[i] != other.operands()[i]) return false;
210 }
211 return true;
212 case kRefI31:
213 case kAnyConvertExtern:
214 case kExternConvertAny:
215 return operands_[0] == other.operands_[0];
216 }
217 }

219 V8_INLINE bool operator!=(const WasmInitExpr& other) const {

223 static WasmInitExpr DefaultValue(ValueType type) {
224 // No initializer, emit a default value.
225 switch (type.kind()) {
226 case kI8:
227 case kI16:
228 case kI32:
229 return WasmInitExpr(int32_t{0});
230 case kI64:
231 return WasmInitExpr(int64_t{0});
232 case kF16:
233 case kF32:
234 return WasmInitExpr(0.0f);
235 case kF64:
236 return WasmInitExpr(0.0);
237 case kRefNull:
238 return WasmInitExpr::RefNullConst(type.heap_representation());
239 case kS128: {
240 uint8_t value[kSimd128Size] = {0};
241 return WasmInitExpr(value);
242 }
243 case kVoid:
244 case kTop:
245 case kBottom:
246 case kRef:
247 case kRtt:
248 UNREACHABLE();
249 }
250 }

258 : kind_(kind),
259 operands_(zone->New<ZoneVector<WasmInitExpr>>(operands, zone)) {}
260 Immediate immediate_;
261 Operator kind_;
262 const ZoneVector<WasmInitExpr>* operands_;
263};
