151void Builtins::Generate_WasmInterpreterEntry(MacroAssembler* masm) {
152 // Input registers:
153 // x7 (kWasmImplicitArgRegister): wasm_instance
154 // x12: array_start
155 // w15: function_index
156 Register array_start = x12;
157 Register function_index = x15;
158
159 // Set up the stackframe:
160 //
161 // fp-0x10 wasm_instance
162 // fp-0x08 Marker(StackFrame::WASM_INTERPRETER_ENTRY)
163 // fp Old RBP
164 __ EnterFrame(StackFrame::WASM_INTERPRETER_ENTRY);
165
166 __ Str(kWasmImplicitArgRegister, MemOperand(sp, 0));
167 __ Push(function_index, array_start);
168 __ Mov(kWasmImplicitArgRegister, xzr);
169 __ CallRuntime(Runtime::kWasmRunInterpreter, 3);
170
171 // Deconstruct the stack frame.
172 __ LeaveFrame(StackFrame::WASM_INTERPRETER_ENTRY);
173 __ Ret();
174}

243class RegisterAllocator {
244 public:
245 class Scoped {
246 public:
247 Scoped(RegisterAllocator* allocator, Register* reg)
248 : allocator_(allocator), reg_(reg) {}
249 ~Scoped() { allocator_->Free(reg_); }
250
251 private:
252 RegisterAllocator* allocator_;
253 Register* reg_;
254 };
255
256 explicit RegisterAllocator(const CPURegList& registers)
257 : initial_(registers), available_(registers) {}
258 void Ask(Register* reg) {
259 DCHECK_EQ(*reg, no_reg);
260 DCHECK(!available_.IsEmpty());
261 *reg = available_.PopLowestIndex().X();
262 allocated_registers_.push_back(reg);
263 }
264
265 void Pinned(const Register& requested, Register* reg) {
266 DCHECK(available_.IncludesAliasOf(requested));
267 *reg = requested;
268 Reserve(requested);
269 allocated_registers_.push_back(reg);
270 }
271
272 void Free(Register* reg) {
273 DCHECK_NE(*reg, no_reg);
274 available_.Combine(*reg);
275 *reg = no_reg;
276 allocated_registers_.erase(
277 find(allocated_registers_.begin(), allocated_registers_.end(), reg));
278 }
279
280 void Reserve(const Register& reg) {
281 if (reg == NoReg) {
282 return;
283 }
284 DCHECK(available_.IncludesAliasOf(reg));
285 available_.Remove(reg);
286 }
287
288 void Reserve(const Register& reg1, const Register& reg2,
289 const Register& reg3 = NoReg, const Register& reg4 = NoReg,
290 const Register& reg5 = NoReg, const Register& reg6 = NoReg) {
291 Reserve(reg1);
292 Reserve(reg2);
293 Reserve(reg3);
294 Reserve(reg4);
295 Reserve(reg5);
296 Reserve(reg6);
297 }
298
299 bool IsUsed(const Register& reg) {
300 return initial_.IncludesAliasOf(reg) && !available_.IncludesAliasOf(reg);
301 }
302
303 void ResetExcept(const Register& reg1 = NoReg, const Register& reg2 = NoReg,
304 const Register& reg3 = NoReg, const Register& reg4 = NoReg,
305 const Register& reg5 = NoReg, const Register& reg6 = NoReg,
306 const Register& reg7 = NoReg) {
307 available_ = initial_;
308 if (reg1 != NoReg) {
309 available_.Remove(reg1, reg2, reg3, reg4);
310 }
311 if (reg5 != NoReg) {
312 available_.Remove(reg5, reg6, reg7);
313 }
314 auto it = allocated_registers_.begin();
315 while (it != allocated_registers_.end()) {
316 if (available_.IncludesAliasOf(**it)) {
317 **it = no_reg;
318 allocated_registers_.erase(it);
319 } else {
320 it++;
321 }
322 }
323 }

325 static RegisterAllocator WithAllocatableGeneralRegisters() {
326 CPURegList list(kXRegSizeInBits, RegList());
327 // Only use registers x0-x15, which are volatile (caller-saved).
328 // Mksnapshot would fail to compile the GenericJSToWasmInterpreterWrapper
329 // and GenericWasmToJSInterpreterWrapper if they needed more registers.
330 list.set_bits(0xffff); // (The default value is 0x0bf8ffff).
331 return RegisterAllocator(list);
332 }
333
334 private:
335 std::vector<Register*> allocated_registers_;
336 const CPURegList initial_;
337 CPURegList available_;
338};

364// TODO(paolosev@microsoft.com): this should be converted into a Torque builtin,
365// like it was done for GenericJSToWasmWrapper.
366void Builtins::Generate_GenericJSToWasmInterpreterWrapper(
367 MacroAssembler* masm) {
368 auto regs = RegisterAllocator::WithAllocatableGeneralRegisters();
369 // Set up the stackframe.
370 __ EnterFrame(StackFrame::JS_TO_WASM);
371
372 // -------------------------------------------
373 // Compute offsets and prepare for GC.
374 // -------------------------------------------
375 // GenericJSToWasmInterpreterWrapperFrame:
376 // fp-N Args/retvals array for Wasm call
377 // ... ...
378 // fp-0x58 SignatureData
379 // fp-0x50 CurrentIndex
380 // fp-0x48 ArgRetsIsArgs
381 // fp-0x40 ArgRetsAddress
382 // fp-0x38 ValueTypesArray
383 // fp-0x30 SigReps
384 // fp-0x28 ReturnCount
385 // fp-0x20 ParamCount
386 // fp-0x18 InParamCount
387 // fp-0x10 GCScanSlotCount
388 // fp-0x08 Marker(StackFrame::JS_TO_WASM)
389 // fp Old RBP
390
391 constexpr int kMarkerOffset =
392 BuiltinWasmInterpreterWrapperConstants::kGCScanSlotCountOffset +
393 kSystemPointerSize;
394 // The number of parameters passed to this function.
395 constexpr int kInParamCountOffset =
396 BuiltinWasmInterpreterWrapperConstants::kGCScanSlotCountOffset -
397 kSystemPointerSize;
398 // The number of parameters according to the signature.
399 constexpr int kParamCountOffset =
400 BuiltinWasmInterpreterWrapperConstants::kParamCountOffset;
401 constexpr int kReturnCountOffset =
402 BuiltinWasmInterpreterWrapperConstants::kReturnCountOffset;
403 constexpr int kSigRepsOffset =
404 BuiltinWasmInterpreterWrapperConstants::kSigRepsOffset;
405 constexpr int kValueTypesArrayStartOffset =
406 BuiltinWasmInterpreterWrapperConstants::kValueTypesArrayStartOffset;
407 // Array for arguments and return values. They will be scanned by GC.
408 constexpr int kArgRetsAddressOffset =
409 BuiltinWasmInterpreterWrapperConstants::kArgRetsAddressOffset;
410 // Arg/Return arrays use the same stack address. So, we should keep a flag
411 // whether we are using the array for args or returns. (1 = Args, 0 = Rets)
412 constexpr int kArgRetsIsArgsOffset =
413 BuiltinWasmInterpreterWrapperConstants::kArgRetsIsArgsOffset;
414 // The index of the argument being converted.
415 constexpr int kCurrentIndexOffset =
416 BuiltinWasmInterpreterWrapperConstants::kCurrentIndexOffset;
417 // Precomputed signature data, a uint32_t with the format:
418 // bit 0-14: PackedArgsSize
419 // bit 15: HasRefArgs
420 // bit 16: HasRefRets
421 constexpr int kSignatureDataOffset =
422 BuiltinWasmInterpreterWrapperConstants::kSignatureDataOffset;
423 // We set and use this slot only when moving parameters into the parameter
424 // registers (so no GC scan is needed).
425 constexpr int kNumSpillSlots =
426 (kMarkerOffset - kSignatureDataOffset) / kSystemPointerSize;
427 constexpr int kNum16BytesAlignedSpillSlots = 2 * ((kNumSpillSlots + 1) / 2);
428
429 __ Sub(sp, sp, Immediate(kNum16BytesAlignedSpillSlots * kSystemPointerSize));
430 // Put the in_parameter count on the stack, we only need it at the very end
431 // when we pop the parameters off the stack.
432 __ Sub(kJavaScriptCallArgCountRegister, kJavaScriptCallArgCountRegister, 1);
433 __ Str(kJavaScriptCallArgCountRegister, MemOperand(fp, kInParamCountOffset));
434
435 // -------------------------------------------
436 // Load the Wasm exported function data and the Wasm instance.
437 // -------------------------------------------
438 DEFINE_PINNED(function_data, kJSFunctionRegister); // x1
439 DEFINE_PINNED(wasm_instance, kWasmImplicitArgRegister); // x7
440 LoadFunctionDataAndWasmInstance(masm, function_data, wasm_instance);
441
442 regs.ResetExcept(function_data, wasm_instance);
443
444 // -------------------------------------------
445 // Load values from the signature.
446 // -------------------------------------------
447
448 // Param should be x0 for calling Runtime in the conversion loop.
449 DEFINE_PINNED(param, x0);
450 // These registers stays alive until we load params to param registers.
451 // To prevent aliasing assign higher register here.
452 DEFINE_PINNED(valuetypes_array_ptr, x11);
453
454 DEFINE_REG(return_count);
455 DEFINE_REG(param_count);
456 DEFINE_REG(signature_data);
457 DEFINE_REG(scratch);
458
459 // -------------------------------------------
460 // Load values from the signature.
461 // -------------------------------------------
462 LoadValueTypesArray(masm, function_data, valuetypes_array_ptr, return_count,
463 param_count, signature_data);
464 __ Str(signature_data, MemOperand(fp, kSignatureDataOffset));
465 Register array_size = signature_data;
466 __ And(array_size, array_size,
467 Immediate(wasm::WasmInterpreterRuntime::PackedArgsSizeField::kMask));
468 // -------------------------------------------
469 // Store signature-related values to the stack.
470 // -------------------------------------------
471 // We store values on the stack to restore them after function calls.
472 // We cannot push values onto the stack right before the wasm call. The wasm
473 // function expects the parameters, that didn't fit into the registers, on the
474 // top of the stack.
475 __ Str(param_count, MemOperand(fp, kParamCountOffset));
476 __ Str(return_count, MemOperand(fp, kReturnCountOffset));
477 __ Str(valuetypes_array_ptr, MemOperand(fp, kSigRepsOffset));
478 __ Str(valuetypes_array_ptr, MemOperand(fp, kValueTypesArrayStartOffset));
479
480 // -------------------------------------------
481 // Allocate array for args and return value.
482 // -------------------------------------------
483
484 // Leave space for WasmInstance.
485 __ Add(array_size, array_size, Immediate(kSystemPointerSize));
486 // Ensure that the array is 16-bytes aligned.
487 __ Add(scratch, array_size, Immediate(8));
488 __ And(array_size, scratch, Immediate(-16));
489
490 DEFINE_PINNED(array_start, x12);
491 __ Sub(array_start, sp, array_size);
492 __ Mov(sp, array_start);
493
494 __ Mov(scratch, 1);
495 __ Str(scratch, MemOperand(fp, kArgRetsIsArgsOffset));
496
497 __ Str(xzr, MemOperand(fp, kCurrentIndexOffset));
498
499 // Set the current_param_slot to point to the start of the section, after the
500 // WasmInstance object.
501 DEFINE_PINNED(current_param_slot, x13);
502 __ Add(current_param_slot, array_start, Immediate(kSystemPointerSize));
503 __ Str(current_param_slot, MemOperand(fp, kArgRetsAddressOffset));
504
505 Label prepare_for_wasm_call;
506 __ Cmp(param_count, 0);
507
508 // IF we have 0 params: jump through parameter handling.
509 __ B(&prepare_for_wasm_call, eq);
510
511 // Create a section on the stack to pass the evaluated parameters to the
512 // interpreter and to receive the results. This section represents the array
513 // expected as argument by the Runtime_WasmRunInterpreter.
514 // Arguments are stored one after the other without holes, starting at the
515 // beginning of the array, and the interpreter puts the returned values in the
516 // same array, also starting at the beginning.
517
518 // Loop through the params starting with the first.
519 // 'fp + kFPOnStackSize + kPCOnStackSize + kReceiverOnStackSize' points to the
520 // first JS parameter we are processing.
521
522 // We have to check the types of the params. The ValueType array contains
523 // first the return then the param types.
524
525 // Set the ValueType array pointer to point to the first parameter.
526 constexpr int kValueTypeSize = sizeof(wasm::ValueType);
527 static_assert(kValueTypeSize == 4);
528 const int32_t kValueTypeSizeLog2 = log2(kValueTypeSize);
529 // Set the ValueType array pointer to point to the first parameter.
530 __ Add(valuetypes_array_ptr, valuetypes_array_ptr,
531 Operand(return_count, LSL, kValueTypeSizeLog2));
532
533 DEFINE_REG(current_index);
534 __ Mov(current_index, xzr);
535
536 // -------------------------------------------
537 // Param evaluation loop.
538 // -------------------------------------------
539 Label loop_through_params;
540 __ bind(&loop_through_params);
541
542 constexpr int kReceiverOnStackSize = kSystemPointerSize;
543 constexpr int kArgsOffset =
544 kFPOnStackSize + kPCOnStackSize + kReceiverOnStackSize;
545 // Read JS argument into 'param'.
546 __ Add(scratch, fp, kArgsOffset);
547 __ Ldr(param,
548 MemOperand(scratch, current_index, LSL, kSystemPointerSizeLog2));
549 __ Str(current_index, MemOperand(fp, kCurrentIndexOffset));
550
551 DEFINE_REG_W(valuetype);
552 __ Ldr(valuetype,
553 MemOperand(valuetypes_array_ptr, wasm::ValueType::bit_field_offset()));
554
555 // -------------------------------------------
556 // Param conversion.
557 // -------------------------------------------
558 // If param is a Smi we can easily convert it. Otherwise we'll call a builtin
559 // for conversion.
560 Label param_conversion_done;
561 Label check_ref_param;
562 Label convert_param;
563 __ cmp(valuetype, Immediate(wasm::kWasmI32.raw_bit_field()));
564 __ B(&check_ref_param, ne);
565 __ JumpIfNotSmi(param, &convert_param);
566
567 // Change the param from Smi to int32.
568 __ SmiUntag(param);
569 // Place the param into the proper slot in Integer section.
570 __ Str(param, MemOperand(current_param_slot, 0));
571 __ Add(current_param_slot, current_param_slot, Immediate(sizeof(int32_t)));
572 __ jmp(&param_conversion_done);
573
574 Label handle_ref_param;
575 __ bind(&check_ref_param);
576
577 // wasm::ValueKind::kRefNull is not representable as a cmp immediate operand.
578 __ And(valuetype, valuetype, Immediate(wasm::kWasmValueKindBitsMask));
579 __ cmp(valuetype, Immediate(wasm::ValueKind::kRefNull));
580 __ B(&handle_ref_param, eq);
581 __ cmp(valuetype, Immediate(wasm::ValueKind::kRef));
582 __ B(&convert_param, ne);
583
584 // Place the reference param into the proper slot.
585 __ bind(&handle_ref_param);
586 // Make sure slot for ref args are 64-bit aligned.
587 __ And(scratch, current_param_slot, Immediate(0x04));
588 __ Add(current_param_slot, current_param_slot, scratch);
589 __ Str(param, MemOperand(current_param_slot, 0));
590 __ Add(current_param_slot, current_param_slot, Immediate(kSystemPointerSize));
591
592 // -------------------------------------------
593 // Param conversion done.
594 // -------------------------------------------
595 __ bind(&param_conversion_done);
596
597 __ Add(valuetypes_array_ptr, valuetypes_array_ptr, kValueTypeSize);
598
599 __ Ldr(current_index, MemOperand(fp, kCurrentIndexOffset));
600 __ Ldr(scratch, MemOperand(fp, kParamCountOffset));
601 __ Add(current_index, current_index, 1);
602 __ cmp(current_index, scratch);
603 __ B(&loop_through_params, lt);
604 __ Str(current_index, MemOperand(fp, kCurrentIndexOffset));
605 __ jmp(&prepare_for_wasm_call);
606
607 // -------------------------------------------
608 // Param conversion builtins.
609 // -------------------------------------------
610 __ bind(&convert_param);
611 // The order of pushes is important. We want the heap objects, that should be
612 // scanned by GC, to be on the top of the stack.
613 // We have to set the indicating value for the GC to the number of values on
614 // the top of the stack that have to be scanned before calling the builtin
615 // function.
616 // We don't need the JS context for these builtin calls.
617 // The builtin expects the parameter to be in register param = rax.
618
619 PrepareForJsToWasmConversionBuiltinCall(masm, current_param_slot,
620 valuetypes_array_ptr, wasm_instance,
621 function_data);
622
623 Label param_kWasmI32_not_smi, param_kWasmI64, param_kWasmF32, param_kWasmF64,
624 throw_type_error;
625
626 __ cmp(valuetype, Immediate(wasm::kWasmI32.raw_bit_field()));
627 __ B(&param_kWasmI32_not_smi, eq);
628 __ cmp(valuetype, Immediate(wasm::kWasmI64.raw_bit_field()));

640 __ DebugBreak();

643 __ Call(BUILTIN_CODE(masm->isolate(), WasmTaggedNonSmiToInt32),
644 RelocInfo::CODE_TARGET);

646 RestoreAfterJsToWasmConversionBuiltinCall(masm, function_data, wasm_instance,
647 valuetypes_array_ptr,

680 __ jmp(&param_conversion_done);
681

684 __ CallRuntime(Runtime::kWasmThrowJSTypeError);
685 __ DebugBreak(); // Should not return.

688 // Prepare for the Wasm call.
689 // -------------------------------------------
690
691 regs.ResetExcept(function_data, wasm_instance, array_start, scratch);
692
693 __ bind(&prepare_for_wasm_call);
694
695 // Set thread_in_wasm_flag.
696 DEFINE_REG_W(scratch32);
697 __ Ldr(scratch, MemOperand(kRootRegister,
698 Isolate::thread_in_wasm_flag_address_offset()));
699 __ Mov(scratch32, 1); // 32 bit.
700 __ Str(scratch32, MemOperand(scratch, 0));
701
702 DEFINE_PINNED(function_index, w15);
703 __ Ldr(
704 function_index,
705 MemOperand(function_data, WasmExportedFunctionData::kFunctionIndexOffset -
706 kHeapObjectTag));
707 // We pass function_index as Smi.
708
709 // One tagged object (the wasm_instance) to be visited if there is a GC
710 // during the call.
711 constexpr int kWasmCallGCScanSlotCount = 1;
712 __ Mov(scratch, kWasmCallGCScanSlotCount);
713 __ Str(
714 scratch,
715 MemOperand(
716 fp, BuiltinWasmInterpreterWrapperConstants::kGCScanSlotCountOffset));
717
718 // -------------------------------------------
719 // Call the Wasm function.
720 // -------------------------------------------
721
722 // Here array_start == sp.
723 __ Str(wasm_instance, MemOperand(sp));
724 // Skip wasm_instance.
725 __ Ldr(array_start, MemOperand(fp, kArgRetsAddressOffset));
726 // Here array_start == sp + kSystemPointerSize.
727 __ Call(BUILTIN_CODE(masm->isolate(), WasmInterpreterEntry),

730 __ Ldr(array_start, MemOperand(fp, kArgRetsAddressOffset));
731
732 __ Str(xzr, MemOperand(fp, kArgRetsIsArgsOffset));
733
734 // Unset thread_in_wasm_flag.
735 __ Ldr(scratch, MemOperand(kRootRegister,
736 Isolate::thread_in_wasm_flag_address_offset()));

738
739 regs.ResetExcept(wasm_instance, array_start, scratch);
740
741 // -------------------------------------------
742 // Return handling.
743 // -------------------------------------------
744 DEFINE_PINNED(return_value, kReturnRegister0); // x0
745 ASSIGN_REG(return_count);
746 __ Ldr(return_count, MemOperand(fp, kReturnCountOffset));
747
748 // All return values are already in the packed array.
749 __ Str(return_count,
750 MemOperand(
751 fp, BuiltinWasmInterpreterWrapperConstants::kCurrentIndexOffset));
752
753 DEFINE_PINNED(fixed_array, x14);
754 __ Mov(fixed_array, xzr);
755 DEFINE_PINNED(jsarray, x15);
756 __ Mov(jsarray, xzr);
757
758 Label all_results_conversion_done, start_return_conversion, return_jsarray;
759
760 __ cmp(return_count, 1);
761 __ B(&start_return_conversion, eq);
762 __ B(&return_jsarray, gt);
763
764 // If no return value, load undefined.
765 __ LoadRoot(return_value, RootIndex::kUndefinedValue);
766 __ jmp(&all_results_conversion_done);
767
768 // If we have more than one return value, we need to return a JSArray.
769 __ bind(&return_jsarray);
770 PrepareForBuiltinCall(masm, array_start, return_count, wasm_instance);
771 __ Mov(return_value, return_count);
772 __ SmiTag(return_value);
773
774 // Create JSArray to hold results.
775 __ Call(BUILTIN_CODE(masm->isolate(), WasmAllocateJSArray),
776 RelocInfo::CODE_TARGET);
777 __ Mov(jsarray, return_value);
778
779 RestoreAfterBuiltinCall(masm, wasm_instance, return_count, array_start);
780 __ LoadTaggedField(fixed_array, MemOperand(jsarray, JSArray::kElementsOffset -
781 kHeapObjectTag));
782
783 __ bind(&start_return_conversion);
784 Register current_return_slot = array_start;
785
786 DEFINE_PINNED(result_index, x13);
787 __ Mov(result_index, xzr);
788
789 // -------------------------------------------
790 // Return conversions.
791 // -------------------------------------------
792 Label convert_return_value;
793 __ bind(&convert_return_value);
794 // We have to make sure that the kGCScanSlotCount is set correctly when we
795 // call the builtins for conversion. For these builtins it's the same as for
796 // the Wasm call, that is, kGCScanSlotCount = 0, so we don't have to reset it.
797 // We don't need the JS context for these builtin calls.
798
799 ASSIGN_REG(valuetypes_array_ptr);
800 __ Ldr(valuetypes_array_ptr, MemOperand(fp, kValueTypesArrayStartOffset));
801
802 // The first valuetype of the array is the return's valuetype.
803 ASSIGN_REG_W(valuetype);
804 __ Ldr(valuetype,
805 MemOperand(valuetypes_array_ptr, wasm::ValueType::bit_field_offset()));
806
807 Label return_kWasmI32, return_kWasmI64, return_kWasmF32, return_kWasmF64,
808 return_kWasmRef;
809
810 __ cmp(valuetype, Immediate(wasm::kWasmI32.raw_bit_field()));
811 __ B(&return_kWasmI32, eq);
812 __ cmp(valuetype, Immediate(wasm::kWasmI64.raw_bit_field()));
813 __ B(&return_kWasmI64, eq);
814 __ cmp(valuetype, Immediate(wasm::kWasmF32.raw_bit_field()));
815 __ B(&return_kWasmF32, eq);
816 __ cmp(valuetype, Immediate(wasm::kWasmF64.raw_bit_field()));
817 __ B(&return_kWasmF64, eq);
818
819 {
820 __ And(valuetype, valuetype, Immediate(wasm::kWasmValueKindBitsMask));
821 __ cmp(valuetype, Immediate(wasm::ValueKind::kRefNull));
822 __ B(&return_kWasmRef, eq);
823 __ cmp(valuetype, Immediate(wasm::ValueKind::kRef));
824 __ B(&return_kWasmRef, eq);
825
826 // Invalid type. Wasm cannot return Simd results to JavaScript.
827 __ DebugBreak();
828 }
829
830 Label return_value_done;
831
832 Label to_heapnumber;
833 {
834 __ bind(&return_kWasmI32);
835 __ Ldr(return_value, MemOperand(current_return_slot, 0));
836 __ Add(current_return_slot, current_return_slot,
837 Immediate(sizeof(int32_t)));
838 // If pointer compression is disabled, we can convert the return to a smi.
839 if (SmiValuesAre32Bits()) {
840 __ SmiTag(return_value);
841 } else {
842 // Double the return value to test if it can be a Smi.
843 __ Adds(wzr, return_value.W(), return_value.W());
844 // If there was overflow, convert the return value to a HeapNumber.
845 __ B(&to_heapnumber, vs);
846 // If there was no overflow, we can convert to Smi.
847 __ SmiTag(return_value);
848 }
849 }
850 __ jmp(&return_value_done);
851
852 // Handle the conversion of the I32 return value to HeapNumber when it cannot
853 // be a smi.
854 __ bind(&to_heapnumber);
855
856 PrepareForWasmToJsConversionBuiltinCall(
857 masm, return_count, result_index, current_return_slot,
858 valuetypes_array_ptr, wasm_instance, fixed_array, jsarray);
859 __ Call(BUILTIN_CODE(masm->isolate(), WasmInt32ToHeapNumber),
860 RelocInfo::CODE_TARGET);
861 RestoreAfterWasmToJsConversionBuiltinCall(
862 masm, jsarray, fixed_array, wasm_instance, valuetypes_array_ptr,
863 current_return_slot, result_index, return_count);
864 __ jmp(&return_value_done);
865
866 __ bind(&return_kWasmI64);
867 __ Ldr(return_value, MemOperand(current_return_slot, 0));
868 __ Add(current_return_slot, current_return_slot, Immediate(sizeof(int64_t)));
869 PrepareForWasmToJsConversionBuiltinCall(
870 masm, return_count, result_index, current_return_slot,
871 valuetypes_array_ptr, wasm_instance, fixed_array, jsarray);
872 __ Call(BUILTIN_CODE(masm->isolate(), I64ToBigInt), RelocInfo::CODE_TARGET);
873 RestoreAfterWasmToJsConversionBuiltinCall(
874 masm, jsarray, fixed_array, wasm_instance, valuetypes_array_ptr,
875 current_return_slot, result_index, return_count);
876 __ jmp(&return_value_done);
877
878 __ bind(&return_kWasmF32);
879 __ Ldr(v0, MemOperand(current_return_slot, 0));
880 __ Add(current_return_slot, current_return_slot, Immediate(sizeof(float)));
881 PrepareForWasmToJsConversionBuiltinCall(
882 masm, return_count, result_index, current_return_slot,
883 valuetypes_array_ptr, wasm_instance, fixed_array, jsarray);
884 __ Call(BUILTIN_CODE(masm->isolate(), WasmFloat32ToNumber),
885 RelocInfo::CODE_TARGET);
886 RestoreAfterWasmToJsConversionBuiltinCall(
887 masm, jsarray, fixed_array, wasm_instance, valuetypes_array_ptr,
888 current_return_slot, result_index, return_count);
889 __ jmp(&return_value_done);
890
891 __ bind(&return_kWasmF64);
892 __ Ldr(d0, MemOperand(current_return_slot, 0));
893 __ Add(current_return_slot, current_return_slot, Immediate(sizeof(double)));
894 PrepareForWasmToJsConversionBuiltinCall(
895 masm, return_count, result_index, current_return_slot,
896 valuetypes_array_ptr, wasm_instance, fixed_array, jsarray);
897 __ Call(BUILTIN_CODE(masm->isolate(), WasmFloat64ToNumber),
898 RelocInfo::CODE_TARGET);
899 RestoreAfterWasmToJsConversionBuiltinCall(
900 masm, jsarray, fixed_array, wasm_instance, valuetypes_array_ptr,
901 current_return_slot, result_index, return_count);
902 __ jmp(&return_value_done);
903
904 __ bind(&return_kWasmRef);
905 // Make sure slot for ref args are 64-bit aligned.
906 __ And(scratch, current_return_slot, Immediate(0x04));
907 __ Add(current_return_slot, current_return_slot, scratch);
908 __ Ldr(return_value, MemOperand(current_return_slot, 0));
909 __ Add(current_return_slot, current_return_slot,
910 Immediate(kSystemPointerSize));
911 // It might be cleaner to call Builtins_WasmFuncRefToJS here to extract
912 // func.external from the ref object if the type is kWasmFuncRef.
913
914 Label next_return_value;
915
916 __ bind(&return_value_done);
917 __ Add(valuetypes_array_ptr, valuetypes_array_ptr, Immediate(kValueTypeSize));
918 __ Str(valuetypes_array_ptr, MemOperand(fp, kValueTypesArrayStartOffset));
919 __ cmp(fixed_array, xzr);
920 __ B(&next_return_value, eq);
921
922 // Store result in JSArray
923 DEFINE_REG(array_items);
924 __ Add(array_items, fixed_array,
925 OFFSET_OF_DATA_START(FixedArray) - kHeapObjectTag);
926 __ StoreTaggedField(return_value, MemOperand(array_items, result_index, LSL,
927 kTaggedSizeLog2));
928
929 __ bind(&next_return_value);
930 __ Add(result_index, result_index, 1);
931 __ cmp(result_index, return_count);
932 __ B(&convert_return_value, lt);
933
934 __ bind(&all_results_conversion_done);
935 ASSIGN_REG(param_count);
936 __ Ldr(param_count, MemOperand(fp, kParamCountOffset)); // ???
937
938 Label do_return;
939 __ cmp(fixed_array, xzr);
940 __ B(&do_return, eq);
941 // The result is jsarray.
942 __ Mov(return_value, jsarray);
943
944 __ bind(&do_return);
945 // Calculate the number of parameters we have to pop off the stack. This
946 // number is max(in_param_count, param_count).
947 DEFINE_REG(in_param_count);
948 __ Ldr(in_param_count, MemOperand(fp, kInParamCountOffset));
949 __ cmp(param_count, in_param_count);
950 __ csel(param_count, in_param_count, param_count, lt);
951
952 // -------------------------------------------
953 // Deconstruct the stack frame.
954 // -------------------------------------------
955 __ LeaveFrame(StackFrame::JS_TO_WASM);
956
957 // We have to remove the caller frame slots:
958 // - JS arguments
959 // - the receiver
960 // and transfer the control to the return address (the return address is
961 // expected to be on the top of the stack).
962 // We cannot use just the ret instruction for this, because we cannot pass the
963 // number of slots to remove in a Register as an argument.
964 __ DropArguments(param_count);
965 __ Ret(lr);
966}

968void Builtins::Generate_WasmInterpreterCWasmEntry(MacroAssembler* masm) {
969 Label invoke, handler_entry, exit;
970
971 __ EnterFrame(StackFrame::C_WASM_ENTRY);
972
973 // Space to store c_entry_fp and current sp (used by exception handler).
974 __ Push(xzr, xzr);
975
976 {
977 NoRootArrayScope no_root_array(masm);
978
979 // Push callee saved registers.
980 __ Push(d14, d15);
981 __ Push(d12, d13);
982 __ Push(d10, d11);
983 __ Push(d8, d9);
984 __ Push(x27, x28);
985 __ Push(x25, x26);
986 __ Push(x23, x24);
987 __ Push(x21, x22);
988 __ Push(x19, x20);
989
990 // Set up the reserved register for 0.0.
991 __ Fmov(fp_zero, 0.0);
992
993 // Initialize the root register.
994 __ Mov(kRootRegister, x2);
995
996#ifdef V8_COMPRESS_POINTERS_IN_SHARED_CAGE
997 // Initialize the pointer cage base register.
998 __ LoadRootRelative(kPtrComprCageBaseRegister,
999 IsolateData::cage_base_offset());
1000#endif
1001 }
1002
1003 {
1004 UseScratchRegisterScope temps(masm);
1005 Register scratch = temps.AcquireX();
1006 __ Mov(scratch, sp);
1007 __ Str(scratch,
1008 MemOperand(fp, WasmInterpreterCWasmEntryConstants::kSPFPOffset));
1009 }
1010
1011 __ Mov(x11, ExternalReference::Create(IsolateAddressId::kCEntryFPAddress,
1012 masm->isolate()));
1013 __ Ldr(x10, MemOperand(x11)); // x10 = C entry FP.
1014
1015 __ Str(x10,
1016 MemOperand(fp, WasmInterpreterCWasmEntryConstants::kCEntryFPOffset));
1017
1018 // Jump to a faked try block that does the invoke, with a faked catch
1019 // block that sets the pending exception.
1020 __ B(&invoke);
1021
1022 // Prevent the constant pool from being emitted between the record of the
1023 // handler_entry position and the first instruction of the sequence here.
1024 // There is no risk because Assembler::Emit() emits the instruction before
1025 // checking for constant pool emission, but we do not want to depend on
1026 // that.
1027 {
1028 Assembler::BlockPoolsScope block_pools(masm);
1029
1030 __ BindExceptionHandler(&handler_entry);
1031
1032 // Store the current pc as the handler offset. It's used later to create the
1033 // handler table.
1034 masm->isolate()->builtins()->SetCWasmInterpreterEntryHandlerOffset(
1035 handler_entry.pos());
1036 }
1037 __ B(&exit);
1038
1039 // Invoke: Link this frame into the handler chain.
1040 __ Bind(&invoke);
1041
1042 // Link the current handler as the next handler.
1043 __ Mov(x11, ExternalReference::Create(IsolateAddressId::kHandlerAddress,
1044 masm->isolate()));
1045 __ Ldr(x10, MemOperand(x11));
1046 __ Push(padreg, x10);
1047
1048 // Set this new handler as the current one.
1049 {
1050 UseScratchRegisterScope temps(masm);
1051 Register scratch = temps.AcquireX();
1052 __ Mov(scratch, sp);
1053 __ Str(scratch, MemOperand(x11));
1054 }
1055
1056 // Invoke the JS function through the GenericWasmToJSInterpreterWrapper.
1057 __ Call(BUILTIN_CODE(masm->isolate(), GenericWasmToJSInterpreterWrapper),
1058 RelocInfo::CODE_TARGET);
1059
1060 // Pop the stack handler and unlink this frame from the handler chain.
1061 static_assert(StackHandlerConstants::kNextOffset == 0 * kSystemPointerSize,
1062 "Unexpected offset for StackHandlerConstants::kNextOffset");
1063 __ Pop(x10, padreg);
1064 __ Mov(x11, ExternalReference::Create(IsolateAddressId::kHandlerAddress,
1065 masm->isolate()));
1066 __ Drop(StackHandlerConstants::kSlotCount - 2);
1067 __ Str(x10, MemOperand(x11));
1068
1069 __ Bind(&exit);
1070
1071 // Pop callee saved registers.
1072 __ Pop(x20, x19);
1073 __ Pop(x22, x21);
1074 __ Pop(x24, x23);
1075 __ Pop(x26, x25);
1076 __ Pop(x28, x27);
1077 __ Pop(d9, d8);
1078 __ Pop(d11, d10);
1079 __ Pop(d13, d12);
1080 __ Pop(d15, d14);
1081
1082 // Deconstruct the stack frame.
1083 __ LeaveFrame(StackFrame::C_WASM_ENTRY);
1084 __ Ret();
1085}

1087void Builtins::Generate_GenericWasmToJSInterpreterWrapper(
1088 MacroAssembler* masm) {
1089 auto regs = RegisterAllocator::WithAllocatableGeneralRegisters();
1090
1091 DEFINE_PINNED(target_js_function, x0);
1092 DEFINE_PINNED(packed_args, x1);
1093 DEFINE_PINNED(signature, x3);
1094 DEFINE_PINNED(callable, x5);
1095
1096 // Set up the stackframe.
1097 __ EnterFrame(StackFrame::WASM_TO_JS);
1098
1099 // -------------------------------------------
1100 // Compute offsets and prepare for GC.
1101 // -------------------------------------------
1102 // GenericJSToWasmInterpreterWrapperFrame:
1103 // sp = fp-N receiver ^
1104 // ... JS arg 0 |
1105 // ... ... | Tagged
1106 // ... JS arg n-1 | objects
1107 // ... (padding if num args is odd) |
1108 // ... context |
1109 // fp-0x58 callable v
1110 // -------------------------------------------
1111 // fp-0x50 current_param_offset/current_result_offset
1112 // fp-0x48 valuetypes_array_ptr
1113 //
1114 // fp-0x40 param_index/return_index
1115 // fp-0x38 signature
1116 //
1117 // fp-0x30 param_count
1118 // fp-0x28 return_count
1119 //
1120 // fp-0x20 packed_array
1121 // fp-0x18 GC_SP (not used)
1122 //
1123 // fp-0x10 GCScanSlotLimit
1124 // fp-0x08 Marker(StackFrame::WASM_TO_JS)
1125 //
1126 // fp Old fp
1127 // fp+0x08 return address
1128
1129 static_assert(WasmToJSInterpreterFrameConstants::kGCSPOffset ==
1130 WasmToJSInterpreterFrameConstants::kGCScanSlotLimitOffset -
1131 kSystemPointerSize);
1132 constexpr int kPackedArrayOffset =
1133 WasmToJSInterpreterFrameConstants::kGCSPOffset - kSystemPointerSize;
1134 constexpr int kReturnCountOffset = kPackedArrayOffset - kSystemPointerSize;
1135 constexpr int kParamCountOffset = kReturnCountOffset - kSystemPointerSize;
1136 constexpr int kSignatureOffset = kParamCountOffset - kSystemPointerSize;
1137 constexpr int kParamIndexOffset = kSignatureOffset - kSystemPointerSize;
1138 // Reuse this slot when iterating over return values.
1139 constexpr int kResultIndexOffset = kParamIndexOffset;
1140 constexpr int kValueTypesArrayStartOffset =
1141 kParamIndexOffset - kSystemPointerSize;
1142 constexpr int kCurrentParamOffset =
1143 kValueTypesArrayStartOffset - kSystemPointerSize;
1144 // Reuse this slot when iterating over return values.
1145 constexpr int kCurrentResultOffset = kCurrentParamOffset;
1146 constexpr int kNumSpillSlots =
1147 (WasmToJSInterpreterFrameConstants::kGCScanSlotLimitOffset -
1148 kCurrentParamOffset) /
1149 kSystemPointerSize;
1150 static_assert((kNumSpillSlots % 2) == 0); // 16-bytes aligned.
1151
1152 constexpr int kCallableOffset = kCurrentParamOffset - kSystemPointerSize;
1153
1154 __ Sub(sp, sp, Immediate(kNumSpillSlots * kSystemPointerSize));
1155
1156 __ Str(packed_args, MemOperand(fp, kPackedArrayOffset));
1157
1158 // Not used
1159 __ Str(xzr, MemOperand(fp, WasmToJSInterpreterFrameConstants::kGCSPOffset));
1160
1161 __ Str(xzr,
1162 MemOperand(fp,
1163 WasmToJSInterpreterFrameConstants::kGCScanSlotLimitOffset));
1164
1165 DEFINE_REG(shared_function_info);
1166 __ LoadTaggedField(
1167 shared_function_info,
1168 MemOperand(
1169 target_js_function,
1170 wasm::ObjectAccess::SharedFunctionInfoOffsetInTaggedJSFunction()));
1171
1172 // Set the context of the function; the call has to run in the function
1173 // context.
1174 DEFINE_REG(context);
1175 __ LoadTaggedField(
1176 context, FieldMemOperand(target_js_function, JSFunction::kContextOffset));
1177 __ Mov(cp, context);
1178
1179 // Load global receiver if sloppy else use undefined.
1180 Label receiver_undefined;
1181 Label calculate_js_function_arity;
1182 DEFINE_REG(receiver);
1183 DEFINE_REG(flags);
1184 __ Ldr(flags, FieldMemOperand(shared_function_info,
1185 SharedFunctionInfo::kFlagsOffset));
1186 __ Tst(flags, Immediate(SharedFunctionInfo::IsNativeBit::kMask |
1187 SharedFunctionInfo::IsStrictBit::kMask));
1188 FREE_REG(flags);
1189 __ B(&receiver_undefined, ne);
1190 __ LoadGlobalProxy(receiver);
1191 __ B(&calculate_js_function_arity);
1192
1193 __ bind(&receiver_undefined);
1194 __ LoadRoot(receiver, RootIndex::kUndefinedValue);
1195
1196 __ bind(&calculate_js_function_arity);
1197
1198 // Load values from the signature.
1199 DEFINE_REG(return_count);
1200 DEFINE_REG(param_count);
1201 __ Str(signature, MemOperand(fp, kSignatureOffset));
1202 Register valuetypes_array_ptr = signature;
1203 LoadFromSignature(masm, valuetypes_array_ptr, return_count, param_count);
1204 __ Str(param_count, MemOperand(fp, kParamCountOffset));
1205 FREE_REG(shared_function_info);
1206
1207 // Make room to pass the args and the receiver.
1208 DEFINE_REG(array_size);
1209 DEFINE_REG(scratch);
1210
1211 // Points to the end of the stack frame area that contains tagged objects
1212 // that need to be visited during GC.
1213 __ Mov(scratch, sp);
1214 __ Str(scratch,
1215 MemOperand(fp,
1216 WasmToJSInterpreterFrameConstants::kGCScanSlotLimitOffset));
