42// Tries to allocate a BackingStore given the input configuration. Either
43// returns the BackingStore or a message template that should be thrown as
44// RangeError.
45std::pair<std::unique_ptr<BackingStore>, std::optional<MessageTemplate>>
46TryAllocateBackingStore(Isolate* isolate, SharedFlag shared,
47 ResizableFlag resizable, DirectHandle<Object> length,
48 DirectHandle<Object> max_length,
49 InitializedFlag initialized) {
50 DisallowJavascriptExecution no_js(isolate);
51
52 size_t byte_length;
53 size_t max_byte_length = 0;
54 std::unique_ptr<BackingStore> backing_store;
55
56 if (!TryNumberToSize(*length, &byte_length) ||
57 byte_length > JSArrayBuffer::kMaxByteLength) {
58 return {nullptr, MessageTemplate::kInvalidArrayBufferLength};
59 }
60
61 switch (resizable) {
62 case ResizableFlag::kNotResizable:
63 backing_store =
64 BackingStore::Allocate(isolate, byte_length, shared, initialized);
65 break;
66 case ResizableFlag::kResizable:
67 static_assert(JSArrayBuffer::kMaxByteLength ==
68 JSTypedArray::kMaxByteLength);
69 if (!TryNumberToSize(*max_length, &max_byte_length) ||
70 max_byte_length > JSArrayBuffer::kMaxByteLength) {
71 return {nullptr, MessageTemplate::kInvalidArrayBufferMaxLength};
72 }
73 if (byte_length > max_byte_length) {
74 return {nullptr, MessageTemplate::kInvalidArrayBufferMaxLength};
75 }
76
77 size_t page_size, initial_pages, max_pages;
78 const auto maybe_range_error =
79 JSArrayBuffer::GetResizableBackingStorePageConfigurationImpl(
80 isolate, byte_length, max_byte_length, &page_size, &initial_pages,
81 &max_pages);
82 if (maybe_range_error.has_value()) {
83 return {nullptr, maybe_range_error.value()};
84 }
85 backing_store = BackingStore::TryAllocateAndPartiallyCommitMemory(
86 isolate, byte_length, max_byte_length, page_size, initial_pages,
87 max_pages, WasmMemoryFlag::kNotWasm, shared);
88 break;
89 }
90
91 // Range errors bailed out earlier; only the failing allocation needs to be
92 // caught here.
93 if (!backing_store) {
94 return {nullptr, MessageTemplate::kArrayBufferAllocationFailed};
95 }
96 return {std::move(backing_store), std::nullopt};
97}

99Tagged<Object> ConstructBuffer(Isolate* isolate,
100 DirectHandle<JSFunction> target,
101 DirectHandle<JSReceiver> new_target,
102 DirectHandle<Object> length,
103 DirectHandle<Object> max_length,
104 InitializedFlag initialized) {
105 // We first try to convert the sizes and collect any possible range errors. If
106 // no errors are observable we create the BackingStore before the
107 // JSArrayBuffer to avoid a complex dance during setup. We then always create
108 // the AB before throwing a possible error as the creation is observable.
109 const SharedFlag shared =
110 *target != target->native_context()->array_buffer_fun()
111 ? SharedFlag::kShared
112 : SharedFlag::kNotShared;
113 const ResizableFlag resizable = max_length.is_null()
114 ? ResizableFlag::kNotResizable
115 : ResizableFlag::kResizable;
116 // BackingStore allocation may GC which is not observable itself.
117 auto [backing_store, range_error] = TryAllocateBackingStore(
118 isolate, shared, resizable, length, max_length, initialized);
119 DirectHandle<JSObject> result;
120 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
121 isolate, result,
122 JSObject::New(target, new_target, {}, NewJSObjectType::kAPIWrapper));
123 auto array_buffer = Cast<JSArrayBuffer>(result);
124 const bool backing_store_creation_failed = !backing_store;
125 array_buffer->Setup(shared, resizable, std::move(backing_store), isolate);
126 if (backing_store_creation_failed) {
127 CHECK(range_error.has_value());
128 THROW_NEW_ERROR_RETURN_FAILURE(isolate, NewRangeError(range_error.value()));
129 }
130 return *array_buffer;
131}

136BUILTIN(ArrayBufferConstructor) {
137 HandleScope scope(isolate);
138 DirectHandle<JSFunction> target = args.target();
139 DCHECK(*target == target->native_context()->array_buffer_fun() ||
140 *target == target->native_context()->shared_array_buffer_fun());
141 if (IsUndefined(*args.new_target(), isolate)) { // [[Call]]
142 THROW_NEW_ERROR_RETURN_FAILURE(
143 isolate,
144 NewTypeError(MessageTemplate::kConstructorNotFunction,
145 direct_handle(target->shared()->Name(), isolate)));
146 }
147 // [[Construct]]
148 DirectHandle<JSReceiver> new_target = Cast<JSReceiver>(args.new_target());
149 DirectHandle<Object> length = args.atOrUndefined(isolate, 1);
150
151 DirectHandle<Object> number_length;
152 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, number_length,
153 Object::ToInteger(isolate, length));
154 if (Object::NumberValue(*number_length) < 0.0) {
155 THROW_NEW_ERROR_RETURN_FAILURE(
156 isolate, NewRangeError(MessageTemplate::kInvalidArrayBufferLength));
157 }
158
159 DirectHandle<Object> number_max_length;
160 DirectHandle<Object> max_length;
161 DirectHandle<Object> options = args.atOrUndefined(isolate, 2);
162 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
163 isolate, max_length,
164 JSObject::ReadFromOptionsBag(
165 options, isolate->factory()->max_byte_length_string(), isolate));
166
167 if (!IsUndefined(*max_length, isolate)) {
168 if (*target == target->native_context()->array_buffer_fun()) {
169 isolate->CountUsage(
170 v8::Isolate::UseCounterFeature::kResizableArrayBuffer);
171 } else {
172 isolate->CountUsage(
173 v8::Isolate::UseCounterFeature::kGrowableSharedArrayBuffer);
174 }
175 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, number_max_length,
176 Object::ToInteger(isolate, max_length));
177 }
178 return ConstructBuffer(isolate, target, new_target, number_length,
179 number_max_length, InitializedFlag::kZeroInitialized);
180}

194static Tagged<Object> SliceHelper(BuiltinArguments args, Isolate* isolate,
195 const char* kMethodName, bool is_shared) {
196 HandleScope scope(isolate);
197 DirectHandle<Object> start = args.at(1);
198 DirectHandle<Object> end = args.atOrUndefined(isolate, 2);
199
200 // * If Type(O) is not Object, throw a TypeError exception.
201 // * If O does not have an [[ArrayBufferData]] internal slot, throw a
202 // TypeError exception.
203 CHECK_RECEIVER(JSArrayBuffer, array_buffer, kMethodName);
204 // * [AB] If IsSharedArrayBuffer(O) is true, throw a TypeError exception.
205 // * [SAB] If IsSharedArrayBuffer(O) is false, throw a TypeError exception.
206 CHECK_SHARED(is_shared, array_buffer, kMethodName);
207
208 // * [AB] If IsDetachedBuffer(buffer) is true, throw a TypeError exception.
209 if (!is_shared && array_buffer->was_detached()) {
210 THROW_NEW_ERROR_RETURN_FAILURE(
211 isolate, NewTypeError(MessageTemplate::kDetachedOperation,
212 isolate->factory()->NewStringFromAsciiChecked(
213 kMethodName)));
214 }
215
216 // * [AB] Let len be O.[[ArrayBufferByteLength]].
217 // * [SAB] Let len be O.[[ArrayBufferByteLength]].
218 double const len = array_buffer->GetByteLength();
219
220 // * Let relativeStart be ? ToInteger(start).
221 DirectHandle<Object> relative_start;
222 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, relative_start,
223 Object::ToInteger(isolate, start));
224
225 // * If relativeStart < 0, let first be max((len + relativeStart), 0); else
226 // let first be min(relativeStart, len).
227 double const first =
228 (Object::NumberValue(*relative_start) < 0)
229 ? std::max(len + Object::NumberValue(*relative_start), 0.0)
230 : std::min(Object::NumberValue(*relative_start), len);
231
232 // * If end is undefined, let relativeEnd be len; else let relativeEnd be ?
233 // ToInteger(end).
234 double relative_end;
235 if (IsUndefined(*end, isolate)) {
236 relative_end = len;
237 } else {
238 DirectHandle<Object> relative_end_obj;
239 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, relative_end_obj,
240 Object::ToInteger(isolate, end));
241 relative_end = Object::NumberValue(*relative_end_obj);
242 }
243
244 // * If relativeEnd < 0, let final be max((len + relativeEnd), 0); else let
245 // final be min(relativeEnd, len).
246 double const final_ = (relative_end < 0) ? std::max(len + relative_end, 0.0)
247 : std::min(relative_end, len);
248
249 // * Let newLen be max(final-first, 0).
250 double const new_len = std::max(final_ - first, 0.0);
251 DirectHandle<Object> new_len_obj = isolate->factory()->NewNumber(new_len);
252
253 // * [AB] Let ctor be ? SpeciesConstructor(O, %ArrayBuffer%).
254 // * [SAB] Let ctor be ? SpeciesConstructor(O, %SharedArrayBuffer%).
255 Handle<JSFunction> constructor_fun = is_shared
256 ? isolate->shared_array_buffer_fun()
257 : isolate->array_buffer_fun();
258 DirectHandle<Object> ctor;
259 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
260 isolate, ctor,
261 Object::SpeciesConstructor(isolate, Cast<JSReceiver>(args.receiver()),
262 constructor_fun));
263
264 // * Let new be ? Construct(ctor, newLen).
265 DirectHandle<JSReceiver> new_;
266 {
267 constexpr int argc = 1;
268 std::array<DirectHandle<Object>, argc> args = {new_len_obj};
269
270 DirectHandle<Object> new_obj;
271 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
272 isolate, new_obj, Execution::New(isolate, ctor, base::VectorOf(args)));
273
274 new_ = Cast<JSReceiver>(new_obj);
275 }
276
277 // * If new does not have an [[ArrayBufferData]] internal slot, throw a
278 // TypeError exception.
279 if (!IsJSArrayBuffer(*new_)) {
280 THROW_NEW_ERROR_RETURN_FAILURE(
281 isolate,
282 NewTypeError(MessageTemplate::kIncompatibleMethodReceiver,
283 isolate->factory()->NewStringFromAsciiChecked(kMethodName),
284 new_));
285 }
286
287 // * [AB] If IsSharedArrayBuffer(new) is true, throw a TypeError exception.
288 // * [SAB] If IsSharedArrayBuffer(new) is false, throw a TypeError exception.
289 DirectHandle<JSArrayBuffer> new_array_buffer = Cast<JSArrayBuffer>(new_);
290 CHECK_SHARED(is_shared, new_array_buffer, kMethodName);
291
292 // The created ArrayBuffer might or might not be resizable, since the species
293 // constructor might return a non-resizable or a resizable buffer.
294
295 // * [AB] If IsDetachedBuffer(new) is true, throw a TypeError exception.
296 if (!is_shared && new_array_buffer->was_detached()) {
297 THROW_NEW_ERROR_RETURN_FAILURE(
298 isolate, NewTypeError(MessageTemplate::kDetachedOperation,
299 isolate->factory()->NewStringFromAsciiChecked(
300 kMethodName)));
301 }
302
303 // * [AB] If SameValue(new, O) is true, throw a TypeError exception.
304 if (!is_shared && Object::SameValue(*new_, *args.receiver())) {
305 THROW_NEW_ERROR_RETURN_FAILURE(
306 isolate, NewTypeError(MessageTemplate::kArrayBufferSpeciesThis));
307 }
308
309 // * [SAB] If new.[[ArrayBufferData]] and O.[[ArrayBufferData]] are the same
310 // Shared Data Block values, throw a TypeError exception.
311 if (is_shared &&
312 new_array_buffer->backing_store() == array_buffer->backing_store()) {
313 THROW_NEW_ERROR_RETURN_FAILURE(
314 isolate, NewTypeError(MessageTemplate::kSharedArrayBufferSpeciesThis));
315 }
316
317 // * If new.[[ArrayBufferByteLength]] < newLen, throw a TypeError exception.
318 size_t new_array_buffer_byte_length = new_array_buffer->GetByteLength();
319 if (new_array_buffer_byte_length < new_len) {
320 THROW_NEW_ERROR_RETURN_FAILURE(
321 isolate,
322 NewTypeError(is_shared ? MessageTemplate::kSharedArrayBufferTooShort
323 : MessageTemplate::kArrayBufferTooShort));
324 }
325
326 // * [AB] NOTE: Side-effects of the above steps may have detached O.
327 // * [AB] If IsDetachedBuffer(O) is true, throw a TypeError exception.
328 if (!is_shared && array_buffer->was_detached()) {
329 THROW_NEW_ERROR_RETURN_FAILURE(
330 isolate, NewTypeError(MessageTemplate::kDetachedOperation,
331 isolate->factory()->NewStringFromAsciiChecked(
332 kMethodName)));
333 }
334
335 // * Let fromBuf be O.[[ArrayBufferData]].
336 // * Let toBuf be new.[[ArrayBufferData]].
337 // * Perform CopyDataBlockBytes(toBuf, 0, fromBuf, first, newLen).
338 size_t first_size = first;
339 size_t new_len_size = new_len;
340 DCHECK(new_array_buffer_byte_length >= new_len_size);
341
342 if (new_len_size != 0) {
343 size_t from_byte_length = array_buffer->GetByteLength();
344 if (V8_UNLIKELY(!is_shared && array_buffer->is_resizable_by_js())) {
345 // The above steps might have resized the underlying buffer. In that case,
346 // only copy the still-accessible portion of the underlying data.
347 if (first_size > from_byte_length) {
348 return *new_; // Nothing to copy.
349 }
350 if (new_len_size > from_byte_length - first_size) {
351 new_len_size = from_byte_length - first_size;
352 }
353 }
354 DCHECK(first_size <= from_byte_length);
355 DCHECK(from_byte_length - first_size >= new_len_size);
356 uint8_t* from_data =
357 reinterpret_cast<uint8_t*>(array_buffer->backing_store()) + first_size;
358 uint8_t* to_data =
359 reinterpret_cast<uint8_t*>(new_array_buffer->backing_store());
360 if (is_shared) {
361 base::Relaxed_Memcpy(reinterpret_cast<base::Atomic8*>(to_data),
362 reinterpret_cast<base::Atomic8*>(from_data),
363 new_len_size);
364 } else {
365 CopyBytes(to_data, from_data, new_len_size);
366 }
367 }
368
369 return *new_;
370}

385static Tagged<Object> ResizeHelper(BuiltinArguments args, Isolate* isolate,
386 const char* kMethodName, bool is_shared) {
387 HandleScope scope(isolate);
388
389 // 1 Let O be the this value.
390 // 2. Perform ? RequireInternalSlot(O, [[ArrayBufferMaxByteLength]]).
391 CHECK_RECEIVER(JSArrayBuffer, array_buffer, kMethodName);
392 CHECK_RESIZABLE(true, array_buffer, kMethodName);
393
394 // [RAB] 3. If IsSharedArrayBuffer(O) is true, throw a *TypeError* exception
395 // [GSAB] 3. If IsSharedArrayBuffer(O) is false, throw a *TypeError* exception
396 CHECK_SHARED(is_shared, array_buffer, kMethodName);
397
398 // Let newByteLength to ? ToIntegerOrInfinity(newLength).
399 DirectHandle<Object> new_length = args.at(1);
400 DirectHandle<Object> number_new_byte_length;
401 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, number_new_byte_length,
402 Object::ToInteger(isolate, new_length));
403
404 // [RAB] If IsDetachedBuffer(O) is true, throw a TypeError exception.
405 if (!is_shared && array_buffer->was_detached()) {
406 THROW_NEW_ERROR_RETURN_FAILURE(
407 isolate, NewTypeError(MessageTemplate::kDetachedOperation,
408 isolate->factory()->NewStringFromAsciiChecked(
409 kMethodName)));
410 }
411
412 // [RAB] If newByteLength < 0 or newByteLength >
413 // O.[[ArrayBufferMaxByteLength]], throw a RangeError exception.
414
415 // [GSAB] If newByteLength < currentByteLength or newByteLength >
416 // O.[[ArrayBufferMaxByteLength]], throw a RangeError exception.
417 size_t new_byte_length;
418 if (!TryNumberToSize(*number_new_byte_length, &new_byte_length)) {
419 THROW_NEW_ERROR_RETURN_FAILURE(
420 isolate, NewRangeError(MessageTemplate::kInvalidArrayBufferResizeLength,
421 isolate->factory()->NewStringFromAsciiChecked(
422 kMethodName)));
423 }
424
425 if (is_shared && new_byte_length < array_buffer->byte_length()) {
426 // GrowableSharedArrayBuffer is only allowed to grow.
427 THROW_NEW_ERROR_RETURN_FAILURE(
428 isolate, NewRangeError(MessageTemplate::kInvalidArrayBufferResizeLength,
429 isolate->factory()->NewStringFromAsciiChecked(
430 kMethodName)));
431 }
432
433 if (new_byte_length > array_buffer->max_byte_length()) {
434 THROW_NEW_ERROR_RETURN_FAILURE(
435 isolate, NewRangeError(MessageTemplate::kInvalidArrayBufferResizeLength,
436 isolate->factory()->NewStringFromAsciiChecked(
437 kMethodName)));
438 }
439
440 // [RAB] Let hostHandled be ? HostResizeArrayBuffer(O, newByteLength).
441 // [GSAB] Let hostHandled be ? HostGrowArrayBuffer(O, newByteLength).
442 // If hostHandled is handled, return undefined.
443
444 // TODO(v8:11111, v8:12746): Wasm integration.
445
446 if (!is_shared) {
447 // [RAB] Let oldBlock be O.[[ArrayBufferData]].
448 // [RAB] Let newBlock be ? CreateByteDataBlock(newByteLength).
449 // [RAB] Let copyLength be min(newByteLength, O.[[ArrayBufferByteLength]]).
450 // [RAB] Perform CopyDataBlockBytes(newBlock, 0, oldBlock, 0, copyLength).
451 // [RAB] NOTE: Neither creation of the new Data Block nor copying from the
452 // old Data Block are observable. Implementations reserve the right to
453 // implement this method as in-place growth or shrinkage.
454 if (array_buffer->GetBackingStore()->ResizeInPlace(isolate,
455 new_byte_length) !=
456 BackingStore::ResizeOrGrowResult::kSuccess) {
457 THROW_NEW_ERROR_RETURN_FAILURE(
458 isolate, NewRangeError(MessageTemplate::kOutOfMemory,
459 isolate->factory()->NewStringFromAsciiChecked(
460 kMethodName)));
461 }
462
463 // TypedsArrays in optimized code may go out of bounds. Trigger deopts
464 // through the ArrayBufferDetaching protector.
465 if (new_byte_length < array_buffer->byte_length()) {
466 if (Protectors::IsArrayBufferDetachingIntact(isolate)) {
467 Protectors::InvalidateArrayBufferDetaching(isolate);
468 }
469 }
470
471 isolate->heap()->ResizeArrayBufferExtension(
472 array_buffer->extension(),
473 static_cast<int64_t>(new_byte_length) - array_buffer->byte_length());
474
475 // [RAB] Set O.[[ArrayBufferByteLength]] to newLength.
476 array_buffer->set_byte_length(new_byte_length);
477 } else {
478 // [GSAB] (Detailed description of the algorithm omitted.)
479 auto result =
480 array_buffer->GetBackingStore()->GrowInPlace(isolate, new_byte_length);
481 if (result == BackingStore::ResizeOrGrowResult::kFailure) {
482 THROW_NEW_ERROR_RETURN_FAILURE(
483 isolate, NewRangeError(MessageTemplate::kOutOfMemory,
484 isolate->factory()->NewStringFromAsciiChecked(
485 kMethodName)));
486 }
487 if (result == BackingStore::ResizeOrGrowResult::kRace) {
488 THROW_NEW_ERROR_RETURN_FAILURE(
489 isolate,
490 NewRangeError(
491 MessageTemplate::kInvalidArrayBufferResizeLength,
492 isolate->factory()->NewStringFromAsciiChecked(kMethodName)));
493 }
494 // Invariant: byte_length for a GSAB is 0 (it needs to be read from the
495 // BackingStore).
496 CHECK_EQ(0, array_buffer->byte_length());
497 }
498 return ReadOnlyRoots(isolate).undefined_value();
499}

533
534Tagged<Object> ArrayBufferTransfer(Isolate* isolate,
535 DirectHandle<JSArrayBuffer> array_buffer,
536 DirectHandle<Object> new_length,
537 PreserveResizability preserve_resizability,
538 const char* method_name) {
539 // 2. If IsSharedArrayBuffer(arrayBuffer) is true, throw a TypeError
540 // exception.
541 CHECK_SHARED(false, array_buffer, method_name);
542
543 size_t new_byte_length;
544 if (IsUndefined(*new_length, isolate)) {
545 // 3. If newLength is undefined, then
546 // a. Let newByteLength be arrayBuffer.[[ArrayBufferByteLength]].
547 new_byte_length = array_buffer->GetByteLength();
548 } else {
549 // 4. Else,
550 // a. Let newByteLength be ? ToIndex(newLength).
551 DirectHandle<Object> number_new_byte_length;
552 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, number_new_byte_length,
553 Object::ToInteger(isolate, new_length));
554 if (Object::NumberValue(*number_new_byte_length) < 0.0) {
555 THROW_NEW_ERROR_RETURN_FAILURE(
556 isolate, NewRangeError(MessageTemplate::kInvalidArrayBufferLength));
557 }
558 if (!TryNumberToSize(*number_new_byte_length, &new_byte_length) ||
559 new_byte_length > JSArrayBuffer::kMaxByteLength) {
560 THROW_NEW_ERROR_RETURN_FAILURE(
561 isolate,
562 NewRangeError(
563 MessageTemplate::kInvalidArrayBufferResizeLength,
564 isolate->factory()->NewStringFromAsciiChecked(method_name)));
565 }
566 }
567
568 // 5. If IsDetachedBuffer(arrayBuffer) is true, throw a TypeError exception.
569 if (array_buffer->was_detached()) {
570 THROW_NEW_ERROR_RETURN_FAILURE(
571 isolate, NewTypeError(MessageTemplate::kDetachedOperation,
572 isolate->factory()->NewStringFromAsciiChecked(
573 method_name)));
574 }
575
576 ResizableFlag resizable;
577 size_t new_max_byte_length;
578 if (preserve_resizability == kPreserveResizability &&
579 array_buffer->is_resizable_by_js()) {
580 // 6. If preserveResizability is preserve-resizability and
581 // IsResizableArrayBuffer(arrayBuffer) is true, then
582 // a. Let newMaxByteLength be arrayBuffer.[[ArrayBufferMaxByteLength]].
583 new_max_byte_length = array_buffer->max_byte_length();
584 resizable = ResizableFlag::kResizable;
585 } else {
586 // 7. Else,
587 // a. Let newMaxByteLength be empty.
588 new_max_byte_length = new_byte_length;
589 resizable = ResizableFlag::kNotResizable;
590 }
591
592 // 8. If arrayBuffer.[[ArrayBufferDetachKey]] is not undefined, throw a
593 // TypeError exception.
594
595 if (!array_buffer->is_detachable()) {
596 THROW_NEW_ERROR_RETURN_FAILURE(
597 isolate,
598 NewTypeError(MessageTemplate::kDataCloneErrorNonDetachableArrayBuffer));
599 }
600
601 // After this point the steps are not observable and are performed out of
602 // spec order.
603
604 // Case 1: We don't need a BackingStore.
605 if (new_byte_length == 0) {
606 // 15. Perform ! DetachArrayBuffer(arrayBuffer).
607 JSArrayBuffer::Detach(array_buffer).Check();
608
609 // 9. Let newBuffer be ? AllocateArrayBuffer(%ArrayBuffer%, newByteLength,
610 // newMaxByteLength).
611 //
612 // Nothing to do for steps 10-14.
613 //
614 // 16. Return newBuffer.
615 return *isolate->factory()
616 ->NewJSArrayBufferAndBackingStore(
617 0, new_max_byte_length, InitializedFlag::kUninitialized,
618 resizable)
619 .ToHandleChecked();
620 }
621
622 // Case 2: We can reuse the same BackingStore.
623 auto from_backing_store = array_buffer->GetBackingStore();
624 if (from_backing_store && !from_backing_store->is_resizable_by_js() &&
625 resizable == ResizableFlag::kNotResizable &&
626 new_byte_length == array_buffer->GetByteLength()) {
627 // TODO(syg): Consider realloc when the default ArrayBuffer allocator's
628 // Reallocate does better than copy.
629 //
630 // See https://crbug.com/330575496#comment27
631
632 // 15. Perform ! DetachArrayBuffer(arrayBuffer).
633 JSArrayBuffer::Detach(array_buffer).Check();
634
635 // 9. Let newBuffer be ? AllocateArrayBuffer(%ArrayBuffer%, newByteLength,
636 // newMaxByteLength).
637 // 16. Return newBuffer.
638 return *isolate->factory()->NewJSArrayBuffer(std::move(from_backing_store));
639 }
640
641 // Case 3: We can't reuse the same BackingStore. Copy the buffer.
642
643 if (new_byte_length > new_max_byte_length) {
644 THROW_NEW_ERROR_RETURN_FAILURE(
645 isolate, NewRangeError(MessageTemplate::kInvalidArrayBufferLength));
646 }
647
648 // 9. Let newBuffer be ? AllocateArrayBuffer(%ArrayBuffer%, newByteLength,
649 // newMaxByteLength).
650 DirectHandle<JSArrayBuffer> new_buffer;
651 MaybeDirectHandle<JSArrayBuffer> result =
652 isolate->factory()->NewJSArrayBufferAndBackingStore(
653 new_byte_length, new_max_byte_length, InitializedFlag::kUninitialized,
654 resizable);
655 if (!result.ToHandle(&new_buffer)) {
656 THROW_NEW_ERROR_RETURN_FAILURE(
657 isolate, NewRangeError(MessageTemplate::kArrayBufferAllocationFailed));
658 }
659
660 // 10. Let copyLength be min(newByteLength,
661 // arrayBuffer.[[ArrayBufferByteLength]]).
662 //
663 // (Size comparison is done manually below instead of using min.)
664
665 // 11. Let fromBlock be arrayBuffer.[[ArrayBufferData]].
666 uint8_t* from_data =
667 reinterpret_cast<uint8_t*>(array_buffer->backing_store());
668
669 // 12. Let toBlock be newBuffer.[[ArrayBufferData]].
670 uint8_t* to_data = reinterpret_cast<uint8_t*>(new_buffer->backing_store());
671
672 // 13. Perform CopyDataBlockBytes(toBlock, 0, fromBlock, 0, copyLength).
673 // 14. NOTE: Neither creation of the new Data Block nor copying from the old
674 // Data Block are observable. Implementations reserve the right to
675 // implement this method as a zero-copy move or a realloc.
676 size_t from_byte_length = array_buffer->GetByteLength();
677 if (new_byte_length <= from_byte_length) {
678 CopyBytes(to_data, from_data, new_byte_length);
679 } else {
680 CopyBytes(to_data, from_data, from_byte_length);
681 memset(to_data + from_byte_length, 0, new_byte_length - from_byte_length);
682 }
683
684 // 15. Perform ! DetachArrayBuffer(arrayBuffer).
685 JSArrayBuffer::Detach(array_buffer).Check();
686
687 // 16. Return newBuffer.
688 return *new_buffer;
689}
