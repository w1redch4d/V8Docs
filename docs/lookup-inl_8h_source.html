69LookupIterator::LookupIterator(Isolate* isolate, DirectHandle<JSAny> receiver,
70 DirectHandle<Name> name, size_t index,
71 DirectHandle<JSAny> lookup_start_object,
72 Configuration configuration)
73 : configuration_(ComputeConfiguration(isolate, configuration, name)),
74 isolate_(isolate),
75 name_(name),
76 receiver_(receiver),
77 lookup_start_object_(lookup_start_object),
78 index_(index) {
79 if (IsElement()) {
80 // If we're not looking at a TypedArray, we will need the key represented
81 // as an internalized string.
82 if (index_ > JSObject::kMaxElementIndex &&
83 !IsJSTypedArray(*lookup_start_object, isolate_)
84#if V8_ENABLE_WEBASSEMBLY
85 && !IsWasmArray(*lookup_start_object, isolate_)
86#endif // V8_ENABLE_WEBASSEMBLY
87 ) {
88 if (name_.is_null()) {
89 name_ = isolate->factory()->SizeToString(index_);
90 }
91 name_ = isolate->factory()->InternalizeName(name_);
92 } else if (!name_.is_null() && !IsInternalizedString(*name_)) {
93 // Maintain the invariant that if name_ is present, it is internalized.
94 name_ = DirectHandle<Name>();
95 }
96 Start<true>();
97 } else {
98 DCHECK(!name_.is_null());
99 name_ = isolate->factory()->InternalizeName(name_);
100#ifdef DEBUG
101 // Assert that the name is not an index.
102 // If we're not looking at the prototype chain and the lookup start object
103 // is not a typed array, then this means "array index", otherwise we need to
104 // ensure the full generality so that typed arrays are handled correctly.
105 if (!check_prototype_chain() && !IsJSTypedArray(*lookup_start_object)) {
106 uint32_t array_index;
107 DCHECK(!name_->AsArrayIndex(&array_index));
108 } else {
109 size_t integer_index;
110 DCHECK(!name_->AsIntegerIndex(&integer_index));
111 }
112#endif // DEBUG
113 Start<false>();
114 }
115}

135PropertyKey::PropertyKey(Isolate* isolate, double index) {
136 DCHECK_EQ(index, static_cast<uint64_t>(index));
137#if V8_TARGET_ARCH_32_BIT
138 if (index <= JSObject::kMaxElementIndex) {
139 static_assert(JSObject::kMaxElementIndex <=
140 std::numeric_limits<size_t>::max());
141 index_ = static_cast<size_t>(index);
142 } else {
143 index_ = LookupIterator::kInvalidIndex;
144 name_ = isolate->factory()->InternalizeString(

147 }
148#else

150#endif

153template <template <typename> typename HandleType>
154 requires(std::is_convertible_v<HandleType<Name>, DirectHandle<Name>>)
155PropertyKey::PropertyKey(Isolate* isolate, HandleType<Name> name, size_t index)
156 : name_(name), index_(index) {
157 DCHECK_IMPLIES(index_ == LookupIterator::kInvalidIndex, !name_.is_null());
158#if V8_TARGET_ARCH_32_BIT
159 DCHECK_IMPLIES(index_ != LookupIterator::kInvalidIndex,
160 index_ <= JSObject::kMaxElementIndex);
161#endif
162#if DEBUG
163 if (index_ != LookupIterator::kInvalidIndex && !name_.is_null()) {
164 // If both valid index and name are given then the name is a string
165 // representation of the same index.
166 size_t integer_index;
167 CHECK(name_->AsIntegerIndex(&integer_index));
168 CHECK_EQ(index_, integer_index);
169 } else if (index_ == LookupIterator::kInvalidIndex) {
170 // If only name is given it must not be a string representing an integer
171 // index.
172 size_t integer_index;
173 CHECK(!name_->AsIntegerIndex(&integer_index));
174 }
175#endif
176}

181 if (name->AsIntegerIndex(&index_)) {

189template <typename T, template <typename> typename HandleType>
190 requires(std::is_convertible_v<HandleType<T>, DirectHandle<T>>)
191PropertyKey::PropertyKey(Isolate* isolate, HandleType<T> valid_key) {
192 HandleType<Object> valid_obj = Cast<Object>(valid_key);
193 DCHECK(IsName(*valid_obj) || IsNumber(*valid_obj));
194 if (Object::ToIntegerIndex(*valid_obj, &index_)) return;
195 if (IsNumber(*valid_obj)) {
196 // Negative or out of range -> treat as named property.
197 valid_obj = isolate->factory()->NumberToString(valid_obj);
198 }
199 DCHECK(IsName(*valid_obj));
200 name_ = Cast<Name>(valid_obj);
201 if (!name_->AsIntegerIndex(&index_)) {
202 index_ = LookupIterator::kInvalidIndex;
203 name_ = isolate->factory()->InternalizeName(name_);
204 }
205}

207template <typename T, template <typename> typename HandleType>
208 requires(std::is_convertible_v<HandleType<T>, DirectHandle<T>>)
209PropertyKey::PropertyKey(Isolate* isolate, HandleType<T> key, bool* success) {
210 if (Object::ToIntegerIndex(*key, &index_)) {
211 *success = true;
212 return;
213 }
214 *success = Object::ToName(isolate, key).ToHandle(&name_);
215 if (!*success) {
216 DCHECK(isolate->has_exception());
217 index_ = LookupIterator::kInvalidIndex;
218 return;
219 }
220 if (!name_->AsIntegerIndex(&index_)) {
221 // Make sure the name is internalized.
222 name_ = isolate->factory()->InternalizeName(name_);
223 // {AsIntegerIndex} may modify {index_} before deciding to fail.
224 index_ = LookupIterator::kInvalidIndex;
225 }

287bool LookupIterator::ExtendingNonExtensible(DirectHandle<JSReceiver> receiver) {
288 DCHECK(receiver.is_identical_to(GetStoreTarget<JSReceiver>()));
289 if (receiver->map(isolate_)->is_extensible()) {
290 return false;
291 }
292 // Extending with elements and non-private properties is not allowed.
293 if (IsElement() || !name_->IsPrivate()) {
294 return true;
295 }
296 // Shared objects have fixed layout. No properties may be added to them, not
297 // even private symbols.
298 if (IsAlwaysSharedSpaceJSObject(*receiver)) {
299 return true;
300 }
301 // Extending non-extensible objects with private fields is allowed.
302 DCHECK(!receiver->map(isolate_)->is_extensible());
303 DCHECK(name_->IsPrivate());
304 if (name_->IsPrivateName()) {
305 isolate()->CountUsage(v8::Isolate::kExtendingNonExtensibleWithPrivate);
306 }
307 return false;
308}

310bool LookupIterator::IsCacheableTransition() {
311 DCHECK_EQ(TRANSITION, state_);

318// static
319void LookupIterator::UpdateProtector(Isolate* isolate,
320 DirectHandle<JSAny> receiver,
321 DirectHandle<Name> name) {
322 RCS_SCOPE(isolate, RuntimeCallCounterId::kUpdateProtector);
323 DCHECK(IsInternalizedString(*name) || IsSymbol(*name));
324
325 // This check must be kept in sync with
326 // CodeStubAssembler::CheckForAssociatedProtector!
327 ReadOnlyRoots roots(isolate);
328 bool maybe_protector = roots.IsNameForProtector(*name);
329
330#if DEBUG
331 bool debug_maybe_protector =
332 *name == roots.constructor_string() || *name == roots.next_string() ||
333 *name == roots.resolve_string() || *name == roots.then_string() ||
334 *name == roots.is_concat_spreadable_symbol() ||
335 *name == roots.iterator_symbol() || *name == roots.species_symbol() ||
336 *name == roots.match_all_symbol() || *name == roots.replace_symbol() ||
337 *name == roots.split_symbol() || *name == roots.to_primitive_symbol() ||
338 *name == roots.valueOf_string();
339 DCHECK_EQ(maybe_protector, debug_maybe_protector);
340#endif // DEBUG
341
342 if (maybe_protector) {
343 InternalUpdateProtector(isolate, receiver, name);
344 }
345}

386template <class T>
387DirectHandle<T> LookupIterator::GetStoreTarget() const {
388 DCHECK(IsJSReceiver(*receiver_, isolate_));
389 if (IsJSGlobalProxy(*receiver_, isolate_)) {
390 Tagged<HeapObject> prototype =
391 Cast<JSGlobalProxy>(*receiver_)->map(isolate_)->prototype(isolate_);
392 if (IsJSGlobalObject(prototype, isolate_)) {
393 return direct_handle(Cast<JSGlobalObject>(prototype), isolate_);
394 }
395 }
396 return Cast<T>(receiver_);
397}

409inline DirectHandle<InterceptorInfo> LookupIterator::GetInterceptor() const {
410 DCHECK_EQ(INTERCEPTOR, state_);
411 Tagged<JSObject> holder = Cast<JSObject>(*holder_);
412 Tagged<InterceptorInfo> result = IsElement(holder)
413 ? GetInterceptor<true>(holder)
414 : GetInterceptor<false>(holder);
