65// The V8 compiler API.
66//
67// This is the central hub for dispatching to the various compilers within V8.
68// Logic for which compiler to choose and how to wire compilation results into
69// the object heap should be kept inside this class.

71// General strategy: Scripts are translated into anonymous functions w/o
72// parameters which then can be executed. If the source code contains other
73// functions, they might be compiled and allocated as part of the compilation

290// A base class for compilation jobs intended to run concurrent to the main
291// thread. The current state of the job can be checked using {state()}.
292class V8_EXPORT_PRIVATE CompilationJob {
293 public:
294 enum Status { SUCCEEDED, FAILED, RETRY_ON_MAIN_THREAD };
295 enum class State {
296 kReadyToPrepare,
297 kReadyToExecute,
298 kReadyToFinalize,
299 kSucceeded,
300 kFailed,
301 };
302
303 explicit CompilationJob(State initial_state) : state_(initial_state) {}
304 virtual ~CompilationJob() = default;
305
306 State state() const { return state_; }
307
308 protected:
309 V8_WARN_UNUSED_RESULT Status UpdateState(Status status, State next_state) {
310 switch (status) {
311 case SUCCEEDED:
312 state_ = next_state;
313 break;
314 case FAILED:
315 state_ = State::kFailed;
316 break;
317 case RETRY_ON_MAIN_THREAD:
318 // Don't change the state, we'll re-try on the main thread.
319 break;
320 }
321 return status;
322 }
323
324 private:
325 State state_;
326};

398// A base class for optimized compilation jobs.
399//
400// The job is split into three phases which are called in sequence on
401// different threads and with different limitations:
402// 1) PrepareJob: Runs on main thread. No major limitations.
403// 2) ExecuteJob: Runs concurrently. No heap allocation or handle derefs.
404// 3) FinalizeJob: Runs on main thread. No dependency changes.
405//
406// Each of the three phases can either fail or succeed.
407class OptimizedCompilationJob : public CompilationJob {
408 public:
409 OptimizedCompilationJob(const char* compiler_name, State initial_state)
410 : CompilationJob(initial_state), compiler_name_(compiler_name) {

418 V8_EXPORT_PRIVATE V8_WARN_UNUSED_RESULT Status

420
421 // Finalizes the compile job. Must be called on the main thread.
422 V8_EXPORT_PRIVATE V8_WARN_UNUSED_RESULT Status FinalizeJob(Isolate* isolate);
423
424 const char* compiler_name() const { return compiler_name_; }
425
426 double prepare_in_ms() const {

445 virtual Status FinalizeJobImpl(Isolate* isolate) = 0;

455 base::TimeDelta time_taken_to_prepare_;
456 base::TimeDelta time_taken_to_execute_;
457 base::TimeDelta time_taken_to_finalize_;

459 base::ElapsedTimer timer_;

461 private:
462 const char* const compiler_name_;
