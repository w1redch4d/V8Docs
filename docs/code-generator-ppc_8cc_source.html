29// Adds PPC-specific methods to convert InstructionOperands.
30class PPCOperandConverter final : public InstructionOperandConverter {
31 public:
32 PPCOperandConverter(CodeGenerator* gen, Instruction* instr)
33 : InstructionOperandConverter(gen, instr) {}
34
35 size_t OutputCount() { return instr_->OutputCount(); }
36
37 RCBit OutputRCBit() const {
38 switch (instr_->flags_mode()) {
39 case kFlags_branch:
40 case kFlags_conditional_branch:
41 case kFlags_deoptimize:
42 case kFlags_set:
43 case kFlags_conditional_set:
44 case kFlags_trap:
45 case kFlags_select:
46 return SetRC;
47 case kFlags_none:
48 return LeaveRC;
49 }
50 UNREACHABLE();
51 }
52
53 bool CompareLogical() const {
54 switch (instr_->flags_condition()) {
55 case kUnsignedLessThan:
56 case kUnsignedGreaterThanOrEqual:
57 case kUnsignedLessThanOrEqual:
58 case kUnsignedGreaterThan:
59 return true;
60 default:
61 return false;
62 }
63 UNREACHABLE();
64 }
65
66 Operand InputImmediate(size_t index) {
67 Constant constant = ToConstant(instr_->InputAt(index));
68 switch (constant.type()) {
69 case Constant::kInt32:
70 return Operand(constant.ToInt32());
71 case Constant::kFloat32:
72 return Operand::EmbeddedNumber(constant.ToFloat32());
73 case Constant::kFloat64:
74 return Operand::EmbeddedNumber(constant.ToFloat64().value());
75 case Constant::kInt64:
76 return Operand(constant.ToInt64());
77 case Constant::kExternalReference:
78 return Operand(constant.ToExternalReference());
79 case Constant::kCompressedHeapObject: {
80 RootIndex root_index;
81 if (gen_->isolate()->roots_table().IsRootHandle(constant.ToHeapObject(),
82 &root_index)) {
83 CHECK(COMPRESS_POINTERS_BOOL);
84 CHECK(V8_STATIC_ROOTS_BOOL || !gen_->isolate()->bootstrapper());
85 Tagged_t ptr =
86 MacroAssemblerBase::ReadOnlyRootPtr(root_index, gen_->isolate());
87 return Operand(ptr);
88 }
89 return Operand(constant.ToHeapObject());
90 }
91 case Constant::kHeapObject:
92 case Constant::kRpoNumber:
93 break;
94 }
95 UNREACHABLE();
96 }
97
98 MemOperand MemoryOperand(AddressingMode* mode, size_t* first_index) {
99 const size_t index = *first_index;
100 AddressingMode addr_mode = AddressingModeField::decode(instr_->opcode());
101 if (mode) *mode = addr_mode;
102 switch (addr_mode) {
103 case kMode_None:
104 break;
105 case kMode_MRI:
106 *first_index += 2;
107 return MemOperand(InputRegister(index + 0), InputInt64(index + 1));
108 case kMode_MRR:
109 *first_index += 2;
110 return MemOperand(InputRegister(index + 0), InputRegister(index + 1));
111 case kMode_Root:
112 *first_index += 1;
113 return MemOperand(kRootRegister, InputRegister(index));
114 }
115 UNREACHABLE();
116 }
117
118 MemOperand MemoryOperand(AddressingMode* mode = NULL,
119 size_t first_index = 0) {
120 return MemoryOperand(mode, &first_index);
121 }
122
123 MemOperand ToMemOperand(InstructionOperand* op) const {
124 DCHECK_NOT_NULL(op);
125 DCHECK(op->IsStackSlot() || op->IsFPStackSlot());
126 return SlotToMemOperand(AllocatedOperand::cast(op)->index());
127 }
128
129 MemOperand SlotToMemOperand(int slot) const {
130 FrameOffset offset = frame_access_state()->GetFrameOffset(slot);
131 return MemOperand(offset.from_stack_pointer() ? sp : fp, offset.offset());
132 }
133};

189 void Generate() final {
190 ConstantPoolUnavailableScope constant_pool_unavailable(masm());
191 // When storing an indirect pointer, the value will always be a
192 // full/decompressed pointer.
193 if (COMPRESS_POINTERS_BOOL &&
194 mode_ != RecordWriteMode::kValueIsIndirectPointer) {
195 __ DecompressTagged(value_, value_);
196 }
197 __ CheckPageFlag(value_, scratch0_,
198 MemoryChunk::kPointersToHereAreInterestingMask, eq,
199 exit());
200 if (offset_ == no_reg) {
201 __ addi(scratch1_, object_, Operand(offset_immediate_));
202 } else {
203 DCHECK_EQ(0, offset_immediate_);
204 __ add(scratch1_, object_, offset_);
205 }
206 SaveFPRegsMode const save_fp_mode = frame()->DidAllocateDoubleRegisters()
207 ? SaveFPRegsMode::kSave
208 : SaveFPRegsMode::kIgnore;
209 if (must_save_lr_) {
210 // We need to save and restore lr if the frame was elided.
211 __ mflr(scratch0_);
212 __ Push(scratch0_);
213 unwinding_info_writer_->MarkLinkRegisterOnTopOfStack(__ pc_offset());
214 }
215 if (mode_ == RecordWriteMode::kValueIsEphemeronKey) {
216 __ CallEphemeronKeyBarrier(object_, scratch1_, save_fp_mode);
217 } else if (mode_ == RecordWriteMode::kValueIsIndirectPointer) {
218 DCHECK(IsValidIndirectPointerTag(indirect_pointer_tag_));
219 __ CallIndirectPointerBarrier(object_, scratch1_, save_fp_mode,
220 indirect_pointer_tag_);
221#if V8_ENABLE_WEBASSEMBLY
222 } else if (stub_mode_ == StubCallMode::kCallWasmRuntimeStub) {
223 __ CallRecordWriteStubSaveRegisters(object_, scratch1_, save_fp_mode,
224 StubCallMode::kCallWasmRuntimeStub);
225#endif // V8_ENABLE_WEBASSEMBLY
226 } else {
227 __ CallRecordWriteStubSaveRegisters(object_, scratch1_, save_fp_mode);
228 }
229 if (must_save_lr_) {
230 // We need to save and restore lr if the frame was elided.
231 __ Pop(scratch0_);
232 __ mtlr(scratch0_);
233 unwinding_info_writer_->MarkPopLinkRegisterFromTopOfStack(__ pc_offset());
234 }
235 }
236
237 private:
238 Register const object_;
239 Register const offset_;
240 int32_t const offset_immediate_; // Valid if offset_ == no_reg.
241 Register const value_;
242 Register const scratch0_;
243 Register const scratch1_;
244 RecordWriteMode const mode_;
245#if V8_ENABLE_WEBASSEMBLY
246 StubCallMode stub_mode_;
247#endif // V8_ENABLE_WEBASSEMBLY
248 bool must_save_lr_;
249 UnwindingInfoWriter* const unwinding_info_writer_;
250 Zone* zone_;
251 IndirectPointerTag indirect_pointer_tag_;
252};

254Condition FlagsConditionToCondition(FlagsCondition condition, ArchOpcode op) {
255 switch (condition) {
256 case kEqual:
257 return eq;
258 case kNotEqual:
259 return ne;
260 case kSignedLessThan:
261 case kUnsignedLessThan:
262 return lt;
263 case kSignedGreaterThanOrEqual:
264 case kUnsignedGreaterThanOrEqual:
265 return ge;
266 case kSignedLessThanOrEqual:
267 case kUnsignedLessThanOrEqual:
268 return le;
269 case kSignedGreaterThan:
270 case kUnsignedGreaterThan:
271 return gt;
272 case kOverflow:
273 // Overflow checked for add/sub only.
274 switch (op) {
275 case kPPC_Add32:
276 case kPPC_Add64:
277 case kPPC_Sub:
278 case kPPC_AddWithOverflow32:
279 case kPPC_SubWithOverflow32:
280 return lt;
281 default:
282 break;
283 }
284 break;
285 case kNotOverflow:
286 switch (op) {
287 case kPPC_Add32:
288 case kPPC_Add64:
289 case kPPC_Sub:
290 case kPPC_AddWithOverflow32:
291 case kPPC_SubWithOverflow32:
292 return ge;
293 default:
294 break;
295 }
296 break;
297 default:
298 break;
299 }
300 UNREACHABLE();
301}

697void FlushPendingPushRegisters(MacroAssembler* masm,
698 FrameAccessState* frame_access_state,
699 ZoneVector<Register>* pending_pushes) {
700 switch (pending_pushes->size()) {
701 case 0:
702 break;
703 case 1:
704 masm->Push((*pending_pushes)[0]);
705 break;
706 case 2:
707 masm->Push((*pending_pushes)[0], (*pending_pushes)[1]);
708 break;
709 case 3:
710 masm->Push((*pending_pushes)[0], (*pending_pushes)[1],
711 (*pending_pushes)[2]);
712 break;
713 default:
714 UNREACHABLE();
715 }
716 frame_access_state->IncreaseSPDelta(pending_pushes->size());
717 pending_pushes->clear();
718}

720void AdjustStackPointerForTailCall(
721 MacroAssembler* masm, FrameAccessState* state, int new_slot_above_sp,
722 ZoneVector<Register>* pending_pushes = nullptr,
723 bool allow_shrinkage = true) {
724 int current_sp_offset = state->GetSPToFPSlotCount() +
725 StandardFrameConstants::kFixedSlotCountAboveFp;
726 int stack_slot_delta = new_slot_above_sp - current_sp_offset;
727 if (stack_slot_delta > 0) {
728 if (pending_pushes != nullptr) {
729 FlushPendingPushRegisters(masm, state, pending_pushes);
730 }
731 masm->AddS64(sp, sp, Operand(-stack_slot_delta * kSystemPointerSize), r0);
732 state->IncreaseSPDelta(stack_slot_delta);
733 } else if (allow_shrinkage && stack_slot_delta < 0) {
734 if (pending_pushes != nullptr) {
735 FlushPendingPushRegisters(masm, state, pending_pushes);
736 }
737 masm->AddS64(sp, sp, Operand(-stack_slot_delta * kSystemPointerSize), r0);
738 state->IncreaseSPDelta(stack_slot_delta);
739 }
740}

744void CodeGenerator::AssembleTailCallBeforeGap(Instruction* instr,
745 int first_unused_slot_offset) {
746 ZoneVector<MoveOperands*> pushes(zone());
747 GetPushCompatibleMoves(instr, kRegisterPush, &pushes);
748
749 if (!pushes.empty() &&
750 (LocationOperand::cast(pushes.back()->destination()).index() + 1 ==
751 first_unused_slot_offset)) {
752 PPCOperandConverter g(this, instr);
753 ZoneVector<Register> pending_pushes(zone());
754 for (auto move : pushes) {
755 LocationOperand destination_location(
756 LocationOperand::cast(move->destination()));
757 InstructionOperand source(move->source());
758 AdjustStackPointerForTailCall(
759 masm(), frame_access_state(),
760 destination_location.index() - pending_pushes.size(),
761 &pending_pushes);
762 // Pushes of non-register data types are not supported.
763 DCHECK(source.IsRegister());
764 LocationOperand source_location(LocationOperand::cast(source));
765 pending_pushes.push_back(source_location.GetRegister());
766 // TODO(arm): We can push more than 3 registers at once. Add support in
767 // the macro-assembler for pushing a list of registers.
768 if (pending_pushes.size() == 3) {
769 FlushPendingPushRegisters(masm(), frame_access_state(),
770 &pending_pushes);
771 }
772 move->Eliminate();
773 }
774 FlushPendingPushRegisters(masm(), frame_access_state(), &pending_pushes);
775 }
776 AdjustStackPointerForTailCall(masm(), frame_access_state(),
777 first_unused_slot_offset, nullptr, false);
778}

802// Assembles an instruction after register allocation, producing machine code.
803CodeGenerator::CodeGenResult CodeGenerator::AssembleArchInstruction(
804 Instruction* instr) {
805 PPCOperandConverter i(this, instr);
806 ArchOpcode opcode = ArchOpcodeField::decode(instr->opcode());
807
808 switch (opcode) {
809 case kArchCallCodeObject: {
810 v8::internal::Assembler::BlockTrampolinePoolScope block_trampoline_pool(
811 masm());
812 if (HasRegisterInput(instr, 0)) {
813 Register reg = i.InputRegister(0);
814 DCHECK_IMPLIES(
815 instr->HasCallDescriptorFlag(CallDescriptor::kFixedTargetRegister),
816 reg == kJavaScriptCallCodeStartRegister);
817 __ CallCodeObject(reg);
818 } else {
819 __ Call(i.InputCode(0), RelocInfo::CODE_TARGET);
820 }
821 RecordCallPosition(instr);
822 DCHECK_EQ(LeaveRC, i.OutputRCBit());
823 frame_access_state()->ClearSPDelta();
824 break;
825 }
826 case kArchCallBuiltinPointer: {
827 DCHECK(!instr->InputAt(0)->IsImmediate());
828 Register builtin_index = i.InputRegister(0);
829 Register target =
830 instr->HasCallDescriptorFlag(CallDescriptor::kFixedTargetRegister)
831 ? kJavaScriptCallCodeStartRegister
832 : builtin_index;
833 __ CallBuiltinByIndex(builtin_index, target);
834 RecordCallPosition(instr);
835 frame_access_state()->ClearSPDelta();
836 break;
837 }
838#if V8_ENABLE_WEBASSEMBLY
839 case kArchCallWasmFunction:
840 case kArchCallWasmFunctionIndirect: {
841 // We must not share code targets for calls to builtins for wasm code, as
842 // they might need to be patched individually.
843 if (instr->InputAt(0)->IsImmediate()) {
844 DCHECK_EQ(opcode, kArchCallWasmFunction);
845 Constant constant = i.ToConstant(instr->InputAt(0));
846 Address wasm_code = static_cast<Address>(constant.ToInt64());
847 __ Call(wasm_code, constant.rmode());
848 } else if (opcode == kArchCallWasmFunctionIndirect) {
849 __ CallWasmCodePointer(i.InputRegister(0));
850 } else {
851 __ Call(i.InputRegister(0));
852 }
853 RecordCallPosition(instr);
854 DCHECK_EQ(LeaveRC, i.OutputRCBit());
855 frame_access_state()->ClearSPDelta();
856 break;
857 }
858 case kArchTailCallWasm:
859 case kArchTailCallWasmIndirect: {
860 // We must not share code targets for calls to builtins for wasm code, as
861 // they might need to be patched individually.
862 if (instr->InputAt(0)->IsImmediate()) {
863 DCHECK_EQ(opcode, kArchTailCallWasm);
864 Constant constant = i.ToConstant(instr->InputAt(0));
865 Address wasm_code = static_cast<Address>(constant.ToInt64());
866 __ Jump(wasm_code, constant.rmode());
867 } else if (opcode == kArchTailCallWasmIndirect) {
868 __ CallWasmCodePointer(i.InputRegister(0), CallJumpMode::kTailCall);
869 } else {
870 __ Jump(i.InputRegister(0));
871 }
872 DCHECK_EQ(LeaveRC, i.OutputRCBit());
873 frame_access_state()->ClearSPDelta();
874 frame_access_state()->SetFrameAccessToDefault();
875 break;
876 }
877#endif // V8_ENABLE_WEBASSEMBLY
878 case kArchTailCallCodeObject: {
879 if (HasRegisterInput(instr, 0)) {
880 Register reg = i.InputRegister(0);
881 DCHECK_IMPLIES(
882 instr->HasCallDescriptorFlag(CallDescriptor::kFixedTargetRegister),
883 reg == kJavaScriptCallCodeStartRegister);
884 __ JumpCodeObject(reg);
885 } else {
886 // We cannot use the constant pool to load the target since
887 // we've already restored the caller's frame.
888 ConstantPoolUnavailableScope constant_pool_unavailable(masm());
889 __ Jump(i.InputCode(0), RelocInfo::CODE_TARGET);
890 }
891 DCHECK_EQ(LeaveRC, i.OutputRCBit());
892 frame_access_state()->ClearSPDelta();
893 frame_access_state()->SetFrameAccessToDefault();
894 break;
895 }
896 case kArchTailCallAddress: {
897 CHECK(!instr->InputAt(0)->IsImmediate());
898 Register reg = i.InputRegister(0);
899 DCHECK_IMPLIES(
900 instr->HasCallDescriptorFlag(CallDescriptor::kFixedTargetRegister),
901 reg == kJavaScriptCallCodeStartRegister);
902 __ Jump(reg);
903 frame_access_state()->ClearSPDelta();
904 frame_access_state()->SetFrameAccessToDefault();
905 break;
906 }
907 case kArchCallJSFunction: {
908 v8::internal::Assembler::BlockTrampolinePoolScope block_trampoline_pool(
909 masm());
910 Register func = i.InputRegister(0);
911 if (v8_flags.debug_code) {
912 // Check the function's context matches the context argument.
913 __ LoadTaggedField(
914 kScratchReg, FieldMemOperand(func, JSFunction::kContextOffset), r0);
915 __ CmpS64(cp, kScratchReg);
916 __ Assert(eq, AbortReason::kWrongFunctionContext);
917 }
918 uint32_t num_arguments =
919 i.InputUint32(instr->JSCallArgumentCountInputIndex());
920 __ CallJSFunction(func, num_arguments, kScratchReg);
921 RecordCallPosition(instr);
922 DCHECK_EQ(LeaveRC, i.OutputRCBit());
923 frame_access_state()->ClearSPDelta();
924 break;
925 }
926 case kArchPrepareCallCFunction: {
927 int const num_gp_parameters = ParamField::decode(instr->opcode());
928 int const num_fp_parameters = FPParamField::decode(instr->opcode());
929 __ PrepareCallCFunction(num_gp_parameters + num_fp_parameters,
930 kScratchReg);
931 // Frame alignment requires using FP-relative frame addressing.
932 frame_access_state()->SetFrameAccessToFP();
933 break;
934 }
935 case kArchSaveCallerRegisters: {
936 fp_mode_ =
937 static_cast<SaveFPRegsMode>(MiscField::decode(instr->opcode()));
938 DCHECK(fp_mode_ == SaveFPRegsMode::kIgnore ||
939 fp_mode_ == SaveFPRegsMode::kSave);
940 // kReturnRegister0 should have been saved before entering the stub.
941 int bytes = __ PushCallerSaved(fp_mode_, ip, r0, kReturnRegister0);
942 DCHECK(IsAligned(bytes, kSystemPointerSize));
943 DCHECK_EQ(0, frame_access_state()->sp_delta());
944 frame_access_state()->IncreaseSPDelta(bytes / kSystemPointerSize);
945 DCHECK(!caller_registers_saved_);
946 caller_registers_saved_ = true;
947 break;
948 }
949 case kArchRestoreCallerRegisters: {
950 DCHECK(fp_mode_ ==
951 static_cast<SaveFPRegsMode>(MiscField::decode(instr->opcode())));
952 DCHECK(fp_mode_ == SaveFPRegsMode::kIgnore ||
953 fp_mode_ == SaveFPRegsMode::kSave);
954 // Don't overwrite the returned value.
955 int bytes = __ PopCallerSaved(fp_mode_, ip, r0, kReturnRegister0);
956 frame_access_state()->IncreaseSPDelta(-(bytes / kSystemPointerSize));
957 DCHECK_EQ(0, frame_access_state()->sp_delta());
958 DCHECK(caller_registers_saved_);
959 caller_registers_saved_ = false;
960 break;
961 }
962 case kArchPrepareTailCall:
963 AssemblePrepareTailCall();
964 break;
965 case kArchComment:
966 __ RecordComment(reinterpret_cast<const char*>(i.InputInt64(0)),
967 SourceLocation());
968 break;
969 case kArchCallCFunctionWithFrameState:
970 case kArchCallCFunction: {
971 int const num_gp_parameters = ParamField::decode(instr->opcode());
972 int const fp_param_field = FPParamField::decode(instr->opcode());
973 int num_fp_parameters = fp_param_field;
974 bool has_function_descriptor = false;
975 SetIsolateDataSlots set_isolate_data_slots = SetIsolateDataSlots::kYes;
976#if ABI_USES_FUNCTION_DESCRIPTORS
977 // AIX/PPC64BE Linux uses a function descriptor
978 int kNumFPParametersMask = kHasFunctionDescriptorBitMask - 1;
979 num_fp_parameters = kNumFPParametersMask & fp_param_field;
980 has_function_descriptor =
981 (fp_param_field & kHasFunctionDescriptorBitMask) != 0;
982#endif
983#if V8_ENABLE_WEBASSEMBLY
984 Label start_call;
985 int start_pc_offset = 0;
986 bool isWasmCapiFunction =
987 linkage()->GetIncomingDescriptor()->IsWasmCapiFunction();
988 if (isWasmCapiFunction) {
989 __ mflr(r0);
990 __ LoadPC(kScratchReg);
991 __ bind(&start_call);
992 start_pc_offset = __ pc_offset();
993 // We are going to patch this instruction after emitting
994 // CallCFunction, using a zero offset here as placeholder for now.
995 // patch_pc_address assumes `addi` is used here to
996 // add the offset to pc.
997 __ addi(kScratchReg, kScratchReg, Operand::Zero());
998 __ StoreU64(kScratchReg,
999 MemOperand(fp, WasmExitFrameConstants::kCallingPCOffset));
1000 __ mtlr(r0);
1001 set_isolate_data_slots = SetIsolateDataSlots::kNo;
1002 }
1003#endif // V8_ENABLE_WEBASSEMBLY
1004 int pc_offset;
1005 if (instr->InputAt(0)->IsImmediate()) {
1006 ExternalReference ref = i.InputExternalReference(0);
1007 pc_offset =
1008 __ CallCFunction(ref, num_gp_parameters, num_fp_parameters,
1009 set_isolate_data_slots, has_function_descriptor);
1010 } else {
1011 Register func = i.InputRegister(0);
1012 pc_offset =
1013 __ CallCFunction(func, num_gp_parameters, num_fp_parameters,
1014 set_isolate_data_slots, has_function_descriptor);
1015 }
1016#if V8_ENABLE_WEBASSEMBLY
1017 if (isWasmCapiFunction) {
1018 int offset_since_start_call = pc_offset - start_pc_offset;
1019 // Here we are going to patch the `addi` instruction above to use the
1020 // correct offset.
1021 // LoadPC emits two instructions and pc is the address of its second
1022 // emitted instruction. Add one more to the offset to point to after the
1023 // Call.
1024 offset_since_start_call += kInstrSize;
1025 __ patch_pc_address(kScratchReg, start_pc_offset,
1026 offset_since_start_call);
1027 }
1028#endif // V8_ENABLE_WEBASSEMBLY
1029 RecordSafepoint(instr->reference_map(), pc_offset);
1030
1031 bool const needs_frame_state =
1032 (opcode == kArchCallCFunctionWithFrameState);
1033 if (needs_frame_state) {
1034 RecordDeoptInfo(instr, pc_offset);
1035 }
1036
1037 frame_access_state()->SetFrameAccessToDefault();
1038 // Ideally, we should decrement SP delta to match the change of stack
1039 // pointer in CallCFunction. However, for certain architectures (e.g.
1040 // ARM), there may be more strict alignment requirement, causing old SP
1041 // to be saved on the stack. In those cases, we can not calculate the SP
1042 // delta statically.
1043 frame_access_state()->ClearSPDelta();
1044 if (caller_registers_saved_) {
1045 // Need to re-sync SP delta introduced in kArchSaveCallerRegisters.
1046 // Here, we assume the sequence to be:
1047 // kArchSaveCallerRegisters;
1048 // kArchCallCFunction;
1049 // kArchRestoreCallerRegisters;
1050 int bytes =
1051 __ RequiredStackSizeForCallerSaved(fp_mode_, kReturnRegister0);
1052 frame_access_state()->IncreaseSPDelta(bytes / kSystemPointerSize);
1053 }
1054 break;
1055 }
1056 case kArchJmp:
1057 AssembleArchJump(i.InputRpo(0));
1058 DCHECK_EQ(LeaveRC, i.OutputRCBit());
1059 break;
1060 case kArchBinarySearchSwitch:
1061 AssembleArchBinarySearchSwitch(instr);
1062 break;
1063 case kArchTableSwitch:
1064 AssembleArchTableSwitch(instr);
1065 DCHECK_EQ(LeaveRC, i.OutputRCBit());
1066 break;
1067 case kArchAbortCSADcheck:
1068 DCHECK(i.InputRegister(0) == r4);
1069 {
1070 // We don't actually want to generate a pile of code for this, so just
1071 // claim there is a stack frame, without generating one.
1072 FrameScope scope(masm(), StackFrame::NO_FRAME_TYPE);
1073 __ CallBuiltin(Builtin::kAbortCSADcheck);
1074 }
1075 __ stop();
1076 break;
1077 case kArchDebugBreak:
1078 __ DebugBreak();
1079 break;
1080 case kArchNop:
1081 case kArchThrowTerminator:
1082 // don't emit code for nops.
1083 DCHECK_EQ(LeaveRC, i.OutputRCBit());
1084 break;
1085 case kArchDeoptimize: {
1086 DeoptimizationExit* exit =
1087 BuildTranslation(instr, -1, 0, 0, OutputFrameStateCombine::Ignore());
1088 __ b(exit->label());
1089 break;
1090 }
1091 case kArchRet:
1092 AssembleReturn(instr->InputAt(0));
1093 DCHECK_EQ(LeaveRC, i.OutputRCBit());
1094 break;
1095 case kArchFramePointer:
1096 __ mr(i.OutputRegister(), fp);
1097 DCHECK_EQ(LeaveRC, i.OutputRCBit());
1098 break;
1099 case kArchParentFramePointer:
1100 if (frame_access_state()->has_frame()) {
1101 __ LoadU64(i.OutputRegister(), MemOperand(fp, 0));
1102 } else {
1103 __ mr(i.OutputRegister(), fp);
1104 }
1105 break;
1106#if V8_ENABLE_WEBASSEMBLY
1107 case kArchStackPointer:
1108 __ mr(i.OutputRegister(), sp);
1109 break;
1110 case kArchSetStackPointer: {
1111 DCHECK(instr->InputAt(0)->IsRegister());
1112 __ mr(sp, i.InputRegister(0));
1113 break;
1114 }
1115#endif // V8_ENABLE_WEBASSEMBLY
1116 case kArchStackPointerGreaterThan: {
1117 // Potentially apply an offset to the current stack pointer before the
1118 // comparison to consider the size difference of an optimized frame versus
1119 // the contained unoptimized frames.
1120
1121 Register lhs_register = sp;
1122 uint32_t offset;
1123
1124 if (ShouldApplyOffsetToStackCheck(instr, &offset)) {
1125 lhs_register = i.TempRegister(0);
1126 __ SubS64(lhs_register, sp, Operand(offset), kScratchReg);
1127 }
1128
1129 constexpr size_t kValueIndex = 0;
1130 DCHECK(instr->InputAt(kValueIndex)->IsRegister());
1131 __ CmpU64(lhs_register, i.InputRegister(kValueIndex), cr0);
1132 break;
1133 }
1134 case kArchStackCheckOffset:
1135 __ LoadSmiLiteral(i.OutputRegister(),
1136 Smi::FromInt(GetStackCheckOffset()));
1137 break;
1138 case kArchTruncateDoubleToI:
1139 __ TruncateDoubleToI(isolate(), zone(), i.OutputRegister(),
1140 i.InputDoubleRegister(0), DetermineStubCallMode());
1141 DCHECK_EQ(LeaveRC, i.OutputRCBit());
1142 break;
1143 case kArchStoreWithWriteBarrier: {
1144 RecordWriteMode mode = RecordWriteModeField::decode(instr->opcode());
1145 // Indirect pointer writes must use a different opcode.
1146 DCHECK_NE(mode, RecordWriteMode::kValueIsIndirectPointer);
1147 Register object = i.InputRegister(0);
1148 Register value = i.InputRegister(2);
1149 Register scratch0 = i.TempRegister(0);
1150 Register scratch1 = i.TempRegister(1);
1151 OutOfLineRecordWrite* ool;
1152
1153 if (v8_flags.debug_code) {
1154 // Checking that |value| is not a cleared weakref: our write barrier
1155 // does not support that for now.
1156 __ CmpS64(value, Operand(kClearedWeakHeapObjectLower32), kScratchReg);
1157 __ Check(ne, AbortReason::kOperandIsCleared);
1158 }
1159
1160 AddressingMode addressing_mode =
1161 AddressingModeField::decode(instr->opcode());
1162 if (addressing_mode == kMode_MRI) {
1163 int32_t offset = i.InputInt32(1);
1164 ool = zone()->New<OutOfLineRecordWrite>(
1165 this, object, offset, value, scratch0, scratch1, mode,
1166 DetermineStubCallMode(), &unwinding_info_writer_);
1167 __ StoreTaggedField(value, MemOperand(object, offset), r0);
1168 } else {
1169 DCHECK_EQ(kMode_MRR, addressing_mode);
1170 Register offset(i.InputRegister(1));
1171 ool = zone()->New<OutOfLineRecordWrite>(
1172 this, object, offset, value, scratch0, scratch1, mode,
1173 DetermineStubCallMode(), &unwinding_info_writer_);
1174 __ StoreTaggedField(value, MemOperand(object, offset), r0);
1175 }
1176 // Skip the write barrier if the value is a Smi. However, this is only
1177 // valid if the value isn't an indirect pointer. Otherwise the value will
1178 // be a pointer table index, which will always look like a Smi (but
1179 // actually reference a pointer in the pointer table).
1180 if (mode > RecordWriteMode::kValueIsIndirectPointer) {
1181 __ JumpIfSmi(value, ool->exit());
1182 }
1183 __ CheckPageFlag(object, scratch0,
1184 MemoryChunk::kPointersFromHereAreInterestingMask, ne,
1185 ool->entry());
1186 __ bind(ool->exit());
1187 break;
1188 }
1189 case kArchStoreIndirectWithWriteBarrier: {
1190 RecordWriteMode mode = RecordWriteModeField::decode(instr->opcode());
1191 Register scratch0 = i.TempRegister(0);
1192 Register scratch1 = i.TempRegister(1);
1193 OutOfLineRecordWrite* ool;
1194 DCHECK_EQ(mode, RecordWriteMode::kValueIsIndirectPointer);
1195 AddressingMode addressing_mode =
1196 AddressingModeField::decode(instr->opcode());
1197 Register object = i.InputRegister(0);
1198 Register value = i.InputRegister(2);
1199 IndirectPointerTag tag = static_cast<IndirectPointerTag>(i.InputInt64(3));
1200 DCHECK(IsValidIndirectPointerTag(tag));
1201 if (addressing_mode == kMode_MRI) {
1202 uint64_t offset = i.InputInt64(1);
1203 ool = zone()->New<OutOfLineRecordWrite>(
1204 this, object, offset, value, scratch0, scratch1, mode,
1205 DetermineStubCallMode(), &unwinding_info_writer_, tag);
1206 __ StoreIndirectPointerField(value, MemOperand(object, offset), r0);
1207 } else {
1208 DCHECK_EQ(addressing_mode, kMode_MRR);
1209 Register offset(i.InputRegister(1));
1210 ool = zone()->New<OutOfLineRecordWrite>(
1211 this, object, offset, value, scratch0, scratch1, mode,
1212 DetermineStubCallMode(), &unwinding_info_writer_, tag);
1213 __ StoreIndirectPointerField(value, MemOperand(object, offset), r0);
1214 }
1215 __ CheckPageFlag(object, scratch0,
1216 MemoryChunk::kPointersFromHereAreInterestingMask, ne,
1217 ool->entry());
1218 __ bind(ool->exit());
1219 break;
1220 }
1221 case kArchStackSlot: {
1222 FrameOffset offset =
1223 frame_access_state()->GetFrameOffset(i.InputInt32(0));
1224 __ AddS64(i.OutputRegister(), offset.from_stack_pointer() ? sp : fp,
1225 Operand(offset.offset()), r0);
1226 break;
1227 }
1228 case kPPC_Peek: {
1229 int reverse_slot = i.InputInt32(0);
1230 int offset =
1231 FrameSlotToFPOffset(frame()->GetTotalFrameSlotCount() - reverse_slot);

2863 PPCOperandConverter i(this, instr);
2864 Label* tlabel = branch->true_label;
2865 Label* flabel = branch->false_label;
2866 ArchOpcode op = instr->arch_opcode();
2867 FlagsCondition condition = branch->condition;
2868 CRegister cr = cr0;
2869
2870 Condition cond = FlagsConditionToCondition(condition, op);
2871 if (op == kPPC_CmpDouble) {
2872 // check for unordered if necessary
2873 if (cond == le) {
2874 __ bunordered(flabel, cr);
2875 // Unnecessary for eq/lt since only FU bit will be set.
2876 } else if (cond == gt) {
2877 __ bunordered(tlabel, cr);
2878 // Unnecessary for ne/ge since only FU bit will be set.
2879 }
2880 }
2881 __ b(cond, tlabel, cr);
2882 if (!branch->fallthru) __ b(flabel); // no fallthru to flabel.
2883}
