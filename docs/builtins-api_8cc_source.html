23// Returns the holder JSObject if the function can legally be called with this
24// receiver. Returns nullptr if the call is illegal.
25// TODO(dcarney): CallOptimization duplicates this logic, merge.
26Tagged<JSReceiver> GetCompatibleReceiver(Isolate* isolate,
27 Tagged<FunctionTemplateInfo> info,
28 Tagged<JSReceiver> receiver) {
29 RCS_SCOPE(isolate, RuntimeCallCounterId::kGetCompatibleReceiver);
30 Tagged<Object> recv_type = info->signature();
31 // No signature, return holder.
32 if (!IsFunctionTemplateInfo(recv_type)) return receiver;
33 // A Proxy cannot have been created from the signature template.
34 if (!IsJSObject(receiver)) return JSReceiver();
35
36 Tagged<JSObject> js_obj_receiver = Cast<JSObject>(receiver);
37 Tagged<FunctionTemplateInfo> signature =
38 Cast<FunctionTemplateInfo>(recv_type);
39
40 // Check the receiver.
41 if (signature->IsTemplateFor(js_obj_receiver)) return receiver;
42
43 // The JSGlobalProxy might have a hidden prototype.
44 if (V8_UNLIKELY(IsJSGlobalProxy(js_obj_receiver))) {
45 Tagged<HeapObject> prototype = js_obj_receiver->map()->prototype();
46 if (!IsNull(prototype, isolate)) {
47 Tagged<JSObject> js_obj_prototype = Cast<JSObject>(prototype);
48 if (signature->IsTemplateFor(js_obj_prototype)) return js_obj_prototype;
49 }
50 }
51 return JSReceiver();
52}

54// argv and argc are the same as those passed to FunctionCallbackInfo:
55// - argc is the number of arguments excluding the receiver
56// - argv is the array arguments. The receiver is stored at argv[-1].
57template <bool is_construct>
58V8_WARN_UNUSED_RESULT MaybeHandle<Object> HandleApiCallHelper(
59 Isolate* isolate, DirectHandle<HeapObject> new_target,
60 DirectHandle<FunctionTemplateInfo> fun_data, DirectHandle<Object> receiver,
61 Address* argv, int argc) {
62 Handle<JSReceiver> js_receiver;
63 Tagged<JSReceiver> raw_holder;
64 if (is_construct) {
65 DCHECK(IsTheHole(*receiver, isolate));
66 if (IsUndefined(fun_data->GetInstanceTemplate(), isolate)) {
67 v8::Local<ObjectTemplate> templ =
68 ObjectTemplate::New(reinterpret_cast<v8::Isolate*>(isolate),
69 ToApiHandle<v8::FunctionTemplate>(fun_data));
70 FunctionTemplateInfo::SetInstanceTemplate(
71 isolate, fun_data, Utils::OpenDirectHandle(*templ));
72 }
73 DirectHandle<ObjectTemplateInfo> instance_template(
74 Cast<ObjectTemplateInfo>(fun_data->GetInstanceTemplate()), isolate);
75 ASSIGN_RETURN_ON_EXCEPTION(
76 isolate, js_receiver,
77 ApiNatives::InstantiateObject(isolate, instance_template,
78 Cast<JSReceiver>(new_target)));
79 argv[BuiltinArguments::kReceiverArgsIndex] = js_receiver->ptr();
80 raw_holder = *js_receiver;
81 } else {
82 DCHECK(IsJSReceiver(*receiver));
83 js_receiver = indirect_handle(Cast<JSReceiver>(receiver), isolate);
84
85 if (!fun_data->accept_any_receiver() && IsAccessCheckNeeded(*js_receiver)) {
86 // Proxies never need access checks.
87 DCHECK(IsJSObject(*js_receiver));
88 DirectHandle<JSObject> js_object = Cast<JSObject>(js_receiver);
89 if (!isolate->MayAccess(isolate->native_context(), js_object)) {
90 RETURN_ON_EXCEPTION(isolate,
91 isolate->ReportFailedAccessCheck(js_object));
92 UNREACHABLE();
93 }
94 }
95
96 raw_holder = GetCompatibleReceiver(isolate, *fun_data, *js_receiver);
97
98 if (raw_holder.is_null()) {
99 // This function cannot be called with the given receiver. Abort!
100 THROW_NEW_ERROR(isolate,
101 NewTypeError(MessageTemplate::kIllegalInvocation));
102 }
103 }
104
105 if (fun_data->has_callback(isolate)) {
106 FunctionCallbackArguments custom(isolate, *fun_data, raw_holder,
107 *new_target, argv, argc);
108 DirectHandle<Object> result =
109 custom.CallOrConstruct(*fun_data, is_construct);
110
111 RETURN_EXCEPTION_IF_EXCEPTION(isolate);
112 if (result.is_null()) {
113 if (is_construct) return js_receiver;
114 return isolate->factory()->undefined_value();
115 }
116 // Rebox the result.
117 {
118 DisallowGarbageCollection no_gc;
119 Tagged<Object> raw_result = *result;
120 DCHECK(Is<JSAny>(raw_result));
121 if (!is_construct || IsJSReceiver(raw_result))
122 return handle(raw_result, isolate);
123 }
124 }
125
126 return js_receiver;
127}

170MaybeHandle<Object> Builtins::InvokeApiFunction(
171 Isolate* isolate, bool is_construct,
172 DirectHandle<FunctionTemplateInfo> function, DirectHandle<Object> receiver,
173 base::Vector<const DirectHandle<Object>> args,
174 DirectHandle<HeapObject> new_target) {
175 RCS_SCOPE(isolate, RuntimeCallCounterId::kInvokeApiFunction);
176
177 // Do proper receiver conversion for non-strict mode api functions.
178 if (!is_construct && !IsJSReceiver(*receiver)) {
179 ASSIGN_RETURN_ON_EXCEPTION(isolate, receiver,
180 Object::ConvertReceiver(isolate, receiver));
181 }
182
183 // We assume that all lazy accessor pairs have been instantiated when setting
184 // a break point on any API function.
185 DCHECK(!Cast<FunctionTemplateInfo>(function)->BreakAtEntry(isolate));
186
187 int argc = static_cast<int>(args.size());
188 base::SmallVector<Address, 32> argv(argc + 1);
189 argv[0] = (*receiver).ptr();
190 for (int i = 0; i < argc; ++i) {
191 argv[i + 1] = (*args[i]).ptr();
192 }
193
194 RelocatableArguments arguments(isolate, argv.size(), argv.data());
195 if (is_construct) {
196 return HandleApiCallHelper<true>(isolate, new_target, function, receiver,
197 argv.data() + 1, argc);
198 }
199 return HandleApiCallHelper<false>(isolate, new_target, function, receiver,
200 argv.data() + 1, argc);
201}

203// Helper function to handle calls to non-function objects created through the
204// API. The object can be called as either a constructor (using new) or just as
205// a function (without new).
206V8_WARN_UNUSED_RESULT static Tagged<Object>
207HandleApiCallAsFunctionOrConstructorDelegate(Isolate* isolate,
208 bool is_construct_call,
209 BuiltinArguments args) {
210 DirectHandle<Object> receiver = args.receiver();
211
212 // Get the object called.
213 Tagged<JSObject> obj = Cast<JSObject>(*receiver);
214
215 // Set the new target.
216 Tagged<HeapObject> new_target;
217 if (is_construct_call) {
218 // TODO(adamk): This should be passed through in args instead of
219 // being patched in here. We need to set a non-undefined value
220 // for v8::FunctionCallbackInfo::IsConstructCall() to get the
221 // right answer.
222 new_target = obj;
223 } else {
224 new_target = ReadOnlyRoots(isolate).undefined_value();
225 }
226
227 // Get the invocation callback from the function descriptor that was
228 // used to create the called object.
229 DCHECK(obj->map()->is_callable());
230 Tagged<JSFunction> constructor =
231 Cast<JSFunction>(obj->map()->GetConstructor());
232 DCHECK(constructor->shared()->IsApiFunction());
233 Tagged<Object> handler =
234 constructor->shared()->api_func_data()->GetInstanceCallHandler();
235 DCHECK(!IsUndefined(handler, isolate));
236 Tagged<FunctionTemplateInfo> templ = Cast<FunctionTemplateInfo>(handler);
237 DCHECK(templ->is_object_template_call_handler());
238 DCHECK(templ->has_callback(isolate));
239
240 // Get the data for the call and perform the callback.
241 Tagged<Object> result;
242 {
243 HandleScope scope(isolate);
244 FunctionCallbackArguments custom(isolate, templ, obj, new_target,
245 args.address_of_first_argument(),
246 args.length() - 1);
247 DirectHandle<Object> result_handle =
248 custom.CallOrConstruct(templ, is_construct_call);
249 if (result_handle.is_null()) {
250 result = ReadOnlyRoots(isolate).undefined_value();
251 } else {
252 result = *result_handle;
253 }
254 // Check for exceptions and return result.
255 RETURN_FAILURE_IF_EXCEPTION(isolate);
256 }
257 return result;
258}

262BUILTIN(HandleApiCallAsFunctionDelegate) {

267// Handle calls to non-function objects created through the API. This delegate
