24OptimizedCompilationInfo::OptimizedCompilationInfo(
25 Zone* zone, Isolate* isolate, IndirectHandle<SharedFunctionInfo> shared,
26 IndirectHandle<JSFunction> closure, CodeKind code_kind,
27 BytecodeOffset osr_offset)
28 : isolate_unsafe_(isolate),
29 code_kind_(code_kind),
30 osr_offset_(osr_offset),
31 zone_(zone),
32 optimization_id_(isolate->NextOptimizationId()) {
33 DCHECK_EQ(*shared, closure->shared());
34 DCHECK(shared->is_compiled());
35 DCHECK_IMPLIES(is_osr(), IsOptimizing());
36 bytecode_array_ = handle(shared->GetBytecodeArray(isolate), isolate);
37 shared_info_ = shared;
38 closure_ = closure;
39 canonical_handles_ = std::make_unique<CanonicalHandlesMap>(
40 isolate->heap(), ZoneAllocationPolicy(zone));
41
42 // Collect source positions for optimized code when profiling or if debugger
43 // is active, to be able to get more precise source positions at the price of
44 // more memory consumption.
45 if (isolate->NeedsDetailedOptimizedCodeLineInfo()) {
46 set_source_positions();
47 }
48
49 SetTracingFlags(shared->PassesFilter(v8_flags.trace_turbo_filter));
50 ConfigureFlags();
51
52 if (isolate->node_observer()) {
53 SetNodeObserver(isolate->node_observer());
54 }
55}

57OptimizedCompilationInfo::OptimizedCompilationInfo(
58 base::Vector<const char> debug_name, Zone* zone, CodeKind code_kind,
59 Builtin builtin)
60 : isolate_unsafe_(nullptr),
61 code_kind_(code_kind),
62 builtin_(builtin),
63 zone_(zone),
64 optimization_id_(kNoOptimizationId),
65 debug_name_(debug_name) {
66 DCHECK_IMPLIES(builtin_ != Builtin::kNoBuiltinId,
67 (code_kind_ == CodeKind::BUILTIN ||
68 code_kind_ == CodeKind::BYTECODE_HANDLER));
69 SetTracingFlags(
70 PassesFilter(debug_name, base::CStrVector(v8_flags.trace_turbo_filter)));
71 ConfigureFlags();
72 DCHECK(!has_shared_info());
73}

75void OptimizedCompilationInfo::ConfigureFlags() {
76 if (v8_flags.turbo_inline_js_wasm_calls) set_inline_js_wasm_calls();
77
78 if (v8_flags.cet_compatible) {
79 set_shadow_stack_compliant_lazy_deopt();
80 }
81
82 switch (code_kind_) {
83 case CodeKind::TURBOFAN_JS:
84 set_called_with_code_start_register();
85 set_switch_jump_table();
86 if (v8_flags.analyze_environment_liveness) {
87 set_analyze_environment_liveness();
88 }
89 if (v8_flags.turbo_splitting) set_splitting();
90 break;
91 case CodeKind::BYTECODE_HANDLER:
92 set_called_with_code_start_register();
93 if (v8_flags.turbo_splitting) set_splitting();
94 if (v8_flags.enable_allocation_folding) set_allocation_folding();
95 break;
96 case CodeKind::BUILTIN:
97#ifdef V8_ENABLE_BUILTIN_JUMP_TABLE_SWITCH
98 set_switch_jump_table();
99#endif // V8_TARGET_ARCH_X64
100 [[fallthrough]];
101 case CodeKind::FOR_TESTING:
102 if (v8_flags.turbo_splitting) set_splitting();
103 if (v8_flags.enable_allocation_folding) set_allocation_folding();
104#if ENABLE_GDB_JIT_INTERFACE && DEBUG
105 set_source_positions();
106#endif // ENABLE_GDB_JIT_INTERFACE && DEBUG
107 break;
108 case CodeKind::WASM_FUNCTION:
109 case CodeKind::WASM_TO_CAPI_FUNCTION:
110 set_switch_jump_table();
111 break;
112 case CodeKind::C_WASM_ENTRY:
113 case CodeKind::JS_TO_WASM_FUNCTION:
114 case CodeKind::WASM_TO_JS_FUNCTION:
115 break;
116 case CodeKind::BASELINE:
117 case CodeKind::MAGLEV:
118 case CodeKind::INTERPRETED_FUNCTION:
119 case CodeKind::REGEXP:
120 UNREACHABLE();
121 }
122}

131void OptimizedCompilationInfo::ReopenAndCanonicalizeHandlesInNewScope(
132 Isolate* isolate) {
133 if (!shared_info_.is_null()) {
134 shared_info_ = CanonicalHandle(*shared_info_, isolate);
135 }
136 if (!bytecode_array_.is_null()) {
137 bytecode_array_ = CanonicalHandle(*bytecode_array_, isolate);
138 }
139 if (!closure_.is_null()) {
140 closure_ = CanonicalHandle(*closure_, isolate);
141 }
142 DCHECK(code_.is_null());
143}

145void OptimizedCompilationInfo::AbortOptimization(BailoutReason reason) {
146 DCHECK_NE(reason, BailoutReason::kNoReason);
147 if (bailout_reason_ == BailoutReason::kNoReason) {
148 bailout_reason_ = reason;
149 }
150 set_disable_future_optimization();
151}

153void OptimizedCompilationInfo::RetryOptimization(BailoutReason reason) {
154 DCHECK_NE(reason, BailoutReason::kNoReason);
155 if (disable_future_optimization()) return;
156 bailout_reason_ = reason;

159std::unique_ptr<char[]> OptimizedCompilationInfo::GetDebugName() const {
160 if (!shared_info().is_null()) {
161 return shared_info()->DebugNameCStr();
162 }
163 base::Vector<const char> name_vec = debug_name_;
164 if (name_vec.empty()) name_vec = base::ArrayVector("unknown");
165 std::unique_ptr<char[]> name(new char[name_vec.length() + 1]);
166 memcpy(name.get(), name_vec.begin(), name_vec.length());
167 name[name_vec.length()] = '\0';
168 return name;
169}

171StackFrame::Type OptimizedCompilationInfo::GetOutputStackFrameType() const {
172 switch (code_kind()) {
173 case CodeKind::FOR_TESTING:
174 case CodeKind::BYTECODE_HANDLER:
175 case CodeKind::BUILTIN:
176 return StackFrame::STUB;
177#if V8_ENABLE_WEBASSEMBLY
178 case CodeKind::WASM_FUNCTION:
179 return StackFrame::WASM;
180 case CodeKind::WASM_TO_CAPI_FUNCTION:
181 return StackFrame::WASM_EXIT;
182 case CodeKind::JS_TO_WASM_FUNCTION:
183 return StackFrame::JS_TO_WASM;
184 case CodeKind::WASM_TO_JS_FUNCTION:
185 return StackFrame::WASM_TO_JS;
186 case CodeKind::C_WASM_ENTRY:
187 return StackFrame::C_WASM_ENTRY;
188#endif // V8_ENABLE_WEBASSEMBLY
189 default:
190 UNIMPLEMENTED();
191 }
192}
