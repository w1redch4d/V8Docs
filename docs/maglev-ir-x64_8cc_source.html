202void Int32MultiplyWithOverflow::GenerateCode(MaglevAssembler* masm,
203 const ProcessingState& state) {
204 Register result = ToRegister(this->result());
205 Register right = ToRegister(right_input());
206 DCHECK_EQ(result, ToRegister(left_input()));
207
208 MaglevAssembler::TemporaryRegisterScope temps(masm);
209 Register saved_left = temps.Acquire();
210 __ movl(saved_left, result);
211 // TODO(leszeks): peephole optimise multiplication by a constant.
212 __ imull(result, right);
213 // None of the mutated input registers should be a register input into the
214 // eager deopt info.
215 DCHECK_REGLIST_EMPTY(RegList{saved_left, result} &
216 GetGeneralRegistersUsedAsInputs(eager_deopt_info()));
217 __ EmitEagerDeoptIf(overflow, DeoptimizeReason::kOverflow, this);
218
219 // If the result is zero, check if either lhs or rhs is negative.
220 Label end;
221 __ cmpl(result, Immediate(0));
222 __ j(not_zero, &end);
223 {
224 __ orl(saved_left, right);
225 __ cmpl(saved_left, Immediate(0));
226 // If one of them is negative, we must have a -0 result, which is non-int32,
227 // so deopt.
228 // TODO(leszeks): Consider splitting these deopts to have distinct deopt
229 // reasons. Otherwise, the reason has to match the above.
230 __ EmitEagerDeoptIf(less, DeoptimizeReason::kOverflow, this);
231 }
232 __ bind(&end);
233}

244void Int32ModulusWithOverflow::GenerateCode(MaglevAssembler* masm,
245 const ProcessingState& state) {
246 // If AreAliased(lhs, rhs):
247 // deopt if lhs < 0 // Minus zero.
248 // 0
249 //
250 // Otherwise, use the same algorithm as in EffectControlLinearizer:
251 // if rhs <= 0 then
252 // rhs = -rhs
253 // deopt if rhs == 0
254 // if lhs < 0 then
255 // let lhs_abs = -lhs in
256 // let res = lhs_abs % rhs in
257 // deopt if res == 0
258 // -res
259 // else
260 // let msk = rhs - 1 in
261 // if rhs & msk == 0 then
262 // lhs & msk
263 // else
264 // lhs % rhs
265
266 Register lhs = ToRegister(left_input());
267 Register rhs = ToRegister(right_input());
268
269 static constexpr DeoptimizeReason deopt_reason =
270 DeoptimizeReason::kDivisionByZero;
271
272 if (lhs == rhs) {
273 // For the modulus algorithm described above, lhs and rhs must not alias
274 // each other.
275 __ testl(lhs, lhs);
276 // TODO(victorgomes): This ideally should be kMinusZero, but Maglev only
277 // allows one deopt reason per IR.
278 __ EmitEagerDeoptIf(negative, deopt_reason, this);
279 __ Move(ToRegister(result()), 0);
280 return;
281 }
282
283 DCHECK(!AreAliased(lhs, rhs, rax, rdx));
284
285 ZoneLabelRef done(masm);
286 ZoneLabelRef rhs_checked(masm);
287
288 __ cmpl(rhs, Immediate(0));
289 __ JumpToDeferredIf(
290 less_equal,
291 [](MaglevAssembler* masm, ZoneLabelRef rhs_checked, Register rhs,
292 Int32ModulusWithOverflow* node) {
293 __ negl(rhs);
294 __ j(not_zero, *rhs_checked);
295 __ EmitEagerDeopt(node, deopt_reason);
296 },
297 rhs_checked, rhs, this);
298 __ bind(*rhs_checked);
299
300 __ cmpl(lhs, Immediate(0));
301 __ JumpToDeferredIf(
302 less,
303 [](MaglevAssembler* masm, ZoneLabelRef done, Register lhs, Register rhs,
304 Int32ModulusWithOverflow* node) {
305 // `divl(divisor)` divides rdx:rax by the divisor and stores the
306 // quotient in rax, the remainder in rdx.
307 __ movl(rax, lhs);
308 __ negl(rax);
309 __ xorl(rdx, rdx);
310 __ divl(rhs);
311 __ negl(rdx);
312 __ j(not_zero, *done);
313 // TODO(victorgomes): This ideally should be kMinusZero, but Maglev only
314 // allows one deopt reason per IR.
315 __ EmitEagerDeopt(node, deopt_reason);
316 },
317 done, lhs, rhs, this);
318
319 Label rhs_not_power_of_2;
320 Register mask = rax;
321 __ leal(mask, Operand(rhs, -1));
322 __ testl(rhs, mask);
323 __ j(not_zero, &rhs_not_power_of_2, Label::kNear);
324
325 // {rhs} is power of 2.
326 __ andl(mask, lhs);
327 __ movl(ToRegister(result()), mask);
328 __ jmp(*done, Label::kNear);
329
330 __ bind(&rhs_not_power_of_2);
331 // `divl(divisor)` divides rdx:rax by the divisor and stores the
332 // quotient in rax, the remainder in rdx.
333 __ movl(rax, lhs);
334 __ xorl(rdx, rdx);
335 __ divl(rhs);
336 // Result is implicitly written to rdx.
337 DCHECK_EQ(ToRegister(result()), rdx);
338
339 __ bind(*done);
340}

351void Int32DivideWithOverflow::GenerateCode(MaglevAssembler* masm,
352 const ProcessingState& state) {
353 Register left = ToRegister(left_input());
354 Register right = ToRegister(right_input());
355 __ movl(rax, left);
356
357 // TODO(leszeks): peephole optimise division by a constant.
358
359 // Sign extend eax into edx.
360 __ cdq();
361
362 // Pre-check for overflow, since idiv throws a division exception on overflow
363 // rather than setting the overflow flag. Logic copied from
364 // effect-control-linearizer.cc
365
366 // Check if {right} is positive (and not zero).
367 __ cmpl(right, Immediate(0));
368 ZoneLabelRef done(masm);
369 __ JumpToDeferredIf(
370 less_equal,
371 [](MaglevAssembler* masm, ZoneLabelRef done, Register right,
372 Int32DivideWithOverflow* node) {
373 // {right} is negative or zero.
374
375 // Check if {right} is zero.
376 // We've already done the compare and flags won't be cleared yet.
377 // TODO(leszeks): Using kNotInt32 here, but kDivisionByZero would be
378 // better. Right now all eager deopts in a node have to be the same --
379 // we should allow a node to emit multiple eager deopts with different
380 // reasons.
381 __ EmitEagerDeoptIf(equal, DeoptimizeReason::kNotInt32, node);
382
383 // Check if {left} is zero, as that would produce minus zero. Left is in
384 // rax already.
385 __ cmpl(rax, Immediate(0));
386 // TODO(leszeks): Better DeoptimizeReason = kMinusZero.
387 __ EmitEagerDeoptIf(equal, DeoptimizeReason::kNotInt32, node);
388
389 // Check if {left} is kMinInt and {right} is -1, in which case we'd have
390 // to return -kMinInt, which is not representable as Int32.
391 __ cmpl(rax, Immediate(kMinInt));
392 __ j(not_equal, *done);
393 __ cmpl(right, Immediate(-1));
394 __ j(not_equal, *done);
395 // TODO(leszeks): Better DeoptimizeReason = kOverflow, but
396 // eager_deopt_info is already configured as kNotInt32.
397 __ EmitEagerDeopt(node, DeoptimizeReason::kNotInt32);
398 },
399 done, right, this);
400 __ bind(*done);
401
402 // Perform the actual integer division.
403 __ idivl(right);
404
405 // Check that the remainder is zero.
406 __ cmpl(rdx, Immediate(0));
407 // None of the mutated input registers should be a register input into the
408 // eager deopt info.
409 DCHECK_REGLIST_EMPTY(RegList{rax, rdx} &
410 GetGeneralRegistersUsedAsInputs(eager_deopt_info()));
411 __ EmitEagerDeoptIf(not_equal, DeoptimizeReason::kNotInt32, this);
412 DCHECK_EQ(ToRegister(result()), rax);
413}

584void Float64Modulus::GenerateCode(MaglevAssembler* masm,
585 const ProcessingState& state) {
586 // Approach copied from code-generator-x64.cc
587 // Allocate space to use fld to move the value to the FPU stack.
588 __ AllocateStackSpace(kDoubleSize);
589 Operand scratch_stack_space = Operand(rsp, 0);
590 __ Movsd(scratch_stack_space, ToDoubleRegister(right_input()));
591 __ fld_d(scratch_stack_space);
592 __ Movsd(scratch_stack_space, ToDoubleRegister(left_input()));
593 __ fld_d(scratch_stack_space);
594 // Loop while fprem isn't done.
595 Label mod_loop;
596 __ bind(&mod_loop);
597 // This instructions traps on all kinds inputs, but we are assuming the
598 // floating point control word is set to ignore them all.
599 __ fprem();
600 // The following 2 instruction implicitly use rax.
601 __ fnstsw_ax();
602 if (CpuFeatures::IsSupported(SAHF)) {
603 CpuFeatureScope sahf_scope(masm, SAHF);
604 __ sahf();
605 } else {
606 __ shrl(rax, Immediate(8));
607 __ andl(rax, Immediate(0xFF));
608 __ pushq(rax);
609 __ popfq();
610 }
611 __ j(parity_even, &mod_loop);
612 // Move output to stack and clean up.
613 __ fstp(1);
614 __ fstp_d(scratch_stack_space);
615 __ Movsd(ToDoubleRegister(result()), scratch_stack_space);
616 __ addq(rsp, Immediate(kDoubleSize));
617}

636void Float64Round::GenerateCode(MaglevAssembler* masm,
637 const ProcessingState& state) {
638 DoubleRegister in = ToDoubleRegister(input());
639 DoubleRegister out = ToDoubleRegister(result());
640
641 if (kind_ == Kind::kNearest) {
642 MaglevAssembler::TemporaryRegisterScope temps(masm);
643 DoubleRegister temp = temps.AcquireDouble();
644 __ Move(temp, in);
645 __ Roundsd(out, in, kRoundToNearest);
646 // RoundToNearest rounds to even on tie, while JS expects it to round
647 // towards +Infinity. Fix the difference by checking if we rounded down by
648 // exactly 0.5, and if so, round to the other side.
649 __ Subsd(temp, out);
650 __ Move(kScratchDoubleReg, 0.5);
651 Label done;
652 __ Ucomisd(temp, kScratchDoubleReg);
653 __ JumpIf(not_equal, &done, Label::kNear);
654 // Fix wrong tie-to-even by adding 0.5 twice.
655 __ Addsd(out, kScratchDoubleReg);
656 __ Addsd(out, kScratchDoubleReg);
657 __ bind(&done);
658 } else if (kind_ == Kind::kFloor) {
659 __ Roundsd(out, in, kRoundDown);
660 } else if (kind_ == Kind::kCeil) {
661 __ Roundsd(out, in, kRoundUp);
662 }

709enum class ReduceInterruptBudgetType { kLoop, kReturn };
710
711void HandleInterruptsAndTiering(MaglevAssembler* masm, ZoneLabelRef done,
712 Node* node, ReduceInterruptBudgetType type,
713 Register scratch0) {
714 // For loops, first check for interrupts. Don't do this for returns, as we
715 // can't lazy deopt to the end of a return.
716 if (type == ReduceInterruptBudgetType::kLoop) {
717 Label next;
718
719 // Here, we only care about interrupts since we've already guarded against
720 // real stack overflows on function entry.
721 __ cmpq(rsp, __ StackLimitAsOperand(StackLimitKind::kInterruptStackLimit));
722 __ j(above, &next);
723
724 // An interrupt has been requested and we must call into runtime to handle
725 // it; since we already pay the call cost, combine with the TieringManager
726 // call.
727 {
728 SaveRegisterStateForCall save_register_state(masm,
729 node->register_snapshot());
730 __ Move(kContextRegister, masm->native_context().object());
731 __ Push(MemOperand(rbp, StandardFrameConstants::kFunctionOffset));
732 __ CallRuntime(Runtime::kBytecodeBudgetInterruptWithStackCheck_Maglev, 1);
733 save_register_state.DefineSafepointWithLazyDeopt(node->lazy_deopt_info());
734 }
735 __ jmp(*done); // All done, continue.
736
737 __ bind(&next);
738 }
739
740 // No pending interrupts. Call into the TieringManager if needed.
741 {
742 SaveRegisterStateForCall save_register_state(masm,
743 node->register_snapshot());
744 __ Move(kContextRegister, masm->native_context().object());
745 __ Push(MemOperand(rbp, StandardFrameConstants::kFunctionOffset));
746 // Note: must not cause a lazy deopt!
747 __ CallRuntime(Runtime::kBytecodeBudgetInterrupt_Maglev, 1);
748 save_register_state.DefineSafepoint();
749 }
750 __ jmp(*done);
751}

753void GenerateReduceInterruptBudget(MaglevAssembler* masm, Node* node,
754 ReduceInterruptBudgetType type, int amount) {
755 MaglevAssembler::TemporaryRegisterScope temps(masm);
756 Register scratch = temps.Acquire();
757 __ movq(scratch, MemOperand(rbp, StandardFrameConstants::kFunctionOffset));
758 __ LoadTaggedField(scratch,
759 FieldOperand(scratch, JSFunction::kFeedbackCellOffset));
760 __ subl(FieldOperand(scratch, FeedbackCell::kInterruptBudgetOffset),
761 Immediate(amount));
762 ZoneLabelRef done(masm);
763 __ JumpToDeferredIf(less, HandleInterruptsAndTiering, done, node, type,
764 scratch);
765 __ bind(*done);
766}

796void Return::GenerateCode(MaglevAssembler* masm, const ProcessingState& state) {
797 DCHECK_EQ(ToRegister(value_input()), kReturnRegister0);
798
799 // Read the formal number of parameters from the top level compilation unit
800 // (i.e. the outermost, non inlined function).
801 int formal_params_size =
802 masm->compilation_info()->toplevel_compilation_unit()->parameter_count();
803
804 // We're not going to continue execution, so we can use an arbitrary register
805 // here instead of relying on temporaries from the register allocator.
806 Register actual_params_size = r8;
807
808 // Compute the size of the actual parameters + receiver (in bytes).
809 // TODO(leszeks): Consider making this an input into Return to re-use the
810 // incoming argc's register (if it's still valid).
811 __ movq(actual_params_size,
812 MemOperand(rbp, StandardFrameConstants::kArgCOffset));
813
814 // Leave the frame.
815 __ LeaveFrame(StackFrame::MAGLEV);
816
817 // If actual is bigger than formal, then we should use it to free up the stack
818 // arguments.
819 Label drop_dynamic_arg_size;
820 __ cmpq(actual_params_size, Immediate(formal_params_size));
821 __ j(greater, &drop_dynamic_arg_size);
822
823 // Drop receiver + arguments according to static formal arguments size.
824 __ Ret(formal_params_size * kSystemPointerSize, kScratchRegister);
825
826 __ bind(&drop_dynamic_arg_size);
827 // Drop receiver + arguments according to dynamic arguments size.
828 __ DropArguments(actual_params_size, r9);
829 __ Ret();
830}
