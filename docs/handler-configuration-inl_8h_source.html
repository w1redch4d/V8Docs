225Handle<Smi> StoreHandler::StoreField(Isolate* isolate, Kind kind,
226 int descriptor, FieldIndex field_index,
227 Representation representation) {
228 DCHECK(!representation.IsNone());
229 DCHECK(kind == Kind::kField || kind == Kind::kConstField ||
230 kind == Kind::kSharedStructField);
231
232 int config = KindBits::encode(kind) |
233 IsInobjectBits::encode(field_index.is_inobject()) |
234 RepresentationBits::encode(representation.kind()) |
235 DescriptorBits::encode(descriptor) |
236 FieldIndexBits::encode(field_index.index());
237 return handle(Smi::FromInt(config), isolate);
238}
