89template <typename T>
90bool CompareNum(T x, T y) {
91 if (x < y) {
92 return true;
93 } else if (x > y) {
94 return false;
95 } else if (!std::is_integral_v<T>) {
96 double _x = x, _y = y;
97 if (x == 0 && x == y) {
98 /* -0.0 is less than +0.0 */
99 return std::signbit(_x) && !std::signbit(_y);
100 } else if (!std::isnan(_x) && std::isnan(_y)) {
101 /* number is less than NaN */
102 return true;
103 }
104 }
105 return false;
106}

110RUNTIME_FUNCTION(Runtime_TypedArraySortFast) {
111 HandleScope scope(isolate);
112 DCHECK_EQ(1, args.length());
113
114 // Validation is handled in the Torque builtin.
115 DirectHandle<JSTypedArray> array = args.at<JSTypedArray>(0);
116 DCHECK(!array->WasDetached());
117 DCHECK(!array->IsOutOfBounds());
118
119#ifdef V8_OS_LINUX
120 if (v8_flags.multi_mapped_mock_allocator) {
121 // Sorting is meaningless with the mock allocator, and std::sort
122 // might crash (because aliasing elements violate its assumptions).
123 return *array;
124 }
125#endif
126
127 size_t length = array->GetLength();
128 DCHECK_LT(1, length);
129
130 // In case of a SAB, the data is copied into temporary memory, as
131 // std::sort might crash in case the underlying data is concurrently
132 // modified while sorting.
133 CHECK(IsJSArrayBuffer(array->buffer()));
134 DirectHandle<JSArrayBuffer> buffer(Cast<JSArrayBuffer>(array->buffer()),
135 isolate);
136 const bool copy_data = buffer->is_shared();
137
138 DirectHandle<ByteArray> array_copy;
139 std::vector<uint8_t> offheap_copy;
140 void* data_copy_ptr = nullptr;
141 if (copy_data) {
142 const size_t bytes = array->GetByteLength();
143 if (bytes <= static_cast<unsigned>(
144 ByteArray::LengthFor(kMaxRegularHeapObjectSize))) {
145 array_copy = isolate->factory()->NewByteArray(static_cast<int>(bytes));
146 data_copy_ptr = array_copy->begin();
147 } else {
148 // Allocate copy in C++ heap.
149 offheap_copy.resize(bytes);
150 data_copy_ptr = &offheap_copy[0];
151 }
152 base::Relaxed_Memcpy(static_cast<base::Atomic8*>(data_copy_ptr),
153 static_cast<base::Atomic8*>(array->DataPtr()), bytes);
154 }
155
156 DisallowGarbageCollection no_gc;
157
158 switch (array->type()) {
159#define TYPED_ARRAY_SORT(Type, type, TYPE, ctype) \
160 case kExternal##Type##Array: { \
161 ctype* data = copy_data ? reinterpret_cast<ctype*>(data_copy_ptr) \
162 : static_cast<ctype*>(array->DataPtr()); \
163 if (kExternal##Type##Array == kExternalFloat64Array || \
164 kExternal##Type##Array == kExternalFloat32Array || \
165 kExternal##Type##Array == kExternalFloat16Array) { \
166 if (COMPRESS_POINTERS_BOOL && alignof(ctype) > kTaggedSize) { \
167 /* TODO(ishell, v8:8875): See UnalignedSlot<T> for details. */ \
168 std::sort(UnalignedSlot<ctype>(data), \
169 UnalignedSlot<ctype>(data + length), CompareNum<ctype>); \
170 } else { \
171 std::sort(data, data + length, CompareNum<ctype>); \
172 } \
173 } else { \
174 if (COMPRESS_POINTERS_BOOL && alignof(ctype) > kTaggedSize) { \
175 /* TODO(ishell, v8:8875): See UnalignedSlot<T> for details. */ \
176 std::sort(UnalignedSlot<ctype>(data), \
177 UnalignedSlot<ctype>(data + length)); \
178 } else { \
179 std::sort(data, data + length); \
180 } \
181 } \
182 break; \
183 }
184
185 TYPED_ARRAYS(TYPED_ARRAY_SORT)
186#undef TYPED_ARRAY_SORT
187 }
188
189 if (copy_data) {
190 DCHECK_NOT_NULL(data_copy_ptr);
191 DCHECK_NE(array_copy.is_null(), offheap_copy.empty());
192 const size_t bytes = array->GetByteLength();
193 base::Relaxed_Memcpy(static_cast<base::Atomic8*>(array->DataPtr()),
194 static_cast<base::Atomic8*>(data_copy_ptr), bytes);
195 }
196
197 return *array;
198}
