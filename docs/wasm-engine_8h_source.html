70// Native modules cached by their wire bytes and compile-time imports.
71class NativeModuleCache {
72 public:
73 struct Key {
74 Key(size_t prefix_hash, CompileTimeImports compile_imports,
75 const base::Vector<const uint8_t>& bytes)
76 : prefix_hash(prefix_hash),
77 compile_imports(std::move(compile_imports)),
78 bytes(bytes) {}
79
80 // Store the prefix hash as part of the key for faster lookup, and to
81 // quickly check existing prefixes for streaming compilation.
82 size_t prefix_hash;
83 CompileTimeImports compile_imports;
84 base::Vector<const uint8_t> bytes;
85
86 bool operator==(const Key& other) const {
87 bool eq = bytes == other.bytes &&
88 compile_imports.compare(other.compile_imports) == 0;
89 DCHECK_IMPLIES(eq, prefix_hash == other.prefix_hash);
90 return eq;
91 }
92
93 bool operator<(const Key& other) const {
94 if (prefix_hash != other.prefix_hash) {
95 DCHECK_IMPLIES(!bytes.empty() && !other.bytes.empty(),
96 bytes != other.bytes);
97 return prefix_hash < other.prefix_hash;
98 }
99 if (bytes.size() != other.bytes.size()) {
100 return bytes.size() < other.bytes.size();
101 }
102 if (int cmp = compile_imports.compare(other.compile_imports)) {
103 return cmp < 0;
104 }
105 // Fast path when the base pointers are the same.
106 // Also handles the {nullptr} case which would be UB for memcmp.
107 if (bytes.begin() == other.bytes.begin()) {
108 DCHECK_EQ(prefix_hash, other.prefix_hash);
109 return false;
110 }
111 DCHECK_NOT_NULL(bytes.begin());
112 DCHECK_NOT_NULL(other.bytes.begin());
113 return memcmp(bytes.begin(), other.bytes.begin(), bytes.size()) < 0;
114 }
115 };

131 // avoid streaming compilation of modules that are likely already in the

136 private:
137 // Each key points to the corresponding native module's wire bytes, so they
138 // should always be valid as long as the native module is alive. When
139 // the native module dies, {FreeNativeModule} deletes the entry from the
140 // map, so that we do not leave any dangling key pointing to an expired
141 // weak_ptr. This also serves as a way to regularly clean up the map, which
142 // would otherwise accumulate expired entries.
143 // A {nullopt} value is inserted to indicate that this native module is
144 // currently being created in some thread, and that other threads should wait
145 // before trying to get it from the cache.
146 // By contrast, an expired {weak_ptr} indicates that the native module died
147 // and will soon be cleaned up from the cache.
148 std::map<Key, std::optional<std::weak_ptr<NativeModule>>> map_;

153 // module. Only one thread will create the {NativeModule}. Other threads
154 // will wait on this variable until the first thread wakes them up.

311 // Try getting a cached {NativeModule}, or get ownership for its creation.
312 // Return {nullptr} if no {NativeModule} exists for these bytes. In this case,
313 // a {nullopt} entry is added to let other threads know that a {NativeModule}
314 // for these bytes is currently being created. The caller should eventually
315 // call {UpdateNativeModuleCache} to update the entry and wake up other
316 // threads. The {wire_bytes}' underlying array should be valid at least until
317 // the call to {UpdateNativeModuleCache}.
318 // The provided {CompileTimeImports} are considered part of the caching key,
