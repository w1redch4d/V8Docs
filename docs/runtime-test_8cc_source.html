307bool CanOptimizeFunction(CodeKind target_kind,
308 DirectHandle<JSFunction> function, Isolate* isolate,
309 IsCompiledScope* is_compiled_scope) {
310 // The following conditions were lifted (in part) from the DCHECK inside
311 // JSFunction::MarkForOptimization().
312
313 // If function isn't compiled, compile it now.
314 if (!is_compiled_scope->is_compiled() &&
315 !Compiler::Compile(isolate, function, Compiler::CLEAR_EXCEPTION,
316 is_compiled_scope)) {
317 return CrashUnlessFuzzingReturnFalse(isolate);
318 }
319
320 if (target_kind == CodeKind::TURBOFAN_JS && !v8_flags.turbofan) return false;
321 if (target_kind == CodeKind::MAGLEV && !maglev::IsMaglevEnabled()) {
322 return false;
323 }
324
325 if (function->shared()->optimization_disabled() &&
326 function->shared()->disabled_optimization_reason() ==
327 BailoutReason::kNeverOptimize) {
328 return CrashUnlessFuzzingReturnFalse(isolate);
329 }
330
331 if (IsAsmWasmFunction(isolate, *function)) {
332 return CrashUnlessFuzzingReturnFalse(isolate);
333 }
334
335 if (v8_flags.testing_d8_test_runner) {
336 if (!CheckMarkedForManualOptimization(isolate, *function)) {
337 return CrashUnlessFuzzingReturnFalse(isolate);
338 }
339 }
340
341 if (function->is_compiled(isolate) &&
342 !function->HasAvailableCodeKind(isolate,
343 CodeKind::INTERPRETED_FUNCTION)) {
344 return CrashUnlessFuzzingReturnFalse(isolate);
345 }
346
347 if (function->HasAvailableCodeKind(isolate, target_kind) ||
348 function->HasAvailableHigherTierCodeThan(isolate, target_kind)) {
349 DCHECK(function->HasAttachedOptimizedCode(isolate) ||
350 function->ChecksTieringState(isolate));
351 return false;
352 }
353
354 return true;
355}

357Tagged<Object> OptimizeFunctionOnNextCall(RuntimeArguments& args,
358 Isolate* isolate,
359 CodeKind target_kind) {
360 if (args.length() != 1 && args.length() != 2) {
361 return CrashUnlessFuzzing(isolate);
362 }
363
364 DirectHandle<Object> function_object = args.at(0);
365 if (!IsJSFunction(*function_object)) return CrashUnlessFuzzing(isolate);
366 DirectHandle<JSFunction> function = Cast<JSFunction>(function_object);
367
368 IsCompiledScope is_compiled_scope(
369 function->shared()->is_compiled_scope(isolate));
370 if (!CanOptimizeFunction(target_kind, function, isolate,
371 &is_compiled_scope)) {
372 return ReadOnlyRoots(isolate).undefined_value();
373 }
374
375 ConcurrencyMode concurrency_mode = ConcurrencyMode::kSynchronous;
376 if (args.length() == 2) {
377 DirectHandle<Object> type = args.at(1);
378 if (!IsString(*type)) return CrashUnlessFuzzing(isolate);
379 if (Cast<String>(type)->IsOneByteEqualTo(
380 base::StaticCharVector("concurrent")) &&
381 isolate->concurrent_recompilation_enabled()) {
382 concurrency_mode = ConcurrencyMode::kConcurrent;
383 }
384 }
385
386 // This function may not have been lazily compiled yet, even though its shared
387 // function has.
388 if (!function->is_compiled(isolate)) {
389 DCHECK(function->shared()->HasBytecodeArray());
390 Tagged<Code> code = *BUILTIN_CODE(isolate, InterpreterEntryTrampoline);
391 if (function->shared()->HasBaselineCode()) {
392 code = function->shared()->baseline_code(kAcquireLoad);
393 }
394 function->UpdateCode(code);
395 }
396
397 TraceManualRecompile(*function, target_kind, concurrency_mode);
398 JSFunction::EnsureFeedbackVector(isolate, function, &is_compiled_scope);
399 if (function->GetActiveTier(isolate) != target_kind) {
400 function->RequestOptimization(isolate, target_kind, concurrency_mode);
401 }
402
403 return ReadOnlyRoots(isolate).undefined_value();
404}

406bool EnsureCompiledAndFeedbackVector(Isolate* isolate,
407 DirectHandle<JSFunction> function,
408 IsCompiledScope* is_compiled_scope) {
409 *is_compiled_scope =
410 function->shared()->is_compiled_scope(function->GetIsolate());
411
412 // If function isn't compiled, compile it now.
413 if (!is_compiled_scope->is_compiled()) {
414 // Check function allows lazy compilation.
415 DCHECK(function->shared()->allows_lazy_compilation());
416 if (!Compiler::Compile(isolate, function, Compiler::CLEAR_EXCEPTION,
417 is_compiled_scope)) {
418 return false;
419 }
420 }
421
422 // Ensure function has a feedback vector to hold type feedback for
423 // optimization.
424 if (!function->shared()->HasFeedbackMetadata()) {
425 return false;
426 }
427 JSFunction::EnsureFeedbackVector(isolate, function, is_compiled_scope);
428 return true;
429}

656BytecodeOffset OffsetOfNextJumpLoop(Isolate* isolate,
657 Handle<BytecodeArray> bytecode_array,
658 int current_offset) {
659 interpreter::BytecodeArrayIterator it(bytecode_array, current_offset);
660
661 // First, look for a loop that contains the current bytecode offset.
662 for (; !it.done(); it.Advance()) {
663 if (it.current_bytecode() != interpreter::Bytecode::kJumpLoop) {
664 continue;
665 }
666 if (!base::IsInRange(current_offset, it.GetJumpTargetOffset(),
667 it.current_offset())) {
668 continue;
669 }
670
671 return BytecodeOffset(it.current_offset());
672 }
673
674 // Fall back to any loop after the current offset.
675 it.SetOffset(current_offset);
676 for (; !it.done(); it.Advance()) {
677 if (it.current_bytecode() == interpreter::Bytecode::kJumpLoop) {
678 return BytecodeOffset(it.current_offset());
679 }
680 }
681
682 return BytecodeOffset::None();
683}

687RUNTIME_FUNCTION(Runtime_OptimizeOsr) {
688 HandleScope handle_scope(isolate);
689
690 DirectHandle<JSFunction> function;
691
692 // The optional parameter determines the frame being targeted.
693 int stack_depth = 0;
694 if (args.length() == 1) {
695 if (!IsSmi(args[0])) return CrashUnlessFuzzing(isolate);
696 stack_depth = args.smi_value_at(0);
697 }
698
699 // Find the JavaScript function on the top of the stack.
700 JavaScriptStackFrameIterator it(isolate);
701 while (!it.done() && stack_depth--) it.Advance();
702 if (!it.done()) {
703 if (it.frame()->is_turbofan()) {
704 if (v8_flags.trace_osr) {
705 CodeTracer::Scope scope(isolate->GetCodeTracer());
706 PrintF(scope.file(),
707 "[OSR - %%OptimizeOsr failed because the current function could "
708 "not be found.]\n");
709 }
710 // This can happen if %OptimizeOsr is in inlined function.
711 return ReadOnlyRoots(isolate).undefined_value();
712 } else if (it.frame()->is_maglev()) {
713 function = MaglevFrame::cast(it.frame())->GetInnermostFunction();
714 } else {
715 function = direct_handle(it.frame()->function(), isolate);
716 }
717 }
718 if (function.is_null()) return CrashUnlessFuzzing(isolate);
719
720 if (V8_UNLIKELY((!v8_flags.turbofan && !maglev::IsMaglevEnabled()) ||
721 (!v8_flags.use_osr && !maglev::IsMaglevOsrEnabled()))) {
722 return ReadOnlyRoots(isolate).undefined_value();
723 }
724
725 if (!function->shared()->allows_lazy_compilation()) {
726 return CrashUnlessFuzzing(isolate);
727 }
728
729 if (function->shared()->optimization_disabled() &&
730 function->shared()->disabled_optimization_reason() ==
731 BailoutReason::kNeverOptimize) {
732 return CrashUnlessFuzzing(isolate);
733 }
734
735 if (v8_flags.testing_d8_test_runner) {
736 if (!CheckMarkedForManualOptimization(isolate, *function)) {
737 return CrashUnlessFuzzing(isolate);
738 }
739 }
740
741 if (function->HasAvailableOptimizedCode(isolate) &&
742 (!function->code(isolate)->is_maglevved() || !v8_flags.osr_from_maglev)) {
743 DCHECK(function->HasAttachedOptimizedCode(isolate) ||
744 function->ChecksTieringState(isolate));
745 // If function is already optimized, return.
746 return ReadOnlyRoots(isolate).undefined_value();
747 }
748
749 if (!it.frame()->is_unoptimized() &&
750 (!it.frame()->is_maglev() || !v8_flags.osr_from_maglev)) {
751 // Nothing to be done.
752 return ReadOnlyRoots(isolate).undefined_value();
753 }
754
755 IsCompiledScope is_compiled_scope(
756 function->shared()->is_compiled_scope(isolate));
757 JSFunction::EnsureFeedbackVector(isolate, function, &is_compiled_scope);
758 isolate->tiering_manager()->RequestOsrAtNextOpportunity(*function);
759
760 // If concurrent OSR is enabled, the testing workflow is a bit tricky. We
761 // must guarantee that the next JumpLoop installs the finished OSR'd code
762 // object, but we still want to exercise concurrent code paths. To do so,
763 // we attempt to find the next JumpLoop, start an OSR job for it now, and
764 // immediately force finalization.
765 // If this succeeds and we correctly match up the next JumpLoop, once we
766 // reach the JumpLoop we'll hit the OSR cache and install the generated code.
767 // If not (e.g. because we enter a nested loop first), the next JumpLoop will
768 // see the cached OSR code with a mismatched offset, and trigger
769 // non-concurrent OSR compilation and installation.
770 // To tier up from Maglev to TF we always do this, because the non-concurrent
771 // recompilation in `CompileOptimizedOSRFromMaglev` is broken. See the comment
772 // in `runtime-compiler.cc`.
773 bool concurrent_osr =
774 isolate->concurrent_recompilation_enabled() && v8_flags.concurrent_osr;
775 bool is_maglev = false;
776 if (it.frame()->is_maglev() || concurrent_osr) {
777 BytecodeOffset osr_offset = BytecodeOffset::None();
778 if (it.frame()->is_unoptimized()) {
779 UnoptimizedJSFrame* frame = UnoptimizedJSFrame::cast(it.frame());
780 Handle<BytecodeArray> bytecode_array(frame->GetBytecodeArray(), isolate);
781 const int current_offset = frame->GetBytecodeOffset();
782 osr_offset =
783 OffsetOfNextJumpLoop(isolate, bytecode_array, current_offset);
784 } else {
785 MaglevFrame* frame = MaglevFrame::cast(it.frame());
786 Handle<BytecodeArray> bytecode_array(
787 function->shared()->GetBytecodeArray(isolate), isolate);
788 const BytecodeOffset current_offset = frame->GetBytecodeOffsetForOSR();
789 osr_offset = OffsetOfNextJumpLoop(
790 isolate, bytecode_array,
791 current_offset.IsNone() ? 0 : current_offset.ToInt());
792 is_maglev = true;
793 }
794
795 if (osr_offset.IsNone()) {
796 // The loop may have been elided by bytecode generation (e.g. for
797 // patterns such as `do { ... } while (false);` or we are in an inlined
798 // constructor stub.
799 return ReadOnlyRoots(isolate).undefined_value();
800 }
801
802 // Finalize first to ensure all pending tasks are done (since we can't
803 // queue more than one OSR job for each function).
804 if (concurrent_osr) {
805 FinalizeOptimization(isolate);
806 }
807
808 // Queue the job.
809 auto unused_result = Compiler::CompileOptimizedOSR(
810 isolate, function, osr_offset,
811 concurrent_osr ? ConcurrencyMode::kConcurrent
812 : ConcurrencyMode::kSynchronous,
813 (maglev::IsMaglevOsrEnabled() && !it.frame()->is_maglev())
814 ? CodeKind::MAGLEV
815 : CodeKind::TURBOFAN_JS);
816 USE(unused_result);
817
818 // Finalize again to finish the queued job. The next call into
819 // Runtime::kCompileOptimizedOSR will pick up the cached InstructionStream
820 // object.
821 if (concurrent_osr) {
822 FinalizeOptimization(isolate);
823 }
824
825 if (is_maglev) {
826 // Maglev ignores the maybe_has_optimized_osr_code flag, thus we also need
827 // to set a maximum urgency.
828 function->feedback_vector()->set_osr_urgency(
829 FeedbackVector::kMaxOsrUrgency);
830 }
831 }
832
833 return ReadOnlyRoots(isolate).undefined_value();
834}

897RUNTIME_FUNCTION(Runtime_GetOptimizationStatus) {
898 HandleScope scope(isolate);
899 DCHECK_EQ(args.length(), 1);
900
901 int status = 0;
902 if (v8_flags.lite_mode || v8_flags.jitless || !V8_ENABLE_TURBOFAN_BOOL) {
903 // These modes cannot optimize. Unit tests should handle these the same
904 // way.
905 status |= static_cast<int>(OptimizationStatus::kLiteMode);
906 }
907 if (!isolate->use_optimizer()) {
908 status |= static_cast<int>(OptimizationStatus::kNeverOptimize);
909 }
910 if (v8_flags.always_turbofan || v8_flags.prepare_always_turbofan) {
911 status |= static_cast<int>(OptimizationStatus::kAlwaysOptimize);
912 }
913 if (v8_flags.deopt_every_n_times) {
914 status |= static_cast<int>(OptimizationStatus::kMaybeDeopted);
915 }
916 if (v8_flags.optimize_on_next_call_optimizes_to_maglev) {
917 status |= static_cast<int>(
918 OptimizationStatus::kOptimizeOnNextCallOptimizesToMaglev);
919 }
920
921 Handle<Object> function_object = args.at(0);
922 if (IsUndefined(*function_object)) return Smi::FromInt(status);
923 if (!IsJSFunction(*function_object)) return CrashUnlessFuzzing(isolate);
924
925 auto function = Cast<JSFunction>(function_object);
926 status |= static_cast<int>(OptimizationStatus::kIsFunction);
927
928 if (function->has_feedback_vector()) {
929 if (function->tiering_in_progress()) {
930 status |= static_cast<int>(OptimizationStatus::kOptimizingConcurrently);
931 } else if (function->GetRequestedOptimizationIfAny(
932 isolate, ConcurrencyMode::kConcurrent) == CodeKind::MAGLEV) {
933 status |= static_cast<int>(
934 OptimizationStatus::kMarkedForConcurrentMaglevOptimization);
935 } else if (function->GetRequestedOptimizationIfAny(
936 isolate, ConcurrencyMode::kSynchronous) ==
937 CodeKind::MAGLEV) {
938 status |=
939 static_cast<int>(OptimizationStatus::kMarkedForMaglevOptimization);
940 } else if (function->GetRequestedOptimizationIfAny(
941 isolate, ConcurrencyMode::kConcurrent) ==
942 CodeKind::TURBOFAN_JS) {
943 status |= static_cast<int>(
944 OptimizationStatus::kMarkedForConcurrentOptimization);
945 } else if (function->GetRequestedOptimizationIfAny(
946 isolate, ConcurrencyMode::kSynchronous) ==
947 CodeKind::TURBOFAN_JS) {
948 status |= static_cast<int>(OptimizationStatus::kMarkedForOptimization);
949 }
950 }
951
952 if (function->HasAttachedOptimizedCode(isolate)) {
953 Tagged<Code> code = function->code(isolate);
954 if (code->marked_for_deoptimization()) {
955 status |= static_cast<int>(OptimizationStatus::kMarkedForDeoptimization);
956 } else {
957 status |= static_cast<int>(OptimizationStatus::kOptimized);
958 }
959 if (code->is_maglevved()) {
960 status |= static_cast<int>(OptimizationStatus::kMaglevved);
961 } else if (code->is_turbofanned()) {
962 status |= static_cast<int>(OptimizationStatus::kTurboFanned);
963 }
964 }
965 if (function->HasAttachedCodeKind(isolate, CodeKind::BASELINE)) {
966 status |= static_cast<int>(OptimizationStatus::kBaseline);
967 }
968 if (function->ActiveTierIsIgnition(isolate)) {
969 status |= static_cast<int>(OptimizationStatus::kInterpreted);
970 }
971 if (!function->is_compiled(isolate)) {
972 status |= static_cast<int>(OptimizationStatus::kIsLazy);
973 }
974
975 // Additionally, detect activations of this frame on the stack, and report the
976 // status of the topmost frame.
977 JavaScriptFrame* frame = nullptr;
978 JavaScriptStackFrameIterator it(isolate);
979 while (!it.done()) {
980 if (it.frame()->function() == *function) {
981 frame = it.frame();
982 break;
983 }
984 it.Advance();
985 }
986 if (frame != nullptr) {
987 status |= static_cast<int>(OptimizationStatus::kIsExecuting);
988 if (frame->is_turbofan()) {
989 status |=
990 static_cast<int>(OptimizationStatus::kTopmostFrameIsTurboFanned);
991 } else if (frame->is_interpreted()) {
992 status |=
993 static_cast<int>(OptimizationStatus::kTopmostFrameIsInterpreted);
994 } else if (frame->is_baseline()) {
995 status |= static_cast<int>(OptimizationStatus::kTopmostFrameIsBaseline);
996 } else if (frame->is_maglev()) {
997 status |= static_cast<int>(OptimizationStatus::kTopmostFrameIsMaglev);
998 }
999 }
1000
1001 return Smi::FromInt(status);
1002}
