103class StateBase {
104 public:
105 // Objects can either be hidden/visible, or depend on some other nodes while

114 EmbedderNode* node, bool visited)

120 DCHECK_NE(Visibility::kDependentVisibility, visibility);
121 }

125 // processed, see also IsPending() below.
126 bool IsVisited() const { return visited_; }
127
128 // Pending objects are currently being processed as part of the same SCC.
129 bool IsPending() const { return pending_; }

132 auto v = GetVisibility();
133 CHECK_NE(Visibility::kDependentVisibility, v);
134 return v == Visibility::kVisible;

138 CHECK_EQ(Visibility::kVisible, GetVisibility());
139 DCHECK_NULL(node_);

141 }

144 CHECK_EQ(Visibility::kVisible, GetVisibility());
145 return node_;
146 }
147
148 protected:
149 const void* key_;
150 // State count keeps track of node processing order. It is used to create only
151 // dependencies on ancestors in the sub graph which ensures that there will be
152 // no cycles in dependencies.
153 const size_t state_count_;
154
155 Visibility visibility_;
156 StateBase* visibility_dependency_ = nullptr;
157 EmbedderNode* node_;
158 bool visited_;
159 bool pending_ = false;
160
161 Visibility GetVisibility() {
162 FollowDependencies();
163 return visibility_;
164 }
165
166 StateBase* FollowDependencies() {
167 if (visibility_ != Visibility::kDependentVisibility) {
168 CHECK_NULL(visibility_dependency_);
169 return this;
170 }
171 StateBase* current = this;
172 std::vector<StateBase*> dependencies;
173 while (current->visibility_dependency_ &&
174 current->visibility_dependency_ != current) {
175 DCHECK_EQ(Visibility::kDependentVisibility, current->visibility_);
176 dependencies.push_back(current);
177 current = current->visibility_dependency_;
178 }
179 auto new_visibility = Visibility::kDependentVisibility;
180 auto* new_visibility_dependency = current;
181 if (current->visibility_ == Visibility::kVisible) {
182 new_visibility = Visibility::kVisible;
183 new_visibility_dependency = nullptr;
184 } else if (!IsPending()) {
185 DCHECK(IsVisited());
186 // The object was not visible (above case). Having a dependency on itself
187 // or null means no visible object was found.
188 new_visibility = Visibility::kHidden;
189 new_visibility_dependency = nullptr;
190 }
191 current->visibility_ = new_visibility;
192 current->visibility_dependency_ = new_visibility_dependency;
193 for (auto* state : dependencies) {
194 state->visibility_ = new_visibility;
195 state->visibility_dependency_ = new_visibility_dependency;
196 }
197 return current;
198 }
199
200 friend class State;
201};

223 void MarkDependentVisibility(StateBase* dependency) {
224 // Follow and update dependencies as much as possible.
225 dependency = dependency->FollowDependencies();
226 DCHECK(dependency->IsVisited());
227 if (visibility_ == StateBase::Visibility::kVisible) {
228 // Already visible, no dependency needed.
229 DCHECK_NULL(visibility_dependency_);
230 return;
231 }
232 if (dependency->visibility_ == Visibility::kVisible) {
233 // Simple case: Dependency is visible.
234 visibility_ = Visibility::kVisible;
235 visibility_dependency_ = nullptr;
236 return;
237 }
238 if ((visibility_dependency_ &&
239 (visibility_dependency_->state_count_ > dependency->state_count_)) ||
240 (!visibility_dependency_ &&
241 (state_count_ > dependency->state_count_))) {
242 // Only update when new state_count_ < original state_count_. This
243 // ensures that we pick an ancestor as dependency and not a child which
244 // is guaranteed to converge to an answer.
245 //
246 // Dependency is now
247 // a) either pending with unknown visibility (same call chain), or
248 // b) not pending and has defined visibility.
249 //
250 // It's not possible to point to a state that is not pending but has
251 // dependent visibility because dependencies are updated to the top-most
252 // dependency at the beginning of method.
253 if (dependency->IsPending()) {
254 visibility_ = Visibility::kDependentVisibility;
255 visibility_dependency_ = dependency;
256 } else {
257 CHECK_NE(Visibility::kDependentVisibility, dependency->visibility_);
258 if (dependency->visibility_ == Visibility::kVisible) {
259 visibility_ = Visibility::kVisible;
260 visibility_dependency_ = nullptr;
261 }
262 }
263 }

403// The following implements a snapshotting algorithm for C++ objects that also
404// filters strongly-connected components (SCCs) of only "hidden" objects that
405// are not (transitively) referencing any non-hidden objects.
406//
407// C++ objects come in two versions.
408// a. Named objects that have been assigned a name through NameProvider.
409// b. Unnamed objects, that are potentially hidden if the build configuration
410// requires Oilpan to hide such names. Hidden objects have their name
411// set to NameProvider::kHiddenName.
412//
413// The main challenge for the algorithm is to avoid blowing up the final object
414// graph with hidden nodes that do not carry information. For that reason, the
415// algorithm filters SCCs of only hidden objects, e.g.:
416// ... -> (object) -> (object) -> (hidden) -> (hidden)
417// In this case the (hidden) objects are filtered from the graph. The trickiest
418// part is maintaining visibility state for objects referencing other objects
419// that are currently being processed.
420//
421// Main algorithm idea (two passes):
422// 1. First pass marks all non-hidden objects and those that transitively reach
423// non-hidden objects as visible. Details:
424// - Iterate over all objects.
425// - If object is non-hidden mark it as visible and also mark parent as
426// visible if needed.
427// - If object is hidden, traverse children as DFS to find non-hidden
428// objects. Post-order process the objects and mark those objects as
429// visible that have child nodes that are visible themselves.
430// - Maintain an epoch counter (StateStorage::state_count_) to allow
431// deferring the visibility decision to other objects in the same SCC. This
432// is similar to the "lowlink" value in Tarjan's algorithm for SCC.
433// - After the first pass it is guaranteed that all deferred visibility
434// decisions can be resolved.
435// 2. Second pass adds nodes and edges for all visible objects.
436// - Upon first checking the visibility state of an object, all deferred
437// visibility states are resolved.
438//
439// For practical reasons, the recursion is transformed into an iteration. We do
440// do not use plain Tarjan's algorithm to avoid another pass over all nodes to
441// create SCCs.

475 void AddEdge(State& parent, const HeapObjectHeader& header,
476 const std::string& edge_name) {
477 DCHECK(parent.IsVisibleNotDependent());
478 auto& current = states_.GetExistingState(header);
479 if (!current.IsVisibleNotDependent()) return;
480
481 // Both states are visible. Create nodes in case this is the first edge
482 // created for any of them.
483 if (!parent.get_node()) {
484 parent.set_node(AddNode(*parent.header()));
485 }
486 if (!current.get_node()) {
487 current.set_node(AddNode(header));
488 }
489
490 if (!edge_name.empty()) {
491 graph_.AddEdge(parent.get_node(), current.get_node(),
492 parent.get_node()->InternalizeEdgeName(edge_name));
493 } else {
494 graph_.AddEdge(parent.get_node(), current.get_node());

498 void AddEdge(State& parent, const TracedReferenceBase& ref,
499 const std::string& edge_name) {
500 DCHECK(parent.IsVisibleNotDependent());
501 v8::Local<v8::Data> v8_data =
502 ref.Get(reinterpret_cast<v8::Isolate*>(cpp_heap_.isolate()));
503 if (v8_data.IsEmpty()) return;
504
505 if (!parent.get_node()) {
506 parent.set_node(AddNode(*parent.header()));
507 }
508 auto* v8_node = graph_.V8Node(v8_data);
509 if (!edge_name.empty()) {
510 graph_.AddEdge(parent.get_node(), v8_node,
511 parent.get_node()->InternalizeEdgeName(edge_name));
512 } else {
513 graph_.AddEdge(parent.get_node(), v8_node);
514 }
515
516 // Don't merge nodes if edges have a name.
517 if (!edge_name.empty()) return;
518
519 // Try to extract the back reference. If the back reference matches
520 // `parent`, then the nodes are merged.
521 void* back_reference_object = ExtractEmbedderDataBackref(
522 reinterpret_cast<v8::internal::Isolate*>(cpp_heap_.isolate()),
523 cpp_heap_, v8_data);
524 if (!back_reference_object) return;
525 // Only JS objects have back references.
526 DCHECK(v8_data->IsValue() && v8_data.As<v8::Value>()->IsObject());
527
528 auto& back_header = HeapObjectHeader::FromObject(back_reference_object);
529 auto& back_state = states_.GetExistingState(back_header);
530
531 // If the back reference doesn't point to the same header, just return. In
532 // such a case we have stand-alone references to a wrapper.
533 if (parent.header() != back_state.header()) return;
534
535 // Back reference points to parents header. In this case, the nodes should
536 // be merged and query the detachedness state of the embedder.
537 if (!back_state.get_node()) {
538 back_state.set_node(AddNode(back_header));
539 }
540 back_state.get_node()->SetWrapperNode(v8_node);
541
542 auto* profiler =
543 reinterpret_cast<Isolate*>(cpp_heap_.isolate())->heap_profiler();
544 if (profiler->HasGetDetachednessCallback()) {
545 back_state.get_node()->SetDetachedness(
546 profiler->GetDetachedness(v8_data.As<v8::Value>(), 0));
547 }
548 }

617// This visitor can be used stand-alone to handle fully weak and ephemeron
618// containers or as part of the VisibilityVisitor that recursively traverses
619// the object graph.
620class WeakVisitor : public JSVisitor {
621 public:
622 explicit WeakVisitor(CppGraphBuilderImpl& graph_builder)
623 : JSVisitor(cppgc::internal::VisitorFactory::CreateKey()),
624 graph_builder_(graph_builder) {}
625
626 void VisitWeakContainer(const void* object,
627 cppgc::TraceDescriptor strong_desc,
628 cppgc::TraceDescriptor weak_desc, cppgc::WeakCallback,
629 const void*) final {
630 const auto& container_header =
631 HeapObjectHeader::FromObject(strong_desc.base_object_payload);
632 WeakContainerScope weak_container_scope(*this, container_header);
633
634 graph_builder_.VisitWeakContainerForVisibility(container_header);
635
636 if (!weak_desc.callback) {
637 // Weak container does not contribute to liveness.
638 return;
639 }
640 // Heap snapshot is always run after a GC so we know there are no dead
641 // entries in the container.
642 if (object) {
643 // The container will itself be traced strongly via the regular Visit()
644 // handling that iterates over all live objects. The visibility visitor
645 // will thus see (because of strongly treating the container):
646 // 1. the container itself;
647 // 2. for each {key} in container: container->key;
648 // 3. for each {key, value} in container: key->value;
649 //
650 // In case the visitor is used stand-alone, we trace through the container
651 // here to create the same state as we would when the container is traced
652 // separately.
653 container_header.Trace(this);
654 }
655 }
656 void VisitEphemeron(const void* key, const void* value,
657 cppgc::TraceDescriptor value_desc) final {
658 // For ephemerons, the key retains the value.
659 // Key always must be a GarbageCollected object.
660 auto& key_header = HeapObjectHeader::FromObject(key);
661 if (current_weak_container_header_) {
662 graph_builder_.RecordEphemeronKey(*current_weak_container_header_,
663 key_header);
664 }
665 if (!value_desc.base_object_payload) {
666 // Value does not represent an actual GarbageCollected object but rather
667 // should be traced eagerly.
668 graph_builder_.VisitEphemeronWithNonGarbageCollectedValueForVisibility(
669 key_header, value, value_desc);
670 return;
671 }
672 // Regular path where both key and value are GarbageCollected objects.
673 graph_builder_.VisitEphemeronForVisibility(
674 key_header, HeapObjectHeader::FromObject(value));
675 }

829 // lazily, as the parent's visibility depends on its children.
830 graph_builder.workstack_.push_back(std::unique_ptr<WorkstackItemBase>{
831 new VisitationDoneItem(parent_, current_)});

840 current_.UnmarkPending();

845void CppGraphBuilderImpl::VisitForVisibility(State* parent,
846 const HeapObjectHeader& header) {
847 auto& current = states_.GetOrCreateState(header);
848
849 if (current.IsVisited()) {
850 // Avoid traversing into already visited subgraphs and just update the state
851 // based on a previous result.
852 if (parent) {
853 parent->MarkDependentVisibility(&current);
854 }
855 return;
856 }
857
858 current.MarkVisited();
859 if (header.GetName().name_was_hidden) {
860 current.MarkPending();
861 workstack_.push_back(std::unique_ptr<WorkstackItemBase>{
862 new VisitationItem(parent, current)});
863 } else {
864 // No need to mark/unmark pending as the node is immediately processed.
865 current.MarkVisible();
866 // In case the names are visible, the graph is not traversed in this phase.
867 // Explicitly trace one level to handle weak containers.
868 WeakVisitor weak_visitor(*this);
869 header.Trace(&weak_visitor);
870 if (parent) {
871 // Eagerly update a parent object as its visibility state is now fixed.
872 parent->MarkVisible();
873 }
874 }

989void CppGraphBuilderImpl::Run() {
990 // Sweeping from a previous GC might still be running, in which case not all
991 // pages have been returned to spaces yet.
992 cpp_heap_.sweeper().FinishIfRunning();
993 cppgc::subtle::DisallowGarbageCollectionScope no_gc(
994 cpp_heap_.GetHeapHandle());
995 // First pass: Figure out which objects should be included in the graph -- see
996 // class-level comment on CppGraphBuilder.
997 LiveObjectsForVisibilityIterator visitor(*this);
998 visitor.Traverse(cpp_heap_.raw_heap());
999 // Second pass: Add graph nodes for objects that must be shown.
1000 states_.ForAllVisibleStates([this](StateBase* state_base) {
1001 // No roots have been created so far, so all StateBase objects are State.
1002 State& state = *static_cast<State*>(state_base);
1003
1004 // Emit no edges for the contents of the weak containers. For both, fully
1005 // weak and ephemeron containers, the contents should be retained from
1006 // somewhere else.
1007 if (state.IsWeakContainer()) return;
1008
1009 ParentScope parent_scope(state);
1010 GraphBuildingVisitor object_visitor(*this, parent_scope);
1011 if (!state.header()->IsInConstruction()) {
1012 // TODO(mlippautz): Handle in-construction objects.
1013 state.header()->Trace(&object_visitor);
1014 }
1015 state.ForAllEphemeronEdges([this, &state](const HeapObjectHeader& value) {
1016 AddEdge(state, value, "part of key -> value pair in ephemeron table");
1017 });
1018 object_visitor.set_edge_name(
1019 "part of key -> value pair in ephemeron table");
1020 state.ForAllEagerEphemeronEdges(
1021 [&object_visitor](const void* value, cppgc::TraceCallback callback) {
1022 callback(&object_visitor, value);
1023 });
1024 });
1025 // Add roots.
1026 {
1027 ParentScope parent_scope(
1028 states_.CreateRootState(AddRootNode("C++ Persistent roots")));
1029 GraphBuildingRootVisitor root_object_visitor(*this, parent_scope);
1030 cpp_heap_.GetStrongPersistentRegion().Iterate(root_object_visitor);
1031 }
1032 {
1033 ParentScope parent_scope(states_.CreateRootState(
1034 AddRootNode("C++ CrossThreadPersistent roots")));
1035 GraphBuildingRootVisitor root_object_visitor(*this, parent_scope);
1036 cppgc::internal::PersistentRegionLock guard;
1037 cpp_heap_.GetStrongCrossThreadPersistentRegion().Iterate(
1038 root_object_visitor);
1039 }
1040 // Only add stack roots in case the callback is not run from generating a
1041 // snapshot without stack. This avoids adding false-positive edges when
1042 // conservatively scanning the stack.
1043 if (cpp_heap_.isolate()->heap()->IsGCWithMainThreadStack()) {
1044 ParentScope parent_scope(
1045 states_.CreateRootState(AddRootNode("C++ native stack roots")));
1046 GraphBuildingRootVisitor root_object_visitor(*this, parent_scope);
1047 GraphBuildingStackVisitor stack_visitor(*this, cpp_heap_,
1048 root_object_visitor);
1049 cpp_heap_.stack()->IteratePointersUntilMarker(&stack_visitor);
1050 }
1051}
