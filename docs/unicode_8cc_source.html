244void Utf16::ReplaceUnpairedSurrogates(const uint16_t* source_code_units,
245 uint16_t* dest_code_units,
246 size_t length) {
247 // U+FFFD (REPLACEMENT CHARACTER)
248 constexpr uint16_t kReplacement = 0xFFFD;
249
250 for (size_t i = 0; i < length; i++) {
251 const uint16_t source_code_unit = source_code_units[i];
252 const size_t copy_index = i;
253 uint16_t dest_code_unit = source_code_unit;
254 if (IsLeadSurrogate(source_code_unit)) {
255 // The current code unit is a leading surrogate. If it's not followed by a
256 // trailing surrogate, replace it with the replacement character.
257 if (i == length - 1 || !IsTrailSurrogate(source_code_units[i + 1])) {
258 dest_code_unit = kReplacement;
259 } else {
260 // Copy the paired trailing surrogate. The paired leading surrogate will
261 // be copied below.
262 ++i;
263 dest_code_units[i] = source_code_units[i];
264 }
265 } else if (IsTrailSurrogate(source_code_unit)) {
266 // All paired trailing surrogates are skipped above, so this branch is
267 // only for those that are unpaired.
268 dest_code_unit = kReplacement;
269 }
270 dest_code_units[copy_index] = dest_code_unit;
271 }

298 // A surrogate codepoint is encoded in a three-byte sequence:
299 //
300 // 0xED [0xA0,0xBF] [0x80,0xBF]
301 //
302 // If the first byte is 0xED, you already have a 50% chance of the value being
303 // a surrogate; you just have to check the second byte. (There are
304 // three-byte non-surrogates starting with 0xED whose second byte is in
305 // [0x80,0x9F].) Could speed this up with SWAR; most likely case is that no
306 // byte in the array is 0xED.
307 const uint8_t kWtf8SurrogateFirstByte = 0xED;
308 const uint8_t kWtf8SurrogateSecondByteHighBit = 0x20;
309
310 for (size_t i = 0; i < wtf8.size(); i++) {
311 if (wtf8[i] == kWtf8SurrogateFirstByte &&
312 (wtf8[i + 1] & kWtf8SurrogateSecondByteHighBit)) {
313 // Record the byte offset of the encoded surrogate.
