49void GeneratorBuiltinsAssembler::InnerResume(
50 CodeStubArguments* args, TNode<JSGeneratorObject> receiver,
51 TNode<Object> value, TNode<Context> context,
52 JSGeneratorObject::ResumeMode resume_mode, char const* const method_name) {
53 // Check if the {receiver} is running or already closed.
54 TNode<Smi> receiver_continuation =
55 LoadObjectField<Smi>(receiver, JSGeneratorObject::kContinuationOffset);
56 Label if_receiverisclosed(this, Label::kDeferred),
57 if_receiverisrunning(this, Label::kDeferred);
58 TNode<Smi> closed = SmiConstant(JSGeneratorObject::kGeneratorClosed);
59 GotoIf(SmiEqual(receiver_continuation, closed), &if_receiverisclosed);
60 DCHECK_LT(JSGeneratorObject::kGeneratorExecuting,
61 JSGeneratorObject::kGeneratorClosed);
62 GotoIf(SmiLessThan(receiver_continuation, closed), &if_receiverisrunning);
63
64 // Remember the {resume_mode} for the {receiver}.
65 StoreObjectFieldNoWriteBarrier(receiver, JSGeneratorObject::kResumeModeOffset,
66 SmiConstant(resume_mode));
67
68 // Resume the {receiver} using our trampoline.
69 // Close the generator if there was an exception.
70 TVARIABLE(Object, var_exception);
71 Label if_exception(this, Label::kDeferred), if_final_return(this);
72 TNode<Object> result;
73 {
74 compiler::ScopedExceptionHandler handler(this, &if_exception,
75 &var_exception);
76 result = CallBuiltin(Builtin::kResumeGeneratorTrampoline, context, value,
77 receiver);
78 }
79
80 // If the generator is not suspended (i.e., its state is 'executing'),
81 // close it and wrap the return value in IteratorResult.
82 TNode<Smi> result_continuation =
83 LoadObjectField<Smi>(receiver, JSGeneratorObject::kContinuationOffset);
84
85 // The generator function should not close the generator by itself, let's
86 // check it is indeed not closed yet.
87 CSA_DCHECK(this, SmiNotEqual(result_continuation, closed));
88
89 TNode<Smi> executing = SmiConstant(JSGeneratorObject::kGeneratorExecuting);
90 GotoIf(SmiEqual(result_continuation, executing), &if_final_return);
91
92 args->PopAndReturn(result);
93
94 BIND(&if_final_return);
95 {
96 // Close the generator.
97 StoreObjectFieldNoWriteBarrier(
98 receiver, JSGeneratorObject::kContinuationOffset, closed);
99 // Return the wrapped result.
100 args->PopAndReturn(CallBuiltin(Builtin::kCreateIterResultObject, context,
101 result, TrueConstant()));
102 }
103
104 BIND(&if_receiverisclosed);
105 {
106 // The {receiver} is closed already.
107 TNode<Object> builtin_result;
108 switch (resume_mode) {
109 case JSGeneratorObject::kNext:
110 builtin_result = CallBuiltin(Builtin::kCreateIterResultObject, context,
111 UndefinedConstant(), TrueConstant());
112 break;
113 case JSGeneratorObject::kReturn:
114 builtin_result = CallBuiltin(Builtin::kCreateIterResultObject, context,
115 value, TrueConstant());
116 break;
117 case JSGeneratorObject::kThrow:
118 builtin_result = CallRuntime(Runtime::kThrow, context, value);
119 break;
120 case JSGeneratorObject::kRethrow:
121 // Currently only async generators use this mode.
122 UNREACHABLE();
123 }
124 args->PopAndReturn(builtin_result);
125 }
126
127 BIND(&if_receiverisrunning);
128 { ThrowTypeError(context, MessageTemplate::kGeneratorRunning); }
129
130 BIND(&if_exception);
131 {
132 StoreObjectFieldNoWriteBarrier(
133 receiver, JSGeneratorObject::kContinuationOffset, closed);
134 CallRuntime(Runtime::kReThrow, context, var_exception.value());
135 Unreachable();
136 }
137}

219TF_BUILTIN(SuspendGeneratorBaseline, GeneratorBuiltinsAssembler) {
220 auto generator = Parameter<JSGeneratorObject>(Descriptor::kGeneratorObject);
221 auto context = LoadContextFromBaseline();
222 StoreJSGeneratorObjectContext(generator, context);
223 auto parameter_count = LoadParameterCountWithoutReceiverFromBaseline();
224 auto suspend_id = SmiTag(UncheckedParameter<IntPtrT>(Descriptor::kSuspendId));
225 StoreJSGeneratorObjectContinuation(generator, suspend_id);
226 // Store the bytecode offset in the [input_or_debug_pos] field, to be used by
227 // the inspector.
228 auto bytecode_offset =
229 SmiTag(UncheckedParameter<IntPtrT>(Descriptor::kBytecodeOffset));
230 // Avoid the write barrier by using the generic helper.
231 StoreObjectFieldNoWriteBarrier(
232 generator, JSGeneratorObject::kInputOrDebugPosOffset, bytecode_offset);
233
234 TNode<FixedArray> parameters_and_registers =
235 LoadJSGeneratorObjectParametersAndRegisters(generator);
236 auto parameters_and_registers_length =
237 LoadAndUntagFixedArrayBaseLength(parameters_and_registers);
238
239 // Copy over the function parameters
240 auto parameter_base_index = IntPtrConstant(
241 interpreter::Register::FromParameterIndex(0).ToOperand() + 1);
242 CSA_CHECK(this,
243 UintPtrLessThan(parameter_count, parameters_and_registers_length));
244 auto parent_frame_pointer = LoadParentFramePointer();
245 BuildFastLoop<IntPtrT>(
246 IntPtrConstant(0), parameter_count,
247 [=, this](TNode<IntPtrT> index) {
248 auto reg_index = IntPtrAdd(parameter_base_index, index);
249 TNode<Object> value = LoadFullTagged(parent_frame_pointer,
250 TimesSystemPointerSize(reg_index));
251 UnsafeStoreFixedArrayElement(parameters_and_registers, index, value);
252 },
253 1, LoopUnrollingMode::kNo, IndexAdvanceMode::kPost);
254
255 // Iterate over register file and write values into array.
256 // The mapping of register to array index must match that used in
257 // BytecodeGraphBuilder::VisitResumeGenerator.
258 auto register_base_index = IntPtrAdd(
259 parameter_count, IntPtrConstant(interpreter::Register(0).ToOperand()));
260 auto register_count = UncheckedParameter<IntPtrT>(Descriptor::kRegisterCount);
261 auto end_index = IntPtrAdd(parameter_count, register_count);
262 CSA_CHECK(this, UintPtrLessThan(end_index, parameters_and_registers_length));
263 BuildFastLoop<IntPtrT>(
264 parameter_count, end_index,
265 [=, this](TNode<IntPtrT> index) {
266 auto reg_index = IntPtrSub(register_base_index, index);
267 TNode<Object> value = LoadFullTagged(parent_frame_pointer,
268 TimesSystemPointerSize(reg_index));
269 UnsafeStoreFixedArrayElement(parameters_and_registers, index, value);
270 },
271 1, LoopUnrollingMode::kNo, IndexAdvanceMode::kPost);
272
273 // The return value is unused, defaulting to undefined.
274 Return(UndefinedConstant());
275}

278TF_BUILTIN(ResumeGeneratorBaseline, GeneratorBuiltinsAssembler) {
279 auto generator = Parameter<JSGeneratorObject>(Descriptor::kGeneratorObject);
280 auto parameter_count = LoadParameterCountWithoutReceiverFromBaseline();
281
282 TNode<FixedArray> parameters_and_registers =
283 LoadJSGeneratorObjectParametersAndRegisters(generator);
284
285 // Iterate over array and write values into register file. Also erase the
286 // array contents to not keep them alive artificially.
287 auto register_base_index = IntPtrAdd(
288 parameter_count, IntPtrConstant(interpreter::Register(0).ToOperand()));
289 auto register_count = UncheckedParameter<IntPtrT>(Descriptor::kRegisterCount);
290 auto end_index = IntPtrAdd(parameter_count, register_count);
291 auto parameters_and_registers_length =
292 LoadAndUntagFixedArrayBaseLength(parameters_and_registers);
293 CSA_CHECK(this, UintPtrLessThan(end_index, parameters_and_registers_length));
294 auto parent_frame_pointer = LoadParentFramePointer();
295 BuildFastLoop<IntPtrT>(
296 parameter_count, end_index,
297 [=, this](TNode<IntPtrT> index) {
298 TNode<Object> value =
299 UnsafeLoadFixedArrayElement(parameters_and_registers, index);
300 auto reg_index = IntPtrSub(register_base_index, index);
301 StoreFullTaggedNoWriteBarrier(parent_frame_pointer,
302 TimesSystemPointerSize(reg_index), value);
303 UnsafeStoreFixedArrayElement(parameters_and_registers, index,
304 StaleRegisterConstant(),
305 SKIP_WRITE_BARRIER);
306 },
307 1, LoopUnrollingMode::kNo, IndexAdvanceMode::kPost);
308
309 Return(LoadJSGeneratorObjectInputOrDebugPos(generator));
310}
