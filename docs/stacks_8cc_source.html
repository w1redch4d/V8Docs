31StackMemory::StackMemory() : owned_(true) {
32 static std::atomic<int> next_id(1);
33 id_ = next_id.fetch_add(1);
34 size_t kJsStackSizeKB = v8_flags.wasm_stack_switching_stack_size;
35 // v8_flags.stack_size is a size of the central stack and maximum
36 // size of a secondary stack to grow.
37 const size_t size_limit = v8_flags.stack_size;
38 PageAllocator* allocator = GetPlatformPageAllocator();
39 auto page_size = allocator->AllocatePageSize();
40 size_t initial_size =
41 std::min<size_t>(size_limit, kJsStackSizeKB + kJSLimitOffsetKB) * KB;
42 first_segment_ =
43 new StackSegment(RoundUp(initial_size, page_size) / page_size);
44 active_segment_ = first_segment_;
45 size_ = first_segment_->size_;
46 limit_ = first_segment_->limit_;
47 if (v8_flags.trace_wasm_stack_switching) {
48 PrintF("Allocate stack #%d (limit: %p, base: %p, size: %zu)\n", id_, limit_,
49 limit_ + size_, size_);
50 }
51}

59StackMemory::StackSegment::StackSegment(size_t pages) {
60 DCHECK_GE(pages, 1);

75bool StackMemory::Grow(Address current_fp) {
76 DCHECK(owned_);
77 if (active_segment_->next_segment_ != nullptr) {
78 active_segment_ = active_segment_->next_segment_;
79 } else {
80 PageAllocator* allocator = GetPlatformPageAllocator();
81 auto page_size = allocator->AllocatePageSize();
82 const size_t size_limit = RoundUp(v8_flags.stack_size * KB, page_size);
83 DCHECK_GE(size_limit, size_);
84 size_t room_to_grow = size_limit - size_;
85 size_t new_size = std::min(2 * active_segment_->size_, room_to_grow);
86 if (new_size < page_size) {
87 // We cannot grow less than page size.
88 if (v8_flags.trace_wasm_stack_switching) {
89 PrintF("Stack #%d reached the grow limit %zu bytes\n", id_, size_limit);
90 }
91 return false;
92 }
93 auto new_segment = new StackSegment(new_size / page_size);
94 new_segment->prev_segment_ = active_segment_;
95 active_segment_->next_segment_ = new_segment;
96 active_segment_ = new_segment;
97 }
98 active_segment_->old_fp = current_fp;
99 size_ += active_segment_->size_;
100 if (v8_flags.trace_wasm_stack_switching) {
101 PrintF("Grow stack #%d by %zu bytes (limit: %p, base: %p)\n", id_,
102 active_segment_->size_, active_segment_->limit_,
103 active_segment_->limit_ + active_segment_->size_);
104 }
105 return true;
106}

108Address StackMemory::Shrink() {
109 DCHECK(owned_);
110 DCHECK_NE(active_segment_->prev_segment_, nullptr);
111 Address old_fp = active_segment_->old_fp;
112 size_ -= active_segment_->size_;
113 active_segment_->old_fp = 0;
114 active_segment_ = active_segment_->prev_segment_;
115 if (v8_flags.trace_wasm_stack_switching) {
116 PrintF("Shrink stack #%d (limit: %p, base: %p)\n", id_,
117 active_segment_->limit_,
118 active_segment_->limit_ + active_segment_->size_);
119 }
120 return old_fp;
121}

128std::unique_ptr<StackMemory> StackPool::GetOrAllocate() {
129 while (size_ > kMaxSize) {
130 size_ -= freelist_.back()->allocated_size();
131 freelist_.pop_back();
132 }
133 std::unique_ptr<StackMemory> stack;
134 if (freelist_.empty()) {
135 stack = StackMemory::New();
136 } else {
137 stack = std::move(freelist_.back());
138 freelist_.pop_back();
139 size_ -= stack->allocated_size();
140 }
141#if DEBUG
142 constexpr uint8_t kZapValue = 0xab;
143 stack->FillWith(kZapValue);
144#endif
145 return stack;
146}

148void StackPool::Add(std::unique_ptr<StackMemory> stack) {
149 // Add the stack to the pool regardless of kMaxSize, because the stack might
150 // still be in use by the unwinder.
151 // Shrink the freelist lazily when we get the next stack instead.
152 size_ += stack->allocated_size();
153 stack->Reset();
154 freelist_.push_back(std::move(stack));
155}

157void StackPool::ReleaseFinishedStacks() {
158 size_ = 0;
159 freelist_.clear();
160}

162size_t StackPool::Size() const {
