97// Parameter |roots| only here for compatibility with HashTable<...>::ToKey.
98template <class Derived, int entrysize>
99bool OrderedHashTable<Derived, entrysize>::ToKey(ReadOnlyRoots roots,
100 InternalIndex entry,
101 Tagged<Object>* out_key) {
102 Tagged<Object> k = KeyAt(entry);
103 if (!IsKey(roots, k)) return false;
104 *out_key = k;
105 return true;

179template <class Derived>
180void SmallOrderedHashTable<Derived>::SetDataEntry(int entry, int relative_index,
181 Tagged<Object> value) {
182 DCHECK_NE(kNotFound, entry);
183 int entry_offset = GetDataEntryOffset(entry, relative_index);
184 RELAXED_WRITE_FIELD(*this, entry_offset, value);
185 WRITE_BARRIER(*this, entry_offset, value);
186}

188template <class Derived, class TableType>
189Tagged<Object> OrderedHashTableIterator<Derived, TableType>::CurrentKey() {
190 Tagged<TableType> table = Cast<TableType>(this->table());
191 int index = Smi::ToInt(this->index());
192 DCHECK_LE(0, index);
193 InternalIndex entry(index);
194 Tagged<Object> key = table->KeyAt(entry);
195 DCHECK(!IsHashTableHole(key));
196 return key;
197}
