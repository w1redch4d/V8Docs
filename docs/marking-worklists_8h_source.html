170template <AccessMode mode>
171void MarkingWorklists::ExternalMarkingWorklist::Push(HeapObjectHeader* object) {
172 DCHECK_NOT_NULL(object);
173 ConditionalMutexGuard<mode> guard(&lock_);
174 objects_.insert(object);
175}
176
177template <AccessMode mode>
178bool MarkingWorklists::ExternalMarkingWorklist::Contains(
179 HeapObjectHeader* object) {
180 ConditionalMutexGuard<mode> guard(&lock_);
181 return objects_.find(object) != objects_.end();
182}
183
184template <AccessMode mode>
185std::unordered_set<HeapObjectHeader*>
186MarkingWorklists::ExternalMarkingWorklist::Extract() {
187 ConditionalMutexGuard<mode> guard(&lock_);
188 std::unordered_set<HeapObjectHeader*> extracted;
189 std::swap(extracted, objects_);
190 DCHECK(objects_.empty());
191 return extracted;
192}
193
194template <AccessMode mode>
195void MarkingWorklists::ExternalMarkingWorklist::Clear() {
196 ConditionalMutexGuard<mode> guard(&lock_);
197 objects_.clear();
198}

200template <AccessMode mode>
201bool MarkingWorklists::ExternalMarkingWorklist::IsEmpty() {
