33// HeapObjectHeader contains meta data per object and is prepended to each
34// object.
35//
36// +-----------------+------+------------------------------------------+
37// | name | bits | |
38// +-----------------+------+------------------------------------------+
39// | padding | 32 | Only present on 64-bit platform. |
40// +-----------------+------+------------------------------------------+
41// | GCInfoIndex | 14 | |
42// | unused | 1 | |
43// | in construction | 1 | In construction encoded as |false|. |
44// +-----------------+------+------------------------------------------+
45// | size | 15 | 17 bits because allocations are aligned. |
46// | mark bit | 1 | |
47// +-----------------+------+------------------------------------------+
48//
49// Notes:
50// - See |GCInfoTable| for constraints on GCInfoIndex.
51// - |size| for regular objects is encoded with 15 bits but can actually
52// represent sizes up to |kBlinkPageSize| (2^17) because allocations are
53// always 4 byte aligned (see kAllocationGranularity) on 32bit. 64bit uses
54// 8 byte aligned allocations which leaves 1 bit unused.
55// - |size| for large objects is encoded as 0. The size of a large object is
56// stored in |LargeObjectPage::PayloadSize()|.
57// - |mark bit| and |in construction| bits are located in separate 16-bit halves
58// to allow potentially accessing them non-atomically.
59class HeapObjectHeader {
60 public:

62 static constexpr size_t kMaxSize = (size_t{1} << kSizeLog2) - 1;
63 static constexpr uint16_t kLargeObjectSizeInHeader = 0;

126 private:
127 enum class EncodedHalf : uint8_t { kLow, kHigh };
128
129 // Used in |encoded_high_|.
130 using FullyConstructedField = v8::base::BitField16<bool, 0, 1>;
131 using UnusedField1 = FullyConstructedField::Next<bool, 1>;
132 using GCInfoIndexField = UnusedField1::Next<GCInfoIndex, 14>;
133 // Used in |encoded_low_|.
134 using MarkBitField = v8::base::BitField16<bool, 0, 1>;
135 using SizeField =
136 MarkBitField::Next<size_t, 15>; // Use EncodeSize/DecodeSize instead.
137
138 static constexpr size_t DecodeSize(uint16_t encoded) {
139 // Essentially, gets optimized to << 1.
140 return SizeField::decode(encoded) * kAllocationGranularity;
141 }
142
143 static constexpr uint16_t EncodeSize(size_t size) {
144 // Essentially, gets optimized to >> 1.
145 return SizeField::encode(size / kAllocationGranularity);
146 }

186HeapObjectHeader::HeapObjectHeader(size_t size, GCInfoIndex gc_info_index) {
187#if defined(V8_HOST_ARCH_64_BIT) && !defined(CPPGC_CAGED_HEAP)
188 USE(padding_);
189#endif // defined(V8_HOST_ARCH_64_BIT) && !defined(CPPGC_CAGED_HEAP)
190 DCHECK_LT(gc_info_index, GCInfoTable::kMaxIndex);
191 DCHECK_EQ(0u, size & (sizeof(HeapObjectHeader) - 1));
192 DCHECK_GE(kMaxSize, size);
193 encoded_low_ = EncodeSize(size);
194 // Objects may get published to the marker without any other synchronization
195 // (e.g., write barrier) in which case the in-construction bit is read
196 // concurrently which requires reading encoded_high_ atomically. It is ok if
197 // this write is not observed by the marker, since the sweeper sets the
198 // in-construction bit to 0 and we can rely on that to guarantee a correct
199 // answer when checking if objects are in-construction.
200 v8::base::AsAtomicPtr(&encoded_high_)
201 ->store(GCInfoIndexField::encode(gc_info_index),
202 std::memory_order_relaxed);
203 DCHECK(IsInConstruction());
204#ifdef DEBUG
205 CheckApiConstants();
206#endif // DEBUG
207}

228template <AccessMode mode>
229size_t HeapObjectHeader::AllocatedSize() const {
230 // Size is immutable after construction while either marking or sweeping
231 // is running so relaxed load (if mode == kAtomic) is enough.
232 uint16_t encoded_low_value =
233 LoadEncoded<mode, EncodedHalf::kLow, std::memory_order_relaxed>();
234 const size_t size = DecodeSize(encoded_low_value);
235 return size;
236}

238void HeapObjectHeader::SetAllocatedSize(size_t size) {
239#if !defined(CPPGC_YOUNG_GENERATION)
240 // With sticky bits, marked objects correspond to old objects.
241 // TODO(bikineev:1029379): Consider disallowing old/marked objects to be
242 // resized.
243 DCHECK(!IsMarked());
244#endif
245 // The object may be marked (i.e. old, in case young generation is enabled).
246 // Make sure to not overwrite the mark bit.
247 encoded_low_ &= ~SizeField::encode(SizeField::kMax);
248 encoded_low_ |= EncodeSize(size);
249}

289bool HeapObjectHeader::TryMarkAtomic() {
290 auto* atomic_encoded = v8::base::AsAtomicPtr(&encoded_low_);
291 uint16_t old_value = atomic_encoded->load(std::memory_order_relaxed);
292 const uint16_t new_value = old_value | MarkBitField::encode(true);
293 if (new_value == old_value) {
294 return false;
295 }
296 return atomic_encoded->compare_exchange_strong(old_value, new_value,
297 std::memory_order_relaxed);
298}

355template <AccessMode mode, HeapObjectHeader::EncodedHalf part,
356 std::memory_order memory_order>
357uint16_t HeapObjectHeader::LoadEncoded() const {
358 const uint16_t& half =
359 part == EncodedHalf::kLow ? encoded_low_ : encoded_high_;
360 if (mode == AccessMode::kNonAtomic) return half;
361 return v8::base::AsAtomicPtr(&half)->load(memory_order);
362}

364template <AccessMode mode, HeapObjectHeader::EncodedHalf part,
365 std::memory_order memory_order>
366void HeapObjectHeader::StoreEncoded(uint16_t bits, uint16_t mask) {
367 // Caveat: Not all changes to HeapObjectHeader's bitfields go through
368 // StoreEncoded. The following have their own implementations and need to be
369 // kept in sync:
370 // - HeapObjectHeader::TryMarkAtomic
371 // - MarkObjectAsFullyConstructed (API)
372 DCHECK_EQ(0u, bits & ~mask);
373 uint16_t& half = part == EncodedHalf::kLow ? encoded_low_ : encoded_high_;
374 if (mode == AccessMode::kNonAtomic) {
375 half = (half & ~mask) | bits;
376 return;
377 }
378 // We don't perform CAS loop here assuming that only none of the info that
379 // shares the same encoded halfs change at the same time.
380 auto* atomic_encoded = v8::base::AsAtomicPtr(&half);
381 uint16_t value = atomic_encoded->load(std::memory_order_relaxed);
382 value = (value & ~mask) | bits;
383 atomic_encoded->store(value, memory_order);
384}
