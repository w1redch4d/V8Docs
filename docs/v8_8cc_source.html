55enum class V8StartupState {
56 kIdle,
57 kPlatformInitializing,
58 kPlatformInitialized,
59 kV8Initializing,
60 kV8Initialized,
61 kV8Disposing,
62 kV8Disposed,
63 kPlatformDisposing,
64 kPlatformDisposed
65};
66
67std::atomic<V8StartupState> v8_startup_state_(V8StartupState::kIdle);
68
69void AdvanceStartupState(V8StartupState expected_next_state) {
70 V8StartupState current_state = v8_startup_state_;
71 CHECK_NE(current_state, V8StartupState::kPlatformDisposed);
72 V8StartupState next_state =
73 static_cast<V8StartupState>(static_cast<int>(current_state) + 1);
74 if (next_state != expected_next_state) {
75 // Ensure the following order:
76 // v8::V8::InitializePlatform(platform);
77 // v8::V8::Initialize();
78 // v8::Isolate* isolate = v8::Isolate::New(...);
79 // ...
80 // isolate->Dispose();
81 // v8::V8::Dispose();
82 // v8::V8::DisposePlatform();
83 FATAL("Wrong initialization order: from %d to %d, expected to %d!",
84 static_cast<int>(current_state), static_cast<int>(next_state),
85 static_cast<int>(expected_next_state));
86 }
87 if (!v8_startup_state_.compare_exchange_strong(current_state, next_state)) {
88 FATAL(
89 "Multiple threads are initializating V8 in the wrong order: expected "
90 "%d got %d!",
91 static_cast<int>(current_state),
92 static_cast<int>(v8_startup_state_.load()));
93 }
94}

103void V8::InitializePlatform(v8::Platform* platform) {
104 AdvanceStartupState(V8StartupState::kPlatformInitializing);
105 CHECK(!platform_);
106 CHECK_NOT_NULL(platform);
107 platform_ = platform;
108 v8::base::SetPrintStackTrace(platform_->GetStackTracePrinter());
109 v8::tracing::TracingCategoryObserver::SetUp();
110#if defined(V8_ENABLE_ETW_STACK_WALKING)
111 if (v8_flags.enable_etw_stack_walking ||
112 v8_flags.enable_etw_by_custom_filter_only) {
113 v8::internal::ETWJITInterface::Register();
114 }
115#endif // V8_ENABLE_ETW_STACK_WALKING
116
117 // Initialization needs to happen on platform-level, as this sets up some
118 // cppgc internals that are needed to allow gracefully failing during cppgc
119 // platform setup.
120 CppHeap::InitializeOncePerProcess();
121
122 AdvanceStartupState(V8StartupState::kPlatformInitialized);
123}

142void V8::Initialize() {
143 AdvanceStartupState(V8StartupState::kV8Initializing);
144 CHECK(platform_);
145
146 FlagList::EnforceFlagImplications();
147
148 // Initialize the default FlagList::Hash.
149 FlagList::Hash();
150
151 // Before initializing internals, freeze the flags such that further changes
152 // are not allowed. Global initialization of the Isolate or the WasmEngine
153 // already reads flags, so they should not be changed afterwards.
154 if (v8_flags.freeze_flags_after_init) FlagList::FreezeFlags();
155
156 if (v8_flags.trace_turbo) {
157 // Create an empty file shared by the process (e.g. the wasm engine).
158 std::ofstream(Isolate::GetTurboCfgFileName(nullptr).c_str(),
159 std::ios_base::trunc);
160 }
161
162 // The --jitless and --interpreted-frames-native-stack flags are incompatible
163 // since the latter requires code generation while the former prohibits code
164 // generation.
165 CHECK(!v8_flags.interpreted_frames_native_stack || !v8_flags.jitless);
166
167 base::AbortMode abort_mode = base::AbortMode::kDefault;
168
169 if (v8_flags.sandbox_fuzzing || v8_flags.hole_fuzzing) {
170 // In this mode, controlled crashes are harmless. Furthermore, DCHECK
171 // failures should be ignored (and execution should continue past them) as
172 // they may otherwise hide issues.
173 abort_mode = base::AbortMode::kExitWithFailureAndIgnoreDcheckFailures;
174 } else if (v8_flags.sandbox_testing) {
175 // Similar to the above case, but here we want to exit with a status
176 // indicating success (e.g. zero on unix). This is useful for example for
177 // sandbox regression tests, which should "pass" if they crash in a
178 // controlled fashion (e.g. in a SBXCHECK).
179 abort_mode = base::AbortMode::kExitWithSuccessAndIgnoreDcheckFailures;
180 } else if (v8_flags.hard_abort) {
181 abort_mode = base::AbortMode::kImmediateCrash;
182 }
183
184 base::OS::Initialize(abort_mode, v8_flags.gc_fake_mmap);
185
186 if (v8_flags.random_seed) {
187 GetPlatformPageAllocator()->SetRandomMmapSeed(v8_flags.random_seed);
188 GetPlatformVirtualAddressSpace()->SetRandomSeed(v8_flags.random_seed);
189 }
190
191 if (v8_flags.print_flag_values) FlagList::PrintValues();
192
193 // Fetch the ThreadIsolatedAllocator once since we need to keep the pointer in
194 // protected memory.
195 ThreadIsolation::Initialize(
196 GetCurrentPlatform()->GetThreadIsolatedAllocator());
197
198#ifdef V8_ENABLE_SANDBOX
199 // If enabled, the sandbox must be initialized first.
200 Sandbox::InitializeDefaultOncePerProcess(GetPlatformVirtualAddressSpace());
201 CHECK_EQ(kSandboxSize, Sandbox::current()->size());
202
203 // Enable sandbox testing mode if requested.
204 //
205 // This will install the sandbox crash filter to ignore all crashes that do
206 // not represent sandbox violations.
207 //
208 // Note: this should happen before the Wasm trap handler is installed, so that
209 // the wasm trap handler is invoked first (and can handle Wasm OOB accesses),
210 // then forwards all "real" crashes to the sandbox crash filter.
211 if (v8_flags.sandbox_testing || v8_flags.sandbox_fuzzing) {
212 SandboxTesting::Mode mode = v8_flags.sandbox_testing
213 ? SandboxTesting::Mode::kForTesting
214 : SandboxTesting::Mode::kForFuzzing;
215 SandboxTesting::Enable(mode);
216 }
217#endif // V8_ENABLE_SANDBOX
218
219#if defined(V8_USE_PERFETTO)
220 if (perfetto::Tracing::IsInitialized()) {
221 TrackEvent::Register();
222 if (v8_flags.perfetto_code_logger) {
223 v8::internal::CodeDataSource::Register();
224 }
225 }
226#endif
227 IsolateGroup::InitializeOncePerProcess();
228 Isolate::InitializeOncePerProcess();
229
230#if defined(USE_SIMULATOR)
231 Simulator::InitializeOncePerProcess();
232#endif
233 CpuFeatures::Probe(false);
234 ElementsAccessor::InitializeOncePerProcess();
235 Bootstrapper::InitializeOncePerProcess();
236 CallDescriptors::InitializeOncePerProcess();
237
238#if V8_ENABLE_WEBASSEMBLY
239 wasm::WasmEngine::InitializeOncePerProcess();
240#endif // V8_ENABLE_WEBASSEMBLY
241
242 ExternalReferenceTable::InitializeOncePerIsolateGroup(
243 IsolateGroup::current()->external_ref_table());
244 AdvanceStartupState(V8StartupState::kV8Initialized);
245}

249void V8::Dispose() {
250 AdvanceStartupState(V8StartupState::kV8Disposing);
251 CHECK(platform_);
252#if V8_ENABLE_WEBASSEMBLY
253 wasm::WasmEngine::GlobalTearDown();
254#endif // V8_ENABLE_WEBASSEMBLY
255#if defined(USE_SIMULATOR)
256 Simulator::GlobalTearDown();
257#endif
258 CallDescriptors::TearDown();
259 ElementsAccessor::TearDown();
260 RegisteredExtension::UnregisterAll();
261 FlagList::ReleaseDynamicAllocations();
262 AdvanceStartupState(V8StartupState::kV8Disposed);
263}

265void V8::DisposePlatform() {
266 AdvanceStartupState(V8StartupState::kPlatformDisposing);
267 CHECK(platform_);
268#if defined(V8_OS_WIN) && defined(V8_ENABLE_ETW_STACK_WALKING)
269 if (v8_flags.enable_etw_stack_walking ||
270 v8_flags.enable_etw_by_custom_filter_only) {
271 v8::internal::ETWJITInterface::Unregister();
272 }
273#endif
274 v8::tracing::TracingCategoryObserver::TearDown();
275 v8::base::SetPrintStackTrace(nullptr);
276
277#ifdef V8_ENABLE_SANDBOX
278 Sandbox::TearDownDefault();
279#endif // V8_ENABLE_SANDBOX
280
281 platform_ = nullptr;
282
283#if DEBUG
284 internal::ThreadIsolation::CheckTrackedMemoryEmpty();
285#endif
286
287 AdvanceStartupState(V8StartupState::kPlatformDisposed);
288}
