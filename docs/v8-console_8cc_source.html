103 void reportCall(ConsoleAPIType type,
104 v8::MemorySpan<const v8::Local<v8::Value>> arguments) {
105 if (!groupId()) return;
106 // Depending on the type of the console message, we capture only parts of
107 // the stack trace, or no stack trace at all.
108 std::unique_ptr<V8StackTraceImpl> stackTrace;
109 switch (type) {
110 case ConsoleAPIType::kTrace:
111 // The purpose of `console.trace()` is to output a stack trace to the
112 // developer tools console, therefore we should always strive to
113 // capture a full stack trace, even before any debugger is attached.
114 stackTrace = m_inspector->debugger()->captureStackTrace(true);
115 break;
116
117 case ConsoleAPIType::kTimeEnd:
118 // The `console.time()` and `console.timeEnd()` APIs are meant for
119 // performance investigations, and therefore it's important to reduce
120 // the total overhead of these calls, but also make sure these APIs
121 // have consistent performance overhead. In order to guarantee that,
122 // we always capture only the top frame, otherwise the performance
123 // characteristics of `console.timeEnd()` would differ based on the
124 // current call depth, which would skew the results.
125 //
126 // See https://crbug.com/41433391 for more information.
127 stackTrace = V8StackTraceImpl::capture(m_inspector->debugger(), 1);
128 break;
129
130 default:
131 // All other APIs get a full stack trace only when the debugger is
132 // attached, otherwise record only the top frame.
133 stackTrace = m_inspector->debugger()->captureStackTrace(false);
134 break;
135 }
136 std::unique_ptr<V8ConsoleMessage> message =
137 V8ConsoleMessage::createForConsoleAPI(
138 context(), contextId(), groupId(), m_inspector,
139 m_inspector->client()->currentTimeMS(), type, arguments,
140 consoleContextToString(isolate(), m_consoleContext),
141 std::move(stackTrace));
142 consoleMessageStorage()->addMessage(std::move(message));
143 }

506void V8Console::createTask(const v8::FunctionCallbackInfo<v8::Value>& info) {
507 v8::Isolate* isolate = info.GetIsolate();
508
509 v8::debug::RecordAsyncStackTaggingCreateTaskCall(isolate);
510
511 if (info.Length() < 1 || !info[0]->IsString() ||
512 !info[0].As<v8::String>()->Length()) {
513 isolate->ThrowError("First argument must be a non-empty string.");
514 return;
515 }
516
517 v8::Local<v8::Object> task = taskTemplate()
518 ->NewInstance(isolate->GetCurrentContext())
519 .ToLocalChecked();
520
521 auto taskInfo = std::make_unique<TaskInfo>(isolate, this, task);
522 void* taskId = taskInfo->Id();
523 auto [iter, inserted] = m_tasks.emplace(taskId, std::move(taskInfo));
524 CHECK(inserted);
525
526 String16 nameArgument = toProtocolString(isolate, info[0].As<v8::String>());
527 StringView taskName =
528 StringView(nameArgument.characters16(), nameArgument.length());
529 m_inspector->asyncTaskScheduled(taskName, taskId, /* recurring */ true);
530
531 info.GetReturnValue().Set(task);
532}

534void V8Console::runTask(const v8::FunctionCallbackInfo<v8::Value>& info) {
535 v8::Isolate* isolate = info.GetIsolate();
536 if (info.Length() < 1 || !info[0]->IsFunction()) {
537 isolate->ThrowError("First argument must be a function.");
538 return;
539 }
540 v8::Local<v8::Function> function = info[0].As<v8::Function>();
541
542 v8::Local<v8::Object> task = info.This();
543 v8::Local<v8::Value> maybeTaskExternal;
544 if (!task->GetPrivate(isolate->GetCurrentContext(), taskInfoKey())
545 .ToLocal(&maybeTaskExternal)) {
546 // An exception is already thrown.
547 return;
548 }
549
550 if (!maybeTaskExternal->IsExternal()) {
551 isolate->ThrowError("'run' called with illegal receiver.");
552 return;
553 }
554
555 v8::Local<v8::External> taskExternal = maybeTaskExternal.As<v8::External>();
556 TaskInfo* taskInfo = reinterpret_cast<TaskInfo*>(taskExternal->Value());
557
558 m_inspector->asyncTaskStarted(taskInfo->Id());
559 {
560 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.inspector"),
561 "V8Console::runTask");
562 v8::Local<v8::Value> result;
563 if (function
564 ->Call(isolate->GetCurrentContext(), v8::Undefined(isolate), 0, {})
565 .ToLocal(&result)) {
566 info.GetReturnValue().Set(result);
567 }
568 }
569 m_inspector->asyncTaskFinished(taskInfo->Id());
570}

747 v8::Local<v8::Value> value, int sessionId,

767 if (V8InspectorSessionImpl* session = helper.session(sessionId)) {
