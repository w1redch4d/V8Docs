76void MarkingStateBase::MarkAndPush(HeapObjectHeader& header,
77 TraceDescriptor desc) {
78 DCHECK_NOT_NULL(desc.callback);
79
80 if (header.IsInConstruction<AccessMode::kAtomic>()) {
81 not_fully_constructed_worklist_.Push<AccessMode::kAtomic>(&header);
82 } else if (MarkNoPush(header)) {
83 PushMarked(header, desc);
84 }
85}

87bool MarkingStateBase::MarkNoPush(HeapObjectHeader& header) {
88 // A GC should only mark the objects that belong in its heap.
89 DCHECK_EQ(&heap_, &BasePage::FromPayload(&header)->heap());
90 // Never mark free space objects. This would e.g. hint to marking a promptly
91 // freed backing store.
92 DCHECK(!header.IsFree<AccessMode::kAtomic>());
93 return header.TryMarkAtomic();
94}

106 DCHECK(!header.IsInConstruction<AccessMode::kAtomic>());
107 DCHECK_NOT_NULL(desc.callback);

234void BasicMarkingState::RegisterWeakReferenceIfNeeded(
235 const void* object, TraceDescriptor desc, WeakCallback weak_callback,
236 const void* parameter) {
237 // Filter out already marked values. The write barrier for WeakMember
238 // ensures that any newly set value after this point is kept alive and does
239 // not require the callback.
240 const HeapObjectHeader& header =
241 HeapObjectHeader::FromObject(desc.base_object_payload);
242 if (!header.IsInConstruction<AccessMode::kAtomic>() &&
243 header.IsMarked<AccessMode::kAtomic>())
244 return;
245 parallel_weak_callback_worklist_.Push({weak_callback, parameter});
246}

264void BasicMarkingState::ProcessWeakContainer(const void* object,
265 TraceDescriptor desc,
266 WeakCallback callback,
267 const void* data) {
268 DCHECK_NOT_NULL(object);
269
270 HeapObjectHeader& header =
271 HeapObjectHeader::FromObject(const_cast<void*>(object));
272
273 if (header.IsInConstruction<AccessMode::kAtomic>()) {
274 not_fully_constructed_worklist_.Push<AccessMode::kAtomic>(&header);
275 return;
276 }
277
278 RegisterWeakContainer(header);
279
280 // Only mark the container initially. Its buckets will be processed after
281 // marking.
282 if (!MarkNoPush(header)) return;
283
284 // Register final weak processing of the backing store.
285 RegisterWeakContainerCallback(callback, data);
286
287 // Weak containers might not require tracing. In such cases the callback in
288 // the TraceDescriptor will be nullptr. For ephemerons the callback will be
289 // non-nullptr so that the container is traced and the ephemeron pairs are
290 // processed.
291 if (desc.callback) {
292 PushMarked(header, desc);
293 } else {
294 // For weak containers, there's no trace callback and no processing loop to
295 // update the marked bytes, hence inline that here.
296 AccountMarkedBytes(header);
297 }
298}

300void BasicMarkingState::ProcessEphemeron(const void* key, const void* value,
301 TraceDescriptor value_desc,
302 Visitor& visitor) {
303 // ProcessEphemeron is not expected to find new ephemerons recursively, which
304 // would break the main marking loop.
305 DCHECK(!in_ephemeron_processing_);
306 in_ephemeron_processing_ = true;
307 // Keys are considered live even in incremental/concurrent marking settings
308 // because the write barrier for WeakMember ensures that any newly set value
309 // after this point is kept alive and does not require the callback.
310 const bool key_in_construction =
311 HeapObjectHeader::FromObject(key).IsInConstruction<AccessMode::kAtomic>();
312 const bool key_considered_as_live =
313 key_in_construction
314 ? in_atomic_pause_
315 : HeapObjectHeader::FromObject(key).IsMarked<AccessMode::kAtomic>();
316 DCHECK_IMPLIES(
317 key_in_construction && in_atomic_pause_,
318 HeapObjectHeader::FromObject(key).IsMarked<AccessMode::kAtomic>());
319 if (key_considered_as_live) {
320 if (value_desc.base_object_payload) {
321 MarkAndPush(value_desc.base_object_payload, value_desc);
322 } else {
323 // If value_desc.base_object_payload is nullptr, the value is not GCed and
324 // should be immediately traced.
325 value_desc.callback(&visitor, value);
326 }
327 } else {
328 discovered_ephemeron_pairs_worklist_.Push({key, value, value_desc});
329 discovered_new_ephemeron_pairs_ = true;
330 }
331 in_ephemeron_processing_ = false;
332}

334void BasicMarkingState::AccountMarkedBytes(const HeapObjectHeader& header) {
335 const size_t marked_bytes =
336 header.IsLargeObject<AccessMode::kAtomic>()
337 ? reinterpret_cast<const LargePage*>(BasePage::FromPayload(&header))
338 ->PayloadSize()
339 : header.AllocatedSize<AccessMode::kAtomic>();
340 auto* base_page =
341 BasePage::FromPayload(&const_cast<HeapObjectHeader&>(header));
342 AccountMarkedBytes(base_page, marked_bytes);
343}

345void BasicMarkingState::AccountMarkedBytes(BasePage* base_page,
346 size_t marked_bytes) {
347 marked_bytes_ += marked_bytes;
348 marked_bytes_map_[base_page] += static_cast<int64_t>(marked_bytes);

443 const bool result =
444 weak_containers_worklist_.Contains<AccessMode::kAtomic>(&header) &&

458void MutatorMarkingState::RecentlyRetracedWeakContainers::Insert(
459 const HeapObjectHeader* header) {
460 last_used_index_ = (last_used_index_ + 1) % kMaxCacheSize;
461 if (recently_retraced_cache_.size() <= last_used_index_)

481 inline void AccountDeferredMarkedBytes(BasePage* base_page,
482 size_t deferred_bytes) {
483 // AccountDeferredMarkedBytes is called from Trace methods, which are always
484 // called after AccountMarkedBytes, so there should be no underflow here.
485 DCHECK_LE(deferred_bytes, marked_bytes_);
486 marked_bytes_ -= deferred_bytes;
487 marked_bytes_map_[base_page] -= static_cast<int64_t>(deferred_bytes);
488 }

494template <size_t deadline_check_interval, typename WorklistLocal,
495 typename Callback, typename Predicate>
496bool DrainWorklistWithPredicate(Predicate should_yield,
497 WorklistLocal& worklist_local,
498 Callback callback) {
499 if (worklist_local.IsLocalAndGlobalEmpty()) return true;
500 // For concurrent markers, should_yield also reports marked bytes.
501 if (should_yield()) return false;
502 size_t processed_callback_count = deadline_check_interval;
503 typename WorklistLocal::ItemType item;
504 while (worklist_local.Pop(&item)) {
505 callback(item);
506 if (--processed_callback_count == 0) {
507 if (should_yield()) {
508 return false;
509 }
510 processed_callback_count = deadline_check_interval;
511 }
512 }
513 return true;
514}
