29template <class T, int s>
30int Mapping<T, s>::get(uchar c, uchar n, uchar* result) {
31 CacheEntry entry = entries_[c & kMask];
32 if (entry.code_point_ == c) {
33 if (entry.offset_ == 0) {
34 return 0;
35 } else {
36 result[0] = c + entry.offset_;
37 return 1;
38 }
39 } else {
40 return CalculateValue(c, n, result);
41 }

44template <class T, int s>
45int Mapping<T, s>::CalculateValue(uchar c, uchar n, uchar* result) {
46 bool allow_caching = true;
47 int length = T::Convert(c, n, result, &allow_caching);
48 if (allow_caching) {
49 if (length == 1) {
50 entries_[c & kMask] = CacheEntry(c, result[0] - c);
51 return 1;
52 } else {
53 entries_[c & kMask] = CacheEntry(c, 0);
54 return 0;
55 }
56 } else {
57 return length;
58 }

62bool Utf16::HasUnpairedSurrogate(const uint16_t* code_units, size_t length) {
63 for (size_t i = 0; i < length; ++i) {
64 const int code_unit = code_units[i];
65 if (IsLeadSurrogate(code_unit)) {
66 // The current code unit is a leading surrogate. Check if it is followed
67 // by a trailing surrogate.
68 if (i == length - 1) return true;
69 if (!IsTrailSurrogate(code_units[i + 1])) return true;
70 // Skip the paired trailing surrogate.
71 ++i;
72 } else if (IsTrailSurrogate(code_unit)) {
73 // All paired trailing surrogates are skipped above, so this branch is
74 // only for those that are unpaired.
75 return true;
76 }
77 }
78 return false;
79}

81// Decodes UTF-8 bytes incrementally, allowing the decoding of bytes as they
82// stream in. This **must** be followed by a call to ValueOfIncrementalFinish
83// when the stream is complete, to ensure incomplete sequences are handled.
84uchar Utf8::ValueOfIncremental(const uint8_t** cursor, State* state,
85 Utf8IncrementalBuffer* buffer) {
86 DCHECK_NOT_NULL(buffer);
87 State old_state = *state;
88 uint8_t next = **cursor;
89 *cursor += 1;
90
91 if (V8_LIKELY(next <= kMaxOneByteChar && old_state == State::kAccept)) {
92 DCHECK_EQ(0u, *buffer);
93 return static_cast<uchar>(next);
94 }
95
96 // So we're at the lead byte of a 2/3/4 sequence, or we're at a continuation
97 // char in that sequence.
98 Utf8DfaDecoder::Decode(next, state, buffer);
99
100 switch (*state) {
101 case State::kAccept: {
102 uchar t = *buffer;
103 *buffer = 0;
104 return t;
105 }
106
107 case State::kReject:
108 *state = State::kAccept;
109 *buffer = 0;
110
111 // If we hit a bad byte, we need to determine if we were trying to start
112 // a sequence or continue one. If we were trying to start a sequence,
113 // that means it's just an invalid lead byte and we need to continue to
114 // the next (which we already did above). If we were already in a
115 // sequence, we need to reprocess this same byte after resetting to the
116 // initial state.
117 if (old_state != State::kAccept) {
118 // We were trying to continue a sequence, so let's reprocess this byte
119 // next time.
120 *cursor -= 1;
121 }
122 return kBadChar;
123
124 default:
125 return kIncomplete;
126 }
127}

129unsigned Utf8::EncodeOneByte(char* str, uint8_t c) {
130 static const int kMask = ~(1 << 6);
131 if (c <= kMaxOneByteChar) {
132 str[0] = c;
133 return 1;
134 } else {
135 str[0] = 0xC0 | (c >> 6);
136 str[1] = 0x80 | (c & kMask);
137 return 2;
138 }
139}

141// Encode encodes the UTF-16 code units c and previous into the given str
142// buffer, and combines surrogate code units into single code points. If
143// replace_invalid is set to true, orphan surrogate code units will be replaced
144// with kBadChar.
145unsigned Utf8::Encode(char* str, uchar c, int previous, bool replace_invalid) {
146 static const int kMask = ~(1 << 6);
147 if (c <= kMaxOneByteChar) {
148 str[0] = c;
149 return 1;
150 } else if (c <= kMaxTwoByteChar) {
151 str[0] = 0xC0 | (c >> 6);
152 str[1] = 0x80 | (c & kMask);
153 return 2;
154 } else if (c <= kMaxThreeByteChar) {
155 DCHECK(!Utf16::IsLeadSurrogate(Utf16::kNoPreviousCharacter));
156 if (Utf16::IsSurrogatePair(previous, c)) {
157 const int kUnmatchedSize = kSizeOfUnmatchedSurrogate;
158 return Encode(str - kUnmatchedSize,
159 Utf16::CombineSurrogatePair(previous, c),
160 Utf16::kNoPreviousCharacter, replace_invalid) -
161 kUnmatchedSize;
162 } else if (replace_invalid &&
163 (Utf16::IsLeadSurrogate(c) || Utf16::IsTrailSurrogate(c))) {
164 c = kBadChar;
165 }
166 str[0] = 0xE0 | (c >> 12);
167 str[1] = 0x80 | ((c >> 6) & kMask);
168 str[2] = 0x80 | (c & kMask);
169 return 3;
170 } else {
171 str[0] = 0xF0 | (c >> 18);
172 str[1] = 0x80 | ((c >> 12) & kMask);
173 str[2] = 0x80 | ((c >> 6) & kMask);
174 str[3] = 0x80 | (c & kMask);
175 return 4;
176 }
177}

179uchar Utf8::ValueOf(const uint8_t* bytes, size_t length, size_t* cursor) {
180 if (length == 0) return kBadChar;
181 uint8_t first = bytes[0];
182 // Characters between 0000 and 007F are encoded as a single character
183 if (V8_LIKELY(first <= kMaxOneByteChar)) {
184 *cursor += 1;
185 return first;
186 }
187 return CalculateValue(bytes, length, cursor);
188}

198unsigned Utf8::Length(uchar c, int previous) {
199 if (c <= kMaxOneByteChar) {
200 return 1;
201 } else if (c <= kMaxTwoByteChar) {
202 return 2;
203 } else if (c <= kMaxThreeByteChar) {
204 DCHECK(!Utf16::IsLeadSurrogate(Utf16::kNoPreviousCharacter));
205 if (Utf16::IsSurrogatePair(previous, c)) {
206 return kSizeOfUnmatchedSurrogate - kBytesSavedByCombiningSurrogates;
207 }
208 return 3;
209 } else {
210 return 4;
211 }

214bool Utf8::IsValidCharacter(uchar c) {
215 return c < 0xD800u || (c >= 0xE000u && c < 0xFDD0u) ||
216 (c > 0xFDEFu && c <= 0x10FFFFu && (c & 0xFFFEu) != 0xFFFEu &&
217 c != kBadChar);
218}

220template <typename Char>
221Utf8::EncodingResult Utf8::Encode(v8::base::Vector<const Char> string,
222 char* buffer, size_t capacity,
223 bool write_null, bool replace_invalid_utf8) {
224 constexpr bool kSourceIsOneByte = sizeof(Char) == 1;
225
226 if constexpr (kSourceIsOneByte) {
227 // Only 16-bit characters can contain invalid unicode.
228 replace_invalid_utf8 = false;
229 }
230
231 size_t write_index = 0;
232 const Char* characters = string.begin();
233 size_t content_capacity = capacity - write_null;
234 CHECK_LE(content_capacity, capacity);
235 uint16_t last = Utf16::kNoPreviousCharacter;
236 size_t read_index = 0;
237 for (; read_index < string.size(); read_index++) {
238 Char character = characters[read_index];
239
240 size_t required_capacity;
241 if constexpr (kSourceIsOneByte) {
242 required_capacity = Utf8::LengthOneByte(character);
243 } else {
244 required_capacity = Utf8::Length(character, last);
245 }
246 size_t remaining_capacity = content_capacity - write_index;
247 if (remaining_capacity < required_capacity) {
248 // Not enough space left, so stop here.
249 if (replace_invalid_utf8 && Utf16::IsLeadSurrogate(last)) {
250 DCHECK_GE(write_index, Utf8::kSizeOfUnmatchedSurrogate);
251 // We're in the middle of a surrogate pair. Delete the first part again.
252 write_index -= Utf8::kSizeOfUnmatchedSurrogate;
253 }
254 break;
255 }
256
257 if constexpr (kSourceIsOneByte) {
258 write_index += Utf8::EncodeOneByte(buffer + write_index, character);
259 } else {
260 write_index += Utf8::Encode(buffer + write_index, character, last,
261 replace_invalid_utf8);
262 }
263
264 last = character;
265 }
266 DCHECK_LE(write_index, capacity);
267
268 if (write_null) {
269 DCHECK_LT(write_index, capacity);
270 buffer[write_index++] = '\0';
271 }
272
273 size_t bytes_written = write_index;
274 size_t characters_processed = read_index;
275 return {bytes_written, characters_processed};
276}
