72template <typename Impl>
73template <AllocationType allocation>
74Handle<Number> FactoryBase<Impl>::NewNumberFromSize(size_t value) {
75 // We can't use Smi::IsValid() here because that operates on a signed
76 // intptr_t, and casting from size_t could create a bogus sign bit.
77 if (value <= static_cast<size_t>(Smi::kMaxValue)) {
78 return handle(Smi::FromIntptr(static_cast<intptr_t>(value)), isolate());
79 }
80 return NewHeapNumber<allocation>(static_cast<double>(value));
81}

83template <typename Impl>
84template <AllocationType allocation>
85Handle<Number> FactoryBase<Impl>::NewNumberFromInt64(int64_t value) {
86 if (value <= std::numeric_limits<int32_t>::max() &&
87 value >= std::numeric_limits<int32_t>::min() &&
88 Smi::IsValid(static_cast<int32_t>(value))) {
89 return handle(Smi::FromInt(static_cast<int32_t>(value)), isolate());
90 }
91 return NewHeapNumber<allocation>(static_cast<double>(value));

135template <typename Impl>
136Tagged<Struct> FactoryBase<Impl>::NewStructInternal(ReadOnlyRoots roots,
137 Tagged<Map> map, int size,
138 AllocationType allocation) {
139 DCHECK_EQ(size, map->instance_size());
140 Tagged<HeapObject> result = AllocateRawWithImmortalMap(size, allocation, map);
141 Tagged<Struct> str = Cast<Struct>(result);
142 Tagged<Undefined> undefined = roots.undefined_value();
143 int length = (size >> kTaggedSizeLog2) - 1;
144 MemsetTagged(str->RawField(Struct::kHeaderSize), undefined, length);
145 return str;
146}
