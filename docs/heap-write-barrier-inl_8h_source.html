20// static
21void WriteBarrier::CombinedWriteBarrierInternal(Tagged<HeapObject> host,
22 HeapObjectSlot slot,
23 Tagged<HeapObject> value,
24 WriteBarrierMode mode) {
25 DCHECK_EQ(mode, UPDATE_WRITE_BARRIER);
26
27 MemoryChunk* host_chunk = MemoryChunk::FromHeapObject(host);
28 MemoryChunk* value_chunk = MemoryChunk::FromHeapObject(value);
29
30 const bool is_marking = host_chunk->IsMarking();
31
32 if (v8_flags.sticky_mark_bits) {
33 // TODO(333906585): Support shared barrier.
34 if (!HeapLayout::InYoungGeneration(host_chunk, host) &&
35 HeapLayout::InYoungGeneration(value_chunk, value)) {
36 // Generational or shared heap write barrier (old-to-new or
37 // old-to-shared).
38 CombinedGenerationalAndSharedBarrierSlow(host, slot.address(), value);
39 }
40 } else {
41 const bool pointers_from_here_are_interesting =
42 !host_chunk->IsYoungOrSharedChunk();
43 if (pointers_from_here_are_interesting &&
44 value_chunk->IsYoungOrSharedChunk()) {
45 // Generational or shared heap write barrier (old-to-new or
46 // old-to-shared).
47 CombinedGenerationalAndSharedBarrierSlow(host, slot.address(), value);
48 }
49 }
50
51 // Marking barrier: mark value & record slots when marking is on.
52 if (V8_UNLIKELY(is_marking)) {
53 MarkingSlow(host, HeapObjectSlot(slot), value);
54 }
55}

57// static
58inline WriteBarrierMode WriteBarrier::GetWriteBarrierModeForObject(
59 Tagged<HeapObject> object, const DisallowGarbageCollection& promise) {
60 if (v8_flags.disable_write_barriers) {
61 return SKIP_WRITE_BARRIER;
62 }
63 DCHECK(PageFlagsAreConsistent(object));
64 MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
65 if (chunk->IsMarking()) {
66 return UPDATE_WRITE_BARRIER;
67 }
68 if (HeapLayout::InYoungGeneration(chunk, object)) {
69 return SKIP_WRITE_BARRIER;
70 }
71 return UPDATE_WRITE_BARRIER;
72}

80// static
81void WriteBarrier::ForRelocInfo(Tagged<InstructionStream> host,
82 RelocInfo* rinfo, Tagged<HeapObject> value,
83 WriteBarrierMode mode) {
84 if (mode == SKIP_WRITE_BARRIER) {
85 SLOW_DCHECK(!WriteBarrier::IsRequired(host, value));
86 return;
87 }
88
89 // Used during InstructionStream initialization where we update the write
90 // barriers together separate from the field writes.
91 if (mode == UNSAFE_SKIP_WRITE_BARRIER) {
92 DCHECK(!DisallowGarbageCollection::IsAllowed());
93 return;
94 }
95
96 DCHECK_EQ(mode, UPDATE_WRITE_BARRIER);
97 GenerationalForRelocInfo(host, rinfo, value);
98 SharedForRelocInfo(host, rinfo, value);
99 MarkingForRelocInfo(host, rinfo, value);
100}

133// static
134void WriteBarrier::ForEphemeronHashTable(Tagged<EphemeronHashTable> host,
135 ObjectSlot slot, Tagged<Object> value,
136 WriteBarrierMode mode) {
137 if (mode == SKIP_WRITE_BARRIER) {
138 SLOW_DCHECK(!WriteBarrier::IsRequired(host, value));
139 return;
140 }
141
142 DCHECK_EQ(mode, UPDATE_WRITE_BARRIER);
143 if (!value.IsHeapObject()) return;
144
145 MemoryChunk* host_chunk = MemoryChunk::FromHeapObject(host);
146
147 Tagged<HeapObject> heap_object_value = Cast<HeapObject>(value);
148 MemoryChunk* value_chunk = MemoryChunk::FromHeapObject(heap_object_value);
149
150 const bool pointers_from_here_are_interesting =
151 !host_chunk->IsYoungOrSharedChunk();
152 const bool is_marking = host_chunk->IsMarking();
153
154 if (pointers_from_here_are_interesting &&
155 value_chunk->IsYoungOrSharedChunk()) {
156 CombinedGenerationalAndSharedEphemeronBarrierSlow(host, slot.address(),
157 heap_object_value);
158 }
159
160 // Marking barrier: mark value & record slots when marking is on.
161 if (is_marking) {
162 MarkingSlow(host, HeapObjectSlot(slot), heap_object_value);
163 }
164}

252void WriteBarrier::Marking(Tagged<HeapObject> host, MaybeObjectSlot slot,
253 Tagged<MaybeObject> value) {
254 Tagged<HeapObject> value_heap_object;
255 if (!value.GetHeapObject(&value_heap_object)) {
256 return;
257 }
258 // This barrier is called from generated code and from C++ code.
259 // There must be no stores of InstructionStream values from generated code and
260 // all stores of InstructionStream values in C++ must be handled by
261 // CombinedWriteBarrierInternal().

347// static
348void WriteBarrier::ForCppHeapPointer(Tagged<JSObject> host,
349 CppHeapPointerSlot slot, void* value) {
350 // Note: this is currently a combined barrier for marking both the
351 // CppHeapPointerTable entry and the referenced object.
352
353 if (V8_LIKELY(!IsMarking(host))) {
354#if defined(CPPGC_YOUNG_GENERATION)
355 // There is no young-gen CppHeapPointerTable space so we should not mark
356 // the table entry in this case.
357 if (value) {
358 GenerationalBarrierForCppHeapPointer(host, value);
359 }
360#endif
361 return;
362 }
363 MarkingBarrier* marking_barrier = CurrentMarkingBarrier(host);
364 if (marking_barrier->is_minor()) {
365 // TODO(v8:13012): We do not currently mark Oilpan objects while MinorMS is
366 // active. Once Oilpan uses a generational GC with incremental marking and
367 // unified heap, this barrier will be needed again.
368 return;
369 }
370
371 MarkingSlowFromCppHeapWrappable(marking_barrier->heap(), host, slot, value);
372}

374// static
375void WriteBarrier::GenerationalBarrierForCppHeapPointer(Tagged<JSObject> host,
376 void* value) {
377 if (!value) {
378 return;
379 }
380 auto* memory_chunk = MemoryChunk::FromHeapObject(host);
381 if (V8_LIKELY(HeapLayout::InYoungGeneration(memory_chunk, host))) {
382 return;
383 }
384 auto* cpp_heap = memory_chunk->GetHeap()->cpp_heap();
385 v8::internal::CppHeap::From(cpp_heap)->RememberCrossHeapReferenceIfNeeded(
386 host, value);
387}

392bool WriteBarrier::IsRequired(Tagged<HeapObject> host, T value) {
393 if (HeapLayout::InYoungGeneration(host)) {
394 return false;
