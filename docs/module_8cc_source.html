104void Module::RecordError(Isolate* isolate, Tagged<Object> error) {
105 DisallowGarbageCollection no_gc;
106 // Allow overriding exceptions with termination exceptions.
107 DCHECK_IMPLIES(isolate->is_catchable_by_javascript(error),
108 IsTheHole(exception(), isolate));
109 DCHECK(!IsTheHole(error, isolate));
110 if (IsSourceTextModule(*this)) {
111 // Revert to minmal SFI in case we have already been instantiating or
112 // evaluating.
113 auto self = Cast<SourceTextModule>(*this);
114 self->set_code(self->GetSharedFunctionInfo());
115 }
116 SetStatusInternal(*this, Module::kErrored);
117 if (isolate->is_catchable_by_javascript(error)) {
118 set_exception(error);
119 } else {
120 // v8::TryCatch uses `null` for termination exceptions.
121 set_exception(ReadOnlyRoots(isolate).null_value());
122 }

125void Module::ResetGraph(Isolate* isolate, DirectHandle<Module> module) {
126 DCHECK_NE(module->status(), kEvaluating);
127 if (module->status() != kPreLinking && module->status() != kLinking) {
128 return;
129 }
130
131 DirectHandle<FixedArray> requested_modules =
132 IsSourceTextModule(*module)
133 ? DirectHandle<FixedArray>(
134 Cast<SourceTextModule>(*module)->requested_modules(), isolate)
135 : DirectHandle<FixedArray>();
136 Reset(isolate, module);
137
138 if (!IsSourceTextModule(*module)) {
139 DCHECK(IsSyntheticModule(*module));
140 return;
141 }
142 for (int i = 0; i < requested_modules->length(); ++i) {
143 DirectHandle<Object> descendant(requested_modules->get(i), isolate);
144 if (IsModule(*descendant)) {
145 ResetGraph(isolate, Cast<Module>(descendant));
146 } else {
147 // The requested module is either an undefined or a WasmModule object.
148#if V8_ENABLE_WEBASSEMBLY
149 DCHECK(IsUndefined(*descendant, isolate) ||
150 IsWasmModuleObject(*descendant));
151#else
152 DCHECK(IsUndefined(*descendant, isolate));
153#endif
154 }
155 }

158void Module::Reset(Isolate* isolate, DirectHandle<Module> module) {
159 DCHECK(module->status() == kPreLinking || module->status() == kLinking);
160 DCHECK(IsTheHole(module->exception(), isolate));
161 // The namespace object cannot exist, because it would have been created
162 // by RunInitializationCode, which is called only after this module's SCC
163 // succeeds instantiation.
164 DCHECK(!IsJSModuleNamespace(module->module_namespace()));
165 const int export_count =
166 IsSourceTextModule(*module)
167 ? Cast<SourceTextModule>(*module)->regular_exports()->length()
168 : Cast<SyntheticModule>(*module)->export_names()->length();
169 DirectHandle<ObjectHashTable> exports =
170 ObjectHashTable::New(isolate, export_count);
171
172 if (IsSourceTextModule(*module)) {
173 SourceTextModule::Reset(isolate, Cast<SourceTextModule>(module));
174 }
175
176 module->set_exports(*exports);
177 SetStatusInternal(*module, kUnlinked);
178}

206bool Module::Instantiate(Isolate* isolate, Handle<Module> module,
207 v8::Local<v8::Context> context,
208 v8::Module::ResolveModuleCallback module_callback,
209 v8::Module::ResolveSourceCallback source_callback) {
210#ifdef DEBUG
211 PrintStatusMessage(*module, "Instantiating module ");
212#endif // DEBUG
213
214 if (!PrepareInstantiate(isolate, module, context, module_callback,
215 source_callback)) {
216 ResetGraph(isolate, module);
217 DCHECK_EQ(module->status(), kUnlinked);
218 return false;
219 }
220 Zone zone(isolate->allocator(), ZONE_NAME);
221 ZoneForwardList<Handle<SourceTextModule>> stack(&zone);
222 unsigned dfs_index = 0;
223 if (!FinishInstantiate(isolate, module, &stack, &dfs_index, &zone)) {
224 ResetGraph(isolate, module);
225 DCHECK_EQ(module->status(), kUnlinked);
226 return false;
227 }
228 DCHECK(module->status() == kLinked || module->status() == kEvaluated ||
229 module->status() == kEvaluatingAsync || module->status() == kErrored);
230 DCHECK(stack.empty());
231 return true;
232}

272MaybeHandle<Object> Module::Evaluate(Isolate* isolate, Handle<Module> module) {
273#ifdef DEBUG
274 PrintStatusMessage(*module, "Evaluating module ");
275#endif // DEBUG
276 int module_status = module->status();
277
278 // In the event of errored evaluation, return a rejected promise.
279 if (module_status == kErrored) {
280 // If we have a top level capability we assume it has already been
281 // rejected, and return it here. Otherwise create a new promise and
282 // reject it with the module's exception.
283 if (IsJSPromise(module->top_level_capability())) {
284 Handle<JSPromise> top_level_capability(
285 Cast<JSPromise>(module->top_level_capability()), isolate);
286 DCHECK(top_level_capability->status() == Promise::kRejected &&
287 top_level_capability->result() == module->exception());
288 return top_level_capability;
289 }
290 Handle<JSPromise> capability = isolate->factory()->NewJSPromise();
291 JSPromise::Reject(capability, direct_handle(module->exception(), isolate));
292 return capability;
293 }
294
295 // Start of Evaluate () Concrete Method
296 // 2. Assert: module.[[Status]] is one of LINKED, EVALUATING-ASYNC, or
297 // EVALUATED.
298 CHECK(module_status == kLinked || module_status == kEvaluatingAsync ||
299 module_status == kEvaluated);
300
301 // 3. If module.[[Status]] is either EVALUATING-ASYNC or EVALUATED, set module
302 // to module.[[CycleRoot]].
303 // A Synthetic Module has no children so it is its own cycle root.
304 if (module_status >= kEvaluatingAsync && IsSourceTextModule(*module)) {
305 module = Cast<SourceTextModule>(module)->GetCycleRoot(isolate);
306 }
307
308 // 4. If module.[[TopLevelCapability]] is not EMPTY, then
309 // a. Return module.[[TopLevelCapability]].[[Promise]].
310 if (IsJSPromise(module->top_level_capability())) {
311 return handle(Cast<JSPromise>(module->top_level_capability()), isolate);
312 }
313 DCHECK(IsUndefined(module->top_level_capability()));
314
315 if (IsSourceTextModule(*module)) {
316 return SourceTextModule::Evaluate(isolate, Cast<SourceTextModule>(module));
317 } else {
318 return SyntheticModule::Evaluate(isolate, Cast<SyntheticModule>(module));
319 }
320}

322Handle<JSModuleNamespace> Module::GetModuleNamespace(Isolate* isolate,
323 Handle<Module> module) {
324 Handle<HeapObject> object(module->module_namespace(), isolate);
325 ReadOnlyRoots roots(isolate);
326 if (!IsUndefined(*object, roots)) {
327 // Namespace object already exists.
328 return Cast<JSModuleNamespace>(object);
329 }
330
331 // Collect the export names.
332 Zone zone(isolate->allocator(), ZONE_NAME);
333 UnorderedModuleSet visited(&zone);
334
335 if (IsSourceTextModule(*module)) {
336 SourceTextModule::FetchStarExports(isolate, Cast<SourceTextModule>(module),
337 &zone, &visited);
338 }
339
340 DirectHandle<ObjectHashTable> exports(module->exports(), isolate);
341 ZoneVector<IndirectHandle<String>> names(&zone);
342 names.reserve(exports->NumberOfElements());
343 for (InternalIndex i : exports->IterateEntries()) {
344 Tagged<Object> key;
345 if (!exports->ToKey(roots, i, &key)) continue;
346 names.push_back(handle(Cast<String>(key), isolate));
347 }
348 DCHECK_EQ(static_cast<int>(names.size()), exports->NumberOfElements());
349
350 // Sort them alphabetically.
351 std::sort(names.begin(), names.end(),
352 [&isolate](IndirectHandle<String> a, IndirectHandle<String> b) {
353 return String::Compare(isolate, a, b) ==
354 ComparisonResult::kLessThan;
355 });
356
357 // Create the namespace object (initially empty).
358 Handle<JSModuleNamespace> ns = isolate->factory()->NewJSModuleNamespace();
359 ns->set_module(*module);
360 module->set_module_namespace(*ns);
361
362 // Create the properties in the namespace object. Transition the object
363 // to dictionary mode so that property addition is faster.
364 PropertyAttributes attr = DONT_DELETE;
365 JSObject::NormalizeProperties(isolate, ns, CLEAR_INOBJECT_PROPERTIES,
366 static_cast<int>(names.size()),
367 "JSModuleNamespace");
368 JSObject::NormalizeElements(ns);
369 for (const auto& name : names) {
370 uint32_t index = 0;
371 if (name->AsArrayIndex(&index)) {
372 JSObject::SetNormalizedElement(
373 ns, index, Accessors::MakeModuleNamespaceEntryInfo(isolate, name),
374 PropertyDetails(PropertyKind::kAccessor, attr,
375 PropertyCellType::kMutable));
376 } else {
377 JSObject::SetNormalizedProperty(
378 ns, name, Accessors::MakeModuleNamespaceEntryInfo(isolate, name),
379 PropertyDetails(PropertyKind::kAccessor, attr,
380 PropertyCellType::kMutable));
381 }
382 }
383 JSObject::PreventExtensions(isolate, ns, kThrowOnError).ToChecked();
384
385 // Optimize the namespace object as a prototype, for two reasons:
386 // - The object's map is guaranteed not to be shared. ICs rely on this.
387 // - We can store a pointer from the map back to the namespace object.
388 // Turbofan can use this for inlining the access.
389 JSObject::OptimizeAsPrototype(ns);
390
391 DirectHandle<PrototypeInfo> proto_info =
392 Map::GetOrCreatePrototypeInfo(ns, isolate);
393 proto_info->set_module_namespace(*ns);
394 return ns;
395}

402MaybeHandle<Object> JSModuleNamespace::GetExport(Isolate* isolate,
403 DirectHandle<String> name) {
404 DirectHandle<Object> object(module()->exports()->Lookup(name), isolate);
405 if (IsTheHole(*object, isolate)) {
406 return isolate->factory()->undefined_value();
407 }
408
409 Handle<Object> value(Cast<Cell>(*object)->value(), isolate);
410 if (IsTheHole(*value, isolate)) {
411 // According to https://tc39.es/ecma262/#sec-InnerModuleLinking
412 // step 10 and
413 // https://tc39.es/ecma262/#sec-source-text-module-record-initialize-environment
414 // step 8-25, variables must be declared in Link. And according to
415 // https://tc39.es/ecma262/#sec-module-namespace-exotic-objects-get-p-receiver,
416 // here accessing uninitialized variable error should be throwed.
417 THROW_NEW_ERROR(isolate,
418 NewReferenceError(
419 MessageTemplate::kAccessedUninitializedVariable, name));
420 }
421
422 return value;
423}

425Maybe<PropertyAttributes> JSModuleNamespace::GetPropertyAttributes(
426 LookupIterator* it) {
427 DirectHandle<JSModuleNamespace> object = it->GetHolder<JSModuleNamespace>();
428 DirectHandle<String> name = Cast<String>(it->GetName());
429 DCHECK_EQ(it->state(), LookupIterator::ACCESSOR);
430
431 Isolate* isolate = it->isolate();
432

441 return Nothing<PropertyAttributes>();
442 }
443
444 return Just(it->property_attributes());

447// ES
448// https://tc39.es/ecma262/#sec-module-namespace-exotic-objects-defineownproperty-p-desc
449// static
450Maybe<bool> JSModuleNamespace::DefineOwnProperty(
451 Isolate* isolate, DirectHandle<JSModuleNamespace> object,
452 DirectHandle<Object> key, PropertyDescriptor* desc,
453 Maybe<ShouldThrow> should_throw) {
454 // 1. If Type(P) is Symbol, return OrdinaryDefineOwnProperty(O, P, Desc).
455 if (IsSymbol(*key)) {
456 return OrdinaryDefineOwnProperty(isolate, object, key, desc, should_throw);
457 }
458
459 // 2. Let current be ? O.[[GetOwnProperty]](P).
460 PropertyKey lookup_key(isolate, key);
461 LookupIterator it(isolate, object, lookup_key, LookupIterator::OWN);
462 PropertyDescriptor current;
463 Maybe<bool> has_own = GetOwnPropertyDescriptor(&it, &current);
464 MAYBE_RETURN(has_own, Nothing<bool>());
465
466 // 3. If current is undefined, return false.
467 // 4. If Desc.[[Configurable]] is present and has value true, return false.
468 // 5. If Desc.[[Enumerable]] is present and has value false, return false.
469 // 6. If ! IsAccessorDescriptor(Desc) is true, return false.
470 // 7. If Desc.[[Writable]] is present and has value false, return false.
471 // 8. If Desc.[[Value]] is present, return
472 // SameValue(Desc.[[Value]], current.[[Value]]).
473 if (!has_own.FromJust() ||
474 (desc->has_configurable() && desc->configurable()) ||
475 (desc->has_enumerable() && !desc->enumerable()) ||
476 PropertyDescriptor::IsAccessorDescriptor(desc) ||
477 (desc->has_writable() && !desc->writable()) ||
478 (desc->has_value() &&
479 !Object::SameValue(*desc->value(), *current.value()))) {
480 RETURN_FAILURE(isolate, GetShouldThrow(isolate, should_throw),
481 NewTypeError(MessageTemplate::kRedefineDisallowed, key));
482 }
483
484 return Just(true);
485}

487bool Module::IsGraphAsync(Isolate* isolate) const {
488 DisallowGarbageCollection no_gc;
489
490 // Only SourceTextModules may be async.
491 if (!IsSourceTextModule(*this)) return false;
492 Tagged<SourceTextModule> root = Cast<SourceTextModule>(*this);
493
494 Zone zone(isolate->allocator(), ZONE_NAME);
495 const size_t bucket_count = 2;
496 ZoneUnorderedSet<Tagged<Module>, Module::Hash> visited(&zone, bucket_count);
497 ZoneVector<Tagged<SourceTextModule>> worklist(&zone);
498 visited.insert(root);
499 worklist.push_back(root);
500
501 do {
502 Tagged<SourceTextModule> current = worklist.back();
503 worklist.pop_back();
504 DCHECK_GE(current->status(), kLinked);
505
506 if (current->has_toplevel_await()) return true;
507 Tagged<FixedArray> requested_modules = current->requested_modules();
508 for (int i = 0, length = requested_modules->length(); i < length; ++i) {
509 Tagged<Module> descendant = Cast<Module>(requested_modules->get(i));
510 if (IsSourceTextModule(descendant)) {
511 const bool cycle = !visited.insert(descendant).second;
512 if (!cycle) worklist.push_back(Cast<SourceTextModule>(descendant));
513 }
514 }
515 } while (!worklist.empty());
516
517 return false;
518}
