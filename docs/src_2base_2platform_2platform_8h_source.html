5// This module contains the platform-specific code. This make the rest of the
6// code less dependent on operating system, compilers and runtime libraries.
7// This module does specifically not deal with differences between different
8// processor architecture.
9// The platform classes have the same definition for all platforms. The
10// implementation for a particular platform is put in platform_<os>.cc.
11// The build system then uses the implementation for the target platform.
12//
13// This design has been chosen because it is simple and fast. Alternatively,
14// the platform dependent classes could have been implemented using abstract
15// superclasses with virtual methods and having specializations for each
16// platform. This design was rejected because it was more complicated and
17// slower. It would require factory methods for selecting the right
18// implementation and the overhead of virtual methods for performance
19// sensitive like mutex locking/unlocking.
20

88V8_INLINE intptr_t InternalGetExistingThreadLocal(intptr_t index) {
89 const intptr_t kTibInlineTlsOffset = 0xE10;
90 const intptr_t kTibExtraTlsOffset = 0xF94;
91 const intptr_t kMaxInlineSlots = 64;
92 const intptr_t kMaxSlots = kMaxInlineSlots + 1024;
93 const intptr_t kSystemPointerSize = sizeof(void*);
94 DCHECK(0 <= index && index < kMaxSlots);
95 USE(kMaxSlots);
96 if (index < kMaxInlineSlots) {
97 return static_cast<intptr_t>(
98 __readfsdword(kTibInlineTlsOffset + kSystemPointerSize * index));
99 }
100 intptr_t extra = static_cast<intptr_t>(__readfsdword(kTibExtraTlsOffset));
101 if (!extra) return 0;
102 return *reinterpret_cast<intptr_t*>(extra + kSystemPointerSize *
103 (index - kMaxInlineSlots));
104}

107// across major releases.

443// AddressSpaceReservation
444//
445// This class provides the same memory management functions as OS but operates
446// inside a previously reserved contiguous region of virtual address space.
447//
448// Reserved address space in which no pages have been allocated is guaranteed
449// to be inaccessible and cause a fault on access. As such, creating guard
450// regions requires no further action.
451class V8_BASE_EXPORT AddressSpaceReservation {
452 public:
453 using Address = uintptr_t;
454
455 void* base() const { return base_; }
456 size_t size() const { return size_; }
457
458 bool Contains(void* region_addr, size_t region_size) const {
459 Address base = reinterpret_cast<Address>(base_);
460 Address region_base = reinterpret_cast<Address>(region_addr);
461 return (region_base >= base) &&
462 ((region_base + region_size) <= (base + size_));
463 }

677 // Returns the real stack frame if slot is part of a fake frame, and slot
678 // otherwise.
679 static StackSlot GetRealStackAddressForSlot(StackSlot slot) {
680#ifdef V8_USE_ADDRESS_SANITIZER
681 // ASAN fetches the real stack deeper in the __asan_addr_is_in_fake_stack()
682 // call (precisely, deeper in __asan_stack_malloc_()), which results in a
683 // real frame that could be outside of stack bounds. Adjust for this
684 // impreciseness here.
685 constexpr size_t kAsanRealFrameOffsetBytes = 32;
686 void* real_frame = __asan_addr_is_in_fake_stack(
687 __asan_get_current_fake_stack(), slot, nullptr, nullptr);
688 return real_frame ? StackSlot(static_cast<char*>(real_frame) +
