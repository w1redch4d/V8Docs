17// Returns either a FixedArray or, if the given {receiver} has an enum cache
18// that contains all enumerable properties of the {receiver} and its prototypes
19// have none, the map of the {receiver}. This is used to speed up the check for
20// deletions during a for-in.
21MaybeDirectHandle<HeapObject> Enumerate(Isolate* isolate,
22 DirectHandle<JSReceiver> receiver) {
23 JSObject::MakePrototypesFast(receiver, kStartAtReceiver, isolate);
24 FastKeyAccumulator accumulator(isolate, receiver,
25 KeyCollectionMode::kIncludePrototypes,
26 ENUMERABLE_STRINGS, true);
27 // Test if we have an enum cache for {receiver}.
28 if (!accumulator.is_receiver_simple_enum()) {
29 DirectHandle<FixedArray> keys;
30 ASSIGN_RETURN_ON_EXCEPTION(
31 isolate, keys,
32 accumulator.GetKeys(accumulator.may_have_elements()
33 ? GetKeysConversion::kConvertToString
34 : GetKeysConversion::kNoNumbers));
35 // Test again, since cache may have been built by GetKeys() calls above.
36 if (!accumulator.is_receiver_simple_enum()) return keys;
37 }
38 DCHECK(!IsJSModuleNamespace(*receiver));
39 return direct_handle(receiver->map(), isolate);
40}

42// This is a slight modification of JSReceiver::HasProperty, dealing with
43// the oddities of JSProxy and JSModuleNamespace in for-in filter.
44MaybeDirectHandle<Object> HasEnumerableProperty(
45 Isolate* isolate, DirectHandle<JSReceiver> receiver,
46 DirectHandle<Object> key) {
47 bool success = false;
48 Maybe<PropertyAttributes> result = Just(ABSENT);
49 PropertyKey lookup_key(isolate, key, &success);
50 if (!success) return isolate->factory()->undefined_value();
51 LookupIterator it(isolate, receiver, lookup_key);
52 for (;; it.Next()) {
53 switch (it.state()) {
54 case LookupIterator::TRANSITION:
55 UNREACHABLE();
56 case LookupIterator::JSPROXY: {
57 // For proxies we have to invoke the [[GetOwnProperty]] trap.
58 result = JSProxy::GetPropertyAttributes(&it);
59 if (result.IsNothing()) return MaybeDirectHandle<Object>();
60 if (result.FromJust() == ABSENT) {
61 // Continue lookup on the proxy's prototype.
62 DirectHandle<JSProxy> proxy = it.GetHolder<JSProxy>();
63 DirectHandle<Object> prototype;
64 ASSIGN_RETURN_ON_EXCEPTION(isolate, prototype,
65 JSProxy::GetPrototype(proxy));
66 if (IsNull(*prototype, isolate)) {
67 return isolate->factory()->undefined_value();
68 }
69 // We already have a stack-check in JSProxy::GetPrototype.
70 return HasEnumerableProperty(isolate, Cast<JSReceiver>(prototype),
71 key);
72 } else if (result.FromJust() & DONT_ENUM) {
73 return isolate->factory()->undefined_value();
74 } else {
75 return it.GetName();
76 }
77 }
78 case LookupIterator::WASM_OBJECT:
79 THROW_NEW_ERROR(isolate,
80 NewTypeError(MessageTemplate::kWasmObjectsAreOpaque));
81 case LookupIterator::INTERCEPTOR: {
82 result = JSObject::GetPropertyAttributesWithInterceptor(&it);
83 if (result.IsNothing()) return MaybeDirectHandle<Object>();
84 if (result.FromJust() != ABSENT) return it.GetName();
85 continue;
86 }
87 case LookupIterator::ACCESS_CHECK: {
88 if (it.HasAccess()) continue;
89 result = JSObject::GetPropertyAttributesWithFailedAccessCheck(&it);
90 if (result.IsNothing()) return MaybeDirectHandle<Object>();
91 if (result.FromJust() != ABSENT) return it.GetName();
92 return isolate->factory()->undefined_value();
93 }
94 case LookupIterator::TYPED_ARRAY_INDEX_NOT_FOUND:
95 // TypedArray out-of-bounds access.
96 return isolate->factory()->undefined_value();
97 case LookupIterator::ACCESSOR: {
98 if (IsJSModuleNamespace(*it.GetHolder<Object>())) {
99 result = JSModuleNamespace::GetPropertyAttributes(&it);
100 if (result.IsNothing()) return MaybeDirectHandle<Object>();
101 DCHECK_EQ(0, result.FromJust() & DONT_ENUM);
102 }
103 return it.GetName();
104 }
105 case LookupIterator::DATA:
106 return it.GetName();
107 case LookupIterator::NOT_FOUND:
108 return isolate->factory()->undefined_value();
109 }
110 UNREACHABLE();
111 }
112}

125RUNTIME_FUNCTION(Runtime_ForInHasProperty) {
126 HandleScope scope(isolate);
127 DCHECK_EQ(2, args.length());
128 DirectHandle<JSReceiver> receiver = args.at<JSReceiver>(0);
129 DirectHandle<Object> key = args.at(1);
130 DirectHandle<Object> result;
131 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
132 isolate, result, HasEnumerableProperty(isolate, receiver, key));
133 return isolate->heap()->ToBoolean(!IsUndefined(*result, isolate));
