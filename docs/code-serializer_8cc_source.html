34AlignedCachedData::AlignedCachedData(const uint8_t* data, int length)
35 : owns_data_(false), rejected_(false), data_(data), length_(length) {
36 if (!IsAligned(reinterpret_cast<intptr_t>(data), kPointerAlignment)) {
37 uint8_t* copy = NewArray<uint8_t>(length);
38 DCHECK(IsAligned(reinterpret_cast<intptr_t>(copy), kPointerAlignment));
39 CopyBytes(copy, data, length);
40 data_ = copy;
41 AcquireDataOwnership();
42 }

50ScriptCompiler::CachedData* CodeSerializer::Serialize(
51 Isolate* isolate, Handle<SharedFunctionInfo> info) {
52 TRACE_EVENT_CALL_STATS_SCOPED(isolate, "v8", "V8.Execute");
53 NestedTimedHistogramScope histogram_timer(
54 isolate->counters()->compile_serialize());
55 RCS_SCOPE(isolate, RuntimeCallCounterId::kCompileSerialize);
56 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"), "V8.CompileSerialize");
57
58 base::ElapsedTimer timer;
59 if (v8_flags.profile_deserialization) timer.Start();
60 DirectHandle<Script> script(Cast<Script>(info->script()), isolate);
61 if (v8_flags.trace_serializer) {
62 PrintF("[Serializing from");
63 ShortPrint(script->name());
64 PrintF("]\n");
65 }
66#if V8_ENABLE_WEBASSEMBLY
67 // TODO(7110): Enable serialization of Asm modules once the AsmWasmData is
68 // context independent.
69 if (script->ContainsAsmModule()) return nullptr;
70#endif // V8_ENABLE_WEBASSEMBLY
71
72 // Serialize code object.
73 DirectHandle<String> source(Cast<String>(script->source()), isolate);
74 DirectHandle<FixedArray> wrapped_arguments;
75 if (script->is_wrapped()) {
76 wrapped_arguments =
77 DirectHandle<FixedArray>(script->wrapped_arguments(), isolate);
78 }
79
80 HandleScope scope(isolate);
81 CodeSerializer cs(isolate,
82 SerializedCodeData::SourceHash(source, wrapped_arguments,
83 script->origin_options()));
84 DisallowGarbageCollection no_gc;
85
86#ifndef DEBUG
87 cs.reference_map()->AddAttachedReference(*source);
88#endif
89 AlignedCachedData* cached_data = cs.SerializeSharedFunctionInfo(info);
90
91 if (v8_flags.profile_deserialization) {
92 double ms = timer.Elapsed().InMillisecondsF();
93 int length = cached_data->length();
94 PrintF("[Serializing to %d bytes took %0.3f ms]\n", length, ms);
95 }
96
97 ScriptCompiler::CachedData* result =
98 new ScriptCompiler::CachedData(cached_data->data(), cached_data->length(),
99 ScriptCompiler::CachedData::BufferOwned);
100 cached_data->ReleaseDataOwnership();
101 delete cached_data;
102
103 return result;
104}

120void CodeSerializer::SerializeObjectImpl(Handle<HeapObject> obj,
121 SlotType slot_type) {
122 ReadOnlyRoots roots(isolate());
123 InstanceType instance_type;
124 {
125 DisallowGarbageCollection no_gc;
126 Tagged<HeapObject> raw = *obj;
127 if (SerializeHotObject(raw)) return;
128 if (SerializeRoot(raw)) return;
129 if (SerializeBackReference(raw)) return;
130 if (SerializeReadOnlyObjectReference(raw, &sink_)) return;
131
132 instance_type = raw->map()->instance_type();
133 CHECK(!InstanceTypeChecker::IsInstructionStream(instance_type));
134 }
135
136 if (InstanceTypeChecker::IsScript(instance_type)) {
137 DirectHandle<FixedArray> host_options;
138 DirectHandle<UnionOf<Smi, Symbol, Undefined>> context_data;
139 {
140 DisallowGarbageCollection no_gc;
141 Tagged<Script> script_obj = Cast<Script>(*obj);
142 DCHECK_NE(script_obj->compilation_type(), Script::CompilationType::kEval);
143 // We want to differentiate between undefined and uninitialized_symbol for
144 // context_data for now. It is hack to allow debugging for scripts that
145 // are included as a part of custom snapshot. (see
146 // debug::Script::IsEmbedded())
147 Tagged<UnionOf<Smi, Symbol, Undefined>> raw_context_data =
148 script_obj->context_data();
149 if (raw_context_data != roots.undefined_value() &&
150 raw_context_data != roots.uninitialized_symbol()) {
151 script_obj->set_context_data(roots.undefined_value());
152 }
153 context_data = direct_handle(raw_context_data, isolate());
154 // We don't want to serialize host options to avoid serializing
155 // unnecessary object graph.
156 host_options =
157 direct_handle(script_obj->host_defined_options(), isolate());
158 script_obj->set_host_defined_options(roots.empty_fixed_array());
159 }
160 SerializeGeneric(obj, slot_type);
161 {
162 DisallowGarbageCollection no_gc;
163 Tagged<Script> script_obj = Cast<Script>(*obj);
164 script_obj->set_host_defined_options(*host_options);
165 script_obj->set_context_data(*context_data);
166 }
167 return;
168 } else if (InstanceTypeChecker::IsSharedFunctionInfo(instance_type)) {
169 DirectHandle<DebugInfo> debug_info;
170 CachedTieringDecision cached_tiering_decision;
171 bool restore_bytecode = false;
172 {
173 DisallowGarbageCollection no_gc;
174 Tagged<SharedFunctionInfo> sfi = Cast<SharedFunctionInfo>(*obj);
175 DCHECK(!sfi->IsApiFunction());
176#if V8_ENABLE_WEBASSEMBLY
177 // TODO(7110): Enable serializing of Asm modules once the AsmWasmData
178 // is context independent.
179 DCHECK(!sfi->HasAsmWasmData());
180#endif // V8_ENABLE_WEBASSEMBLY
181
182 if (auto maybe_debug_info = sfi->TryGetDebugInfo(isolate())) {
183 debug_info = direct_handle(maybe_debug_info.value(), isolate());
184 // Clear debug info.
185 if (debug_info->HasInstrumentedBytecodeArray()) {
186 restore_bytecode = true;
187 sfi->SetActiveBytecodeArray(
188 debug_info->OriginalBytecodeArray(isolate()), isolate());
189 }
190 }
191 if (v8_flags.profile_guided_optimization) {
192 cached_tiering_decision = sfi->cached_tiering_decision();
193 if (cached_tiering_decision > CachedTieringDecision::kEarlySparkplug) {
194 sfi->set_cached_tiering_decision(
195 CachedTieringDecision::kEarlySparkplug);
196 }
197 }
198 }
199 SerializeGeneric(obj, slot_type);
200 DisallowGarbageCollection no_gc;
201 Tagged<SharedFunctionInfo> sfi = Cast<SharedFunctionInfo>(*obj);
202 if (restore_bytecode) {
203 sfi->SetActiveBytecodeArray(debug_info->DebugBytecodeArray(isolate()),
204 isolate());
205 }
206 if (v8_flags.profile_guided_optimization &&
207 cached_tiering_decision > CachedTieringDecision::kEarlySparkplug) {
208 sfi->set_cached_tiering_decision(cached_tiering_decision);
209 }
210 return;
211 } else if (InstanceTypeChecker::IsUncompiledDataWithoutPreparseDataWithJob(
212 instance_type)) {
213 Handle<UncompiledDataWithoutPreparseDataWithJob> data =
214 Cast<UncompiledDataWithoutPreparseDataWithJob>(obj);
215 Address job = data->job();
216 data->set_job(kNullAddress);
217 SerializeGeneric(data, slot_type);
218 data->set_job(job);
219 return;
220 } else if (InstanceTypeChecker::IsUncompiledDataWithPreparseDataAndJob(
221 instance_type)) {
222 Handle<UncompiledDataWithPreparseDataAndJob> data =
223 Cast<UncompiledDataWithPreparseDataAndJob>(obj);
224 Address job = data->job();
225 data->set_job(kNullAddress);
226 SerializeGeneric(data, slot_type);
227 data->set_job(job);
228 return;
229 } else if (InstanceTypeChecker::IsScopeInfo(instance_type)) {
230 // TODO(ishell): define a dedicated instance type for DependentCode and
231 // serialize DependentCode objects as an empty_dependent_code instead
232 // of customizing ScopeInfo serialization.
233 static_assert(DEPENDENT_CODE_TYPE == WEAK_ARRAY_LIST_TYPE);
234 Handle<ScopeInfo> scope_info = Cast<ScopeInfo>(obj);
235 DirectHandle<DependentCode> dependent_code;
236 bool restore_dependent_code = false;
237 if (scope_info->SloppyEvalCanExtendVars()) {
238 // If |scope_info| has a dependent code field, serialize it as an empty
239 // dependent code in order to avoid accidental serialization of optimized
240 // code.
241 Tagged<DependentCode> empty_dependent_code =
242 DependentCode::empty_dependent_code(ReadOnlyRoots(isolate()));
243 if (scope_info->dependent_code() != empty_dependent_code) {
244 dependent_code = direct_handle(scope_info->dependent_code(), isolate());
245 restore_dependent_code = true;
246 scope_info->set_dependent_code(empty_dependent_code);
247 }
248 }
249 SerializeGeneric(scope_info, slot_type);
250 if (restore_dependent_code) {
251 scope_info->set_dependent_code(*dependent_code);
252 }
253 return;
254 }
255
256 // NOTE(mmarchini): If we try to serialize an InterpreterData our process
257 // will crash since it stores a code object. Instead, we serialize the
258 // bytecode array stored within the InterpreterData, which is the important
259 // information. On deserialization we'll create our code objects again, if
260 // --interpreted-frames-native-stack is on. See v8:9122 for more context
261 if (V8_UNLIKELY(isolate()->interpreted_frames_native_stack()) &&
262 IsInterpreterData(*obj)) {
263 obj = handle(Cast<InterpreterData>(*obj)->bytecode_array(), isolate());
264 }
265
266 // Past this point we should not see any (context-specific) maps anymore.
267 CHECK(!InstanceTypeChecker::IsMap(instance_type));
268 // There should be no references to the global object embedded.
269 CHECK(!InstanceTypeChecker::IsJSGlobalProxy(instance_type) &&
270 !InstanceTypeChecker::IsJSGlobalObject(instance_type));
271 // Embedded FixedArrays that need rehashing must support rehashing.
272 CHECK_IMPLIES(obj->NeedsRehashing(cage_base()),
273 obj->CanBeRehashed(cage_base()));
274 // We expect no instantiated function objects or contexts.
275 CHECK(!InstanceTypeChecker::IsJSFunction(instance_type) &&
276 !InstanceTypeChecker::IsContext(instance_type));
277
278 SerializeGeneric(obj, slot_type);
279}

290// NOTE(mmarchini): when v8_flags.interpreted_frames_native_stack is on, we want
291// to create duplicates of InterpreterEntryTrampoline for the deserialized
292// functions, otherwise we'll call the builtin IET for those functions (which
293// is not what a user of this flag wants).
294void CreateInterpreterDataForDeserializedCode(
295 Isolate* isolate, DirectHandle<SharedFunctionInfo> result_sfi,
296 bool log_code_creation) {
297 DCHECK_IMPLIES(log_code_creation, isolate->NeedsSourcePositions());
298
299 DirectHandle<Script> script(Cast<Script>(result_sfi->script()), isolate);
300 if (log_code_creation) Script::InitLineEnds(isolate, script);
301
302 Tagged<String> name = ReadOnlyRoots(isolate).empty_string();
303 if (IsString(script->name())) name = Cast<String>(script->name());
304 Handle<String> name_handle(name, isolate);
305
306 SharedFunctionInfo::ScriptIterator iter(isolate, *script);
307 for (Tagged<SharedFunctionInfo> shared_info = iter.Next();
308 !shared_info.is_null(); shared_info = iter.Next()) {
309 IsCompiledScope is_compiled(shared_info, isolate);
310 if (!is_compiled.is_compiled()) continue;
311 DCHECK(shared_info->HasBytecodeArray());
312 Handle<SharedFunctionInfo> sfi = handle(shared_info, isolate);
313
314 DirectHandle<BytecodeArray> bytecode(sfi->GetBytecodeArray(isolate),
315 isolate);
316 Handle<Code> code =
317 Builtins::CreateInterpreterEntryTrampolineForProfiling(isolate);
318 DirectHandle<InterpreterData> interpreter_data =
319 isolate->factory()->NewInterpreterData(bytecode, code);
320
321 if (sfi->HasBaselineCode()) {
322 sfi->baseline_code(kAcquireLoad)
323 ->set_bytecode_or_interpreter_data(*interpreter_data);
324 } else {
325 sfi->set_interpreter_data(isolate, *interpreter_data);
326 }
327
328 if (!log_code_creation) continue;
329
330 Handle<AbstractCode> abstract_code = Cast<AbstractCode>(code);
331 Script::PositionInfo info;
332 Script::GetPositionInfo(script, sfi->StartPosition(), &info);
333 int line_num = info.line_start + 1;
334 int column_num = info.line_end + 1;
335 PROFILE(isolate,
336 CodeCreateEvent(LogEventListener::CodeTag::kFunction, abstract_code,
337 sfi, name_handle, line_num, column_num));
338 }

341class StressOffThreadDeserializeThread final : public base::Thread {
342 public:
343 explicit StressOffThreadDeserializeThread(Isolate* isolate,
344 AlignedCachedData* cached_data)
345 : Thread(
346 base::Thread::Options("StressOffThreadDeserializeThread", 2 * MB)),
347 isolate_(isolate),
348 cached_data_(cached_data) {}

350 void Run() final {
351 LocalIsolate local_isolate(isolate_, ThreadKind::kBackground);
352 UnparkedScope unparked_scope(&local_isolate);
353 LocalHandleScope handle_scope(&local_isolate);

356 }

364 }

366 private:
367 Isolate* isolate_;
368 AlignedCachedData* cached_data_;
369 CodeSerializer::OffThreadDeserializeData off_thread_data_;
370};

372void FinalizeDeserialization(Isolate* isolate,
373 DirectHandle<SharedFunctionInfo> result,
374 const base::ElapsedTimer& timer,
375 const ScriptDetails& script_details) {
376 // Devtools can report time in this function as profiler overhead, since none
377 // of the following tasks would need to happen normally.
378 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
379 "V8.FinalizeDeserialization");
380
381 const bool log_code_creation = isolate->IsLoggingCodeCreation();
382
383 if (V8_UNLIKELY(isolate->interpreted_frames_native_stack())) {
384 CreateInterpreterDataForDeserializedCode(isolate, result,
385 log_code_creation);
386 }
387
388 DirectHandle<Script> script(Cast<Script>(result->script()), isolate);
389 // Reset the script details, including host-defined options.
390 {
391 DisallowGarbageCollection no_gc;
392 SetScriptFieldsFromDetails(isolate, *script, script_details, &no_gc);
393 }
394
395 bool needs_source_positions = isolate->NeedsSourcePositions();
396 if (!log_code_creation && !needs_source_positions) return;
397
398 if (needs_source_positions) {
399 Script::InitLineEnds(isolate, script);
400 }
401
402 Handle<String> name(IsString(script->name())
403 ? Cast<String>(script->name())
404 : ReadOnlyRoots(isolate).empty_string(),
405 isolate);
406
407 if (V8_UNLIKELY(v8_flags.log_function_events)) {
408 LOG(isolate,
409 FunctionEvent("deserialize", script->id(),
410 timer.Elapsed().InMillisecondsF(),
411 result->StartPosition(), result->EndPosition(), *name));
412 }
413
414 SharedFunctionInfo::ScriptIterator iter(isolate, *script);
415 for (Tagged<SharedFunctionInfo> info = iter.Next(); !info.is_null();
416 info = iter.Next()) {
417 if (!info->is_compiled()) continue;
418 Handle<SharedFunctionInfo> shared_info(info, isolate);
419 if (needs_source_positions) {
420 SharedFunctionInfo::EnsureSourcePositionsAvailable(isolate, shared_info);
421 }
422 Script::PositionInfo pos_info;
423 Script::GetPositionInfo(script, shared_info->StartPosition(), &pos_info);
424 int line_num = pos_info.line + 1;
425 int column_num = pos_info.column + 1;
426 PROFILE(isolate, CodeCreateEvent(
427 shared_info->is_toplevel()
428 ? LogEventListener::CodeTag::kScript
429 : LogEventListener::CodeTag::kFunction,
430 handle(shared_info->abstract_code(isolate), isolate),
431 shared_info, name, line_num, column_num));
432 }
433}

480MaybeDirectHandle<SharedFunctionInfo> CodeSerializer::Deserialize(
481 Isolate* isolate, AlignedCachedData* cached_data,
482 DirectHandle<String> source, const ScriptDetails& script_details,
483 MaybeDirectHandle<Script> maybe_cached_script) {
484 if (v8_flags.stress_background_compile) {
485 StressOffThreadDeserializeThread thread(isolate, cached_data);
486 CHECK(thread.Start());
487 thread.Join();
488 return thread.Finalize(isolate, source, script_details);
489 // TODO(leszeks): Compare off-thread deserialized data to on-thread.
490 }
491
492 base::ElapsedTimer timer;
493 if (v8_flags.profile_deserialization || v8_flags.log_function_events) {
494 timer.Start();
495 }
496
497 HandleScope scope(isolate);
498
499 DirectHandle<FixedArray> wrapped_arguments;
500 if (!script_details.wrapped_arguments.is_null()) {
501 wrapped_arguments = script_details.wrapped_arguments.ToHandleChecked();
502 }
503
504 SerializedCodeSanityCheckResult sanity_check_result =
505 SerializedCodeSanityCheckResult::kSuccess;
506 const SerializedCodeData scd = SerializedCodeData::FromCachedData(
507 isolate, cached_data,
508 SerializedCodeData::SourceHash(source, wrapped_arguments,
509 script_details.origin_options),
510 &sanity_check_result);
511 if (sanity_check_result != SerializedCodeSanityCheckResult::kSuccess) {
512 if (v8_flags.profile_deserialization) {
513 PrintF("[Cached code failed check: %s]\n", ToString(sanity_check_result));
514 }
515 DCHECK(cached_data->rejected());
516 isolate->counters()->code_cache_reject_reason()->AddSample(
517 static_cast<int>(sanity_check_result));
518 return MaybeDirectHandle<SharedFunctionInfo>();
519 }
520
521 // Deserialize.
522 MaybeDirectHandle<SharedFunctionInfo> maybe_result =
523 ObjectDeserializer::DeserializeSharedFunctionInfo(isolate, &scd, source);
524
525 DirectHandle<SharedFunctionInfo> result;
526 if (!maybe_result.ToHandle(&result)) {
527 // Deserializing may fail if the reservations cannot be fulfilled.
528 if (v8_flags.profile_deserialization) PrintF("[Deserializing failed]\n");
529 return MaybeDirectHandle<SharedFunctionInfo>();
530 }
531
532 // Check whether the newly deserialized data should be merged into an
533 // existing Script from the Isolate compilation cache. If so, perform
534 // the merge in a single-threaded manner since this deserialization was
535 // single-threaded.
536 if (DirectHandle<Script> cached_script;
537 maybe_cached_script.ToHandle(&cached_script)) {
538 BackgroundMergeTask merge;
539 merge.SetUpOnMainThread(isolate, cached_script);
540 CHECK(merge.HasPendingBackgroundWork());
541 DirectHandle<Script> new_script(Cast<Script>(result->script()), isolate);
542 merge.BeginMergeInBackground(isolate->AsLocalIsolate(), new_script);
543 CHECK(merge.HasPendingForegroundWork());
544 result = merge.CompleteMergeInForeground(isolate, new_script);
545 }
546
547 Tagged<Script> script = Cast<Script>(result->script());
548 script->set_deserialized(true);
549 BaselineBatchCompileIfSparkplugCompiled(isolate, script);
550 if (v8_flags.profile_deserialization) {
551 double ms = timer.Elapsed().InMillisecondsF();
552 int length = cached_data->length();
553 PrintF("[Deserializing from %d bytes took %0.3f ms]\n", length, ms);
554 }
555
556 FinalizeDeserialization(isolate, result, timer, script_details);
557
558 return scope.CloseAndEscape(result);
559}

561Handle<Script> CodeSerializer::OffThreadDeserializeData::GetOnlyScript(
562 LocalHeap* heap) {
563 std::unique_ptr<PersistentHandles> previous_persistent_handles =
564 heap->DetachPersistentHandles();
565 heap->AttachPersistentHandles(std::move(persistent_handles));
566
567 DCHECK_EQ(scripts.size(), 1);
568 // Make a non-persistent handle to return.
569 Handle<Script> script = handle(*scripts[0], heap);
570 DCHECK_EQ(*script, maybe_result.ToHandleChecked()->script());
571
572 persistent_handles = heap->DetachPersistentHandles();
573 if (previous_persistent_handles) {
574 heap->AttachPersistentHandles(std::move(previous_persistent_handles));
575 }
576
577 return script;

580CodeSerializer::OffThreadDeserializeData
581CodeSerializer::StartDeserializeOffThread(LocalIsolate* local_isolate,
582 AlignedCachedData* cached_data) {
583 OffThreadDeserializeData result;
584
585 DCHECK(!local_isolate->heap()->HasPersistentHandles());
586
587 const SerializedCodeData scd =
588 SerializedCodeData::FromCachedDataWithoutSource(
589 local_isolate, cached_data, &result.sanity_check_result);
590 if (result.sanity_check_result != SerializedCodeSanityCheckResult::kSuccess) {
591 // Exit early but don't report yet, we'll re-check this when finishing on
592 // the main thread
593 DCHECK(cached_data->rejected());
594 return result;
595 }
596
597 MaybeDirectHandle<SharedFunctionInfo> local_maybe_result =
598 OffThreadObjectDeserializer::DeserializeSharedFunctionInfo(
599 local_isolate, &scd, &result.scripts);
600
601 result.maybe_result =
602 local_isolate->heap()->NewPersistentMaybeHandle(local_maybe_result);
603 result.persistent_handles = local_isolate->heap()->DetachPersistentHandles();
604
605 return result;
606}

608MaybeHandle<SharedFunctionInfo> CodeSerializer::FinishOffThreadDeserialize(
609 Isolate* isolate, OffThreadDeserializeData&& data,
610 AlignedCachedData* cached_data, DirectHandle<String> source,
611 const ScriptDetails& script_details,
612 BackgroundMergeTask* background_merge_task) {
613 base::ElapsedTimer timer;
614 if (v8_flags.profile_deserialization || v8_flags.log_function_events) {
615 timer.Start();
616 }
617
618 HandleScope scope(isolate);
619
620 DirectHandle<FixedArray> wrapped_arguments;
621 if (!script_details.wrapped_arguments.is_null()) {
622 wrapped_arguments = script_details.wrapped_arguments.ToHandleChecked();
623 }
624
625 // Do a source sanity check now that we have the source. It's important for
626 // FromPartiallySanityCheckedCachedData call that the sanity_check_result
627 // holds the result of the off-thread sanity check.
628 SerializedCodeSanityCheckResult sanity_check_result =
629 data.sanity_check_result;
630 const SerializedCodeData scd =
631 SerializedCodeData::FromPartiallySanityCheckedCachedData(
632 cached_data,
633 SerializedCodeData::SourceHash(source, wrapped_arguments,
634 script_details.origin_options),
635 &sanity_check_result);
636 if (sanity_check_result != SerializedCodeSanityCheckResult::kSuccess) {
637 // The only case where the deserialization result could exist despite a
638 // check failure is on a source mismatch, since we can't test for this
639 // off-thread.
640 DCHECK_IMPLIES(!data.maybe_result.is_null(),
641 sanity_check_result ==
642 SerializedCodeSanityCheckResult::kSourceMismatch);
643 // The only kind of sanity check we can't test for off-thread is a source
644 // mismatch.
645 DCHECK_IMPLIES(sanity_check_result != data.sanity_check_result,
646 sanity_check_result ==
647 SerializedCodeSanityCheckResult::kSourceMismatch);
648 if (v8_flags.profile_deserialization) {
649 PrintF("[Cached code failed check: %s]\n", ToString(sanity_check_result));
650 }
651 DCHECK(cached_data->rejected());
652 isolate->counters()->code_cache_reject_reason()->AddSample(
653 static_cast<int>(sanity_check_result));
654 return MaybeHandle<SharedFunctionInfo>();
655 }
656
657 Handle<SharedFunctionInfo> result;
658 if (!data.maybe_result.ToHandle(&result)) {
659 // Deserializing may fail if the reservations cannot be fulfilled.
660 if (v8_flags.profile_deserialization) {
661 PrintF("[Off-thread deserializing failed]\n");
662 }
663 return MaybeHandle<SharedFunctionInfo>();
664 }
665
666 // Change the result persistent handle into a regular handle.
667 DCHECK(data.persistent_handles->Contains(result.location()));
668 result = handle(*result, isolate);
669
670 if (background_merge_task &&
671 background_merge_task->HasPendingForegroundWork()) {
672 DCHECK_EQ(data.scripts.size(), 1);
673 DirectHandle<Script> new_script = data.scripts[0];
674 result =
675 background_merge_task->CompleteMergeInForeground(isolate, new_script);
676 DCHECK(Object::StrictEquals(Cast<Script>(result->script())->source(),
677 *source));
678 DCHECK(isolate->factory()->script_list()->Contains(
679 MakeWeak(result->script())));
680 } else {
681 DirectHandle<Script> script(Cast<Script>(result->script()), isolate);
682 // Fix up the source on the script. This should be the only deserialized
683 // script, and the off-thread deserializer should have set its source to the
684 // empty string. In debug mode the code cache does contain the original
685 // source.
686 DCHECK_EQ(data.scripts.size(), 1);
687 DCHECK_EQ(*script, *data.scripts[0]);
688#ifdef DEBUG
689 if (!Cast<String>(script->source())->Equals(*source)) {
690 isolate->PushStackTraceAndDie(
691 reinterpret_cast<void*>(script->source().ptr()),
692 reinterpret_cast<void*>(source->ptr()));
693 }
694#else
695 CHECK_EQ(script->source(), ReadOnlyRoots(isolate).empty_string());
696#endif
697 Script::SetSource(isolate, script, source);
698
699 // Fix up the script list to include the newly deserialized script.
700 Handle<WeakArrayList> list = isolate->factory()->script_list();
701 for (Handle<Script> script : data.scripts) {
702 script->set_deserialized(true);
703 BaselineBatchCompileIfSparkplugCompiled(isolate, *script);
704 DCHECK(data.persistent_handles->Contains(script.location()));
705 list = WeakArrayList::AddToEnd(isolate, list,
706 MaybeObjectDirectHandle::Weak(script));
707 }
708 isolate->heap()->SetRootScriptList(*list);
709 }
710
711 if (v8_flags.profile_deserialization) {
712 double ms = timer.Elapsed().InMillisecondsF();
713 int length = cached_data->length();
714 PrintF("[Finishing off-thread deserialize from %d bytes took %0.3f ms]\n",
715 length, ms);
716 }
717
718 FinalizeDeserialization(isolate, result, timer, script_details);
719
720 DCHECK(!background_merge_task ||
721 !background_merge_task->HasPendingForegroundWork());
722
723 return scope.CloseAndEscape(result);
724}

726SerializedCodeData::SerializedCodeData(const std::vector<uint8_t>* payload,
727 const CodeSerializer* cs) {
728 DisallowGarbageCollection no_gc;
729
730 // Calculate sizes.
731 uint32_t size = kHeaderSize + static_cast<uint32_t>(payload->size());
732 DCHECK(IsAligned(size, kPointerAlignment));
733
734 // Allocate backing store and create result data.
735 AllocateData(size);
736
737 // Zero out pre-payload data. Part of that is only used for padding.
738 memset(data_, 0, kHeaderSize);
739
740 // Set header values.
741 SetMagicNumber();
742 SetHeaderValue(kVersionHashOffset, Version::Hash());
743 SetHeaderValue(kSourceHashOffset, cs->source_hash());
744 SetHeaderValue(kFlagHashOffset, FlagList::Hash());
745 SetHeaderValue(kReadOnlySnapshotChecksumOffset,
746 Snapshot::ExtractReadOnlySnapshotChecksum(
747 cs->isolate()->snapshot_blob()));
748 SetHeaderValue(kPayloadLengthOffset, static_cast<uint32_t>(payload->size()));
749
750 // Zero out any padding in the header.
751 memset(data_ + kUnalignedHeaderSize, 0, kHeaderSize - kUnalignedHeaderSize);
752
753 // Copy serialized data.
754 CopyBytes(data_ + kHeaderSize, payload->data(),
755 static_cast<size_t>(payload->size()));
756 uint32_t checksum =
757 v8_flags.verify_snapshot_checksum ? Checksum(ChecksummedContent()) : 0;
758 SetHeaderValue(kChecksumOffset, checksum);
759}

761SerializedCodeSanityCheckResult SerializedCodeData::SanityCheck(
762 uint32_t expected_ro_snapshot_checksum,
763 uint32_t expected_source_hash) const {
764 SerializedCodeSanityCheckResult result =
765 SanityCheckWithoutSource(expected_ro_snapshot_checksum);
766 if (result != SerializedCodeSanityCheckResult::kSuccess) return result;
767 return SanityCheckJustSource(expected_source_hash);
768}
769
770SerializedCodeSanityCheckResult SerializedCodeData::SanityCheckJustSource(
771 uint32_t expected_source_hash) const {
772 uint32_t source_hash = GetHeaderValue(kSourceHashOffset);
773 if (source_hash != expected_source_hash) {
774 return SerializedCodeSanityCheckResult::kSourceMismatch;
775 }
776 return SerializedCodeSanityCheckResult::kSuccess;
777}
778
779SerializedCodeSanityCheckResult SerializedCodeData::SanityCheckWithoutSource(
780 uint32_t expected_ro_snapshot_checksum) const {
781 if (size_ < kHeaderSize) {
782 return SerializedCodeSanityCheckResult::kInvalidHeader;
783 }
784 uint32_t magic_number = GetMagicNumber();
785 if (magic_number != kMagicNumber) {
786 return SerializedCodeSanityCheckResult::kMagicNumberMismatch;
787 }
788 uint32_t version_hash = GetHeaderValue(kVersionHashOffset);
789 if (version_hash != Version::Hash()) {
790 return SerializedCodeSanityCheckResult::kVersionMismatch;
791 }
792 uint32_t flags_hash = GetHeaderValue(kFlagHashOffset);
793 if (flags_hash != FlagList::Hash()) {
794 return SerializedCodeSanityCheckResult::kFlagsMismatch;
795 }
796 uint32_t ro_snapshot_checksum =
797 GetHeaderValue(kReadOnlySnapshotChecksumOffset);
798 if (ro_snapshot_checksum != expected_ro_snapshot_checksum) {
799 return SerializedCodeSanityCheckResult::kReadOnlySnapshotChecksumMismatch;
800 }
801 uint32_t payload_length = GetHeaderValue(kPayloadLengthOffset);
802 uint32_t max_payload_length = size_ - kHeaderSize;
803 if (payload_length > max_payload_length) {
804 return SerializedCodeSanityCheckResult::kLengthMismatch;
805 }
806 if (v8_flags.verify_snapshot_checksum) {
807 uint32_t checksum = GetHeaderValue(kChecksumOffset);
808 if (Checksum(ChecksummedContent()) != checksum) {
809 return SerializedCodeSanityCheckResult::kChecksumMismatch;
810 }
811 }
812 return SerializedCodeSanityCheckResult::kSuccess;
813}

815uint32_t SerializedCodeData::SourceHash(
816 DirectHandle<String> source, DirectHandle<FixedArray> wrapped_arguments,
817 ScriptOriginOptions origin_options) {
818 using LengthField = base::BitField<uint32_t, 0, 29>;
819 static_assert(String::kMaxLength <= LengthField::kMax,
820 "String length must fit into a LengthField");
821 using HasWrappedArgumentsField = LengthField::Next<bool, 1>;
822 using IsModuleField = HasWrappedArgumentsField::Next<bool, 1>;
823
824 uint32_t hash = 0;
825 hash = LengthField::update(hash, source->length());
826 hash = HasWrappedArgumentsField::update(hash, !wrapped_arguments.is_null());
827 hash = IsModuleField::update(hash, origin_options.IsModule());
828 return hash;
829}

883SerializedCodeData SerializedCodeData::FromPartiallySanityCheckedCachedData(
884 AlignedCachedData* cached_data, uint32_t expected_source_hash,
885 SerializedCodeSanityCheckResult* rejection_result) {
886 DisallowGarbageCollection no_gc;
887 // The previous call to FromCachedDataWithoutSource may have already rejected
888 // the cached data, so re-use the previous rejection result if it's not a
889 // success.
890 if (*rejection_result != SerializedCodeSanityCheckResult::kSuccess) {
891 // FromCachedDataWithoutSource doesn't check the source, so there can't be
892 // a source mismatch.
893 DCHECK_NE(*rejection_result,
894 SerializedCodeSanityCheckResult::kSourceMismatch);
895 cached_data->Reject();
896 return SerializedCodeData(nullptr, 0);
897 }
898 SerializedCodeData scd(cached_data);
899 *rejection_result = scd.SanityCheckJustSource(expected_source_hash);
900 if (*rejection_result != SerializedCodeSanityCheckResult::kSuccess) {
901 // This check only checks the source, so the only possible failure is a
902 // source mismatch.
903 DCHECK_EQ(*rejection_result,
904 SerializedCodeSanityCheckResult::kSourceMismatch);
905 cached_data->Reject();
906 return SerializedCodeData(nullptr, 0);
907 }
908 return scd;
909}
