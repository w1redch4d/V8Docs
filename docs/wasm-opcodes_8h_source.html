896// Prefixed opcodes are encoded as 1 prefix byte, followed by LEB encoded
897// opcode bytes. We internally encode them as {WasmOpcode} as follows:
898// 1) non-prefixed opcodes use the opcode itself as {WasmOpcode} enum value;
899// 2) prefixed opcodes in [0, 0xff] use {(prefix << 8) | opcode};
900// 3) prefixed opcodes in [0x100, 0xfff] use {(prefix << 12) | opcode} (this is
901// only used for relaxed simd so far).
902//
903// This encoding is bijective (i.e. a one-to-one mapping in both directions).
904// The used opcode ranges are:
905// 1) [0, 0xff] -> no prefix, 8 bits opcode
906// 2) [0xfb00, 0xfe00] -> prefix shifted by 8 bits, and 8 bits opcode
907// 3) [0xfd100, 0xfdfff] -> prefix shifted by 12 bits, and 12 bits opcode
908// (only [0xfd100, 0xfd1ff] used so far)
909//
910// This allows to compute back the prefix and the non-prefixed opcode from each
911// WasmOpcode, see {WasmOpcodes::ExtractPrefix} and
912// {ExtractPrefixedOpcodeBytes} (for testing).
913enum WasmOpcode {
914// Declare expression opcodes.
915#define DECLARE_NAMED_ENUM(name, opcode, ...) kExpr##name = opcode,
916 FOREACH_OPCODE(DECLARE_NAMED_ENUM)
917#undef DECLARE_NAMED_ENUM
918#define DECLARE_PREFIX(name, opcode) k##name##Prefix = opcode,
919 FOREACH_PREFIX(DECLARE_PREFIX)
920#undef DECLARE_PREFIX
921};
