31void TypeCanonicalizer::AddRecursiveGroup(WasmModule* module, uint32_t size) {
32 if (size == 0) return;
33 // If the caller knows statically that {size == 1}, it should have called
34 // {AddRecursiveSingletonGroup} directly. For cases where this is not
35 // statically determined we add this dispatch here.
36 if (size == 1) return AddRecursiveSingletonGroup(module);
37
38 uint32_t start_index = static_cast<uint32_t>(module->types.size() - size);
39
40 // Multiple threads could try to register recursive groups concurrently.
41 // TODO(manoskouk): Investigate if we can fine-grain the synchronization.
42 base::SpinningMutexGuard mutex_guard(&mutex_);
43 // Compute the first canonical index in the recgroup in the case that it does
44 // not already exist.
45 CanonicalTypeIndex first_new_canonical_index{
46 static_cast<uint32_t>(canonical_supertypes_.size())};
47
48 // Create a snapshot of the zone; this will be restored in case we find a
49 // matching recursion group.
50 ZoneSnapshot zone_snapshot = zone_.Snapshot();
51
52 DCHECK_GE(module->types.size(), start_index + size);
53 CanonicalGroup group{&zone_, size, first_new_canonical_index};
54 for (uint32_t i = 0; i < size; i++) {
55 group.types[i] = CanonicalizeTypeDef(
56 module, ModuleTypeIndex{start_index + i}, ModuleTypeIndex{start_index},
57 first_new_canonical_index);
58 }
59 if (CanonicalTypeIndex canonical_index = FindCanonicalGroup(group);
60 canonical_index.valid()) {
61 // Delete zone memory from {CanonicalizeTypeDef} and {CanonicalGroup}.
62 zone_snapshot.Restore(&zone_);
63
64 // Identical group found. Map new types to the old types's canonical
65 // representatives.
66 for (uint32_t i = 0; i < size; i++) {
67 CanonicalTypeIndex existing_type_index =
68 CanonicalTypeIndex{canonical_index.index + i};
69 module->isorecursive_canonical_type_ids[start_index + i] =
70 existing_type_index;
71 }
72 return;
73 }
74 canonical_supertypes_.resize(first_new_canonical_index.index + size);
75 CheckMaxCanonicalIndex();
76 for (uint32_t i = 0; i < size; i++) {
77 CanonicalType& canonical_type = group.types[i];
78 canonical_supertypes_[first_new_canonical_index.index + i] =
79 canonical_type.supertype;
80 CanonicalTypeIndex canonical_id{first_new_canonical_index.index + i};
81 module->isorecursive_canonical_type_ids[start_index + i] = canonical_id;
82 if (canonical_type.kind == CanonicalType::kFunction) {
83 const CanonicalSig* sig = canonical_type.function_sig;
84 CHECK(canonical_function_sigs_.emplace(canonical_id, sig).second);
85 }
86 }
87 // Check that this canonical ID is not used yet.
88 DCHECK(std::none_of(
89 canonical_singleton_groups_.begin(), canonical_singleton_groups_.end(),
90 [=](auto& entry) { return entry.index == first_new_canonical_index; }));
91 DCHECK(std::none_of(
92 canonical_groups_.begin(), canonical_groups_.end(),
93 [=](auto& entry) { return entry.first == first_new_canonical_index; }));
94 canonical_groups_.emplace(group);
95}

97void TypeCanonicalizer::AddRecursiveSingletonGroup(WasmModule* module) {
98 DCHECK(!module->types.empty());
99 uint32_t type_index = static_cast<uint32_t>(module->types.size() - 1);
100 base::SpinningMutexGuard guard(&mutex_);
101 CanonicalTypeIndex new_canonical_index{
102 static_cast<uint32_t>(canonical_supertypes_.size())};
103 // Snapshot the zone before allocating the new type; the zone will be reset if
104 // we find an identical type.
105 ZoneSnapshot zone_snapshot = zone_.Snapshot();
106 CanonicalType type =
107 CanonicalizeTypeDef(module, ModuleTypeIndex{type_index},
108 ModuleTypeIndex{type_index}, new_canonical_index);
109 CanonicalSingletonGroup group{type, new_canonical_index};
110 if (CanonicalTypeIndex index = FindCanonicalGroup(group); index.valid()) {
111 // Make sure this signature can be looked up later.
112 DCHECK_IMPLIES(type.kind == CanonicalType::kFunction,
113 canonical_function_sigs_.count(index));
114 zone_snapshot.Restore(&zone_);
115 module->isorecursive_canonical_type_ids[type_index] = index;
116 return;
117 }
118 // Check that the new canonical ID is not used yet.
119 DCHECK(std::none_of(
120 canonical_singleton_groups_.begin(), canonical_singleton_groups_.end(),
121 [=](auto& entry) { return entry.index == new_canonical_index; }));
122 DCHECK(std::none_of(
123 canonical_groups_.begin(), canonical_groups_.end(),
124 [=](auto& entry) { return entry.first == new_canonical_index; }));
125 canonical_singleton_groups_.emplace(group);
126 canonical_supertypes_.push_back(type.supertype);
127 if (type.kind == CanonicalType::kFunction) {
128 const CanonicalSig* sig = type.function_sig;
129 CHECK(canonical_function_sigs_.emplace(new_canonical_index, sig).second);
130 }
131 CheckMaxCanonicalIndex();
132 module->isorecursive_canonical_type_ids[type_index] = new_canonical_index;
133}

135CanonicalTypeIndex TypeCanonicalizer::AddRecursiveGroup(
136 const FunctionSig* sig) {
137// Types in the signature must be module-independent.
138#if DEBUG
139 for (ValueType type : sig->all()) DCHECK(!type.has_index());
140#endif
141 const bool kFinal = true;
142 const bool kNotShared = false;
143 // Because of the checks above, we can treat the type_def as canonical.
144 // TODO(366180605): It would be nice to not have to rely on a cast here.
145 // Is there a way to avoid it? In the meantime, these asserts provide at
146 // least partial assurances that the cast is safe:
147 static_assert(sizeof(CanonicalValueType) == sizeof(ValueType));
148 static_assert(CanonicalValueType::Primitive(kI32).raw_bit_field() ==
149 ValueType::Primitive(kI32).raw_bit_field());
150 CanonicalType canonical{reinterpret_cast<const CanonicalSig*>(sig),
151 CanonicalTypeIndex{kNoSuperType}, kFinal, kNotShared};
152 base::SpinningMutexGuard guard(&mutex_);
153 // Fast path lookup before canonicalizing (== copying into the
154 // TypeCanonicalizer's zone) the function signature.
155 CanonicalTypeIndex new_canonical_index{
156 static_cast<uint32_t>(canonical_supertypes_.size())};
157 CanonicalTypeIndex index = FindCanonicalGroup(
158 CanonicalSingletonGroup{canonical, new_canonical_index});
159 if (index.valid()) return index;
160
161 // Copy into this class's zone to store this as a new canonical function type.
162 CanonicalSig::Builder builder(&zone_, sig->return_count(),
163 sig->parameter_count());
164 for (ValueType ret : sig->returns()) {
165 builder.AddReturn(CanonicalValueType{ret});
166 }
167 for (ValueType param : sig->parameters()) {
168 builder.AddParam(CanonicalValueType{param});
169 }
170 canonical.function_sig = builder.Get();
171
172 CanonicalSingletonGroup group{canonical, new_canonical_index};
173 // Copying the signature shouldn't make a difference: There is no match.
174 DCHECK(!FindCanonicalGroup(group).valid());
175 // Check that the new canonical ID is not used yet.
176 DCHECK(std::none_of(
177 canonical_singleton_groups_.begin(), canonical_singleton_groups_.end(),
178 [=](auto& entry) { return entry.index == new_canonical_index; }));
179 DCHECK(std::none_of(
180 canonical_groups_.begin(), canonical_groups_.end(),
181 [=](auto& entry) { return entry.first == new_canonical_index; }));
182 canonical_singleton_groups_.emplace(group);
183 canonical_supertypes_.push_back(CanonicalTypeIndex{kNoSuperType});
184 CHECK(canonical_function_sigs_
185 .emplace(new_canonical_index, canonical.function_sig)
186 .second);
187 CheckMaxCanonicalIndex();
188 return new_canonical_index;
189}

199void TypeCanonicalizer::AddPredefinedArrayTypes() {
200 static constexpr std::pair<CanonicalTypeIndex, CanonicalValueType>
201 kPredefinedArrayTypes[] = {{kPredefinedArrayI8Index, {kWasmI8}},
202 {kPredefinedArrayI16Index, {kWasmI16}}};
203 for (auto [index, element_type] : kPredefinedArrayTypes) {
204 DCHECK_EQ(index.index, canonical_singleton_groups_.size());
205 static constexpr bool kMutable = true;
206 // TODO(jkummerow): Decide whether this should be final or nonfinal.
207 static constexpr bool kFinal = true;
208 static constexpr bool kShared = false; // TODO(14616): Fix this.
209 CanonicalArrayType* type =
210 zone_.New<CanonicalArrayType>(element_type, kMutable);
211 CanonicalSingletonGroup group{
212 .type = CanonicalType(type, CanonicalTypeIndex{kNoSuperType}, kFinal,
213 kShared),
214 .index = index};
215 canonical_singleton_groups_.emplace(group);
216 canonical_supertypes_.emplace_back(CanonicalTypeIndex{kNoSuperType});
217 DCHECK_LE(canonical_supertypes_.size(), kMaxCanonicalTypes);
218 }
219}

221bool TypeCanonicalizer::IsCanonicalSubtype(CanonicalTypeIndex sub_index,
222 CanonicalTypeIndex super_index) {
223 // Fast path without synchronization:
224 if (sub_index == super_index) return true;
225
226 // Multiple threads could try to register and access recursive groups
227 // concurrently.
228 // TODO(manoskouk): Investigate if we can improve this synchronization.
229 base::SpinningMutexGuard mutex_guard(&mutex_);
230 while (sub_index.valid()) {
231 if (sub_index == super_index) return true;
232 sub_index = canonical_supertypes_[sub_index.index];
233 }
234 return false;
235}

257TypeCanonicalizer::CanonicalType TypeCanonicalizer::CanonicalizeTypeDef(
258 const WasmModule* module, ModuleTypeIndex module_type_idx,
259 ModuleTypeIndex recgroup_start,
260 CanonicalTypeIndex canonical_recgroup_start) {
261 mutex_.AssertHeld(); // The caller must hold the mutex.
262
263 auto CanonicalizeTypeIndex = [=](ModuleTypeIndex type_index) {
264 DCHECK(type_index.valid());
265 return type_index < recgroup_start
266 // This references a type from an earlier recgroup; use the
267 // already-canonicalized type index.
268 ? module->canonical_type_id(type_index)
269 // For types within the same recgroup, generate indexes assuming
270 // that this is a new canonical recgroup.
271 : CanonicalTypeIndex{canonical_recgroup_start.index +
272 (type_index.index - recgroup_start.index)};
273 };
274
275 auto CanonicalizeValueType = [=](ValueType type) {
276 if (!type.has_index()) return CanonicalValueType{type};
277 static_assert(kMaxCanonicalTypes <= (1u << ValueType::kHeapTypeBits));
278 return CanonicalValueType::FromIndex(
279 type.kind(), CanonicalizeTypeIndex(type.ref_index()));
280 };
281
282 TypeDefinition type = module->type(module_type_idx);
283 CanonicalTypeIndex supertype = type.supertype.valid()
284 ? CanonicalizeTypeIndex(type.supertype)
285 : CanonicalTypeIndex::Invalid();
286 switch (type.kind) {
287 case TypeDefinition::kFunction: {
288 const FunctionSig* original_sig = type.function_sig;
289 CanonicalSig::Builder builder(&zone_, original_sig->return_count(),
290 original_sig->parameter_count());
291 for (ValueType ret : original_sig->returns()) {
292 builder.AddReturn(CanonicalizeValueType(ret));
293 }
294 for (ValueType param : original_sig->parameters()) {
295 builder.AddParam(CanonicalizeValueType(param));
296 }
297 return CanonicalType(builder.Get(), supertype, type.is_final,
298 type.is_shared);
299 }
300 case TypeDefinition::kStruct: {
301 const StructType* original_type = type.struct_type;
302 CanonicalStructType::Builder builder(&zone_,
303 original_type->field_count());
304 for (uint32_t i = 0; i < original_type->field_count(); i++) {
305 builder.AddField(CanonicalizeValueType(original_type->field(i)),
306 original_type->mutability(i),
307 original_type->field_offset(i));
308 }
309 builder.set_total_fields_size(original_type->total_fields_size());
310 return CanonicalType(
311 builder.Build(CanonicalStructType::Builder::kUseProvidedOffsets),
312 supertype, type.is_final, type.is_shared);
313 }
314 case TypeDefinition::kArray: {
315 CanonicalValueType element_type =
316 CanonicalizeValueType(type.array_type->element_type());
317 CanonicalArrayType* array_type = zone_.New<CanonicalArrayType>(
318 element_type, type.array_type->mutability());
319 return CanonicalType(array_type, supertype, type.is_final,
320 type.is_shared);
321 }
322 }
323}

325// Returns the index of the canonical representative of the first type in this
326// group if it exists, and `CanonicalTypeIndex::Invalid()` otherwise.
327CanonicalTypeIndex TypeCanonicalizer::FindCanonicalGroup(
328 const CanonicalGroup& group) const {
329 // Groups of size 0 do not make sense here; groups of size 1 should use
330 // {CanonicalSingletonGroup} (see below).
331 DCHECK_LT(1, group.types.size());
332 auto it = canonical_groups_.find(group);
333 return it == canonical_groups_.end() ? CanonicalTypeIndex::Invalid()
334 : it->first;
335}
336
337// Returns the canonical index of the given group if it already exists.
338CanonicalTypeIndex TypeCanonicalizer::FindCanonicalGroup(
339 const CanonicalSingletonGroup& group) const {
340 auto it = canonical_singleton_groups_.find(group);
341 static_assert(kMaxCanonicalTypes <= kMaxInt);
342 return it == canonical_singleton_groups_.end() ? CanonicalTypeIndex::Invalid()
343 : it->index;
344}

346size_t TypeCanonicalizer::EstimateCurrentMemoryConsumption() const {
347 UPDATE_WHEN_CLASS_CHANGES(TypeCanonicalizer, 256);
348 // The storage of the canonical group's types is accounted for via the
349 // allocator below (which tracks the zone memory).
350 base::SpinningMutexGuard mutex_guard(&mutex_);
351 size_t result = ContentSize(canonical_supertypes_);
352 result += ContentSize(canonical_groups_);
353 result += ContentSize(canonical_singleton_groups_);
354 result += ContentSize(canonical_function_sigs_);
355 result += allocator_.GetCurrentMemoryUsage();
356 if (v8_flags.trace_wasm_offheap_memory) {
357 PrintF("TypeCanonicalizer: %zu\n", result);
358 }
359 return result;
360}

367// static
368void TypeCanonicalizer::PrepareForCanonicalTypeId(Isolate* isolate,
369 CanonicalTypeIndex id) {
370 if (!id.valid()) return;
371 Heap* heap = isolate->heap();
372 // {2 * (id + 1)} needs to fit in an int.
373 CHECK_LE(id.index, kMaxInt / 2 - 1);
374 // Canonical types and wrappers are zero-indexed.
375 const int length = id.index + 1;
376 // The fast path is non-handlified.
377 Tagged<WeakFixedArray> old_rtts_raw = heap->wasm_canonical_rtts();
378 Tagged<WeakFixedArray> old_wrappers_raw = heap->js_to_wasm_wrappers();
379
380 // Fast path: Lengths are sufficient.
381 int old_length = old_rtts_raw->length();
382 DCHECK_EQ(old_length, old_wrappers_raw->length());
383 if (old_length >= length) return;
384
385 // Allocate bigger WeakFixedArrays for rtts and wrappers. Grow them
386 // exponentially.
387 const int new_length = std::max(old_length * 3 / 2, length);
388 CHECK_LT(old_length, new_length);
389
390 // Allocation can invalidate previous unhandled pointers.
391 DirectHandle<WeakFixedArray> old_rtts{old_rtts_raw, isolate};
392 DirectHandle<WeakFixedArray> old_wrappers{old_wrappers_raw, isolate};
393 old_rtts_raw = old_wrappers_raw = {};
394
395 // We allocate the WeakFixedArray filled with undefined values, as we cannot
396 // pass the cleared value in a Handle (see https://crbug.com/364591622). We
397 // overwrite the new entries via {MemsetTagged} afterwards.
398 DirectHandle<WeakFixedArray> new_rtts =
399 WeakFixedArray::New(isolate, new_length, AllocationType::kOld);
400 WeakFixedArray::CopyElements(isolate, *new_rtts, 0, *old_rtts, 0, old_length);
401 MemsetTagged(new_rtts->RawFieldOfFirstElement() + old_length,
402 ClearedValue(isolate), new_length - old_length);
403 DirectHandle<WeakFixedArray> new_wrappers =
404 WeakFixedArray::New(isolate, new_length, AllocationType::kOld);
405 WeakFixedArray::CopyElements(isolate, *new_wrappers, 0, *old_wrappers, 0,
406 old_length);
407 MemsetTagged(new_wrappers->RawFieldOfFirstElement() + old_length,
408 ClearedValue(isolate), new_length - old_length);
409 heap->SetWasmCanonicalRttsAndJSToWasmWrappers(*new_rtts, *new_wrappers);
410}
