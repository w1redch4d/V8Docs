157void Factory::NumberToStringCacheSet(DirectHandle<Object> number, int hash,
158 DirectHandle<String> js_string) {
159 if (!IsUndefined(number_string_cache()->get(hash * 2), isolate()) &&
160 !v8_flags.optimize_for_size) {
161 int full_size = isolate()->heap()->MaxNumberToStringCacheSize();
162 if (number_string_cache()->length() != full_size) {
163 DirectHandle<FixedArray> new_cache =
164 NewFixedArray(full_size, AllocationType::kOld);
165 isolate()->heap()->set_number_string_cache(*new_cache);
166 return;
167 }
168 }
169 DisallowGarbageCollection no_gc;
170 Tagged<FixedArray> cache = *number_string_cache();
171 cache->set(hash * 2, *number);
172 cache->set(hash * 2 + 1, *js_string);
173}
174
175Handle<Object> Factory::NumberToStringCacheGet(Tagged<Object> number,
176 int hash) {
177 DisallowGarbageCollection no_gc;
178 Tagged<FixedArray> cache = *number_string_cache();
179 Tagged<Object> key = cache->get(hash * 2);
180 if (key == number ||
181 (IsHeapNumber(key) && IsHeapNumber(number) &&
182 Cast<HeapNumber>(key)->value() == Cast<HeapNumber>(number)->value())) {
183 return Handle<String>(Cast<String>(cache->get(hash * 2 + 1)), isolate());
184 }
185 return undefined_value();
186}
