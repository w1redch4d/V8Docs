66TNode<Object> ArrayBuiltinsAssembler::TypedArrayMapProcessor(
67 TNode<Object> k_value, TNode<UintPtrT> k) {
68 // 7c. Let mapped_value be ? Call(callbackfn, T, Â« kValue, k, O Â»).
69 TNode<Number> k_number = ChangeUintPtrToTagged(k);
70 TNode<Object> mapped_value =
71 Call(context(), callbackfn(), this_arg(), k_value, k_number, o());
72 Label fast(this), slow(this), done(this), detached(this, Label::kDeferred);
73
74 // 7d. Perform ? Set(A, Pk, mapped_value, true).
75 // Since we know that A is a TypedArray, this always ends up in
76 // #sec-integer-indexed-exotic-objects-set-p-v-receiver and then
77 // tc39.github.io/ecma262/#sec-integerindexedelementset .
78 Branch(fast_typed_array_target_, &fast, &slow);
79
80 BIND(&fast);
81 // #sec-integerindexedelementset
82 // 2. If arrayTypeName is "BigUint64Array" or "BigInt64Array", let
83 // numValue be ? ToBigInt(v).
84 // 3. Otherwise, let numValue be ? ToNumber(value).
85 TNode<Object> num_value;
86 if (IsBigIntTypedArrayElementsKind(source_elements_kind_)) {
87 num_value = ToBigInt(context(), mapped_value);
88 } else {
89 num_value = ToNumber_Inline(context(), mapped_value);
90 }
91
92 // The only way how this can bailout is because of a detached or out of bounds
93 // buffer.
94 // TODO(v8:4153): Consider checking IsDetachedBuffer() and calling
95 // TypedArrayBuiltinsAssembler::StoreJSTypedArrayElementFromNumeric() here
96 // instead to avoid converting k_number back to UintPtrT.
97
98 // Using source_elements_kind_ (not "target elements kind") is correct here,
99 // because the fast branch is taken only when the source and the target
100 // elements kinds match.
101 EmitElementStore(CAST(a()), k_number, num_value, source_elements_kind_,
102 KeyedAccessStoreMode::kInBounds, &detached, context());
103 Goto(&done);
104
105 BIND(&slow);
106 {
107 SetPropertyStrict(context(), a(), k_number, mapped_value);
108 Goto(&done);
109 }
110
111 BIND(&detached);
112 // tc39.github.io/ecma262/#sec-integerindexedelementset
113 // 8. If IsDetachedBuffer(buffer) is true, throw a TypeError exception.
114 ThrowTypeError(context_, MessageTemplate::kDetachedOperation, name_);
115
116 BIND(&done);
117 return a();

139void ArrayBuiltinsAssembler::GenerateIteratingTypedArrayBuiltinBody(
140 const char* name, const BuiltinResultGenerator& generator,
141 const CallResultProcessor& processor, ForEachDirection direction) {
142 name_ = name;
143
144 // ValidateTypedArray: tc39.github.io/ecma262/#sec-validatetypedarray
145
146 Label throw_not_typed_array(this, Label::kDeferred);
147
148 GotoIf(TaggedIsSmi(receiver_), &throw_not_typed_array);
149 TNode<Map> typed_array_map = LoadMap(CAST(receiver_));
150 GotoIfNot(IsJSTypedArrayMap(typed_array_map), &throw_not_typed_array);
151
152 TNode<JSTypedArray> typed_array = CAST(receiver_);
153 o_ = typed_array;
154
155 Label throw_detached(this, Label::kDeferred);
156 len_ = LoadJSTypedArrayLengthAndCheckDetached(typed_array, &throw_detached);
157
158 Label throw_not_callable(this, Label::kDeferred);
159 Label distinguish_types(this);
160 GotoIf(TaggedIsSmi(callbackfn_), &throw_not_callable);
161 Branch(IsCallableMap(LoadMap(CAST(callbackfn_))), &distinguish_types,
162 &throw_not_callable);
163
164 BIND(&throw_not_typed_array);
165 ThrowTypeError(context_, MessageTemplate::kNotTypedArray);
166
167 BIND(&throw_not_callable);
168 ThrowTypeError(context_, MessageTemplate::kCalledNonCallable, callbackfn_);
169
170 BIND(&throw_detached);
171 ThrowTypeError(context_, MessageTemplate::kDetachedOperation, name_);
172
173 Label unexpected_instance_type(this);
174 BIND(&unexpected_instance_type);
175 Unreachable();
176
177 std::vector<int32_t> elements_kinds = {
178#define ELEMENTS_KIND(Type, type, TYPE, ctype) TYPE##_ELEMENTS,
179 TYPED_ARRAYS(ELEMENTS_KIND) RAB_GSAB_TYPED_ARRAYS(ELEMENTS_KIND)
180#undef ELEMENTS_KIND
181 };
182 std::list<Label> labels;
183 for (size_t i = 0; i < elements_kinds.size(); ++i) {
184 labels.emplace_back(this);
185 }
186 std::vector<Label*> label_ptrs;
187 for (Label& label : labels) {
188 label_ptrs.push_back(&label);
189 }
190
191 BIND(&distinguish_types);
192
193 generator(this);
194
195 TNode<JSArrayBuffer> array_buffer = LoadJSArrayBufferViewBuffer(typed_array);
196 TNode<Int32T> elements_kind = LoadMapElementsKind(typed_array_map);
197 Switch(elements_kind, &unexpected_instance_type, elements_kinds.data(),
198 label_ptrs.data(), labels.size());
199
200 size_t i = 0;
201 for (auto it = labels.begin(); it != labels.end(); ++i, ++it) {
202 BIND(&*it);
203 source_elements_kind_ = static_cast<ElementsKind>(elements_kinds[i]);
204 VisitAllTypedArrayElements(array_buffer, processor, direction, typed_array);
205 ReturnFromBuiltin(a_.value());
206 }
207}

209void ArrayBuiltinsAssembler::VisitAllTypedArrayElements(
210 TNode<JSArrayBuffer> array_buffer, const CallResultProcessor& processor,
211 ForEachDirection direction, TNode<JSTypedArray> typed_array) {
212 VariableList list({&a_, &k_}, zone());
213
214 TNode<UintPtrT> start = UintPtrConstant(0);
215 TNode<UintPtrT> end = len_;
216 IndexAdvanceMode advance_mode = IndexAdvanceMode::kPost;
217 int incr = 1;
218 if (direction == ForEachDirection::kReverse) {
219 std::swap(start, end);
220 advance_mode = IndexAdvanceMode::kPre;
221 incr = -1;
222 }
223 k_ = start;
224
225 // TODO(v8:11111): Only RAB-backed TAs need special handling here since the
226 // backing store can shrink mid-iteration. This implementation has an
227 // overzealous check for GSAB-backed length-tracking TAs. Then again, the
228 // non-RAB/GSAB code also has an overzealous detached check for SABs.
229 ElementsKind effective_elements_kind = source_elements_kind_;
230 bool is_rab_gsab = IsRabGsabTypedArrayElementsKind(effective_elements_kind);
231 if (is_rab_gsab) {
232 effective_elements_kind =
233 GetCorrespondingNonRabGsabElementsKind(effective_elements_kind);
234 }
235 BuildFastLoop<UintPtrT>(
236 list, start, end,
237 [&](TNode<UintPtrT> index) {
238 TVARIABLE(Object, value);
239 Label detached(this, Label::kDeferred);
240 Label process(this);
241 if (is_rab_gsab) {
242 // If `index` is out of bounds, Get returns undefined.
243 CheckJSTypedArrayIndex(typed_array, index, &detached);
244 } else {
245 GotoIf(IsDetachedBuffer(array_buffer), &detached);
246 }
247 {
248 TNode<RawPtrT> data_ptr = LoadJSTypedArrayDataPtr(typed_array);
249 value = LoadFixedTypedArrayElementAsTagged(data_ptr, index,
250 effective_elements_kind);
251 Goto(&process);
252 }
253
254 BIND(&detached);
255 {
256 value = UndefinedConstant();
257 Goto(&process);
258 }
259
260 BIND(&process);
261 {
262 k_ = index;
263 a_ = processor(this, value.value(), index);
264 }
265 },
266 incr, LoopUnrollingMode::kNo, advance_mode);
267}

269TF_BUILTIN(ArrayPrototypePop, CodeStubAssembler) {
270 auto argc = UncheckedParameter<Int32T>(Descriptor::kJSActualArgumentsCount);
271 auto context = Parameter<Context>(Descriptor::kContext);
272 CSA_DCHECK(this, IsUndefined(Parameter<Object>(Descriptor::kJSNewTarget)));
273
274 CodeStubArguments args(this, argc);
275 TNode<Object> receiver = args.GetReceiver();
276
277 Label runtime(this, Label::kDeferred);
278 Label fast(this);
279
280 // Only pop in this stub if
281 // 1) the array has fast elements
282 // 2) the length is writable,
283 // 3) the elements backing store isn't copy-on-write,
284 // 4) we aren't supposed to shrink the backing store.
285
286 // 1) Check that the array has fast elements.
287 BranchIfFastJSArray(receiver, context, &fast, &runtime);
288
289 BIND(&fast);
290 {
291 TNode<JSArray> array_receiver = CAST(receiver);
292 CSA_DCHECK(this, TaggedIsPositiveSmi(LoadJSArrayLength(array_receiver)));
293 TNode<Int32T> length =
294 LoadAndUntagToWord32ObjectField(array_receiver, JSArray::kLengthOffset);
295 Label return_undefined(this), fast_elements(this);
296
297 // 2) Ensure that the length is writable.
298 EnsureArrayLengthWritable(context, LoadMap(array_receiver), &runtime);
299
300 GotoIf(Word32Equal(length, Int32Constant(0)), &return_undefined);
301
302 // 3) Check that the elements backing store isn't copy-on-write.
303 TNode<FixedArrayBase> elements = LoadElements(array_receiver);
304 GotoIf(TaggedEqual(LoadMap(elements), FixedCOWArrayMapConstant()),
305 &runtime);
306
307 TNode<Int32T> new_length = Int32Sub(length, Int32Constant(1));
308
309 // 4) Check that we're not supposed to shrink the backing store, as
310 // implemented in elements.cc:ElementsAccessorBase::SetLengthImpl.
311 TNode<Int32T> capacity = SmiToInt32(LoadFixedArrayBaseLength(elements));
312 GotoIf(Int32LessThan(
313 Int32Add(Int32Add(new_length, new_length),
314 Int32Constant(JSObject::kMinAddedElementsCapacity)),
315 capacity),
316 &runtime);
317
318 TNode<IntPtrT> new_length_intptr = ChangePositiveInt32ToIntPtr(new_length);
319 StoreObjectFieldNoWriteBarrier(array_receiver, JSArray::kLengthOffset,
320 SmiTag(new_length_intptr));
321
322 TNode<Int32T> elements_kind = LoadElementsKind(array_receiver);
323 GotoIf(Int32LessThanOrEqual(elements_kind,
324 Int32Constant(TERMINAL_FAST_ELEMENTS_KIND)),
325 &fast_elements);
326
327 {
328 TNode<FixedDoubleArray> elements_known_double_array =
329 ReinterpretCast<FixedDoubleArray>(elements);
330 TNode<Float64T> value = LoadFixedDoubleArrayElement(
331 elements_known_double_array, new_length_intptr, &return_undefined);
332
333 StoreFixedDoubleArrayHole(elements_known_double_array, new_length_intptr);
334 args.PopAndReturn(AllocateHeapNumberWithValue(value));
335 }
336
337 BIND(&fast_elements);
338 {
339 TNode<FixedArray> elements_known_fixed_array = CAST(elements);
340 TNode<Object> value =
341 LoadFixedArrayElement(elements_known_fixed_array, new_length_intptr);
342 StoreFixedArrayElement(elements_known_fixed_array, new_length_intptr,
343 TheHoleConstant());
344 GotoIf(TaggedEqual(value, TheHoleConstant()), &return_undefined);
345 args.PopAndReturn(value);
346 }
347
348 BIND(&return_undefined);
349 { args.PopAndReturn(UndefinedConstant()); }
350 }
351
352 BIND(&runtime);
353 {
354 // We are not using Parameter(Descriptor::kJSTarget) and loading the value
355 // from the current frame here in order to reduce register pressure on the
356 // fast path.
357 TNode<JSFunction> target = LoadTargetFromFrame();
358 TailCallJSBuiltin(Builtin::kArrayPop, context, target, UndefinedConstant(),
359 argc, InvalidDispatchHandleConstant());
360 }

658void ArrayIncludesIndexofAssembler::Generate(SearchVariant variant,
659 TNode<IntPtrT> argc,
660 TNode<Context> context) {
661 const int kSearchElementArg = 0;
662 const int kFromIndexArg = 1;
663
664 CodeStubArguments args(this, argc);
665
666 TNode<Object> receiver = args.GetReceiver();
667 TNode<Object> search_element =
668 args.GetOptionalArgumentValue(kSearchElementArg);
669
670 TNode<IntPtrT> intptr_zero = IntPtrConstant(0);
671
672 Label init_index(this), return_not_found(this), call_runtime(this);
673
674 // Take slow path if not a JSArray, if retrieving elements requires
675 // traversing prototype, or if access checks are required.
676 BranchIfFastJSArrayForRead(receiver, context, &init_index, &call_runtime);
677
678 BIND(&init_index);
679 TVARIABLE(IntPtrT, index_var, intptr_zero);
680 TNode<JSArray> array = CAST(receiver);
681
682 // JSArray length is always a positive Smi for fast arrays.
683 CSA_DCHECK(this, TaggedIsPositiveSmi(LoadJSArrayLength(array)));
684 TNode<Smi> array_length = LoadFastJSArrayLength(array);
685 TNode<IntPtrT> array_length_untagged = PositiveSmiUntag(array_length);
686
687 {
688 // Initialize fromIndex.
689 Label is_smi(this), is_nonsmi(this), done(this);
690
691 // If no fromIndex was passed, default to 0.
692 GotoIf(IntPtrLessThanOrEqual(args.GetLengthWithoutReceiver(),
693 IntPtrConstant(kFromIndexArg)),
694 &done);
695
696 TNode<Object> start_from = args.AtIndex(kFromIndexArg);
697 // Handle Smis and undefined here and everything else in runtime.
698 // We must be very careful with side effects from the ToInteger conversion,
699 // as the side effects might render previously checked assumptions about
700 // the receiver being a fast JSArray and its length invalid.
701 Branch(TaggedIsSmi(start_from), &is_smi, &is_nonsmi);
702
703 BIND(&is_nonsmi);
704 {
705 GotoIfNot(IsUndefined(start_from), &call_runtime);
706 Goto(&done);
707 }
708 BIND(&is_smi);
709 {
710 TNode<IntPtrT> intptr_start_from = SmiUntag(CAST(start_from));
711 index_var = intptr_start_from;
712
713 GotoIf(IntPtrGreaterThanOrEqual(index_var.value(), intptr_zero), &done);
714 // The fromIndex is negative: add it to the array's length.
715 index_var = IntPtrAdd(array_length_untagged, index_var.value());
716 // Clamp negative results at zero.
717 GotoIf(IntPtrGreaterThanOrEqual(index_var.value(), intptr_zero), &done);
718 index_var = intptr_zero;
719 Goto(&done);
720 }
721 BIND(&done);
722 }
723
724 // Fail early if startIndex >= array.length.
725 GotoIf(IntPtrGreaterThanOrEqual(index_var.value(), array_length_untagged),
726 &return_not_found);
727
728 Label if_smi(this), if_smiorobjects(this), if_packed_doubles(this),
729 if_holey_doubles(this);
730
731 TNode<Int32T> elements_kind = LoadElementsKind(array);
732 TNode<FixedArrayBase> elements = LoadElements(array);
733 static_assert(PACKED_SMI_ELEMENTS == 0);
734 static_assert(HOLEY_SMI_ELEMENTS == 1);
735 static_assert(PACKED_ELEMENTS == 2);
736 static_assert(HOLEY_ELEMENTS == 3);
737 GotoIf(IsElementsKindLessThanOrEqual(elements_kind, HOLEY_SMI_ELEMENTS),
738 &if_smi);
739 GotoIf(IsElementsKindLessThanOrEqual(elements_kind, HOLEY_ELEMENTS),
740 &if_smiorobjects);
741 GotoIf(
742 ElementsKindEqual(elements_kind, Int32Constant(PACKED_DOUBLE_ELEMENTS)),
743 &if_packed_doubles);
744 GotoIf(ElementsKindEqual(elements_kind, Int32Constant(HOLEY_DOUBLE_ELEMENTS)),
745 &if_holey_doubles);
746 GotoIf(IsElementsKindLessThanOrEqual(elements_kind,
747 LAST_ANY_NONEXTENSIBLE_ELEMENTS_KIND),
748 &if_smiorobjects);
749 Goto(&return_not_found);
750
751 BIND(&if_smi);
752 {
753 Builtin builtin = (variant == kIncludes) ? Builtin::kArrayIncludesSmi
754 : Builtin::kArrayIndexOfSmi;
755 TNode<Object> result =
756 CallBuiltin(builtin, context, elements, search_element, array_length,
757 SmiTag(index_var.value()));
758 args.PopAndReturn(result);
759 }
760
761 BIND(&if_smiorobjects);
762 {
763 Builtin builtin = (variant == kIncludes)
764 ? Builtin::kArrayIncludesSmiOrObject
765 : Builtin::kArrayIndexOfSmiOrObject;
766 TNode<Object> result =
767 CallBuiltin(builtin, context, elements, search_element, array_length,
768 SmiTag(index_var.value()));
769 args.PopAndReturn(result);
770 }
771
772 BIND(&if_packed_doubles);
773 {
774 Builtin builtin = (variant == kIncludes)
775 ? Builtin::kArrayIncludesPackedDoubles
776 : Builtin::kArrayIndexOfPackedDoubles;
777 TNode<Object> result =
778 CallBuiltin(builtin, context, elements, search_element, array_length,
779 SmiTag(index_var.value()));
780 args.PopAndReturn(result);
781 }
782
783 BIND(&if_holey_doubles);
784 {
785 Builtin builtin = (variant == kIncludes)
786 ? Builtin::kArrayIncludesHoleyDoubles
787 : Builtin::kArrayIndexOfHoleyDoubles;
788 TNode<Object> result =
789 CallBuiltin(builtin, context, elements, search_element, array_length,
790 SmiTag(index_var.value()));
791 args.PopAndReturn(result);
792 }
793
794 BIND(&return_not_found);
795 if (variant == kIncludes) {
796 args.PopAndReturn(FalseConstant());
797 } else {
798 args.PopAndReturn(NumberConstant(-1));
799 }
800
801 BIND(&call_runtime);
802 {
803 TNode<Object> start_from = args.GetOptionalArgumentValue(kFromIndexArg);
804 Runtime::FunctionId function = variant == kIncludes
805 ? Runtime::kArrayIncludes_Slow
806 : Runtime::kArrayIndexOf;
807 args.PopAndReturn(
808 CallRuntime(function, context, array, search_element, start_from));
809 }

812void ArrayIncludesIndexofAssembler::GenerateSmiOrObject(
813 SearchVariant variant, TNode<Context> context, TNode<FixedArray> elements,
814 TNode<Object> search_element, TNode<Smi> array_length,
815 TNode<Smi> from_index, SimpleElementKind array_kind) {
816 TVARIABLE(IntPtrT, index_var, SmiUntag(from_index));
817 TVARIABLE(Float64T, search_num);
818 TNode<IntPtrT> array_length_untagged = PositiveSmiUntag(array_length);
819
820 Label ident_loop(this, &index_var), heap_num_loop(this, &search_num),
821 string_loop(this), bigint_loop(this, &index_var),
822 undef_loop(this, &index_var), not_smi(this), not_heap_num(this),
823 return_found(this), return_not_found(this);
824
825 GotoIfNot(TaggedIsSmi(search_element), &not_smi);
826 search_num = SmiToFloat64(CAST(search_element));
827 Goto(&heap_num_loop);
828
829 BIND(&not_smi);
830 if (variant == kIncludes) {
831 GotoIf(IsUndefined(search_element), &undef_loop);
832 }
833 TNode<Map> map = LoadMap(CAST(search_element));
834 GotoIfNot(IsHeapNumberMap(map), &not_heap_num);
835 search_num = LoadHeapNumberValue(CAST(search_element));
836 Goto(&heap_num_loop);
837
838 BIND(&not_heap_num);
839 TNode<Uint16T> search_type = LoadMapInstanceType(map);
840 GotoIf(IsStringInstanceType(search_type), &string_loop);
841 GotoIf(IsBigIntInstanceType(search_type), &bigint_loop);
842
843 // Use UniqueInt32Constant instead of BoolConstant here in order to ensure
844 // that the graph structure does not depend on the value of the predicate
845 // (BoolConstant uses cached nodes).
846 GotoIfNot(UniqueInt32Constant(kCanVectorize), &ident_loop);
847 {
848 Label simd_call(this);
849 Branch(
850 UintPtrLessThan(array_length_untagged, IntPtrConstant(kSIMDThreshold)),
851 &ident_loop, &simd_call);
852 BIND(&simd_call);
853 TNode<ExternalReference> simd_function = ExternalConstant(
854 ExternalReference::array_indexof_includes_smi_or_object());
855 TNode<IntPtrT> result = UncheckedCast<IntPtrT>(CallCFunction(
856 simd_function, MachineType::UintPtr(),
857 std::make_pair(MachineType::TaggedPointer(), elements),
858 std::make_pair(MachineType::UintPtr(), array_length_untagged),
859 std::make_pair(MachineType::UintPtr(), index_var.value()),
860 std::make_pair(MachineType::TaggedPointer(), search_element)));
861 index_var = ReinterpretCast<IntPtrT>(result);
862 Branch(IntPtrLessThan(index_var.value(), IntPtrConstant(0)),
863 &return_not_found, &return_found);
864 }
865
866 BIND(&ident_loop);
867 {
868 GotoIfNot(UintPtrLessThan(index_var.value(), array_length_untagged),
869 &return_not_found);
870 TNode<Object> element_k =
871 UnsafeLoadFixedArrayElement(elements, index_var.value());
872 GotoIf(TaggedEqual(element_k, search_element), &return_found);
873
874 Increment(&index_var);
875 Goto(&ident_loop);
876 }
877
878 if (variant == kIncludes) {
879 BIND(&undef_loop);
880
881 GotoIfNot(UintPtrLessThan(index_var.value(), array_length_untagged),
882 &return_not_found);
883 TNode<Object> element_k =
884 UnsafeLoadFixedArrayElement(elements, index_var.value());
885 GotoIf(IsUndefined(element_k), &return_found);
886 GotoIf(IsTheHole(element_k), &return_found);
887
888 Increment(&index_var);
889 Goto(&undef_loop);
890 }
891
892 BIND(&heap_num_loop);
893 {
894 Label nan_loop(this, &index_var), not_nan_loop(this, &index_var);
895 Label* nan_handling = variant == kIncludes ? &nan_loop : &return_not_found;
896 GotoIfNot(Float64Equal(search_num.value(), search_num.value()),
897 nan_handling);
898
899 // Use UniqueInt32Constant instead of BoolConstant here in order to ensure
900 // that the graph structure does not depend on the value of the predicate
901 // (BoolConstant uses cached nodes).
902 GotoIfNot(UniqueInt32Constant(kCanVectorize &&
903 array_kind == SimpleElementKind::kSmiOrHole),
904 &not_nan_loop);
905 {
906 Label smi_check(this), simd_call(this);
907 Branch(UintPtrLessThan(array_length_untagged,
908 IntPtrConstant(kSIMDThreshold)),
909 &not_nan_loop, &smi_check);
910 BIND(&smi_check);
911 Branch(TaggedIsSmi(search_element), &simd_call, &not_nan_loop);
912 BIND(&simd_call);
913 TNode<ExternalReference> simd_function = ExternalConstant(
914 ExternalReference::array_indexof_includes_smi_or_object());
915 TNode<IntPtrT> result = UncheckedCast<IntPtrT>(CallCFunction(
916 simd_function, MachineType::UintPtr(),
917 std::make_pair(MachineType::TaggedPointer(), elements),
918 std::make_pair(MachineType::UintPtr(), array_length_untagged),
919 std::make_pair(MachineType::UintPtr(), index_var.value()),
920 std::make_pair(MachineType::TaggedPointer(), search_element)));
921 index_var = ReinterpretCast<IntPtrT>(result);
922 Branch(IntPtrLessThan(index_var.value(), IntPtrConstant(0)),
923 &return_not_found, &return_found);
924 }
925
926 BIND(&not_nan_loop);
927 {
928 Label continue_loop(this), element_k_not_smi(this);
929 GotoIfNot(UintPtrLessThan(index_var.value(), array_length_untagged),
930 &return_not_found);
931 TNode<Object> element_k =
932 UnsafeLoadFixedArrayElement(elements, index_var.value());
933 GotoIfNot(TaggedIsSmi(element_k), &element_k_not_smi);
934 Branch(Float64Equal(search_num.value(), SmiToFloat64(CAST(element_k))),
935 &return_found, &continue_loop);
936
937 BIND(&element_k_not_smi);
938 GotoIfNot(IsHeapNumber(CAST(element_k)), &continue_loop);
939 Branch(Float64Equal(search_num.value(),
940 LoadHeapNumberValue(CAST(element_k))),
941 &return_found, &continue_loop);
942
943 BIND(&continue_loop);
944 Increment(&index_var);
945 Goto(&not_nan_loop);
946 }
947
948 // Array.p.includes uses SameValueZero comparisons, where NaN == NaN.
949 if (variant == kIncludes) {
950 BIND(&nan_loop);
951 Label continue_loop(this);
952 GotoIfNot(UintPtrLessThan(index_var.value(), array_length_untagged),
953 &return_not_found);
954 TNode<Object> element_k =
955 UnsafeLoadFixedArrayElement(elements, index_var.value());
956 GotoIf(TaggedIsSmi(element_k), &continue_loop);
957 GotoIfNot(IsHeapNumber(CAST(element_k)), &continue_loop);
958 BranchIfFloat64IsNaN(LoadHeapNumberValue(CAST(element_k)), &return_found,
959 &continue_loop);
960
961 BIND(&continue_loop);
962 Increment(&index_var);
963 Goto(&nan_loop);
964 }
965 }
966
967 BIND(&string_loop);
968 {
969 TNode<String> search_element_string = CAST(search_element);
970 Label continue_loop(this), next_iteration(this, &index_var),
971 slow_compare(this), runtime(this, Label::kDeferred);
972 TNode<IntPtrT> search_length =
973 LoadStringLengthAsWord(search_element_string);
974 Goto(&next_iteration);
975 BIND(&next_iteration);
976 GotoIfNot(UintPtrLessThan(index_var.value(), array_length_untagged),
977 &return_not_found);
978 TNode<Object> element_k =
979 UnsafeLoadFixedArrayElement(elements, index_var.value());
980 GotoIf(TaggedIsSmi(element_k), &continue_loop);
981 GotoIf(TaggedEqual(search_element_string, element_k), &return_found);
982 TNode<Uint16T> element_k_type = LoadInstanceType(CAST(element_k));
983 GotoIfNot(IsStringInstanceType(element_k_type), &continue_loop);
984 Branch(IntPtrEqual(search_length, LoadStringLengthAsWord(CAST(element_k))),
985 &slow_compare, &continue_loop);
986
987 BIND(&slow_compare);
988 StringBuiltinsAssembler string_asm(state());
989 string_asm.StringEqual_Core(search_element_string, search_type,
990 CAST(element_k), element_k_type, search_length,
991 &return_found, &continue_loop, &runtime);
992 BIND(&runtime);
993 TNode<Object> result = CallRuntime(Runtime::kStringEqual, context,
994 search_element_string, element_k);
995 Branch(TaggedEqual(result, TrueConstant()), &return_found, &continue_loop);
996
997 BIND(&continue_loop);
998 Increment(&index_var);
999 Goto(&next_iteration);
1000 }
1001
1002 BIND(&bigint_loop);
1003 {
1004 GotoIfNot(UintPtrLessThan(index_var.value(), array_length_untagged),
1005 &return_not_found);
1006
1007 TNode<Object> element_k =
1008 UnsafeLoadFixedArrayElement(elements, index_var.value());
1009 Label continue_loop(this);
1010 GotoIf(TaggedIsSmi(element_k), &continue_loop);
1011 GotoIfNot(IsBigInt(CAST(element_k)), &continue_loop);
1012 TNode<Object> result = CallRuntime(Runtime::kBigIntEqualToBigInt, context,
1013 search_element, element_k);
1014 Branch(TaggedEqual(result, TrueConstant()), &return_found, &continue_loop);
1015
1016 BIND(&continue_loop);
1017 Increment(&index_var);
1018 Goto(&bigint_loop);
1019 }
1020 BIND(&return_found);
1021 if (variant == kIncludes) {
1022 Return(TrueConstant());
1023 } else {
1024 Return(SmiTag(index_var.value()));
1025 }
1026
1027 BIND(&return_not_found);
1028 if (variant == kIncludes) {
1029 Return(FalseConstant());
1030 } else {
1031 Return(NumberConstant(-1));
1032 }
1033}

1200 Label continue_loop(this);
1201 GotoIfNot(UintPtrLessThan(index_var.value(), array_length_untagged),
1202 &return_not_found);
1203
1204 // Load double value or continue if it's the hole NaN.
1205 TNode<Float64T> element_k = LoadFixedDoubleArrayElement(
1206 elements, index_var.value(), &continue_loop);
1207
1208 BranchIfFloat64IsNaN(element_k, &return_found, &continue_loop);
1209 BIND(&continue_loop);
1210 Increment(&index_var);
1211 Goto(&nan_loop);
1212 }
1213
1214 // Array.p.includes treats the hole as undefined.
1215 if (variant == kIncludes) {
1216 BIND(&hole_loop);
1217 GotoIfNot(UintPtrLessThan(index_var.value(), array_length_untagged),
1218 &return_not_found);
1219
1220 // Check if the element is a double hole, but don't load it.
1221 LoadFixedDoubleArrayElement(elements, index_var.value(), &return_found,
1222 MachineType::None());
1223
1224 Increment(&index_var);
1225 Goto(&hole_loop);
1226 }
1227
1228 BIND(&return_found);
1229 if (variant == kIncludes) {
1230 Return(TrueConstant());
1231 } else {
1232 Return(SmiTag(index_var.value()));
1233 }
1234
1235 BIND(&return_not_found);
1236 if (variant == kIncludes) {
1237 Return(FalseConstant());
1238 } else {
1239 Return(NumberConstant(-1));
1240 }
1241}

1643 TNode<Context> context, TNode<JSFunction> target, TNode<Int32T> argc,

1656 DCHECK(allocation_site);

1687 // iterating over all expected elements kinds.

1708 Abort(AbortReason::kUnexpectedElementsKindInArrayConstructor);
1709 }
