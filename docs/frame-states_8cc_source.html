100// Lazy deopt points where the frame state is associated with a call get an
101// additional parameter for the return result from the call. The return result
102// is added by the deoptimizer and not explicitly specified in the frame state.
103// Lazy deopt points which can catch exceptions further get an additional
104// parameter, namely the exception thrown. The exception is also added by the
105// deoptimizer.
106uint8_t DeoptimizerParameterCountFor(ContinuationFrameStateMode mode) {
107 switch (mode) {
108 case ContinuationFrameStateMode::EAGER:
109 return 0;
110 case ContinuationFrameStateMode::LAZY:
111 return 1;
112 case ContinuationFrameStateMode::LAZY_WITH_CATCH:
113 return 2;
114 }
115 UNREACHABLE();
116}

118FrameState CreateBuiltinContinuationFrameStateCommon(
119 JSGraph* jsgraph, FrameStateType frame_type, Builtin name, Node* closure,
120 Node* context, Node** parameters, int parameter_count,
121 Node* outer_frame_state,
122 Handle<SharedFunctionInfo> shared = Handle<SharedFunctionInfo>(),
123 const wasm::CanonicalSig* signature = nullptr) {
124 Graph* const graph = jsgraph->graph();
125 CommonOperatorBuilder* const common = jsgraph->common();
126
127 const Operator* op_param =
128 common->StateValues(parameter_count, SparseInputMask::Dense());
129 Node* params_node = graph->NewNode(op_param, parameter_count, parameters);
130
131 BytecodeOffset bailout_id = Builtins::GetContinuationBytecodeOffset(name);
132#if V8_ENABLE_WEBASSEMBLY
133 const FrameStateFunctionInfo* state_info =
134 signature ? common->CreateJSToWasmFrameStateFunctionInfo(
135 frame_type, parameter_count, 0, shared, signature)
136 : common->CreateFrameStateFunctionInfo(
137 frame_type, parameter_count, 0, 0, shared, {});
138#else
139 DCHECK_NULL(signature);
140 const FrameStateFunctionInfo* state_info =
141 common->CreateFrameStateFunctionInfo(frame_type, parameter_count, 0, 0,
142 shared, {});
143#endif // V8_ENABLE_WEBASSEMBLY
144
145 const Operator* op = common->FrameState(
146 bailout_id, OutputFrameStateCombine::Ignore(), state_info);
147 return FrameState(graph->NewNode(op, params_node, jsgraph->EmptyStateValues(),
148 jsgraph->EmptyStateValues(), context,
149 closure, outer_frame_state));
150}

154FrameState CreateStubBuiltinContinuationFrameState(
155 JSGraph* jsgraph, Builtin name, Node* context, Node* const* parameters,
156 int parameter_count, Node* outer_frame_state,
157 ContinuationFrameStateMode mode, const wasm::CanonicalSig* signature) {
158 Callable callable = Builtins::CallableFor(jsgraph->isolate(), name);
159 CallInterfaceDescriptor descriptor = callable.descriptor();
160
161 std::vector<Node*> actual_parameters;
162 // Stack parameters first. Depending on {mode}, final parameters are added
163 // by the deoptimizer and aren't explicitly passed in the frame state.
164 int stack_parameter_count =
165 descriptor.GetStackParameterCount() - DeoptimizerParameterCountFor(mode);
166
167 // Ensure the parameters added by the deoptimizer are passed on the stack.
168 // This check prevents using TFS builtins as continuations while doing the
169 // lazy deopt. Use TFC or TFJ builtin as a lazy deopt continuation which
170 // would pass the result parameter on the stack.
171 DCHECK_GE(stack_parameter_count, 0);
172
173 // Reserving space in the vector.
174 actual_parameters.reserve(stack_parameter_count +
175 descriptor.GetRegisterParameterCount());
176 for (int i = 0; i < stack_parameter_count; ++i) {
177 actual_parameters.push_back(
178 parameters[descriptor.GetRegisterParameterCount() + i]);
179 }
180 // Register parameters follow, context will be added by instruction selector
181 // during FrameState translation.
182 for (int i = 0; i < descriptor.GetRegisterParameterCount(); ++i) {
183 actual_parameters.push_back(parameters[i]);
184 }
185
186 FrameStateType frame_state_type = FrameStateType::kBuiltinContinuation;
187#if V8_ENABLE_WEBASSEMBLY
188 if (name == Builtin::kJSToWasmLazyDeoptContinuation) {
189 CHECK_NOT_NULL(signature);
190 frame_state_type = FrameStateType::kJSToWasmBuiltinContinuation;
191 }
192#endif // V8_ENABLE_WEBASSEMBLY
193 return CreateBuiltinContinuationFrameStateCommon(
194 jsgraph, frame_state_type, name, jsgraph->UndefinedConstant(), context,
195 actual_parameters.data(), static_cast<int>(actual_parameters.size()),
196 outer_frame_state, Handle<SharedFunctionInfo>(), signature);
197}

215FrameState CreateJavaScriptBuiltinContinuationFrameState(
216 JSGraph* jsgraph, SharedFunctionInfoRef shared, Builtin name, Node* target,
217 Node* context, Node* const* stack_parameters, int stack_parameter_count,
218 Node* outer_frame_state, ContinuationFrameStateMode mode) {
219 // Depending on {mode}, final parameters are added by the deoptimizer
220 // and aren't explicitly passed in the frame state.
221 DCHECK_EQ(Builtins::GetStackParameterCount(name),
222 stack_parameter_count + DeoptimizerParameterCountFor(mode));
223
224 Node* argc = jsgraph->ConstantNoHole(Builtins::GetStackParameterCount(name));
225
226 // Stack parameters first. They must be first because the receiver is expected
227 // to be the second value in the translation when creating stack crawls
228 // (e.g. Error.stack) of optimized JavaScript frames.
229 std::vector<Node*> actual_parameters;
230 actual_parameters.reserve(stack_parameter_count);
231 for (int i = 0; i < stack_parameter_count; ++i) {
232 actual_parameters.push_back(stack_parameters[i]);
233 }
234
235 Node* new_target = jsgraph->UndefinedConstant();
236
237 // Register parameters follow stack parameters. The context will be added by
238 // instruction selector during FrameState translation.
239 DCHECK_EQ(
240 Builtins::CallInterfaceDescriptorFor(name).GetRegisterParameterCount(),
241 V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE_BOOL ? 4 : 3);
242 actual_parameters.push_back(target); // kJavaScriptCallTargetRegister
243 actual_parameters.push_back(new_target); // kJavaScriptCallNewTargetRegister
244 actual_parameters.push_back(argc); // kJavaScriptCallArgCountRegister
245#ifdef V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE
246 // The dispatch handle isn't used by the continuation builtins.
247 Node* handle = jsgraph->ConstantNoHole(kInvalidDispatchHandle.value());
248 actual_parameters.push_back(handle); // kJavaScriptDispatchHandleRegister
249#endif
250
251 return CreateBuiltinContinuationFrameStateCommon(
252 jsgraph,
253 mode == ContinuationFrameStateMode::LAZY_WITH_CATCH
254 ? FrameStateType::kJavaScriptBuiltinContinuationWithCatch
255 : FrameStateType::kJavaScriptBuiltinContinuation,
256 name, target, context, &actual_parameters[0],
257 static_cast<int>(actual_parameters.size()), outer_frame_state,
258 shared.object());
259}

280FrameState CloneFrameState(JSGraph* jsgraph, FrameState frame_state,
281 OutputFrameStateCombine changed_state_combine) {
282 Graph* graph = jsgraph->graph();
283 CommonOperatorBuilder* common = jsgraph->common();
284
285 DCHECK_EQ(IrOpcode::kFrameState, frame_state->op()->opcode());
286
287 const Operator* op = common->FrameState(
288 frame_state.frame_state_info().bailout_id(), changed_state_combine,
289 frame_state.frame_state_info().function_info());
290 return FrameState(
291 graph->NewNode(op, frame_state.parameters(), frame_state.locals(),
292 frame_state.stack(), frame_state.context(),
293 frame_state.function(), frame_state.outer_frame_state()));
294}
