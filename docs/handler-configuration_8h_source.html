49 enum class Kind {
50 kElement,
51 kIndexedString,
52 kNormal,
53 kGlobal,
54 kField,

68 // Defines whether access rights check should be done on lookup start object.
69 // Applicable to named property kinds only when loading value from prototype
70 // chain. Ignored when loading from lookup start object.
71 using DoAccessCheckOnLookupStartObjectBits = KindBits::Next<bool, 1>;
72
73 // Defines whether a lookup should be done on lookup start object before
74 // proceeding to the prototype chain. Applicable to named property kinds only
75 // when loading value from prototype chain. Ignored when loading from lookup
76 // start object.
77 using LookupOnLookupStartObjectBits =
78 DoAccessCheckOnLookupStartObjectBits::Next<bool, 1>;
79
80 //
81 // Encoding when KindBits contains kNativeDataProperty.
82 //
83
84 // Index of a value entry in the descriptor array.
85 using DescriptorBits =
86 LookupOnLookupStartObjectBits::Next<unsigned, kDescriptorIndexBitCount>;
87 // Make sure we don't overflow the smi.
88 static_assert(DescriptorBits::kLastUsedBit < kSmiValueSize);
89
90 //
91 // Encoding when KindBits contains kField.
92 //
93 using IsWasmStructBits = LookupOnLookupStartObjectBits::Next<bool, 1>;
94
95 //
96 // Encoding when KindBits contains kField and IsWasmStructBits is 0.
97 //
98 using IsInobjectBits = IsWasmStructBits::Next<bool, 1>;
99 using IsDoubleBits = IsInobjectBits::Next<bool, 1>;
100 // +1 here is to cover all possible JSObject header sizes.
101 using FieldIndexBits =
102 IsDoubleBits::Next<unsigned, kDescriptorIndexBitCount + 1>;
103 // Make sure we don't overflow the smi.
104 static_assert(FieldIndexBits::kLastUsedBit < kSmiValueSize);
105
106 //
107 // Encoding when KindBits contains kField and IsWasmStructBits is 1.
108 //
109 using WasmFieldTypeBits = IsWasmStructBits::Next<WasmValueType, 4>;
110 using WasmFieldOffsetBits = WasmFieldTypeBits::Next<unsigned, 20>;
111 // Make sure we don't overflow the smi.
112 static_assert(WasmFieldOffsetBits::kLastUsedBit < kSmiValueSize);
113
114 //
115 // Encoding when KindBits contains kElement or kIndexedString.
116 //
117 using AllowOutOfBoundsBits = LookupOnLookupStartObjectBits::Next<bool, 1>;
118
119 //
120 // Encoding when KindBits contains kElement.
121 //
122 using IsWasmArrayBits = AllowOutOfBoundsBits::Next<bool, 1>;
123
124 //
125 // Encoding when KindBits contains kElement and IsWasmArrayBits is 0.
126 //
127 using IsJsArrayBits = IsWasmArrayBits::Next<bool, 1>;
128 using AllowHandlingHole = IsJsArrayBits::Next<bool, 1>;
129 using ElementsKindBits = AllowHandlingHole::Next<ElementsKind, 8>;
130 // Make sure we don't overflow the smi.
131 static_assert(ElementsKindBits::kLastUsedBit < kSmiValueSize);
132
133 //
134 // Encoding when KindBits contains kElement and IsWasmArrayBits is 1.
135 //
136 using WasmArrayTypeBits = IsWasmArrayBits::Next<WasmValueType, 4>;
137 // Make sure we don't overflow the smi.
138 static_assert(WasmArrayTypeBits::kLastUsedBit < kSmiValueSize);
139
140 //
141 // Encoding when KindBits contains kModuleExport.
142 //
143 using ExportsIndexBits = LookupOnLookupStartObjectBits::Next<
144 unsigned,
145 kSmiValueSize - LookupOnLookupStartObjectBits::kLastUsedBit - 1>;
146 static_assert(ExportsIndexBits::kLastUsedBit < kSmiValueSize);
