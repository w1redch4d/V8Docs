46// https://arai-a.github.io/ecma262-compare/?pr=3000&id=sec-disposeresources
47MaybeHandle<Object> JSDisposableStackBase::DisposeResources(
48 Isolate* isolate, DirectHandle<JSDisposableStackBase> disposable_stack,
49 MaybeHandle<Object> maybe_continuation_error,
50 DisposableStackResourcesType resources_type) {
51 DCHECK(!IsUndefined(disposable_stack->stack()));
52
53 DirectHandle<FixedArray> stack(disposable_stack->stack(), isolate);
54
55 // 1. Let needsAwait be false.
56 // 2. Let hasAwaited be false.
57 // `false` is assigned to both in the initialization of the DisposableStack.
58
59 int length = disposable_stack->length();
60
61 MaybeDirectHandle<Object> result;
62 Handle<Object> continuation_error;
63
64 if (maybe_continuation_error.ToHandle(&continuation_error)) {
65 disposable_stack->set_error(*continuation_error);
66 disposable_stack->set_error_message(isolate->pending_message());
67 }
68
69 // 3. For each element resource of
70 // disposeCapability.[[DisposableResourceStack]], in reverse list order, do
71 while (length > 0) {
72 // a. Let value be resource.[[ResourceValue]].
73 // b. Let hint be resource.[[Hint]].
74 // c. Let method be resource.[[DisposeMethod]].
75 Tagged<Object> stack_type = stack->get(--length);
76
77 Tagged<Object> tagged_method = stack->get(--length);
78 DirectHandle<Object> method(tagged_method, isolate);
79
80 Tagged<Object> tagged_value = stack->get(--length);
81 DirectHandle<Object> value(tagged_value, isolate);
82
83 DirectHandle<Object> args[] = {value};
84
85 auto stack_type_case = static_cast<int>(Cast<Smi>(stack_type).value());
86 DisposeMethodCallType call_type =
87 DisposeCallTypeBit::decode(stack_type_case);
88 DisposeMethodHint hint = DisposeHintBit::decode(stack_type_case);
89
90 // d. If hint is sync-dispose and needsAwait is true and hasAwaited is
91 // false, then
92 // i. Perform ! Await(undefined).
93 // ii. Set needsAwait to false.
94
95 if (hint == DisposeMethodHint::kSyncDispose &&
96 disposable_stack->needs_await() == true &&
97 disposable_stack->has_awaited() == false) {
98 // i. Perform ! Await(undefined).
99 // ii. Set needsAwait to false.
100 disposable_stack->set_needs_await(false);
101
102 return ResolveAPromiseWithValueAndReturnIt(
103 isolate, isolate->factory()->undefined_value());
104 }
105
106 // e. If method is not undefined, then
107 if (!IsUndefined(*method)) {
108 // i. Let result be Completion(Call(method, value)).
109
110 if (call_type == DisposeMethodCallType::kValueIsReceiver) {
111 result = Execution::Call(isolate, method, value, {});
112 } else if (call_type == DisposeMethodCallType::kValueIsArgument) {
113 result = Execution::Call(isolate, method,
114 isolate->factory()->undefined_value(),
115 base::VectorOf(args));
116 }
117
118 DirectHandle<Object> result_handle;
119 // ii. If result is a normal completion and hint is async-dispose, then
120 // 1. Set result to Completion(Await(result.[[Value]])).
121 // 2. Set hasAwaited to true.
122 if (result.ToHandle(&result_handle)) {
123 if (hint == DisposeMethodHint::kAsyncDispose) {
124 DCHECK_NE(resources_type, DisposableStackResourcesType::kAllSync);
125 disposable_stack->set_length(length);
126
127 disposable_stack->set_has_awaited(true);
128
129 MaybeHandle<JSReceiver> resolved_promise =
130 ResolveAPromiseWithValueAndReturnIt(isolate, result_handle);
131
132 if (resolved_promise.is_null()) {
133 // iii. If result is a throw completion, then
134 // 1. If completion is a throw completion, then
135 // a. Set result to result.[[Value]].
136 // b. Let suppressed be completion.[[Value]].
137 // c. Let error be a newly created SuppressedError object.
138 // d. Perform CreateNonEnumerableDataPropertyOrThrow(error,
139 // "error", result).
140 // e. Perform CreateNonEnumerableDataPropertyOrThrow(error,
141 // "suppressed", suppressed).
142 // f. Set completion to ThrowCompletion(error).
143 // 2. Else,
144 // a. Set completion to result.
145 CHECK_EXCEPTION_ON_DISPOSAL(isolate, disposable_stack, {});
146 } else {
147 return resolved_promise;
148 }
149 }
150 } else {
151 CHECK_EXCEPTION_ON_DISPOSAL(isolate, disposable_stack, {});
152 }
153 } else {
154 // Else,
155 // i. Assert: hint is async-dispose.
156 DCHECK_EQ(hint, DisposeMethodHint::kAsyncDispose);
157 // ii. Set needsAwait to true.
158 // iii. NOTE: This can only indicate a case where either null or
159 // undefined was the initialized value of an await using declaration.
160 disposable_stack->set_length(length);
161 disposable_stack->set_needs_await(true);
162 }
163 }
164
165 // 4. If needsAwait is true and hasAwaited is false, then
166 // a. Perform ! Await(undefined).
167 if (disposable_stack->needs_await() == true &&
168 disposable_stack->has_awaited() == false) {
169 disposable_stack->set_length(length);
170 disposable_stack->set_has_awaited(true);
171
172 return ResolveAPromiseWithValueAndReturnIt(
173 isolate, isolate->factory()->undefined_value());
174 }
175
176 // 5. NOTE: After disposeCapability has been disposed, it will never be used
177 // again. The contents of disposeCapability.[[DisposableResourceStack]] can be
178 // discarded in implementations, such as by garbage collection, at this point.
179 // 6. Set disposeCapability.[[DisposableResourceStack]] to a new empty List.
180 disposable_stack->set_stack(ReadOnlyRoots(isolate).empty_fixed_array());
181 disposable_stack->set_length(0);
182 disposable_stack->set_state(DisposableStackState::kDisposed);
183
184 Handle<Object> existing_error_handle(disposable_stack->error(), isolate);
185 DirectHandle<Object> existing_error_message_handle(
186 disposable_stack->error_message(), isolate);
187 disposable_stack->set_error(*(isolate->factory()->uninitialized_value()));
188 disposable_stack->set_error_message(
189 *(isolate->factory()->uninitialized_value()));
190
191 // 7. Return ? completion.
192 if (!IsUninitialized(*existing_error_handle) &&
193 !(existing_error_handle.equals(continuation_error))) {
194 if (disposable_stack->suppressed_error_created() == true) {
195 // Created SuppressedError is intentionally suppressed here for debug.
196 SuppressDebug while_processing(isolate->debug());
197 isolate->Throw(*existing_error_handle);
198 } else {
199 isolate->ReThrow(*existing_error_handle, *existing_error_message_handle);
200 }
201 return MaybeHandle<Object>();
202 }
203 return isolate->factory()->true_value();
204}

206MaybeHandle<JSReceiver>
207JSDisposableStackBase::ResolveAPromiseWithValueAndReturnIt(
208 Isolate* isolate, DirectHandle<Object> value) {
209 DirectHandle<JSFunction> promise_function = isolate->promise_function();
210 DirectHandle<Object> args[] = {value};
211 Handle<Object> result;
212 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
213 isolate, result,
214 Execution::CallBuiltin(isolate, isolate->promise_resolve(),
215 promise_function, base::VectorOf(args)),
216 MaybeHandle<JSReceiver>());
217 return Cast<JSReceiver>(result);

220Maybe<bool> JSAsyncDisposableStack::NextDisposeAsyncIteration(
221 Isolate* isolate,
222 DirectHandle<JSDisposableStackBase> async_disposable_stack,
223 DirectHandle<JSPromise> outer_promise) {
224 MaybeDirectHandle<Object> result;
225
226 bool done;
227 do {
228 done = true;
229
230 // 6. Let result be
231 // DisposeResources(asyncDisposableStack.[[DisposeCapability]],
232 // NormalCompletion(undefined)).
233 result =
234 DisposeResources(isolate, async_disposable_stack, MaybeHandle<Object>(),
235 DisposableStackResourcesType::kAtLeastOneAsync);
236
237 DirectHandle<Object> result_handle;
238
239 if (result.ToHandle(&result_handle)) {
240 if (!IsTrue(*result_handle)) {
241 DirectHandle<Context> async_disposable_stack_context =
242 isolate->factory()->NewBuiltinContext(
243 isolate->native_context(),
244 static_cast<int>(
245 JSDisposableStackBase::AsyncDisposableStackContextSlots::
246 kLength));
247 async_disposable_stack_context->set(
248 static_cast<int>(JSDisposableStackBase::
249 AsyncDisposableStackContextSlots::kStack),
250 *async_disposable_stack);
251 async_disposable_stack_context->set(
252 static_cast<int>(
253 JSDisposableStackBase::AsyncDisposableStackContextSlots::
254 kOuterPromise),
255 *outer_promise);
256
257 DirectHandle<JSFunction> on_fulfilled =
258 Factory::JSFunctionBuilder{
259 isolate,
260 isolate->factory()
261 ->async_disposable_stack_on_fulfilled_shared_fun(),
262 async_disposable_stack_context}
263 .Build();
264
265 DirectHandle<JSFunction> on_rejected =
266 Factory::JSFunctionBuilder{
267 isolate,
268 isolate->factory()
269 ->async_disposable_stack_on_rejected_shared_fun(),
270 async_disposable_stack_context}
271 .Build();
272
273 DirectHandle<Object> args[] = {on_fulfilled, on_rejected};
274 if (Execution::CallBuiltin(isolate, isolate->perform_promise_then(),
275 Cast<JSPromise>(result_handle),
276 base::VectorOf(args))
277 .is_null()) {
278 CHECK_EXCEPTION_ON_DISPOSAL(isolate, async_disposable_stack,
279 Nothing<bool>());
280 done = false;
281 }
282 } else {
283 // 8. Perform ! Call(promiseCapability.[[Resolve]], undefined, Â« result
284 // Â»).
285 if (JSPromise::Resolve(outer_promise,
286 isolate->factory()->undefined_value())
287 .is_null()) {
288 CHECK_EXCEPTION_ON_DISPOSAL(isolate, async_disposable_stack,
289 Nothing<bool>());
290 done = false;
291 }
292 }
293 } else {
294 // 7. IfAbruptRejectPromise(result, promiseCapability).
295 DirectHandle<Object> exception(isolate->exception(), isolate);
296 if (!isolate->is_catchable_by_javascript(*exception)) {
297 return Nothing<bool>();
298 }
299 isolate->clear_internal_exception();
300 isolate->clear_pending_message();
301 JSPromise::Reject(outer_promise, exception);
302 }
303 } while (!done);
304
305 return Just(true);
306}
