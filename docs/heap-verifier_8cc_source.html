58// Visitor class to verify interior pointers in spaces that do not contain
59// or care about inter-generational references. All heap object pointers have to
60// point into the heap to a location that has a map pointer at its first word.
61// Caveat: Heap::Contains is an approximation because it can return true for
62// objects in a heap space but above the allocation pointer.
63class VerifyPointersVisitor : public HeapVisitor<VerifyPointersVisitor>,
64 public RootVisitor {
65 public:
66 V8_INLINE explicit VerifyPointersVisitor(Heap* heap)
67 : HeapVisitor(heap), heap_(heap) {}
68
69 void VisitPointers(Tagged<HeapObject> host, ObjectSlot start,
70 ObjectSlot end) override;
71 void VisitPointers(Tagged<HeapObject> host, MaybeObjectSlot start,
72 MaybeObjectSlot end) override;
73 void VisitInstructionStreamPointer(Tagged<Code> host,
74 InstructionStreamSlot slot) override;
75 void VisitCodeTarget(Tagged<InstructionStream> host,
76 RelocInfo* rinfo) override;
77 void VisitEmbeddedPointer(Tagged<InstructionStream> host,
78 RelocInfo* rinfo) override;
79
80 void VisitRootPointers(Root root, const char* description,
81 FullObjectSlot start, FullObjectSlot end) override;
82 void VisitRootPointers(Root root, const char* description,
83 OffHeapObjectSlot start,
84 OffHeapObjectSlot end) override;
85 void VisitMapPointer(Tagged<HeapObject> host) override;
86
87 protected:
88 V8_INLINE void VerifyHeapObjectImpl(Tagged<HeapObject> heap_object);
89 V8_INLINE void VerifyCodeObjectImpl(Tagged<HeapObject> heap_object);

141void VerifyPointersVisitor::VerifyHeapObjectImpl(
142 Tagged<HeapObject> heap_object) {
143 CHECK(IsValidHeapObject(heap_, heap_object));
144 CHECK(IsMap(heap_object->map(cage_base())));
145 // Heap::InToPage() is not available with sticky mark-bits.
146 CHECK_IMPLIES(
147 !v8_flags.sticky_mark_bits && HeapLayout::InYoungGeneration(heap_object),
148 Heap::InToPage(heap_object));

151void VerifyPointersVisitor::VerifyCodeObjectImpl(
152 Tagged<HeapObject> heap_object) {
153 CHECK(IsValidCodeObject(heap_, heap_object));
154 CHECK(IsMap(heap_object->map(cage_base())));
155 CHECK(heap_object->map(cage_base())->instance_type() ==
156 INSTRUCTION_STREAM_TYPE);

159template <typename TSlot>
160void VerifyPointersVisitor::VerifyPointersImpl(TSlot start, TSlot end) {
161 for (TSlot slot = start; slot < end; ++slot) {
162 typename TSlot::TObject object = slot.load(cage_base());
163#ifdef V8_ENABLE_DIRECT_HANDLE
164 if (object.ptr() == kTaggedNullAddress) continue;
165#endif
166 Tagged<HeapObject> heap_object;
167 if (object.GetHeapObject(&heap_object)) {
168 VerifyHeapObjectImpl(heap_object);
169 } else {
170 CHECK(IsSmi(object) || object.IsCleared() ||
171 MapWord::IsPacked(object.ptr()));
172 }
173 }
174}

176void VerifyPointersVisitor::VerifyPointers(Tagged<HeapObject> host,
177 MaybeObjectSlot start,

180 // to one of objects in DATA_ONLY_VISITOR_ID_LIST. You can fix
181 // this by moving that object to POINTER_VISITOR_ID_LIST.
182 CHECK_EQ(ObjectFields::kMaybePointers,
183 Map::ObjectFieldsFrom(host->map(cage_base())->visitor_id()));
184 VerifyPointersImpl(start, end);

221class VerifySharedHeapObjectVisitor : public VerifyPointersVisitor {
222 public:
223 explicit VerifySharedHeapObjectVisitor(Heap* heap)
224 : VerifyPointersVisitor(heap),
225 shared_space_(heap->shared_space()),
226 shared_trusted_space_(heap->shared_trusted_space()),
227 shared_lo_space_(heap->shared_lo_space()),
228 shared_trusted_lo_space_(heap->shared_trusted_lo_space()) {
229 CHECK_NOT_NULL(shared_space_);
230 CHECK_NOT_NULL(shared_lo_space_);
231 }
232
233 protected:
234 void VerifyPointers(Tagged<HeapObject> host, MaybeObjectSlot start,
235 MaybeObjectSlot end) override {
236 if (!host.is_null()) {
237 Tagged<Map> map = host->map();
238 CHECK(ReadOnlyHeap::Contains(map) || shared_space_->Contains(map));
239 }
240 VerifyPointersVisitor::VerifyPointers(host, start, end);
241
242 for (MaybeObjectSlot current = start; current < end; ++current) {
243 Tagged<HeapObject> heap_object;
244 if ((*current).GetHeapObject(&heap_object)) {
245 MemoryChunk* chunk = MemoryChunk::FromHeapObject(heap_object);
246 CHECK(chunk->InReadOnlySpace() || chunk->InWritableSharedSpace());
247 CHECK(ReadOnlyHeap::Contains(heap_object) ||
248 shared_space_->Contains(heap_object) ||
249 shared_lo_space_->Contains(heap_object) ||
250 shared_trusted_space_->Contains(heap_object) ||
251 shared_trusted_lo_space_->Contains(heap_object));
252 }
253 }
254 }
255
256 private:
257 SharedSpace* shared_space_;
258 SharedTrustedSpace* shared_trusted_space_;
259 SharedLargeObjectSpace* shared_lo_space_;
260 SharedTrustedLargeObjectSpace* shared_trusted_lo_space_;
261};

321void HeapVerification::Verify() {
322 CHECK(heap()->HasBeenSetUp());
323 AllowGarbageCollection allow_gc;
324 SafepointKind safepoint_kind = isolate()->is_shared_space_isolate()
325 ? SafepointKind::kGlobal
326 : SafepointKind::kIsolate;
327 SafepointScope safepoint_scope(isolate(), safepoint_kind);
328 HandleScope scope(isolate());
329
330 heap()->MakeHeapIterable();
331 heap()->FreeLinearAllocationAreas();
332
333 // TODO(v8:13257): Currently we don't iterate through the stack conservatively
334 // when verifying the heap.
335 VerifyPointersVisitor visitor(heap());
336 heap()->IterateRoots(&visitor,
337 base::EnumSet<SkipRoot>{SkipRoot::kConservativeStack});
338
339 if (!isolate()->context().is_null() &&
340 !isolate()->raw_native_context().is_null()) {
341 Tagged<Object> normalized_map_cache =
342 isolate()->raw_native_context()->normalized_map_cache();
343
344 if (IsNormalizedMapCache(normalized_map_cache)) {
345 Cast<NormalizedMapCache>(normalized_map_cache)
346 ->NormalizedMapCacheVerify(isolate());
347 }
348 }
349
350#if V8_ENABLE_WEBASSEMBLY
351 // wasm_canonical_rtts holds weak references to maps or (strong) undefined.
352 Tagged<WeakFixedArray> canonical_rtts = heap()->wasm_canonical_rtts();
353 for (int i = 0, e = canonical_rtts->length(); i < e; ++i) {
354 Tagged<MaybeObject> maybe_rtt = canonical_rtts->get(i);
355 if (maybe_rtt.IsCleared()) continue;
356 CHECK(maybe_rtt.IsWeak());
357 CHECK(IsMap(maybe_rtt.GetHeapObjectAssumeWeak()));
358 }
359
360 // js_to_wasm_wrappers holds weak references to code or cleared values.
361 Tagged<WeakFixedArray> wrappers = heap()->js_to_wasm_wrappers();
362 for (int i = 0, e = wrappers->length(); i < e; ++i) {
363 Tagged<MaybeObject> maybe_wrapper = wrappers->get(i);
364 if (maybe_wrapper.IsCleared()) continue;
365 CHECK(maybe_wrapper.IsWeak());
366 CHECK(IsCodeWrapper(maybe_wrapper.GetHeapObjectAssumeWeak()));
367 }
368#endif // V8_ENABLE_WEBASSEMBLY
369
370 // The heap verifier can't deal with partially deserialized objects, so
371 // disable it if a deserializer is active.
372 // TODO(leszeks): Enable verification during deserialization, e.g. by only
373 // blocklisting objects that are in a partially deserialized state.
374 if (isolate()->has_active_deserializer()) return;
375
376 VerifySmisVisitor smis_visitor;
377 heap()->IterateSmiRoots(&smis_visitor);
378
379 VerifySpace(new_space());
380
381 VerifySpace(old_space());
382 VerifySpace(shared_space());
383 VerifySpace(code_space());
384
385 VerifySpace(lo_space());
386 VerifySpace(new_lo_space());
387 VerifySpace(shared_lo_space());
388 VerifySpace(code_lo_space());
389
390 VerifySpace(trusted_space());
391 VerifySpace(shared_trusted_space());
392 VerifySpace(trusted_lo_space());
393 VerifySpace(shared_trusted_lo_space());
394
395 isolate()->string_table()->VerifyIfOwnedBy(isolate());
396
397#if DEBUG
398 heap()->VerifyCommittedPhysicalMemory();
399#endif // DEBUG
400}

409void HeapVerification::VerifyPage(const MemoryChunkMetadata* chunk_metadata) {
410 const MemoryChunk* chunk = chunk_metadata->Chunk();
411
412 CHECK(!current_chunk_.has_value());
413 CHECK(!chunk->IsFlagSet(MemoryChunk::PAGE_NEW_OLD_PROMOTION));
414 CHECK(!chunk->IsFlagSet(MemoryChunk::FROM_PAGE));
415 CHECK(!chunk->IsQuarantined());
416 if (chunk->InReadOnlySpace()) {
417 CHECK_NULL(chunk_metadata->owner());
418 } else {
419 CHECK_EQ(chunk_metadata->heap(), heap());
420 CHECK_EQ(chunk_metadata->owner()->identity(), current_space_identity());
421 }
422 current_chunk_ = chunk_metadata;
423}
424
425void HeapVerification::VerifyPageDone(const MemoryChunkMetadata* chunk) {
426 CHECK_EQ(chunk, *current_chunk_);
427 current_chunk_.reset();
428}

430void HeapVerification::VerifyObject(Tagged<HeapObject> object) {
431 CHECK_EQ(MemoryChunkMetadata::FromHeapObject(object), *current_chunk_);
432
433 // Verify object map.
434 VerifyObjectMap(object);
435
436 // The object itself should look OK.
437 Object::ObjectVerify(object, isolate_);
438
439 // Verify outgoing references.
440 VerifyOutgoingPointers(object);
441
442 // Verify remembered set.
443 if (!heap_->incremental_marking()->IsMinorMarking()) {
444 // Minor incremental marking "steals" the remembered sets from pages.
445 VerifyRememberedSetFor(object);
446 }
447}

449void HeapVerification::VerifyOutgoingPointers(Tagged<HeapObject> object) {
450 switch (current_space_identity()) {
451 case RO_SPACE: {
452 VerifyReadOnlyPointersVisitor visitor(heap());
453 visitor.Visit(object);
454 break;
455 }
456
457 case SHARED_SPACE:
458 case SHARED_TRUSTED_SPACE:
459 case SHARED_LO_SPACE:
460 case SHARED_TRUSTED_LO_SPACE: {
461 VerifySharedHeapObjectVisitor visitor(heap());
462 visitor.Visit(object);
463 break;
464 }
465
466 case NEW_SPACE:
467 case OLD_SPACE:
468 case TRUSTED_SPACE:
469 case CODE_SPACE:
470 case LO_SPACE:
471 case NEW_LO_SPACE:
472 case CODE_LO_SPACE:
473 case TRUSTED_LO_SPACE: {
474 VerifyPointersVisitor visitor(heap());
475 visitor.Visit(object);
476 break;
477 }
478 }
479}

481void HeapVerification::VerifyObjectMap(Tagged<HeapObject> object) {
482 // The first word should be a map, and we expect all map pointers to be
483 // in map space or read-only space.
484 Tagged<Map> map = object->map(cage_base_);
485 CHECK(IsMap(map, cage_base_));
486 CHECK(ReadOnlyHeap::Contains(map) || old_space()->Contains(map) ||
487 (shared_space() && shared_space()->Contains(map)));
488
489 if (HeapLayout::InYoungGeneration(object)) {
490 // The object should not be code or a map.
491 CHECK(!IsMap(object, cage_base_));
492 CHECK(!IsAbstractCode(object, cage_base_));
493 } else if (current_space_identity() == RO_SPACE) {
494 CHECK(!IsExternalString(object));
495 CHECK(!IsJSArrayBuffer(object));
496 }
497}

504class SlotVerifyingVisitor : public HeapVisitor<SlotVerifyingVisitor> {
505 public:
506 SlotVerifyingVisitor(Isolate* isolate, std::set<Address>* untyped,
507 std::set<std::pair<SlotType, Address>>* typed,
508 std::set<Address>* protected_pointer)
509 : HeapVisitor(isolate),

514 virtual bool ShouldHaveBeenRecorded(Tagged<HeapObject> host,
515 Tagged<MaybeObject> target) = 0;

517 void VisitPointers(Tagged<HeapObject> host, ObjectSlot start,
518 ObjectSlot end) override {
519#ifdef DEBUG
520 for (ObjectSlot slot = start; slot < end; ++slot) {
521 Tagged<Object> obj = slot.load(cage_base());
522 CHECK(!MapWord::IsPacked(obj.ptr()) || !HasWeakHeapObjectTag(obj));
523 }
524#endif // DEBUG
525 VisitPointers(host, MaybeObjectSlot(start), MaybeObjectSlot(end));

528 void VisitPointers(Tagged<HeapObject> host, MaybeObjectSlot start,
529 MaybeObjectSlot end) final {
530 for (MaybeObjectSlot slot = start; slot < end; ++slot) {
531 if (ShouldHaveBeenRecorded(host, slot.load(cage_base()))) {
532 CHECK_GT(untyped_->count(slot.address()), 0);
533 }
534 }
535 }
536
537 void VisitInstructionStreamPointer(Tagged<Code> host,
538 InstructionStreamSlot slot) override {
539 if (ShouldHaveBeenRecorded(host, slot.load(code_cage_base()))) {
540 CHECK_GT(untyped_->count(slot.address()), 0);
541 }
542 }
543
544 void VisitCodeTarget(Tagged<InstructionStream> host,
545 RelocInfo* rinfo) override {
546 Tagged<Object> target =
547 InstructionStream::FromTargetAddress(rinfo->target_address());
548 if (ShouldHaveBeenRecorded(host, target)) {
549 CHECK(InTypedSet(SlotType::kCodeEntry, rinfo->pc()) ||
550 (rinfo->IsInConstantPool() &&
551 InTypedSet(SlotType::kConstPoolCodeEntry,
552 rinfo->constant_pool_entry_address())));
553 }
554 }
555
556 void VisitEmbeddedPointer(Tagged<InstructionStream> host,
557 RelocInfo* rinfo) override {
558 Tagged<Object> target = rinfo->target_object(cage_base());
559 if (ShouldHaveBeenRecorded(host, target)) {
560 CHECK(InTypedSet(SlotType::kEmbeddedObjectFull, rinfo->pc()) ||
561 InTypedSet(SlotType::kEmbeddedObjectCompressed, rinfo->pc()) ||
562 (rinfo->IsInConstantPool() &&
563 InTypedSet(SlotType::kConstPoolEmbeddedObjectCompressed,
564 rinfo->constant_pool_entry_address())) ||
565 (rinfo->IsInConstantPool() &&
566 InTypedSet(SlotType::kConstPoolEmbeddedObjectFull,
567 rinfo->constant_pool_entry_address())));
568 }
569 }
570
571 void VisitProtectedPointer(Tagged<TrustedObject> host,
572 ProtectedPointerSlot slot) override {
573 if (ShouldHaveBeenRecorded(host, slot.load())) {
574 CHECK_NOT_NULL(protected_);
575 CHECK_GT(protected_->count(slot.address()), 0);
576 }
577 }
578
579 void VisitProtectedPointer(Tagged<TrustedObject> host,
580 ProtectedMaybeObjectSlot slot) override {
581 if (ShouldHaveBeenRecorded(host, slot.load())) {
582 CHECK_NOT_NULL(protected_);
583 CHECK_GT(protected_->count(slot.address()), 0);
584 }
585 }
586
587 void VisitMapPointer(Tagged<HeapObject> object) override {
588 VisitPointers(object, object->map_slot(), object->map_slot() + 1);
589 }
590
591 protected:

593 return untyped_->count(slot.address()) > 0;
594 }

596 private:

598 return typed_->count(std::make_pair(type, slot)) > 0;
599 }

666template <RememberedSetType direction>
667void CollectSlots(MutablePageMetadata* chunk, Address start, Address end,
668 std::set<Address>* untyped,
669 std::set<std::pair<SlotType, Address>>* typed) {
670 RememberedSet<direction>::Iterate(
671 chunk,
672 [start, end, untyped](MaybeObjectSlot slot) {
673 if (start <= slot.address() && slot.address() < end) {
674 untyped->insert(slot.address());
675 }
676 return KEEP_SLOT;
677 },
678 SlotSet::FREE_EMPTY_BUCKETS);
679 RememberedSet<direction>::IterateTyped(
680 chunk, [=](SlotType type, Address slot) {
681 if (start <= slot && slot < end) {
682 typed->insert(std::make_pair(type, slot));
683 }
684 return KEEP_SLOT;
685 });
686}

740void HeapVerification::VerifyRememberedSetFor(Tagged<HeapObject> object) {
741 if (current_space_identity() == RO_SPACE ||
742 v8_flags.verify_heap_skip_remembered_set) {
743 return;
744 }
745
746 MutablePageMetadata* chunk = MutablePageMetadata::FromHeapObject(object);
747
748 Address start = object.address();
749 Address end = start + object->Size(cage_base_);
750
751 std::set<Address> old_to_new;
752 std::set<std::pair<SlotType, Address>> typed_old_to_new;
753 CollectSlots<OLD_TO_NEW>(chunk, start, end, &old_to_new, &typed_old_to_new);
754 CollectSlots<OLD_TO_NEW_BACKGROUND>(chunk, start, end, &old_to_new,
755 &typed_old_to_new);
756
757 OldToNewSlotVerifyingVisitor old_to_new_visitor(
758 isolate(), &old_to_new, &typed_old_to_new,
759 heap()->ephemeron_remembered_set()->tables());
760 old_to_new_visitor.Visit(object);
761
762 std::set<Address> old_to_shared;
763 std::set<std::pair<SlotType, Address>> typed_old_to_shared;
764 CollectSlots<OLD_TO_SHARED>(chunk, start, end, &old_to_shared,
765 &typed_old_to_shared);
766 std::set<Address> trusted_to_shared_trusted;
767 CollectSlots<TRUSTED_TO_SHARED_TRUSTED>(chunk, start, end,
768 &trusted_to_shared_trusted, nullptr);
769 OldToSharedSlotVerifyingVisitor old_to_shared_visitor(
770 isolate(), &old_to_shared, &typed_old_to_shared,
771 &trusted_to_shared_trusted);
772 old_to_shared_visitor.Visit(object);
773
774 if (!MemoryChunk::FromHeapObject(object)->IsTrusted()) {
775 CHECK_NULL(chunk->slot_set<TRUSTED_TO_TRUSTED>());
776 CHECK_NULL(chunk->slot_set<TRUSTED_TO_SHARED_TRUSTED>());
777 }
778
779 if (HeapLayout::InWritableSharedSpace(object)) {
780 CHECK_NULL(chunk->slot_set<OLD_TO_SHARED>());
781 CHECK_NULL(chunk->typed_slot_set<OLD_TO_SHARED>());
782
783 CHECK_NULL(chunk->slot_set<TRUSTED_TO_SHARED_TRUSTED>());
784 CHECK_NULL(chunk->typed_slot_set<TRUSTED_TO_SHARED_TRUSTED>());
785
786 CHECK_NULL(chunk->slot_set<OLD_TO_NEW>());
787 CHECK_NULL(chunk->typed_slot_set<OLD_TO_NEW>());
788
789 CHECK_NULL(chunk->slot_set<OLD_TO_NEW_BACKGROUND>());
790 CHECK_NULL(chunk->typed_slot_set<OLD_TO_NEW_BACKGROUND>());
791 }
792
793 if (!v8_flags.sticky_mark_bits && HeapLayout::InYoungGeneration(object)) {
794 CHECK_NULL(chunk->slot_set<OLD_TO_NEW>());
795 CHECK_NULL(chunk->typed_slot_set<OLD_TO_NEW>());
796
797 CHECK_NULL(chunk->slot_set<OLD_TO_NEW_BACKGROUND>());
798 CHECK_NULL(chunk->typed_slot_set<OLD_TO_NEW_BACKGROUND>());
799
800 CHECK_NULL(chunk->slot_set<OLD_TO_OLD>());
801 CHECK_NULL(chunk->typed_slot_set<OLD_TO_OLD>());
802
803 CHECK_NULL(chunk->slot_set<OLD_TO_SHARED>());
804 CHECK_NULL(chunk->typed_slot_set<OLD_TO_SHARED>());
805 }
806
807 // TODO(v8:11797): Add old to old slot set verification once all weak objects
808 // have their own instance types and slots are recorded for all weak fields.
809}

851// static
852void HeapVerifier::VerifyObjectLayoutChange(Heap* heap,
853 Tagged<HeapObject> object,
854 Tagged<Map> new_map) {
855 // Object layout changes are currently not supported on background threads.
856 CHECK_NULL(LocalHeap::Current());
857
858 if (!v8_flags.verify_heap) return;
859
860 VerifyObjectLayoutChangeIsAllowed(heap, object);
861
862 PtrComprCageBase cage_base(heap->isolate());
863
864 // Check that Heap::NotifyObjectLayoutChange was called for object transitions
865 // that are not safe for concurrent marking.
866 // If you see this check triggering for a freshly allocated object,
867 // use object->set_map_after_allocation() to initialize its map.
868 if (pending_layout_change_object.is_null()) {
869 VerifySafeMapTransition(heap, object, new_map);
870 } else {
871 CHECK_EQ(pending_layout_change_object, object);
872 pending_layout_change_object = HeapObject();
873 }
874}

876// static
877void HeapVerifier::VerifySafeMapTransition(Heap* heap,
878 Tagged<HeapObject> object,
879 Tagged<Map> new_map) {
880 PtrComprCageBase cage_base(heap->isolate());
881
882 if (IsJSObject(object, cage_base)) {
883 // Without double unboxing all in-object fields of a JSObject are tagged.
884 return;
885 }
886
887 if (IsString(object, cage_base) &&
888 (new_map == ReadOnlyRoots(heap).thin_two_byte_string_map() ||
889 new_map == ReadOnlyRoots(heap).thin_one_byte_string_map())) {
890 // When transitioning a string to ThinString,
891 // Heap::NotifyObjectLayoutChange doesn't need to be invoked because only
892 // tagged fields are introduced.
893 return;
894 }
895
896 if (v8_flags.shared_string_table && IsString(object, cage_base) &&
897 InstanceTypeChecker::IsInternalizedString(new_map->instance_type())) {
898 // In-place internalization does not change a string's fields.
899 //
900 // When sharing the string table, the setting and re-setting of maps below
901 // can race when there are parallel internalization operations, causing
902 // CHECKs to fail.
903 return;
904 }
905
906 // Check that the set of slots before and after the transition match.
907 SlotCollectingVisitor old_visitor(heap->isolate());
908 old_visitor.Visit(object);
909 MapWord old_map_word = object->map_word(cage_base, kRelaxedLoad);
910 // Temporarily set the new map to iterate new slots.
911 object->set_map_word(new_map, kRelaxedStore);
912 SlotCollectingVisitor new_visitor(heap->isolate());
913 new_visitor.Visit(object);
914 // Restore the old map.
915 object->set_map_word(old_map_word.ToMap(), kRelaxedStore);
916 CHECK_EQ(new_visitor.number_of_slots(), old_visitor.number_of_slots());
917 for (int i = 0; i < new_visitor.number_of_slots(); i++) {
918 CHECK_EQ(new_visitor.slot(i), old_visitor.slot(i));
919 }
920#ifdef V8_EXTERNAL_CODE_SPACE
921 CHECK_EQ(new_visitor.number_of_code_slots(),
922 old_visitor.number_of_code_slots());
923 for (int i = 0; i < new_visitor.number_of_code_slots(); i++) {
924 CHECK_EQ(new_visitor.code_slot(i), old_visitor.code_slot(i));
925 }
926#endif // V8_EXTERNAL_CODE_SPACE
927}
