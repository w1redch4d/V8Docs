31// A representation of a constant expression. The most common expression types
32// are hard-coded, while the rest are represented as a {WireBytesRef}.
33class ConstantExpression {
34 public:
35 enum Kind {
36 kEmpty,
37 kI32Const,
38 kRefNull,
39 kRefFunc,
40 kWireBytesRef,
41 kLastKind = kWireBytesRef
42 };
43
44 constexpr ConstantExpression() = default;
45
46 static constexpr ConstantExpression I32Const(int32_t value) {
47 return ConstantExpression(ValueField::encode(value) |
48 KindField::encode(kI32Const));
49 }
50 static constexpr ConstantExpression RefFunc(uint32_t index) {
51 return ConstantExpression(ValueField::encode(index) |
52 KindField::encode(kRefFunc));
53 }
54 static constexpr ConstantExpression RefNull(HeapType::Representation repr) {
55 return ConstantExpression(ValueField::encode(repr) |
56 KindField::encode(kRefNull));
57 }
58 static constexpr ConstantExpression WireBytes(uint32_t offset,
59 uint32_t length) {
60 return ConstantExpression(OffsetField::encode(offset) |
61 LengthField::encode(length) |
62 KindField::encode(kWireBytesRef));
63 }
64
65 constexpr Kind kind() const { return KindField::decode(bit_field_); }
66
67 constexpr bool is_set() const { return kind() != kEmpty; }
68
69 constexpr uint32_t index() const {
70 DCHECK_EQ(kind(), kRefFunc);
71 return ValueField::decode(bit_field_);
72 }
73
74 constexpr HeapType::Representation repr() const {
75 DCHECK_EQ(kind(), kRefNull);
76 return static_cast<HeapType::Representation>(
77 ValueField::decode(bit_field_));
78 }
79
80 constexpr int32_t i32_value() const {
81 DCHECK_EQ(kind(), kI32Const);
82 return ValueField::decode(bit_field_);
83 }
84
85 V8_EXPORT_PRIVATE WireBytesRef wire_bytes_ref() const;
86
87 private:
88 static constexpr int kValueBits = 32;
89 static constexpr int kLengthBits = 30;
90 static constexpr int kOffsetBits = 30;
91 static constexpr int kKindBits = 3;
92
93 // There are two possible combinations of fields: offset + length + kind if
94 // kind = kWireBytesRef, or value + kind for anything else.
95 using ValueField = base::BitField<uint32_t, 0, kValueBits, uint64_t>;
96 using OffsetField = base::BitField<uint32_t, 0, kOffsetBits, uint64_t>;
97 using LengthField = OffsetField::Next<uint32_t, kLengthBits>;
98 using KindField = LengthField::Next<Kind, kKindBits>;
99
100 // Make sure we reserve enough bits for a {WireBytesRef}'s length and offset.
101 static_assert(kV8MaxWasmModuleSize <= LengthField::kMax + 1);
102 static_assert(kV8MaxWasmModuleSize <= OffsetField::kMax + 1);
103 // Make sure kind fits in kKindBits.
104 static_assert(kLastKind <= KindField::kMax + 1);
105
106 explicit constexpr ConstantExpression(uint64_t bit_field)
107 : bit_field_(bit_field) {}
108
109 uint64_t bit_field_ = 0;
110};
111
112// Verify that the default constructor initializes the {kind()} to {kEmpty}.
113static_assert(ConstantExpression{}.kind() == ConstantExpression::kEmpty);
114
115// We want to keep {ConstantExpression} small to reduce memory usage during
116// compilation/instantiation.
117static_assert(sizeof(ConstantExpression) <= 8);
