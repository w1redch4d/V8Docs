58 template <TurboshaftPhase Phase, typename... Args>
59 auto Run(Args&&... args) {
60 // Setup run scope.
61 PhaseScope phase_scope(data_->pipeline_statistics(), Phase::phase_name());
62 ZoneWithName<Phase::kPhaseName> temp_zone(data_->zone_stats(),
63 Phase::phase_name());
64 NodeOriginTable::PhaseScope origin_scope(data_->node_origins(),
65 Phase::phase_name());
66#ifdef V8_RUNTIME_CALL_STATS
67 RuntimeCallTimerScope runtime_call_timer_scope(data_->runtime_call_stats(),
68 Phase::kRuntimeCallCounterId,
69 Phase::kCounterMode);
70#endif
71
72 Phase phase;
73 using result_t =
74 decltype(phase.Run(data_, temp_zone, std::forward<Args>(args)...));
75 if constexpr (std::is_same_v<result_t, void>) {
76 phase.Run(data_, temp_zone, std::forward<Args>(args)...);
77 if constexpr (produces_printable_graph<Phase>::value) {
78 PrintGraph(temp_zone, Phase::phase_name());
79 }
80 return;
81 } else {
82 auto result = phase.Run(data_, temp_zone, std::forward<Args>(args)...);
83 if constexpr (produces_printable_graph<Phase>::value) {
84 PrintGraph(temp_zone, Phase::phase_name());
85 }
86 return result;
87 }
88 UNREACHABLE();
89 }

163 bool OptimizeTurboshaftGraph(Linkage* linkage) {
164 UnparkedScopeIfNeeded scope(data_->broker(),
165 v8_flags.turboshaft_trace_reduction ||
166 v8_flags.turboshaft_trace_emitted);
167
168 turboshaft::Tracing::Scope tracing_scope(data_->info());
169
170#ifdef V8_ENABLE_WEBASSEMBLY
171 // TODO(dlehmann,353475584): Once the Wasm-in-JS TS inlining MVP is feature-
172 // complete and cleaned-up, move its reducer into the beginning of the
173 // `MachineLoweringPhase` since we can reuse the `DataViewLoweringReducer`
174 // there and avoid a separate phase.
175 if (v8_flags.turboshaft_wasm_in_js_inlining) {
176 Run<turboshaft::WasmInJSInliningPhase>();
177 }
178#endif // !V8_ENABLE_WEBASSEMBLY
179
180 Run<turboshaft::MachineLoweringPhase>();
181
182 // TODO(dmercadier): find a way to merge LoopPeeling and LoopUnrolling. It's
183 // not currently possible for 2 reasons. First, LoopPeeling reduces the
184 // number of iteration of a loop, thus invalidating LoopUnrolling's
185 // analysis. This could probably be worked around fairly easily though.
186 // Second, LoopPeeling has to emit the non-peeled header of peeled loops, in
187 // order to fix their loop phis (because their 1st input should be replace
188 // by their 2nd input coming from the peeled iteration), but LoopUnrolling
189 // has to be triggered before emitting the loop header. This could be fixed
190 // by changing LoopUnrolling start unrolling after the 1st header has been
191 // emitted, but this would also require updating CloneSubgraph.
192 if (v8_flags.turboshaft_loop_peeling) {
193 Run<turboshaft::LoopPeelingPhase>();
194 }
195
196 if (v8_flags.turboshaft_loop_unrolling) {
197 Run<turboshaft::LoopUnrollingPhase>();
198 }
199
200 if (v8_flags.turbo_store_elimination) {
201 Run<turboshaft::StoreStoreEliminationPhase>();
202 }
203
204 Run<turboshaft::OptimizePhase>();
205
206 if (v8_flags.turboshaft_typed_optimizations) {
207 Run<turboshaft::TypedOptimizationsPhase>();
208 }
209
210 if (v8_flags.turboshaft_assert_types) {
211 Run<turboshaft::TypeAssertionsPhase>();
212 }
213
214 // Perform dead code elimination, reduce stack checks, simplify loads on
215 // platforms where required, ...
216 Run<turboshaft::CodeEliminationAndSimplificationPhase>();
217
218#ifdef V8_ENABLE_DEBUG_CODE
219 if (V8_UNLIKELY(v8_flags.turboshaft_enable_debug_features)) {
220 // This phase has to run very late to allow all previous phases to use
221 // debug features.
222 Run<turboshaft::DebugFeatureLoweringPhase>();
223 }
224#endif // V8_ENABLE_DEBUG_CODE
225
226 return true;
227 }

229 void PrepareForInstructionSelection(
230 const ProfileDataFromFile* profile = nullptr) {
231 if (V8_UNLIKELY(data()->pipeline_kind() == TurboshaftPipelineKind::kCSA ||
232 data()->pipeline_kind() ==
233 TurboshaftPipelineKind::kTSABuiltin)) {
234 if (profile) {
235 Run<ProfileApplicationPhase>(profile);
236 }
237
238 if (v8_flags.reorder_builtins &&
239 Builtins::IsBuiltinId(info()->builtin())) {
240 UnparkedScopeIfNeeded unparked_scope(data()->broker());
241 BasicBlockCallGraphProfiler::StoreCallGraph(info(), data()->graph());
242 }
243
244 if (v8_flags.turbo_profiling) {
245 UnparkedScopeIfNeeded unparked_scope(data()->broker());
246
247 // Basic block profiling disables concurrent compilation, so handle
248 // deref is fine.
249 AllowHandleDereference allow_handle_dereference;
250 const size_t block_count = data()->graph().block_count();
251 BasicBlockProfilerData* profiler_data =
252 BasicBlockProfiler::Get()->NewData(block_count);
253
254 // Set the function name.
255 profiler_data->SetFunctionName(info()->GetDebugName());
256 // Capture the schedule string before instrumentation.
257 if (v8_flags.turbo_profiling_verbose) {
258 std::ostringstream os;
259 os << data()->graph();
260 profiler_data->SetSchedule(os);
261 }
262
263 info()->set_profiler_data(profiler_data);
264
265 Run<BlockInstrumentationPhase>();
266 } else {
267 // We run an empty copying phase to make sure that we have the same
268 // control flow as when taking the profile.
269 ZoneWithName<kTempZoneName> temp_zone(data()->zone_stats(),
270 kTempZoneName);
271 CopyingPhase<>::Run(data(), temp_zone);
272 }
273 }
274
275 // DecompressionOptimization has to run as the last phase because it
276 // constructs an (slightly) invalid graph that mixes Tagged and Compressed
277 // representations.
278 Run<DecompressionOptimizationPhase>();
279
280 Run<SpecialRPOSchedulingPhase>();
281 }

332 bool AllocateRegisters(CallDescriptor* call_descriptor) {
333 BeginPhaseKind("V8.TFRegisterAllocation");
334
335 bool run_verifier = v8_flags.turbo_verify_allocation;
336
337 // Allocate registers.
338 const RegisterConfiguration* config = RegisterConfiguration::Default();
339 std::unique_ptr<const RegisterConfiguration> restricted_config;
340 if (call_descriptor->HasRestrictedAllocatableRegisters()) {
341 RegList registers = call_descriptor->AllocatableRegisters();
342 DCHECK_LT(0, registers.Count());
343 restricted_config.reset(
344 RegisterConfiguration::RestrictGeneralRegisters(registers));
345 config = restricted_config.get();
346 }
347 AllocateRegisters(config, call_descriptor, run_verifier);
348
349 // Verify the instruction sequence has the same hash in two stages.
350 VerifyGeneratedCodeIsIdempotent();
351
352 Run<FrameElisionPhase>();
353
354 // TODO(mtrofin): move this off to the register allocator.
355 bool generate_frame_at_start =
356 data_->sequence()->instruction_blocks().front()->must_construct_frame();
357 // Optimimize jumps.
358 if (v8_flags.turbo_jt) {
359 Run<JumpThreadingPhase>(generate_frame_at_start);
360 }
361
362 EndPhaseKind();
363
364 return true;
365 }

372 void VerifyGeneratedCodeIsIdempotent() {
373 JumpOptimizationInfo* jump_opt = data()->jump_optimization_info();
374 if (jump_opt == nullptr) return;
375
376 InstructionSequence* code = data()->sequence();
377 int instruction_blocks = code->InstructionBlockCount();
378 int virtual_registers = code->VirtualRegisterCount();
379 size_t hash_code =
380 base::hash_combine(instruction_blocks, virtual_registers);
381 for (Instruction* instr : *code) {
382 hash_code = base::hash_combine(hash_code, instr->opcode(),
383 instr->InputCount(), instr->OutputCount());
384 }
385 for (int i = 0; i < virtual_registers; i++) {
386 hash_code = base::hash_combine(hash_code, code->GetRepresentation(i));
387 }
388 if (jump_opt->is_collecting()) {
389 jump_opt->hash_code = hash_code;
390 } else {
391 CHECK_EQ(hash_code, jump_opt->hash_code);
392 }
393 }

395 void AllocateRegisters(const RegisterConfiguration* config,
396 CallDescriptor* call_descriptor, bool run_verifier) {
397 // Don't track usage for this zone in compiler stats.
398 std::unique_ptr<Zone> verifier_zone;
399 RegisterAllocatorVerifier* verifier = nullptr;
400 if (run_verifier) {
401 AccountingAllocator* allocator = data()->allocator();
402 DCHECK_NOT_NULL(allocator);
403 verifier_zone.reset(
404 new Zone(allocator, kRegisterAllocatorVerifierZoneName));
405 verifier = verifier_zone->New<RegisterAllocatorVerifier>(
406 verifier_zone.get(), config, data()->sequence(), data()->frame());
407 }
408
409#ifdef DEBUG
410 data_->sequence()->ValidateEdgeSplitForm();
411 data_->sequence()->ValidateDeferredBlockEntryPaths();
412 data_->sequence()->ValidateDeferredBlockExitPaths();
413#endif
414
415 data_->InitializeRegisterComponent(config, call_descriptor);
416
417 Run<MeetRegisterConstraintsPhase>();
418 Run<ResolvePhisPhase>();
419 Run<BuildLiveRangesPhase>();
420 Run<BuildLiveRangeBundlesPhase>();
421
422 TraceSequence("before register allocation");
423 if (verifier != nullptr) {
424 CHECK(!data_->register_allocation_data()->ExistsUseWithoutDefinition());
425 CHECK(data_->register_allocation_data()
426 ->RangesDefinedInDeferredStayInDeferred());
427 }
428
429 if (data_->info()->trace_turbo_json() && !MayHaveUnverifiableGraph()) {
430 TurboCfgFile tcf(data_->isolate());
431 tcf << AsC1VRegisterAllocationData("PreAllocation",
432 data_->register_allocation_data());
433 }
434
435 Run<AllocateGeneralRegistersPhase<LinearScanAllocator>>();
436
437 if (data_->sequence()->HasFPVirtualRegisters()) {
438 Run<AllocateFPRegistersPhase<LinearScanAllocator>>();
439 }
440
441 if (data_->sequence()->HasSimd128VirtualRegisters() &&
442 (kFPAliasing == AliasingKind::kIndependent)) {
443 Run<AllocateSimd128RegistersPhase<LinearScanAllocator>>();
444 }
445
446 Run<DecideSpillingModePhase>();
447 Run<AssignSpillSlotsPhase>();
448 Run<CommitAssignmentPhase>();
449
450 // TODO(chromium:725559): remove this check once
451 // we understand the cause of the bug. We keep just the
452 // check at the end of the allocation.
453 if (verifier != nullptr) {
454 verifier->VerifyAssignment("Immediately after CommitAssignmentPhase.");
455 }
456
457 Run<ConnectRangesPhase>();
458
459 Run<ResolveControlFlowPhase>();
460
461 Run<PopulateReferenceMapsPhase>();
462
463 if (v8_flags.turbo_move_optimization) {
464 Run<OptimizeMovesPhase>();
465 }
466
467 TraceSequence("after register allocation");
468
469 if (verifier != nullptr) {
470 verifier->VerifyAssignment("End of regalloc pipeline.");
471 verifier->VerifyGapMoves();
472 }
473
474 if (data_->info()->trace_turbo_json() && !MayHaveUnverifiableGraph()) {
475 TurboCfgFile tcf(data_->isolate());
476 tcf << AsC1VRegisterAllocationData("CodeGen",
477 data_->register_allocation_data());
478 }
479
480 data()->ClearRegisterComponent();
481 }

504 MaybeHandle<Code> GenerateCode(CallDescriptor* call_descriptor) {
505 Linkage linkage(call_descriptor);

525 MaybeIndirectHandle<Code> FinalizeCode(bool retire_broker = true) {
526 BeginPhaseKind("V8.TFFinalizeCode");
527 if (data_->broker() && retire_broker) {
528 data_->broker()->Retire();
529 }
530 Run<FinalizeCodePhase>();
531
532 MaybeIndirectHandle<Code> maybe_code = data_->code();
533 IndirectHandle<Code> code;
534 if (!maybe_code.ToHandle(&code)) {
535 return maybe_code;
536 }
537
538 data_->info()->SetCode(code);
539 PrintCode(data_->isolate(), code, data_->info());
540
541 // Functions with many inline candidates are sensitive to correct call
542 // frequency feedback and should therefore not be tiered up early.
543 if (v8_flags.profile_guided_optimization &&
544 info()->could_not_inline_all_candidates() &&
545 info()->shared_info()->cached_tiering_decision() !=
546 CachedTieringDecision::kDelayMaglev) {
547 info()->shared_info()->set_cached_tiering_decision(
548 CachedTieringDecision::kNormal);
549 }
550
551 if (info()->trace_turbo_json()) {
552 TurboJsonFile json_of(info(), std::ios_base::app);
553
554 json_of << "{\"name\":\"disassembly\",\"type\":\"disassembly\""
555 << BlockStartsAsJSON{&data_->code_generator()->block_starts()}
556 << "\"data\":\"";
557#ifdef ENABLE_DISASSEMBLER
558 std::stringstream disassembly_stream;
559 code->Disassemble(nullptr, disassembly_stream, data_->isolate());
560 std::string disassembly_string(disassembly_stream.str());
561 for (const auto& c : disassembly_string) {
562 json_of << AsEscapedUC16ForJSON(c);
563 }
564#endif // ENABLE_DISASSEMBLER
565 json_of << "\"}\n],\n";
566 json_of << "\"nodePositions\":";
567 // TODO(nicohartmann): We should try to always provide source positions.
568 json_of << (data_->source_position_output().empty()
569 ? "{}"
570 : data_->source_position_output())
571 << ",\n";
572 JsonPrintAllSourceWithPositions(json_of, data_->info(), data_->isolate());
573 if (info()->has_bytecode_array()) {
574 json_of << ",\n";
575 JsonPrintAllBytecodeSources(json_of, info());
576 }
577 json_of << "\n}";
578 }
579 if (info()->trace_turbo_json() || info()->trace_turbo_graph()) {
580 CodeTracer::StreamScope tracing_scope(data_->GetCodeTracer());
581 tracing_scope.stream()
582 << "---------------------------------------------------\n"
583 << "Finished compiling method " << info()->GetDebugName().get()
584 << " using TurboFan" << std::endl;
585 }
586 EndPhaseKind();
587 return code;
588 }
