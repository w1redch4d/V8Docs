31 public:
32 WasmIntoJSInlinerImpl(Zone* zone, const wasm::WasmModule* module,
33 MachineGraph* mcgraph, const wasm::FunctionBody& body,
34 base::Vector<const uint8_t> bytes,
35 SourcePositionTable* source_position_table,
36 int inlining_id)
37 : wasm::Decoder(bytes.begin(), bytes.end()),
38 module_(module),
39 mcgraph_(mcgraph),
40 body_(body),
41 graph_(mcgraph->graph()),
42 gasm_(mcgraph, zone),
43 source_position_table_(source_position_table),
44 inlining_id_(inlining_id) {
45 // +1 for instance node.
46 size_t params = body.sig->parameter_count() + 1;
47 Node* start =
48 graph_->NewNode(mcgraph->common()->Start(static_cast<int>(params)));
49 graph_->SetStart(start);
50 graph_->SetEnd(graph_->NewNode(mcgraph->common()->End(0)));
51 gasm_.InitializeEffectControl(start, start);
52
53 // Initialize parameter nodes.
54 // We have to add another +1 as the minimum parameter index is actually
55 // -1, not 0...
56 size_t params_extended = params + 1;
57 parameters_ = zone->AllocateArray<Node*>(params_extended);
58 for (unsigned i = 0; i < params_extended; i++) {
59 parameters_[i] = nullptr;
60 }
61 // Instance node at parameter 0.
62 trusted_data_node_ = Param(wasm::kWasmInstanceDataParameterIndex);
63 }

65 Node* Param(int index, const char* debug_name = nullptr) {
66 DCHECK_NOT_NULL(graph_->start());
67 // Turbofan allows negative parameter indices.
68 DCHECK_GE(index, kMinParameterIndex);
69 int array_index = index - kMinParameterIndex;
70 if (parameters_[array_index] == nullptr) {
71 Node* param = graph_->NewNode(
72 mcgraph_->common()->Parameter(index, debug_name), graph_->start());
73 if (index > wasm::kWasmInstanceDataParameterIndex) {
74 // Add a type guard to keep type information based on the inlinee's
75 // signature.
76 wasm::ValueType type = body_.sig->GetParam(index - 1);
77 Type tf_type = compiler::Type::Wasm(type, module_, graph_->zone());
78 param = gasm_.TypeGuard(tf_type, param);
79 }
80 parameters_[array_index] = param;
81 }
82 return parameters_[array_index];
83 }

85 bool TryInlining() {
86 if (body_.sig->return_count() > 1) {
87 return false; // Multi-return is not supported.
88 }
89 // Parse locals.
90 if (consume_u32v() != 0) {
91 // Functions with locals are not supported.
92 return false;
93 }
94 // Parse body.
95 base::SmallVector<Value, 4> stack;
96 while (is_inlineable_) {
97 WasmOpcode opcode = ReadOpcode();
98 switch (opcode) {
99 case wasm::kExprAnyConvertExtern:
100 DCHECK(!stack.empty());
101 stack.back() = ParseAnyConvertExtern(stack.back());
102 continue;
103 case wasm::kExprExternConvertAny:
104 DCHECK(!stack.empty());
105 stack.back() = ParseExternConvertAny(stack.back());
106 continue;
107 case wasm::kExprRefCast:
108 case wasm::kExprRefCastNull:
109 DCHECK(!stack.empty());
110 stack.back() =
111 ParseRefCast(stack.back(), opcode == wasm::kExprRefCastNull);
112 continue;
113 case wasm::kExprArrayLen:
114 DCHECK(!stack.empty());
115 stack.back() = ParseArrayLen(stack.back());
116 continue;
117 case wasm::kExprArrayGet:
118 case wasm::kExprArrayGetS:
119 case wasm::kExprArrayGetU: {
120 DCHECK_GE(stack.size(), 2);
121 Value index = stack.back();
122 stack.pop_back();
123 Value array = stack.back();
124 stack.back() = ParseArrayGet(array, index, opcode);
125 continue;
126 }
127 case wasm::kExprArraySet: {
128 DCHECK_GE(stack.size(), 3);
129 Value value = stack.back();
130 stack.pop_back();
131 Value index = stack.back();
132 stack.pop_back();
133 Value array = stack.back();
134 stack.pop_back();
135 ParseArraySet(array, index, value);
136 continue;
137 }
138 case wasm::kExprStructGet:
139 case wasm::kExprStructGetS:
140 case wasm::kExprStructGetU:
141 DCHECK(!stack.empty());
142 stack.back() = ParseStructGet(stack.back(), opcode);
143 continue;
144 case wasm::kExprStructSet: {
145 DCHECK_GE(stack.size(), 2);
146 Value value = stack.back();
147 stack.pop_back();
148 Value wasm_struct = stack.back();
149 stack.pop_back();
150 ParseStructSet(wasm_struct, value);
151 continue;
152 }
153 case wasm::kExprLocalGet:
154 stack.push_back(ParseLocalGet());
155 continue;
156 case wasm::kExprDrop:
157 DCHECK(!stack.empty());
158 stack.pop_back();
159 continue;
160 case wasm::kExprEnd: {
161 DCHECK_LT(stack.size(), 2);
162 int return_count = static_cast<int>(stack.size());
163 base::SmallVector<Node*, 8> buf(return_count + 3);
164 buf[0] = mcgraph_->Int32Constant(0);
165 if (return_count) {
166 buf[1] = stack.back().node;
167 }
168 buf[return_count + 1] = gasm_.effect();
169 buf[return_count + 2] = gasm_.control();
170 Node* ret = graph_->NewNode(mcgraph_->common()->Return(return_count),
171 return_count + 3, buf.data());
172
173 gasm_.MergeControlToEnd(ret);
174 return true;
175 }
176 default:
177 // Instruction not supported for inlining.
178 return false;
179 }
180 }
181 // The decoder found an instruction it couldn't inline successfully.
182 return false;
183 }

186 Value ParseAnyConvertExtern(Value input) {
187 DCHECK(input.type.is_reference_to(wasm::HeapType::kExtern) ||
188 input.type.is_reference_to(wasm::HeapType::kNoExtern));
189 wasm::ValueType result_type = wasm::ValueType::RefMaybeNull(
190 wasm::HeapType::kAny, input.type.is_nullable()
191 ? wasm::Nullability::kNullable
192 : wasm::Nullability::kNonNullable);
193 Node* internalized = gasm_.WasmAnyConvertExtern(input.node);
194 return TypeNode(internalized, result_type);
195 }
196
197 Value ParseExternConvertAny(Value input) {
198 DCHECK(input.type.is_reference());
199 wasm::ValueType result_type = wasm::ValueType::RefMaybeNull(
200 wasm::HeapType::kExtern, input.type.is_nullable()
201 ? wasm::Nullability::kNullable
202 : wasm::Nullability::kNonNullable);
203 Node* internalized = gasm_.WasmExternConvertAny(input.node);
204 return TypeNode(internalized, result_type);
205 }

207 Value ParseLocalGet() {
208 uint32_t index = consume_u32v();
209 DCHECK_LT(index, body_.sig->parameter_count());
210 return TypeNode(Param(index + 1), body_.sig->GetParam(index));
211 }

213 Value ParseStructGet(Value struct_val, WasmOpcode opcode) {
214 wasm::ModuleTypeIndex struct_index{consume_u32v()};
215 DCHECK(module_->has_struct(struct_index));
216 const wasm::StructType* struct_type = module_->struct_type(struct_index);
217 uint32_t field_index = consume_u32v();
218 DCHECK_GT(struct_type->field_count(), field_index);
219 const bool is_signed = opcode == wasm::kExprStructGetS;
220 const CheckForNull null_check =
221 struct_val.type.is_nullable() ? kWithNullCheck : kWithoutNullCheck;
222 Node* member = gasm_.StructGet(struct_val.node, struct_type, field_index,
223 is_signed, null_check);
224 SetSourcePosition(member);
225 return TypeNode(member, struct_type->field(field_index).Unpacked());
226 }
227
228 void ParseStructSet(Value wasm_struct, Value value) {
229 wasm::ModuleTypeIndex struct_index{consume_u32v()};
230 DCHECK(module_->has_struct(struct_index));
231 const wasm::StructType* struct_type = module_->struct_type(struct_index);
232 uint32_t field_index = consume_u32v();
233 DCHECK_GT(struct_type->field_count(), field_index);
234 const CheckForNull null_check =
235 wasm_struct.type.is_nullable() ? kWithNullCheck : kWithoutNullCheck;
236 gasm_.StructSet(wasm_struct.node, value.node, struct_type, field_index,
237 null_check);
238 SetSourcePosition(gasm_.effect());
239 }

241 Value ParseRefCast(Value input, bool null_succeeds) {
242 auto [heap_index, length] = read_i33v<ValidationTag>(pc_);
243 pc_ += length;
244 if (heap_index < 0) {
245 if ((heap_index & 0x7f) != wasm::kArrayRefCode) {
246 // Abstract casts for non array type are not supported.
247 is_inlineable_ = false;
248 return {};
249 }
250 auto done = gasm_.MakeLabel();
251 // Abstract cast to array.
252 if (input.type.is_nullable() && null_succeeds) {
253 gasm_.GotoIf(gasm_.IsNull(input.node, input.type), &done);
254 }
255 gasm_.TrapIf(gasm_.IsSmi(input.node), TrapId::kTrapIllegalCast);
256 gasm_.TrapUnless(gasm_.HasInstanceType(input.node, WASM_ARRAY_TYPE),
257 TrapId::kTrapIllegalCast);
258 SetSourcePosition(gasm_.effect());
259 gasm_.Goto(&done);
260 gasm_.Bind(&done);
261 // Add TypeGuard for graph typing.
262 Graph* graph = mcgraph_->graph();
263 wasm::ValueType result_type = wasm::ValueType::RefMaybeNull(
264 wasm::HeapType::kArray,
265 null_succeeds ? wasm::kNullable : wasm::kNonNullable);
266 Node* type_guard =
267 graph->NewNode(mcgraph_->common()->TypeGuard(
268 Type::Wasm(result_type, module_, graph->zone())),
269 input.node, gasm_.effect(), gasm_.control());
270 gasm_.InitializeEffectControl(type_guard, gasm_.control());
271 return TypeNode(type_guard, result_type);
272 }
273 if (module_->has_signature(
274 wasm::ModuleTypeIndex{static_cast<uint32_t>(heap_index)})) {
275 is_inlineable_ = false;
276 return {};
277 }
278 wasm::ValueType target_type = wasm::ValueType::RefMaybeNull(
279 wasm::ModuleTypeIndex{static_cast<uint32_t>(heap_index)},
280 null_succeeds ? wasm::kNullable : wasm::kNonNullable);
281 Node* rtt = mcgraph_->graph()->NewNode(
282 gasm_.simplified()->RttCanon(target_type.ref_index()),
283 trusted_data_node_);
284 TypeNode(rtt, wasm::ValueType::Rtt(target_type.ref_index()));
285 Node* cast = gasm_.WasmTypeCast(input.node, rtt, {input.type, target_type});
286 SetSourcePosition(cast);
287 return TypeNode(cast, target_type);
288 }

290 Value ParseArrayLen(Value input) {
291 DCHECK(wasm::IsHeapSubtypeOf(input.type.heap_type(),
292 wasm::HeapType(wasm::HeapType::kArray),
293 module_));
294 const CheckForNull null_check =
295 input.type.is_nullable() ? kWithNullCheck : kWithoutNullCheck;
296 Node* len = gasm_.ArrayLength(input.node, null_check);
297 SetSourcePosition(len);
298 return TypeNode(len, wasm::kWasmI32);
299 }

301 Value ParseArrayGet(Value array, Value index, WasmOpcode opcode) {
302 wasm::ModuleTypeIndex array_index{consume_u32v()};
303 DCHECK(module_->has_array(array_index));
304 const wasm::ArrayType* array_type = module_->array_type(array_index);
305 const bool is_signed = opcode == WasmOpcode::kExprArrayGetS;
306 const CheckForNull null_check =
307 array.type.is_nullable() ? kWithNullCheck : kWithoutNullCheck;
308 // Perform bounds check.
309 Node* length = gasm_.ArrayLength(array.node, null_check);
310 SetSourcePosition(length);
311 gasm_.TrapUnless(gasm_.Uint32LessThan(index.node, length),
312 TrapId::kTrapArrayOutOfBounds);
313 SetSourcePosition(gasm_.effect());
314 // Perform array.get.
315 Node* element =
316 gasm_.ArrayGet(array.node, index.node, array_type, is_signed);
317 return TypeNode(element, array_type->element_type().Unpacked());
318 }
319
320 void ParseArraySet(Value array, Value index, Value value) {
321 wasm::ModuleTypeIndex array_index{consume_u32v()};
322 DCHECK(module_->has_array(array_index));
323 const wasm::ArrayType* array_type = module_->array_type(array_index);
324 const CheckForNull null_check =
325 array.type.is_nullable() ? kWithNullCheck : kWithoutNullCheck;
326 // Perform bounds check.
327 Node* length = gasm_.ArrayLength(array.node, null_check);
328 SetSourcePosition(length);
329 gasm_.TrapUnless(gasm_.Uint32LessThan(index.node, length),
330 TrapId::kTrapArrayOutOfBounds);
331 SetSourcePosition(gasm_.effect());
332 // Perform array.set.
333 gasm_.ArraySet(array.node, index.node, value.node, array_type);
334 }

336 WasmOpcode ReadOpcode() {
337 DCHECK_LT(pc_, end_);
338 instruction_start_ = pc();
339 WasmOpcode opcode = static_cast<WasmOpcode>(*pc_);
340 if (!WasmOpcodes::IsPrefixOpcode(opcode)) {
341 ++pc_;
342 return opcode;
343 }
344 auto [opcode_with_prefix, length] =
345 read_prefixed_opcode<ValidationTag>(pc_);
346 pc_ += length;
347 return opcode_with_prefix;
348 }

350 Value TypeNode(Node* node, wasm::ValueType type) {
351 compiler::NodeProperties::SetType(
352 node, compiler::Type::Wasm(type, module_, graph_->zone()));
353 return {node, type};
354 }
355
356 void SetSourcePosition(Node* node) {
357 if (!source_position_table_->IsEnabled()) return;
358 int offset = static_cast<int>(instruction_start_ - start());
359 source_position_table_->SetSourcePosition(
360 node, SourcePosition(offset, inlining_id_));
361 }

363 const wasm::WasmModule* module_;
364 MachineGraph* mcgraph_;
365 const wasm::FunctionBody& body_;
366 Node** parameters_;
367 Graph* graph_;
368 Node* trusted_data_node_;
369 WasmGraphAssembler gasm_;
370 SourcePositionTable* source_position_table_ = nullptr;
371 const uint8_t* instruction_start_ = pc_;
372 int inlining_id_;
373 bool is_inlineable_ = true;
