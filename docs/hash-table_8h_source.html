25// HashTable is a subclass of FixedArray that implements a hash table
26// that uses open addressing and quadratic probing.
27//
28// In order for the quadratic probing to work, elements that have not
29// yet been used and elements that have been deleted are
30// distinguished. Probing continues when deleted elements are
31// encountered and stops when unused elements are encountered.
32//
33// - Elements with key == undefined have not been used yet.
34// - Elements with key == the_hole have been deleted.
35//
36// The hash table class is parameterized with a Shape.
37// Shape must be a class with the following interface:
38// class ExampleShape {
39// public:
40// // Tells whether key matches other.
41// static bool IsMatch(Key key, Tagged<Object> other);
42// // Returns the hash value for key.
43// static uint32_t Hash(ReadOnlyRoots roots, Key key);
44// // Returns the hash value for object.
45// static uint32_t HashForObject(ReadOnlyRoots roots,
46// Tagged<Object> object);
47// // Convert key to an object.
48// static inline DirectHandle<Object> AsHandle(Isolate* isolate, Key key);
49// // The prefix size indicates number of elements in the beginning
50// // of the backing storage.
51// static const int kPrefixSize = ..;
52// // The Element size indicates number of elements per entry.
53// static const int kEntrySize = ..;
54// // Indicates whether IsMatch can deal with other being the_hole (a
55// // deleted entry).
56// static const bool kMatchNeedsHoleCheck = ..;
57// };
58// The prefix size indicates an amount of memory in the
59// beginning of the backing storage that can be used for non-element
60// information by subclasses.
61
62template <typename KeyT>
63class V8_EXPORT_PRIVATE BaseShape {
64 public:
65 using Key = KeyT;
66 static Tagged<Object> Unwrap(Tagged<Object> key) { return key; }
67};

114 // Returns probe entry.
115 inline static InternalIndex FirstProbe(uint32_t hash, uint32_t size) {
116 if (size <= (1 << PropertyArray::HashField::kSize)) {
117 return InternalIndex(hash & (size - 1));
118 }
119 // The hash stored in JSReceiver (see PropertyArray::HashField) only has 20
120 // bits. This is suboptimal, if the hash table size is larger than 1 << 20.
121 // This distributes the probes more evenly even in that case.
122 return InternalIndex((hash * (size >> PropertyArray::HashField::kSize)) &
123 (size - 1));
124 }
125
126 inline static InternalIndex NextProbe(InternalIndex last, uint32_t number,
127 uint32_t size) {
128 return InternalIndex((last.as_uint32() + number) & (size - 1));
129 }
