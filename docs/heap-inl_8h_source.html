45template <typename T>
46Tagged<T> ForwardingAddress(Tagged<T> heap_obj) {
47 MapWord map_word = Cast<HeapObject>(heap_obj)->map_word(kRelaxedLoad);
48
49 if (map_word.IsForwardingAddress()) {
50 return Cast<T>(map_word.ToForwardingAddress(heap_obj));
51 } else if (Heap::InFromPage(heap_obj)) {
52 DCHECK(!v8_flags.minor_ms);
53 return Tagged<T>();
54 } else {
55 return heap_obj;
56 }
57}

99#define ROOT_ACCESSOR(type, name, CamelName) \
100 void Heap::set_##name(Tagged<type> value) { \
101 /* The deserializer makes use of the fact that these common roots are */ \
102 /* never in new space and never on a page that is being compacted. */ \
103 DCHECK_IMPLIES(deserialization_complete(), \
104 !RootsTable::IsImmortalImmovable(RootIndex::k##CamelName)); \
105 if constexpr (RootsTable::IsImmortalImmovable(RootIndex::k##CamelName)) { \

291bool Heap::IsPendingAllocationInternal(Tagged<HeapObject> object) {
292 DCHECK(deserialization_complete());
293
294 MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
295 if (chunk->InReadOnlySpace()) return false;
296
297 BaseSpace* base_space = chunk->Metadata()->owner();
298 Address addr = object.address();
299
300 switch (base_space->identity()) {
301 case NEW_SPACE: {
302 return allocator()->new_space_allocator()->IsPendingAllocation(addr);
303 }
304
305 case OLD_SPACE: {
306 return allocator()->old_space_allocator()->IsPendingAllocation(addr);
307 }
308
309 case CODE_SPACE: {
310 return allocator()->code_space_allocator()->IsPendingAllocation(addr);
311 }
312
313 case TRUSTED_SPACE: {
314 return allocator()->trusted_space_allocator()->IsPendingAllocation(addr);
315 }
316
317 case LO_SPACE:
318 case CODE_LO_SPACE:
319 case TRUSTED_LO_SPACE:
320 case NEW_LO_SPACE: {
321 LargeObjectSpace* large_space =
322 static_cast<LargeObjectSpace*>(base_space);
323 base::SharedMutexGuard<base::kShared> guard(
324 large_space->pending_allocation_mutex());
325 return addr == large_space->pending_object();
326 }
327
328 case SHARED_SPACE:
329 case SHARED_LO_SPACE:
330 case SHARED_TRUSTED_SPACE:
331 case SHARED_TRUSTED_LO_SPACE:
332 // TODO(v8:13267): Ensure that all shared space objects have a memory
333 // barrier after initialization.
334 return false;
335
336 case RO_SPACE:
337 UNREACHABLE();
338 }
339
340 UNREACHABLE();
341}

356void Heap::ExternalStringTable::AddString(Tagged<String> string) {
357 std::optional<base::SpinningMutexGuard> guard;
358
359 // With --shared-string-table client isolates may insert into the main
360 // isolate's table concurrently.
361 if (v8_flags.shared_string_table &&
362 heap_->isolate()->is_shared_space_isolate()) {
363 guard.emplace(&mutex_);
364 }
365
366 DCHECK(IsExternalString(string));
367 DCHECK(!Contains(string));
368
369 if (HeapLayout::InYoungGeneration(string)) {
370 young_strings_.push_back(string);
371 } else {
372 old_strings_.push_back(string);
373 }
374}

387int Heap::MaxNumberToStringCacheSize() const {
388 // Compute the size of the number string cache based on the max newspace size.
389 // The number string cache has a minimum size based on twice the initial cache
390 // size to ensure that it is bigger after being made 'full size'.
391 size_t number_string_cache_size = max_semi_space_size_ / 512;
392 number_string_cache_size =
393 std::max(static_cast<size_t>(kInitialNumberStringCacheSize * 2),
394 std::min(static_cast<size_t>(0x4000), number_string_cache_size));
395 // There is a string and a number per entry so the length is twice the number
396 // of entries.
397 return static_cast<int>(number_string_cache_size * 2);
398}
