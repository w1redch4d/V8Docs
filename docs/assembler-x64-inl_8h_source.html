152void Assembler::emit_vex_prefix(XMMRegister reg, XMMRegister vreg,
153 XMMRegister rm, VectorLength l, SIMDPrefix pp,
154 LeadingOpcode mm, VexW w) {
155 if (rm.high_bit() || mm != k0F || w != kW0) {
156 emit_vex3_byte0();
157 emit_vex3_byte1(reg, rm, mm);
158 emit_vex3_byte2(w, vreg, l, pp);
159 } else {
160 emit_vex2_byte0();
161 emit_vex2_byte1(reg, vreg, l, pp);
162 }
163}

265// The modes possibly affected by apply must be in kApplyMask.
266void WritableRelocInfo::apply(intptr_t delta) {
267 if (IsCodeTarget(rmode_) || IsNearBuiltinEntry(rmode_) ||
268 IsWasmStubCall(rmode_)) {
269 jit_allocation_.WriteUnalignedValue(
270 pc_, ReadUnalignedValue<int32_t>(pc_) - static_cast<int32_t>(delta));
271 } else if (IsInternalReference(rmode_)) {
272 // Absolute code pointer inside code object moves with the code object.
273 jit_allocation_.WriteUnalignedValue(
274 pc_, ReadUnalignedValue<Address>(pc_) + delta);
275 }
276}

302Tagged<HeapObject> RelocInfo::target_object(PtrComprCageBase cage_base) {
303 DCHECK(IsCodeTarget(rmode_) || IsEmbeddedObjectMode(rmode_));
304 if (IsCompressedEmbeddedObject(rmode_)) {
305 Tagged_t compressed = ReadUnalignedValue<Tagged_t>(pc_);
306 DCHECK(!HAS_SMI_TAG(compressed));
307 Tagged<Object> obj(
308 V8HeapCompressionScheme::DecompressTagged(cage_base, compressed));
309 return Cast<HeapObject>(obj);
310 }
311 DCHECK(IsFullEmbeddedObject(rmode_));
312 return Cast<HeapObject>(Tagged<Object>(ReadUnalignedValue<Address>(pc_)));
313}

371void WritableRelocInfo::set_target_object(Tagged<HeapObject> target,
372 ICacheFlushMode icache_flush_mode) {
373 DCHECK(IsCodeTarget(rmode_) || IsEmbeddedObjectMode(rmode_));
374 if (IsCompressedEmbeddedObject(rmode_)) {
375 DCHECK(COMPRESS_POINTERS_BOOL);
376 // We must not compress pointers to objects outside of the main pointer
377 // compression cage as we wouldn't be able to decompress them with the
378 // correct cage base.
379 DCHECK_IMPLIES(V8_ENABLE_SANDBOX_BOOL, !HeapLayout::InTrustedSpace(target));
380 DCHECK_IMPLIES(V8_EXTERNAL_CODE_SPACE_BOOL,
381 !HeapLayout::InCodeSpace(target));
382 Tagged_t tagged = V8HeapCompressionScheme::CompressObject(target.ptr());
383 jit_allocation_.WriteUnalignedValue(pc_, tagged);
384 } else {
385 DCHECK(IsFullEmbeddedObject(rmode_));
386 jit_allocation_.WriteUnalignedValue(pc_, target.ptr());
387 }
388 if (icache_flush_mode != SKIP_ICACHE_FLUSH) {
389 FlushInstructionCache(pc_, sizeof(Address));
390 }
391}
