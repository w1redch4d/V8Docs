34void Code::ClearEmbeddedObjects(Heap* heap) {
35 DisallowGarbageCollection no_gc;
36 Tagged<HeapObject> undefined = ReadOnlyRoots(heap).undefined_value();
37 Tagged<InstructionStream> istream = unchecked_instruction_stream();
38 int mode_mask = RelocInfo::EmbeddedObjectModeMask();
39 {
40 WritableJitAllocation jit_allocation = ThreadIsolation::LookupJitAllocation(
41 istream->address(), istream->Size(),
42 ThreadIsolation::JitAllocationType::kInstructionStream, true);
43 for (WritableRelocIterator it(jit_allocation, istream, constant_pool(),
44 mode_mask);
45 !it.done(); it.next()) {
46 DCHECK(RelocInfo::IsEmbeddedObjectMode(it.rinfo()->rmode()));
47 it.rinfo()->set_target_object(istream, undefined, SKIP_WRITE_BARRIER);
48 }
49 }
50 set_embedded_objects_cleared(true);
51}

57int Code::SourcePosition(int offset) const {
58 CHECK_NE(kind(), CodeKind::BASELINE);
59
60 // Subtract one because the current PC is one instruction after the call site.
61 offset--;
62
63 int position = 0;
64 if (!has_source_position_table()) return position;
65 for (SourcePositionTableIterator it(
66 source_position_table(),
67 SourcePositionTableIterator::kJavaScriptOnly,
68 SourcePositionTableIterator::kDontSkipFunctionEntry);
69 !it.done() && it.code_offset() <= offset; it.Advance()) {
70 position = it.source_position().ScriptOffset();
71 }
72 return position;
73}

105bool Code::IsIsolateIndependent(Isolate* isolate) {
106 static constexpr int kModeMask =
107 RelocInfo::AllRealModesMask() &
108 ~RelocInfo::ModeMask(RelocInfo::CONST_POOL) &
109 ~RelocInfo::ModeMask(RelocInfo::OFF_HEAP_TARGET) &
110 ~RelocInfo::ModeMask(RelocInfo::VENEER_POOL) &
111 ~RelocInfo::ModeMask(RelocInfo::WASM_CANONICAL_SIG_ID) &
112 ~RelocInfo::ModeMask(RelocInfo::WASM_CODE_POINTER_TABLE_ENTRY);
113 static_assert(kModeMask ==
114 (RelocInfo::ModeMask(RelocInfo::CODE_TARGET) |
115 RelocInfo::ModeMask(RelocInfo::RELATIVE_CODE_TARGET) |
116 RelocInfo::ModeMask(RelocInfo::COMPRESSED_EMBEDDED_OBJECT) |
117 RelocInfo::ModeMask(RelocInfo::FULL_EMBEDDED_OBJECT) |
118 RelocInfo::ModeMask(RelocInfo::EXTERNAL_REFERENCE) |
119 RelocInfo::ModeMask(RelocInfo::INTERNAL_REFERENCE) |
120 RelocInfo::ModeMask(RelocInfo::INTERNAL_REFERENCE_ENCODED) |
121 RelocInfo::ModeMask(RelocInfo::JS_DISPATCH_HANDLE) |
122 RelocInfo::ModeMask(RelocInfo::NEAR_BUILTIN_ENTRY) |
123 RelocInfo::ModeMask(RelocInfo::WASM_CALL) |
124 RelocInfo::ModeMask(RelocInfo::WASM_STUB_CALL)));
125
126#if defined(V8_TARGET_ARCH_PPC64) || defined(V8_TARGET_ARCH_MIPS64)
127 return RelocIterator(*this, kModeMask).done();
128#elif defined(V8_TARGET_ARCH_X64) || defined(V8_TARGET_ARCH_ARM64) || \
129 defined(V8_TARGET_ARCH_ARM) || defined(V8_TARGET_ARCH_S390X) || \
130 defined(V8_TARGET_ARCH_IA32) || defined(V8_TARGET_ARCH_RISCV64) || \
131 defined(V8_TARGET_ARCH_LOONG64) || defined(V8_TARGET_ARCH_RISCV32)
132 for (RelocIterator it(*this, kModeMask); !it.done(); it.next()) {
133 // On these platforms we emit relative builtin-to-builtin
134 // jumps for isolate independent builtins in the snapshot. They are later
135 // rewritten as pc-relative jumps to the off-heap instruction stream and are
136 // thus process-independent. See also: FinalizeEmbeddedCodeTargets.
137 if (RelocInfo::IsCodeTargetMode(it.rinfo()->rmode())) {
138 Address target_address = it.rinfo()->target_address();
139 if (OffHeapInstructionStream::PcIsOffHeap(isolate, target_address))
140 continue;
141
142 Tagged<Code> target = Code::FromTargetAddress(target_address);
143 if (Builtins::IsIsolateIndependentBuiltin(target)) {
144 continue;
145 }
146 }
147 return false;
148 }
149 return true;
150#else
151#error Unsupported architecture.
152#endif
153}

155bool Code::Inlines(Tagged<SharedFunctionInfo> sfi) {
156 // We can only check for inlining for optimized code.
157 DCHECK(is_optimized_code());
158 DisallowGarbageCollection no_gc;
159 Tagged<DeoptimizationData> const data =
160 Cast<DeoptimizationData>(deoptimization_data());
161 if (data->length() == 0) return false;
162 if (data->GetSharedFunctionInfo() == sfi) return true;
163 Tagged<DeoptimizationLiteralArray> const literals = data->LiteralArray();
164 int const inlined_count = data->InlinedFunctionCount().value();
165 for (int i = 0; i < inlined_count; ++i) {
166 if (Cast<SharedFunctionInfo>(literals->get(i)) == sfi) return true;
167 }
168 return false;
169}

196void Disassemble(const char* name, std::ostream& os, Isolate* isolate,
197 Tagged<Code> code, Address current_pc) {
198 CodeKind kind = code->kind();
199 os << "kind = " << CodeKindToString(kind) << "\n";
200 if (name == nullptr && code->is_builtin()) {
201 name = Builtins::name(code->builtin_id());
202 }
203 if ((name != nullptr) && (name[0] != '\0')) {
204 os << "name = " << name << "\n";
205 }
206 os << "compiler = "
207 << (code->is_turbofanned() ? "turbofan"
208 : code->is_maglevved() ? "maglev"
209 : kind == CodeKind::BASELINE ? "baseline"
210 : "unknown")
211 << "\n";
212 os << "address = " << reinterpret_cast<void*>(code.ptr()) << "\n\n";
213
214 {
215 int code_size = code->instruction_size();
216 os << "Instructions (size = " << code_size << ")\n";
217 DisassembleCodeRange(isolate, os, code, code->instruction_start(),
218 code_size, current_pc);
219
220 if (int pool_size = code->constant_pool_size()) {
221 DCHECK_EQ(pool_size & kPointerAlignmentMask, 0);
222 os << "\nConstant Pool (size = " << pool_size << ")\n";
223 base::Vector<char> buf = base::Vector<char>::New(50);
224 intptr_t* ptr = reinterpret_cast<intptr_t*>(code->constant_pool());
225 for (int i = 0; i < pool_size; i += kSystemPointerSize, ptr++) {
226 SNPrintF(buf, "%4d %08" V8PRIxPTR, i, *ptr);
227 os << static_cast<const void*>(ptr) << " " << buf.begin() << "\n";
228 }
229 }
230 }
231 os << "\n";
232
233 // TODO(cbruni): add support for baseline code.
234 if (code->has_source_position_table()) {
235 {
236 SourcePositionTableIterator it(
237 code->source_position_table(),
238 SourcePositionTableIterator::kJavaScriptOnly);
239 if (!it.done()) {
240 os << "Source positions:\n pc offset position\n";
241 for (; !it.done(); it.Advance()) {
242 os << std::setw(10) << std::hex << it.code_offset() << std::dec
243 << std::setw(10) << it.source_position().ScriptOffset()
244 << (it.is_statement() ? " statement" : "") << "\n";
245 }
246 os << "\n";
247 }
248 }
249
250 {
251 SourcePositionTableIterator it(
252 code->source_position_table(),
253 SourcePositionTableIterator::kExternalOnly);
254 if (!it.done()) {
255 os << "External Source positions:\n pc offset fileid line\n";
256 for (; !it.done(); it.Advance()) {
257 DCHECK(it.source_position().IsExternal());
258 os << std::setw(10) << std::hex << it.code_offset() << std::dec
259 << std::setw(10) << it.source_position().ExternalFileId()
260 << std::setw(10) << it.source_position().ExternalLine() << "\n";
261 }
262 os << "\n";
263 }
264 }
265 }
266
267 if (code->uses_deoptimization_data()) {
268 Tagged<DeoptimizationData> data =
269 Cast<DeoptimizationData>(code->deoptimization_data());
270 data->PrintDeoptimizationData(os);
271 }
272 os << "\n";
273
274 if (code->uses_safepoint_table()) {
275 if (code->is_maglevved()) {
276 MaglevSafepointTable table(isolate, current_pc, code);
277 table.Print(os);
278 } else {
279 SafepointTable table(isolate, current_pc, code);
280 table.Print(os);
281 }
282 os << "\n";
283 }
284
285 if (code->has_handler_table()) {
286 HandlerTable table(code);
287 os << "Handler Table (size = " << table.NumberOfReturnEntries() << ")\n";
288 if (CodeKindIsOptimizedJSFunction(kind)) {
289 table.HandlerTableReturnPrint(os);
290 }
291 os << "\n";
292 }
293
294 os << "RelocInfo (size = " << code->relocation_size() << ")\n";
295 if (code->has_instruction_stream()) {
296 for (RelocIterator it(code); !it.done(); it.next()) {
297 it.rinfo()->Print(isolate, os);
298 }
299 }
300 os << "\n";
301
302 if (code->has_unwinding_info()) {
303 os << "UnwindingInfo (size = " << code->unwinding_info_size() << ")\n";
304 EhFrameDisassembler eh_frame_disassembler(
305 reinterpret_cast<uint8_t*>(code->unwinding_info_start()),
306 reinterpret_cast<uint8_t*>(code->unwinding_info_end()));
307 eh_frame_disassembler.DisassembleToStream(os);
308 os << "\n";
309 }
310}
