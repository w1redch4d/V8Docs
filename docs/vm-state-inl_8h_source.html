17// VMState class implementation. A simple stack of VM states held by the logger
18// and partially threaded through the call stack. States are pushed by VMState
19// construction and popped by destruction.
20inline const char* StateToString(StateTag state) {
21 switch (state) {
22 case JS:
23 return "JS";
24 case GC:
25 return "GC";
26 case PARSER:
27 return "PARSER";
28 case BYTECODE_COMPILER:

33 return "OTHER";

35 return "EXTERNAL";
36 case ATOMICS_WAIT:
37 return "ATOMICS_WAIT";

56ExternalCallbackScope::ExternalCallbackScope(
57 Isolate* isolate, Address callback, v8::ExceptionContext exception_context,
58 const void* callback_info)
59 : callback_(callback),
60 callback_info_(callback_info),
61 previous_scope_(isolate->external_callback_scope()),
62 vm_state_(isolate),
63 exception_context_(exception_context),
64 pause_timed_histogram_scope_(isolate->counters()->execute()) {
65#if USE_SIMULATOR || V8_USE_ADDRESS_SANITIZER || V8_USE_SAFE_STACK
66 js_stack_comparable_address_ =
67 i::SimulatorStack::RegisterJSStackComparableAddress(isolate);
68#endif
69 vm_state_.isolate_->set_external_callback_scope(this);
70#ifdef V8_RUNTIME_CALL_STATS
71 TRACE_EVENT_BEGIN0(TRACE_DISABLED_BY_DEFAULT("v8.runtime"),
72 "V8.ExternalCallback");
73#endif
74 // The external callback might be called via different code paths and on some
75 // of them it's not guaranteed that the topmost_script_having_context value
76 // is still valid (in particular, when the callback call is initiated by
77 // embedder via V8 Api). So, clear it to ensure correctness of
78 // Isolate::GetIncumbentContext().
79 vm_state_.isolate_->clear_topmost_script_having_context();
80}
81
82ExternalCallbackScope::~ExternalCallbackScope() {
83 vm_state_.isolate_->set_external_callback_scope(previous_scope_);
84 // JS code might have been executed by the callback and it could have changed
85 // {topmost_script_having_context}, clear it to ensure correctness of
86 // Isolate::GetIncumbentContext() in case it'll be called after returning
87 // from the callback.
88 vm_state_.isolate_->clear_topmost_script_having_context();
89#ifdef V8_RUNTIME_CALL_STATS
90 TRACE_EVENT_END0(TRACE_DISABLED_BY_DEFAULT("v8.runtime"),
91 "V8.ExternalCallback");
92#endif
93#if USE_SIMULATOR || V8_USE_ADDRESS_SANITIZER || V8_USE_SAFE_STACK
94 i::SimulatorStack::UnregisterJSStackComparableAddress(vm_state_.isolate_);
95#endif
96}
