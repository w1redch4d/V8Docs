29 // The buffer passed into OnBytesReceived is owned by the caller.
30 void OnBytesReceived(base::Vector<const uint8_t> bytes) override {
31 buffer_.emplace_back(bytes.size());
32 CHECK_EQ(buffer_.back().size(), bytes.size());
33 std::memcpy(buffer_.back().data(), bytes.data(), bytes.size());
34 buffer_size_ += bytes.size();
35 }
36
37 void Finish(bool can_use_compiled_module) override {
38 // We copy all received chunks into one byte buffer.
39 auto bytes = base::OwnedVector<uint8_t>::NewForOverwrite(buffer_size_);
40 uint8_t* destination = bytes.begin();
41 for (auto& chunk : buffer_) {
42 std::memcpy(destination, chunk.data(), chunk.size());
43 destination += chunk.size();
44 }
45 CHECK_EQ(destination - bytes.begin(), buffer_size_);
46
47 // Check if we can deserialize the module from cache.
48 if (can_use_compiled_module && deserializing()) {
49 HandleScope scope(isolate_);
50 SaveAndSwitchContext saved_context(isolate_, *context_);
51
52 MaybeHandle<WasmModuleObject> module_object = DeserializeNativeModule(
53 isolate_, compiled_module_bytes_, bytes.as_vector(), compile_imports_,
54 base::VectorOf(url()));
55
56 if (!module_object.is_null()) {
57 Handle<WasmModuleObject> module = module_object.ToHandleChecked();
58 resolver_->OnCompilationSucceeded(module);
59 return;
60 }
61 }
62
63 // Compile the received bytes synchronously.
64 ErrorThrower thrower(isolate_, api_method_name_for_errors_);
65 MaybeHandle<WasmModuleObject> module_object = GetWasmEngine()->SyncCompile(
66 isolate_, enabled_, std::move(compile_imports_), &thrower,
67 std::move(bytes));
68 if (thrower.error()) {
69 resolver_->OnCompilationFailed(thrower.Reify());
70 return;
71 }
72 Handle<WasmModuleObject> module = module_object.ToHandleChecked();
73 resolver_->OnCompilationSucceeded(module);
74 }
75
76 void Abort() override {
77 // Abort is fully handled by the API, we only clear the buffer.
78 buffer_.clear();
79 }
80
81 void NotifyCompilationDiscarded() override { buffer_.clear(); }
82
83 void NotifyNativeModuleCreated(
84 const std::shared_ptr<NativeModule>&) override {
85 // This function is only called from the {AsyncCompileJob}.
86 UNREACHABLE();
87 }
88
89 private:
90 Isolate* isolate_;
91 const WasmEnabledFeatures enabled_;
92 CompileTimeImports compile_imports_;
93 Handle<Context> context_;
94 const char* api_method_name_for_errors_;
95 std::shared_ptr<CompilationResultResolver> resolver_;
96
97 std::vector<std::vector<uint8_t>> buffer_;
98 size_t buffer_size_ = 0;
99};
