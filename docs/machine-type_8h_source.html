19enum class MachineRepresentation : uint8_t {
20 kNone,
21 kBit,
22 // Integral representations must be consecutive, in order of increasing order.
23 kWord8,
24 kWord16,
25 kWord32,
26 kWord64,
27 // (uncompressed) MapWord
28 // kMapWord is the representation of a map word, i.e. a map in the header
29 // of a HeapObject.
30 // If V8_MAP_PACKING is disabled, a map word is just the map itself. Hence
31 // kMapWord is equivalent to kTaggedPointer -- in fact it will be
32 // translated to kTaggedPointer during memory lowering.
33 // If V8_MAP_PACKING is enabled, a map word is a Smi-like encoding of a map
34 // and some meta data. Memory lowering of kMapWord loads/stores
35 // produces low-level kTagged loads/stores plus the necessary
36 // decode/encode operations.
37 // In either case, the kMapWord representation is not used after memory
38 // lowering.
39 kMapWord,
40 kTaggedSigned, // (uncompressed) Smi
41 kTaggedPointer, // (uncompressed) HeapObject
42 kTagged, // (uncompressed) Object (Smi or HeapObject)
43 kCompressedPointer, // (compressed) HeapObject
44 kCompressed, // (compressed) Object (Smi or HeapObject)
45 kProtectedPointer, // (uncompressed) TrustedObject
46 kIndirectPointer, // (indirect) HeapObject
47 // A 64-bit pointer encoded in a way (e.g. as offset) that guarantees it will
48 // point into the sandbox.
49 kSandboxedPointer,
50 // FP and SIMD representations must be last, and in order of increasing size.
51 kFloat16,
52 kFloat32,
53 kFloat64,
54 kSimd128,
55 kSimd256,
56 kFirstFPRepresentation = kFloat16,
57 kLastRepresentation = kSimd256
58};

63 static_assert(static_cast<uint8_t>(MachineRepresentation::k##rep1) + 1 == \

276 static MachineType TypeForRepresentation(const MachineRepresentation& rep,
277 bool isSigned = true) {
278 switch (rep) {
279 case MachineRepresentation::kNone:
280 return MachineType::None();
281 case MachineRepresentation::kBit:
282 return MachineType::Bool();
283 case MachineRepresentation::kWord8:
284 return isSigned ? MachineType::Int8() : MachineType::Uint8();
285 case MachineRepresentation::kWord16:
286 return isSigned ? MachineType::Int16() : MachineType::Uint16();
287 case MachineRepresentation::kWord32:
288 return isSigned ? MachineType::Int32() : MachineType::Uint32();
289 case MachineRepresentation::kWord64:
290 return isSigned ? MachineType::Int64() : MachineType::Uint64();
291 case MachineRepresentation::kFloat16:
292 return MachineType::Float16();
293 case MachineRepresentation::kFloat32:
294 return MachineType::Float32();
295 case MachineRepresentation::kFloat64:
296 return MachineType::Float64();
297 case MachineRepresentation::kSimd128:
298 return MachineType::Simd128();
299 case MachineRepresentation::kSimd256:
300 return MachineType::Simd256();
301 case MachineRepresentation::kTagged:
302 return MachineType::AnyTagged();
303 case MachineRepresentation::kTaggedSigned:
304 return MachineType::TaggedSigned();
305 case MachineRepresentation::kTaggedPointer:
306 return MachineType::TaggedPointer();
307 case MachineRepresentation::kCompressed:
308 return MachineType::AnyCompressed();
309 case MachineRepresentation::kIndirectPointer:
310 return MachineType::IndirectPointer();
311 case MachineRepresentation::kCompressedPointer:
312 return MachineType::CompressedPointer();
313 case MachineRepresentation::kSandboxedPointer:
314 return MachineType::SandboxedPointer();
315 default:
316 UNREACHABLE();
317 }
318 }

320 static MachineType TypeForCType(const CTypeInfo& type) {
321 switch (type.GetType()) {
322 case CTypeInfo::Type::kVoid:
323 return MachineType::AnyTagged();
324 case CTypeInfo::Type::kBool:
325 return MachineType::Bool();
326 case CTypeInfo::Type::kUint8:
327 return MachineType::Uint8();
328 case CTypeInfo::Type::kInt32:
329 return MachineType::Int32();
330 case CTypeInfo::Type::kUint32:
331 return MachineType::Uint32();
332 case CTypeInfo::Type::kInt64:
333 return MachineType::Int64();
334 case CTypeInfo::Type::kAny:
335 static_assert(
336 sizeof(AnyCType) == kInt64Size,
337 "CTypeInfo::Type::kAny is assumed to be of size 64 bits.");
338 return MachineType::Int64();
339 case CTypeInfo::Type::kUint64:
340 return MachineType::Uint64();
341 case CTypeInfo::Type::kFloat32:
342 return MachineType::Float32();
343 case CTypeInfo::Type::kFloat64:
344 return MachineType::Float64();
345 case CTypeInfo::Type::kPointer:
346 return MachineType::Pointer();
347 case CTypeInfo::Type::kV8Value:
348 case CTypeInfo::Type::kSeqOneByteString:
349 case CTypeInfo::Type::kApiObject:
350 return MachineType::AnyTagged();
351 }

436// Gets the log2 of the element size in bytes of the machine type.
437V8_EXPORT_PRIVATE inline constexpr int ElementSizeLog2Of(
438 MachineRepresentation rep) {
439 switch (rep) {
440 case MachineRepresentation::kBit:
441 case MachineRepresentation::kWord8:
442 return 0;
443 case MachineRepresentation::kWord16:
444 case MachineRepresentation::kFloat16:
445 return 1;
446 case MachineRepresentation::kWord32:
447 case MachineRepresentation::kFloat32:
448 case MachineRepresentation::kIndirectPointer:
449 return 2;
450 case MachineRepresentation::kWord64:
451 case MachineRepresentation::kFloat64:
452 return 3;
453 case MachineRepresentation::kSimd128:
454 return 4;
455 case MachineRepresentation::kSimd256:
456 return 5;
457 case MachineRepresentation::kTaggedSigned:
458 case MachineRepresentation::kTaggedPointer:
459 case MachineRepresentation::kTagged:
460 case MachineRepresentation::kMapWord:
461 case MachineRepresentation::kCompressedPointer:
462 case MachineRepresentation::kCompressed:
463 case MachineRepresentation::kProtectedPointer:
464 return kTaggedSizeLog2;
465 case MachineRepresentation::kSandboxedPointer:
466 return kSystemPointerSizeLog2;
467 default:
468 UNREACHABLE();
469 }
470}
471
472constexpr int kMaximumReprSizeLog2 =
473 ElementSizeLog2Of(MachineRepresentation::kSimd128);
474constexpr int kMaximumReprSizeInBytes = 1 << kMaximumReprSizeLog2;
475
476static_assert(kMaximumReprSizeLog2 >=
477 ElementSizeLog2Of(MachineRepresentation::kTagged));
478static_assert(kMaximumReprSizeLog2 >=
479 ElementSizeLog2Of(MachineRepresentation::kWord64));
