39 // If we found a regular object we are done. In case of free space, we

51bool LiveObjectRange::iterator::AdvanceToNextMarkedObject() {
52 // The following block moves the iterator to the next cell from the current
53 // object. This means skipping all possibly set mark bits (in case of black
54 // allocation).
55 if (!current_object_.is_null()) {
56 // Compute an end address that is inclusive. This allows clearing the cell
57 // up and including the end address. This works for one word fillers as
58 // well as other objects.
59 Address next_object = current_object_.address() + current_size_;
60 current_object_ = HeapObject();
61 if (MemoryChunk::IsAligned(next_object)) {
62 return false;
63 }
64 // Area end may not be exactly aligned to kAlignment. We don't need to bail
65 // out for area_end() though as we are guaranteed to have a bit for the
66 // whole page.
67 DCHECK_LE(next_object, page_->area_end());
68 // Move to the corresponding cell of the end index.
69 const auto next_markbit_index = MarkingBitmap::AddressToIndex(next_object);
70 DCHECK_GE(MarkingBitmap::IndexToCell(next_markbit_index),
71 current_cell_index_);
72 current_cell_index_ = MarkingBitmap::IndexToCell(next_markbit_index);
73 DCHECK_LT(current_cell_index_, MarkingBitmap::kCellsCount);
74 // Mask out lower addresses in the cell.
75 const MarkBit::CellType mask =
76 MarkingBitmap::IndexInCellMask(next_markbit_index);
77 current_cell_ = cells_[current_cell_index_] & ~(mask - 1);
78 }
79 // The next block finds any marked object starting from the current cell.
80 const MemoryChunk* chunk = page_->Chunk();
81 while (true) {
82 if (current_cell_) {
83 const auto trailing_zeros = base::bits::CountTrailingZeros(current_cell_);
84 Address current_cell_base =
85 chunk->address() + MarkingBitmap::CellToBase(current_cell_index_);
86 Address object_address = current_cell_base + trailing_zeros * kTaggedSize;
87 // The object may be a filler which we want to skip.
88 current_object_ = HeapObject::FromAddress(object_address);
89 current_map_ = current_object_->map(cage_base_, kAcquireLoad);
90 DCHECK(MapWord::IsMapOrForwarded(current_map_));
91 current_size_ = ALIGN_TO_ALLOCATION_ALIGNMENT(
92 current_object_->SizeFromMap(current_map_));
93 CHECK(page_->ContainsLimit(object_address + current_size_));
94 return true;
95 }
96 if (++current_cell_index_ >= MarkingBitmap::kCellsCount) break;
97 current_cell_ = cells_[current_cell_index_];
98 }
99 return false;
100}
