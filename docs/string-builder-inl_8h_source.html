36// static
37inline void ReplacementStringBuilder::AddSubjectSlice(
38 FixedArrayBuilder* builder, int from, int to) {
39 DCHECK_GE(from, 0);
40 int length = to - from;
41 DCHECK_GT(length, 0);
42 if (StringBuilderSubstringLength::is_valid(length) &&
43 StringBuilderSubstringPosition::is_valid(from)) {
44 int encoded_slice = StringBuilderSubstringLength::encode(length) |
45 StringBuilderSubstringPosition::encode(from);
46 builder->Add(Smi::FromInt(encoded_slice));
47 } else {
48 // Otherwise encode as two smis.
49 builder->Add(Smi::FromInt(-length));
50 builder->Add(Smi::FromInt(from));
51 }
52}

60template <typename SrcChar, typename DestChar>
61void IncrementalStringBuilder::Append(SrcChar c) {
62 DCHECK_EQ(encoding_ == String::ONE_BYTE_ENCODING, sizeof(DestChar) == 1);
63 if (sizeof(DestChar) == 1) {
64 DCHECK_EQ(String::ONE_BYTE_ENCODING, encoding_);
65 Cast<SeqOneByteString>(*current_part_)
66 ->SeqOneByteStringSet(current_index_++, c);
67 } else {
68 DCHECK_EQ(String::TWO_BYTE_ENCODING, encoding_);
69 Cast<SeqTwoByteString>(*current_part_)
70 ->SeqTwoByteStringSet(current_index_++, c);
71 }
72 if (current_index_ == part_length_) Extend();
73 DCHECK(HasValidCurrentIndex());
74}

84template <int N>
85V8_INLINE void IncrementalStringBuilder::AppendCStringLiteral(
86 const char (&literal)[N]) {
87 // Note that the literal contains the zero char.
88 const int length = N - 1;
89 static_assert(length > 0);
90 if (length == 1) return AppendCharacter(literal[0]);
91 if (encoding_ == String::ONE_BYTE_ENCODING && CurrentPartCanFit(N)) {
92 const uint8_t* chars = reinterpret_cast<const uint8_t*>(literal);
93 Cast<SeqOneByteString>(*current_part_)
94 ->SeqOneByteStringSetChars(current_index_, chars, length);
95 current_index_ += length;
96 if (current_index_ == part_length_) Extend();
97 DCHECK(HasValidCurrentIndex());
98 return;
99 }
100 return AppendCString(literal);
101}

103template <typename SrcChar>
104V8_INLINE void IncrementalStringBuilder::AppendCString(const SrcChar* s) {
105 if (encoding_ == String::ONE_BYTE_ENCODING) {
106 while (*s != '\0') Append<SrcChar, uint8_t>(*s++);
107 } else {
108 while (*s != '\0') Append<SrcChar, base::uc16>(*s++);
109 }
110}

112V8_INLINE void IncrementalStringBuilder::AppendString(std::string_view str) {
113 uint32_t length = static_cast<uint32_t>(str.length());
114 if (encoding_ == String::ONE_BYTE_ENCODING && CurrentPartCanFit(length)) {
115 Cast<SeqOneByteString>(*current_part_)
116 ->SeqOneByteStringSetChars(current_index_,
117 reinterpret_cast<const uint8_t*>(str.data()),
118 length);
119 current_index_ += str.length();
120 if (current_index_ == part_length_) Extend();
121 DCHECK(HasValidCurrentIndex());
122 } else {
123 for (size_t i = 0; i < str.length(); i++) {
124 AppendCharacter(str[i]);
125 }
126 }
127}

135V8_INLINE int IncrementalStringBuilder::EscapedLengthIfCurrentPartFits(
136 int length) {
137 if (length > kMaxPartLength) return 0;
138 // The worst case length of an escaped character is 6. Shifting the remaining
139 // string length right by 3 is a more pessimistic estimate, but faster to
140 // calculate.
141 static_assert((kMaxPartLength << 3) <= String::kMaxLength);
142 // This shift will not overflow because length is already less than the
143 // maximum part length.
144 int worst_case_length = length << 3;
145 return CurrentPartCanFit(worst_case_length) ? worst_case_length : 0;
146}
