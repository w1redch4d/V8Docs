67bool SemiSpace::EnsureCapacity(size_t capacity) {
68 DCHECK_IMPLIES(!IsCommitted(), CommittedMemory() == 0);
69 const size_t quarantined_pages =
70 id_ == kToSpace ? quarantined_pages_count_ : 0;
71 const int num_pages =
72 static_cast<int>((capacity / PageMetadata::kPageSize) -
73 memory_chunk_list_.size() + quarantined_pages);
74 if (num_pages >= 0) {
75 for (int pages_added = 0; pages_added < num_pages; pages_added++) {
76 // Pages in the new spaces can be moved to the old space by the full
77 // collector. Therefore, they must be initialized with the same FreeList
78 // as old pages.
79 if (!AllocateFreshPage()) {
80 if (pages_added) RewindPages(pages_added);
81 DCHECK(!IsCommitted());
82 return false;
83 }
84 }
85 } else {
86 // Due to previously quarantined pages, we already have more pages then
87 // needed. Free the redundant pages.
88 RewindPages(-num_pages);
89 }
90 DCHECK_EQ(capacity + quarantined_pages * PageMetadata::kPageSize,
91 CommittedMemory());
92 DCHECK_EQ(
93 static_cast<int>(capacity / PageMetadata::kPageSize) + quarantined_pages,
94 memory_chunk_list_.size());
95 return true;
96}

98bool SemiSpace::Commit() {
99 if (!EnsureCapacity(target_capacity_)) {
100 return false;
101 }
102 Reset();
103 DCHECK_EQ(target_capacity_, CommittedMemory());
104 if (age_mark_ == kNullAddress) {
105 age_mark_ = first_page()->area_start();
106 }
107 DCHECK(IsCommitted());
108 return true;
109}

111void SemiSpace::Uncommit() {
112 DCHECK(IsCommitted());
113 DCHECK_EQ(CommittedMemory(),
114 memory_chunk_list_.size() * PageMetadata::kPageSize);
115 RewindPages(static_cast<int>(memory_chunk_list_.size()));
116 current_page_ = nullptr;
117 current_capacity_ = 0;
118 quarantined_pages_count_ = 0;
119 DCHECK_EQ(CommittedPhysicalMemory(), 0);
120 DCHECK_EQ(CommittedMemory(), 0);
121 DCHECK(!IsCommitted());
122}

144bool SemiSpace::AllocateFreshPage() {
145 PageMetadata* new_page = heap()->memory_allocator()->AllocatePage(
146 MemoryAllocator::AllocationMode::kUsePool, this, NOT_EXECUTABLE);
147 if (new_page == nullptr) {
148 return false;
149 }
150 memory_chunk_list_.PushBack(new_page);
151 CHECK(new_page->IsLivenessClear());
152 IncrementCommittedPhysicalMemory(new_page->CommittedPhysicalMemory());
153 AccountCommitted(PageMetadata::kPageSize);
154 heap()->CreateFillerObjectAt(new_page->area_start(),
155 static_cast<int>(new_page->area_size()));
156 return true;
157}

159void SemiSpace::RewindPages(int num_pages) {
160 DCHECK_GT(num_pages, 0);
161 DCHECK(last_page());
162 AccountUncommitted(num_pages * PageMetadata::kPageSize);
163 size_t uncommitted_physical_memory = 0;
164 while (num_pages > 0) {
165 MutablePageMetadata* last = last_page();
166 CHECK_NOT_NULL(last);
167 uncommitted_physical_memory += last->CommittedPhysicalMemory();
168 memory_chunk_list_.Remove(last);
169 heap()->memory_allocator()->Free(MemoryAllocator::FreeMode::kPool, last);
170 num_pages--;
171 }
172 DecrementCommittedPhysicalMemory(uncommitted_physical_memory);
173}

185void SemiSpace::FixPagesFlags() {
186 const auto to_space_flags =
187 MemoryChunk::YoungGenerationPageFlags(
188 heap()->incremental_marking()->marking_mode()) |
189 MemoryChunk::TO_PAGE;
190 for (PageMetadata* page : *this) {
191 MemoryChunk* chunk = page->Chunk();
192 page->set_owner(this);
193 if (id_ == kToSpace) {
194 chunk->SetFlagsNonExecutable(to_space_flags);
195 } else {
196 DCHECK_EQ(id_, kFromSpace);
197 // From space must preserve `NEW_SPACE_BELOW_AGE_MARK` which is used for
198 // deciding on whether to copy or promote an object.
199 chunk->SetFlagNonExecutable(MemoryChunk::FROM_PAGE);
200 chunk->ClearFlagNonExecutable(MemoryChunk::TO_PAGE);
201 }
202 DCHECK(chunk->InYoungGeneration());
203 }

214void SemiSpace::RemovePage(PageMetadata* page) {
215 if (current_page_ == page) {
216 if (page->prev_page()) {
217 current_page_ = page->prev_page();
218 }
219 }
220 memory_chunk_list_.Remove(page);
221 AccountUncommitted(PageMetadata::kPageSize);
222 DecrementCommittedPhysicalMemory(page->CommittedPhysicalMemory());
223 ForAll<ExternalBackingStoreType>(
224 [this, page](ExternalBackingStoreType type, int index) {
225 DecrementExternalBackingStoreBytes(
226 type, page->ExternalBackingStoreBytes(type));
227 });
228}

231 DCHECK_EQ(page->owner(), this);
232 memory_chunk_list_.Remove(page);
233 memory_chunk_list_.PushBack(page);
234 current_page_ = page;

237void SemiSpace::Swap(SemiSpace* from, SemiSpace* to) {
238 // We won't be swapping semispaces without data in them.
239 DCHECK(from->first_page());
240 DCHECK(to->first_page());
241 DCHECK_EQ(from->maximum_capacity_, to->maximum_capacity_);
242 DCHECK_EQ(from->minimum_capacity_, to->minimum_capacity_);
243 // We swap all properties but id_.
244 std::swap(from->target_capacity_, to->target_capacity_);
245 std::swap(from->age_mark_, to->age_mark_);
246 std::swap(from->memory_chunk_list_, to->memory_chunk_list_);
247 std::swap(from->current_page_, to->current_page_);
248 ForAll<ExternalBackingStoreType>(
249 [from, to](ExternalBackingStoreType type, int index) {
250 const size_t tmp = from->external_backing_store_bytes_[index].load(
251 std::memory_order_relaxed);
252 from->external_backing_store_bytes_[index].store(
253 to->external_backing_store_bytes_[index].load(
254 std::memory_order_relaxed),
255 std::memory_order_relaxed);
256 to->external_backing_store_bytes_[index].store(
257 tmp, std::memory_order_relaxed);
258 });
259 std::swap(from->committed_physical_memory_, to->committed_physical_memory_);
260 {
261 // Swap committed atomic counters.
262 size_t to_commited = to->committed_.load();
263 to->committed_.store(from->committed_.load());
264 from->committed_.store(to_commited);
265 }
266 std::swap(from->quarantined_pages_count_, to->quarantined_pages_count_);
267
268 // Swapping the `memory_cunk_list_` essentially swaps out the pages (actual
269 // payload) from to and from space.
270 to->FixPagesFlags();
271 from->FixPagesFlags();
272}

302void SemiSpace::set_age_mark(Address mark) {
303 age_mark_ = mark;
304 PageMetadata* age_mark_page = PageMetadata::FromAllocationAreaAddress(mark);
305 size_t below_age_mark_pages = 0;
306 DCHECK_EQ(age_mark_page->owner(), this);
307 // Mark all pages up to the one containing mark.
308 for (PageMetadata* p : *this) {
309 below_age_mark_pages++;
310 p->Chunk()->SetFlagNonExecutable(MemoryChunk::NEW_SPACE_BELOW_AGE_MARK);
311 if (p == age_mark_page) break;
312 }
313 DCHECK_LT(quarantined_pages_count_, below_age_mark_pages);
314 USE(below_age_mark_pages);

326#ifdef VERIFY_HEAP
327void SemiSpace::VerifyPageMetadata() const {
328 bool is_from_space = (id_ == kFromSpace);
329 size_t external_backing_store_bytes[static_cast<int>(
330 ExternalBackingStoreType::kNumValues)] = {0};
331
332 size_t actual_pages = 0;
333 size_t computed_committed_physical_memory = 0;
334
335 for (const PageMetadata* page : *this) {
336 const MemoryChunk* chunk = page->Chunk();
337 CHECK_EQ(page->owner(), this);
338 CHECK(chunk->InNewSpace());
339 CHECK(chunk->IsFlagSet(is_from_space ? MemoryChunk::FROM_PAGE
340 : MemoryChunk::TO_PAGE));
341 CHECK(!chunk->IsFlagSet(is_from_space ? MemoryChunk::TO_PAGE
342 : MemoryChunk::FROM_PAGE));
343 CHECK(chunk->IsFlagSet(MemoryChunk::POINTERS_TO_HERE_ARE_INTERESTING));
344 CHECK(!chunk->IsQuarantined());
345 if (!is_from_space) {
346 // The pointers-from-here-are-interesting flag isn't updated dynamically
347 // on from-space pages, so it might be out of sync with the marking state.
348 if (page->heap()->incremental_marking()->IsMarking()) {
349 CHECK(page->heap()->incremental_marking()->IsMajorMarking());
350 CHECK(
351 chunk->IsFlagSet(MemoryChunk::POINTERS_FROM_HERE_ARE_INTERESTING));
352 } else {
353 CHECK(
354 !chunk->IsFlagSet(MemoryChunk::POINTERS_FROM_HERE_ARE_INTERESTING));
355 }
356 CHECK_IMPLIES(actual_pages < quarantined_pages_count_,
357 chunk->IsFlagSet(MemoryChunk::NEW_SPACE_BELOW_AGE_MARK));
358 }
359 ForAll<ExternalBackingStoreType>(
360 [&external_backing_store_bytes, page](ExternalBackingStoreType type,
361 int index) {
362 external_backing_store_bytes[index] +=
363 page->ExternalBackingStoreBytes(type);
364 });
365
366 computed_committed_physical_memory += page->CommittedPhysicalMemory();
367
368 CHECK_IMPLIES(page->list_node().prev(),
369 page->list_node().prev()->list_node().next() == page);
370 actual_pages++;
371 }
372 CHECK_EQ(actual_pages * PageMetadata::kPageSize, CommittedMemory());
373 CHECK_EQ(computed_committed_physical_memory, CommittedPhysicalMemory());
374 ForAll<ExternalBackingStoreType>(
375 [this, external_backing_store_bytes](ExternalBackingStoreType type,
376 int index) {
377 CHECK_EQ(external_backing_store_bytes[index],
378 ExternalBackingStoreBytes(type));
379 });
380}
381#endif // VERIFY_HEAP

404void SemiSpace::MoveQuarantinedPage(MemoryChunk* chunk) {
405 // Quarantining pages happens at the end of scavenge, after the semi spaces
406 // have been swapped. Thus, the quarantined page originates from "from space"
407 // to is moved to "to space" to keep pinned objects as live.
408 DCHECK_EQ(kToSpace, id_);
409 DCHECK(chunk->IsQuarantined());
410 DCHECK(chunk->IsFromPage());
411 PageMetadata* page = PageMetadata::cast(chunk->Metadata());
412 SemiSpace& from_space = heap_->semi_space_new_space()->from_space();
413 DCHECK_EQ(&from_space, page->owner());
414 DCHECK_NE(&from_space, this);
415 from_space.RemovePage(page);
416 chunk->ClearFlagNonExecutable(MemoryChunk::IS_QUARANTINED);
417 chunk->ClearFlagNonExecutable(MemoryChunk::FROM_PAGE);
418 chunk->SetFlagNonExecutable(MemoryChunk::TO_PAGE);
419 page->set_owner(this);
420 AccountCommitted(PageMetadata::kPageSize);
421 IncrementCommittedPhysicalMemory(page->CommittedPhysicalMemory());
422 DCHECK(!memory_chunk_list_.Empty());
423 memory_chunk_list_.PushFront(page);
424 quarantined_pages_count_++;
425}

461void SemiSpaceNewSpace::Grow() {
462 heap()->safepoint()->AssertActive();
463 // Double the semispace size but only up to maximum capacity.
464 DCHECK(TotalCapacity() < MaximumCapacity());
465 size_t new_capacity = std::min(
466 MaximumCapacity(),
467 static_cast<size_t>(v8_flags.semi_space_growth_factor) * TotalCapacity());
468 if (to_space_.GrowTo(new_capacity)) {
469 // Only grow from space if we managed to grow to-space.
470 if (!from_space_.GrowTo(new_capacity)) {
471 // If we managed to grow to-space but couldn't grow from-space,
472 // attempt to shrink to-space.
473 to_space_.ShrinkTo(from_space_.target_capacity());
474 }
475 }
476 DCHECK_SEMISPACE_ALLOCATION_TOP(allocation_top(), to_space_);
477}

483void SemiSpaceNewSpace::Shrink() {
484 size_t new_capacity = std::max(InitialTotalCapacity(), 2 * Size());
485 size_t rounded_new_capacity =
486 ::RoundUp(new_capacity, PageMetadata::kPageSize);
487 if (rounded_new_capacity < TotalCapacity()) {
488 to_space_.ShrinkTo(rounded_new_capacity);
489 // Only shrink from-space if we managed to shrink to-space.
490 if (from_space_.IsCommitted()) from_space_.Reset();
491 from_space_.ShrinkTo(rounded_new_capacity);
492 }
493 DCHECK_SEMISPACE_ALLOCATION_TOP(allocation_top(), to_space_);
494 if (!from_space_.IsCommitted()) return;
495 from_space_.Uncommit();
496}

549bool SemiSpaceNewSpace::AddParkedAllocationBuffer(
550 int size_in_bytes, AllocationAlignment alignment) {
551 int parked_size = 0;
552 Address start = 0;
553 for (auto it = parked_allocation_buffers_.begin();
554 it != parked_allocation_buffers_.end();) {
555 parked_size = it->first;
556 start = it->second;
557 int filler_size = Heap::GetFillToAlign(start, alignment);
558 if (size_in_bytes + filler_size <= parked_size) {
559 parked_allocation_buffers_.erase(it);
560 PageMetadata* page = PageMetadata::FromAddress(start);
561 // We move a page with a parked allocation to the end of the pages list
562 // to maintain the invariant that the last page is the used one.
563 to_space_.MovePageToTheEnd(page);
564 SetAllocationTop(start);
565 return true;
566 } else {
567 it++;
568 }
569 }
570 return false;
571}

601// We do not use the SemiSpaceObjectIterator because verification doesn't assume
602// that it works (it depends on the invariants we are checking).
603void SemiSpaceNewSpace::VerifyObjects(Isolate* isolate,
604 SpaceVerificationVisitor* visitor) const {
605 size_t external_space_bytes[static_cast<int>(
606 ExternalBackingStoreType::kNumValues)] = {0};
607 PtrComprCageBase cage_base(isolate);
608 for (const PageMetadata* page = to_space_.first_page(); page;
609 page = page->next_page()) {
610 visitor->VerifyPage(page);
611
612 Address current_address = page->area_start();
613
614 while (!PageMetadata::IsAlignedToPageSize(current_address)) {
615 Tagged<HeapObject> object = HeapObject::FromAddress(current_address);
616
617 // The first word should be a map, and we expect all map pointers to
618 // be in map space or read-only space.
619 int size = object->Size(cage_base);
620
621 visitor->VerifyObject(object);
622
623 if (IsExternalString(object, cage_base)) {
624 Tagged<ExternalString> external_string = Cast<ExternalString>(object);
625 size_t string_size = external_string->ExternalPayloadSize();
626 external_space_bytes[static_cast<int>(
627 ExternalBackingStoreType::kExternalString)] += string_size;
628 }
629
630 current_address += ALIGN_TO_ALLOCATION_ALIGNMENT(size);
631 }
632
633 visitor->VerifyPageDone(page);
634 }
635
636 ForAll<ExternalBackingStoreType>(
637 [this, external_space_bytes](ExternalBackingStoreType type, int index) {
638 if (type == ExternalBackingStoreType::kArrayBuffer) {
639 return;
640 }
641 CHECK_EQ(external_space_bytes[index], ExternalBackingStoreBytes(type));
642 });
643
644 if (!v8_flags.concurrent_array_buffer_sweeping) {
645 size_t bytes = heap()->array_buffer_sweeper()->young().BytesSlow();
646 CHECK_EQ(bytes,
647 ExternalBackingStoreBytes(ExternalBackingStoreType::kArrayBuffer));
648 }
649}

714size_t SemiSpaceNewSpace::AllocatedSinceLastGC() const {
715 const Address age_mark = to_space_.age_mark();
716 DCHECK_NE(age_mark, kNullAddress);
717 DCHECK_NE(allocation_top(), kNullAddress);
718 PageMetadata* const age_mark_page =
719 PageMetadata::FromAllocationAreaAddress(age_mark);
720 PageMetadata* const last_page =
721 PageMetadata::FromAllocationAreaAddress(allocation_top());
722 PageMetadata* current_page = age_mark_page;
723 size_t allocated = 0;
724 if (current_page != last_page) {
725 DCHECK_EQ(current_page, age_mark_page);
726 DCHECK_GE(age_mark_page->area_end(), age_mark);
727 allocated += age_mark_page->area_end() - age_mark;
728 current_page = current_page->next_page();
729 } else {
730 DCHECK_GE(allocation_top(), age_mark);
731 return allocation_top() - age_mark;
732 }
733 while (current_page != last_page) {
734 DCHECK_NE(current_page, age_mark_page);
735 allocated += MemoryChunkLayout::AllocatableMemoryInDataPage();
736 current_page = current_page->next_page();
737 }
738 DCHECK_GE(allocation_top(), current_page->area_start());
739 allocated += allocation_top() - current_page->area_start();
740 DCHECK_LE(allocated, Size());
741 return allocated;
742}

744void SemiSpaceNewSpace::GarbageCollectionPrologue() {
745 ResetParkedAllocationBuffers();
746
747 // We need to commit from space here to be able to check for from/to space
748 // pages later on in the GC. We need to commit before sweeping starts to avoid
749 // empty pages being reused for commiting from space and thus ending up with
750 // remembered set entries that point to from space instead of freed memory.
751 if (!from_space_.Commit()) {
752 heap_->FatalProcessOutOfMemory("Committing semi space failed.");
753 }
754}

756void SemiSpaceNewSpace::EvacuatePrologue() {
757 // Flip the semispaces. After flipping, to space is empty and from space has
758 // live objects.
759 SemiSpace::Swap(&from_space_, &to_space_);
760 ResetCurrentSpace();
761 DCHECK_EQ(0u, Size());
762}

797std::optional<std::pair<Address, Address>> SemiSpaceNewSpace::Allocate(
798 int size_in_bytes, AllocationAlignment alignment) {
799 size_in_bytes = ALIGN_TO_ALLOCATION_ALIGNMENT(size_in_bytes);
800 DCHECK_SEMISPACE_ALLOCATION_TOP(allocation_top(), to_space_);
801
802 Address top = allocation_top();
803 Address high = to_space_.page_high();
804 int filler_size = Heap::GetFillToAlign(top, alignment);
805 int aligned_size_in_bytes = size_in_bytes + filler_size;
806
807 if (top + aligned_size_in_bytes <= high) {
808 IncrementAllocationTop(high);
809 return std::pair(top, high);
810 }
811
812 int remaining_in_page = static_cast<int>(high - top);
813 heap()->CreateFillerObjectAt(top, remaining_in_page);
814 SetAllocationTop(high);
815
816 // We park unused allocation buffer space of allocations happening from the
817 // mutator.
818 if (v8_flags.allocation_buffer_parking &&
819 heap()->gc_state() == Heap::NOT_IN_GC &&
820 remaining_in_page >= kAllocationBufferParkingThreshold) {
821 parked_allocation_buffers_.push_back(
822 ParkedAllocationBuffer(remaining_in_page, top));
823 }
824
825 if (AddFreshPage()) {
826 Address start = allocation_top();
827 Address end = to_space_.page_high();
828 DCHECK_EQ(0, Heap::GetFillToAlign(start, alignment));
829 IncrementAllocationTop(end);
830 return std::pair(start, end);
831 }
832
833 if (v8_flags.allocation_buffer_parking &&
834 AddParkedAllocationBuffer(size_in_bytes, alignment)) {
835 Address start = allocation_top();
836 Address end = to_space_.page_high();
837 DCHECK_LT(start, end);
838 IncrementAllocationTop(end);
839 return std::pair(start, end);
840 }
841
842 return std::nullopt;
843}
