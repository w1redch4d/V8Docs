54static int AddKey(Tagged<Object> key, DirectHandle<FixedArray> combined_keys,
55 DirectHandle<DescriptorArray> descs, int nof_descriptors,
56 int target) {

64static Handle<FixedArray> CombineKeys(Isolate* isolate,
65 Handle<FixedArray> own_keys,
66 Handle<FixedArray> prototype_chain_keys,
67 DirectHandle<JSReceiver> receiver,
68 bool may_have_elements) {
69 int prototype_chain_keys_length = prototype_chain_keys->length();
70 if (prototype_chain_keys_length == 0) return own_keys;
71
72 Tagged<Map> map = receiver->map();
73 int nof_descriptors = map->NumberOfOwnDescriptors();
74 if (nof_descriptors == 0 && !may_have_elements) return prototype_chain_keys;
75
76 DirectHandle<DescriptorArray> descs(map->instance_descriptors(isolate),
77 isolate);
78 int own_keys_length = own_keys.is_null() ? 0 : own_keys->length();
79 Handle<FixedArray> combined_keys = isolate->factory()->NewFixedArray(
80 own_keys_length + prototype_chain_keys_length);
81 if (own_keys_length != 0) {
82 FixedArray::CopyElements(isolate, *combined_keys, 0, *own_keys, 0,
83 own_keys_length);
84 }
85 int target_keys_length = own_keys_length;
86 for (int i = 0; i < prototype_chain_keys_length; i++) {
87 target_keys_length += AddKey(prototype_chain_keys->get(i), combined_keys,
88 descs, nof_descriptors, target_keys_length);
89 }
90 return FixedArray::RightTrimOrEmpty(isolate, combined_keys,
91 target_keys_length);
92}

134ExceptionStatus KeyAccumulator::AddKey(DirectHandle<Object> key,
135 AddKeyConversion convert) {
136 if (filter_ == PRIVATE_NAMES_ONLY) {
137 if (!IsSymbol(*key)) return ExceptionStatus::kSuccess;
138 if (!Cast<Symbol>(*key)->is_private_name())
139 return ExceptionStatus::kSuccess;
140 } else if (IsSymbol(*key)) {
141 if (filter_ & SKIP_SYMBOLS) return ExceptionStatus::kSuccess;
142 if (Cast<Symbol>(*key)->is_private()) return ExceptionStatus::kSuccess;
143 } else if (filter_ & SKIP_STRINGS) {
144 return ExceptionStatus::kSuccess;
145 }
146
147 if (IsShadowed(key)) return ExceptionStatus::kSuccess;
148 if (keys_.is_null()) {
149 keys_ = OrderedHashSet::Allocate(isolate_, 16).ToHandleChecked();
150 }
151 uint32_t index;
152 if (convert == CONVERT_TO_ARRAY_INDEX && IsString(*key) &&
153 Cast<String>(key)->AsArrayIndex(&index)) {
154 key = isolate_->factory()->NewNumberFromUint(index);
155 }
156 MaybeHandle<OrderedHashSet> new_set_candidate =
157 OrderedHashSet::Add(isolate(), keys(), key);
158 Handle<OrderedHashSet> new_set;
159 if (!new_set_candidate.ToHandle(&new_set)) {
160 CHECK(isolate_->has_exception());
161 return ExceptionStatus::kException;
162 }
163 if (*new_set != *keys_) {
164 // The keys_ Set is converted directly to a FixedArray in GetKeys which can
165 // be left-trimmer. Hence the previous Set should not keep a pointer to the
166 // new one.
167 keys_->set(OrderedHashSet::NextTableIndex(), Smi::zero());
168 keys_ = new_set;
169 }
170 return ExceptionStatus::kSuccess;
171}

190MaybeDirectHandle<FixedArray> FilterProxyKeys(KeyAccumulator* accumulator,
191 DirectHandle<JSProxy> owner,
192 DirectHandle<FixedArray> keys,
193 PropertyFilter filter,
194 bool skip_indices) {
195 if (filter == ALL_PROPERTIES) {
196 // Nothing to do.
197 return keys;
198 }
199 Isolate* isolate = accumulator->isolate();
200 int store_position = 0;
201 for (int i = 0; i < keys->length(); ++i) {
202 DirectHandle<Name> key(Cast<Name>(keys->get(i)), isolate);
203 if (Object::FilterKey(*key, filter)) continue; // Skip this key.
204 if (skip_indices) {
205 uint32_t index;
206 if (key->AsArrayIndex(&index)) continue; // Skip this key.
207 }
208 if (filter & ONLY_ENUMERABLE) {
209 PropertyDescriptor desc;
210 Maybe<bool> found =
211 JSProxy::GetOwnPropertyDescriptor(isolate, owner, key, &desc);
212 MAYBE_RETURN(found, MaybeDirectHandle<FixedArray>());
213 if (!found.FromJust()) continue;
214 if (!desc.enumerable()) {
215 accumulator->AddShadowingKey(key);
216 continue;
217 }
218 }
219 // Keep this key.
220 if (store_position != i) {
221 keys->set(store_position, *key);
222 }
223 store_position++;
224 }
225 return FixedArray::RightTrimOrEmpty(isolate, keys, store_position);
226}

228// Returns "nothing" in case of exception, "true" on success.
229Maybe<bool> KeyAccumulator::AddKeysFromJSProxy(DirectHandle<JSProxy> proxy,
230 DirectHandle<FixedArray> keys) {
231 // Postpone the enumerable check for for-in to the ForInFilter step.
232 if (!is_for_in_) {
233 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
234 isolate_, keys,
235 FilterProxyKeys(this, proxy, keys, filter_, skip_indices_),
236 Nothing<bool>());
237 }
238 // https://tc39.es/ecma262/#sec-proxy-object-internal-methods-and-internal-slots-ownpropertykeys
239 // As of 10.5.11.9 says, the keys collected from Proxy should not contain
240 // any duplicates. And the order of the keys is preserved by the
241 // OrderedHashTable.
242 RETURN_NOTHING_IF_NOT_SUCCESSFUL(AddKeys(keys, CONVERT_TO_ARRAY_INDEX));
243 return Just(true);
244}

246Maybe<bool> KeyAccumulator::CollectKeys(DirectHandle<JSReceiver> receiver,
247 DirectHandle<JSReceiver> object) {
248 // Proxies have no hidden prototype and we should not trigger the
249 // [[GetPrototypeOf]] trap on the last iteration when using
250 // AdvanceFollowingProxies.
251 if (mode_ == KeyCollectionMode::kOwnOnly && IsJSProxy(*object)) {
252 MAYBE_RETURN(CollectOwnJSProxyKeys(receiver, Cast<JSProxy>(object)),
253 Nothing<bool>());
254 return Just(true);
255 }
256
257 PrototypeIterator::WhereToEnd end = mode_ == KeyCollectionMode::kOwnOnly
258 ? PrototypeIterator::END_AT_NON_HIDDEN
259 : PrototypeIterator::END_AT_NULL;
260 for (PrototypeIterator iter(isolate_, object, kStartAtReceiver, end);
261 !iter.IsAtEnd();) {
262 // Start the shadow checks only after the first prototype has added
263 // shadowing keys.
264 if (HasShadowingKeys()) skip_shadow_check_ = false;
265 DirectHandle<JSReceiver> current =
266 PrototypeIterator::GetCurrent<JSReceiver>(iter);
267 Maybe<bool> result = Just(false); // Dummy initialization.
268 if (IsJSProxy(*current)) {
269 result = CollectOwnJSProxyKeys(receiver, Cast<JSProxy>(current));
270 } else if (IsWasmObject(*current)) {
271 if (mode_ == KeyCollectionMode::kIncludePrototypes) {
272 RETURN_FAILURE(isolate_, kThrowOnError,
273 NewTypeError(MessageTemplate::kWasmObjectsAreOpaque));
274 } else {
275 DCHECK_EQ(KeyCollectionMode::kOwnOnly, mode_);
276 DCHECK_EQ(result, Just(false)); // Stop iterating.
277 }
278 } else {
279 DCHECK(IsJSObject(*current));
280 result = CollectOwnKeys(receiver, Cast<JSObject>(current));
281 }
282 MAYBE_RETURN(result, Nothing<bool>());
283 if (!result.FromJust()) break; // |false| means "stop iterating".
284 // Iterate through proxies but ignore access checks case on API objects for
285 // OWN_ONLY keys handled in CollectOwnKeys.
286 if (!iter.AdvanceFollowingProxiesIgnoringAccessChecks()) {
287 return Nothing<bool>();
288 }
289 if (!last_non_empty_prototype_.is_null() &&
290 *last_non_empty_prototype_ == *current) {
291 break;
292 }
293 }
294 return Just(true);
295}

339void FastKeyAccumulator::Prepare() {
340 DisallowGarbageCollection no_gc;
341 // Directly go for the fast path for OWN_ONLY keys.
342 if (mode_ == KeyCollectionMode::kOwnOnly) return;
343 // Fully walk the prototype chain and find the last prototype with keys.
344 is_receiver_simple_enum_ = false;
345 has_empty_prototype_ = true;
346 only_own_has_simple_elements_ =
347 !IsCustomElementsReceiverMap(receiver_->map());
348 Tagged<JSReceiver> last_prototype;
349 may_have_elements_ = MayHaveElements(*receiver_);
350 for (PrototypeIterator iter(isolate_, *receiver_); !iter.IsAtEnd();
351 iter.Advance()) {
352 Tagged<JSReceiver> current = iter.GetCurrent<JSReceiver>();
353 if (!may_have_elements_ || only_own_has_simple_elements_) {
354 if (MayHaveElements(current)) {
355 may_have_elements_ = true;
356 only_own_has_simple_elements_ = false;
357 }
358 }
359 bool has_no_properties = CheckAndInitializeEmptyEnumCache(current);
360 if (has_no_properties) continue;
361 last_prototype = current;
362 has_empty_prototype_ = false;
363 }
364 // Check if we should try to create/use prototype info cache.
365 try_prototype_info_cache_ = TryPrototypeInfoCache(receiver_);
366 if (has_prototype_info_cache_) return;
367 if (has_empty_prototype_) {
368 is_receiver_simple_enum_ =
369 receiver_->map()->EnumLength() != kInvalidEnumCacheSentinel &&
370 !Cast<JSObject>(*receiver_)->HasEnumerableElements();
371 } else if (!last_prototype.is_null()) {
372 last_non_empty_prototype_ = direct_handle(last_prototype, isolate_);
373 }
374}

385// Initializes and directly returns the enum cache. Users of this function
386// have to make sure to never directly leak the enum cache.
387Handle<FixedArray> GetFastEnumPropertyKeys(Isolate* isolate,
388 DirectHandle<JSObject> object) {
389 DirectHandle<Map> map(object->map(), isolate);
390 Handle<FixedArray> keys(
391 map->instance_descriptors(isolate)->enum_cache()->keys(), isolate);
392
393 // Check if the {map} has a valid enum length, which implies that it
394 // must have a valid enum cache as well.
395 int enum_length = map->EnumLength();
396 if (enum_length != kInvalidEnumCacheSentinel) {
397 DCHECK(map->OnlyHasSimpleProperties());
398 DCHECK_LE(enum_length, keys->length());
399 DCHECK_EQ(enum_length, map->NumberOfEnumerableProperties());
400 isolate->counters()->enum_cache_hits()->Increment();
401 return ReduceFixedArrayTo(isolate, keys, enum_length);
402 }
403
404 // Determine the actual number of enumerable properties of the {map}.
405 enum_length = map->NumberOfEnumerableProperties();
406
407 // Check if there's already a shared enum cache on the {map}s
408 // DescriptorArray with sufficient number of entries.
409 if (enum_length <= keys->length()) {
410 if (map->OnlyHasSimpleProperties()) map->SetEnumLength(enum_length);
411 isolate->counters()->enum_cache_hits()->Increment();
412 return ReduceFixedArrayTo(isolate, keys, enum_length);
413 }
414
415 return FastKeyAccumulator::InitializeFastPropertyEnumCache(isolate, map,
416 enum_length);
417}

419template <bool fast_properties>
420MaybeHandle<FixedArray> GetOwnKeysWithElements(Isolate* isolate,
421 DirectHandle<JSObject> object,
422 GetKeysConversion convert,
423 bool skip_indices) {
424 Handle<FixedArray> keys;
425 if (fast_properties) {
426 keys = GetFastEnumPropertyKeys(isolate, object);
427 } else {
428 // TODO(cbruni): preallocate big enough array to also hold elements.
429 keys = KeyAccumulator::GetOwnEnumPropertyKeys(isolate, object);
430 }
431
432 MaybeHandle<FixedArray> result;
433 if (skip_indices) {
434 result = keys;
435 } else {
436 ElementsAccessor* accessor = object->GetElementsAccessor(isolate);
437 result = accessor->PrependElementIndices(isolate, object, keys, convert,
438 ONLY_ENUMERABLE);
439 }
440
441 if (v8_flags.trace_for_in_enumerate) {
442 PrintF("| strings=%d symbols=0 elements=%u || prototypes>=1 ||\n",
443 keys->length(), result.ToHandleChecked()->length() - keys->length());
444 }
445 return result;
446}

471MaybeHandle<FixedArray> FastKeyAccumulator::GetKeysFast(
472 GetKeysConversion keys_conversion) {
473 bool own_only = has_empty_prototype_ || mode_ == KeyCollectionMode::kOwnOnly;
474 Tagged<Map> map = receiver_->map();
475 if (!own_only || IsCustomElementsReceiverMap(map)) {
476 return MaybeHandle<FixedArray>();
477 }
478
479 // From this point on we are certain to only collect own keys.
480 DCHECK(IsJSObject(*receiver_));
481 DirectHandle<JSObject> object = Cast<JSObject>(receiver_);
482
483 // Do not try to use the enum-cache for dict-mode objects.
484 if (map->is_dictionary_map()) {
485 return GetOwnKeysWithElements<false>(isolate_, object, keys_conversion,
486 skip_indices_);
487 }
488 int enum_length = receiver_->map()->EnumLength();
489 if (enum_length == kInvalidEnumCacheSentinel) {
490 Handle<FixedArray> keys;
491 // Try initializing the enum cache and return own properties.
492 if (GetOwnKeysWithUninitializedEnumLength().ToHandle(&keys)) {
493 if (v8_flags.trace_for_in_enumerate) {
494 PrintF("| strings=%d symbols=0 elements=0 || prototypes>=1 ||\n",
495 keys->length());
496 }
497 is_receiver_simple_enum_ =
498 object->map()->EnumLength() != kInvalidEnumCacheSentinel;
499 return keys;
500 }
501 }
502 // The properties-only case failed because there were probably elements on the
503 // receiver.
504 return GetOwnKeysWithElements<true>(isolate_, object, keys_conversion,
505 skip_indices_);
506}

508// static
509Handle<FixedArray> FastKeyAccumulator::InitializeFastPropertyEnumCache(
510 Isolate* isolate, DirectHandle<Map> map, int enum_length,
511 AllocationType allocation) {
512 DCHECK_EQ(kInvalidEnumCacheSentinel, map->EnumLength());
513 DCHECK_GT(enum_length, 0);
514 DCHECK_EQ(enum_length, map->NumberOfEnumerableProperties());
515 DCHECK(!map->is_dictionary_map());
516
517 DirectHandle<DescriptorArray> descriptors(map->instance_descriptors(isolate),
518 isolate);
519
520 // The enum cache should have been a hit if the number of enumerable
521 // properties is fewer than what's already in the cache.
522 DCHECK_LT(descriptors->enum_cache()->keys()->length(), enum_length);
523 isolate->counters()->enum_cache_misses()->Increment();
524
525 // Create the keys array.
526 int index = 0;
527 bool fields_only = true;
528 Handle<FixedArray> keys =
529 isolate->factory()->NewFixedArray(enum_length, allocation);
530 for (InternalIndex i : map->IterateOwnDescriptors()) {
531 DisallowGarbageCollection no_gc;
532 PropertyDetails details = descriptors->GetDetails(i);
533 if (details.IsDontEnum()) continue;
534 Tagged<Object> key = descriptors->GetKey(i);
535 if (IsSymbol(key)) continue;
536 keys->set(index, key);
537 if (details.location() != PropertyLocation::kField) fields_only = false;
538 index++;
539 }
540 DCHECK_EQ(index, keys->length());
541
542 // Optionally also create the indices array.
543 DirectHandle<FixedArray> indices = isolate->factory()->empty_fixed_array();
544 if (fields_only) {
545 indices = isolate->factory()->NewFixedArray(enum_length, allocation);
546 index = 0;
547 DisallowGarbageCollection no_gc;
548 Tagged<Map> raw_map = *map;
549 Tagged<FixedArray> raw_indices = *indices;
550 Tagged<DescriptorArray> raw_descriptors = *descriptors;
551 for (InternalIndex i : raw_map->IterateOwnDescriptors()) {
552 PropertyDetails details = raw_descriptors->GetDetails(i);
553 if (details.IsDontEnum()) continue;
554 Tagged<Object> key = raw_descriptors->GetKey(i);
555 if (IsSymbol(key)) continue;
556 DCHECK_EQ(PropertyKind::kData, details.kind());
557 DCHECK_EQ(PropertyLocation::kField, details.location());
558 FieldIndex field_index = FieldIndex::ForDetails(raw_map, details);
559 raw_indices->set(index, Smi::FromInt(field_index.GetLoadByFieldIndex()));
560 index++;
561 }
562 DCHECK_EQ(index, indices->length());
563 }
564
565 DescriptorArray::InitializeOrChangeEnumCache(descriptors, isolate, keys,
566 indices, allocation);
567 if (map->OnlyHasSimpleProperties()) map->SetEnumLength(enum_length);
568 return keys;
569}

610MaybeHandle<FixedArray> FastKeyAccumulator::GetKeysWithPrototypeInfoCache(
611 GetKeysConversion keys_conversion) {
612 Handle<FixedArray> own_keys;
613 if (may_have_elements_) {
614 MaybeHandle<FixedArray> maybe_own_keys;
615 if (receiver_->map()->is_dictionary_map()) {
616 maybe_own_keys = GetOwnKeysWithElements<false>(
617 isolate_, Cast<JSObject>(receiver_), keys_conversion, skip_indices_);

622 ASSIGN_RETURN_ON_EXCEPTION(isolate_, own_keys, maybe_own_keys);
623 } else {
624 own_keys = KeyAccumulator::GetOwnEnumPropertyKeys(
625 isolate_, Cast<JSObject>(receiver_));
626 }
627 Handle<FixedArray> prototype_chain_keys;
628 if (has_prototype_info_cache_) {
629 prototype_chain_keys =
630 handle(Cast<FixedArray>(
631 Cast<PrototypeInfo>(first_prototype_map_->prototype_info())
632 ->prototype_chain_enum_cache()),
633 isolate_);
634 } else {
635 KeyAccumulator accumulator(isolate_, mode_, filter_);
636 accumulator.set_is_for_in(is_for_in_);
637 accumulator.set_skip_indices(skip_indices_);
638 accumulator.set_last_non_empty_prototype(last_non_empty_prototype_);
639 accumulator.set_may_have_elements(may_have_elements_);
640 accumulator.set_receiver(receiver_);
641 accumulator.set_first_prototype_map(first_prototype_map_);
642 accumulator.set_try_prototype_info_cache(try_prototype_info_cache_);
643 MAYBE_RETURN(accumulator.CollectKeys(first_prototype_, first_prototype_),
644 MaybeHandle<FixedArray>());
645 prototype_chain_keys = accumulator.GetKeys(keys_conversion);
646 }
647 Handle<FixedArray> result = CombineKeys(
648 isolate_, own_keys, prototype_chain_keys, receiver_, may_have_elements_);
649 if (is_for_in_ && own_keys.is_identical_to(result)) {
650 // Don't leak the enumeration cache without the receiver since it might get
651 // trimmed otherwise.
652 return isolate_->factory()->CopyFixedArrayUpTo(result, result->length());
653 }
654 return result;
655}

665bool FastKeyAccumulator::TryPrototypeInfoCache(
666 DirectHandle<JSReceiver> receiver) {
667 if (may_have_elements_ && !only_own_has_simple_elements_) return false;
668 DirectHandle<JSObject> object = Cast<JSObject>(receiver);
669 if (!object->HasFastProperties()) return false;

684 first_prototype_map_ = direct_handle(maybe_proto_map, isolate_);
685 has_prototype_info_cache_ =
686 maybe_proto_map->IsPrototypeValidityCellValid() &&
687 IsFixedArray(prototype_info->prototype_chain_enum_cache());
688 return true;

691V8_WARN_UNUSED_RESULT ExceptionStatus
692KeyAccumulator::FilterForEnumerableProperties(
693 DirectHandle<JSReceiver> receiver, DirectHandle<JSObject> object,
694 DirectHandle<InterceptorInfo> interceptor, DirectHandle<JSObject> result,
695 IndexedOrNamed type) {
696 DCHECK(IsJSArray(*result) || result->HasSloppyArgumentsElements());
697 ElementsAccessor* accessor = result->GetElementsAccessor();
698
699 size_t length = accessor->GetCapacity(*result, result->elements());
700 for (InternalIndex entry : InternalIndex::Range(length)) {
701 if (!accessor->HasEntry(*result, entry)) continue;
702
703 // args are invalid after args.Call(), create a new one in every iteration.
704 // Query callbacks are not expected to have side effects.
705 PropertyCallbackArguments args(isolate_, interceptor->data(), *receiver,
706 *object, Just(kDontThrow));
707 DirectHandle<Object> element = accessor->Get(isolate_, result, entry);
708 DirectHandle<Object> attributes;
709 if (type == kIndexed) {
710 uint32_t number;
711 CHECK(Object::ToUint32(*element, &number));
712 attributes = args.CallIndexedQuery(interceptor, number);
713 } else {
714 CHECK(IsName(*element));
715 attributes = args.CallNamedQuery(interceptor, Cast<Name>(element));
716 }
717 // An exception was thrown in the interceptor. Propagate.
718 RETURN_VALUE_IF_EXCEPTION(isolate_, ExceptionStatus::kException);
719
720 if (!attributes.is_null()) {
721 int32_t value;
722 CHECK(Object::ToInt32(*attributes, &value));
723 if ((value & DONT_ENUM) == 0) {
724 RETURN_FAILURE_IF_NOT_SUCCESSFUL(AddKey(element, DO_NOT_CONVERT));
725 }
726 }
727 }
728 return ExceptionStatus::kSuccess;
729}

731// Returns |true| on success, |nothing| on exception.
732Maybe<bool> KeyAccumulator::CollectInterceptorKeysInternal(
733 DirectHandle<JSReceiver> receiver, DirectHandle<JSObject> object,
734 DirectHandle<InterceptorInfo> interceptor, IndexedOrNamed type) {
735 PropertyCallbackArguments enum_args(isolate_, interceptor->data(), *receiver,
736 *object, Just(kDontThrow));
737
738 if (IsUndefined(interceptor->enumerator(), isolate_)) {
739 return Just(true);
740 }
741 DirectHandle<JSObjectOrUndefined> maybe_result;
742 if (type == kIndexed) {
743 maybe_result = enum_args.CallIndexedEnumerator(interceptor);
744 } else {
745 DCHECK_EQ(type, kNamed);
746 maybe_result = enum_args.CallNamedEnumerator(interceptor);
747 }
748 // An exception was thrown in the interceptor. Propagate.
749 RETURN_VALUE_IF_EXCEPTION_DETECTOR(isolate_, enum_args, Nothing<bool>());
750 if (IsUndefined(*maybe_result)) return Just(true);
751 DCHECK(IsJSObject(*maybe_result));
752 DirectHandle<JSObject> result = Cast<JSObject>(maybe_result);
753
754 // Request was successfully intercepted, so accept potential side effects
755 // happened up to this point.
756 enum_args.AcceptSideEffects();
757
758 if ((filter_ & ONLY_ENUMERABLE) &&
759 !IsUndefined(interceptor->query(), isolate_)) {
760 RETURN_NOTHING_IF_NOT_SUCCESSFUL(FilterForEnumerableProperties(
761 receiver, object, interceptor, result, type));
762 } else {
763 RETURN_NOTHING_IF_NOT_SUCCESSFUL(AddKeys(
764 result, type == kIndexed ? CONVERT_TO_ARRAY_INDEX : DO_NOT_CONVERT));
765 }
766 return Just(true);
767}

777 DirectHandle<InterceptorInfo> interceptor(

796template <bool skip_symbols>
797std::optional<int> CollectOwnPropertyNamesInternal(
798 DirectHandle<JSObject> object, KeyAccumulator* keys,
799 DirectHandle<DescriptorArray> descs, int start_index, int limit) {
800 AllowGarbageCollection allow_gc;
801 int first_skipped = -1;
802 PropertyFilter filter = keys->filter();
803 KeyCollectionMode mode = keys->mode();
804 for (InternalIndex i : InternalIndex::Range(start_index, limit)) {
805 bool is_shadowing_key = false;
806 PropertyDetails details = descs->GetDetails(i);
807
808 if ((int{details.attributes()} & filter) != 0) {
809 if (mode == KeyCollectionMode::kIncludePrototypes) {
810 is_shadowing_key = true;
811 } else {
812 continue;
813 }
814 }
815
816 Tagged<Name> key = descs->GetKey(i);
817 if (skip_symbols == IsSymbol(key)) {
818 if (first_skipped == -1) first_skipped = i.as_int();
819 continue;
820 }
821 if (Object::FilterKey(key, keys->filter())) continue;
822
823 if (is_shadowing_key) {
824 // This might allocate, but {key} is not used afterwards.
825 keys->AddShadowingKey(key, &allow_gc);
826 continue;
827 } else {
828 if (keys->AddKey(key, DO_NOT_CONVERT) != ExceptionStatus::kSuccess) {
829 return std::optional<int>();
830 }
831 }
832 }
833 return first_skipped;
834}

836// Logic shared between different specializations of CopyEnumKeysTo.
837template <typename Dictionary>
838void CommonCopyEnumKeysTo(Isolate* isolate, DirectHandle<Dictionary> dictionary,
839 DirectHandle<FixedArray> storage,
840 KeyCollectionMode mode, KeyAccumulator* accumulator) {
841 DCHECK_IMPLIES(mode != KeyCollectionMode::kOwnOnly, accumulator != nullptr);
842 int length = storage->length();
843 int properties = 0;
844 ReadOnlyRoots roots(isolate);
845
846 AllowGarbageCollection allow_gc;
847 for (InternalIndex i : dictionary->IterateEntries()) {
848 Tagged<Object> key;
849 if (!dictionary->ToKey(roots, i, &key)) continue;
850 bool is_shadowing_key = false;
851 if (IsSymbol(key)) continue;
852 PropertyDetails details = dictionary->DetailsAt(i);
853 if (details.IsDontEnum()) {
854 if (mode == KeyCollectionMode::kIncludePrototypes) {
855 is_shadowing_key = true;
856 } else {
857 continue;
858 }
859 }
860 if (is_shadowing_key) {
861 // This might allocate, but {key} is not used afterwards.
862 accumulator->AddShadowingKey(key, &allow_gc);
863 continue;
864 } else {
865 if (Dictionary::kIsOrderedDictionaryType) {
866 storage->set(properties, Cast<Name>(key));
867 } else {
868 // If the dictionary does not store elements in enumeration order,
869 // we need to sort it afterwards in CopyEnumKeysTo. To enable this we
870 // need to store indices at this point, rather than the values at the
871 // given indices.
872 storage->set(properties, Smi::FromInt(i.as_int()));
873 }
874 }
875 properties++;
876 if (mode == KeyCollectionMode::kOwnOnly && properties == length) break;
877 }
878
879 CHECK_EQ(length, properties);
880}

882// Copies enumerable keys to preallocated fixed array.
883// Does not throw for uninitialized exports in module namespace objects, so
884// this has to be checked separately.
885template <typename Dictionary>
886void CopyEnumKeysTo(Isolate* isolate, DirectHandle<Dictionary> dictionary,
887 DirectHandle<FixedArray> storage, KeyCollectionMode mode,
888 KeyAccumulator* accumulator) {
889 static_assert(!Dictionary::kIsOrderedDictionaryType);
890
891 CommonCopyEnumKeysTo<Dictionary>(isolate, dictionary, storage, mode,
892 accumulator);
893
894 int length = storage->length();
895
896 DisallowGarbageCollection no_gc;
897 Tagged<Dictionary> raw_dictionary = *dictionary;
898 Tagged<FixedArray> raw_storage = *storage;
899 EnumIndexComparator<Dictionary> cmp(raw_dictionary);
900 // Use AtomicSlot wrapper to ensure that std::sort uses atomic load and
901 // store operations that are safe for concurrent marking.
902 AtomicSlot start(storage->RawFieldOfFirstElement());
903 std::sort(start, start + length, cmp);
904 for (int i = 0; i < length; i++) {
905 InternalIndex index(Smi::ToInt(raw_storage->get(i)));
906 raw_storage->set(i, raw_dictionary->NameAt(index));
907 }
908}

938// Collect the keys from |dictionary| into |keys|, in ascending chronological
939// order of property creation.
940template <typename Dictionary>
941ExceptionStatus CollectKeysFromDictionary(DirectHandle<Dictionary> dictionary,
942 KeyAccumulator* keys) {
943 Isolate* isolate = keys->isolate();
944 ReadOnlyRoots roots(isolate);
945 // TODO(jkummerow): Consider using a std::unique_ptr<InternalIndex[]> instead.
946 DirectHandle<FixedArray> array =
947 isolate->factory()->NewFixedArray(dictionary->NumberOfElements());
948 int array_size = 0;
949 PropertyFilter filter = keys->filter();
950 // Handle enumerable strings in CopyEnumKeysTo.
951 DCHECK_NE(keys->filter(), ENUMERABLE_STRINGS);
952 {
953 DisallowGarbageCollection no_gc;
954 for (InternalIndex i : dictionary->IterateEntries()) {
955 Tagged<Object> key;
956 Tagged<Dictionary> raw_dictionary = *dictionary;
957 if (!raw_dictionary->ToKey(roots, i, &key)) continue;
958 if (Object::FilterKey(key, filter)) continue;
959 PropertyDetails details = raw_dictionary->DetailsAt(i);
960 if ((int{details.attributes()} & filter) != 0) {
961 AllowGarbageCollection gc;
962 // This might allocate, but {key} is not used afterwards.
963 keys->AddShadowingKey(key, &gc);
964 continue;
965 }
966 // TODO(emrich): consider storing keys instead of indices into the array
967 // in case of ordered dictionary type.
968 array->set(array_size++, Smi::FromInt(i.as_int()));
969 }
970 if (!Dictionary::kIsOrderedDictionaryType) {
971 // Sorting only needed if it's an unordered dictionary,
972 // otherwise we traversed elements in insertion order
973
974 EnumIndexComparator<Dictionary> cmp(*dictionary);
975 // Use AtomicSlot wrapper to ensure that std::sort uses atomic load and
976 // store operations that are safe for concurrent marking.
977 AtomicSlot start(array->RawFieldOfFirstElement());
978 std::sort(start, start + array_size, cmp);
979 }
980 }
981
982 bool has_seen_symbol = false;
983 for (int i = 0; i < array_size; i++) {
984 InternalIndex index(Smi::ToInt(array->get(i)));
985 Tagged<Object> key = dictionary->NameAt(index);
986 if (IsSymbol(key)) {
987 has_seen_symbol = true;
988 continue;
989 }
990 ExceptionStatus status = keys->AddKey(key, DO_NOT_CONVERT);
991 if (!status) return status;
992 }
993 if (has_seen_symbol) {
994 for (int i = 0; i < array_size; i++) {
995 InternalIndex index(Smi::ToInt(array->get(i)));
996 Tagged<Object> key = dictionary->NameAt(index);
997 if (!IsSymbol(key)) continue;
998 ExceptionStatus status = keys->AddKey(key, DO_NOT_CONVERT);
999 if (!status) return status;
1000 }
1001 }
1002 return ExceptionStatus::kSuccess;
1003}

1007Maybe<bool> KeyAccumulator::CollectOwnPropertyNames(
1008 DirectHandle<JSReceiver> receiver, DirectHandle<JSObject> object) {
1009 if (filter_ == ENUMERABLE_STRINGS) {
1010 DirectHandle<FixedArray> enum_keys;
1011 if (object->HasFastProperties()) {
1012 enum_keys = KeyAccumulator::GetOwnEnumPropertyKeys(isolate_, object);
1013 // If the number of properties equals the length of enumerable properties
1014 // we do not have to filter out non-enumerable ones
1015 Tagged<Map> map = object->map();
1016 int nof_descriptors = map->NumberOfOwnDescriptors();
1017 if (enum_keys->length() != nof_descriptors) {
1018 if (map->prototype(isolate_) != ReadOnlyRoots(isolate_).null_value()) {
1019 AllowGarbageCollection allow_gc;
1020 DirectHandle<DescriptorArray> descs(
1021 map->instance_descriptors(isolate_), isolate_);
1022 for (InternalIndex i : InternalIndex::Range(nof_descriptors)) {
1023 PropertyDetails details = descs->GetDetails(i);
1024 if (!details.IsDontEnum()) continue;
1025 this->AddShadowingKey(descs->GetKey(i), &allow_gc);
1026 }
1027 }
1028 }
1029 } else if (IsJSGlobalObject(*object)) {
1030 enum_keys = GetOwnEnumPropertyDictionaryKeys(
1031 isolate_, mode_, this, object,
1032 Cast<JSGlobalObject>(*object)->global_dictionary(kAcquireLoad));
1033 } else if (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
1034 enum_keys = GetOwnEnumPropertyDictionaryKeys(
1035 isolate_, mode_, this, object, object->property_dictionary_swiss());
1036 } else {
1037 enum_keys = GetOwnEnumPropertyDictionaryKeys(
1038 isolate_, mode_, this, object, object->property_dictionary());
1039 }
1040 if (IsJSModuleNamespace(*object)) {
1041 // Simulate [[GetOwnProperty]] for establishing enumerability, which
1042 // throws for uninitialized exports.
1043 for (int i = 0, n = enum_keys->length(); i < n; ++i) {
1044 DirectHandle<String> key(Cast<String>(enum_keys->get(i)), isolate_);
1045 if (Cast<JSModuleNamespace>(object)
1046 ->GetExport(isolate(), key)
1047 .is_null()) {
1048 return Nothing<bool>();
1049 }
1050 }
1051 }
1052 RETURN_NOTHING_IF_NOT_SUCCESSFUL(AddKeys(enum_keys, DO_NOT_CONVERT));
1053 } else {
1054 if (object->HasFastProperties()) {
1055 int limit = object->map()->NumberOfOwnDescriptors();
1056 DirectHandle<DescriptorArray> descs(
1057 object->map()->instance_descriptors(isolate_), isolate_);
1058 // First collect the strings,
1059 std::optional<int> first_symbol =
1060 CollectOwnPropertyNamesInternal<true>(object, this, descs, 0, limit);
1061 // then the symbols.
1062 RETURN_NOTHING_IF_NOT_SUCCESSFUL(first_symbol);
1063 if (first_symbol.value() != -1) {
1064 RETURN_NOTHING_IF_NOT_SUCCESSFUL(CollectOwnPropertyNamesInternal<false>(
1065 object, this, descs, first_symbol.value(), limit));
1066 }
1067 } else if (IsJSGlobalObject(*object)) {
1068 RETURN_NOTHING_IF_NOT_SUCCESSFUL(CollectKeysFromDictionary(
1069 direct_handle(
1070 Cast<JSGlobalObject>(*object)->global_dictionary(kAcquireLoad),
1071 isolate_),
1072 this));
1073 } else if (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
1074 RETURN_NOTHING_IF_NOT_SUCCESSFUL(CollectKeysFromDictionary(
1075 direct_handle(object->property_dictionary_swiss(), isolate_), this));
1076 } else {
1077 RETURN_NOTHING_IF_NOT_SUCCESSFUL(CollectKeysFromDictionary(
1078 direct_handle(object->property_dictionary(), isolate_), this));
1079 }
1080 }
1081 // Add the property keys from the interceptor.
1082 return CollectInterceptorKeys(receiver, object, kNamed);
1083}

1133// |nothing| if an exception was thrown.

1135 DirectHandle<JSObject> object) {
1136 // Check access rights if required.
1137 if (IsAccessCheckNeeded(*object) &&
1138 !isolate_->MayAccess(isolate_->native_context(), object)) {

1140 // iterator when it doesn't have access...

1157 access_check_info->named_interceptor() != Tagged<Object>()) {
1158 MAYBE_RETURN(CollectAccessCheckInterceptorKeys(access_check_info,
1159 receiver, object),
1160 Nothing<bool>());

1163 }
1164 if (filter_ & PRIVATE_NAMES_ONLY) {
1165 RETURN_NOTHING_IF_NOT_SUCCESSFUL(CollectPrivateNames(receiver, object));
1166 return Just(true);
1167 }
1168
1169 if (may_have_elements_) {
1170 MAYBE_RETURN(CollectOwnElementIndices(receiver, object), Nothing<bool>());
1171 }
1172 MAYBE_RETURN(CollectOwnPropertyNames(receiver, object), Nothing<bool>());
1173 return Just(true);

1215// Returns |true| on success, |nothing| in case of exception.
1216Maybe<bool> KeyAccumulator::CollectOwnJSProxyKeys(
1217 DirectHandle<JSReceiver> receiver, DirectHandle<JSProxy> proxy) {
1218 STACK_CHECK(isolate_, Nothing<bool>());
1219 if (filter_ == PRIVATE_NAMES_ONLY) {
1220 if (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
1221 RETURN_NOTHING_IF_NOT_SUCCESSFUL(CollectKeysFromDictionary(
1222 direct_handle(proxy->property_dictionary_swiss(), isolate_), this));
1223 } else {
1224 RETURN_NOTHING_IF_NOT_SUCCESSFUL(CollectKeysFromDictionary(
1225 direct_handle(proxy->property_dictionary(), isolate_), this));
1226 }

1228 }

1230 // 1. Let handler be the value of the [[ProxyHandler]] internal slot of O.
1231 DirectHandle<Object> handler(proxy->handler(), isolate_);
1232 // 2. If handler is null, throw a TypeError exception.
1233 // 3. Assert: Type(handler) is Object.
