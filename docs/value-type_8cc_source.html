28bool EquivalentNumericSig(const CanonicalSig* a, const FunctionSig* b) {
29 if (a->parameter_count() != b->parameter_count()) return false;
30 if (a->return_count() != b->return_count()) return false;
31 base::Vector<const CanonicalValueType> a_types = a->all();
32 base::Vector<const ValueType> b_types = b->all();
33 for (size_t i = 0; i < a_types.size(); i++) {
34 if (!a_types[i].is_numeric()) return false;
35 if (a_types[i].kind() != b_types[i].kind()) return false;
36 }
37 return true;
38}

56const wasm::FunctionSig* ReplaceTypeInSig(Zone* zone,
57 const wasm::FunctionSig* sig,
58 wasm::ValueType from,
59 wasm::ValueType to,
60 size_t num_replacements) {
61 size_t param_occurences =
62 std::count(sig->parameters().begin(), sig->parameters().end(), from);
63 size_t return_occurences =
64 std::count(sig->returns().begin(), sig->returns().end(), from);
65 if (param_occurences == 0 && return_occurences == 0) return sig;
66
67 wasm::FunctionSig::Builder builder(
68 zone, sig->return_count() + return_occurences * (num_replacements - 1),
69 sig->parameter_count() + param_occurences * (num_replacements - 1));
70
71 for (wasm::ValueType ret : sig->returns()) {
72 if (ret == from) {
73 for (size_t i = 0; i < num_replacements; i++) builder.AddReturn(to);
74 } else {
75 builder.AddReturn(ret);
76 }
77 }
78
79 for (wasm::ValueType param : sig->parameters()) {
80 if (param == from) {
81 for (size_t i = 0; i < num_replacements; i++) builder.AddParam(to);
82 } else {
83 builder.AddParam(param);
84 }
85 }
86
87 return builder.Get();
88}
