21// `ConcreteVisitor` must implement the following methods:
22// 1) FilterPage(const MemoryChunk*) - returns true if a page may contain
23// interesting objects for the GC (e.g. a young page in a young gen GC).
24// 2) FilterLargeObject(Tagged<HeapObject>, MapWord) - returns true if the
25// object should be handled by conservative stack scanning.
26// 3) FilterNormalObject(Tagged<HeapObject>, MapWord) - returns true if the
27// object should be handled by conservative stack scanning.
28// 4) HandleObjectFound(Tagged<HeapObject>, size_t) - Callback whenever
29// FindBasePtr finds a new object.
30template <typename ConcreteVisitor>
31class V8_EXPORT_PRIVATE ConservativeStackVisitorBase
32 : public ::heap::base::StackVisitor {
33 public:
34 ConservativeStackVisitorBase(Isolate* isolate, RootVisitor* root_visitor);
35
36 void VisitPointer(const void* pointer) final;
37
38 // This method finds an object header based on a `maybe_inner_ptr`. It returns
39 // `kNullAddress` if the parameter does not point to (the interior of) a valid
40 // heap object. The allocator_ field is used to provide the set of valid heap
41 // pages. The collector_ field is used to determine which kind of heap objects
42 // we are interested in. For MARK_COMPACTOR all heap objects are considered,
43 // whereas for young generation collectors we only consider objects in the
44 // young generation.
45 Address FindBasePtr(Address maybe_inner_ptr,
46 PtrComprCageBase cage_base) const;
47
48 private:
49 void VisitConservativelyIfPointer(Address address);
50 void VisitConservativelyIfPointer(Address address,
51 PtrComprCageBase cage_base);
52
53#ifdef V8_COMPRESS_POINTERS
54 bool IsInterestingCage(PtrComprCageBase cage_base) const;
55#endif
56
57 // The "interesting" cages where we conservatively scan pointers are:
58 // - The regular cage for the V8 heap.
59 // - The cage used for code objects, if an external code space is used.
60 // - The trusted space cage.
61 const PtrComprCageBase cage_base_;
62#ifdef V8_EXTERNAL_CODE_SPACE
63 const PtrComprCageBase code_cage_base_;
64 base::AddressRegion code_address_region_;
65#endif
66#ifdef V8_ENABLE_SANDBOX
67 const PtrComprCageBase trusted_cage_base_;
68#endif
69
70 RootVisitor* const root_visitor_;
71 MemoryAllocator* const allocator_;
72};
