26WasmCompilationResult WasmCompilationUnit::ExecuteCompilation(
27 CompilationEnv* env, const WireBytesStorage* wire_bytes_storage,
28 Counters* counters, WasmDetectedFeatures* detected) {
29 DCHECK_GE(func_index_, static_cast<int>(env->module->num_imported_functions));
30 const WasmFunction* func = &env->module->functions[func_index_];
31 base::Vector<const uint8_t> code = wire_bytes_storage->GetCode(func->code);
32 bool is_shared = env->module->type(func->sig_index).is_shared;
33 wasm::FunctionBody func_body{func->sig, func->code.offset(), code.begin(),
34 code.end(), is_shared};
35
36 std::optional<TimedHistogramScope> wasm_compile_function_time_scope;
37 std::optional<TimedHistogramScope> wasm_compile_huge_function_time_scope;
38 if (counters && base::TimeTicks::IsHighResolution()) {
39 if (func_body.end - func_body.start >= 100 * KB) {
40 auto huge_size_histogram = SELECT_WASM_COUNTER(
41 counters, env->module->origin, wasm, huge_function_size_bytes);
42 huge_size_histogram->AddSample(
43 static_cast<int>(func_body.end - func_body.start));
44 wasm_compile_huge_function_time_scope.emplace(
45 counters->wasm_compile_huge_function_time());
46 }
47 auto timed_histogram = SELECT_WASM_COUNTER(counters, env->module->origin,
48 wasm_compile, function_time);
49 wasm_compile_function_time_scope.emplace(timed_histogram);
50 }
51
52 // Before executing compilation, make sure that the function was validated.
53 // Both Liftoff and TurboFan compilation do not perform validation, so can
54 // only run on valid functions.
55 if (V8_UNLIKELY(!env->module->function_was_validated(func_index_))) {
56 // This code path can only be reached in
57 // - eager compilation mode,
58 // - with lazy validation,
59 // - with PGO (which compiles some functions eagerly), or
60 // - with compilation hints (which also compiles some functions eagerly).
61 DCHECK(!v8_flags.wasm_lazy_compilation || v8_flags.wasm_lazy_validation ||
62 v8_flags.experimental_wasm_pgo_from_file ||
63 v8_flags.experimental_wasm_compilation_hints);
64 Zone validation_zone{GetWasmEngine()->allocator(), ZONE_NAME};
65 if (ValidateFunctionBody(&validation_zone, env->enabled_features,
66 env->module, detected, func_body)
67 .failed()) {
68 return {};
69 }
70 env->module->set_function_validated(func_index_);
71 }
72
73 if (v8_flags.trace_wasm_compiler) {
74 PrintF("Compiling wasm function %d with %s\n", func_index_,
75 ExecutionTierToString(tier_));
76 }
77
78 WasmCompilationResult result;
79 int declared_index = declared_function_index(env->module, func_index_);
80
81 switch (tier_) {
82 case ExecutionTier::kNone:
83#if V8_ENABLE_DRUMBRAKE
84 case ExecutionTier::kInterpreter:
85#endif // V8_ENABLE_DRUMBRAKE
86 UNREACHABLE();
87
88 case ExecutionTier::kLiftoff: {
89 // The --wasm-tier-mask-for-testing flag can force functions to be
90 // compiled with TurboFan, and the --wasm-debug-mask-for-testing can force
91 // them to be compiled for debugging, see documentation.
92 bool try_liftoff = true;
93 if (V8_UNLIKELY(v8_flags.wasm_tier_mask_for_testing != 0)) {
94 bool must_use_liftoff =
95 v8_flags.liftoff_only ||
96 for_debugging_ != ForDebugging::kNotForDebugging;
97 bool tiering_requested =
98 declared_index < 32 &&
99 (v8_flags.wasm_tier_mask_for_testing & (1 << declared_index));
100 if (!must_use_liftoff && tiering_requested) try_liftoff = false;
101 }
102
103 if (V8_LIKELY(try_liftoff)) {
104 auto options = LiftoffOptions{}
105 .set_func_index(func_index_)
106 .set_for_debugging(for_debugging_)
107 .set_counters(counters)
108 .set_detected_features(detected);
109 // We do not use the debug side table, we only (optionally) pass it to
110 // cover different code paths in Liftoff for testing.
111 std::unique_ptr<DebugSideTable> unused_debug_sidetable;
112 if (V8_UNLIKELY(declared_index < 32 &&
113 (v8_flags.wasm_debug_mask_for_testing &
114 (1 << declared_index)) != 0)) {
115 options.set_debug_sidetable(&unused_debug_sidetable);
116 if (!for_debugging_) options.set_for_debugging(kForDebugging);
117 }
118 result = ExecuteLiftoffCompilation(env, func_body, options);
119 if (result.succeeded()) break;
120 }
121
122 // If --liftoff-only, do not fall back to turbofan, even if compilation
123 // failed.
124 if (v8_flags.liftoff_only) break;
125
126 // If Liftoff failed, fall back to TurboFan.
127 // TODO(wasm): We could actually stop or remove the tiering unit for this
128 // function to avoid compiling it twice with TurboFan.
129 [[fallthrough]];
130 }
131 case ExecutionTier::kTurbofan: {
132 compiler::WasmCompilationData data(func_body);
133 data.func_index = func_index_;
134 data.wire_bytes_storage = wire_bytes_storage;
135 result = compiler::turboshaft::ExecuteTurboshaftWasmCompilation(
136 env, data, detected);
137 // In exceptional cases it can happen that compilation requests for
138 // debugging end up being executed by Turbofan, e.g. if Liftoff bails out
139 // because of unsupported features or the --wasm-tier-mask-for-testing is
140 // set. In that case we set the for_debugging field for the TurboFan
141 // result to match the requested for_debugging_.
142 result.for_debugging = for_debugging_;
143 break;
144 }
145 }
146
147 DCHECK(result.succeeded());
148 if (counters) {
149 counters->wasm_generated_code_size()->Increment(
150 result.code_desc.instr_size);
151 counters->wasm_reloc_size()->Increment(result.code_desc.reloc_size);
152 counters->wasm_deopt_data_size()->Increment(
153 static_cast<int>(result.deopt_data.size()));
154 }
155
156 result.func_index = func_index_;
157
158 return result;
159}

161// static
162void WasmCompilationUnit::CompileWasmFunction(Counters* counters,
163 NativeModule* native_module,
164 WasmDetectedFeatures* detected,
165 const WasmFunction* function,
166 ExecutionTier tier) {
167 ModuleWireBytes wire_bytes(native_module->wire_bytes());
168 bool is_shared = native_module->module()->type(function->sig_index).is_shared;
169 FunctionBody function_body{function->sig, function->code.offset(),
170 wire_bytes.start() + function->code.offset(),
171 wire_bytes.start() + function->code.end_offset(),
172 is_shared};
173
174 DCHECK_LE(native_module->num_imported_functions(), function->func_index);
175 DCHECK_LT(function->func_index, native_module->num_functions());
176 WasmCompilationUnit unit(function->func_index, tier, kNotForDebugging);
177 CompilationEnv env = CompilationEnv::ForModule(native_module);
178 WasmCompilationResult result = unit.ExecuteCompilation(
179 &env, native_module->compilation_state()->GetWireBytesStorage().get(),
180 counters, detected);
181 if (result.succeeded()) {
182 WasmCodeRefScope code_ref_scope;
183 native_module->PublishCode(native_module->AddCompiledCode(result));
184 } else {
185 native_module->compilation_state()->SetError();
186 }
187}

220Handle<Code> JSToWasmWrapperCompilationUnit::Finalize() {
221#if V8_ENABLE_DRUMBRAKE
222 if (v8_flags.wasm_jitless) {
223 return isolate_->builtins()->code_handle(
224 Builtin::kGenericJSToWasmInterpreterWrapper);
225 }
226#endif // V8_ENABLE_DRUMBRAKE
227
228 CompilationJob::Status status = job_->FinalizeJob(isolate_);
229 CHECK_EQ(status, CompilationJob::SUCCEEDED);
230 OptimizedCompilationInfo* info =
231 static_cast<compiler::turboshaft::TurboshaftCompilationJob*>(job_.get())
232 ->compilation_info();
233 Handle<Code> code = info->code();
234 if (isolate_->IsLoggingCodeCreation()) {
235 Handle<String> name = isolate_->factory()->NewStringFromAsciiChecked(
236 info->GetDebugName().get());
237 PROFILE(isolate_, CodeCreateEvent(LogEventListener::CodeTag::kStub,
238 Cast<AbstractCode>(code), name));
239 }
240 // We should always have checked the cache before compiling a wrapper.
241 Tagged<WeakFixedArray> cache = isolate_->heap()->js_to_wasm_wrappers();
242 DCHECK(cache->get(sig_index_.index).IsCleared());
243 // Install the compiled wrapper in the cache now.
244 cache->set(sig_index_.index, MakeWeak(code->wrapper()));
245 Counters* counters = isolate_->counters();
246 counters->wasm_generated_code_size()->Increment(code->body_size());
247 counters->wasm_reloc_size()->Increment(code->relocation_size());
248 counters->wasm_compiled_export_wrapper()->Increment(1);
249 return code;
250}

255 // Run the compilation unit synchronously.
256 JSToWasmWrapperCompilationUnit unit(isolate, sig, sig_index);
257 unit.Execute();
258 return unit.Finalize();
