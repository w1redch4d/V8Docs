238 DCHECK(node->opcode() == IrOpcode::kStore ||

247 StoreRepresentation stored_rep() const {
248 switch (node_->opcode()) {
249 case IrOpcode::kStore:
250 case IrOpcode::kProtectedStore:
251 case IrOpcode::kStoreTrapOnNull:
252 case IrOpcode::kStoreIndirectPointer:
253 return StoreRepresentationOf(node_->op());
254 case IrOpcode::kUnalignedStore:
255 return {UnalignedStoreRepresentationOf(node_->op()),
256 WriteBarrierKind::kNoWriteBarrier};
257 case IrOpcode::kWord32AtomicStore:
258 case IrOpcode::kWord64AtomicStore:
259 return AtomicStoreParametersOf(node_->op()).store_representation();
260 default:
261 UNREACHABLE();
262 }
263 }

265 switch (node_->opcode()) {
266 case IrOpcode::kStore:
267 case IrOpcode::kProtectedStore:
268 case IrOpcode::kStoreTrapOnNull:
269 case IrOpcode::kStoreIndirectPointer:
270 case IrOpcode::kUnalignedStore:
271 return std::nullopt;
272 case IrOpcode::kWord32AtomicStore:
273 case IrOpcode::kWord64AtomicStore:
274 return AtomicStoreParametersOf(node_->op()).order();
275 default:
276 UNREACHABLE();
277 }
278 }
279 MemoryAccessKind access_kind() const {
280 switch (node_->opcode()) {
281 case IrOpcode::kStore:
282 case IrOpcode::kStoreIndirectPointer:
283 case IrOpcode::kUnalignedStore:
284 return MemoryAccessKind::kNormal;
285 case IrOpcode::kProtectedStore:
286 case IrOpcode::kStoreTrapOnNull:
287 return MemoryAccessKind::kProtectedByTrapHandler;
288 case IrOpcode::kWord32AtomicStore:
289 case IrOpcode::kWord64AtomicStore:
290 return AtomicStoreParametersOf(node_->op()).kind();
291 default:
292 UNREACHABLE();
293 }
294 }

296 return node_->opcode() == IrOpcode::kWord32AtomicStore ||
297 node_->opcode() == IrOpcode::kWord64AtomicStore;

1302 const turboshaft::Operation& value_op = graph_->Get(value);
1303 const turboshaft::Operation& user_op = graph_->Get(user);
1304 size_t use_count = base::count_if(
1305 user_op.inputs(),
1306 [value](turboshaft::OpIndex input) { return input == value; });
1307 if (V8_UNLIKELY(use_count == 0)) {
1308 // We have a special case here:
1309 //
1310 // value
1311 // |
1312 // TruncateWord64ToWord32
1313 // |
1314 // user
1315 //
1316 // If emitting user performs the truncation implicitly, we end up calling
1317 // CanCover with value and user such that user might have no (direct) uses
1318 // of value. There are cases of other unnecessary operations that can lead
1319 // to the same situation (e.g. bitwise and, ...). In this case, we still
1320 // cover if value has only a single use and this is one of the direct
1321 // inputs of user, which also only has a single use (in user).
1322 // TODO(nicohartmann@): We might generalize this further if we see use
1323 // cases.
1324 if (!value_op.saturated_use_count.IsOne()) return false;
1325 for (auto input : user_op.inputs()) {
1326 const turboshaft::Operation& input_op = graph_->Get(input);

1328 input_op.inputs(),
1329 [value](turboshaft::OpIndex input) { return input == value; });
1330 if (indirect_use_count > 0) {
1331 return input_op.saturated_use_count.IsOne();
1332 }
1333 }
1334 return false;
1335 }

1337 // Projections always have a Tuple use, but it shouldn't count as a use as
1338 // far as is_exclusive_user_of is concerned, since no instructions are
1339 // emitted for the TupleOp, which is just a Turboshaft "meta operation".
1340 // We thus increase the use_count by 1, to attribute the TupleOp use to
1341 // the current operation.
1342 use_count++;
1343 }
