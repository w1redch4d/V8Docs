50class MatchPrototypePredicate : public v8::QueryObjectPredicate {
51 public:
52 MatchPrototypePredicate(V8InspectorImpl* inspector,
53 v8::Local<v8::Context> context,
54 v8::Local<v8::Object> prototype)
55 : m_inspector(inspector), m_context(context), m_prototype(prototype) {}
56
57 bool Filter(v8::Local<v8::Object> object) override {
58 if (object->IsModuleNamespaceObject()) return false;
59 v8::Local<v8::Context> objectContext;
60 if (!v8::debug::GetCreationContext(object).ToLocal(&objectContext)) {
61 return false;
62 }
63 if (objectContext != m_context) return false;
64 if (!m_inspector->client()->isInspectableHeapObject(object)) return false;
65 // Get prototype chain for current object until first visited prototype.
66 for (v8::Local<v8::Value> prototype = object->GetPrototype();
67 prototype->IsObject();
68 prototype = prototype.As<v8::Object>()->GetPrototype()) {
69 if (m_prototype == prototype) return true;
70 }
71 return false;
72 }
73
74 private:
75 V8InspectorImpl* m_inspector;
76 v8::Local<v8::Context> m_context;
77 v8::Local<v8::Value> m_prototype;
78};

123void V8Debugger::disable() {
124 if (isPaused()) {
125 bool scheduledOOMBreak = m_scheduledOOMBreak;
126 bool hasAgentAcceptsPause = false;
127
128 if (m_instrumentationPause) {
129 quitMessageLoopIfAgentsFinishedInstrumentation();
130 } else {
131 m_inspector->forEachSession(
132 m_pausedContextGroupId, [&scheduledOOMBreak, &hasAgentAcceptsPause](
133 V8InspectorSessionImpl* session) {
134 if (session->debuggerAgent()->acceptsPause(scheduledOOMBreak)) {
135 hasAgentAcceptsPause = true;

154 m_originalHeapLimit);

164std::vector<std::unique_ptr<V8DebuggerScript>> V8Debugger::getCompiledScripts(
165 int contextGroupId, V8DebuggerAgentImpl* agent) {
166 std::vector<std::unique_ptr<V8DebuggerScript>> result;
167 v8::HandleScope scope(m_isolate);
168 std::vector<v8::Global<v8::debug::Script>> scripts;
169 v8::debug::GetLoadedScripts(m_isolate, scripts);
170 for (size_t i = 0; i < scripts.size(); ++i) {
171 v8::Local<v8::debug::Script> script = scripts[i].Get(m_isolate);
172 if (!script->WasCompiled()) continue;
173 if (!script->IsEmbedded()) {
174 int contextId;
175 if (!script->ContextId().To(&contextId)) continue;
176 if (m_inspector->contextGroupId(contextId) != contextGroupId) continue;
177 }
178 result.push_back(V8DebuggerScript::Create(m_isolate, script, false, agent,
179 m_inspector->client()));
180 }
181 return result;
182}

374void V8Debugger::reportTermination() {
375 if (m_terminateExecutionReported) {
376 DCHECK(m_terminateExecutionCallbackContext.IsEmpty());
377 return;
378 }
379 v8::HandleScope handles(m_isolate);
380 m_isolate->RemoveCallCompletedCallback(
381 &V8Debugger::terminateExecutionCompletedCallback);
382 if (!m_terminateExecutionCallbackContext.IsEmpty()) {
383 v8::MicrotaskQueue* microtask_queue =
384 m_terminateExecutionCallbackContext.Get(m_isolate)->GetMicrotaskQueue();
385 if (microtask_queue) {
386 microtask_queue->RemoveMicrotasksCompletedCallback(
387 &V8Debugger::terminateExecutionCompletedCallbackIgnoringData,
388 microtask_queue);
389 }
390 }
391 m_isolate->CancelTerminateExecution();
392 if (m_terminateExecutionCallback) {
393 m_terminateExecutionCallback->sendSuccess();
394 m_terminateExecutionCallback.reset();
395 }
396 m_terminateExecutionCallbackContext.Reset();
397 m_terminateExecutionReported = true;
398}

478void V8Debugger::handleProgramBreak(
479 v8::Local<v8::Context> pausedContext, v8::Local<v8::Value> exception,
480 const std::vector<v8::debug::BreakpointId>& breakpointIds,
481 v8::debug::BreakReasons breakReasons,
482 v8::debug::ExceptionType exceptionType, bool isUncaught) {
483 // Don't allow nested breaks.
484 if (isPaused()) return;
485
486 int contextGroupId = m_inspector->contextGroupId(pausedContext);
487 if (m_targetContextGroupId && contextGroupId != m_targetContextGroupId) {
488 v8::debug::PrepareStep(m_isolate, v8::debug::StepOut);
489 return;
490 }
491
492 DCHECK(hasScheduledBreakOnNextFunctionCall() ==
493 (m_taskWithScheduledBreakPauseRequested ||
494 m_externalAsyncTaskPauseRequested || m_pauseOnNextCallRequested));
495 if (m_taskWithScheduledBreakPauseRequested ||
496 m_externalAsyncTaskPauseRequested)
497 breakReasons.Add(v8::debug::BreakReason::kAsyncStep);
498 if (m_pauseOnNextCallRequested)
499 breakReasons.Add(v8::debug::BreakReason::kAgent);
500
501 m_targetContextGroupId = 0;
502 m_pauseOnNextCallRequested = false;
503 m_pauseOnAsyncCall = false;
504 m_taskWithScheduledBreak = nullptr;
505 m_externalAsyncTaskPauseRequested = false;
506 m_taskWithScheduledBreakPauseRequested = false;
507
508 bool scheduledOOMBreak = m_scheduledOOMBreak;
509 DCHECK(scheduledOOMBreak ==
510 breakReasons.contains(v8::debug::BreakReason::kOOM));
511 bool hasAgents = false;
512
513 m_inspector->forEachSession(
514 contextGroupId,
515 [&scheduledOOMBreak, &hasAgents](V8InspectorSessionImpl* session) {
516 if (session->debuggerAgent()->acceptsPause(scheduledOOMBreak))
517 hasAgents = true;
518 });
519 if (!hasAgents) return;
520
521 if (breakpointIds.size() == 1 &&
522 breakpointIds[0] == m_continueToLocationBreakpointId) {
523 v8::Context::Scope contextScope(pausedContext);
524 if (!shouldContinueToCurrentLocation()) return;
525 }
526 clearContinueToLocation();
527
528 DCHECK(contextGroupId);
529 m_pausedContextGroupId = contextGroupId;
530
531 m_inspector->forEachSession(
532 contextGroupId,
533 [&pausedContext, &exception, &breakpointIds, &exceptionType, &isUncaught,
534 &scheduledOOMBreak, &breakReasons](V8InspectorSessionImpl* session) {
535 if (session->debuggerAgent()->acceptsPause(scheduledOOMBreak)) {
536 session->debuggerAgent()->didPause(
537 InspectedContext::contextId(pausedContext), exception,
538 breakpointIds, exceptionType, isUncaught, breakReasons);
539 }
540 });
541 {
542 v8::Context::Scope scope(pausedContext);
543
544 m_inspector->forEachSession(
545 contextGroupId, [](V8InspectorSessionImpl* session) {
546 if (session->heapProfilerAgent()) {
547 session->heapProfilerAgent()->takePendingHeapSnapshots();
548 }
549 });
550
551 m_inspector->client()->runMessageLoopOnPause(contextGroupId);
552 m_pausedContextGroupId = 0;
553 }
554 m_inspector->forEachSession(contextGroupId,
555 [](V8InspectorSessionImpl* session) {
556 if (session->debuggerAgent()->enabled()) {
557 session->debuggerAgent()->clearBreakDetails();
558 session->debuggerAgent()->didContinue();
559 }
560 });
561
562 if (m_scheduledOOMBreak) m_isolate->RestoreOriginalHeapLimit();
563 m_scheduledOOMBreak = false;
564}

576size_t V8Debugger::nearHeapLimitCallback(void* data, size_t current_heap_limit,
577 size_t initial_heap_limit) {
578 V8Debugger* thisPtr = static_cast<V8Debugger*>(data);
579 thisPtr->m_originalHeapLimit = current_heap_limit;
580 thisPtr->m_scheduledOOMBreak = true;
581 v8::Local<v8::Context> context =
582 thisPtr->m_isolate->GetEnteredOrMicrotaskContext();
583 thisPtr->m_targetContextGroupId =
584 context.IsEmpty() ? 0 : thisPtr->m_inspector->contextGroupId(context);
585 thisPtr->m_isolate->RequestInterrupt(

587 // There's a redundancy between setting `m_scheduledOOMBreak` and

590 // for being explicit, we still pass the break reason along.
591 v8::debug::BreakRightNow(
592 isolate, v8::debug::BreakReasons({v8::debug::BreakReason::kOOM}));
593 },
594 nullptr);
595 return HeapLimitForDebugging(initial_heap_limit);

621V8Debugger::ActionAfterInstrumentation V8Debugger::BreakOnInstrumentation(
622 v8::Local<v8::Context> pausedContext,
623 v8::debug::BreakpointId instrumentationId) {
624 // Don't allow nested breaks.
625 if (isPaused()) return ActionAfterInstrumentation::kPauseIfBreakpointsHit;
626
627 int contextGroupId = m_inspector->contextGroupId(pausedContext);
628 bool hasAgents = false;
629 m_inspector->forEachSession(
630 contextGroupId, [&hasAgents](V8InspectorSessionImpl* session) {
631 if (session->debuggerAgent()->acceptsPause(false /* isOOMBreak */))
632 hasAgents = true;
633 });
634 if (!hasAgents) return ActionAfterInstrumentation::kPauseIfBreakpointsHit;
635
636 m_pausedContextGroupId = contextGroupId;
637 m_instrumentationPause = true;
638 m_inspector->forEachSession(
639 contextGroupId, [instrumentationId](V8InspectorSessionImpl* session) {

756void V8Debugger::AsyncEventOccurred(v8::debug::DebugAsyncActionType type,
757 int id, bool isBlackboxed) {
758 // Async task events from Promises are given misaligned pointers to prevent
759 // from overlapping with other Blink task identifiers.
760 void* task = reinterpret_cast<void*>(id * 2 + 1);
761 switch (type) {
762 case v8::debug::kDebugPromiseThen:
763 asyncTaskScheduledForStack(toStringView("Promise.then"), task, false);
764 if (!isBlackboxed) asyncTaskCandidateForStepping(task);
765 break;
766 case v8::debug::kDebugPromiseCatch:
767 asyncTaskScheduledForStack(toStringView("Promise.catch"), task, false);
768 if (!isBlackboxed) asyncTaskCandidateForStepping(task);
769 break;
770 case v8::debug::kDebugPromiseFinally:
771 asyncTaskScheduledForStack(toStringView("Promise.finally"), task, false);
772 if (!isBlackboxed) asyncTaskCandidateForStepping(task);
773 break;
774 case v8::debug::kDebugWillHandle:
775 asyncTaskStartedForStack(task);
776 asyncTaskStartedForStepping(task);
777 break;
778 case v8::debug::kDebugDidHandle:
779 asyncTaskFinishedForStack(task);
780 asyncTaskFinishedForStepping(task);
781 break;
782 case v8::debug::kDebugAwait:
783 asyncTaskScheduledForStack(toStringView("await"), task, false, true);
784 break;
785 case v8::debug::kDebugStackTraceCaptured:
786 asyncStackTraceCaptured(id);
787 break;
788 }
789}

800v8::MaybeLocal<v8::Value> V8Debugger::getTargetScopes(
801 v8::Local<v8::Context> context, v8::Local<v8::Value> value,
802 ScopeTargetKind kind) {
803 std::unique_ptr<v8::debug::ScopeIterator> iterator;
804 switch (kind) {
805 case FUNCTION:
806 iterator = v8::debug::ScopeIterator::CreateForFunction(
807 m_isolate, value.As<v8::Function>());
808 break;
809 case GENERATOR:
810 v8::Local<v8::debug::GeneratorObject> generatorObject =
811 v8::debug::GeneratorObject::Cast(value);
812 if (!generatorObject->IsSuspended()) return v8::MaybeLocal<v8::Value>();
813
814 iterator = v8::debug::ScopeIterator::CreateForGeneratorObject(
815 m_isolate, value.As<v8::Object>());
816 break;
817 }
818 if (!iterator) return v8::MaybeLocal<v8::Value>();
819 v8::Local<v8::Array> result = v8::Array::New(m_isolate);
820 if (!result->SetPrototypeV2(context, v8::Null(m_isolate)).FromMaybe(false)) {
821 return v8::MaybeLocal<v8::Value>();
822 }
823
824 for (; !iterator->Done(); iterator->Advance()) {
825 v8::Local<v8::Object> scope = v8::Object::New(m_isolate);
826 if (!addInternalObject(context, scope, V8InternalValueType::kScope))
827 return v8::MaybeLocal<v8::Value>();
828 String16 nameSuffix = toProtocolStringWithTypeCheck(
829 m_isolate, iterator->GetFunctionDebugName());
830 String16 description;
831 if (nameSuffix.length()) nameSuffix = " (" + nameSuffix + ")";
832 switch (iterator->GetType()) {
833 case v8::debug::ScopeIterator::ScopeTypeGlobal:
834 description = "Global" + nameSuffix;
835 break;
836 case v8::debug::ScopeIterator::ScopeTypeLocal:
837 description = "Local" + nameSuffix;
838 break;
839 case v8::debug::ScopeIterator::ScopeTypeWith:
840 description = "With Block" + nameSuffix;
841 break;
842 case v8::debug::ScopeIterator::ScopeTypeClosure:
843 description = "Closure" + nameSuffix;
844 break;
845 case v8::debug::ScopeIterator::ScopeTypeCatch:
846 description = "Catch" + nameSuffix;
847 break;
848 case v8::debug::ScopeIterator::ScopeTypeBlock:
849 description = "Block" + nameSuffix;
850 break;
851 case v8::debug::ScopeIterator::ScopeTypeScript:
852 description = "Script" + nameSuffix;
853 break;
854 case v8::debug::ScopeIterator::ScopeTypeEval:
855 description = "Eval" + nameSuffix;
856 break;
857 case v8::debug::ScopeIterator::ScopeTypeModule:
858 description = "Module" + nameSuffix;
859 break;
860 case v8::debug::ScopeIterator::ScopeTypeWasmExpressionStack:
861 description = "Wasm Expression Stack" + nameSuffix;
862 break;
863 }
864 v8::Local<v8::Object> object = iterator->GetObject();
865 createDataProperty(context, scope,
866 toV8StringInternalized(m_isolate, "description"),
867 toV8String(m_isolate, description));
868 createDataProperty(context, scope,
869 toV8StringInternalized(m_isolate, "object"), object);
870 createDataProperty(context, result, result->Length(), scope);
871 }
872 if (!addInternalObject(context, result, V8InternalValueType::kScopeList))
873 return v8::MaybeLocal<v8::Value>();
874 return result;
875}

887v8::MaybeLocal<v8::Array> V8Debugger::collectionsEntries(
888 v8::Local<v8::Context> context, v8::Local<v8::Value> collection) {
889 v8::Isolate* isolate = context->GetIsolate();
890 v8::Local<v8::Array> entries;
891 bool isKeyValue = false;
892 if (!collection->IsObject() || !collection.As<v8::Object>()
893 ->PreviewEntries(&isKeyValue)
894 .ToLocal(&entries)) {
895 return v8::MaybeLocal<v8::Array>();
896 }
897
898 v8::Local<v8::Array> wrappedEntries = v8::Array::New(isolate);
899 CHECK(!isKeyValue || wrappedEntries->Length() % 2 == 0);
900 if (!wrappedEntries->SetPrototypeV2(context, v8::Null(isolate))
901 .FromMaybe(false))
902 return v8::MaybeLocal<v8::Array>();
903 for (uint32_t i = 0; i < entries->Length(); i += isKeyValue ? 2 : 1) {
904 v8::Local<v8::Value> item;
905 if (!entries->Get(context, i).ToLocal(&item)) continue;
906 v8::Local<v8::Value> value;
907 if (isKeyValue && !entries->Get(context, i + 1).ToLocal(&value)) continue;
908 v8::Local<v8::Object> wrapper = v8::Object::New(isolate);
909 if (!wrapper->SetPrototypeV2(context, v8::Null(isolate)).FromMaybe(false))
910 continue;
911 createDataProperty(
912 context, wrapper,
913 toV8StringInternalized(isolate, isKeyValue ? "key" : "value"), item);
914 if (isKeyValue) {
915 createDataProperty(context, wrapper,
916 toV8StringInternalized(isolate, "value"), value);
917 }
918 if (!addInternalObject(context, wrapper, V8InternalValueType::kEntry))
919 continue;
920 createDataProperty(context, wrappedEntries, wrappedEntries->Length(),
921 wrapper);
922 }
923 return wrappedEntries;
924}

926v8::MaybeLocal<v8::Array> V8Debugger::privateMethods(
927 v8::Local<v8::Context> context, v8::Local<v8::Value> receiver) {
928 if (!receiver->IsObject()) {
929 return v8::MaybeLocal<v8::Array>();
930 }
931 v8::Isolate* isolate = context->GetIsolate();
932 v8::LocalVector<v8::Value> names(isolate);
933 v8::LocalVector<v8::Value> values(isolate);
934 int filter =
935 static_cast<int>(v8::debug::PrivateMemberFilter::kPrivateMethods);
936 if (!v8::debug::GetPrivateMembers(context, receiver.As<v8::Object>(), filter,
937 &names, &values) ||
938 names.empty()) {
939 return v8::MaybeLocal<v8::Array>();
940 }
941
942 v8::Local<v8::Array> result = v8::Array::New(isolate);
943 if (!result->SetPrototypeV2(context, v8::Null(isolate)).FromMaybe(false))
944 return v8::MaybeLocal<v8::Array>();

946 v8::Local<v8::Value> name = names[i];
947 v8::Local<v8::Value> value = values[i];
948 DCHECK(value->IsFunction());
949 v8::Local<v8::Object> wrapper = v8::Object::New(isolate);
950 if (!wrapper->SetPrototypeV2(context, v8::Null(isolate)).FromMaybe(false))

953 toV8StringInternalized(isolate, "name"), name);
954 createDataProperty(context, wrapper,
955 toV8StringInternalized(isolate, "value"), value);
956 if (!addInternalObject(context, wrapper,
957 V8InternalValueType::kPrivateMethod))
958 continue;
959 createDataProperty(context, result, result->Length(), wrapper);
960 }
961
962 if (!addInternalObject(context, result,
963 V8InternalValueType::kPrivateMethodList))
964 return v8::MaybeLocal<v8::Array>();
965 return result;

1053void V8Debugger::setMaxCallStackSizeToCapture(V8RuntimeAgentImpl* agent,
1054 int size) {
1055 if (size < 0) {
1056 m_maxCallStackSizeToCaptureMap.erase(agent);
1057 } else {
1058 m_maxCallStackSizeToCaptureMap[agent] = size;
1059 }
1060
1061 // The following logic is a bit complicated to decipher because we
1062 // want to retain backwards compatible semantics:
1063 //
1064 // (a) When no `Runtime` domain is enabled, we stick to the default
1065 // maximum call stack size, but don't let V8 collect stack traces
1066 // for uncaught exceptions.
1067 // (b) When `Runtime` is enabled for at least one front-end, we compute
1068 // the maximum of the requested maximum call stack sizes of all the
1069 // front-ends whose `Runtime` domains are enabled (which might be 0),
1070 // and ask V8 to collect stack traces for uncaught exceptions.
1071 //
1072 // The latter allows performance test automation infrastructure to drive
1073 // browser via `Runtime` domain while still minimizing the performance
1074 // overhead of having the inspector attached - see the relevant design
1075 // document https://bit.ly/v8-cheaper-inspector-stack-traces for more
1076 if (m_maxCallStackSizeToCaptureMap.empty()) {
1077 m_maxCallStackSizeToCapture =
1078 V8StackTraceImpl::kDefaultMaxCallStackSizeToCapture;
1079 m_isolate->SetCaptureStackTraceForUncaughtExceptions(false);
1080 } else {
1081 m_maxCallStackSizeToCapture = 0;
1082 for (auto const& pair : m_maxCallStackSizeToCaptureMap) {
1083 if (m_maxCallStackSizeToCapture < pair.second)
1084 m_maxCallStackSizeToCapture = pair.second;
1085 }
1086 m_isolate->SetCaptureStackTraceForUncaughtExceptions(
1087 m_maxCallStackSizeToCapture > 0, m_maxCallStackSizeToCapture);
1088 }
1089}

1366 if (m_allAsyncStacks.size() <= m_maxAsyncCallStacks) return;
1367 size_t halfOfLimitRoundedUp =
1368 m_maxAsyncCallStacks / 2 + m_maxAsyncCallStacks % 2;

1370 m_allAsyncStacks.pop_front();

1377 if (m_asyncTaskStacks.find(*it) == m_asyncTaskStacks.end()) {
1378 it = m_recurringTasks.erase(it);

1391std::shared_ptr<StackFrame> V8Debugger::symbolize(
1392 v8::Local<v8::StackFrame> v8Frame) {
1393 int scriptId = v8Frame->GetScriptId();
1394 auto location = v8Frame->GetLocation();
1395 int lineNumber = location.GetLineNumber();
1396 int columnNumber = location.GetColumnNumber();
1397 CachedStackFrameKey key{scriptId, lineNumber, columnNumber};
1398 auto functionName = toProtocolString(isolate(), v8Frame->GetFunctionName());
1399 auto it = m_cachedStackFrames.find(key);
1400 if (it != m_cachedStackFrames.end() && !it->second.expired()) {
1401 auto stackFrame = it->second.lock();
1402 if (stackFrame->functionName() == functionName) {
1403 DCHECK_EQ(
1404 stackFrame->sourceURL(),
1405 toProtocolString(isolate(), v8Frame->GetScriptNameOrSourceURL()));
1406 return stackFrame;
1407 }
1408 }
1409 auto sourceURL =
1410 toProtocolString(isolate(), v8Frame->GetScriptNameOrSourceURL());
1411 auto hasSourceURLComment =
1412 v8Frame->GetScriptName() != v8Frame->GetScriptNameOrSourceURL();
1413 auto stackFrame = std::make_shared<StackFrame>(
1414 std::move(functionName), scriptId, std::move(sourceURL), lineNumber,
1415 columnNumber, hasSourceURLComment);
1416 m_cachedStackFrames.emplace(key, stackFrame);
1417 return stackFrame;
