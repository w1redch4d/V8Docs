410// Addressing modes represent the "shape" of inputs to an instruction.
411// Many instructions support multiple addressing modes. Addressing modes
412// are encoded into the InstructionCode of the instruction and tell the
413// code generator after register allocation which assembler method to call.
414//
415// We use the following local notation for addressing modes:
416//
417// M = memory operand
418// R = base register
419// N = index register * N for N in {1, 2, 4, 8}
420// I = immediate displacement (32-bit signed integer)
421
422#define TARGET_ADDRESSING_MODE_LIST(V) \
423 V(MR) /* [%r1 ] */ \
424 V(MRI) /* [%r1 + K] */ \
425 V(MR1) /* [%r1 + %r2*1 ] */ \
426 V(MR2) /* [%r1 + %r2*2 ] */ \
427 V(MR4) /* [%r1 + %r2*4 ] */ \
428 V(MR8) /* [%r1 + %r2*8 ] */ \
429 V(MR1I) /* [%r1 + %r2*1 + K] */ \
430 V(MR2I) /* [%r1 + %r2*2 + K] */ \
431 V(MR4I) /* [%r1 + %r2*4 + K] */ \
432 V(MR8I) /* [%r1 + %r2*8 + K] */ \
433 V(M1) /* [ %r2*1 ] */ \
434 V(M2) /* [ %r2*2 ] */ \
435 V(M4) /* [ %r2*4 ] */ \
436 V(M8) /* [ %r2*8 ] */ \
437 V(M1I) /* [ %r2*1 + K] */ \
438 V(M2I) /* [ %r2*2 + K] */ \
439 V(M4I) /* [ %r2*4 + K] */ \
440 V(M8I) /* [ %r2*8 + K] */ \
441 V(Root) /* [%root + K] */ \
442 V(MCR) /* [%compressed_base + %r1] */ \
443 V(MCRI) /* [%compressed_base + %r1 + K] */
