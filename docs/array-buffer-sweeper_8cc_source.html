22size_t ArrayBufferList::Append(ArrayBufferExtension* extension) {
23 if (head_ == nullptr) {
24 DCHECK_NULL(tail_);
25 head_ = tail_ = extension;
26 } else {
27 tail_->set_next(extension);
28 tail_ = extension;
29 }
30
31 const size_t accounting_length = [&] {
32 if (age_ == ArrayBufferExtension::Age::kOld) {
33 return extension->SetOld().accounting_length();
34 } else {
35 return extension->SetYoung().accounting_length();
36 }
37 }();
38 DCHECK_GE(bytes_ + accounting_length, bytes_);
39 bytes_ += accounting_length;
40 extension->set_next(nullptr);
41 return accounting_length;
42}

44void ArrayBufferList::Append(ArrayBufferList& list) {
45 DCHECK_EQ(age_, list.age_);
46
47 if (head_ == nullptr) {
48 DCHECK_NULL(tail_);
49 head_ = list.head_;
50 tail_ = list.tail_;
51 } else if (list.head_) {
52 DCHECK_NOT_NULL(list.tail_);
53 tail_->set_next(list.head_);
54 tail_ = list.tail_;
55 } else {
56 DCHECK_NULL(list.tail_);
57 }
58
59 bytes_ += list.ApproximateBytes();
60 list = ArrayBufferList(age_);
61}

71size_t ArrayBufferList::BytesSlow() const {
72 ArrayBufferExtension* current = head_;
73 size_t sum = 0;
74 while (current) {
75 sum += current->accounting_length();
76 current = current->next();
77 }
78 DCHECK_GE(sum, ApproximateBytes());

88class ArrayBufferSweeper::SweepingState final {
89 enum class Status { kInProgress, kDone };
90
91 public:
92 SweepingState(Heap* heap, ArrayBufferList young, ArrayBufferList old,
93 SweepingType type,
94 TreatAllYoungAsPromoted treat_all_young_as_promoted,
95 uint64_t trace_id);
96
97 ~SweepingState() { DCHECK(job_handle_ && !job_handle_->IsValid()); }
98
99 void SetDone() { status_.store(Status::kDone, std::memory_order_relaxed); }
100 bool IsDone() const {
101 return status_.load(std::memory_order_relaxed) == Status::kDone;
102 }
103
104 void MergeTo(ArrayBufferSweeper* sweeper) {
105 // the worker may see a difference between `young/old_bytes_accounted_` and
106 // `initial_young/old_bytes_` due to concurrent main thread adjustments
107 // (resizing).
108 sweeper->young_bytes_adjustment_while_sweeping_ +=
109 initial_young_bytes_ - young_bytes_accounted_;
110 sweeper->old_bytes_adjustment_while_sweeping_ +=
111 initial_old_bytes_ - old_bytes_accounted_;
112 DCHECK_GE(new_young_.bytes_ +
113 sweeper->young_bytes_adjustment_while_sweeping_ +
114 sweeper->young_.bytes_,
115 0);
116 DCHECK_GE(new_old_.bytes_ + sweeper->old_bytes_adjustment_while_sweeping_ +
117 sweeper->old_.bytes_,
118 0);
119 sweeper->young_.Append(new_young_);
120 sweeper->old_.Append(new_old_);
121 // Apply pending adjustments from resizing and detaching.
122 sweeper->young_.bytes_ +=
123 std::exchange(sweeper->young_bytes_adjustment_while_sweeping_, 0);
124 sweeper->old_.bytes_ +=
125 std::exchange(sweeper->old_bytes_adjustment_while_sweeping_, 0);
126 sweeper->DecrementExternalMemoryCounters(freed_bytes_);
127 }
128
129 void StartBackgroundSweeping() { job_handle_->NotifyConcurrencyIncrease(); }
130 void FinishSweeping() {
131 DCHECK(job_handle_ && job_handle_->IsValid());
132 job_handle_->Join();
133 }
134
135 private:
136 class SweepingJob;
137
138 std::atomic<Status> status_{Status::kInProgress};
139 ArrayBufferList new_young_{ArrayBufferList::Age::kYoung};
140 ArrayBufferList new_old_{ArrayBufferList::Age::kOld};
141 size_t freed_bytes_{0};
142 const uint64_t initial_young_bytes_{0};
143 const uint64_t initial_old_bytes_{0};
144 // Track bytes accounted bytes during sweeping, including freed and promoted
145 // bytes. This is used to compute adjustment when sweeping finishes.
146 uint64_t young_bytes_accounted_{0};
147 uint64_t old_bytes_accounted_{0};
148 std::unique_ptr<JobHandle> job_handle_;
149};

196void ArrayBufferSweeper::SweepingState::SweepingJob::Run(
197 JobDelegate* delegate) {
198 const ThreadKind thread_kind =
199 delegate->IsJoiningThread() ? ThreadKind::kMain : ThreadKind::kBackground;
200 if (treat_all_young_as_promoted_ == TreatAllYoungAsPromoted::kNo) {
201 // Waiting for promoted page iteration is only needed when not all young
202 // array buffers are promoted.
203 GCTracer::Scope::ScopeId scope_id =
204 type_ == SweepingType::kYoung
205 ? thread_kind == ThreadKind::kMain
206 ? GCTracer::Scope::MINOR_MS_SWEEP
207 : GCTracer::Scope::MINOR_MS_BACKGROUND_SWEEPING
208 : thread_kind == ThreadKind::kMain
209 ? GCTracer::Scope::MC_SWEEP
210 : GCTracer::Scope::MC_BACKGROUND_SWEEPING;
211 TRACE_GC_EPOCH_WITH_FLOW(
212 heap_->tracer(), scope_id, thread_kind,
213 heap_->sweeper()->GetTraceIdForFlowEvent(scope_id),
214 TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
215 const bool finished =
216 local_sweeper_.ContributeAndWaitForPromotedPagesIteration(delegate);
217 DCHECK_IMPLIES(delegate->IsJoiningThread(), finished);
218 if (!finished) return;
219 DCHECK(!heap_->sweeper()->IsIteratingPromotedPages());
220 }
221 GCTracer::Scope::ScopeId scope_id =
222 type_ == SweepingType::kYoung
223 ? thread_kind == ThreadKind::kMain
224 ? GCTracer::Scope::YOUNG_ARRAY_BUFFER_SWEEP
225 : GCTracer::Scope::BACKGROUND_YOUNG_ARRAY_BUFFER_SWEEP
226 : thread_kind == ThreadKind::kMain
227 ? GCTracer::Scope::FULL_ARRAY_BUFFER_SWEEP
228 : GCTracer::Scope::BACKGROUND_FULL_ARRAY_BUFFER_SWEEP;
229 TRACE_GC_EPOCH_WITH_FLOW(heap_->tracer(), scope_id, thread_kind, trace_id_,
230 TRACE_EVENT_FLAG_FLOW_IN);
231 Sweep(delegate);
232}

278void ArrayBufferSweeper::RequestSweep(
279 SweepingType type, TreatAllYoungAsPromoted treat_all_young_as_promoted) {
280 DCHECK(!sweeping_in_progress());
281
282 if (young_.IsEmpty() && (old_.IsEmpty() || type == SweepingType::kYoung))
283 return;
284
285 GCTracer::Scope::ScopeId scope_id =
286 type == SweepingType::kYoung
287 ? v8_flags.minor_ms
288 ? GCTracer::Scope::MINOR_MS_FINISH_SWEEP_ARRAY_BUFFERS
289 : GCTracer::Scope::SCAVENGER_SWEEP_ARRAY_BUFFERS
290 : GCTracer::Scope::MC_FINISH_SWEEP_ARRAY_BUFFERS;
291 auto trace_id = GetTraceIdForFlowEvent(scope_id);
292 TRACE_GC_WITH_FLOW(heap_->tracer(), scope_id, trace_id,
293 TRACE_EVENT_FLAG_FLOW_OUT);
294 Prepare(type, treat_all_young_as_promoted, trace_id);
295 DCHECK_IMPLIES(v8_flags.minor_ms && type == SweepingType::kYoung,
296 !heap_->ShouldReduceMemory());
297 if (!heap_->IsTearingDown() && !heap_->ShouldReduceMemory() &&
298 v8_flags.concurrent_array_buffer_sweeping &&
299 heap_->ShouldUseBackgroundThreads()) {
300 state_->StartBackgroundSweeping();
301 } else {
302 Finish();
303 }
304}

306void ArrayBufferSweeper::Prepare(
307 SweepingType type, TreatAllYoungAsPromoted treat_all_young_as_promoted,
308 uint64_t trace_id) {
309 DCHECK(!sweeping_in_progress());
310 DCHECK_IMPLIES(type == SweepingType::kFull,
311 treat_all_young_as_promoted == TreatAllYoungAsPromoted::kYes);
312 switch (type) {
313 case SweepingType::kYoung: {
314 state_ = std::make_unique<SweepingState>(
315 heap_, std::move(young_), ArrayBufferList(ArrayBufferList::Age::kOld),
316 type, treat_all_young_as_promoted, trace_id);
317 young_ = ArrayBufferList(ArrayBufferList::Age::kYoung);
318 } break;
319 case SweepingType::kFull: {
320 state_ = std::make_unique<SweepingState>(
321 heap_, std::move(young_), std::move(old_), type,
322 treat_all_young_as_promoted, trace_id);
323 young_ = ArrayBufferList(ArrayBufferList::Age::kYoung);
324 old_ = ArrayBufferList(ArrayBufferList::Age::kOld);
325 } break;
326 }
327 DCHECK(sweeping_in_progress());
328}

338void ArrayBufferSweeper::ReleaseAll(ArrayBufferList* list) {
339 ArrayBufferExtension* current = list->head_;
340 while (current) {
341 ArrayBufferExtension* next = current->next();
342 const size_t bytes = current->ClearAccountingLength().accounting_length();
343 DecrementExternalMemoryCounters(bytes);
344 FinalizeAndDelete(current);
345 current = next;
346 }
347 *list = ArrayBufferList(list->age_);

350void ArrayBufferSweeper::Append(ArrayBufferExtension* extension) {
351 size_t bytes = extension->accounting_length();
352
353 FinishIfDone();
354
355 switch (extension->age()) {
356 case ArrayBufferExtension::Age::kYoung:

367void ArrayBufferSweeper::Resize(ArrayBufferExtension* extension,
368 int64_t delta) {
369 FinishIfDone();
370
371 const ArrayBufferExtension::AccountingState previous_value =
372 extension->UpdateAccountingLength(delta);
373 UpdateApproximateBytes(delta, previous_value.age());
374 if (delta > 0) {
375 IncrementExternalMemoryCounters(delta);
376 } else {
377 DecrementExternalMemoryCounters(-delta);
378 }
379}

381void ArrayBufferSweeper::Detach(ArrayBufferExtension* extension) {
382 // Finish sweeping here first such that the code below is guaranteed to
383 // observe the same sweeping state.
384 FinishIfDone();
385
386 ArrayBufferExtension::AccountingState previous_value =
387 extension->ClearAccountingLength();
388
389 // We cannot free the extension eagerly here, since extensions are tracked in
390 // a singly linked list. The next GC will remove it automatically.
391
392 UpdateApproximateBytes(-previous_value.accounting_length(),
393 previous_value.age());
394 DecrementExternalMemoryCounters(previous_value.accounting_length());
395}

397void ArrayBufferSweeper::UpdateApproximateBytes(int64_t delta,
398 ArrayBufferExtension::Age age) {
399 switch (age) {
400 case ArrayBufferExtension::Age::kYoung:
401 if (!sweeping_in_progress()) {
402 DCHECK_GE(young_.bytes_, -delta);
403 young_.bytes_ += delta;
404 } else {
405 young_bytes_adjustment_while_sweeping_ += delta;
406 }
407 break;
408 case ArrayBufferExtension::Age::kOld:
409 if (!sweeping_in_progress()) {
410 DCHECK_GE(old_.bytes_, -delta);
411 old_.bytes_ += delta;
412 } else {
413 old_bytes_adjustment_while_sweeping_ += delta;
414 }
415 }
416}

468bool ArrayBufferSweeper::SweepingState::SweepingJob::SweepListFull(
469 JobDelegate* delegate, ArrayBufferList& list,
470 ArrayBufferExtension::Age age) {
471 static constexpr size_t kYieldCheckInterval = 256;
472 static_assert(base::bits::IsPowerOfTwo(kYieldCheckInterval),
473 "kYieldCheckInterval must be power of 2");
474
475 ArrayBufferExtension* current = list.head_;
476
477 ArrayBufferList& new_old = state_.new_old_;
478 size_t freed_bytes = 0;
479 size_t accounted_bytes = 0;
480 size_t swept_extensions = 0;
481
482 while (current) {
483 DCHECK_EQ(list.age_, current->age());
484 if ((swept_extensions++ & (kYieldCheckInterval - 1)) == 0) {
485 if (delegate->ShouldYield()) break;
486 }
487 ArrayBufferExtension* next = current->next();
488
489 if (!current->IsMarked()) {
490 freed_bytes += current->accounting_length();
491 FinalizeAndDelete(current);
492 } else {
493 current->Unmark();
494 accounted_bytes += new_old.Append(current);
495 }
496
497 current = next;
498 }
499
500 state_.freed_bytes_ += freed_bytes;
501 if (age == ArrayBufferExtension::Age::kYoung) {
502 state_.young_bytes_accounted_ += (freed_bytes + accounted_bytes);
503 } else {
504 state_.old_bytes_accounted_ += (freed_bytes + accounted_bytes);
505 }
506
507 list.head_ = current;
508 return !current;
509}

511bool ArrayBufferSweeper::SweepingState::SweepingJob::SweepYoung(
512 JobDelegate* delegate) {
513 static constexpr size_t kYieldCheckInterval = 256;
514 static_assert(base::bits::IsPowerOfTwo(kYieldCheckInterval),
515 "kYieldCheckInterval must be power of 2");
516
517 DCHECK_EQ(SweepingType::kYoung, type_);
518 ArrayBufferExtension* current = young_.head_;
519
520 ArrayBufferList& new_old = state_.new_old_;
521 ArrayBufferList& new_young = state_.new_young_;
522 size_t freed_bytes = 0;
523 size_t accounted_bytes = 0;
524 size_t swept_extensions = 0;
525
526 while (current) {
527 DCHECK_EQ(ArrayBufferExtension::Age::kYoung, current->age());
528 if ((swept_extensions++ & (kYieldCheckInterval - 1)) == 0) {
529 if (delegate->ShouldYield()) break;
530 }
531 ArrayBufferExtension* next = current->next();
532
533 if (!current->IsYoungMarked()) {
534 const size_t bytes = current->accounting_length();
535 FinalizeAndDelete(current);
536 if (bytes) freed_bytes += bytes;
537 } else {
538 if ((treat_all_young_as_promoted_ == TreatAllYoungAsPromoted::kYes) ||
539 current->IsYoungPromoted()) {
540 current->YoungUnmark();
541 accounted_bytes += new_old.Append(current);
542 } else {
543 current->YoungUnmark();
544 accounted_bytes += new_young.Append(current);
545 }
546 }
547
548 current = next;
549 }
550
551 state_.freed_bytes_ += freed_bytes;
552 // Update young/old_bytes_accounted_; the worker may see a difference between
553 // this and `initial_young/old_bytes_` due to concurrent main thread
554 // adjustments.
555 state_.young_bytes_accounted_ += (freed_bytes + accounted_bytes);
556
557 young_.head_ = current;
558 return !current;
559}
