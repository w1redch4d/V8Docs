15TracedNode* TracedNodeBlock::AllocateNode() {
16 DCHECK_NE(used_, capacity_);
17 DCHECK_NE(first_free_node_, kInvalidFreeListNodeIndex);
18 auto* node = at(first_free_node_);
19 first_free_node_ = node->next_free();
20 used_++;
21 DCHECK(!node->is_in_use());
22 return node;
23}

25std::pair<TracedNodeBlock*, TracedNode*> TracedHandles::AllocateNode() {
26 if (V8_UNLIKELY(usable_blocks_.empty())) {
27 RefillUsableNodeBlocks();
28 }
29 TracedNodeBlock* block = usable_blocks_.Front();
30 auto* node = block->AllocateNode();
31 DCHECK(node->IsMetadataCleared());
32 if (V8_UNLIKELY(block->IsFull())) {
33 usable_blocks_.Remove(block);
34 }
35 used_nodes_++;
36 return std::make_pair(block, node);
37}

54bool TracedHandles::IsCppGCHostOld(CppHeap& cpp_heap, Address host) const {
55 DCHECK(host);
56 DCHECK(cpp_heap.generational_gc_supported());
57 auto* host_ptr = reinterpret_cast<void*>(host);
58 auto* page = cppgc::internal::BasePage::FromInnerAddress(&cpp_heap, host_ptr);
59 // TracedReference may be created on stack, in which case assume it's young
60 // and doesn't need to be remembered, since it'll anyway be scanned.
61 if (!page) return false;
62 return !page->ObjectHeaderFromInnerAddress(host_ptr).IsYoung();
63}

65bool TracedHandles::NeedsToBeRemembered(
66 Tagged<Object> object, TracedNode* node, Address* slot,
67 TracedReferenceStoreMode store_mode) const {
68 DCHECK(!node->has_old_host());
69
70 auto* cpp_heap = GetCppHeapIfUnifiedYoungGC(isolate_);
71 if (!cpp_heap) {
72 return false;
73 }
74 if (store_mode == TracedReferenceStoreMode::kInitializingStore) {
75 // Don't record initializing stores.
76 return false;
77 }
78 if (is_marking_) {
79 // If marking is in progress, the marking barrier will be issued later.
80 return false;
81 }
82 if (!HeapLayout::InYoungGeneration(object)) {
83 return false;
84 }
85 return IsCppGCHostOld(*cpp_heap, reinterpret_cast<Address>(slot));
86}

88// Publishes all internal state to be consumed by other threads.
89FullObjectSlot TracedNode::Publish(Tagged<Object> object,
90 bool needs_young_bit_update,
91 bool needs_black_allocation,
92 bool has_old_host, bool is_droppable_value) {
93 DCHECK(IsMetadataCleared());
94
95 flags_ = needs_young_bit_update << IsInYoungList::kShift |
96 has_old_host << HasOldHost::kShift |
97 is_droppable_value << IsDroppable::kShift | 1 << IsInUse::kShift;
98 if (needs_black_allocation) set_markbit();
99 reinterpret_cast<std::atomic<Address>*>(&object_)->store(
100 object.ptr(), std::memory_order_release);
101 return FullObjectSlot(&object_);
102}

104FullObjectSlot TracedHandles::Create(
105 Address value, Address* slot, TracedReferenceStoreMode store_mode,
106 TracedReferenceHandling reference_handling) {
107 DCHECK_NOT_NULL(slot);
108 Tagged<Object> object(value);
109 auto [block, node] = AllocateNode();
110 const bool needs_young_bit_update = NeedsTrackingInYoungNodes(object, node);
111 const bool has_old_host = NeedsToBeRemembered(object, node, slot, store_mode);
112 const bool needs_black_allocation =
113 is_marking_ && store_mode != TracedReferenceStoreMode::kInitializingStore;
114 const bool is_droppable =
115 reference_handling == TracedReferenceHandling::kDroppable;
116 auto result_slot =
117 node->Publish(object, needs_young_bit_update, needs_black_allocation,
118 has_old_host, is_droppable);
119 // Write barrier and young node tracking may be reordered, so move them below
120 // `Publish()`.
121 if (needs_young_bit_update && !block->InYoungList()) {
122 young_blocks_.PushFront(block);
123 block->SetInYoungList(true);
124 DCHECK(block->InYoungList());
125 }
126 if (needs_black_allocation) {
127 WriteBarrier::MarkingFromTracedHandle(object);
128 }
129#ifdef VERIFY_HEAP
130 if (i::v8_flags.verify_heap) {
131 Object::ObjectVerify(*result_slot, isolate_);
132 }
133#endif // VERIFY_HEAP
134 return result_slot;
135}
