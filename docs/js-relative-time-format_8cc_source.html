70 Isolate* isolate, DirectHandle<Map> map, DirectHandle<Object> locales,
71 DirectHandle<Object> input_options) {
72 // 1. Let requestedLocales be ? CanonicalizeLocaleList(locales).
73 Maybe<std::vector<std::string>> maybe_requested_locales =
74 Intl::CanonicalizeLocaleList(isolate, locales);
75 MAYBE_RETURN(maybe_requested_locales, Handle<JSRelativeTimeFormat>());
76 std::vector<std::string> requested_locales =
77 maybe_requested_locales.FromJust();
78
79 // 2. Set options to ? CoerceOptionsToObject(options).
80 DirectHandle<JSReceiver> options;
81 const char* service = "Intl.RelativeTimeFormat";
82 ASSIGN_RETURN_ON_EXCEPTION(
83 isolate, options, CoerceOptionsToObject(isolate, input_options, service));
84
85 // 4. Let opt be a new Record.
86 // 5. Let matcher be ? GetOption(options, "localeMatcher", "string", Â«
87 // "lookup", "best fit" Â», "best fit").
88 // 6. Set opt.[[localeMatcher]] to matcher.
89 Maybe<Intl::MatcherOption> maybe_locale_matcher =
90 Intl::GetLocaleMatcher(isolate, options, service);
91 MAYBE_RETURN(maybe_locale_matcher, MaybeHandle<JSRelativeTimeFormat>());
92 Intl::MatcherOption matcher = maybe_locale_matcher.FromJust();
93
94 // 7. Let _numberingSystem_ be ? GetOption(_options_, `"numberingSystem"`,
95 // `"string"`, *undefined*, *undefined*).
96 std::unique_ptr<char[]> numbering_system_str = nullptr;
97 Maybe<bool> maybe_numberingSystem = Intl::GetNumberingSystem(
98 isolate, options, service, &numbering_system_str);
99 // 8. If _numberingSystem_ is not *undefined*, then
100 // a. If _numberingSystem_ does not match the
101 // `(3*8alphanum) *("-" (3*8alphanum))` sequence, throw a *RangeError*
102 // exception.
103 MAYBE_RETURN(maybe_numberingSystem, MaybeHandle<JSRelativeTimeFormat>());
104
105 // 9. Set _opt_.[[nu]] to _numberingSystem_.
106
107 // 10. Let localeData be %RelativeTimeFormat%.[[LocaleData]].
108 // 11. Let r be
109 // ResolveLocale(%RelativeTimeFormat%.[[AvailableLocales]],
110 // requestedLocales, opt,
111 // %RelativeTimeFormat%.[[RelevantExtensionKeys]], localeData).
112 Maybe<Intl::ResolvedLocale> maybe_resolve_locale =
113 Intl::ResolveLocale(isolate, JSRelativeTimeFormat::GetAvailableLocales(),
114 requested_locales, matcher, {"nu"});
115 if (maybe_resolve_locale.IsNothing()) {
116 THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
117 }
118 Intl::ResolvedLocale r = maybe_resolve_locale.FromJust();
119
120 UErrorCode status = U_ZERO_ERROR;
121
122 icu::Locale icu_locale = r.icu_locale;
123 if (numbering_system_str != nullptr) {
124 auto nu_extension_it = r.extensions.find("nu");
125 if (nu_extension_it != r.extensions.end() &&
126 nu_extension_it->second != numbering_system_str.get()) {
127 icu_locale.setUnicodeKeywordValue("nu", nullptr, status);
128 DCHECK(U_SUCCESS(status));
129 }
130 }
131 // 12. Let locale be r.[[Locale]].
132 Maybe<std::string> maybe_locale_str = Intl::ToLanguageTag(icu_locale);
133 MAYBE_RETURN(maybe_locale_str, MaybeHandle<JSRelativeTimeFormat>());
134
135 // 13. Set relativeTimeFormat.[[Locale]] to locale.
136 DirectHandle<String> locale_str =
137 isolate->factory()->NewStringFromAsciiChecked(
138 maybe_locale_str.FromJust().c_str());
139
140 // 14. Set relativeTimeFormat.[[NumberingSystem]] to r.[[nu]].
141 if (numbering_system_str != nullptr &&
142 Intl::IsValidNumberingSystem(numbering_system_str.get())) {
143 icu_locale.setUnicodeKeywordValue("nu", numbering_system_str.get(), status);
144 DCHECK(U_SUCCESS(status));
145 }
146 // 15. Let dataLocale be r.[[DataLocale]].
147
148 // 16. Let s be ? GetOption(options, "style", "string",
149 // Â«"long", "short", "narrow"Â», "long").
150 Maybe<Style> maybe_style = GetStringOption<Style>(
151 isolate, options, "style", service, {"long", "short", "narrow"},
152 {Style::LONG, Style::SHORT, Style::NARROW}, Style::LONG);
153 MAYBE_RETURN(maybe_style, MaybeHandle<JSRelativeTimeFormat>());
154 Style style_enum = maybe_style.FromJust();
155
156 // 17. Set relativeTimeFormat.[[Style]] to s.
157
158 // 18. Let numeric be ? GetOption(options, "numeric", "string",
159 // Â«"always", "auto"Â», "always").
160 Maybe<Numeric> maybe_numeric = GetStringOption<Numeric>(
161 isolate, options, "numeric", service, {"always", "auto"},
162 {Numeric::ALWAYS, Numeric::AUTO}, Numeric::ALWAYS);
163 MAYBE_RETURN(maybe_numeric, MaybeHandle<JSRelativeTimeFormat>());
164 Numeric numeric_enum = maybe_numeric.FromJust();
165
166 // 19. Set relativeTimeFormat.[[Numeric]] to numeric.
167
168 // 23. Let relativeTimeFormat.[[NumberFormat]] be
169 // ? Construct(%NumberFormat%, Â« nfLocale, nfOptions Â»).
170 icu::NumberFormat* number_format =
171 icu::NumberFormat::createInstance(icu_locale, UNUM_DECIMAL, status);

173 // Data build filter files excluded data in "rbnf_tree" since ECMA402 does
174 // not support "algorithmic" numbering systems. Therefore we may get the
175 // U_MISSING_RESOURCE_ERROR here. Fallback to locale without the numbering
176 // system and create the object again.
177 if (status == U_MISSING_RESOURCE_ERROR) {
178 delete number_format;
179 status = U_ZERO_ERROR;
180 icu_locale.setUnicodeKeywordValue("nu", nullptr, status);
181 DCHECK(U_SUCCESS(status));
182 number_format =
183 icu::NumberFormat::createInstance(icu_locale, UNUM_DECIMAL, status);
184 }
185 if (U_FAILURE(status) || number_format == nullptr) {
186 delete number_format;
187 THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
188 }
189 }
190
191 if (number_format->getDynamicClassID() ==
192 icu::DecimalFormat::getStaticClassID()) {
193 icu::DecimalFormat* decimal_format =
194 static_cast<icu::DecimalFormat*>(number_format);
195 decimal_format->setMinimumGroupingDigits(-2);
196 }
197
198 // Change UDISPCTX_CAPITALIZATION_NONE to other values if
199 // ECMA402 later include option to change capitalization.
200 // Ref: https://github.com/tc39/proposal-intl-relative-time/issues/11
201 std::shared_ptr<icu::RelativeDateTimeFormatter> icu_formatter =
202 std::make_shared<icu::RelativeDateTimeFormatter>(
203 icu_locale, number_format, toIcuStyle(style_enum),
204 UDISPCTX_CAPITALIZATION_NONE, status);
205 if (U_FAILURE(status) || icu_formatter == nullptr) {
206 THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
207 }
208
209 DirectHandle<String> numbering_system_string =
210 isolate->factory()->NewStringFromAsciiChecked(
211 Intl::GetNumberingSystem(icu_locale).c_str());
212
213 DirectHandle<Managed<icu::RelativeDateTimeFormatter>> managed_formatter =
214 Managed<icu::RelativeDateTimeFormatter>::From(isolate, 0,
215 std::move(icu_formatter));
216
217 Handle<JSRelativeTimeFormat> relative_time_format_holder =
218 Cast<JSRelativeTimeFormat>(
219 isolate->factory()->NewFastOrSlowJSObjectFromMap(map));
220
221 DisallowGarbageCollection no_gc;
222 relative_time_format_holder->set_flags(0);
223 relative_time_format_holder->set_locale(*locale_str);
224 relative_time_format_holder->set_numberingSystem(*numbering_system_string);
225 relative_time_format_holder->set_numeric(numeric_enum);
226 relative_time_format_holder->set_icu_formatter(*managed_formatter);
227
228 // 25. Return relativeTimeFormat.
229 return relative_time_format_holder;
230}

306bool GetURelativeDateTimeUnit(DirectHandle<String> unit,
307 URelativeDateTimeUnit* unit_enum) {
308 std::unique_ptr<char[]> unit_str = unit->ToCString();
309 if ((strcmp("second", unit_str.get()) == 0) ||
310 (strcmp("seconds", unit_str.get()) == 0)) {
311 *unit_enum = UDAT_REL_UNIT_SECOND;
312 } else if ((strcmp("minute", unit_str.get()) == 0) ||
313 (strcmp("minutes", unit_str.get()) == 0)) {
314 *unit_enum = UDAT_REL_UNIT_MINUTE;
315 } else if ((strcmp("hour", unit_str.get()) == 0) ||
316 (strcmp("hours", unit_str.get()) == 0)) {
317 *unit_enum = UDAT_REL_UNIT_HOUR;
318 } else if ((strcmp("day", unit_str.get()) == 0) ||
319 (strcmp("days", unit_str.get()) == 0)) {
320 *unit_enum = UDAT_REL_UNIT_DAY;
321 } else if ((strcmp("week", unit_str.get()) == 0) ||
322 (strcmp("weeks", unit_str.get()) == 0)) {
323 *unit_enum = UDAT_REL_UNIT_WEEK;
324 } else if ((strcmp("month", unit_str.get()) == 0) ||
325 (strcmp("months", unit_str.get()) == 0)) {
326 *unit_enum = UDAT_REL_UNIT_MONTH;
327 } else if ((strcmp("quarter", unit_str.get()) == 0) ||
328 (strcmp("quarters", unit_str.get()) == 0)) {
329 *unit_enum = UDAT_REL_UNIT_QUARTER;
330 } else if ((strcmp("year", unit_str.get()) == 0) ||
331 (strcmp("years", unit_str.get()) == 0)) {
332 *unit_enum = UDAT_REL_UNIT_YEAR;
333 } else {
334 return false;
335 }
336 return true;
337}

339template <typename T>
340MaybeHandle<T> FormatCommon(
341 Isolate* isolate, DirectHandle<JSRelativeTimeFormat> format,
342 Handle<Object> value_obj, Handle<Object> unit_obj, const char* func_name,
343 MaybeHandle<T> (*formatToResult)(Isolate*,
344 const icu::FormattedRelativeDateTime&,
345 DirectHandle<String>, bool)) {
346 // 3. Let value be ? ToNumber(value).
347 DirectHandle<Object> value;
348 ASSIGN_RETURN_ON_EXCEPTION(isolate, value,
349 Object::ToNumber(isolate, value_obj));
350 double number = Object::NumberValue(*value);
351 // 4. Let unit be ? ToString(unit).
352 Handle<String> unit;
353 ASSIGN_RETURN_ON_EXCEPTION(isolate, unit,
354 Object::ToString(isolate, unit_obj));
355 // 4. If isFinite(value) is false, then throw a RangeError exception.
356 if (!std::isfinite(number)) {
357 THROW_NEW_ERROR(
358 isolate, NewRangeError(
359 MessageTemplate::kNotFiniteNumber,
360 isolate->factory()->NewStringFromAsciiChecked(func_name)));
361 }
362 icu::RelativeDateTimeFormatter* formatter = format->icu_formatter()->raw();
363 DCHECK_NOT_NULL(formatter);
364 URelativeDateTimeUnit unit_enum;
365 if (!GetURelativeDateTimeUnit(unit, &unit_enum)) {
366 THROW_NEW_ERROR(
367 isolate,
368 NewRangeError(MessageTemplate::kInvalidUnit,
369 isolate->factory()->NewStringFromAsciiChecked(func_name),
370 unit));
371 }
372 UErrorCode status = U_ZERO_ERROR;
373 icu::FormattedRelativeDateTime formatted =
374 (format->numeric() == JSRelativeTimeFormat::Numeric::ALWAYS)
375 ? formatter->formatNumericToValue(number, unit_enum, status)
376 : formatter->formatToValue(number, unit_enum, status);
377 if (U_FAILURE(status)) {
378 THROW_NEW_ERROR(isolate, NewTypeError(MessageTemplate::kIcuError));
379 }
380 return formatToResult(isolate, formatted, UnitAsString(isolate, unit_enum),
381 IsNaN(*value));
382}

422MaybeHandle<JSArray> FormatToJSArray(
423 Isolate* isolate, const icu::FormattedRelativeDateTime& formatted,
424 DirectHandle<String> unit, bool is_nan) {
425 UErrorCode status = U_ZERO_ERROR;
426 icu::UnicodeString string = formatted.toString(status);
427
428 Factory* factory = isolate->factory();
429 Handle<JSArray> array = factory->NewJSArray(0);
430 icu::ConstrainedFieldPosition cfpos;
431 cfpos.constrainCategory(UFIELD_CATEGORY_NUMBER);
432 int32_t index = 0;
433
434 int32_t previous_end = 0;
435 DirectHandle<String> substring;
436 std::vector<std::pair<int32_t, int32_t>> groups;
437 while (formatted.nextPosition(cfpos, status) && U_SUCCESS(status)) {
438 int32_t category = cfpos.getCategory();
439 int32_t field = cfpos.getField();
440 int32_t start = cfpos.getStart();
441 int32_t limit = cfpos.getLimit();
442 if (category == UFIELD_CATEGORY_NUMBER) {
443 if (field == UNUM_GROUPING_SEPARATOR_FIELD) {
444 groups.push_back(std::pair<int32_t, int32_t>(start, limit));
445 continue;
446 }
447 if (start > previous_end) {
448 Maybe<bool> maybe_added =
449 AddLiteral(isolate, array, string, index++, previous_end, start);
450 MAYBE_RETURN(maybe_added, Handle<JSArray>());
451 }
452 if (field == UNUM_INTEGER_FIELD) {
453 for (auto start_limit : groups) {
454 if (start_limit.first > start) {
455 Maybe<bool> maybe_added =
456 AddUnit(isolate, array, string, index++,
457 NumberFormatSpan(field, start, start_limit.first), unit,
458 is_nan);
459 MAYBE_RETURN(maybe_added, Handle<JSArray>());
460 maybe_added =
461 AddUnit(isolate, array, string, index++,
462 NumberFormatSpan(UNUM_GROUPING_SEPARATOR_FIELD,
463 start_limit.first, start_limit.second),
464 unit, is_nan);
465 MAYBE_RETURN(maybe_added, Handle<JSArray>());
466 start = start_limit.second;
467 }
468 }
469 }
470 Maybe<bool> maybe_added =
471 AddUnit(isolate, array, string, index++,
472 NumberFormatSpan(field, start, limit), unit, is_nan);
473 MAYBE_RETURN(maybe_added, Handle<JSArray>());
474 previous_end = limit;
475 }
476 }
477 if (U_FAILURE(status)) {
478 THROW_NEW_ERROR(isolate, NewTypeError(MessageTemplate::kIcuError));
479 }
480 if (string.length() > previous_end) {
481 Maybe<bool> maybe_added = AddLiteral(isolate, array, string, index,
482 previous_end, string.length());
483 MAYBE_RETURN(maybe_added, Handle<JSArray>());
484 }
485
486 JSObject::ValidateElements(*array);
487 return array;
488}
