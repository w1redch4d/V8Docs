82// static
83int HashTableBase::ComputeCapacity(int at_least_space_for) {
84 // Add 50% slack to make slot collisions sufficiently unlikely.
85 // See matching computation in HashTable::HasSufficientCapacityToAdd().
86 // Must be kept in sync with CodeStubAssembler::HashTableComputeCapacity().
87 int raw_cap = at_least_space_for + (at_least_space_for >> 1);
88 int capacity = base::bits::RoundUpToPowerOfTwo32(raw_cap);
89 return std::max({capacity, kMinCapacity});
90}

133// Find entry for key otherwise return kNotFound.
134template <typename Derived, typename Shape>
135InternalIndex HashTable<Derived, Shape>::FindEntry(PtrComprCageBase cage_base,
136 ReadOnlyRoots roots, Key key,
137 int32_t hash) {
138 DisallowGarbageCollection no_gc;
139 uint32_t capacity = Capacity();
140 uint32_t count = 1;
141 Tagged<Object> undefined = roots.undefined_value();
142 Tagged<Object> the_hole = roots.the_hole_value();
143 DCHECK_EQ(TodoShape::Hash(roots, key), static_cast<uint32_t>(hash));
144 // EnsureCapacity will guarantee the hash table is never full.
145 for (InternalIndex entry = FirstProbe(hash, capacity);;
146 entry = NextProbe(entry, count++, capacity)) {
147 Tagged<Object> element = KeyAt(cage_base, entry);
148 // Empty entry. Uses raw unchecked accessors because it is called by the
149 // string table during bootstrapping.
150 if (element == undefined) return InternalIndex::NotFound();
151 if (TodoShape::kMatchNeedsHoleCheck && element == the_hole) continue;
152 if (TodoShape::IsMatch(key, element)) return entry;
153 }
154}

302template <typename IsolateT>
303Handle<NameToIndexHashTable> NameToIndexHashTable::Add(
304 IsolateT* isolate, Handle<NameToIndexHashTable> table,
305 DirectHandle<Name> key, int32_t index) {
306 DCHECK_GE(index, 0);
307 // Validate that the key is absent.
308 SLOW_DCHECK(table->FindEntry(isolate, key).is_not_found());
309 // Check whether the dictionary should be extended.
310 table = EnsureCapacity(isolate, table);
311 DisallowGarbageCollection no_gc;
312 Tagged<NameToIndexHashTable> raw_table = *table;
313 // Compute the key object.
314 InternalIndex entry = raw_table->FindInsertionEntry(isolate, key->hash());
315 raw_table->set(EntryToIndex(entry), *key);
316 raw_table->set(EntryToValueIndex(entry), Smi::FromInt(index));
317 raw_table->ElementAdded();
318 return table;
319}
