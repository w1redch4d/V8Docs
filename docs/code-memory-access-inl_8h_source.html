48WritableJitAllocation::WritableJitAllocation(
49 Address addr, size_t size, ThreadIsolation::JitAllocationType type,
50 JitAllocationSource source, bool enforce_write_api)
51 : address_(addr),
52 // The order of these is important. We need to create the write scope
53 // before we lookup the Jit page, since the latter will take a mutex in
54 // protected memory.
55 write_scope_("WritableJitAllocation"),
56 page_ref_(ThreadIsolation::LookupJitPage(addr, size)),
57 allocation_(source == JitAllocationSource::kRegister
58 ? page_ref_->RegisterAllocation(addr, size, type)
59 : page_ref_->LookupAllocation(addr, size, type)),
60 enforce_write_api_(enforce_write_api) {
61#ifdef DEBUG
62 if (enforce_write_api_) {
63 // Reset the write scope for debugging. We'll create fine-grained scopes in
64 // all Write functions of this class instead.
65 write_scope_.reset();
66 }
67#else
68 // Suppress -Wunused-private-field warning.
69 (void)enforce_write_api_;
70#endif
71}

130template <typename T, size_t offset>
131void WritableJitAllocation::WriteHeaderSlot(T value) {
132 std::optional<RwxMemoryWriteScope> write_scope =
133 WriteScopeForApiEnforcement();
134 // This assert is no strict requirement, it just guards against
135 // non-implemented functionality.
136 static_assert(!is_taggable_v<T>);
137
138 if constexpr (offset == HeapObject::kMapOffset) {
139 TaggedField<T, offset>::Relaxed_Store_Map_Word(
140 HeapObject::FromAddress(address_), value);
141 } else {
142 WriteMaybeUnalignedValue<T>(address_ + offset, value);
143 }
144}

191template <typename T>
192V8_INLINE void WritableJitAllocation::WriteHeaderSlot(Address address, T value,
193 RelaxedStoreTag tag) {
194 CHECK_EQ(allocation_.Type(),
195 ThreadIsolation::JitAllocationType::kInstructionStream);
196 size_t offset = address - address_;
197 Tagged<T> tagged(value);
198 switch (offset) {
199 case InstructionStream::kCodeOffset:
200 WriteProtectedPointerHeaderSlot<T, InstructionStream::kCodeOffset>(tagged,
201 tag);
202 break;
203 case InstructionStream::kRelocationInfoOffset:
204 WriteProtectedPointerHeaderSlot<T,
205 InstructionStream::kRelocationInfoOffset>(
206 tagged, tag);
207 break;
208 default:
209 UNREACHABLE();
210 }
211}
