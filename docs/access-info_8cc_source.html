34bool CanInlinePropertyAccess(MapRef map, AccessMode access_mode) {
35 // We can inline property access to prototypes of all primitives, except
36 // the special Oddball ones that have no wrapper counterparts (i.e. Null,
37 // Undefined and TheHole).
38 // We can only inline accesses to dictionary mode holders if the access is a
39 // load and the holder is a prototype. The latter ensures a 1:1
40 // relationship between the map and the object (and therefore the property
41 // dictionary).
42 static_assert(ODDBALL_TYPE == LAST_PRIMITIVE_HEAP_OBJECT_TYPE);
43 if (IsBooleanMap(*map.object())) return true;
44 if (map.instance_type() < LAST_PRIMITIVE_HEAP_OBJECT_TYPE) return true;
45 if (IsJSObjectMap(*map.object())) {
46 if (map.is_dictionary_map()) {
47 if (!V8_DICT_PROPERTY_CONST_TRACKING_BOOL) return false;
48 return access_mode == AccessMode::kLoad &&
49 map.object()->is_prototype_map();
50 }
51 return !map.object()->has_named_interceptor() &&
52 // TODO(verwaest): Allowlist contexts to which we have access.
53 !map.is_access_check_needed();
54 }
55 return false;
56}

277bool PropertyAccessInfo::Merge(PropertyAccessInfo const* that,
278 AccessMode access_mode, Zone* zone) {
279 if (kind_ != that->kind_) return false;
280 if (!OptionalRefEquals(holder_, that->holder_)) return false;
281
282 switch (kind_) {
283 case kInvalid:
284 DCHECK_EQ(that->kind_, kInvalid);
285 return true;
286
287 case kDataField:
288 case kFastDataConstant: {
289 // Check if we actually access the same field (we use the
290 // GetFieldAccessStubKey method here just like the ICs do
291 // since that way we only compare the relevant bits of the
292 // field indices).
293 if (field_index_.GetFieldAccessStubKey() !=
294 that->field_index_.GetFieldAccessStubKey()) {
295 return false;
296 }
297
298 switch (access_mode) {
299 case AccessMode::kHas:
300 case AccessMode::kLoad: {
301 if (!field_representation_.Equals(that->field_representation_)) {
302 if (field_representation_.IsDouble() ||
303 that->field_representation_.IsDouble()) {
304 return false;
305 }
306 field_representation_ = Representation::Tagged();
307 }
308 if (!OptionalRefEquals(field_map_, that->field_map_)) {
309 field_map_ = {};
310 }
311 break;
312 }
313 case AccessMode::kStore:
314 case AccessMode::kStoreInLiteral:
315 case AccessMode::kDefine: {
316 // For stores, the field map and field representation information
317 // must match exactly, otherwise we cannot merge the stores. We
318 // also need to make sure that in case of transitioning stores,
319 // the transition targets match.
320 if (!OptionalRefEquals(field_map_, that->field_map_) ||
321 !field_representation_.Equals(that->field_representation_) ||
322 !OptionalRefEquals(transition_map_, that->transition_map_)) {
323 return false;
324 }
325 break;
326 }
327 }
328
329 field_type_ = Type::Union(field_type_, that->field_type_, zone);
330 AppendVector(&lookup_start_object_maps_, that->lookup_start_object_maps_);
331 AppendVector(&unrecorded_dependencies_, that->unrecorded_dependencies_);
332 return true;
333 }
334
335 case kDictionaryProtoAccessorConstant:
336 case kFastAccessorConstant: {
337 // Check if we actually access the same constant.
338 if (!OptionalRefEquals(constant_, that->constant_)) return false;
339
340 DCHECK(unrecorded_dependencies_.empty());
341 DCHECK(that->unrecorded_dependencies_.empty());
342 AppendVector(&lookup_start_object_maps_, that->lookup_start_object_maps_);
343 return true;
344 }
345
346 case kDictionaryProtoDataConstant: {
347 DCHECK_EQ(AccessMode::kLoad, access_mode);
348 if (dictionary_index_ != that->dictionary_index_) return false;
349 AppendVector(&lookup_start_object_maps_, that->lookup_start_object_maps_);
350 return true;
351 }
352
353 case kNotFound:
354 case kStringLength:
355 case kStringWrapperLength: {
356 DCHECK(unrecorded_dependencies_.empty());
357 DCHECK(that->unrecorded_dependencies_.empty());
358 AppendVector(&lookup_start_object_maps_, that->lookup_start_object_maps_);
359 return true;
360 }
361 case kModuleExport:
362 return false;
363 }
364}

380bool AccessInfoFactory::ComputeElementAccessInfos(
381 ElementAccessFeedback const& feedback,
382 ZoneVector<ElementAccessInfo>* access_infos) const {
383 AccessMode access_mode = feedback.keyed_mode().access_mode();
384 if (access_mode == AccessMode::kLoad || access_mode == AccessMode::kHas) {
385 // For polymorphic loads of similar elements kinds (i.e. all tagged or all
386 // double), always use the "worst case" code without a transition. This is
387 // much faster than transitioning the elements to the worst case, trading a
388 // TransitionElementsKind for a CheckMaps, avoiding mutation of the array.
389 std::optional<ElementAccessInfo> access_info =
390 ConsolidateElementLoad(feedback);
391 if (access_info.has_value()) {
392 access_infos->push_back(*access_info);
393 return true;
394 }
395 }
396
397 for (auto const& group : feedback.transition_groups()) {
398 DCHECK(!group.empty());
399 OptionalMapRef target = group.front();
400 std::optional<ElementAccessInfo> access_info =
401 ComputeElementAccessInfo(target.value(), access_mode);
402 if (!access_info.has_value()) return false;
403
404 for (size_t i = 1; i < group.size(); ++i) {
405 OptionalMapRef map_ref = group[i];
406 if (!map_ref.has_value()) continue;
407 access_info->AddTransitionSource(map_ref.value());
408 }
409 access_infos->push_back(*access_info);
410 }
411 return true;
412}

414PropertyAccessInfo AccessInfoFactory::ComputeDataFieldAccessInfo(
415 MapRef receiver_map, MapRef map, NameRef name, OptionalJSObjectRef holder,
416 InternalIndex descriptor, AccessMode access_mode) const {
417 DCHECK(descriptor.is_found());
418 // TODO(jgruber,v8:7790): Use DescriptorArrayRef instead.
419 DirectHandle<DescriptorArray> descriptors =
420 map.instance_descriptors(broker()).object();
421 PropertyDetails const details = descriptors->GetDetails(descriptor);
422 int index = descriptors->GetFieldIndex(descriptor);
423 Representation details_representation = details.representation();
424 if (details_representation.IsNone()) {
425 // The ICs collect feedback in PREMONOMORPHIC state already,
426 // but at this point the {receiver_map} might still contain
427 // fields for which the representation has not yet been
428 // determined by the runtime. So we need to catch this case
429 // here and fall back to use the regular IC logic instead.
430 return Invalid();
431 }
432 FieldIndex field_index = FieldIndex::ForPropertyIndex(*map.object(), index,
433 details_representation);
434 // Private brands are used when loading private methods, which are stored in a
435 // BlockContext, an internal object.
436 Type field_type = name.object()->IsPrivateBrand() ? Type::OtherInternal()
437 : Type::NonInternal();
438 OptionalMapRef field_map;
439
440 ZoneVector<CompilationDependency const*> unrecorded_dependencies(zone());
441
442 Handle<FieldType> descriptors_field_type =
443 broker()->CanonicalPersistentHandle(
444 descriptors->GetFieldType(descriptor));
445 OptionalObjectRef descriptors_field_type_ref =
446 TryMakeRef<Object>(broker(), descriptors_field_type);
447 if (!descriptors_field_type_ref.has_value()) return Invalid();
448
449 // Note: FindFieldOwner may be called multiple times throughout one
450 // compilation. This is safe since its result is fixed for a given map and
451 // descriptor.
452 MapRef field_owner_map = map.FindFieldOwner(broker(), descriptor);
453
454 if (details_representation.IsSmi()) {
455 field_type = Type::SignedSmall();
456 unrecorded_dependencies.push_back(
457 dependencies()->FieldRepresentationDependencyOffTheRecord(
458 map, field_owner_map, descriptor, details_representation));
459 } else if (details_representation.IsDouble()) {
460 field_type = type_cache_->kFloat64;
461 unrecorded_dependencies.push_back(
462 dependencies()->FieldRepresentationDependencyOffTheRecord(
463 map, field_owner_map, descriptor, details_representation));
464 } else if (details_representation.IsHeapObject()) {
465 if (IsNone(*descriptors_field_type)) {
466 // Cleared field-types are pre-monomorphic states. The field type was
467 // garbge collected and we need to record an updated type.
468 static_assert(FieldType::kFieldTypesCanBeClearedOnGC);
469 switch (access_mode) {
470 case AccessMode::kStore:
471 case AccessMode::kStoreInLiteral:
472 case AccessMode::kDefine:
473 return Invalid();
474 case AccessMode::kLoad:
475 case AccessMode::kHas:
476 break;
477 }
478 }
479 unrecorded_dependencies.push_back(
480 dependencies()->FieldRepresentationDependencyOffTheRecord(
481 map, field_owner_map, descriptor, details_representation));
482 if (IsClass(*descriptors_field_type)) {
483 // Remember the field map, and try to infer a useful type.
484 OptionalMapRef maybe_field_map =
485 TryMakeRef(broker(), FieldType::AsClass(*descriptors_field_type));
486 if (!maybe_field_map.has_value()) return Invalid();
487 field_type = Type::For(maybe_field_map.value(), broker());
488 field_map = maybe_field_map;
489 }
490 } else {
491 CHECK(details_representation.IsTagged());
492 }
493 // TODO(turbofan): We may want to do this only depending on the use
494 // of the access info.
495 unrecorded_dependencies.push_back(
496 dependencies()->FieldTypeDependencyOffTheRecord(
497 map, field_owner_map, descriptor,
498 descriptors_field_type_ref.value()));
499
500 PropertyConstness constness =
501 dependencies()->DependOnFieldConstness(map, field_owner_map, descriptor);
502
503 switch (constness) {
504 case PropertyConstness::kMutable:
505 return PropertyAccessInfo::DataField(
506 broker(), zone(), receiver_map, std::move(unrecorded_dependencies),
507 field_index, details_representation, field_type, field_owner_map,
508 field_map, holder, {});
509
510 case PropertyConstness::kConst:
511 return PropertyAccessInfo::FastDataConstant(
512 zone(), receiver_map, std::move(unrecorded_dependencies), field_index,
513 details_representation, field_type, field_owner_map, field_map,
514 holder, {});
515 }
516 UNREACHABLE();
517}

523PropertyAccessInfo AccessorAccessInfoHelper(
524 Isolate* isolate, Zone* zone, JSHeapBroker* broker,
525 const AccessInfoFactory* ai_factory, MapRef receiver_map, NameRef name,
526 MapRef holder_map, OptionalJSObjectRef holder, AccessMode access_mode,
527 AccessorsObjectGetter get_accessors) {
528 if (holder_map.instance_type() == JS_MODULE_NAMESPACE_TYPE) {
529 DCHECK(holder_map.object()->is_prototype_map());
530 DirectHandle<PrototypeInfo> proto_info = broker->CanonicalPersistentHandle(
531 Cast<PrototypeInfo>(holder_map.object()->prototype_info()));
532 DirectHandle<JSModuleNamespace> module_namespace =
533 broker->CanonicalPersistentHandle(
534 Cast<JSModuleNamespace>(proto_info->module_namespace()));
535 Handle<Cell> cell = broker->CanonicalPersistentHandle(
536 Cast<Cell>(module_namespace->module()->exports()->Lookup(
537 isolate, name.object(),
538 Smi::ToInt(Object::GetHash(*name.object())))));
539 if (IsAnyStore(access_mode)) {
540 // ES#sec-module-namespace-exotic-objects-set-p-v-receiver
541 // ES#sec-module-namespace-exotic-objects-defineownproperty-p-desc
542 //
543 // Storing to a module namespace object is always an error or a no-op in
544 // JS.
545 return PropertyAccessInfo::Invalid(zone);
546 }
547 if (IsTheHole(cell->value(kRelaxedLoad), isolate)) {
548 // This module has not been fully initialized yet.
549 return PropertyAccessInfo::Invalid(zone);
550 }
551 OptionalCellRef cell_ref = TryMakeRef(broker, cell);
552 if (!cell_ref.has_value()) {
553 return PropertyAccessInfo::Invalid(zone);
554 }
555 return PropertyAccessInfo::ModuleExport(zone, receiver_map,
556 cell_ref.value());
557 }
558 if (access_mode == AccessMode::kHas) {
559 // kHas is not supported for dictionary mode objects.
560 DCHECK(!holder_map.is_dictionary_map());
561
562 // HasProperty checks don't call getter/setters, existence is sufficient.
563 return PropertyAccessInfo::FastAccessorConstant(zone, receiver_map, holder,
564 {}, {});
565 }
566 DirectHandle<Object> maybe_accessors = get_accessors();
567 if (!IsAccessorPair(*maybe_accessors)) {
568 return PropertyAccessInfo::Invalid(zone);
569 }
570 DirectHandle<AccessorPair> accessors = Cast<AccessorPair>(maybe_accessors);
571 Handle<Object> accessor = broker->CanonicalPersistentHandle(
572 access_mode == AccessMode::kLoad ? accessors->getter(kAcquireLoad)
573 : accessors->setter(kAcquireLoad));
574
575 OptionalObjectRef accessor_ref = TryMakeRef(broker, accessor);
576 if (!accessor_ref.has_value()) return PropertyAccessInfo::Invalid(zone);
577
578 OptionalJSObjectRef api_holder_ref;
579 if (!IsJSFunction(*accessor)) {
580 CallOptimization optimization(broker->local_isolate_or_isolate(), accessor);
581 if (!optimization.is_simple_api_call() ||
582 optimization.IsCrossContextLazyAccessorPair(
583 *broker->target_native_context().object(), *holder_map.object())) {
584 return PropertyAccessInfo::Invalid(zone);
585 }
586 if (DEBUG_BOOL && holder.has_value()) {
587 std::optional<Tagged<NativeContext>> holder_creation_context =
588 holder->object()->GetCreationContext();
589 CHECK(holder_creation_context.has_value());
590 CHECK_EQ(*broker->target_native_context().object(),
591 holder_creation_context.value());
592 }
593
594 CallOptimization::HolderLookup holder_lookup;
595 Handle<JSObject> api_holder = broker->CanonicalPersistentHandle(
596 optimization.LookupHolderOfExpectedType(
597 broker->local_isolate_or_isolate(), receiver_map.object(),
598 &holder_lookup));
599 if (holder_lookup == CallOptimization::kHolderNotFound) {
600 return PropertyAccessInfo::Invalid(zone);
601 }
602 DCHECK_IMPLIES(holder_lookup == CallOptimization::kHolderIsReceiver,
603 api_holder.is_null());
604 DCHECK_IMPLIES(holder_lookup == CallOptimization::kHolderFound,
605 !api_holder.is_null());
606
607 if (!api_holder.is_null()) {
608 api_holder_ref = TryMakeRef(broker, api_holder);
609 if (!api_holder_ref.has_value()) return PropertyAccessInfo::Invalid(zone);
610 }
611 }
612 if (access_mode == AccessMode::kLoad) {
613 std::optional<Tagged<Name>> cached_property_name =
614 FunctionTemplateInfo::TryGetCachedPropertyName(isolate, *accessor);
615 if (cached_property_name.has_value()) {
616 OptionalNameRef cached_property_name_ref =
617 TryMakeRef(broker, cached_property_name.value());
618 if (cached_property_name_ref.has_value()) {
619 PropertyAccessInfo access_info = ai_factory->ComputePropertyAccessInfo(
620 holder_map, cached_property_name_ref.value(), access_mode);
621 if (!access_info.IsInvalid()) return access_info;
622 }
623 }
624 }
625
626 if (holder_map.is_dictionary_map()) {
627 CHECK(!api_holder_ref.has_value());
628 return PropertyAccessInfo::DictionaryProtoAccessorConstant(
629 zone, receiver_map, holder, accessor_ref.value(), api_holder_ref, name);
630 } else {
631 return PropertyAccessInfo::FastAccessorConstant(
632 zone, receiver_map, holder, accessor_ref.value(), api_holder_ref);
633 }
634}

728PropertyAccessInfo AccessInfoFactory::ComputePropertyAccessInfo(
729 MapRef map, NameRef name, AccessMode access_mode) const {
730 CHECK(name.IsUniqueName());
731
732 // Dictionary property const tracking is unsupported with concurrent inlining.
733 CHECK(!V8_DICT_PROPERTY_CONST_TRACKING_BOOL);
734
735 JSHeapBroker::MapUpdaterGuardIfNeeded mumd_scope(broker());
736
737 if (access_mode == AccessMode::kHas && !IsJSReceiverMap(*map.object())) {
738 return Invalid();
739 }
740
741 // Check if it is safe to inline property access for the {map}.
742 if (!CanInlinePropertyAccess(map, access_mode)) {
743 return Invalid();
744 }
745
746 // We support fast inline cases for certain JSObject getters.
747 if (access_mode == AccessMode::kLoad || access_mode == AccessMode::kHas) {
748 PropertyAccessInfo access_info = LookupSpecialFieldAccessor(map, name);
749 if (!access_info.IsInvalid()) return access_info;
750 }
751
752 // Only relevant if V8_DICT_PROPERTY_CONST_TRACKING enabled.
753 bool dictionary_prototype_on_chain = false;
754 bool fast_mode_prototype_on_chain = false;
755
756 // Remember the receiver map. We use {map} as loop variable.
757 MapRef receiver_map = map;
758 OptionalJSObjectRef holder;
759
760 // Perform the implicit ToObject for primitives here.
761 // Implemented according to ES6 section 7.3.2 GetV (V, P).
762 // Note: Keep sync'd with
763 // CompilationDependencies::DependOnStablePrototypeChains.
764 if (receiver_map.IsPrimitiveMap()) {
765 OptionalJSFunctionRef constructor =
766 broker()->target_native_context().GetConstructorFunction(broker(),
767 receiver_map);
768 if (!constructor.has_value()) return Invalid();
769 map = constructor->initial_map(broker());
770 DCHECK(!map.IsPrimitiveMap());
771 }
772
773 while (true) {
774 PropertyDetails details = PropertyDetails::Empty();
775 InternalIndex index = InternalIndex::NotFound();
776 if (!TryLoadPropertyDetails(map, holder, name, &index, &details)) {
777 return Invalid();
778 }
779
780 if (index.is_found()) {
781 if (IsAnyStore(access_mode)) {
782 DCHECK(!map.is_dictionary_map());
783
784 // Don't bother optimizing stores to read-only properties.
785 if (details.IsReadOnly()) return Invalid();
786
787 if (details.kind() == PropertyKind::kData && holder.has_value()) {
788 // This is a store to a property not found on the receiver but on a
789 // prototype. According to ES6 section 9.1.9 [[Set]], we need to
790 // create a new data property on the receiver. We can still optimize
791 // if such a transition already exists.
792 return LookupTransition(receiver_map, name, holder, NONE);
793 }
794 }
795
796 if (IsDefiningStore(access_mode)) {
797 if (details.attributes() != PropertyAttributes::NONE) {
798 // We should store the property with WEC attributes, but that's not
799 // the attributes of the property that we found. We just bail out and
800 // let the runtime figure out what to do (which probably requires
801 // changing the object's map).
802 return Invalid();
803 }
804 }
805
806 if (map.is_dictionary_map()) {
807 DCHECK(V8_DICT_PROPERTY_CONST_TRACKING_BOOL);
808
809 if (fast_mode_prototype_on_chain) {
810 // TODO(v8:11248) While the work on dictionary mode prototypes is in
811 // progress, we may still see fast mode objects on the chain prior to
812 // reaching a dictionary mode prototype holding the property . Due to
813 // this only being an intermediate state, we don't stupport these kind
814 // of heterogenous prototype chains.
815 return Invalid();
816 }
817
818 // TryLoadPropertyDetails only succeeds if we know the holder.
819 return ComputeDictionaryProtoAccessInfo(
820 receiver_map, name, holder.value(), index, access_mode, details);
821 }
822
823 if (dictionary_prototype_on_chain) {
824 // If V8_DICT_PROPERTY_CONST_TRACKING_BOOL was disabled, then a
825 // dictionary prototype would have caused a bailout earlier.
826 DCHECK(V8_DICT_PROPERTY_CONST_TRACKING_BOOL);
827
828 // TODO(v8:11248) We have a fast mode holder, but there was a dictionary
829 // mode prototype earlier on the chain. Note that seeing a fast mode
830 // prototype even though V8_DICT_PROPERTY_CONST_TRACKING is enabled
831 // should only be possible while the implementation of dictionary mode
832 // prototypes is work in progress. Eventually, enabling
833 // V8_DICT_PROPERTY_CONST_TRACKING will guarantee that all prototypes
834 // are always in dictionary mode, making this case unreachable. However,
835 // due to the complications of checking dictionary mode prototypes for
836 // modification, we don't attempt to support dictionary mode prototypes
837 // occuring before a fast mode holder on the chain.
838 return Invalid();
839 }
840 if (details.location() == PropertyLocation::kField) {
841 if (details.kind() == PropertyKind::kData) {
842 return ComputeDataFieldAccessInfo(receiver_map, map, name, holder,
843 index, access_mode);
844 } else {
845 DCHECK_EQ(PropertyKind::kAccessor, details.kind());
846 // TODO(turbofan): Add support for general accessors?
847 return Invalid();
848 }
849 } else {
850 DCHECK_EQ(PropertyLocation::kDescriptor, details.location());
851 DCHECK_EQ(PropertyKind::kAccessor, details.kind());
852 return ComputeAccessorDescriptorAccessInfo(receiver_map, name, map,
853 holder, index, access_mode);
854 }
855
856 UNREACHABLE();
857 }
858
859 // The property wasn't found on {map}. Look on the prototype if appropriate.
860 DCHECK(!index.is_found());
861
862 // Don't search on the prototype chain for special indices in case of
863 // integer indexed exotic objects (see ES6 section 9.4.5).
864 if (IsJSTypedArrayMap(*map.object()) && name.IsString()) {
865 StringRef name_str = name.AsString();
866 SharedStringAccessGuardIfNeeded access_guard(
867 *name_str.object(), broker()->local_isolate_or_isolate());
868 if (IsSpecialIndex(*name_str.object(), access_guard)) return Invalid();
869 }
870
871 // Don't search on the prototype when storing in literals, or performing a
872 // Define operation
873 if (access_mode == AccessMode::kStoreInLiteral ||
874 access_mode == AccessMode::kDefine) {
875 PropertyAttributes attrs = NONE;
876 if (name.object()->IsPrivate()) {
877 // When PrivateNames are added to an object, they are by definition
878 // non-enumerable.
879 attrs = DONT_ENUM;
880 }
881 return LookupTransition(receiver_map, name, holder, attrs);
882 }
883
884 // Don't lookup private symbols on the prototype chain.
885 if (name.object()->IsPrivate()) {
886 return Invalid();
887 }
888
889 if (V8_DICT_PROPERTY_CONST_TRACKING_BOOL && holder.has_value()) {
890 // At this point, we are past the first loop iteration.
891 DCHECK(holder->object()->map()->is_prototype_map());
892 DCHECK(!holder->map(broker()).equals(receiver_map));
893
894 fast_mode_prototype_on_chain =
895 fast_mode_prototype_on_chain || !map.is_dictionary_map();
896 dictionary_prototype_on_chain =
897 dictionary_prototype_on_chain || map.is_dictionary_map();
898 }
899
900 // Walk up the prototype chain.
901 // Load the map's prototype's map to guarantee that every time we use it,
902 // we use the same Map.
903 HeapObjectRef prototype = map.prototype(broker());
904
905 MapRef map_prototype_map = prototype.map(broker());
906 if (!IsJSObjectMap(*map_prototype_map.object())) {
907 // Don't allow proxies on the prototype chain.
908 if (!prototype.IsNull()) {
909 DCHECK(IsJSProxy(*prototype.object()) ||
910 IsWasmObject(*prototype.object()));
911 return Invalid();
912 }
913
914 DCHECK(prototype.IsNull());
915
916 if (dictionary_prototype_on_chain) {
917 // TODO(v8:11248) See earlier comment about
918 // dictionary_prototype_on_chain. We don't support absent properties
919 // with dictionary mode prototypes on the chain, either. This is again
920 // just due to how we currently deal with dependencies for dictionary
921 // properties during finalization.
922 return Invalid();
923 }
924
925 // Store to property not found on the receiver or any prototype, we need
926 // to transition to a new data property.
927 // Implemented according to ES6 section 9.1.9 [[Set]] (P, V, Receiver)
928 if (access_mode == AccessMode::kStore) {
929 return LookupTransition(receiver_map, name, holder, NONE);
930 }
931
932 // The property was not found (access returns undefined or throws
933 // depending on the language mode of the load operation.
934 // Implemented according to ES6 section 9.1.8 [[Get]] (P, Receiver)
935 return PropertyAccessInfo::NotFound(zone(), receiver_map, holder);
936 }
937
938 CHECK(prototype.IsJSObject());
939 holder = prototype.AsJSObject();
940 map = map_prototype_map;
941
942 if (!CanInlinePropertyAccess(map, access_mode)) {
943 return Invalid();
944 }
945
946 // Successful lookup on prototype chain needs to guarantee that all the
947 // prototypes up to the holder have stable maps, except for dictionary-mode
948 // prototypes. We currently do this by taking a
949 // DependOnStablePrototypeChains dependency in the caller.
950 //
951 // TODO(jgruber): This is brittle and easy to miss. Consider a refactor
952 // that moves the responsibility of taking the dependency into
953 // AccessInfoFactory.
954 }
955 UNREACHABLE();
956}

1068 // Check for String::length field accessor.
1069 if (IsStringMap(*map.object())) {
1070 if (Name::Equals(isolate(), name.object(),
1071 isolate()->factory()->length_string())) {
1072 return PropertyAccessInfo::StringLength(zone(), map);

1091 DCHECK(Name::Equals(isolate(), isolate()->factory()->length_string(),

1094 // [0, FixedDoubleArray::kMaxLength] in case of fast double
1095 // elements, a smi in the range [0, FixedArray::kMaxLength]
1096 // in case of other fast elements, and [0, kMaxUInt32] in
1097 // case of other arrays.
1098 if (IsDoubleElementsKind(map.elements_kind())) {
1099 field_type = type_cache_->kFixedDoubleArrayLengthType;
1100 field_representation = Representation::Smi();
1101 } else if (IsFastElementsKind(map.elements_kind())) {
1102 field_type = type_cache_->kFixedArrayLengthType;
1103 field_representation = Representation::Smi();
1104 } else {
1105 field_type = type_cache_->kJSArrayLengthType;
1106 }
1107 }
1108 // Special fields are always mutable.
1109 return PropertyAccessInfo::DataField(broker(), zone(), map, {{}, zone()},
1110 field_index, field_representation,
1111 field_type, map, {}, {}, {});
1112 }
1113 return Invalid();

1116PropertyAccessInfo AccessInfoFactory::LookupTransition(
1117 MapRef map, NameRef name, OptionalJSObjectRef holder,
1118 PropertyAttributes attrs) const {
1119 // Check if the {map} has a data transition with the given {name}.
1120 Tagged<Map> transition =
1121 TransitionsAccessor(isolate(), *map.object(), true)
1122 .SearchTransition(*name.object(), PropertyKind::kData, attrs);
1123 if (transition.is_null()) return Invalid();
1124 OptionalMapRef maybe_transition_map = TryMakeRef(broker(), transition);
1125 if (!maybe_transition_map.has_value()) return Invalid();
1126 MapRef transition_map = maybe_transition_map.value();
1127
1128 InternalIndex const number = transition_map.object()->LastAdded();
1129 DirectHandle<DescriptorArray> descriptors =
1130 transition_map.instance_descriptors(broker()).object();
1131 PropertyDetails const details = descriptors->GetDetails(number);
1132
1133 // Don't bother optimizing stores to read-only properties.
1134 if (details.IsReadOnly()) return Invalid();
1135
1136 // TODO(bmeurer): Handle transition to data constant?
1137 if (details.location() != PropertyLocation::kField) return Invalid();
1138
1139 int const index = details.field_index();
1140 Representation details_representation = details.representation();
1141 if (details_representation.IsNone()) return Invalid();
1142
1143 FieldIndex field_index = FieldIndex::ForPropertyIndex(
1144 *transition_map.object(), index, details_representation);
1145 Type field_type = Type::NonInternal();
1146 OptionalMapRef field_map;
1147
1148 DCHECK_EQ(transition_map, transition_map.FindFieldOwner(broker(), number));
1149
1150 ZoneVector<CompilationDependency const*> unrecorded_dependencies(zone());
1151 if (details_representation.IsSmi()) {
1152 field_type = Type::SignedSmall();
1153 unrecorded_dependencies.push_back(
1154 dependencies()->FieldRepresentationDependencyOffTheRecord(
1155 transition_map, transition_map, number, details_representation));
1156 } else if (details_representation.IsDouble()) {
1157 field_type = type_cache_->kFloat64;
1158 unrecorded_dependencies.push_back(
1159 dependencies()->FieldRepresentationDependencyOffTheRecord(
1160 transition_map, transition_map, number, details_representation));
1161 } else if (details_representation.IsHeapObject()) {
1162 // Extract the field type from the property details (make sure its
1163 // representation is TaggedPointer to reflect the heap object case).
1164 // TODO(jgruber,v8:7790): Use DescriptorArrayRef instead.
1165 Handle<FieldType> descriptors_field_type =
1166 broker()->CanonicalPersistentHandle(descriptors->GetFieldType(number));
1167 OptionalObjectRef descriptors_field_type_ref =
1168 TryMakeRef<Object>(broker(), descriptors_field_type);
1169 if (!descriptors_field_type_ref.has_value()) return Invalid();
1170
1171 if (IsNone(*descriptors_field_type)) {
1172 // Cleared field-types are pre-monomorphic states. The field type was
1173 // garbge collected and we need to record an updated type.
1174 static_assert(FieldType::kFieldTypesCanBeClearedOnGC);
1175 return Invalid();
1176 }
1177 unrecorded_dependencies.push_back(
1178 dependencies()->FieldRepresentationDependencyOffTheRecord(
1179 transition_map, transition_map, number, details_representation));
1180 if (IsClass(*descriptors_field_type)) {
1181 unrecorded_dependencies.push_back(
1182 dependencies()->FieldTypeDependencyOffTheRecord(
1183 transition_map, transition_map, number,
1184 *descriptors_field_type_ref));
1185 // Remember the field map, and try to infer a useful type.
1186 OptionalMapRef maybe_field_map =
1187 TryMakeRef(broker(), FieldType::AsClass(*descriptors_field_type));
1188 if (!maybe_field_map.has_value()) return Invalid();
1189 field_type = Type::For(maybe_field_map.value(), broker());
1190 field_map = maybe_field_map;
1191 }
1192 }
1193
1194 unrecorded_dependencies.push_back(
1195 dependencies()->TransitionDependencyOffTheRecord(transition_map));
1196 // Transitioning stores *may* store to const fields. The resulting
1197 // DataConstant access infos can be distinguished from later, i.e. redundant,
1198 // stores to the same constant field by the presence of a transition map.
1199 switch (dependencies()->DependOnFieldConstness(transition_map, transition_map,
1200 number)) {
1201 case PropertyConstness::kMutable:
1202 return PropertyAccessInfo::DataField(
1203 broker(), zone(), map, std::move(unrecorded_dependencies),
1204 field_index, details_representation, field_type, transition_map,
1205 field_map, holder, transition_map);
1206 case PropertyConstness::kConst:
1207 return PropertyAccessInfo::FastDataConstant(
1208 zone(), map, std::move(unrecorded_dependencies), field_index,
1209 details_representation, field_type, transition_map, field_map, holder,
1210 transition_map);
1211 }
1212 UNREACHABLE();
1213}
