56template <class T>
57static T ComputeRounding(T a, int mode) {
58 switch (mode) {
59 case ROUND_TO_NEAREST_AWAY_FROM_0:
60 return std::round(a);
61 case ROUND_TO_NEAREST_TO_EVEN:
62 return std::nearbyint(a);
63 case ROUND_TOWARD_0:
64 return std::trunc(a);
65 case ROUND_TOWARD_POS_INF:
66 return std::ceil(a);
67 case ROUND_TOWARD_NEG_INF:
68 return std::floor(a);
69 default:
70 UNIMPLEMENTED();
71 }
72 return 0;
73}

273 template <typename T>
274 void SetS390ConditionCode(T lhs, T rhs) {
275 condition_reg_ = 0;
276 if (lhs == rhs) {
277 condition_reg_ |= CC_EQ;
278 } else if (lhs < rhs) {
279 condition_reg_ |= CC_LT;
280 } else if (lhs > rhs) {
281 condition_reg_ |= CC_GT;
282 }
283
284 // We get down here only for floating point
285 // comparisons and the values are unordered
286 // i.e. NaN
287 if (condition_reg_ == 0) condition_reg_ = unordered;
288 }
289
290 // Used by arithmetic operations that use carry.
291 template <typename T>
292 void SetS390ConditionCodeCarry(T result, bool overflow) {
293 condition_reg_ = 0;
294 bool zero_result = (result == static_cast<T>(0));
295 if (zero_result && !overflow) {
296 condition_reg_ |= 8;
297 } else if (!zero_result && !overflow) {
298 condition_reg_ |= 4;
299 } else if (zero_result && overflow) {
300 condition_reg_ |= 2;
301 } else if (!zero_result && overflow) {
302 condition_reg_ |= 1;
303 }
304 if (condition_reg_ == 0) UNREACHABLE();
305 }

378 // Vector register lane numbers on IBM machines are reversed compared to
379 // x64. For example, doing an I32x4 extract_lane with lane number 0 on x64
380 // will be equal to lane number 3 on IBM machines. Vector registers are only
381 // used for compiling Wasm code at the moment. Wasm is also little endian
382 // enforced. On s390 native, we manually do a reverse byte whenever values are
383 // loaded/stored from memory to a Simd register. On the simulator however, we
384 // do not reverse the bytes and data is just copied as is from one memory
385 // location to another location which represents a register. To keep the Wasm
386 // simulation accurate, we need to make sure accessing a lane is correctly
387 // simulated and as such we reverse the lane number on the getters and setters
388 // below. We need to be careful when getting/setting values on the Low or High
389 // side of a simulated register. In the simulation, "Low" is equal to the MSB
390 // and "High" is equal to the LSB on memory. "force_ibm_lane_numbering" could
391 // be used to disabled automatic lane number reversal and help with accessing
392 // the Low or High side of a simulated register.
393 template <class T>
394 T get_simd_register_by_lane(int reg, int lane,
395 bool force_ibm_lane_numbering = true) {
396 if (force_ibm_lane_numbering) {
397 lane = (kSimd128Size / sizeof(T)) - 1 - lane;
398 }
399 CHECK_LE(lane, kSimd128Size / sizeof(T));
400 CHECK_LT(reg, kNumFPRs);
401 CHECK_GE(lane, 0);
402 CHECK_GE(reg, 0);
403 return (reinterpret_cast<T*>(&fp_registers_[reg]))[lane];
404 }
405
406 template <class T>
407 void set_simd_register_by_lane(int reg, int lane, const T& value,
408 bool force_ibm_lane_numbering = true) {
409 if (force_ibm_lane_numbering) {
410 lane = (kSimd128Size / sizeof(T)) - 1 - lane;
411 }
412 CHECK_LE(lane, kSimd128Size / sizeof(T));
413 CHECK_LT(reg, kNumFPRs);
414 CHECK_GE(lane, 0);
415 CHECK_GE(reg, 0);
416 (reinterpret_cast<T*>(&fp_registers_[reg]))[lane] = value;
417 }
