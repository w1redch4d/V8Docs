176bool Script::HasValidSource() {
177 Tagged<Object> src = this->source();
178 if (!IsString(src)) return true;
179 Tagged<String> src_str = Cast<String>(src);
180 if (!StringShape(src_str).IsExternal()) return true;
181 if (src_str->IsOneByteRepresentation()) {
182 return Cast<ExternalOneByteString>(src)->resource() != nullptr;
183 } else if (src_str->IsTwoByteRepresentation()) {
184 return Cast<ExternalTwoByteString>(src)->resource() != nullptr;
185 }
186 return true;
187}

221Tagged<Script> Script::GetEvalOrigin() {
222 DisallowGarbageCollection no_gc;
223 Tagged<Script> origin_script = *this;
224 while (origin_script->has_eval_from_shared()) {
225 Tagged<HeapObject> maybe_script =
226 origin_script->eval_from_shared()->script();
227 CHECK(IsScript(maybe_script));
228 origin_script = Cast<Script>(maybe_script);
229 }
230 return origin_script;
231}
