25void WritableRelocInfo::apply(intptr_t delta) {
26 // On arm64 only internal references and immediate branches need extra work.
27 if (RelocInfo::IsInternalReference(rmode_)) {
28 // Absolute code pointer inside code object moves with the code object.
29 intptr_t internal_ref = ReadUnalignedValue<intptr_t>(pc_);
30 internal_ref += delta; // Relocate entry.
31 jit_allocation_.WriteUnalignedValue<intptr_t>(pc_, internal_ref);
32 } else {
33 Instruction* instr = reinterpret_cast<Instruction*>(pc_);
34 if (instr->IsBranchAndLink() || instr->IsUnconditionalBranch()) {
35 Address old_target =
36 reinterpret_cast<Address>(instr->ImmPCOffsetTarget());
37 Address new_target = old_target - delta;
38 instr->SetBranchImmTarget<UncondBranchType>(
39 reinterpret_cast<Instruction*>(new_target), &jit_allocation_);
40 }
41 }
42}

561void Assembler::set_target_address_at(Address pc, Address constant_pool,
562 Address target,
563 WritableJitAllocation* jit_allocation,
564 ICacheFlushMode icache_flush_mode) {
565 Instruction* instr = reinterpret_cast<Instruction*>(pc);
566 if (instr->IsLdrLiteralX()) {
567 if (jit_allocation) {
568 jit_allocation->WriteValue<Address>(target_pointer_address_at(pc),
569 target);
570 } else {
571 Memory<Address>(target_pointer_address_at(pc)) = target;
572 }
573 // Intuitively, we would think it is necessary to always flush the
574 // instruction cache after patching a target address in the code. However,
575 // in this case, only the constant pool contents change. The instruction
576 // accessing the constant pool remains unchanged, so a flush is not
577 // required.
578 } else {
579 DCHECK(instr->IsBranchAndLink() || instr->IsUnconditionalBranch());
580 if (target == 0) {
581 // We are simply wiping the target out for serialization. Set the offset
582 // to zero instead.
583 target = pc;
584 }
585 instr->SetBranchImmTarget<UncondBranchType>(
586 reinterpret_cast<Instruction*>(target), jit_allocation);
587 if (icache_flush_mode != SKIP_ICACHE_FLUSH) {
588 FlushInstructionCache(pc, kInstrSize);
589 }
590 }
591}

621Address RelocInfo::target_address_address() {
622 DCHECK(HasTargetAddressAddress());
623 Instruction* instr = reinterpret_cast<Instruction*>(pc_);
624 // Read the address of the word containing the target_address in an
625 // instruction stream.
626 // The only architecture-independent user of this function is the serializer.
627 // The serializer uses it to find out how many raw bytes of instruction to
628 // output before the next target.
629 // For an instruction like B/BL, where the target bits are mixed into the
630 // instruction bits, the size of the target will be zero, indicating that the
631 // serializer should not step forward in memory after a target is resolved
632 // and written.
633 // For LDR literal instructions, we can skip up to the constant pool entry
634 // address. We make sure that RelocInfo is ordered by the
635 // target_address_address so that we do not skip over any relocatable
636 // instruction sequences.
637 if (instr->IsLdrLiteralX()) {
638 return constant_pool_entry_address();
639 } else {
640 DCHECK(instr->IsBranchAndLink() || instr->IsUnconditionalBranch());
641 return pc_;
642 }
643}
