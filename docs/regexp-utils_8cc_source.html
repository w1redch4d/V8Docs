18Handle<String> RegExpUtils::GenericCaptureGetter(
19 Isolate* isolate, DirectHandle<RegExpMatchInfo> match_info, int capture,
20 bool* ok) {
21 const int capture_start_index = RegExpMatchInfo::capture_start_index(capture);
22 if (capture_start_index >= match_info->number_of_capture_registers()) {
23 if (ok != nullptr) *ok = false;
24 return isolate->factory()->empty_string();
25 }
26
27 const int capture_end_index = RegExpMatchInfo::capture_end_index(capture);
28 const int match_start = match_info->capture(capture_start_index);
29 const int match_end = match_info->capture(capture_end_index);
30 if (match_start == -1 || match_end == -1) {
31 if (ok != nullptr) *ok = false;
32 return isolate->factory()->empty_string();
33 }
34
35 if (ok != nullptr) *ok = true;
36 Handle<String> last_subject(match_info->last_subject(), isolate);
37 return isolate->factory()->NewSubString(last_subject, match_start, match_end);
38}

41bool RegExpUtils::IsMatchedCapture(Tagged<RegExpMatchInfo> match_info,
42 int capture) {
43 // Sentinel used as failure indicator in other functions.
44 if (capture == -1) return false;
45
46 const int capture_start_index = RegExpMatchInfo::capture_start_index(capture);
47 if (capture_start_index >= match_info->number_of_capture_registers()) {
48 return false;
49 }
50
51 const int capture_end_index = RegExpMatchInfo::capture_end_index(capture);
52 const int match_start = match_info->capture(capture_start_index);
53 const int match_end = match_info->capture(capture_end_index);
54 return match_start != -1 && match_end != -1;
55}

91// ES#sec-regexpexec Runtime Semantics: RegExpExec ( R, S )
92// Also takes an optional exec method in case our caller
93// has already fetched exec.
94MaybeHandle<JSAny> RegExpUtils::RegExpExec(Isolate* isolate,
95 Handle<JSReceiver> regexp,
96 DirectHandle<String> string,
97 DirectHandle<Object> exec) {
98 if (IsUndefined(*exec, isolate)) {
99 ASSIGN_RETURN_ON_EXCEPTION(
100 isolate, exec,
101 Object::GetProperty(isolate, regexp,
102 isolate->factory()->exec_string()));
103 }
104
105 if (IsCallable(*exec)) {
106 constexpr int argc = 1;
107 std::array<DirectHandle<Object>, argc> args = {string};
108
109 Handle<JSAny> result;
110 ASSIGN_RETURN_ON_EXCEPTION(
111 isolate, result,
112 Cast<JSAny>(
113 Execution::Call(isolate, exec, regexp, base::VectorOf(args))));
114
115 if (!IsJSReceiver(*result) && !IsNull(*result, isolate)) {
116 THROW_NEW_ERROR(isolate,
117 NewTypeError(MessageTemplate::kInvalidRegExpExecResult));
118 }
119 return result;
120 }
121
122 if (!IsJSRegExp(*regexp)) {
123 THROW_NEW_ERROR(isolate,
124 NewTypeError(MessageTemplate::kIncompatibleMethodReceiver,
125 isolate->factory()->NewStringFromAsciiChecked(
126 "RegExp.prototype.exec"),
127 regexp));
128 }
129
130 {
131 DirectHandle<JSFunction> regexp_exec = isolate->regexp_exec_function();
132
133 constexpr int argc = 1;
134 std::array<DirectHandle<Object>, argc> args = {string};
135
136 return Cast<JSAny>(
137 Execution::Call(isolate, regexp_exec, regexp, base::VectorOf(args)));
138 }

141bool RegExpUtils::IsUnmodifiedRegExp(Isolate* isolate,
142 DirectHandle<Object> obj) {
143#ifdef V8_ENABLE_FORCE_SLOW_PATH
144 if (isolate->force_slow_path()) return false;
145#endif
146
147 if (!IsJSReceiver(*obj)) return false;
148
149 Tagged<JSReceiver> recv = Cast<JSReceiver>(*obj);
150
151 if (!HasInitialRegExpMap(isolate, recv)) return false;
152
153 // Check the receiver's prototype's map.
154 Tagged<Object> proto = recv->map()->prototype();
155 if (!IsJSReceiver(proto)) return false;
156
157 DirectHandle<Map> initial_proto_initial_map = isolate->regexp_prototype_map();
158 Tagged<Map> proto_map = Cast<JSReceiver>(proto)->map();
159 if (proto_map != *initial_proto_initial_map) {
160 return false;
161 }
162
163 // Check that the "exec" method is unmodified.
164 // Check that the index refers to "exec" method (this has to be consistent
165 // with the init order in the bootstrapper).
166 InternalIndex kExecIndex(JSRegExp::kExecFunctionDescriptorIndex);
167 DCHECK_EQ(*(isolate->factory()->exec_string()),
168 proto_map->instance_descriptors(isolate)->GetKey(kExecIndex));
169 if (proto_map->instance_descriptors(isolate)
170 ->GetDetails(kExecIndex)
171 .constness() != PropertyConstness::kConst) {
172 return false;
173 }
174
175 // Note: Unlike the more involved check in CSA (see BranchIfFastRegExp), this
176 // does not go on to check the actual value of the exec property. This would
177 // not be valid since this method is called from places that access the flags
178 // property. Similar spots in CSA would use BranchIfFastRegExp_Strict in this
179 // case.
180
181 if (!Protectors::IsRegExpSpeciesLookupChainIntact(isolate)) return false;
182
183 // The smi check is required to omit ToLength(lastIndex) calls with possible
184 // user-code execution on the fast path.
185 Tagged<Object> last_index = Cast<JSRegExp>(recv)->last_index();
186 return IsSmi(last_index) && Smi::ToInt(last_index) >= 0;
187}

189uint64_t RegExpUtils::AdvanceStringIndex(Tagged<String> string, uint64_t index,
190 bool unicode) {
191 DCHECK_LE(static_cast<double>(index), kMaxSafeInteger);
192 const uint64_t string_length = static_cast<uint64_t>(string->length());
193 if (unicode && index < string_length) {
194 const uint16_t first = string->Get(static_cast<uint32_t>(index));
195 if (first >= 0xD800 && first <= 0xDBFF && index + 1 < string_length) {
196 DCHECK_LT(index, std::numeric_limits<uint64_t>::max());
197 const uint16_t second = string->Get(static_cast<uint32_t>(index + 1));
198 if (second >= 0xDC00 && second <= 0xDFFF) {
199 return index + 2;
200 }
201 }
202 }
203
204 return index + 1;
205}

207MaybeHandle<Object> RegExpUtils::SetAdvancedStringIndex(
208 Isolate* isolate, Handle<JSReceiver> regexp, DirectHandle<String> string,
209 bool unicode) {
210 DirectHandle<Object> last_index_obj;
211 ASSIGN_RETURN_ON_EXCEPTION(
212 isolate, last_index_obj,
213 Object::GetProperty(isolate, regexp,
214 isolate->factory()->lastIndex_string()));
215
216 ASSIGN_RETURN_ON_EXCEPTION(isolate, last_index_obj,
217 Object::ToLength(isolate, last_index_obj));
218 const uint64_t last_index = PositiveNumberToUint64(*last_index_obj);
219 const uint64_t new_last_index =
220 AdvanceStringIndex(*string, last_index, unicode);
221
222 return SetLastIndex(isolate, regexp, new_last_index);
223}
