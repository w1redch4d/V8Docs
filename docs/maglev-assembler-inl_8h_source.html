187 masm->set_allow_allocate(allow_allocate_);

211#endif // DEBUG

216template <typename Function, typename... Args>
217inline Label* MaglevAssembler::MakeDeferredCode(Function&& deferred_code_gen,
218 Args&&... args) {
219 using FunctionPointer =
220 typename detail::FunctionArgumentsTupleHelper<Function>::FunctionPointer;
221 static_assert(
222 std::is_invocable_v<FunctionPointer, MaglevAssembler*,
223 decltype(detail::CopyForDeferred(
224 std::declval<MaglevCompilationInfo*>(),
225 std::declval<Args>()))...>,
226 "Parameters of deferred_code_gen function should match arguments into "
227 "MakeDeferredCode");
228
229 TemporaryRegisterScope scratch_scope(this);
230 using DeferredCodeInfoT = detail::DeferredCodeInfoImpl<Function>;
231 DeferredCodeInfoT* deferred_code =
232 compilation_info()->zone()->New<DeferredCodeInfoT>(
233 compilation_info(), scratch_scope.CopyForDefer(), deferred_code_gen,
234 std::forward<Args>(args)...);
235
236#ifdef DEBUG
237 deferred_code->set_allow_call(allow_deferred_call_);
238 deferred_code->set_allow_allocate(allow_allocate_);
239#endif // DEBUG
240
241 code_gen_state()->PushDeferredCode(deferred_code);
242 return &deferred_code->deferred_code_label;
243}

260template <typename T>
261inline void AllocateSlow(MaglevAssembler* masm,
262 RegisterSnapshot register_snapshot, Register object,
263 Builtin builtin, T size_in_bytes, ZoneLabelRef done) {
264 // Remove {object} from snapshot, since it is the returned allocated
265 // HeapObject.
266 register_snapshot.live_registers.clear(object);
267 register_snapshot.live_tagged_registers.clear(object);
268 {
269 SaveRegisterStateForCall save_register_state(masm, register_snapshot);
270 using D = AllocateDescriptor;
271 masm->Move(D::GetRegisterParameter(D::kRequestedSize), size_in_bytes);
272 masm->CallBuiltin(builtin);
273 save_register_state.DefineSafepoint();
274 masm->Move(object, kReturnRegister0);
275 }
276 masm->Jump(*done);
277}

363inline void MaglevAssembler::Branch(Condition condition, Label* if_true,
364 Label::Distance true_distance,
365 bool fallthrough_when_true, Label* if_false,
366 Label::Distance false_distance,
367 bool fallthrough_when_false) {
368 if (fallthrough_when_false) {
369 if (fallthrough_when_true) {
370 // If both paths are a fallthrough, do nothing.
371 DCHECK_EQ(if_true, if_false);
372 return;
373 }
374 // Jump over the false block if true, otherwise fall through into it.
375 JumpIf(condition, if_true, true_distance);
376 } else {
377 // Jump to the false block if true.
378 JumpIf(NegateCondition(condition), if_false, false_distance);
379 // Jump to the true block if it's not the next block.
380 if (!fallthrough_when_true) {
381 Jump(if_true, true_distance);
382 }
383 }
384}

581template <Builtin kBuiltin, typename... Args>
582void MoveArgumentsForBuiltin(MaglevAssembler* masm, Args&&... args) {
583 using Descriptor = typename CallInterfaceDescriptorFor<kBuiltin>::type;
584
585 // Put the args into a tuple for easier manipulation.
586 std::tuple<Args&&...> args_tuple{std::forward<Args>(args)...};
587
588 // If there is a context, the first argument is the context parameter. Use
589 // the remaining args as the actual arguments. We pass the context first
590 // instead of last to avoid ambiguity around dealing with on-stack
591 // arguments.
592 constexpr size_t context_args = Descriptor::HasContextParameter() ? 1 : 0;
593 static_assert(context_args <= std::tuple_size_v<decltype(args_tuple)>,
594 "Not enough arguments passed in to builtin (are you missing a "
595 "context argument?)");
596 auto args_tuple_without_context = base::tuple_drop<context_args>(args_tuple);
597 CheckArgs<Descriptor>(masm, args_tuple_without_context);
598
599 // Split args into register and stack args.
600 static_assert(Descriptor::GetRegisterParameterCount() <=
601 std::tuple_size_v<decltype(args_tuple_without_context)>,
602 "Not enough arguments passed in to builtin (are you missing a "
603 "context argument?)");
604 auto register_args =
605 base::tuple_head<Descriptor::GetRegisterParameterCount()>(
606 args_tuple_without_context);
607 auto stack_args = base::tuple_drop<Descriptor::GetRegisterParameterCount()>(
608 args_tuple_without_context);
609
610 // Split stack args into fixed and variable.
611 static_assert(
612 Descriptor::GetStackParameterCount() <=
613 std::tuple_size_v<decltype(stack_args)>,
614 "Not enough stack arguments passed in to builtin (are you missing a "
615 "context argument?)");
616 auto fixed_stack_args =
617 base::tuple_head<Descriptor::GetStackParameterCount()>(stack_args);
618 auto vararg_stack_args =
619 base::tuple_drop<Descriptor::GetStackParameterCount()>(stack_args);
620
621 if constexpr (!Descriptor::AllowVarArgs()) {
622 static_assert(std::tuple_size_v<decltype(vararg_stack_args)> == 0,
623 "Too many arguments passed in to builtin that expects no "
624 "vararg stack arguments");
625 }
626
627 // First push stack arguments (if any), since some of these may be in
628 // registers and we don't want to clobber them. This supports any thing
629 // `masm->Push` supports, including iterator ranges, so the tuple size may be
630 // smaller than the number of arguments actually pushed. We push fixed and
631 // vararg stack arguments separately, so that there's an appropriate amount
632 // of padding between them.
633 if (Descriptor::kStackArgumentOrder == StackArgumentOrder::kDefault) {
634 PushArgumentsForBuiltin<Descriptor>(
635 masm, std::forward<decltype(fixed_stack_args)>(fixed_stack_args));
636 PushArgumentsForBuiltin<Descriptor>(
637 masm, std::forward<decltype(vararg_stack_args)>(vararg_stack_args));
638 } else {
639 PushArgumentsForBuiltin<Descriptor>(
640 masm, std::forward<decltype(vararg_stack_args)>(vararg_stack_args));
641 PushArgumentsForBuiltin<Descriptor>(
642 masm, std::forward<decltype(fixed_stack_args)>(fixed_stack_args));
643 }
644
645// Then, set register arguments.
646// TODO(leszeks): Use the parallel move helper to do register moves, instead
647// of detecting clobbering.
648#ifdef DEBUG
649 RegList written_registers = {};
650 DoubleRegList written_double_registers = {};
651#endif // DEBUG
652
653 base::tuple_for_each_with_index(register_args, [&](auto&& arg, auto index) {
654 using Arg = decltype(arg);
655 static_assert(index < Descriptor::GetRegisterParameterCount());
656
657 // Make sure the argument wasn't clobbered by any previous write.
658 DCHECK(!ClobberedBy(written_registers, arg));
659 DCHECK(!ClobberedBy(written_double_registers, arg));
660
661 static constexpr bool use_double_register =
662 IsFloatingPoint(Descriptor::GetParameterType(index).representation());
663 if constexpr (use_double_register) {
664 DoubleRegister target = Descriptor::GetDoubleRegisterParameter(index);
665 if constexpr (std::is_same_v<Input, std::decay_t<Arg>>) {
666 DCHECK_EQ(target, arg.AssignedDoubleRegister());
667 USE(target);
668 } else {
669 masm->Move(target, std::forward<Arg>(arg));
670 }
671#ifdef DEBUG
672 written_double_registers.set(target);
673#endif // DEBUG
674 } else {
675 Register target = Descriptor::GetRegisterParameter(index);
676 if constexpr (std::is_same_v<Input, std::decay_t<Arg>>) {
677 DCHECK_EQ(target, arg.AssignedGeneralRegister());
678 USE(target);
679 } else {
680 masm->Move(target, std::forward<Arg>(arg));
681 }
682#ifdef DEBUG
683 written_registers.set(target);
684#endif // DEBUG
685 }
686
687 // TODO(leszeks): Support iterator range for register args.
688 });
689
690 // Set the context last (to avoid clobbering).
691 if constexpr (Descriptor::HasContextParameter()) {
692 auto&& context = std::get<0>(args_tuple);
693 DCHECK(!ClobberedBy(written_registers, context));
694 DCHECK(!ClobberedBy(written_double_registers, context));
695 DCHECK(MachineTypeMatches(MachineType::AnyTagged(), context));
696
697 if constexpr (std::is_same_v<Input, std::decay_t<decltype(context)>>) {
698 DCHECK_EQ(Descriptor::ContextRegister(),
699 context.AssignedGeneralRegister());
700 } else {
701 // Don't allow raw Register here, force materialisation from a constant.
702 // This is because setting parameters could have clobbered the register.
703 // TODO(leszeks): Include the context register in the parallel moves
704 // described above.
705 static_assert(!std::is_same_v<Register, std::decay_t<decltype(context)>>);
706 masm->Move(Descriptor::ContextRegister(), context);
707 }
708 }
709}
