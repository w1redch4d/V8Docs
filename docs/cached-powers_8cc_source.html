76static const int kCachedPowersOffset = 348; // -1 * the first decimal_exponent.
77static const double kD_1_LOG2_10 = 0.30102999566398114; // 1 / lg(10)
78// Difference between the decimal exponents in the table above.
79const int PowersOfTenCache::kDecimalExponentDistance = 8;
80const int PowersOfTenCache::kMinDecimalExponent = -348;
81const int PowersOfTenCache::kMaxDecimalExponent = 340;

83void PowersOfTenCache::GetCachedPowerForBinaryExponentRange(
84 int min_exponent, int max_exponent, DiyFp* power, int* decimal_exponent) {
85 int kQ = DiyFp::kSignificandSize;
86 // Some platforms return incorrect sign on 0 result. We can ignore that here,
87 // which means we can avoid depending on platform.h.
88 double k = std::ceil((min_exponent + kQ - 1) * kD_1_LOG2_10);
89 int foo = kCachedPowersOffset;
90 int index = (foo + static_cast<int>(k) - 1) / kDecimalExponentDistance + 1;
91 DCHECK(0 <= index && index < kCachedPowersLength);
92 CachedPower cached_power = kCachedPowers[index];
93 DCHECK(min_exponent <= cached_power.binary_exponent);
94 DCHECK(cached_power.binary_exponent <= max_exponent);
95 *decimal_exponent = cached_power.decimal_exponent;
96 *power = DiyFp(cached_power.significand, cached_power.binary_exponent);
97}

99void PowersOfTenCache::GetCachedPowerForDecimalExponent(int requested_exponent,
100 DiyFp* power,
101 int* found_exponent) {
102 DCHECK_LE(kMinDecimalExponent, requested_exponent);
103 DCHECK(requested_exponent < kMaxDecimalExponent + kDecimalExponentDistance);
104 int index =
105 (requested_exponent + kCachedPowersOffset) / kDecimalExponentDistance;
106 CachedPower cached_power = kCachedPowers[index];
107 *power = DiyFp(cached_power.significand, cached_power.binary_exponent);
108 *found_exponent = cached_power.decimal_exponent;
109 DCHECK(*found_exponent <= requested_exponent);
110 DCHECK(requested_exponent < *found_exponent + kDecimalExponentDistance);
111}
