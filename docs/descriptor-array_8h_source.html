37// A DescriptorArray is a custom array that holds instance descriptors.
38// It has the following layout:
39// Header:
40// [16:0 bits]: number_of_all_descriptors (including slack)
41// [32:16 bits]: number_of_descriptors
42// [64:32 bits]: raw_gc_state (used by GC)
43// [kEnumCacheOffset]: enum cache
44// Elements:
45// [kHeaderSize + 0]: first key (and internalized String)
46// [kHeaderSize + 1]: first descriptor details (see PropertyDetails)
47// [kHeaderSize + 2]: first value for constants / Tagged<Smi>(1) when not
48// used
49// Slack:
50// [kHeaderSize + number of descriptors * 3]: start of slack
51// The "value" fields store either values or field types. A field type is either
52// FieldType::None(), FieldType::Any() or a weak reference to a Map. All other
53// references are strong.

252// Custom DescriptorArray marking state for visitors that are allowed to write
253// into the heap. The marking state uses DescriptorArray::raw_gc_state() as
254// storage.
255//
256// The state essentially keeps track of 3 fields:
257// 1. The collector epoch: The rest of the state is only valid if the epoch
258// matches. If the epoch doesn't match, the other fields should be considered
259// invalid. The epoch is necessary, as not all DescriptorArray objects are
260// eventually trimmed in the atomic pause and thus available for resetting
261// the state.
262// 2. Number of already marked descriptors.
263// 3. Delta of to be marked descriptors in this cycle. This must be 0 after
264// marking is done.
265class DescriptorArrayMarkingState final {
266 public:
267#define BIT_FIELD_FIELDS(V, _) \
268 V(Epoch, unsigned, 2, _) \
269 V(Marked, uint16_t, 14, _) \
270 V(Delta, uint16_t, 16, _)
271 DEFINE_BIT_FIELDS(BIT_FIELD_FIELDS)
272#undef BIT_FIELD_FIELDS
273 static_assert(Marked::kMax <= Delta::kMax);
274 static_assert(kMaxNumberOfDescriptors <= Marked::kMax);
275
276 using DescriptorIndex = uint16_t;
277 using RawGCStateType = uint32_t;
278
279 static constexpr RawGCStateType kInitialGCState = 0;
280
281 static constexpr RawGCStateType GetFullyMarkedState(
282 unsigned epoch, DescriptorIndex number_of_descriptors) {
283 return NewState(epoch & Epoch::kMask, number_of_descriptors, 0);
284 }
285
286 // Potentially updates the delta of to be marked descriptors. Returns true if
287 // the update was successful and the object should be processed via a marking
288 // visitor.
289 //
290 // The call issues and Acq/Rel barrier to allow synchronizing other state
291 // (e.g. value of descriptor slots) with it.
292 static inline bool TryUpdateIndicesToMark(unsigned gc_epoch,
293 Tagged<DescriptorArray> array,
294 DescriptorIndex index_to_mark);
295
296 // Used from the visitor when processing a DescriptorArray. Returns a range of
297 // start and end descriptor indices. No processing is required for start ==
298 // end. The method signals the first invocation by returning start == 0, and
299 // end != 0.
300 static inline std::pair<DescriptorIndex, DescriptorIndex>
301 AcquireDescriptorRangeToMark(unsigned gc_epoch,
302 Tagged<DescriptorArray> array);
303
304 private:
305 static constexpr RawGCStateType NewState(unsigned masked_epoch,
306 DescriptorIndex marked,
307 DescriptorIndex delta) {
308 return Epoch::encode(masked_epoch) | Marked::encode(marked) |
309 Delta::encode(delta);
310 }
311
312 static bool SwapState(Tagged<DescriptorArray> array, RawGCStateType old_state,
313 RawGCStateType new_state) {
314 return static_cast<RawGCStateType>(base::AcquireRelease_CompareAndSwap(
315 reinterpret_cast<base::Atomic32*>(
316 FIELD_ADDR(array, DescriptorArray::kRawGcStateOffset)),
317 old_state, new_state)) == old_state;
318 }
