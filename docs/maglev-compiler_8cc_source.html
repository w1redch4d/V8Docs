58// static
59bool MaglevCompiler::Compile(LocalIsolate* local_isolate,
60 MaglevCompilationInfo* compilation_info) {
61 compiler::CurrentHeapBrokerScope current_broker(compilation_info->broker());
62 Graph* graph =
63 Graph::New(compilation_info->zone(),
64 compilation_info->toplevel_compilation_unit()->is_osr());
65
66 bool is_tracing_enabled = false;
67 {
68 UnparkedScopeIfOnBackground unparked_scope(local_isolate->heap());
69
70 // Build graph.
71 if (v8_flags.print_maglev_code || v8_flags.code_comments ||
72 v8_flags.print_maglev_graph || v8_flags.print_maglev_graphs ||
73 v8_flags.trace_maglev_graph_building ||
74 v8_flags.trace_maglev_escape_analysis ||
75 v8_flags.trace_maglev_phi_untagging || v8_flags.trace_maglev_regalloc ||
76 v8_flags.trace_maglev_object_tracking) {
77 is_tracing_enabled = compilation_info->toplevel_compilation_unit()
78 ->shared_function_info()
79 .object()
80 ->PassesFilter(v8_flags.maglev_print_filter);
81 compilation_info->set_graph_labeller(new MaglevGraphLabeller());
82 }
83
84 if (is_tracing_enabled &&
85 (v8_flags.print_maglev_code || v8_flags.print_maglev_graph ||
86 v8_flags.print_maglev_graphs || v8_flags.trace_maglev_graph_building ||
87 v8_flags.trace_maglev_phi_untagging ||
88 v8_flags.trace_maglev_regalloc)) {
89 MaglevCompilationUnit* top_level_unit =
90 compilation_info->toplevel_compilation_unit();
91 std::cout << "Compiling " << Brief(*compilation_info->toplevel_function())
92 << " with Maglev\n";
93 BytecodeArray::Disassemble(top_level_unit->bytecode().object(),
94 std::cout);
95 if (v8_flags.maglev_print_feedback) {
96 Print(*top_level_unit->feedback().object(), std::cout);
97 }
98 }
99
100 MaglevGraphBuilder graph_builder(
101 local_isolate, compilation_info->toplevel_compilation_unit(), graph);
102
103 {
104 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
105 "V8.Maglev.GraphBuilding");
106 graph_builder.Build();
107
108 if (is_tracing_enabled && v8_flags.print_maglev_graphs) {
109 std::cout << "\nAfter graph building" << std::endl;
110 PrintGraph(std::cout, compilation_info, graph);
111 }
112 }
113
114 if (v8_flags.maglev_licm) {
115 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
116 "V8.Maglev.LoopOptimizations");
117
118 GraphProcessor<LoopOptimizationProcessor> loop_optimizations(
119 &graph_builder);
120 loop_optimizations.ProcessGraph(graph);
121
122 if (is_tracing_enabled && v8_flags.print_maglev_graphs) {
123 std::cout << "\nAfter loop optimizations" << std::endl;
124 PrintGraph(std::cout, compilation_info, graph);
125 }
126 }
127
128 if (v8_flags.maglev_untagged_phis) {
129 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
130 "V8.Maglev.PhiUntagging");
131
132 GraphProcessor<MaglevPhiRepresentationSelector> representation_selector(
133 &graph_builder);
134 representation_selector.ProcessGraph(graph);
135
136 if (is_tracing_enabled && v8_flags.print_maglev_graphs) {
137 std::cout << "\nAfter Phi untagging" << std::endl;
138 PrintGraph(std::cout, compilation_info, graph);
139 }
140 }
141 }
142
143#ifdef DEBUG
144 {
145 GraphProcessor<MaglevGraphVerifier> verifier(compilation_info);
146 verifier.ProcessGraph(graph);
147 }
148#endif
149
150 {
151 // Post-hoc optimisation:
152 // - Dead node marking
153 // - Cleaning up identity nodes
154 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
155 "V8.Maglev.DeadCodeMarking");
156 GraphMultiProcessor<AnyUseMarkingProcessor> processor;
157 processor.ProcessGraph(graph);
158 }
159
160 if (is_tracing_enabled && v8_flags.print_maglev_graphs) {
161 UnparkedScopeIfOnBackground unparked_scope(local_isolate->heap());
162 std::cout << "After use marking" << std::endl;
163 PrintGraph(std::cout, compilation_info, graph);
164 }
165
166#ifdef DEBUG
167 {
168 GraphProcessor<MaglevGraphVerifier> verifier(compilation_info);
169 verifier.ProcessGraph(graph);
170 }
171#endif
172
173 {
174 // Preprocessing for register allocation and code gen:
175 // - Remove dead nodes
176 // - Collect input/output location constraints
177 // - Find the maximum number of stack arguments passed to calls
178 // - Collect use information, for SSA liveness and next-use distance.
179 // - Mark
180 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
181 "V8.Maglev.NodeProcessing");
182 GraphMultiProcessor<DeadNodeSweepingProcessor,
183 ValueLocationConstraintProcessor, MaxCallDepthProcessor,
184 LiveRangeAndNextUseProcessor,
185 DecompressedUseMarkingProcessor>
186 processor(DeadNodeSweepingProcessor{compilation_info},
187 LiveRangeAndNextUseProcessor{compilation_info});
188 processor.ProcessGraph(graph);
189 }
190
191 if (is_tracing_enabled && v8_flags.print_maglev_graphs) {
192 UnparkedScopeIfOnBackground unparked_scope(local_isolate->heap());
193 std::cout << "After register allocation pre-processing" << std::endl;
194 PrintGraph(std::cout, compilation_info, graph);
195 }
196
197 {
198 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
199 "V8.Maglev.RegisterAllocation");
200 StraightForwardRegisterAllocator allocator(compilation_info, graph);
201
202 if (is_tracing_enabled &&
203 (v8_flags.print_maglev_graph || v8_flags.print_maglev_graphs)) {
204 UnparkedScopeIfOnBackground unparked_scope(local_isolate->heap());
205 std::cout << "After register allocation" << std::endl;
206 PrintGraph(std::cout, compilation_info, graph);
207 }
208 }
209
210 {
211 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
212 "V8.Maglev.CodeAssembly");
213 UnparkedScopeIfOnBackground unparked_scope(local_isolate->heap());
214 std::unique_ptr<MaglevCodeGenerator> code_generator =
215 std::make_unique<MaglevCodeGenerator>(local_isolate, compilation_info,
216 graph);
217 bool success = code_generator->Assemble();
218 if (!success) {
219 return false;
220 }
221
222 // Stash the compiled code_generator on the compilation info.
223 compilation_info->set_code_generator(std::move(code_generator));
224 }
225
226 return true;
227}

229// static
230MaybeHandle<Code> MaglevCompiler::GenerateCode(
231 Isolate* isolate, MaglevCompilationInfo* compilation_info) {
232 compiler::CurrentHeapBrokerScope current_broker(compilation_info->broker());
233 MaglevCodeGenerator* const code_generator =
234 compilation_info->code_generator();
235 DCHECK_NOT_NULL(code_generator);
236
237 Handle<Code> code;
238 {
239 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
240 "V8.Maglev.CodeGeneration");
241 if (compilation_info->is_detached() ||
242 !code_generator->Generate(isolate).ToHandle(&code)) {
243 compilation_info->toplevel_compilation_unit()
244 ->shared_function_info()
245 .object()
246 ->set_maglev_compilation_failed(true);
247 return {};
248 }
249 }
250
251 {
252 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
253 "V8.Maglev.CommittingDependencies");
254 if (!compilation_info->broker()->dependencies()->Commit(code)) {
255 // Don't `set_maglev_compilation_failed` s.t. we may reattempt
256 // compilation.
257 // TODO(v8:7700): Make this more robust, i.e.: don't recompile endlessly.
258 compilation_info->toplevel_function()->SetInterruptBudget(
259 isolate, BudgetModification::kReduce);
260 return {};
261 }
262 }
263
264 if (v8_flags.print_maglev_code) {
265#ifdef OBJECT_PRINT
266 std::unique_ptr<char[]> debug_name =
267 compilation_info->toplevel_function()->shared()->DebugNameCStr();
268 CodeTracer::StreamScope tracing_scope(isolate->GetCodeTracer());
269 auto& os = tracing_scope.stream();
270 code->CodePrint(os, debug_name.get());
271#else
272 Print(*code);
273#endif
274 }
275
276 return code;
277}
