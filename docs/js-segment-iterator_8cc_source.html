31// ecma402 #sec-createsegmentiterator
32MaybeHandle<JSSegmentIterator> JSSegmentIterator::Create(
33 Isolate* isolate, DirectHandle<String> input_string,
34 icu::BreakIterator* incoming_break_iterator,
35 JSSegmenter::Granularity granularity) {
36 // Clone a copy for both the ownership and not sharing with containing and
37 // other calls to the iterator because icu::BreakIterator keep the iteration
38 // position internally and cannot be shared across multiple calls to
39 // JSSegmentIterator::Create and JSSegments::Containing.
40 std::shared_ptr<icu::BreakIterator> break_iterator{
41 incoming_break_iterator->clone()};
42 DCHECK_NOT_NULL(break_iterator);
43 DirectHandle<Map> map(isolate->native_context()->intl_segment_iterator_map(),
44 isolate);
45
46 // 5. Set iterator.[[IteratedStringNextSegmentCodeUnitIndex]] to 0.
47 break_iterator->first();
48 DirectHandle<Managed<icu::BreakIterator>> managed_break_iterator =
49 Managed<icu::BreakIterator>::From(isolate, 0, break_iterator);
50
51 std::shared_ptr<icu::UnicodeString> string =
52 std::make_shared<icu::UnicodeString>();
53 break_iterator->getText().getText(*string);
54 DirectHandle<Managed<icu::UnicodeString>> unicode_string =
55 Managed<icu::UnicodeString>::From(isolate, 0, string);
56
57 break_iterator->setText(*string);
58
59 // Now all properties are ready, so we can allocate the result object.
60 Handle<JSObject> result = isolate->factory()->NewJSObjectFromMap(map);
61 DisallowGarbageCollection no_gc;
62 Handle<JSSegmentIterator> segment_iterator = Cast<JSSegmentIterator>(result);
63
64 segment_iterator->set_flags(0);
65 segment_iterator->set_granularity(granularity);
66 segment_iterator->set_icu_break_iterator(*managed_break_iterator);
67 segment_iterator->set_raw_string(*input_string);
68 segment_iterator->set_unicode_string(*unicode_string);
69
70 return segment_iterator;
71}

73// ecma402 #sec-%segmentiteratorprototype%.next
74MaybeHandle<JSReceiver> JSSegmentIterator::Next(
75 Isolate* isolate, DirectHandle<JSSegmentIterator> segment_iterator) {
76 // Sketches of ideas for future performance improvements, roughly in order
77 // of difficulty:
78 // - Add a fast path for grapheme segmentation of one-byte strings that
79 // entirely skips calling into ICU.
80 // - When we enter this function, perform a batch of calls into ICU and
81 // stash away the results, so the next couple of invocations can access
82 // them from a (Torque?) builtin without calling into C++.
83 // - Implement compiler support for escape-analyzing the JSSegmentDataObject
84 // and avoid allocating it when possible.
85
86 // TODO(v8:14681): We StackCheck here to break execution in the event of an
87 // interrupt. Ordinarily in JS loops, this stack check should already be
88 // occuring, however some loops implemented within CodeStubAssembler and
89 // Torque builtins do not currently implement these checks. A preferable
90 // solution which would benefit other iterators implemented in C++ include:
91 // 1) Performing the stack check in CEntry, which would provide a solution
92 // for all methods implemented in C++.
93 //
94 // 2) Rewriting the loop to include an outer loop, which performs periodic
95 // stack checks every N loop bodies (where N is some arbitrary heuristic
96 // selected to allow short loop counts to run with few interruptions).
97 STACK_CHECK(isolate, MaybeHandle<JSReceiver>());
98
99 Factory* factory = isolate->factory();
100 icu::BreakIterator* icu_break_iterator =
101 segment_iterator->icu_break_iterator()->raw();
102 // 5. Let startIndex be iterator.[[IteratedStringNextSegmentCodeUnitIndex]].
103 int32_t start_index = icu_break_iterator->current();
104 // 6. Let endIndex be ! FindBoundary(segmenter, string, startIndex, after).
105 int32_t end_index = icu_break_iterator->next();
106
107 // 7. If endIndex is not finite, then
108 if (end_index == icu::BreakIterator::DONE) {
109 // a. Return ! CreateIterResultObject(undefined, true).
110 return factory->NewJSIteratorResult(isolate->factory()->undefined_value(),
111 true);
112 }
113
114 // 8. Set iterator.[[IteratedStringNextSegmentCodeUnitIndex]] to endIndex.
115
116 // 9. Let segmentData be ! CreateSegmentDataObject(segmenter, string,
117 // startIndex, endIndex).
118
119 DirectHandle<JSSegmentDataObject> segment_data;
120 if (segment_iterator->granularity() == JSSegmenter::Granularity::GRAPHEME &&
121 start_index == end_index - 1) {
122 // Fast path: use cached segment string and skip avoidable handle creations.
123 DirectHandle<String> segment;
124 uint16_t code = segment_iterator->raw_string()->Get(start_index);
125 if (code > unibrow::Latin1::kMaxChar) {
126 segment = factory->LookupSingleCharacterStringFromCode(code);
127 }
128 DirectHandle<Number> index;
129 if (!Smi::IsValid(start_index)) index = factory->NewHeapNumber(start_index);
130 DirectHandle<Map> map(
131 isolate->native_context()->intl_segment_data_object_map(), isolate);
132 segment_data = Cast<JSSegmentDataObject>(factory->NewJSObjectFromMap(map));
133 Tagged<JSSegmentDataObject> raw = *segment_data;
134 DisallowHeapAllocation no_gc;
135 // We can skip write barriers because {segment_data} is the last object
136 // that was allocated.
137 raw->set_segment(
138 code <= unibrow::Latin1::kMaxChar
139 ? Cast<String>(factory->single_character_string_table()->get(code))
140 : *segment,
141 SKIP_WRITE_BARRIER);
142 raw->set_index(
143 Smi::IsValid(start_index) ? Smi::FromInt(start_index) : *index,
144 SKIP_WRITE_BARRIER);
145 raw->set_input(segment_iterator->raw_string(), SKIP_WRITE_BARRIER);
146 } else {
147 ASSIGN_RETURN_ON_EXCEPTION(
148 isolate, segment_data,
149 JSSegments::CreateSegmentDataObject(
150 isolate, segment_iterator->granularity(), icu_break_iterator,
151 direct_handle(segment_iterator->raw_string(), isolate),
152 *segment_iterator->unicode_string()->raw(), start_index,
153 end_index));
154 }
155
156 // 10. Return ! CreateIterResultObject(segmentData, false).
157 return factory->NewJSIteratorResult(segment_data, false);
158}
