51void WritableRelocInfo::apply(intptr_t delta) {
52 // Absolute code pointer inside code object moves with the code object.
53 if (IsInternalReference(rmode_)) {
54 // Jump table entry
55 Address target = Memory<Address>(pc_);
56 jit_allocation_.WriteValue(pc_, target + delta);
57 } else if (IsCodeTarget(rmode_)) {
58 SixByteInstr instr =
59 Instruction::InstructionBits(reinterpret_cast<const uint8_t*>(pc_));
60 int32_t dis = static_cast<int32_t>(instr & 0xFFFFFFFF) * 2 // halfwords
61 - static_cast<int32_t>(delta);
62 instr >>= 32; // Clear the 4-byte displacement field.
63 instr <<= 32;
64 instr |= static_cast<uint32_t>(dis / 2);
65 Instruction::SetInstructionBits<SixByteInstr>(
66 reinterpret_cast<uint8_t*>(pc_), instr, &jit_allocation_);
67 } else {
68 // mov sequence
69 DCHECK(IsInternalReferenceEncoded(rmode_));
70 Address target = Assembler::target_address_at(pc_, constant_pool_);
71 Assembler::set_target_address_at(pc_, constant_pool_, target + delta,
72 &jit_allocation_, SKIP_ICACHE_FLUSH);
73 }
74}

98Address RelocInfo::target_address_address() {
99 DCHECK(HasTargetAddressAddress());
100
101 // Read the address of the word containing the target_address in an
102 // instruction stream.
103 // The only architecture-independent user of this function is the serializer.
104 // The serializer uses it to find out how many raw bytes of instruction to
105 // output before the next target.
106 // For an instruction like LIS/ORI where the target bits are mixed into the
107 // instruction bits, the size of the target will be zero, indicating that the
108 // serializer should not step forward in memory after a target is resolved
109 // and written.
110 return pc_;
111}

229// Fetch the 32bit value from the FIXED_SEQUENCE IIHF / IILF
230Address Assembler::target_address_at(Address pc, Address constant_pool) {
231 // S390 Instruction!
232 // We want to check for instructions generated by Asm::mov()
233 Opcode op1 =
234 Instruction::S390OpcodeValue(reinterpret_cast<const uint8_t*>(pc));
235 SixByteInstr instr_1 =
236 Instruction::InstructionBits(reinterpret_cast<const uint8_t*>(pc));
237
238 if (BRASL == op1 || BRCL == op1) {
239 int32_t dis = static_cast<int32_t>(instr_1 & 0xFFFFFFFF) * 2;
240 return pc + dis;
241 }
242
243 int instr1_length =
244 Instruction::InstructionLength(reinterpret_cast<const uint8_t*>(pc));
245 Opcode op2 = Instruction::S390OpcodeValue(
246 reinterpret_cast<const uint8_t*>(pc + instr1_length));
247 SixByteInstr instr_2 = Instruction::InstructionBits(
248 reinterpret_cast<const uint8_t*>(pc + instr1_length));
249 // IIHF for hi_32, IILF for lo_32
250 if (IIHF == op1 && IILF == op2) {
251 return static_cast<Address>(((instr_1 & 0xFFFFFFFF) << 32) |
252 ((instr_2 & 0xFFFFFFFF)));
253 }
254
255 UNIMPLEMENTED();
256 return 0;
257}

275// This code assumes the FIXED_SEQUENCE of IIHF/IILF
276void Assembler::set_target_address_at(Address pc, Address constant_pool,
277 Address target,
278 WritableJitAllocation* jit_allocation,
279 ICacheFlushMode icache_flush_mode) {
280 // Check for instructions generated by Asm::mov()
281 Opcode op1 =
282 Instruction::S390OpcodeValue(reinterpret_cast<const uint8_t*>(pc));
283 SixByteInstr instr_1 =
284 Instruction::InstructionBits(reinterpret_cast<const uint8_t*>(pc));
285 bool patched = false;
286
287 if (BRASL == op1 || BRCL == op1) {
288 instr_1 >>= 32; // Zero out the lower 32-bits
289 instr_1 <<= 32;
290 int32_t halfwords = (target - pc) / 2; // number of halfwords
291 instr_1 |= static_cast<uint32_t>(halfwords);
292 Instruction::SetInstructionBits<SixByteInstr>(
293 reinterpret_cast<uint8_t*>(pc), instr_1, jit_allocation);
294 if (icache_flush_mode != SKIP_ICACHE_FLUSH) {
295 FlushInstructionCache(pc, 6);
296 }
297 patched = true;
298 } else {
299 int instr1_length =
300 Instruction::InstructionLength(reinterpret_cast<const uint8_t*>(pc));
301 Opcode op2 = Instruction::S390OpcodeValue(
302 reinterpret_cast<const uint8_t*>(pc + instr1_length));
303 SixByteInstr instr_2 = Instruction::InstructionBits(
304 reinterpret_cast<const uint8_t*>(pc + instr1_length));
305 // IIHF for hi_32, IILF for lo_32
306 if (IIHF == op1 && IILF == op2) {
307 // IIHF
308 instr_1 >>= 32; // Zero out the lower 32-bits
309 instr_1 <<= 32;
310 instr_1 |= reinterpret_cast<uint64_t>(target) >> 32;
311
312 Instruction::SetInstructionBits<SixByteInstr>(
313 reinterpret_cast<uint8_t*>(pc), instr_1, jit_allocation);
314
315 // IILF
316 instr_2 >>= 32;
317 instr_2 <<= 32;
318 instr_2 |= reinterpret_cast<uint64_t>(target) & 0xFFFFFFFF;
319
320 Instruction::SetInstructionBits<SixByteInstr>(
321 reinterpret_cast<uint8_t*>(pc + instr1_length), instr_2,
322 jit_allocation);
323 if (icache_flush_mode != SKIP_ICACHE_FLUSH) {
324 FlushInstructionCache(pc, 12);
325 }
326 patched = true;
327 }
328 }
329 if (!patched) UNREACHABLE();
330}
