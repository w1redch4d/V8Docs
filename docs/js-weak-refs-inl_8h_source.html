29void JSFinalizationRegistry::RegisterWeakCellWithUnregisterToken(
30 DirectHandle<JSFinalizationRegistry> finalization_registry,
31 DirectHandle<WeakCell> weak_cell, Isolate* isolate) {
32 Handle<SimpleNumberDictionary> key_map;
33 if (IsUndefined(finalization_registry->key_map(), isolate)) {
34 key_map = SimpleNumberDictionary::New(isolate, 1);
35 } else {
36 key_map =
37 handle(Cast<SimpleNumberDictionary>(finalization_registry->key_map()),
38 isolate);
39 }
40
41 // Unregister tokens are held weakly as objects are often their own
42 // unregister token. To avoid using an ephemeron map, the map for token
43 // lookup is keyed on the token's identity hash instead of the token itself.
44 uint32_t key =
45 Object::GetOrCreateHash(weak_cell->unregister_token(), isolate).value();
46 InternalIndex entry = key_map->FindEntry(isolate, key);
47 if (entry.is_found()) {
48 Tagged<Object> value = key_map->ValueAt(entry);
49 Tagged<WeakCell> existing_weak_cell = Cast<WeakCell>(value);
50 existing_weak_cell->set_key_list_prev(*weak_cell);
51 weak_cell->set_key_list_next(existing_weak_cell);
52 }
53 key_map = SimpleNumberDictionary::Set(isolate, key_map, key, weak_cell);
54 finalization_registry->set_key_map(*key_map);
55}

57bool JSFinalizationRegistry::Unregister(
58 DirectHandle<JSFinalizationRegistry> finalization_registry,
59 DirectHandle<HeapObject> unregister_token, Isolate* isolate) {
60 // Iterate through the doubly linked list of WeakCells associated with the
61 // key. Each WeakCell will be in the "active_cells" or "cleared_cells" list of
62 // its FinalizationRegistry; remove it from there.
63 return finalization_registry->RemoveUnregisterToken(
64 *unregister_token, isolate, kRemoveMatchedCellsFromRegistry,
65 [](Tagged<HeapObject>, ObjectSlot, Tagged<Object>) {});
66}

68template <typename GCNotifyUpdatedSlotCallback>
69bool JSFinalizationRegistry::RemoveUnregisterToken(
70 Tagged<HeapObject> unregister_token, Isolate* isolate,
71 RemoveUnregisterTokenMode removal_mode,
72 GCNotifyUpdatedSlotCallback gc_notify_updated_slot) {
73 // This method is called from both FinalizationRegistry#unregister and for
74 // removing weakly-held dead unregister tokens. The latter is during GC so
75 // this function cannot GC.
76 DisallowGarbageCollection no_gc;
77 if (IsUndefined(key_map(), isolate)) {
78 return false;
79 }
80
81 Tagged<SimpleNumberDictionary> key_map =
82 Cast<SimpleNumberDictionary>(this->key_map());
83 // If the token doesn't have a hash, it was not used as a key inside any hash
84 // tables.
85 Tagged<Object> hash = Object::GetHash(unregister_token);
86 if (IsUndefined(hash, isolate)) {
87 return false;
88 }
89 uint32_t key = Smi::ToInt(hash);
90 InternalIndex entry = key_map->FindEntry(isolate, key);
91 if (entry.is_not_found()) {
92 return false;
93 }
94
95 Tagged<Object> value = key_map->ValueAt(entry);
96 bool was_present = false;
97 Tagged<Undefined> undefined = ReadOnlyRoots(isolate).undefined_value();
98 Tagged<UnionOf<Undefined, WeakCell>> new_key_list_head = undefined;
99 Tagged<UnionOf<Undefined, WeakCell>> new_key_list_prev = undefined;
100 // Compute a new key list that doesn't have unregister_token. Because
101 // unregister tokens are held weakly, key_map is keyed using the tokens'
102 // identity hashes, and identity hashes may collide.
103 while (!IsUndefined(value, isolate)) {
104 Tagged<WeakCell> weak_cell = Cast<WeakCell>(value);
105 DCHECK(!HeapLayout::InYoungGeneration(weak_cell));
106 value = weak_cell->key_list_next();
107 if (weak_cell->unregister_token() == unregister_token) {
108 // weak_cell has the same unregister token; remove it from the key list.
109 switch (removal_mode) {
110 case kRemoveMatchedCellsFromRegistry:
111 weak_cell->RemoveFromFinalizationRegistryCells(isolate);
112 break;
113 case kKeepMatchedCellsInRegistry:
114 // Do nothing.
115 break;
116 }
117 // Clear unregister token-related fields.
118 weak_cell->set_unregister_token(undefined);
119 weak_cell->set_key_list_prev(undefined);
120 weak_cell->set_key_list_next(undefined);
121 was_present = true;
122 } else {
123 // weak_cell has a different unregister token with the same key (hash
124 // collision); fix up the list.
125 weak_cell->set_key_list_prev(new_key_list_prev);
126 gc_notify_updated_slot(weak_cell,
127 weak_cell->RawField(WeakCell::kKeyListPrevOffset),
128 new_key_list_prev);
129 weak_cell->set_key_list_next(undefined);
130 if (IsUndefined(new_key_list_prev, isolate)) {
131 new_key_list_head = weak_cell;
132 } else {
133 DCHECK(IsWeakCell(new_key_list_head));
134 Tagged<WeakCell> prev_cell = Cast<WeakCell>(new_key_list_prev);
135 prev_cell->set_key_list_next(weak_cell);
136 gc_notify_updated_slot(
137 prev_cell, prev_cell->RawField(WeakCell::kKeyListNextOffset),
138 weak_cell);
139 }
140 new_key_list_prev = weak_cell;
141 }
142 }
143 if (IsUndefined(new_key_list_head, isolate)) {
144 DCHECK(was_present);
145 key_map->ClearEntry(entry);
146 key_map->ElementRemoved();
147 } else {
148 key_map->ValueAtPut(entry, new_key_list_head);
149 gc_notify_updated_slot(key_map, key_map->RawFieldOfValueAt(entry),
150 new_key_list_head);
151 }
152 return was_present;
153}

167template <typename GCNotifyUpdatedSlotCallback>
168void WeakCell::Nullify(Isolate* isolate,
169 GCNotifyUpdatedSlotCallback gc_notify_updated_slot) {
170 // Remove from the WeakCell from the "active_cells" list of its
171 // JSFinalizationRegistry and insert it into the "cleared_cells" list. This is
172 // only called for WeakCells which haven't been unregistered yet, so they will
173 // be in the active_cells list. (The caller must guard against calling this
174 // for unregistered WeakCells by checking that the target is not undefined.)
175 DCHECK(Object::CanBeHeldWeakly(target()));
176 set_target(ReadOnlyRoots(isolate).undefined_value());
177
178 Tagged<JSFinalizationRegistry> fr =
179 Cast<JSFinalizationRegistry>(finalization_registry());
180 if (IsWeakCell(prev())) {
181 DCHECK_NE(fr->active_cells(), *this);
182 Tagged<WeakCell> prev_cell = Cast<WeakCell>(prev());
183 prev_cell->set_next(next());
184 gc_notify_updated_slot(prev_cell,
185 prev_cell->RawField(WeakCell::kNextOffset), next());
186 } else {
187 DCHECK_EQ(fr->active_cells(), *this);
188 fr->set_active_cells(next());
189 gc_notify_updated_slot(
190 fr, fr->RawField(JSFinalizationRegistry::kActiveCellsOffset), next());
191 }
192 if (IsWeakCell(next())) {
193 Tagged<WeakCell> next_cell = Cast<WeakCell>(next());
194 next_cell->set_prev(prev());
195 gc_notify_updated_slot(next_cell,
196 next_cell->RawField(WeakCell::kPrevOffset), prev());
197 }
198
199 set_prev(ReadOnlyRoots(isolate).undefined_value());
200 Tagged<UnionOf<Undefined, WeakCell>> cleared_head = fr->cleared_cells();
201 if (IsWeakCell(cleared_head)) {
202 Tagged<WeakCell> cleared_head_cell = Cast<WeakCell>(cleared_head);
203 cleared_head_cell->set_prev(*this);
204 gc_notify_updated_slot(cleared_head_cell,
205 cleared_head_cell->RawField(WeakCell::kPrevOffset),
206 *this);
207 }
208 set_next(fr->cleared_cells());
209 gc_notify_updated_slot(*this, RawField(WeakCell::kNextOffset), next());
210 fr->set_cleared_cells(*this);
211 gc_notify_updated_slot(
212 fr, fr->RawField(JSFinalizationRegistry::kClearedCellsOffset), *this);
213}

215void WeakCell::RemoveFromFinalizationRegistryCells(Isolate* isolate) {
216 // Remove the WeakCell from the list it's in (either "active_cells" or
217 // "cleared_cells" of its JSFinalizationRegistry).
218
219 // It's important to set_target to undefined here. This guards that we won't
220 // call Nullify (which assumes that the WeakCell is in active_cells).
221 DCHECK(IsUndefined(target()) || Object::CanBeHeldWeakly(target()));
222 set_target(ReadOnlyRoots(isolate).undefined_value());
223
224 Tagged<JSFinalizationRegistry> fr =
225 Cast<JSFinalizationRegistry>(finalization_registry());
226 if (fr->active_cells() == *this) {
227 DCHECK(IsUndefined(prev(), isolate));
228 fr->set_active_cells(next());
229 } else if (fr->cleared_cells() == *this) {
230 DCHECK(!IsWeakCell(prev()));
231 fr->set_cleared_cells(next());
232 } else {
233 DCHECK(IsWeakCell(prev()));
234 Tagged<WeakCell> prev_cell = Cast<WeakCell>(prev());
235 prev_cell->set_next(next());
236 }
237 if (IsWeakCell(next())) {
238 Tagged<WeakCell> next_cell = Cast<WeakCell>(next());
239 next_cell->set_prev(prev());
240 }
241 set_prev(ReadOnlyRoots(isolate).undefined_value());
242 set_next(ReadOnlyRoots(isolate).undefined_value());
243}
