93v8::CodeEventType GetCodeEventTypeForTag(LogEventListener::CodeTag tag) {
94 switch (tag) {
95 case LogEventListener::CodeTag::kLength:
96 // Manually create this switch, since v8::CodeEventType is API expose and
97 // cannot be easily modified.
98 case LogEventListener::CodeTag::kBuiltin:
99 return v8::CodeEventType::kBuiltinType;
100 case LogEventListener::CodeTag::kCallback:
101 return v8::CodeEventType::kCallbackType;
102 case LogEventListener::CodeTag::kEval:
103 return v8::CodeEventType::kEvalType;
104 case LogEventListener::CodeTag::kNativeFunction:
105 case LogEventListener::CodeTag::kFunction:
106 return v8::CodeEventType::kFunctionType;
107 case LogEventListener::CodeTag::kHandler:
108 return v8::CodeEventType::kHandlerType;
109 case LogEventListener::CodeTag::kBytecodeHandler:

111 case LogEventListener::CodeTag::kRegExp:

113 case LogEventListener::CodeTag::kNativeScript:
114 case LogEventListener::CodeTag::kScript:
115 return v8::CodeEventType::kScriptType;

129const char* ComputeMarker(Tagged<SharedFunctionInfo> shared,
130 Tagged<AbstractCode> code) {
131 PtrComprCageBase cage_base = GetPtrComprCageBase(shared);
132 CodeKind kind = code->kind(cage_base);
133 // We record interpreter trampoline builtin copies as having the
134 // "interpreted" marker.
135 if (v8_flags.interpreted_frames_native_stack && kind == CodeKind::BUILTIN &&
136 code->has_instruction_stream(cage_base) &&
137 code->builtin_id(cage_base) == Builtin::kInterpreterEntryTrampoline) {
138 kind = CodeKind::INTERPRETED_FUNCTION;
139 }
140 if (shared->optimization_disabled() &&
141 kind == CodeKind::INTERPRETED_FUNCTION) {
142 return "";
143 }
144 return CodeKindToMarker(
145 kind, IsCode(code) && code->GetCode()->is_context_specialized());
146}

149const char* ComputeMarker(const wasm::WasmCode* code) {
150 switch (code->kind()) {

152 return code->is_liftoff() ? "" : "*";
153 default:
154 return "";
155 }

161class CodeEventLogger::NameBuffer {
162 public:
163 NameBuffer() { Reset(); }
164
165 void Reset() { utf8_pos_ = 0; }
166
167 void Init(CodeTag tag) {
168 Reset();
169 AppendBytes(kCodeTagNames[static_cast<int>(tag)]);
170 AppendByte(':');
171 }
172
173 void AppendName(Tagged<Name> name) {
174 if (IsString(name)) {
175 AppendString(Cast<String>(name));
176 } else {
177 Tagged<Symbol> symbol = Cast<Symbol>(name);
178 AppendBytes("symbol(");
179 if (!IsUndefined(symbol->description())) {
180 AppendBytes("\"");
181 AppendString(Cast<String>(symbol->description()));
182 AppendBytes("\" ");
183 }
184 AppendBytes("hash ");
185 AppendHex(symbol->hash());
186 AppendByte(')');
187 }
188 }
189
190 void AppendString(Tagged<String> str) {
191 if (str.is_null()) return;
192 size_t length = 0;
193 std::unique_ptr<char[]> c_str = str->ToCString(&length);
194 AppendBytes(c_str.get(), length);
195 }
196
197 void AppendBytes(const char* bytes, size_t size) {
198 size = std::min(size, kUtf8BufferSize - utf8_pos_);
199 MemCopy(utf8_buffer_ + utf8_pos_, bytes, size);
200 utf8_pos_ += size;
201 }
202
203 void AppendBytes(const char* bytes) {
204 size_t len = strlen(bytes);
205 DCHECK_GE(kMaxInt, len);
206 AppendBytes(bytes, static_cast<int>(len));
207 }
208
209 void AppendByte(char c) {
210 if (utf8_pos_ >= kUtf8BufferSize) return;
211 utf8_buffer_[utf8_pos_++] = c;
212 }
213
214 void AppendInt(int n) {
215 if (utf8_pos_ >= kUtf8BufferSize) return;
216 size_t space = kUtf8BufferSize - utf8_pos_;
217 base::Vector<char> buffer(utf8_buffer_ + utf8_pos_, space);
218 int size = SNPrintF(buffer, "%d", n);
219 if (size > 0 && utf8_pos_ + size <= kUtf8BufferSize) {
220 utf8_pos_ += size;
221 }
222 }
223
224 void AppendHex(uint32_t n) {
225 if (utf8_pos_ >= kUtf8BufferSize) return;
226 size_t space = kUtf8BufferSize - utf8_pos_;
227 base::Vector<char> buffer(utf8_buffer_ + utf8_pos_, space);
228 int size = SNPrintF(buffer, "%x", n);
229 if (size > 0 && utf8_pos_ + size <= kUtf8BufferSize) {
230 utf8_pos_ += size;
231 }
232 }
233
234 const char* get() { return utf8_buffer_; }
235 size_t size() const { return utf8_pos_; }
236
237 private:
238 static const size_t kUtf8BufferSize = 4096;
239 static const size_t kUtf16BufferSize = kUtf8BufferSize;
240
241 size_t utf8_pos_;
242 char utf8_buffer_[kUtf8BufferSize];
243};

395// The following static variables are protected by
396// LinuxPerfBasicLogger::GetFileMutex().
397uint64_t LinuxPerfBasicLogger::reference_count_ = 0;
398FILE* LinuxPerfBasicLogger::perf_output_handle_ = nullptr;

400LinuxPerfBasicLogger::LinuxPerfBasicLogger(Isolate* isolate)
401 : CodeEventLogger(isolate) {
402 base::LockGuard<base::RecursiveMutex> guard_file(GetFileMutex().Pointer());
403 int process_id_ = base::OS::GetCurrentProcessId();
404 reference_count_++;
405 // If this is the first logger, open the file.
406 if (reference_count_ == 1) {
407 CHECK_NULL(perf_output_handle_);
408 CHECK_NOT_NULL(v8_flags.perf_basic_prof_path);
409 const char* base_dir = v8_flags.perf_basic_prof_path;
410 // Open the perf JIT dump file.
411 base::ScopedVector<char> perf_dump_name(strlen(base_dir) +
412 kFilenameBufferPadding);
413 int size =
414 SNPrintF(perf_dump_name, "%s/perf-%d.map", base_dir, process_id_);
415 CHECK_NE(size, -1);
416 perf_output_handle_ =
417 base::OS::FOpen(perf_dump_name.begin(), base::OS::LogFileOpenMode);
418 CHECK_NOT_NULL(perf_output_handle_);
419 setvbuf(perf_output_handle_, nullptr, _IOLBF, 0);
420 }
421}

429 CHECK_NOT_NULL(perf_output_handle_);

431 perf_output_handle_ = nullptr;

435void LinuxPerfBasicLogger::WriteLogRecordedBuffer(uintptr_t address,
436 size_t size, const char* name,
437 size_t name_length) {
438 // Linux perf expects hex literals without a leading 0x, while some
439 // implementations of printf might prepend one when using the %p format
440 // for pointers, leading to wrongly formatted JIT symbols maps. On the other
441 // hand, Android's simpleperf does expect a leading 0x.
442 //
443 // Instead, we use V8PRIxPTR format string and cast pointer to uintpr_t,
444 // so that we have control over the exact output format.
445 int int_name_length = static_cast<int>(name_length);
446#ifdef V8_OS_ANDROID
447 base::OS::FPrint(perf_output_handle_, "0x%" V8PRIxPTR " 0x%zx %.*s\n",
448 address, size, int_name_length, name);
449#else
450 base::OS::FPrint(perf_output_handle_, "%" V8PRIxPTR " %zx %.*s\n", address,
451 size, int_name_length, name);
452#endif
453}

1053// The Profiler samples pc and sp values for the main thread.
1054// Each sample is appended to a circular buffer.
1055// An independent thread removes data and writes it to the log.
1056// This design minimizes the time spent in the sampler.
1057//
1058class Profiler : public base::Thread {
1059 public:
1060 explicit Profiler(Isolate* isolate);
1061 void Engage();
1062 void Disengage();
1063
1064 // Inserts collected profiling data into buffer.
1065 void Insert(TickSample* sample) {
1066 if (Succ(head_) == static_cast<int>(base::Acquire_Load(&tail_))) {
1067 base::Relaxed_Store(&overflow_, true);
1068 } else {
1069 buffer_[head_] = *sample;
1070 head_ = Succ(head_);
1071 buffer_semaphore_.Signal(); // Tell we have an element.
1072 }
1073 }
1074
1075 void Run() override;
1076
1077 private:
1078 // Waits for a signal and removes profiling data.
1079 bool Remove(TickSample* sample) {
1080 buffer_semaphore_.Wait(); // Wait for an element.
1081 *sample = buffer_[base::Relaxed_Load(&tail_)];
1082 bool result = base::Relaxed_Load(&overflow_);
1083 base::Release_Store(
1084 &tail_, static_cast<base::Atomic32>(Succ(base::Relaxed_Load(&tail_))));
1085 base::Relaxed_Store(&overflow_, false);
1086 return result;
1087 }
1088
1089 // Returns the next index in the cyclic buffer.
1090 int Succ(int index) { return (index + 1) % kBufferSize; }
1091
1092 Isolate* isolate_;
1093 // Cyclic buffer for communicating profiling samples
1094 // between the signal handler and the worker thread.
1095 static const int kBufferSize = 128;
1096 TickSample buffer_[kBufferSize]; // Buffer storage.
1097 int head_; // Index to the buffer head.
1098 base::Atomic32 tail_; // Index to the buffer tail.
1099 base::Atomic32 overflow_; // Tell whether a buffer overflow has occurred.
1100 // Semaphore used for buffer synchronization.
1101 base::Semaphore buffer_semaphore_;
1102
1103 // Tells whether worker thread should continue running.
1104 base::Atomic32 running_;
1105};

1382// We log source code information in the form:
1383//
1384// code-source-info <addr>,<script>,<start>,<end>,<pos>,<inline-pos>,<fns>
1385//
1386// where
1387// <addr> is code object address
1388// <script> is script id
1389// <start> is the starting position inside the script
1390// <end> is the end position inside the script
1391// <pos> is source position table encoded in the string,
1392// it is a sequence of C<code-offset>O<script-offset>[I<inlining-id>]
1393// where
1394// <code-offset> is the offset within the code object
1395// <script-offset> is the position within the script
1396// <inlining-id> is the offset in the <inlining> table
1397// <inlining> table is a sequence of strings of the form
1398// F<function-id>O<script-offset>[I<inlining-id>]
1399// where
1400// <function-id> is an index into the <fns> function table
1401// <fns> is the function table encoded as a sequence of strings
1402// S<shared-function-info-address>
1403
1404void V8FileLogger::LogSourceCodeInformation(
1405 DirectHandle<AbstractCode> code, DirectHandle<SharedFunctionInfo> shared) {
1406 PtrComprCageBase cage_base(isolate_);
1407 Tagged<Object> script_object = shared->script(cage_base);
1408 if (!IsScript(script_object, cage_base)) return;
1409 Tagged<Script> script = Cast<Script>(script_object);
1410 EnsureLogScriptSource(script);
1411
1412 if (!v8_flags.log_source_position) return;
1413 VMStateIfMainThread<LOGGING> state(isolate_);
1414 MSG_BUILDER();
1415 msg << "code-source-info" << V8FileLogger::kNext
1416 << reinterpret_cast<void*>(code->InstructionStart(cage_base))
1417 << V8FileLogger::kNext << script->id() << V8FileLogger::kNext
1418 << shared->StartPosition() << V8FileLogger::kNext << shared->EndPosition()
1419 << V8FileLogger::kNext;
1420 // TODO(v8:11429): Clean-up baseline-replated code in source position
1421 // iteration.
1422 bool hasInlined = false;
1423 if (code->kind(cage_base) != CodeKind::BASELINE) {
1424 SourcePositionTableIterator iterator(
1425 code->SourcePositionTable(isolate_, *shared));
1426 for (; !iterator.done(); iterator.Advance()) {
1427 SourcePosition pos = iterator.source_position();
1428 msg << "C" << iterator.code_offset() << "O" << pos.ScriptOffset();
1429 if (pos.isInlined()) {
1430 msg << "I" << pos.InliningId();
1431 hasInlined = true;
1432 }
1433 }
1434 }
1435 msg << V8FileLogger::kNext;
1436 int maxInlinedId = -1;
1437 if (hasInlined) {
1438 Tagged<TrustedPodArray<InliningPosition>> inlining_positions =
1439 Cast<DeoptimizationData>(Cast<Code>(code)->deoptimization_data())
1440 ->InliningPositions();
1441 for (int i = 0; i < inlining_positions->length(); i++) {
1442 InliningPosition inlining_pos = inlining_positions->get(i);
1443 msg << "F";
1444 if (inlining_pos.inlined_function_id != -1) {
1445 msg << inlining_pos.inlined_function_id;
1446 if (inlining_pos.inlined_function_id > maxInlinedId) {
1447 maxInlinedId = inlining_pos.inlined_function_id;
1448 }
1449 }
1450 SourcePosition pos = inlining_pos.position;
1451 msg << "O" << pos.ScriptOffset();
1452 if (pos.isInlined()) {
1453 msg << "I" << pos.InliningId();
1454 }
1455 }
1456 }
1457 msg << V8FileLogger::kNext;
1458 if (hasInlined) {
1459 Tagged<DeoptimizationData> deopt_data =
1460 Cast<DeoptimizationData>(Cast<Code>(code)->deoptimization_data());
1461 msg << std::hex;
1462 for (int i = 0; i <= maxInlinedId; i++) {
1463 msg << "S"
1464 << reinterpret_cast<void*>(
1465 deopt_data->GetInlinedFunction(i).address());
1466 }
1467 msg << std::dec;
1468 }
1469 msg.WriteToLogFile();
1470}

2084static std::vector<std::pair<Handle<SharedFunctionInfo>, Handle<AbstractCode>>>
2085EnumerateCompiledFunctions(Heap* heap) {
2086 HeapObjectIterator iterator(heap);
2087 DisallowGarbageCollection no_gc;
2088 std::vector<std::pair<Handle<SharedFunctionInfo>, Handle<AbstractCode>>>
2089 compiled_funcs;
2090 Isolate* isolate = heap->isolate();
2091 auto hash =
2092 [](const std::pair<Tagged<SharedFunctionInfo>, Tagged<AbstractCode>>& p) {
2093 return base::hash_combine(p.first.address(), p.second.address());
2094 };
2095 std::unordered_set<
2096 std::pair<Tagged<SharedFunctionInfo>, Tagged<AbstractCode>>,
2097 decltype(hash)>
2098 seen(8, hash);
2099
2100 auto record = [&](Tagged<SharedFunctionInfo> sfi, Tagged<AbstractCode> c) {
2101 if (auto [iter, inserted] = seen.emplace(sfi, c); inserted)
2102 compiled_funcs.emplace_back(handle(sfi, isolate), handle(c, isolate));
2103 };
2104
2105 // Iterate the heap to find JSFunctions and record their optimized code.
2106 for (Tagged<HeapObject> obj = iterator.Next(); !obj.is_null();
2107 obj = iterator.Next()) {
2108 if (IsSharedFunctionInfo(obj)) {
2109 Tagged<SharedFunctionInfo> sfi = Cast<SharedFunctionInfo>(obj);
2110 if (sfi->is_compiled() && !sfi->HasBytecodeArray()) {
2111 record(sfi, Cast<AbstractCode>(sfi->abstract_code(isolate)));
2112 }
2113 } else if (IsJSFunction(obj)) {
2114 // Given that we no longer iterate over all optimized JSFunctions, we need
2115 // to take care of this here.
2116 Tagged<JSFunction> function = Cast<JSFunction>(obj);
2117 // TODO(jarin) This leaves out deoptimized code that might still be on the
2118 // stack. Also note that we will not log optimized code objects that are
2119 // only on a type feedback vector. We should make this more precise.
2120 if (function->HasAttachedOptimizedCode(isolate) &&
2121 Cast<Script>(function->shared()->script())->HasValidSource()) {
2122 record(function->shared(), Cast<AbstractCode>(function->code(isolate)));
2123 }
2124 }
2125 }
2126
2127 Script::Iterator script_iterator(heap->isolate());
2128 for (Tagged<Script> script = script_iterator.Next(); !script.is_null();
2129 script = script_iterator.Next()) {
2130 if (!script->HasValidSource()) continue;
2131
2132 SharedFunctionInfo::ScriptIterator sfi_iterator(heap->isolate(), script);
2133 for (Tagged<SharedFunctionInfo> sfi = sfi_iterator.Next(); !sfi.is_null();
2134 sfi = sfi_iterator.Next()) {
2135 if (sfi->is_compiled()) {
2136 record(sfi, Cast<AbstractCode>(sfi->abstract_code(isolate)));
2137 }
2138 }
2139 }
2140
2141 return compiled_funcs;
2142}

2571void ExistingCodeLogger::LogCompiledFunctions(
2572 bool ensure_source_positions_available) {
2573 Heap* heap = isolate_->heap();
2574 HandleScope scope(isolate_);
2575 std::vector<std::pair<Handle<SharedFunctionInfo>, Handle<AbstractCode>>>
2576 compiled_funcs = EnumerateCompiledFunctions(heap);
2577
2578 // During iteration, there can be heap allocation due to
2579 // GetScriptLineNumber call.
2580 for (auto& pair : compiled_funcs) {
2581 Handle<SharedFunctionInfo> shared = pair.first;
2582
2583 // If the script is a Smi, then the SharedFunctionInfo is in
2584 // the process of being deserialized.
2585 Tagged<Object> script = shared->raw_script(kAcquireLoad);
2586 if (IsSmi(script)) {
2587 DCHECK_EQ(script, Smi::uninitialized_deserialization_value());
2588 continue;
2589 }
2590
2591 if (ensure_source_positions_available) {
2592 SharedFunctionInfo::EnsureSourcePositionsAvailable(isolate_, shared);
2593 }
2594 if (shared->HasInterpreterData(isolate_)) {
2595 LogExistingFunction(
2596 shared,
2597 Handle<AbstractCode>(
2598 Cast<AbstractCode>(shared->InterpreterTrampoline(isolate_)),
2599 isolate_));
2600 }
2601 if (shared->HasBaselineCode()) {
2602 LogExistingFunction(
2603 shared, Handle<AbstractCode>(
2604 Cast<AbstractCode>(shared->baseline_code(kAcquireLoad)),
2605 isolate_));
2606 }
2607 // TODO(saelo): remove the "!IsTrustedSpaceObject" once builtin Code
2608 // objects are also in trusted space. Currently this breaks because we must
2609 // not compare objects in trusted space with ones inside the sandbox.
2610 static_assert(!kAllCodeObjectsLiveInTrustedSpace);
2611 if (!HeapLayout::InTrustedSpace(*pair.second) &&
2612 pair.second.is_identical_to(BUILTIN_CODE(isolate_, CompileLazy))) {
2613 continue;
2614 }
2615 LogExistingFunction(pair.first, pair.second);
2616 }
2617
2618#if V8_ENABLE_WEBASSEMBLY
2619 HeapObjectIterator iterator(heap);
2620 DisallowGarbageCollection no_gc;
2621
2622 for (Tagged<HeapObject> obj = iterator.Next(); !obj.is_null();
