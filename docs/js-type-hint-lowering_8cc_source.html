20bool BinaryOperationHintToNumberOperationHint(
21 BinaryOperationHint binop_hint, NumberOperationHint* number_hint) {
22 switch (binop_hint) {
23 case BinaryOperationHint::kSignedSmall:
24 *number_hint = NumberOperationHint::kSignedSmall;
25 return true;
26 case BinaryOperationHint::kSignedSmallInputs:
27 *number_hint = NumberOperationHint::kSignedSmallInputs;
28 return true;
29 case BinaryOperationHint::kNumber:
30 *number_hint = NumberOperationHint::kNumber;
31 return true;
32 case BinaryOperationHint::kNumberOrOddball:
33 *number_hint = NumberOperationHint::kNumberOrOddball;
34 return true;
35 case BinaryOperationHint::kAny:
36 case BinaryOperationHint::kNone:
37 case BinaryOperationHint::kString:
38 case BinaryOperationHint::kStringOrStringWrapper:
39 case BinaryOperationHint::kBigInt:
40 case BinaryOperationHint::kBigInt64:
41 break;
42 }
43 return false;
44}
45
46bool BinaryOperationHintToBigIntOperationHint(
47 BinaryOperationHint binop_hint, BigIntOperationHint* bigint_hint) {
48 switch (binop_hint) {
49 case BinaryOperationHint::kSignedSmall:
50 case BinaryOperationHint::kSignedSmallInputs:
51 case BinaryOperationHint::kNumber:
52 case BinaryOperationHint::kNumberOrOddball:
53 case BinaryOperationHint::kAny:
54 case BinaryOperationHint::kNone:
55 case BinaryOperationHint::kString:
56 case BinaryOperationHint::kStringOrStringWrapper:
57 return false;
58 case BinaryOperationHint::kBigInt64:
59 *bigint_hint = BigIntOperationHint::kBigInt64;
60 return true;
61 case BinaryOperationHint::kBigInt:
62 *bigint_hint = BigIntOperationHint::kBigInt;
63 return true;
64 }
65 UNREACHABLE();
66}

144 const Operator* SpeculativeNumberOp(NumberOperationHint hint) {
145 switch (op_->opcode()) {
146 case IrOpcode::kJSAdd:
147 if (hint == NumberOperationHint::kSignedSmall) {
148 return simplified()->SpeculativeSafeIntegerAdd(hint);
149 } else {
150 return simplified()->SpeculativeNumberAdd(hint);
151 }
152 case IrOpcode::kJSSubtract:
153 if (hint == NumberOperationHint::kSignedSmall) {
154 return simplified()->SpeculativeSafeIntegerSubtract(hint);
155 } else {
156 return simplified()->SpeculativeNumberSubtract(hint);
157 }
158 case IrOpcode::kJSMultiply:
159 return simplified()->SpeculativeNumberMultiply(hint);
160 case IrOpcode::kJSExponentiate:
161 return simplified()->SpeculativeNumberPow(hint);
162 case IrOpcode::kJSDivide:
163 return simplified()->SpeculativeNumberDivide(hint);
164 case IrOpcode::kJSModulus:
165 return simplified()->SpeculativeNumberModulus(hint);
166 case IrOpcode::kJSBitwiseAnd:
167 return simplified()->SpeculativeNumberBitwiseAnd(hint);
168 case IrOpcode::kJSBitwiseOr:
169 return simplified()->SpeculativeNumberBitwiseOr(hint);
170 case IrOpcode::kJSBitwiseXor:
171 return simplified()->SpeculativeNumberBitwiseXor(hint);
172 case IrOpcode::kJSShiftLeft:
173 return simplified()->SpeculativeNumberShiftLeft(hint);
174 case IrOpcode::kJSShiftRight:
175 return simplified()->SpeculativeNumberShiftRight(hint);
176 case IrOpcode::kJSShiftRightLogical:
177 return simplified()->SpeculativeNumberShiftRightLogical(hint);
178 default:
179 break;
180 }
181 UNREACHABLE();
182 }

212 const Operator* SpeculativeNumberCompareOp(NumberOperationHint hint) {
213 switch (op_->opcode()) {
214 case IrOpcode::kJSEqual:
215 return simplified()->SpeculativeNumberEqual(hint);
216 case IrOpcode::kJSLessThan:
217 return simplified()->SpeculativeNumberLessThan(hint);
218 case IrOpcode::kJSGreaterThan:
219 std::swap(left_, right_); // a > b => b < a
220 return simplified()->SpeculativeNumberLessThan(hint);
221 case IrOpcode::kJSLessThanOrEqual:
222 return simplified()->SpeculativeNumberLessThanOrEqual(hint);
223 case IrOpcode::kJSGreaterThanOrEqual:
224 std::swap(left_, right_); // a >= b => b <= a
225 return simplified()->SpeculativeNumberLessThanOrEqual(hint);
226 default:
227 break;
228 }
229 UNREACHABLE();
230 }

252 Node* BuildSpeculativeOperation(const Operator* op) {
253 DCHECK_EQ(2, op->ValueInputCount());
254 DCHECK_EQ(1, op->EffectInputCount());
255 DCHECK_EQ(1, op->ControlInputCount());
256 DCHECK_EQ(false, OperatorProperties::HasFrameStateInput(op));
257 DCHECK_EQ(false, OperatorProperties::HasContextInput(op));
258 DCHECK_EQ(1, op->EffectOutputCount());
259 DCHECK_EQ(0, op->ControlOutputCount());
260 return graph()->NewNode(op, left_, right_, effect_, control_);
261 }

350JSTypeHintLowering::LoweringResult JSTypeHintLowering::ReduceUnaryOperation(
351 const Operator* op, Node* operand, Node* effect, Node* control,
352 FeedbackSlot slot) const {
353 if (Node* node = BuildDeoptIfFeedbackIsInsufficient(
354 slot, effect, control,
355 DeoptimizeReason::kInsufficientTypeFeedbackForUnaryOperation)) {
356 return LoweringResult::Exit(node);
357 }
358
359 // Note: Unary and binary operations collect the same kind of feedback.
360 FeedbackSource feedback(feedback_vector(), slot);
361
362 Node* node;
363 Node* check = nullptr;
364 switch (op->opcode()) {
365 case IrOpcode::kJSBitwiseNot: {
366 // Lower to a speculative xor with -1 if we have some kind of Number
367 // feedback.
368 JSSpeculativeBinopBuilder b(
369 this, jsgraph()->javascript()->BitwiseXor(feedback), operand,
370 jsgraph()->SmiConstant(-1), effect, control, slot);
371 node = b.TryBuildNumberBinop();
372 break;
373 }
374 case IrOpcode::kJSDecrement: {
375 // Lower to a speculative subtraction of 1 if we have some kind of Number
376 // feedback.
377 JSSpeculativeBinopBuilder b(
378 this, jsgraph()->javascript()->Subtract(feedback), operand,
379 jsgraph()->SmiConstant(1), effect, control, slot);
380 node = b.TryBuildNumberBinop();
381 break;
382 }
383 case IrOpcode::kJSIncrement: {
384 // Lower to a speculative addition of 1 if we have some kind of Number
385 // feedback.
386 JSSpeculativeBinopBuilder b(this, jsgraph()->javascript()->Add(feedback),
387 operand, jsgraph()->SmiConstant(1), effect,
388 control, slot);
389 node = b.TryBuildNumberBinop();
390 break;
391 }
392 case IrOpcode::kJSNegate: {
393 // Lower to a speculative multiplication with -1 if we have some kind of
394 // Number feedback.
395 JSSpeculativeBinopBuilder b(
396 this, jsgraph()->javascript()->Multiply(feedback), operand,
397 jsgraph()->SmiConstant(-1), effect, control, slot);
398 node = b.TryBuildNumberBinop();
399 if (!node) {
400 if (jsgraph()->machine()->Is64()) {
401 if (GetBinaryOperationHint(slot) == BinaryOperationHint::kBigInt) {
402 op = jsgraph()->simplified()->SpeculativeBigIntNegate(
403 BigIntOperationHint::kBigInt);
404 node = jsgraph()->graph()->NewNode(op, operand, effect, control);
405 }
406 }
407 }
408 break;
409 }
410 case IrOpcode::kTypeOf: {
411 TypeOfFeedback::Result hint = broker()->GetFeedbackForTypeOf(feedback);
412 switch (hint) {
413 case TypeOfFeedback::kNumber:
414 check = jsgraph()->graph()->NewNode(
415 jsgraph()->simplified()->CheckNumber(FeedbackSource()), operand,
416 effect, control);
417 node = jsgraph()->ConstantNoHole(broker()->number_string(), broker());
418 break;
419 case TypeOfFeedback::kString:
420 check = jsgraph()->graph()->NewNode(
421 jsgraph()->simplified()->CheckString(FeedbackSource()), operand,
422 effect, control);
423 node = jsgraph()->ConstantNoHole(broker()->string_string(), broker());
424 break;
425 case TypeOfFeedback::kFunction: {
426 Node* condition = jsgraph()->graph()->NewNode(
427 jsgraph()->simplified()->ObjectIsDetectableCallable(), operand);
428 check = jsgraph()->graph()->NewNode(
429 jsgraph()->simplified()->CheckIf(
430 DeoptimizeReason::kNotDetectableReceiver, FeedbackSource()),
431 condition, effect, control);
432 node =
433 jsgraph()->ConstantNoHole(broker()->function_string(), broker());
434 break;
435 }
436 default:
437 node = nullptr;
438 break;
439 }
440 break;
441 }
442 default:
443 UNREACHABLE();
444 }
445
446 if (node != nullptr) {
447 return LoweringResult::SideEffectFree(node, check ? check : node, control);
448 } else {
449 return LoweringResult::NoChange();
450 }
451}

453JSTypeHintLowering::LoweringResult JSTypeHintLowering::ReduceBinaryOperation(
454 const Operator* op, Node* left, Node* right, Node* effect, Node* control,
455 FeedbackSlot slot) const {
456 switch (op->opcode()) {
457 case IrOpcode::kJSStrictEqual: {
458 if (Node* node = BuildDeoptIfFeedbackIsInsufficient(
459 slot, effect, control,
460 DeoptimizeReason::kInsufficientTypeFeedbackForCompareOperation)) {
461 return LoweringResult::Exit(node);
462 }
463 // TODO(turbofan): Should we generally support early lowering of
464 // JSStrictEqual operators here?
465 break;
466 }
467 case IrOpcode::kJSEqual:
468 case IrOpcode::kJSLessThan:
469 case IrOpcode::kJSGreaterThan:
470 case IrOpcode::kJSLessThanOrEqual:
471 case IrOpcode::kJSGreaterThanOrEqual: {
472 if (Node* node = BuildDeoptIfFeedbackIsInsufficient(
473 slot, effect, control,
474 DeoptimizeReason::kInsufficientTypeFeedbackForCompareOperation)) {
475 return LoweringResult::Exit(node);
476 }
477 JSSpeculativeBinopBuilder b(this, op, left, right, effect, control, slot);
478 if (Node* node = b.TryBuildNumberCompare()) {
479 return LoweringResult::SideEffectFree(node, node, control);
480 }
481 if (Node* node = b.TryBuildBigIntCompare()) {
482 return LoweringResult::SideEffectFree(node, node, control);
483 }
484 break;
485 }
486 case IrOpcode::kJSInstanceOf: {
487 if (Node* node = BuildDeoptIfFeedbackIsInsufficient(
488 slot, effect, control,
489 DeoptimizeReason::kInsufficientTypeFeedbackForCompareOperation)) {
490 return LoweringResult::Exit(node);
491 }
492 // TODO(turbofan): Should we generally support early lowering of
493 // JSInstanceOf operators here?
494 break;
495 }
496 case IrOpcode::kJSBitwiseOr:
497 case IrOpcode::kJSBitwiseXor:
498 case IrOpcode::kJSBitwiseAnd:
499 case IrOpcode::kJSShiftLeft:
500 case IrOpcode::kJSShiftRight:
501 case IrOpcode::kJSShiftRightLogical:
502 case IrOpcode::kJSAdd:
503 case IrOpcode::kJSSubtract:
504 case IrOpcode::kJSMultiply:
505 case IrOpcode::kJSDivide:
506 case IrOpcode::kJSModulus:
507 case IrOpcode::kJSExponentiate: {
508 if (Node* node = BuildDeoptIfFeedbackIsInsufficient(
509 slot, effect, control,
510 DeoptimizeReason::kInsufficientTypeFeedbackForBinaryOperation)) {
511 return LoweringResult::Exit(node);
512 }
513 JSSpeculativeBinopBuilder b(this, op, left, right, effect, control, slot);
514 if (Node* node = b.TryBuildNumberBinop()) {
515 return LoweringResult::SideEffectFree(node, node, control);
516 }
517 if (op->opcode() != IrOpcode::kJSShiftRightLogical &&
518 op->opcode() != IrOpcode::kJSExponentiate) {
519 if (Node* node = b.TryBuildBigIntBinop()) {
520 return LoweringResult::SideEffectFree(node, node, control);
521 }
522 }
523 break;
524 }
525 default:
526 UNREACHABLE();
527 }
528 return LoweringResult::NoChange();
529}

666Node* JSTypeHintLowering::BuildDeoptIfFeedbackIsInsufficient(
667 FeedbackSlot slot, Node* effect, Node* control,
668 DeoptimizeReason reason) const {
669 if (!(flags() & kBailoutOnUninitialized)) return nullptr;
670
671 FeedbackSource source(feedback_vector(), slot);
672 if (!broker()->FeedbackIsInsufficient(source)) return nullptr;
673
674 Node* deoptimize = jsgraph()->graph()->NewNode(
675 jsgraph()->common()->Deoptimize(reason, FeedbackSource()),
676 jsgraph()->Dead(), effect, control);
677 Node* frame_state =
