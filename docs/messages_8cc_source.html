98void MessageHandler::ReportMessage(Isolate* isolate, const MessageLocation* loc,
99 DirectHandle<JSMessageObject> message) {
100 v8::Local<v8::Message> api_message_obj = v8::Utils::MessageToLocal(message);
101
102 if (api_message_obj->ErrorLevel() != v8::Isolate::kMessageError) {
103 ReportMessageNoExceptions(isolate, loc, message, v8::Local<v8::Value>());
104 return;
105 }
106
107 // We are calling into embedder's code which can throw exceptions.
108 // Thus we need to save current exception state, reset it to the clean one
109 // and ignore scheduled exceptions callbacks can throw.
110
111 // We pass the exception object into the message handler callback though.
112 DirectHandle<Object> exception = isolate->factory()->undefined_value();
113 if (isolate->has_exception()) {
114 exception = direct_handle(isolate->exception(), isolate);
115 }
116
117 Isolate::ExceptionScope exception_scope(isolate);
118 isolate->clear_pending_message();
119
120 // Turn the exception on the message into a string if it is an object.
121 if (IsJSObject(message->argument())) {
122 HandleScope scope(isolate);
123 DirectHandle<Object> argument(message->argument(), isolate);
124
125 MaybeDirectHandle<Object> maybe_stringified;
126 DirectHandle<Object> stringified;
127 // Make sure we don't leak uncaught internally generated Error objects.
128 if (IsJSError(*argument)) {
129 maybe_stringified = Object::NoSideEffectsToString(isolate, argument);
130 } else {
131 v8::TryCatch catcher(reinterpret_cast<v8::Isolate*>(isolate));
132 catcher.SetVerbose(false);
133 catcher.SetCaptureMessage(false);
134
135 maybe_stringified = Object::ToString(isolate, argument);
136 }
137
138 if (!maybe_stringified.ToHandle(&stringified)) {
139 isolate->clear_pending_message();
140 stringified = isolate->factory()->exception_string();
141 }
142 message->set_argument(*stringified);
143 }
144
145 v8::Local<v8::Value> api_exception_obj = v8::Utils::ToLocal(exception);
146 ReportMessageNoExceptions(isolate, loc, message, api_exception_obj);
147}

149void MessageHandler::ReportMessageNoExceptions(
150 Isolate* isolate, const MessageLocation* loc, DirectHandle<Object> message,
151 v8::Local<v8::Value> api_exception_obj) {
152 v8::Local<v8::Message> api_message_obj = v8::Utils::MessageToLocal(message);
153 int error_level = api_message_obj->ErrorLevel();
154
155 DirectHandle<ArrayList> global_listeners =
156 isolate->factory()->message_listeners();
157 int global_length = global_listeners->length();
158 if (global_length == 0) {
159 DefaultMessageReport(isolate, loc, message);
160 } else {
161 for (int i = 0; i < global_length; i++) {
162 HandleScope scope(isolate);
163 if (IsUndefined(global_listeners->get(i), isolate)) continue;
164 Tagged<FixedArray> listener = Cast<FixedArray>(global_listeners->get(i));
165 Tagged<Foreign> callback_obj = Cast<Foreign>(listener->get(0));
166 int32_t message_levels =
167 static_cast<int32_t>(Smi::ToInt(listener->get(2)));
168 if (!(message_levels & error_level)) {
169 continue;
170 }
171 v8::MessageCallback callback = FUNCTION_CAST<v8::MessageCallback>(
172 callback_obj->foreign_address<kMessageListenerTag>());
173 DirectHandle<Object> callback_data(listener->get(1), isolate);
174 {
175 RCS_SCOPE(isolate, RuntimeCallCounterId::kMessageListenerCallback);
176 // Do not allow exceptions to propagate.
177 v8::TryCatch try_catch(reinterpret_cast<v8::Isolate*>(isolate));
178 callback(api_message_obj, IsUndefined(*callback_data, isolate)
179 ? api_exception_obj
180 : v8::Utils::ToLocal(callback_data));
181 }
182 }
183 }

202// Convert the raw frames as written by Isolate::CaptureSimpleStackTrace into
203// a JSArray of JSCallSite objects.
204MaybeHandle<JSArray> GetStackFrames(Isolate* isolate,
205 DirectHandle<FixedArray> frames) {
206 int frame_count = frames->length();
207 DirectHandle<JSFunction> constructor = isolate->callsite_function();
208 DirectHandle<FixedArray> sites =
209 isolate->factory()->NewFixedArray(frame_count);
210 for (int i = 0; i < frame_count; ++i) {
211 Handle<CallSiteInfo> frame(Cast<CallSiteInfo>(frames->get(i)), isolate);
212 DirectHandle<JSObject> site;
213 ASSIGN_RETURN_ON_EXCEPTION(isolate, site,
214 JSObject::New(constructor, constructor,
215 Handle<AllocationSite>::null()));
216 RETURN_ON_EXCEPTION(
217 isolate, JSObject::SetOwnPropertyIgnoreAttributes(
218 site, isolate->factory()->call_site_info_symbol(), frame,
219 DONT_ENUM));
220 sites->set(i, *site);
221 }
222
223 return isolate->factory()->NewJSArrayWithElements(sites);
224}

226MaybeDirectHandle<Object> AppendErrorString(Isolate* isolate,
227 DirectHandle<Object> error,
228 IncrementalStringBuilder* builder) {
229 v8::TryCatch try_catch(reinterpret_cast<v8::Isolate*>(isolate));
230 try_catch.SetVerbose(false);
231 try_catch.SetCaptureMessage(false);
232 MaybeDirectHandle<String> err_str = ErrorUtils::ToString(
233 isolate, Cast<Object>(error),
234 ErrorUtils::ToStringMessageSource::kPreferOriginalMessage);
235 if (err_str.is_null()) {
236 // Error.toString threw. Try to return a string representation of the thrown
237 // exception instead.
238
239 DCHECK(isolate->has_exception());
240 if (isolate->is_execution_terminating()) {
241 return {};
242 }
243 DirectHandle<Object> exception(isolate->exception(), isolate);
244 try_catch.Reset();
245
246 err_str = ErrorUtils::ToString(
247 isolate, exception,
248 ErrorUtils::ToStringMessageSource::kPreferOriginalMessage);
249 if (err_str.is_null()) {
250 // Formatting the thrown exception threw again, give up.
251 DCHECK(isolate->has_exception());
252 if (isolate->is_execution_terminating()) return {};
253 builder->AppendCStringLiteral("<error>");
254 } else {
255 // Formatted thrown exception successfully, append it.
256 builder->AppendCStringLiteral("<error: ");
257 builder->AppendString(err_str.ToHandleChecked());
258 builder->AppendCharacter('>');
259 }
260 } else {
261 builder->AppendString(err_str.ToHandleChecked());
262 }
263
264 return error;
265}

286MaybeHandle<Object> ErrorUtils::FormatStackTrace(
287 Isolate* isolate, DirectHandle<JSObject> error,
288 DirectHandle<Object> raw_stack) {
289 if (v8_flags.correctness_fuzzer_suppressions) {
290 return isolate->factory()->empty_string();
291 }
292 DCHECK(IsFixedArray(*raw_stack));
293 auto elems = Cast<FixedArray>(raw_stack);
294
295 const bool in_recursion = isolate->formatting_stack_trace();
296 const bool has_overflowed = i::StackLimitCheck{isolate}.HasOverflowed();
297 DirectHandle<NativeContext> error_context;
298 if (!in_recursion && !has_overflowed &&
299 error->GetCreationContext(isolate).ToHandle(&error_context)) {
300 if (isolate->HasPrepareStackTraceCallback()) {
301 PrepareStackTraceScope scope(isolate);
302
303 DirectHandle<JSArray> sites;
304 ASSIGN_RETURN_ON_EXCEPTION(isolate, sites,
305 GetStackFrames(isolate, elems));
306
307 Handle<Object> result;
308 ASSIGN_RETURN_ON_EXCEPTION(
309 isolate, result,
310 isolate->RunPrepareStackTraceCallback(error_context, error, sites));
311 return result;
312 } else {
313 DirectHandle<JSFunction> global_error(error_context->error_function(),
314 isolate);
315
316 // If there's a user-specified "prepareStackTrace" function, call it on
317 // the frames and use its result.
318
319 DirectHandle<Object> prepare_stack_trace;
320 ASSIGN_RETURN_ON_EXCEPTION(
321 isolate, prepare_stack_trace,
322 JSFunction::GetProperty(isolate, global_error, "prepareStackTrace"));
323
324 if (IsJSFunction(*prepare_stack_trace)) {
325 PrepareStackTraceScope scope(isolate);
326
327 isolate->CountUsage(v8::Isolate::kErrorPrepareStackTrace);
328
329 DirectHandle<JSArray> sites;
330 ASSIGN_RETURN_ON_EXCEPTION(isolate, sites,
331 GetStackFrames(isolate, elems));
332
333 constexpr int argc = 2;
334 std::array<DirectHandle<Object>, argc> args;
335 if (V8_UNLIKELY(IsJSGlobalObject(*error))) {
336 // Pass global proxy instead of global object.
337 args[0] = direct_handle(Cast<JSGlobalObject>(*error)->global_proxy(),
338 isolate);
339 } else {
340 args[0] = error;
341 }
342 args[1] = sites;
343
344 Handle<Object> result;
345
346 ASSIGN_RETURN_ON_EXCEPTION(
347 isolate, result,
348 Execution::Call(isolate, prepare_stack_trace, global_error,
349 base::VectorOf(args)));
350 return result;
351 }
352 }
353 }
354
355 // Otherwise, run our internal formatting logic.
356 IncrementalStringBuilder builder(isolate);
357
358 RETURN_ON_EXCEPTION(isolate, AppendErrorString(isolate, error, &builder));
359
360 for (int i = 0; i < elems->length(); ++i) {
361 builder.AppendCStringLiteral("\n at ");
362
363 DirectHandle<CallSiteInfo> frame(Cast<CallSiteInfo>(elems->get(i)),
364 isolate);
365
366 v8::TryCatch try_catch(reinterpret_cast<v8::Isolate*>(isolate));
367 SerializeCallSiteInfo(isolate, frame, &builder);
368
369 if (isolate->has_exception()) {
370 // CallSite.toString threw. Parts of the current frame might have been
371 // stringified already regardless. Still, try to append a string
372 // representation of the thrown exception.
373
374 DirectHandle<Object> exception(isolate->exception(), isolate);
375 try_catch.Reset();
376
377 MaybeDirectHandle<String> exception_string =
378 ErrorUtils::ToString(isolate, exception);
379 if (exception_string.is_null()) {
380 // Formatting the thrown exception threw again, give up.
381
382 builder.AppendCStringLiteral("<error>");
383 } else {
384 // Formatted thrown exception successfully, append it.
385 builder.AppendCStringLiteral("<error: ");
386 builder.AppendString(exception_string.ToHandleChecked());
387 builder.AppendCStringLiteral("<error>");
388 }
389 }
390 }
391
392 return indirect_handle(builder.Finish(), isolate);
393}

395Handle<String> MessageFormatter::Format(
396 Isolate* isolate, MessageTemplate index,
397 base::Vector<const DirectHandle<Object>> args) {
398 constexpr size_t kMaxArgs = 3;
399 DirectHandle<String> arg_strings[kMaxArgs];
400 DCHECK_LE(args.size(), kMaxArgs);
401 for (size_t i = 0; i < args.size(); ++i) {
402 DCHECK(!args[i].is_null());
403 arg_strings[i] = Object::NoSideEffectsToString(isolate, args[i]);
404 }
405 v8::TryCatch try_catch(reinterpret_cast<v8::Isolate*>(isolate));
406 try_catch.SetVerbose(false);
407 try_catch.SetCaptureMessage(false);
408 MaybeHandle<String> maybe_result_string = MessageFormatter::TryFormat(
409 isolate, index, base::VectorOf(arg_strings, args.size()));
410 Handle<String> result_string;
411 if (!maybe_result_string.ToHandle(&result_string)) {
412 DCHECK(isolate->has_exception());
413 return isolate->factory()->InternalizeString(
414 base::StaticCharVector("<error>"));
415 }
416 // A string that has been obtained from JS code in this way is
417 // likely to be a complicated ConsString of some sort. We flatten it
418 // here to improve the efficiency of converting it to a C string and
419 // other operations that are likely to take place (see GetLocalizedMessage
420 // for example).
421 return String::Flatten(isolate, result_string);
422}

437 Isolate* isolate, MessageTemplate index,
438 base::Vector<const DirectHandle<String>> args) {
439 const char* template_string = TemplateString(index);
440
441 IncrementalStringBuilder builder(isolate);
442
443 // TODO(14386): Get this list empty.

453 MessageTemplate::kInvalidDataViewLength,
454 MessageTemplate::kInvalidIndex,
455 MessageTemplate::kInvalidLhsInAssignment,
456 MessageTemplate::kInvalidLhsInFor,
457 MessageTemplate::kInvalidLhsInPostfixOp,
458 MessageTemplate::kInvalidLhsInPrefixOp,
459 MessageTemplate::kInvalidPrivateBrandReinitialization,
460 MessageTemplate::kInvalidPrivateFieldReinitialization,
461 MessageTemplate::kInvalidPrivateMemberWrite,
462 MessageTemplate::kInvalidRegExpExecResult,
463 MessageTemplate::kInvalidTimeValue,
464 MessageTemplate::kInvalidWeakMapKey,
465 MessageTemplate::kInvalidWeakSetValue,
466 MessageTemplate::kIteratorReduceNoInitial,
467 MessageTemplate::kJsonParseShortString,
468 MessageTemplate::kJsonParseUnexpectedEOS,
469 MessageTemplate::kJsonParseUnexpectedTokenEndStringWithContext,
470 MessageTemplate::kJsonParseUnexpectedTokenShortString,
471 MessageTemplate::kJsonParseUnexpectedTokenStartStringWithContext,
472 MessageTemplate::kJsonParseUnexpectedTokenSurroundStringWithContext,
473 MessageTemplate::kMustBePositive,
474 MessageTemplate::kNotIterable,
475 MessageTemplate::kNotTypedArray,
476 MessageTemplate::kProxyNonObject,
477 MessageTemplate::kProxyPrivate,
478 MessageTemplate::kProxyRevoked,
479 MessageTemplate::kProxyTrapReturnedFalsishFor,
480 MessageTemplate::kReduceNoInitial,
481 MessageTemplate::kSpreadIteratorSymbolNonCallable,
482 MessageTemplate::kSymbolIteratorInvalid,
483 MessageTemplate::kTopLevelAwaitStalled,
484 MessageTemplate::kUndefinedOrNullToObject,
485 MessageTemplate::kUnexpectedStrictReserved,
486 MessageTemplate::kUnexpectedTokenIdentifier,
487 MessageTemplate::kWeakRefsCleanupMustBeCallable};
488
489 base::Vector<const DirectHandle<String>> remaining_args = args;
490 for (const char* c = template_string; *c != '\0'; c++) {
491 if (*c == '%') {
492 // %% results in verbatim %.
493 if (*(c + 1) == '%') {
494 c++;
495 builder.AppendCharacter('%');
496 } else {
497 // TODO(14386): Remove this fallback.
498 if (remaining_args.empty()) {
499 if (std::count(std::begin(kTemplatesWithMismatchedArguments),
500 std::end(kTemplatesWithMismatchedArguments), index)) {
501 builder.AppendCString("undefined");
502 } else {
503 FATAL("Missing argument to template (got %zu): %s", args.size(),
504 template_string);
505 }
506 } else {
507 DirectHandle<String> arg = remaining_args[0];
508 remaining_args += 1;
509 builder.AppendString(arg);
510 }
511 }
512 } else {
513 builder.AppendCharacter(*c);
514 }
515 }
516 if (!remaining_args.empty() &&
517 std::count(std::begin(kTemplatesWithMismatchedArguments),
518 std::end(kTemplatesWithMismatchedArguments), index) == 0) {
519 FATAL("Too many arguments to template (expected %zu, got %zu): %s",
520 args.size() - remaining_args.size(), args.size(), template_string);
521 }
522
523 return indirect_handle(builder.Finish(), isolate);
524}

547MaybeHandle<JSObject> ErrorUtils::Construct(
548 Isolate* isolate, DirectHandle<JSFunction> target,
549 DirectHandle<Object> new_target, DirectHandle<Object> message,
550 DirectHandle<Object> options, FrameSkipMode mode,
551 DirectHandle<Object> caller, StackTraceCollection stack_trace_collection) {
552 if (v8_flags.correctness_fuzzer_suppressions) {
553 // Abort range errors in correctness fuzzing, as their causes differ
554 // accross correctness-fuzzing scenarios.
555 if (target.is_identical_to(isolate->range_error_function())) {
556 FATAL("Aborting on range error");
557 }
558 // Ignore error messages in correctness fuzzing, because the spec leaves
559 // room for undefined behavior.
560 message = isolate->factory()->InternalizeUtf8String(
561 "Message suppressed for fuzzers (--correctness-fuzzer-suppressions)");
562 }
563
564 // 1. If NewTarget is undefined, let newTarget be the active function object,
565 // else let newTarget be NewTarget.
566 DirectHandle<JSReceiver> new_target_recv = IsJSReceiver(*new_target)
567 ? Cast<JSReceiver>(new_target)
568 : Cast<JSReceiver>(target);
569
570 // 2. Let O be ? OrdinaryCreateFromConstructor(newTarget, "%ErrorPrototype%",
571 // Â« [[ErrorData]] Â»).
572 Handle<JSObject> err;
573 ASSIGN_RETURN_ON_EXCEPTION(isolate, err,
574 JSObject::New(target, new_target_recv, {}));
575
576 // 3. If message is not undefined, then
577 // a. Let msg be ? ToString(message).
578 // b. Let msgDesc be the PropertyDescriptor{[[Value]]: msg, [[Writable]]:
579 // true, [[Enumerable]]: false, [[Configurable]]: true}.
580 // c. Perform ! DefinePropertyOrThrow(O, "message", msgDesc).
581 // 4. Return O.
582 if (!IsUndefined(*message, isolate)) {
583 Handle<String> msg_string;
584 ASSIGN_RETURN_ON_EXCEPTION(
585 isolate, msg_string,
586 indirect_handle(Object::ToString(isolate, message), isolate));
587 RETURN_ON_EXCEPTION(isolate, JSObject::SetOwnPropertyIgnoreAttributes(
588 err, isolate->factory()->message_string(),
589 msg_string, DONT_ENUM));
590
591 if (v8_flags.use_original_message_for_stack_trace) {
592 RETURN_ON_EXCEPTION(isolate,
593 JSObject::SetOwnPropertyIgnoreAttributes(
594 err, isolate->factory()->error_message_symbol(),
595 msg_string, DONT_ENUM));
596 }
597 }
598
599 if (!IsUndefined(*options, isolate)) {
600 // If Type(options) is Object and ? HasProperty(options, "cause") then
601 // a. Let cause be ? Get(options, "cause").
602 // b. Perform ! CreateNonEnumerableDataPropertyOrThrow(O, "cause", cause).
603 DirectHandle<Name> cause_string = isolate->factory()->cause_string();
604 if (IsJSReceiver(*options)) {
605 DirectHandle<JSReceiver> js_options = Cast<JSReceiver>(options);
606 Maybe<bool> has_cause =
607 JSObject::HasProperty(isolate, js_options, cause_string);
608 if (has_cause.IsNothing()) {
609 DCHECK((isolate)->has_exception());
610 return MaybeHandle<JSObject>();
611 }
612 if (has_cause.ToChecked()) {
613 Handle<Object> cause;
614 ASSIGN_RETURN_ON_EXCEPTION(
615 isolate, cause,
616 JSObject::GetProperty(isolate, js_options, cause_string));
617 RETURN_ON_EXCEPTION(isolate, JSObject::SetOwnPropertyIgnoreAttributes(
618 err, cause_string, cause, DONT_ENUM));
619 }
620 }
621 }
622
623 switch (stack_trace_collection) {
624 case StackTraceCollection::kEnabled:
625 RETURN_ON_EXCEPTION(isolate,
626 isolate->CaptureAndSetErrorStack(
627 err, mode, indirect_handle(caller, isolate)));
628 break;
629 case StackTraceCollection::kDisabled:
630 break;
631 }
632 return err;
633}

671 // 3. Let name be ? Get(O, "name").
672 // 4. If name is undefined, let name be "Error"; otherwise let name be
673 // ? ToString(name).
674 DirectHandle<String> name_key = isolate->factory()->name_string();
675 Handle<String> name_default = isolate->factory()->Error_string();
676 Handle<String> name;
677 ASSIGN_RETURN_ON_EXCEPTION(
678 isolate, name,
679 GetStringPropertyOrDefault(isolate, recv, name_key, name_default));
680
681 // 5. Let msg be ? Get(O, "message").
682 // 6. If msg is undefined, let msg be the empty String; otherwise let msg be
683 // ? ToString(msg).
684 Handle<String> msg;
685 Handle<String> msg_default = isolate->factory()->empty_string();
686 if (message_source == ToStringMessageSource::kPreferOriginalMessage) {
687 // V8-specific extension for Error.stack: Use the original message with
688 // which the Error constructor was called. This keeps Error.stack consistent
689 // w.r.t. "message" property changes regardless of the time when Error.stack
690 // is accessed the first time.
691 //
692 // If |recv| was not constructed with %Error%, use the "message" property.
693 LookupIterator it(isolate, LookupIterator::PROTOTYPE_CHAIN_SKIP_INTERCEPTOR,
694 recv, isolate->factory()->error_message_symbol());
695 Handle<Object> result = JSReceiver::GetDataProperty(&it);
696 if (it.IsFound() && IsUndefined(*result, isolate)) {
697 msg = msg_default;
698 } else if (it.IsFound()) {
699 ASSIGN_RETURN_ON_EXCEPTION(isolate, msg,
700 Object::ToString(isolate, result));
701 }
702 }
703
704 if (msg.is_null()) {
705 DirectHandle<String> msg_key = isolate->factory()->message_string();
706 ASSIGN_RETURN_ON_EXCEPTION(
707 isolate, msg,
708 GetStringPropertyOrDefault(isolate, recv, msg_key, msg_default));
709 }
710
711 // 7. If name is the empty String, return msg.
712 // 8. If msg is the empty String, return name.
713 if (name->length() == 0) return msg;
714 if (msg->length() == 0) return name;
715
716 // 9. Return the result of concatenating name, the code unit 0x003A (COLON),
717 // the code unit 0x0020 (SPACE), and msg.
718 IncrementalStringBuilder builder(isolate);
719 builder.AppendString(name);
720 builder.AppendCStringLiteral(": ");
721 builder.AppendString(msg);
722
723 Handle<String> result;
724 ASSIGN_RETURN_ON_EXCEPTION(isolate, result,
725 indirect_handle(builder.Finish(), isolate));
726 return result;

836 SharedFunctionInfo::EnsureSourcePositionsAvailable(isolate, shared);

840 !(IsUndefined(Cast<Script>(script)->source(), isolate))) {
841 Handle<Script> casted_script = Cast<Script>(script);
842 *target = MessageLocation(casted_script, pos, pos + 1, shared);

849Handle<String> BuildDefaultCallSite(Isolate* isolate,
850 DirectHandle<Object> object) {
851 IncrementalStringBuilder builder(isolate);
852
853 builder.AppendString(Object::TypeOf(isolate, object));

855 builder.AppendCStringLiteral(" \"");

857 // This threshold must be sufficiently far below String::kMaxLength that
858 // the {builder}'s result can never exceed that limit.
859 constexpr int kMaxPrintedStringLength = 100;
860 if (string->length() <= kMaxPrintedStringLength) {
861 builder.AppendString(string);
862 } else {
863 string = isolate->factory()->NewProperSubString(string, 0,
864 kMaxPrintedStringLength);
865 builder.AppendString(string);
866 builder.AppendCStringLiteral("<...>");
867 }
868 builder.AppendCStringLiteral("\"");
869 } else if (IsNull(*object, isolate)) {
870 builder.AppendCStringLiteral(" null");
871 } else if (IsTrue(*object, isolate)) {
872 builder.AppendCStringLiteral(" true");
873 } else if (IsFalse(*object, isolate)) {
874 builder.AppendCStringLiteral(" false");
875 } else if (IsNumber(*object)) {
876 builder.AppendCharacter(' ');
877 builder.AppendString(isolate->factory()->NumberToString(object));
878 }

1004Tagged<Object> ErrorUtils::ThrowLoadFromNullOrUndefined(
1005 Isolate* isolate, DirectHandle<Object> object,
1006 MaybeDirectHandle<Object> key) {
1007 DCHECK(IsNullOrUndefined(*object));
1008
1009 MaybeDirectHandle<String> maybe_property_name;
1010
1011 // Try to extract the property name from the given key, if any.
1012 DirectHandle<Object> key_handle;
1013 if (key.ToHandle(&key_handle)) {
1014 if (IsString(*key_handle)) {
1015 maybe_property_name = Cast<String>(key_handle);
1016 } else {
1017 maybe_property_name =
1018 Object::NoSideEffectsToMaybeString(isolate, key_handle);
1019 }
1020 }
1021
1022 DirectHandle<String> callsite;
1023
1024 // Inline the RenderCallSite logic here so that we can additionally access the
1025 // destructuring property.
1026 bool location_computed = false;
1027 bool is_destructuring = false;
1028 MessageLocation location;
1029 if (ComputeLocation(isolate, &location)) {
1030 location_computed = true;
1031
1032 UnoptimizedCompileFlags flags = UnoptimizedCompileFlags::ForFunctionCompile(
1033 isolate, *location.shared());
1034 flags.set_is_reparse(true);
1035 UnoptimizedCompileState compile_state;
1036 ReusableUnoptimizedCompileState reusable_state(isolate);
1037 ParseInfo info(isolate, flags, &compile_state, &reusable_state);
1038 if (parsing::ParseAny(&info, location.shared(), isolate,
1039 parsing::ReportStatisticsMode::kNo)) {
1040 info.ast_value_factory()->Internalize(isolate);
1041 CallPrinter printer(isolate, location.shared()->IsUserJavaScript());
1042 DirectHandle<String> str =
1043 printer.Print(info.literal(), location.start_pos());
1044
1045 int pos = -1;
1046 is_destructuring = printer.destructuring_assignment() != nullptr;
1047
1048 if (is_destructuring) {
1049 // If we don't have one yet, try to extract the property name from the
1050 // destructuring property in the AST.
1051 ObjectLiteralProperty* destructuring_prop =
1052 printer.destructuring_prop();
1053 if (maybe_property_name.is_null() && destructuring_prop != nullptr &&
1054 destructuring_prop->key()->IsPropertyName()) {
1055 maybe_property_name = destructuring_prop->key()
1056 ->AsLiteral()
1057 ->AsRawPropertyName()
1058 ->string();
1059 // Change the message location to point at the property name.
1060 pos = destructuring_prop->key()->position();
1061 }
1062 if (maybe_property_name.is_null()) {
1063 // Change the message location to point at the destructured value.
1064 pos = printer.destructuring_assignment()->value()->position();
1065 }
1066
1067 // If we updated the pos to a valid pos, rewrite the location.
1068 if (pos != -1) {
1069 location = MessageLocation(location.script(), pos, pos + 1,
1070 location.shared());
1071 }
1072 }
1073
1074 if (str->length() > 0) callsite = str;
1075 }
1076 }
1077
1078 if (callsite.is_null()) {
1079 callsite = BuildDefaultCallSite(isolate, object);
1080 }
1081
1082 DirectHandle<JSObject> error;
1083 DirectHandle<String> property_name;
1084 if (is_destructuring) {
1085 if (maybe_property_name.ToHandle(&property_name)) {
1086 error = isolate->factory()->NewTypeError(
1087 MessageTemplate::kNonCoercibleWithProperty, property_name, callsite,
1088 object);
1089 } else {
1090 error = isolate->factory()->NewTypeError(MessageTemplate::kNonCoercible,
1091 callsite, object);
1092 }
1093 } else {
1094 if (!key.ToHandle(&key_handle) ||
1095 !maybe_property_name.ToHandle(&property_name)) {
1096 error = isolate->factory()->NewTypeError(
1097 MessageTemplate::kNonObjectPropertyLoad, object);
1098 } else if (*key_handle == ReadOnlyRoots(isolate).iterator_symbol()) {
1099 error = NewIteratorError(isolate, object);
1100 } else {
1101 error = isolate->factory()->NewTypeError(
1102 MessageTemplate::kNonObjectPropertyLoadWithProperty, object,
1103 property_name);
1104 }
1105 }
1106
1107 if (location_computed) {
1108 isolate->ThrowAt(error, &location);
1109 } else {
1110 isolate->Throw(*error);
1111 }
1112 return ReadOnlyRoots(isolate).exception();
1113}
