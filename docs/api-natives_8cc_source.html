71MaybeHandle<Object> DefineAccessorProperty(Isolate* isolate,
72 Handle<JSObject> object,
73 DirectHandle<Name> name,
74 DirectHandle<Object> getter,
75 DirectHandle<Object> setter,
76 PropertyAttributes attributes) {
77 DCHECK(!IsFunctionTemplateInfo(*getter) ||
78 Cast<FunctionTemplateInfo>(*getter)->should_cache());
79 DCHECK(!IsFunctionTemplateInfo(*setter) ||
80 Cast<FunctionTemplateInfo>(*setter)->should_cache());
81 if (IsFunctionTemplateInfo(*getter) &&
82 Cast<FunctionTemplateInfo>(*getter)->BreakAtEntry(isolate)) {
83 ASSIGN_RETURN_ON_EXCEPTION(
84 isolate, getter,
85 InstantiateFunction(isolate, Cast<FunctionTemplateInfo>(getter)));
86 DirectHandle<Code> trampoline = BUILTIN_CODE(isolate, DebugBreakTrampoline);
87 Cast<JSFunction>(getter)->UpdateCode(*trampoline);
88 }
89 if (IsFunctionTemplateInfo(*setter) &&
90 Cast<FunctionTemplateInfo>(*setter)->BreakAtEntry(isolate)) {
91 ASSIGN_RETURN_ON_EXCEPTION(
92 isolate, setter,
93 InstantiateFunction(isolate, Cast<FunctionTemplateInfo>(setter)));
94 DirectHandle<Code> trampoline = BUILTIN_CODE(isolate, DebugBreakTrampoline);
95 Cast<JSFunction>(setter)->UpdateCode(*trampoline);
96 }
97 RETURN_ON_EXCEPTION(isolate, JSObject::DefineOwnAccessorIgnoreAttributes(
98 object, name, getter, setter, attributes));
99 return object;
100}

103 DirectHandle<JSObject> object,
104 DirectHandle<Name> name,
105 Handle<Object> prop_data,
106 PropertyAttributes attributes) {
107 Handle<Object> value;

109 Instantiate(isolate, prop_data, name));

112 LookupIterator it(isolate, object, key, LookupIterator::OWN_SKIP_INTERCEPTOR);

114#ifdef DEBUG

116 DCHECK(maybe.IsJust());

122#endif
123
124 MAYBE_RETURN_NULL(Object::AddDataProperty(&it, value, attributes,
125 Just(ShouldThrow::kThrowOnError),
126 StoreOrigin::kNamed));
127 return value;

183template <typename TemplateInfoT>
184MaybeHandle<JSObject> ConfigureInstance(Isolate* isolate, Handle<JSObject> obj,
185 DirectHandle<TemplateInfoT> data) {
186 RCS_SCOPE(isolate, RuntimeCallCounterId::kConfigureInstance);
187 HandleScope scope(isolate);
188 // Disable access checks while instantiating the object.
189 AccessCheckDisableScope access_check_scope(isolate, obj);
190
191 // Walk the inheritance chain and copy all accessors to current object.
192 int max_number_of_properties = 0;
193 Tagged<TemplateInfoT> info = *data;
194 while (!info.is_null()) {
195 Tagged<Object> props = info->property_accessors();
196 if (!IsUndefined(props, isolate)) {
197 max_number_of_properties += Cast<ArrayList>(props)->length();
198 }
199 info = info->GetParent(isolate);
200 }
201
202 if (max_number_of_properties > 0) {
203 int valid_descriptors = 0;
204 // Use a temporary FixedArray to accumulate unique accessors.
205 DirectHandle<FixedArray> array =
206 isolate->factory()->NewFixedArray(max_number_of_properties);
207
208 // TODO(leszeks): Avoid creating unnecessary handles for cases where we
209 // don't need to append anything.
210 for (DirectHandle<TemplateInfoT> temp(*data, isolate); !(*temp).is_null();
211 temp = direct_handle(temp->GetParent(isolate), isolate)) {
212 // Accumulate accessors.
213 Tagged<Object> maybe_properties = temp->property_accessors();
214 if (!IsUndefined(maybe_properties, isolate)) {
215 valid_descriptors = AccessorInfo::AppendUnique(
216 isolate, direct_handle(maybe_properties, isolate), array,
217 valid_descriptors);
218 }
219 }
220
221 // Install accumulated accessors.
222 for (int i = 0; i < valid_descriptors; i++) {
223 DirectHandle<AccessorInfo> accessor(Cast<AccessorInfo>(array->get(i)),
224 isolate);
225 DirectHandle<Name> name(Cast<Name>(accessor->name()), isolate);
226 JSObject::SetAccessor(obj, name, accessor,
227 accessor->initial_property_attributes())
228 .Assert();
229 }
230 }
231
232 Tagged<Object> maybe_property_list = data->property_list();
233 if (IsUndefined(maybe_property_list, isolate)) return obj;
234 DirectHandle<ArrayList> properties(Cast<ArrayList>(maybe_property_list),
235 isolate);
236 if (properties->length() == 0) return obj;
237
238 int i = 0;
239 for (int c = 0; c < data->number_of_properties(); c++) {
240 auto name = direct_handle(Cast<Name>(properties->get(i++)), isolate);
241 Tagged<Object> bit = properties->get(i++);
242 if (IsSmi(bit)) {
243 PropertyDetails details(Cast<Smi>(bit));
244 PropertyAttributes attributes = details.attributes();
245 PropertyKind kind = details.kind();
246
247 if (kind == PropertyKind::kData) {
248 auto prop_data = handle(properties->get(i++), isolate);
249 RETURN_ON_EXCEPTION(isolate, DefineDataProperty(isolate, obj, name,
250 prop_data, attributes));
251 } else {
252 auto getter = direct_handle(properties->get(i++), isolate);
253 auto setter = direct_handle(properties->get(i++), isolate);
254 RETURN_ON_EXCEPTION(
255 isolate, DefineAccessorProperty(isolate, obj, name, getter, setter,
256 attributes));
257 }
258 } else {
259 // Intrinsic data property --- Get appropriate value from the current
260 // context.
261 PropertyDetails details(Cast<Smi>(properties->get(i++)));
262 PropertyAttributes attributes = details.attributes();
263 DCHECK_EQ(PropertyKind::kData, details.kind());
264
265 v8::Intrinsic intrinsic =
266 static_cast<v8::Intrinsic>(Smi::ToInt(properties->get(i++)));
267 auto prop_data = handle(GetIntrinsic(isolate, intrinsic), isolate);
268
269 RETURN_ON_EXCEPTION(isolate, DefineDataProperty(isolate, obj, name,
270 prop_data, attributes));
271 }
272 }
273 return obj;
274}

276bool IsSimpleInstantiation(Isolate* isolate, Tagged<ObjectTemplateInfo> info,
277 Tagged<JSReceiver> new_target) {
278 DisallowGarbageCollection no_gc;
279
280 if (!IsJSFunction(new_target)) return false;
281 Tagged<JSFunction> fun = Cast<JSFunction>(new_target);
282 if (!fun->shared()->IsApiFunction()) return false;
283 if (fun->shared()->api_func_data() != info->constructor()) return false;
284 if (info->immutable_proto()) return false;
285 return fun->native_context() == isolate->raw_native_context();
286}

288MaybeHandle<JSObject> InstantiateObject(Isolate* isolate,
289 DirectHandle<ObjectTemplateInfo> info,
290 DirectHandle<JSReceiver> new_target,
291 bool is_prototype) {
292 RCS_SCOPE(isolate, RuntimeCallCounterId::kInstantiateObject);
293 DirectHandle<JSFunction> constructor;
294 bool should_cache = info->should_cache();
295 if (!new_target.is_null()) {
296 if (IsSimpleInstantiation(isolate, *info, *new_target)) {
297 constructor = Cast<JSFunction>(new_target);
298 } else {
299 // Disable caching for subclass instantiation.
300 should_cache = false;
301 }
302 }
303 // Fast path.
304 Handle<JSObject> result;
305 if (should_cache && info->is_cached()) {
306 if (TemplateInfo::ProbeInstantiationsCache<JSObject>(
307 isolate, isolate->native_context(), info->serial_number(),
308 TemplateInfo::CachingMode::kLimited)
309 .ToHandle(&result)) {
310 return isolate->factory()->CopyJSObject(result);
311 }
312 }
313
314 if (constructor.is_null()) {
315 Tagged<Object> maybe_constructor_info = info->constructor();
316 if (IsUndefined(maybe_constructor_info, isolate)) {
317 constructor = isolate->object_function();
318 } else {
319 // Enter a new scope. Recursion could otherwise create a lot of handles.
320 HandleScope scope(isolate);
321 DirectHandle<FunctionTemplateInfo> cons_templ(
322 Cast<FunctionTemplateInfo>(maybe_constructor_info), isolate);
323 DirectHandle<JSFunction> tmp_constructor;
324 ASSIGN_RETURN_ON_EXCEPTION(isolate, tmp_constructor,
325 InstantiateFunction(isolate, cons_templ));
326 constructor = scope.CloseAndEscape(tmp_constructor);
327 }
328
329 if (new_target.is_null()) new_target = constructor;
330 }
331
332 const auto new_js_object_type =
333 constructor->has_initial_map() &&
334 IsJSApiWrapperObject(constructor->initial_map())
335 ? NewJSObjectType::kAPIWrapper
336 : NewJSObjectType::kNoAPIWrapper;
337 Handle<JSObject> object;
338 ASSIGN_RETURN_ON_EXCEPTION(
339 isolate, object,
340 JSObject::New(constructor, new_target, {}, new_js_object_type));
341
342 if (is_prototype) JSObject::OptimizeAsPrototype(object);
343
344 ASSIGN_RETURN_ON_EXCEPTION(isolate, result,
345 ConfigureInstance(isolate, object, info));
346 if (info->immutable_proto()) {
347 JSObject::SetImmutableProto(isolate, object);
348 }
349 if (!is_prototype) {
350 // Keep prototypes in slow-mode. Let them be lazily turned fast later on.
351 // TODO(dcarney): is this necessary?
352 JSObject::MigrateSlowToFast(result, 0, "ApiNatives::InstantiateObject");
353 // Don't cache prototypes.
354 if (should_cache) {
355 TemplateInfo::CacheTemplateInstantiation<JSObject, ObjectTemplateInfo>(
356 isolate, isolate->native_context(), info,
357 TemplateInfo::CachingMode::kLimited, result);
358 result = isolate->factory()->CopyJSObject(result);
359 }
360 }
361
362 return result;
363}

385MaybeHandle<JSFunction> InstantiateFunction(
386 Isolate* isolate, DirectHandle<NativeContext> native_context,
387 DirectHandle<FunctionTemplateInfo> data,
388 MaybeDirectHandle<Name> maybe_name) {
389 RCS_SCOPE(isolate, RuntimeCallCounterId::kInstantiateFunction);
390 bool should_cache = data->should_cache();
391 if (should_cache && data->is_cached()) {
392 Handle<JSObject> result;
393 if (TemplateInfo::ProbeInstantiationsCache<JSObject>(
394 isolate, native_context, data->serial_number(),
395 TemplateInfo::CachingMode::kUnlimited)
396 .ToHandle(&result)) {
397 return Cast<JSFunction>(result);
398 }
399 }
400 DirectHandle<Object> prototype;
401 if (!data->remove_prototype()) {
402 DirectHandle<Object> prototype_templ(data->GetPrototypeTemplate(), isolate);
403 if (IsUndefined(*prototype_templ, isolate)) {
404 DirectHandle<Object> protoype_provider_templ(
405 data->GetPrototypeProviderTemplate(), isolate);
406 if (IsUndefined(*protoype_provider_templ, isolate)) {
407 prototype = isolate->factory()->NewJSObject(
408 direct_handle(native_context->object_function(), isolate));
409 } else {
410 ASSIGN_RETURN_ON_EXCEPTION(
411 isolate, prototype,
412 GetInstancePrototype(isolate, protoype_provider_templ));
413 }
414 } else {
415 ASSIGN_RETURN_ON_EXCEPTION(
416 isolate, prototype,
417 InstantiateObject(isolate, Cast<ObjectTemplateInfo>(prototype_templ),
418 DirectHandle<JSReceiver>(), true));
419 }
420 DirectHandle<Object> parent(data->GetParentTemplate(), isolate);
421 if (!IsUndefined(*parent, isolate)) {
422 DirectHandle<Object> parent_prototype;
423 ASSIGN_RETURN_ON_EXCEPTION(isolate, parent_prototype,
424 GetInstancePrototype(isolate, parent));
425 DirectHandle<JSPrototype> checked_parent_prototype;
426 CHECK(TryCast(parent_prototype, &checked_parent_prototype));
427 JSObject::ForceSetPrototype(isolate, Cast<JSObject>(prototype),
428 checked_parent_prototype);
429 }
430 }
431 InstanceType function_type = JS_SPECIAL_API_OBJECT_TYPE;
432 if (!data->needs_access_check() &&
433 IsUndefined(data->GetNamedPropertyHandler(), isolate) &&
434 IsUndefined(data->GetIndexedPropertyHandler(), isolate)) {
435 function_type = v8_flags.experimental_embedder_instance_types
436 ? data->GetInstanceType()
437 : JS_API_OBJECT_TYPE;
438 DCHECK(InstanceTypeChecker::IsJSApiObject(function_type));
439 }
440
441 Handle<JSFunction> function = ApiNatives::CreateApiFunction(
442 isolate, native_context, data, prototype, function_type, maybe_name);
443 if (should_cache) {
444 // Cache the function.
445 TemplateInfo::CacheTemplateInstantiation<JSObject, FunctionTemplateInfo>(
446 isolate, native_context, data, TemplateInfo::CachingMode::kUnlimited,
447 function);
448 }
449 MaybeDirectHandle<JSObject> result =
450 ConfigureInstance(isolate, function, data);
451 if (result.is_null()) {
452 // Uncache on error.
453 TemplateInfo::UncacheTemplateInstantiation<FunctionTemplateInfo>(
454 isolate, native_context, data, TemplateInfo::CachingMode::kUnlimited);
455 return {};
456 }
457 data->set_published(true);
458 return function;
459}

598Handle<JSFunction> ApiNatives::CreateApiFunction(
599 Isolate* isolate, DirectHandle<NativeContext> native_context,
600 DirectHandle<FunctionTemplateInfo> obj, DirectHandle<Object> prototype,
601 InstanceType type, MaybeDirectHandle<Name> maybe_name) {
602 RCS_SCOPE(isolate, RuntimeCallCounterId::kCreateApiFunction);
603 DirectHandle<SharedFunctionInfo> shared =
604 FunctionTemplateInfo::GetOrCreateSharedFunctionInfo(isolate, obj,
605 maybe_name);
606 // To simplify things, API functions always have shared name.
607 DCHECK(shared->HasSharedName());
608
609 Handle<JSFunction> result =
610 Factory::JSFunctionBuilder{isolate, shared, native_context}.Build();
611
612 if (obj->remove_prototype()) {
613 DCHECK(prototype.is_null());
614 DCHECK(result->shared()->IsApiFunction());
615 DCHECK(!IsConstructor(*result));
616 DCHECK(!result->has_prototype_slot());
617 return result;
618 }
619
620 // Down from here is only valid for API functions that can be used as a
621 // constructor (don't set the "remove prototype" flag).
622 DCHECK(result->has_prototype_slot());
623
624 if (obj->read_only_prototype()) {
625 result->set_map(isolate,
626 *isolate->sloppy_function_with_readonly_prototype_map());
627 }
628
629 if (IsTheHole(*prototype, isolate)) {
630 prototype = isolate->factory()->NewFunctionPrototype(result);
631 } else if (IsUndefined(obj->GetPrototypeProviderTemplate(), isolate)) {
632 JSObject::AddProperty(isolate, Cast<JSObject>(prototype),
633 isolate->factory()->constructor_string(), result,
634 DONT_ENUM);
635 }
636
637 int embedder_field_count = 0;
638 bool immutable_proto = false;
639 if (!IsUndefined(obj->GetInstanceTemplate(), isolate)) {
640 DirectHandle<ObjectTemplateInfo> GetInstanceTemplate(
641 Cast<ObjectTemplateInfo>(obj->GetInstanceTemplate()), isolate);
642 embedder_field_count = GetInstanceTemplate->embedder_field_count();
643 immutable_proto = GetInstanceTemplate->immutable_proto();
644 }
645
646 // JSFunction requires information about the prototype slot.
647 DCHECK(!InstanceTypeChecker::IsJSFunction(type));
648 int instance_size = JSObject::GetHeaderSize(type) +
649 kEmbedderDataSlotSize * embedder_field_count;
650
651 DirectHandle<Map> map = isolate->factory()->NewContextfulMap(
652 native_context, type, instance_size, TERMINAL_FAST_ELEMENTS_KIND);
653
654 // Mark as undetectable if needed.
655 if (obj->undetectable()) {
656 // We only allow callable undetectable receivers here, since this whole
657 // undetectable business is only to support document.all, which is both
658 // undetectable and callable. If we ever see the need to have an object
659 // that is undetectable but not callable, we need to update the types.h
660 // to allow encoding this.
661 CHECK(!IsUndefined(obj->GetInstanceCallHandler(), isolate));
662
663 if (Protectors::IsNoUndetectableObjectsIntact(isolate)) {
664 Protectors::InvalidateNoUndetectableObjects(isolate);
665 }
666 map->set_is_undetectable(true);
667 }
668
669 // Mark as needs_access_check if needed.
670 if (obj->needs_access_check()) {
671 map->set_is_access_check_needed(true);
672 map->set_may_have_interesting_properties(true);
673 }
674
675 // Set interceptor information in the map.
676 if (!IsUndefined(obj->GetNamedPropertyHandler(), isolate)) {
677 map->set_has_named_interceptor(true);
678 map->set_may_have_interesting_properties(true);
679 }
680 if (!IsUndefined(obj->GetIndexedPropertyHandler(), isolate)) {
681 map->set_has_indexed_interceptor(true);
682 }
683
684 // Mark instance as callable in the map.
685 if (!IsUndefined(obj->GetInstanceCallHandler(), isolate)) {
686 map->set_is_callable(true);
687 map->set_is_constructor(!obj->undetectable());
688 }
689
690 if (immutable_proto) map->set_is_immutable_proto(true);
691
692 JSFunction::SetInitialMap(isolate, result, map, Cast<JSObject>(prototype));
693 return result;
694}
