27// Adds Loong64-specific methods to convert InstructionOperands.
28class Loong64OperandConverter final : public InstructionOperandConverter {
29 public:
30 Loong64OperandConverter(CodeGenerator* gen, Instruction* instr)
31 : InstructionOperandConverter(gen, instr) {}
32
33 FloatRegister OutputSingleRegister(size_t index = 0) {
34 return ToSingleRegister(instr_->OutputAt(index));
35 }
36
37 FloatRegister InputSingleRegister(size_t index) {
38 return ToSingleRegister(instr_->InputAt(index));
39 }
40
41 FloatRegister ToSingleRegister(InstructionOperand* op) {
42 // Single (Float) and Double register namespace is same on LOONG64,
43 // both are typedefs of FPURegister.
44 return ToDoubleRegister(op);
45 }
46
47 Register InputOrZeroRegister(size_t index) {
48 if (instr_->InputAt(index)->IsImmediate()) {
49 DCHECK_EQ(0, InputInt32(index));
50 return zero_reg;
51 }
52 return InputRegister(index);
53 }
54
55 DoubleRegister InputOrZeroDoubleRegister(size_t index) {
56 if (instr_->InputAt(index)->IsImmediate()) return kDoubleRegZero;
57
58 return InputDoubleRegister(index);
59 }
60
61 DoubleRegister InputOrZeroSingleRegister(size_t index) {
62 if (instr_->InputAt(index)->IsImmediate()) return kDoubleRegZero;
63
64 return InputSingleRegister(index);
65 }
66
67 Operand InputImmediate(size_t index) {
68 Constant constant = ToConstant(instr_->InputAt(index));
69 switch (constant.type()) {
70 case Constant::kInt32:
71 return Operand(constant.ToInt32());
72 case Constant::kInt64:
73 return Operand(constant.ToInt64());
74 case Constant::kFloat32:
75 return Operand::EmbeddedNumber(constant.ToFloat32());
76 case Constant::kFloat64:
77 return Operand::EmbeddedNumber(constant.ToFloat64().value());
78 case Constant::kCompressedHeapObject: {
79 RootIndex root_index;
80 if (gen_->isolate()->roots_table().IsRootHandle(constant.ToHeapObject(),
81 &root_index)) {
82 CHECK(COMPRESS_POINTERS_BOOL);
83 CHECK(V8_STATIC_ROOTS_BOOL || !gen_->isolate()->bootstrapper());
84 Tagged_t ptr =
85 MacroAssemblerBase::ReadOnlyRootPtr(root_index, gen_->isolate());
86 return Operand(ptr);
87 }
88 return Operand(constant.ToHeapObject());
89 }
90 case Constant::kExternalReference:
91 case Constant::kHeapObject:
92 break;
93 case Constant::kRpoNumber:
94 UNREACHABLE(); // TODO(titzer): RPO immediates on loong64?
95 }
96 UNREACHABLE();
97 }
98
99 Operand InputOperand(size_t index) {
100 InstructionOperand* op = instr_->InputAt(index);
101 if (op->IsRegister()) {
102 return Operand(ToRegister(op));
103 }
104 return InputImmediate(index);
105 }
106
107 MemOperand MemoryOperand(size_t* first_index) {
108 const size_t index = *first_index;
109 switch (AddressingModeField::decode(instr_->opcode())) {
110 case kMode_None:
111 break;
112 case kMode_Root:
113 *first_index += 1;
114 return MemOperand(kRootRegister, InputInt32(index));
115 case kMode_MRI:
116 *first_index += 2;
117 return MemOperand(InputRegister(index + 0), InputInt32(index + 1));
118 case kMode_MRR:
119 *first_index += 2;
120 return MemOperand(InputRegister(index + 0), InputRegister(index + 1));
121 }
122 UNREACHABLE();
123 }
124
125 MemOperand MemoryOperand(size_t index = 0) { return MemoryOperand(&index); }
126
127 MemOperand ToMemOperand(InstructionOperand* op) const {
128 DCHECK_NOT_NULL(op);
129 DCHECK(op->IsStackSlot() || op->IsFPStackSlot());
130 return SlotToMemOperand(AllocatedOperand::cast(op)->index());
131 }
132
133 MemOperand SlotToMemOperand(int slot) const {
134 FrameOffset offset = frame_access_state()->GetFrameOffset(slot);
135 return MemOperand(offset.from_stack_pointer() ? sp : fp, offset.offset());
136 }
137};

145class OutOfLineRecordWrite final : public OutOfLineCode {
146 public:
147 OutOfLineRecordWrite(
148 CodeGenerator* gen, Register object, Operand offset, Register value,
149 RecordWriteMode mode, StubCallMode stub_mode,
150 IndirectPointerTag indirect_pointer_tag = kIndirectPointerNullTag)
151 : OutOfLineCode(gen),
152 object_(object),
153 offset_(offset),
154 value_(value),
155 mode_(mode),
156#if V8_ENABLE_WEBASSEMBLY
157 stub_mode_(stub_mode),
158#endif // V8_ENABLE_WEBASSEMBLY
159 must_save_lr_(!gen->frame_access_state()->has_frame()),
160 zone_(gen->zone()),
161 indirect_pointer_tag_(indirect_pointer_tag) {
162 }
163
164 void Generate() final {
165 // When storing an indirect pointer, the value will always be a
166 // full/decompressed pointer.
167 if (COMPRESS_POINTERS_BOOL &&
168 mode_ != RecordWriteMode::kValueIsIndirectPointer) {
169 __ DecompressTagged(value_, value_);
170 }
171
172 __ CheckPageFlag(value_, MemoryChunk::kPointersToHereAreInterestingMask, eq,
173 exit());
174
175 SaveFPRegsMode const save_fp_mode = frame()->DidAllocateDoubleRegisters()
176 ? SaveFPRegsMode::kSave
177 : SaveFPRegsMode::kIgnore;
178 if (must_save_lr_) {
179 // We need to save and restore ra if the frame was elided.
180 __ Push(ra);
181 }
182 if (mode_ == RecordWriteMode::kValueIsEphemeronKey) {
183 __ CallEphemeronKeyBarrier(object_, offset_, save_fp_mode);
184 } else if (mode_ == RecordWriteMode::kValueIsIndirectPointer) {
185 DCHECK(IsValidIndirectPointerTag(indirect_pointer_tag_));
186 __ CallIndirectPointerBarrier(object_, offset_, save_fp_mode,
187 indirect_pointer_tag_);
188#if V8_ENABLE_WEBASSEMBLY
189 } else if (stub_mode_ == StubCallMode::kCallWasmRuntimeStub) {
190 // A direct call to a wasm runtime stub defined in this module.
191 // Just encode the stub index. This will be patched when the code
192 // is added to the native module and copied into wasm code space.
193 __ CallRecordWriteStubSaveRegisters(object_, offset_, save_fp_mode,
194 StubCallMode::kCallWasmRuntimeStub);
195#endif // V8_ENABLE_WEBASSEMBLY
196 } else {
197 __ CallRecordWriteStubSaveRegisters(object_, offset_, save_fp_mode);
198 }
199 if (must_save_lr_) {
200 __ Pop(ra);
201 }
202 }
203
204 private:
205 Register const object_;
206 Operand const offset_;
207 Register const value_;
208 RecordWriteMode const mode_;
209#if V8_ENABLE_WEBASSEMBLY
210 StubCallMode const stub_mode_;
211#endif // V8_ENABLE_WEBASSEMBLY
212 bool must_save_lr_;
213 Zone* zone_;
214 IndirectPointerTag indirect_pointer_tag_;
215};

270void RecordTrapInfoIfNeeded(Zone* zone, CodeGenerator* codegen,
271 InstructionCode opcode, Instruction* instr,
272 int pc) {
273 const MemoryAccessMode access_mode = AccessModeField::decode(opcode);
274 if (access_mode == kMemoryAccessProtectedMemOutOfBounds ||
275 access_mode == kMemoryAccessProtectedNullDereference) {
276 ReferenceMap* reference_map =
277 codegen->zone()->New<ReferenceMap>(codegen->zone());
278 // The safepoint has to be recorded at the return address of a call. Address
279 // we use as the fake return address in the case of the trap handler is the
280 // fault address (here `pc`) + 1. Therefore the safepoint here has to be
281 // recorded at pc + 1;
282 codegen->RecordSafepoint(reference_map, pc + 1);
283 codegen->RecordProtectedInstruction(pc);
284 }

588void AdjustStackPointerForTailCall(MacroAssembler* masm,
589 FrameAccessState* state,
590 int new_slot_above_sp,
591 bool allow_shrinkage = true) {
592 int current_sp_offset = state->GetSPToFPSlotCount() +
593 StandardFrameConstants::kFixedSlotCountAboveFp;
594 int stack_slot_delta = new_slot_above_sp - current_sp_offset;
595 if (stack_slot_delta > 0) {
596 masm->Sub_d(sp, sp, stack_slot_delta * kSystemPointerSize);
597 state->IncreaseSPDelta(stack_slot_delta);
598 } else if (allow_shrinkage && stack_slot_delta < 0) {
599 masm->Add_d(sp, sp, -stack_slot_delta * kSystemPointerSize);
600 state->IncreaseSPDelta(stack_slot_delta);
601 }
602}

656// Assembles an instruction after register allocation, producing machine code.
657CodeGenerator::CodeGenResult CodeGenerator::AssembleArchInstruction(
658 Instruction* instr) {
659 Loong64OperandConverter i(this, instr);
660 InstructionCode opcode = instr->opcode();
661 ArchOpcode arch_opcode = ArchOpcodeField::decode(opcode);
662 switch (arch_opcode) {
663 case kArchCallCodeObject: {
664 if (instr->InputAt(0)->IsImmediate()) {
665 __ Call(i.InputCode(0), RelocInfo::CODE_TARGET);
666 } else {
667 Register reg = i.InputRegister(0);
668 CodeEntrypointTag tag =
669 i.InputCodeEntrypointTag(instr->CodeEnrypointTagInputIndex());
670 DCHECK_IMPLIES(
671 instr->HasCallDescriptorFlag(CallDescriptor::kFixedTargetRegister),
672 reg == kJavaScriptCallCodeStartRegister);
673 __ CallCodeObject(reg, tag);
674 }
675 RecordCallPosition(instr);
676 frame_access_state()->ClearSPDelta();
677 break;
678 }
679 case kArchCallBuiltinPointer: {
680 DCHECK(!instr->InputAt(0)->IsImmediate());
681 Register builtin_index = i.InputRegister(0);
682 Register target =
683 instr->HasCallDescriptorFlag(CallDescriptor::kFixedTargetRegister)
684 ? kJavaScriptCallCodeStartRegister
685 : builtin_index;
686 __ CallBuiltinByIndex(builtin_index, target);
687 RecordCallPosition(instr);
688 frame_access_state()->ClearSPDelta();
689 break;
690 }
691#if V8_ENABLE_WEBASSEMBLY
692 case kArchCallWasmFunction:
693 case kArchCallWasmFunctionIndirect: {
694 if (instr->InputAt(0)->IsImmediate()) {
695 DCHECK_EQ(arch_opcode, kArchCallWasmFunction);
696 Constant constant = i.ToConstant(instr->InputAt(0));
697 Address wasm_code = static_cast<Address>(constant.ToInt64());
698 __ Call(wasm_code, constant.rmode());
699 } else if (arch_opcode == kArchCallWasmFunctionIndirect) {
700 __ CallWasmCodePointer(
701 i.InputRegister(0),
702 i.InputInt64(instr->WasmSignatureHashInputIndex()));
703 } else {
704 __ Call(i.InputRegister(0));
705 }
706 RecordCallPosition(instr);
707 frame_access_state()->ClearSPDelta();
708 break;
709 }
710 case kArchTailCallWasm:
711 case kArchTailCallWasmIndirect: {
712 if (instr->InputAt(0)->IsImmediate()) {
713 DCHECK_EQ(arch_opcode, kArchTailCallWasm);
714 Constant constant = i.ToConstant(instr->InputAt(0));
715 Address wasm_code = static_cast<Address>(constant.ToInt64());
716 __ Jump(wasm_code, constant.rmode());
717 } else if (arch_opcode == kArchTailCallWasmIndirect) {
718 __ CallWasmCodePointer(
719 i.InputRegister(0),
720 i.InputInt64(instr->WasmSignatureHashInputIndex()),
721 CallJumpMode::kTailCall);
722 } else {
723 __ Jump(i.InputRegister(0));
724 }
725 frame_access_state()->ClearSPDelta();
726 frame_access_state()->SetFrameAccessToDefault();
727 break;
728 }
729#endif // V8_ENABLE_WEBASSEMBLY
730 case kArchTailCallCodeObject: {
731 if (instr->InputAt(0)->IsImmediate()) {
732 __ Jump(i.InputCode(0), RelocInfo::CODE_TARGET);
733 } else {
734 Register reg = i.InputRegister(0);
735 CodeEntrypointTag tag =
736 i.InputCodeEntrypointTag(instr->CodeEnrypointTagInputIndex());
737 DCHECK_IMPLIES(
738 instr->HasCallDescriptorFlag(CallDescriptor::kFixedTargetRegister),
739 reg == kJavaScriptCallCodeStartRegister);
740 __ JumpCodeObject(reg, tag);
741 }
742 frame_access_state()->ClearSPDelta();
743 frame_access_state()->SetFrameAccessToDefault();
744 break;
745 }
746 case kArchTailCallAddress: {
747 CHECK(!instr->InputAt(0)->IsImmediate());
748 Register reg = i.InputRegister(0);
749 DCHECK_IMPLIES(
750 instr->HasCallDescriptorFlag(CallDescriptor::kFixedTargetRegister),
751 reg == kJavaScriptCallCodeStartRegister);
752 __ Jump(reg);
753 frame_access_state()->ClearSPDelta();
754 frame_access_state()->SetFrameAccessToDefault();
755 break;
756 }
757 case kArchCallJSFunction: {
758 Register func = i.InputRegister(0);
759 if (v8_flags.debug_code) {
760 UseScratchRegisterScope temps(masm());
761 Register scratch = temps.Acquire();
762 // Check the function's context matches the context argument.
763 __ LoadTaggedField(scratch,
764 FieldMemOperand(func, JSFunction::kContextOffset));
765 __ Assert(eq, AbortReason::kWrongFunctionContext, cp, Operand(scratch));
766 }
767 uint32_t num_arguments =
768 i.InputUint32(instr->JSCallArgumentCountInputIndex());
769 __ CallJSFunction(func, num_arguments);
770 RecordCallPosition(instr);
771 frame_access_state()->ClearSPDelta();
772 break;
773 }
774 case kArchPrepareCallCFunction: {
775 UseScratchRegisterScope temps(masm());
776 Register scratch = temps.Acquire();
777 int const num_gp_parameters = ParamField::decode(instr->opcode());
778 int const num_fp_parameters = FPParamField::decode(instr->opcode());
779 __ PrepareCallCFunction(num_gp_parameters, num_fp_parameters, scratch);
780 // Frame alignment requires using FP-relative frame addressing.
781 frame_access_state()->SetFrameAccessToFP();
782 break;
783 }
784 case kArchSaveCallerRegisters: {
785 fp_mode_ =
786 static_cast<SaveFPRegsMode>(MiscField::decode(instr->opcode()));
787 DCHECK(fp_mode_ == SaveFPRegsMode::kIgnore ||
788 fp_mode_ == SaveFPRegsMode::kSave);
789 // kReturnRegister0 should have been saved before entering the stub.
790 int bytes = __ PushCallerSaved(fp_mode_, kReturnRegister0);
791 DCHECK(IsAligned(bytes, kSystemPointerSize));
792 DCHECK_EQ(0, frame_access_state()->sp_delta());
793 frame_access_state()->IncreaseSPDelta(bytes / kSystemPointerSize);
794 DCHECK(!caller_registers_saved_);
795 caller_registers_saved_ = true;
796 break;
797 }
798 case kArchRestoreCallerRegisters: {
799 DCHECK(fp_mode_ ==
800 static_cast<SaveFPRegsMode>(MiscField::decode(instr->opcode())));
801 DCHECK(fp_mode_ == SaveFPRegsMode::kIgnore ||
802 fp_mode_ == SaveFPRegsMode::kSave);
803 // Don't overwrite the returned value.
804 int bytes = __ PopCallerSaved(fp_mode_, kReturnRegister0);
805 frame_access_state()->IncreaseSPDelta(-(bytes / kSystemPointerSize));
806 DCHECK_EQ(0, frame_access_state()->sp_delta());
807 DCHECK(caller_registers_saved_);
808 caller_registers_saved_ = false;
809 break;
810 }
811 case kArchPrepareTailCall:
812 AssemblePrepareTailCall();
813 break;
814 case kArchCallCFunctionWithFrameState:
815 case kArchCallCFunction: {

817 int const num_fp_parameters = FPParamField::decode(instr->opcode());
818 SetIsolateDataSlots set_isolate_data_slots = SetIsolateDataSlots::kYes;
819 Label return_location;
820#if V8_ENABLE_WEBASSEMBLY
821 bool isWasmCapiFunction =
822 linkage()->GetIncomingDescriptor()->IsWasmCapiFunction();

825 __ St_d(t7, MemOperand(fp, WasmExitFrameConstants::kCallingPCOffset));
826 set_isolate_data_slots = SetIsolateDataSlots::kNo;
827 }
828#endif // V8_ENABLE_WEBASSEMBLY
829 int pc_offset;
830 if (instr->InputAt(0)->IsImmediate()) {
831 ExternalReference ref = i.InputExternalReference(0);
832 pc_offset = __ CallCFunction(ref, num_gp_parameters, num_fp_parameters,
833 set_isolate_data_slots, &return_location);
834 } else {
835 Register func = i.InputRegister(0);
836 pc_offset = __ CallCFunction(func, num_gp_parameters, num_fp_parameters,
837 set_isolate_data_slots, &return_location);
838 }
839 RecordSafepoint(instr->reference_map(), pc_offset);
840
841 bool const needs_frame_state =
842 (arch_opcode == kArchCallCFunctionWithFrameState);
843 if (needs_frame_state) {
844 RecordDeoptInfo(instr, pc_offset);
845 }
846
847 frame_access_state()->SetFrameAccessToDefault();
848 // Ideally, we should decrement SP delta to match the change of stack
849 // pointer in CallCFunction. However, for certain architectures (e.g.
850 // ARM), there may be more strict alignment requirement, causing old SP
851 // to be saved on the stack. In those cases, we can not calculate the SP
852 // delta statically.
853 frame_access_state()->ClearSPDelta();
854 if (caller_registers_saved_) {
855 // Need to re-sync SP delta introduced in kArchSaveCallerRegisters.
856 // Here, we assume the sequence to be:
857 // kArchSaveCallerRegisters;
858 // kArchCallCFunction;
859 // kArchRestoreCallerRegisters;
860 int bytes =
861 __ RequiredStackSizeForCallerSaved(fp_mode_, kReturnRegister0);
862 frame_access_state()->IncreaseSPDelta(bytes / kSystemPointerSize);
863 }
864 break;
865 }
866 case kArchJmp:
867 AssembleArchJump(i.InputRpo(0));
868 break;
869 case kArchBinarySearchSwitch:
870 AssembleArchBinarySearchSwitch(instr);
871 break;
872 case kArchTableSwitch:

874 break;
875 case kArchAbortCSADcheck:

879 // claim there is a stack frame, without generating one.

884 break;

887 break;

891 break;

2176 Instruction* instr, FlagsCondition condition,
2177 Label* tlabel, Label* flabel, bool fallthru) {
2178#undef __
2179#define __ masm->
2180 Loong64OperandConverter i(gen, instr);
2181
2182 // LOONG64 does not have condition code flags, so compare and branch are
2183 // implemented differently than on the other arch's. The compare operations
2184 // emit loong64 pseudo-instructions, which are handled here by branch
2185 // instructions that do the actual comparison. Essential that the input
2186 // registers to compare pseudo-op are not modified before this branch op, as
2187 // they are tested here.
2188
2189 if (instr->arch_opcode() == kLoong64Tst) {
2190 Condition cc = FlagsConditionToConditionTst(condition);
2191 __ Branch(tlabel, cc, t8, Operand(zero_reg));
2192 } else if (instr->arch_opcode() == kLoong64Add_d ||
2193 instr->arch_opcode() == kLoong64Sub_d) {
2194 UseScratchRegisterScope temps(masm);
2195 Register scratch = temps.Acquire();
2196 Register scratch2 = temps.Acquire();
2197 Condition cc = FlagsConditionToConditionOvf(condition);
2198 __ srai_d(scratch, i.OutputRegister(), 32);
2199 __ srai_w(scratch2, i.OutputRegister(), 31);
2200 __ Branch(tlabel, cc, scratch2, Operand(scratch));
2201 } else if (instr->arch_opcode() == kLoong64AddOvf_d ||
2202 instr->arch_opcode() == kLoong64SubOvf_d) {
2203 switch (condition) {
2204 // Overflow occurs if overflow register is negative
2205 case kOverflow:
2206 __ Branch(tlabel, lt, t8, Operand(zero_reg));
2207 break;
2208 case kNotOverflow:
2209 __ Branch(tlabel, ge, t8, Operand(zero_reg));
2210 break;
2211 default:
2212 UNSUPPORTED_COND(instr->arch_opcode(), condition);
2213 }
2214 } else if (instr->arch_opcode() == kLoong64MulOvf_w ||
2215 instr->arch_opcode() == kLoong64MulOvf_d) {
2216 // Overflow occurs if overflow register is not zero
2217 switch (condition) {
2218 case kOverflow:
2219 __ Branch(tlabel, ne, t8, Operand(zero_reg));
2220 break;
2221 case kNotOverflow:
2222 __ Branch(tlabel, eq, t8, Operand(zero_reg));
2223 break;
2224 default:
2225 UNSUPPORTED_COND(instr->arch_opcode(), condition);
2226 }
2227 } else if (instr->arch_opcode() == kLoong64Cmp32 ||
2228 instr->arch_opcode() == kLoong64Cmp64) {
2229 Condition cc = FlagsConditionToConditionCmp(condition);
2230 Register left = i.InputRegister(0);
2231 Operand right = i.InputOperand(1);
2232 // Word32Compare has two temp registers.
2233 if (COMPRESS_POINTERS_BOOL && (instr->arch_opcode() == kLoong64Cmp32)) {
2234 Register temp0 = i.TempRegister(0);
2235 Register temp1 = right.is_reg() ? i.TempRegister(1) : no_reg;
2236 SignExtend(masm, instr, &left, &right, &temp0, &temp1);
2237 }
2238 __ Branch(tlabel, cc, left, right);
2239 } else if (instr->arch_opcode() == kArchStackPointerGreaterThan) {
2240 Condition cc = FlagsConditionToConditionCmp(condition);
2241 DCHECK((cc == ls) || (cc == hi));
2242 if (cc == ls) {
2243 __ xori(i.TempRegister(0), i.TempRegister(0), 1);
2244 }
2245 __ Branch(tlabel, ne, i.TempRegister(0), Operand(zero_reg));
2246 } else if (instr->arch_opcode() == kLoong64Float32Cmp ||
2247 instr->arch_opcode() == kLoong64Float64Cmp) {
2248 bool predicate;
2249 FlagsConditionToConditionCmpFPU(&predicate, condition);
2250 if (predicate) {
2251 __ BranchTrueF(tlabel);

2253 __ BranchFalseF(tlabel);
2254 }
2255 } else {

2257 instr->arch_opcode());

2260 if (!fallthru) __ Branch(flabel); // no fallthru to flabel.
2261#undef __

2311 // emit loong64 pseudo-instructions, which are checked and handled here.
2312
2313 if (instr->arch_opcode() == kLoong64Tst) {
2314 Condition cc = FlagsConditionToConditionTst(condition);
2315 if (cc == eq) {
2316 __ Sltu(result, t8, 1);
2317 } else {
2318 __ Sltu(result, zero_reg, t8);
2319 }
2320 return;
2321 } else if (instr->arch_opcode() == kLoong64Add_d ||
2322 instr->arch_opcode() == kLoong64Sub_d) {
2323 UseScratchRegisterScope temps(masm());
2324 Register scratch = temps.Acquire();
2325 Condition cc = FlagsConditionToConditionOvf(condition);
2326 // Check for overflow creates 1 or 0 for result.
2327 __ srli_d(scratch, i.OutputRegister(), 63);
2328 __ srli_w(result, i.OutputRegister(), 31);
2329 __ xor_(result, scratch, result);
2330 if (cc == eq) // Toggle result for not overflow.
2331 __ xori(result, result, 1);
2332 return;
2333 } else if (instr->arch_opcode() == kLoong64AddOvf_d ||
2334 instr->arch_opcode() == kLoong64SubOvf_d) {
2335 // Overflow occurs if overflow register is negative
2336 __ slt(result, t8, zero_reg);
2337 } else if (instr->arch_opcode() == kLoong64MulOvf_w ||
2338 instr->arch_opcode() == kLoong64MulOvf_d) {
2339 // Overflow occurs if overflow register is not zero
2340 __ Sgtu(result, t8, zero_reg);
2341 } else if (instr->arch_opcode() == kLoong64Cmp32 ||
2342 instr->arch_opcode() == kLoong64Cmp64) {
2343 Condition cc = FlagsConditionToConditionCmp(condition);
2344 Register left = i.InputRegister(0);
2345 Operand right = i.InputOperand(1);
2346 if (COMPRESS_POINTERS_BOOL && (instr->arch_opcode() == kLoong64Cmp32)) {
2347 Register temp0 = i.TempRegister(0);
2348 Register temp1 = right.is_reg() ? i.TempRegister(1) : no_reg;
2349 SignExtend(masm(), instr, &left, &right, &temp0, &temp1);
2350 }
2351 __ CompareWord(cc, result, left, right);
2352 return;
2353 } else if (instr->arch_opcode() == kLoong64Float64Cmp ||
2354 instr->arch_opcode() == kLoong64Float32Cmp) {
2355 FPURegister left = i.InputOrZeroDoubleRegister(0);
2356 FPURegister right = i.InputOrZeroDoubleRegister(1);
2357 if ((left == kDoubleRegZero || right == kDoubleRegZero) &&
2358 !__ IsDoubleZeroRegSet()) {
2359 __ Move(kDoubleRegZero, 0.0);
2360 }
2361 bool predicate;
2362 FlagsConditionToConditionCmpFPU(&predicate, condition);
2363 {
2364 __ movcf2gr(result, FCC0);
2365 if (!predicate) {
2366 __ xori(result, result, 1);
2367 }
2368 }
2369 return;
2370 } else if (instr->arch_opcode() == kArchStackPointerGreaterThan) {
2371 Condition cc = FlagsConditionToConditionCmp(condition);
2372 DCHECK((cc == ls) || (cc == hi));
2373 if (cc == ls) {
2374 __ xori(i.OutputRegister(), i.TempRegister(0), 1);
2375 }
2376 return;
2377 } else {
2378 PrintF("AssembleArchBranch Unimplemented arch_opcode is : %d\n",
2379 instr->arch_opcode());
2380 TRACE("UNIMPLEMENTED code_generator_loong64: %s at line %d\n", __FUNCTION__,
2381 __LINE__);
2382 UNIMPLEMENTED();
2383 }
2384}

2446void CodeGenerator::AssembleConstructFrame() {
2447 auto call_descriptor = linkage()->GetIncomingDescriptor();
2448
2449 if (frame_access_state()->has_frame()) {
2450 if (call_descriptor->IsCFunctionCall()) {
2451#if V8_ENABLE_WEBASSEMBLY
2452 if (info()->GetOutputStackFrameType() == StackFrame::C_WASM_ENTRY) {
2453 __ StubPrologue(StackFrame::C_WASM_ENTRY);
2454 // Reserve stack space for saving the c_entry_fp later.
2455 __ Sub_d(sp, sp, Operand(kSystemPointerSize));
2456#else
2457 // For balance.
2458 if (false) {
2459#endif // V8_ENABLE_WEBASSEMBLY
2460 } else {
2461 __ Push(ra, fp);
2462 __ mov(fp, sp);
2463 }
2464 } else if (call_descriptor->IsJSFunctionCall()) {
2465 __ Prologue();
2466 } else {
2467 __ StubPrologue(info()->GetOutputStackFrameType());
2468#if V8_ENABLE_WEBASSEMBLY
2469 if (call_descriptor->IsAnyWasmFunctionCall() ||
2470 call_descriptor->IsWasmImportWrapper() ||
2471 call_descriptor->IsWasmCapiFunction()) {
2472 // For import wrappers and C-API functions, this stack slot is only used
2473 // for printing stack traces in V8. Also, it holds a WasmImportData
2474 // instead of the trusted instance data, which is taken care of in the
2475 // frames accessors.
2476 __ Push(kWasmImplicitArgRegister);
2477 }
2478 if (call_descriptor->IsWasmCapiFunction()) {
2479 // Reserve space for saving the PC later.
2480 __ Sub_d(sp, sp, Operand(kSystemPointerSize));
2481 }
2482#endif // V8_ENABLE_WEBASSEMBLY
2483 }
2484 }
2485
2486 int required_slots =
2487 frame()->GetTotalFrameSlotCount() - frame()->GetFixedSlotCount();
2488
2489 if (info()->is_osr()) {
2490 // TurboFan OSR-compiled functions cannot be entered directly.
2491 __ Abort(AbortReason::kShouldNotDirectlyEnterOsrFunction);
2492
2493 // Unoptimized code jumps directly to this entrypoint while the unoptimized
2494 // frame is still on the stack. Optimized code uses OSR values directly from
2495 // the unoptimized frame. Thus, all that needs to be done is to allocate the
2496 // remaining stack slots.
2497 __ RecordComment("-- OSR entrypoint --");
2498 osr_pc_offset_ = __ pc_offset();
2499 required_slots -= osr_helper()->UnoptimizedFrameSlots();
2500 }
2501
2502 const RegList saves = call_descriptor->CalleeSavedRegisters();
2503 const DoubleRegList saves_fpu = call_descriptor->CalleeSavedFPRegisters();
2504
2505 if (required_slots > 0) {
2506 DCHECK(frame_access_state()->has_frame());
2507#if V8_ENABLE_WEBASSEMBLY
2508 if (info()->IsWasm() && required_slots * kSystemPointerSize > 4 * KB) {
2509 // For WebAssembly functions with big frames we have to do the stack
2510 // overflow check before we construct the frame. Otherwise we may not
2511 // have enough space on the stack to call the runtime for the stack
2512 // overflow.
2513 Label done;
2514
2515 // If the frame is bigger than the stack, we throw the stack overflow
2516 // exception unconditionally. Thereby we can avoid the integer overflow
2517 // check in the condition code.
2518 if (required_slots * kSystemPointerSize < v8_flags.stack_size * KB) {
2519 UseScratchRegisterScope temps(masm());
2520 Register stack_limit = temps.Acquire();
2521 __ LoadStackLimit(stack_limit,
2522 MacroAssembler::StackLimitKind::kRealStackLimit);
2523 __ Add_d(stack_limit, stack_limit,
2524 Operand(required_slots * kSystemPointerSize));
2525 __ Branch(&done, uge, sp, Operand(stack_limit));
2526 }
2527
2528 if (v8_flags.experimental_wasm_growable_stacks) {
2529 RegList regs_to_save;
2530 regs_to_save.set(WasmHandleStackOverflowDescriptor::GapRegister());
2531 regs_to_save.set(
2532 WasmHandleStackOverflowDescriptor::FrameBaseRegister());
2533
2534 for (auto reg : wasm::kGpParamRegisters) regs_to_save.set(reg);
2535 __ MultiPush(regs_to_save);
2536 DoubleRegList fp_regs_to_save;
2537 for (auto reg : wasm::kFpParamRegisters) fp_regs_to_save.set(reg);
2538 __ MultiPushFPU(fp_regs_to_save);
2539 __ li(WasmHandleStackOverflowDescriptor::GapRegister(),
2540 required_slots * kSystemPointerSize);
2541 __ Add_d(
2542 WasmHandleStackOverflowDescriptor::FrameBaseRegister(), fp,
2543 Operand(call_descriptor->ParameterSlotCount() * kSystemPointerSize +
2544 CommonFrameConstants::kFixedFrameSizeAboveFp));
2545 __ CallBuiltin(Builtin::kWasmHandleStackOverflow);
2546 __ MultiPopFPU(fp_regs_to_save);
2547 __ MultiPop(regs_to_save);
2548 } else {
2549 __ Call(static_cast<intptr_t>(Builtin::kWasmStackOverflow),
2550 RelocInfo::WASM_STUB_CALL);
2551 // The call does not return, hence we can ignore any references and just
2552 // define an empty safepoint.
2553 ReferenceMap* reference_map = zone()->New<ReferenceMap>(zone());
2554 RecordSafepoint(reference_map);
2555 if (v8_flags.debug_code) {
2556 __ stop();
2557 }
2558 }
2559
2560 __ bind(&done);
2561 }
2562#endif // V8_ENABLE_WEBASSEMBLY
2563 }
2564
2565 const int returns = frame()->GetReturnSlotCount();
2566
2567 // Skip callee-saved and return slots, which are pushed below.
2568 required_slots -= saves.Count();
2569 required_slots -= saves_fpu.Count();
2570 required_slots -= returns;
2571 if (required_slots > 0) {
2572 __ Sub_d(sp, sp, Operand(required_slots * kSystemPointerSize));
2573 }
2574
2575 if (!saves_fpu.is_empty()) {
2576 // Save callee-saved FPU registers.
2577 __ MultiPushFPU(saves_fpu);
2578 DCHECK_EQ(kNumCalleeSavedFPU, saves_fpu.Count());
2579 }
2580
2581 if (!saves.is_empty()) {
2582 // Save callee-saved registers.
2583 __ MultiPush(saves);
2584 }
2585
2586 if (returns != 0) {
2587 // Create space for returns.
2588 __ Sub_d(sp, sp, Operand(returns * kSystemPointerSize));
2589 }
2590
2591 for (int spill_slot : frame()->tagged_slots()) {
2592 FrameOffset offset = frame_access_state()->GetFrameOffset(spill_slot);
2593 DCHECK(offset.from_frame_pointer());
2594 __ St_d(zero_reg, MemOperand(fp, offset.offset()));
2595 }
2596}

2598void CodeGenerator::AssembleReturn(InstructionOperand* additional_pop_count) {
2599 auto call_descriptor = linkage()->GetIncomingDescriptor();
2600
2601 const int returns = frame()->GetReturnSlotCount();
2602 if (returns != 0) {
2603 __ Add_d(sp, sp, Operand(returns * kSystemPointerSize));
2604 }
2605
2606 // Restore GP registers.
2607 const RegList saves = call_descriptor->CalleeSavedRegisters();
2608 if (!saves.is_empty()) {
2609 __ MultiPop(saves);
2610 }
2611
2612 // Restore FPU registers.
2613 const DoubleRegList saves_fpu = call_descriptor->CalleeSavedFPRegisters();
2614 if (!saves_fpu.is_empty()) {
2615 __ MultiPopFPU(saves_fpu);
2616 }
2617
2618 Loong64OperandConverter g(this, nullptr);
2619
2620 const int parameter_slots =
2621 static_cast<int>(call_descriptor->ParameterSlotCount());
2622
2623 // {aditional_pop_count} is only greater than zero if {parameter_slots = 0}.
2624 // Check RawMachineAssembler::PopAndReturn.
2625 if (parameter_slots != 0) {
2626 if (additional_pop_count->IsImmediate()) {
2627 DCHECK_EQ(g.ToConstant(additional_pop_count).ToInt32(), 0);
2628 } else if (v8_flags.debug_code) {
2629 __ Assert(eq, AbortReason::kUnexpectedAdditionalPopValue,
2630 g.ToRegister(additional_pop_count),
2631 Operand(static_cast<int64_t>(0)));
2632 }
2633 }
2634
2635#if V8_ENABLE_WEBASSEMBLY
2636 if (call_descriptor->IsAnyWasmFunctionCall() &&
2637 v8_flags.experimental_wasm_growable_stacks) {
2638 Label done;
2639 {
2640 UseScratchRegisterScope temps{masm()};
2641 Register scratch = temps.Acquire();
2642 __ Ld_d(scratch, MemOperand(fp, TypedFrameConstants::kFrameTypeOffset));
2643 __ BranchShort(
2644 &done, ne, scratch,
2645 Operand(StackFrame::TypeToMarker(StackFrame::WASM_SEGMENT_START)));
2646 }
2647 RegList regs_to_save;
2648 for (auto reg : wasm::kGpReturnRegisters) regs_to_save.set(reg);
2649 __ MultiPush(regs_to_save);
2650 __ li(kCArgRegs[0], ExternalReference::isolate_address());
2651 {
2652 UseScratchRegisterScope temps{masm()};
2653 Register scratch = temps.Acquire();
2654 __ PrepareCallCFunction(1, scratch);
2655 }
2656 __ CallCFunction(ExternalReference::wasm_shrink_stack(), 1);
2657 __ mov(fp, kReturnRegister0);
2658 __ MultiPop(regs_to_save);
2659 __ bind(&done);
2660 }
2661#endif // V8_ENABLE_WEBASSEMBLY
2662
2663 // Functions with JS linkage have at least one parameter (the receiver).
2664 // If {parameter_slots} == 0, it means it is a builtin with
2665 // kDontAdaptArgumentsSentinel, which takes care of JS arguments popping
2666 // itself.
2667 const bool drop_jsargs = frame_access_state()->has_frame() &&
2668 call_descriptor->IsJSFunctionCall() &&
2669 parameter_slots != 0;
2670
2671 if (call_descriptor->IsCFunctionCall()) {
2672 AssembleDeconstructFrame();
2673 } else if (frame_access_state()->has_frame()) {
2674 // Canonicalize JSFunction return sites for now unless they have an variable
2675 // number of stack slot pops.
2676 if (additional_pop_count->IsImmediate() &&
2677 g.ToConstant(additional_pop_count).ToInt32() == 0) {
2678 if (return_label_.is_bound()) {
2679 __ Branch(&return_label_);
2680 return;
2681 } else {
2682 __ bind(&return_label_);
2683 }
2684 }
2685 if (drop_jsargs) {
2686 // Get the actual argument count
2687 __ Ld_d(t0, MemOperand(fp, StandardFrameConstants::kArgCOffset));
2688 }
2689 AssembleDeconstructFrame();
2690 }
2691 if (drop_jsargs) {
2692 // We must pop all arguments from the stack (including the receiver). This
2693 // number of arguments is given by max(1 + argc_reg, parameter_count).
2694 if (parameter_slots > 1) {
2695 __ li(t1, parameter_slots);
2696 __ slt(t2, t0, t1);
2697 __ Movn(t0, t1, t2);
2698 }
2699 __ Alsl_d(sp, t0, sp, kSystemPointerSizeLog2);
2700 } else if (additional_pop_count->IsImmediate()) {
2701 int additional_count = g.ToConstant(additional_pop_count).ToInt32();
2702 __ Drop(parameter_slots + additional_count);
2703 } else {
2704 Register pop_reg = g.ToRegister(additional_pop_count);
2705 __ Drop(parameter_slots);
2706 __ Alsl_d(sp, pop_reg, sp, kSystemPointerSizeLog2);
2707 }
2708 __ Ret();
2709}

2716AllocatedOperand CodeGenerator::Push(InstructionOperand* source) {
2717 auto rep = LocationOperand::cast(source)->representation();
2718 int new_slots = ElementSizeInPointers(rep);
2719 Loong64OperandConverter g(this, nullptr);
2720 int last_frame_slot_id =
2721 frame_access_state_->frame()->GetTotalFrameSlotCount() - 1;
2722 int sp_delta = frame_access_state_->sp_delta();
2723 int slot_id = last_frame_slot_id + sp_delta + new_slots;
2724 AllocatedOperand stack_slot(LocationOperand::STACK_SLOT, rep, slot_id);
2725 if (source->IsRegister()) {
2726 __ Push(g.ToRegister(source));
2727 frame_access_state()->IncreaseSPDelta(new_slots);
2728 } else if (source->IsStackSlot()) {
2729 UseScratchRegisterScope temps(masm());
2730 Register scratch = temps.Acquire();
2731 __ Ld_d(scratch, g.ToMemOperand(source));
2732 __ Push(scratch);
2733 frame_access_state()->IncreaseSPDelta(new_slots);
2734 } else {
2735 // No push instruction for this operand type. Bump the stack pointer and
2736 // assemble the move.
2737 __ Sub_d(sp, sp, Operand(new_slots * kSystemPointerSize));
2738 frame_access_state()->IncreaseSPDelta(new_slots);
2739 AssembleMove(source, &stack_slot);
2740 }
2741 temp_slots_ += new_slots;
2742 return stack_slot;
2743}

2745void CodeGenerator::Pop(InstructionOperand* dest, MachineRepresentation rep) {
2746 Loong64OperandConverter g(this, nullptr);
2747 int dropped_slots = ElementSizeInPointers(rep);
2748 if (dest->IsRegister()) {
2749 frame_access_state()->IncreaseSPDelta(-dropped_slots);
2750 __ Pop(g.ToRegister(dest));
2751 } else if (dest->IsStackSlot()) {
2752 frame_access_state()->IncreaseSPDelta(-dropped_slots);
2753 UseScratchRegisterScope temps(masm());
2754 Register scratch = temps.Acquire();
2755 __ Pop(scratch);
2756 __ St_d(scratch, g.ToMemOperand(dest));
2757 } else {
2758 int last_frame_slot_id =
2759 frame_access_state_->frame()->GetTotalFrameSlotCount() - 1;
2760 int sp_delta = frame_access_state_->sp_delta();
2761 int slot_id = last_frame_slot_id + sp_delta;
2762 AllocatedOperand stack_slot(LocationOperand::STACK_SLOT, rep, slot_id);
2763 AssembleMove(&stack_slot, dest);
2764 frame_access_state()->IncreaseSPDelta(-dropped_slots);
2765 __ Add_d(sp, sp, Operand(dropped_slots * kSystemPointerSize));
2766 }
2767 temp_slots_ -= dropped_slots;
2768}

2778void CodeGenerator::MoveToTempLocation(InstructionOperand* source,
2779 MachineRepresentation rep) {
2780 // Must be kept in sync with {MoveTempLocationTo}.
2781 DCHECK(!source->IsImmediate());
2782 move_cycle_.temps.emplace(masm());
2783 auto& temps = *move_cycle_.temps;
2784 // Temporarily exclude the reserved scratch registers while we pick one to
2785 // resolve the move cycle. Re-include them immediately afterwards as they
2786 // might be needed for the move to the temp location.
2787 temps.Exclude(move_cycle_.scratch_regs);
2788 temps.ExcludeFp(move_cycle_.scratch_fpregs);
2789 if (!IsFloatingPoint(rep)) {
2790 if (temps.hasAvailable()) {
2791 Register scratch = move_cycle_.temps->Acquire();
2792 move_cycle_.scratch_reg.emplace(scratch);
2793 } else if (temps.hasAvailableFp()) {
2794 // Try to use an FP register if no GP register is available for non-FP
2795 // moves.
2796 FPURegister scratch = move_cycle_.temps->AcquireFp();
2797 move_cycle_.scratch_fpreg.emplace(scratch);
2798 }
2799 } else {
2800 DCHECK(temps.hasAvailableFp());
2801 FPURegister scratch = move_cycle_.temps->AcquireFp();
2802 move_cycle_.scratch_fpreg.emplace(scratch);
2803 }
2804 temps.Include(move_cycle_.scratch_regs);
2805 temps.IncludeFp(move_cycle_.scratch_fpregs);
2806 if (move_cycle_.scratch_reg.has_value()) {
2807 // A scratch register is available for this rep.
2808 AllocatedOperand scratch(LocationOperand::REGISTER, rep,
2809 move_cycle_.scratch_reg->code());
2810 AssembleMove(source, &scratch);
2811 } else if (move_cycle_.scratch_fpreg.has_value()) {
2812 // A scratch fp register is available for this rep.
2813 if (!IsFloatingPoint(rep)) {
2814 AllocatedOperand scratch(LocationOperand::REGISTER,
2815 MachineRepresentation::kFloat64,
2816 move_cycle_.scratch_fpreg->code());
2817 Loong64OperandConverter g(this, nullptr);
2818 if (source->IsStackSlot()) {
2819 __ Fld_d(g.ToDoubleRegister(&scratch), g.ToMemOperand(source));
2820 } else {
2821 DCHECK(source->IsRegister());
2822 __ movgr2fr_d(g.ToDoubleRegister(&scratch), g.ToRegister(source));
2823 }
2824 } else {
2825 AllocatedOperand scratch(LocationOperand::REGISTER, rep,
2826 move_cycle_.scratch_fpreg->code());
2827 AssembleMove(source, &scratch);
2828 }
2829 } else {
2830 // The scratch registers are blocked by pending moves. Use the stack
2831 // instead.
2832 Push(source);
2833 }
2834}

2869void CodeGenerator::SetPendingMove(MoveOperands* move) {
2870 InstructionOperand* src = &move->source();
2871 InstructionOperand* dst = &move->destination();
2872 UseScratchRegisterScope temps(masm());
2873 if (src->IsConstant() || (src->IsStackSlot() && dst->IsStackSlot())) {
2874 Register temp = temps.Acquire();
2875 move_cycle_.scratch_regs.set(temp);
2876 }
2877 if (src->IsAnyStackSlot() || dst->IsAnyStackSlot()) {
2878 Loong64OperandConverter g(this, nullptr);
2879 bool src_need_scratch = false;
2880 bool dst_need_scratch = false;
2881 if (src->IsStackSlot()) {
2882 // Doubleword load/store
2883 MemOperand src_mem = g.ToMemOperand(src);
2884 src_need_scratch =

2928void CodeGenerator::AssembleMove(InstructionOperand* source,
2929 InstructionOperand* destination) {
2930 Loong64OperandConverter g(this, nullptr);
2931 // Dispatch on the source and destination operand kinds. Not all
2932 // combinations are possible.
2933 if (source->IsRegister()) {
2934 DCHECK(destination->IsRegister() || destination->IsStackSlot());
2935 Register src = g.ToRegister(source);
2936 if (destination->IsRegister()) {
2937 __ mov(g.ToRegister(destination), src);
2938 } else {
2939 __ St_d(src, g.ToMemOperand(destination));
2940 }
2941 } else if (source->IsStackSlot()) {
2942 DCHECK(destination->IsRegister() || destination->IsStackSlot());
2943 MemOperand src = g.ToMemOperand(source);
2944 if (destination->IsRegister()) {
2945 if (Use32BitMove(source, destination)) {
2946 __ Ld_w(g.ToRegister(destination), src);
2947 } else {
2948 __ Ld_d(g.ToRegister(destination), src);
2949 }
2950 } else {
2951 UseScratchRegisterScope temps(masm());
2952 Register scratch = temps.Acquire();
2953 __ Ld_d(scratch, src);
2954 __ St_d(scratch, g.ToMemOperand(destination));
2955 }
2956 } else if (source->IsConstant()) {
2957 Constant src = g.ToConstant(source);
2958 if (destination->IsRegister() || destination->IsStackSlot()) {
2959 UseScratchRegisterScope temps(masm());
2960 Register scratch = temps.Acquire();
2961 Register dst =
2962 destination->IsRegister() ? g.ToRegister(destination) : scratch;
2963 switch (src.type()) {
2964 case Constant::kInt32:
2965 __ li(dst, Operand(src.ToInt32(), src.rmode()));
2966 break;
2967 case Constant::kFloat32:
2968 __ li(dst, Operand::EmbeddedNumber(src.ToFloat32()));
2969 break;
2970 case Constant::kInt64:
2971 __ li(dst, Operand(src.ToInt64(), src.rmode()));
2972 break;
2973 case Constant::kFloat64:
2974 __ li(dst, Operand::EmbeddedNumber(src.ToFloat64().value()));
2975 break;
2976 case Constant::kExternalReference:
2977 __ li(dst, src.ToExternalReference());
2978 break;
2979 case Constant::kHeapObject: {
2980 Handle<HeapObject> src_object = src.ToHeapObject();
2981 RootIndex index;
2982 if (IsMaterializableFromRoot(src_object, &index)) {
2983 __ LoadRoot(dst, index);
2984 } else {
2985 __ li(dst, src_object);
2986 }
2987 break;
2988 }
2989 case Constant::kCompressedHeapObject: {
2990 Handle<HeapObject> src_object = src.ToHeapObject();
2991 RootIndex index;
2992 if (IsMaterializableFromRoot(src_object, &index)) {
2993 __ LoadTaggedRoot(dst, index);
2994 } else {
2995 __ li(dst, src_object, RelocInfo::COMPRESSED_EMBEDDED_OBJECT);
2996 }
2997 break;
2998 }
2999 case Constant::kRpoNumber:
3000 UNREACHABLE(); // TODO(titzer): loading RPO numbers on LOONG64.
3001 }
3002 if (destination->IsStackSlot()) __ St_d(dst, g.ToMemOperand(destination));
3003 } else if (src.type() == Constant::kFloat32) {
3004 if (destination->IsFPStackSlot()) {
3005 MemOperand dst = g.ToMemOperand(destination);
3006 if (base::bit_cast<int32_t>(src.ToFloat32()) == 0) {
3007 __ St_d(zero_reg, dst);
3008 } else {
3009 UseScratchRegisterScope temps(masm());
3010 Register scratch = temps.Acquire();
3011 __ li(scratch, Operand(base::bit_cast<int32_t>(src.ToFloat32())));
3012 __ St_d(scratch, dst);
3013 }
3014 } else {
3015 DCHECK(destination->IsFPRegister());
3016 FloatRegister dst = g.ToSingleRegister(destination);
3017 __ Move(dst, src.ToFloat32());
3018 }
3019 } else {
3020 DCHECK_EQ(Constant::kFloat64, src.type());
3021 DoubleRegister dst = destination->IsFPRegister()
3022 ? g.ToDoubleRegister(destination)
3023 : kScratchDoubleReg;
3024 __ Move(dst, src.ToFloat64().value());
3025 if (destination->IsFPStackSlot()) {
3026 __ Fst_d(dst, g.ToMemOperand(destination));
3027 }
3028 }
3029 } else if (source->IsFPRegister()) {
3030 FPURegister src = g.ToDoubleRegister(source);
3031 if (destination->IsFPRegister()) {
3032 FPURegister dst = g.ToDoubleRegister(destination);
3033 __ Move(dst, src);
3034 } else {
3035 DCHECK(destination->IsFPStackSlot());

3037 }
3038 } else if (source->IsFPStackSlot()) {
3039 DCHECK(destination->IsFPRegister() || destination->IsFPStackSlot());

3041 if (destination->IsFPRegister()) {
3042 __ Fld_d(g.ToDoubleRegister(destination), src);
3043 } else {
3044 DCHECK(destination->IsFPStackSlot());

3046 __ Fld_d(temp, src);
3047 __ Fst_d(temp, g.ToMemOperand(destination));
