209int Context::FunctionMapIndex(LanguageMode language_mode, FunctionKind kind,
210 bool has_shared_name) {
211 if (IsClassConstructor(kind)) {
212 // Like the strict function map, but with no 'name' accessor. 'name'
213 // needs to be the last property and it is added during instantiation,
214 // in case a static property with the same name exists"
215 return CLASS_FUNCTION_MAP_INDEX;
216 }
217
218 int base = 0;
219 if (IsGeneratorFunction(kind)) {
220 CHECK_FOLLOWS2(GENERATOR_FUNCTION_MAP_INDEX,
221 GENERATOR_FUNCTION_WITH_NAME_MAP_INDEX);
222 CHECK_FOLLOWS2(ASYNC_GENERATOR_FUNCTION_MAP_INDEX,
223 ASYNC_GENERATOR_FUNCTION_WITH_NAME_MAP_INDEX);
224
225 base = IsAsyncFunction(kind) ? ASYNC_GENERATOR_FUNCTION_MAP_INDEX
226 : GENERATOR_FUNCTION_MAP_INDEX;
227
228 } else if (IsAsyncFunction(kind) || IsModuleWithTopLevelAwait(kind)) {
229 CHECK_FOLLOWS2(ASYNC_FUNCTION_MAP_INDEX,
230 ASYNC_FUNCTION_WITH_NAME_MAP_INDEX);
231
232 base = ASYNC_FUNCTION_MAP_INDEX;
233
234 } else if (IsStrictFunctionWithoutPrototype(kind)) {
235 CHECK_FOLLOWS2(STRICT_FUNCTION_WITHOUT_PROTOTYPE_MAP_INDEX,
236 METHOD_WITH_NAME_MAP_INDEX);
237
238 base = STRICT_FUNCTION_WITHOUT_PROTOTYPE_MAP_INDEX;
239
240 } else {
241 CHECK_FOLLOWS2(SLOPPY_FUNCTION_MAP_INDEX,
242 SLOPPY_FUNCTION_WITH_NAME_MAP_INDEX);
243 CHECK_FOLLOWS2(STRICT_FUNCTION_MAP_INDEX,
244 STRICT_FUNCTION_WITH_NAME_MAP_INDEX);
245
246 base = is_strict(language_mode) ? STRICT_FUNCTION_MAP_INDEX
247 : SLOPPY_FUNCTION_MAP_INDEX;
248 }
249 int offset = static_cast<int>(!has_shared_name);
250 DCHECK_EQ(0, offset & ~1);
251
252 return base + offset;
253}

255#undef CHECK_FOLLOWS2
256#undef CHECK_FOLLOWS4
