19// Minimal SmallVector implementation. Uses inline storage first, switches to
20// dynamic storage when it overflows.

70 SmallVector& operator=(const SmallVector& other) V8_NOEXCEPT {
71 if (this == &other) return *this;
72 size_t other_size = other.size();
73 if (capacity() < other_size) {
74 // Create large-enough heap-allocated storage.
75 if (is_big()) FreeDynamicStorage();
76 begin_ = AllocateDynamicStorage(other_size);
77 end_of_storage_ = begin_ + other_size;
78 }
79 std::copy(other.begin_, other.end_, begin_);
80 end_ = begin_ + other_size;
81 return *this;
82 }

84 SmallVector& operator=(SmallVector&& other) V8_NOEXCEPT {
85 if (this == &other) return *this;
86 if (other.is_big()) {
87 if (is_big()) FreeDynamicStorage();
88 begin_ = other.begin_;
89 end_ = other.end_;
90 end_of_storage_ = other.end_of_storage_;
91 } else {
92 DCHECK_GE(capacity(), other.size()); // Sanity check.
93 size_t other_size = other.size();
94 std::move(other.begin_, other.end_, begin_);
95 end_ = begin_ + other_size;
96 }
97 other.reset_to_inline_storage();
98 return *this;

215 void resize_and_init(size_t new_size, const T& initial_value = {}) {
216 static_assert(std::is_trivially_destructible_v<T>);
217 if (new_size > capacity()) Grow(new_size);
218 T* new_end = begin_ + new_size;
219 if (new_end > end_) {
220 std::uninitialized_fill(end_, new_end, initial_value);
221 }

234 private:
235 // Grows the backing store by a factor of two. Returns the new end of the used
236 // storage (this reduces binary size).
237 V8_NOINLINE V8_PRESERVE_MOST void Grow() { Grow(0); }
238
239 // Grows the backing store by a factor of two, and at least to {min_capacity}.
240 V8_NOINLINE V8_PRESERVE_MOST void Grow(size_t min_capacity) {
241 size_t in_use = end_ - begin_;
242 size_t new_capacity =
243 base::bits::RoundUpToPowerOfTwo(std::max(min_capacity, 2 * capacity()));
244 T* new_storage = AllocateDynamicStorage(new_capacity);
245 if (new_storage == nullptr) {
246 FatalOOM(OOMType::kProcess, "base::SmallVector::Grow");
247 }
248 std::uninitialized_move(begin_, end_, new_storage);
249 if (is_big()) FreeDynamicStorage();
250 begin_ = new_storage;
251 end_ = new_storage + in_use;
252 end_of_storage_ = new_storage + new_capacity;
253 }
254
255 T* AllocateDynamicStorage(size_t number_of_elements) {
256 return allocator_.allocate(number_of_elements);
257 }
258
259 V8_NOINLINE V8_PRESERVE_MOST void FreeDynamicStorage() {
260 DCHECK(is_big());
261 allocator_.deallocate(begin_, end_of_storage_ - begin_);
262 }
263
264 // Clear and go back to inline storage. Dynamic storage is *not* freed. For
265 // internal use only.
266 void reset_to_inline_storage() {
267 begin_ = inline_storage_begin();
268 end_ = begin_;
269 end_of_storage_ = begin_ + kInlineSize;
270 }
271
272 bool is_big() const { return begin_ != inline_storage_begin(); }
273
274 T* inline_storage_begin() { return reinterpret_cast<T*>(&inline_storage_); }
275 const T* inline_storage_begin() const {
276 return reinterpret_cast<const T*>(&inline_storage_);
277 }
278
279 V8_NO_UNIQUE_ADDRESS Allocator allocator_;
280
281 T* begin_ = inline_storage_begin();
282 T* end_ = begin_;
283 T* end_of_storage_ = begin_ + kInlineSize;
284 typename std::aligned_storage<sizeof(T) * kInlineSize, alignof(T)>::type
285 inline_storage_;
