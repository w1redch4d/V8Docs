45// The key in a script cache is a WeakFixedArray containing a weak pointer to
46// the Script. The corresponding value can be either the root SharedFunctionInfo
47// or undefined. The purpose of storing the root SharedFunctionInfo as the value
48// is to keep it alive, not to save a lookup on the Script. A newly added entry
49// always contains the root SharedFunctionInfo. After the root
50// SharedFunctionInfo has aged sufficiently, it is replaced with undefined. In
51// this way, all strong references to large objects are dropped, but there is
52// still a way to get the Script if it happens to still be alive.
53class ScriptCacheKey : public HashTableKey {
54 public:
55 enum Index {
56 kHash,
57 kWeakScript,
58 kEnd,
59 };

76 static std::optional<Tagged<String>> SourceFromObject(Tagged<Object> obj) {
77 DisallowGarbageCollection no_gc;
78 DCHECK(IsWeakFixedArray(obj));
79 Tagged<WeakFixedArray> array = Cast<WeakFixedArray>(obj);
80 DCHECK_EQ(array->length(), kEnd);
81
82 Tagged<MaybeObject> maybe_script = array->get(kWeakScript);
83 if (Tagged<HeapObject> script; maybe_script.GetHeapObjectIfWeak(&script)) {
84 Tagged<PrimitiveHeapObject> source_or_undefined =
85 Cast<Script>(script)->source();
86 // Scripts stored in the script cache should always have a source string.
87 return Cast<String>(source_or_undefined);
88 }
89
90 DCHECK(maybe_script.IsCleared());
91 return {};
92 }
93
94 private:
95 Handle<String> source_;
96 MaybeHandle<Object> name_;
97 int line_offset_;
98 int column_offset_;
99 v8::ScriptOriginOptions origin_options_;
100 MaybeHandle<Object> host_defined_options_;
101 MaybeHandle<FixedArray> wrapped_arguments_;
102 Isolate* isolate_;
103};

110uint32_t CompilationCacheShape::EvalHash(Tagged<String> source,
111 Tagged<SharedFunctionInfo> shared,
112 LanguageMode language_mode,
113 int position) {
114 uint32_t hash = source->EnsureHash();
115 if (shared->HasSourceCode()) {
116 // Instead of using the SharedFunctionInfo pointer in the hash
117 // code computation, we use a combination of the hash of the
118 // script source code and the start position of the calling scope.
119 // We do this to ensure that the cache entries can survive garbage
120 // collection.
121 Tagged<Script> script(Cast<Script>(shared->script()));
122 hash ^= Cast<String>(script->source())->EnsureHash();
123 }
124 static_assert(LanguageModeSize == 2);
125 if (is_strict(language_mode)) hash ^= 0x8000;
126 hash += position;
127 return hash;
128}

130uint32_t CompilationCacheShape::HashForObject(ReadOnlyRoots roots,
131 Tagged<Object> object) {
132 // Eval: The key field contains the hash as a Number.
133 if (IsNumber(object))
134 return static_cast<uint32_t>(Object::NumberValue(object));
135
136 // Code: The key field contains the SFI key.
137 if (IsSharedFunctionInfo(object)) {
138 return Cast<SharedFunctionInfo>(object)->Hash();
139 }
140
141 // Script.
142 if (IsWeakFixedArray(object)) {
143 return static_cast<uint32_t>(Smi::ToInt(
144 Cast<WeakFixedArray>(object)->get(ScriptCacheKey::kHash).ToSmi()));
145 }
146
147 // RegExpData: The key field (and the value field) contains the RegExpData
148 // object.
149 if (IsRegExpDataWrapper(object)) {
150 Tagged<RegExpDataWrapper> re_wrapper = Cast<RegExpDataWrapper>(object);
151 Isolate* isolate = GetIsolateFromWritableObject(re_wrapper);
152 Tagged<RegExpData> data = re_wrapper->data(isolate);
153 return RegExpHash(data->source(), Smi::FromInt(data->flags()));
154 }
155
156 // Eval: See EvalCacheKey::ToHandle for the encoding.
157 Tagged<FixedArray> val = Cast<FixedArray>(object);
158 DCHECK_EQ(val->map(), roots.fixed_cow_array_map());
159 DCHECK_EQ(4, val->length());
160 Tagged<String> source = Cast<String>(val->get(1));
161 int language_unchecked = Smi::ToInt(val->get(2));
162 DCHECK(is_valid_language_mode(language_unchecked));
163 LanguageMode language_mode = static_cast<LanguageMode>(language_unchecked);
164 int position = Smi::ToInt(val->get(3));
165 Tagged<Object> shared = val->get(0);
166 return EvalHash(source, Cast<SharedFunctionInfo>(shared), language_mode,
167 position);
168}
