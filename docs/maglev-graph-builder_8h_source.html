549 // Splits incoming critical edges and labels predecessors.
550 void ProcessMergePointPredecessors(
551 MergePointInterpreterFrameState& merge_state,
552 BasicBlockRef& jump_targets) {
553 // Merges aren't simple fallthroughs, so we should reset state which is
554 // cached directly on the builder instead of on the merge states.
555 ResetBuilderCachedState();
556
557 if (merge_state.is_loop()) {
558 DCHECK_EQ(merge_state.predecessors_so_far(),
559 merge_state.predecessor_count() - 1);
560 } else {
561 DCHECK_EQ(merge_state.predecessors_so_far(),
562 merge_state.predecessor_count());
563 }
564
565 if (merge_state.predecessor_count() == 1) return;
566
567 // Set up edge-split.
568 int predecessor_index = merge_state.predecessor_count() - 1;
569 if (merge_state.is_loop()) {
570 // For loops, the JumpLoop block hasn't been generated yet, and so isn't
571 // in the list of jump targets. IT's the last predecessor, so drop the
572 // index by one.
573 DCHECK(merge_state.is_unmerged_loop());
574 predecessor_index--;
575 }
576 BasicBlockRef* old_jump_targets = jump_targets.Reset();
577 while (old_jump_targets != nullptr) {
578 BasicBlock* predecessor = merge_state.predecessor_at(predecessor_index);
579 CHECK(predecessor);
580 ControlNode* control = predecessor->control_node();
581 if (control->Is<ConditionalControlNode>()) {
582 // CreateEmptyBlock automatically registers itself with the offset.
583 predecessor = CreateEdgeSplitBlock(jump_targets, predecessor);
584 // Set the old predecessor's (the conditional block) reference to
585 // point to the new empty predecessor block.
586 old_jump_targets =
587 old_jump_targets->SetToBlockAndReturnNext(predecessor);
588 merge_state.set_predecessor_at(predecessor_index, predecessor);
589 } else {
590 // Re-register the block in the offset's ref list.
591 old_jump_targets = old_jump_targets->MoveToRefList(&jump_targets);
592 }
593 // We only set the predecessor id after splitting critical edges, to make
594 // sure the edge split blocks pick up the correct predecessor index.
595 predecessor->set_predecessor_id(predecessor_index--);
596 }
597 DCHECK_EQ(predecessor_index, -1);
598 RegisterPhisWithGraphLabeller(merge_state);
599 }

633 void KillPeeledLoopTargets(int peelings) {
634 DCHECK_EQ(iterator_.current_bytecode(), interpreter::Bytecode::kJumpLoop);
635 int target = iterator_.GetJumpTargetOffset();
636 // Since we ended up not peeling we must kill all the doubly accounted
637 // jumps out of the loop.
638 interpreter::BytecodeArrayIterator iterator(bytecode().object());
639 for (iterator.SetOffset(target);
640 iterator.current_offset() < iterator_.current_offset();
641 iterator.Advance()) {
642 interpreter::Bytecode bc = iterator.current_bytecode();
643 DCHECK_NE(bc, interpreter::Bytecode::kJumpLoop);
644 int kill = -1;
645 if (interpreter::Bytecodes::IsJump(bc) &&
646 iterator.GetJumpTargetOffset() > iterator_.current_offset()) {
647 kill = iterator.GetJumpTargetOffset();
648 } else if (is_inline() && interpreter::Bytecodes::Returns(bc)) {
649 kill = inline_exit_offset();
650 }
651 if (kill != -1) {

653 for (int i = 0; i < peelings; ++i) {
654 merge_states_[kill]->MergeDead(*compilation_unit_);
655 }
656 }
657 UpdatePredecessorCount(kill, -peelings);
658 }

662 void MarkBytecodeDead() {
663 DCHECK_NULL(current_block_);
664 if (v8_flags.trace_maglev_graph_building) {
665 std::cout << "== Dead ==\n"
666 << std::setw(4) << iterator_.current_offset() << " : ";
667 interpreter::BytecodeDecoder::Decode(std::cout,
668 iterator_.current_address());
669 std::cout << std::endl;
670 }
671
672 // If the current bytecode is a jump to elsewhere, then this jump is
673 // also dead and we should make sure to merge it as a dead predecessor.
674 interpreter::Bytecode bytecode = iterator_.current_bytecode();
675 if (interpreter::Bytecodes::IsForwardJump(bytecode)) {
676 // Jumps merge into their target, and conditional jumps also merge into
677 // the fallthrough.
678 MergeDeadIntoFrameState(iterator_.GetJumpTargetOffset());
679 if (interpreter::Bytecodes::IsConditionalJump(bytecode)) {
680 MergeDeadIntoFrameState(iterator_.next_offset());
681 }
682 } else if (bytecode == interpreter::Bytecode::kJumpLoop) {
683 // JumpLoop merges into its loop header, which has to be treated
684 // specially by the merge.
685 if (!in_peeled_iteration() || in_optimistic_peeling_iteration()) {
686 MergeDeadLoopIntoFrameState(iterator_.GetJumpTargetOffset());
687 }
688 } else if (interpreter::Bytecodes::IsSwitch(bytecode)) {
689 // Switches merge into their targets, and into the fallthrough.
690 for (auto offset : iterator_.GetJumpTableTargetOffsets()) {
691 MergeDeadIntoFrameState(offset.target_offset);
692 }
693 MergeDeadIntoFrameState(iterator_.next_offset());
694 } else if (!interpreter::Bytecodes::Returns(bytecode) &&
695 !interpreter::Bytecodes::UnconditionallyThrows(bytecode)) {
696 // Any other bytecode that doesn't return or throw will merge into the
697 // fallthrough.
698 MergeDeadIntoFrameState(iterator_.next_offset());
699 } else if (interpreter::Bytecodes::Returns(bytecode) && is_inline()) {
700 MergeDeadIntoFrameState(inline_exit_offset());
701 }
702
703 // TODO(leszeks): We could now continue iterating the bytecode
704 }

725 void VisitSingleBytecode() {
726 if (v8_flags.trace_maglev_graph_building) {
727 std::cout << std::setw(4) << iterator_.current_offset() << " : ";
728 interpreter::BytecodeDecoder::Decode(std::cout,
729 iterator_.current_address());
730 std::cout << std::endl;
731 }
732
733 int offset = iterator_.current_offset();
734 UpdateSourceAndBytecodePosition(offset);
735
736 MergePointInterpreterFrameState* merge_state = merge_states_[offset];
737 if (V8_UNLIKELY(merge_state != nullptr)) {
738 bool preserve_known_node_aspects = in_optimistic_peeling_iteration() &&
739 loop_headers_to_peel_.Contains(offset);
740 if (merge_state->is_resumable_loop()) {
741 current_for_in_state.enum_cache_indices = nullptr;
742 }
743 if (current_block_ != nullptr) {
744 DCHECK(!preserve_known_node_aspects);
745 // TODO(leszeks): Re-evaluate this DCHECK, we might hit it if the only
746 // bytecodes in this basic block were only register juggling.
747 // DCHECK(!current_block_->nodes().is_empty());
748 BasicBlock* predecessor;
749 if (merge_state->is_loop() && !merge_state->is_resumable_loop() &&
750 need_checkpointed_loop_entry()) {
751 predecessor =
752 FinishBlock<CheckpointedJump>({}, &jump_targets_[offset]);
753 } else {
754 predecessor = FinishBlock<Jump>({}, &jump_targets_[offset]);
755 }
756 merge_state->Merge(this, *compilation_unit_, current_interpreter_frame_,
757 predecessor);
758 }
759 if (v8_flags.trace_maglev_graph_building) {
760 auto detail = merge_state->is_exception_handler() ? "exception handler"
761 : merge_state->is_loop() ? "loop header"
762 : "merge";
763 std::cout << "== New block (" << detail << " @" << merge_state
764 << ") at "
765 << compilation_unit()->shared_function_info().object()
766 << "==" << std::endl;
767 PrintVirtualObjects();
768 }
769
770 if (V8_UNLIKELY(merge_state->is_exception_handler())) {
771 CHECK_EQ(predecessor_count(offset), 0);
772 // If we have no reference to this block, then the exception handler is
773 // dead.
774 if (!jump_targets_[offset].has_ref() ||
775 !merge_state->exception_handler_was_used()) {
776 MarkBytecodeDead();
777 return;
778 }
779 ProcessMergePointAtExceptionHandlerStart(offset);
780 } else if (merge_state->is_unmerged_unreachable_loop()) {
781 // We encoutered a loop header that is only reachable by the JumpLoop
782 // back-edge, but the bytecode_analysis didn't notice upfront. This can
783 // e.g. be a loop that is entered on a dead fall-through.
784 static_assert(kLoopsMustBeEnteredThroughHeader);
785 MarkBytecodeDead();
786 return;
787 } else {
788 ProcessMergePoint(offset, preserve_known_node_aspects);
789 }
790
791 if (is_loop_effect_tracking_enabled() && merge_state->is_loop()) {
792 BeginLoopEffects(offset);
793 }
794 // We pass nullptr for the `predecessor` argument of StartNewBlock because
795 // this block is guaranteed to have a merge_state_, and hence to not have
796 // a `predecessor_` field.
797 StartNewBlock(offset, /*predecessor*/ nullptr);
798 } else if (V8_UNLIKELY(current_block_ == nullptr)) {
799 // If we don't have a current block, the bytecode must be dead (because of
800 // some earlier deopt). Mark this bytecode dead too and return.
801 // TODO(leszeks): Merge these two conditions by marking dead states with
802 // a sentinel value.
803 if (predecessor_count(offset) == 1) {
804 CHECK_NULL(merge_state);
805 CHECK(bytecode_analysis().IsLoopHeader(offset));
806 } else {
807 CHECK_EQ(predecessor_count(offset), 0);
808 }
809 MarkBytecodeDead();
810 return;
811 }
812
813 // Handle exceptions if we have a table.
814 if (bytecode().handler_table_size() > 0) {
815 // Pop all entries where offset >= end.
816 while (IsInsideTryBlock()) {
817 HandlerTableEntry& entry = catch_block_stack_.top();
818 if (offset < entry.end) break;
819 catch_block_stack_.pop();
820 }
821 // Push new entries from interpreter handler table where offset >= start
822 // && offset < end.
823 HandlerTable table(*bytecode().object());
824 while (next_handler_table_index_ < table.NumberOfRangeEntries()) {
825 int start = table.GetRangeStart(next_handler_table_index_);
826 if (offset < start) break;
827 int end = table.GetRangeEnd(next_handler_table_index_);
828 if (offset >= end) {
829 next_handler_table_index_++;
830 continue;
831 }
832 int handler = table.GetRangeHandler(next_handler_table_index_);
833 catch_block_stack_.push({end, handler});
834 DCHECK_NOT_NULL(merge_states_[handler]);
835 next_handler_table_index_++;
836 }
837 }
838
839 DCHECK_NOT_NULL(current_block_);
840#ifdef DEBUG
841 // Clear new nodes for the next VisitFoo
842 new_nodes_.clear();
843#endif
844
845 if (iterator_.current_bytecode() == interpreter::Bytecode::kJumpLoop &&
846 iterator_.GetJumpTargetOffset() < entrypoint_) {
847 static_assert(kLoopsMustBeEnteredThroughHeader);
848 CHECK(EmitUnconditionalDeopt(DeoptimizeReason::kOSREarlyExit)
849 .IsDoneWithAbort());
850 MarkBytecodeDead();
851 return;
852 }
853
854 switch (iterator_.current_bytecode()) {
855#define BYTECODE_CASE(name, ...) \
856 case interpreter::Bytecode::k##name: { \
857 if (Visit##name().IsDoneWithAbort()) { \
858 MarkBytecodeDead(); \
859 } \
860 break; \
861 }
862 BYTECODE_LIST(BYTECODE_CASE, BYTECODE_CASE)
863#undef BYTECODE_CASE
864 }
865 }

868 BYTECODE_LIST(BYTECODE_VISITOR, BYTECODE_VISITOR)
869#undef BYTECODE_VISITOR

872 ReduceResult VisitIntrinsic##name(interpreter::RegisterList args);
873 INTRINSICS_LIST(DECLARE_VISITOR)
874#undef DECLARE_VISITOR

907 template <typename NodeT, typename... Args>
908 NodeT* AddNewNodeOrGetEquivalent(std::initializer_list<ValueNode*> raw_inputs,
909 Args&&... args) {
910 DCHECK(v8_flags.maglev_cse);
911 static constexpr Opcode op = Node::opcode_of<NodeT>;
912 static_assert(Node::participate_in_cse(op));
913 using options_result =
914 typename std::invoke_result<decltype(&NodeT::options),
915 const NodeT>::type;
916 static_assert(
917 std::is_assignable<options_result, std::tuple<Args...>>::value,
918 "Instruction participating in CSE needs options() returning "
919 "a tuple matching the constructor arguments");
920 static_assert(IsFixedInputNode<NodeT>());
921 static_assert(NodeT::kInputCount <= 3);
922
923 std::array<ValueNode*, NodeT::kInputCount> inputs;
924 // Nodes with zero input count don't have kInputTypes defined.
925 if constexpr (NodeT::kInputCount > 0) {
926 int i = 0;
927 constexpr UseReprHintRecording hint = ShouldRecordUseReprHint<NodeT>();
928 for (ValueNode* raw_input : raw_inputs) {
929 inputs[i] = ConvertInputTo<hint>(raw_input, NodeT::kInputTypes[i]);
930 i++;
931 }
932 if constexpr (IsCommutativeNode(Node::opcode_of<NodeT>)) {
933 static_assert(NodeT::kInputCount == 2);
934 if (inputs[0] > inputs[1]) {
935 std::swap(inputs[0], inputs[1]);
936 }
937 }
938 }
939
940 uint32_t value_number;
941 {
942 size_t tmp_value_number = base::hash_value(op);
943 (
944 [&] {
945 tmp_value_number =
946 fast_hash_combine(tmp_value_number, gvn_hash_value(args));
947 }(),
948 ...);
949 for (const auto& inp : inputs) {
950 tmp_value_number =
951 fast_hash_combine(tmp_value_number, base::hash_value(inp));
952 }
953 value_number = static_cast<uint32_t>(tmp_value_number);
954 }
955
956 auto exists = known_node_aspects().available_expressions.find(value_number);
957 if (exists != known_node_aspects().available_expressions.end()) {
958 auto candidate = exists->second.node;
959 const bool sanity_check =
960 candidate->Is<NodeT>() &&
961 static_cast<size_t>(candidate->input_count()) == inputs.size();
962 DCHECK_IMPLIES(sanity_check,
963 (StaticPropertiesForOpcode(op) &
964 candidate->properties()) == candidate->properties());
965 const bool epoch_check =
966 !Node::needs_epoch_check(op) ||
967 known_node_aspects().effect_epoch() <= exists->second.effect_epoch;
968 if (sanity_check && epoch_check) {
969 if (static_cast<NodeT*>(candidate)->options() ==
970 std::tuple{std::forward<Args>(args)...}) {
971 int i = 0;
972 for (const auto& inp : inputs) {
973 if (inp != candidate->input(i).node()) {
974 break;
975 }
976 i++;
977 }
978 if (static_cast<size_t>(i) == inputs.size()) {
979 return static_cast<NodeT*>(candidate);
980 }
981 }
982 }
983 if (!epoch_check) {
984 known_node_aspects().available_expressions.erase(exists);
985 }
986 }
987 NodeT* node = NodeBase::New<NodeT>(zone(), inputs.size(),
988 std::forward<Args>(args)...);
989 int i = 0;
990 for (ValueNode* input : inputs) {
991 DCHECK_NOT_NULL(input);
992 node->set_input(i++, input);
993 }
994 DCHECK_EQ(node->options(), std::tuple{std::forward<Args>(args)...});
995 uint32_t epoch = Node::needs_epoch_check(op)
996 ? known_node_aspects().effect_epoch()
997 : KnownNodeAspects::kEffectEpochForPureInstructions;
998 if (epoch != KnownNodeAspects::kEffectEpochOverflow) {
999 known_node_aspects().available_expressions[value_number] = {node, epoch};
1000 }
1001 return AttachExtraInfoAndAddToGraph(node);
1002 }

1697 template <typename NodeT>
1698 void MarkPossibleSideEffect(NodeT* node) {
1699 if constexpr (NodeT::kProperties.can_read() ||
1700 NodeT::kProperties.can_deopt() ||
1701 NodeT::kProperties.can_throw()) {
1702 unobserved_context_slot_stores_.clear();
1703 }
1704
1705 // Don't do anything for nodes without side effects.
1706 if constexpr (!NodeT::kProperties.can_write()) return;
1707
1708 if (v8_flags.maglev_cse) {
1709 known_node_aspects().increment_effect_epoch();
1710 }
1711
1712 // We only need to clear unstable node aspects on the current builder, not
1713 // the parent, since we'll anyway copy the known_node_aspects to the parent
1714 // once we finish the inlined function.
1715
1716 if constexpr (IsElementsArrayWrite(Node::opcode_of<NodeT>)) {
1717 node->ClearElementsProperties(known_node_aspects());
1718 if (is_loop_effect_tracking()) {
1719 loop_effects_->keys_cleared.insert(
1720 KnownNodeAspects::LoadedPropertyMapKey::Elements());
1721 }
1722 } else if constexpr (!IsSimpleFieldStore(Node::opcode_of<NodeT>) &&
1723 !IsTypedArrayStore(Node::opcode_of<NodeT>)) {
1724 // Don't change known node aspects for simple field stores. The only
1725 // relevant side effect on these is writes to objects which invalidate
1726 // loaded properties and context slots, and we invalidate these already as
1727 // part of emitting the store.
1728 node->ClearUnstableNodeAspects(known_node_aspects());
1729 if (is_loop_effect_tracking()) {
1730 loop_effects_->unstable_aspects_cleared = true;
1731 }
1732 }
1733
1734 // Simple field stores can't possibly change or migrate the map.
1735 static constexpr bool is_possible_map_change =
1736 !IsSimpleFieldStore(Node::opcode_of<NodeT>);
1737
1738 // All user-observable side effects need to clear state that is cached on
1739 // the builder. This reset has to be propagated up through the parents.
1740 // TODO(leszeks): What side effects aren't observable? Maybe migrations?
1741 for (MaglevGraphBuilder* builder = this; builder != nullptr;
1742 builder = builder->parent_) {
1743 builder->ResetBuilderCachedState<is_possible_map_change>();
1744 }
1745 }

1839 template <typename ControlNodeT, typename... Args>
1840 BasicBlock* FinishBlock(std::initializer_list<ValueNode*> control_inputs,
1841 Args&&... args) {
1842 ControlNodeT* control_node = NodeBase::New<ControlNodeT>(
1843 zone(), control_inputs.size(), std::forward<Args>(args)...);
1844 SetNodeInputs(control_node, control_inputs);
1845 AttachEagerDeoptInfo(control_node);
1846 AttachDeoptCheckpoint(control_node);
1847 static_assert(!ControlNodeT::kProperties.can_lazy_deopt());
1848 static_assert(!ControlNodeT::kProperties.can_throw());
1849 static_assert(!ControlNodeT::kProperties.can_write());
1850 control_node->set_owner(current_block_);
1851 current_block_->set_control_node(control_node);
1852 // Clear unobserved context slot stores when there is any controlflow.
1853 // TODO(olivf): More precision could be achieved by tracking dominating
1854 // stores within known_node_aspects. For this we could use a stack of
1855 // stores, which we push on split and pop on merge.
1856 unobserved_context_slot_stores_.clear();
1857
1858 BasicBlock* block = current_block_;
1859 current_block_ = nullptr;
1860
1861 graph()->Add(block);
1862 if (has_graph_labeller()) {
1863 graph_labeller()->RegisterNode(control_node, compilation_unit_,
1864 BytecodeOffset(iterator_.current_offset()),
1865 current_source_position_);
1866 graph_labeller()->RegisterBasicBlock(block);
1867 if (v8_flags.trace_maglev_graph_building) {
1868 bool kSkipTargets = true;
1869 std::cout << " " << control_node << " "
1870 << PrintNodeLabel(graph_labeller(), control_node) << ": "
1871 << PrintNode(graph_labeller(), control_node, kSkipTargets)

2767 void CalculatePredecessorCounts() {
2768 // Add 1 after the end of the bytecode so we can always write to the offset
2769 // after the last bytecode.
2770 uint32_t array_length = bytecode().length() + 1;
2771 predecessor_count_ = zone()->AllocateArray<uint32_t>(array_length);
2772 MemsetUint32(predecessor_count_, 0, entrypoint_);
2773 MemsetUint32(predecessor_count_ + entrypoint_, 1,
2774 array_length - entrypoint_);
2775
2776 const int max_peelings = v8_flags.maglev_optimistic_peeled_loops ? 2 : 1;
2777 // We count jumps from peeled loops to outside of the loop twice.
2778 bool is_loop_peeling_iteration = false;
2779 std::optional<int> peeled_loop_end;
2780 interpreter::BytecodeArrayIterator iterator(bytecode().object());
2781 for (iterator.AdvanceTo(entrypoint_); !iterator.done();
2782 iterator.Advance()) {
2783 interpreter::Bytecode bytecode = iterator.current_bytecode();
2784 if (allow_loop_peeling_ &&
2785 bytecode_analysis().IsLoopHeader(iterator.current_offset())) {
2786 const compiler::LoopInfo& loop_info =
2787 bytecode_analysis().GetLoopInfoFor(iterator.current_offset());
2788 // Generators use irreducible control flow, which makes loop peeling too
2789 // complicated.
2790 int size = loop_info.loop_end() - loop_info.loop_start();
2791 if (loop_info.innermost() && !loop_info.resumable() &&
2792 iterator.next_offset() < loop_info.loop_end() &&
2793 size < v8_flags.maglev_loop_peeling_max_size &&
2794 size + graph_->total_peeled_bytecode_size() <
2795 v8_flags.maglev_loop_peeling_max_size_cumulative) {
2796 DCHECK(!is_loop_peeling_iteration);
2797 graph_->add_peeled_bytecode_size(size);
2798 is_loop_peeling_iteration = true;
2799 loop_headers_to_peel_.Add(iterator.current_offset());
2800 peeled_loop_end = bytecode_analysis().GetLoopEndOffsetForInnermost(
2801 iterator.current_offset());
2802 }
2803 }
2804 if (interpreter::Bytecodes::IsJump(bytecode)) {
2805 if (is_loop_peeling_iteration &&
2806 bytecode == interpreter::Bytecode::kJumpLoop) {
2807 DCHECK_EQ(iterator.next_offset(), peeled_loop_end);
2808 is_loop_peeling_iteration = false;
2809 peeled_loop_end = {};
2810 }
2811 if (iterator.GetJumpTargetOffset() < entrypoint_) {
2812 static_assert(kLoopsMustBeEnteredThroughHeader);
2813 if (predecessor_count(iterator.GetJumpTargetOffset()) == 1) {
2814 // We encoutered a JumpLoop whose loop header is not reachable
2815 // otherwise. This loop is either dead or the JumpLoop will bail
2816 // with DeoptimizeReason::kOSREarlyExit.
2817 InitializePredecessorCount(iterator.GetJumpTargetOffset(), 0);
2818 }
2819 } else {
2820 UpdatePredecessorCount(iterator.GetJumpTargetOffset(), 1);
2821 }
2822 if (is_loop_peeling_iteration &&
2823 iterator.GetJumpTargetOffset() >= *peeled_loop_end) {
2824 // Jumps from within the peeled loop to outside need to be counted
2825 // twice, once for the peeled and once for the regular loop body.
2826 UpdatePredecessorCount(iterator.GetJumpTargetOffset(), max_peelings);
2827 }
2828 if (!interpreter::Bytecodes::IsConditionalJump(bytecode)) {
2829 UpdatePredecessorCount(iterator.next_offset(), -1);
2830 }
2831 } else if (interpreter::Bytecodes::IsSwitch(bytecode)) {
2832 for (auto offset : iterator.GetJumpTableTargetOffsets()) {
2833 UpdatePredecessorCount(offset.target_offset, 1);
2834 }
2835 } else if (interpreter::Bytecodes::Returns(bytecode) ||
2836 interpreter::Bytecodes::UnconditionallyThrows(bytecode)) {
2837 UpdatePredecessorCount(iterator.next_offset(), -1);
2838 // Collect inline return jumps in the slot after the last bytecode.
2839 if (is_inline() && interpreter::Bytecodes::Returns(bytecode)) {
2840 UpdatePredecessorCount(array_length - 1, 1);
2841 if (is_loop_peeling_iteration) {
2842 UpdatePredecessorCount(array_length - 1, max_peelings);
2843 }
2844 }
2845 }
2846 // TODO(leszeks): Also consider handler entries (the bytecode analysis)
2847 // will do this automatically I guess if we merge this into that.
2848 }
2849 if (!is_inline()) {
2850 DCHECK_EQ(0, predecessor_count(bytecode().length()));
2851 }
2852 }
