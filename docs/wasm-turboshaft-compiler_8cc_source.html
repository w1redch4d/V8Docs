21wasm::WasmCompilationResult ExecuteTurboshaftWasmCompilation(
22 wasm::CompilationEnv* env, compiler::WasmCompilationData& data,
23 wasm::WasmDetectedFeatures* detected) {
24 // TODO(nicohartmann): We should not allocate TurboFan graph(s) here but
25 // instead use only Turboshaft inside `GenerateWasmCodeFromTurboshaftGraph`.
26 Zone zone(wasm::GetWasmEngine()->allocator(), ZONE_NAME, kCompressGraphZone);
27 compiler::MachineGraph* mcgraph = zone.New<compiler::MachineGraph>(
28 zone.New<compiler::Graph>(&zone), zone.New<CommonOperatorBuilder>(&zone),
29 zone.New<MachineOperatorBuilder>(
30 &zone, MachineType::PointerRepresentation(),
31 InstructionSelector::SupportedMachineOperatorFlags(),
32 InstructionSelector::AlignmentRequirements()));
33
34 OptimizedCompilationInfo info(
35 GetDebugName(&zone, env->module, data.wire_bytes_storage,
36 data.func_index),
37 &zone, CodeKind::WASM_FUNCTION);
38
39 if (info.trace_turbo_json()) {
40 TurboCfgFile tcf;
41 tcf << AsC1VCompilation(&info);
42 }
43
44 if (info.trace_turbo_json()) {
45 data.node_origins = zone.New<NodeOriginTable>(mcgraph->graph());
46 }
47
48 data.source_positions =
49 mcgraph->zone()->New<SourcePositionTable>(mcgraph->graph());
50 data.assumptions = new wasm::AssumptionsJournal();
51 auto call_descriptor = GetWasmCallDescriptor(&zone, data.func_body.sig);
52
53 if (!Pipeline::GenerateWasmCodeFromTurboshaftGraph(
54 &info, env, data, mcgraph, detected, call_descriptor)) {
55 delete data.assumptions;
56 return {};
57 }
58 auto result = info.ReleaseWasmCompilationResult();
59 CHECK_NOT_NULL(result); // Compilation expected to succeed.
60 DCHECK_EQ(wasm::ExecutionTier::kTurbofan, result->result_tier);
61 result->assumptions.reset(data.assumptions);
62 return std::move(*result);
63}
