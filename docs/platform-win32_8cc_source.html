155 // Initialize timezone information. The timezone information is obtained from
156 // windows. If we cannot get the timezone information we fall back to CET.
157 void InitializeIfNeeded() {
158 // Just return if timezone information has already been initialized.
159 if (initialized_) return;
160
161 // Initialize POSIX time zone data.
162 _tzset();
163 // Obtain timezone information from operating system.
164 memset(&tzinfo_, 0, sizeof(tzinfo_));
165 if (GetTimeZoneInformation(&tzinfo_) == TIME_ZONE_ID_INVALID) {
166 // If we cannot get timezone information we fall back to CET.
167 tzinfo_.Bias = -60;
168 tzinfo_.StandardDate.wMonth = 10;
169 tzinfo_.StandardDate.wDay = 5;
170 tzinfo_.StandardDate.wHour = 3;
171 tzinfo_.StandardBias = 0;
172 tzinfo_.DaylightDate.wMonth = 3;
173 tzinfo_.DaylightDate.wDay = 5;
174 tzinfo_.DaylightDate.wHour = 2;
175 tzinfo_.DaylightBias = -60;
176 }
177
178 // Make standard and DST timezone names.
179 WideCharToMultiByte(CP_UTF8, 0, tzinfo_.StandardName, -1, std_tz_name_,
180 kTzNameSize, nullptr, nullptr);
181 std_tz_name_[kTzNameSize - 1] = '\0';
182 WideCharToMultiByte(CP_UTF8, 0, tzinfo_.DaylightName, -1, dst_tz_name_,
183 kTzNameSize, nullptr, nullptr);
184 dst_tz_name_[kTzNameSize - 1] = '\0';
185
186 // If OS returned empty string or resource id (like "@tzres.dll,-211")
187 // simply guess the name from the UTC bias of the timezone.
188 // To properly resolve the resource identifier requires a library load,
189 // which is not possible in a sandbox.
190 if (std_tz_name_[0] == '\0' || std_tz_name_[0] == '@') {
191 OS::SNPrintF(std_tz_name_, kTzNameSize - 1,
192 "%s Standard Time",
193 GuessTimezoneNameFromBias(tzinfo_.Bias));
194 }
195 if (dst_tz_name_[0] == '\0' || dst_tz_name_[0] == '@') {
196 OS::SNPrintF(dst_tz_name_, kTzNameSize - 1,
197 "%s Daylight Time",
198 GuessTimezoneNameFromBias(tzinfo_.Bias));
199 }
200 // Timezone information initialized.
201 initialized_ = true;
202 }
203
204 // Guess the name of the timezone from the bias.
205 // The guess is very biased towards the northern hemisphere.
206 const char* GuessTimezoneNameFromBias(int bias) {
207 static const int kHour = 60;
208 switch (-bias) {
209 case -9*kHour: return "Alaska";
210 case -8*kHour: return "Pacific";
211 case -7*kHour: return "Mountain";
212 case -6*kHour: return "Central";
213 case -5*kHour: return "Eastern";
214 case -4*kHour: return "Atlantic";
215 case 0*kHour: return "GMT";
216 case +1*kHour: return "Central Europe";
217 case +2*kHour: return "Eastern Europe";
218 case +3*kHour: return "Russia";
219 case +5*kHour + 30: return "India";
220 case +8*kHour: return "China";
221 case +9*kHour: return "Japan";
222 case +12*kHour: return "New Zealand";
223 default: return "Local";
224 }
225 }

335// Set timestamp to current time.
336void Win32Time::SetToCurrentTime() {
337 // The default GetSystemTimeAsFileTime has a ~15.5ms resolution.
338 // Because we're fast, we like fast timers which have at least a
339 // 1ms resolution.
340 //
341 // timeGetTime() provides 1ms granularity when combined with
342 // timeBeginPeriod(). If the host application for v8 wants fast
343 // timers, it can use timeBeginPeriod to increase the resolution.
344 //
345 // Using timeGetTime() has a drawback because it is a 32bit value
346 // and hence rolls-over every ~49days.
347 //
348 // To use the clock, we use GetSystemTimeAsFileTime as our base;
349 // and then use timeGetTime to extrapolate current time from the
350 // start time. To deal with rollovers, we resync the clock
351 // any time when more than kMaxClockElapsedTime has passed or
352 // whenever timeGetTime creates a rollover.
353
354 static bool initialized = false;
355 static TimeStamp init_time;
356 static DWORD init_ticks;
357 static const int64_t kHundredNanosecondsPerSecond = 10000000;
358 static const int64_t kMaxClockElapsedTime =
359 60*kHundredNanosecondsPerSecond; // 1 minute
360
361 // If we are uninitialized, we need to resync the clock.
362 bool needs_resync = !initialized;
363
364 // Get the current time.
365 TimeStamp time_now;
366 GetSystemTimeAsFileTime(&time_now.ft_);
367 DWORD ticks_now = timeGetTime();
368
369 // Check if we need to resync due to clock rollover.
370 needs_resync |= ticks_now < init_ticks;
371
372 // Check if we need to resync due to elapsed time.
373 needs_resync |= (time_now.t_ - init_time.t_) > kMaxClockElapsedTime;
374
375 // Check if we need to resync due to backwards time change.
376 needs_resync |= time_now.t_ < init_time.t_;
377
378 // Resync the clock if necessary.
379 if (needs_resync) {
380 GetSystemTimeAsFileTime(&init_time.ft_);
381 init_ticks = ticks_now = timeGetTime();
382 initialized = true;
383 }
384
385 // Finally, compute the actual time. Why is this so hard.
386 DWORD elapsed = ticks_now - init_ticks;
387 this->time_.t_ = init_time.t_ + (static_cast<int64_t>(elapsed) * 10000);
388}

391// Return the local timezone offset in milliseconds east of UTC. This
392// takes into account whether daylight saving is in effect at the time.
393// Only times in the 32-bit Unix range may be passed to this function.
394// Also, adding the time-zone offset to the input must not overflow.
395// The function EquivalentTime() in date.js guarantees this.
396int64_t Win32Time::LocalOffset(WindowsTimezoneCache* cache) {
397 cache->InitializeIfNeeded();
398
399 Win32Time rounded_to_second(*this);
400 rounded_to_second.t() =
401 rounded_to_second.t() / 1000 / kTimeScaler * 1000 * kTimeScaler;
402 // Convert to local time using POSIX localtime function.
403 // Windows XP Service Pack 3 made SystemTimeToTzSpecificLocalTime()
404 // very slow. Other browsers use localtime().
405
406 // Convert from JavaScript milliseconds past 1/1/1970 0:00:00 to
407 // POSIX seconds past 1/1/1970 0:00:00.
408 double unchecked_posix_time = rounded_to_second.ToJSTime() / 1000;
409 if (unchecked_posix_time > INT_MAX || unchecked_posix_time < 0) {
410 return 0;
411 }
412 // Because _USE_32BIT_TIME_T is defined, time_t is a 32-bit int.
413 time_t posix_time = static_cast<time_t>(unchecked_posix_time);
414
415 // Convert to local time, as struct with fields for day, hour, year, etc.
416 tm posix_local_time_struct;
417 if (localtime_s(&posix_local_time_struct, &posix_time)) return 0;
418
419 if (posix_local_time_struct.tm_isdst > 0) {
420 return (cache->tzinfo_.Bias + cache->tzinfo_.DaylightBias) * -kMsPerMinute;
421 } else if (posix_local_time_struct.tm_isdst == 0) {
422 return (cache->tzinfo_.Bias + cache->tzinfo_.StandardBias) * -kMsPerMinute;
423 } else {
424 return cache->tzinfo_.Bias * -kMsPerMinute;
425 }
426}

429// Return whether or not daylight savings time is in effect at this time.
430bool Win32Time::InDST(WindowsTimezoneCache* cache) {
431 cache->InitializeIfNeeded();
432
433 // Determine if DST is in effect at the specified time.
434 bool in_dst = false;
435 if (cache->tzinfo_.StandardDate.wMonth != 0 ||
436 cache->tzinfo_.DaylightDate.wMonth != 0) {
437 // Get the local timezone offset for the timestamp in milliseconds.
438 int64_t offset = LocalOffset(cache);
439
440 // Compute the offset for DST. The bias parameters in the timezone info
441 // are specified in minutes. These must be converted to milliseconds.
442 int64_t dstofs =
443 -(cache->tzinfo_.Bias + cache->tzinfo_.DaylightBias) * kMsPerMinute;
444
445 // If the local time offset equals the timezone bias plus the daylight
446 // bias then DST is in effect.
447 in_dst = offset == dstofs;
448 }
449
450 return in_dst;
451}

559// ----------------------------------------------------------------------------
560// Win32 console output.
561//
562// If a Win32 application is linked as a console application it has a normal
563// standard output and standard error. In this case normal printf works fine
564// for output. However, if the application is linked as a GUI application,
565// the process doesn't have a console, and therefore (debugging) output is lost.
566// This is the case if we are embedded in a windows program (like a browser).
567// In order to be able to get debug output in this case the the debugging
568// facility using OutputDebugString. This output goes to the active debugger
569// for the process (if any). Else the output can be monitored using DBMON.EXE.
570
571enum OutputMode {
572 UNKNOWN, // Output method has not yet been determined.
573 CONSOLE, // Output is written to stdout.
574 ODS // Output is written to debug facility.
575};
576
577static OutputMode output_mode = UNKNOWN; // Current output mode.
578

580// Determine if the process has a console for output.
581static bool HasConsole() {
582 // Only check the first time. Eventual race conditions are not a problem,
583 // because all threads will eventually determine the same mode.
584 if (output_mode == UNKNOWN) {
585 // We cannot just check that the standard output is attached to a console
586 // because this would fail if output is redirected to a file. Therefore we
587 // say that a process does not have an output console if either the
588 // standard output handle is invalid or its file type is unknown.
589 if (GetStdHandle(STD_OUTPUT_HANDLE) != INVALID_HANDLE_VALUE &&
590 GetFileType(GetStdHandle(STD_OUTPUT_HANDLE)) != FILE_TYPE_UNKNOWN)
591 output_mode = CONSOLE;
592 else
593 output_mode = ODS;
594 }
595 return output_mode == CONSOLE;
596}

857// Try not to map pages into the default range that windows loads DLLs
858// Use a multiple of 64k to prevent committing unused memory.

919 DWORD protect) {

928uint8_t* VirtualAllocWithHint(size_t size, DWORD flags, DWORD protect,
929 void* hint) {
930 LPVOID base = VirtualAllocWrapper(hint, size, flags, protect);
931
932 // On failure, let the OS find an address to use.
933 if (hint && base == nullptr) {
934 base = VirtualAllocWrapper(nullptr, size, flags, protect);
935 }
936
937 return reinterpret_cast<uint8_t*>(base);
938}
939
940void* AllocateInternal(void* hint, size_t size, size_t alignment,
941 size_t page_size, DWORD flags, DWORD protect) {
942 // First, try an exact size aligned allocation.
943 uint8_t* base = VirtualAllocWithHint(size, flags, protect, hint);
944 if (base == nullptr) return nullptr; // Can't allocate, we're OOM.
945
946 // If address is suitably aligned, we're done.
947 uint8_t* aligned_base = reinterpret_cast<uint8_t*>(
948 RoundUp(reinterpret_cast<uintptr_t>(base), alignment));
949 if (base == aligned_base) return reinterpret_cast<void*>(base);
950
951 // Otherwise, free it and try a larger allocation.
952 CHECK(VirtualFree(base, 0, MEM_RELEASE));
953
954 // Clear the hint. It's unlikely we can allocate at this address.
955 hint = nullptr;
956
957 // Add the maximum misalignment so we are guaranteed an aligned base address
958 // in the allocated region.
959 size_t padded_size = size + (alignment - page_size);
960 const int kMaxAttempts = 3;
961 aligned_base = nullptr;
962 for (int i = 0; i < kMaxAttempts; ++i) {
963 base = VirtualAllocWithHint(padded_size, flags, protect, hint);
964 if (base == nullptr) return nullptr; // Can't allocate, we're OOM.
965
966 // Try to trim the allocation by freeing the padded allocation and then
967 // calling VirtualAlloc at the aligned base.
968 CHECK(VirtualFree(base, 0, MEM_RELEASE));
969 aligned_base = reinterpret_cast<uint8_t*>(
970 RoundUp(reinterpret_cast<uintptr_t>(base), alignment));
971 base = reinterpret_cast<uint8_t*>(
972 VirtualAllocWrapper(aligned_base, size, flags, protect));
973 // We might not get the reduced allocation due to a race. In that case,
974 // base will be nullptr.
975 if (base != nullptr) break;
976 }
977 DCHECK_IMPLIES(base, base == aligned_base);
978 return reinterpret_cast<void*>(base);
979}

1085 // not guaranteed to be zeroed when returned to the application.

1598static std::vector<OS::SharedLibraryAddress> LoadSymbols(
1599 HANDLE process_handle) {
1600 static std::vector<OS::SharedLibraryAddress> result;
1601
1602 static bool symbols_loaded = false;
1603
1604 if (symbols_loaded) return result;
1605
1606 BOOL ok;
1607
1608 // Initialize the symbol engine.
1609 ok = _SymInitialize(process_handle, // hProcess
1610 nullptr, // UserSearchPath
1611 false); // fInvadeProcess
1612 if (!ok) return result;
1613
1614 DWORD options = _SymGetOptions();
1615 options |= SYMOPT_LOAD_LINES;
1616 options |= SYMOPT_FAIL_CRITICAL_ERRORS;
1617 options = _SymSetOptions(options);
1618
1619 char buf[OS::kStackWalkMaxNameLen] = {0};
1620 ok = _SymGetSearchPath(process_handle, buf, OS::kStackWalkMaxNameLen);
1621 if (!ok) {
1622 int err = GetLastError();
1623 OS::Print("%d\n", err);
1624 return result;
1625 }
1626
1627 HANDLE snapshot = _CreateToolhelp32Snapshot(
1628 TH32CS_SNAPMODULE, // dwFlags
1629 GetCurrentProcessId()); // th32ProcessId
1630 if (snapshot == INVALID_HANDLE_VALUE) return result;
1631 MODULEENTRY32W module_entry;
1632 module_entry.dwSize = sizeof(module_entry); // Set the size of the structure.
1633 BOOL cont = _Module32FirstW(snapshot, &module_entry);
1634 while (cont) {
1635 DWORD64 base;
1636 // NOTE the SymLoadModule64 function has the peculiarity of accepting a
1637 // both unicode and ASCII strings even though the parameter is PSTR.
1638 base = _SymLoadModule64(
1639 process_handle, // hProcess
1640 0, // hFile
1641 reinterpret_cast<PSTR>(module_entry.szExePath), // ImageName
1642 reinterpret_cast<PSTR>(module_entry.szModule), // ModuleName
1643 reinterpret_cast<DWORD64>(module_entry.modBaseAddr), // BaseOfDll
1644 module_entry.modBaseSize); // SizeOfDll
1645 if (base == 0) {
1646 int err = GetLastError();
1647 if (err != ERROR_MOD_NOT_FOUND &&
1648 err != ERROR_INVALID_HANDLE) {
1649 result.clear();
1650 return result;
1651 }
1652 }
1653 int lib_name_length = WideCharToMultiByte(
1654 CP_UTF8, 0, module_entry.szExePath, -1, nullptr, 0, nullptr, nullptr);
1655 std::string lib_name(lib_name_length, 0);
1656 WideCharToMultiByte(CP_UTF8, 0, module_entry.szExePath, -1, &lib_name[0],
1657 lib_name_length, nullptr, nullptr);

1659 lib_name, reinterpret_cast<uintptr_t>(module_entry.modBaseAddr),
1660 reinterpret_cast<uintptr_t>(module_entry.modBaseAddr +
1661 module_entry.modBaseSize)));
1662 cont = _Module32NextW(snapshot, &module_entry);
1663 }
1664 CloseHandle(snapshot);
1665
1666 symbols_loaded = true;
1667 return result;
1668}

1815std::optional<OS::MemoryRange> OS::GetFirstFreeMemoryRangeWithin(
1816 OS::Address boundary_start, OS::Address boundary_end, size_t minimum_size,
1817 size_t alignment) {
1818 // Search for the virtual memory (vm) ranges within the boundary.
1819 // If a range is free and larger than {minimum_size}, then push it to the
1820 // returned vector.
1821 uintptr_t vm_start = RoundUp(boundary_start, alignment);
1822 uintptr_t vm_end = 0;
1823 MEMORY_BASIC_INFORMATION mi;
1824 // This loop will terminate once the scanning reaches the higher address
1825 // to the end of boundary or the function VirtualQuery fails.
1826 while (vm_start < boundary_end &&
1827 VirtualQuery(reinterpret_cast<LPCVOID>(vm_start), &mi, sizeof(mi)) !=
1828 0) {
1829 vm_start = reinterpret_cast<uintptr_t>(mi.BaseAddress);
1830 vm_end = vm_start + mi.RegionSize;
1831 if (mi.State == MEM_FREE) {
1832 // The available area is the overlap of the virtual memory range and
1833 // boundary. Push the overlapped memory range to the vector if there is
1834 // enough space.
1835 const uintptr_t overlap_start =
1836 RoundUp(std::max(vm_start, boundary_start), alignment);
1837 const uintptr_t overlap_end =
1838 RoundDown(std::min(vm_end, boundary_end), alignment);
1839 if (overlap_start < overlap_end &&
1840 overlap_end - overlap_start >= minimum_size) {
1841 return OS::MemoryRange{overlap_start, overlap_end};
1842 }
1843 }
1844 // Continue to visit the next virtual memory range.
1845 vm_start = vm_end;
1846 }
1847
1848 return {};
1849}
