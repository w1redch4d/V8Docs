18// We use the full 16 bits of the instance_type field to encode heap object
19// instance types. All the high-order bits (bits 7-15) are cleared if the object
20// is a string, and contain set bits if it is not a string.
21const uint32_t kIsNotStringMask = ~((1 << 7) - 1);
22const uint32_t kStringTag = 0x0;
23
24// For strings, bits 0-2 indicate the representation of the string. In
25// particular, bit 0 indicates whether the string is direct or indirect.
26const uint32_t kStringRepresentationMask = (1 << 3) - 1;
27enum StringRepresentationTag {
28 kSeqStringTag = 0x0,
29 kConsStringTag = 0x1,
30 kExternalStringTag = 0x2,
31 kSlicedStringTag = 0x3,
32 kThinStringTag = 0x5,
33};
34const uint32_t kIsIndirectStringMask = 1 << 0;
35const uint32_t kIsIndirectStringTag = 1 << 0;
36static_assert((kSeqStringTag & kIsIndirectStringMask) == 0);
37static_assert((kExternalStringTag & kIsIndirectStringMask) == 0);
38static_assert((kConsStringTag & kIsIndirectStringMask) == kIsIndirectStringTag);
39static_assert((kSlicedStringTag & kIsIndirectStringMask) ==
40 kIsIndirectStringTag);
41static_assert((kThinStringTag & kIsIndirectStringMask) == kIsIndirectStringTag);
42
43// For strings, bit 3 indicates whether the string consists of two-byte
44// characters or one-byte characters.
45const uint32_t kStringEncodingMask = 1 << 3;
46const uint32_t kTwoByteStringTag = 0;
47const uint32_t kOneByteStringTag = 1 << 3;
48
49// Combined tags for convenience (add more if needed).
50constexpr uint32_t kStringRepresentationAndEncodingMask =
51 kStringRepresentationMask | kStringEncodingMask;
52constexpr uint32_t kSeqOneByteStringTag = kSeqStringTag | kOneByteStringTag;
53constexpr uint32_t kSeqTwoByteStringTag = kSeqStringTag | kTwoByteStringTag;
54constexpr uint32_t kExternalOneByteStringTag =
55 kExternalStringTag | kOneByteStringTag;
56constexpr uint32_t kExternalTwoByteStringTag =
57 kExternalStringTag | kTwoByteStringTag;
58
59// For strings, bit 4 indicates whether the data pointer of an external string
60// is cached. Note that the string representation is expected to be
61// kExternalStringTag.
62const uint32_t kUncachedExternalStringMask = 1 << 4;
63const uint32_t kUncachedExternalStringTag = 1 << 4;
64
65// For strings, bit 5 indicates that the string is internalized (if not set) or
66// isn't (if set).
67const uint32_t kIsNotInternalizedMask = 1 << 5;
68const uint32_t kNotInternalizedTag = 1 << 5;
69const uint32_t kInternalizedTag = 0;
70
71// For strings, bit 6 indicates that the string is accessible by more than one
72// thread. Note that a string that is allocated in the shared heap is not
73// accessible by more than one thread until it is explicitly shared (e.g. by
74// postMessage).
75//
76// Runtime code that shares strings with other threads directly need to manually
77// set this bit.
78//
79// TODO(v8:12007): External strings cannot be shared yet.
80//
81// TODO(v8:12007): This bit is currently ignored on internalized strings, which
82// are either always shared or always not shared depending on
83// v8_flags.shared_string_table. This will be hardcoded once
84// v8_flags.shared_string_table is removed.
85const uint32_t kSharedStringMask = 1 << 6;
86const uint32_t kSharedStringTag = 1 << 6;
87
88constexpr uint32_t kStringRepresentationEncodingAndSharedMask =
89 kStringRepresentationAndEncodingMask | kSharedStringMask;
90
91// A ConsString with an empty string as the right side is a candidate
92// for being shortcut by the garbage collector. We don't allocate any
93// non-flat internalized strings, so we do not shortcut them thereby
94// avoiding turning internalized strings into strings. The bit-masks
95// below contain the internalized bit as additional safety.
96// See heap.cc, mark-compact.cc and heap-visitor.cc.
97const uint32_t kShortcutTypeMask =
98 kIsNotStringMask | kIsNotInternalizedMask | kStringRepresentationMask;
99const uint32_t kShortcutTypeTag = kConsStringTag | kNotInternalizedTag;
100
101static inline bool IsShortcutCandidate(int type) {
102 return ((type & kShortcutTypeMask) == kShortcutTypeTag);
103}

156// Most instance types are defined in Torque, with the exception of the string
157// types above. They are ordered by inheritance hierarchy so that we can easily
158// use range checks to determine whether an object is an instance of a subclass
159// of any type. There are a few more constraints specified in the Torque type
160// definitions:
161// - Some instance types are exposed in v8.h, so they are locked to specific
162// values to not unnecessarily change the ABI.
163// - JSSpecialObject and JSCustomElementsObject are aligned with the beginning
164// of the JSObject range, so that we can use a larger range check from
165// FIRST_JS_RECEIVER_TYPE to the end of those ranges and include JSProxy too.
166#define MAKE_TORQUE_INSTANCE_TYPE(TYPE, value) TYPE = value,
167 TORQUE_ASSIGNED_INSTANCE_TYPES(MAKE_TORQUE_INSTANCE_TYPE)
168#undef MAKE_TORQUE_INSTANCE_TYPE
169
170 // Pseudo-types
171 FIRST_UNIQUE_NAME_TYPE = INTERNALIZED_TWO_BYTE_STRING_TYPE,
172 LAST_UNIQUE_NAME_TYPE = SYMBOL_TYPE,
173 FIRST_NONSTRING_TYPE = SYMBOL_TYPE,
174 // Callable JS Functions are all JS Functions except class constructors.
175 FIRST_CALLABLE_JS_FUNCTION_TYPE = FIRST_JS_FUNCTION_TYPE,
176 LAST_CALLABLE_JS_FUNCTION_TYPE = JS_CLASS_CONSTRUCTOR_TYPE - 1,
177 // Boundary for testing JSReceivers that need special property lookup handling
178 LAST_SPECIAL_RECEIVER_TYPE = LAST_JS_SPECIAL_OBJECT_TYPE,
179 // Boundary case for testing JSReceivers that may have elements while having
180 // an empty fixed array as elements backing store. This is true for string
181 // wrappers.
182 LAST_CUSTOM_ELEMENTS_RECEIVER = LAST_JS_CUSTOM_ELEMENTS_OBJECT_TYPE,
183
184 // Convenient names for things where the generated name is awkward:
185 FIRST_TYPE = FIRST_HEAP_OBJECT_TYPE,
186 LAST_TYPE = LAST_HEAP_OBJECT_TYPE,
187 BIGINT_TYPE = BIG_INT_BASE_TYPE,
188
189 // TODO(ishell): define a dedicated instance type for DependentCode to
190 // simplify CodeSerializer.
191 DEPENDENT_CODE_TYPE = WEAK_ARRAY_LIST_TYPE,
192};

225// Two ranges don't cleanly follow the inheritance hierarchy. Here we ensure
226// that only expected types fall within these ranges.
227// - From FIRST_JS_RECEIVER_TYPE to LAST_SPECIAL_RECEIVER_TYPE should correspond
228// to the union type JSProxy | JSSpecialObject.
229// - From FIRST_JS_RECEIVER_TYPE to LAST_CUSTOM_ELEMENTS_RECEIVER should
230// correspond to the union type JSProxy | JSCustomElementsObject.
231// Note in particular that these ranges include all subclasses of JSReceiver
232// that are not also subclasses of JSObject (currently only JSProxy).
233// clang-format off
234#define CHECK_INSTANCE_TYPE(TYPE) \
235 static_assert((TYPE >= FIRST_JS_RECEIVER_TYPE && \
236 TYPE <= LAST_SPECIAL_RECEIVER_TYPE) == \
237 (IF_WASM(EXPAND, TYPE == WASM_STRUCT_TYPE || \
238 TYPE == WASM_ARRAY_TYPE ||) \
239 TYPE == JS_PROXY_TYPE || TYPE == JS_GLOBAL_OBJECT_TYPE || \
240 TYPE == JS_GLOBAL_PROXY_TYPE || \
241 TYPE == JS_MODULE_NAMESPACE_TYPE || \
242 TYPE == JS_SPECIAL_API_OBJECT_TYPE)); \
243 static_assert((TYPE >= FIRST_JS_RECEIVER_TYPE && \
244 TYPE <= LAST_CUSTOM_ELEMENTS_RECEIVER) == \
245 (IF_WASM(EXPAND, TYPE == WASM_STRUCT_TYPE || \
246 TYPE == WASM_ARRAY_TYPE ||) \
247 TYPE == JS_PROXY_TYPE || TYPE == JS_GLOBAL_OBJECT_TYPE || \
248 TYPE == JS_GLOBAL_PROXY_TYPE || \
249 TYPE == JS_MODULE_NAMESPACE_TYPE || \
250 TYPE == JS_SPECIAL_API_OBJECT_TYPE || \
251 TYPE == JS_PRIMITIVE_WRAPPER_TYPE));
252// clang-format on
253TORQUE_ASSIGNED_INSTANCE_TYPE_LIST(CHECK_INSTANCE_TYPE)
254#undef CHECK_INSTANCE_TYPE
