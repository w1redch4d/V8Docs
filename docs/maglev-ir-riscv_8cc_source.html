144void BuiltinStringFromCharCode::GenerateCode(MaglevAssembler* masm,
145 const ProcessingState& state) {
146 MaglevAssembler::TemporaryRegisterScope temps(masm);
147 Register scratch = temps.Acquire();
148 Register result_string = ToRegister(result());
149 if (Int32Constant* constant = code_input().node()->TryCast<Int32Constant>()) {
150 int32_t char_code = constant->value() & 0xFFFF;
151 if (0 <= char_code && char_code < String::kMaxOneByteCharCode) {
152 __ LoadSingleCharacterString(result_string, char_code);
153 } else {
154 __ AllocateTwoByteString(register_snapshot(), result_string, 1);
155 __ Move(scratch, char_code);
156 __ Sh(scratch, FieldMemOperand(result_string,
157 OFFSET_OF_DATA_START(SeqTwoByteString)));
158 }
159 } else {
160 __ StringFromCharCode(register_snapshot(), nullptr, result_string,
161 ToRegister(code_input()), scratch,
162 MaglevAssembler::CharCodeMaskMode::kMustApplyMask);
163 }

270void Int32MultiplyWithOverflow::GenerateCode(MaglevAssembler* masm,
271 const ProcessingState& state) {
272 Register left = ToRegister(left_input());
273 Register right = ToRegister(right_input());
274 Register out = ToRegister(result());
275
276 // TODO(leszeks): peephole optimise multiplication by a constant.
277
278 MaglevAssembler::TemporaryRegisterScope temps(masm);
279 bool out_alias_input = out == left || out == right;
280 Register res = out;
281 if (out_alias_input) {
282 res = temps.Acquire();
283 }
284
285 Register scratch = temps.Acquire();
286 __ MulOverflow32(res, left, Operand(right), scratch);
287
288 static_assert(Int32MultiplyWithOverflow::kProperties.can_eager_deopt());
289 // if res != (res[0:31] sign extended to 64 bits), then the multiplication
290 // result is too large for 32 bits.
291 Label* fail = __ GetDeoptLabel(this, DeoptimizeReason::kOverflow);
292 __ RecordComment("-- Jump to eager deopt");
293 __ MacroAssembler::Branch(fail, ne, scratch, Operand(zero_reg));
294
295 // If the result is zero, check if either lhs or rhs is negative.
296 Label end;
297 __ MacroAssembler::Branch(&end, ne, res, Operand(zero_reg),
298 Label::Distance::kNear);
299 {
300 Register maybeNegative = scratch;
301 __ Or(maybeNegative, left, Operand(right));
302 // TODO(Vladimir Kempik): consider usage of bexti instruction if Zbs
303 // extension is available
304 __ And(maybeNegative, maybeNegative, Operand(0x80000000)); // 1 << 31
305 // If one of them is negative, we must have a -0 result, which is non-int32,
306 // so deopt.
307 // TODO(leszeks): Consider splitting these deopts to have distinct deopt
308 // reasons. Otherwise, the reason has to match the above.
309 __ RecordComment("-- Jump to eager deopt if the result is negative zero");
310 Label* deopt_label = __ GetDeoptLabel(this, DeoptimizeReason::kOverflow);
311 __ MacroAssembler::Branch(deopt_label, ne, maybeNegative,
312 Operand(zero_reg));
313 }
314
315 __ bind(&end);
316 if (out_alias_input) {
317 __ Move(out, res);
318 }
319
320 // The output register shouldn't be a register input into the eager deopt
321 // info.
322 DCHECK_REGLIST_EMPTY(RegList{out} &
323 GetGeneralRegistersUsedAsInputs(eager_deopt_info()));
324}

332void Int32DivideWithOverflow::GenerateCode(MaglevAssembler* masm,
333 const ProcessingState& state) {
334 Register left = ToRegister(left_input());
335 Register right = ToRegister(right_input());
336 Register out = ToRegister(result());
337
338 // TODO(leszeks): peephole optimise division by a constant.
339
340 static_assert(Int32DivideWithOverflow::kProperties.can_eager_deopt());
341 ZoneLabelRef done(masm);
342 Label* deferred_overflow_checks = __ MakeDeferredCode(
343 [](MaglevAssembler* masm, ZoneLabelRef done, Register left,
344 Register right, Int32DivideWithOverflow* node) {
345 // {right} is negative or zero.
346
347 // TODO(leszeks): Using kNotInt32 here, but in same places
348 // kDivisionByZerokMinusZero/kMinusZero/kOverflow would be better. Right
349 // now all eager deopts in a node have to be the same -- we should allow
350 // a node to emit multiple eager deopts with different reasons.
351 Label* deopt = __ GetDeoptLabel(node, DeoptimizeReason::kNotInt32);
352
353 // Check if {right} is zero.
354 __ RecordComment("-- Jump to eager deopt if right is zero");
355 __ MacroAssembler::Branch(deopt, eq, right, Operand(zero_reg));
356
357 // Check if {left} is zero, as that would produce minus zero.
358 __ RecordComment("-- Jump to eager deopt if left is zero");
359 __ MacroAssembler::Branch(deopt, eq, left, Operand(zero_reg));
360
361 // Check if {left} is kMinInt and {right} is -1, in which case we'd have
362 // to return -kMinInt, which is not representable as Int32.
363 __ MacroAssembler::Branch(*done, ne, left, Operand(kMinInt));
364 __ MacroAssembler::Branch(*done, ne, right, Operand(-1));
365 __ JumpToDeopt(deopt);
366 },
367 done, left, right, this);
368
369 // Check if {right} is positive and not zero.
370 __ MacroAssembler::Branch(deferred_overflow_checks, less_equal, right,
371 Operand(zero_reg));
372 __ bind(*done);
373
374 // Perform the actual integer division.
375 MaglevAssembler::TemporaryRegisterScope temps(masm);
376 bool out_alias_input = out == left || out == right;
377 Register res = out;
378 if (out_alias_input) {
379 res = temps.Acquire();
380 }
381 __ Div32(res, left, right);
382
383 // Check that the remainder is zero.
384 Register temp = temps.Acquire();
385 __ remw(temp, left, right);
386 Label* deopt = __ GetDeoptLabel(this, DeoptimizeReason::kNotInt32);
387 __ RecordComment("-- Jump to eager deopt if remainder is zero");
388 __ MacroAssembler::Branch(deopt, ne, temp, Operand(zero_reg));
389
390 // The output register shouldn't be a register input into the eager deopt
391 // info.
392 DCHECK_REGLIST_EMPTY(RegList{res} &
393 GetGeneralRegistersUsedAsInputs(eager_deopt_info()));
394 __ Move(out, res);
395}

403void Int32ModulusWithOverflow::GenerateCode(MaglevAssembler* masm,
404 const ProcessingState& state) {
405 // If AreAliased(lhs, rhs):
406 // deopt if lhs < 0 // Minus zero.
407 // 0
408 //
409 // Using same algorithm as in EffectControlLinearizer:
410 // if rhs <= 0 then
411 // rhs = -rhs
412 // deopt if rhs == 0
413 // if lhs < 0 then
414 // let lhs_abs = -lhs in
415 // let res = lhs_abs % rhs in
416 // deopt if res == 0
417 // -res
418 // else
419 // let msk = rhs - 1 in
420 // if rhs & msk == 0 then
421 // lhs & msk
422 // else
423 // lhs % rhs
424
425 Register lhs = ToRegister(left_input());
426 Register rhs = ToRegister(right_input());
427 Register out = ToRegister(result());
428
429 static_assert(Int32ModulusWithOverflow::kProperties.can_eager_deopt());
430 static constexpr DeoptimizeReason deopt_reason =
431 DeoptimizeReason::kDivisionByZero;
432
433 // For the modulus algorithm described above, lhs and rhs must not alias
434 // each other.
435 if (lhs == rhs) {
436 // TODO(victorgomes): This ideally should be kMinusZero, but Maglev only
437 // allows one deopt reason per IR.
438 Label* deopt = __ GetDeoptLabel(this, DeoptimizeReason::kDivisionByZero);
439 __ RecordComment("-- Jump to eager deopt");
440 __ MacroAssembler::Branch(deopt, less, lhs, Operand(zero_reg));
441 __ Move(out, zero_reg);
442 return;
443 }
444
445 DCHECK(!AreAliased(lhs, rhs));
446
447 ZoneLabelRef done(masm);
448 ZoneLabelRef rhs_checked(masm);
449
450 Label* deferred_rhs_check = __ MakeDeferredCode(
451 [](MaglevAssembler* masm, ZoneLabelRef rhs_checked, Register rhs,
452 Int32ModulusWithOverflow* node) {
453 __ negw(rhs, rhs);
454 __ MacroAssembler::Branch(*rhs_checked, ne, rhs, Operand(zero_reg));
455 __ EmitEagerDeopt(node, deopt_reason);
456 },
457 rhs_checked, rhs, this);
458 __ MacroAssembler::Branch(deferred_rhs_check, less_equal, rhs,
459 Operand(zero_reg));
460 __ bind(*rhs_checked);
461
462 Label* deferred_lhs_check = __ MakeDeferredCode(
463 [](MaglevAssembler* masm, ZoneLabelRef done, Register lhs, Register rhs,
464 Register out, Int32ModulusWithOverflow* node) {
465 MaglevAssembler::TemporaryRegisterScope temps(masm);
466 Register lhs_abs = temps.AcquireScratch();
467 __ negw(lhs_abs, lhs);
468 Register res = lhs_abs;
469 __ remw(res, lhs_abs, rhs);
470 __ negw(out, res);
471 __ MacroAssembler::Branch(*done, ne, res, Operand(zero_reg));
472 // TODO(victorgomes): This ideally should be kMinusZero, but Maglev
473 // only allows one deopt reason per IR.
474 __ EmitEagerDeopt(node, deopt_reason);
475 },
476 done, lhs, rhs, out, this);
477 __ MacroAssembler::Branch(deferred_lhs_check, less, lhs, Operand(zero_reg));
478
479 Label rhs_not_power_of_2;
480 MaglevAssembler::TemporaryRegisterScope temps(masm);
481 Register scratch = temps.AcquireScratch();
482 Register msk = temps.AcquireScratch();
483 __ Sub32(msk, rhs, Operand(1));
484 __ And(scratch, rhs, msk);
485 __ MacroAssembler::Branch(&rhs_not_power_of_2, not_equal, scratch,
486 Operand(zero_reg), Label::kNear);
487 // {rhs} is power of 2.
488 __ And(out, lhs, msk);
489 __ MacroAssembler::Branch(*done, Label::kNear);
490
491 __ bind(&rhs_not_power_of_2);
492 __ remw(out, lhs, rhs);
493
494 __ bind(*done);
495}

651void Float64Round::GenerateCode(MaglevAssembler* masm,
652 const ProcessingState& state) {
653 DoubleRegister in = ToDoubleRegister(input());
654 DoubleRegister out = ToDoubleRegister(result());
655 MaglevAssembler::TemporaryRegisterScope temps(masm);
656 DoubleRegister fscratch1 = temps.AcquireScratchDouble();
657
658 if (kind_ == Kind::kNearest) {
659 // RISC-V Rounding Mode RNE means "Round to Nearest, ties to Even", while JS
660 // expects it to round towards +Infinity (see ECMA-262, 20.2.2.28).
661 // The best seems to be to add 0.5 then round with RDN mode.
662
663 DoubleRegister half_one = temps.AcquireDouble(); // available in this mode
664 __ LoadFPRImmediate(half_one, 0.5);
665 DoubleRegister tmp = half_one;
666 __ fadd_d(tmp, in, half_one);
667 __ Floor_d_d(out, tmp, fscratch1);
668 __ fsgnj_d(out, out, in);
669 } else if (kind_ == Kind::kCeil) {
670 __ Ceil_d_d(out, in, fscratch1);
671 } else if (kind_ == Kind::kFloor) {
672 __ Floor_d_d(out, in, fscratch1);
673 } else {
674 UNREACHABLE();
675 }
676}

775
776void HandleInterruptsAndTiering(MaglevAssembler* masm, ZoneLabelRef done,
777 Node* node, ReduceInterruptBudgetType type,
778 Register scratch0) {
779 // For loops, first check for interrupts. Don't do this for returns, as we
780 // can't lazy deopt to the end of a return.
781 if (type == ReduceInterruptBudgetType::kLoop) {
782 Label next;
783 // Here, we only care about interrupts since we've already guarded against
784 // real stack overflows on function entry.
785 {
786 Register stack_limit = scratch0;
787 __ LoadStackLimit(stack_limit, StackLimitKind::kInterruptStackLimit);
788 __ MacroAssembler::Branch(&next, ugt, sp, Operand(stack_limit),
789 Label::Distance::kNear);
790 }
791
792 // An interrupt has been requested and we must call into runtime to handle
793 // it; since we already pay the call cost, combine with the TieringManager
794 // call.
795 {
796 SaveRegisterStateForCall save_register_state(masm,
797 node->register_snapshot());
798 Register function = scratch0;
799 __ LoadWord(function,
800 MemOperand(fp, StandardFrameConstants::kFunctionOffset));
801 __ Push(function);
802 // Move into kContextRegister after the load into scratch0, just in case
803 // scratch0 happens to be kContextRegister.
804 __ Move(kContextRegister, masm->native_context().object());
805 __ CallRuntime(Runtime::kBytecodeBudgetInterruptWithStackCheck_Maglev, 1);
806 save_register_state.DefineSafepointWithLazyDeopt(node->lazy_deopt_info());
807 }
808 __ MacroAssembler::Branch(*done); // All done, continue.
809 __ bind(&next);
810 }
811
812 // No pending interrupts. Call into the TieringManager if needed.
813 {
814 SaveRegisterStateForCall save_register_state(masm,
815 node->register_snapshot());
816 Register function = scratch0;
817 __ LoadWord(function,
818 MemOperand(fp, StandardFrameConstants::kFunctionOffset));
819 __ Push(function);
820 // Move into kContextRegister after the load into scratch0, just in case
821 // scratch0 happens to be kContextRegister.
822 __ Move(kContextRegister, masm->native_context().object());
823 // Note: must not cause a lazy deopt!
824 __ CallRuntime(Runtime::kBytecodeBudgetInterrupt_Maglev, 1);
825 save_register_state.DefineSafepoint();
826 }
827 __ MacroAssembler::Branch(*done);
828}

830void GenerateReduceInterruptBudget(MaglevAssembler* masm, Node* node,
831 ReduceInterruptBudgetType type, int amount) {
832 MaglevAssembler::TemporaryRegisterScope temps(masm);
833 Register scratch = temps.Acquire();
834 Register feedback_cell = scratch;
835 Register budget = temps.Acquire();
836 __ LoadWord(feedback_cell,
837 MemOperand(fp, StandardFrameConstants::kFunctionOffset));
838 __ LoadTaggedField(
839 feedback_cell,
840 FieldMemOperand(feedback_cell, JSFunction::kFeedbackCellOffset));
841 __ Lw(budget,
842 FieldMemOperand(feedback_cell, FeedbackCell::kInterruptBudgetOffset));
843 __ Sub32(budget, budget, Operand(amount));
844 __ Sw(budget,
845 FieldMemOperand(feedback_cell, FeedbackCell::kInterruptBudgetOffset));
846
847 ZoneLabelRef done(masm);
848 Label* deferred_code = __ MakeDeferredCode(
849 [](MaglevAssembler* masm, ZoneLabelRef done, Node* node,
850 ReduceInterruptBudgetType type, Register scratch) {
851 HandleInterruptsAndTiering(masm, done, node, type, scratch);
852 },
853 done, node, type, scratch);
854 __ MacroAssembler::Branch(deferred_code, lt, budget, Operand(zero_reg));
855
856 __ bind(*done);
857}

888void Return::GenerateCode(MaglevAssembler* masm, const ProcessingState& state) {
889 DCHECK_EQ(ToRegister(value_input()), kReturnRegister0);
890 // Read the formal number of parameters from the top level compilation unit
891 // (i.e. the outermost, non inlined function).
892 int formal_params_size =
893 masm->compilation_info()->toplevel_compilation_unit()->parameter_count();
894
895 // We're not going to continue execution, so we can use an arbitrary register
896 // here instead of relying on temporaries from the register allocator.
897 // We cannot use scratch registers, since they're used in LeaveFrame and
898 // DropArguments.
899 Register actual_params_size = a5;
900 Register params_size = a6;
901
902 // Compute the size of the actual parameters + receiver (in bytes).
903 // TODO(leszeks): Consider making this an input into Return to re-use the
904 // incoming argc's register (if it's still valid).
905 __ LoadWord(actual_params_size,
906 MemOperand(fp, StandardFrameConstants::kArgCOffset));
907 __ Move(params_size, formal_params_size);
908
909 // If actual is bigger than formal, then we should use it to free up the stack
910 // arguments.
911 Label corrected_args_count;
912 __ MacroAssembler::Branch(&corrected_args_count, ge, params_size,
913 Operand(actual_params_size),
914 Label::Distance::kNear);
915 __ Move(params_size, actual_params_size);
916 __ bind(&corrected_args_count);
917
918 // Leave the frame.
919 __ LeaveFrame(StackFrame::MAGLEV);
920
921 // Drop receiver + arguments according to dynamic arguments size.
922 __ DropArguments(params_size);
923 __ Ret();
924}
