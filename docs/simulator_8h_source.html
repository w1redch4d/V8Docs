41// Running with a simulator.
42
43// The simulator has its own stack. Thus it has a different stack limit from
44// the C-based native code. The JS-based limit normally points near the end of
45// the simulator stack. When the C-based limit is exhausted we reflect that by
46// lowering the JS-based limit as well, to make stack checks trigger.
47class SimulatorStack : public v8::internal::AllStatic {
48 public:
49 static inline uintptr_t JsLimitFromCLimit(v8::internal::Isolate* isolate,
50 uintptr_t c_limit) {
51 return Simulator::current(isolate)->StackLimit(c_limit);
52 }
53
54#if V8_ENABLE_WEBASSEMBLY
55 static inline base::Vector<uint8_t> GetCentralStackView(
56 v8::internal::Isolate* isolate) {
57 return Simulator::current(isolate)->GetCentralStackView();
58 }
59#endif
60
61 // When running on the simulator, we should leave the C stack limits alone
62 // when switching stacks for Wasm.
63 static inline bool ShouldSwitchCStackForWasmStackSwitching() { return false; }
64
65 // Returns the current stack address on the simulator stack frame.
66 // The returned address is comparable with JS stack address.
67 static inline uintptr_t RegisterJSStackComparableAddress(
68 v8::internal::Isolate* isolate) {
69 // The value of |kPlaceHolder| is actually not used. It just occupies a
70 // single word on the stack frame of the simulator.
71 const uintptr_t kPlaceHolder = 0x4A535350u; // "JSSP" in ASCII
72 return Simulator::current(isolate)->PushAddress(kPlaceHolder);
73 }
74
75 static inline void UnregisterJSStackComparableAddress(
76 v8::internal::Isolate* isolate) {
77 Simulator::current(isolate)->PopAddress();
78 }
79};

81#else // defined(USE_SIMULATOR)

83
84// The stack limit beyond which we will throw stack overflow errors in
85// generated code. Because generated code uses the C stack, we just use
86// the C stack limit.
87class SimulatorStack : public v8::internal::AllStatic {
88 public:
89 static inline uintptr_t JsLimitFromCLimit(v8::internal::Isolate* isolate,
90 uintptr_t c_limit) {
91 USE(isolate);
92 return c_limit;
93 }
94
95#if V8_ENABLE_WEBASSEMBLY
96 static inline base::Vector<uint8_t> GetCentralStackView(
97 v8::internal::Isolate* isolate) {
98 uintptr_t upper_bound = base::Stack::GetStackStart();
99 size_t size =
100 v8_flags.stack_size * KB + wasm::StackMemory::kJSLimitOffsetKB * KB;
101 uintptr_t lower_bound = upper_bound - size;
102 return base::VectorOf(reinterpret_cast<uint8_t*>(lower_bound), size);
103 }
104#endif
105
106 // When running on real hardware, we should also switch the C stack limit
107 // when switching stacks for Wasm.
108 static inline bool ShouldSwitchCStackForWasmStackSwitching() { return true; }
109
110 // Returns the current stack address on the native stack frame.
111 // The returned address is comparable with JS stack address.
112 static inline uintptr_t RegisterJSStackComparableAddress(
113 v8::internal::Isolate* isolate) {
114 USE(isolate);
115 return internal::GetCurrentStackPosition();
116 }
117
118 static inline void UnregisterJSStackComparableAddress(
119 v8::internal::Isolate* isolate) {
120 USE(isolate);
121 }
122};

126// Use this class either as {GeneratedCode<ret, arg1, arg2>} or
127// {GeneratedCode<ret(arg1, arg2)>} (see specialization below).
128template <typename Return, typename... Args>
129class GeneratedCode {
130 public:
131 using Signature = Return(Args...);
132
133 static GeneratedCode FromAddress(Isolate* isolate, Address addr) {
134 return GeneratedCode(isolate, reinterpret_cast<Signature*>(addr));
135 }
136
137 static GeneratedCode FromBuffer(Isolate* isolate, uint8_t* buffer) {
138 return GeneratedCode(isolate, reinterpret_cast<Signature*>(buffer));
139 }
140
141 static GeneratedCode FromCode(Isolate* isolate, Tagged<Code> code) {
142 return FromAddress(isolate, code->instruction_start());
143 }
144
145#ifdef USE_SIMULATOR
146 // Defined in simulator-base.h.
147 Return Call(Args... args) {
148// Starboard is a platform abstraction interface that also include Windows
149// platforms like UWP.
150#if defined(V8_TARGET_OS_WIN) && !defined(V8_OS_WIN) && \
151 !defined(V8_OS_STARBOARD) && !defined(V8_TARGET_ARCH_ARM)
152 FATAL(
153 "Generated code execution not possible during cross-compilation."
154 "Also, generic C function calls are not implemented on 32-bit arm "
155 "yet.");
156#endif // defined(V8_TARGET_OS_WIN) && !defined(V8_OS_WIN) &&
157 // !defined(V8_OS_STARBOARD) && !defined(V8_TARGET_ARCH_ARM)
158 return Simulator::current(isolate_)->template Call<Return>(
159 reinterpret_cast<Address>(fn_ptr_), args...);
160 }
161#else
162
163 DISABLE_CFI_ICALL Return Call(Args... args) {
164 // When running without a simulator we call the entry directly.
165// Starboard is a platform abstraction interface that also include Windows
166// platforms like UWP.
167#if defined(V8_TARGET_OS_WIN) && !defined(V8_OS_WIN) && \
168 !defined(V8_OS_STARBOARD)
169 FATAL("Generated code execution not possible during cross-compilation.");
170#endif // defined(V8_TARGET_OS_WIN) && !defined(V8_OS_WIN)
171#if ABI_USES_FUNCTION_DESCRIPTORS
172#if V8_OS_ZOS
173 // z/OS ABI requires function descriptors (FD). Artificially create a pseudo
174 // FD to ensure correct dispatch to generated code.
175 void* function_desc[2] = {0, reinterpret_cast<void*>(fn_ptr_)};
176 asm volatile(" stg 5,%0 " : "=m"(function_desc[0])::"r5");
177 Signature* fn = reinterpret_cast<Signature*>(function_desc);
178 return fn(args...);
179#else
180 // AIX ABI requires function descriptors (FD). Artificially create a pseudo
181 // FD to ensure correct dispatch to generated code. The 'volatile'
182 // declaration is required to avoid the compiler from not observing the
183 // alias of the pseudo FD to the function pointer, and hence, optimizing the
184 // pseudo FD declaration/initialization away.
185 volatile Address function_desc[] = {reinterpret_cast<Address>(fn_ptr_), 0,
186 0};
187 Signature* fn = reinterpret_cast<Signature*>(function_desc);
188 return fn(args...);
189#endif // V8_OS_ZOS
190#else
191 return fn_ptr_(args...);
192#endif // ABI_USES_FUNCTION_DESCRIPTORS
193 }
194#endif // USE_SIMULATOR
195
196 private:
197 friend class GeneratedCode<Return(Args...)>;
198 Isolate* isolate_;
199 Signature* fn_ptr_;
200 GeneratedCode(Isolate* isolate, Signature* fn_ptr)
201 : isolate_(isolate), fn_ptr_(fn_ptr) {}
202};
