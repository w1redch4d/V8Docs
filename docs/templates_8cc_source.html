32Handle<SharedFunctionInfo> FunctionTemplateInfo::GetOrCreateSharedFunctionInfo(
33 Isolate* isolate, DirectHandle<FunctionTemplateInfo> info,
34 MaybeDirectHandle<Name> maybe_name) {
35 Tagged<Object> current_info = info->shared_function_info();
36 if (IsSharedFunctionInfo(current_info)) {
37 return handle(Cast<SharedFunctionInfo>(current_info), isolate);
38 }
39 DirectHandle<Name> name;
40 DirectHandle<String> name_string;
41 if (maybe_name.ToHandle(&name) && IsString(*name)) {
42 name_string = Cast<String>(name);
43 } else if (IsString(info->class_name())) {
44 name_string = direct_handle(Cast<String>(info->class_name()), isolate);
45 } else {
46 name_string = isolate->factory()->empty_string();
47 }
48 FunctionKind function_kind;
49 if (info->remove_prototype()) {
50 function_kind = FunctionKind::kConciseMethod;
51 } else {
52 function_kind = FunctionKind::kNormalFunction;
53 }
54 Handle<SharedFunctionInfo> sfi =
55 isolate->factory()->NewSharedFunctionInfoForApiFunction(name_string, info,
56 function_kind);
57 DCHECK(sfi->IsApiFunction());
58 info->set_shared_function_info(*sfi);
59 return sfi;
60}

62bool FunctionTemplateInfo::IsTemplateFor(Tagged<Map> map) const {
63 RCS_SCOPE(
64 LocalHeap::Current() == nullptr
65 ? GetIsolateChecked()->counters()->runtime_call_stats()
66 : LocalIsolate::FromHeap(LocalHeap::Current())->runtime_call_stats(),
67 RuntimeCallCounterId::kIsTemplateFor);
68
69 // There is a constraint on the object; check.
70 if (!IsJSObjectMap(map)) return false;
71
72 if (v8_flags.experimental_embedder_instance_types) {
73 DCHECK_IMPLIES(allowed_receiver_instance_type_range_start() == 0,
74 allowed_receiver_instance_type_range_end() == 0);
75 if (base::IsInRange(map->instance_type(),
76 allowed_receiver_instance_type_range_start(),
77 allowed_receiver_instance_type_range_end())) {
78 return true;
79 }
80 }
81
82 // Fetch the constructor function of the object.
83 Tagged<Object> cons_obj = map->GetConstructor();
84 Tagged<Object> type;
85 if (IsJSFunction(cons_obj)) {
86 Tagged<JSFunction> fun = Cast<JSFunction>(cons_obj);
87 if (!fun->shared()->IsApiFunction()) return false;
88 type = fun->shared()->api_func_data();
89 } else if (IsFunctionTemplateInfo(cons_obj)) {
90 type = Cast<FunctionTemplateInfo>(cons_obj);
91 } else {
92 return false;
93 }
94 DCHECK(IsFunctionTemplateInfo(type));
95 // Iterate through the chain of inheriting function templates to
96 // see if the required one occurs.
97 while (IsFunctionTemplateInfo(type)) {
98 if (type == *this) return true;
99 type = Cast<FunctionTemplateInfo>(type)->GetParentTemplate();
100 }
101 // Didn't find the required type in the inheritance chain.
102 return false;
103}

105bool FunctionTemplateInfo::IsLeafTemplateForApiObject(
106 Tagged<Object> object) const {
107 i::DisallowGarbageCollection no_gc;
108
109 if (!IsJSApiObject(object)) {
110 return false;
111 }
112
113 bool result = false;
114 Tagged<Map> map = Cast<HeapObject>(object)->map();
115 Tagged<Object> constructor_obj = map->GetConstructor();
116 if (IsJSFunction(constructor_obj)) {
117 Tagged<JSFunction> fun = Cast<JSFunction>(constructor_obj);
118 result = (*this == fun->shared()->api_func_data());
119 } else if (IsFunctionTemplateInfo(constructor_obj)) {
120 result = (*this == constructor_obj);
121 }
122 DCHECK_IMPLIES(result, IsTemplateFor(map));
123 return result;
124}

176Handle<DictionaryTemplateInfo> DictionaryTemplateInfo::Create(
177 Isolate* isolate, const v8::MemorySpan<const std::string_view>& names) {
178 DirectHandle<FixedArray> property_names = isolate->factory()->NewFixedArray(
179 static_cast<int>(names.size()), AllocationType::kOld);
180 int index = 0;
181 uint32_t unused_array_index;
182 for (const std::string_view& name : names) {
183 DirectHandle<String> internalized_name =
184 isolate->factory()->InternalizeString(
185 base::Vector<const char>(name.data(), name.length()));
186 // Check that property name cannot be used as index.
187 CHECK(!internalized_name->AsArrayIndex(&unused_array_index));
188 property_names->set(index, *internalized_name);
189 ++index;
190 }
191 return isolate->factory()->NewDictionaryTemplateInfo(property_names);
192}

196Handle<JSObject> CreateSlowJSObjectWithProperties(
197 Isolate* isolate, DirectHandle<FixedArray> property_names,
198 const MemorySpan<MaybeLocal<Value>>& property_values,
199 int num_properties_set) {
200 Handle<JSObject> object = isolate->factory()->NewSlowJSObjectFromMap(
201 isolate->slow_object_with_object_prototype_map(), num_properties_set,
202 AllocationType::kYoung);
203 Handle<Object> properties = handle(object->raw_properties_or_hash(), isolate);
204 for (int i = 0; i < static_cast<int>(property_values.size()); ++i) {
205 Local<Value> property_value;
206 if (!property_values[i].ToLocal(&property_value)) {
207 continue;
208 }
209 properties = PropertyDictionary::Add(
210 isolate, Cast<PropertyDictionary>(properties),
211 Cast<String>(handle(property_names->get(i), isolate)),
212 Utils::OpenDirectHandle(*property_value), PropertyDetails::Empty());
213 }
214 object->set_raw_properties_or_hash(*properties);
215 return object;
216}

220// static
221Handle<JSObject> DictionaryTemplateInfo::NewInstance(
222 DirectHandle<NativeContext> context,
223 DirectHandle<DictionaryTemplateInfo> self,
224 const MemorySpan<MaybeLocal<Value>>& property_values) {
225 Isolate* isolate = context->GetIsolate();
226 DirectHandle<FixedArray> property_names(self->property_names(), isolate);
227
228 const int property_names_len = property_names->length();
229 CHECK_EQ(property_names_len, static_cast<int>(property_values.size()));
230 const int num_properties_set = static_cast<int>(std::count_if(
231 property_values.begin(), property_values.end(),
232 [](const auto& maybe_value) { return !maybe_value.IsEmpty(); }));
233
234 if (V8_UNLIKELY(num_properties_set > JSObject::kMaxInObjectProperties)) {
235 return CreateSlowJSObjectWithProperties(
236 isolate, property_names, property_values, num_properties_set);
237 }
238
239 const bool can_use_map_cache = num_properties_set == property_names_len;
240 MaybeDirectHandle<Map> maybe_cached_map;
241 if (V8_LIKELY(can_use_map_cache)) {
242 maybe_cached_map = TemplateInfo::ProbeInstantiationsCache<Map>(
243 isolate, context, self->serial_number(),
244 TemplateInfo::CachingMode::kUnlimited);
245 }
246 DirectHandle<Map> cached_map;
247 if (V8_LIKELY(can_use_map_cache && maybe_cached_map.ToHandle(&cached_map))) {
248 DCHECK(!cached_map->is_dictionary_map());
249 bool can_use_cached_map = !cached_map->is_deprecated();
250 if (V8_LIKELY(can_use_cached_map)) {
251 // Verify that the cached map can be reused.
252 auto descriptors = handle(cached_map->instance_descriptors(), isolate);
253 for (int i = 0; i < static_cast<int>(property_values.size()); ++i) {
254 DirectHandle<Object> value =
255 Utils::OpenDirectHandle(*property_values[i].ToLocalChecked());
256 InternalIndex descriptor{static_cast<size_t>(i)};
257 const auto details = descriptors->GetDetails(descriptor);
258
259 if (!Object::FitsRepresentation(*value, details.representation()) ||
260 !FieldType::NowContains(descriptors->GetFieldType(descriptor),
261 value)) {
262 can_use_cached_map = false;
263 break;
264 }
265 // Double representation means mutable heap number. In this case we need
266 // to allocate a new heap number to put in the dictionary.
267 if (details.representation().Equals(Representation::Double())) {
268 // We allowed coercion in `FitsRepresentation` above which means that
269 // we may deal with a Smi here.
270 property_values[i] =
271 ToApiHandle<v8::Object>(isolate->factory()->NewHeapNumber(
272 Object::NumberValue(Cast<Number>(*value))));
273 }
274 }
275 if (V8_LIKELY(can_use_cached_map)) {
276 // Create the object from the cached map.
277 CHECK(!cached_map->is_deprecated());
278 CHECK_EQ(context->object_function_prototype(), cached_map->prototype());
279 auto object = isolate->factory()->NewJSObjectFromMap(
280 cached_map, AllocationType::kYoung);
281 DisallowGarbageCollection no_gc;
282 for (int i = 0; i < static_cast<int>(property_values.size()); ++i) {
283 Local<Value> property_value = property_values[i].ToLocalChecked();
284 DirectHandle<Object> value = Utils::OpenDirectHandle(*property_value);
285 const FieldIndex index = FieldIndex::ForPropertyIndex(
286 *cached_map, i, Representation::Tagged());
287 object->FastPropertyAtPut(index, *value,
288 WriteBarrierMode::SKIP_WRITE_BARRIER);
289 }
290 return object;
291 }
292 }
293 // A cached map was either deprecated or the descriptors changed in
294 // incompatible ways. We clear the cached map and continue with the generic
295 // path.
296 TemplateInfo::UncacheTemplateInstantiation(
297 isolate, context, self, TemplateInfo::CachingMode::kUnlimited);
298 }
299
300 // General case: We either don't have a cached map, or it is unusuable for the
301 // values provided.
302 DirectHandle<Map> current_map = isolate->factory()->ObjectLiteralMapFromCache(
303 context, num_properties_set);
304 Handle<JSObject> object = isolate->factory()->NewJSObjectFromMap(current_map);
305 int current_property_index = 0;
306 for (int i = 0; i < static_cast<int>(property_values.size()); ++i) {
307 Local<Value> property_value;
308 if (!property_values[i].ToLocal(&property_value)) {
309 continue;
310 }
311 auto name = Cast<String>(handle(property_names->get(i), isolate));
312 DirectHandle<Object> value = Utils::OpenDirectHandle(*property_value);
313 constexpr PropertyAttributes attributes = PropertyAttributes::NONE;
314 constexpr PropertyConstness constness = PropertyConstness::kConst;
315 current_map = Map::TransitionToDataProperty(isolate, current_map, name,
316 value, attributes, constness,
317 StoreOrigin::kNamed);
318 if (current_map->is_dictionary_map()) {
319 return CreateSlowJSObjectWithProperties(
320 isolate, property_names, property_values, num_properties_set);
321 }
322 JSObject::MigrateToMap(isolate, object, current_map);
323 PropertyDetails details = current_map->GetLastDescriptorDetails(isolate);
324 object->WriteToField(InternalIndex(current_property_index), details,
325 *value);
326 current_property_index++;
327 }
328 if (V8_LIKELY(can_use_map_cache)) {
329 TemplateInfo::CacheTemplateInstantiation(
330 isolate, context, self, TemplateInfo::CachingMode::kUnlimited,
331 handle(object->map(), isolate));
332 }
333 return object;
334}
