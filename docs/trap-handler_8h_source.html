122// Initially false, set to true if when trap handlers are enabled. Never goes
123// back to false then.
124TH_EXPORT_PRIVATE extern bool g_is_trap_handler_enabled;

126// Initially true, set to false when either {IsTrapHandlerEnabled} or
127// {EnableTrapHandler} is called to prevent calling {EnableTrapHandler}
128// repeatedly, or after {IsTrapHandlerEnabled}. Needs to be atomic because
129// {IsTrapHandlerEnabled} can be called from any thread. Updated using relaxed
130// semantics, since it's not used for synchronization.
131TH_EXPORT_PRIVATE extern std::atomic<bool> g_can_enable_trap_handler;

143inline bool IsTrapHandlerEnabled() {
144 TH_DCHECK(!g_is_trap_handler_enabled || V8_TRAP_HANDLER_SUPPORTED);
145 // Disallow enabling the trap handler after retrieving the current value.
146 // Re-enabling them late can produce issues because code or objects might have
147 // been generated under the assumption that trap handlers are disabled.
148 // Note: We test before setting to avoid contention by an unconditional write.
149 if (g_can_enable_trap_handler.load(std::memory_order_relaxed)) {
150 g_can_enable_trap_handler.store(false, std::memory_order_relaxed);
151 }
152 return g_is_trap_handler_enabled;
153}

168// On Windows, asan installs its own exception handler which maps shadow
169// memory. Since our exception handler may be executed before the asan exception
170// handler, we have to make sure that asan shadow memory is not accessed here.
171TH_DISABLE_ASAN inline bool IsThreadInWasm() { return g_thread_in_wasm_code; }

173inline void SetThreadInWasm() {
174 if (IsTrapHandlerEnabled()) {
175 TH_DCHECK(!IsThreadInWasm());
176 g_thread_in_wasm_code = true;
177 }
178}

180inline void ClearThreadInWasm() {
181 if (IsTrapHandlerEnabled()) {
182 TH_DCHECK(IsThreadInWasm());
183 g_thread_in_wasm_code = false;
184 }
