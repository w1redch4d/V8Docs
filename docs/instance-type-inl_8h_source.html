130// This one is very sneaky. String maps are laid out sequentially, and
131// alternate between two-byte and one-byte. Since they're sequential, each
132// address is one Map::kSize larger than the previous. This means that the LSB
133// of the map size alternates being set and unset for alternating string map
134// addresses, and therefore is on/off for all two-byte/one-byte strings. Which
135// of the two has the on-bit depends on the current RO heap layout, so just
136// sniff this by checking an arbitrary one-byte map's value.
137static constexpr int kStringMapEncodingMask =
138 1 << base::bits::CountTrailingZerosNonZero(Map::kSize);
139static constexpr int kOneByteStringMapBit =
140 StaticReadOnlyRoot::kSeqOneByteStringMap & kStringMapEncodingMask;
141static constexpr int kTwoByteStringMapBit =
142 StaticReadOnlyRoot::kSeqTwoByteStringMap & kStringMapEncodingMask;
143
144inline constexpr std::optional<TaggedAddressRange>
145UniqueMapRangeOfInstanceTypeRange(InstanceType first, InstanceType last) {
146 // Doesn't use range based for loop due to LLVM <11 bug re. constexpr
147 // functions.
148 for (size_t i = 0; i < kUniqueMapRangeOfInstanceTypeRangeList.size(); ++i) {
149 if (kUniqueMapRangeOfInstanceTypeRangeList[i].first.first == first &&
150 kUniqueMapRangeOfInstanceTypeRangeList[i].first.second == last) {
151 return {kUniqueMapRangeOfInstanceTypeRangeList[i].second};
152 }
153 }
154 return {};
155}
156
157inline constexpr std::optional<TaggedAddressRange> UniqueMapRangeOfInstanceType(
158 InstanceType type) {
159 return UniqueMapRangeOfInstanceTypeRange(type, type);
160}
161
162inline bool MayHaveMapCheckFastCase(InstanceType type) {
163 if (UniqueMapOfInstanceType(type)) return true;
164 for (auto& el : kUniqueMapRangeOfInstanceTypeRangeList) {
165 if (el.first.first <= type && type <= el.first.second) {
166 return true;
167 }
168 }
169 return false;
170}

189// Define type checkers for classes with single instance type.
190// INSTANCE_TYPE_CHECKER1 is to be used if the instance type is already loaded.
191// INSTANCE_TYPE_CHECKER2 is preferred since it can sometimes avoid loading the
192// instance type from the map, if the checked instance type corresponds to a
193// known map or range of maps.
194
195#define INSTANCE_TYPE_CHECKER1(type, forinstancetype) \
196 V8_INLINE constexpr bool Is##type(InstanceType instance_type) { \
197 return instance_type == forinstancetype; \
198 }

202#define INSTANCE_TYPE_CHECKER2(type, forinstancetype_) \
203 V8_INLINE bool Is##type(Tagged<Map> map_object) { \
204 constexpr InstanceType forinstancetype = \
205 static_cast<InstanceType>(forinstancetype_); \
206 if constexpr (constexpr std::optional<RootIndex> index = \
207 UniqueMapOfInstanceType(forinstancetype)) { \

211 UniqueMapRangeOfInstanceType(forinstancetype)) { \

231// Checks if value is in range [lower_limit, higher_limit] using a single
232// branch. Assumes that the input instance type is valid.
233template <InstanceType lower_limit, InstanceType upper_limit>
234struct InstanceRangeChecker {
235 static constexpr bool Check(InstanceType value) {
236 return base::IsInRange(value, lower_limit, upper_limit);
237 }
238};
239template <InstanceType upper_limit>
240struct InstanceRangeChecker<FIRST_TYPE, upper_limit> {
241 static constexpr bool Check(InstanceType value) {
242 DCHECK_LE(FIRST_TYPE, value);
243 return value <= upper_limit;
244 }
245};
246template <InstanceType lower_limit>
247struct InstanceRangeChecker<lower_limit, LAST_TYPE> {
248 static constexpr bool Check(InstanceType value) {
249 DCHECK_GE(LAST_TYPE, value);
250 return value >= lower_limit;
251 }
252};

465// Returns true for those heap object types that must be tied to some native
466// context.
467V8_INLINE constexpr bool IsNativeContextSpecific(InstanceType instance_type) {
468 // All context map are tied to some native context.
469 if (IsContext(instance_type)) return true;
470 // All non-JSReceivers are never tied to any native context.
471 if (!IsJSReceiver(instance_type)) return false;
472
473 // Most of the JSReceivers are tied to some native context modulo the
474 // following exceptions.
475 if (instance_type == JS_MESSAGE_OBJECT_TYPE ||
476 instance_type == JS_EXTERNAL_OBJECT_TYPE) {
477 // These JSObject types are wrappers around a set of primitive values
478 // and exist only for the purpose of passing the data across V8 Api.
479 // Thus they are not tied to any native context.
480 return false;
481
482 } else if (InstanceTypeChecker::IsAlwaysSharedSpaceJSObject(instance_type)) {
483 // JSObjects allocated in shared space are never tied to a native context.
484 return false;
485 }
486 return true;
487}
488
489V8_INLINE bool IsNativeContextSpecificMap(Tagged<Map> map_object) {
490 return IsNativeContextSpecific(map_object->instance_type());
491}
