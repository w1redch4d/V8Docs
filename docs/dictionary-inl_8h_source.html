51template <typename Derived, typename Shape>
52std::optional<Tagged<Object>> Dictionary<Derived, Shape>::TryValueAt(
53 InternalIndex entry) {
54#if DEBUG
55 Isolate* isolate;
56 GetIsolateFromHeapObject(this, &isolate);
57 DCHECK_NE(isolate, nullptr);
58 SLOW_DCHECK(!isolate->heap()->IsPendingAllocation(Tagged(this)));
59#endif // DEBUG
60 // We can read length() in a non-atomic way since we are reading an
61 // initialized object which is not pending allocation.
62 if (DerivedHashTable::EntryToIndex(entry) + Derived::kEntryValueIndex >=
63 this->length()) {
64 return {};
65 }
66 return ValueAt(entry);
67}

138bool NumberDictionary::requires_slow_elements() {
139 Tagged<Object> max_index_object = get(kMaxNumberKeyIndex);
140 if (!IsSmi(max_index_object)) return false;
141 return 0 != (Smi::ToInt(max_index_object) & kRequiresSlowElementsMask);

144uint32_t NumberDictionary::max_number_key() {

147 if (!IsSmi(max_index_object)) return 0;
148 uint32_t value = static_cast<uint32_t>(Smi::ToInt(max_index_object));
149 return value >> kRequiresSlowElementsTagSize;

163template <typename Derived, typename Shape>
164void Dictionary<Derived, Shape>::SetEntry(InternalIndex entry,
165 Tagged<Object> key,
166 Tagged<Object> value,
167 PropertyDetails details) {
168 DCHECK(Dictionary::kEntrySize == 2 || Dictionary::kEntrySize == 3);
169 DCHECK(!IsName(key) || details.dictionary_index() > 0);
170 int index = DerivedHashTable::EntryToIndex(entry);
171 DisallowGarbageCollection no_gc;
172 WriteBarrierMode mode = this->GetWriteBarrierMode(no_gc);
173 this->set(index + Derived::kEntryKeyIndex, key, mode);
174 this->set(index + Derived::kEntryValueIndex, value, mode);
175 if (Shape::kHasDetails) DetailsAtPut(entry, details);
176}
