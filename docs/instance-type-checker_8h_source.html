57// Maps for primitive objects and a select few JS objects are allocated in r/o
58// space. All JS_RECEIVER maps must come after primitive object maps, i.e. they
59// have a compressed address above the last primitive object map root. If we
60// have a receiver and need to distinguish whether it is either a primitive
61// object or a JS receiver, it suffices to check if its map is allocated above
62// the following limit address.
63constexpr Tagged_t kNonJsReceiverMapLimit =
64 StaticReadOnlyRootsPointerTable[static_cast<size_t>(
65 RootIndex::kFirstJSReceiverMapRoot)] &
66 ~0xFFF;
67
68// Maps for strings allocated as the first maps in r/o space, so their lower
69// bound is zero.
70constexpr Tagged_t kStringMapLowerBound = 0;
71// If we have a receiver and need to distinguish whether it is a string or not,
72// it suffices to check whether it is less-than-equal to the following value.
73constexpr Tagged_t kStringMapUpperBound =
74 StaticReadOnlyRoot::kThinOneByteStringMap;
75
76#define ASSERT_IS_LAST_STRING_MAP(instance_type, size, name, Name) \
77 static_assert(StaticReadOnlyRoot::k##Name##Map <= kStringMapUpperBound);
78STRING_TYPE_LIST(ASSERT_IS_LAST_STRING_MAP)
79#undef ASSERT_IS_LAST_STRING_MAP
80
81// For performance, the limit is chosen to be encodable as an Arm64
82// constant. See Assembler::IsImmAddSub in assembler-arm64.cc.
83//
84// If this assert fails, then you have perturbed the allocation pattern in
85// Heap::CreateReadOnlyHeapObjects. Currently this limit is ensured to exist by
86// allocating the first JSReceiver map in RO space a sufficiently large distance
87// away from the last non-JSReceiver map.
88static_assert(kNonJsReceiverMapLimit != 0 &&
89 is_uint12(kNonJsReceiverMapLimit >> 12) &&
90 ((kNonJsReceiverMapLimit & 0xFFF) == 0));
