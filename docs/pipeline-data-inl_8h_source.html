47inline Maybe<OuterContext> GetModuleContext(OptimizedCompilationInfo* info) {
48 Tagged<Context> current = info->closure()->context();
49 size_t distance = 0;
50 while (!IsNativeContext(*current)) {
51 if (current->IsModuleContext()) {
52 return Just(OuterContext(
53 info->CanonicalHandle(current, current->GetIsolate()), distance));
54 }
55 current = current->previous();
56 distance++;
57 }
58 return Nothing<OuterContext>();
59}

258 template <typename T>
259 using GraphZonePointer = turboshaft::ZoneWithNamePointer<T, kGraphZoneName>;
260 void InitializeWithGraphZone(
261 turboshaft::ZoneWithName<kGraphZoneName> graph_zone,
262 GraphZonePointer<SourcePositionTable> source_positions,
263 GraphZonePointer<NodeOriginTable> node_origins,
264 size_t node_count_hint = 0) {
265 // Delete the old zone first.
266 DeleteGraphZone();
267
268 // Take ownership of the new zone and the existing pointers.
269 graph_zone_ = std::move(graph_zone);
270 source_positions_ = source_positions;
271 node_origins_ = node_origins;
272
273 // Allocate a new graph and schedule.
274 graph_ = graph_zone_.New<Graph>(graph_zone_);
275 schedule_ = graph_zone_.New<Schedule>(graph_zone_, node_count_hint);
276
277 // Initialize node builders.
278 javascript_ = graph_zone_.New<JSOperatorBuilder>(graph_zone_);
279 common_ = graph_zone_.New<CommonOperatorBuilder>(graph_zone_);
280 simplified_ = graph_zone_.New<SimplifiedOperatorBuilder>(graph_zone_);
281 machine_ = graph_zone_.New<MachineOperatorBuilder>(
282 graph_zone_, MachineType::PointerRepresentation(),
283 InstructionSelector::SupportedMachineOperatorFlags(),
284 InstructionSelector::AlignmentRequirements());
285 }
286 turboshaft::ZoneWithName<kGraphZoneName> ReleaseGraphZone() {
287 turboshaft::ZoneWithName<kGraphZoneName> temp = std::move(graph_zone_);
288 // Call `DeleteGraphZone` to reset all pointers. The actual zone is not
289 // released because we moved it away.
290 DeleteGraphZone();
291 return temp;
292 }
