38 NodeIdT first_id() const {
39 if (has_phi()) return phis()->first()->id();
40 if (nodes_.is_empty()) {
41 return control_node()->id();
42 }
43 auto node = nodes_.first();
44 while (node && node->Is<Identity>()) {
45 node = node->NextNode();
46 }

48 }

55 if (node->Is<Identity>()) continue;

151 template <typename Func>
152 void ForEachPredecessor(Func&& functor) const {
153 if (type_ == kEdgeSplit || type_ == kOther) {
154 BasicBlock* predecessor_block = predecessor();
155 if (predecessor_block) {
156 functor(predecessor_block);
157 }
158 } else {
159 for (int i = 0; i < predecessor_count(); i++) {
160 functor(predecessor_at(i));
161 }
162 }
163 }

165 template <typename Func>
166 void ForEachSuccessor(Func&& functor) const {
167 ControlNode* control = control_node();
168 if (auto unconditional_control =
169 control->TryCast<UnconditionalControlNode>()) {
170 functor(unconditional_control->target());
171 } else if (auto branch = control->TryCast<BranchControlNode>()) {
172 functor(branch->if_true());
173 functor(branch->if_false());
174 } else if (auto switch_node = control->TryCast<Switch>()) {
175 for (int i = 0; i < switch_node->size(); i++) {
176 functor(switch_node->targets()[i].block_ptr());
177 }
178 if (switch_node->has_fallthrough()) {
179 functor(switch_node->fallthrough());
180 }
181 }
182 }

210 // If the basic block is an empty (unnecessary) block containing only an
211 // unconditional jump to the successor block, return the successor block.
212 BasicBlock* RealJumpTarget() {
213 if (real_jump_target_cache_ != nullptr) {
214 return real_jump_target_cache_;
215 }
216
217 BasicBlock* current = this;
218 while (true) {
219 if (!current->nodes_.is_empty() || current->is_loop() ||
220 current->is_exception_handler_block() ||
221 current->HasPhisOrRegisterMerges()) {
222 break;
223 }
224 Jump* control = current->control_node()->TryCast<Jump>();
225 if (!control) {
226 break;
227 }
228 BasicBlock* next = control->target();
229 if (next->HasPhisOrRegisterMerges()) {
230 break;
231 }
232 current = next;
233 }
234 real_jump_target_cache_ = current;
235 return current;
236 }

241 private:
242 bool HasPhisOrRegisterMerges() const {
243 if (!has_state()) {
244 return false;
245 }
246 if (has_phi()) {
247 return true;
248 }
249 bool has_register_merge = false;
250#ifdef V8_ENABLE_MAGLEV
251 if (!state()->register_state().is_initialized()) {
252 // This can happen when the graph has disconnected blocks; bail out and
253 // don't jump thread them.
254 return true;
255 }
256
257 state()->register_state().ForEachGeneralRegister(
258 [&](Register reg, RegisterState& state) {
259 ValueNode* node;
260 RegisterMerge* merge;
261 if (LoadMergeState(state, &node, &merge)) {
262 has_register_merge = true;
263 }
264 });
265 state()->register_state().ForEachDoubleRegister(
266 [&](DoubleRegister reg, RegisterState& state) {
267 ValueNode* node;
268 RegisterMerge* merge;
269 if (LoadMergeState(state, &node, &merge)) {
270 has_register_merge = true;
271 }
272 });
273#endif // V8_ENABLE_MAGLEV
274 return has_register_merge;
275 }

303inline base::SmallVector<BasicBlock*, 2> BasicBlock::successors() const {
304 ControlNode* control = control_node();
305 if (auto unconditional_control =
306 control->TryCast<UnconditionalControlNode>()) {
307 return {unconditional_control->target()};
308 } else if (auto branch = control->TryCast<BranchControlNode>()) {
309 return {branch->if_true(), branch->if_false()};
310 } else if (auto switch_node = control->TryCast<Switch>()) {
311 base::SmallVector<BasicBlock*, 2> succs;
312 for (int i = 0; i < switch_node->size(); i++) {
313 succs.push_back(switch_node->targets()[i].block_ptr());
314 }
315 if (switch_node->has_fallthrough()) {
316 succs.push_back(switch_node->fallthrough());
317 }
318 return succs;
319 } else {
320 return base::SmallVector<BasicBlock*, 2>();
321 }
322}
