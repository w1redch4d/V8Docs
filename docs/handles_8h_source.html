133// ----------------------------------------------------------------------------
134// A Handle provides a reference to an object that survives relocation by
135// the garbage collector.
136//
137// Handles are only valid within a HandleScope. When a handle is created
138// for an object a cell is allocated in the current HandleScope.
139//
140// Also note that Handles do not provide default equality comparison or hashing
141// operators on purpose. Such operators would be misleading, because intended
142// semantics is ambiguous between Handle location and object identity. Instead
143// use either {is_identical_to} or {location} explicitly.
144template <typename T>
145class Handle final : public HandleBase {
146 public:
147 // Handles denote strong references.
148 static_assert(!is_maybe_weak_v<T>);
149
150 V8_INLINE Handle() : HandleBase(nullptr) {}
151
152 V8_INLINE explicit Handle(Address* location) : HandleBase(location) {
153 // TODO(jkummerow): Runtime type check here as a SLOW_DCHECK?
154 }
155
156 V8_INLINE Handle(Tagged<T> object, Isolate* isolate);
157 V8_INLINE Handle(Tagged<T> object, LocalIsolate* isolate);
158 V8_INLINE Handle(Tagged<T> object, LocalHeap* local_heap);
159
160 // Allocate a new handle for the object.
161 V8_INLINE static Handle<T> New(Tagged<T> object, Isolate* isolate);
162
163 // Constructor for handling automatic up casting.
164 // Ex. Handle<JSFunction> can be passed when Handle<Object> is expected.
165 template <typename S>
166 V8_INLINE Handle(Handle<S> handle)
167 requires(is_subtype_v<S, T>)
168 : HandleBase(handle) {}
169
170 // Access a member of the T object referenced by this handle.
171 //
172 // This is actually a double dereference -- first it dereferences the Handle
173 // pointing to a Tagged<T>, and then continues through Tagged<T>::operator->.
174 // This means that this is only permitted for Tagged<T> with an operator->,
175 // i.e. for on-heap object T.
176 V8_INLINE Tagged<T> operator->() const {
177 // For non-HeapObjects, there's no on-heap object to dereference, so
178 // disallow using operator->.
179 //
180 // If you got an error here and want to access the Tagged<T>, use
181 // operator* -- e.g. for `Tagged<Smi>::value()`, use `(*handle).value()`.
182 static_assert(
183 is_subtype_v<T, HeapObject>,
184 "This handle does not reference a heap object. Use `(*handle).foo`.");
185 return **this;
186 }
187
188 V8_INLINE Tagged<T> operator*() const {
189 // This static type check also fails for forward class declarations. We
190 // check on access instead of on construction to allow Handles to forward
191 // declared types.
192 static_assert(is_taggable_v<T>, "static type violation");
193 // Direct construction of Tagged from address, without a type check, because
194 // we rather trust Handle<T> to contain a T than include all the respective
195 // -inl.h headers for SLOW_DCHECKs.
196 SLOW_DCHECK(IsDereferenceAllowed());
197 return Tagged<T>(*location());
198 }
199
200 // Consider declaring values that contain empty handles as
201 // MaybeHandle to force validation before being used as handles.
202 static const Handle<T> null() { return Handle<T>(); }
203
204 // Location equality.
205 bool equals(Handle<T> other) const { return address() == other.address(); }
206
207 // Patches this Handle's value, in-place, with a new value. All indirect
208 // handles with the same location will see this update.
209 void PatchValue(Tagged<T> new_value) {
210 SLOW_DCHECK(location_ != nullptr && IsDereferenceAllowed());
211 *location_ = new_value.ptr();
212 }
213
214 // Provide function object for location equality comparison.
215 struct equal_to {
216 V8_INLINE bool operator()(Handle<T> lhs, Handle<T> rhs) const {
217 return lhs.equals(rhs);
218 }
219 };
220
221 // Provide function object for location hashing.
222 struct hash {
223 V8_INLINE size_t operator()(Handle<T> const& handle) const {
224 return base::hash<Address>()(handle.address());
225 }
226 };
227
228 using MaybeType = MaybeHandle<T>;
229
230 private:
231 // Handles of different classes are allowed to access each other's location_.
232 template <typename>
233 friend class Handle;
234 // MaybeHandle is allowed to access location_.
235 template <typename>
236 friend class MaybeHandle;
237 // Casts are allowed to access location_.
238 template <typename To, typename From>
239 friend inline Handle<To> UncheckedCast(Handle<From> value);
240};

245// ----------------------------------------------------------------------------
246// A stack-allocated class that governs a number of local handles.
247// After a handle scope has been created, all local handles will be
248// allocated within that handle scope until either the handle scope is
249// deleted or another handle scope is created. If there is already a
250// handle scope and a new one is created, all allocations will take
251// place in the new handle scope until it is deleted. After that,
252// new handles will again be allocated in the original handle scope.
253//
254// After the handle scope of a local handle has been deleted the
255// garbage collector will no longer track the object stored in the
256// handle and may deallocate it. The behavior of accessing a handle
257// for which the handle scope has been deleted is undefined.
258class V8_NODISCARD HandleScope {
259 public:
260 explicit V8_INLINE HandleScope(Isolate* isolate);
261 inline HandleScope(HandleScope&& other) V8_NOEXCEPT;
262 HandleScope(const HandleScope&) = delete;
263 HandleScope& operator=(const HandleScope&) = delete;
264
265 // Allow placement new.
266 void* operator new(size_t size, void* storage) {
267 return ::operator new(size, storage);
268 }
269
270 // Prevent heap allocation or illegal handle scopes.
271 void* operator new(size_t size) = delete;
272 void operator delete(void* size_t) = delete;
273
274 V8_INLINE ~HandleScope();
275
276 inline HandleScope& operator=(HandleScope&& other) V8_NOEXCEPT;
277
278 // Counts the number of allocated handles.
279 V8_EXPORT_PRIVATE static int NumberOfHandles(Isolate* isolate);
280
281 // Creates a new handle with the given value.
282 V8_INLINE static Address* CreateHandle(Isolate* isolate, Address value);
283
284 // Deallocates any extensions used by the current scope.
285 V8_EXPORT_PRIVATE static void DeleteExtensions(Isolate* isolate);
286
287 static Address current_next_address(Isolate* isolate);
288 static Address current_limit_address(Isolate* isolate);
289 static Address current_level_address(Isolate* isolate);
290
291 // Closes the HandleScope (invalidating all handles
292 // created in the scope of the HandleScope) and returns
293 // a Handle backed by the parent scope holding the
294 // value of the argument handle.
295 //
296 // TODO(42203211): When direct handles are enabled, the version with
297 // HandleType = DirectHandle does not need to be called, as it simply
298 // closes the scope (which is done by the scope's destructor anyway)
299 // and returns its parameter. This will be cleaned up after direct
300 // handles ship.
301 template <typename T, template <typename> typename HandleType>
302 requires(std::is_convertible_v<HandleType<T>, DirectHandle<T>>)
303 HandleType<T> CloseAndEscape(HandleType<T> handle_value);
304
305 Isolate* isolate() { return isolate_; }
306
307 // Limit for number of handles with --check-handle-count. This is
308 // large enough to compile natives and pass unit tests with some
309 // slack for future changes to natives.
310 static const int kCheckHandleThreshold = 30 * 1024;
311
312 private:
313 Isolate* isolate_;
314 Address* prev_next_;
315 Address* prev_limit_;
316
317#ifdef V8_ENABLE_CHECKS
318 int scope_level_ = 0;
319#endif
320
321 // Close the handle scope resetting limits to a previous state.
322 static V8_INLINE void CloseScope(Isolate* isolate, Address* prev_next,
323 Address* prev_limit);
324
325 // Extend the handle scope making room for more handles.
326 V8_EXPORT_PRIVATE V8_NOINLINE static Address* Extend(Isolate* isolate);
327
328#if defined(ENABLE_GLOBAL_HANDLE_ZAPPING) || \
329 defined(ENABLE_LOCAL_HANDLE_ZAPPING)
330 // Zaps the handles in the half-open interval [start, end).
331 V8_EXPORT_PRIVATE static void ZapRange(Address* start, Address* end,
332 uintptr_t value = kHandleZapValue);
333#endif
334
335 friend class v8::HandleScope;
336 friend class HandleScopeImplementer;
337 friend class Isolate;
338 friend class LocalHandles;
339 friend class LocalHandleScope;
340 friend class PersistentHandles;
341};

490// ----------------------------------------------------------------------------
491// A DirectHandle provides a reference to an object without an intermediate
492// pointer.
493//
494// A DirectHandle is a simple wrapper around a tagged pointer to a heap object
495// or a SMI. Its methods are symmetrical with Handle, so that Handles can be
496// easily migrated.
497//
498// DirectHandles are intended to be used with conservative stack scanning, as
499// they do not provide a mechanism for keeping an object alive across a garbage
500// collection.
501//
502// Further motivation is explained in the design doc:
503// https://docs.google.com/document/d/1uRGYQM76vk1fc_aDqDH3pm2qhaJtnK2oyzeVng4cS6I/
504template <typename T>
505class DirectHandle : public DirectHandleBase {
506 public:
507 // Handles denote strong references.
508 static_assert(!is_maybe_weak_v<T>);
509
510 V8_INLINE DirectHandle() : DirectHandle(kTaggedNullAddress) {}
511
512 V8_INLINE explicit DirectHandle(Address object) : DirectHandleBase(object) {}
513
514 V8_INLINE DirectHandle(Tagged<T> object, Isolate* isolate)
515 : DirectHandle(object) {}
516 V8_INLINE DirectHandle(Tagged<T> object, LocalIsolate* isolate)
517 : DirectHandle(object) {}
518 V8_INLINE DirectHandle(Tagged<T> object, LocalHeap* local_heap)
519 : DirectHandle(object) {}
520
521 V8_INLINE static DirectHandle<T> New(Tagged<T> object, Isolate* isolate) {
522 return DirectHandle<T>(object);
523 }
524
525 // Constructor for handling automatic up casting.
526 // Ex. DirectHandle<JSFunction> can be passed when DirectHandle<Object> is
527 // expected.
528 template <typename S, typename = std::enable_if_t<is_subtype_v<S, T>>>
529 V8_INLINE DirectHandle(DirectHandle<S> handle) : DirectHandle(handle.obj_) {}
530
531 template <typename S, typename = std::enable_if_t<is_subtype_v<S, T>>>
532 V8_INLINE DirectHandle(Handle<S> handle)
533 : DirectHandle(handle.location() != nullptr ? *handle.location()
534 : kTaggedNullAddress) {}
535
536 V8_INLINE Tagged<T> operator->() const {
537 if constexpr (is_subtype_v<T, HeapObject>) {
538 return **this;
539 } else {
540 // For non-HeapObjects, there's no on-heap object to dereference, so
541 // disallow using operator->.
542 //
543 // If you got an error here and want to access the Tagged<T>, use
544 // operator* -- e.g. for `Tagged<Smi>::value()`, use `(*handle).value()`.
545 static_assert(
546 false,
547 "This handle does not reference a heap object. Use `(*handle).foo`.");
548 }
549 }
550
551 V8_INLINE Tagged<T> operator*() const {
552 // This static type check also fails for forward class declarations. We
553 // check on access instead of on construction to allow DirectHandles to
554 // forward declared types.
555 static_assert(is_taggable_v<T>, "static type violation");
556 // Direct construction of Tagged from address, without a type check, because
557 // we rather trust DirectHandle<T> to contain a T than include all the
558 // respective -inl.h headers for SLOW_DCHECKs.
559 SLOW_DCHECK(IsDereferenceAllowed());
560 return Tagged<T>(address());
561 }
562
563 // Consider declaring values that contain empty handles as
564 // MaybeDirectHandle to force validation before being used as handles.
565 V8_INLINE static const DirectHandle<T> null() { return DirectHandle<T>(); }
566
567 // Address equality.
568 bool equals(DirectHandle<T> other) const {
569 return address() == other.address();
570 }
571
572 // Sets this DirectHandle's value. This is equivalent to handle assignment,
573 // except for the check that is equivalent to that performed in
574 // Handle<T>::PatchValue.
575 // TODO(42203211): Calls to this method will eventually be replaced by direct
576 // handle assignments, when the migration to direct handles is complete.
577 void SetValue(Tagged<T> new_value) {
578 SLOW_DCHECK(obj_ != kTaggedNullAddress && IsDereferenceAllowed());
579 obj_ = new_value.ptr();
580 }
581
582 using MaybeType = MaybeDirectHandle<T>;
583
584 private:
585 // DirectHandles of different classes are allowed to access each other's
586 // obj_.
587 template <typename>
588 friend class DirectHandle;
589 // MaybeDirectHandle is allowed to access obj_.
590 template <typename>
591 friend class MaybeDirectHandle;
592 friend class DirectHandleUnchecked<T>;
593 // Casts are allowed to access obj_.
594 template <typename To, typename From>
595 friend inline DirectHandle<To> UncheckedCast(DirectHandle<From> value);
596
597 V8_INLINE explicit DirectHandle(Tagged<T> object);
598
599 explicit DirectHandle(no_checking_tag do_not_check)
600 : DirectHandleBase(kTaggedNullAddress, do_not_check) {}
601 explicit DirectHandle(const DirectHandle<T>& other,
602 no_checking_tag do_not_check)
603 : DirectHandleBase(other.obj_, do_not_check) {}
604};
