56// This class exists in order to solve a shutdown ordering problem.
57// The basic situation is that the process-global WasmEngine has, for each
58// Isolate that it knows about, a map from NativeModule to Script, using
59// WeakScriptHandles to make sure that the NativeModules, which are shared
60// across the process, don't keep the (Isolate-specific) Scripts alive.
61// In the other direction, the Scripts keep the NativeModule alive, IOW
62// usually the Scripts die first, and the WeakScriptHandles are cleared
63// before being freed.
64// In case of asm.js modules and in case of Isolate shutdown, it can happen
65// that the NativeModule dies first, so the WeakScriptHandles are no longer
66// needed and should be destroyed. That can only happen on the main thread of
67// the Isolate they belong to, whereas the last thread that releases a
68// NativeModule might be any other thread, so we post a
69// ClearWeakScriptHandleTask to that isolate's foreground task runner.
70// In case of Isolate shutdown at an inconvenient moment, this task runner can
71// destroy all waiting tasks; and *afterwards* global handles are freed, which
72// writes to the memory location backing the handle, so this bit of memory must
73// not be owned by (and die with) the ClearWeakScriptHandleTask.
74// The solution is this class here: its instances form a linked list owned by
75// the Isolate to which the referenced Scripts belong. Its name refers to the
76// fact that it stores global handles that used to have a purpose but are now
77// just waiting for the right thread to destroy them.
78// If the ClearWeakScriptHandleTask gets to run (i.e. in the regular case),
79// it destroys the weak global handle and then the WasmOrphanedGlobalHandle
80// container, removing it from the isolate's list.
81// If the ClearWeakScriptHandleTask is destroyed before it runs, the isolate's
82// list of WasmOrphanedGlobalHandles isn't modified, so the indirection cell
83// is still around when all remaining global handles are freed; nevertheless
84// it won't leak because the Isolate owns it and will free it.
85class WasmOrphanedGlobalHandle {
86 public:
87 WasmOrphanedGlobalHandle() = default;
88
89 void InitializeLocation(std::unique_ptr<Address*> location) {
90 location_ = std::move(location);
91 }
92
93 static void Destroy(WasmOrphanedGlobalHandle* that) {
94 // Destroy the global handle if it still exists.
95 Address** location = that->location_.get();
96 if (location) GlobalHandles::Destroy(*location);
97 that->location_.reset();
98 // Unlink and free the container.
99 *that->prev_ptr_ = that->next_;
100 if (that->next_ != nullptr) that->next_->prev_ptr_ = that->prev_ptr_;
101 // This function could be a non-static method, but then the next line
102 // would read "delete this", which is UB.
103 delete that;
104 }
105
106 private:
107 friend class WasmEngine;
108
109 // This is a doubly linked list with a twist: the {next_} pointer is just
110 // what you would expect, whereas {prev_ptr_} points at the slot inside
111 // the previous element that's pointing at the current element. The purpose
112 // of this design is to make it possible for the previous element to be
113 // the {Isolate::wasm_orphaned_handle_} field, without requiring any
114 // special-casing in the insert and delete operations.
115 WasmOrphanedGlobalHandle* next_ = nullptr;
116 WasmOrphanedGlobalHandle** prev_ptr_ = nullptr;
117 std::unique_ptr<Address*> location_;
118};
119
120// static
121WasmOrphanedGlobalHandle* WasmEngine::NewOrphanedGlobalHandle(
122 WasmOrphanedGlobalHandle** pointer) {
123 // No need for additional locking: this is only ever called indirectly
124 // from {WasmEngine::ClearWeakScriptHandle()}, which holds the engine-wide
125 // {mutex_}.
126 WasmOrphanedGlobalHandle* orphan = new WasmOrphanedGlobalHandle();
127 orphan->next_ = *pointer;
128 orphan->prev_ptr_ = pointer;
129 if (orphan->next_ != nullptr) orphan->next_->prev_ptr_ = &orphan->next_;
130 *pointer = orphan;
131 return orphan;
132}

134// static
135void WasmEngine::FreeAllOrphanedGlobalHandles(WasmOrphanedGlobalHandle* start) {
136 // This is meant to be called from ~Isolate, so we no longer care about
137 // maintaining invariants: the only task is to free memory to prevent leaks.
138 while (start != nullptr) {
139 WasmOrphanedGlobalHandle* next = start->next_;
140 delete start;
141 start = next;
142 }
143}

215class WeakScriptHandle {
216 public:
217 WeakScriptHandle(DirectHandle<Script> script, Isolate* isolate)
218 : script_id_(script->id()), isolate_(isolate) {
219 DCHECK(IsString(script->name()) || IsUndefined(script->name()));
220 if (IsString(script->name())) {
221 source_url_ = Cast<String>(script->name())->ToCString();
222 }
223 auto global_handle =
224 script->GetIsolate()->global_handles()->Create(*script);
225 location_ = std::make_unique<Address*>(global_handle.location());
226 GlobalHandles::MakeWeak(location_.get());
227 }
228
229 ~WeakScriptHandle() {
230 // Usually the destructor of this class is called after the weak callback,
231 // because the Script keeps the NativeModule alive. In that case,
232 // {location_} is already cleared, and there is nothing to do.
233 if (location_ == nullptr || *location_ == nullptr) return;
234 // For asm.js modules, the Script usually outlives the NativeModule.
235 // We must destroy the GlobalHandle before freeing the memory that's
236 // backing {location_}, so that when the Script does die eventually, there
237 // is no lingering weak GlobalHandle that would try to clear {location_}.
238 // We can't do that from arbitrary threads, so we must post a task to the
239 // main thread.
240 GetWasmEngine()->ClearWeakScriptHandle(isolate_, std::move(location_));
241 }
242
243 WeakScriptHandle(WeakScriptHandle&&) V8_NOEXCEPT = default;
244
245 Handle<Script> handle() const { return Handle<Script>(*location_); }
246
247 // Called by ~IsolateInfo. When the Isolate is shutting down, cleaning
248 // up properly is both no longer necessary and no longer safe to do.
249 void Clear() { location_.reset(); }
250
251 int script_id() const { return script_id_; }
252
253 const std::shared_ptr<const char[]>& source_url() const {
254 return source_url_;
255 }
256
257 private:
258 // Store the location in a unique_ptr so that its address stays the same even
259 // when this object is moved/copied.
260 std::unique_ptr<Address*> location_;
261
262 // Store the script ID independent of the weak handle, such that it's always
263 // available.
264 int script_id_;
265
266 // Similar for the source URL. We cannot dereference the Handle from arbitrary
267 // threads, but we need the URL available for code logging.
268 // The shared pointer is kept alive by unlogged code, even if this entry is
269 // collected in the meantime.
270 // TODO(chromium:1132260): Revisit this for huge URLs.
271 std::shared_ptr<const char[]> source_url_;
272
273 // The Isolate that the handled script belongs to.
274 Isolate* isolate_;
275};

285std::shared_ptr<NativeModule> NativeModuleCache::MaybeGetNativeModule(
286 ModuleOrigin origin, base::Vector<const uint8_t> wire_bytes,
287 const CompileTimeImports& compile_imports) {
288 if (!v8_flags.wasm_native_module_cache) return nullptr;
289 if (origin != kWasmOrigin) return nullptr;
290 base::MutexGuard lock(&mutex_);
291 size_t prefix_hash = PrefixHash(wire_bytes);
292 NativeModuleCache::Key key{prefix_hash, compile_imports, wire_bytes};
293 while (true) {
294 auto it = map_.find(key);
295 if (it == map_.end()) {
296 // Even though this exact key is not in the cache, there might be a
297 // matching prefix hash indicating that a streaming compilation is
298 // currently compiling a module with the same prefix. {OnFinishedStream}
299 // happens on the main thread too, so waiting for streaming compilation to
300 // finish would create a deadlock. Instead, compile the module twice and
301 // handle the conflict in {UpdateNativeModuleCache}.
302
303 // Insert a {nullopt} entry to let other threads know that this
304 // {NativeModule} is already being created on another thread.
305 [[maybe_unused]] auto [iterator, inserted] =
306 map_.emplace(key, std::nullopt);
307 DCHECK(inserted);
308 return nullptr;
309 }
310 if (it->second.has_value()) {
311 if (auto shared_native_module = it->second.value().lock()) {
312 DCHECK_EQ(
313 shared_native_module->compile_imports().compare(compile_imports),
314 0);
315 DCHECK_EQ(shared_native_module->wire_bytes(), wire_bytes);
316 return shared_native_module;
317 }
318 }
319 // TODO(11858): This deadlocks in predictable mode, because there is only a
320 // single thread.
321 cache_cv_.Wait(&mutex_);
322 }
323}

325bool NativeModuleCache::GetStreamingCompilationOwnership(
326 size_t prefix_hash, const CompileTimeImports& compile_imports) {
327 if (!v8_flags.wasm_native_module_cache) return true;
328 base::MutexGuard lock(&mutex_);
329 auto it = map_.lower_bound(Key{prefix_hash, compile_imports, {}});
330 if (it != map_.end() && it->first.prefix_hash == prefix_hash) {
331 DCHECK_IMPLIES(!it->first.bytes.empty(),
332 PrefixHash(it->first.bytes) == prefix_hash);
333 return false;
334 }
335 Key key{prefix_hash, compile_imports, {}};
336 DCHECK_EQ(0, map_.count(key));
337 map_.emplace(key, std::nullopt);
338 return true;
339}

350std::shared_ptr<NativeModule> NativeModuleCache::Update(
351 std::shared_ptr<NativeModule> native_module, bool error) {
352 DCHECK_NOT_NULL(native_module);
353 if (!v8_flags.wasm_native_module_cache) return native_module;
354 if (native_module->module()->origin != kWasmOrigin) return native_module;
355 base::Vector<const uint8_t> wire_bytes = native_module->wire_bytes();
356 DCHECK(!wire_bytes.empty());
357 size_t prefix_hash = PrefixHash(native_module->wire_bytes());
358 base::MutexGuard lock(&mutex_);
359 const CompileTimeImports& compile_imports = native_module->compile_imports();
360 map_.erase(Key{prefix_hash, compile_imports, {}});
361 const Key key{prefix_hash, compile_imports, wire_bytes};
362 auto it = map_.find(key);
363 if (it != map_.end()) {
364 if (it->second.has_value()) {
365 auto conflicting_module = it->second.value().lock();
366 if (conflicting_module != nullptr) {
367 DCHECK_EQ(conflicting_module->wire_bytes(), wire_bytes);
368 // This return might delete {native_module} if we were the last holder.
369 // That in turn can call {NativeModuleCache::Erase}, which takes the
370 // mutex. This is not a problem though, since the {MutexGuard} above is
371 // released before the {native_module}, per the definition order.
372 return conflicting_module;
373 }
374 }
375 map_.erase(it);
376 }
377 if (!error) {
378 // The key now points to the new native module's owned copy of the bytes,
379 // so that it stays valid until the native module is freed and erased from
380 // the map.
381 [[maybe_unused]] auto [iterator, inserted] = map_.emplace(
382 key, std::optional<std::weak_ptr<NativeModule>>(native_module));
383 DCHECK(inserted);
384 }
385 cache_cv_.NotifyAll();
386 return native_module;
387}

389void NativeModuleCache::Erase(NativeModule* native_module) {
390 if (!v8_flags.wasm_native_module_cache) return;
391 if (native_module->module()->origin != kWasmOrigin) return;
392 // Happens in some tests where bytes are set directly.
393 if (native_module->wire_bytes().empty()) return;
394 base::MutexGuard lock(&mutex_);
395 size_t prefix_hash = PrefixHash(native_module->wire_bytes());
396 map_.erase(Key{prefix_hash, native_module->compile_imports(),
397 native_module->wire_bytes()});
398 cache_cv_.NotifyAll();
399}

401// static
402size_t NativeModuleCache::PrefixHash(base::Vector<const uint8_t> wire_bytes) {
403 // Compute the hash as a combined hash of the sections up to the code section
404 // header, to mirror the way streaming compilation does it.
405 Decoder decoder(wire_bytes.begin(), wire_bytes.end());
406 decoder.consume_bytes(8, "module header");
407 SectionCode section_id = SectionCode::kUnknownSectionCode;
408 base::Hasher hasher;
409 while (decoder.ok() && decoder.more()) {
410 section_id = static_cast<SectionCode>(decoder.consume_u8());
411 uint32_t section_size = decoder.consume_u32v("section size");
412 if (section_id == SectionCode::kCodeSectionCode) {
413 hasher.Add(section_size);
414 break;
415 }
416 const uint8_t* payload_start = decoder.pc();
417 decoder.consume_bytes(section_size, "section payload");
418 hasher.AddRange(base::VectorOf(payload_start, section_size));
419 }
420 return hasher.hash();
421}

424 explicit CurrentGCInfo(int8_t gc_sequence_index)

533WasmEngine::~WasmEngine() {
534#ifdef V8_ENABLE_WASM_GDB_REMOTE_DEBUGGING
535 // Synchronize on the GDB-remote thread, if running.
536 gdb_server_.reset();
537#endif // V8_ENABLE_WASM_GDB_REMOTE_DEBUGGING
538
539 if (V8_UNLIKELY(v8_flags.print_wasm_offheap_memory_size)) {
540 PrintCurrentMemoryConsumptionEstimate();
541 }
542
543 // Free all modules that were kept alive for collecting PGO. This is to avoid
544 // memory leaks.
545 if (V8_UNLIKELY(native_modules_kept_alive_for_pgo)) {
546 delete native_modules_kept_alive_for_pgo;
547 }
548
549 operations_barrier_->CancelAndWait();
550
551 // All code should have been deleted already, but wrappers managed by the
552 // WasmImportWrapperCache are placed in {potentially_dead_code_} when they
553 // are no longer referenced, and we don't want to wait for the next
554 // Wasm Code GC cycle to remove them from that set.
555 for (WasmCode* code : potentially_dead_code_) {
556 code->DcheckRefCountIsOne();
557 // The actual instructions will get thrown out when the global
558 // WasmImportWrapperCache's {code_allocator_} frees its memory region.
559 // Here we just pacify LSan.
560 delete code;
561 }
562
563 // All AsyncCompileJobs have been canceled.
564 DCHECK(async_compile_jobs_.empty());
565 // All Isolates have been deregistered.
566 DCHECK(isolates_.empty());
567 // All NativeModules did die.
568 DCHECK(native_modules_.empty());
569 // Native module cache does not leak.
570 DCHECK(native_module_cache_.empty());
571}

598 compilation_id);
599 ModuleOrigin origin = language_mode == LanguageMode::kSloppy
600 ? kAsmJsSloppyOrigin
601 : kAsmJsStrictOrigin;
602 // TODO(leszeks): If we want asm.js in UKM, we should figure out a way to pass
603 // the context id in here.
604 v8::metrics::Recorder::ContextId context_id =
605 v8::metrics::Recorder::ContextId::Empty();
606 WasmDetectedFeatures detected_features;
607 ModuleResult result = DecodeWasmModule(
608 WasmEnabledFeatures::ForAsmjs(), bytes.as_vector(), false, origin,
609 isolate->counters(), isolate->metrics_recorder(), context_id,
610 DecodingMethod::kSync, &detected_features);
611 if (result.failed()) {
612 // This happens once in a while when we have missed some limit check
613 // in the asm parser. Output an error message to help diagnose, but crash.
614 std::cout << result.error().message();
615 UNREACHABLE();
616 }
617
618 result.value()->asm_js_offset_information =
619 std::make_unique<AsmJsOffsetInformation>(asm_js_offset_table_bytes);
620
621 // Transfer ownership of the WasmModule to the {Managed<WasmModule>} generated
622 // in {CompileToNativeModule}.
623 constexpr ProfileInformation* kNoProfileInformation = nullptr;
624 std::shared_ptr<NativeModule> native_module = CompileToNativeModule(
625 isolate, WasmEnabledFeatures::ForAsmjs(), detected_features,
626 CompileTimeImports{}, thrower, std::move(result).value(),
627 std::move(bytes), compilation_id, context_id, kNoProfileInformation);
628 if (!native_module) return {};
629
630 native_module->LogWasmCodes(isolate, *script);
631 {
632 // Register the script with the isolate. We do this unconditionally for
633 // consistency; it is in particular required for logging lazy-compiled code.
634 base::SpinningMutexGuard guard(&mutex_);
635 DCHECK_EQ(1, isolates_.count(isolate));
636 auto& scripts = isolates_[isolate]->scripts;
637 // If the same asm.js module is instantiated repeatedly, then we
638 // deduplicate the NativeModule, so the script exists already.
639 if (scripts.count(native_module.get()) == 0) {
640 scripts.emplace(native_module.get(), WeakScriptHandle(script, isolate));
641 }
642 }
643
644 return AsmWasmData::New(isolate, std::move(native_module), uses_bitset);
645}

657MaybeHandle<WasmModuleObject> WasmEngine::SyncCompile(
658 Isolate* isolate, WasmEnabledFeatures enabled_features,
659 CompileTimeImports compile_imports, ErrorThrower* thrower,
660 base::OwnedVector<const uint8_t> bytes) {
661 int compilation_id = next_compilation_id_.fetch_add(1);
662 TRACE_EVENT1("v8.wasm", "wasm.SyncCompile", "id", compilation_id);
663 v8::metrics::Recorder::ContextId context_id =
664 isolate->GetOrRegisterRecorderContextId(isolate->native_context());
665 std::shared_ptr<WasmModule> module;
666 WasmDetectedFeatures detected_features;
667 {
668 // Normally modules are validated in {CompileToNativeModule} but in jitless
669 // mode the only opportunity of validatiom is during decoding.
670 bool validate_module = v8_flags.wasm_jitless;
671 ModuleResult result = DecodeWasmModule(
672 enabled_features, bytes.as_vector(), validate_module, kWasmOrigin,
673 isolate->counters(), isolate->metrics_recorder(), context_id,
674 DecodingMethod::kSync, &detected_features);
675 if (result.failed()) {
676 thrower->CompileFailed(result.error());
677 return {};
678 }
679 module = std::move(result).value();
680 if (WasmError error =
681 ValidateAndSetBuiltinImports(module.get(), bytes.as_vector(),
682 compile_imports, &detected_features)) {
683 thrower->CompileError("%s @+%u", error.message().c_str(), error.offset());
684 return {};
685 }
686 }
687
688 // If experimental PGO via files is enabled, load profile information now.
689 std::unique_ptr<ProfileInformation> pgo_info;
690 if (V8_UNLIKELY(v8_flags.experimental_wasm_pgo_from_file)) {
691 pgo_info = LoadProfileFromFile(module.get(), bytes.as_vector());
692 }
693
694 // Transfer ownership of the WasmModule to the {Managed<WasmModule>} generated
695 // in {CompileToNativeModule}.
696 std::shared_ptr<NativeModule> native_module = CompileToNativeModule(
697 isolate, enabled_features, detected_features, std::move(compile_imports),
698 thrower, std::move(module), std::move(bytes), compilation_id, context_id,
699 pgo_info.get());
700 if (!native_module) return {};
701
702#ifdef DEBUG
703 // Ensure that code GC will check this isolate for live code.
704 {
705 base::SpinningMutexGuard lock(&mutex_);
706 DCHECK_EQ(1, isolates_.count(isolate));
707 DCHECK_EQ(1, isolates_[isolate]->native_modules.count(native_module.get()));
708 DCHECK_EQ(1, native_modules_.count(native_module.get()));
709 DCHECK_EQ(1, native_modules_[native_module.get()]->isolates.count(isolate));
710 }
711#endif
712
713 constexpr base::Vector<const char> kNoSourceUrl;
714 DirectHandle<Script> script =
715 GetOrCreateScript(isolate, native_module, kNoSourceUrl);
716
717 native_module->LogWasmCodes(isolate, *script);
718
719 // Create the compiled module object and populate with compiled functions
720 // and information needed at instantiation time. This object needs to be
721 // serializable. Instantiation may occur off a deserialized version of this
722 // object.
723 Handle<WasmModuleObject> module_object =
724 WasmModuleObject::New(isolate, std::move(native_module), script);
725
726 // Finish the Wasm script now and make it public to the debugger.
727 isolate->debug()->OnAfterCompile(script);
728 return module_object;
729}

741 Isolate* isolate, std::unique_ptr<InstantiationResultResolver> resolver,
742 Handle<WasmModuleObject> module_object, MaybeHandle<JSReceiver> imports) {
743 ErrorThrower thrower(isolate, "WebAssembly.instantiate()");
744 TRACE_EVENT0("v8.wasm", "wasm.AsyncInstantiate");
745 // Instantiate a TryCatch so that caught exceptions won't progagate out.
746 // They will still be set as exceptions on the isolate.
747 // TODO(clemensb): Avoid TryCatch, use Execution::TryCall internally to invoke
748 // start function and report thrown exception explicitly via out argument.

758 return;
759 }
760

762 thrower.Reset();
763 if (isolate->is_execution_terminating()) return;

765 // We have to move the exception to the promise chain.
766 Handle<Object> exception(isolate->exception(), isolate);
767 DCHECK(!IsHole(*exception));
768 isolate->clear_exception();
769 resolver->OnInstantiationFailed(exception);
770 } else {
771 DCHECK(thrower.error());
772 resolver->OnInstantiationFailed(thrower.Reify());
773 }

776void WasmEngine::AsyncCompile(
777 Isolate* isolate, WasmEnabledFeatures enabled,
778 CompileTimeImports compile_imports,
779 std::shared_ptr<CompilationResultResolver> resolver,
780 base::OwnedVector<const uint8_t> bytes,
781 const char* api_method_name_for_errors) {
782 int compilation_id = next_compilation_id_.fetch_add(1);
783 TRACE_EVENT1("v8.wasm", "wasm.AsyncCompile", "id", compilation_id);
784
785 if (!v8_flags.wasm_async_compilation || v8_flags.wasm_jitless) {
786 // Asynchronous compilation disabled; fall back on synchronous compilation.
787 ErrorThrower thrower(isolate, api_method_name_for_errors);
788 MaybeHandle<WasmModuleObject> module_object;
789 module_object = SyncCompile(isolate, enabled, std::move(compile_imports),
790 &thrower, std::move(bytes));
791 if (thrower.error()) {
792 resolver->OnCompilationFailed(thrower.Reify());
793 return;
794 }
795 Handle<WasmModuleObject> module = module_object.ToHandleChecked();
796 resolver->OnCompilationSucceeded(module);
797 return;
798 }
799
800 if (v8_flags.wasm_test_streaming) {
801 std::shared_ptr<StreamingDecoder> streaming_decoder =
802 StartStreamingCompilation(isolate, enabled, std::move(compile_imports),
803 handle(isolate->context(), isolate),
804 api_method_name_for_errors,
805 std::move(resolver));
806
807 auto* rng = isolate->random_number_generator();
808 base::SmallVector<base::Vector<const uint8_t>, 16> ranges;
809 if (!bytes.empty()) ranges.push_back(bytes.as_vector());
810 // Split into up to 16 ranges (2^4).
811 for (int round = 0; round < 4; ++round) {
812 for (auto it = ranges.begin(); it != ranges.end(); ++it) {
813 auto range = *it;
814 if (range.size() < 2 || !rng->NextBool()) continue; // Do not split.
815 // Choose split point within [1, range.size() - 1].
816 static_assert(kV8MaxWasmModuleSize <= kMaxInt);
817 size_t split_point =
818 1 + rng->NextInt(static_cast<int>(range.size() - 1));
819 // Insert first sub-range *before* {it} and make {it} point after it.
820 it = ranges.insert(it, range.SubVector(0, split_point)) + 1;
821 *it = range.SubVectorFrom(split_point);
822 }
823 }
824 for (auto range : ranges) {
825 streaming_decoder->OnBytesReceived(range);
826 }
827 streaming_decoder->Finish();
828 return;
829 }
830
831 AsyncCompileJob* job = CreateAsyncCompileJob(
832 isolate, enabled, std::move(compile_imports), std::move(bytes),
833 isolate->native_context(), api_method_name_for_errors,
834 std::move(resolver), compilation_id);
835 job->Start();
836}

869void WasmEngine::EnterDebuggingForIsolate(Isolate* isolate) {
870 if (v8_flags.wasm_jitless) return;
871
872 std::vector<std::shared_ptr<NativeModule>> native_modules;
873 // {mutex_} gets taken both here and in {RemoveCompiledCode} in
874 // {AddPotentiallyDeadCode}. Therefore {RemoveCompiledCode} has to be
875 // called outside the lock.
876 {
877 base::SpinningMutexGuard lock(&mutex_);
878 if (isolates_[isolate]->keep_in_debug_state) return;
879 isolates_[isolate]->keep_in_debug_state = true;
880 for (auto* native_module : isolates_[isolate]->native_modules) {
881 DCHECK_EQ(1, native_modules_.count(native_module));
882 if (auto shared_ptr = native_modules_[native_module]->weak_ptr.lock()) {
883 native_modules.emplace_back(std::move(shared_ptr));
884 }
885 native_module->SetDebugState(kDebugging);
886 }
887 }
888 WasmCodeRefScope ref_scope;
889 for (auto& native_module : native_modules) {
890 native_module->RemoveCompiledCode(
891 NativeModule::RemoveFilter::kRemoveNonDebugCode);
892 }

894
895void WasmEngine::LeaveDebuggingForIsolate(Isolate* isolate) {
896 // Only trigger recompilation after releasing the mutex, otherwise we risk
897 // deadlocks because of lock inversion. The bool tells whether the module
898 // needs recompilation for tier up.
899 std::vector<std::pair<std::shared_ptr<NativeModule>, bool>> native_modules;
900 {
901 base::SpinningMutexGuard lock(&mutex_);
902 isolates_[isolate]->keep_in_debug_state = false;
903 auto can_remove_debug_code = [this](NativeModule* native_module) {
904 DCHECK_EQ(1, native_modules_.count(native_module));
905 for (auto* isolate : native_modules_[native_module]->isolates) {
906 DCHECK_EQ(1, isolates_.count(isolate));
907 if (isolates_[isolate]->keep_in_debug_state) return false;
908 }
909 return true;
910 };
911 for (auto* native_module : isolates_[isolate]->native_modules) {
912 DCHECK_EQ(1, native_modules_.count(native_module));
913 auto shared_ptr = native_modules_[native_module]->weak_ptr.lock();
914 if (!shared_ptr) continue; // The module is not used any more.
915 if (!native_module->IsInDebugState()) continue;
916 // Only start tier-up if no other isolate needs this module in tiered
917 // down state.
918 bool remove_debug_code = can_remove_debug_code(native_module);
919 if (remove_debug_code) native_module->SetDebugState(kNotDebugging);
920 native_modules.emplace_back(std::move(shared_ptr), remove_debug_code);
921 }
922 }
923 for (auto& entry : native_modules) {
924 auto& native_module = entry.first;
925 bool remove_debug_code = entry.second;
926 // Remove all breakpoints set by this isolate.
927 if (native_module->HasDebugInfo()) {
928 native_module->GetDebugInfo()->RemoveIsolate(isolate);
929 }
930 if (remove_debug_code) {
931 WasmCodeRefScope ref_scope;
932 native_module->RemoveCompiledCode(
933 NativeModule::RemoveFilter::kRemoveDebugCode);
934 }
935 }

939Handle<Script> CreateWasmScript(Isolate* isolate,
940 std::shared_ptr<NativeModule> native_module,
941 base::Vector<const char> source_url) {
942 base::Vector<const uint8_t> wire_bytes = native_module->wire_bytes();
943
944 // The source URL of the script is
945 // - the original source URL if available (from the streaming API),
946 // - wasm://wasm/<module name>-<hash> if a module name has been set, or
947 // - wasm://wasm/<hash> otherwise.
948 const WasmModule* module = native_module->module();
949 Handle<String> url_str;
950 if (!source_url.empty()) {
951 url_str = isolate->factory()
952 ->NewStringFromUtf8(source_url, AllocationType::kOld)
953 .ToHandleChecked();
954 } else {
955 // Limit the printed hash to 8 characters.
956 uint32_t hash = static_cast<uint32_t>(GetWireBytesHash(wire_bytes));
957 base::EmbeddedVector<char, 32> buffer;
958 if (module->name.is_empty()) {
959 // Build the URL in the form "wasm://wasm/<hash>".
960 int url_len = SNPrintF(buffer, "wasm://wasm/%08x", hash);
961 DCHECK(url_len >= 0 && url_len < buffer.length());
962 url_str = isolate->factory()
963 ->NewStringFromUtf8(buffer.SubVector(0, url_len),
964 AllocationType::kOld)
965 .ToHandleChecked();
966 } else {
967 // Build the URL in the form "wasm://wasm/<module name>-<hash>".
968 int hash_len = SNPrintF(buffer, "-%08x", hash);
969 DCHECK(hash_len >= 0 && hash_len < buffer.length());
970 Handle<String> prefix =
971 isolate->factory()->NewStringFromStaticChars("wasm://wasm/");
972 Handle<String> module_name =
973 WasmModuleObject::ExtractUtf8StringFromModuleBytes(
974 isolate, wire_bytes, module->name, kNoInternalize);
975 Handle<String> hash_str =
976 isolate->factory()
977 ->NewStringFromUtf8(buffer.SubVector(0, hash_len))
978 .ToHandleChecked();
979 // Concatenate the three parts.
980 url_str = isolate->factory()
981 ->NewConsString(prefix, module_name)
982 .ToHandleChecked();
983 url_str = isolate->factory()
984 ->NewConsString(url_str, hash_str)
985 .ToHandleChecked();
986 }
987 }
988 DirectHandle<PrimitiveHeapObject> source_map_url =
989 isolate->factory()->undefined_value();
990 if (module->debug_symbols[WasmDebugSymbols::Type::SourceMap].type !=
991 WasmDebugSymbols::Type::None) {
992 auto source_map_symbols =
993 module->debug_symbols[WasmDebugSymbols::Type::SourceMap];
994 base::Vector<const char> external_url =
995 ModuleWireBytes(wire_bytes)
996 .GetNameOrNull(source_map_symbols.external_url);
997 MaybeDirectHandle<String> src_map_str =
998 isolate->factory()->NewStringFromUtf8(external_url,
999 AllocationType::kOld);
1000 source_map_url = src_map_str.ToHandleChecked();
1001 }
1002
1003 // Use the given shared {NativeModule}, but increase its reference count by
1004 // allocating a new {Managed<T>} that the {Script} references.
1005 size_t code_size_estimate = native_module->committed_code_space();
1006 size_t memory_estimate =
1007 code_size_estimate +
1008 wasm::WasmCodeManager::EstimateNativeModuleMetaDataSize(module);
1009 DirectHandle<Managed<wasm::NativeModule>> managed_native_module =
1010 Managed<wasm::NativeModule>::From(isolate, memory_estimate,
1011 std::move(native_module));
1012
1013 Handle<Script> script =
1014 isolate->factory()->NewScript(isolate->factory()->undefined_value());
1015 {
1016 DisallowGarbageCollection no_gc;
1017 Tagged<Script> raw_script = *script;
1018 raw_script->set_compilation_state(Script::CompilationState::kCompiled);
1019 raw_script->set_context_data(isolate->native_context()->debug_context_id());
1020 raw_script->set_name(*url_str);
1021 raw_script->set_type(Script::Type::kWasm);
1022 raw_script->set_source_mapping_url(*source_map_url);
1023 raw_script->set_line_ends(ReadOnlyRoots(isolate).empty_fixed_array(),
1024 SKIP_WRITE_BARRIER);
1025 raw_script->set_wasm_managed_native_module(*managed_native_module);
1026 raw_script->set_wasm_breakpoint_infos(
1027 ReadOnlyRoots(isolate).empty_fixed_array(), SKIP_WRITE_BARRIER);
1028 raw_script->set_wasm_weak_instance_list(
1029 ReadOnlyRoots(isolate).empty_weak_array_list(), SKIP_WRITE_BARRIER);
1030
1031 // For correct exception handling (in particular, the onunhandledrejection
1032 // callback), we must set the origin options from the nearest calling JS
1033 // frame.
1034 // Considering all Wasm modules as shared across origins isn't a privacy
1035 // issue, because in order to instantiate and use them, a site needs to
1036 // already have access to their wire bytes anyway.
1037 static constexpr bool kIsSharedCrossOrigin = true;
1038 static constexpr bool kIsOpaque = false;
1039 static constexpr bool kIsWasm = true;
1040 static constexpr bool kIsModule = false;
1041 raw_script->set_origin_options(ScriptOriginOptions(
1042 kIsSharedCrossOrigin, kIsOpaque, kIsWasm, kIsModule));
1043 }
1044
1045 return script;
1046}

1076std::pair<size_t, size_t> WasmEngine::FlushLiftoffCode() {
1077 // Keep the NativeModules alive until after the destructor of the
1078 // `WasmCodeRefScope`, which still needs to access the code and the
1079 // NativeModule.
1080 std::vector<std::shared_ptr<NativeModule>> native_modules_with_dead_code;
1081 WasmCodeRefScope ref_scope;
1082 base::SpinningMutexGuard guard(&mutex_);
1083 size_t removed_code_size = 0;
1084 size_t removed_metadata_size = 0;
1085 for (auto& [native_module, info] : native_modules_) {
1086 std::shared_ptr<NativeModule> shared = info->weak_ptr.lock();
1087 if (!shared) continue; // The NativeModule is dying anyway.
1088 auto [code_size, metadata_size] = native_module->RemoveCompiledCode(
1089 NativeModule::RemoveFilter::kRemoveLiftoffCode);
1090 DCHECK_EQ(code_size == 0, metadata_size == 0);
1091 if (code_size == 0) continue;
1092 native_modules_with_dead_code.emplace_back(std::move(shared));
1093 removed_code_size += code_size;
1094 removed_metadata_size += metadata_size;
1095 }
1096 return {removed_code_size, removed_metadata_size};
1097}

1195void WasmEngine::DeleteCompileJobsOnIsolate(Isolate* isolate) {
1196 // Under the mutex get all jobs to delete. Then delete them without holding
1197 // the mutex, such that deletion can reenter the WasmEngine.
1198 std::vector<std::unique_ptr<AsyncCompileJob>> jobs_to_delete;
1199 std::vector<std::weak_ptr<NativeModule>> modules_in_isolate;
1200 {
1201 base::SpinningMutexGuard guard(&mutex_);
1202 for (auto it = async_compile_jobs_.begin();
1203 it != async_compile_jobs_.end();) {
1204 if (it->first->isolate() != isolate) {
1205 ++it;
1206 continue;
1207 }
1208 jobs_to_delete.push_back(std::move(it->second));
1209 it = async_compile_jobs_.erase(it);
1210 }
1211 DCHECK_EQ(1, isolates_.count(isolate));
1212 auto* isolate_info = isolates_[isolate].get();
1213 for (auto* native_module : isolate_info->native_modules) {
1214 DCHECK_EQ(1, native_modules_.count(native_module));
1215 modules_in_isolate.emplace_back(native_modules_[native_module]->weak_ptr);
1216 }
1217 }
1218
1219 // All modules that have not finished initial compilation yet cannot be
1220 // shared with other isolates. Hence we cancel their compilation. In
1221 // particular, this will cancel wrapper compilation which is bound to this
1222 // isolate (this would be a UAF otherwise).
1223 for (auto& weak_module : modules_in_isolate) {
1224 if (auto shared_module = weak_module.lock()) {
1225 shared_module->compilation_state()->CancelInitialCompilation();
1226 }
1227 }

1230void WasmEngine::AddIsolate(Isolate* isolate) {
1231 const bool log_code = WasmCode::ShouldBeLogged(isolate);
1232 // Create the IsolateInfo.
1233 {
1234 // Create the IsolateInfo outside the mutex to reduce the size of the
1235 // critical section and to avoid lock-order-inversion issues.
1236 auto isolate_info = std::make_unique<IsolateInfo>(isolate, log_code);
1237 base::SpinningMutexGuard guard(&mutex_);
1238 DCHECK_EQ(0, isolates_.count(isolate));
1239 isolates_.emplace(isolate, std::move(isolate_info));
1240 }
1241 if (log_code) {
1242 // Log existing wrappers (which are shared across isolates).
1243 GetWasmImportWrapperCache()->LogForIsolate(isolate);
1244 }
1245
1246 // Install sampling GC callback.
1247 // TODO(v8:7424): For now we sample module sizes in a GC callback. This will
1248 // bias samples towards apps with high memory pressure. We should switch to
1249 // using sampling based on regular intervals independent of the GC.
1250 auto callback = [](v8::Isolate* v8_isolate, v8::GCType type,
1251 v8::GCCallbackFlags flags, void* data) {
1252 Isolate* isolate = reinterpret_cast<Isolate*>(v8_isolate);
1253 Counters* counters = isolate->counters();
1254 WasmEngine* engine = GetWasmEngine();
1255 {
1256 base::SpinningMutexGuard lock(&engine->mutex_);
1257 DCHECK_EQ(1, engine->isolates_.count(isolate));
1258 for (auto* native_module : engine->isolates_[isolate]->native_modules) {
1259 native_module->SampleCodeSize(counters);
1260 }
1261 }
1262 // Also sample overall metadata size (this includes the metadata size of
1263 // individual NativeModules; we are summing that up twice, which could be
1264 // improved performance-wise).
1265 // The engine-wide metadata also includes global storage e.g. for the type
1266 // canonicalizer.
1267 Histogram* metadata_histogram = counters->wasm_engine_metadata_size_kb();
1268 if (metadata_histogram->Enabled()) {
1269 size_t engine_meta_data = engine->EstimateCurrentMemoryConsumption();
1270 metadata_histogram->AddSample(static_cast<int>(engine_meta_data / KB));
1271 }
1272 };
1273 isolate->heap()->AddGCEpilogueCallback(callback, v8::kGCTypeMarkSweepCompact,
1274 nullptr);
1275
1276#ifdef V8_ENABLE_WASM_GDB_REMOTE_DEBUGGING
1277 if (gdb_server_) {
1278 gdb_server_->AddIsolate(isolate);
1279 }
1280#endif // V8_ENABLE_WASM_GDB_REMOTE_DEBUGGING
1281}

1283void WasmEngine::RemoveIsolate(Isolate* isolate) {
1284#ifdef V8_ENABLE_WASM_GDB_REMOTE_DEBUGGING
1285 if (gdb_server_) {
1286 gdb_server_->RemoveIsolate(isolate);
1287 }
1288#endif // V8_ENABLE_WASM_GDB_REMOTE_DEBUGGING
1289
1290 // Keep a WasmCodeRefScope which dies after the {mutex_} is released, to avoid
1291 // deadlock when code actually dies, as that requires taking the {mutex_}.
1292 // Also, keep the NativeModules themselves alive. The isolate is shutting
1293 // down, so the heap will not do that any more.
1294 std::map<NativeModule*, std::shared_ptr<NativeModule>>
1295 native_modules_with_code_to_log;
1296 WasmCodeRefScope code_ref_scope_for_dead_code;
1297
1298 base::SpinningMutexGuard guard(&mutex_);
1299
1300 // Lookup the IsolateInfo; do not remove it yet (that happens below).
1301 auto isolates_it = isolates_.find(isolate);
1302 DCHECK_NE(isolates_.end(), isolates_it);
1303 IsolateInfo* isolate_info = isolates_it->second.get();
1304
1305 // Remove the isolate from the per-native-module info, and other cleanup.
1306 for (auto* native_module : isolate_info->native_modules) {
1307 DCHECK_EQ(1, native_modules_.count(native_module));
1308 NativeModuleInfo* native_module_info =
1309 native_modules_.find(native_module)->second.get();
1310
1311 // Check that the {NativeModule::log_code_} field has the expected value,
1312 // and update if the dying isolate was the last one with code logging
1313 // enabled.
1314 auto has_isolate_with_code_logging = [this, native_module_info] {
1315 return std::any_of(native_module_info->isolates.begin(),
1316 native_module_info->isolates.end(),
1317 [this](Isolate* isolate) {
1318 return isolates_.find(isolate)->second->log_codes;
1319 });
1320 };
1321 DCHECK_EQ(native_module->log_code(), has_isolate_with_code_logging());
1322 DCHECK_EQ(1, native_module_info->isolates.count(isolate));
1323 native_module_info->isolates.erase(isolate);
1324 if (native_module->log_code() && !has_isolate_with_code_logging()) {
1325 DisableCodeLogging(native_module);
1326 }
1327
1328 // Remove any debug code and other info for this isolate.
1329 if (native_module->HasDebugInfo()) {
1330 native_module->GetDebugInfo()->RemoveIsolate(isolate);
1331 }
1332 }
1333
1334 // Abort any outstanding GC.
1335 if (current_gc_info_) {
1336 if (RemoveIsolateFromCurrentGC(isolate)) PotentiallyFinishCurrentGC();
1337 }
1338
1339 // Clear the {code_to_log} vector.
1340 for (auto& [script_id, code_to_log] : isolate_info->code_to_log) {
1341 for (WasmCode* code : code_to_log.code) {
1342 if (code->native_module() == nullptr) {
1343 // Wrapper code objects have neither Script nor NativeModule.
1344 DCHECK_EQ(script_id, -1);
1345 } else if (!native_modules_with_code_to_log.count(
1346 code->native_module())) {
1347 std::shared_ptr<NativeModule> shared_native_module =
1348 native_modules_[code->native_module()]->weak_ptr.lock();
1349 if (!shared_native_module) {
1350 // The module is dying already; there's no need to decrement the ref
1351 // count and add the code to the WasmCodeRefScope.
1352 continue;
1353 }
1354 native_modules_with_code_to_log.insert(std::make_pair(
1355 code->native_module(), std::move(shared_native_module)));
1356 }
1357 // Keep a reference in the {code_ref_scope_for_dead_code} such that the
1358 // code cannot become dead immediately.
1359 WasmCodeRefScope::AddRef(code);
1360 code->DecRefOnLiveCode();
1361 }
1362 }
1363 isolate_info->code_to_log.clear();
1364
1365 // Finally remove the {IsolateInfo} for this isolate.
1366 isolates_.erase(isolates_it);
1367}

1369void WasmEngine::LogCode(base::Vector<WasmCode*> code_vec) {
1370 if (code_vec.empty()) return;
1371 NativeModule* native_module = code_vec[0]->native_module();
1372 if (!native_module->log_code()) return;
1373 using TaskToSchedule =
1374 std::pair<std::shared_ptr<v8::TaskRunner>, std::unique_ptr<LogCodesTask>>;
1375 std::vector<TaskToSchedule> to_schedule;
1376 {
1377 base::SpinningMutexGuard guard(&mutex_);
1378 DCHECK_EQ(1, native_modules_.count(native_module));
1379 NativeModuleInfo* native_module_info =
1380 native_modules_.find(native_module)->second.get();
1381 for (Isolate* isolate : native_module_info->isolates) {
1382 DCHECK_EQ(1, isolates_.count(isolate));
1383 IsolateInfo* info = isolates_[isolate].get();
1384 if (info->log_codes == false) continue;
1385
1386 auto script_it = info->scripts.find(native_module);
1387 // If the script does not yet exist, logging will happen later. If the
1388 // weak handle is cleared already, we also don't need to log any more.
1389 if (script_it == info->scripts.end()) continue;
1390
1391 // If there is no code scheduled to be logged already in that isolate,
1392 // then schedule a new task and also set an interrupt to log the newly
1393 // added code as soon as possible.
1394 if (info->code_to_log.empty()) {
1395 isolate->stack_guard()->RequestLogWasmCode();
1396 to_schedule.emplace_back(info->foreground_task_runner,
1397 std::make_unique<LogCodesTask>(isolate));
1398 }
1399
1400 WeakScriptHandle& weak_script_handle = script_it->second;
1401 auto& log_entry = info->code_to_log[weak_script_handle.script_id()];
1402 if (!log_entry.source_url) {
1403 log_entry.source_url = weak_script_handle.source_url();
1404 }
1405 log_entry.code.insert(log_entry.code.end(), code_vec.begin(),
1406 code_vec.end());
1407
1408 // Increment the reference count for the added {log_entry.code} entries.
1409 for (WasmCode* code : code_vec) {
1410 DCHECK_EQ(native_module, code->native_module());
1411 code->IncRef();
1412 }
1413 }
1414 }
1415 for (auto& [runner, task] : to_schedule) {
1416 runner->PostTask(std::move(task));
1417 }
1418}

1420bool WasmEngine::LogWrapperCode(WasmCode* code) {
1421 // Wrappers don't belong to any particular NativeModule.
1422 DCHECK_NULL(code->native_module());
1423 // Fast path:
1424 if (!num_modules_with_code_logging_.load(std::memory_order_relaxed)) {
1425 return false;
1426 }
1427
1428 using TaskToSchedule =
1429 std::pair<std::shared_ptr<v8::TaskRunner>, std::unique_ptr<LogCodesTask>>;
1430 std::vector<TaskToSchedule> to_schedule;
1431 bool did_trigger_code_logging = false;
1432 {
1433 base::SpinningMutexGuard guard(&mutex_);
1434 for (const auto& entry : isolates_) {
1435 Isolate* isolate = entry.first;
1436 IsolateInfo* info = entry.second.get();
1437 if (info->log_codes == false) continue;
1438 did_trigger_code_logging = true;
1439
1440 // If this is the first code to log in that isolate, request an interrupt
1441 // to log the newly added code as soon as possible.
1442 if (info->code_to_log.empty()) {
1443 isolate->stack_guard()->RequestLogWasmCode();
1444 to_schedule.emplace_back(info->foreground_task_runner,
1445 std::make_unique<LogCodesTask>(isolate));
1446 }
1447
1448 constexpr int kNoScriptId = -1;
1449 auto& log_entry = info->code_to_log[kNoScriptId];
1450 log_entry.code.push_back(code);
1451
1452 // Increment the reference count for the added {log_entry.code} entry.
1453 // TODO(jkummerow): It might be nice to have a custom smart pointer
1454 // that manages updating the refcount for the WasmCode it holds.
1455 code->IncRef();
1456 }
1457 DCHECK_EQ(did_trigger_code_logging, num_modules_with_code_logging_.load(
1458 std::memory_order_relaxed) > 0);
1459 }
1460 for (auto& [runner, task] : to_schedule) {
1461 runner->PostTask(std::move(task));
1462 }
1463
1464 return did_trigger_code_logging;
1465}

1511void WasmEngine::LogOutstandingCodesForIsolate(Isolate* isolate) {
1512 // Under the mutex, get the vector of wasm code to log. Then log and decrement
1513 // the ref count without holding the mutex.
1514 std::unordered_map<int, IsolateInfo::CodeToLogPerScript> code_to_log_map;
1515 {
1516 base::SpinningMutexGuard guard(&mutex_);
1517 DCHECK_EQ(1, isolates_.count(isolate));
1518 code_to_log_map.swap(isolates_[isolate]->code_to_log);

1523

1525 for (auto& [script_id, code_to_log] : code_to_log_map) {
1526 for (WasmCode* code : code_to_log.code) {
1527 if (should_log) {
1528 const char* source_url = code_to_log.source_url.get();
1529 // The source URL can be empty for eval()'ed scripts.
1530 if (!source_url) source_url = "";
1531 code->LogCode(isolate, source_url, script_id);
1532 }
1533 }
1534 WasmCode::DecrementRefCount(base::VectorOf(code_to_log.code));
1535 }

1686void WasmEngine::FreeNativeModule(NativeModule* native_module) {
1687 base::SpinningMutexGuard guard(&mutex_);
1688 auto module = native_modules_.find(native_module);
1689 DCHECK_NE(native_modules_.end(), module);
1690 auto part_of_native_module = [native_module](WasmCode* code) {
1691 return code->native_module() == native_module;
1692 };
1693 for (Isolate* isolate : module->second->isolates) {
1694 DCHECK_EQ(1, isolates_.count(isolate));
1695 IsolateInfo* info = isolates_[isolate].get();
1696 DCHECK_EQ(1, info->native_modules.count(native_module));
1697 info->native_modules.erase(native_module);
1698 info->scripts.erase(native_module);
1699
1700 // Flush the Wasm code lookup cache, since it may refer to some
1701 // code within native modules that we are going to release (if a
1702 // Managed<wasm::NativeModule> object is no longer referenced).
1703 GetWasmCodeManager()->FlushCodeLookupCache(isolate);
1704
1705 // If there are {WasmCode} objects of the deleted {NativeModule}
1706 // outstanding to be logged in this isolate, remove them. Decrementing the
1707 // ref count is not needed, since the {NativeModule} dies anyway.
1708 for (auto& log_entry : info->code_to_log) {
1709 std::vector<WasmCode*>& code = log_entry.second.code;
1710 auto new_end =
1711 std::remove_if(code.begin(), code.end(), part_of_native_module);
1712 code.erase(new_end, code.end());
1713 }
1714 // Now remove empty entries in {code_to_log}.
1715 for (auto it = info->code_to_log.begin(), end = info->code_to_log.end();
1716 it != end;) {
1717 if (it->second.code.empty()) {
1718 it = info->code_to_log.erase(it);
1719 } else {
1720 ++it;
1721 }
1722 }
1723 }
1724 // If there is a GC running which has references to code contained in the
1725 // deleted {NativeModule}, remove those references.
1726 if (current_gc_info_) {
1727 for (auto it = current_gc_info_->dead_code.begin(),
1728 end = current_gc_info_->dead_code.end();
1729 it != end;) {
1730 if ((*it)->native_module() == native_module) {
1731 it = current_gc_info_->dead_code.erase(it);
1732 } else {
1733 ++it;
1734 }
1735 }
1736 TRACE_CODE_GC("Native module %p died, reducing dead code objects to %zu.\n",
1737 native_module, current_gc_info_->dead_code.size());
1738 }
1739 // If any code objects are currently tracked as dead or near-dead, remove
1740 // references belonging to the NativeModule that's being deleted.
1741 std::erase_if(dead_code_, part_of_native_module);
1742 std::erase_if(potentially_dead_code_, part_of_native_module);
1743
1744 if (native_module->log_code()) DisableCodeLogging(native_module);
1745
1746 native_module_cache_.Erase(native_module);
1747 native_modules_.erase(module);
1748}

1793 wasm::WasmCodeRefScope code_ref_scope;

1825bool WasmEngine::AddPotentiallyDeadCode(WasmCode* code) {
1826 base::SpinningMutexGuard guard(&mutex_);
1827 if (dead_code_.contains(code)) return false; // Code is already dead.
1828 auto added = potentially_dead_code_.insert(code);
1829 if (!added.second) return false; // An entry already existed.
1830 new_potentially_dead_code_size_ += code->instructions().size();
1831 if (v8_flags.wasm_code_gc) {
1832 // Trigger a GC if 64kB plus 10% of committed code are potentially dead.
1833 size_t dead_code_limit =
1834 v8_flags.stress_wasm_code_gc
1835 ? 0
1836 : 64 * KB + GetWasmCodeManager()->committed_code_space() / 10;
1837 if (new_potentially_dead_code_size_ > dead_code_limit) {
1838 bool inc_gc_count =
1839 num_code_gcs_triggered_ < std::numeric_limits<int8_t>::max();
1840 if (current_gc_info_ == nullptr) {
1841 if (inc_gc_count) ++num_code_gcs_triggered_;
1842 TRACE_CODE_GC(
1843 "Triggering GC (potentially dead: %zu bytes; limit: %zu bytes).\n",
1844 new_potentially_dead_code_size_, dead_code_limit);
1845 TriggerGC(num_code_gcs_triggered_);
1846 } else if (current_gc_info_->next_gc_sequence_index == 0) {
1847 if (inc_gc_count) ++num_code_gcs_triggered_;
1848 TRACE_CODE_GC(
1849 "Scheduling another GC after the current one (potentially dead: "
1850 "%zu bytes; limit: %zu bytes).\n",
1851 new_potentially_dead_code_size_, dead_code_limit);
1852 current_gc_info_->next_gc_sequence_index = num_code_gcs_triggered_;
1853 DCHECK_NE(0, current_gc_info_->next_gc_sequence_index);
1854 }
1855 }
1856 }
1857 return true;
1858}

1926void WasmEngine::TriggerGC(int8_t gc_sequence_index) {
1927 mutex_.AssertHeld();
1928 DCHECK_NULL(current_gc_info_);
1929 DCHECK(v8_flags.wasm_code_gc);
1930 new_potentially_dead_code_size_ = 0;
1931 current_gc_info_.reset(new CurrentGCInfo(gc_sequence_index));
1932 // Add all potentially dead code to this GC, and trigger a GC task in each
1933 // known isolate. We can't limit the isolates to those that contributed
1934 // potentially-dead WasmCode objects, because wrappers don't point back
1935 // at a NativeModule or Isolate.
1936 for (WasmCode* code : potentially_dead_code_) {
1937 current_gc_info_->dead_code.insert(code);
1938 }
1939 for (const auto& entry : isolates_) {
1940 Isolate* isolate = entry.first;
1941 auto& gc_task = current_gc_info_->outstanding_isolates[isolate];
1942 if (!gc_task) {
1943 auto new_task = std::make_unique<WasmGCForegroundTask>(isolate);
1944 gc_task = new_task.get();
1945 DCHECK_EQ(1, isolates_.count(isolate));
1946 isolates_[isolate]->foreground_task_runner->PostTask(std::move(new_task));
1947 }
1948 isolate->stack_guard()->RequestWasmCodeGC();
1949 }
1950 TRACE_CODE_GC(
1951 "Starting GC (nr %d). Number of potentially dead code objects: %zu\n",
1952 current_gc_info_->gc_sequence_index, current_gc_info_->dead_code.size());
1953 // Ensure that there are outstanding isolates that will eventually finish this
1954 // GC. If there are no outstanding isolates, we finish the GC immediately.
1955 PotentiallyFinishCurrentGC();
1956 DCHECK(current_gc_info_ == nullptr ||
1957 !current_gc_info_->outstanding_isolates.empty());
1958}

1966void WasmEngine::PotentiallyFinishCurrentGC() {
1967 mutex_.AssertHeld();
1968 TRACE_CODE_GC(
1969 "Remaining dead code objects: %zu; outstanding isolates: %zu.\n",
1970 current_gc_info_->dead_code.size(),
1971 current_gc_info_->outstanding_isolates.size());
1972
1973 // If there are more outstanding isolates, return immediately.
1974 if (!current_gc_info_->outstanding_isolates.empty()) return;
1975
1976 // All remaining code in {current_gc_info->dead_code} is really dead.
1977 // Move it from the set of potentially dead code to the set of dead code,
1978 // and decrement its ref count.
1979 size_t num_freed = 0;
1980 DeadCodeMap dead_code;
1981 std::vector<WasmCode*> dead_wrappers;
1982 for (WasmCode* code : current_gc_info_->dead_code) {
1983 DCHECK(potentially_dead_code_.contains(code));
1984 potentially_dead_code_.erase(code);
1985 DCHECK(!dead_code_.contains(code));
1986 dead_code_.insert(code);
1987 if (code->DecRefOnDeadCode()) {
1988 NativeModule* native_module = code->native_module();
1989 if (native_module) {
1990 dead_code[native_module].push_back(code);
1991 } else {
1992 dead_wrappers.push_back(code);
1993 }
1994 ++num_freed;
1995 }
1996 }
1997
1998 FreeDeadCodeLocked(dead_code, dead_wrappers);
1999
2000 TRACE_CODE_GC("Found %zu dead code objects, freed %zu.\n",
2001 current_gc_info_->dead_code.size(), num_freed);
2002 USE(num_freed);
2003
2004 int8_t next_gc_sequence_index = current_gc_info_->next_gc_sequence_index;
2005 current_gc_info_.reset();
2006 if (next_gc_sequence_index != 0) TriggerGC(next_gc_sequence_index);
2007}
