51bool PersistentHandles::Contains(Address* location) {
52 auto it = ordered_blocks_.upper_bound(location);
53 if (it == ordered_blocks_.begin()) return false;
54 --it;
55 DCHECK_LE(*it, location);
56 if (*it == blocks_.back()) {
57 // The last block is a special case because it may have
58 // less than block_size_ handles.
59 return location < block_next_;
60 }
61 return location < *it + kHandleBlockSize;
62}
63#endif

65void PersistentHandles::AddBlock() {
66 DCHECK_EQ(block_next_, block_limit_);
67
68 Address* block_start = NewArray<Address>(kHandleBlockSize);
69 blocks_.push_back(block_start);
70
71 block_next_ = block_start;
72 block_limit_ = block_start + kHandleBlockSize;
73
74#ifdef DEBUG
75 ordered_blocks_.insert(block_start);
76#endif
77}

79Address* PersistentHandles::GetHandle(Address value) {
80 if (block_next_ == block_limit_) {
81 AddBlock();
82 }
83
84 DCHECK_LT(block_next_, block_limit_);
85 *block_next_ = value;

89void PersistentHandles::Iterate(RootVisitor* visitor) {
90 for (int i = 0; i < static_cast<int>(blocks_.size()) - 1; i++) {
91 Address* block_start = blocks_[i];
92 Address* block_end = block_start + kHandleBlockSize;
93 visitor->VisitRootPointers(Root::kHandleScope, nullptr,
94 FullObjectSlot(block_start),
95 FullObjectSlot(block_end));
96 }
97
98 if (!blocks_.empty()) {
99 Address* block_start = blocks_.back();
100 visitor->VisitRootPointers(Root::kHandleScope, nullptr,
101 FullObjectSlot(block_start),
102 FullObjectSlot(block_next_));
103 }
104}

106void PersistentHandlesList::Add(PersistentHandles* persistent_handles) {
107 base::SpinningMutexGuard guard(&persistent_handles_mutex_);
108 if (persistent_handles_head_)
109 persistent_handles_head_->prev_ = persistent_handles;

115void PersistentHandlesList::Remove(PersistentHandles* persistent_handles) {
116 base::SpinningMutexGuard guard(&persistent_handles_mutex_);
117 if (persistent_handles->next_)
118 persistent_handles->next_->prev_ = persistent_handles->prev_;
119 if (persistent_handles->prev_)
120 persistent_handles->prev_->next_ = persistent_handles->next_;
121 else
122 persistent_handles_head_ = persistent_handles->next_;
123}

125void PersistentHandlesList::Iterate(RootVisitor* visitor, Isolate* isolate) {
126 isolate->heap()->safepoint()->AssertActive();
127 base::SpinningMutexGuard guard(&persistent_handles_mutex_);
128 for (PersistentHandles* current = persistent_handles_head_; current;
129 current = current->next_) {
130 current->Iterate(visitor);

134PersistentHandlesScope::PersistentHandlesScope(Isolate* isolate)
135 : impl_(isolate->handle_scope_implementer()) {
136 impl_->BeginPersistentScope();
137 HandleScopeData* data = impl_->isolate()->handle_scope_data();
138 Address* new_next = impl_->GetSpareOrNewBlock();
139 Address* new_limit = &new_next[kHandleBlockSize];
140 impl_->blocks()->push_back(new_next);
141
142#ifdef DEBUG
143 prev_level_ = data->level;
144#endif
145 data->level++;
146 first_block_ = new_next;
147 prev_limit_ = data->limit;
148 prev_next_ = data->next;
149 data->next = new_next;
150 data->limit = new_limit;
151}
