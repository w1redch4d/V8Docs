564class ArrayBufferViewAccessBuilder {
565 public:
566 explicit ArrayBufferViewAccessBuilder(JSGraphAssembler* assembler,
567 InstanceType instance_type,
568 std::set<ElementsKind> candidates)
569 : assembler_(assembler),
570 instance_type_(instance_type),
571 candidates_(std::move(candidates)) {
572 DCHECK_NOT_NULL(assembler_);
573 // TODO(v8:11111): Optimize for JS_RAB_GSAB_DATA_VIEW_TYPE too.
574 DCHECK(instance_type_ == JS_DATA_VIEW_TYPE ||
575 instance_type_ == JS_TYPED_ARRAY_TYPE);
576 }
577
578 bool maybe_rab_gsab() const {
579 if (candidates_.empty()) return true;
580 return !base::all_of(candidates_, [](auto e) {
581 return !IsRabGsabTypedArrayElementsKind(e);
582 });
583 }
584
585 std::optional<int> TryComputeStaticElementShift() {
586 DCHECK(instance_type_ != JS_RAB_GSAB_DATA_VIEW_TYPE);
587 if (instance_type_ == JS_DATA_VIEW_TYPE) return 0;
588 if (candidates_.empty()) return std::nullopt;
589 int shift = ElementsKindToShiftSize(*candidates_.begin());
590 if (!base::all_of(candidates_, [shift](auto e) {
591 return ElementsKindToShiftSize(e) == shift;
592 })) {
593 return std::nullopt;
594 }
595 return shift;
596 }
597
598 std::optional<int> TryComputeStaticElementSize() {
599 DCHECK(instance_type_ != JS_RAB_GSAB_DATA_VIEW_TYPE);
600 if (instance_type_ == JS_DATA_VIEW_TYPE) return 1;
601 if (candidates_.empty()) return std::nullopt;
602 int size = ElementsKindToByteSize(*candidates_.begin());
603 if (!base::all_of(candidates_, [size](auto e) {
604 return ElementsKindToByteSize(e) == size;
605 })) {
606 return std::nullopt;
607 }
608 return size;
609 }

611 TNode<UintPtrT> BuildLength(TNode<JSArrayBufferView> view,
612 TNode<Context> context) {
613 auto& a = *assembler_;
614
615 // Case 1: Normal (backed by AB/SAB) or non-length tracking backed by GSAB
616 // (can't go oob once constructed)
617 auto GsabFixedOrNormal = [&]() {
618 return MachineLoadField<UintPtrT>(AccessBuilder::ForJSTypedArrayLength(),
619 view, UseInfo::Word());
620 };
621
622 // If we statically know we cannot have rab/gsab backed, we can simply
623 // load from the view.
624 if (!maybe_rab_gsab()) {
625 return GsabFixedOrNormal();
626 }
627
628 // Otherwise, we need to generate the checks for the view's bitfield.
629 TNode<Word32T> bitfield = a.EnterMachineGraph<Word32T>(
630 a.LoadField<Word32T>(AccessBuilder::ForJSArrayBufferViewBitField(),
631 view),
632 UseInfo::TruncatingWord32());
633 TNode<Word32T> length_tracking_bit = a.Word32And(
634 bitfield, a.Uint32Constant(JSArrayBufferView::kIsLengthTracking));
635 TNode<Word32T> backed_by_rab_bit = a.Word32And(
636 bitfield, a.Uint32Constant(JSArrayBufferView::kIsBackedByRab));
637
638 // Load the underlying buffer.
639 TNode<HeapObject> buffer = a.LoadField<HeapObject>(
640 AccessBuilder::ForJSArrayBufferViewBuffer(), view);
641
642 // Compute the element size.
643 TNode<Uint32T> element_size;
644 if (auto size_opt = TryComputeStaticElementSize()) {
645 element_size = a.Uint32Constant(*size_opt);
646 } else {
647 DCHECK_EQ(instance_type_, JS_TYPED_ARRAY_TYPE);
648 TNode<Map> typed_array_map = a.LoadField<Map>(
649 AccessBuilder::ForMap(WriteBarrierKind::kNoWriteBarrier), view);
650 TNode<Uint32T> elements_kind = a.LoadElementsKind(typed_array_map);
651 element_size = a.LookupByteSizeForElementsKind(elements_kind);
652 }
653
654 // 2) Fixed length backed by RAB (can go oob once constructed)
655 auto RabFixed = [&]() {
656 TNode<UintPtrT> unchecked_byte_length = MachineLoadField<UintPtrT>(
657 AccessBuilder::ForJSArrayBufferViewByteLength(), view,
658 UseInfo::Word());
659 TNode<UintPtrT> underlying_byte_length = MachineLoadField<UintPtrT>(
660 AccessBuilder::ForJSArrayBufferByteLength(), buffer, UseInfo::Word());
661 TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
662 AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
663 UseInfo::Word());
664
665 TNode<UintPtrT> byte_length =
666 a
667 .MachineSelectIf<UintPtrT>(a.UintPtrLessThanOrEqual(
668 a.UintPtrAdd(byte_offset, unchecked_byte_length),
669 underlying_byte_length))
670 .Then([&]() { return unchecked_byte_length; })
671 .Else([&]() { return a.UintPtrConstant(0); })
672 .Value();
673 return a.UintPtrDiv(byte_length, a.ChangeUint32ToUintPtr(element_size));
674 };
675
676 // 3) Length-tracking backed by RAB (JSArrayBuffer stores the length)
677 auto RabTracking = [&]() {
678 TNode<UintPtrT> byte_length = MachineLoadField<UintPtrT>(
679 AccessBuilder::ForJSArrayBufferByteLength(), buffer, UseInfo::Word());
680 TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
681 AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
682 UseInfo::Word());
683
684 return a
685 .MachineSelectIf<UintPtrT>(
686 a.UintPtrLessThanOrEqual(byte_offset, byte_length))
687 .Then([&]() {
688 // length = floor((byte_length - byte_offset) / element_size)
689 return a.UintPtrDiv(a.UintPtrSub(byte_length, byte_offset),
690 a.ChangeUint32ToUintPtr(element_size));
691 })
692 .Else([&]() { return a.UintPtrConstant(0); })
693 .ExpectTrue()
694 .Value();
695 };
696
697 // 4) Length-tracking backed by GSAB (BackingStore stores the length)
698 auto GsabTracking = [&]() {
699 TNode<Number> temp = TNode<Number>::UncheckedCast(a.TypeGuard(
700 TypeCache::Get()->kJSArrayBufferViewByteLengthType,
701 a.JSCallRuntime1(Runtime::kGrowableSharedArrayBufferByteLength,
702 buffer, context, std::nullopt, Operator::kNoWrite)));
703 TNode<UintPtrT> byte_length =
704 a.EnterMachineGraph<UintPtrT>(temp, UseInfo::Word());
705 TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
706 AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
707 UseInfo::Word());
708
709 return a
710 .MachineSelectIf<UintPtrT>(
711 a.UintPtrLessThanOrEqual(byte_offset, byte_length))
712 .Then([&]() {
713 // length = floor((byte_length - byte_offset) / element_size)
714 return a.UintPtrDiv(a.UintPtrSub(byte_length, byte_offset),
715 a.ChangeUint32ToUintPtr(element_size));
716 })
717 .Else([&]() { return a.UintPtrConstant(0); })
718 .ExpectTrue()
719 .Value();
720 };
721
722 return a.MachineSelectIf<UintPtrT>(length_tracking_bit)
723 .Then([&]() {
724 return a.MachineSelectIf<UintPtrT>(backed_by_rab_bit)
725 .Then(RabTracking)
726 .Else(GsabTracking)
727 .Value();
728 })
729 .Else([&]() {
730 return a.MachineSelectIf<UintPtrT>(backed_by_rab_bit)
731 .Then(RabFixed)
732 .Else(GsabFixedOrNormal)
733 .Value();
734 })
735 .Value();
736 }

738 TNode<UintPtrT> BuildByteLength(TNode<JSArrayBufferView> view,
739 TNode<Context> context) {
740 auto& a = *assembler_;
741
742 // Case 1: Normal (backed by AB/SAB) or non-length tracking backed by GSAB
743 // (can't go oob once constructed)
744 auto GsabFixedOrNormal = [&]() {
745 return MachineLoadField<UintPtrT>(
746 AccessBuilder::ForJSArrayBufferViewByteLength(), view,
747 UseInfo::Word());
748 };
749
750 // If we statically know we cannot have rab/gsab backed, we can simply
751 // use load from the view.
752 if (!maybe_rab_gsab()) {
753 return GsabFixedOrNormal();
754 }
755
756 // Otherwise, we need to generate the checks for the view's bitfield.
757 TNode<Word32T> bitfield = a.EnterMachineGraph<Word32T>(
758 a.LoadField<Word32T>(AccessBuilder::ForJSArrayBufferViewBitField(),
759 view),
760 UseInfo::TruncatingWord32());
761 TNode<Word32T> length_tracking_bit = a.Word32And(
762 bitfield, a.Uint32Constant(JSArrayBufferView::kIsLengthTracking));
763 TNode<Word32T> backed_by_rab_bit = a.Word32And(
764 bitfield, a.Uint32Constant(JSArrayBufferView::kIsBackedByRab));
765
766 // Load the underlying buffer.
767 TNode<HeapObject> buffer = a.LoadField<HeapObject>(
768 AccessBuilder::ForJSArrayBufferViewBuffer(), view);
769
770 // Case 2: Fixed length backed by RAB (can go oob once constructed)
771 auto RabFixed = [&]() {
772 TNode<UintPtrT> unchecked_byte_length = MachineLoadField<UintPtrT>(
773 AccessBuilder::ForJSArrayBufferViewByteLength(), view,
774 UseInfo::Word());
775 TNode<UintPtrT> underlying_byte_length = MachineLoadField<UintPtrT>(
776 AccessBuilder::ForJSArrayBufferByteLength(), buffer, UseInfo::Word());
777 TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
778 AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
779 UseInfo::Word());
780
781 return a
782 .MachineSelectIf<UintPtrT>(a.UintPtrLessThanOrEqual(
783 a.UintPtrAdd(byte_offset, unchecked_byte_length),
784 underlying_byte_length))
785 .Then([&]() { return unchecked_byte_length; })
786 .Else([&]() { return a.UintPtrConstant(0); })
787 .Value();
788 };
789
790 auto RoundDownToElementSize = [&](TNode<UintPtrT> byte_size) {
791 if (auto shift_opt = TryComputeStaticElementShift()) {
792 constexpr uintptr_t all_bits = static_cast<uintptr_t>(-1);
793 if (*shift_opt == 0) return byte_size;
794 return TNode<UintPtrT>::UncheckedCast(
795 a.WordAnd(byte_size, a.UintPtrConstant(all_bits << (*shift_opt))));
796 }
797 DCHECK_EQ(instance_type_, JS_TYPED_ARRAY_TYPE);
798 TNode<Map> typed_array_map = a.LoadField<Map>(
799 AccessBuilder::ForMap(WriteBarrierKind::kNoWriteBarrier), view);
800 TNode<Uint32T> elements_kind = a.LoadElementsKind(typed_array_map);
801 TNode<Uint32T> element_shift =
802 a.LookupByteShiftForElementsKind(elements_kind);
803 return TNode<UintPtrT>::UncheckedCast(
804 a.WordShl(a.WordShr(byte_size, element_shift), element_shift));
805 };
806
807 // Case 3: Length-tracking backed by RAB (JSArrayBuffer stores the length)
808 auto RabTracking = [&]() {
809 TNode<UintPtrT> byte_length = MachineLoadField<UintPtrT>(
810 AccessBuilder::ForJSArrayBufferByteLength(), buffer, UseInfo::Word());
811 TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
812 AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
813 UseInfo::Word());
814
815 return a
816 .MachineSelectIf<UintPtrT>(
817 a.UintPtrLessThanOrEqual(byte_offset, byte_length))
818 .Then([&]() {
819 return RoundDownToElementSize(
820 a.UintPtrSub(byte_length, byte_offset));
821 })
822 .Else([&]() { return a.UintPtrConstant(0); })
823 .ExpectTrue()
824 .Value();
825 };
826
827 // Case 4: Length-tracking backed by GSAB (BackingStore stores the length)
828 auto GsabTracking = [&]() {
829 TNode<Number> temp = TNode<Number>::UncheckedCast(a.TypeGuard(
830 TypeCache::Get()->kJSArrayBufferViewByteLengthType,
831 a.JSCallRuntime1(Runtime::kGrowableSharedArrayBufferByteLength,
832 buffer, context, std::nullopt, Operator::kNoWrite)));
833 TNode<UintPtrT> byte_length =
834 a.EnterMachineGraph<UintPtrT>(temp, UseInfo::Word());
835 TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
836 AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
837 UseInfo::Word());
838
839 return a
840 .MachineSelectIf<UintPtrT>(
841 a.UintPtrLessThanOrEqual(byte_offset, byte_length))
842 .Then([&]() {
843 return RoundDownToElementSize(
844 a.UintPtrSub(byte_length, byte_offset));
845 })
846 .Else([&]() { return a.UintPtrConstant(0); })
847 .ExpectTrue()
848 .Value();
849 };
850
851 return a.MachineSelectIf<UintPtrT>(length_tracking_bit)
852 .Then([&]() {
853 return a.MachineSelectIf<UintPtrT>(backed_by_rab_bit)
854 .Then(RabTracking)
855 .Else(GsabTracking)
856 .Value();
857 })
858 .Else([&]() {
859 return a.MachineSelectIf<UintPtrT>(backed_by_rab_bit)
860 .Then(RabFixed)
861 .Else(GsabFixedOrNormal)
862 .Value();
863 })
864 .Value();
865 }

867 TNode<Word32T> BuildDetachedCheck(TNode<JSArrayBufferView> view) {
868 auto& a = *assembler_;
869
870 // Load the underlying buffer and its bitfield.
871 TNode<HeapObject> buffer = a.LoadField<HeapObject>(
872 AccessBuilder::ForJSArrayBufferViewBuffer(), view);
873 TNode<Word32T> buffer_bit_field =
874 MachineLoadField<Word32T>(AccessBuilder::ForJSArrayBufferBitField(),
875 buffer, UseInfo::TruncatingWord32());
876 // Mask the detached bit.
877 TNode<Word32T> detached_bit =
878 a.Word32And(buffer_bit_field,
879 a.Uint32Constant(JSArrayBuffer::WasDetachedBit::kMask));
880
881 // If we statically know we cannot have rab/gsab backed, we are done here.
882 if (!maybe_rab_gsab()) {
883 return detached_bit;
884 }
885
886 // Otherwise, we need to generate the checks for the view's bitfield.
887 TNode<Word32T> bitfield = a.EnterMachineGraph<Word32T>(
888 a.LoadField<Word32T>(AccessBuilder::ForJSArrayBufferViewBitField(),
889 view),
890 UseInfo::TruncatingWord32());
891 TNode<Word32T> length_tracking_bit = a.Word32And(
892 bitfield, a.Uint32Constant(JSArrayBufferView::kIsLengthTracking));
893 TNode<Word32T> backed_by_rab_bit = a.Word32And(
894 bitfield, a.Uint32Constant(JSArrayBufferView::kIsBackedByRab));
895
896 auto RabLengthTracking = [&]() {
897 TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
898 AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
899 UseInfo::Word());
900
901 TNode<UintPtrT> underlying_byte_length = MachineLoadField<UintPtrT>(
902 AccessBuilder::ForJSArrayBufferByteLength(), buffer, UseInfo::Word());
903
904 return a.Word32Or(detached_bit,
905 a.UintPtrLessThan(underlying_byte_length, byte_offset));
906 };
907
908 auto RabFixed = [&]() {
909 TNode<UintPtrT> unchecked_byte_length = MachineLoadField<UintPtrT>(
910 AccessBuilder::ForJSArrayBufferViewByteLength(), view,
911 UseInfo::Word());
912 TNode<UintPtrT> byte_offset = MachineLoadField<UintPtrT>(
913 AccessBuilder::ForJSArrayBufferViewByteOffset(), view,
914 UseInfo::Word());
915
916 TNode<UintPtrT> underlying_byte_length = MachineLoadField<UintPtrT>(
917 AccessBuilder::ForJSArrayBufferByteLength(), buffer, UseInfo::Word());
918
919 return a.Word32Or(
920 detached_bit,
921 a.UintPtrLessThan(underlying_byte_length,
922 a.UintPtrAdd(byte_offset, unchecked_byte_length)));
923 };
924
925 // Dispatch depending on rab/gsab and length tracking.
926 return a.MachineSelectIf<Word32T>(backed_by_rab_bit)
927 .Then([&]() {
928 return a.MachineSelectIf<Word32T>(length_tracking_bit)
929 .Then(RabLengthTracking)
930 .Else(RabFixed)
931 .Value();
932 })
933 .Else([&]() { return detached_bit; })
934 .Value();
935 }

1275Node* GraphAssembler::AddNode(Node* node) {
1276 if (!inline_reducers_.empty() && !inline_reductions_blocked_) {
1277 // Reducers may add new nodes to the graph using this graph assembler,
1278 // however they should never introduce nodes that need further reduction,
1279 // so block reduction
1280 BlockInlineReduction scope(this);
1281 Reduction reduction;
1282 for (auto reducer : inline_reducers_) {
1283 reduction = reducer->Reduce(node, nullptr);
1284 if (reduction.Changed()) break;
1285 }
1286 if (reduction.Changed()) {
1287 Node* replacement = reduction.replacement();
1288 if (replacement != node) {
1289 // Replace all uses of node and kill the node to make sure we don't
1290 // leave dangling dead uses.
1291 NodeProperties::ReplaceUses(node, replacement, effect(), control());
1292 node->Kill();
1293 return replacement;
1294 }
1295 }
1296 }
1297
1298 if (node->opcode() == IrOpcode::kTerminate) {
1299 return node;
1300 }
1301
1302 UpdateEffectControlWith(node);
1303 return node;
1304}
