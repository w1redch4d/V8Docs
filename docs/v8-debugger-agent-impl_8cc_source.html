138bool parseBreakpointId(const String16& breakpointId, BreakpointType* type,
139 String16* scriptSelector = nullptr,
140 int* lineNumber = nullptr, int* columnNumber = nullptr) {
141 size_t typeLineSeparator = breakpointId.find(':');
142 if (typeLineSeparator == String16::kNotFound) return false;
143
144 int rawType = breakpointId.substring(0, typeLineSeparator).toInteger();
145 if (rawType < static_cast<int>(BreakpointType::kByUrl) ||
146 rawType > static_cast<int>(BreakpointType::kInstrumentationBreakpoint)) {
147 return false;
148 }
149 if (type) *type = static_cast<BreakpointType>(rawType);
150 if (rawType == static_cast<int>(BreakpointType::kDebugCommand) ||
151 rawType == static_cast<int>(BreakpointType::kMonitorCommand) ||
152 rawType == static_cast<int>(BreakpointType::kBreakpointAtEntry) ||
153 rawType == static_cast<int>(BreakpointType::kInstrumentationBreakpoint)) {
154 // The script and source position are not encoded in this case.
155 return true;
156 }
157
158 size_t lineColumnSeparator = breakpointId.find(':', typeLineSeparator + 1);
159 if (lineColumnSeparator == String16::kNotFound) return false;
160 size_t columnSelectorSeparator =
161 breakpointId.find(':', lineColumnSeparator + 1);
162 if (columnSelectorSeparator == String16::kNotFound) return false;
163 if (scriptSelector) {
164 *scriptSelector = breakpointId.substring(columnSelectorSeparator + 1);
165 }
166 if (lineNumber) {
167 *lineNumber = breakpointId
168 .substring(typeLineSeparator + 1,
169 lineColumnSeparator - typeLineSeparator - 1)
170 .toInteger();
171 }
172 if (columnNumber) {
173 *columnNumber =
174 breakpointId
175 .substring(lineColumnSeparator + 1,
176 columnSelectorSeparator - lineColumnSeparator - 1)
177 .toInteger();
178 }
179 return true;
180}

188std::unique_ptr<protocol::DictionaryValue> breakpointHint(
189 const V8DebuggerScript& script, int breakpointLineNumber,
190 int breakpointColumnNumber, int actualLineNumber, int actualColumnNumber) {
191 int actualOffset;
192 int breakpointOffset;
193 if (!script.offset(actualLineNumber, actualColumnNumber).To(&actualOffset) ||
194 !script.offset(breakpointLineNumber, breakpointColumnNumber)
195 .To(&breakpointOffset)) {
196 return {};
197 }
198
199 auto hintObject = protocol::DictionaryValue::create();
200 String16 rawHint = script.source(actualOffset, kBreakpointHintMaxLength);
201 std::pair<size_t, size_t> offsetAndLength =
202 rawHint.getTrimmedOffsetAndLength();
203 String16 hint =
204 rawHint.substring(offsetAndLength.first, offsetAndLength.second);
205 for (size_t i = 0; i < hint.length(); ++i) {
206 if (hint[i] == '\r' || hint[i] == '\n' || hint[i] == ';') {
207 hint = hint.substring(0, i);
208 break;
209 }
210 }
211 hintObject->setString(DebuggerAgentState::breakpointHintText, hint);
212
213 // Also store the hash of the text between the requested breakpoint location
214 // and the actual breakpoint location. If we see the same prefix text next
215 // time, we will keep the breakpoint at the same location (so that
216 // breakpoints do not slide around on reloads without any edits).
217 if (breakpointOffset <= actualOffset) {
218 size_t length = actualOffset - breakpointOffset + offsetAndLength.first;
219 String16 prefix = script.source(breakpointOffset, length);
220 int crc32 = computeCrc32(prefix);
221 hintObject->setInteger(DebuggerAgentState::breakpointHintPrefixHash, crc32);
222 hintObject->setInteger(DebuggerAgentState::breakpointHintPrefixLength,
223 v8::base::checked_cast<int32_t>(length));
224 }
225 return hintObject;
226}

228void adjustBreakpointLocation(const V8DebuggerScript& script,
229 const protocol::DictionaryValue* hintObject,
230 int* lineNumber, int* columnNumber) {
231 if (*lineNumber < script.startLine() || *lineNumber > script.endLine())
232 return;
233 if (*lineNumber == script.startLine() &&
234 *columnNumber < script.startColumn()) {
235 return;
236 }
237 if (*lineNumber == script.endLine() && script.endColumn() < *columnNumber) {
238 return;
239 }
240
241 int sourceOffset;
242 if (!script.offset(*lineNumber, *columnNumber).To(&sourceOffset)) return;
243
244 int prefixLength = 0;
245 hintObject->getInteger(DebuggerAgentState::breakpointHintPrefixLength,
246 &prefixLength);
247 String16 hint;
248 if (!hintObject->getString(DebuggerAgentState::breakpointHintText, &hint) ||
249 hint.isEmpty())
250 return;
251
252 intptr_t searchRegionOffset = std::max(
253 sourceOffset - kBreakpointHintMaxSearchOffset, static_cast<intptr_t>(0));
254 size_t offset = sourceOffset - searchRegionOffset;
255 size_t searchRegionSize =
256 offset + std::max(kBreakpointHintMaxSearchOffset,
257 static_cast<intptr_t>(prefixLength + hint.length()));
258
259 String16 searchArea = script.source(searchRegionOffset, searchRegionSize);
260
261 // Let us see if the breakpoint hint text appears at the same location
262 // as before, with the same prefix text in between. If yes, then we just use
263 // that position.
264 int prefixHash;
265 if (hintObject->getInteger(DebuggerAgentState::breakpointHintPrefixHash,
266 &prefixHash) &&
267 offset + prefixLength + hint.length() <= searchArea.length() &&
268 searchArea.substring(offset + prefixLength, hint.length()) == hint &&
269 computeCrc32(searchArea.substring(offset, prefixLength)) == prefixHash) {
270 v8::debug::Location hintPosition = script.location(
271 static_cast<int>(searchRegionOffset + offset + prefixLength));
272 *lineNumber = hintPosition.GetLineNumber();
273 *columnNumber = hintPosition.GetColumnNumber();
274 return;
275 }
276
277 size_t nextMatch = searchArea.find(hint, offset);
278 size_t prevMatch = searchArea.reverseFind(hint, offset);
279 if (nextMatch == String16::kNotFound && prevMatch == String16::kNotFound) {
280 return;
281 }
282 size_t bestMatch;
283 if (nextMatch == String16::kNotFound ||
284 nextMatch > offset + kBreakpointHintMaxSearchOffset) {
285 bestMatch = prevMatch;
286 } else if (prevMatch == String16::kNotFound) {
287 bestMatch = nextMatch;
288 } else {
289 bestMatch = nextMatch - offset < offset - prevMatch ? nextMatch : prevMatch;
290 }
291 bestMatch += searchRegionOffset;
292 v8::debug::Location hintPosition =
293 script.location(static_cast<int>(bestMatch));
294 if (hintPosition.IsEmpty()) return;
295 *lineNumber = hintPosition.GetLineNumber();
296 *columnNumber = hintPosition.GetColumnNumber();
297}

339Response buildScopes(v8::Isolate* isolate, v8::debug::ScopeIterator* iterator,
340 InjectedScript* injectedScript,
341 std::unique_ptr<Array<Scope>>* scopes) {
342 *scopes = std::make_unique<Array<Scope>>();
343 if (!injectedScript) return Response::Success();
344 if (iterator->Done()) return Response::Success();
345
346 String16 scriptId = String16::fromInteger(iterator->GetScriptId());
347
348 for (; !iterator->Done(); iterator->Advance()) {
349 std::unique_ptr<RemoteObject> object;
350 Response result =
351 injectedScript->wrapObject(iterator->GetObject(), kBacktraceObjectGroup,
352 WrapOptions({WrapMode::kIdOnly}), &object);
353 if (!result.IsSuccess()) return result;
354
355 auto scope = Scope::create()
356 .setType(scopeType(iterator->GetType()))
357 .setObject(std::move(object))
358 .build();
359
360 String16 name = toProtocolStringWithTypeCheck(
361 isolate, iterator->GetFunctionDebugName());
362 if (!name.isEmpty()) scope->setName(name);
363
364 if (iterator->HasLocationInfo()) {
365 v8::debug::Location start = iterator->GetStartLocation();
366 scope->setStartLocation(protocol::Debugger::Location::create()
367 .setScriptId(scriptId)
368 .setLineNumber(start.GetLineNumber())
369 .setColumnNumber(start.GetColumnNumber())
370 .build());
371
372 v8::debug::Location end = iterator->GetEndLocation();
373 scope->setEndLocation(protocol::Debugger::Location::create()
374 .setScriptId(scriptId)
375 .setLineNumber(end.GetLineNumber())
376 .setColumnNumber(end.GetColumnNumber())
377 .build());
378 }
379 (*scopes)->emplace_back(std::move(scope));
380 }
381 return Response::Success();
382}

616Response V8DebuggerAgentImpl::setBreakpointByUrl(
617 int lineNumber, std::optional<String16> optionalURL,
618 std::optional<String16> optionalURLRegex,
619 std::optional<String16> optionalScriptHash,
620 std::optional<int> optionalColumnNumber,
621 std::optional<String16> optionalCondition, String16* outBreakpointId,
622 std::unique_ptr<protocol::Array<protocol::Debugger::Location>>* locations) {
623 if (!enabled()) return Response::ServerError(kDebuggerNotEnabled);
624
625 *locations = std::make_unique<Array<protocol::Debugger::Location>>();
626
627 int specified = (optionalURL.has_value() ? 1 : 0) +
628 (optionalURLRegex.has_value() ? 1 : 0) +
629 (optionalScriptHash.has_value() ? 1 : 0);
630 if (specified != 1) {
631 return Response::ServerError(
632 "Either url or urlRegex or scriptHash must be specified.");
633 }
634 int columnNumber = 0;
635 if (optionalColumnNumber.has_value()) {
636 columnNumber = optionalColumnNumber.value();
637 if (columnNumber < 0)
638 return Response::ServerError("Incorrect column number");
639 }
640
641 BreakpointType type = BreakpointType::kByUrl;
642 String16 selector;
643 if (optionalURLRegex.has_value()) {
644 selector = optionalURLRegex.value();
645 type = BreakpointType::kByUrlRegex;
646 } else if (optionalURL.has_value()) {
647 selector = optionalURL.value();
648 type = BreakpointType::kByUrl;
649 } else if (optionalScriptHash.has_value()) {
650 selector = optionalScriptHash.value();
651 type = BreakpointType::kByScriptHash;
652 }
653
654 // Note: This constructor can call into JavaScript.
655 Matcher matcher(m_inspector, type, selector);
656
657 String16 condition = optionalCondition.value_or(String16());
658 String16 breakpointId =
659 generateBreakpointId(type, selector, lineNumber, columnNumber);
660 protocol::DictionaryValue* breakpoints;
661 switch (type) {
662 case BreakpointType::kByUrlRegex:
663 breakpoints =
664 getOrCreateObject(m_state, DebuggerAgentState::breakpointsByRegex);
665 break;
666 case BreakpointType::kByUrl:
667 breakpoints = getOrCreateObject(
668 getOrCreateObject(m_state, DebuggerAgentState::breakpointsByUrl),
669 selector);
670 break;
671 case BreakpointType::kByScriptHash:
672 breakpoints = getOrCreateObject(
673 getOrCreateObject(m_state,
674 DebuggerAgentState::breakpointsByScriptHash),
675 selector);
676 break;
677 default:
678 UNREACHABLE();
679 }
680 if (breakpoints->get(breakpointId)) {
681 return Response::ServerError(
682 "Breakpoint at specified location already exists.");
683 }
684
685 std::unique_ptr<protocol::DictionaryValue> hint;
686 for (const auto& script : m_scripts) {
687 if (!matcher.matches(*script.second)) continue;
688 // Make sure the session was not disabled by some re-entrant call
689 // in the script matcher.
690 DCHECK(enabled());
691 int adjustedLineNumber = lineNumber;
692 int adjustedColumnNumber = columnNumber;
693 if (hint) {
694 adjustBreakpointLocation(*script.second, hint.get(), &adjustedLineNumber,
695 &adjustedColumnNumber);
696 }
697 std::unique_ptr<protocol::Debugger::Location> location =
698 setBreakpointImpl(breakpointId, script.first, condition,
699 adjustedLineNumber, adjustedColumnNumber);
700 if (location && type != BreakpointType::kByUrlRegex) {
701 hint = breakpointHint(*script.second, lineNumber, columnNumber,
702 location->getLineNumber(),
703 location->getColumnNumber(adjustedColumnNumber));
704 }
705 if (location) (*locations)->emplace_back(std::move(location));
706 }
707 breakpoints->setString(breakpointId, condition);
708 if (hint) {
709 protocol::DictionaryValue* breakpointHints =
710 getOrCreateObject(m_state, DebuggerAgentState::breakpointHints);
711 breakpointHints->setObject(breakpointId, std::move(hint));
712 }
713 *outBreakpointId = breakpointId;
714 return Response::Success();
715}

783Response V8DebuggerAgentImpl::removeBreakpoint(const String16& breakpointId) {
784 if (!enabled()) return Response::ServerError(kDebuggerNotEnabled);
785 BreakpointType type;
786 String16 selector;
787 if (!parseBreakpointId(breakpointId, &type, &selector)) {
788 return Response::Success();
789 }
790 Matcher matcher(m_inspector, type, selector);
791 protocol::DictionaryValue* breakpoints = nullptr;
792 switch (type) {
793 case BreakpointType::kByUrl: {
794 protocol::DictionaryValue* breakpointsByUrl =
795 m_state->getObject(DebuggerAgentState::breakpointsByUrl);
796 if (breakpointsByUrl) {
797 breakpoints = breakpointsByUrl->getObject(selector);
798 }
799 } break;
800 case BreakpointType::kByScriptHash: {
801 protocol::DictionaryValue* breakpointsByScriptHash =
802 m_state->getObject(DebuggerAgentState::breakpointsByScriptHash);
803 if (breakpointsByScriptHash) {
804 breakpoints = breakpointsByScriptHash->getObject(selector);
805 }
806 } break;
807 case BreakpointType::kByUrlRegex:
808 breakpoints = m_state->getObject(DebuggerAgentState::breakpointsByRegex);
809 break;
810 case BreakpointType::kInstrumentationBreakpoint:
811 breakpoints =
812 m_state->getObject(DebuggerAgentState::instrumentationBreakpoints);
813 break;
814 default:
815 break;
816 }
817 if (breakpoints) breakpoints->remove(breakpointId);
818 protocol::DictionaryValue* breakpointHints =
819 m_state->getObject(DebuggerAgentState::breakpointHints);
820 if (breakpointHints) breakpointHints->remove(breakpointId);
821
822 // Get a list of scripts to remove breakpoints.
823 // TODO(duongn): we can do better here if from breakpoint id we can tell it is
824 // not Wasm breakpoint.
825 std::vector<V8DebuggerScript*> scripts;
826 for (const auto& scriptIter : m_scripts) {
827 const bool scriptSelectorMatch = matcher.matches(*scriptIter.second);
828 // Make sure the session was not disabled by some re-entrant call
829 // in the script matcher.
830 DCHECK(enabled());
831 const bool isInstrumentation =
832 type == BreakpointType::kInstrumentationBreakpoint;
833 if (!scriptSelectorMatch && !isInstrumentation) continue;
834 V8DebuggerScript* script = scriptIter.second.get();
835 if (script->getLanguage() == V8DebuggerScript::Language::WebAssembly) {
836 scripts.push_back(script);
837 }
838 }
839 removeBreakpointImpl(breakpointId, scripts);
840
841 return Response::Success();
842}

845 const String16& breakpointId,
846 const std::vector<V8DebuggerScript*>& scripts) {
847 DCHECK(enabled());
848 BreakpointIdToDebuggerBreakpointIdsMap::iterator
849 debuggerBreakpointIdsIterator =

985bool V8DebuggerAgentImpl::isFunctionBlackboxed(const String16& scriptId,
986 const v8::debug::Location& start,
987 const v8::debug::Location& end) {
988 ScriptsMap::iterator it = m_scripts.find(scriptId);
989 if (it == m_scripts.end()) {
990 // Unknown scripts are blackboxed.
991 return true;
992 }
993 const String16& scriptSourceURL = it->second->sourceURL();
994 if (m_blackboxPattern && !scriptSourceURL.isEmpty()
995 && m_blackboxPattern->match(scriptSourceURL) != -1) {
996 return true;
997 }
998 if (m_skipAnonymousScripts && scriptSourceURL.isEmpty()) {
999 return true;
1000 }
1001 if (!m_blackboxedExecutionContexts.empty()) {
1002 int contextId = it->second->executionContextId();
1003 InspectedContext* inspected = m_inspector->getContext(contextId);
1004 if (inspected && m_blackboxedExecutionContexts.count(
1005 inspected->uniqueId().toString()) > 0) {
1006 return true;
1007 }
1008 }
1009 auto itBlackboxedPositions = m_blackboxedPositions.find(scriptId);
1010 if (itBlackboxedPositions == m_blackboxedPositions.end()) return false;
1011
1012 const std::vector<std::pair<int, int>>& ranges =
1013 itBlackboxedPositions->second;
1014 auto itStartRange = std::lower_bound(
1015 ranges.begin(), ranges.end(),
1016 std::make_pair(start.GetLineNumber(), start.GetColumnNumber()),
1017 positionComparator);
1018 auto itEndRange = std::lower_bound(
1019 itStartRange, ranges.end(),
1020 std::make_pair(end.GetLineNumber(), end.GetColumnNumber()),
1021 positionComparator);
1022 // Ranges array contains positions in script where blackbox state is changed.
1023 // [(0,0) ... ranges[0]) isn't blackboxed, [ranges[0] ... ranges[1]) is
1024 // blackboxed...
1025 return itStartRange == itEndRange &&
1026 std::distance(ranges.begin(), itStartRange) % 2;
1027}

1029bool V8DebuggerAgentImpl::shouldBeSkipped(const String16& scriptId, int line,
1030 int column) {
1031 if (m_skipList.empty()) return false;
1032
1033 auto it = m_skipList.find(scriptId);
1034 if (it == m_skipList.end()) return false;
1035
1036 const std::vector<std::pair<int, int>>& ranges = it->second;
1037 DCHECK(!ranges.empty());
1038 const std::pair<int, int> location = std::make_pair(line, column);
1039 auto itLowerBound = std::lower_bound(ranges.begin(), ranges.end(), location,
1040 positionComparator);
1041
1042 bool shouldSkip = false;
1043 if (itLowerBound != ranges.end()) {
1044 // Skip lists are defined as pairs of locations that specify the
1045 // start and the end of ranges to skip: [ranges[0], ranges[1], ..], where
1046 // locations in [ranges[0], ranges[1]) should be skipped, i.e.
1047 // [(lineStart, columnStart), (lineEnd, columnEnd)).
1048 const bool isSameAsLowerBound = location == *itLowerBound;
1049 const bool isUnevenIndex = (itLowerBound - ranges.begin()) % 2;
1050 shouldSkip = isSameAsLowerBound ^ isUnevenIndex;
1051 }
1052
1053 return shouldSkip;
1054}

1246 return Response::ServerError("No script for id: " + scriptId.utf8());
1247 }
1248 *scriptSource = it->second->source(0);
1249#if V8_ENABLE_WEBASSEMBLY

1251 if (it->second->wasmBytecode().To(&span)) {
1252 if (span.size() > kWasmBytecodeMaxLength) {
1253 return Response::ServerError(kWasmBytecodeExceedsTransferLimit);
1254 }
1255 *bytecode = protocol::Binary::fromSpan(span);
1256 }
1257#endif // V8_ENABLE_WEBASSEMBLY
1258 return Response::Success();
1259}

1261struct DisassemblyChunk {
1262 DisassemblyChunk() = default;
1263 DisassemblyChunk(const DisassemblyChunk& other) = delete;
1264 DisassemblyChunk& operator=(const DisassemblyChunk& other) = delete;
1265 DisassemblyChunk(DisassemblyChunk&& other) V8_NOEXCEPT = default;
1266 DisassemblyChunk& operator=(DisassemblyChunk&& other) V8_NOEXCEPT = default;
1267
1268 std::vector<String16> lines;
1269 std::vector<int> lineOffsets;
1270
1271 void Reserve(size_t size) {
1272 lines.reserve(size);
1273 lineOffsets.reserve(size);
1274 }
1275};
1276
1277class DisassemblyCollectorImpl final : public v8::debug::DisassemblyCollector {
1278 public:
1279 DisassemblyCollectorImpl() = default;
1280
1281 void ReserveLineCount(size_t count) override {
1282 if (count == 0) return;
1283 size_t num_chunks = (count + kLinesPerChunk - 1) / kLinesPerChunk;
1284 chunks_.resize(num_chunks);
1285 for (size_t i = 0; i < num_chunks - 1; i++) {
1286 chunks_[i].Reserve(kLinesPerChunk);
1287 }
1288 size_t last = num_chunks - 1;
1289 size_t last_size = count % kLinesPerChunk;
1290 if (last_size == 0) last_size = kLinesPerChunk;
1291 chunks_[last].Reserve(last_size);
1292 }
1293
1294 void AddLine(const char* src, size_t length,
1295 uint32_t bytecode_offset) override {
1296 chunks_[writing_chunk_index_].lines.emplace_back(src, length);
1297 chunks_[writing_chunk_index_].lineOffsets.push_back(
1298 static_cast<int>(bytecode_offset));
1299 if (chunks_[writing_chunk_index_].lines.size() == kLinesPerChunk) {
1300 writing_chunk_index_++;
1301 }
1302 total_number_of_lines_++;
1303 }
1304
1305 size_t total_number_of_lines() { return total_number_of_lines_; }
1306
1307 bool HasNextChunk() { return reading_chunk_index_ < chunks_.size(); }
1308 DisassemblyChunk NextChunk() {
1309 return std::move(chunks_[reading_chunk_index_++]);
1310 }
1311
1312 private:
1313 // For a large Ritz module, the average is about 50 chars per line,
1314 // so (with 2-byte String16 chars) this should give approximately 20 MB
1315 // per chunk.
1316 static constexpr size_t kLinesPerChunk = 200'000;
1317
1318 size_t writing_chunk_index_ = 0;
1319 size_t reading_chunk_index_ = 0;
1320 size_t total_number_of_lines_ = 0;
1321 std::vector<DisassemblyChunk> chunks_;
1322};

1573Response V8DebuggerAgentImpl::evaluateOnCallFrame(
1574 const String16& callFrameId, const String16& expression,
1575 std::optional<String16> objectGroup,
1576 std::optional<bool> includeCommandLineAPI, std::optional<bool> silent,
1577 std::optional<bool> returnByValue, std::optional<bool> generatePreview,
1578 std::optional<bool> throwOnSideEffect, std::optional<double> timeout,
1579 std::unique_ptr<RemoteObject>* result,
1580 std::unique_ptr<protocol::Runtime::ExceptionDetails>* exceptionDetails) {
1581 if (!isPaused()) return Response::ServerError(kDebuggerNotPaused);
1582 InjectedScript::CallFrameScope scope(m_session, callFrameId);
1583 Response response = scope.initialize();
1584 if (!response.IsSuccess()) return response;
1585 if (includeCommandLineAPI.value_or(false)) scope.installCommandLineAPI();
1586 if (silent.value_or(false)) scope.ignoreExceptionsAndMuteConsole();
1587
1588 int frameOrdinal = static_cast<int>(scope.frameOrdinal());
1589 auto it = v8::debug::StackTraceIterator::Create(m_isolate, frameOrdinal);
1590 if (it->Done()) {
1591 return Response::ServerError("Could not find call frame with given id");
1592 }
1593
1594 v8::MaybeLocal<v8::Value> maybeResultValue;
1595 {
1596 V8InspectorImpl::EvaluateScope evaluateScope(scope);
1597 if (timeout.has_value()) {
1598 response = evaluateScope.setTimeout(timeout.value() / 1000.0);
1599 if (!response.IsSuccess()) return response;
1600 }
1601 maybeResultValue = it->Evaluate(toV8String(m_isolate, expression),
1602 throwOnSideEffect.value_or(false));
1603 }
1604 // Re-initialize after running client's code, as it could have destroyed
1605 // context or session.
1606 response = scope.initialize();
1607 if (!response.IsSuccess()) return response;
1608 WrapOptions wrapOptions = generatePreview.value_or(false)
1609 ? WrapOptions({WrapMode::kPreview})
1610 : WrapOptions({WrapMode::kIdOnly});
1611 if (returnByValue.value_or(false))
1612 wrapOptions = WrapOptions({WrapMode::kJson});
1613 return scope.injectedScript()->wrapEvaluateResult(
1614 maybeResultValue, scope.tryCatch(), objectGroup.value_or(""), wrapOptions,
1615 throwOnSideEffect.value_or(false), result, exceptionDetails);
1616}

1773Response V8DebuggerAgentImpl::currentCallFrames(
1774 std::unique_ptr<Array<CallFrame>>* result) {
1775 if (!isPaused()) {
1776 *result = std::make_unique<Array<CallFrame>>();
1777 return Response::Success();
1778 }
1779 v8::HandleScope handles(m_isolate);
1780 *result = std::make_unique<Array<CallFrame>>();
1781 auto iterator = v8::debug::StackTraceIterator::Create(m_isolate);
1782 int frameOrdinal = 0;
1783 for (; !iterator->Done(); iterator->Advance(), frameOrdinal++) {
1784 int contextId = iterator->GetContextId();
1785 InjectedScript* injectedScript = nullptr;
1786 if (contextId) m_session->findInjectedScript(contextId, injectedScript);
1787 String16 callFrameId = RemoteCallFrameId::serialize(
1788 m_inspector->isolateId(), contextId, frameOrdinal);
1789
1790 v8::debug::Location loc = iterator->GetSourceLocation();
1791
1792 std::unique_ptr<Array<Scope>> scopes;
1793 auto scopeIterator = iterator->GetScopeIterator();
1794 Response res =
1795 buildScopes(m_isolate, scopeIterator.get(), injectedScript, &scopes);
1796 if (!res.IsSuccess()) return res;
1797
1798 std::unique_ptr<RemoteObject> protocolReceiver;
1799 if (injectedScript) {
1800 v8::Local<v8::Value> receiver;
1801 if (iterator->GetReceiver().ToLocal(&receiver)) {
1802 res = injectedScript->wrapObject(receiver, kBacktraceObjectGroup,
1803 WrapOptions({WrapMode::kIdOnly}),
1804 &protocolReceiver);
1805 if (!res.IsSuccess()) return res;
1806 }
1807 }
1808 if (!protocolReceiver) {
1809 protocolReceiver = RemoteObject::create()
1810 .setType(RemoteObject::TypeEnum::Undefined)
1811 .build();
1812 }
1813
1814 v8::Local<v8::debug::Script> script = iterator->GetScript();
1815 DCHECK(!script.IsEmpty());
1816 std::unique_ptr<protocol::Debugger::Location> location =
1817 protocol::Debugger::Location::create()
1818 .setScriptId(String16::fromInteger(script->Id()))
1819 .setLineNumber(loc.GetLineNumber())
1820 .setColumnNumber(loc.GetColumnNumber())
1821 .build();
1822
1823 auto frame = CallFrame::create()
1824 .setCallFrameId(callFrameId)
1825 .setFunctionName(toProtocolString(
1826 m_isolate, iterator->GetFunctionDebugName()))
1827 .setLocation(std::move(location))
1828 .setUrl(String16())
1829 .setScopeChain(std::move(scopes))
1830 .setThis(std::move(protocolReceiver))
1831 .setCanBeRestarted(iterator->CanBeRestarted())
1832 .build();
1833
1834 v8::debug::Location func_loc = iterator->GetFunctionLocation();
