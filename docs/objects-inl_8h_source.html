296template <class T>
297T HeapObject::Relaxed_ReadField(size_t offset) const
298 requires((std::is_arithmetic_v<T> || std::is_enum_v<T>) &&
299 !std::is_floating_point_v<T>)
300{
301 // Pointer compression causes types larger than kTaggedSize to be
302 // unaligned. Atomic loads must be aligned.
303 DCHECK_IMPLIES(COMPRESS_POINTERS_BOOL, sizeof(T) <= kTaggedSize);
304 using AtomicT = typename base::AtomicTypeFromByteWidth<sizeof(T)>::type;
305 return static_cast<T>(base::AsAtomicImpl<AtomicT>::Relaxed_Load(
306 reinterpret_cast<AtomicT*>(field_address(offset))));
307}

309template <class T>
310void HeapObject::Relaxed_WriteField(size_t offset, T value)
311 requires((std::is_arithmetic_v<T> || std::is_enum_v<T>) &&
312 !std::is_floating_point_v<T>)
313{
314 // Pointer compression causes types larger than kTaggedSize to be
315 // unaligned. Atomic stores must be aligned.
316 DCHECK_IMPLIES(COMPRESS_POINTERS_BOOL, sizeof(T) <= kTaggedSize);
317 using AtomicT = typename base::AtomicTypeFromByteWidth<sizeof(T)>::type;
318 base::AsAtomicImpl<AtomicT>::Relaxed_Store(
319 reinterpret_cast<AtomicT*>(field_address(offset)),
320 static_cast<AtomicT>(value));
321}

337template <typename CompareAndSwapImpl>
338Tagged<Object> HeapObject::SeqCst_CompareAndSwapField(
339 Tagged<Object> expected, Tagged<Object> value,
340 CompareAndSwapImpl compare_and_swap_impl) {
341 Tagged<Object> actual_expected = expected;
342 do {
343 Tagged<Object> old_value = compare_and_swap_impl(actual_expected, value);
344 if (old_value == actual_expected || !IsNumber(old_value) ||
345 !IsNumber(actual_expected)) {
346 return old_value;
347 }
348 if (!Object::SameNumberValue(
349 Object::NumberValue(Cast<Number>(old_value)),
350 Object::NumberValue(Cast<Number>(actual_expected)))) {
351 return old_value;
352 }
353 // The pointer comparison failed, but the numbers are equal. This can
354 // happen even if both numbers are HeapNumbers with the same value.
355 // Try again in the next iteration.
356 actual_expected = old_value;
357 } while (true);
358}

517DEF_HEAP_OBJECT_PREDICATE(HeapObject, IsDeoptimizationData) {
518 // Must be a (protected) fixed array.
519 if (!IsProtectedFixedArray(obj, cage_base)) return false;
520
521 // There's no sure way to detect the difference between a fixed array and
522 // a deoptimization data array. Since this is used for asserts we can
523 // check that the length is zero or else the fixed size plus a multiple of
524 // the entry size.
525 int length = Cast<ProtectedFixedArray>(obj)->length();
526 if (length == 0) return true;
527
528 length -= DeoptimizationData::kFirstDeoptEntryIndex;
529 return length >= 0 && length % DeoptimizationData::kDeoptEntrySize == 0;
530}

675// static
676bool Object::SameNumberValue(double value1, double value2) {
677 // Compare values bitwise, to cover -0 being different from 0 -- we'd need to
678 // look at sign bits anyway if we'd done a double comparison, so we may as
679 // well compare bitwise immediately.
680 uint64_t value1_bits = base::bit_cast<uint64_t>(value1);
681 uint64_t value2_bits = base::bit_cast<uint64_t>(value2);
682 if (value1_bits == value2_bits) {
683 return true;
684 }
685 // SameNumberValue(NaN, NaN) is true even for NaNs with different bit
686 // representations.
687 return std::isnan(value1) && std::isnan(value2);
688}

988template <ExternalPointerTag tag>
989void HeapObject::WriteLazilyInitializedExternalPointerField(
990 size_t offset, IsolateForSandbox isolate, Address value) {
991#ifdef V8_ENABLE_SANDBOX
992 static_assert(tag != kExternalPointerNullTag);
993 ExternalPointerTable& table = isolate.GetExternalPointerTableFor(tag);
994 auto location =
995 reinterpret_cast<ExternalPointerHandle*>(field_address(offset));
996 ExternalPointerHandle handle = base::AsAtomic32::Relaxed_Load(location);

998 // Field has not been initialized yet.
999 ExternalPointerHandle handle = table.AllocateAndInitializeEntry(
1000 isolate.GetExternalPointerTableSpaceFor(tag, address()), value, tag);
1001 base::AsAtomic32::Release_Store(location, handle);
1002 // In this case, we're adding a reference from an existing object to a new
1003 // table entry, so we always require a write barrier.
1004 EXTERNAL_POINTER_WRITE_BARRIER(*this, static_cast<int>(offset), tag);
1005 } else {
1006 table.Set(handle, value, tag);
1007 }
1008#else
1009 WriteMaybeUnalignedValue<Address>(field_address(offset), value);
1010#endif // V8_ENABLE_SANDBOX

1217 // Smi representing a diff from the source object address in kObjectAlignment
1218 // chunks.
1219 intptr_t diff = static_cast<intptr_t>(object.ptr() - map_word_host.ptr());
1220 DCHECK(IsAligned(diff, kObjectAlignment));
1221 MapWord map_word(Smi::FromIntptr(diff / kObjectAlignment).ptr());
1222 DCHECK(map_word.IsForwardingAddress());
1223 return map_word;
1224#else
1225 return MapWord(object.ptr() - kHeapObjectTag);
1226#endif // V8_EXTERNAL_CODE_SPACE
1227}

1234 // pointers are encoded as Smi representing a diff from the source object
1235 // address in kObjectAlignment chunks. This is required as we are using
1236 // multiple pointer compression cages in these scenarios.

1386template <HeapObject::EmitWriteBarrier emit_write_barrier, typename MemoryOrder,
1387 typename IsolateT>
1388void HeapObject::set_map(IsolateT* isolate, Tagged<Map> value,
1389 MemoryOrder order, VerificationMode mode) {
1390#if V8_ENABLE_WEBASSEMBLY
1391 // In {WasmGraphBuilder::SetMap} and {WasmGraphBuilder::LoadMap}, we treat
1392 // maps as immutable. Therefore we are not allowed to mutate them here.
1393 DCHECK(!IsWasmStructMap(value) && !IsWasmArrayMap(value));
1394#endif
1395 // Object layout changes are currently not supported on background threads.
1396 // This method might change object layout and therefore can't be used on
1397 // background threads.
1398 DCHECK_IMPLIES(mode != VerificationMode::kSafeMapTransition,
1399 !LocalHeap::Current());
1400 if (v8_flags.verify_heap && !value.is_null()) {
1401 if (mode == VerificationMode::kSafeMapTransition) {
1402 HeapVerifier::VerifySafeMapTransition(isolate->heap()->AsHeap(), *this,
1403 value);
1404 } else {
1405 DCHECK_EQ(mode, VerificationMode::kPotentialLayoutChange);
1406 HeapVerifier::VerifyObjectLayoutChange(isolate->heap()->AsHeap(), *this,
1407 value);
1408 }
1409 }
1410 set_map_word(value, order);
1411 Heap::NotifyObjectLayoutChangeDone(*this);
1412#ifndef V8_DISABLE_WRITE_BARRIERS
1413 if (!value.is_null()) {
1414 if (emit_write_barrier == EmitWriteBarrier::kYes) {
1415 WriteBarrier::ForValue(*this, MaybeObjectSlot(map_slot()), value,
1416 UPDATE_WRITE_BARRIER);
1417 } else {
1418 DCHECK_EQ(emit_write_barrier, EmitWriteBarrier::kNo);
1419 SLOW_DCHECK(!WriteBarrier::IsRequired(*this, value));
1420 }
1421 }
1422#endif
1423}

1601 *index = result;
1602 return true;
1603 }

1744// static
1745Tagged<Object> Object::GetSimpleHash(Tagged<Object> object) {
1746 DisallowGarbageCollection no_gc;
1747 if (IsSmi(object)) {
1748 uint32_t hash = ComputeUnseededHash(Smi::ToInt(object));
1749 return Smi::FromInt(hash & Smi::kMaxValue);
1750 }
1751 auto instance_type = Cast<HeapObject>(object)->map()->instance_type();
1752 if (InstanceTypeChecker::IsHeapNumber(instance_type)) {
1753 double num = Cast<HeapNumber>(object)->value();
1754 if (std::isnan(num)) return Smi::FromInt(Smi::kMaxValue);
1755 // Use ComputeUnseededHash for all values in Signed32 range, including -0,
1756 // which is considered equal to 0 because collections use SameValueZero.
1757 uint32_t hash;
1758 // Check range before conversion to avoid undefined behavior.
1759 if (num >= kMinInt && num <= kMaxInt && FastI2D(FastD2I(num)) == num) {
1760 hash = ComputeUnseededHash(FastD2I(num));
1761 } else {
1762 hash = ComputeLongHash(base::double_to_uint64(num));
1763 }
1764 return Smi::FromInt(hash & Smi::kMaxValue);
1765 } else if (InstanceTypeChecker::IsName(instance_type)) {
1766 uint32_t hash = Cast<Name>(object)->EnsureHash();
1767 return Smi::FromInt(hash);
1768 } else if (InstanceTypeChecker::IsOddball(instance_type)) {
1769 uint32_t hash = Cast<Oddball>(object)->to_string()->EnsureHash();
1770 return Smi::FromInt(hash);
1771 } else if (InstanceTypeChecker::IsBigInt(instance_type)) {
1772 uint32_t hash = Cast<BigInt>(object)->Hash();
1773 return Smi::FromInt(hash & Smi::kMaxValue);
1774 } else if (InstanceTypeChecker::IsSharedFunctionInfo(instance_type)) {
1775 uint32_t hash = Cast<SharedFunctionInfo>(object)->Hash();
1776 return Smi::FromInt(hash & Smi::kMaxValue);
1777 } else if (InstanceTypeChecker::IsScopeInfo(instance_type)) {
1778 uint32_t hash = Cast<ScopeInfo>(object)->Hash();
1779 return Smi::FromInt(hash & Smi::kMaxValue);
1780 } else if (InstanceTypeChecker::IsScript(instance_type)) {
1781 int id = Cast<Script>(object)->id();
1782 return Smi::FromInt(ComputeUnseededHash(id) & Smi::kMaxValue);

1802bool IsShared(Tagged<Object> obj) {
1803 // This logic should be kept in sync with fast paths in
1804 // CodeStubAssembler::SharedValueBarrier.
1805
1806 // Smis are trivially shared.
1807 if (IsSmi(obj)) return true;
1808
1809 Tagged<HeapObject> object = Cast<HeapObject>(obj);
1810
1811 // RO objects are shared when the RO space is shared.
1812 if (HeapLayout::InReadOnlySpace(object)) {
1813 return true;
1814 }
1815
1816 // Check if this object is already shared.
1817 InstanceType instance_type = object->map()->instance_type();
1818 if (InstanceTypeChecker::IsAlwaysSharedSpaceJSObject(instance_type)) {
1819 DCHECK(HeapLayout::InAnySharedSpace(object));
1820 return true;
1821 }
1822 switch (instance_type) {
1823 case SHARED_SEQ_TWO_BYTE_STRING_TYPE:
1824 case SHARED_SEQ_ONE_BYTE_STRING_TYPE:
1825 case SHARED_EXTERNAL_TWO_BYTE_STRING_TYPE:
1826 case SHARED_EXTERNAL_ONE_BYTE_STRING_TYPE:
1827 case SHARED_UNCACHED_EXTERNAL_TWO_BYTE_STRING_TYPE:
1828 case SHARED_UNCACHED_EXTERNAL_ONE_BYTE_STRING_TYPE:
1829 DCHECK(HeapLayout::InAnySharedSpace(object));
1830 return true;
1831 case INTERNALIZED_TWO_BYTE_STRING_TYPE:
1832 case INTERNALIZED_ONE_BYTE_STRING_TYPE:
1833 case EXTERNAL_INTERNALIZED_TWO_BYTE_STRING_TYPE:
1834 case EXTERNAL_INTERNALIZED_ONE_BYTE_STRING_TYPE:
1835 case UNCACHED_EXTERNAL_INTERNALIZED_TWO_BYTE_STRING_TYPE:
1836 case UNCACHED_EXTERNAL_INTERNALIZED_ONE_BYTE_STRING_TYPE:
1837 if (v8_flags.shared_string_table) {
1838 DCHECK(HeapLayout::InAnySharedSpace(object));
1839 return true;
1840 }
1841 return false;
1842 case HEAP_NUMBER_TYPE:
1843 return HeapLayout::InWritableSharedSpace(object);
1844 default:
1845 return false;
1846 }
1847}
