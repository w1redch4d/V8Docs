17uint32_t CppHeapPointerTable::SweepAndCompact(Space* space,
18 Counters* counters) {
19 DCHECK(space->BelongsTo(this));
20
21 // Lock the space. Technically this is not necessary since no other thread can
22 // allocate entries at this point, but some of the methods we call on the
23 // space assert that the lock is held.
24 base::SpinningMutexGuard guard(&space->mutex_);
25 // Same for the invalidated fields mutex.
26 base::SpinningMutexGuard invalidated_fields_guard(
27 &space->invalidated_fields_mutex_);
28
29 // There must not be any entry allocations while the table is being swept as
30 // that would not be safe. Set the freelist to this special marker value to
31 // easily catch any violation of this requirement.
32 space->freelist_head_.store(kEntryAllocationIsForbiddenMarker,
33 std::memory_order_relaxed);
34
35 // When compacting, we can compute the number of unused segments at the end of
36 // the table and skip those during sweeping.
37 uint32_t start_of_evacuation_area =
38 space->start_of_evacuation_area_.load(std::memory_order_relaxed);
39 bool evacuation_was_successful = false;
40 if (space->IsCompacting()) {
41 if (space->CompactingWasAborted()) {
42 // Extract the original start_of_evacuation_area value so that the
43 // DCHECKs below and in TryResolveEvacuationEntryDuringSweeping work.
44 start_of_evacuation_area &= ~Space::kCompactionAbortedMarker;
45 } else {
46 evacuation_was_successful = true;
47 }
48 DCHECK(IsAligned(start_of_evacuation_area, kEntriesPerSegment));
49
50 space->StopCompacting();
51 }
52
53 // Sweep top to bottom and rebuild the freelist from newly dead and
54 // previously freed entries while also clearing the marking bit on live
55 // entries and resolving evacuation entries table when compacting the table.
56 // This way, the freelist ends up sorted by index which already makes the
57 // table somewhat self-compacting and is required for the compaction
58 // algorithm so that evacuated entries are evacuated to the start of a space.
59 // This method must run either on the mutator thread or while the mutator is
60 // stopped.
61 uint32_t current_freelist_head = 0;
62 uint32_t current_freelist_length = 0;
63 auto AddToFreelist = [&](uint32_t entry_index) {
64 at(entry_index).MakeFreelistEntry(current_freelist_head);
65 current_freelist_head = entry_index;
66 current_freelist_length++;
67 };
68
69 std::vector<Segment> segments_to_deallocate;
70 for (auto segment : base::Reversed(space->segments_)) {
71 bool segment_will_be_evacuated =
72 evacuation_was_successful &&
73 segment.first_entry() >= start_of_evacuation_area;
74 // Remember the state of the freelist before this segment in case this
75 // segment turns out to be completely empty and we deallocate it.
76 uint32_t previous_freelist_head = current_freelist_head;
77 uint32_t previous_freelist_length = current_freelist_length;
78
79 // Process every entry in this segment, again going top to bottom.
80 for (uint32_t i = segment.last_entry(); i >= segment.first_entry(); i--) {
81 auto payload = at(i).GetRawPayload();
82 if (payload.ContainsEvacuationEntry()) {
83 // Segments that will be evacuated cannot contain evacuation entries
84 // into which other entries would be evacuated.
85 DCHECK(!segment_will_be_evacuated);
86
87 // An evacuation entry contains the address of the slot that owns the
88 // entry that is to be evacuated.
89 Address handle_location =
90 payload.ExtractEvacuationEntryHandleLocation();
91
92 // The CppHeapPointerTable does not support field invalidation.
93 DCHECK(!space->FieldWasInvalidated(handle_location));
94
95 // Resolve the evacuation entry: take the pointer to the handle from the
96 // evacuation entry, copy the entry to its new location, and finally
97 // update the handle to point to the new entry.
98 //
99 // While we now know that the entry being evacuated is free, we don't
100 // add it to (the start of) the freelist because that would immediately
101 // cause new fragmentation when the next entry is allocated. Instead, we
102 // assume that the segments out of which entries are evacuated will all
103 // be decommitted anyway after this loop, which is usually the case
104 // unless compaction was already aborted during marking.
105 ResolveEvacuationEntryDuringSweeping(
106 i, reinterpret_cast<CppHeapPointerHandle*>(handle_location),
107 start_of_evacuation_area);
108
109 // The entry must now contain a pointer and be unmarked as the entry
110 // that was evacuated must have been processed already (it is in an
111 // evacuated segment, which are processed first as they are at the end
112 // of the space). This will have cleared the marking bit.
113 DCHECK(at(i).GetRawPayload().ContainsPointer());
114 DCHECK(!at(i).GetRawPayload().HasMarkBitSet());
115 } else if (!payload.HasMarkBitSet()) {
116 AddToFreelist(i);
117 } else {
118 auto new_payload = payload;
119 new_payload.ClearMarkBit();
120 at(i).SetRawPayload(new_payload);
121 }
122
123 // We must have resolved all evacuation entries. Otherwise, we'll try to
124 // process them again during the next GC, which would cause problems.
125 DCHECK(!at(i).HasEvacuationEntry());
126 }
127
128 // If a segment is completely empty, or if all live entries will be
129 // evacuated out of it at the end of this loop, free the segment.
130 // Note: for segments that will be evacuated, we could avoid building up a
131 // freelist, but it's probably not worth the effort.
132 uint32_t free_entries = current_freelist_length - previous_freelist_length;
133 bool segment_is_empty = free_entries == kEntriesPerSegment;
134 if (segment_is_empty || segment_will_be_evacuated) {
135 segments_to_deallocate.push_back(segment);
136 // Restore the state of the freelist before this segment.
137 current_freelist_head = previous_freelist_head;
138 current_freelist_length = previous_freelist_length;
139 }
140 }
141
142 // We cannot deallocate the segments during the above loop, so do it now.
143 for (auto segment : segments_to_deallocate) {
144 FreeTableSegment(segment);
145 space->segments_.erase(segment);
146 }
147
148 FreelistHead new_freelist(current_freelist_head, current_freelist_length);
149 space->freelist_head_.store(new_freelist, std::memory_order_release);
150 DCHECK_EQ(space->freelist_length(), current_freelist_length);
151
152 uint32_t num_live_entries = space->capacity() - current_freelist_length;
153 counters->cppheap_pointers_count()->AddSample(num_live_entries);
154 return num_live_entries;
155}

157void CppHeapPointerTable::ResolveEvacuationEntryDuringSweeping(
158 uint32_t new_index, CppHeapPointerHandle* handle_location,
159 uint32_t start_of_evacuation_area) {
160 CppHeapPointerHandle old_handle = *handle_location;
161 CHECK(IsValidHandle(old_handle));
162
163 uint32_t old_index = HandleToIndex(old_handle);
164 CppHeapPointerHandle new_handle = IndexToHandle(new_index);
165
166 // The compaction algorithm always moves an entry from the evacuation area to
167 // the front of the table. These DCHECKs verify this invariant.
168 DCHECK_GE(old_index, start_of_evacuation_area);
169 DCHECK_LT(new_index, start_of_evacuation_area);
170 auto& new_entry = at(new_index);
171 at(old_index).Evacuate(new_entry);
172 *handle_location = new_handle;
173}
