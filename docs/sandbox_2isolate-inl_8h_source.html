90ExternalPointerTable::Space*
91IsolateForPointerCompression::GetExternalPointerTableSpaceFor(
92 ExternalPointerTagRange tag_range, Address host) {
93 DCHECK(!tag_range.IsEmpty());
94
95 if (V8_UNLIKELY(IsSharedExternalPointerType(tag_range))) {
96 DCHECK(!ReadOnlyHeap::Contains(host));
97 return isolate_->shared_external_pointer_space();
98 }
99
100 if (V8_UNLIKELY(IsMaybeReadOnlyExternalPointerType(tag_range) &&
101 ReadOnlyHeap::Contains(host))) {
102 return isolate_->heap()->read_only_external_pointer_space();
103 }
104
105 if (HeapLayout::InYoungGeneration(HeapObject::FromAddress(host))) {
106 return isolate_->heap()->young_external_pointer_space();
107 }
108
109 return isolate_->heap()->old_external_pointer_space();
110}
