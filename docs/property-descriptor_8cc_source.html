21// Helper function for ToPropertyDescriptor. Comments describe steps for
22// "enumerable", other properties are handled the same way.
23// Returns false if an exception was thrown.
24bool GetPropertyIfPresent(Isolate* isolate, DirectHandle<JSReceiver> receiver,
25 DirectHandle<String> name, Handle<JSAny>* value) {
26 LookupIterator it(isolate, receiver, name, receiver);
27 // 4. Let hasEnumerable be HasProperty(Obj, "enumerable").
28 Maybe<bool> has_property = JSReceiver::HasProperty(&it);
29 // 5. ReturnIfAbrupt(hasEnumerable).
30 if (has_property.IsNothing()) return false;
31 // 6. If hasEnumerable is true, then
32 if (has_property.FromJust() == true) {
33 // 6a. Let enum be ToBoolean(Get(Obj, "enumerable")).
34 // 6b. ReturnIfAbrupt(enum).
35 if (!Cast<JSAny>(Object::GetProperty(&it)).ToHandle(value)) return false;
36 }
37 return true;
38}

40// Helper function for ToPropertyDescriptor. Handles the case of "simple"
41// objects: nothing on the prototype chain, just own fast data properties.
42// Must not have observable side effects, because the slow path will restart
43// the entire conversion!
44bool ToPropertyDescriptorFastPath(Isolate* isolate,
45 DirectHandle<JSReceiver> obj,
46 PropertyDescriptor* desc) {
47 {
48 DisallowGarbageCollection no_gc;
49 Tagged<JSReceiver> raw_obj = *obj;
50 if (!IsJSObject(*raw_obj)) return false;
51 Tagged<Map> raw_map = raw_obj->map(isolate);
52 if (raw_map->instance_type() != JS_OBJECT_TYPE) return false;
53 if (raw_map->is_access_check_needed()) return false;
54 if (raw_map->prototype() != *isolate->initial_object_prototype())
55 return false;
56 // During bootstrapping, the object_function_prototype_map hasn't been
57 // set up yet.
58 if (isolate->bootstrapper()->IsActive()) return false;
59 if (Cast<JSObject>(raw_map->prototype())->map() !=
60 isolate->raw_native_context()->object_function_prototype_map()) {
61 return false;
62 }
63 // TODO(jkummerow): support dictionary properties?
64 if (raw_map->is_dictionary_map()) return false;
65 }
66
67 DirectHandle<Map> map(obj->map(isolate), isolate);
68
69 DirectHandle<DescriptorArray> descs(map->instance_descriptors(isolate),
70 isolate);
71 ReadOnlyRoots roots(isolate);
72 for (InternalIndex i : map->IterateOwnDescriptors()) {
73 PropertyDetails details = descs->GetDetails(i);
74 DirectHandle<Object> value;
75 if (details.location() == PropertyLocation::kField) {
76 if (details.kind() == PropertyKind::kData) {
77 value = JSObject::FastPropertyAt(isolate, Cast<JSObject>(obj),
78 details.representation(),
79 FieldIndex::ForDetails(*map, details));
80 } else {
81 DCHECK_EQ(PropertyKind::kAccessor, details.kind());
82 // Bail out to slow path.
83 return false;
84 }
85
86 } else {
87 DCHECK_EQ(PropertyLocation::kDescriptor, details.location());
88 if (details.kind() == PropertyKind::kData) {
89 value = direct_handle(descs->GetStrongValue(i), isolate);
90 } else {
91 DCHECK_EQ(PropertyKind::kAccessor, details.kind());
92 // Bail out to slow path.
93 return false;
94 }
95 }
96 Tagged<Name> key = descs->GetKey(i);
97 if (key == roots.enumerable_string()) {
98 desc->set_enumerable(Object::BooleanValue(*value, isolate));
99 } else if (key == roots.configurable_string()) {
100 desc->set_configurable(Object::BooleanValue(*value, isolate));
101 } else if (key == roots.value_string()) {
102 desc->set_value(Cast<JSAny>(value));
103 } else if (key == roots.writable_string()) {
104 desc->set_writable(Object::BooleanValue(*value, isolate));
105 } else if (key == roots.get_string()) {
106 // Bail out to slow path to throw an exception if necessary.
107 if (!IsCallable(*value)) return false;
108 desc->set_get(Cast<JSAny>(value));
109 } else if (key == roots.set_string()) {
110 // Bail out to slow path to throw an exception if necessary.
111 if (!IsCallable(*value)) return false;
112 desc->set_set(Cast<JSAny>(value));
113 }
114 }
115 if ((desc->has_get() || desc->has_set()) &&
116 (desc->has_value() || desc->has_writable())) {
117 // Bail out to slow path to throw an exception.
118 return false;
119 }
120 return true;
121}

132// ES6 6.2.4.4 "FromPropertyDescriptor"
133Handle<JSObject> PropertyDescriptor::ToObject(Isolate* isolate) {
134 DCHECK(!(PropertyDescriptor::IsAccessorDescriptor(this) &&
135 PropertyDescriptor::IsDataDescriptor(this)));
136 Factory* factory = isolate->factory();
137 if (IsRegularAccessorProperty()) {
138 // Fast case for regular accessor properties.
139 Handle<JSObject> result = factory->NewJSObjectFromMap(
140 isolate->accessor_property_descriptor_map());
141 result->InObjectPropertyAtPut(JSAccessorPropertyDescriptor::kGetIndex,
142 *get());
143 result->InObjectPropertyAtPut(JSAccessorPropertyDescriptor::kSetIndex,
144 *set());
145 result->InObjectPropertyAtPut(
146 JSAccessorPropertyDescriptor::kEnumerableIndex,
147 isolate->heap()->ToBoolean(enumerable()));
148 result->InObjectPropertyAtPut(
149 JSAccessorPropertyDescriptor::kConfigurableIndex,
150 isolate->heap()->ToBoolean(configurable()));
151 return result;
152 }
153 if (IsRegularDataProperty()) {
154 // Fast case for regular data properties.
155 Handle<JSObject> result =
156 factory->NewJSObjectFromMap(isolate->data_property_descriptor_map());
157 result->InObjectPropertyAtPut(JSDataPropertyDescriptor::kValueIndex,
158 *value());
159 result->InObjectPropertyAtPut(JSDataPropertyDescriptor::kWritableIndex,
160 isolate->heap()->ToBoolean(writable()));
161 result->InObjectPropertyAtPut(JSDataPropertyDescriptor::kEnumerableIndex,
162 isolate->heap()->ToBoolean(enumerable()));
163 result->InObjectPropertyAtPut(JSDataPropertyDescriptor::kConfigurableIndex,
164 isolate->heap()->ToBoolean(configurable()));
165 return result;
166 }
167 Handle<JSObject> result = factory->NewJSObject(isolate->object_function());
168 if (has_value()) {
169 CreateDataProperty(isolate, result, factory->value_string(), value());
170 }
171 if (has_writable()) {
172 CreateDataProperty(isolate, result, factory->writable_string(),
173 factory->ToBoolean(writable()));
174 }
175 if (has_get()) {
176 CreateDataProperty(isolate, result, factory->get_string(), get());
177 }
178 if (has_set()) {
179 CreateDataProperty(isolate, result, factory->set_string(), set());
180 }
181 if (has_enumerable()) {
182 CreateDataProperty(isolate, result, factory->enumerable_string(),
183 factory->ToBoolean(enumerable()));
184 }
185 if (has_configurable()) {
186 CreateDataProperty(isolate, result, factory->configurable_string(),
187 factory->ToBoolean(configurable()));
188 }
189 return result;
190}

192// ES6 6.2.4.5
193// Returns false in case of exception.
194// static
195bool PropertyDescriptor::ToPropertyDescriptor(Isolate* isolate,
196 Handle<JSAny> obj,
197 PropertyDescriptor* desc) {
198 // 1. ReturnIfAbrupt(Obj).
199 // 2. If Type(Obj) is not Object, throw a TypeError exception.
200 if (!IsJSReceiver(*obj)) {
201 isolate->Throw(*isolate->factory()->NewTypeError(
202 MessageTemplate::kPropertyDescObject, obj));
203 return false;
204 }
205 // 3. Let desc be a new Property Descriptor that initially has no fields.
206 DCHECK(desc->is_empty());
207
208 DirectHandle<JSReceiver> receiver = Cast<JSReceiver>(obj);
209 if (ToPropertyDescriptorFastPath(isolate, receiver, desc)) {
210 return true;
211 }
212
213 // enumerable?
214 Handle<JSAny> enumerable;
215 // 4 through 6b.
216 if (!GetPropertyIfPresent(isolate, receiver,
217 isolate->factory()->enumerable_string(),
218 &enumerable)) {
219 return false;
220 }
221 // 6c. Set the [[Enumerable]] field of desc to enum.
222 if (!enumerable.is_null()) {
223 desc->set_enumerable(Object::BooleanValue(*enumerable, isolate));
224 }
225
226 // configurable?
227 Handle<JSAny> configurable;
228 // 7 through 9b.
229 if (!GetPropertyIfPresent(isolate, receiver,
230 isolate->factory()->configurable_string(),
231 &configurable)) {
232 return false;
233 }
234 // 9c. Set the [[Configurable]] field of desc to conf.
235 if (!configurable.is_null()) {
236 desc->set_configurable(Object::BooleanValue(*configurable, isolate));
237 }
238
239 // value?
240 Handle<JSAny> value;
241 // 10 through 12b.
242 if (!GetPropertyIfPresent(isolate, receiver,
243 isolate->factory()->value_string(), &value)) {
244 return false;
245 }
246 // 12c. Set the [[Value]] field of desc to value.
247 if (!value.is_null()) desc->set_value(value);
248
249 // writable?
250 Handle<JSAny> writable;
251 // 13 through 15b.
252 if (!GetPropertyIfPresent(isolate, receiver,
253 isolate->factory()->writable_string(), &writable)) {
254 return false;
255 }
256 // 15c. Set the [[Writable]] field of desc to writable.
257 if (!writable.is_null())
258 desc->set_writable(Object::BooleanValue(*writable, isolate));
259
260 // getter?
261 Handle<JSAny> getter;
262 // 16 through 18b.
263 if (!GetPropertyIfPresent(isolate, receiver, isolate->factory()->get_string(),
264 &getter)) {
265 return false;
266 }
267 if (!getter.is_null()) {
268 // 18c. If IsCallable(getter) is false and getter is not undefined,
269 // throw a TypeError exception.
270 if (!IsCallable(*getter) && !IsUndefined(*getter, isolate)) {
271 isolate->Throw(*isolate->factory()->NewTypeError(
272 MessageTemplate::kObjectGetterCallable, getter));
273 return false;
274 }
275 // 18d. Set the [[Get]] field of desc to getter.
276 desc->set_get(getter);
277 }
278 // setter?
279 Handle<JSAny> setter;
280 // 19 through 21b.
281 if (!GetPropertyIfPresent(isolate, receiver, isolate->factory()->set_string(),
282 &setter)) {
283 return false;
284 }
285 if (!setter.is_null()) {
286 // 21c. If IsCallable(setter) is false and setter is not undefined,
287 // throw a TypeError exception.
288 if (!IsCallable(*setter) && !IsUndefined(*setter, isolate)) {
289 isolate->Throw(*isolate->factory()->NewTypeError(
290 MessageTemplate::kObjectSetterCallable, setter));
291 return false;
292 }
293 // 21d. Set the [[Set]] field of desc to setter.
294 desc->set_set(setter);
295 }
296
297 // 22. If either desc.[[Get]] or desc.[[Set]] is present, then
298 // 22a. If either desc.[[Value]] or desc.[[Writable]] is present,
299 // throw a TypeError exception.
300 if ((desc->has_get() || desc->has_set()) &&
301 (desc->has_value() || desc->has_writable())) {
302 isolate->Throw(*isolate->factory()->NewTypeError(
303 MessageTemplate::kValueAndAccessor, obj));
304 return false;
305 }
306
307 // 23. Return desc.
308 return true;
309}

311// ES6 6.2.4.6
312// static
313void PropertyDescriptor::CompletePropertyDescriptor(Isolate* isolate,
314 PropertyDescriptor* desc) {
315 // 1. ReturnIfAbrupt(Desc).
316 // 2. Assert: Desc is a Property Descriptor.
317 // 3. Let like be Record{
318 // [[Value]]: undefined, [[Writable]]: false,
319 // [[Get]]: undefined, [[Set]]: undefined,
320 // [[Enumerable]]: false, [[Configurable]]: false}.
321 // 4. If either IsGenericDescriptor(Desc) or IsDataDescriptor(Desc) is true,
322 // then:
323 if (!IsAccessorDescriptor(desc)) {
324 // 4a. If Desc does not have a [[Value]] field, set Desc.[[Value]] to
325 // like.[[Value]].
326 if (!desc->has_value()) {
327 desc->set_value(isolate->factory()->undefined_value());
328 }
329 // 4b. If Desc does not have a [[Writable]] field, set Desc.[[Writable]]
330 // to like.[[Writable]].
331 if (!desc->has_writable()) desc->set_writable(false);
332 } else {
333 // 5. Else,
334 // 5a. If Desc does not have a [[Get]] field, set Desc.[[Get]] to
335 // like.[[Get]].
336 if (!desc->has_get()) {
337 desc->set_get(isolate->factory()->undefined_value());
338 }
339 // 5b. If Desc does not have a [[Set]] field, set Desc.[[Set]] to
340 // like.[[Set]].
341 if (!desc->has_set()) {
342 desc->set_set(isolate->factory()->undefined_value());
343 }
344 }
345 // 6. If Desc does not have an [[Enumerable]] field, set
346 // Desc.[[Enumerable]] to like.[[Enumerable]].
347 if (!desc->has_enumerable()) desc->set_enumerable(false);
348 // 7. If Desc does not have a [[Configurable]] field, set
349 // Desc.[[Configurable]] to like.[[Configurable]].
350 if (!desc->has_configurable()) desc->set_configurable(false);
351 // 8. Return Desc.
352}

354Handle<PropertyDescriptorObject> PropertyDescriptor::ToPropertyDescriptorObject(
355 Isolate* isolate) {
356 Handle<PropertyDescriptorObject> obj =
357 isolate->factory()->NewPropertyDescriptorObject();
358
359 int flags =
360 PropertyDescriptorObject::IsEnumerableBit::encode(enumerable_) |
361 PropertyDescriptorObject::HasEnumerableBit::encode(has_enumerable_) |
362 PropertyDescriptorObject::IsConfigurableBit::encode(configurable_) |
363 PropertyDescriptorObject::HasConfigurableBit::encode(has_configurable_) |
364 PropertyDescriptorObject::IsWritableBit::encode(writable_) |
365 PropertyDescriptorObject::HasWritableBit::encode(has_writable_) |
366 PropertyDescriptorObject::HasValueBit::encode(has_value()) |
367 PropertyDescriptorObject::HasGetBit::encode(has_get()) |
368 PropertyDescriptorObject::HasSetBit::encode(has_set());
369
370 obj->set_flags(flags);
371
372 if (has_value()) obj->set_value(*value_);
373 if (has_get()) obj->set_get(*get_);
374 if (has_set()) obj->set_set(*set_);
375
376 return obj;
377}
