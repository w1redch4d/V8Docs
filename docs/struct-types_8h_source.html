20class StructTypeBase : public ZoneObject {
21 public:
22 StructTypeBase(uint32_t field_count, uint32_t* field_offsets,
23 const ValueTypeBase* reps, const bool* mutabilities)
24 : field_count_(field_count),
25 field_offsets_(field_offsets),
26 reps_(reps),
27 mutabilities_(mutabilities) {}
28
29 uint32_t field_count() const { return field_count_; }
30

33 return reps_[index];
34 }
35

37 DCHECK_LT(index, field_count_);
38 return mutabilities_[index];
39 }

43 return {reps_, reps_ + field_count_};
44 }
45 base::iterator_range<const bool*> mutabilities() const {
46 return {mutabilities_, mutabilities_ + field_count_};

49 // Returns the offset of this field in the runtime representation of the
50 // object, from the start of the object fields (disregarding the object
51 // header).
52 uint32_t field_offset(uint32_t index) const {
53 DCHECK_LT(index, field_count());
54 if (index == 0) return 0;
55 DCHECK(offsets_initialized_);
56 return field_offsets_[index - 1];
57 }
58 uint32_t total_fields_size() const {
59 return field_count() == 0 ? 0 : field_offsets_[field_count() - 1];
60 }
61
62 uint32_t Align(uint32_t offset, uint32_t alignment) {
63 return RoundUp(offset, std::min(alignment, uint32_t{kTaggedSize}));
64 }
65
66 void InitializeOffsets() {
67 if (field_count() == 0) return;
68 DCHECK(!offsets_initialized_);
69 uint32_t offset = field(0).value_kind_size();
70 // Optimization: we track the last gap that was introduced by alignment,
71 // and place any sufficiently-small fields in it.
72 // It's important that the algorithm that assigns offsets to fields is
73 // subtyping-safe, i.e. two lists of fields with a common prefix must
74 // always compute the same offsets for the fields in this common prefix.
75 uint32_t gap_position = 0;
76 uint32_t gap_size = 0;
77 for (uint32_t i = 1; i < field_count(); i++) {
78 uint32_t field_size = field(i).value_kind_size();
79 if (field_size <= gap_size) {
80 uint32_t aligned_gap = Align(gap_position, field_size);
81 uint32_t gap_before = aligned_gap - gap_position;
82 uint32_t aligned_gap_size = gap_size - gap_before;
83 if (field_size <= aligned_gap_size) {
84 field_offsets_[i - 1] = aligned_gap;
85 uint32_t gap_after = aligned_gap_size - field_size;
86 if (gap_before > gap_after) {
87 // Keep old {gap_position}.
88 gap_size = gap_before;
89 } else {
90 gap_position = aligned_gap + field_size;
91 gap_size = gap_after;
92 }
93 continue; // Successfully placed the field in the gap.
94 }
95 }
96 uint32_t old_offset = offset;
97 offset = Align(offset, field_size);
98 uint32_t gap = offset - old_offset;
99 if (gap > gap_size) {
100 gap_size = gap;
101 gap_position = old_offset;
102 }
103 field_offsets_[i - 1] = offset;
104 offset += field_size;
105 }
106 offset = RoundUp(offset, kTaggedSize);
107 field_offsets_[field_count() - 1] = offset;
108#if DEBUG
109 offsets_initialized_ = true;
110#endif
111 }

113 // For incrementally building StructTypes.
114 template <class Subclass, class ValueTypeSubclass>
115 class BuilderImpl {
116 public:
117 enum ComputeOffsets : bool {
118 kComputeOffsets = true,
119 kUseProvidedOffsets = false
120 };
121
122 BuilderImpl(Zone* zone, uint32_t field_count)
123 : zone_(zone),
124 field_count_(field_count),
125 cursor_(0),
126 field_offsets_(zone_->AllocateArray<uint32_t>(field_count_)),
127 buffer_(zone->AllocateArray<ValueTypeSubclass>(
128 static_cast<int>(field_count))),
129 mutabilities_(
130 zone->AllocateArray<bool>(static_cast<int>(field_count))) {}
131
132 void AddField(ValueTypeSubclass type, bool mutability,
133 uint32_t offset = 0) {
134 DCHECK_LT(cursor_, field_count_);
135 if (cursor_ > 0) {
136 field_offsets_[cursor_ - 1] = offset;
137 } else {
138 DCHECK_EQ(0, offset); // First field always has offset 0.
139 }
140 mutabilities_[cursor_] = mutability;
141 buffer_[cursor_++] = type;
142 }
143
144 void set_total_fields_size(uint32_t size) {
145 if (field_count_ == 0) {
146 DCHECK_EQ(0, size);
147 return;
148 }
149 field_offsets_[field_count_ - 1] = size;
150 }
151
152 Subclass* Build(ComputeOffsets compute_offsets = kComputeOffsets) {
153 DCHECK_EQ(cursor_, field_count_);
154 Subclass* result = zone_->New<Subclass>(field_count_, field_offsets_,
155 buffer_, mutabilities_);
156 if (compute_offsets == kComputeOffsets) {
157 result->InitializeOffsets();
158 } else {
159#if DEBUG
160 bool offsets_specified = true;
161 for (uint32_t i = 0; i < field_count_; i++) {
162 if (field_offsets_[i] == 0) {
163 offsets_specified = false;
164 break;
165 }
166 }
167 result->offsets_initialized_ = offsets_specified;
168#endif
169 }
170 return result;
171 }
172
173 private:
174 Zone* const zone_;
175 const uint32_t field_count_;
176 uint32_t cursor_;
177 uint32_t* field_offsets_;
178 ValueTypeSubclass* const buffer_;
179 bool* const mutabilities_;
180 };
181
182 static const size_t kMaxFieldOffset =
183 (kV8MaxWasmStructFields - 1) * kMaxValueTypeSize;

185 private:

189 const uint32_t field_count_;
190#if DEBUG
191 bool offsets_initialized_ = false;
192#endif
193 uint32_t* const field_offsets_;
194 const ValueTypeBase* const reps_;
195 const bool* const mutabilities_;
196};

199class StructType : public StructTypeBase {
200 public:
201 using Builder = StructTypeBase::BuilderImpl<StructType, ValueType>;
202
203 StructType(uint32_t field_count, uint32_t* field_offsets,
204 const ValueType* reps, const bool* mutabilities)
205 : StructTypeBase(field_count, field_offsets, reps, mutabilities) {}
206
207 bool operator==(const StructType& other) const {
208 if (this == &other) return true;
209 if (field_count() != other.field_count()) return false;
210 return std::equal(fields().begin(), fields().end(),
211 other.fields().begin()) &&
212 std::equal(mutabilities().begin(), mutabilities().end(),
213 other.mutabilities().begin());
214 }
215 bool operator!=(const StructType& other) const { return !(*this == other); }
216
217 ValueType field(uint32_t index) const {
218 return ValueType{StructTypeBase::field(index)};
219 }
220
221 base::iterator_range<const ValueType*> fields() const {
222 const ValueType* cast_reps = static_cast<const ValueType*>(reps_);
223 return {cast_reps, cast_reps + field_count_};
224 }
225};
