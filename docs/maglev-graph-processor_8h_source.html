20// The GraphProcessor takes a NodeProcessor, and applies it to each Node in the
21// Graph by calling NodeProcessor::Process on each Node.
22//
23// The GraphProcessor also keeps track of the current ProcessingState, and
24// passes this to the Process method.
25//
26// It expects a NodeProcessor class with:
27//
28// // A function that processes the graph before the nodes are walked.
29// void PreProcessGraph(Graph* graph);
30//
31// // A function that processes the graph after the nodes are walked.
32// void PostProcessGraph(Graph* graph);
33//
34// // A function that processes each basic block before its nodes are walked.
35// BlockProcessResult PreProcessBasicBlock(BasicBlock* block);
36//
37// // Process methods for each Node type. The GraphProcessor switches over
38// // the Node's opcode, casts it to the appropriate FooNode, and dispatches
39// // to NodeProcessor::Process. It's then up to the NodeProcessor to provide
40// // either distinct Process methods per Node type, or using templates or
41// // overloading as appropriate to group node processing.
42// void Process(FooNode* node, const ProcessingState& state) {}
43//
44template <typename NodeProcessor, bool visit_identity_nodes = false>
45class GraphProcessor;
46
47enum class BlockProcessResult {
48 kContinue, // Process exited normally.
49 kSkip, // Skip processing this block (no MultiProcessor support).
50};
51
52enum class ProcessResult {
53 kContinue, // Process exited normally, and the following processors will be
54 // called on the node.
55 kRemove, // Remove the current node from the graph (and do not call the
56 // following processors).
57 kHoist, // Hoist the current instruction to the parent basic block
58 // and reset the current instruction to the beginning of the
59 // block. Parent block must be dominating.
60 kAbort, // Stop processing now, do not process subsequent nodes/blocks.
61 // Should not be used when processing Constants.
62 kSkipBlock, // Stop processing this block and skip the remaining nodes (no
63 // MultiProcessor support).
64};

66class ProcessingState {
67 public:
68 explicit ProcessingState(BlockConstIterator block_it,
69 NodeIterator* node_it = nullptr)
70 : block_it_(block_it), node_it_(node_it) {}
71
72 // Disallow copies, since the underlying frame states stay mutable.
73 ProcessingState(const ProcessingState&) = delete;
74 ProcessingState& operator=(const ProcessingState&) = delete;
75
76 BasicBlock* block() const { return *block_it_; }
77 BasicBlock* next_block() const { return *(block_it_ + 1); }
78
79 NodeIterator* node_it() const {
80 DCHECK_NOT_NULL(node_it_);
81 return node_it_;
82 }
83
84 private:
85 BlockConstIterator block_it_;
86 NodeIterator* node_it_;
87};

89template <typename NodeProcessor, bool visit_identity_nodes>
90class GraphProcessor {
91 public:
92 template <typename... Args>
93 explicit GraphProcessor(Args&&... args)
94 : node_processor_(std::forward<Args>(args)...) {}
95
96 void ProcessGraph(Graph* graph) {
97 graph_ = graph;
98
99 node_processor_.PreProcessGraph(graph);
100
101 auto process_constants = [&](auto& map) {
102 for (auto it = map.begin(); it != map.end();) {
103 ProcessResult result =
104 node_processor_.Process(it->second, GetCurrentState());
105 switch (result) {
106 [[likely]] case ProcessResult::kContinue:
107 ++it;
108 break;
109 case ProcessResult::kRemove:
110 it = map.erase(it);
111 break;
112 case ProcessResult::kHoist:
113 case ProcessResult::kAbort:
114 case ProcessResult::kSkipBlock:
115 UNREACHABLE();
116 }
117 }
118 };
119 process_constants(graph->constants());
120 process_constants(graph->root());
121 process_constants(graph->smi());
122 process_constants(graph->tagged_index());
123 process_constants(graph->int32());
124 process_constants(graph->uint32());
125 process_constants(graph->float64());
126 process_constants(graph->external_references());
127 process_constants(graph->trusted_constants());
128
129 for (block_it_ = graph->begin(); block_it_ != graph->end(); ++block_it_) {
130 BasicBlock* block = *block_it_;
131
132 BlockProcessResult preprocess_result =
133 node_processor_.PreProcessBasicBlock(block);
134 switch (preprocess_result) {
135 [[likely]] case BlockProcessResult::kContinue:
136 break;
137 case BlockProcessResult::kSkip:
138 continue;
139 }
140
141 if (block->has_phi()) {
142 auto& phis = *block->phis();
143 for (auto it = phis.begin(); it != phis.end();) {
144 Phi* phi = *it;
145 ProcessResult result =
146 node_processor_.Process(phi, GetCurrentState());
147 switch (result) {
148 [[likely]] case ProcessResult::kContinue:
149 ++it;
150 break;
151 case ProcessResult::kRemove:
152 it = phis.RemoveAt(it);
153 break;
154 case ProcessResult::kAbort:
155 return;
156 case ProcessResult::kSkipBlock:
157 goto skip_block;
158 case ProcessResult::kHoist:
159 UNREACHABLE();
160 }
161 }
162 }
163
164 node_processor_.PostPhiProcessing();
165
166 for (node_it_ = block->nodes().begin();
167 node_it_ != block->nodes().end();) {
168 Node* node = *node_it_;
169 ProcessResult result = ProcessNodeBase(node, GetCurrentState());
170 switch (result) {
171 [[likely]] case ProcessResult::kContinue:
172 ++node_it_;
173 break;
174 case ProcessResult::kRemove:
175 node_it_ = block->nodes().RemoveAt(node_it_);
176 break;
177 case ProcessResult::kHoist: {
178 DCHECK(block->predecessor_count() == 1 ||
179 (block->predecessor_count() == 2 && block->is_loop()));
180 BasicBlock* target = block->predecessor_at(0);
181 DCHECK(target->successors().size() == 1);
182 Node* cur = *node_it_;
183 cur->set_owner(target);
184 block->nodes().RemoveAt(node_it_);
185 target->nodes().Add(cur);
186 node_it_ = block->nodes().begin();
187 break;
188 }
189 case ProcessResult::kAbort:
190 return;
191 case ProcessResult::kSkipBlock:
192 goto skip_block;
193 }
194 }
195
196 {
197 ProcessResult control_result =
198 ProcessNodeBase(block->control_node(), GetCurrentState());
199 switch (control_result) {
200 [[likely]] case ProcessResult::kContinue:
201 case ProcessResult::kSkipBlock:
202 break;
203 case ProcessResult::kAbort:
204 return;
205 case ProcessResult::kRemove:
206 case ProcessResult::kHoist:
207 UNREACHABLE();
208 }
209 }
210 skip_block:
211 continue;
212 }
213
214 node_processor_.PostProcessGraph(graph);
215 }
216
217 NodeProcessor& node_processor() { return node_processor_; }
218 const NodeProcessor& node_processor() const { return node_processor_; }
219
220 private:
221 ProcessingState GetCurrentState() {
222 return ProcessingState(block_it_, &node_it_);
223 }
224
225 ProcessResult ProcessNodeBase(NodeBase* node, const ProcessingState& state) {
226 switch (node->opcode()) {
227#define CASE(OPCODE) \
228 case Opcode::k##OPCODE: \
229 if constexpr (!visit_identity_nodes && \
230 Opcode::k##OPCODE == Opcode::kIdentity) { \
231 return ProcessResult::kContinue; \
232 } \
233 PreProcess(node->Cast<OPCODE>(), state); \
234 return node_processor_.Process(node->Cast<OPCODE>(), state);
235
236 NODE_BASE_LIST(CASE)
237#undef CASE
238 }
239 }
240
241 void PreProcess(NodeBase* node, const ProcessingState& state) {}
242
243 NodeProcessor node_processor_;
244 Graph* graph_;
245 BlockConstIterator block_it_;
246 NodeIterator node_it_;
247};

249// A NodeProcessor that wraps multiple NodeProcessors, and forwards to each of
250// them iteratively.
251template <typename... Processors>
252class NodeMultiProcessor;

254template <>
255class NodeMultiProcessor<> {
256 public:
257 void PreProcessGraph(Graph* graph) {}
258 void PostProcessGraph(Graph* graph) {}
259 BlockProcessResult PreProcessBasicBlock(BasicBlock* block) {
260 return BlockProcessResult::kContinue;

269template <typename Processor, typename... Processors>
270class NodeMultiProcessor<Processor, Processors...>
271 : NodeMultiProcessor<Processors...> {
272 using Base = NodeMultiProcessor<Processors...>;
273
274 public:
275 template <typename... Args>
276 explicit NodeMultiProcessor(Processor&& processor, Args&&... processors)
277 : Base(std::forward<Args>(processors)...),
278 processor_(std::forward<Processor>(processor)) {}
279 template <typename... Args>
280 explicit NodeMultiProcessor(Args&&... processors)
281 : Base(std::forward<Args>(processors)...) {}
282
283 template <typename Node>
284 ProcessResult Process(Node* node, const ProcessingState& state) {
285 auto res = processor_.Process(node, state);
286 switch (res) {
287 [[likely]] case ProcessResult::kContinue:
288 return Base::Process(node, state);
289 case ProcessResult::kAbort:
290 case ProcessResult::kRemove:
291 return res;
292 case ProcessResult::kHoist:
293 case ProcessResult::kSkipBlock:
294 // TODO(olivf): How to combine these with multiple processors depends on
295 // the needs of the actual processors. Implement once needed.
296 UNREACHABLE();
297 }
298 }
299 void PreProcessGraph(Graph* graph) {
300 processor_.PreProcessGraph(graph);
301 Base::PreProcessGraph(graph);
302 }
303 void PostProcessGraph(Graph* graph) {
304 // Post process in reverse order because that kind of makes sense.
305 Base::PostProcessGraph(graph);
306 processor_.PostProcessGraph(graph);
307 }
308 BlockProcessResult PreProcessBasicBlock(BasicBlock* block) {
309 BlockProcessResult res = processor_.PreProcessBasicBlock(block);
310 switch (res) {
311 [[likely]] case BlockProcessResult::kContinue:
312 return Base::PreProcessBasicBlock(block);
313 case BlockProcessResult::kSkip:
314 // TODO(olivf): How to combine this with multiple processors depends on
315 // the needs of the actual processors. Implement once needed.
316 UNREACHABLE();
317 }
318 }
319 void PostPhiProcessing() {
320 processor_.PostPhiProcessing();
321 Base::PostPhiProcessing();
322 }
323
324 private:
325 Processor processor_;
326};
