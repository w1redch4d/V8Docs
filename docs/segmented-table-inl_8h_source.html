59template <typename Entry, size_t size>
60void SegmentedTable<Entry, size>::Initialize() {
61 DCHECK(!is_initialized());
62 DCHECK_EQ(vas_, nullptr);
63
64 VirtualAddressSpace* root_space = GetPlatformVirtualAddressSpace();
65
66#ifdef V8_TARGET_ARCH_64_BIT
67 static_assert(kUseContiguousMemory);
68 DCHECK(IsAligned(kReservationSize, root_space->allocation_granularity()));
69
70 if (root_space->CanAllocateSubspaces()) {
71 auto subspace = root_space->AllocateSubspace(VirtualAddressSpace::kNoHint,
72 kReservationSize, kSegmentSize,
73 PagePermissions::kReadWrite);
74 vas_ = subspace.release();
75 } else {
76 // This may be required on old Windows versions that don't support
77 // VirtualAlloc2, which is required for subspaces. In that case, just use a
78 // fully-backed emulated subspace.
79 Address reservation_base = root_space->AllocatePages(
80 VirtualAddressSpace::kNoHint, kReservationSize, kSegmentSize,
81 PagePermissions::kNoAccess);
82 if (reservation_base) {
83 vas_ = new base::EmulatedVirtualAddressSubspace(
84 root_space, reservation_base, kReservationSize, kReservationSize);
85 }
86 }
87 if (!vas_) {
88 V8::FatalProcessOutOfMemory(
89 nullptr, "SegmentedTable::InitializeTable (subspace allocation)");
90 }
91#else // V8_TARGET_ARCH_64_BIT
92 static_assert(!kUseContiguousMemory);
93 vas_ = root_space;
94#endif
95
96 base_ = reinterpret_cast<Entry*>(vas_->base());
97
98 if constexpr (kUseContiguousMemory && kIsWriteProtected) {
99 CHECK(ThreadIsolation::WriteProtectMemory(
100 base(), size, PageAllocator::Permission::kNoAccess));
101 }
102}

115template <typename Entry, size_t size>
116typename SegmentedTable<Entry, size>::FreelistHead
117SegmentedTable<Entry, size>::InitializeFreeList(Segment segment,
118 uint32_t start_offset) {
119 DCHECK_LT(start_offset, kEntriesPerSegment);
120 uint32_t num_entries = kEntriesPerSegment - start_offset;
121
122 uint32_t first = segment.first_entry() + start_offset;
123 uint32_t last = segment.last_entry();
124 {
125 WriteIterator it = iter_at(first);
126 while (it.index() != last) {
127 it->MakeFreelistEntry(it.index() + 1);
128 ++it;
129 }
130 it->MakeFreelistEntry(0);
131 }
132
133 return FreelistHead(first, num_entries);
134}

136template <typename Entry, size_t size>
137std::pair<typename SegmentedTable<Entry, size>::Segment,
138 typename SegmentedTable<Entry, size>::FreelistHead>
139SegmentedTable<Entry, size>::AllocateAndInitializeSegment() {
140 Address start =
141 vas_->AllocatePages(VirtualAddressSpace::kNoHint, kSegmentSize,
142 kSegmentSize, PagePermissions::kReadWrite);
143 if (!start) {
144 V8::FatalProcessOutOfMemory(nullptr, "SegmentedTable::AllocateSegment");
145 }
146 uint32_t offset = static_cast<uint32_t>((start - vas_->base()));
147 Segment segment = Segment::At(offset);
148
149 FreelistHead freelist = InitializeFreeList(segment);
150
151 return {segment, freelist};
152}
