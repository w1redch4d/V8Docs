64// Creates a Map for the given debug proxy |id| using the |create_template_fn|
65// on-demand and caches this map in the global object. The map is derived from
66// the FunctionTemplate returned by |create_template_fn| and has its prototype
67// set to |null| and is marked non-extensible (by default).
68// TODO(bmeurer): remove the extensibility opt-out and replace it with a proper
69// way to add non-intercepted named properties.
70Handle<Map> GetOrCreateDebugProxyMap(
71 Isolate* isolate, DebugProxyId id,
72 v8::Local<v8::FunctionTemplate> (*create_template_fn)(v8::Isolate*),
73 bool make_non_extensible = true) {
74 auto maps = GetOrCreateDebugMaps(isolate);
75 CHECK_LE(kNumProxies, maps->length());
76 if (!maps->is_the_hole(isolate, id)) {
77 return handle(Cast<Map>(maps->get(id)), isolate);
78 }
79 auto tmp = (*create_template_fn)(reinterpret_cast<v8::Isolate*>(isolate));
80 auto fun =
81 ApiNatives::InstantiateFunction(isolate, Utils::OpenDirectHandle(*tmp))
82 .ToHandleChecked();
83 auto map = JSFunction::GetDerivedMap(isolate, fun, fun).ToHandleChecked();
84 Map::SetPrototype(isolate, map, isolate->factory()->null_value());
85 if (make_non_extensible) {
86 map->set_is_extensible(false);
87 }
88 maps->set(id, *map);
89 return map;
90}

92// Base class for debug proxies, offers indexed access. The subclasses
93// need to implement |Count| and |Get| methods appropriately.
94template <typename T, DebugProxyId id, typename Provider>
95struct IndexedDebugProxy {
96 static constexpr DebugProxyId kId = id;

98 static Handle<JSObject> Create(Isolate* isolate, Handle<Provider> provider,
99 bool make_map_non_extensible = true) {
100 auto object_map = GetOrCreateDebugProxyMap(isolate, kId, &T::CreateTemplate,
101 make_map_non_extensible);
102 auto object = isolate->factory()->NewFastOrSlowJSObjectFromMap(
103 object_map, 0, AllocationType::kYoung,
104 DirectHandle<AllocationSite>::null(), NewJSObjectType::kAPIWrapper);
105 object->SetEmbedderField(kProviderField, *provider);
106 return object;
107 }

114 static v8::Local<v8::FunctionTemplate> CreateTemplate(v8::Isolate* isolate) {
115 Local<v8::FunctionTemplate> templ = v8::FunctionTemplate::New(isolate);
116 templ->SetClassName(
117 v8::String::NewFromUtf8(isolate, T::kClassName).ToLocalChecked());
118 templ->InstanceTemplate()->SetInternalFieldCount(T::kFieldCount);

148 static v8::Intercepted IndexedGetter(
149 uint32_t index, const PropertyCallbackInfo<v8::Value>& info) {
150 auto isolate = GetIsolate(info);
151 auto provider = GetProvider(info);
152 if (index < T::Count(isolate, provider)) {
153 auto value = T::Get(isolate, provider, index);
154 info.GetReturnValue().Set(Utils::ToLocal(value));
155 return v8::Intercepted::kYes;
156 }
157 return v8::Intercepted::kNo;
158 }
159
160 static v8::Intercepted IndexedDescriptor(
161 uint32_t index, const PropertyCallbackInfo<v8::Value>& info) {
162 auto isolate = GetIsolate(info);
163 auto provider = GetProvider(info);
164 if (index < T::Count(isolate, provider)) {
165 PropertyDescriptor descriptor;
166 descriptor.set_configurable(false);
167 descriptor.set_enumerable(true);
168 descriptor.set_writable(false);
169 descriptor.set_value(Cast<JSAny>(T::Get(isolate, provider, index)));
170 info.GetReturnValue().Set(Utils::ToLocal(descriptor.ToObject(isolate)));
171 return v8::Intercepted::kYes;
172 }
173 return v8::Intercepted::kNo;
174 }
175
176 static v8::Intercepted IndexedQuery(
177 uint32_t index, const PropertyCallbackInfo<v8::Integer>& info) {
178 if (index < T::Count(GetIsolate(info), GetProvider(info))) {
179 info.GetReturnValue().Set(Integer::New(
180 info.GetIsolate(),
181 PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly));
182 return v8::Intercepted::kYes;
183 }
184 return v8::Intercepted::kNo;
185 }
186
187 static void IndexedEnumerator(const PropertyCallbackInfo<v8::Array>& info) {
188 auto isolate = GetIsolate(info);
189 auto count = T::Count(isolate, GetProvider(info));
190 auto indices = isolate->factory()->NewFixedArray(count);
191 for (uint32_t index = 0; index < count; ++index) {
192 indices->set(index, Smi::FromInt(index));
193 }
194 info.GetReturnValue().Set(
195 Utils::ToLocal(isolate->factory()->NewJSArrayWithElements(
196 indices, PACKED_SMI_ELEMENTS)));
197 }

200// Extends |IndexedDebugProxy| with named access, where the names are computed
201// on-demand, and all names are assumed to start with a dollar char ($). This
202// is important in order to scale to Wasm modules with hundreds of thousands
203// of functions in them.
204template <typename T, DebugProxyId id, typename Provider = WasmInstanceObject>
205struct NamedDebugProxy : IndexedDebugProxy<T, id, Provider> {
206 static v8::Local<v8::FunctionTemplate> CreateTemplate(v8::Isolate* isolate) {
207 auto templ = IndexedDebugProxy<T, id, Provider>::CreateTemplate(isolate);
208 templ->InstanceTemplate()->SetHandler(v8::NamedPropertyHandlerConfiguration(
209 &T::NamedGetter, {}, &T::NamedQuery, {}, &T::NamedEnumerator, {},
210 &T::NamedDescriptor, {}, v8::PropertyHandlerFlags::kHasNoSideEffect));
211 return templ;
212 }
213
214 static void IndexedEnumerator(const PropertyCallbackInfo<v8::Array>& info) {
215 info.GetReturnValue().Set(v8::Array::New(info.GetIsolate()));
216 }

218 static Handle<NameDictionary> GetNameTable(Handle<JSObject> holder,
219 Isolate* isolate) {
220 Handle<Symbol> symbol = isolate->factory()->wasm_debug_proxy_names_symbol();
221 Handle<Object> table_or_undefined =
222 JSObject::GetProperty(isolate, holder, symbol).ToHandleChecked();
223 if (!IsUndefined(*table_or_undefined, isolate)) {
224 return Cast<NameDictionary>(table_or_undefined);
225 }
226 auto provider = T::GetProvider(holder, isolate);
227 auto count = T::Count(isolate, provider);
228 auto table = NameDictionary::New(isolate, count);
229 for (uint32_t index = 0; index < count; ++index) {
230 HandleScope scope(isolate);
231 auto key = T::GetName(isolate, provider, index);
232 if (table->FindEntry(isolate, key).is_found()) continue;
233 Handle<Smi> value(Smi::FromInt(index), isolate);
234 table = NameDictionary::Add(isolate, table, key, value,
235 PropertyDetails::Empty());
236 }
237 Object::SetProperty(isolate, holder, symbol, table).Check();
238 return table;
239 }

241 template <typename V>
242 static std::optional<uint32_t> FindName(Local<v8::Name> name,
243 const PropertyCallbackInfo<V>& info) {
244 if (!name->IsString()) return {};
245 auto name_str = Utils::OpenHandle(*name.As<v8::String>());
246 if (name_str->length() == 0 || name_str->Get(0) != '$') return {};
247 auto isolate = T::GetIsolate(info);
248 auto table = GetNameTable(T::GetHolder(info), isolate);
249 auto entry = table->FindEntry(isolate, name_str);
250 if (entry.is_found()) return Smi::ToInt(table->ValueAt(entry));
251 return {};
252 }
253
254 static v8::Intercepted NamedGetter(
255 Local<v8::Name> name, const PropertyCallbackInfo<v8::Value>& info) {
256 if (auto index = FindName(name, info)) {
257 return T::IndexedGetter(*index, info);
258 }
259 return v8::Intercepted::kNo;
260 }
261
262 static v8::Intercepted NamedQuery(
263 Local<v8::Name> name, const PropertyCallbackInfo<v8::Integer>& info) {
264 if (auto index = FindName(name, info)) {
265 return T::IndexedQuery(*index, info);
266 }
267 return v8::Intercepted::kNo;
268 }
269
270 static v8::Intercepted NamedDescriptor(
271 Local<v8::Name> name, const PropertyCallbackInfo<v8::Value>& info) {
272 if (auto index = FindName(name, info)) {
273 return T::IndexedDescriptor(*index, info);
274 }
275 return v8::Intercepted::kNo;
276 }
277
278 static void NamedEnumerator(const PropertyCallbackInfo<v8::Array>& info) {
279 auto isolate = T::GetIsolate(info);
280 auto table = GetNameTable(T::GetHolder(info), isolate);
281 auto names = NameDictionary::IterationIndices(isolate, table);
282 for (int i = 0; i < names->length(); ++i) {
283 InternalIndex entry(Smi::ToInt(names->get(i)));
284 names->set(i, table->NameAt(entry));
285 }
286 info.GetReturnValue().Set(Utils::ToLocal(
287 isolate->factory()->NewJSArrayWithElements(names, PACKED_ELEMENTS)));
288 }
289};

522// This class implements the debug proxy for a given Wasm frame. The debug
523// proxy is used when evaluating JavaScript expressions on a wasm frame via
524// the inspector |Runtime.evaluateOnCallFrame()| API and enables developers
525// and extensions to inspect the WebAssembly engine state from JavaScript.
526// The proxy provides the following interface:
527//
528// type WasmValue = {
529// type: string;
530// value: number | bigint | object | string;
531// };
532// type WasmFunction = (... args : WasmValue[]) = > WasmValue;
533// interface WasmInterface {
534// $globalX: WasmValue;
535// $varX: WasmValue;
536// $funcX(a : WasmValue /*, ...*/) : WasmValue;
537// readonly $memoryX : WebAssembly.Memory;
538// readonly $tableX : WebAssembly.Table;
539//
540// readonly instance : WebAssembly.Instance;
541// readonly module : WebAssembly.Module;
542//
543// readonly memories : {[nameOrIndex:string | number] : WebAssembly.Memory};
544// readonly tables : {[nameOrIndex:string | number] : WebAssembly.Table};
545// readonly stack : WasmValue[];
546// readonly globals : {[nameOrIndex:string | number] : WasmValue};
547// readonly locals : {[nameOrIndex:string | number] : WasmValue};
548// readonly functions : {[nameOrIndex:string | number] : WasmFunction};
549// }
550//
551// The wasm index spaces memories, tables, stack, globals, locals, and
552// functions are JSObjects with interceptors that lazily produce values
553// either by index or by name (except for stack).
554// Only the names are reported by APIs such as Object.keys() and
555// Object.getOwnPropertyNames(), since the indices are not meant to be
556// used interactively by developers (in Chrome DevTools), but are provided
557// for WebAssembly language extensions. Also note that these JSObjects
558// all have null prototypes, to not confuse context lookup and to make
559// their purpose as dictionaries clear.
560//
561// See http://doc/1VZOJrU2VsqOZe3IUzbwQWQQSZwgGySsm5119Ust1gUA and
562// http://bit.ly/devtools-wasm-entities for more details.
563class ContextProxyPrototype {
564 public:
565 static Handle<JSObject> Create(Isolate* isolate) {
566 auto object_map =
567 GetOrCreateDebugProxyMap(isolate, kContextProxy, &CreateTemplate);
568 return isolate->factory()->NewJSObjectFromMap(
569 object_map, AllocationType::kYoung,
570 DirectHandle<AllocationSite>::null(), NewJSObjectType::kAPIWrapper);
571 }
572
573 private:
574 static v8::Local<v8::FunctionTemplate> CreateTemplate(v8::Isolate* isolate) {
575 Local<v8::FunctionTemplate> templ = v8::FunctionTemplate::New(isolate);
576 templ->InstanceTemplate()->SetHandler(v8::NamedPropertyHandlerConfiguration(
577 &NamedGetter, {}, {}, {}, {}, {}, {}, {},
578 static_cast<v8::PropertyHandlerFlags>(
579 static_cast<unsigned>(
580 v8::PropertyHandlerFlags::kOnlyInterceptStrings) |
581 static_cast<unsigned>(
582 v8::PropertyHandlerFlags::kHasNoSideEffect))));
583 return templ;
584 }

586 static MaybeHandle<Object> GetNamedProperty(Isolate* isolate,
587 DirectHandle<JSObject> receiver,
588 DirectHandle<String> name) {
589 if (name->length() != 0 && name->Get(0) == '$') {
590 const char* kDelegateNames[] = {"memories", "locals", "tables",
591 "functions", "globals"};
592 for (auto delegate_name : kDelegateNames) {
593 DirectHandle<JSAny> delegate;
594 ASSIGN_RETURN_ON_EXCEPTION(isolate, delegate,
595 Cast<JSAny>(JSObject::GetProperty(
596 isolate, receiver, delegate_name)));
597 if (!IsUndefined(*delegate, isolate)) {
598 Handle<Object> value;
599 ASSIGN_RETURN_ON_EXCEPTION(
600 isolate, value, Object::GetProperty(isolate, delegate, name));
601 if (!IsUndefined(*value, isolate)) return value;
602 }
603 }
604 }
605 return {};
606 }

608 static v8::Intercepted NamedGetter(
609 Local<v8::Name> name, const PropertyCallbackInfo<v8::Value>& info) {
610 auto name_string = Cast<String>(Utils::OpenHandle(*name));
611 auto isolate = reinterpret_cast<Isolate*>(info.GetIsolate());
612 auto receiver = Cast<JSObject>(Utils::OpenHandle(*info.This()));
613 DirectHandle<Object> value;
614 if (GetNamedProperty(isolate, receiver, name_string).ToHandle(&value)) {
615 info.GetReturnValue().Set(Utils::ToLocal(value));
616 return v8::Intercepted::kYes;
617 }
618 return v8::Intercepted::kNo;
619 }

1012// static
1013Handle<WasmValueObject> WasmValueObject::New(
1014 Isolate* isolate, const wasm::WasmValue& value,
1015 DirectHandle<WasmModuleObject> module_object) {
1016 DirectHandle<String> t;
1017 DirectHandle<Object> v;
1018 switch (value.type().kind()) {
1019 case wasm::kI8: {
1020 // This can't be reached for most "top-level" things, only via nested
1021 // calls for struct/array fields.
1022 t = isolate->factory()->InternalizeString(base::StaticCharVector("i8"));
1023 v = isolate->factory()->NewNumber(value.to_i8_unchecked());
1024 break;
1025 }
1026 case wasm::kI16: {
1027 // This can't be reached for most "top-level" things, only via nested
1028 // calls for struct/array fields.
1029 t = isolate->factory()->InternalizeString(base::StaticCharVector("i16"));
1030 v = isolate->factory()->NewNumber(value.to_i16_unchecked());
1031 break;
1032 }
1033 case wasm::kI32: {
1034 t = isolate->factory()->InternalizeString(base::StaticCharVector("i32"));
1035 v = isolate->factory()->NewNumberFromInt(value.to_i32_unchecked());
1036 break;
1037 }
1038 case wasm::kI64: {
1039 t = isolate->factory()->InternalizeString(base::StaticCharVector("i64"));
1040 v = BigInt::FromInt64(isolate, value.to_i64_unchecked());
1041 break;
1042 }
1043 case wasm::kF16: {
1044 // This can't be reached for most "top-level" things, only via nested
1045 // calls for struct/array fields.
1046 t = isolate->factory()->InternalizeString(base::StaticCharVector("f16"));
1047 v = isolate->factory()->NewNumber(value.to_f16_unchecked());
1048 break;
1049 }
1050 case wasm::kF32: {
1051 t = isolate->factory()->InternalizeString(base::StaticCharVector("f32"));
1052 v = isolate->factory()->NewNumber(value.to_f32_unchecked());
1053 break;
1054 }
1055 case wasm::kF64: {
1056 t = isolate->factory()->InternalizeString(base::StaticCharVector("f64"));
1057 v = isolate->factory()->NewNumber(value.to_f64_unchecked());
1058 break;
1059 }
1060 case wasm::kS128: {
1061 t = isolate->factory()->InternalizeString(base::StaticCharVector("v128"));
1062 v = WasmSimd128ToString(isolate, value.to_s128_unchecked());
1063 break;
1064 }
1065 case wasm::kRefNull:
1066 case wasm::kRef: {
1067 DirectHandle<Object> ref = value.to_ref();
1068 if (value.type().heap_type().representation() == wasm::HeapType::kExn) {
1069 t = isolate->factory()->InternalizeString(
1070 base::StaticCharVector("exnref"));
1071 v = ref;
1072 } else if (IsWasmStruct(*ref)) {
1073 Tagged<WasmTypeInfo> type_info =
1074 Cast<HeapObject>(*ref)->map()->wasm_type_info();
1075 wasm::ValueType type = wasm::ValueType::Ref(type_info->type_index());
1076 // Getting the trusted data is safe; structs always have the instance
1077 // data defined.
1078 DirectHandle<WasmTrustedInstanceData> wtid(
1079 type_info->trusted_data(isolate), isolate);
1080 t = GetRefTypeName(isolate, type, wtid->native_module());
1081 v = StructProxy::Create(isolate, Cast<WasmStruct>(ref),
1082 direct_handle(wtid->module_object(), isolate));
1083 } else if (IsWasmArray(*ref)) {
1084 Tagged<WasmTypeInfo> type_info =
1085 Cast<HeapObject>(*ref)->map()->wasm_type_info();
1086 wasm::ValueType type = wasm::ValueType::Ref(type_info->type_index());
1087 // Getting the trusted data is safe; arrays always have the instance
1088 // data defined.
1089 DirectHandle<WasmTrustedInstanceData> wtid(
1090 type_info->trusted_data(isolate), isolate);
1091 t = GetRefTypeName(isolate, type, wtid->native_module());
1092 v = ArrayProxy::Create(isolate, Cast<WasmArray>(ref),
1093 direct_handle(wtid->module_object(), isolate));
1094 } else if (IsWasmFuncRef(*ref)) {
1095 DirectHandle<WasmInternalFunction> internal_fct{
1096 Cast<WasmFuncRef>(*ref)->internal(isolate), isolate};
1097 v = WasmInternalFunction::GetOrCreateExternal(internal_fct);
1098 // If the module is not provided by the caller, retrieve it from the
1099 // instance object. If the function was created in JavaScript using
1100 // `new WebAssembly.Function(...)`, a module for name resolution is not
1101 // available.
1102 if (module_object.is_null() &&
1103 IsWasmTrustedInstanceData(internal_fct->implicit_arg())) {
1104 module_object = direct_handle(
1105 Cast<WasmTrustedInstanceData>(internal_fct->implicit_arg())
1106 ->module_object(),
1107 isolate);
1108 }
1109 t = GetRefTypeName(isolate, value.type(), module_object);
1110 } else if (IsWasmNull(*ref)) {
1111 // TODO(manoskouk): Is this value correct?
1112 v = isolate->factory()->null_value();
1113 t = GetRefTypeName(isolate, value.type(), module_object);
1114 } else if (IsJSFunction(*ref) || IsSmi(*ref) || IsNull(*ref) ||
1115 IsString(*ref) ||
1116 value.type().is_reference_to(wasm::HeapType::kExtern) ||
1117 value.type().is_reference_to(wasm::HeapType::kAny)) {
1118 t = GetRefTypeName(isolate, value.type(), module_object);
1119 v = ref;
1120 } else {
1121 // Fail gracefully.
1122 base::EmbeddedVector<char, 64> error;
1123 int len = SNPrintF(error, "unimplemented object type: %d",
1124 Cast<HeapObject>(*ref)->map()->instance_type());
1125 t = GetRefTypeName(isolate, value.type(), module_object);
1126 v = isolate->factory()->InternalizeString(error.SubVector(0, len));
1127 }
1128 break;
1129 }
1130 case wasm::kRtt:
1131 case wasm::kVoid:
1132 case wasm::kTop:
1133 case wasm::kBottom:
1134 UNREACHABLE();
1135 }
1136 return New(isolate, t, v);
1137}
