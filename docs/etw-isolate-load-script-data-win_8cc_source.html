61// static
62void IsolateLoadScriptData::UpdateAllIsolates(bool etw_enabled,
63 uint32_t options) {
64 ETWTRACEDBG << "UpdateAllIsolates with etw_enabled==" << etw_enabled
65 << " and options==" << options << " acquiring mutex" << std::endl;
66 base::MutexGuard guard(isolates_mutex.Pointer());
67 ETWTRACEDBG << "UpdateAllIsolates Isolate count=="
68 << isolate_map.Pointer()->size() << std::endl;
69 auto monitor = std::make_shared<EtwIsolateCaptureStateMonitor>(
70 isolates_mutex.Pointer(), isolate_map.Pointer()->size());
71 bool capture_state =
72 (options & kJitCodeEventEnumExisting) == kJitCodeEventEnumExisting;
73 std::weak_ptr<EtwIsolateCaptureStateMonitor> weak_monitor = monitor;
74 std::for_each(
75 isolate_map.Pointer()->begin(), isolate_map.Pointer()->end(),
76 [etw_enabled, weak_monitor, options](auto& pair) {
77 auto& isolate_data = pair.second;
78 if (etw_enabled) {
79 ETWTRACEDBG << "UpdateAllIsolates enqueing enablelog" << std::endl;
80 isolate_data.EnqueueEnableLog(weak_monitor, options);
81 } else {
82 ETWTRACEDBG << "UpdateAllIsolates enqueing disablelog" << std::endl;
83 isolate_data.EnqueueDisableLog();
84 }
85 });
86
87 if (!capture_state) {
88 return;
89 }
90
91 ETWTRACEDBG << "UpdateAllIsolates starting WaitFor" << std::endl;
92 bool timeout = !monitor->WaitFor(kCaptureStateTimeout);
93 ETWTRACEDBG << "UpdateAllIsolates WaitFor "
94 << (timeout ? "timeout" : "completed") << std::endl;
95}

109// static
110void IsolateLoadScriptData::EnableLog(
111 Isolate* isolate, size_t event_id,
112 std::weak_ptr<EtwIsolateCaptureStateMonitor> weak_monitor,
113 uint32_t options) {
114 {
115 ETWTRACEDBG << "EnableLog called with event_id==" << event_id
116 << " and options==" << options << " taking mutex" << std::endl;
117 base::MutexGuard guard(isolates_mutex.Pointer());
118 auto& data = GetData(isolate);
119 if (event_id > 0 && data.CurrentEventId() != event_id) {
120 // This interrupt was canceled by a newer interrupt.
121 return;
122 }
123
124 // Cause all SourceLoad events to be re-emitted.
125 if (options & kJitCodeEventEnumExisting) {
126 data.RemoveAllLoadedScripts();
127 }
128 }
129
130 ETWTRACEDBG << "Mutex released with event_id==" << event_id << std::endl;
131
132 // This cannot be done while isolate_mutex is locked, as it can call
133 // EventHandler while in the call for all the existing code.
134 EtwIsolateOperations::Instance()->SetEtwCodeEventHandler(isolate, options);
135 isolate->SetETWTracingEnabled(true);
136
137 // Notify waiting thread if a monitor was provided.
138 if (auto monitor = weak_monitor.lock()) {
139 ETWTRACEDBG << "monitor->Notify with event_id==" << event_id << std::endl;
140 monitor->Notify();
141 }
142}

144// static
145void IsolateLoadScriptData::EnableLogWithFilterDataOnAllIsolates(
146 const uint8_t* data, size_t size, uint32_t options) {
147 base::MutexGuard guard(isolates_mutex.Pointer());
148
149 std::string etw_filter_payload;
150 etw_filter_payload.assign(data, data + size);
151 auto monitor = std::make_shared<EtwIsolateCaptureStateMonitor>(
152 isolates_mutex.Pointer(), isolate_map.Pointer()->size());
153 bool capture_state =
154 (options & kJitCodeEventEnumExisting) == kJitCodeEventEnumExisting;
155 std::weak_ptr<EtwIsolateCaptureStateMonitor> weak_monitor = monitor;
156 std::for_each(isolate_map.Pointer()->begin(), isolate_map.Pointer()->end(),
157 [&etw_filter_payload, weak_monitor, options](auto& pair) {
158 auto& isolate_data = pair.second;
159 isolate_data.EnqueueEnableLogWithFilterData(
160 etw_filter_payload, weak_monitor, options);
161 });
162
163 if (!capture_state) {
164 return;
165 }
166
167 bool timeout = !monitor->WaitFor(kCaptureStateTimeout);
168 ETWTRACEDBG << "EnableLogWithFilterDataOnAllIsolates WaitFor "
169 << (timeout ? "timeout" : "completed") << std::endl;
170}

173void IsolateLoadScriptData::DisableLog(Isolate* isolate, size_t event_id) {
174 {
175 base::MutexGuard guard(isolates_mutex.Pointer());
176 auto& data = GetData(isolate);
177 if (event_id > 0 && data.CurrentEventId() != event_id) {

187// static
188void IsolateLoadScriptData::EnableLogWithFilterData(
189 Isolate* isolate, size_t event_id, const std::string& etw_filter_payload,
190 std::weak_ptr<EtwIsolateCaptureStateMonitor> weak_monitor,
191 uint32_t options) {
192 bool filter_did_match = false;
193 DCHECK(!etw_filter_payload.empty());
194
195 {
196 ETWTRACEDBG << "EnableLogWithFilterData called with event_id==" << event_id
197 << " and options==" << options << " taking mutex" << std::endl;
198 base::MutexGuard guard(isolates_mutex.Pointer());
199
200 auto& data = GetData(isolate);
201 if (event_id > 0 && data.CurrentEventId() != event_id) {
202 // This interrupt was canceled by a newer interrupt.
203 return;
204 }
205
206 FilterETWSessionByURLResult filter_etw_session_by_url_result =
207 EtwIsolateOperations::Instance()->RunFilterETWSessionByURLCallback(
208 isolate, etw_filter_payload);
209 filter_did_match = filter_etw_session_by_url_result.enable_etw_tracing;
210 if (filter_did_match) {
211 if (filter_etw_session_by_url_result.trace_interpreter_frames) {
212 isolate->set_etw_trace_interpreted_frames();
213 }
214
215 // Cause all SourceLoad events to be re-emitted.
216 if (options & kJitCodeEventEnumExisting) {
217 data.RemoveAllLoadedScripts();
218 }
219 }
220 }
221
222 if (filter_did_match) {
223 ETWTRACEDBG << "Filter was matched with event_id==" << event_id
224 << std::endl;
225 EtwIsolateOperations::Instance()->SetEtwCodeEventHandler(isolate, options);
226 isolate->SetETWTracingEnabled(true);
227 }
228
229 // Notify waiting thread if a monitor was provided.
230 if (auto monitor = weak_monitor.lock()) {
231 ETWTRACEDBG << "monitor->Notify with event_id==" << event_id << std::endl;
232 monitor->Notify();
233 }
234}

241void IsolateLoadScriptData::EnqueueEnableLog(
242 std::weak_ptr<EtwIsolateCaptureStateMonitor> weak_monitor,
243 uint32_t options) {
244 size_t event_id = event_id_.fetch_add(1);
245 EtwIsolateOperations::Instance()->RequestInterrupt(
246 isolate_,
247 // Executed in the isolate thread.
248 [](v8::Isolate* v8_isolate, void* data) {
249 std::unique_ptr<EnableInterruptData> interrupt_data(
250 reinterpret_cast<EnableInterruptData*>(data));
251 size_t event_id = interrupt_data->event_id;
252 auto weak_monitor = interrupt_data->weak_monitor;
253 uint32_t options = interrupt_data->options;
254 EnableLog(reinterpret_cast<Isolate*>(v8_isolate), event_id,
255 weak_monitor, options);
256 },
257 new EnableInterruptData{event_id + 1, weak_monitor, options});
258}

260void IsolateLoadScriptData::EnqueueDisableLog() {
261 size_t event_id = event_id_.fetch_add(1);
262 EtwIsolateOperations::Instance()->RequestInterrupt(
263 isolate_,
264 // Executed in the isolate thread.
265 [](v8::Isolate* v8_isolate, void* data) {
266 DisableLog(reinterpret_cast<Isolate*>(v8_isolate),
267 reinterpret_cast<size_t>(data));
268 },
269 reinterpret_cast<void*>(event_id + 1));
270}
