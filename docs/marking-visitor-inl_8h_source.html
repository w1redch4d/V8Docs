45 Tagged<HeapObject> retainer, Tagged<HeapObject> object,
46 MarkingHelper::WorklistTarget target_worklist) {
47 DCHECK(heap_->Contains(object));
48 SynchronizePageAccess(object);

52 concrete_visitor()->marking_state(),
53 target_worklist, object);

57template <typename ConcreteVisitor>
58// method template arguments
59template <typename THeapObjectSlot>
60void MarkingVisitorBase<ConcreteVisitor>::ProcessStrongHeapObject(
61 Tagged<HeapObject> host, THeapObjectSlot slot,
62 Tagged<HeapObject> heap_object) {
63 SynchronizePageAccess(heap_object);
64 const auto target_worklist =
65 MarkingHelper::ShouldMarkObject(heap_, heap_object);
66 if (!target_worklist) {
67 return;
68 }
69 // TODO(chromium:1495151): Remove after diagnosing.
70 if (V8_UNLIKELY(!MemoryChunk::FromHeapObject(heap_object)->IsMarking() &&
71 IsFreeSpaceOrFiller(
72 heap_object, ObjectVisitorWithCageBases::cage_base()))) {
73 heap_->isolate()->PushStackTraceAndDie(
74 reinterpret_cast<void*>(host->map().ptr()),
75 reinterpret_cast<void*>(host->address()),
76 reinterpret_cast<void*>(slot.address()),
77 reinterpret_cast<void*>(MemoryChunkMetadata::FromHeapObject(heap_object)
78 ->owner()
79 ->identity()));
80 }
81 MarkObject(host, heap_object, target_worklist.value());
82 concrete_visitor()->RecordSlot(host, slot, heap_object);
83}

94// class template arguments
95template <typename ConcreteVisitor>
96// method template arguments
97template <typename THeapObjectSlot>
98void MarkingVisitorBase<ConcreteVisitor>::ProcessWeakHeapObject(
99 Tagged<HeapObject> host, THeapObjectSlot slot,
100 Tagged<HeapObject> heap_object) {
101 SynchronizePageAccess(heap_object);
102 concrete_visitor()->AddWeakReferenceForReferenceSummarizer(host, heap_object);
103 const auto target_worklist =
104 MarkingHelper::ShouldMarkObject(heap_, heap_object);
105 if (!target_worklist) {
106 return;
107 }
108 if (concrete_visitor()->marking_state()->IsMarked(heap_object)) {
109 // Weak references with live values are directly processed here to
110 // reduce the processing time of weak cells during the main GC
111 // pause.
112 concrete_visitor()->RecordSlot(host, slot, heap_object);
113 } else {
114 // If we do not know about liveness of the value, we have to process
115 // the reference when we know the liveness of the whole transitive
116 // closure.
117 // Distinguish trivial cases (non involving custom weakness) from
118 // non-trivial ones. The latter are maps in host objects of type Map,
119 // TransitionArray and DescriptorArray.
120 if constexpr (SlotHoldsTrustedPointerV<THeapObjectSlot>) {
121 local_weak_objects_->weak_references_trusted_local.Push(
122 TrustedObjectAndSlot{host, slot});
123 } else if (V8_LIKELY(IsTrivialWeakReferenceValue(host, heap_object))) {
124 local_weak_objects_->weak_references_trivial_local.Push(
125 HeapObjectAndSlot{host, slot});
126 } else {
127 local_weak_objects_->weak_references_non_trivial_local.Push(
128 HeapObjectAndSlot{host, slot});
129 }
130 }
131}

133// class template arguments
134template <typename ConcreteVisitor>
135// method template arguments
136template <typename TSlot>
137void MarkingVisitorBase<ConcreteVisitor>::VisitPointersImpl(
138 Tagged<HeapObject> host, TSlot start, TSlot end) {
139 using THeapObjectSlot = typename TSlot::THeapObjectSlot;
140 for (TSlot slot = start; slot < end; ++slot) {
141 typename TSlot::TObject object;
142 if constexpr (SlotHoldsTrustedPointerV<TSlot>) {
143 // The fast check doesn't support the trusted cage, so skip it.
144 object = slot.Relaxed_Load();
145 } else {
146 const std::optional<Tagged<Object>> optional_object =
147 this->GetObjectFilterReadOnlyAndSmiFast(slot);
148 if (!optional_object) {
149 continue;
150 }
151 object = *optional_object;
152 }
153 Tagged<HeapObject> heap_object;
154 if (object.GetHeapObjectIfStrong(&heap_object)) {
155 // If the reference changes concurrently from strong to weak, the write
156 // barrier will treat the weak reference as strong, so we won't miss the
157 // weak reference.
158 ProcessStrongHeapObject(host, THeapObjectSlot(slot), heap_object);
159 } else if (TSlot::kCanBeWeak && object.GetHeapObjectIfWeak(&heap_object)) {
160 ProcessWeakHeapObject(host, THeapObjectSlot(slot), heap_object);
161 }
162 }
163}

180template <typename ConcreteVisitor>
181void MarkingVisitorBase<ConcreteVisitor>::VisitEmbeddedPointer(
182 Tagged<InstructionStream> host, RelocInfo* rinfo) {
183 DCHECK(RelocInfo::IsEmbeddedObjectMode(rinfo->rmode()));
184 Tagged<HeapObject> object =
185 rinfo->target_object(ObjectVisitorWithCageBases::cage_base());
186 const auto target_worklist = MarkingHelper::ShouldMarkObject(heap_, object);
187 if (!target_worklist) {
188 return;
189 }
190
191 if (!concrete_visitor()->marking_state()->IsMarked(object)) {
192 Tagged<Code> code = UncheckedCast<Code>(host->raw_code(kAcquireLoad));
193 if (code->IsWeakObject(object)) {
194 local_weak_objects_->weak_objects_in_code_local.Push(
195 HeapObjectAndCode{object, code});
196 concrete_visitor()->AddWeakReferenceForReferenceSummarizer(host, object);
197 } else {
198 MarkObject(host, object, target_worklist.value());
199 }
200 }
201 concrete_visitor()->RecordRelocSlot(host, rinfo, object);
202}

219template <typename ConcreteVisitor>
220void MarkingVisitorBase<ConcreteVisitor>::VisitExternalPointer(
221 Tagged<HeapObject> host, ExternalPointerSlot slot) {
222#ifdef V8_COMPRESS_POINTERS
223 DCHECK(!slot.tag_range().IsEmpty());
224 if (slot.HasExternalPointerHandle()) {
225 ExternalPointerHandle handle = slot.Relaxed_LoadHandle();
226 ExternalPointerTable* table;
227 ExternalPointerTable::Space* space;
228 if (IsSharedExternalPointerType(slot.tag_range())) {
229 table = shared_external_pointer_table_;
230 space = shared_external_pointer_space_;
231 } else {
232 table = external_pointer_table_;
233 if (v8_flags.sticky_mark_bits) {
234 // Everything is considered old during major GC.
235 DCHECK(!HeapLayout::InYoungGeneration(host));
236 if (handle == kNullExternalPointerHandle) return;
237 // The object may either be in young or old EPT.
238 if (table->Contains(heap_->young_external_pointer_space(), handle)) {
239 space = heap_->young_external_pointer_space();
240 } else {
241 DCHECK(table->Contains(heap_->old_external_pointer_space(), handle));
242 space = heap_->old_external_pointer_space();
243 }
244 } else {
245 space = HeapLayout::InYoungGeneration(host)
246 ? heap_->young_external_pointer_space()
247 : heap_->old_external_pointer_space();
248 }
249 }
250 table->Mark(space, handle, slot.address());
251 }
252#endif // V8_COMPRESS_POINTERS
253}

274template <typename ConcreteVisitor>
275void MarkingVisitorBase<ConcreteVisitor>::VisitIndirectPointer(
276 Tagged<HeapObject> host, IndirectPointerSlot slot,
277 IndirectPointerMode mode) {
278#ifdef V8_ENABLE_SANDBOX
279 if (mode == IndirectPointerMode::kStrong) {
280 // Load the referenced object (if the slot is initialized) and mark it as
281 // alive if necessary. Indirect pointers never have to be added to a
282 // remembered set because the referenced object will update the pointer
283 // table entry when it is relocated.
284 Tagged<Object> value = slot.Relaxed_Load(heap_->isolate());
285 if (IsHeapObject(value)) {
286 Tagged<HeapObject> obj = Cast<HeapObject>(value);
287 SynchronizePageAccess(obj);
288 const auto target_worklist = MarkingHelper::ShouldMarkObject(heap_, obj);
289 if (!target_worklist) {
290 return;
291 }
292 MarkObject(host, obj, target_worklist.value());
293 }
294 }
295#else
296 UNREACHABLE();
297#endif
298}

330template <typename ConcreteVisitor>
331size_t MarkingVisitorBase<ConcreteVisitor>::VisitJSFunction(
332 Tagged<Map> map, Tagged<JSFunction> js_function,
333 MaybeObjectSize maybe_object_size) {
334 if (ShouldFlushBaselineCode(js_function)) {
335 DCHECK(IsBaselineCodeFlushingEnabled(code_flush_mode_));
336#ifndef V8_ENABLE_LEAPTIERING
337 local_weak_objects_->baseline_flushing_candidates_local.Push(js_function);
338#endif // !V8_ENABLE_LEAPTIERING
339 return Base::VisitJSFunction(map, js_function, maybe_object_size);
340 }
341
342 // We're not flushing the Code, so mark it as alive.
343#ifdef V8_ENABLE_LEAPTIERING
344 // Here we can see JSFunctions that aren't fully initialized (e.g. during
345 // deserialization) so we need to check for the null handle.
346 JSDispatchHandle handle(
347 js_function->Relaxed_ReadField<JSDispatchHandle::underlying_type>(
348 JSFunction::kDispatchHandleOffset));
349 if (handle != kNullJSDispatchHandle) {
350 Tagged<HeapObject> obj =
351 IsolateGroup::current()->js_dispatch_table()->GetCode(handle);
352 // TODO(saelo): maybe factor out common code with VisitIndirectPointer
353 // into a helper routine?
354 SynchronizePageAccess(obj);
355 const auto target_worklist = MarkingHelper::ShouldMarkObject(heap_, obj);
356 if (target_worklist) {
357 MarkObject(js_function, obj, target_worklist.value());
358 }
359 }
360#else
361
362#ifdef V8_ENABLE_SANDBOX
363 VisitIndirectPointer(js_function,
364 js_function->RawIndirectPointerField(
365 JSFunction::kCodeOffset, kCodeIndirectPointerTag),
366 IndirectPointerMode::kStrong);
367#else
368 VisitPointer(js_function, js_function->RawField(JSFunction::kCodeOffset));
369#endif // V8_ENABLE_SANDBOX
370
371#endif // V8_ENABLE_LEAPTIERING
372
373 // TODO(mythria): Consider updating the check for ShouldFlushBaselineCode to
374 // also include cases where there is old bytecode even when there is no
375 // baseline code and remove this check here.
376 if (IsByteCodeFlushingEnabled(code_flush_mode_) &&
377 js_function->NeedsResetDueToFlushedBytecode(heap_->isolate())) {
378 local_weak_objects_->flushed_js_functions_local.Push(js_function);
379 }
380
381 return Base::VisitJSFunction(map, js_function, maybe_object_size);
382}

384template <typename ConcreteVisitor>
385size_t MarkingVisitorBase<ConcreteVisitor>::VisitSharedFunctionInfo(
386 Tagged<Map> map, Tagged<SharedFunctionInfo> shared_info,
387 MaybeObjectSize maybe_object_size) {
388 const bool can_flush_bytecode = HasBytecodeArrayForFlushing(shared_info);
389
390 // We found a BytecodeArray that can be flushed. Increment the age of the SFI.
391 if (can_flush_bytecode && !should_keep_ages_unchanged_) {
392 MakeOlder(shared_info);
393 }
394
395 if (!can_flush_bytecode || !ShouldFlushCode(shared_info)) {
396 // If the SharedFunctionInfo doesn't have old bytecode visit the function
397 // data strongly.
398#ifdef V8_ENABLE_SANDBOX
399 VisitIndirectPointer(shared_info,
400 shared_info->RawIndirectPointerField(
401 SharedFunctionInfo::kTrustedFunctionDataOffset,
402 kUnknownIndirectPointerTag),
403 IndirectPointerMode::kStrong);
404#else
405 VisitPointer(
406 shared_info,
407 shared_info->RawField(SharedFunctionInfo::kTrustedFunctionDataOffset));
408#endif
409 VisitPointer(shared_info,
410 shared_info->RawField(
411 SharedFunctionInfo::kUntrustedFunctionDataOffset));
412 } else if (!IsByteCodeFlushingEnabled(code_flush_mode_)) {
413 // If bytecode flushing is disabled but baseline code flushing is enabled
414 // then we have to visit the bytecode but not the baseline code.
415 DCHECK(IsBaselineCodeFlushingEnabled(code_flush_mode_));
416 Tagged<Code> baseline_code = shared_info->baseline_code(kAcquireLoad);
417 // Visit the bytecode hanging off baseline code.
418 VisitProtectedPointer(
419 baseline_code, baseline_code->RawProtectedPointerField(
420 Code::kDeoptimizationDataOrInterpreterDataOffset));
421 local_weak_objects_->code_flushing_candidates_local.Push(shared_info);
422 } else {
423 // In other cases, record as a flushing candidate since we have old
424 // bytecode.
425 local_weak_objects_->code_flushing_candidates_local.Push(shared_info);
426 }
427 return Base::VisitSharedFunctionInfo(map, shared_info, maybe_object_size);
428}

430template <typename ConcreteVisitor>
431bool MarkingVisitorBase<ConcreteVisitor>::HasBytecodeArrayForFlushing(
432 Tagged<SharedFunctionInfo> sfi) const {
433 if (IsFlushingDisabled(code_flush_mode_)) return false;
434
435 // TODO(rmcilroy): Enable bytecode flushing for resumable functions.
436 if (IsResumableFunction(sfi->kind()) || !sfi->allows_lazy_compilation()) {
437 return false;
438 }
439
440 // Get a snapshot of the function data field, and if it is a bytecode array,
441 // check if it is old. Note, this is done this way since this function can be
442 // called by the concurrent marker.
443 Tagged<Object> data = sfi->GetTrustedData(heap_->isolate());
444 if (IsCode(data)) {
445 Tagged<Code> baseline_code = Cast<Code>(data);
446 DCHECK_EQ(baseline_code->kind(), CodeKind::BASELINE);
447 // If baseline code flushing isn't enabled and we have baseline data on SFI
448 // we cannot flush baseline / bytecode.
449 if (!IsBaselineCodeFlushingEnabled(code_flush_mode_)) return false;
450 data = baseline_code->bytecode_or_interpreter_data();
451 } else if (!IsByteCodeFlushingEnabled(code_flush_mode_)) {
452 // If bytecode flushing isn't enabled and there is no baseline code there is
453 // nothing to flush.
454 return false;
455 }
456
457 return IsBytecodeArray(data);
458}

466template <typename ConcreteVisitor>
467bool MarkingVisitorBase<ConcreteVisitor>::IsOld(
468 Tagged<SharedFunctionInfo> sfi) const {
469 if (v8_flags.flush_code_based_on_time) {
470 return sfi->age() >= v8_flags.bytecode_old_time;
471 } else if (v8_flags.flush_code_based_on_tab_visibility) {
472 return isolate_in_background_ ||
473 V8_UNLIKELY(sfi->age() == SharedFunctionInfo::kMaxAge);
474 } else {
475 return sfi->age() >= v8_flags.bytecode_old_age;
476 }
477}

479template <typename ConcreteVisitor>
480void MarkingVisitorBase<ConcreteVisitor>::MakeOlder(
481 Tagged<SharedFunctionInfo> sfi) const {
482 if (v8_flags.flush_code_based_on_time) {
483 if (code_flushing_increase_ == 0) {
484 return;
485 }
486
487 uint16_t current_age;
488 uint16_t updated_age;
489 do {
490 current_age = sfi->age();
491 // When the age is 0, it was reset by the function prologue in
492 // Ignition/Sparkplug. But that might have been some time after the last
493 // full GC. So in this case we don't increment the value like we normally
494 // would but just set the age to 1. All non-0 values can be incremented as
495 // expected (we add the number of seconds since the last GC) as they were
496 // definitely last executed before the last full GC.
497 updated_age = current_age == 0
498 ? 1
499 : SaturateAdd(current_age, code_flushing_increase_);
500 } while (sfi->CompareExchangeAge(current_age, updated_age) != current_age);
501 } else if (v8_flags.flush_code_based_on_tab_visibility) {
502 // No need to increment age.
503 } else {
504 uint16_t age = sfi->age();
505 if (age < v8_flags.bytecode_old_age) {
506 sfi->CompareExchangeAge(age, age + 1);
507 }
508 DCHECK_LE(sfi->age(), v8_flags.bytecode_old_age);
509 }
510}

512template <typename ConcreteVisitor>
513bool MarkingVisitorBase<ConcreteVisitor>::ShouldFlushBaselineCode(
514 Tagged<JSFunction> js_function) const {
515 if (!IsBaselineCodeFlushingEnabled(code_flush_mode_)) return false;
516 // Do a raw read for shared and code fields here since this function may be
517 // called on a concurrent thread. JSFunction itself should be fully
518 // initialized here but the SharedFunctionInfo, InstructionStream objects may
519 // not be initialized. We read using acquire loads to defend against that.
520 Tagged<Object> maybe_shared =
521 ACQUIRE_READ_FIELD(js_function, JSFunction::kSharedFunctionInfoOffset);
522 if (!IsSharedFunctionInfo(maybe_shared)) return false;
523
524 // See crbug.com/v8/11972 for more details on acquire / release semantics for
525 // code field. We don't use release stores when copying code pointers from
526 // SFI / FV to JSFunction but it is safe in practice.
527 Tagged<Object> maybe_code =
528 js_function->raw_code(heap_->isolate(), kAcquireLoad);
529
530#ifdef THREAD_SANITIZER
531 // This is needed because TSAN does not process the memory fence
532 // emitted after page initialization.
533 MemoryChunk::FromAddress(maybe_code.ptr())->SynchronizedLoad();
534#endif
535 if (!IsCode(maybe_code)) return false;
536 Tagged<Code> code = Cast<Code>(maybe_code);
537 if (code->kind() != CodeKind::BASELINE) return false;
538
539 Tagged<SharedFunctionInfo> shared = Cast<SharedFunctionInfo>(maybe_shared);
540 return HasBytecodeArrayForFlushing(shared) && ShouldFlushCode(shared);
541}

547template <typename ConcreteVisitor>
548size_t MarkingVisitorBase<ConcreteVisitor>::VisitFixedArrayWithProgressTracker(
549 Tagged<Map> map, Tagged<FixedArray> object,
550 MarkingProgressTracker& progress_tracker) {
551 static_assert(kMaxRegularHeapObjectSize % kTaggedSize == 0);
552 static constexpr size_t kMaxQueuedWorklistItems = 8u;
553 DCHECK(concrete_visitor()->marking_state()->IsMarked(object));
554
555 const size_t size = FixedArray::BodyDescriptor::SizeOf(map, object);
556 const size_t chunk = progress_tracker.GetNextChunkToMark();
557 const size_t total_chunks = progress_tracker.TotalNumberOfChunks();
558 size_t start = 0;
559 size_t end = 0;
560 if (chunk == 0) {
561 // We just started marking the fixed array. Push the total number of chunks
562 // to the marking worklist and publish it so that other markers can
563 // participate.
564 if (const auto target_worklist =
565 MarkingHelper::ShouldMarkObject(heap_, object)) {
566 DCHECK_EQ(target_worklist.value(),
567 MarkingHelper::WorklistTarget::kRegular);
568 const size_t scheduled_chunks =
569 std::min(total_chunks, kMaxQueuedWorklistItems);
570 DCHECK_GT(scheduled_chunks, 0);
571 for (size_t i = 1; i < scheduled_chunks; ++i) {
572 local_marking_worklists_->Push(object);
573 // Publish each chunk into a new segment so that other markers would be
574 // able to steal work. This is probabilistic (a single marker can be
575 // fast and steal multiple segments), but it works well in practice.
576 local_marking_worklists_->ShareWork();
577 }
578 }
579 concrete_visitor()
580 ->template VisitMapPointerIfNeeded<VisitorId::kVisitFixedArray>(object);
581 start = FixedArray::BodyDescriptor::kStartOffset;
582 end = std::min(size, MarkingProgressTracker::kChunkSize);
583 } else {
584 start = chunk * MarkingProgressTracker::kChunkSize;
585 end = std::min(size, start + MarkingProgressTracker::kChunkSize);
586 }
587
588 // Repost the task if needed.
589 if (chunk + kMaxQueuedWorklistItems < total_chunks) {
590 if (const auto target_worklist =
591 MarkingHelper::ShouldMarkObject(heap_, object)) {
592 local_marking_worklists_->Push(object);
593 local_marking_worklists_->ShareWork();
594 }
595 }
596
597 if (start < end) {
598 VisitPointers(object,
599 Cast<HeapObject>(object)->RawField(static_cast<int>(start)),
600 Cast<HeapObject>(object)->RawField(static_cast<int>(end)));
601 }
602
603 return end - start;
604}

634template <typename ConcreteVisitor>
635size_t MarkingVisitorBase<ConcreteVisitor>::VisitEphemeronHashTable(
636 Tagged<Map> map, Tagged<EphemeronHashTable> table, MaybeObjectSize) {
637 local_weak_objects_->ephemeron_hash_tables_local.Push(table);
638
639 for (InternalIndex i : table->IterateEntries()) {
640 ObjectSlot key_slot =
641 table->RawFieldOfElementAt(EphemeronHashTable::EntryToIndex(i));
642 Tagged<HeapObject> key = Cast<HeapObject>(table->KeyAt(i, kRelaxedLoad));
643
644 SynchronizePageAccess(key);
645 concrete_visitor()->RecordSlot(table, key_slot, key);
646 concrete_visitor()->AddWeakReferenceForReferenceSummarizer(table, key);
647
648 ObjectSlot value_slot =
649 table->RawFieldOfElementAt(EphemeronHashTable::EntryToValueIndex(i));
650
651 // Objects in the shared heap are prohibited from being used as keys in
652 // WeakMaps and WeakSets and therefore cannot be ephemeron keys. See also
653 // MarkCompactCollector::ProcessEphemeron.
654 DCHECK(!HeapLayout::InWritableSharedSpace(key));
655 if (MarkingHelper::IsMarkedOrAlwaysLive(
656 heap_, concrete_visitor()->marking_state(), key)) {
657 VisitPointer(table, value_slot);
658 } else {
659 Tagged<Object> value_obj = table->ValueAt(i);
660
661 if (IsHeapObject(value_obj)) {
662 Tagged<HeapObject> value = Cast<HeapObject>(value_obj);
663 SynchronizePageAccess(value);
664 concrete_visitor()->RecordSlot(table, value_slot, value);
665 concrete_visitor()->AddWeakReferenceForReferenceSummarizer(table,
666 value);
667
668 const auto target_worklist =
669 MarkingHelper::ShouldMarkObject(heap_, value);
670 if (!target_worklist) {
671 continue;
672 }
673
674 // Revisit ephemerons with both key and value unreachable at end
675 // of concurrent marking cycle.
676 if (concrete_visitor()->marking_state()->IsUnmarked(value)) {
677 local_weak_objects_->discovered_ephemerons_local.Push(
678 Ephemeron{key, value});
679 }
680 }
681 }
682 }
683 return table->SizeFromMap(map);
684}

686template <typename ConcreteVisitor>
687size_t MarkingVisitorBase<ConcreteVisitor>::VisitJSWeakRef(
688 Tagged<Map> map, Tagged<JSWeakRef> weak_ref,
689 MaybeObjectSize maybe_object_size) {
690 if (IsHeapObject(weak_ref->target())) {
691 Tagged<HeapObject> target = Cast<HeapObject>(weak_ref->target());
692 SynchronizePageAccess(target);
693 concrete_visitor()->AddWeakReferenceForReferenceSummarizer(weak_ref,
694 target);
695 if (MarkingHelper::IsMarkedOrAlwaysLive(
696 heap_, concrete_visitor()->marking_state(), target)) {
697 // Record the slot inside the JSWeakRef, since the VisitJSWeakRef above
698 // didn't visit it.
699 ObjectSlot slot = weak_ref->RawField(JSWeakRef::kTargetOffset);
700 concrete_visitor()->RecordSlot(weak_ref, slot, target);
701 } else {
702 // JSWeakRef points to a potentially dead object. We have to process them
703 // when we know the liveness of the whole transitive closure.
704 local_weak_objects_->js_weak_refs_local.Push(weak_ref);
705 }
706 }
707 return Base::VisitJSWeakRef(map, weak_ref, maybe_object_size);
708}

710template <typename ConcreteVisitor>
711size_t MarkingVisitorBase<ConcreteVisitor>::VisitWeakCell(
712 Tagged<Map> map, Tagged<WeakCell> weak_cell,
713 MaybeObjectSize maybe_object_size) {
714 Tagged<HeapObject> target = weak_cell->relaxed_target();
715 Tagged<HeapObject> unregister_token = weak_cell->relaxed_unregister_token();
716 SynchronizePageAccess(target);
717 SynchronizePageAccess(unregister_token);
718 if (MarkingHelper::IsMarkedOrAlwaysLive(
719 heap_, concrete_visitor()->marking_state(), target) &&
720 MarkingHelper::IsMarkedOrAlwaysLive(
721 heap_, concrete_visitor()->marking_state(), unregister_token)) {
722 // Record the slots inside the WeakCell, since its IterateBody doesn't visit
723 // it.
724 ObjectSlot slot = weak_cell->RawField(WeakCell::kTargetOffset);
725 concrete_visitor()->RecordSlot(weak_cell, slot, target);
726 slot = weak_cell->RawField(WeakCell::kUnregisterTokenOffset);
727 concrete_visitor()->RecordSlot(weak_cell, slot, unregister_token);
728 } else {
729 // WeakCell points to a potentially dead object or a dead unregister
730 // token. We have to process them when we know the liveness of the whole
731 // transitive closure.
732 local_weak_objects_->weak_cells_local.Push(weak_cell);
733 concrete_visitor()->AddWeakReferenceForReferenceSummarizer(weak_cell,
734 target);
735 concrete_visitor()->AddWeakReferenceForReferenceSummarizer(
736 weak_cell, unregister_token);
737 }
738 return Base::VisitWeakCell(map, weak_cell, maybe_object_size);
739}

745template <typename ConcreteVisitor>
746size_t MarkingVisitorBase<ConcreteVisitor>::VisitDescriptorArrayStrongly(
747 Tagged<Map> map, Tagged<DescriptorArray> array, MaybeObjectSize) {
748 this->template VisitMapPointerIfNeeded<VisitorId::kVisitDescriptorArray>(
749 array);
750 const int size = DescriptorArray::BodyDescriptor::SizeOf(map, array);
751 VisitPointers(array, array->GetFirstPointerSlot(),
752 array->GetDescriptorSlot(0));
753 VisitPointers(array, MaybeObjectSlot(array->GetDescriptorSlot(0)),
754 MaybeObjectSlot(
755 array->GetDescriptorSlot(array->number_of_descriptors())));
756 return size;
757}
758
759template <typename ConcreteVisitor>
760size_t MarkingVisitorBase<ConcreteVisitor>::VisitDescriptorArray(
761 Tagged<Map> map, Tagged<DescriptorArray> array,
762 MaybeObjectSize maybe_object_size) {
763 if (!concrete_visitor()->CanUpdateValuesInHeap()) {
764 // If we cannot update the values in the heap, we just treat the array
765 // strongly.
766 return VisitDescriptorArrayStrongly(map, array, maybe_object_size);
767 }
768
769 // The markbit is not used anymore. This is different from a checked
770 // transition in that the array is re-added to the worklist and thus there's
771 // many invocations of this transition. All cases (roots, marking via map,
772 // write barrier) are handled here as they all update the state accordingly.
773 const auto [start, end] =
774 DescriptorArrayMarkingState::AcquireDescriptorRangeToMark(
775 mark_compact_epoch_, array);
776 if (start != end) {
777 DCHECK_LT(start, end);
778 VisitPointers(array, MaybeObjectSlot(array->GetDescriptorSlot(start)),
779 MaybeObjectSlot(array->GetDescriptorSlot(end)));
780 if (start == 0) {
781 // We are processing the object the first time. Visit the header and
782 // return a size for accounting.
783 size_t size = DescriptorArray::BodyDescriptor::SizeOf(map, array);
784 VisitPointers(array, array->GetFirstPointerSlot(),
785 array->GetDescriptorSlot(0));
786 concrete_visitor()
787 ->template VisitMapPointerIfNeeded<VisitorId::kVisitDescriptorArray>(
788 array);
789 return size;
790 }
791 }
792 return 0;
793}

795template <typename ConcreteVisitor>
796void MarkingVisitorBase<ConcreteVisitor>::VisitDescriptorsForMap(
797 Tagged<Map> map) {
798 if (!concrete_visitor()->CanUpdateValuesInHeap() || !map->CanTransition())
799 return;
800
801 // Maps that can transition share their descriptor arrays and require
802 // special visiting logic to avoid memory leaks.
803 // Since descriptor arrays are potentially shared, ensure that only the
804 // descriptors that belong to this map are marked. The first time a
805 // non-empty descriptor array is marked, its header is also visited. The
806 // slot holding the descriptor array will be implicitly recorded when the
807 // pointer fields of this map are visited.
808 Tagged<Object> maybe_descriptors =
809 TaggedField<Object, Map::kInstanceDescriptorsOffset>::Acquire_Load(
810 heap_->isolate(), map);
811
812 // If the descriptors are a Smi, then this Map is in the process of being
813 // deserialized, and doesn't yet have an initialized descriptor field.
814 if (IsSmi(maybe_descriptors)) {
815 DCHECK_EQ(maybe_descriptors, Smi::uninitialized_deserialization_value());
816 return;
817 }
818
819 Tagged<DescriptorArray> descriptors =
820 Cast<DescriptorArray>(maybe_descriptors);
821 // Synchronize reading of page flags for tsan.
822 SynchronizePageAccess(descriptors);
823 // Normal processing of descriptor arrays through the pointers iteration that
824 // follows this call:
825 // - Array in read only space;
826 // - Array in a black allocated page;
827 // - StrongDescriptor array;
828 if (HeapLayout::InReadOnlySpace(descriptors) ||
829 IsStrongDescriptorArray(descriptors)) {
830 return;
831 }
832
833 if (v8_flags.black_allocated_pages &&
834 HeapLayout::InBlackAllocatedPage(descriptors)) {
835 return;
836 }
837
838 const int number_of_own_descriptors = map->NumberOfOwnDescriptors();
839 if (number_of_own_descriptors) {
840 // It is possible that the concurrent marker observes the
841 // number_of_own_descriptors out of sync with the descriptors. In that
842 // case the marking write barrier for the descriptor array will ensure
843 // that all required descriptors are marked. The concurrent marker
844 // just should avoid crashing in that case. That's why we need the
845 // std::min<int>() below.
846 const auto descriptors_to_mark = std::min<int>(
847 number_of_own_descriptors, descriptors->number_of_descriptors());
848 concrete_visitor()->marking_state()->TryMark(descriptors);
849 if (DescriptorArrayMarkingState::TryUpdateIndicesToMark(
850 mark_compact_epoch_, descriptors, descriptors_to_mark)) {
851#ifdef DEBUG
852 const auto target_worklist =
853 MarkingHelper::ShouldMarkObject(heap_, descriptors);
854 DCHECK(target_worklist);
855 DCHECK_EQ(target_worklist.value(),
856 MarkingHelper::WorklistTarget::kRegular);
857#endif // DEBUG
858 local_marking_worklists_->Push(descriptors);
859 }
860 }
861}

904 TrustedPointerTable::Space* space =
905 use_shared_table

909 }
