84 // Since this pass changes the representation of Phis, it makes some untagging
85 // operations outdated: if we've decided that a Phi should have Int32
86 // representation, then we don't need to do a kCheckedSmiUntag before using
87 // it. UpdateNodeInputs(n) removes such untagging from {n}'s input (and insert
88 // new conversions if needed, from Int32 to Float64 for instance).
89 template <class NodeT>
90 ProcessResult UpdateNodeInputs(NodeT* n, const ProcessingState* state) {
91 NodeBase* node = static_cast<NodeBase*>(n);
92
93 ProcessResult result = ProcessResult::kContinue;
94 if (IsUntagging(n->opcode())) {
95 if (node->input(0).node()->Is<Phi>() &&
96 node->input(0).node()->value_representation() !=
97 ValueRepresentation::kTagged) {
98 DCHECK_EQ(node->input_count(), 1);
99 // This untagging conversion is outdated, since its input has been
100 // untagged. Depending on the conversion, it might need to be replaced
101 // by another untagged->untagged conversion, or it might need to be
102 // removed alltogether (or rather, replaced by an identity node).
103 UpdateUntaggingOfPhi(node->input(0).node()->Cast<Phi>(),
104 n->template Cast<ValueNode>());
105 }
106 } else {
107 result = UpdateNonUntaggingNodeInputs(n, state);
108 }
109
110 // It's important to check the properties of {node} rather than the static
111 // properties of `NodeT`, because `UpdateUntaggingOfPhi` could have changed
112 // the opcode of {node}, potentially converting a deopting node into a
113 // non-deopting one.
114 if (node->properties().can_eager_deopt()) {
115 BypassIdentities(node->eager_deopt_info());
116 }
117 if (node->properties().can_lazy_deopt()) {
118 BypassIdentities(node->lazy_deopt_info());
119 }
120
121 return result;
122 }
123
124 template <class NodeT>
125 ProcessResult UpdateNonUntaggingNodeInputs(NodeT* n,
126 const ProcessingState* state) {
127 NodeBase* node = static_cast<NodeBase*>(n);
128
129 // It would be bad to re-tag the input of an untagging node, so this
130 // function should never be called on untagging nodes.
131 DCHECK(!IsUntagging(n->opcode()));
132
133 for (int i = 0; i < n->input_count(); i++) {
134 ValueNode* input = node->input(i).node();
135 if (input->Is<Identity>()) {
136 // Bypassing the identity
137 node->change_input(i, input->input(0).node());
138 } else if (Phi* phi = input->TryCast<Phi>()) {
139 // If the input is a Phi and it was used without any untagging, then
140 // we need to retag it (with some additional checks/changes for some
141 // nodes, cf the overload of UpdateNodePhiInput).
142 ProcessResult result = UpdateNodePhiInput(n, phi, i, state);
143 if (V8_UNLIKELY(result == ProcessResult::kRemove)) {
144 return ProcessResult::kRemove;
145 }
146 }
147 }
148
149 return ProcessResult::kContinue;
150 }
