48 auto task =

63 if (handle_.IsCanceled() || (collector_->epoch() != saved_epoch_)) return;

65 collector_->set_override_stack_state(EmbedderStackState::kNoHeapPointers);
66 collector_->CollectGarbage(config_);

98void GCInvoker::GCInvokerImpl::CollectGarbage(GCConfig config) {
99 DCHECK_EQ(config.marking_type, cppgc::Heap::MarkingType::kAtomic);
100 if ((config.stack_state == StackState::kNoHeapPointers) ||
101 (stack_support_ ==
102 cppgc::Heap::StackSupport::kSupportsConservativeStackScan)) {
103 collector_->CollectGarbage(config);
104 } else if (platform_->GetForegroundTaskRunner() &&
105 platform_->GetForegroundTaskRunner()->NonNestableTasksEnabled()) {
106 if (!gc_task_handle_) {
107 // Force a precise GC since it will run in a non-nestable task.
108 config.stack_state = StackState::kNoHeapPointers;
109 DCHECK_NE(cppgc::Heap::StackSupport::kSupportsConservativeStackScan,
110 stack_support_);
111 gc_task_handle_ = GCTask::Post(
112 collector_, platform_->GetForegroundTaskRunner().get(), config);
113 }
114 }
115}

117void GCInvoker::GCInvokerImpl::StartIncrementalGarbageCollection(
118 GCConfig config) {
119 DCHECK_NE(config.marking_type, cppgc::Heap::MarkingType::kAtomic);
120 if ((stack_support_ !=
121 cppgc::Heap::StackSupport::kSupportsConservativeStackScan) &&
122 (!platform_->GetForegroundTaskRunner() ||
123 !platform_->GetForegroundTaskRunner()->NonNestableTasksEnabled())) {
124 // In this configuration the GC finalization can only be triggered through
125 // ForceGarbageCollectionSlow. If incremental GC is started, there is no
126 // way to know how long it will remain enabled (and the write barrier with
127 // it). For that reason, we do not support running incremental GCs in this
128 // configuration.
129 return;
130 }
131 // No need to postpone starting incremental GC since the stack is not scanned
132 // until GC finalization.
133 collector_->StartIncrementalGarbageCollection(config);
134}
