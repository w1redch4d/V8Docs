34static size_t CountTotalHolesSize(Heap* heap) {
35 size_t holes_size = 0;
36 PagedSpaceIterator spaces(heap);
37 for (PagedSpace* space = spaces.Next(); space != nullptr;
38 space = spaces.Next()) {
39 DCHECK_GE(holes_size + space->Waste() + space->Available(), holes_size);
40 holes_size += space->Waste() + space->Available();
41 }
42 return holes_size;
43}

96GCTracer::RecordGCPhasesInfo::RecordGCPhasesInfo(
97 Heap* heap, GarbageCollector collector, GarbageCollectionReason reason) {
98 if (Heap::IsYoungGenerationCollector(collector)) {
99 type_timer_ = nullptr;
100 type_priority_timer_ = nullptr;
101 if (!v8_flags.minor_ms) {
102 mode_ = Mode::Scavenger;
103 trace_event_name_ = "V8.GCScavenger";
104 } else {
105 mode_ = Mode::None;
106 trace_event_name_ = "V8.GCMinorMS";
107 }
108 } else {
109 DCHECK_EQ(GarbageCollector::MARK_COMPACTOR, collector);
110 Counters* counters = heap->isolate()->counters();
111 const bool in_background = heap->isolate()->is_backgrounded();
112 const bool is_incremental = !heap->incremental_marking()->IsStopped();
113 mode_ = Mode::None;
114 // The following block selects histogram counters to emit. The trace event
115 // name should be changed when metrics are updated.
116 //
117 // Memory reducing GCs take priority over memory measurement GCs. They can
118 // happen at the same time when measuring memory is folded into a memory
119 // reducing GC.
120 if (is_incremental) {
121 if (heap->ShouldReduceMemory()) {
122 type_timer_ = counters->gc_finalize_incremental_memory_reducing();
123 type_priority_timer_ =
124 in_background
125 ? counters->gc_finalize_incremental_memory_reducing_background()
126 : counters
127 ->gc_finalize_incremental_memory_reducing_foreground();
128 trace_event_name_ = "V8.GCFinalizeMCReduceMemory";
129 } else if (reason == GarbageCollectionReason::kMeasureMemory) {
130 type_timer_ = counters->gc_finalize_incremental_memory_measure();
131 type_priority_timer_ =
132 in_background
133 ? counters->gc_finalize_incremental_memory_measure_background()
134 : counters->gc_finalize_incremental_memory_measure_foreground();
135 trace_event_name_ = "V8.GCFinalizeMCMeasureMemory";
136 } else {
137 type_timer_ = counters->gc_finalize_incremental_regular();
138 type_priority_timer_ =
139 in_background
140 ? counters->gc_finalize_incremental_regular_background()
141 : counters->gc_finalize_incremental_regular_foreground();
142 trace_event_name_ = "V8.GCFinalizeMC";
143 mode_ = Mode::Finalize;
144 }
145 } else {
146 trace_event_name_ = "V8.GCCompactor";
147 if (heap->ShouldReduceMemory()) {
148 type_timer_ = counters->gc_finalize_non_incremental_memory_reducing();
149 type_priority_timer_ =
150 in_background
151 ? counters
152 ->gc_finalize_non_incremental_memory_reducing_background()
153 : counters
154 ->gc_finalize_non_incremental_memory_reducing_foreground();
155 } else if (reason == GarbageCollectionReason::kMeasureMemory) {
156 type_timer_ = counters->gc_finalize_non_incremental_memory_measure();
157 type_priority_timer_ =
158 in_background
159 ? counters
160 ->gc_finalize_non_incremental_memory_measure_background()
161 : counters
162 ->gc_finalize_non_incremental_memory_measure_foreground();
163 } else {
164 type_timer_ = counters->gc_finalize_non_incremental_regular();
165 type_priority_timer_ =
166 in_background
167 ? counters->gc_finalize_non_incremental_regular_background()
168 : counters->gc_finalize_non_incremental_regular_foreground();
169 }
170 }
171 }

228void GCTracer::StartCycle(GarbageCollector collector,
229 GarbageCollectionReason gc_reason,
230 const char* collector_reason, MarkingType marking) {
231 // We cannot start a new cycle while there's another one in its atomic pause.
232 DCHECK_NE(Event::State::ATOMIC, current_.state);
233 // We cannot start a new cycle while a young generation GC cycle has
234 // already interrupted a full GC cycle.
235 DCHECK(!young_gc_while_full_gc_);
236
237 young_gc_while_full_gc_ = current_.state != Event::State::NOT_RUNNING;
238 CHECK_IMPLIES(v8_flags.separate_gc_phases && young_gc_while_full_gc_,
239 current_.state == Event::State::SWEEPING);
240 if (young_gc_while_full_gc_) {
241 // The cases for interruption are: Scavenger, MinorMS interrupting sweeping.
242 // In both cases we are fine with fetching background counters now and
243 // fixing them up later in StopAtomicPause().
244 FetchBackgroundCounters();
245 }
246
247 DCHECK_IMPLIES(young_gc_while_full_gc_,
248 Heap::IsYoungGenerationCollector(collector));
249 DCHECK_IMPLIES(young_gc_while_full_gc_,
250 !Event::IsYoungGenerationEvent(current_.type));
251
252 Event::Type type;
253 switch (collector) {
254 case GarbageCollector::SCAVENGER:
255 type = Event::Type::SCAVENGER;
256 break;
257 case GarbageCollector::MINOR_MARK_SWEEPER:
258 type = marking == MarkingType::kIncremental
259 ? Event::Type::INCREMENTAL_MINOR_MARK_SWEEPER
260 : Event::Type::MINOR_MARK_SWEEPER;
261 break;
262 case GarbageCollector::MARK_COMPACTOR:
263 type = marking == MarkingType::kIncremental
264 ? Event::Type::INCREMENTAL_MARK_COMPACTOR
265 : Event::Type::MARK_COMPACTOR;
266 break;
267 }
268
269 DCHECK_IMPLIES(!young_gc_while_full_gc_,
270 current_.state == Event::State::NOT_RUNNING);
271 DCHECK_EQ(Event::State::NOT_RUNNING, previous_.state);
272
273 previous_ = current_;
274 current_ = Event(type, Event::State::MARKING, gc_reason, collector_reason,
275 heap_->isolate()->priority());
276
277 switch (marking) {
278 case MarkingType::kAtomic:
279 DCHECK(IsInObservablePause());
280 // TODO(chromium:1154636): The start_time of the current event contains
281 // currently the start time of the observable pause. This should be
282 // reconsidered.
283 current_.start_time = start_of_observable_pause_.value();
284 current_.reduce_memory = heap_->ShouldReduceMemory();
285 break;
286 case MarkingType::kIncremental:
287 // The current event will be updated later.
288 DCHECK_IMPLIES(Heap::IsYoungGenerationCollector(collector),
289 (v8_flags.minor_ms &&
290 collector == GarbageCollector::MINOR_MARK_SWEEPER));
291 DCHECK(!IsInObservablePause());
292 break;
293 }
294
295 if (Heap::IsYoungGenerationCollector(collector)) {
296 epoch_young_ = next_epoch();
297 } else {
298 epoch_full_ = next_epoch();
299 }
300}

338void GCTracer::StopObservablePause(GarbageCollector collector,
339 base::TimeTicks time) {
340 DCHECK(IsConsistentWithCollector(collector));
341 DCHECK(IsInObservablePause());
342 start_of_observable_pause_.reset();
343
344 // TODO(chromium:1154636): The end_time of the current event contains
345 // currently the end time of the observable pause. This should be
346 // reconsidered.
347 current_.end_time = time;
348
349 FetchBackgroundCounters();
350
351 const base::TimeDelta duration = current_.end_time - current_.start_time;
352 auto* long_task_stats = heap_->isolate()->GetCurrentLongTaskStats();
353 const bool is_young = Heap::IsYoungGenerationCollector(collector);
354 if (is_young) {
355 recorded_minor_gc_atomic_pause_.Push(
356 BytesAndDuration(current_.survived_young_object_size, duration));
357 long_task_stats->gc_young_wall_clock_duration_us +=
358 duration.InMicroseconds();
359 } else {
360 if (current_.type == Event::Type::INCREMENTAL_MARK_COMPACTOR) {
361 RecordIncrementalMarkingSpeed(current_.incremental_marking_bytes,
362 current_.incremental_marking_duration);
363 recorded_incremental_mark_compacts_.Push(
364 BytesAndDuration(current_.end_object_size, duration));
365 for (int i = 0; i < Scope::NUMBER_OF_INCREMENTAL_SCOPES; i++) {
366 current_.incremental_scopes[i] = incremental_scopes_[i];
367 current_.scopes[i] = incremental_scopes_[i].duration;
368 new (&incremental_scopes_[i]) IncrementalInfos;
369 }
370 } else {
371 recorded_mark_compacts_.Push(
372 BytesAndDuration(current_.end_object_size, duration));
373 DCHECK_EQ(0u, current_.incremental_marking_bytes);
374 DCHECK(current_.incremental_marking_duration.IsZero());
375 }
376 RecordGCSumCounters();
377 combined_mark_compact_speed_cache_ = std::nullopt;
378 long_task_stats->gc_full_atomic_wall_clock_duration_us +=
379 duration.InMicroseconds();
380 RecordMutatorUtilization(current_.end_time,
381 duration + current_.incremental_marking_duration);
382 }
383
384 heap_->UpdateTotalGCTime(duration);
385
386 if (v8_flags.trace_gc_ignore_scavenger && is_young) return;
387
388 if (v8_flags.trace_gc_nvp) {
389 PrintNVP();
390 } else {
391 Print();
392 }
393
394 if (v8_flags.trace_gc) {
395 heap_->PrintShortHeapStatistics();
396 }
397
398 if (V8_UNLIKELY(TracingFlags::gc.load(std::memory_order_relaxed) &
399 v8::tracing::TracingCategoryObserver::ENABLED_BY_TRACING)) {
400 TRACE_GC_NOTE("V8.GC_HEAP_DUMP_STATISTICS");
401 std::stringstream heap_stats;
402 heap_->DumpJSONHeapStatistics(heap_stats);
403
404 TRACE_EVENT_INSTANT1(TRACE_DISABLED_BY_DEFAULT("v8.gc"), "V8.GC_Heap_Stats",
405 TRACE_EVENT_SCOPE_THREAD, "stats",
406 TRACE_STR_COPY(heap_stats.str().c_str()));
407 }
408}

410void GCTracer::UpdateMemoryBalancerGCSpeed() {
411 DCHECK(v8_flags.memory_balancer);
412 size_t major_gc_bytes = current_.start_object_size;
413 const base::TimeDelta atomic_pause_duration =
414 current_.end_atomic_pause_time - current_.start_atomic_pause_time;
415 const base::TimeDelta blocked_time_taken =
416 atomic_pause_duration + current_.incremental_marking_duration;
417 base::TimeDelta concurrent_gc_time;
418 {
419 base::SpinningMutexGuard guard(&background_scopes_mutex_);
420 concurrent_gc_time =
421 background_scopes_[Scope::MC_BACKGROUND_EVACUATE_COPY] +
422 background_scopes_[Scope::MC_BACKGROUND_EVACUATE_UPDATE_POINTERS] +
423 background_scopes_[Scope::MC_BACKGROUND_MARKING] +
424 background_scopes_[Scope::MC_BACKGROUND_SWEEPING];
425 }
426 const base::TimeDelta major_gc_duration =
427 blocked_time_taken + concurrent_gc_time;
428 const base::TimeDelta major_allocation_duration =
429 (current_.end_atomic_pause_time - previous_mark_compact_end_time_) -
430 blocked_time_taken;
431 CHECK_GE(major_allocation_duration, base::TimeDelta());
432
433 heap_->mb_->UpdateGCSpeed(major_gc_bytes, major_gc_duration);
434}

458void GCTracer::StopCycle(GarbageCollector collector) {
459 DCHECK_EQ(Event::State::SWEEPING, current_.state);
460 current_.state = Event::State::NOT_RUNNING;
461
462 DCHECK(IsConsistentWithCollector(collector));
463
464 FetchBackgroundCounters();
465
466 if (Heap::IsYoungGenerationCollector(collector)) {
467 ReportYoungCycleToRecorder();
468
469 const v8::base::TimeDelta per_thread_wall_time =
470 YoungGenerationWallTime(current_) / current_.concurrency_estimate;
471 recorded_minor_gc_per_thread_.Push(BytesAndDuration(
472 current_.survived_young_object_size, per_thread_wall_time));
473
474 // If a young generation GC interrupted an unfinished full GC cycle, restore
475 // the event corresponding to the full GC cycle.
476 if (young_gc_while_full_gc_) {
477 // Sweeping for full GC could have occured during the young GC. Copy over
478 // any sweeping scope values to the previous_ event. The full GC sweeping
479 // scopes are never reported by young cycles.
480 previous_.scopes[Scope::MC_SWEEP] += current_.scopes[Scope::MC_SWEEP];
481 previous_.scopes[Scope::MC_BACKGROUND_SWEEPING] +=
482 current_.scopes[Scope::MC_BACKGROUND_SWEEPING];
483 std::swap(current_, previous_);
484 young_gc_while_full_gc_ = false;
485 }
486 } else {
487 ReportFullCycleToRecorder();
488
489 heap_->isolate()->counters()->mark_compact_reason()->AddSample(
490 static_cast<int>(current_.gc_reason));
491
492 if (v8_flags.trace_gc_freelists) {
493 PrintIsolate(heap_->isolate(),
494 "FreeLists statistics before collection:\n");
495 heap_->PrintFreeListsStats();
496 }
497 }
498}

510void GCTracer::StopYoungCycleIfNeeded() {
511 DCHECK(Event::IsYoungGenerationEvent(current_.type));
512 if (current_.state != Event::State::SWEEPING) return;
513 if ((current_.type == Event::Type::MINOR_MARK_SWEEPER ||
514 current_.type == Event::Type::INCREMENTAL_MINOR_MARK_SWEEPER) &&
515 !notified_young_sweeping_completed_)
516 return;
517 // Check if young cppgc was scheduled but hasn't completed yet.
518 if (heap_->cpp_heap() && notified_young_cppgc_running_ &&
519 !notified_young_cppgc_completed_)
520 return;
521 bool was_young_gc_while_full_gc_ = young_gc_while_full_gc_;
522 StopCycle(current_.type == Event::Type::SCAVENGER
523 ? GarbageCollector::SCAVENGER
524 : GarbageCollector::MINOR_MARK_SWEEPER);
525 notified_young_sweeping_completed_ = false;
526 notified_young_cppgc_running_ = false;
527 notified_young_cppgc_completed_ = false;
528 if (was_young_gc_while_full_gc_) {
529 // Check if the full gc cycle is ready to be stopped.
530 StopFullCycleIfNeeded();
531 }

534void GCTracer::NotifyFullSweepingCompleted() {
535 // Notifying twice that V8 sweeping is finished for the same cycle is possible
536 // only if Oilpan sweeping is still in progress.
537 DCHECK_IMPLIES(
538 notified_full_sweeping_completed_,
539 !notified_full_cppgc_completed_ || full_cppgc_completed_during_minor_gc_);
540
541 if (Event::IsYoungGenerationEvent(current_.type)) {
542 bool was_young_gc_while_full_gc = young_gc_while_full_gc_;
543 bool was_full_sweeping_notified = notified_full_sweeping_completed_;
544 NotifyYoungSweepingCompleted();
545 // NotifyYoungSweepingCompleted checks if the full cycle needs to be stopped
546 // as well. If full sweeping was already notified, nothing more needs to be
547 // done here.
548 if (!was_young_gc_while_full_gc || was_full_sweeping_notified) return;
549 }
550
551 DCHECK(!Event::IsYoungGenerationEvent(current_.type));
552 // Sweeping finalization can also be triggered from inside a full GC cycle's
553 // atomic pause.
554 DCHECK(current_.state == Event::State::SWEEPING ||
555 current_.state == Event::State::ATOMIC);
556
557 // Stop a full GC cycle only when both v8 and cppgc (if available) GCs have
558 // finished sweeping. This method is invoked by v8.
559 if (v8_flags.trace_gc_freelists) {
560 PrintIsolate(heap_->isolate(),
561 "FreeLists statistics after sweeping completed:\n");
562 heap_->PrintFreeListsStats();
563 }
564 notified_full_sweeping_completed_ = true;
565 StopFullCycleIfNeeded();
566}

586void GCTracer::NotifyFullCppGCCompleted() {
587 // Stop a full GC cycle only when both v8 and cppgc (if available) GCs have
588 // finished sweeping. This method is invoked by cppgc.
589 DCHECK(heap_->cpp_heap());
590 const auto* metric_recorder =
591 CppHeap::From(heap_->cpp_heap())->GetMetricRecorder();
592 USE(metric_recorder);
593 DCHECK(metric_recorder->FullGCMetricsReportPending());
594 DCHECK(!notified_full_cppgc_completed_);
595 notified_full_cppgc_completed_ = true;

678void GCTracer::NotifyMarkingStart() {
679 const auto marking_start = base::TimeTicks::Now();
680
681 // Handle code flushing time deltas. Times are incremented conservatively:
682 // 1. The first delta is 0s.
683 // 2. Any delta is rounded downwards to a full second.
684 // 3. 0s-deltas are carried over to the next GC with their precise diff. This
685 // allows for frequent GCs (within a single second) to be attributed
686 // correctly later on.
687 // 4. The first non-zero increment after a reset always just increments by 1s.
688 using SFIAgeType = decltype(code_flushing_increase_s_);
689 static_assert(SharedFunctionInfo::kAgeSize == sizeof(SFIAgeType));
690 static constexpr auto kMaxDeltaForSFIAge =
691 base::TimeDelta::FromSeconds(std::numeric_limits<SFIAgeType>::max());
692 SFIAgeType code_flushing_increase_s = 0;
693 if (last_marking_start_time_for_code_flushing_.has_value()) {
694 const auto diff =
695 marking_start - last_marking_start_time_for_code_flushing_.value();
696 if (diff > kMaxDeltaForSFIAge) {
697 code_flushing_increase_s = std::numeric_limits<SFIAgeType>::max();
698 } else {
699 code_flushing_increase_s = static_cast<SFIAgeType>(diff.InSeconds());
700 }
701 }
702 DCHECK_LE(code_flushing_increase_s, std::numeric_limits<SFIAgeType>::max());
703 code_flushing_increase_s_ = code_flushing_increase_s;
704 if (!last_marking_start_time_for_code_flushing_.has_value() ||
705 code_flushing_increase_s > 0) {
706 last_marking_start_time_for_code_flushing_ = marking_start;
707 }
708 if (V8_UNLIKELY(v8_flags.trace_flush_code)) {
709 PrintIsolate(heap_->isolate(), "code flushing: increasing time: %u s\n",
710 code_flushing_increase_s_);
711 }
712}

715 return code_flushing_increase_s_;

1320 double speed2 =
1321 FinalIncrementalMarkCompactSpeedInBytesPerMillisecond().value_or(0.0);
1322 if (speed1 < kMinimumMarkingSpeed || speed2 < kMinimumMarkingSpeed) {
1323 // No data for the incremental marking speed.
1324 // Return the non-incremental mark-compact speed.
1325 combined_mark_compact_speed_cache_ =
1326 MarkCompactSpeedInBytesPerMillisecond();
1327 } else {
1328 // Combine the speed of incremental step and the speed of the final step.
1329 // 1 / (1 / speed1 + 1 / speed2) = speed1 * speed2 / (speed1 + speed2).
1330 combined_mark_compact_speed_cache_ = speed1 * speed2 / (speed1 + speed2);
1331 }
1332 return combined_mark_compact_speed_cache_;

1583 optional_cppgc_event =

1600 event.collection_rate_cpp_in_percent =

1637 current_.scopes[Scope::MC_BACKGROUND_MARKING];

1639 current_.scopes[Scope::MC_BACKGROUND_SWEEPING];

1642 current_.scopes[Scope::MC_BACKGROUND_EVACUATE_UPDATE_POINTERS];

1645 compact_background_duration;

1675 event.main_thread_atomic.compact_wall_clock_duration_in_us =
1676 event.main_thread.compact_wall_clock_duration_in_us =
1677 compact_duration.InMicroseconds();

1681 atomic_sweeping_duration.InMicroseconds();

1683 sweeping_duration.InMicroseconds();

1697 // even if it is zero.
1698 event.main_thread_incremental.sweep_wall_clock_duration_in_us =
1699 incremental_sweeping.InMicroseconds();

1701 // Objects:
1702 event.objects.bytes_before = current_.start_object_size;
1703 event.objects.bytes_after = current_.end_object_size;
1704 event.objects.bytes_freed =
1705 current_.start_object_size - current_.end_object_size;

1708 event.memory.bytes_after = current_.end_memory_size;

1719 event.objects.bytes_before;

1736 static_cast<double>(event.objects.bytes_freed) /
1737 event.main_thread.total_wall_clock_duration_in_us;
1738 }
1739 if (total_duration_since_last_mark_compact_.IsZero()) {
1740 event.collection_weight_in_percent = 0;
1741 event.main_thread_collection_weight_in_percent = 0;
1742 } else {
1743 event.collection_weight_in_percent =
1744 static_cast<double>(event.total.total_wall_clock_duration_in_us) /
1745 total_duration_since_last_mark_compact_.InMicroseconds();

1747 static_cast<double>(event.main_thread.total_wall_clock_duration_in_us) /

1828 CopySizeMetrics(event.memory_cpp, cppgc_event.memory);
1829 DCHECK_NE(-1, cppgc_event.collection_rate_in_percent);
1830 event.collection_rate_cpp_in_percent =
1831 cppgc_event.collection_rate_in_percent;
1832 DCHECK_NE(-1, cppgc_event.efficiency_in_bytes_per_us);
1833 event.efficiency_cpp_in_bytes_per_us =
1834 cppgc_event.efficiency_in_bytes_per_us;

1836 event.main_thread_efficiency_cpp_in_bytes_per_us =
1837 cppgc_event.main_thread_efficiency_in_bytes_per_us;
1838 }
1839 }
1840#endif // defined(CPPGC_YOUNG_GENERATION)
1841
1842 // Total:
1843 const base::TimeDelta total_wall_clock_duration =
1844 YoungGenerationWallTime(current_);
1845
1846 // TODO(chromium:1154636): Consider adding BACKGROUND_YOUNG_ARRAY_BUFFER_SWEEP
1847 // (both for the case of the scavenger and the minor mark-sweeper).
1848 event.total_wall_clock_duration_in_us =

1858 event.collection_rate_in_percent = 0;

1877 : freed_bytes / main_thread_wall_clock_duration.InMicroseconds();
1878 recorder->AddMainThreadEvent(event, GetContextId(heap_->isolate()));
