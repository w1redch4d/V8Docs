23enum class StoreMode {
24 // kSet implements [[Set]] in the spec and traverses the prototype
25 // chain to invoke setters. it's used by KeyedStoreIC and StoreIC to
26 // set the properties when there is no feedback.
27 kSet,
28 // kDefineKeyedOwnInLiteral implements [[CreateDataProperty]] in the spec,
29 // and it assumes that the receiver is a JSObject that is created by us.
30 // It is used by Object.fromEntries(), CloneObjectIC and
31 // StoreInArrayLiteralIC to define a property in an object without
32 // traversing the prototype chain.
33 // TODO(v8:12548): merge this into the more generic kDefineKeyedOwn.
34 kDefineKeyedOwnInLiteral,
35 // kDefineNamedOwn implements [[CreateDataProperty]] but it can deal with
36 // user-defined receivers such as a JSProxy. It also assumes that the key
37 // is statically known. It's used to initialize named roperties in object
38 // literals and named public class fields.
39 kDefineNamedOwn,
40 // kDefineKeyedOwn implements [[CreateDataProperty]], but it can deal with
41 // user-defined receivers such as a JSProxy, and for private class fields,
42 // it will throw if the field does already exist. It's different from
43 // kDefineNamedOwn in that it does not assume the key is statically known.
44 // It's used to initialized computed public class fields and private
45 // class fields.
46 kDefineKeyedOwn
47};

266void KeyedStoreGenericAssembler::BranchIfPrototypesMayHaveReadOnlyElements(
267 TNode<Map> receiver_map, Label* maybe_read_only_elements,
268 Label* only_fast_writable_elements) {
269 TVARIABLE(Map, var_map);
270 var_map = receiver_map;
271 Label loop_body(this, &var_map);
272 Goto(&loop_body);
273
274 BIND(&loop_body);
275 {
276 TNode<Map> map = var_map.value();
277 TNode<HeapObject> prototype = LoadMapPrototype(map);
278 GotoIf(IsNull(prototype), only_fast_writable_elements);
279 TNode<Map> prototype_map = LoadMap(prototype);
280 var_map = prototype_map;
281 TNode<Uint16T> instance_type = LoadMapInstanceType(prototype_map);
282 GotoIf(IsCustomElementsReceiverInstanceType(instance_type),
283 maybe_read_only_elements);
284 TNode<Int32T> elements_kind = LoadMapElementsKind(prototype_map);
285 GotoIf(IsFastOrNonExtensibleOrSealedElementsKind(elements_kind),
286 &loop_body);
287 GotoIf(Word32Equal(elements_kind, Int32Constant(NO_ELEMENTS)), &loop_body);
288 Goto(maybe_read_only_elements);
289 }

292void KeyedStoreGenericAssembler::TryRewriteElements(
293 TNode<JSObject> receiver, TNode<Map> receiver_map,
294 TNode<FixedArrayBase> elements, TNode<NativeContext> native_context,
295 ElementsKind from_kind, ElementsKind to_kind, Label* bailout) {
296 DCHECK(IsFastPackedElementsKind(from_kind));
297 ElementsKind holey_from_kind = GetHoleyElementsKind(from_kind);
298 ElementsKind holey_to_kind = GetHoleyElementsKind(to_kind);
299 if (AllocationSite::ShouldTrack(from_kind, to_kind)) {
300 TrapAllocationMemento(receiver, bailout);
301 }
302 Label perform_transition(this), check_holey_map(this);
303 TVARIABLE(Map, var_target_map);
304 // Check if the receiver has the default |from_kind| map.
305 {
306 TNode<Map> packed_map = LoadJSArrayElementsMap(from_kind, native_context);
307 GotoIf(TaggedNotEqual(receiver_map, packed_map), &check_holey_map);
308 var_target_map = CAST(
309 LoadContextElement(native_context, Context::ArrayMapIndex(to_kind)));
310 Goto(&perform_transition);
311 }
312
313 // Check if the receiver has the default |holey_from_kind| map.
314 BIND(&check_holey_map);
315 {
316 TNode<Object> holey_map = LoadContextElement(
317 native_context, Context::ArrayMapIndex(holey_from_kind));
318 GotoIf(TaggedNotEqual(receiver_map, holey_map), bailout);
319 var_target_map = CAST(LoadContextElement(
320 native_context, Context::ArrayMapIndex(holey_to_kind)));
321 Goto(&perform_transition);
322 }
323
324 // Found a supported transition target map, perform the transition!
325 BIND(&perform_transition);
326 {
327 if (IsDoubleElementsKind(from_kind) != IsDoubleElementsKind(to_kind)) {
328 TNode<IntPtrT> capacity = LoadAndUntagFixedArrayBaseLength(elements);
329 GrowElementsCapacity(receiver, elements, from_kind, to_kind, capacity,
330 capacity, bailout);
331 }
332 StoreMap(receiver, var_target_map.value());
333 }
334}

336void KeyedStoreGenericAssembler::TryChangeToHoleyMapHelper(

410void KeyedStoreGenericAssembler::StoreElementWithCapacity(
411 TNode<JSObject> receiver, TNode<Map> receiver_map,
412 TNode<FixedArrayBase> elements, TNode<Word32T> elements_kind,
413 TNode<IntPtrT> index, TNode<Object> value, TNode<Context> context,
414 Label* slow, UpdateLength update_length) {
415 if (update_length != kDontChangeLength) {
416 CSA_DCHECK(this, IsJSArrayMap(receiver_map));
417 // Check if the length property is writable. The fast check is only
418 // supported for fast properties.
419 GotoIf(IsDictionaryMap(receiver_map), slow);
420 // The length property is non-configurable, so it's guaranteed to always
421 // be the first property.
422 TNode<DescriptorArray> descriptors = LoadMapDescriptors(receiver_map);
423 TNode<Uint32T> details = LoadDetailsByDescriptorEntry(descriptors, 0);
424 GotoIf(IsSetWord32(details, PropertyDetails::kAttributesReadOnlyMask),
425 slow);
426 }
427 static_assert(OFFSET_OF_DATA_START(FixedArray) ==
428 OFFSET_OF_DATA_START(FixedDoubleArray));
429 const int kHeaderSize = OFFSET_OF_DATA_START(FixedArray) - kHeapObjectTag;
430
431 Label check_double_elements(this), check_cow_elements(this);
432 TNode<Map> elements_map = LoadMap(elements);
433 GotoIf(IsNotFixedArrayMap(elements_map), &check_double_elements);
434
435 // FixedArray backing store -> Smi or object elements.
436 {
437 TNode<IntPtrT> offset =
438 ElementOffsetFromIndex(index, PACKED_ELEMENTS, kHeaderSize);
439 if (!IsDefineKeyedOwnInLiteral()) {
440 // Check if we're about to overwrite the hole. We can safely do that
441 // only if there can be no setters on the prototype chain.
442 // If we know that we're storing beyond the previous array length, we
443 // can skip the hole check (and always assume the hole).
444 {
445 Label hole_check_passed(this);
446 if (update_length == kDontChangeLength) {
447 TNode<Object> element =
448 CAST(Load(MachineType::AnyTagged(), elements, offset));
449 GotoIf(IsNotTheHole(element), &hole_check_passed);
450 }
451 BranchIfPrototypesMayHaveReadOnlyElements(receiver_map, slow,
452 &hole_check_passed);
453 BIND(&hole_check_passed);
454 }
455 }
456
457 // Check if the value we're storing matches the elements_kind. Smis
458 // can always be stored.
459 {
460 Label non_smi_value(this);
461 GotoIfNot(TaggedIsSmi(value), &non_smi_value);
462 // If we're about to introduce holes, ensure holey elements.
463 if (update_length == kBumpLengthWithGap) {
464 TryChangeToHoleyMapMulti(receiver, receiver_map, elements_kind, context,
465 PACKED_SMI_ELEMENTS, PACKED_ELEMENTS, slow);
466 }
467 StoreNoWriteBarrier(MachineRepresentation::kTaggedSigned, elements,
468 offset, value);
469 MaybeUpdateLengthAndReturn(receiver, index, value, update_length);
470
471 BIND(&non_smi_value);
472 }
473
474 // Check if we already have object elements; just do the store if so.
475 {
476 Label must_transition(this);
477 static_assert(PACKED_SMI_ELEMENTS == 0);
478 static_assert(HOLEY_SMI_ELEMENTS == 1);
479 GotoIf(Int32LessThanOrEqual(elements_kind,
480 Int32Constant(HOLEY_SMI_ELEMENTS)),
481 &must_transition);
482 if (update_length == kBumpLengthWithGap) {
483 TryChangeToHoleyMap(receiver, receiver_map, elements_kind, context,
484 PACKED_ELEMENTS, slow);
485 }
486 Store(elements, offset, value);
487 MaybeUpdateLengthAndReturn(receiver, index, value, update_length);
488
489 BIND(&must_transition);
490 }
491
492 // Transition to the required ElementsKind.
493 {
494 Label transition_to_double(this), transition_to_object(this);
495 TNode<NativeContext> native_context = LoadNativeContext(context);
496 Branch(IsHeapNumber(CAST(value)), &transition_to_double,
497 &transition_to_object);
498 BIND(&transition_to_double);
499 {
500 // If we're adding holes at the end, always transition to a holey
501 // elements kind, otherwise try to remain packed.
502 ElementsKind target_kind = update_length == kBumpLengthWithGap
503 ? HOLEY_DOUBLE_ELEMENTS
504 : PACKED_DOUBLE_ELEMENTS;
505 TryRewriteElements(receiver, receiver_map, elements, native_context,
506 PACKED_SMI_ELEMENTS, target_kind, slow);
507 // Reload migrated elements.
508 TNode<FixedArrayBase> double_elements = LoadElements(receiver);
509 TNode<IntPtrT> double_offset =
510 ElementOffsetFromIndex(index, PACKED_DOUBLE_ELEMENTS, kHeaderSize);
511 // Make sure we do not store signalling NaNs into double arrays.
512 TNode<Float64T> double_value =
513 Float64SilenceNaN(LoadHeapNumberValue(CAST(value)));
514 StoreNoWriteBarrier(MachineRepresentation::kFloat64, double_elements,
515 double_offset, double_value);
516 MaybeUpdateLengthAndReturn(receiver, index, value, update_length);
517 }
518
519 BIND(&transition_to_object);
520 {
521 // If we're adding holes at the end, always transition to a holey
522 // elements kind, otherwise try to remain packed.
523 ElementsKind target_kind = update_length == kBumpLengthWithGap
524 ? HOLEY_ELEMENTS
525 : PACKED_ELEMENTS;
526 TryRewriteElements(receiver, receiver_map, elements, native_context,
527 PACKED_SMI_ELEMENTS, target_kind, slow);
528 // The elements backing store didn't change, no reload necessary.
529 CSA_DCHECK(this, TaggedEqual(elements, LoadElements(receiver)));
530 Store(elements, offset, value);
531 MaybeUpdateLengthAndReturn(receiver, index, value, update_length);
532 }
533 }
534 }
535
536 BIND(&check_double_elements);
537 GotoIf(IsNotFixedDoubleArrayMap(elements_map), &check_cow_elements);
538 // FixedDoubleArray backing store -> double elements.
539 {
540 TNode<IntPtrT> offset =
541 ElementOffsetFromIndex(index, PACKED_DOUBLE_ELEMENTS, kHeaderSize);
542 if (!IsDefineKeyedOwnInLiteral()) {
543 // Check if we're about to overwrite the hole. We can safely do that
544 // only if there can be no setters on the prototype chain.
545 {
546 Label hole_check_passed(this);
547 // If we know that we're storing beyond the previous array length, we
548 // can skip the hole check (and always assume the hole).
549 if (update_length == kDontChangeLength) {
550 Label found_hole(this);
551 LoadDoubleWithHoleCheck(elements, offset, &found_hole,
552 MachineType::None());
553 Goto(&hole_check_passed);
554 BIND(&found_hole);
555 }
556 BranchIfPrototypesMayHaveReadOnlyElements(receiver_map, slow,
557 &hole_check_passed);
558 BIND(&hole_check_passed);
559 }
560 }
561
562 // Try to store the value as a double.
563 {
564 Label non_number_value(this);
565 TNode<Float64T> double_value =
566 TryTaggedToFloat64(value, &non_number_value);
567
568 // Make sure we do not store signalling NaNs into double arrays.
569 double_value = Float64SilenceNaN(double_value);
570 // If we're about to introduce holes, ensure holey elements.
571 if (update_length == kBumpLengthWithGap) {
572 TryChangeToHoleyMap(receiver, receiver_map, elements_kind, context,
573 PACKED_DOUBLE_ELEMENTS, slow);
574 }
575 StoreNoWriteBarrier(MachineRepresentation::kFloat64, elements, offset,
576 double_value);
577 MaybeUpdateLengthAndReturn(receiver, index, value, update_length);
578
579 BIND(&non_number_value);
580 }
581
582 // Transition to object elements.
583 {
584 TNode<NativeContext> native_context = LoadNativeContext(context);
585 ElementsKind target_kind = update_length == kBumpLengthWithGap
586 ? HOLEY_ELEMENTS
587 : PACKED_ELEMENTS;
588 TryRewriteElements(receiver, receiver_map, elements, native_context,
589 PACKED_DOUBLE_ELEMENTS, target_kind, slow);
590 // Reload migrated elements.
591 TNode<FixedArrayBase> fast_elements = LoadElements(receiver);
592 TNode<IntPtrT> fast_offset =

606void KeyedStoreGenericAssembler::EmitGenericElementStore(
607 TNode<JSObject> receiver, TNode<Map> receiver_map,
608 TNode<Uint16T> instance_type, TNode<IntPtrT> index, TNode<Object> value,
609 TNode<Context> context, Label* slow) {
610 Label if_fast(this), if_in_bounds(this), if_increment_length_by_one(this),
611 if_bump_length_with_gap(this), if_grow(this), if_nonfast(this),
612 if_typed_array(this), if_dictionary(this), if_shared_array(this);
613 TNode<FixedArrayBase> elements = LoadElements(receiver);
614 TNode<Int32T> elements_kind = LoadMapElementsKind(receiver_map);
615 Branch(IsFastElementsKind(elements_kind), &if_fast, &if_nonfast);
616 BIND(&if_fast);
617 Label if_array(this);
618 GotoIf(IsJSArrayInstanceType(instance_type), &if_array);
619 {
620 TNode<IntPtrT> capacity = LoadAndUntagFixedArrayBaseLength(elements);
621 Branch(UintPtrLessThan(index, capacity), &if_in_bounds, &if_grow);
622 }
623 BIND(&if_array);
624 {
625 TNode<IntPtrT> length =
626 PositiveSmiUntag(LoadFastJSArrayLength(CAST(receiver)));
627 GotoIf(UintPtrLessThan(index, length), &if_in_bounds);
628 TNode<IntPtrT> capacity = LoadAndUntagFixedArrayBaseLength(elements);
629 GotoIf(UintPtrGreaterThanOrEqual(index, capacity), &if_grow);
630 Branch(WordEqual(index, length), &if_increment_length_by_one,
631 &if_bump_length_with_gap);
632 }
633
634 BIND(&if_in_bounds);
635 {
636 StoreElementWithCapacity(receiver, receiver_map, elements, elements_kind,
637 index, value, context, slow, kDontChangeLength);
638 }
639
640 BIND(&if_increment_length_by_one);
641 {
642 StoreElementWithCapacity(receiver, receiver_map, elements, elements_kind,
643 index, value, context, slow,
644 kIncrementLengthByOne);
645 }
646
647 BIND(&if_bump_length_with_gap);
648 {
649 StoreElementWithCapacity(receiver, receiver_map, elements, elements_kind,
650 index, value, context, slow, kBumpLengthWithGap);
651 }
652
653 // Out-of-capacity accesses (index >= capacity) jump here. Additionally,
654 // an ElementsKind transition might be necessary.
655 // The index can also be negative or larger than kMaxElementIndex at this
656 // point! Jump to the runtime in that case to convert it to a named property.
657 BIND(&if_grow);
658 {
659 Comment("Grow backing store");
660 // TODO(jkummerow): Support inline backing store growth.
661 Goto(slow);
662 }
663
664 // Any ElementsKind > LAST_FAST_ELEMENTS_KIND jumps here for further
665 // dispatch.
666 BIND(&if_nonfast);
667 {
668 static_assert(LAST_ELEMENTS_KIND ==
669 LAST_RAB_GSAB_FIXED_TYPED_ARRAY_ELEMENTS_KIND);
670 GotoIf(Int32GreaterThanOrEqual(
671 elements_kind,
672 Int32Constant(FIRST_FIXED_TYPED_ARRAY_ELEMENTS_KIND)),
673 &if_typed_array);
674 GotoIf(Word32Equal(elements_kind, Int32Constant(DICTIONARY_ELEMENTS)),
675 &if_dictionary);
676 GotoIf(Word32Equal(elements_kind, Int32Constant(SHARED_ARRAY_ELEMENTS)),
677 &if_shared_array);
678 Goto(slow);
679 }
680
681 BIND(&if_dictionary);
682 {
683 Comment("Dictionary");
684 // TODO(jkummerow): Support storing to dictionary elements.
685 Goto(slow);
686 }
687
688 BIND(&if_typed_array);
689 {
690 Comment("Typed array");
691 // TODO(jkummerow): Support typed arrays. Note: RAB / GSAB backed typed
692 // arrays end up here too.
693 Goto(slow);
694 }
695
696 BIND(&if_shared_array);
697 {
698 TNode<IntPtrT> length = LoadAndUntagFixedArrayBaseLength(elements);
699 GotoIf(UintPtrGreaterThanOrEqual(index, length), slow);
700 StoreSharedArrayElement(context, elements, index, value);
701 }

704void KeyedStoreGenericAssembler::LookupPropertyOnPrototypeChain(
705 TNode<Map> receiver_map, TNode<Name> name, Label* accessor,
706 TVariable<Object>* var_accessor_pair,
707 TVariable<HeapObject>* var_accessor_holder, Label* readonly,
708 Label* bailout) {
709 Label ok_to_write(this);
710 TVARIABLE(HeapObject, var_holder);
711 TVARIABLE(Map, var_holder_map);
712 var_holder = LoadMapPrototype(receiver_map);
713 var_holder_map = LoadMap(var_holder.value());
714
715 Label loop(this, {&var_holder, &var_holder_map});
716 Goto(&loop);
717 BIND(&loop);
718 {
719 TNode<HeapObject> holder = var_holder.value();
720 GotoIf(IsNull(holder), &ok_to_write);
721 TNode<Map> holder_map = var_holder_map.value();

726 TVARIABLE(HeapObject, var_meta_storage);

733 TNode<DescriptorArray> descriptors = CAST(var_meta_storage.value());
734 TNode<IntPtrT> name_index = var_entry.value();
735 TNode<Uint32T> details = LoadDetailsByKeyIndex(descriptors, name_index);
736 JumpIfDataProperty(details, &ok_to_write, readonly);
737
738 // Accessor case.
739 // TODO(jkummerow): Implement a trimmed-down
740 // LoadAccessorFromFastObject.
741 LoadPropertyFromFastObject(holder, holder_map, descriptors, name_index,
742 details, var_accessor_pair);
743 *var_accessor_holder = holder;
744 Goto(accessor);
745 }
746
747 BIND(&found_dict);
748 {
749 TNode<PropertyDictionary> dictionary = CAST(var_meta_storage.value());
750 TNode<IntPtrT> entry = var_entry.value();
751 TNode<Uint32T> details = LoadDetailsByKeyIndex(dictionary, entry);
752 JumpIfDataProperty(details, &ok_to_write, readonly);
753
754 if (accessor != nullptr) {
755 // Accessor case.
756 *var_accessor_pair = LoadValueByKeyIndex(dictionary, entry);
757 *var_accessor_holder = holder;
758 Goto(accessor);
759 } else {
760 Goto(&ok_to_write);
761 }
762 }
763
764 BIND(&found_global);
765 {
766 TNode<GlobalDictionary> dictionary = CAST(var_meta_storage.value());
767 TNode<IntPtrT> entry = var_entry.value();
768 TNode<PropertyCell> property_cell =
769 CAST(LoadValueByKeyIndex(dictionary, entry));
770 TNode<Object> value =
771 LoadObjectField(property_cell, PropertyCell::kValueOffset);
772 GotoIf(TaggedEqual(value, TheHoleConstant()), &next_proto);
773 TNode<Uint32T> details = Unsigned(LoadAndUntagToWord32ObjectField(
774 property_cell, PropertyCell::kPropertyDetailsRawOffset));
775 JumpIfDataProperty(details, &ok_to_write, readonly);
776
777 if (accessor != nullptr) {
778 // Accessor case.
779 *var_accessor_pair = value;
780 *var_accessor_holder = holder;
781 Goto(accessor);
782 } else {
783 Goto(&ok_to_write);
784 }
785 }
786 }
787
788 BIND(&next_proto);
789 // Bailout if it can be an integer indexed exotic case.
790 GotoIf(IsJSTypedArrayInstanceType(instance_type), bailout);
791 TNode<HeapObject> proto = LoadMapPrototype(holder_map);
792 GotoIf(IsNull(proto), &ok_to_write);
793 var_holder = proto;
794 var_holder_map = LoadMap(proto);
795 Goto(&loop);
796 }
797 BIND(&ok_to_write);

800TNode<Map> KeyedStoreGenericAssembler::FindCandidateStoreICTransitionMapHandler(
801 TNode<Map> map, TNode<Name> name, Label* slow) {
802 TVARIABLE(Map, var_transition_map);
803 Label simple_transition(this), transition_array(this),
804 found_handler_candidate(this);
805
806 TNode<MaybeObject> maybe_handler =
807 LoadMaybeWeakObjectField(map, Map::kTransitionsOrPrototypeInfoOffset);
808
809 // Smi -> slow,
810 // Cleared weak reference -> slow
811 // weak reference -> simple_transition
812 // strong reference -> transition_array
813 TVARIABLE(Object, var_transition_map_or_array);
814 DispatchMaybeObject(maybe_handler, slow, slow, &simple_transition,
815 &transition_array, &var_transition_map_or_array);
816
817 BIND(&simple_transition);
818 {
819 var_transition_map = CAST(var_transition_map_or_array.value());
820 Goto(&found_handler_candidate);
821 }
822
823 BIND(&transition_array);
824 {
825 TNode<Map> maybe_handler_map =
826 LoadMap(CAST(var_transition_map_or_array.value()));
827 GotoIfNot(IsTransitionArrayMap(maybe_handler_map), slow);
828
829 TVARIABLE(IntPtrT, var_name_index);
830 Label if_found_candidate(this);
831 TNode<TransitionArray> transitions =
832 CAST(var_transition_map_or_array.value());
833 TransitionLookup(name, transitions, &if_found_candidate, &var_name_index,
834 slow);
835
836 BIND(&if_found_candidate);
837 {
838 // Given that
839 // 1) transitions with the same name are ordered in the transition
840 // array by PropertyKind and then by PropertyAttributes values,
841 // 2) kData < kAccessor,
842 // 3) NONE == 0,
843 // 4) properties with private symbol names are guaranteed to be
844 // non-enumerable (so DONT_ENUM bit in attributes is always set),
845 // the resulting map of transitioning store if it exists in the
846 // transition array is expected to be the first among the transitions
847 // with the same name.
848 // See TransitionArray::CompareDetails() for details.
849 static_assert(static_cast<int>(PropertyKind::kData) == 0);
850 static_assert(NONE == 0);
851 const int kKeyToTargetOffset = (TransitionArray::kEntryTargetIndex -
852 TransitionArray::kEntryKeyIndex) *
853 kTaggedSize;
854 var_transition_map = CAST(GetHeapObjectAssumeWeak(
855 LoadArrayElement(transitions, OFFSET_OF_DATA_START(WeakFixedArray),
856 var_name_index.value(), kKeyToTargetOffset)));
857 Goto(&found_handler_candidate);
858 }
859 }
860
861 BIND(&found_handler_candidate);
862 return var_transition_map.value();
863}

865void KeyedStoreGenericAssembler::EmitGenericPropertyStore(
866 TNode<JSReceiver> receiver, TNode<Map> receiver_map,
867 TNode<Uint16T> instance_type, const StoreICParameters* p,
868 ExitPoint* exit_point, Label* slow, Maybe<LanguageMode> maybe_language_mode,
869 UseStubCache use_stub_cache) {
870 CSA_DCHECK(this, IsSimpleObjectMap(receiver_map));
871 // TODO(rmcilroy) Type as Struct once we use a trimmed down
872 // LoadAccessorFromFastObject instead of LoadPropertyFromFastObject.
873 TVARIABLE(Object, var_accessor_pair);
874 TVARIABLE(HeapObject, var_accessor_holder);
875 Label fast_properties(this), dictionary_properties(this), accessor(this),
876 readonly(this), try_stub_cache(this);
877 TNode<Uint32T> bitfield3 = LoadMapBitField3(receiver_map);
878 TNode<Name> name = CAST(p->name());
879 Branch(IsSetWord32<Map::Bits3::IsDictionaryMapBit>(bitfield3),
880 &dictionary_properties, &fast_properties);
881
882 BIND(&fast_properties);
883 {
884 Comment("fast property store");
885 TNode<DescriptorArray> descriptors = LoadMapDescriptors(receiver_map);
886 Label descriptor_found(this), lookup_transition(this);
887 TVARIABLE(IntPtrT, var_name_index);
888 DescriptorLookup(name, descriptors, bitfield3,
889 IsAnyDefineOwn() ? slow : &descriptor_found,
890 &var_name_index, &lookup_transition);
891
892 // When dealing with class fields defined with DefineKeyedOwnIC or
893 // DefineNamedOwnIC, use the slow path to check the existing property.
894 if (!IsAnyDefineOwn()) {
895 BIND(&descriptor_found);
896 {
897 TNode<IntPtrT> name_index = var_name_index.value();
898 TNode<Uint32T> details = LoadDetailsByKeyIndex(descriptors, name_index);
899 Label data_property(this);
900 JumpIfDataProperty(details, &data_property,
901 ShouldReconfigureExisting() ? nullptr : &readonly);
902
903 if (ShouldCallSetter()) {
904 // Accessor case.
905 // TODO(jkummerow): Implement a trimmed-down
906 // LoadAccessorFromFastObject.
907 LoadPropertyFromFastObject(receiver, receiver_map, descriptors,
908 name_index, details, &var_accessor_pair);
909 var_accessor_holder = receiver;
910 Goto(&accessor);
911 } else {
912 // Handle accessor to data property reconfiguration in runtime.
913 Goto(slow);
914 }
915
916 BIND(&data_property);
917 {
918 Label shared(this);
919 GotoIf(IsJSSharedStructInstanceType(instance_type), &shared);
920
921 CheckForAssociatedProtector(name, slow);
922 OverwriteExistingFastDataProperty(receiver, receiver_map, descriptors,
923 name_index, details, p->value(),
924 slow, false);
925 exit_point->Return(p->value());
926
927 BIND(&shared);
928 {
929 StoreJSSharedStructField(p->context(), receiver, receiver_map,
930 descriptors, name_index, details,
931 p->value());
932 exit_point->Return(p->value());
933 }
934 }
935 }
936 }
937
938 BIND(&lookup_transition);
939 {
940 Comment("lookup transition");
941 CheckForAssociatedProtector(name, slow);
942
943 DCHECK_IMPLIES(use_stub_cache == kUseStubCache, IsSet());
944 Label* if_not_found =
945 use_stub_cache == kUseStubCache ? &try_stub_cache : slow;
946
947 TNode<Map> transition_map = FindCandidateStoreICTransitionMapHandler(
948 receiver_map, name, if_not_found);
949
950 // Validate the transition handler candidate and apply the transition.
951 StoreTransitionMapFlags flags = kValidateTransitionHandler;
952 if (ShouldCheckPrototypeValidity()) {
953 flags = StoreTransitionMapFlags(flags | kCheckPrototypeValidity);
954 }
955 HandleStoreICTransitionMapHandlerCase(p, transition_map, slow, flags);
956 exit_point->Return(p->value());
957 }
958 }
959
960 BIND(&dictionary_properties);
961 {
962 Comment("dictionary property store");
963 // We checked for LAST_CUSTOM_ELEMENTS_RECEIVER before, which rules out
964 // seeing global objects here (which would need special handling).
965
966 TVARIABLE(IntPtrT, var_name_index);
967 Label dictionary_found(this, &var_name_index),
968 not_found(this, &var_name_index);
969 TNode<PropertyDictionary> properties = CAST(LoadSlowProperties(receiver));
970
971 // When dealing with class fields defined with DefineKeyedOwnIC or
972 // DefineNamedOwnIC, use the slow path to check the existing property.
973 NameDictionaryLookup<PropertyDictionary>(
974 properties, name, IsAnyDefineOwn() ? slow : &dictionary_found,
975 &var_name_index, &not_found, kFindExistingOrInsertionIndex);
976
977 if (!IsAnyDefineOwn()) {
978 BIND(&dictionary_found);
979 {
980 Label check_const(this), overwrite(this), done(this);
981 TNode<Uint32T> details =
982 LoadDetailsByKeyIndex(properties, var_name_index.value());
983 JumpIfDataProperty(details, &check_const,
984 ShouldReconfigureExisting() ? nullptr : &readonly);
985
986 if (ShouldCallSetter()) {
987 // Accessor case.
988 var_accessor_pair =
989 LoadValueByKeyIndex(properties, var_name_index.value());
990 var_accessor_holder = receiver;
991 Goto(&accessor);
992 } else {
993 // We must reconfigure an accessor property to a data property
994 // here, let the runtime take care of that.
995 Goto(slow);
996 }
997
998 BIND(&check_const);
999 {
1000 if (V8_DICT_PROPERTY_CONST_TRACKING_BOOL) {
1001 GotoIfNot(IsPropertyDetailsConst(details), &overwrite);
1002 TNode<Object> prev_value =
1003 LoadValueByKeyIndex(properties, var_name_index.value());
1004
1005 Branch(TaggedEqual(prev_value, p->value()), &done, slow);
1006 } else {
1007 Goto(&overwrite);
1008 }
1009 }
1010
1011 BIND(&overwrite);
1012 {
1013 CheckForAssociatedProtector(name, slow);
1014 StoreValueByKeyIndex<PropertyDictionary>(
1015 properties, var_name_index.value(), p->value());
1016 Goto(&done);
1017 }
1018
1019 BIND(&done);
1020 exit_point->Return(p->value());
1021 }
1022 }
1023
1024 BIND(&not_found);
1025 {
1026 // TODO(jkummerow): Also add support to correctly handle integer exotic
1027 // cases for typed arrays and remove this check here.
1028 GotoIf(IsJSTypedArrayMap(receiver_map), slow);
1029 CheckForAssociatedProtector(name, slow);
1030 Label extensible(this), is_private_symbol(this);
1031 GotoIf(IsPrivateSymbol(name), &is_private_symbol);
1032 Branch(IsSetWord32<Map::Bits3::IsExtensibleBit>(bitfield3), &extensible,
1033 slow);
1034
1035 BIND(&is_private_symbol);
1036 {
1037 CSA_DCHECK(this, IsPrivateSymbol(name));
1038 // For private names, we miss to the runtime which will throw.
1039 // For private symbols, we extend and store an own property.
1040 Branch(IsPrivateName(CAST(name)), slow, &extensible);
1041 }
1042
1043 BIND(&extensible);
1044 if (ShouldCheckPrototype()) {
1045 DCHECK(ShouldCallSetter());
1046 LookupPropertyOnPrototypeChain(
1047 receiver_map, name, &accessor, &var_accessor_pair,
1048 &var_accessor_holder,
1049 ShouldReconfigureExisting() ? nullptr : &readonly, slow);
1050 }
1051 Label add_dictionary_property_slow(this);
1052 InvalidateValidityCellIfPrototype(receiver_map, bitfield3);
1053 UpdateMayHaveInterestingProperty(properties, name);
1054 AddToDictionary<PropertyDictionary>(properties, name, p->value(),
1055 &add_dictionary_property_slow,
1056 var_name_index.value());
1057 exit_point->Return(p->value());
1058
1059 BIND(&add_dictionary_property_slow);
1060 exit_point->ReturnCallRuntime(Runtime::kAddDictionaryProperty,
1061 p->context(), p->receiver(), name,
1062 p->value());
1063 }
1064 }
1065
1066 if (ShouldCallSetter()) {
1067 BIND(&accessor);
1068 {
1069 Label not_callable(this);
1070 TNode<HeapObject> accessor_pair = CAST(var_accessor_pair.value());
1071 GotoIf(IsAccessorInfo(accessor_pair), slow);
1072 CSA_DCHECK(this, IsAccessorPair(accessor_pair));

1074 CAST(LoadObjectField(accessor_pair, AccessorPair::kSetterOffset));
1075 TNode<Map> setter_map = LoadMap(setter);
1076 // FunctionTemplateInfo setters are not supported yet.
1077 GotoIf(IsFunctionTemplateInfoMap(setter_map), slow);
1078 GotoIfNot(IsCallableMap(setter_map), &not_callable);
1079
1080 Call(p->context(), setter, receiver, p->value());
1081 exit_point->Return(p->value());
1082
1083 BIND(&not_callable);
1084 {
1085 LanguageMode language_mode;
1086 if (maybe_language_mode.To(&language_mode)) {
1087 if (language_mode == LanguageMode::kStrict) {
1088 exit_point->ReturnCallRuntime(
1089 Runtime::kThrowTypeError, p->context(),
1090 SmiConstant(MessageTemplate::kNoSetterInCallback), name,
1091 var_accessor_holder.value());
1092 } else {
1093 exit_point->Return(p->value());
1094 }
1095 } else {
1096 CallRuntime(Runtime::kThrowTypeErrorIfStrict, p->context(),
1097 SmiConstant(MessageTemplate::kNoSetterInCallback), name,
1098 var_accessor_holder.value());
1099 exit_point->Return(p->value());
1100 }
1101 }
1102 }
1103 }
1104
1105 if (!ShouldReconfigureExisting() && !IsAnyDefineOwn()) {
1106 BIND(&readonly);
1107 {
1108 LanguageMode language_mode;
1109 if (maybe_language_mode.To(&language_mode)) {
1110 if (language_mode == LanguageMode::kStrict) {
1111 TNode<String> type = Typeof(p->receiver());

1113 name, type, p->receiver());
1114 } else {
1115 exit_point->Return(p->value());
1116 }
1117 } else {
1118 CallRuntime(Runtime::kThrowTypeErrorIfStrict, p->context(),
1119 SmiConstant(MessageTemplate::kStrictReadOnlyProperty), name,
1120 Typeof(p->receiver()), p->receiver());
1121 exit_point->Return(p->value());
1122 }
1123 }
1124 }
1125
1126 if (use_stub_cache == kUseStubCache) {
1127 DCHECK(IsSet());
1128 BIND(&try_stub_cache);
1129 // Do megamorphic cache lookup only for Api objects where it definitely
1130 // pays off.
1131 GotoIfNot(IsJSApiObjectInstanceType(instance_type), slow);
1132
1133 Comment("stub cache probe");
1134 TVARIABLE(MaybeObject, var_handler);
1135 Label found_handler(this, &var_handler), stub_cache_miss(this);
1136
1137 TryProbeStubCache(p->stub_cache(isolate()), receiver, name, &found_handler,
1138 &var_handler, &stub_cache_miss);
1139
1140 BIND(&found_handler);
1141 {
1142 Comment("KeyedStoreGeneric found handler");
1143 HandleStoreICHandlerCase(p, var_handler.value(), &stub_cache_miss,
1144 ICMode::kNonGlobalIC);
1145 }
1146 BIND(&stub_cache_miss);
1147 {
1148 Comment("KeyedStoreGeneric_miss");
1149 TailCallRuntime(Runtime::kKeyedStoreIC_Miss, p->context(), p->value(),
1150 p->slot(), p->vector(), p->receiver(), name);
1151 }
1152 }
