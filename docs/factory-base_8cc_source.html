80template <typename Impl>
81Handle<Code> FactoryBase<Impl>::NewCode(const NewCodeOptions& options) {
82 DirectHandle<CodeWrapper> wrapper = NewCodeWrapper();
83 Tagged<Map> map = read_only_roots().code_map();
84 int size = map->instance_size();
85 Tagged<Code> code = Cast<Code>(
86 AllocateRawWithImmortalMap(size, AllocationType::kTrusted, map));
87 DisallowGarbageCollection no_gc;
88 code->init_self_indirect_pointer(isolate());
89 code->initialize_flags(options.kind, options.is_context_specialized,
90 options.is_turbofanned);
91 code->set_builtin_id(options.builtin);
92 code->set_instruction_size(options.instruction_size);
93 code->set_metadata_size(options.metadata_size);
94 code->set_inlined_bytecode_size(options.inlined_bytecode_size);
95 code->set_osr_offset(options.osr_offset);
96 code->set_handler_table_offset(options.handler_table_offset);
97 code->set_constant_pool_offset(options.constant_pool_offset);
98 code->set_code_comments_offset(options.code_comments_offset);
99 code->set_builtin_jump_table_info_offset(
100 options.builtin_jump_table_info_offset);
101 code->set_unwinding_info_offset(options.unwinding_info_offset);
102 code->set_parameter_count(options.parameter_count);
103#ifdef V8_ENABLE_LEAPTIERING
104 code->set_js_dispatch_handle(kNullJSDispatchHandle);
105#endif // V8_ENABLE_LEAPTIERING
106
107 // Set bytecode/interpreter data or deoptimization data.
108 if (CodeKindUsesBytecodeOrInterpreterData(options.kind)) {
109 DCHECK(options.deoptimization_data.is_null());
110 Tagged<TrustedObject> data =
111 *options.bytecode_or_interpreter_data.ToHandleChecked();
112 DCHECK(IsBytecodeArray(data) || IsInterpreterData(data));
113 code->set_bytecode_or_interpreter_data(data);
114 } else if (CodeKindUsesDeoptimizationData(options.kind)) {
115 DCHECK(options.bytecode_or_interpreter_data.is_null());
116 code->set_deoptimization_data(
117 *options.deoptimization_data.ToHandleChecked());
118 } else {
119 DCHECK(options.deoptimization_data.is_null());
120 DCHECK(options.bytecode_or_interpreter_data.is_null());
121 code->clear_deoptimization_data_and_interpreter_data();
122 }
123
124 // Set bytecode offset table or source position table.
125 if (CodeKindUsesBytecodeOffsetTable(options.kind)) {
126 DCHECK(options.source_position_table.is_null());
127 code->set_bytecode_offset_table(
128 *options.bytecode_offset_table.ToHandleChecked());
129 } else if (CodeKindMayLackSourcePositionTable(options.kind)) {
130 DCHECK(options.bytecode_offset_table.is_null());
131 Handle<TrustedByteArray> table;
132 if (options.source_position_table.ToHandle(&table)) {
133 code->set_source_position_table(*table);
134 } else {
135 code->clear_source_position_table_and_bytecode_offset_table();
136 }
137 } else {
138 DCHECK(options.bytecode_offset_table.is_null());
139 code->set_source_position_table(
140 *options.source_position_table.ToHandleChecked());
141 }
142
143 // Set instruction stream and entrypoint.
144 Handle<InstructionStream> istream;
145 if (options.instruction_stream.ToHandle(&istream)) {
146 DCHECK_EQ(options.instruction_start, kNullAddress);
147 code->SetInstructionStreamAndInstructionStart(isolate(), *istream);
148 } else {
149 DCHECK_NE(options.instruction_start, kNullAddress);
150 code->set_raw_instruction_stream(Smi::zero(), SKIP_WRITE_BARRIER);
151 code->SetInstructionStartForOffHeapBuiltin(isolate(),
152 options.instruction_start);
153 }
154
155 wrapper->set_code(code);
156 code->set_wrapper(*wrapper);
157
158 code->clear_padding();
159 return handle(code, isolate());
160}

325template <typename Impl>
326Handle<BytecodeArray> FactoryBase<Impl>::NewBytecodeArray(
327 int length, const uint8_t* raw_bytecodes, int frame_size,
328 uint16_t parameter_count, uint16_t max_arguments,
329 DirectHandle<TrustedFixedArray> constant_pool,
330 DirectHandle<TrustedByteArray> handler_table, AllocationType allocation) {
331 DCHECK(allocation == AllocationType::kTrusted ||
332 allocation == AllocationType::kSharedTrusted);
333 if (length < 0 || length > BytecodeArray::kMaxLength) {
334 FATAL("Fatal JavaScript invalid size error %d", length);
335 UNREACHABLE();
336 }
337 DirectHandle<BytecodeWrapper> wrapper = NewBytecodeWrapper();
338 int size = BytecodeArray::SizeFor(length);
339 Tagged<HeapObject> result = AllocateRawWithImmortalMap(
340 size, allocation, read_only_roots().bytecode_array_map());
341 DisallowGarbageCollection no_gc;
342 Tagged<BytecodeArray> instance = Cast<BytecodeArray>(result);
343 instance->init_self_indirect_pointer(isolate());
344 instance->set_length(length);
345 instance->set_frame_size(frame_size);
346 instance->set_parameter_count(parameter_count);
347 instance->set_max_arguments(max_arguments);
348 instance->set_incoming_new_target_or_generator_register(
349 interpreter::Register::invalid_value());
350 instance->set_constant_pool(*constant_pool);
351 instance->set_handler_table(*handler_table);
352 instance->clear_source_position_table(kReleaseStore);
353 instance->set_wrapper(*wrapper);
354 CopyBytes(reinterpret_cast<uint8_t*>(instance->GetFirstBytecodeAddress()),
355 raw_bytecodes, length);
356 instance->clear_padding();
357 wrapper->set_bytecode(instance);
358 return handle(instance, isolate());
359}

548template <typename Impl>
549Handle<SharedFunctionInfo> FactoryBase<Impl>::NewSharedFunctionInfo(
550 MaybeDirectHandle<String> maybe_name,
551 MaybeDirectHandle<HeapObject> maybe_function_data, Builtin builtin, int len,
552 AdaptArguments adapt, FunctionKind kind) {
553 Handle<SharedFunctionInfo> shared =
554 NewSharedFunctionInfo(AllocationType::kOld);
555 DisallowGarbageCollection no_gc;
556 Tagged<SharedFunctionInfo> raw = *shared;
557 // Function names are assumed to be flat elsewhere.
558 DirectHandle<String> shared_name;
559 bool has_shared_name = maybe_name.ToHandle(&shared_name);
560 if (has_shared_name) {
561 DCHECK(shared_name->IsFlat());
562 raw->set_name_or_scope_info(*shared_name, kReleaseStore);
563 } else {
564 DCHECK_EQ(raw->name_or_scope_info(kAcquireLoad),
565 SharedFunctionInfo::kNoSharedNameSentinel);
566 }
567
568 DirectHandle<HeapObject> function_data;
569 if (maybe_function_data.ToHandle(&function_data)) {
570 // If we pass function_data then we shouldn't pass a builtin index, and
571 // the function_data should not be code with a builtin.
572 DCHECK(!Builtins::IsBuiltinId(builtin));
573 DCHECK(!IsInstructionStream(*function_data));
574 DCHECK(!IsCode(*function_data));
575 if (IsExposedTrustedObject(*function_data)) {
576 raw->SetTrustedData(Cast<ExposedTrustedObject>(*function_data));
577 } else {
578 raw->SetUntrustedData(*function_data);
579 }
580 } else if (Builtins::IsBuiltinId(builtin)) {
581 raw->set_builtin_id(builtin);
582 } else {
583 DCHECK(raw->HasBuiltinId());
584 DCHECK_EQ(Builtin::kIllegal, raw->builtin_id());
585 }
586
587 raw->CalculateConstructAsBuiltin();
588 raw->set_kind(kind);
589
590 switch (adapt) {
591 case AdaptArguments::kYes:
592 raw->set_formal_parameter_count(JSParameterCount(len));
593 break;
594 case AdaptArguments::kNo:
595 raw->DontAdaptArguments();
596 break;
597 }

862template <typename Impl>
863MaybeHandle<String> FactoryBase<Impl>::NewConsString(
864 Handle<String> left, Handle<String> right, AllocationType allocation) {
865 if (IsThinString(*left)) {
866 left = handle(Cast<ThinString>(*left)->actual(), isolate());
867 }
868 if (IsThinString(*right)) {
869 right = handle(Cast<ThinString>(*right)->actual(), isolate());
870 }
871 uint32_t left_length = left->length();
872 if (left_length == 0) return right;
873 uint32_t right_length = right->length();
874 if (right_length == 0) return left;
875
876 uint32_t length = left_length + right_length;
877
878 if (length == 2) {
879 uint16_t c1 = left->Get(0, isolate());
880 uint16_t c2 = right->Get(0, isolate());
881 return MakeOrFindTwoCharacterString(c1, c2);
882 }
883
884 // Make sure that an out of memory exception is thrown if the length
885 // of the new cons string is too large.
886 if (length > String::kMaxLength || length < 0) {
887 THROW_NEW_ERROR(isolate(), NewInvalidStringLengthError());
888 }
889
890 bool left_is_one_byte = left->IsOneByteRepresentation();
891 bool right_is_one_byte = right->IsOneByteRepresentation();
892 bool is_one_byte = left_is_one_byte && right_is_one_byte;
893
894 // If the resulting string is small make a flat string.
895 if (length < ConsString::kMinLength) {
896 // Note that neither of the two inputs can be a slice because:
897 static_assert(ConsString::kMinLength <= SlicedString::kMinLength);
898 DCHECK(left->IsFlat());
899 DCHECK(right->IsFlat());
900
901 static_assert(ConsString::kMinLength <= String::kMaxLength);
902 if (is_one_byte) {
903 Handle<SeqOneByteString> result =
904 NewRawOneByteString(length, allocation).ToHandleChecked();
905 DisallowGarbageCollection no_gc;
906 SharedStringAccessGuardIfNeeded access_guard(isolate());
907 uint8_t* dest = result->GetChars(no_gc, access_guard);
908 // Copy left part.
909 {
910 const uint8_t* src =
911 left->template GetDirectStringChars<uint8_t>(no_gc, access_guard);
912 CopyChars(dest, src, left_length);
913 }
914 // Copy right part.
915 {
916 const uint8_t* src =
917 right->template GetDirectStringChars<uint8_t>(no_gc, access_guard);
918 CopyChars(dest + left_length, src, right_length);
919 }
920 return result;
921 }
922
923 Handle<SeqTwoByteString> result =
924 NewRawTwoByteString(length, allocation).ToHandleChecked();
925
926 DisallowGarbageCollection no_gc;
927 SharedStringAccessGuardIfNeeded access_guard(isolate());
928 base::uc16* sink = result->GetChars(no_gc, access_guard);
929 String::WriteToFlat(*left, sink, 0, left->length(), access_guard);
930 String::WriteToFlat(*right, sink + left->length(), 0, right->length(),
931 access_guard);
932 return result;
933 }
934
935 return NewConsString(left, right, length, is_one_byte, allocation);
936}

1061 ? 0
1062 : impl()->NumberToStringCacheHash(number);
1063

1095 }
