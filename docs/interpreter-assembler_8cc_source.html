27 Bytecode bytecode,
28 OperandScale operand_scale)

31 operand_scale_(operand_scale),
32 TVARIABLE_CONSTRUCTOR(interpreted_frame_pointer_),
33 TVARIABLE_CONSTRUCTOR(bytecode_array_,
34 Parameter<BytecodeArray>(
35 InterpreterDispatchDescriptor::kBytecodeArray)),
36 TVARIABLE_CONSTRUCTOR(

38 UncheckedParameter<IntPtrT>(
39 InterpreterDispatchDescriptor::kBytecodeOffset)),
40 TVARIABLE_CONSTRUCTOR(dispatch_table_,
41 UncheckedParameter<ExternalReference>(
42 InterpreterDispatchDescriptor::kDispatchTable)),
43 TVARIABLE_CONSTRUCTOR(
44 accumulator_,
45 Parameter<Object>(InterpreterDispatchDescriptor::kAccumulator)),
46 implicit_register_use_(ImplicitRegisterUse::kNone),
47 made_call_(false),
48 reloaded_frame_ptr_(false),
49 bytecode_array_valid_(true) {
50#ifdef V8_TRACE_UNOPTIMIZED
51 TraceBytecode(Runtime::kTraceUnoptimizedBytecodeEntry);
52#endif
53 RegisterCallGenerationCallbacks([this] { CallPrologue(); },
54 [this] { CallEpilogue(); });

56 // Save the bytecode offset immediately if bytecode will make a call along
57 // the critical path, or it is a return bytecode.
58 if (Bytecodes::MakesCallAlongCriticalPath(bytecode) ||
59 Bytecodes::Returns(bytecode)) {
60 SaveBytecodeOffset();
61 }

95 TNode<IntPtrT> offset = LoadAndUntagRegister(Register::bytecode_offset());

104void InterpreterAssembler::SaveBytecodeOffset() {
105 TNode<IntPtrT> bytecode_offset = BytecodeOffset();
106 if (operand_scale() != OperandScale::kSingle) {
107 // Subtract one from the bytecode_offset such that it points to the Wide /
108 // ExtraWide prefix bytecode.
109 bytecode_offset = IntPtrSub(BytecodeOffset(), IntPtrConstant(1));
110 }
111 int store_offset =
112 Register::bytecode_offset().ToOperand() * kSystemPointerSize;
113 TNode<RawPtrT> base = GetInterpretedFramePointer();
114
115 if (SmiValuesAre32Bits()) {
116 int zero_offset = store_offset + 4;
117 int payload_offset = store_offset;
118#if V8_TARGET_LITTLE_ENDIAN
119 std::swap(zero_offset, payload_offset);
120#endif
121 StoreNoWriteBarrier(MachineRepresentation::kWord32, base,
122 IntPtrConstant(zero_offset), Int32Constant(0));
123 StoreNoWriteBarrier(MachineRepresentation::kWord32, base,
124 IntPtrConstant(payload_offset),
125 TruncateIntPtrToInt32(bytecode_offset));
126 } else {
127 StoreFullTaggedNoWriteBarrier(base, IntPtrConstant(store_offset),
128 SmiTag(bytecode_offset));
129 }
130}

186TNode<Context> InterpreterAssembler::GetContextAtDepth(TNode<Context> context,
187 TNode<Uint32T> depth) {
188 TVARIABLE(Context, cur_context, context);
189 TVARIABLE(Uint32T, cur_depth, depth);
190
191 Label context_found(this);
192
193 Label context_search(this, {&cur_depth, &cur_context});
194
195 // Fast path if the depth is 0.
196 Branch(Word32Equal(depth, Int32Constant(0)), &context_found, &context_search);
197
198 // Loop until the depth is 0.
199 BIND(&context_search);
200 {
201 cur_depth = Unsigned(Int32Sub(cur_depth.value(), Int32Constant(1)));
202 cur_context =
203 CAST(LoadContextElement(cur_context.value(), Context::PREVIOUS_INDEX));
204
205 Branch(Word32Equal(cur_depth.value(), Int32Constant(0)), &context_found,
206 &context_search);
207 }
208
209 BIND(&context_found);
210 return cur_context.value();
211}

304void InterpreterAssembler::StoreRegisterForShortStar(TNode<Object> value,
305 TNode<WordT> opcode) {
306 DCHECK(Bytecodes::IsShortStar(bytecode_));
307 implicit_register_use_ =
308 implicit_register_use_ | ImplicitRegisterUse::kWriteShortStar;
309
310 CSA_DCHECK(
311 this, UintPtrGreaterThanOrEqual(opcode, UintPtrConstant(static_cast<int>(
312 Bytecode::kFirstShortStar))));
313 CSA_DCHECK(
314 this,
315 UintPtrLessThanOrEqual(
316 opcode, UintPtrConstant(static_cast<int>(Bytecode::kLastShortStar))));
317
318 // Compute the constant that we can add to a Bytecode value to map the range
319 // [Bytecode::kStar15, Bytecode::kStar0] to the range
320 // [Register(15).ToOperand(), Register(0).ToOperand()].
321 constexpr int short_star_to_operand =
322 Register(0).ToOperand() - static_cast<int>(Bytecode::kStar0);
323 // Make sure the values count in the right direction.
324 static_assert(short_star_to_operand ==
325 Register(1).ToOperand() - static_cast<int>(Bytecode::kStar1));
326
327 TNode<IntPtrT> offset =
328 IntPtrAdd(RegisterFrameOffset(Signed(opcode)),
329 IntPtrConstant(short_star_to_operand * kSystemPointerSize));
330 StoreFullTaggedNoWriteBarrier(GetInterpretedFramePointer(), offset, value);
331}

392TNode<Word32T> InterpreterAssembler::BytecodeOperandReadUnaligned(
393 int relative_offset, MachineType result_type) {
394 static const int kMaxCount = 4;
395 DCHECK(!TargetSupportsUnalignedAccess());
396
397 int count;
398 switch (result_type.representation()) {
399 case MachineRepresentation::kWord16:
400 count = 2;
401 break;
402 case MachineRepresentation::kWord32:
403 count = 4;
404 break;
405 default:
406 UNREACHABLE();
407 }
408 MachineType msb_type =
409 result_type.IsSigned() ? MachineType::Int8() : MachineType::Uint8();
410
411#if V8_TARGET_LITTLE_ENDIAN
412 const int kStep = -1;
413 int msb_offset = count - 1;
414#elif V8_TARGET_BIG_ENDIAN
415 const int kStep = 1;
416 int msb_offset = 0;
417#else
418#error "Unknown Architecture"
419#endif
420
421 // Read the most signicant bytecode into bytes[0] and then in order
422 // down to least significant in bytes[count - 1].
423 DCHECK_LE(count, kMaxCount);
424 TNode<Word32T> bytes[kMaxCount];
425 for (int i = 0; i < count; i++) {
426 MachineType machine_type = (i == 0) ? msb_type : MachineType::Uint8();
427 TNode<IntPtrT> offset =
428 IntPtrConstant(relative_offset + msb_offset + i * kStep);
429 TNode<IntPtrT> array_offset = IntPtrAdd(BytecodeOffset(), offset);
430 bytes[i] = UncheckedCast<Word32T>(
431 Load(machine_type, BytecodeArrayTaggedPointer(), array_offset));
432 }
433
434 // Pack LSB to MSB.
435 TNode<Word32T> result = bytes[--count];
436 for (int i = 1; --count >= 0; i++) {
437 TNode<Int32T> shift = Int32Constant(i * kBitsPerByte);
438 TNode<Word32T> value = Word32Shl(bytes[count], shift);
439 result = Word32Or(value, result);
440 }
441 return result;
442}

879TNode<Object> InterpreterAssembler::ConstructWithSpread(
880 TNode<Object> target, TNode<Context> context, TNode<Object> new_target,
881 const RegListNodePair& args, TNode<UintPtrT> slot_id) {
882 // TODO(bmeurer): Unify this with the Construct bytecode feedback
883 // above once we have a way to pass the AllocationSite to the Array
884 // constructor _and_ spread the last argument at the same time.
885 DCHECK(Bytecodes::MakesCallAlongCriticalPath(bytecode_));
886
887#ifndef V8_JITLESS
888 // TODO(syg): Is the feedback collection logic here the same as
889 // CollectConstructFeedback?
890 Label extra_checks(this, Label::kDeferred), construct(this);
891 TNode<HeapObject> maybe_feedback_vector = LoadFeedbackVector();
892 GotoIf(IsUndefined(maybe_feedback_vector), &construct);
893 TNode<FeedbackVector> feedback_vector = CAST(maybe_feedback_vector);
894
895 // Increment the call count.
896 IncrementCallCount(feedback_vector, slot_id);
897
898 // Check if we have monomorphic {new_target} feedback already.
899 TNode<HeapObjectReference> feedback =
900 CAST(LoadFeedbackVectorSlot(feedback_vector, slot_id));
901 Branch(IsWeakReferenceToObject(feedback, new_target), &construct,
902 &extra_checks);
903
904 BIND(&extra_checks);
905 {
906 Label check_initialized(this), initialize(this), mark_megamorphic(this);
907
908 // Check if it is a megamorphic {new_target}.
909 Comment("check if megamorphic");
910 TNode<BoolT> is_megamorphic = TaggedEqual(
911 feedback,
912 HeapConstantNoHole(FeedbackVector::MegamorphicSentinel(isolate())));
913 GotoIf(is_megamorphic, &construct);
914
915 Comment("check if weak reference");
916 GotoIfNot(IsWeakOrCleared(feedback), &check_initialized);
917
918 // If the weak reference is cleared, we have a new chance to become
919 // monomorphic.
920 Comment("check if weak reference is cleared");
921 Branch(IsCleared(feedback), &initialize, &mark_megamorphic);
922
923 BIND(&check_initialized);
924 {
925 // Check if it is uninitialized.
926 Comment("check if uninitialized");
927 TNode<BoolT> is_uninitialized =
928 TaggedEqual(feedback, UninitializedSymbolConstant());
929 Branch(is_uninitialized, &initialize, &mark_megamorphic);
930 }
931
932 BIND(&initialize);
933 {
934 Comment("check if function in same native context");
935 GotoIf(TaggedIsSmi(new_target), &mark_megamorphic);
936 // Check if the {new_target} is a JSFunction or JSBoundFunction
937 // in the current native context.
938 TVARIABLE(HeapObject, var_current, CAST(new_target));
939 Label loop(this, &var_current), done_loop(this);
940 Goto(&loop);
941 BIND(&loop);
942 {
943 Label if_boundfunction(this), if_function(this);
944 TNode<HeapObject> current = var_current.value();
945 TNode<Uint16T> current_instance_type = LoadInstanceType(current);
946 GotoIf(InstanceTypeEqual(current_instance_type, JS_BOUND_FUNCTION_TYPE),
947 &if_boundfunction);
948 Branch(IsJSFunctionInstanceType(current_instance_type), &if_function,
949 &mark_megamorphic);
950
951 BIND(&if_function);
952 {
953 // Check that the JSFunction {current} is in the current native
954 // context.
955 TNode<Context> current_context =
956 CAST(LoadObjectField(current, JSFunction::kContextOffset));
957 TNode<NativeContext> current_native_context =
958 LoadNativeContext(current_context);
959 Branch(
960 TaggedEqual(LoadNativeContext(context), current_native_context),
961 &done_loop, &mark_megamorphic);
962 }
963
964 BIND(&if_boundfunction);
965 {
966 // Continue with the [[BoundTargetFunction]] of {current}.
967 var_current = LoadObjectField<HeapObject>(
968 current, JSBoundFunction::kBoundTargetFunctionOffset);
969 Goto(&loop);
970 }
971 }
972 BIND(&done_loop);
973 StoreWeakReferenceInFeedbackVector(feedback_vector, slot_id,
974 CAST(new_target));
975 ReportFeedbackUpdate(feedback_vector, slot_id,
976 "ConstructWithSpread:Initialize");
977 Goto(&construct);
978 }
979
980 BIND(&mark_megamorphic);
981 {
982 // MegamorphicSentinel is an immortal immovable object so
983 // write-barrier is not needed.
984 Comment("transition to megamorphic");
985 DCHECK(RootsTable::IsImmortalImmovable(RootIndex::kmegamorphic_symbol));
986 StoreFeedbackVectorSlot(
987 feedback_vector, slot_id,
988 HeapConstantNoHole(FeedbackVector::MegamorphicSentinel(isolate())),
989 SKIP_WRITE_BARRIER);
990 ReportFeedbackUpdate(feedback_vector, slot_id,
991 "ConstructWithSpread:TransitionMegamorphic");
992 Goto(&construct);
993 }
994 }
995
996 BIND(&construct);
997#endif // !V8_JITLESS
998 Comment("call using ConstructWithSpread builtin");
999 Builtin builtin = Builtins::InterpreterPushArgsThenConstruct(
1000 InterpreterPushArgsMode::kWithFinalSpread);
1001 TNode<Word32T> args_count = JSParameterCount(args.reg_count());
1002 return CallBuiltin(builtin, context, args_count, args.base_reg_location(),
1003 target, new_target, UndefinedConstant());
1004}

1228void InterpreterAssembler::StarDispatchLookahead(TNode<WordT> target_bytecode) {
1229 Label do_inline_star(this), done(this);
1230
1231 // Check whether the following opcode is one of the short Star codes. All
1232 // opcodes higher than the short Star variants are invalid, and invalid
1233 // opcodes are never deliberately written, so we can use a one-sided check.
1234 // This is no less secure than the normal-length Star handler, which performs
1235 // no validation on its operand.
1236 static_assert(static_cast<int>(Bytecode::kLastShortStar) + 1 ==
1237 static_cast<int>(Bytecode::kIllegal));
1238 static_assert(Bytecode::kIllegal == Bytecode::kLast);
1239 TNode<Int32T> first_short_star_bytecode =
1240 Int32Constant(static_cast<int>(Bytecode::kFirstShortStar));
1241 TNode<BoolT> is_star = Uint32GreaterThanOrEqual(
1242 TruncateWordToInt32(target_bytecode), first_short_star_bytecode);
1243 Branch(is_star, &do_inline_star, &done);
1244
1245 BIND(&do_inline_star);
1246 {
1247 InlineShortStar(target_bytecode);
1248
1249 // Rather than merging control flow to a single indirect jump, we can get
1250 // better branch prediction by duplicating it. This is because the
1251 // instruction following a merged X + StarN is a bad predictor of the
1252 // instruction following a non-merged X, and vice versa.
1253 DispatchToBytecode(LoadBytecode(BytecodeOffset()), BytecodeOffset());
1254 }
1255 BIND(&done);
1256}

1263 // kStar0 has the right attributes (namely, no operands) for any of the short
1264 // Star codes.

1317void InterpreterAssembler::DispatchWide(OperandScale operand_scale) {
1318 // Dispatching a wide bytecode requires treating the prefix
1319 // bytecode a base pointer into the dispatch table and dispatching
1320 // the bytecode that follows relative to this base.
1321 //
1322 // Indices 0-255 correspond to bytecodes with operand_scale == 0
1323 // Indices 256-511 correspond to bytecodes with operand_scale == 1
1324 // Indices 512-767 correspond to bytecodes with operand_scale == 2
1325 DCHECK_IMPLIES(Bytecodes::MakesCallAlongCriticalPath(bytecode_), made_call_);

1330 TraceBytecodeDispatch(next_bytecode);

1337 break;

1339 base_index = IntPtrConstant(2 << kBitsPerByte);
1340 break;

1342 UNREACHABLE();
1343 }

1345 TNode<RawPtrT> target_code_entry = Load<RawPtrT>(
1346 DispatchTablePointer(), TimesSystemPointerSize(target_index));

1351void InterpreterAssembler::UpdateInterruptBudgetOnReturn() {
1352 // TODO(rmcilroy): Investigate whether it is worth supporting self
1353 // optimization of primitive functions like FullCodegen.
1354

1356 // current bytecode and the start of the first one, to simulate backedge to
1357 // start of function.

1360 //
1361 // <---------- simulated backedge ----------
1362 // | header | first bytecode | .... | return bytecode |
1363 // |<------ current offset ------->
1364 // ^ tagged bytecode array pointer

1367 // length of the back-edge, so we just have to correct for the non-zero offset
1368 // of the first bytecode.

1371 Int32Sub(TruncateIntPtrToInt32(BytecodeOffset()),
1372 Int32Constant(kFirstBytecodeOffset));

1402void InterpreterAssembler::OnStackReplacement(
1403 TNode<Context> context, TNode<FeedbackVector> feedback_vector,
1404 TNode<IntPtrT> relative_jump, TNode<Int32T> loop_depth,
1405 TNode<IntPtrT> feedback_slot, TNode<Int8T> osr_state,
1406 OnStackReplacementParams params) {
1407 // Three cases may cause us to attempt OSR, in the following order:
1408 //
1409 // 1) Presence of cached OSR Turbofan/Maglev code.
1410 // 2) Presence of cached OSR Sparkplug code.
1411 // 3) The OSR urgency exceeds the current loop depth - in that case, trigger
1412 // a Turbofan OSR compilation.
1413
1414 TVARIABLE(Object, maybe_target_code, SmiConstant(0));
1415 Label osr_to_opt(this), osr_to_sparkplug(this);
1416
1417 // Case 1).
1418 {
1419 Label next(this);
1420 TNode<MaybeObject> maybe_cached_osr_code =
1421 LoadFeedbackVectorSlot(feedback_vector, feedback_slot);
1422 GotoIf(IsCleared(maybe_cached_osr_code), &next);
1423 maybe_target_code = GetHeapObjectAssumeWeak(maybe_cached_osr_code);
1424
1425 // Is it marked_for_deoptimization? If yes, clear the slot.
1426 TNode<CodeWrapper> code_wrapper = CAST(maybe_target_code.value());
1427 maybe_target_code =
1428 LoadCodePointerFromObject(code_wrapper, CodeWrapper::kCodeOffset);
1429 GotoIfNot(IsMarkedForDeoptimization(CAST(maybe_target_code.value())),
1430 &osr_to_opt);
1431 StoreFeedbackVectorSlot(feedback_vector, Unsigned(feedback_slot),
1432 ClearedValue(), UNSAFE_SKIP_WRITE_BARRIER);
1433 maybe_target_code = SmiConstant(0);
1434
1435 Goto(&next);
1436 BIND(&next);
1437 }
1438
1439 // Case 2).
1440 if (params == OnStackReplacementParams::kBaselineCodeIsCached) {
1441 Goto(&osr_to_sparkplug);
1442 } else {
1443 DCHECK_EQ(params, OnStackReplacementParams::kDefault);
1444 TNode<SharedFunctionInfo> sfi = LoadObjectField<SharedFunctionInfo>(
1445 LoadFunctionClosure(), JSFunction::kSharedFunctionInfoOffset);
1446 GotoIf(SharedFunctionInfoHasBaselineCode(sfi), &osr_to_sparkplug);
1447
1448 // Case 3).
1449 {
1450 static_assert(FeedbackVector::OsrUrgencyBits::kShift == 0);
1451 TNode<Int32T> osr_urgency = Word32And(
1452 osr_state, Int32Constant(FeedbackVector::OsrUrgencyBits::kMask));
1453 GotoIf(Uint32LessThan(loop_depth, osr_urgency), &osr_to_opt);
1454 JumpBackward(relative_jump);
1455 }
1456 }
1457
1458 BIND(&osr_to_opt);
1459 {
1460 TNode<BytecodeArray> bytecode = BytecodeArrayTaggedPointer();
1461 TNode<Uint32T> length = LoadAndUntagBytecodeArrayLength(bytecode);
1462 TNode<Uint32T> weight =
1463 Uint32Mul(length, Uint32Constant(v8_flags.osr_to_tierup));
1464 DecreaseInterruptBudget(Signed(weight), kDisableStackCheck);
1465 TNode<Smi> expected_param_count =
1466 SmiFromInt32(LoadBytecodeArrayParameterCount(bytecode));
1467 CallBuiltin(Builtin::kInterpreterOnStackReplacement, context,
1468 maybe_target_code.value(), expected_param_count);
1469 UpdateInterruptBudget(Int32Mul(Signed(weight), Int32Constant(-1)));
1470 JumpBackward(relative_jump);
1471 }
1472
1473 BIND(&osr_to_sparkplug);
1474 {
1475 // We already compiled the baseline code, so we don't need to handle failed
1476 // compilation as in the Ignition -> Turbofan case. Therefore we can just
1477 // tailcall to the OSR builtin.
1478 SaveBytecodeOffset();
1479 TailCallBuiltin(Builtin::kInterpreterOnStackReplacement_ToBaseline,
1480 context);
1481 }
1482}
