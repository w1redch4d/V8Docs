48void JSArray::SetContent(DirectHandle<JSArray> array,
49 DirectHandle<FixedArrayBase> storage) {
50 EnsureCanContainElements(array, storage, storage->length(),
51 ALLOW_COPIED_DOUBLE_ELEMENTS);
52#ifdef DEBUG
53 ReadOnlyRoots roots = GetReadOnlyRoots();
54 Tagged<Map> map = storage->map();
55 if (map == roots.fixed_double_array_map()) {
56 DCHECK(IsDoubleElementsKind(array->GetElementsKind()));
57 } else {
58 DCHECK_NE(map, roots.fixed_double_array_map());
59 if (IsSmiElementsKind(array->GetElementsKind())) {
60 auto elems = Cast<FixedArray>(storage);
61 Tagged<Object> the_hole = roots.the_hole_value();
62 for (int i = 0; i < elems->length(); i++) {
63 Tagged<Object> candidate = elems->get(i);
64 DCHECK(IsSmi(candidate) || candidate == the_hole);
65 }
66 } else {
67 DCHECK(IsObjectElementsKind(array->GetElementsKind()));
68 }
69 }
70#endif // DEBUG
71 array->set_elements(*storage);
72 array->set_length(Smi::FromInt(storage->length()));
73}
