317 template <class T, int size, int lo, int hi>
318 inline T getfield(T value) {
319 DCHECK(lo < hi);
320 DCHECK_GT(size, 0);
321 int mask = hi - lo;
322 int shift = size * 8 - hi;
323 uint32_t mask_value = (mask == 32) ? 0xffffffff : (1 << mask) - 1;
324 return (value & mask_value) << shift;
325 }

1431 // Helper to emit the binary encoding of a 6 byte instruction
1432 void emit6bytes(uint64_t x) {
1433 CheckBuffer();
1434#if V8_TARGET_LITTLE_ENDIAN
1435 // We need to emit instructions in big endian format as disassembler /
1436 // simulator require the first byte of the instruction in order to decode
1437 // the instruction length. Swap the bytes.
1438 x = (static_cast<uint64_t>(x & 0xFF) << 40) |
1439 (static_cast<uint64_t>((x >> 8) & 0xFF) << 32) |
1440 (static_cast<uint64_t>((x >> 16) & 0xFF) << 24) |
1441 (static_cast<uint64_t>((x >> 24) & 0xFF) << 16) |
1442 (static_cast<uint64_t>((x >> 32) & 0xFF) << 8) |
1443 (static_cast<uint64_t>((x >> 40) & 0xFF));
1444 x |= (*reinterpret_cast<uint64_t*>(pc_) >> 48) << 48;
1445#else
1446 // We need to pad two bytes of zeros in order to get the 6-bytes
1447 // stored from low address.
1448 x = x << 16;
1449 x |= *reinterpret_cast<uint64_t*>(pc_) & 0xFFFF;
1450#endif
1451 // It is safe to store 8-bytes, as CheckBuffer() guarantees we have kGap
1452 // space left over.
1453 *reinterpret_cast<uint64_t*>(pc_) = x;
1454 pc_ += 6;
1455 }
