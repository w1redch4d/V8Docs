23// Represents a linked list that threads through the nodes in the linked list.
24// Entries in the list are pointers to nodes. By default nodes need to have a
25// T** next() method that returns the location where the next value is stored.
26// The kSupportsUnsafeInsertion flag defines whether the list supports insertion
27// of new elements into the list by just rewiring the next pointers without
28// updating the list object itself. Such an insertion might invalidate the
29// pointer to list tail and thus requires additional steps to recover the
30// pointer to the tail.
31// The default can be overwritten by providing a ThreadedTraits class.
32template <typename T, typename BaseClass,
33 typename TLTraits = ThreadedListTraits<T>,
34 bool kSupportsUnsafeInsertion = false>
35class ThreadedListBase final : public BaseClass {
36 public:
37 ThreadedListBase() : head_(nullptr), tail_(&head_) {}
38 ThreadedListBase(const ThreadedListBase&) = delete;
39 ThreadedListBase& operator=(const ThreadedListBase&) = delete;
40
41 void Add(T* v) {
42 EnsureValidTail();
43 DCHECK_NULL(*tail_);
44 DCHECK_NULL(*TLTraits::next(v));
45 *tail_ = v;
46 tail_ = TLTraits::next(v);
47 // Check that only one element was added (and that hasn't created a cycle).
48 DCHECK_NULL(*tail_);
49 }
50
51 void AddFront(T* v) {
52 DCHECK_NULL(*TLTraits::next(v));
53 DCHECK_NOT_NULL(v);
54 T** const next = TLTraits::next(v);
55
56 *next = head_;
57 if (head_ == nullptr) tail_ = next;
58 head_ = v;
59 }
60
61 // This temporarily breaks the tail_ invariant, and it should only be called
62 // if we support unsafe insertions.
63 static void AddAfter(T* after_this, T* v) {
64 DCHECK(kSupportsUnsafeInsertion);
65 DCHECK_NULL(*TLTraits::next(v));
66 *TLTraits::next(v) = *TLTraits::next(after_this);
67 *TLTraits::next(after_this) = v;
68 }
69
70 void DropHead() {
71 DCHECK_NOT_NULL(head_);
72
73 T* old_head = head_;
74 head_ = *TLTraits::next(head_);
75 if (head_ == nullptr) tail_ = &head_;
76 *TLTraits::next(old_head) = nullptr;
77 }
78
79 bool Contains(T* v) {
80 for (Iterator it = begin(); it != end(); ++it) {
81 if (*it == v) return true;
82 }
83 return false;
84 }
85
86 void Append(ThreadedListBase&& list) {
87 if (list.is_empty()) return;
88
89 EnsureValidTail();
90 *tail_ = list.head_;
91 tail_ = list.tail_;
92 list.Clear();
93 }
94
95 void Prepend(ThreadedListBase&& list) {
96 if (list.head_ == nullptr) return;
97
98 EnsureValidTail();
99 T* new_head = list.head_;
100 *list.tail_ = head_;
101 if (head_ == nullptr) {
102 tail_ = list.tail_;
103 }
104 head_ = new_head;
105 list.Clear();
106 }
107
108 void Clear() {
109 head_ = nullptr;
110 tail_ = &head_;
111 }
112
113 ThreadedListBase& operator=(ThreadedListBase&& other) V8_NOEXCEPT {
114 head_ = other.head_;
115 tail_ = other.head_ ? other.tail_ : &head_;
116#ifdef DEBUG
117 other.Clear();
118#endif
119 return *this;
120 }
121
122 ThreadedListBase(ThreadedListBase&& other) V8_NOEXCEPT
123 : head_(other.head_),
124 tail_(other.head_ ? other.tail_ : &head_) {
125#ifdef DEBUG
126 other.Clear();
127#endif
128 }
129
130 bool Remove(T* v) {
131 T* current = first();
132 if (current == v) {
133 DropHead();
134 return true;
135 }
136
137 EnsureValidTail();
138 while (current != nullptr) {
139 T* next = *TLTraits::next(current);
140 if (next == v) {
141 *TLTraits::next(current) = *TLTraits::next(next);
142 *TLTraits::next(next) = nullptr;
143
144 if (TLTraits::next(next) == tail_) {
145 tail_ = TLTraits::next(current);
146 }
147 return true;
148 }
149 current = next;
150 }
151 return false;
152 }

246 // Rewinds the list's tail to the reset point, i.e., cutting of the rest of
247 // the list, including the reset_point.
248 void Rewind(Iterator reset_point) {
249 tail_ = reset_point.entry_;
250 *tail_ = nullptr;
251 }
252
253 // Moves the tail of the from_list, starting at the from_location, to the end
254 // of this list.
255 void MoveTail(ThreadedListBase* from_list, Iterator from_location) {
256 if (from_list->end() != from_location) {
257 DCHECK_NULL(*tail_);
258 *tail_ = *from_location;
259 tail_ = from_list->tail_;
260 from_list->Rewind(from_location);
261 }
262 }
263
264 // Removes the element at `it`, and returns a new iterator pointing to the
265 // element following the removed element (if `it` was pointing to the last
266 // element, then `end()` is returned). The head and the tail are updated. `it`
267 // should not be `end()`. Iterators that are currently on the same element as
268 // `it` are invalidated. Other iterators are not affected. If the last element
269 // is removed, existing `end()` iterators will be invalidated.
270 Iterator RemoveAt(Iterator it) {
271 if (*it.entry_ == head_) {
272 DropHead();
273 return begin();
274 } else if (tail_ == TLTraits::next(*it.entry_)) {
275 tail_ = it.entry_;
276 *it.entry_ = nullptr;
277 return end();
278 } else {
279 T* old_entry = *it.entry_;
280 *it.entry_ = *TLTraits::next(*it.entry_);
281 *TLTraits::next(old_entry) = nullptr;
282 return Iterator(it.entry_);
283 }
284 }

303 bool Verify() const {
304 T* last = this->first();
305 if (last == nullptr) {
306 CHECK_EQ(&head_, tail_);
307 } else {
308 while (*TLTraits::next(last) != nullptr) {
309 last = *TLTraits::next(last);
310 }
311 CHECK_EQ(TLTraits::next(last), tail_);
312 }
313 return true;
314 }
315
316 inline void EnsureValidTail() const {
317 if (!kSupportsUnsafeInsertion) {
318 DCHECK_EQ(*tail_, nullptr);
319 return;
320 }
321 // If kSupportsUnsafeInsertion, then we support adding a new element by
322 // using the pointer to a certain element. E.g., imagine list A -> B -> C,
323 // we can add D after B, by just moving the pointer of B to D and D to
324 // whatever B used to point to. We do not need to know the beginning of the
325 // list (ie. to have a pointer to the ThreadList class). This however might
326 // break the tail_ invariant. We ensure this here, by manually looking for
327 // the tail of the list.
328 if (*tail_ == nullptr) return;
329 T* last = *tail_;
330 if (last != nullptr) {
331 while (*TLTraits::next(last) != nullptr) {
332 last = *TLTraits::next(last);
333 }
334 tail_ = TLTraits::next(last);
335 }
336 }
337
338 private:
339 T* head_;
340 mutable T** tail_; // We need to ensure a valid `tail_` even when using a
341 // const Iterator.
342};
