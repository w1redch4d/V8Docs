44RUNTIME_FUNCTION(Runtime_NewArray) {
45 HandleScope scope(isolate);
46 DCHECK_LE(3, args.length());
47 int const argc = args.length() - 3;
48 // argv points to the arguments constructed by the JavaScript call.
49 JavaScriptArguments argv(argc, args.address_of_arg_at(0));
50 DirectHandle<JSFunction> constructor = args.at<JSFunction>(argc);
51 DirectHandle<JSReceiver> new_target = args.at<JSReceiver>(argc + 1);
52 Handle<HeapObject> type_info = args.at<HeapObject>(argc + 2);
53 // TODO(bmeurer): Use MaybeHandle to pass around the AllocationSite.
54 DirectHandle<AllocationSite> site = IsAllocationSite(*type_info)
55 ? Cast<AllocationSite>(type_info)
56 : Handle<AllocationSite>::null();
57
58 Factory* factory = isolate->factory();
59
60 // If called through new, new.target can be:
61 // - a subclass of constructor,
62 // - a proxy wrapper around constructor, or
63 // - the constructor itself.
64 // If called through Reflect.construct, it's guaranteed to be a constructor by
65 // REFLECT_CONSTRUCT_PREPARE.
66 DCHECK(IsConstructor(*new_target));
67
68 bool holey = false;
69 bool can_use_type_feedback = !site.is_null();
70 bool can_inline_array_constructor = true;
71 if (argv.length() == 1) {
72 DirectHandle<Object> argument_one = argv.at<Object>(0);
73 if (IsSmi(*argument_one)) {
74 int value = Cast<Smi>(*argument_one).value();
75 if (value < 0 ||
76 JSArray::SetLengthWouldNormalize(isolate->heap(), value)) {
77 // the array is a dictionary in this case.
78 can_use_type_feedback = false;
79 } else if (value != 0) {
80 holey = true;
81 if (value >= JSArray::kInitialMaxFastElementArray) {
82 can_inline_array_constructor = false;
83 }
84 }
85 } else {
86 // Non-smi length argument produces a dictionary
87 can_use_type_feedback = false;
88 }
89 }
90
91 DirectHandle<Map> initial_map;
92 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
93 isolate, initial_map,
94 JSFunction::GetDerivedMap(isolate, constructor, new_target));
95
96 ElementsKind to_kind = can_use_type_feedback ? site->GetElementsKind()
97 : initial_map->elements_kind();
98 if (holey && !IsHoleyElementsKind(to_kind)) {
99 to_kind = GetHoleyElementsKind(to_kind);
100 // Update the allocation site info to reflect the advice alteration.
101 if (!site.is_null()) site->SetElementsKind(to_kind);
102 }
103
104 // We should allocate with an initial map that reflects the allocation site
105 // advice. Therefore we use AllocateJSObjectFromMap instead of passing
106 // the constructor.
107 initial_map = Map::AsElementsKind(isolate, initial_map, to_kind);
108
109 // If we don't care to track arrays of to_kind ElementsKind, then
110 // don't emit a memento for them.
111 DirectHandle<AllocationSite> allocation_site;
112 if (AllocationSite::ShouldTrack(to_kind)) {
113 allocation_site = site;
114 }
115
116 DirectHandle<JSArray> array = Cast<JSArray>(factory->NewJSObjectFromMap(
117 initial_map, AllocationType::kYoung, allocation_site));
118
119 factory->NewJSArrayStorage(
120 array, 0, 0, ArrayStorageAllocationMode::DONT_INITIALIZE_ARRAY_ELEMENTS);
121
122 ElementsKind old_kind = array->GetElementsKind();
123 RETURN_FAILURE_ON_EXCEPTION(isolate,
124 ArrayConstructInitializeElements(array, &argv));
125 if (!site.is_null()) {
126 if ((old_kind != array->GetElementsKind() || !can_use_type_feedback ||
127 !can_inline_array_constructor)) {
128 // The arguments passed in caused a transition. This kind of complexity
129 // can't be dealt with in the inlined optimized array constructor case.
130 // We must mark the allocationsite as un-inlinable.
131 site->SetDoNotInlineCall();
132 }
133 } else {
134 if (old_kind != array->GetElementsKind() || !can_inline_array_constructor) {
135 // We don't have an AllocationSite for this Array constructor invocation,
136 // i.e. it might a call from Array#map or from an Array subclass, so we
137 // just flip the bit on the global protector cell instead.
138 // TODO(bmeurer): Find a better way to mark this. Global protectors
139 // tend to back-fire over time...
140 if (Protectors::IsArrayConstructorIntact(isolate)) {
141 Protectors::InvalidateArrayConstructor(isolate);
142 }
143 }
144 }
145
146 return *array;
147}

159// GrowArrayElements grows fast kind elements and returns a sentinel Smi if the
160// object was normalized or if the key is negative.
161RUNTIME_FUNCTION(Runtime_GrowArrayElements) {
162 HandleScope scope(isolate);
163 DCHECK_EQ(2, args.length());
164 DirectHandle<JSObject> object = args.at<JSObject>(0);
165 DirectHandle<Object> key = args.at(1);
166 ElementsKind kind = object->GetElementsKind();
167 CHECK(IsFastElementsKind(kind));
168 uint32_t index;
169 if (IsSmi(*key)) {
170 int value = Smi::ToInt(*key);
171 if (value < 0) return Smi::zero();
172 index = static_cast<uint32_t>(value);
173 } else {
174 CHECK(IsHeapNumber(*key));
175 double value = Cast<HeapNumber>(*key)->value();
176 if (value < 0 || value > std::numeric_limits<uint32_t>::max()) {
177 return Smi::zero();
178 }
179 index = static_cast<uint32_t>(value);
180 }
181
182 uint32_t capacity = static_cast<uint32_t>(object->elements()->length());
183
184 if (index >= capacity) {
185 bool has_grown;
186 MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
187 isolate, has_grown,
188 object->GetElementsAccessor()->GrowCapacity(object, index));
189 if (!has_grown) {
190 return Smi::zero();
191 }
192 }
193
194 return object->elements();
195}

223RUNTIME_FUNCTION(Runtime_ArrayIncludes_Slow) {
224 HandleScope shs(isolate);
225 DCHECK_EQ(3, args.length());
226 DirectHandle<Object> search_element = args.at(1);
227 Handle<Object> from_index = args.at(2);
228
229 // Let O be ? ToObject(this value).
230 DirectHandle<JSReceiver> object;
231 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
232 isolate, object,
233 Object::ToObject(isolate, Handle<Object>(args[0], isolate)));
234
235 // Let len be ? ToLength(? Get(O, "length")).
236 int64_t len;
237 {
238 if (object->map()->instance_type() == JS_ARRAY_TYPE) {
239 uint32_t len32 = 0;
240 bool success =
241 Object::ToArrayLength(Cast<JSArray>(*object)->length(), &len32);
242 DCHECK(success);
243 USE(success);
244 len = len32;
245 } else {
246 DirectHandle<Object> len_;
247 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
248 isolate, len_,
249 Object::GetProperty(isolate, object,
250 isolate->factory()->length_string()));
251
252 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, len_,
253 Object::ToLength(isolate, len_));
254 len = static_cast<int64_t>(Object::NumberValue(*len_));
255 DCHECK_EQ(len, Object::NumberValue(*len_));
256 }
257 }
258
259 if (len == 0) return ReadOnlyRoots(isolate).false_value();
260
261 // Let n be ? ToInteger(fromIndex). (If fromIndex is undefined, this step
262 // produces the value 0.)
263 int64_t index = 0;
264 if (!IsUndefined(*from_index, isolate)) {
265 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, from_index,
266 Object::ToInteger(isolate, from_index));
267
268 if (V8_LIKELY(IsSmi(*from_index))) {
269 int start_from = Smi::ToInt(*from_index);
270 if (start_from < 0) {
271 index = std::max<int64_t>(len + start_from, 0);
272 } else {
273 index = start_from;
274 }
275 } else {
276 DCHECK(IsHeapNumber(*from_index));
277 double start_from = Object::NumberValue(*from_index);
278 if (start_from >= len) return ReadOnlyRoots(isolate).false_value();
279 if (V8_LIKELY(std::isfinite(start_from))) {
280 if (start_from < 0) {
281 index = static_cast<int64_t>(std::max<double>(start_from + len, 0));
282 } else {
283 index = start_from;
284 }
285 }
286 }
287
288 DCHECK_GE(index, 0);
289 }
290
291 // If the receiver is not a special receiver type, and the length is a valid
292 // element index, perform fast operation tailored to specific ElementsKinds.
293 if (!IsSpecialReceiverMap(object->map()) &&
294 len <= JSObject::kMaxElementCount &&
295 JSObject::PrototypeHasNoElements(isolate, Cast<JSObject>(*object))) {
296 DirectHandle<JSObject> obj = Cast<JSObject>(object);
297 ElementsAccessor* elements = obj->GetElementsAccessor();
298 Maybe<bool> result =
299 elements->IncludesValue(isolate, obj, search_element, index, len);
300 MAYBE_RETURN(result, ReadOnlyRoots(isolate).exception());
301 return *isolate->factory()->ToBoolean(result.FromJust());
302 }
303
304 // Otherwise, perform slow lookups for special receiver types.
305 for (; index < len; ++index) {
306 HandleScope iteration_hs(isolate);
307
308 // Let elementK be the result of ? Get(O, ! ToString(k)).
309 DirectHandle<Object> element_k;
310 {
311 PropertyKey key(isolate, static_cast<double>(index));
312 LookupIterator it(isolate, object, key);
313 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, element_k,
314 Object::GetProperty(&it));
315 }
316
317 // If SameValueZero(searchElement, elementK) is true, return true.
318 if (Object::SameValueZero(*search_element, *element_k)) {
319 return ReadOnlyRoots(isolate).true_value();
320 }
321 }
322 return ReadOnlyRoots(isolate).false_value();
323}

325RUNTIME_FUNCTION(Runtime_ArrayIndexOf) {
326 HandleScope hs(isolate);
327 DCHECK_EQ(3, args.length());
328 DirectHandle<Object> search_element = args.at(1);
329 Handle<Object> from_index = args.at(2);
330
331 // Let O be ? ToObject(this value).
332 DirectHandle<JSReceiver> object;
333 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
334 isolate, object,
335 Object::ToObject(isolate, args.at(0), "Array.prototype.indexOf"));
336
337 // Let len be ? ToLength(? Get(O, "length")).
338 int64_t len;
339 {
340 if (IsJSArray(*object)) {
341 uint32_t len32 = 0;
342 bool success =
343 Object::ToArrayLength(Cast<JSArray>(*object)->length(), &len32);
344 DCHECK(success);
345 USE(success);
346 len = len32;
347 } else {
348 DirectHandle<Object> len_;
349 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
350 isolate, len_,
351 Object::GetProperty(isolate, object,
352 isolate->factory()->length_string()));
353
354 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, len_,
355 Object::ToLength(isolate, len_));
356 len = static_cast<int64_t>(Object::NumberValue(*len_));
357 DCHECK_EQ(len, Object::NumberValue(*len_));
358 }
359 }
360
361 if (len == 0) return Smi::FromInt(-1);
362
363 // Let n be ? ToInteger(fromIndex). (If fromIndex is undefined, this step
364 // produces the value 0.)
365 int64_t start_from;
366 {
367 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, from_index,
368 Object::ToInteger(isolate, from_index));
369 double fp = Object::NumberValue(*from_index);
370 if (fp > len) return Smi::FromInt(-1);
371 if (V8_LIKELY(fp >=
372 static_cast<double>(std::numeric_limits<int64_t>::min()))) {
373 DCHECK(fp < static_cast<double>(std::numeric_limits<int64_t>::max()));
374 start_from = static_cast<int64_t>(fp);
375 } else {
376 start_from = std::numeric_limits<int64_t>::min();
377 }
378 }
379
380 int64_t index;
381 if (start_from >= 0) {
382 index = start_from;
383 } else {
384 index = len + start_from;
385 if (index < 0) {
386 index = 0;
387 }
388 }
389
390 // If the receiver is not a special receiver type, and the length fits
391 // uint32_t, perform fast operation tailored to specific ElementsKinds.
392 if (!IsSpecialReceiverMap(object->map()) && len <= kMaxUInt32 &&
393 JSObject::PrototypeHasNoElements(isolate, Cast<JSObject>(*object))) {
394 DirectHandle<JSObject> obj = Cast<JSObject>(object);
395 ElementsAccessor* elements = obj->GetElementsAccessor();
396 Maybe<int64_t> result = elements->IndexOfValue(isolate, obj, search_element,
397 static_cast<uint32_t>(index),
398 static_cast<uint32_t>(len));
399 MAYBE_RETURN(result, ReadOnlyRoots(isolate).exception());
400 return *isolate->factory()->NewNumberFromInt64(result.FromJust());
401 }
402
403 // Otherwise, perform slow lookups for special receiver types.
404 for (; index < len; ++index) {
405 HandleScope iteration_hs(isolate);
406 // Let elementK be the result of ? Get(O, ! ToString(k)).
407 DirectHandle<Object> element_k;
408 {
409 PropertyKey key(isolate, static_cast<double>(index));
410 LookupIterator it(isolate, object, key);
411 Maybe<bool> present = JSReceiver::HasProperty(&it);
412 MAYBE_RETURN(present, ReadOnlyRoots(isolate).exception());
413 if (!present.FromJust()) continue;
414 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, element_k,
415 Object::GetProperty(&it));
416 if (Object::StrictEquals(*search_element, *element_k)) {
417 return *isolate->factory()->NewNumberFromInt64(index);
418 }
419 }
420 }
421 return Smi::FromInt(-1);
422}
