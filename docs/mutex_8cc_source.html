42// The YIELD_PROCESSOR macro wraps an architecture specific-instruction that
43// informs the processor we're in a busy wait, so it can handle the branch more
44// intelligently and e.g. reduce power to our core or give more resources to the
45// other hyper-thread on this core. See the following for context:
46// https://software.intel.com/en-us/articles/benefitting-power-and-performance-sleep-loops

51void SpinningMutex::AcquireSpinThenBlock() {
52 int tries = 0;
53 int backoff = 1;
54 do {
55 if (TryLock()) [[likely]] {
56 return;
57 }
58 // Note: Per the intel optimization manual
59 // (https://software.intel.com/content/dam/develop/public/us/en/documents/64-ia-32-architectures-optimization-manual.pdf),
60 // the "pause" instruction is more costly on Skylake Client than on previous
61 // architectures. The latency is found to be 141 cycles
62 // there (from ~10 on previous ones, nice 14x).
63 //
64 // According to Agner Fog's instruction tables, the latency is still >100
65 // cycles on Ice Lake, and from other sources, seems to be high as well on
66 // Alder Lake. Separately, it is (from
67 // https://agner.org/optimize/instruction_tables.pdf) also high on AMD Zen 3
68 // (~65). So just assume that it's this way for most x86_64 architectures.
69 //
70 // Also, loop several times here, following the guidelines in section 2.3.4
71 // of the manual, "Pause latency in Skylake Client Microarchitecture".
72 for (int yields = 0; yields < backoff; yields++) {
73 YIELD_PROCESSOR;
74 tries++;
75 }
76 constexpr int kMaxBackoff = 16;
77 backoff = std::min(kMaxBackoff, backoff << 1);
78 } while (tries < kSpinCount);
79
80 LockSlow();
81}

85void SpinningMutex::FutexWait() {
86 // Save and restore errno.
87 int saved_errno = errno;
88 // Don't check the return value, as we will not be awaken by a timeout, since
89 // none is specified.
90 //
91 // Ignoring the return value doesn't impact correctness, as this acts as an
92 // immediate wakeup. For completeness, the possible errors for FUTEX_WAIT are:
93 // - EACCES: state_ is not readable. Should not happen.
94 // - EAGAIN: the value is not as expected, that is not |kLockedContended|, in
95 // which case retrying the loop is the right behavior.
96 // - EINTR: signal, looping is the right behavior.
97 // - EINVAL: invalid argument.
98 //
99 // Note: not checking the return value is the approach used in bionic and
100 // glibc as well.
101 //
102 // Will return immediately if |state_| is no longer equal to
103 // |kLockedContended|. Otherwise, sleeps and wakes up when |state_| may not be
104 // |kLockedContended| anymore. Note that even without spurious wakeups, the
105 // value of |state_| is not guaranteed when this returns, as another thread
106 // may get the lock before we get to run.
107 if (syscall(SYS_futex, &state_, FUTEX_WAIT | FUTEX_PRIVATE_FLAG,
108 kLockedContended, nullptr, nullptr, 0) != 0) {
109 // These are programming error, check them.
110 DCHECK_NE(EACCES, errno);
111 DCHECK_NE(EINVAL, errno);
112 }
113 errno = saved_errno;
114}

116void SpinningMutex::FutexWake() {
117 int saved_errno = errno;
118 CHECK_NE(-1, syscall(SYS_futex, &state_, FUTEX_WAKE | FUTEX_PRIVATE_FLAG,
119 1 /* wake up a single waiter */, nullptr, nullptr, 0));
120 errno = saved_errno;
121}

123void SpinningMutex::LockSlow() {
124 // If this thread gets awaken but another one got the lock first, then go back
125 // to sleeping. See comments in |FutexWait()| to see why a loop is required.
126 while (state_.exchange(kLockedContended, std::memory_order_acquire) !=
127 kUnlocked) {
128 FutexWait();
129 }
130}

174void SpinningMutex::LockSlow() {
175 int yield_thread_count = 0;
176 do {
177 if (yield_thread_count < 10) {
178 V8_YIELD_THREAD;
179 yield_thread_count++;
180 } else {
181 // At this point, it's likely that the lock is held by a lower priority
182 // thread that is unavailable to finish its work because of higher
183 // priority threads spinning here. Sleeping should ensure that they make
184 // progress.
185 base::PlatformThread::Sleep(base::Milliseconds(1));
186 }
187 } while (!TryLock());
188}

194// Used for asserts to guarantee we are not re-locking a mutex on the same
195// thread. If this thread has only one held shared mutex (common case), we use
196// {single_held_shared_mutex}. If it has more than one we allocate a set for it.
197// Said set has to manually be constructed and destroyed.
198thread_local base::SharedMutex* single_held_shared_mutex = nullptr;
199using TSet = std::unordered_set<base::SharedMutex*>;
200thread_local TSet* held_shared_mutexes = nullptr;
201
202// Returns true iff {shared_mutex} is not a held mutex.
203bool SharedMutexNotHeld(SharedMutex* shared_mutex) {
204 DCHECK_NOT_NULL(shared_mutex);
205 return single_held_shared_mutex != shared_mutex &&
206 (!held_shared_mutexes ||
207 held_shared_mutexes->count(shared_mutex) == 0);
208}
209
210// Tries to hold {shared_mutex}. Returns true iff it hadn't been held prior to
211// this function call.
212bool TryHoldSharedMutex(SharedMutex* shared_mutex) {
213 DCHECK_NOT_NULL(shared_mutex);
214 if (single_held_shared_mutex) {
215 if (shared_mutex == single_held_shared_mutex) {
216 return false;
217 }
218 DCHECK_NULL(held_shared_mutexes);
219 held_shared_mutexes = new TSet({single_held_shared_mutex, shared_mutex});
220 single_held_shared_mutex = nullptr;
221 return true;
222 } else if (held_shared_mutexes) {
223 return held_shared_mutexes->insert(shared_mutex).second;
224 } else {
225 DCHECK_NULL(single_held_shared_mutex);
226 single_held_shared_mutex = shared_mutex;
227 return true;
228 }
229}
230
231// Tries to release {shared_mutex}. Returns true iff it had been held prior to
232// this function call.
233bool TryReleaseSharedMutex(SharedMutex* shared_mutex) {
234 DCHECK_NOT_NULL(shared_mutex);
235 if (single_held_shared_mutex == shared_mutex) {
236 single_held_shared_mutex = nullptr;
237 return true;
238 }
239 if (held_shared_mutexes && held_shared_mutexes->erase(shared_mutex)) {
240 if (held_shared_mutexes->empty()) {
241 delete held_shared_mutexes;
242 held_shared_mutexes = nullptr;
243 }
244 return true;
245 }
246 return false;
247}
248} // namespace
249#endif // DEBUG

253static V8_INLINE void InitializeNativeHandle(pthread_mutex_t* mutex) {
254 int result;
255#if defined(DEBUG)
256 // Use an error checking mutex in debug mode.
257 pthread_mutexattr_t attr;
258 result = pthread_mutexattr_init(&attr);
259 DCHECK_EQ(0, result);
260 result = pthread_mutexattr_settype(&attr, PTHREAD_MUTEX_ERRORCHECK);
261 DCHECK_EQ(0, result);
262 result = pthread_mutex_init(mutex, &attr);
263 DCHECK_EQ(0, result);
264 result = pthread_mutexattr_destroy(&attr);
265#else
266 // Use a fast mutex (default attributes).
267 result = pthread_mutex_init(mutex, nullptr);
268#endif // defined(DEBUG)
269 DCHECK_EQ(0, result);
270 USE(result);
271}
