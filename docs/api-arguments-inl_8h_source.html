79Handle<Object> FunctionCallbackArguments::CallOrConstruct(
80 Tagged<FunctionTemplateInfo> function, bool is_construct) {
81 Isolate* isolate = this->isolate();
82 RCS_SCOPE(isolate, RuntimeCallCounterId::kFunctionCallback);
83 v8::FunctionCallback f =
84 reinterpret_cast<v8::FunctionCallback>(function->callback(isolate));
85 if (isolate->should_check_side_effects() &&
86 !isolate->debug()->PerformSideEffectCheckForCallback(
87 handle(function, isolate))) {
88 return {};
89 }
90 FunctionCallbackInfo<v8::Value> info(values_, argv_, argc_);
91 ExternalCallbackScope call_scope(isolate, FUNCTION_ADDR(f),
92 is_construct ? ExceptionContext::kConstructor
93 : ExceptionContext::kOperation,
94 &info);
95 f(info);
96 return GetReturnValue<Object>(isolate);
97}

110Maybe<InterceptorResult> PropertyCallbackArguments::GetBooleanReturnValue(
111 v8::Intercepted intercepted, const char* callback_kind_for_error_message,
112 bool ignore_return_value) {
113 Isolate* isolate = this->isolate();
114 if (isolate->has_exception()) {
115 // TODO(ishell, 328490288): fix Node.js which has Setter/Definer
116 // interceptor callbacks not returning v8::Intercepted::kYes on exceptions.
117 if ((false) && DEBUG_BOOL && (intercepted == v8::Intercepted::kNo)) {
118 FATAL(
119 "Check failed: %s interceptor callback has thrown an "
120 "exception but hasn't returned v8::Intercepted::kYes.",
121 callback_kind_for_error_message);
122 }
123 return Nothing<InterceptorResult>();
124 }
125
126 if (intercepted == v8::Intercepted::kNo) {
127 // Not intercepted, there must be no side effects including exceptions.
128 DCHECK(!isolate->has_exception());
129 return Just(InterceptorResult::kNotIntercepted);
130 }
131 DCHECK_EQ(intercepted, v8::Intercepted::kYes);
132 AcceptSideEffects();
133
134 if (ignore_return_value) return Just(InterceptorResult::kTrue);
135
136 bool result = IsTrue(*GetReturnValue<Boolean>(isolate), isolate);
137
138 // TODO(ishell, 348688196): ensure callbacks comply with this and
139 // enable the check.
140 if ((false) && DEBUG_BOOL && !result && ShouldThrowOnError()) {
141 FATAL(
142 "Check failed: %s interceptor callback hasn't thrown an "
143 "exception on failure as requested.",
144 callback_kind_for_error_message);
145 }
146 return Just(result ? InterceptorResult::kTrue : InterceptorResult::kFalse);
147}

394Handle<JSObjectOrUndefined> PropertyCallbackArguments::CallPropertyEnumerator(
395 DirectHandle<InterceptorInfo> interceptor) {
396 // Named and indexed enumerator callbacks have same signatures.
397 static_assert(std::is_same<NamedPropertyEnumeratorCallback,
398 IndexedPropertyEnumeratorCallback>::value);
399 Isolate* isolate = this->isolate();
400 slot_at(kPropertyKeyIndex).store(Smi::zero()); // not relevant
401 // Enumerator callback's return value is initialized with undefined even
402 // though it's supposed to return v8::Array.
403 slot_at(kReturnValueIndex).store(ReadOnlyRoots(isolate).undefined_value());
404 // TODO(ishell): consider making it return v8::Intercepted to indicate
405 // whether the result was set or not.
406 IndexedPropertyEnumeratorCallback f =
407 v8::ToCData<IndexedPropertyEnumeratorCallback,
408 kApiIndexedPropertyEnumeratorCallbackTag>(
409 isolate, interceptor->enumerator());
410 PREPARE_CALLBACK_INFO_INTERCEPTOR(isolate, f, v8::Array, interceptor,
411 ExceptionContext::kNamedEnumerator);
412 f(callback_info);
413 Handle<JSAny> result = GetReturnValue<JSAny>(isolate);
414 DCHECK(IsUndefined(*result) || IsJSObject(*result));
415 return Cast<JSObjectOrUndefined>(result);
416}

440bool PropertyCallbackArguments::CallAccessorSetter(
441 DirectHandle<AccessorInfo> accessor_info, DirectHandle<Name> name,
442 DirectHandle<Object> value) {
443 Isolate* isolate = this->isolate();
444 RCS_SCOPE(isolate, RuntimeCallCounterId::kAccessorSetterCallback);
445 // Unlike interceptor callbacks we know that the property exists, so
446 // the callback is allowed to have side effects.
447 AcceptSideEffects();
448
449 slot_at(kPropertyKeyIndex).store(*name);
450 slot_at(kReturnValueIndex).store(ReadOnlyRoots(isolate).true_value());
451 // The actual type of setter callback is either
452 // v8::AccessorNameSetterCallback or
453 // i::Accesors::AccessorNameBooleanSetterCallback, depending on whether the
454 // AccessorInfo was created by the API or internally (see accessors.cc).
455 // Here we handle both cases using the AccessorNameSetterCallback signature
456 // and checking whether the returned result is set to default value
457 // (the undefined value).
458 // TODO(ishell, 348660658): update V8 Api to allow setter callbacks provide
459 // the result of [[Set]] operation according to JavaScript semantics.
460 AccessorNameSetterCallback f = reinterpret_cast<AccessorNameSetterCallback>(
461 accessor_info->setter(isolate));
462 PREPARE_CALLBACK_INFO_ACCESSOR(
463 isolate, f, void, accessor_info, direct_handle(receiver(), isolate),
464 ACCESSOR_SETTER, ExceptionContext::kAttributeSet);
465 f(v8::Utils::ToLocal(name), v8::Utils::ToLocal(value), callback_info);
466 // Historically, in case of v8::AccessorNameSetterCallback it wasn't allowed
467 // to set the result and not setting the result was treated as successful
468 // execution.
469 // During interceptors Api refactoring it was temporarily allowed to call
470 // v8::ReturnValue<void>::Set[NonEmpty](Local<S>) and the result was just
471 // converted to v8::Boolean which was then treated as a result of [[Set]].
472 // In case of AccessorNameBooleanSetterCallback, the result is always
473 // set to v8::Boolean or an exception is be thrown (in which case the
474 // result is ignored anyway). So, regardless of whether the signature was
475 // v8::AccessorNameSetterCallback or AccessorNameBooleanSetterCallback
476 // the result is guaranteed to be v8::Boolean value indicating success or
477 // failure.
478 DirectHandle<Boolean> result = GetReturnValue<Boolean>(isolate);
479 return IsTrue(*result, isolate);
480}
