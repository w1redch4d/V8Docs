22// Represents a tree of inlining decisions.
23// A node in the tree represents a function frame, and `function_calls_`
24// represent all direct/call_ref/call_indirect function calls in this frame.
25// Each element of `function_calls_` is itself a `Vector` of `InliningTree`s,
26// corresponding to the different speculative candidates for a
27// call_ref/call_indirect; for a direct call, it has a single element.
28// If a transitive element of `function_calls_` has its `is_inlined_` field set,
29// it should be inlined into the caller.
30// We have this additional datastructure for Turboshaft, since nodes in the
31// Turboshaft IR aren't easily expanded incrementally, so all the inlining
32// decisions are already made before graph building on this abstracted form of
33// the code.
34class InliningTree : public ZoneObject {
35 private:
36 struct Data;

41 static InliningTree* CreateRoot(Zone* zone, const WasmModule* module,
42 uint32_t function_index) {
43 InliningTree* tree = zone->New<InliningTree>(
44 zone->New<Data>(zone, module, function_index), function_index,
45 0, // Call count.
46 0, // Wire byte size. `0` causes the root node to always get
47 // expanded, regardless of budget.
48 -1, -1, -1, // Caller, feedback slot, case.
49 0 // Inlining depth.
50 );
51 tree->FullyExpand();
52 return tree;
53 }

55 // This should stay roughly in sync with the full logic below, but not rely
56 // on having observed any call counts. Since it therefore can't simulate
57 // regular behavior accurately anyway, it may be a very coarse approximation.
58 static int NoLiftoffBudget(const WasmModule* module, uint32_t func_index) {
59 size_t wirebytes = module->functions[func_index].code.length();
60 double scaled = BudgetScaleFactor(module);
61 // TODO(jkummerow): When TF is gone, remove this adjustment by folding
62 // it into the flag's default value.
63 constexpr int kTurboshaftAdjustment = 2;
64 int high_growth =
65 static_cast<int>(v8_flags.wasm_inlining_factor) + kTurboshaftAdjustment;
66 constexpr int kLowestUsefulValue = 2;
67 int low_growth = std::max(kLowestUsefulValue, high_growth - 3);
68 double max_growth_factor = low_growth * (1 - scaled) + high_growth * scaled;
69 return std::max(static_cast<int>(v8_flags.wasm_inlining_min_budget),
70 static_cast<int>(max_growth_factor * wirebytes));
71 }

73 int64_t score() const {
74 // Note that the zero-point is arbitrary. Functions with negative score
75 // can still get inlined.
76 constexpr int count_factor = 2;
77 constexpr int size_factor = 3;
78 return int64_t{call_count_} * count_factor -
79 int64_t{wire_byte_size_} * size_factor;
80 }

99 static double BudgetScaleFactor(const WasmModule* module) {
100 // If there are few small functions, that indicates that the toolchain
101 // already performed significant inlining, so we reduce the budget
102 // significantly as further inlining has diminishing benefits.
103 // For both major knobs, we apply a smoothened step function based on
104 // the module's percentage of small functions (sfp):
105 // sfp <= 25%: use "low" budget
106 // sfp >= 50%: use "high" budget
107 // 25% < sfp < 50%: interpolate linearly between both budgets.
108 double small_function_percentage =
109 module->num_small_functions * 100.0 / module->num_declared_functions;
110 if (small_function_percentage <= 25) {
111 return 0;
112 } else if (small_function_percentage >= 50) {
113 return 1;
114 } else {
115 return (small_function_percentage - 25) / 25;
116 }
117 }

119 struct Data {
120 Data(Zone* zone, const WasmModule* module, uint32_t topmost_caller_index)
121 : zone(zone),
122 module(module),
123 topmost_caller_index(topmost_caller_index) {
124 double scaled = BudgetScaleFactor(module);
125 // We found experimentally that we need to allow a larger growth factor
126 // for Turboshaft to achieve similar inlining decisions as in Turbofan;
127 // presumably because some functions that have a small wire size of their
128 // own still need to be allowed to inline some callees.
129 // TODO(jkummerow): When TF is gone, remove this adjustment by folding
130 // it into the flag's default value.
131 constexpr int kTurboshaftAdjustment = 2;
132 int high_growth = static_cast<int>(v8_flags.wasm_inlining_factor) +
133 kTurboshaftAdjustment;
134 // A value of 1 would be equivalent to disabling inlining entirely.
135 constexpr int kLowestUsefulValue = 2;
136 int low_growth = std::max(kLowestUsefulValue, high_growth - 3);
137 max_growth_factor = low_growth * (1 - scaled) + high_growth * scaled;
138 // The {wasm_inlining_budget} value has been tuned for Turbofan node
139 // counts. Turboshaft looks at wire bytes instead, and on average there
140 // are about 0.74 TF nodes per wire byte, so we apply a small factor to
141 // account for the difference, so we get similar inlining decisions in
142 // both compilers.
143 // TODO(jkummerow): When TF is gone, remove this factor by folding it
144 // into the flag's default value.
145 constexpr double kTurboshaftCorrectionFactor = 1.4;
146 double high_cap =
147 v8_flags.wasm_inlining_budget * kTurboshaftCorrectionFactor;
148 double low_cap = high_cap / 10;
149 budget_cap = low_cap * (1 - scaled) + high_cap * scaled;
150 }

152 Zone* zone;
153 const WasmModule* module;
154 double max_growth_factor;
155 size_t budget_cap;
156 uint32_t topmost_caller_index;
157 };

171 // Recursively expand the tree by expanding this node and children nodes etc.
172 // Nodes are prioritized by their `score`. Expansion continues until
173 // `kMaxInlinedCount` nodes are expanded or `budget` (in wire-bytes size) is

207void InliningTree::Inline() {
208 is_inlined_ = true;
209 auto feedback =
210 data_->module->type_feedback.feedback_for_function.find(function_index_);
211 if (feedback != data_->module->type_feedback.feedback_for_function.end() &&
212 feedback->second.feedback_vector.size() ==
213 feedback->second.call_targets.size()) {
214 std::vector<CallSiteFeedback>& type_feedback =
215 feedback->second.feedback_vector;
216 feedback_found_ = true;
217 function_calls_ =
218 data_->zone->AllocateVector<CasesPerCallSite>(type_feedback.size());
219 has_non_inlineable_targets_ =
220 data_->zone->AllocateVector<bool>(type_feedback.size());
221 for (size_t i = 0; i < type_feedback.size(); i++) {
222 function_calls_[i] = data_->zone->AllocateVector<InliningTree*>(
223 type_feedback[i].num_cases());
224 has_non_inlineable_targets_[i] =
225 type_feedback[i].has_non_inlineable_targets();
226 for (int the_case = 0; the_case < type_feedback[i].num_cases();
227 the_case++) {
228 uint32_t callee_index = type_feedback[i].function_index(the_case);
229 // TODO(jkummerow): Experiment with propagating relative call counts
230 // into the nested InliningTree, and weighting scores there accordingly.
231 function_calls_[i][the_case] = data_->zone->New<InliningTree>(
232 data_, callee_index, type_feedback[i].call_count(the_case),
233 data_->module->functions[callee_index].code.length(),
234 function_index_, static_cast<int>(i), the_case, depth_ + 1);
235 }
236 }
237 }

240struct TreeNodeOrdering {
241 bool operator()(InliningTree* t1, InliningTree* t2) {
242 // Prefer callees with a higher score, and if the scores are equal,
243 // those with a lower function index (to make the queue ordering strict).
244 return std::make_pair(t1->score(), t2->function_index()) <
245 std::make_pair(t2->score(), t1->function_index());
246 }
247};

249void InliningTree::FullyExpand() {
250 DCHECK_EQ(this->function_index_, data_->topmost_caller_index);
251 size_t initial_wire_byte_size =
252 data_->module->functions[function_index_].code.length();
253 size_t inlined_wire_byte_count = 0;
254 std::priority_queue<InliningTree*, std::vector<InliningTree*>,
255 TreeNodeOrdering>
256 queue;
257 queue.push(this);
258 int inlined_count = 0;
259 base::SharedMutexGuard<base::kShared> mutex_guard(
260 &data_->module->type_feedback.mutex);
261 while (!queue.empty() && inlined_count < kMaxInlinedCount) {
262 InliningTree* top = queue.top();
263 if (v8_flags.trace_wasm_inlining) {
264 if (top != this) {
265 PrintF(
266 "[function %d: in function %d, considering call #%d, case #%d, to "
267 "function %d (count=%d, size=%d, score=%lld)... ",
268 data_->topmost_caller_index, top->caller_index_,
269 top->feedback_slot_, static_cast<int>(top->case_),
270 static_cast<int>(top->function_index_), top->call_count_,
271 top->wire_byte_size_, static_cast<long long>(top->score()));
272 } else {
273 PrintF("[function %d: expanding topmost caller... ",
274 data_->topmost_caller_index);
275 }
276 }
277 queue.pop();
278 if (top->function_index_ < data_->module->num_imported_functions) {
279 if (v8_flags.trace_wasm_inlining && top != this) {
280 PrintF("imported function]\n");
281 }
282 continue;
283 }
284 if (is_asmjs_module(data_->module)) {
285 if (v8_flags.trace_wasm_inlining) {
286 PrintF("cannot inline asm.js function]\n");
287 }
288 continue;
289 }
290
291 // Key idea: inlining hot calls is good, inlining big functions is bad,
292 // so inline when a candidate is "hotter than it is big". Exception:
293 // tiny candidates can get inlined regardless of their call count.
294 if (top != this && top->wire_byte_size_ >= 12 &&
295 !v8_flags.wasm_inlining_ignore_call_counts) {
296 if (top->call_count_ < top->wire_byte_size_ / 2) {
297 if (v8_flags.trace_wasm_inlining) {
298 PrintF("not called often enough]\n");
299 }
300 continue;
301 }
302 }
303
304 if (!top->SmallEnoughToInline(initial_wire_byte_size,
305 inlined_wire_byte_count)) {
306 if (v8_flags.trace_wasm_inlining && top != this) {
307 PrintF("not enough inlining budget]\n");
308 }
309 continue;
310 }
311 if (v8_flags.trace_wasm_inlining && top != this) {
312 PrintF("decided to inline! ");
313 }
314 top->Inline();
315 inlined_count++;
316 // For tiny functions, inlining may actually decrease generated code size
317 // because we have one less call and don't need to push arguments, etc.
318 // Subtract a little bit from the code size increase, such that inlining
319 // these tiny functions doesn't use up any of the budget.
320 constexpr int kOneLessCall = 6; // Guesstimated savings per call.
321 inlined_wire_byte_count += std::max(top->wire_byte_size_ - kOneLessCall, 0);
322
323 if (top->feedback_found()) {
324 if (top->depth_ < kMaxInliningNestingDepth) {
325 if (v8_flags.trace_wasm_inlining) PrintF("queueing callees]\n");
326 for (CasesPerCallSite cases : top->function_calls_) {
327 for (InliningTree* call : cases) {
328 if (call != nullptr) {
329 queue.push(call);
330 }
331 }
332 }
333 } else if (v8_flags.trace_wasm_inlining) {
334 PrintF("max inlining depth reached]\n");
335 }
336 } else {
337 if (v8_flags.trace_wasm_inlining) PrintF("feedback not found]\n");
338 }
339 }
340 if (v8_flags.trace_wasm_inlining && !queue.empty()) {
341 PrintF("[function %d: too many inlining candidates, stopping...]\n",
342 data_->topmost_caller_index);
343 }
344}

346// Returns true if there is still enough budget left to inline the current
347// candidate given the initial graph size and the already inlined wire bytes.
348bool InliningTree::SmallEnoughToInline(size_t initial_wire_byte_size,
349 size_t inlined_wire_byte_count) {
350 if (wire_byte_size_ > static_cast<int>(v8_flags.wasm_inlining_max_size)) {
351 return false;
352 }
353 // For tiny functions, let's be a bit more generous.
354 // TODO(dlehmann): Since we don't use up budget (i.e., increase
355 // `inlined_wire_byte_count` see above) for very tiny functions, we might be
356 // able to remove/simplify this code in the future.
357 if (wire_byte_size_ < 12) {
358 if (inlined_wire_byte_count > 100) {
359 inlined_wire_byte_count -= 100;
360 } else {
361 inlined_wire_byte_count = 0;
362 }
363 }
364 // For small-ish functions, the inlining budget is defined by the larger of
365 // 1) the wasm_inlining_min_budget and
366 // 2) the max_growth_factor * initial_wire_byte_size.
367 // Inlining a little bit should always be fine even for tiny functions (1),
368 // otherwise (2) makes sure that the budget scales in relation with the
369 // original function size, to limit the compile time increase caused by
370 // inlining.
371 size_t budget_small_function =
372 std::max<size_t>(v8_flags.wasm_inlining_min_budget,
373 data_->max_growth_factor * initial_wire_byte_size);
374
375 // For large functions, growing by the same factor would add too much
376 // compilation effort, so we also apply a fixed cap. However, independent
377 // of the budget cap, for large functions we should still allow a little
378 // inlining, which is why we allow 10% of the graph size is the minimal
379 // budget even for large functions that exceed the regular budget.
380 //
381 // Note for future tuning: it might make sense to allow 20% here, and in
382 // turn perhaps lower --wasm-inlining-budget. The drawback is that this
383 // would allow truly huge functions to grow even bigger; the benefit is
384 // that we wouldn't fall off as steep a cliff when hitting the cap.
385 size_t budget_large_function =
386 std::max<size_t>(data_->budget_cap, initial_wire_byte_size * 1.1);
387 size_t total_size = initial_wire_byte_size + inlined_wire_byte_count +
388 static_cast<size_t>(wire_byte_size_);
389 if (v8_flags.trace_wasm_inlining) {
390 PrintF("budget=min(%zu, %zu), size %zu->%zu ", budget_small_function,
391 budget_large_function,
392 (initial_wire_byte_size + inlined_wire_byte_count), total_size);
393 }
394 return total_size <
395 std::min<size_t>(budget_small_function, budget_large_function);
396}
