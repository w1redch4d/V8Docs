27 OpIndex REDUCE(FastApiCall)(
28 V<FrameState> frame_state, V<Object> data_argument, V<Context> context,
29 base::Vector<const OpIndex> arguments,
30 const FastApiCallParameters* parameters,
31 base::Vector<const RegisterRepresentation> out_reps) {
32 FastApiCallFunction c_function = parameters->c_function;
33 const auto& c_signature = parameters->c_signature();
34 const int c_arg_count = c_signature->ArgumentCount();
35 DCHECK_EQ(c_arg_count, arguments.size());
36
37 Label<> handle_error(this);
38 Label<Word32> done(this);
39 Variable result = __ NewVariable(RegisterRepresentation::FromCTypeInfo(
40 c_signature->ReturnInfo(), c_signature->GetInt64Representation()));
41
42 OpIndex callee = __ ExternalConstant(ExternalReference::Create(
43 c_function.address, ExternalReference::FAST_C_CALL));
44
45 base::SmallVector<OpIndex, 16> args;
46 for (int i = 0; i < c_arg_count; ++i) {
47 CTypeInfo type = c_signature->ArgumentInfo(i);
48 args.push_back(AdaptFastCallArgument(arguments[i], type, handle_error));
49 }
50
51 // While adapting the arguments, we might have noticed an inconsistency that
52 // lead to unconditionally jumping to {handle_error}. If this happens, then
53 // we don't emit the call.
54 if (V8_LIKELY(!__ generating_unreachable_operations())) {
55 MachineSignature::Builder builder(
56 __ graph_zone(), 1,
57 c_arg_count + (c_signature->HasOptions() ? 1 : 0));
58
59 builder.AddReturn(MachineType::TypeForCType(c_signature->ReturnInfo()));
60
61 for (int i = 0; i < c_arg_count; ++i) {
62 CTypeInfo type = c_signature->ArgumentInfo(i);
63 MachineType machine_type =
64 type.GetSequenceType() == CTypeInfo::SequenceType::kScalar
65 ? MachineType::TypeForCType(type)
66 : MachineType::AnyTagged();
67 builder.AddParam(machine_type);
68 }
69
70 OpIndex stack_slot;
71 if (c_signature->HasOptions()) {
72 const int kAlign = alignof(v8::FastApiCallbackOptions);
73 const int kSize = sizeof(v8::FastApiCallbackOptions);
74 // If this check fails, you've probably added new fields to
75 // v8::FastApiCallbackOptions, which means you'll need to write code
76 // that initializes and reads from them too.
77 static_assert(kSize == sizeof(uintptr_t) * 2);
78 stack_slot = __ StackSlot(kSize, kAlign);
79
80 // isolate
81 __ StoreOffHeap(
82 stack_slot,
83 __ ExternalConstant(ExternalReference::isolate_address()),
84 MemoryRepresentation::UintPtr(),
85 offsetof(v8::FastApiCallbackOptions, isolate));
86 // data = data_argument
87 OpIndex data_argument_to_pass = __ AdaptLocalArgument(data_argument);
88 __ StoreOffHeap(stack_slot, data_argument_to_pass,
89 MemoryRepresentation::UintPtr(),
90 offsetof(v8::FastApiCallbackOptions, data));
91
92 args.push_back(stack_slot);
93 builder.AddParam(MachineType::Pointer());
94 }
95
96 // Build the actual call.
97 const TSCallDescriptor* call_descriptor = TSCallDescriptor::Create(
98 Linkage::GetSimplifiedCDescriptor(__ graph_zone(), builder.Get(),
99 CallDescriptor::kNeedsFrameState),
100 CanThrow::kNo, LazyDeoptOnThrow::kNo, __ graph_zone());
101 OpIndex c_call_result = WrapFastCall(call_descriptor, callee, frame_state,
102 context, base::VectorOf(args));
103
104 Label<> trigger_exception(this);
105
106 V<Object> exception =
107 __ Load(__ ExternalConstant(ExternalReference::Create(
108 IsolateAddressId::kExceptionAddress, isolate_)),
109 LoadOp::Kind::RawAligned(), MemoryRepresentation::UintPtr());
110 GOTO_IF_NOT(LIKELY(__ TaggedEqual(
111 exception,
112 __ HeapConstant(isolate_->factory()->the_hole_value()))),
113 trigger_exception);
114
115 V<Any> fast_call_result = ConvertReturnValue(c_signature, c_call_result);
116 __ SetVariable(result, fast_call_result);
117
118 GOTO(done, FastApiCallOp::kSuccessValue);
119 BIND(trigger_exception);
120 __ template CallRuntime<
121 typename RuntimeCallDescriptor::PropagateException>(
122 isolate_, frame_state, __ NoContextConstant(), LazyDeoptOnThrow::kNo,
123 {});
124
125 __ Unreachable();
126 }
127
128 if (BIND(handle_error)) {
129 __ SetVariable(result, DefaultReturnValue(c_signature));
130 // We pass Tagged<Smi>(0) as the value here, although this should never be
131 // visible when calling code reacts to `kFailureValue` properly.
132 GOTO(done, FastApiCallOp::kFailureValue);
133 }
134
135 BIND(done, state);
136 return __ Tuple(state, __ GetVariable(result));
137 }

141 V<T> Checked(V<Tuple<T, Word32>> result, Label<>& otherwise) {
142 V<Word32> result_state = __ template Projection<1>(result);
143 GOTO_IF_NOT(__ Word32Equal(result_state, TryChangeOp::kSuccessValue),
144 otherwise);
145 return __ template Projection<0>(result);
146 }

148 OpIndex AdaptFastCallArgument(OpIndex argument, CTypeInfo arg_type,
149 Label<>& handle_error) {
150 switch (arg_type.GetSequenceType()) {
151 case CTypeInfo::SequenceType::kScalar: {
152 uint8_t flags = static_cast<uint8_t>(arg_type.GetFlags());
153 if (flags & static_cast<uint8_t>(CTypeInfo::Flags::kEnforceRangeBit)) {
154 switch (arg_type.GetType()) {
155 case CTypeInfo::Type::kInt32: {
156 auto result = __ TryTruncateFloat64ToInt32(argument);
157 return Checked(result, handle_error);
158 }
159 case CTypeInfo::Type::kUint32: {
160 auto result = __ TryTruncateFloat64ToUint32(argument);
161 return Checked(result, handle_error);
162 }
163 case CTypeInfo::Type::kInt64: {
164 auto result = __ TryTruncateFloat64ToInt64(argument);
165 return Checked(result, handle_error);
166 }
167 case CTypeInfo::Type::kUint64: {
168 auto result = __ TryTruncateFloat64ToUint64(argument);
169 return Checked(result, handle_error);
170 }
171 default: {
172 GOTO(handle_error);
173 return argument;
174 }
175 }
176 } else if (flags & static_cast<uint8_t>(CTypeInfo::Flags::kClampBit)) {
177 return ClampFastCallArgument(argument, arg_type.GetType());
178 } else {
179 switch (arg_type.GetType()) {
180 case CTypeInfo::Type::kV8Value: {
181 return __ AdaptLocalArgument(argument);
182 }
183 case CTypeInfo::Type::kFloat32: {
184 return __ TruncateFloat64ToFloat32(argument);
185 }
186 case CTypeInfo::Type::kPointer: {
187 // Check that the value is a HeapObject.
188 GOTO_IF(__ ObjectIsSmi(argument), handle_error);
189 Label<WordPtr> done(this);
190
191 // Check if the value is null.
192 GOTO_IF(UNLIKELY(__ TaggedEqual(
193 argument, __ HeapConstant(factory_->null_value()))),
194 done, 0);
195
196 // Check that the value is a JSExternalObject.
197 GOTO_IF_NOT(
198 __ TaggedEqual(__ LoadMapField(argument),
199 __ HeapConstant(factory_->external_map())),
200 handle_error);
201
202 GOTO(done, __ template LoadField<WordPtr>(
203 V<HeapObject>::Cast(argument),
204 AccessBuilder::ForJSExternalObjectValue()));
205
206 BIND(done, result);
207 return result;
208 }
209 case CTypeInfo::Type::kSeqOneByteString: {
210 // Check that the value is a HeapObject.
211 GOTO_IF(__ ObjectIsSmi(argument), handle_error);
212 V<HeapObject> argument_obj = V<HeapObject>::Cast(argument);
213
214 V<Map> map = __ LoadMapField(argument_obj);
215 V<Word32> instance_type = __ LoadInstanceTypeField(map);
216
217 V<Word32> encoding = __ Word32BitwiseAnd(
218 instance_type, kStringRepresentationAndEncodingMask);
219 GOTO_IF_NOT(__ Word32Equal(encoding, kSeqOneByteStringTag),
220 handle_error);
221
222 V<WordPtr> length_in_bytes = __ template LoadField<WordPtr>(
223 argument_obj, AccessBuilder::ForStringLength());
224 V<WordPtr> data_ptr = __ GetElementStartPointer(
225 argument_obj, AccessBuilder::ForSeqOneByteStringCharacter());
226
227 constexpr int kAlign = alignof(FastOneByteString);
228 constexpr int kSize = sizeof(FastOneByteString);
229 static_assert(kSize == sizeof(uintptr_t) + sizeof(size_t),
230 "The size of "
231 "FastOneByteString isn't equal to the sum of its "
232 "expected members.");
233 OpIndex stack_slot = __ StackSlot(kSize, kAlign);
234 __ StoreOffHeap(stack_slot, data_ptr,
235 MemoryRepresentation::UintPtr());
236 __ StoreOffHeap(stack_slot, length_in_bytes,
237 MemoryRepresentation::Uint32(), sizeof(size_t));
238 static_assert(sizeof(uintptr_t) == sizeof(size_t),
239 "The string length can't "
240 "fit the PointerRepresentation used to store it.");
241 return stack_slot;
242 }
243 default: {
244 return argument;
245 }
246 }
247 }
248 }
249 case CTypeInfo::SequenceType::kIsSequence: {
250 CHECK_EQ(arg_type.GetType(), CTypeInfo::Type::kVoid);
251
252 // Check that the value is a HeapObject.
253 GOTO_IF(__ ObjectIsSmi(argument), handle_error);
254
255 // Check that the value is a JSArray.
256 V<Map> map = __ LoadMapField(argument);
257 V<Word32> instance_type = __ LoadInstanceTypeField(map);
258 GOTO_IF_NOT(__ Word32Equal(instance_type, JS_ARRAY_TYPE), handle_error);
259
260 return __ AdaptLocalArgument(argument);
261 }
262 START_ALLOW_USE_DEPRECATED()
263 case CTypeInfo::SequenceType::kIsTypedArray:
264 UNREACHABLE();
265 END_ALLOW_USE_DEPRECATED()
266 default: {
267 UNREACHABLE();
268 }
269 }

272 OpIndex ClampFastCallArgument(V<Float64> argument,
273 CTypeInfo::Type scalar_type) {
274 double min, max;
275 switch (scalar_type) {
276 case CTypeInfo::Type::kInt32:
277 min = std::numeric_limits<int32_t>::min();
278 max = std::numeric_limits<int32_t>::max();
279 break;
280 case CTypeInfo::Type::kUint32:
281 min = 0;
282 max = std::numeric_limits<uint32_t>::max();
283 break;
284 case CTypeInfo::Type::kInt64:
285 min = kMinSafeInteger;
286 max = kMaxSafeInteger;
287 break;
288 case CTypeInfo::Type::kUint64:
289 min = 0;
290 max = kMaxSafeInteger;
291 break;
292 default:
293 UNREACHABLE();
294 }
295
296 V<Float64> clamped =
297 __ Conditional(__ Float64LessThan(min, argument),
298 __ Conditional(__ Float64LessThan(argument, max),
299 argument, __ Float64Constant(max)),
300 __ Float64Constant(min));
301
302 Label<Float64> done(this);
303 V<Float64> rounded = __ Float64RoundTiesEven(clamped);
304 GOTO_IF(__ Float64IsNaN(rounded), done, 0.0);
305 GOTO(done, rounded);
306
307 BIND(done, rounded_result);
308 switch (scalar_type) {
309 case CTypeInfo::Type::kInt32:
310 return __ ReversibleFloat64ToInt32(rounded_result);
311 case CTypeInfo::Type::kUint32:
312 return __ ReversibleFloat64ToUint32(rounded_result);
313 case CTypeInfo::Type::kInt64:
314 return __ ReversibleFloat64ToInt64(rounded_result);
315 case CTypeInfo::Type::kUint64:
316 return __ ReversibleFloat64ToUint64(rounded_result);
317 default:
318 UNREACHABLE();
319 }
320 }

322 V<Any> DefaultReturnValue(const CFunctionInfo* c_signature) {
323 switch (c_signature->ReturnInfo().GetType()) {
324 case CTypeInfo::Type::kVoid:
325 return __ HeapConstant(factory_->undefined_value());
326 case CTypeInfo::Type::kBool:
327 case CTypeInfo::Type::kInt32:
328 case CTypeInfo::Type::kUint32:
329 return __ Word32Constant(0);
330 case CTypeInfo::Type::kInt64:
331 case CTypeInfo::Type::kUint64: {
332 CFunctionInfo::Int64Representation repr =
333 c_signature->GetInt64Representation();
334 if (repr == CFunctionInfo::Int64Representation::kBigInt) {
335 return __ Word64Constant(int64_t{0});
336 }
337 DCHECK_EQ(repr, CFunctionInfo::Int64Representation::kNumber);
338 return __ Float64Constant(0);
339 }
340 case CTypeInfo::Type::kFloat32:
341 return __ Float32Constant(0);
342 case CTypeInfo::Type::kFloat64:
343 return __ Float64Constant(0);
344 case CTypeInfo::Type::kPointer:
345 return __ HeapConstant(factory_->undefined_value());
346 case CTypeInfo::Type::kAny:
347 case CTypeInfo::Type::kSeqOneByteString:
348 case CTypeInfo::Type::kV8Value:
349 case CTypeInfo::Type::kApiObject:
350 case CTypeInfo::Type::kUint8:

355 V<Any> ConvertReturnValue(const CFunctionInfo* c_signature, OpIndex result) {
356 switch (c_signature->ReturnInfo().GetType()) {
357 case CTypeInfo::Type::kVoid:
358 return __ HeapConstant(factory_->undefined_value());
359 case CTypeInfo::Type::kBool:
360 static_assert(sizeof(bool) == 1, "unsupported bool size");
361 return __ Word32BitwiseAnd(result, __ Word32Constant(0xFF));
362 case CTypeInfo::Type::kInt32:
363 case CTypeInfo::Type::kUint32:
364 case CTypeInfo::Type::kFloat32:
365 case CTypeInfo::Type::kFloat64:
366 return result;
367 case CTypeInfo::Type::kInt64: {
368 CFunctionInfo::Int64Representation repr =
369 c_signature->GetInt64Representation();
370 if (repr == CFunctionInfo::Int64Representation::kBigInt) {
371 return result;
372 }
373 DCHECK_EQ(repr, CFunctionInfo::Int64Representation::kNumber);
374 return __ ChangeInt64ToFloat64(result);
375 }
376 case CTypeInfo::Type::kUint64: {
377 CFunctionInfo::Int64Representation repr =
378 c_signature->GetInt64Representation();
379 if (repr == CFunctionInfo::Int64Representation::kBigInt) {
380 return result;
381 }
382 DCHECK_EQ(repr, CFunctionInfo::Int64Representation::kNumber);
383 return __ ChangeUint64ToFloat64(result);
384 }
385
386 case CTypeInfo::Type::kPointer:
387 return BuildAllocateJSExternalObject(result);
388 case CTypeInfo::Type::kAny:
389 case CTypeInfo::Type::kSeqOneByteString:
390 case CTypeInfo::Type::kV8Value:
391 case CTypeInfo::Type::kApiObject:
392 case CTypeInfo::Type::kUint8:
393 UNREACHABLE();
394 }
395 }

397 V<HeapObject> BuildAllocateJSExternalObject(V<WordPtr> pointer) {
398 Label<HeapObject> done(this);
399
400 // Check if the pointer is a null pointer.
401 GOTO_IF(__ WordPtrEqual(pointer, 0), done,
402 __ HeapConstant(factory_->null_value()));
403
404 Uninitialized<HeapObject> external =
405 __ Allocate(JSExternalObject::kHeaderSize, AllocationType::kYoung);
406 __ InitializeField(external, AccessBuilder::ForMap(),
407 __ HeapConstant(factory_->external_map()));
408 V<FixedArray> empty_fixed_array =
409 __ HeapConstant(factory_->empty_fixed_array());
410 __ InitializeField(external, AccessBuilder::ForJSObjectPropertiesOrHash(),
411 empty_fixed_array);
412 __ InitializeField(external, AccessBuilder::ForJSObjectElements(),
413 empty_fixed_array);
414
415#ifdef V8_ENABLE_SANDBOX
416 OpIndex isolate_ptr =
417 __ ExternalConstant(ExternalReference::isolate_address());
418 MachineSignature::Builder builder(__ graph_zone(), 1, 2);
419 builder.AddReturn(MachineType::Uint32());
420 builder.AddParam(MachineType::Pointer());
421 builder.AddParam(MachineType::Pointer());
422 OpIndex allocate_and_initialize_young_external_pointer_table_entry =
423 __ ExternalConstant(
424 ExternalReference::
425 allocate_and_initialize_young_external_pointer_table_entry());
426 auto call_descriptor =
427 Linkage::GetSimplifiedCDescriptor(__ graph_zone(), builder.Get());
428 OpIndex handle = __ Call(
429 allocate_and_initialize_young_external_pointer_table_entry,
430 {isolate_ptr, pointer},
431 TSCallDescriptor::Create(call_descriptor, CanThrow::kNo,
432 LazyDeoptOnThrow::kNo, __ graph_zone()));
433 __ InitializeField(
434 external, AccessBuilder::ForJSExternalObjectPointerHandle(), handle);
435#else
436 __ InitializeField(external, AccessBuilder::ForJSExternalObjectValue(),
437 pointer);
438#endif // V8_ENABLE_SANDBOX
439 GOTO(done, __ FinishInitialization(std::move(external)));
440
441 BIND(done, result);
442 return result;
443 }

445 OpIndex WrapFastCall(const TSCallDescriptor* descriptor, OpIndex callee,
446 V<FrameState> frame_state, V<Context> context,
447 base::Vector<const OpIndex> arguments) {
448 // CPU profiler support.
449 OpIndex target_address =
450 __ IsolateField(IsolateFieldId::kFastApiCallTarget);
451 __ StoreOffHeap(target_address, __ BitcastHeapObjectToWordPtr(callee),
452 MemoryRepresentation::UintPtr());
453
454 OpIndex context_address = __ ExternalConstant(
455 ExternalReference::Create(IsolateAddressId::kContextAddress, isolate_));
456
457 __ StoreOffHeap(context_address, __ BitcastHeapObjectToWordPtr(context),
458 MemoryRepresentation::UintPtr());
459
460 // Create the fast call.
461 OpIndex result = __ Call(callee, frame_state, arguments, descriptor);
462
463 // Reset the CPU profiler target address.
464 __ StoreOffHeap(target_address, __ IntPtrConstant(0),
465 MemoryRepresentation::UintPtr());
466
467#if DEBUG
468 // Reset the context again after the call, to make sure nobody is using the
469 // leftover context in the isolate.
470 __ StoreOffHeap(context_address,
471 __ WordPtrConstant(Context::kInvalidContext),
472 MemoryRepresentation::UintPtr());
473#endif
474
475 return result;
476 }
