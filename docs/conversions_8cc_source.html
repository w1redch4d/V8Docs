41// Helper class for building result strings in a character buffer. The
42// purpose of the class is to use safe operations that checks the
43// buffer bounds on all operations in debug mode.
44// This simple base class does not allow formatted output.
45class SimpleStringBuilder final {
46 public:
47 // Create a string builder with a buffer of the given size. The
48 // buffer is allocated through NewArray<char> and must be
49 // deallocated by the caller of Finalize().
50 explicit SimpleStringBuilder(int size) {
51 buffer_ = base::Vector<char>::New(size);
52 cursor_ = buffer_.begin();
53 }
54
55 SimpleStringBuilder(char* buffer, int size)
56 : buffer_(buffer, size), cursor_(buffer) {}
57
58 ~SimpleStringBuilder() {
59 if (V8_UNLIKELY(!is_finalized())) Finalize();
60 }
61
62 // Get the current position in the builder.
63 int position() const {
64 DCHECK(!is_finalized());
65 size_t position = cursor_ - buffer_.begin();
66 DCHECK_GE(kMaxInt, position);
67 return static_cast<int>(position);
68 }
69
70 // Add a single character to the builder. It is not allowed to add
71 // 0-characters; use the Finalize() method to terminate the string
72 // instead.
73 V8_INLINE void AddCharacter(char c) {
74 DCHECK_NE(c, '\0');
75 DCHECK(!is_finalized());
76 DCHECK_LT(position(), buffer_.length());
77 *cursor_++ = c;
78 }
79
80 // Add an entire string to the builder. 'len' must be equal to strlen().
81 V8_INLINE void AddString(const char* s, int len) {
82 DCHECK_EQ(static_cast<size_t>(len), strlen(s));
83 AddSubstring(s, len);
84 }
85
86 // Add a string literal to the builder.
87 template <int N>
88 V8_INLINE void AddStringLiteral(const char (&s)[N]) {
89 AddSubstring(s, N - 1);
90 }
91
92 // Add the first 'n' characters of the given 0-terminated string 's' to the
93 // builder. The input string must have enough characters.
94 V8_INLINE void AddSubstring(const char* s, int n) {
95 DCHECK(!is_finalized());
96 DCHECK_LE(position() + n, buffer_.length());
97 DCHECK_LE(n, strlen(s));
98 MemCopy(cursor_, s, n * kCharSize);
99 cursor_ += n;
100 }
101
102 // Add character padding to the builder. If count is non-positive,
103 // nothing is added to the builder.
104 V8_INLINE void AddPadding(char c, int count) {
105 DCHECK(!is_finalized());
106 DCHECK_LE(position() + count, buffer_.length());
107 cursor_ = std::fill_n(cursor_, count, c);
108 }
109
110 // Add the decimal representation of the value.
111 void AddDecimalInteger(int value) {
112 uint32_t number = static_cast<uint32_t>(value);
113 if (value < 0) {
114 AddCharacter('-');
115 number = static_cast<uint32_t>(-value);
116 }
117 int digits = 1;
118 for (uint32_t factor = 10; digits < 10; digits++, factor *= 10) {
119 if (factor > number) break;
120 }
121 cursor_ += digits;
122 for (int i = 1; i <= digits; i++) {
123 *(cursor_ - i) = '0' + static_cast<char>(number % 10);
124 number /= 10;
125 }
126 }
127
128 // Finalize the string by, checking that there is no null-character in the
129 // content. Returns a pointer one past the last character.
130 char* Finalize() {
131 DCHECK(!is_finalized());
132 DCHECK_LE(position(), buffer_.length());
133#ifdef DEBUG
134 // Make sure nobody managed to add a 0-character to the
135 // buffer while building the string.
136 for (const char* buf = buffer_.begin(); buf != cursor_; buf++) {
137 DCHECK_NE(*buf, '\0');
138 }
139#endif
140 char* ret = cursor_;
141 cursor_ = nullptr;
142 DCHECK(is_finalized());
143 return ret;
144 }
145
146 private:
147 base::Vector<char> buffer_;
148 char* cursor_;
149
150 bool is_finalized() const { return cursor_ == nullptr; }
151
152 DISALLOW_IMPLICIT_CONSTRUCTORS(SimpleStringBuilder);
153};

194// Parsing integers with radix 2, 4, 8, 16, 32. Assumes current != end.
195template <int radix_log_2, class Char>
196double InternalStringToIntDouble(const Char* start, const Char* end,
197 bool negative, bool allow_trailing_junk) {
198 const Char* current = start;
199 DCHECK_NE(current, end);
200
201 // Skip leading 0s.
202 while (*current == '0') {
203 ++current;
204 if (current == end) return SignedZero(negative);
205 }
206
207 int64_t number = 0;
208 int exponent = 0;
209 constexpr int radix = (1 << radix_log_2);
210
211 constexpr int lim_0 = '0' + (radix < 10 ? radix : 10);
212 constexpr int lim_a = 'a' + (radix - 10);
213 constexpr int lim_A = 'A' + (radix - 10);
214
215 do {
216 int digit;
217 if (*current >= '0' && *current < lim_0) {
218 digit = static_cast<char>(*current) - '0';
219 } else if (*current >= 'a' && *current < lim_a) {
220 digit = static_cast<char>(*current) - 'a' + 10;
221 } else if (*current >= 'A' && *current < lim_A) {
222 digit = static_cast<char>(*current) - 'A' + 10;
223 } else {
224 // We've not found any digits, this must be junk.
225 if (current == start) return JunkStringValue();
226 if (allow_trailing_junk || !AdvanceToNonspace(&current, end)) break;
227 return JunkStringValue();
228 }
229
230 number = number * radix + digit;
231 int overflow = static_cast<int>(number >> 53);
232 if (overflow != 0) {
233 // Overflow occurred. Need to determine which direction to round the
234 // result.
235 int overflow_bits_count = 1;
236 while (overflow > 1) {
237 overflow_bits_count++;
238 overflow >>= 1;
239 }
240
241 int dropped_bits_mask = ((1 << overflow_bits_count) - 1);
242 int dropped_bits = static_cast<int>(number) & dropped_bits_mask;
243 number >>= overflow_bits_count;
244 exponent = overflow_bits_count;
245
246 bool zero_tail = true;
247 while (true) {
248 ++current;
249 if (current == end || !isDigit(*current, radix)) break;
250 zero_tail = zero_tail && *current == '0';
251 exponent += radix_log_2;
252 }
253
254 if (!allow_trailing_junk && AdvanceToNonspace(&current, end)) {
255 return JunkStringValue();
256 }
257
258 int middle_value = (1 << (overflow_bits_count - 1));
259 if (dropped_bits > middle_value) {
260 number++; // Rounding up.
261 } else if (dropped_bits == middle_value) {
262 // Rounding to even to consistency with decimals: half-way case rounds
263 // up if significant part is odd and down otherwise.
264 if ((number & 1) != 0 || !zero_tail) {
265 number++; // Rounding up.
266 }
267 }
268
269 // Rounding up may cause overflow.
270 if ((number & (static_cast<int64_t>(1) << 53)) != 0) {
271 exponent++;
272 number >>= 1;
273 }
274 break;
275 }
276 ++current;
277 } while (current != end);
278
279 DCHECK(number < ((int64_t)1 << 53));
280 DCHECK(static_cast<int64_t>(static_cast<double>(number)) == number);
281
282 if (exponent == 0) {
283 if (negative) {
284 if (number == 0) return -0.0;
285 number = -number;
286 }
287 return static_cast<double>(number);
288 }
289
290 DCHECK_NE(number, 0);
291 return std::ldexp(static_cast<double>(negative ? -number : number), exponent);
292}

409template <class Char>
410void StringToIntHelper::DetectRadixInternal(const Char* current, int length) {
411 const Char* start = current;
412 length_ = length;
413 const Char* end = start + length;
414
415 if (!AdvanceToNonspace(&current, end)) {
416 return set_state(State::kEmpty);
417 }
418
419 if (*current == '+') {
420 // Ignore leading sign; skip following spaces.
421 ++current;
422 if (current == end) {
423 return set_state(State::kJunk);
424 }
425 sign_ = Sign::kPositive;
426 } else if (*current == '-') {
427 ++current;
428 if (current == end) {
429 return set_state(State::kJunk);
430 }
431 sign_ = Sign::kNegative;
432 }
433
434 if (radix_ == 0) {
435 // Radix detection.
436 radix_ = 10;
437 if (*current == '0') {
438 ++current;
439 if (current == end) return set_state(State::kZero);
440 if (*current == 'x' || *current == 'X') {
441 radix_ = 16;
442 ++current;
443 if (current == end) return set_state(State::kJunk);
444 } else if (allow_binary_and_octal_prefixes_ &&
445 (*current == 'o' || *current == 'O')) {
446 radix_ = 8;
447 ++current;
448 if (current == end) return set_state(State::kJunk);
449 } else if (allow_binary_and_octal_prefixes_ &&
450 (*current == 'b' || *current == 'B')) {
451 radix_ = 2;
452 ++current;
453 if (current == end) return set_state(State::kJunk);
454 } else {
455 leading_zero_ = true;
456 }
457 }
458 } else if (radix_ == 16) {
459 if (*current == '0') {
460 // Allow "0x" prefix.
461 ++current;
462 if (current == end) return set_state(State::kZero);
463 if (*current == 'x' || *current == 'X') {
464 ++current;
465 if (current == end) return set_state(State::kJunk);
466 } else {
467 leading_zero_ = true;
468 }
469 }
470 }
471 // Skip leading zeros.
472 while (*current == '0') {
473 leading_zero_ = true;
474 ++current;
475 if (current == end) return set_state(State::kZero);
476 }
477 // Detect leading zeros with junk after them, if allowed.
478 if (leading_zero_ && allow_trailing_junk_ && !isDigit(*current, radix_)) {
479 return set_state(State::kZero);
480 }
481
482 if (!leading_zero_ && !isDigit(*current, radix_)) {
483 return set_state(State::kJunk);
484 }
485
486 DCHECK(radix_ >= 2 && radix_ <= 36);
487 static_assert(String::kMaxLength <= INT_MAX);
488 cursor_ = static_cast<int>(current - start);
489}

569 template <class Char>
570 void HandleBaseTenCase(const Char* current, const Char* end) {
571 // Parsing with strtod.
572 const int kMaxSignificantDigits = 309; // Doubles are less than 1.8e308.
573 // The buffer may contain up to kMaxSignificantDigits + 1 digits and a zero
574 // end.
575 const int kBufferSize = kMaxSignificantDigits + 2;
576 char buffer[kBufferSize];
577 int buffer_pos = 0;
578 while (*current >= '0' && *current <= '9') {
579 if (buffer_pos <= kMaxSignificantDigits) {
580 // If the number has more than kMaxSignificantDigits it will be parsed
581 // as infinity.
582 DCHECK_LT(buffer_pos, kBufferSize);
583 buffer[buffer_pos++] = static_cast<char>(*current);
584 }
585 ++current;
586 if (current == end) break;
587 }
588
589 SLOW_DCHECK(buffer_pos < kBufferSize);
590 buffer[buffer_pos] = '\0';
591 base::Vector<const char> buffer_vector(buffer, buffer_pos);
592 result_ = Strtod(buffer_vector, 0);
593 set_state(State::kDone);
594 }
595
596 double result_ = 0;

599template <class Char>
600void NumberParseIntHelper::HandleGenericCase(const Char* current,
601 const Char* end) {
602 // The following code causes accumulating rounding error for numbers greater
603 // than ~2^56. It's explicitly allowed in the spec: "if R is not 2, 4, 8, 10,
604 // 16, or 32, then mathInt may be an implementation-dependent approximation to
605 // the mathematical integer value" (15.1.2.2).
606
607 int lim_0 = '0' + (radix() < 10 ? radix() : 10);
608 int lim_a = 'a' + (radix() - 10);
609 int lim_A = 'A' + (radix() - 10);
610
611 // NOTE: The code for computing the value may seem a bit complex at
612 // first glance. It is structured to use 32-bit multiply-and-add
613 // loops as long as possible to avoid losing precision.
614
615 bool done = false;
616 do {
617 // Parse the longest part of the string starting at {current}
618 // possible while keeping the multiplier, and thus the part
619 // itself, within 32 bits.
620 uint32_t part = 0, multiplier = 1;
621 while (true) {
622 uint32_t d;
623 if (*current >= '0' && *current < lim_0) {
624 d = *current - '0';
625 } else if (*current >= 'a' && *current < lim_a) {
626 d = *current - 'a' + 10;
627 } else if (*current >= 'A' && *current < lim_A) {
628 d = *current - 'A' + 10;
629 } else {
630 done = true;
631 break;
632 }
633
634 // Update the value of the part as long as the multiplier fits
635 // in 32 bits. When we can't guarantee that the next iteration
636 // will not overflow the multiplier, we stop parsing the part
637 // by leaving the loop.
638 const uint32_t kMaximumMultiplier = 0xFFFFFFFFU / 36;
639 uint32_t m = multiplier * static_cast<uint32_t>(radix());
640 if (m > kMaximumMultiplier) break;
641 part = part * radix() + d;
642 multiplier = m;
643 DCHECK(multiplier > part);
644
645 ++current;
646 if (current == end) {
647 done = true;
648 break;
649 }
650 }
651 result_ = result_ * multiplier + part;
652 } while (!done);
653
654 if (!allow_trailing_junk() && AdvanceToNonspace(&current, end)) {
655 return set_state(State::kJunk);
656 }
657 return set_state(State::kDone);
658}

660// Converts a string to a double value.
661template <class Char>
662double InternalStringToDouble(const Char* current, const Char* end,
663 ConversionFlag flag, double empty_string_val) {
664 // To make sure that iterator dereferencing is valid the following
665 // convention is used:
666 // 1. Each '++current' statement is followed by check for equality to 'end'.
667 // 2. If AdvanceToNonspace returned false then current == end.
668 // 3. If 'current' becomes be equal to 'end' the function returns or goes to
669 // 'parsing_done'.
670 // 4. 'current' is not dereferenced after the 'parsing_done' label.
671 // 5. Code before 'parsing_done' may rely on 'current != end'.
672 if (!AdvanceToNonspace(&current, end)) {
673 return empty_string_val;
674 }
675
676 // The non-decimal prefix has to be the first thing after any whitespace,
677 // so check for this first.
678 if (flag == ALLOW_NON_DECIMAL_PREFIX) {
679 // Copy the current iterator, so that on a failure to find the prefix, we
680 // rewind to the start.
681 const Char* prefixed = current;
682 if (*prefixed == '0') {
683 ++prefixed;
684 if (prefixed == end) return 0;
685
686 if (*prefixed == 'x' || *prefixed == 'X') {
687 ++prefixed;
688 if (prefixed == end) return JunkStringValue(); // "0x".
689 return InternalStringToIntDouble<4>(prefixed, end, false, false);
690 } else if (*prefixed == 'o' || *prefixed == 'O') {
691 ++prefixed;
692 if (prefixed == end) return JunkStringValue(); // "0o".
693 return InternalStringToIntDouble<3>(prefixed, end, false, false);
694 } else if (*prefixed == 'b' || *prefixed == 'B') {
695 ++prefixed;
696 if (prefixed == end) return JunkStringValue(); // "0b".
697 return InternalStringToIntDouble<1>(prefixed, end, false, false);
698 }
699 }
700 }
701
702 // From here we are parsing a StrDecimalLiteral, as per
703 // https://tc39.es/ecma262/#sec-tonumber-applied-to-the-string-type
704 const bool allow_trailing_junk = flag == ALLOW_TRAILING_JUNK;
705
706 double value;
707 // fast_float takes a char/char16_t instead of a uint8_t/uint16_t. Cast the
708 // pointers to match.
709 using UC = std::conditional_t<std::is_same_v<Char, uint8_t>, char, char16_t>;
710 static_assert(sizeof(UC) == sizeof(Char));
711 const UC* current_uc = reinterpret_cast<const UC*>(current);
712 const UC* end_uc = reinterpret_cast<const UC*>(end);
713 auto ret = fast_float::from_chars(current_uc, end_uc, value,
714 static_cast<fast_float::chars_format>(
715 fast_float::chars_format::general |
716 fast_float::chars_format::no_infnan));
717 if (ret.ptr == end_uc) return value;
718 if (ret.ptr > current_uc) {
719 current = reinterpret_cast<const Char*>(ret.ptr);
720 if (!allow_trailing_junk && AdvanceToNonspace(&current, end)) {
721 return JunkStringValue();
722 }
723 return value;
724 }
725
726 // Failed to parse any number -- handle Â±Infinity before giving up.
727 DCHECK_EQ(ret.ptr, current_uc);
728 DCHECK_NE(current, end);
729 static constexpr char kInfinityString[] = "Infinity";
730 switch (*current) {
731 case '+':
732 // Ignore leading plus sign.
733 ++current;
734 if (current == end) return JunkStringValue();
735 if (*current != kInfinityString[0]) return JunkStringValue();
736 [[fallthrough]];
737 case kInfinityString[0]:
738 if (!SubStringEquals(&current, end, kInfinityString)) {
739 return JunkStringValue();
740 }
741 if (!allow_trailing_junk && AdvanceToNonspace(&current, end)) {
742 return JunkStringValue();
743 }
744 return V8_INFINITY;
745
746 case '-':
747 ++current;
748 if (current == end) return JunkStringValue();
749 if (*current != kInfinityString[0]) return JunkStringValue();
750 if (!SubStringEquals(&current, end, kInfinityString)) {
751 return JunkStringValue();
752 }
753 if (!allow_trailing_junk && AdvanceToNonspace(&current, end)) {
754 return JunkStringValue();
755 }
756 return -V8_INFINITY;
757
758 default:
759 return JunkStringValue();
760 }
761}

946std::string_view DoubleToStringView(double v, base::Vector<char> buffer) {
947 switch (FPCLASSIFY_NAMESPACE::fpclassify(v)) {
948 case FP_NAN:
949 return "NaN";
950 case FP_INFINITE:
951 return (v < 0.0 ? "-Infinity" : "Infinity");
952 case FP_ZERO:
953 return "0";
954 default: {
955 if (IsInt32Double(v)) {
956 // This will trigger if v is -0 and -0.0 is stringified to "0".
957 // (see ES section 7.1.12.1 #sec-tostring-applied-to-the-number-type)
958 return IntToStringView(FastD2I(v), buffer);
959 }
960 SimpleStringBuilder builder(buffer.begin(), buffer.length());
961 int decimal_point;
962 int sign;
963 const int kV8DtoaBufferCapacity = base::kBase10MaximalLength + 1;
964 char decimal_rep[kV8DtoaBufferCapacity];
965 int length;
966
967 base::DoubleToAscii(
968 v, base::DTOA_SHORTEST, 0,
969 base::Vector<char>(decimal_rep, kV8DtoaBufferCapacity), &sign,
970 &length, &decimal_point);
971
972 if (sign) builder.AddCharacter('-');
973
974 if (length <= decimal_point && decimal_point <= 21) {
975 // ECMA-262 section 9.8.1 step 6.
976 builder.AddString(decimal_rep, length);
977 builder.AddPadding('0', decimal_point - length);
978
979 } else if (0 < decimal_point && decimal_point <= 21) {
980 // ECMA-262 section 9.8.1 step 7.
981 builder.AddSubstring(decimal_rep, decimal_point);
982 builder.AddCharacter('.');
983 builder.AddString(decimal_rep + decimal_point, length - decimal_point);
984
985 } else if (decimal_point <= 0 && decimal_point > -6) {
986 // ECMA-262 section 9.8.1 step 8.
987 builder.AddStringLiteral("0.");
988 builder.AddPadding('0', -decimal_point);
989 builder.AddString(decimal_rep, length);
990
991 } else {
992 // ECMA-262 section 9.8.1 step 9 and 10 combined.
993 builder.AddCharacter(decimal_rep[0]);
994 if (length != 1) {
995 builder.AddCharacter('.');
996 builder.AddString(decimal_rep + 1, length - 1);
997 }
998 builder.AddCharacter('e');
999 builder.AddCharacter((decimal_point >= 0) ? '+' : '-');
1000 int exponent = decimal_point - 1;
1001 if (exponent < 0) exponent = -exponent;
1002 builder.AddDecimalInteger(exponent);
1003 }
1004 return {buffer.begin(), builder.Finalize()};
1005 }
1006 }

1009std::string_view IntToStringView(int n, base::Vector<char> buffer) {
1010 bool negative = true;
1011 if (n >= 0) {
1012 n = -n;
1013 negative = false;
1014 }
1015 // Build the string backwards from the least significant digit.
1016 int i = buffer.length();
1017 do {
1018 // We ensured n <= 0, so the subtraction does the right addition.
1019 buffer[--i] = '0' - (n % 10);
1020 n /= 10;
1021 } while (n);
1022 if (negative) buffer[--i] = '-';
1023 return {buffer.begin() + i, buffer.end()};
1024}

1026std::string_view DoubleToFixedStringView(double value, int f,
1027 base::Vector<char> buffer) {
1028 const double kFirstNonFixed = 1e21;
1029 DCHECK_GE(f, 0);
1030 DCHECK_LE(f, kMaxFractionDigits);
1031
1032 bool negative = false;
1033 double abs_value = value;
1034 if (value < 0) {
1035 abs_value = -value;
1036 negative = true;
1037 }
1038
1039 // If abs_value has more than kDoubleToFixedMaxDigitsBeforePoint digits before
1040 // the point use the non-fixed conversion routine.
1041 if (abs_value >= kFirstNonFixed) {
1042 return DoubleToStringView(value, buffer);
1043 }
1044
1045 // Find a sufficiently precise decimal representation of n.
1046 int decimal_point;
1047 int sign;
1048 // Add space for the '\0' byte.
1049 const int kDecimalRepCapacity =
1050 kDoubleToFixedMaxDigitsBeforePoint + kMaxFractionDigits + 1;
1051 char decimal_rep[kDecimalRepCapacity];
1052 int decimal_rep_length;
1053 base::DoubleToAscii(value, base::DTOA_FIXED, f,
1054 base::Vector<char>(decimal_rep, kDecimalRepCapacity),
1055 &sign, &decimal_rep_length, &decimal_point);
1056
1057 // Create a representation that is padded with zeros if needed.
1058 int zero_prefix_length = 0;
1059 int zero_postfix_length = 0;
1060
1061 if (decimal_point <= 0) {
1062 zero_prefix_length = -decimal_point + 1;
1063 decimal_point = 1;
1064 }
1065
1066 if (zero_prefix_length + decimal_rep_length < decimal_point + f) {
1067 zero_postfix_length =
1068 decimal_point + f - decimal_rep_length - zero_prefix_length;
1069 }
1070
1071 unsigned rep_length =
1072 zero_prefix_length + decimal_rep_length + zero_postfix_length;
1073 // TODO(pthier): Get rid of this intermediate string builder.
1074 base::Vector<char> rep_buffer = base::Vector<char>::New(rep_length + 1);
1075 SimpleStringBuilder rep_builder(rep_buffer.begin(), rep_buffer.length());
1076 rep_builder.AddPadding('0', zero_prefix_length);
1077 rep_builder.AddString(decimal_rep, decimal_rep_length);
1078 rep_builder.AddPadding('0', zero_postfix_length);
1079 char* rep_end = rep_builder.Finalize();
1080 // AddSubstring requires a null-terminated string (for DCHECKs only).
1081 *rep_end = '\0';
1082
1083 // Create the result string by appending a minus and putting in a
1084 // decimal point if needed.
1085 SimpleStringBuilder builder(buffer.begin(), buffer.length());
1086 if (negative) builder.AddCharacter('-');
1087 builder.AddSubstring(rep_buffer.begin(), decimal_point);
1088 if (f > 0) {
1089 builder.AddCharacter('.');
1090 builder.AddSubstring(rep_buffer.begin() + decimal_point, f);
1091 }
1092 DeleteArray(rep_buffer.begin());
1093 return {buffer.begin(), builder.Finalize()};
1094}

1096static std::string_view CreateExponentialRepresentation(
1097 char* decimal_rep, int rep_length, int exponent, bool negative,
1098 int significant_digits, base::Vector<char> buffer) {
1099 bool negative_exponent = false;
1100 if (exponent < 0) {
1101 negative_exponent = true;
1102 exponent = -exponent;
1103 }
1104
1105 SimpleStringBuilder builder(buffer.begin(), buffer.length());
1106
1107 if (negative) builder.AddCharacter('-');
1108 builder.AddCharacter(decimal_rep[0]);
1109 if (significant_digits != 1) {
1110 builder.AddCharacter('.');
1111 DCHECK_EQ(rep_length, strlen(decimal_rep));
1112 DCHECK_GE(significant_digits, rep_length);
1113 builder.AddString(decimal_rep + 1, rep_length - 1);
1114 builder.AddPadding('0', significant_digits - rep_length);
1115 }
1116
1117 builder.AddCharacter('e');
1118 builder.AddCharacter(negative_exponent ? '-' : '+');
1119 builder.AddDecimalInteger(exponent);
1120 return {buffer.begin(), builder.Finalize()};

1123std::string_view DoubleToExponentialStringView(double value, int f,
1124 base::Vector<char> buffer) {
1125 // f might be -1 to signal that f was undefined in JavaScript.
1126 DCHECK(f >= -1 && f <= kMaxFractionDigits);
1127
1128 bool negative = false;
1129 if (value < 0) {
1130 value = -value;
1131 negative = true;
1132 }
1133
1134 // Find a sufficiently precise decimal representation of n.
1135 int decimal_point;
1136 int sign;
1137 // f corresponds to the digits after the point. There is always one digit
1138 // before the point. The number of requested_digits equals hence f + 1.
1139 // And we have to add one character for the null-terminator.
1140 const int kV8DtoaBufferCapacity = kMaxFractionDigits + 1 + 1;
1141 // Make sure that the buffer is big enough, even if we fall back to the
1142 // shortest representation (which happens when f equals -1).
1143 DCHECK_LE(base::kBase10MaximalLength, kMaxFractionDigits + 1);
1144 char decimal_rep[kV8DtoaBufferCapacity];
1145 int decimal_rep_length;
1146
1147 if (f == -1) {
1148 base::DoubleToAscii(value, base::DTOA_SHORTEST, 0,
1149 base::Vector<char>(decimal_rep, kV8DtoaBufferCapacity),
1150 &sign, &decimal_rep_length, &decimal_point);
1151 f = decimal_rep_length - 1;
1152 } else {
1153 base::DoubleToAscii(value, base::DTOA_PRECISION, f + 1,
1154 base::Vector<char>(decimal_rep, kV8DtoaBufferCapacity),
1155 &sign, &decimal_rep_length, &decimal_point);
1156 }
1157 DCHECK_GT(decimal_rep_length, 0);
1158 DCHECK(decimal_rep_length <= f + 1);
1159
1160 int exponent = decimal_point - 1;
1161 return CreateExponentialRepresentation(decimal_rep, decimal_rep_length,
1162 exponent, negative, f + 1, buffer);
1163}

1165std::string_view DoubleToPrecisionStringView(double value, int p,
1166 base::Vector<char> buffer) {
1167 const int kMinimalDigits = 1;
1168 DCHECK(p >= kMinimalDigits && p <= kMaxFractionDigits);
1169 USE(kMinimalDigits);
1170
1171 bool negative = false;
1172 if (value < 0) {
1173 value = -value;
1174 negative = true;
1175 }
1176
1177 // Find a sufficiently precise decimal representation of n.
1178 int decimal_point;
1179 int sign;
1180 // Add one for the terminating null character.
1181 const int kV8DtoaBufferCapacity = kMaxFractionDigits + 1;
1182 char decimal_rep[kV8DtoaBufferCapacity];
1183 int decimal_rep_length;
1184
1185 base::DoubleToAscii(value, base::DTOA_PRECISION, p,
1186 base::Vector<char>(decimal_rep, kV8DtoaBufferCapacity),
1187 &sign, &decimal_rep_length, &decimal_point);
1188 DCHECK(decimal_rep_length <= p);
1189
1190 int exponent = decimal_point - 1;
1191
1192 std::string_view result;
1193
1194 if (exponent < -6 || exponent >= p) {
1195 result = CreateExponentialRepresentation(decimal_rep, decimal_rep_length,
1196 exponent, negative, p, buffer);
1197 } else {
1198 // Use fixed notation.
1199 SimpleStringBuilder builder(buffer.begin(), buffer.length());
1200 if (negative) builder.AddCharacter('-');
1201 if (decimal_point <= 0) {
1202 builder.AddStringLiteral("0.");
1203 builder.AddPadding('0', -decimal_point);
1204 builder.AddString(decimal_rep, decimal_rep_length);
1205 builder.AddPadding('0', p - decimal_rep_length);
1206 } else {
1207 const int m = std::min(decimal_rep_length, decimal_point);
1208 builder.AddSubstring(decimal_rep, m);
1209 builder.AddPadding('0', decimal_point - decimal_rep_length);
1210 if (decimal_point < p) {
1211 builder.AddCharacter('.');
1212 const int extra = negative ? 2 : 1;
1213 if (decimal_rep_length > decimal_point) {
1214 DCHECK_EQ(decimal_rep_length - decimal_point,
1215 strlen(decimal_rep + decimal_point));
1216 const int len = decimal_rep_length - decimal_point;
1217 const int n = std::min(len, p - (builder.position() - extra));
1218 builder.AddSubstring(decimal_rep + decimal_point, n);
1219 }
1220 builder.AddPadding('0', extra + (p - builder.position()));
1221 }
1222 }
1223 result = {buffer.begin(), builder.Finalize()};
1224 }
1225
1226 return result;
1227}

1229std::string_view DoubleToRadixStringView(double value, int radix,
1230 base::Vector<char> buffer) {
1231 DCHECK(radix >= 2 && radix <= 36);
1232 DCHECK(std::isfinite(value));
1233 DCHECK_NE(0.0, value);
1234 // Character array used for conversion.
1235 static const char chars[] = "0123456789abcdefghijklmnopqrstuvwxyz";
1236
1237 int integer_cursor = buffer.length() / 2;
1238 int fraction_cursor = integer_cursor;
1239
1240 bool negative = value < 0;
1241 if (negative) value = -value;
1242
1243 // Split the value into an integer part and a fractional part.
1244 double integer = std::floor(value);
1245 double fraction = value - integer;
1246 // We only compute fractional digits up to the input double's precision.
1247 double delta = 0.5 * (base::Double(value).NextDouble() - value);
1248 delta = std::max(base::Double(0.0).NextDouble(), delta);
1249 DCHECK_GT(delta, 0.0);
1250 if (fraction >= delta) {
1251 // Insert decimal point.
1252 buffer[fraction_cursor++] = '.';
1253 do {
1254 // Shift up by one digit.
1255 fraction *= radix;
1256 delta *= radix;
1257 // Write digit.
1258 int digit = static_cast<int>(fraction);
1259 buffer[fraction_cursor++] = chars[digit];
1260 // Calculate remainder.
1261 fraction -= digit;
1262 // Round to even.
1263 if (fraction > 0.5 || (fraction == 0.5 && (digit & 1))) {
1264 if (fraction + delta > 1) {
1265 // We need to back trace already written digits in case of carry-over.
1266 while (true) {
1267 fraction_cursor--;
1268 if (fraction_cursor == buffer.length() / 2) {
1269 CHECK_EQ('.', buffer[fraction_cursor]);
1270 // Carry over to the integer part.
1271 integer += 1;
1272 break;
1273 }
1274 char c = buffer[fraction_cursor];
1275 // Reconstruct digit.
1276 digit = c > '9' ? (c - 'a' + 10) : (c - '0');
1277 if (digit + 1 < radix) {
1278 buffer[fraction_cursor++] = chars[digit + 1];
1279 break;
1280 }
1281 }
1282 break;
1283 }
1284 }
1285 } while (fraction >= delta);
1286 }
1287
1288 // Compute integer digits. Fill unrepresented digits with zero.
1289 while (base::Double(integer / radix).Exponent() > 0) {
1290 integer /= radix;
1291 buffer[--integer_cursor] = '0';
1292 }
1293 do {
1294 double remainder = Modulo(integer, radix);
1295 buffer[--integer_cursor] = chars[static_cast<int>(remainder)];
1296 integer = (integer - remainder) / radix;
1297 } while (integer > 0);
1298
1299 // Add sign and terminate string.
1300 if (negative) buffer[--integer_cursor] = '-';
1301 DCHECK_LE(0, integer_cursor);
1302 DCHECK_GT(fraction_cursor - integer_cursor, 0);
1303 return {buffer.begin() + integer_cursor,
1304 static_cast<size_t>(fraction_cursor - integer_cursor)};
1305}

1374bool IsSpecialIndex(Tagged<String> string,
1375 SharedStringAccessGuardIfNeeded& access_guard) {
1376 // Max length of canonical double: -X.XXXXXXXXXXXXXXXXX-eXXX
1377 const uint32_t kBufferSize = 24;
1378 const uint32_t length = string->length();
1379 if (length == 0 || length > kBufferSize) return false;
1380 uint16_t buffer[kBufferSize];
1381 String::WriteToFlat(string, buffer, 0, length, access_guard);
1382 // If the first char is not a digit or a '-' or we can't match 'NaN' or
1383 // '(-)Infinity', bailout immediately.
1384 uint32_t offset = 0;
1385 if (!IsDecimalDigit(buffer[0])) {
1386 if (buffer[0] == '-') {
1387 if (length == 1) return false; // Just '-' is bad.
1388 if (!IsDecimalDigit(buffer[1])) {
1389 if (buffer[1] == 'I' && length == 9) {
1390 // Allow matching of '-Infinity' below.
1391 } else {
1392 return false;
1393 }
1394 }
1395 offset++;
1396 } else if (buffer[0] == 'I' && length == 8) {
1397 // Allow matching of 'Infinity' below.
1398 } else if (buffer[0] == 'N' && length == 3) {
1399 // Match NaN.
1400 return buffer[1] == 'a' && buffer[2] == 'N';
1401 } else {
1402 return false;
1403 }
1404 }
1405 // Expected fast path: key is an integer.
1406 static const uint32_t kRepresentableIntegerLength = 15; // (-)XXXXXXXXXXXXXXX
1407 if (length - offset <= kRepresentableIntegerLength) {
1408 const uint32_t initial_offset = offset;
1409 bool matches = true;
1410 for (; offset < length; offset++) {
1411 matches &= IsDecimalDigit(buffer[offset]);
1412 }
1413 if (matches) {
1414 // Match 0 and -0.
1415 if (buffer[initial_offset] == '0') return initial_offset == length - 1;
1416 return true;
1417 }
1418 }
1419 // Slow path: test DoubleToString(StringToDouble(string)) == string.
1420 base::Vector<const uint16_t> vector(buffer, length);
1421 double d = StringToDouble(vector, NO_CONVERSION_FLAG);
1422 if (std::isnan(d)) return false;
1423 // Compute reverse string.
1424 char reverse_buffer[kBufferSize + 1]; // Result will be /0 terminated.
1425 base::Vector<char> reverse_vector(reverse_buffer, arraysize(reverse_buffer));
1426 std::string_view reverse_string = DoubleToStringView(d, reverse_vector);
1427
1428 if (reverse_string.length() != length) return false;
1429 for (uint32_t i = 0; i < length; ++i) {
1430 if (static_cast<uint16_t>(reverse_string[i]) != buffer[i]) return false;
1431 }
1432 return true;
1433}
