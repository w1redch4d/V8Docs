14void PeelWasmLoop(Node* loop_node, ZoneUnorderedSet<Node*>* loop, Graph* graph,
15 CommonOperatorBuilder* common, Zone* tmp_zone,
16 SourcePositionTable* source_positions,
17 NodeOriginTable* node_origins) {
18 DCHECK_EQ(loop_node->opcode(), IrOpcode::kLoop);
19 DCHECK_NOT_NULL(loop);
20 // No back-jump to the loop header means this is not really a loop.
21 if (loop_node->InputCount() < 2) return;
22
23 uint32_t copied_size = static_cast<uint32_t>(loop->size()) * 2;
24
25 NodeVector copied_nodes(tmp_zone);
26
27 NodeCopier copier(graph, copied_size, &copied_nodes, 1);
28 source_positions->AddDecorator();
29 copier.CopyNodes(graph, tmp_zone, graph->NewNode(common->Dead()),
30 base::make_iterator_range(loop->begin(), loop->end()),
31 source_positions, node_origins);
32 source_positions->RemoveDecorator();
33
34 Node* peeled_iteration_header = copier.map(loop_node);
35
36 // The terminator nodes in the copies need to get connected to the graph's end
37 // node, except Terminate nodes which will be deleted anyway.
38 for (Node* node : copied_nodes) {
39 if (IrOpcode::IsGraphTerminator(node->opcode()) &&
40 node->opcode() != IrOpcode::kTerminate && node->UseCount() == 0) {
41 NodeProperties::MergeControlToEnd(graph, common, node);
42 }
43 }
44
45 // Step 1: Create merges for loop exits.
46 for (Node* node : loop_node->uses()) {
47 // We do not need the Terminate node for the peeled iteration.
48 if (node->opcode() == IrOpcode::kTerminate) {
49 copier.map(node)->Kill();
50 continue;
51 }
52 if (node->opcode() != IrOpcode::kLoopExit) continue;
53 DCHECK_EQ(node->InputAt(1), loop_node);
54 // Create a merge node for the peeled iteration and main loop. Skip the
55 // LoopExit node in the peeled iteration, use its control input instead.
56 Node* merge_node =
57 graph->NewNode(common->Merge(2), node, copier.map(node)->InputAt(0));
58 // Replace all uses of the loop exit with the merge node.
59 for (Edge use_edge : node->use_edges()) {
60 Node* use = use_edge.from();
61 if (loop->count(use) == 1) {
62 // Uses within the loop will be LoopExitEffects and LoopExitValues.
63 // Those are used by nodes outside the loop. We need to create phis from
64 // the main loop and peeled iteration to replace loop exits.
65 DCHECK(use->opcode() == IrOpcode::kLoopExitEffect ||
66 use->opcode() == IrOpcode::kLoopExitValue);
67 const Operator* phi_operator =
68 use->opcode() == IrOpcode::kLoopExitEffect
69 ? common->EffectPhi(2)
70 : common->Phi(LoopExitValueRepresentationOf(use->op()), 2);
71 Node* phi = graph->NewNode(phi_operator, use,
72 copier.map(use)->InputAt(0), merge_node);
73 use->ReplaceUses(phi);
74 // Fix the input of phi we just broke.
75 phi->ReplaceInput(0, use);
76 copier.map(use)->Kill();
77 } else if (use != merge_node) {
78 // For uses outside the loop, simply redirect them to the merge.
79 use->ReplaceInput(use_edge.index(), merge_node);
80 }
81 }
82 copier.map(node)->Kill();
83 }
84
85 // Step 2: The peeled iteration is not a loop anymore. Any control uses of
86 // its loop header should now point to its non-recursive input. Any phi uses
87 // should use the value coming from outside the loop.
88 for (Edge use_edge : peeled_iteration_header->use_edges()) {
89 if (NodeProperties::IsPhi(use_edge.from())) {
90 use_edge.from()->ReplaceUses(use_edge.from()->InputAt(0));
91 } else {
92 use_edge.UpdateTo(loop_node->InputAt(0));
93 }
94 }
95
96 // We are now left with an unconnected subgraph of the peeled Loop node and
97 // its phi uses.
98
99 // Step 3: Rewire the peeled iteration to flow into the main loop.
100
101 // We are reusing the Loop node of the peeled iteration and its phis as the
102 // merge and phis which flow from the peeled iteration into the main loop.
103 // First, remove the non-recursive input.
104 peeled_iteration_header->RemoveInput(0);
105 NodeProperties::ChangeOp(
106 peeled_iteration_header,
107 common->Merge(peeled_iteration_header->InputCount()));
108
109 // Remove the non-recursive input.
110 for (Edge use_edge : peeled_iteration_header->use_edges()) {
111 DCHECK(NodeProperties::IsPhi(use_edge.from()));
112 use_edge.from()->RemoveInput(0);
113 const Operator* phi = common->ResizeMergeOrPhi(
114 use_edge.from()->op(),
115 use_edge.from()->InputCount() - /* control input */ 1);
116 NodeProperties::ChangeOp(use_edge.from(), phi);
117 }
118
119 // In the main loop, change inputs to the merge and phis above.
120 loop_node->ReplaceInput(0, peeled_iteration_header);
121 for (Edge use_edge : loop_node->use_edges()) {
122 if (NodeProperties::IsPhi(use_edge.from())) {
123 use_edge.from()->ReplaceInput(0, copier.map(use_edge.from()));
124 }
125 }
126}
