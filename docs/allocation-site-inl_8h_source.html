130inline bool AllocationSite::CanTrack(InstanceType type) {
131 if (!V8_ALLOCATION_SITE_TRACKING_BOOL) return false;
132 if (v8_flags.allocation_site_pretenuring) {
133 // TurboFan doesn't care at all about String pretenuring feedback,
134 // so don't bother even trying to track that.
135 return type == JS_ARRAY_TYPE || type == JS_OBJECT_TYPE;
136 }
137 return type == JS_ARRAY_TYPE;

212template <AllocationSiteUpdateMode update_or_check>
213bool AllocationSite::DigestTransitionFeedback(DirectHandle<AllocationSite> site,
214 ElementsKind to_kind) {
215 Isolate* isolate = site->GetIsolate();
216 bool result = false;
217
218 if (site->PointsToLiteral() && IsJSArray(site->boilerplate())) {
219 DirectHandle<JSArray> boilerplate(Cast<JSArray>(site->boilerplate()),
220 isolate);
221 ElementsKind kind = boilerplate->GetElementsKind();
222 // if kind is holey ensure that to_kind is as well.
223 if (IsHoleyElementsKind(kind)) {
224 to_kind = GetHoleyElementsKind(to_kind);
225 }
226 if (IsMoreGeneralElementsKindTransition(kind, to_kind)) {
227 // If the array is huge, it's not likely to be defined in a local
228 // function, so we shouldn't make new instances of it very often.
229 uint32_t length = 0;
230 CHECK(Object::ToArrayLength(boilerplate->length(), &length));
231 if (length <= kMaximumArrayBytesToPretransition) {
232 if (update_or_check == AllocationSiteUpdateMode::kCheckOnly) {
233 return true;
234 }
235 if (v8_flags.trace_track_allocation_sites) {
236 bool is_nested = site->IsNested();
237 PrintF("AllocationSite: JSArray %p boilerplate %supdated %s->%s\n",
238 reinterpret_cast<void*>(site->ptr()),
239 is_nested ? "(nested)" : " ", ElementsKindToString(kind),
240 ElementsKindToString(to_kind));
241 }
242 CHECK_NE(to_kind, DICTIONARY_ELEMENTS);
243 JSObject::TransitionElementsKind(boilerplate, to_kind);
244 DependentCode::DeoptimizeDependencyGroups(
245 isolate, *site,
246 DependentCode::kAllocationSiteTransitionChangedGroup);
247 result = true;
248 }
249 }
250 } else {
251 // The AllocationSite is for a constructed Array.
252 ElementsKind kind = site->GetElementsKind();
253 // if kind is holey ensure that to_kind is as well.
254 if (IsHoleyElementsKind(kind)) {
255 to_kind = GetHoleyElementsKind(to_kind);
256 }
257 if (IsMoreGeneralElementsKindTransition(kind, to_kind)) {
258 if (update_or_check == AllocationSiteUpdateMode::kCheckOnly) return true;
259 if (v8_flags.trace_track_allocation_sites) {
260 PrintF("AllocationSite: JSArray %p site updated %s->%s\n",
261 reinterpret_cast<void*>(site->ptr()), ElementsKindToString(kind),
262 ElementsKindToString(to_kind));
263 }
264 site->SetElementsKind(to_kind);
265 DependentCode::DeoptimizeDependencyGroups(
266 isolate, *site, DependentCode::kAllocationSiteTransitionChangedGroup);
267 result = true;
268 }
269 }
270 return result;
271}
