65// This may return an empty MaybeHandle if an exception is thrown or
66// we abort due to reaching the recursion limit.
67MaybeHandle<String> StringReplaceOneCharWithString(
68 Isolate* isolate, Handle<String> subject, Handle<String> search,
69 Handle<String> replace, bool* found, int recursion_limit) {
70 StackLimitCheck stackLimitCheck(isolate);
71 if (stackLimitCheck.HasOverflowed() || (recursion_limit == 0)) {
72 return MaybeHandle<String>();
73 }
74 recursion_limit--;
75 if (IsConsString(*subject)) {
76 Tagged<ConsString> cons = Cast<ConsString>(*subject);
77 Handle<String> first = handle(cons->first(), isolate);
78 Handle<String> second = handle(cons->second(), isolate);
79 Handle<String> new_first;
80 if (!StringReplaceOneCharWithString(isolate, first, search, replace, found,
81 recursion_limit).ToHandle(&new_first)) {
82 return MaybeHandle<String>();
83 }
84 if (*found) return isolate->factory()->NewConsString(new_first, second);
85
86 Handle<String> new_second;
87 if (!StringReplaceOneCharWithString(isolate, second, search, replace, found,
88 recursion_limit)
89 .ToHandle(&new_second)) {
90 return MaybeHandle<String>();
91 }
92 if (*found) return isolate->factory()->NewConsString(first, new_second);
93
94 return subject;
95 } else {
96 int index = String::IndexOf(isolate, subject, search, 0);
97 if (index == -1) return subject;
98 *found = true;
99 Handle<String> first = isolate->factory()->NewSubString(subject, 0, index);
100 Handle<String> cons1;
101 ASSIGN_RETURN_ON_EXCEPTION(
102 isolate, cons1, isolate->factory()->NewConsString(first, replace));
103 Handle<String> second =
104 isolate->factory()->NewSubString(subject, index + 1, subject->length());
105 return isolate->factory()->NewConsString(cons1, second);
106 }
107}

109RUNTIME_FUNCTION(Runtime_StringReplaceOneCharWithString) {
110 HandleScope scope(isolate);
111 DCHECK_EQ(3, args.length());
112 Handle<String> subject = args.at<String>(0);
113 Handle<String> search = args.at<String>(1);
114 Handle<String> replace = args.at<String>(2);
115
116 // If the cons string tree is too deep, we simply abort the recursion and
117 // retry with a flattened subject string.
118 const int kRecursionLimit = 0x1000;
119 bool found = false;
120 DirectHandle<String> result;
121 if (StringReplaceOneCharWithString(isolate, subject, search, replace, &found,
122 kRecursionLimit).ToHandle(&result)) {
123 return *result;
124 }
125 if (isolate->has_exception()) return ReadOnlyRoots(isolate).exception();
126
127 subject = String::Flatten(isolate, subject);
128 if (StringReplaceOneCharWithString(isolate, subject, search, replace, &found,
129 kRecursionLimit).ToHandle(&result)) {
130 return *result;
131 }
132 if (isolate->has_exception()) return ReadOnlyRoots(isolate).exception();
133 // In case of empty handle and no exception we have stack overflow.
134 return isolate->StackOverflow();
135}

194RUNTIME_FUNCTION(Runtime_StringCodePointAt) {
195 HandleScope handle_scope(isolate);
196 DCHECK_EQ(2, args.length());
197
198 Handle<String> subject = args.at<String>(0);
199 uint32_t i = NumberToUint32(args[1]);
200
201 // Flatten the string. If someone wants to get a char at an index
202 // in a cons string, it is likely that more indices will be
203 // accessed.
204 subject = String::Flatten(isolate, subject);
205
206 if (i >= static_cast<uint32_t>(subject->length())) {
207 return ReadOnlyRoots(isolate).nan_value();
208 }
209
210 int first_code_point = subject->Get(i);
211 if ((first_code_point & 0xFC00) != 0xD800) {
212 return Smi::FromInt(first_code_point);
213 }
214
215 if (i + 1 >= static_cast<uint32_t>(subject->length())) {
216 return Smi::FromInt(first_code_point);
217 }
218
219 int second_code_point = subject->Get(i + 1);
220 if ((second_code_point & 0xFC00) != 0xDC00) {
221 return Smi::FromInt(first_code_point);
222 }
223
224 int surrogate_offset = 0x10000 - (0xD800 << 10) - 0xDC00;
225 return Smi::FromInt((first_code_point << 10) +
226 (second_code_point + surrogate_offset));
227}

229RUNTIME_FUNCTION(Runtime_StringBuilderConcat) {
230 HandleScope scope(isolate);
231 DCHECK_EQ(3, args.length());
232 DirectHandle<FixedArray> array = args.at<FixedArray>(0);
233
234 int array_length = args.smi_value_at(1);
235
236 DirectHandle<String> special = args.at<String>(2);
237
238 // This assumption is used by the slice encoding in one or two smis.
239 DCHECK_GE(Smi::kMaxValue, String::kMaxLength);
240
241 int special_length = special->length();
242
243 int length;
244 bool one_byte = special->IsOneByteRepresentation();
245
246 {
247 DisallowGarbageCollection no_gc;
248 Tagged<FixedArray> fixed_array = *array;
249
250 if (array_length == 0) {
251 return ReadOnlyRoots(isolate).empty_string();
252 } else if (array_length == 1) {
253 Tagged<Object> first = fixed_array->get(0);
254 if (IsString(first)) return first;
255 }
256 length = StringBuilderConcatLength(special_length, fixed_array,
257 array_length, &one_byte);
258 }
259
260 if (length == -1) {
261 return isolate->Throw(ReadOnlyRoots(isolate).illegal_argument_string());
262 }
263 if (length == 0) {
264 return ReadOnlyRoots(isolate).empty_string();
265 }
266
267 if (one_byte) {
268 DirectHandle<SeqOneByteString> answer;
269 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
270 isolate, answer, isolate->factory()->NewRawOneByteString(length));
271 DisallowGarbageCollection no_gc;
272 StringBuilderConcatHelper(*special, answer->GetChars(no_gc), *array,
273 array_length);
274 return *answer;
275 } else {
276 DirectHandle<SeqTwoByteString> answer;
277 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
278 isolate, answer, isolate->factory()->NewRawTwoByteString(length));
279 DisallowGarbageCollection no_gc;
280 StringBuilderConcatHelper(*special, answer->GetChars(no_gc), *array,
281 array_length);
282 return *answer;
283 }

286// Converts a String to JSArray.
287// For example, "foo" => ["f", "o", "o"].
288RUNTIME_FUNCTION(Runtime_StringToArray) {
289 HandleScope scope(isolate);
290 DCHECK_EQ(2, args.length());
291 Handle<String> s = args.at<String>(0);
292 uint32_t limit = NumberToUint32(args[1]);
293
294 s = String::Flatten(isolate, s);
295 const int length =
296 static_cast<int>(std::min(static_cast<uint32_t>(s->length()), limit));
297
298 DirectHandle<FixedArray> elements = isolate->factory()->NewFixedArray(length);
299 bool elements_are_initialized = false;
300
301 if (s->IsFlat() && s->IsOneByteRepresentation()) {
302 DisallowGarbageCollection no_gc;
303 String::FlatContent content = s->GetFlatContent(no_gc);
304 // Use pre-initialized single characters to intialize all the elements.
305 // This can be false if the string is sliced from an externalized
306 // two-byte string that has only one-byte chars, in that case we will do
307 // a LookupSingleCharacterStringFromCode for each of the characters.
308 if (content.IsOneByte()) {
309 base::Vector<const uint8_t> chars = content.ToOneByteVector();
310 Tagged<FixedArray> one_byte_table =
311 isolate->heap()->single_character_string_table();
312 for (int i = 0; i < length; ++i) {
313 Tagged<Object> value = one_byte_table->get(chars[i]);
314 DCHECK(IsString(value));
315 DCHECK(ReadOnlyHeap::Contains(Cast<HeapObject>(value)));
316 // The single-character strings are in RO space so it should
317 // be safe to skip the write barriers.
318 elements->set(i, value, SKIP_WRITE_BARRIER);
319 }
320 elements_are_initialized = true;
321 }
322 }
323
324 if (!elements_are_initialized) {
325 for (int i = 0; i < length; ++i) {
326 DirectHandle<Object> str =
327 isolate->factory()->LookupSingleCharacterStringFromCode(s->Get(i));
328 elements->set(i, *str);
329 }
330 }
331
332#ifdef DEBUG
333 for (int i = 0; i < length; ++i) {
334 DCHECK_EQ(Cast<String>(elements->get(i))->length(), 1);
335 }
336#endif
337
338 return *isolate->factory()->NewJSArrayWithElements(elements);
339}

417RUNTIME_FUNCTION(Runtime_StringEscapeQuotes) {
418 HandleScope handle_scope(isolate);
419 DCHECK_EQ(1, args.length());
420 DirectHandle<String> string = args.at<String>(0);
421
422 // Equivalent to global replacement `string.replace(/"/g, "&quot")`, but this
423 // does not modify any global state (e.g. the regexp match info).
424
425 const int string_length = string->length();
426 DirectHandle<String> quotes =
427 isolate->factory()->LookupSingleCharacterStringFromCode('"');
428
429 int quote_index = String::IndexOf(isolate, string, quotes, 0);
430
431 // No quotes, nothing to do.
432 if (quote_index == -1) return *string;
433
434 // Find all quotes.
435 std::vector<int> indices = {quote_index};
436 while (quote_index + 1 < string_length) {
437 quote_index = String::IndexOf(isolate, string, quotes, quote_index + 1);
438 if (quote_index == -1) break;
439 indices.emplace_back(quote_index);
440 }
441
442 // Build the replacement string.
443 DirectHandle<String> replacement =
444 isolate->factory()->NewStringFromAsciiChecked("&quot;");
445 const int estimated_part_count = static_cast<int>(indices.size()) * 2 + 1;
446 ReplacementStringBuilder builder(isolate->heap(), string,
447 estimated_part_count);
448
449 int prev_index = -1; // Start at -1 to avoid special-casing the first match.
450 for (int index : indices) {
451 const int slice_start = prev_index + 1;
452 const int slice_end = index;
453 if (slice_end > slice_start) {
454 builder.AddSubjectSlice(slice_start, slice_end);
455 }
456 builder.AddString(replacement);
457 prev_index = index;
458 }
459
460 if (prev_index < string_length - 1) {
461 builder.AddSubjectSlice(prev_index + 1, string_length);
462 }
463
464 DirectHandle<String> result;
465 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(isolate, result, builder.ToString());
466 return *result;
467}
