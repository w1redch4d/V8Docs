72bool PrototypeIterator::HasAccess() const {
73 // We can only perform access check in the handlified version of the
74 // PrototypeIterator.
75 DCHECK(!handle_.is_null());
76 if (IsAccessCheckNeeded(*handle_)) {
77 return isolate_->MayAccess(isolate_->native_context(),
78 Cast<JSObject>(handle_));
79 }
80 return true;

83void PrototypeIterator::Advance() {
84 if (handle_.is_null() && IsJSProxy(object_)) {
85 is_at_end_ = true;
86 object_ = ReadOnlyRoots(isolate_).null_value();
87 return;
88 } else if (!handle_.is_null() && IsJSProxy(*handle_)) {
89 is_at_end_ = true;
90 handle_ = isolate_->factory()->null_value();
91 return;
92 }
93 AdvanceIgnoringProxies();
94}

96void PrototypeIterator::AdvanceIgnoringProxies() {
97 Tagged<JSPrototype> object = handle_.is_null() ? object_ : *handle_;
98 Tagged<Map> map = object->map();
99
100 Tagged<JSPrototype> prototype = map->prototype();
101 is_at_end_ = IsNull(prototype, isolate_) ||
102 (where_to_end_ == END_AT_NON_HIDDEN && !IsJSGlobalProxyMap(map));
103
104 if (handle_.is_null()) {
105 object_ = prototype;
106 } else {
107 handle_ = handle(prototype, isolate_);
108 }
109}

111V8_WARN_UNUSED_RESULT bool PrototypeIterator::AdvanceFollowingProxies() {
112 DCHECK(!(handle_.is_null() && IsJSProxy(object_)));
113 if (!HasAccess()) {
114 // Abort the lookup if we do not have access to the current object.
115 handle_ = isolate_->factory()->null_value();
116 is_at_end_ = true;
117 return true;
118 }
119 return AdvanceFollowingProxiesIgnoringAccessChecks();
120}

122V8_WARN_UNUSED_RESULT bool
123PrototypeIterator::AdvanceFollowingProxiesIgnoringAccessChecks() {
124 if (handle_.is_null() || !IsJSProxy(*handle_)) {
125 AdvanceIgnoringProxies();
126 return true;
127 }
128
129 // Due to possible __proto__ recursion limit the number of Proxies
130 // we visit to an arbitrarily chosen large number.
131 seen_proxies_++;
132 if (seen_proxies_ > JSProxy::kMaxIterationLimit) {
133 isolate_->StackOverflow();
134 return false;
135 }
136 MaybeDirectHandle<JSPrototype> proto =
137 JSProxy::GetPrototype(Cast<JSProxy>(handle_));
138
139 // TODO(372390038): This can be again simplified when handle_ migrates to a
140 // direct handle.
141 DirectHandle<JSPrototype> proto_direct_handle;
142 bool ok = proto.ToHandle(&proto_direct_handle);
143 handle_ = indirect_handle(proto_direct_handle, isolate_);
144 if (!ok) return false;
145
146 is_at_end_ = where_to_end_ == END_AT_NON_HIDDEN || IsNull(*handle_, isolate_);
147 return true;
148}
