51 switch (node->opcode()) {
52 case IrOpcode::kWasmTypeCheck:
53 return ReduceWasmTypeCheck(node);
54 case IrOpcode::kWasmTypeCheckAbstract:
55 return ReduceWasmTypeCheckAbstract(node);
56 case IrOpcode::kWasmTypeCast:

58 case IrOpcode::kWasmTypeCastAbstract:
59 return ReduceWasmTypeCastAbstract(node);

72 case IrOpcode::kWasmAnyConvertExtern:

74 case IrOpcode::kWasmExternConvertAny:

76 case IrOpcode::kWasmStructGet:

78 case IrOpcode::kWasmStructSet:

118Reduction WasmGCLowering::ReduceWasmTypeCheck(Node* node) {
119 DCHECK_EQ(node->opcode(), IrOpcode::kWasmTypeCheck);
120
121 Node* object = node->InputAt(0);
122 Node* rtt = node->InputAt(1);
123 Node* effect_input = NodeProperties::GetEffectInput(node);
124 Node* control_input = NodeProperties::GetControlInput(node);
125 auto config = OpParameter<WasmTypeCheckConfig>(node->op());
126 int rtt_depth = wasm::GetSubtypingDepth(module_, config.to.ref_index());
127 bool object_can_be_null = config.from.is_nullable();
128 bool object_can_be_i31 =
129 wasm::IsSubtypeOf(wasm::kWasmI31Ref.AsNonNull(), config.from, module_);
130
131 gasm_.InitializeEffectControl(effect_input, control_input);
132
133 auto end_label = gasm_.MakeLabel(MachineRepresentation::kWord32);
134 bool is_cast_from_any = config.from.is_reference_to(wasm::HeapType::kAny);
135
136 // If we are casting from any and null results in check failure, then the
137 // {IsDataRefMap} check below subsumes the null check. Otherwise, perform
138 // an explicit null check now.
139 if (object_can_be_null && (!is_cast_from_any || config.to.is_nullable())) {
140 const int kResult = config.to.is_nullable() ? 1 : 0;
141 gasm_.GotoIf(IsNull(object, wasm::kWasmAnyRef), &end_label,
142 BranchHint::kFalse, gasm_.Int32Constant(kResult));
143 }
144
145 if (object_can_be_i31) {
146 gasm_.GotoIf(gasm_.IsSmi(object), &end_label, gasm_.Int32Constant(0));
147 }
148
149 Node* map = gasm_.LoadMap(object);
150
151 if (module_->type(config.to.ref_index()).is_final) {
152 gasm_.Goto(&end_label, gasm_.TaggedEqual(map, rtt));
153 } else {
154 // First, check if types happen to be equal. This has been shown to give
155 // large speedups.
156 gasm_.GotoIf(gasm_.TaggedEqual(map, rtt), &end_label, BranchHint::kTrue,
157 gasm_.Int32Constant(1));
158
159 // Check if map instance type identifies a wasm object.
160 if (is_cast_from_any) {
161 Node* is_wasm_obj = gasm_.IsDataRefMap(map);
162 gasm_.GotoIfNot(is_wasm_obj, &end_label, BranchHint::kTrue,
163 gasm_.Int32Constant(0));
164 }
165
166 Node* type_info = gasm_.LoadWasmTypeInfo(map);
167 DCHECK_GE(rtt_depth, 0);
168 // If the depth of the rtt is known to be less that the minimum supertype
169 // array length, we can access the supertype without bounds-checking the
170 // supertype array.
171 if (static_cast<uint32_t>(rtt_depth) >= wasm::kMinimumSupertypeArraySize) {
172 Node* supertypes_length =
173 gasm_.BuildChangeSmiToIntPtr(gasm_.LoadImmutableFromObject(
174 MachineType::TaggedSigned(), type_info,
175 wasm::ObjectAccess::ToTagged(
176 WasmTypeInfo::kSupertypesLengthOffset)));
177 gasm_.GotoIfNot(gasm_.UintLessThan(gasm_.IntPtrConstant(rtt_depth),
178 supertypes_length),
179 &end_label, BranchHint::kTrue, gasm_.Int32Constant(0));
180 }
181
182 Node* maybe_match = gasm_.LoadImmutableFromObject(
183 MachineType::TaggedPointer(), type_info,
184 wasm::ObjectAccess::ToTagged(WasmTypeInfo::kSupertypesOffset +
185 kTaggedSize * rtt_depth));
186
187 gasm_.Goto(&end_label, gasm_.TaggedEqual(maybe_match, rtt));
188 }
189
190 gasm_.Bind(&end_label);
191
192 ReplaceWithValue(node, end_label.PhiAt(0), gasm_.effect(), gasm_.control());
193 node->Kill();
194 return Replace(end_label.PhiAt(0)); // Meaningless argument.
195}

197Reduction WasmGCLowering::ReduceWasmTypeCheckAbstract(Node* node) {
198 DCHECK_EQ(node->opcode(), IrOpcode::kWasmTypeCheckAbstract);
199
200 Node* object = node->InputAt(0);
201 Node* effect_input = NodeProperties::GetEffectInput(node);
202 Node* control_input = NodeProperties::GetControlInput(node);
203 WasmTypeCheckConfig config = OpParameter<WasmTypeCheckConfig>(node->op());
204 const bool object_can_be_null = config.from.is_nullable();
205 const bool null_succeeds = config.to.is_nullable();
206 const bool object_can_be_i31 =
207 wasm::IsSubtypeOf(wasm::kWasmI31Ref.AsNonNull(), config.from, module_) ||
208 config.from.heap_representation() == wasm::HeapType::kExtern;
209
210 gasm_.InitializeEffectControl(effect_input, control_input);
211
212 Node* result = nullptr;
213 auto end_label = gasm_.MakeLabel(MachineRepresentation::kWord32);
214
215 wasm::HeapType::Representation to_rep = config.to.heap_representation();
216 do {
217 // The none-types only perform a null check. They need no control flow.
218 if (to_rep == wasm::HeapType::kNone ||
219 to_rep == wasm::HeapType::kNoExtern ||
220 to_rep == wasm::HeapType::kNoFunc || to_rep == wasm::HeapType::kNoExn) {
221 result = IsNull(object, config.from);
222 break;
223 }
224 // Null checks performed by any other type check need control flow. We can
225 // skip the null check if null fails, because it's covered by the Smi check
226 // or instance type check we'll do later.
227 if (object_can_be_null && null_succeeds) {
228 const int kResult = null_succeeds ? 1 : 0;
229 gasm_.GotoIf(IsNull(object, wasm::kWasmAnyRef), &end_label,
230 BranchHint::kFalse, gasm_.Int32Constant(kResult));
231 }
232 // i31 is special in that the Smi check is the last thing to do.
233 if (to_rep == wasm::HeapType::kI31) {
234 // If earlier optimization passes reached the limit of possible graph
235 // transformations, we could DCHECK(object_can_be_i31) here.
236 result = object_can_be_i31 ? gasm_.IsSmi(object) : gasm_.Int32Constant(0);
237 break;
238 }
239 if (to_rep == wasm::HeapType::kEq) {
240 if (object_can_be_i31) {
241 gasm_.GotoIf(gasm_.IsSmi(object), &end_label, BranchHint::kFalse,
242 gasm_.Int32Constant(1));
243 }
244 result = gasm_.IsDataRefMap(gasm_.LoadMap(object));
245 break;
246 }
247 // array, struct, string: i31 fails.
248 if (object_can_be_i31) {
249 gasm_.GotoIf(gasm_.IsSmi(object), &end_label, BranchHint::kFalse,
250 gasm_.Int32Constant(0));
251 }
252 if (to_rep == wasm::HeapType::kArray) {
253 result = gasm_.HasInstanceType(object, WASM_ARRAY_TYPE);
254 break;
255 }
256 if (to_rep == wasm::HeapType::kStruct) {
257 result = gasm_.HasInstanceType(object, WASM_STRUCT_TYPE);
258 break;
259 }
260 if (to_rep == wasm::HeapType::kString ||
261 to_rep == wasm::HeapType::kExternString) {
262 Node* instance_type = gasm_.LoadInstanceType(gasm_.LoadMap(object));
263 result = gasm_.Uint32LessThan(instance_type,
264 gasm_.Uint32Constant(FIRST_NONSTRING_TYPE));
265 break;
266 }
267 UNREACHABLE();
268 } while (false);
269
270 DCHECK_NOT_NULL(result);
271 if (end_label.IsUsed()) {
272 gasm_.Goto(&end_label, result);
273 gasm_.Bind(&end_label);
274 result = end_label.PhiAt(0);
275 }
276
277 ReplaceWithValue(node, result, gasm_.effect(), gasm_.control());
278 node->Kill();
279 return Replace(result); // Meaningless argument.
280}

282Reduction WasmGCLowering::ReduceWasmTypeCast(Node* node) {
283 DCHECK_EQ(node->opcode(), IrOpcode::kWasmTypeCast);
284
285 Node* object = node->InputAt(0);
286 Node* rtt = node->InputAt(1);
287 Node* effect_input = NodeProperties::GetEffectInput(node);
288 Node* control_input = NodeProperties::GetControlInput(node);
289 auto config = OpParameter<WasmTypeCheckConfig>(node->op());
290 int rtt_depth = wasm::GetSubtypingDepth(module_, config.to.ref_index());
291 bool object_can_be_null = config.from.is_nullable();
292 bool object_can_be_i31 =
293 wasm::IsSubtypeOf(wasm::kWasmI31Ref.AsNonNull(), config.from, module_);
294
295 gasm_.InitializeEffectControl(effect_input, control_input);
296
297 auto end_label = gasm_.MakeLabel();
298 bool is_cast_from_any = config.from.is_reference_to(wasm::HeapType::kAny);
299
300 // If we are casting from any and null results in check failure, then the
301 // {IsDataRefMap} check below subsumes the null check. Otherwise, perform
302 // an explicit null check now.
303 if (object_can_be_null && (!is_cast_from_any || config.to.is_nullable())) {
304 Node* is_null = IsNull(object, wasm::kWasmAnyRef);
305 if (config.to.is_nullable()) {
306 gasm_.GotoIf(is_null, &end_label, BranchHint::kFalse);
307 } else if (!v8_flags.experimental_wasm_skip_null_checks) {
308 gasm_.TrapIf(is_null, TrapId::kTrapIllegalCast);
309 UpdateSourcePosition(gasm_.effect(), node);
310 }
311 }
312
313 if (object_can_be_i31) {
314 gasm_.TrapIf(gasm_.IsSmi(object), TrapId::kTrapIllegalCast);
315 UpdateSourcePosition(gasm_.effect(), node);
316 }
317
318 Node* map = gasm_.LoadMap(object);
319
320 if (module_->type(config.to.ref_index()).is_final) {
321 gasm_.TrapUnless(gasm_.TaggedEqual(map, rtt), TrapId::kTrapIllegalCast);
322 UpdateSourcePosition(gasm_.effect(), node);
323 gasm_.Goto(&end_label);
324 } else {
325 // First, check if types happen to be equal. This has been shown to give
326 // large speedups.
327 gasm_.GotoIf(gasm_.TaggedEqual(map, rtt), &end_label, BranchHint::kTrue);
328
329 // Check if map instance type identifies a wasm object.
330 if (is_cast_from_any) {
331 Node* is_wasm_obj = gasm_.IsDataRefMap(map);
332 gasm_.TrapUnless(is_wasm_obj, TrapId::kTrapIllegalCast);
333 UpdateSourcePosition(gasm_.effect(), node);
334 }
335
336 Node* type_info = gasm_.LoadWasmTypeInfo(map);
337 DCHECK_GE(rtt_depth, 0);
338 // If the depth of the rtt is known to be less that the minimum supertype
339 // array length, we can access the supertype without bounds-checking the
340 // supertype array.
341 if (static_cast<uint32_t>(rtt_depth) >= wasm::kMinimumSupertypeArraySize) {
342 Node* supertypes_length =
343 gasm_.BuildChangeSmiToIntPtr(gasm_.LoadImmutableFromObject(
344 MachineType::TaggedSigned(), type_info,
345 wasm::ObjectAccess::ToTagged(
346 WasmTypeInfo::kSupertypesLengthOffset)));
347 gasm_.TrapUnless(gasm_.UintLessThan(gasm_.IntPtrConstant(rtt_depth),
348 supertypes_length),
349 TrapId::kTrapIllegalCast);
350 UpdateSourcePosition(gasm_.effect(), node);
351 }
352
353 Node* maybe_match = gasm_.LoadImmutableFromObject(
354 MachineType::TaggedPointer(), type_info,
355 wasm::ObjectAccess::ToTagged(WasmTypeInfo::kSupertypesOffset +
356 kTaggedSize * rtt_depth));
357
358 gasm_.TrapUnless(gasm_.TaggedEqual(maybe_match, rtt),
359 TrapId::kTrapIllegalCast);
360 UpdateSourcePosition(gasm_.effect(), node);
361 gasm_.Goto(&end_label);
362 }
363
364 gasm_.Bind(&end_label);
365
366 ReplaceWithValue(node, object, gasm_.effect(), gasm_.control());
367 node->Kill();
368 return Replace(object);
369}

373
374 Node* object = node->InputAt(0);
375 Node* effect_input = NodeProperties::GetEffectInput(node);
376 Node* control_input = NodeProperties::GetControlInput(node);

378 const bool object_can_be_null = config.from.is_nullable();
379 const bool null_succeeds = config.to.is_nullable();

387

395 gasm_.TrapUnless(IsNull(object, config.from), TrapId::kTrapIllegalCast);
396 UpdateSourcePosition(gasm_.effect(), node);
397 break;
398 }

413 break;
414 }

417 gasm_.GotoIf(gasm_.IsSmi(object), &end_label, BranchHint::kFalse);

420 TrapId::kTrapIllegalCast);
421 UpdateSourcePosition(gasm_.effect(), node);

431 TrapId::kTrapIllegalCast);
432 UpdateSourcePosition(gasm_.effect(), node);
433 break;
434 }
435 if (to_rep == wasm::HeapType::kStruct) {

444 gasm_.TrapUnless(
445 gasm_.Uint32LessThan(instance_type,

447 TrapId::kTrapIllegalCast);
448 UpdateSourcePosition(gasm_.effect(), node);
449 break;
450 }

452 } while (false);
453
454 if (end_label.IsUsed()) {
455 gasm_.Goto(&end_label);
456 gasm_.Bind(&end_label);
457 }

464Reduction WasmGCLowering::ReduceAssertNotNull(Node* node) {
465 DCHECK_EQ(node->opcode(), IrOpcode::kAssertNotNull);
466 Node* effect = NodeProperties::GetEffectInput(node);
467 Node* control = NodeProperties::GetControlInput(node);
468 Node* object = NodeProperties::GetValueInput(node, 0);
469 gasm_.InitializeEffectControl(effect, control);
470 auto op_parameter = OpParameter<AssertNotNullParameters>(node->op());
471 // When able, implement a non-null assertion by loading from the object just
472 // after the map word. This will trap for null and be handled by the trap
473 // handler.
474 if (op_parameter.trap_id == TrapId::kTrapNullDereference) {
475 if (!v8_flags.experimental_wasm_skip_null_checks) {
476 // For supertypes of i31ref, we would need to check for i31ref anyway
477 // before loading from the object, so we might as well just check directly
478 // for null.
479 // For subtypes of externref, we use JS null, so we have to check
480 // explicitly.
481 if (null_check_strategy_ == NullCheckStrategy::kExplicit ||
482 wasm::IsSubtypeOf(wasm::kWasmI31Ref.AsNonNull(), op_parameter.type,
483 module_) ||
484 !op_parameter.type.use_wasm_null()) {
485 gasm_.TrapIf(IsNull(object, op_parameter.type), op_parameter.trap_id);
486 UpdateSourcePosition(gasm_.effect(), node);
487 } else {
488 static_assert(WasmStruct::kHeaderSize > kTaggedSize);
489 static_assert(WasmArray::kHeaderSize > kTaggedSize);
490 static_assert(WasmInternalFunction::kHeaderSize > kTaggedSize);
491 Node* trap_null = gasm_.LoadTrapOnNull(
492 MachineType::Int32(), object,
493 gasm_.IntPtrConstant(wasm::ObjectAccess::ToTagged(kTaggedSize)));
494 UpdateSourcePosition(trap_null, node);
495 }
496 }
497 } else {
498 gasm_.TrapIf(IsNull(object, op_parameter.type), op_parameter.trap_id);
499 UpdateSourcePosition(gasm_.effect(), node);
500 }
501
502 ReplaceWithValue(node, object, gasm_.effect(), gasm_.control());
503 node->Kill();
504 return Replace(object);
505}

553Reduction WasmGCLowering::ReduceWasmAnyConvertExtern(Node* node) {

560 auto end_label = gasm_.MakeLabel(MachineRepresentation::kTagged);
561 auto null_label = gasm_.MakeLabel();
562 auto smi_label = gasm_.MakeLabel();
563 auto int_to_smi_label = gasm_.MakeLabel();
564 auto heap_number_label = gasm_.MakeLabel();
565
566 gasm_.GotoIf(IsNull(input, wasm::kWasmExternRef), &null_label);
567 gasm_.GotoIf(gasm_.IsSmi(input), &smi_label);
568 Node* is_heap_number = gasm_.HasInstanceType(input, HEAP_NUMBER_TYPE);
569 gasm_.GotoIf(is_heap_number, &heap_number_label);
570 // For anything else, just pass through the value.
571 gasm_.Goto(&end_label, input);
572
573 gasm_.Bind(&null_label);
574 gasm_.Goto(&end_label, Null(wasm::kWasmNullRef));
575
576 // Canonicalize SMI.
577 gasm_.Bind(&smi_label);
578 if constexpr (SmiValuesAre31Bits()) {
579 gasm_.Goto(&end_label, input);
580 } else {
581 auto to_heap_number_label = gasm_.MakeLabel();
582 Node* int_value = gasm_.BuildChangeSmiToInt32(input);
583
584 // Convert to heap number if the int32 does not fit into an i31ref.
585 gasm_.GotoIf(
586 gasm_.Int32LessThan(gasm_.Int32Constant(kInt31MaxValue), int_value),
587 &to_heap_number_label);
588 gasm_.GotoIf(
589 gasm_.Int32LessThan(int_value, gasm_.Int32Constant(kInt31MinValue)),
590 &to_heap_number_label);
591 gasm_.Goto(&end_label, input);
592
593 gasm_.Bind(&to_heap_number_label);
594 Node* heap_number = gasm_.CallBuiltin(Builtin::kWasmInt32ToHeapNumber,
595 Operator::kPure, int_value);
596 gasm_.Goto(&end_label, heap_number);
597 }
598
599 // Convert HeapNumber to SMI if possible.
600 gasm_.Bind(&heap_number_label);
601 Node* float_value = gasm_.LoadFromObject(
602 MachineType::Float64(), input,
603 wasm::ObjectAccess::ToTagged(AccessBuilder::ForHeapNumberValue().offset));
604 // Check range of float value.
605 gasm_.GotoIf(
606 gasm_.Float64LessThan(float_value, gasm_.Float64Constant(kInt31MinValue)),
607 &end_label, input);
608 gasm_.GotoIf(
609 gasm_.Float64LessThan(gasm_.Float64Constant(kInt31MaxValue), float_value),
610 &end_label, input);
611 // Check if value is -0.
612 Node* is_minus_zero = nullptr;
613 if (mcgraph_->machine()->Is64()) {
614 Node* minus_zero = gasm_.Int64Constant(base::bit_cast<int64_t>(-0.0));
615 Node* float_bits = gasm_.BitcastFloat64ToInt64(float_value);
616 is_minus_zero = gasm_.Word64Equal(float_bits, minus_zero);
617 } else {
618 constexpr int32_t kMinusZeroLoBits = static_cast<int32_t>(0);
619 constexpr int32_t kMinusZeroHiBits = static_cast<int32_t>(1) << 31;
620 auto done = gasm_.MakeLabel(MachineRepresentation::kBit);
621
622 Node* value_lo = gasm_.Float64ExtractLowWord32(float_value);
623 gasm_.GotoIfNot(
624 gasm_.Word32Equal(value_lo, gasm_.Int32Constant(kMinusZeroLoBits)),
625 &done, gasm_.Int32Constant(0));
626 Node* value_hi = gasm_.Float64ExtractHighWord32(float_value);
627 gasm_.Goto(&done, gasm_.Word32Equal(value_hi,
628 gasm_.Int32Constant(kMinusZeroHiBits)));
629 gasm_.Bind(&done);
630 is_minus_zero = done.PhiAt(0);
631 }
632 gasm_.GotoIf(is_minus_zero, &end_label, input);
633 // Check if value is integral.
634 Node* int_value = gasm_.ChangeFloat64ToInt32(float_value);
635 gasm_.GotoIf(
636 gasm_.Float64Equal(float_value, gasm_.ChangeInt32ToFloat64(int_value)),
637 &int_to_smi_label);
638 gasm_.Goto(&end_label, input);
639
640 gasm_.Bind(&int_to_smi_label);
641 gasm_.Goto(&end_label, gasm_.BuildChangeInt32ToSmi(int_value));
642
643 gasm_.Bind(&end_label);
644 ReplaceWithValue(node, end_label.PhiAt(0), gasm_.effect(), gasm_.control());
645 node->Kill();
646 return Replace(end_label.PhiAt(0));
647}

870Reduction WasmGCLowering::ReduceStringPrepareForGetCodeunit(Node* node) {
871 DCHECK_EQ(node->opcode(), IrOpcode::kStringPrepareForGetCodeunit);
872 Node* effect = NodeProperties::GetEffectInput(node);
873 Node* control = NodeProperties::GetControlInput(node);
874 Node* original_string = NodeProperties::GetValueInput(node, 0);
875
876 gasm_.InitializeEffectControl(effect, control);
877
878 auto dispatch =
879 gasm_.MakeLoopLabel(MachineRepresentation::kTaggedPointer, // String.
880 MachineRepresentation::kWord32, // Instance type.
881 MachineRepresentation::kWord32); // Offset.
882 auto next = gasm_.MakeLabel(MachineRepresentation::kTaggedPointer, // String.
883 MachineRepresentation::kWord32, // Instance type.
884 MachineRepresentation::kWord32); // Offset.
885 auto direct_string =
886 gasm_.MakeLabel(MachineRepresentation::kTaggedPointer, // String.
887 MachineRepresentation::kWord32, // Instance type.
888 MachineRepresentation::kWord32); // Offset.
889
890 // These values will be used to replace the original node's projections.
891 // The first, "string", is either a SeqString or Tagged<Smi>(0) (in case of
892 // external string). Notably this makes it GC-safe: if that string moves, this
893 // pointer will be updated accordingly. The second, "offset", has full
894 // register width so that it can be used to store external pointers: for
895 // external strings, we add up the character backing store's base address and
896 // any slice offset. The third, "character width", is a shift width, i.e. it
897 // is 0 for one-byte strings, 1 for two-byte strings,
898 // kCharWidthBailoutSentinel for uncached external strings (for which
899 // "string"/"offset" are invalid and unusable).
900 auto done =
901 gasm_.MakeLabel(MachineRepresentation::kTagged, // String.
902 MachineType::PointerRepresentation(), // Offset.
903 MachineRepresentation::kWord32); // Character width.
904
905 Node* original_type = gasm_.LoadInstanceType(gasm_.LoadMap(original_string));
906 gasm_.Goto(&dispatch, original_string, original_type, gasm_.Int32Constant(0));
907
908 gasm_.Bind(&dispatch);
909 {
910 auto thin_string = gasm_.MakeLabel();
911 auto cons_string = gasm_.MakeLabel();
912
913 Node* string = dispatch.PhiAt(0);
914 Node* instance_type = dispatch.PhiAt(1);
915 Node* offset = dispatch.PhiAt(2);
916 static_assert(kIsIndirectStringTag == 1);
917 static constexpr int kIsDirectStringTag = 0;
918 gasm_.GotoIf(gasm_.Word32Equal(
919 gasm_.Word32And(instance_type, gasm_.Int32Constant(
920 kIsIndirectStringMask)),
921 gasm_.Int32Constant(kIsDirectStringTag)),
922 &direct_string, string, instance_type, offset);
923
924 // Handle indirect strings.
925 Node* string_representation = gasm_.Word32And(
926 instance_type, gasm_.Int32Constant(kStringRepresentationMask));
927 gasm_.GotoIf(gasm_.Word32Equal(string_representation,
928 gasm_.Int32Constant(kThinStringTag)),
929 &thin_string);
930 gasm_.GotoIf(gasm_.Word32Equal(string_representation,
931 gasm_.Int32Constant(kConsStringTag)),
932 &cons_string);
933
934 // Sliced string.
935 Node* new_offset = gasm_.Int32Add(
936 offset, gasm_.BuildChangeSmiToInt32(gasm_.LoadImmutableFromObject(
937 MachineType::TaggedSigned(), string,
938 TaggedOffset(AccessBuilder::ForSlicedStringOffset()))));
939 Node* parent = gasm_.LoadImmutableFromObject(
940 MachineType::TaggedPointer(), string,
941 TaggedOffset(AccessBuilder::ForSlicedStringParent()));
942 Node* parent_type = gasm_.LoadInstanceType(gasm_.LoadMap(parent));
943 gasm_.Goto(&next, parent, parent_type, new_offset);
944
945 // Thin string.
946 gasm_.Bind(&thin_string);
947 Node* actual = gasm_.LoadImmutableFromObject(
948 MachineType::TaggedPointer(), string,
949 TaggedOffset(AccessBuilder::ForThinStringActual()));
950 Node* actual_type = gasm_.LoadInstanceType(gasm_.LoadMap(actual));
951 // ThinStrings always reference (internalized) direct strings.
952 gasm_.Goto(&direct_string, actual, actual_type, offset);
953
954 // Flat cons string. (Non-flat cons strings are ruled out by
955 // string.as_wtf16.)
956 gasm_.Bind(&cons_string);
957 Node* first = gasm_.LoadImmutableFromObject(
958 MachineType::TaggedPointer(), string,
959 TaggedOffset(AccessBuilder::ForConsStringFirst()));
960 Node* first_type = gasm_.LoadInstanceType(gasm_.LoadMap(first));
961 gasm_.Goto(&next, first, first_type, offset);
962
963 gasm_.Bind(&next);
964 gasm_.Goto(&dispatch, next.PhiAt(0), next.PhiAt(1), next.PhiAt(2));
965 }
966
967 gasm_.Bind(&direct_string);
968 {
969 Node* string = direct_string.PhiAt(0);
970 Node* instance_type = direct_string.PhiAt(1);
971 Node* offset = direct_string.PhiAt(2);
972
973 Node* is_onebyte = gasm_.Word32And(
974 instance_type, gasm_.Int32Constant(kStringEncodingMask));
975 // Char width shift is 1 - (is_onebyte).
976 static_assert(kStringEncodingMask == 1 << 3);
977 Node* charwidth_shift =
978 gasm_.Int32Sub(gasm_.Int32Constant(1),
979 gasm_.Word32Shr(is_onebyte, gasm_.Int32Constant(3)));
980
981 auto external = gasm_.MakeLabel();
982 Node* string_representation = gasm_.Word32And(
983 instance_type, gasm_.Int32Constant(kStringRepresentationMask));
984 gasm_.GotoIf(gasm_.Word32Equal(string_representation,
985 gasm_.Int32Constant(kExternalStringTag)),
986 &external);
987
988 // Sequential string.
989 DCHECK_EQ(AccessBuilder::ForSeqOneByteStringCharacter().header_size,
990 AccessBuilder::ForSeqTwoByteStringCharacter().header_size);
991 const int chars_start_offset =
992 AccessBuilder::ForSeqOneByteStringCharacter().header_size;
993 Node* final_offset = gasm_.Int32Add(
994 gasm_.Int32Constant(wasm::ObjectAccess::ToTagged(chars_start_offset)),
995 gasm_.Word32Shl(offset, charwidth_shift));
996 gasm_.Goto(&done, string, gasm_.BuildChangeInt32ToIntPtr(final_offset),
997 charwidth_shift);
998
999 // External string.
1000 gasm_.Bind(&external);
1001 gasm_.GotoIf(
1002 gasm_.Word32And(instance_type,
1003 gasm_.Int32Constant(kUncachedExternalStringMask)),
1004 &done, string, gasm_.IntPtrConstant(0),
1005 gasm_.Int32Constant(kCharWidthBailoutSentinel));
1006 Node* resource = gasm_.BuildLoadExternalPointerFromObject(
1007 string, AccessBuilder::ForExternalStringResourceData().offset,
1008 kExternalStringResourceDataTag, gasm_.LoadRootRegister());
1009 Node* shifted_offset = gasm_.Word32Shl(offset, charwidth_shift);
1010 final_offset = gasm_.IntPtrAdd(
1011 resource, gasm_.BuildChangeInt32ToIntPtr(shifted_offset));
1012 gasm_.Goto(&done, gasm_.SmiConstant(0), final_offset, charwidth_shift);
1013 }
1014
1015 gasm_.Bind(&done);
1016 Node* base = done.PhiAt(0);
1017 Node* final_offset = done.PhiAt(1);
1018 Node* charwidth_shift = done.PhiAt(2);
1019
1020 Node* base_proj = NodeProperties::FindProjection(node, 0);
1021 Node* offset_proj = NodeProperties::FindProjection(node, 1);
1022 Node* charwidth_proj = NodeProperties::FindProjection(node, 2);
1023 if (base_proj) {
1024 ReplaceWithValue(base_proj, base, gasm_.effect(), gasm_.control());
1025 base_proj->Kill();
1026 }
1027 if (offset_proj) {
1028 ReplaceWithValue(offset_proj, final_offset, gasm_.effect(),
1029 gasm_.control());
1030 offset_proj->Kill();
1031 }
1032 if (charwidth_proj) {
1033 ReplaceWithValue(charwidth_proj, charwidth_shift, gasm_.effect(),
1034 gasm_.control());
1035 charwidth_proj->Kill();
1036 }
1037
1038 // Wire up the dangling end of the new effect chain.
1039 ReplaceWithValue(node, node, gasm_.effect(), gasm_.control());
1040
1041 node->Kill();
1042 return Replace(base);
1043}
