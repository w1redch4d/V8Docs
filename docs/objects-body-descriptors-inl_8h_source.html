68template <typename ObjectVisitor>
69void BodyDescriptorBase::IterateJSObjectBodyImpl(Tagged<Map> map,
70 Tagged<HeapObject> obj,
71 int start_offset,
72 int end_offset,
73 ObjectVisitor* v) {
74#ifdef V8_COMPRESS_POINTERS
75 static_assert(kEmbedderDataSlotSize == 2 * kTaggedSize);
76 int header_end_offset = JSObject::GetHeaderSize(map);
77 int inobject_fields_start_offset = map->GetInObjectPropertyOffset(0);
78 // We are always requested to process header and embedder fields.
79 DCHECK_LE(inobject_fields_start_offset, end_offset);
80 // Embedder fields are located between header and inobject properties.
81 if (header_end_offset < inobject_fields_start_offset) {
82 // There are embedder fields.
83 DCHECK_EQ(header_end_offset, JSObject::GetEmbedderFieldsStartOffset(map));
84 IteratePointers(obj, start_offset, header_end_offset, v);
85 for (int offset = header_end_offset; offset < inobject_fields_start_offset;
86 offset += kEmbedderDataSlotSize) {
87 IteratePointer(obj, offset + EmbedderDataSlot::kTaggedPayloadOffset, v);
88 v->VisitExternalPointer(
89 obj, obj->RawExternalPointerField(
90 offset + EmbedderDataSlot::kExternalPointerOffset,
91 kEmbedderDataSlotPayloadTag));
92 }
93 // Proceed processing inobject properties.
94 start_offset = inobject_fields_start_offset;
95 }
96#else
97 // We store raw aligned pointers as Smis, so it's safe to iterate the whole
98 // embedder field area as tagged slots.
99 static_assert(kEmbedderDataSlotSize == kTaggedSize);
100#endif
101 IteratePointers(obj, start_offset, end_offset, v);
102}

115// This is a BodyDescriptor helper for usage within JSAPIObjectWithEmbedderSlots
116// and JSSpecialObject. The class hierarchies are separate but
117// `kCppHeapWrappableOffset` is the same for both.
118class JSAPIObjectWithEmbedderSlotsOrJSSpecialObjectBodyDescriptor
119 : public BodyDescriptorBase {
120 public:
121 template <typename ObjectVisitor>
122 static inline void IterateJSAPIObjectWithEmbedderSlotsHeader(
123 Tagged<Map> map, Tagged<HeapObject> obj, int object_size,
124 ObjectVisitor* v) {
125 // Visit JSObject header.
126 IteratePointers(obj, JSObject::kPropertiesOrHashOffset,
127 JSObject::kEndOfStrongFieldsOffset, v);
128
129 // Visit JSAPIObjectWithEmbedderSlots or JSSpecialObject header.
130 static_assert(JSObject::kEndOfStrongFieldsOffset ==
131 JSAPIObjectWithEmbedderSlots::kCppHeapWrappableOffset);
132 static_assert(JSAPIObjectWithEmbedderSlots::kCppHeapWrappableOffset ==
133 JSSpecialObject::kCppHeapWrappableOffset);
134 static_assert(JSAPIObjectWithEmbedderSlots::kCppHeapWrappableOffsetEnd +
135 1 ==
136 JSAPIObjectWithEmbedderSlots::kHeaderSize);
137 v->VisitCppHeapPointer(
138 obj, obj->RawCppHeapPointerField(
139 JSAPIObjectWithEmbedderSlots::kCppHeapWrappableOffset));
140 }
141
142 template <typename ConcreteType, typename ObjectVisitor>
143 static inline void IterateJSAPIObjectWithEmbedderSlotsTail(
144 Tagged<Map> map, Tagged<HeapObject> obj, int object_size,
145 ObjectVisitor* v) {
146 // Visit the tail of JSObject with possible embedder fields and in-object
147 // properties. Note that embedder fields are processed in the JSObject base
148 // class as there's other object hierarchies that contain embedder fields as
149 // well.
150 IterateJSObjectBodyImpl(map, obj, ConcreteType::kHeaderSize, object_size,
151 v);
152 }
153
154 template <typename ConcreteType, typename ObjectVisitor>
155 static inline void IterateJSAPIObjectWithoutEmbedderSlotsTail(
156 Tagged<Map> map, Tagged<HeapObject> obj, int object_size,
157 ObjectVisitor* v) {
158 IterateJSObjectBodyWithoutEmbedderFieldsImpl(
159 map, obj, ConcreteType::kHeaderSize, object_size, v);
160 }
161
162 static constexpr int kHeaderSize = JSSpecialObject::kHeaderSize;
163
164 static_assert(JSAPIObjectWithEmbedderSlots::kHeaderSize ==
165 JSSpecialObject::kHeaderSize);
166 static_assert(Internals::kJSAPIObjectWithEmbedderSlotsHeaderSize ==
167 JSSpecialObject::kHeaderSize);
