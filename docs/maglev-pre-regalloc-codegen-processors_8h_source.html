53class MaxCallDepthProcessor {
54 public:
55 void PreProcessGraph(Graph* graph) {}
56 void PostProcessGraph(Graph* graph) {
57 graph->set_max_call_stack_args(max_call_stack_args_);
58 graph->set_max_deopted_stack_size(max_deopted_stack_size_);
59 }
60 BlockProcessResult PreProcessBasicBlock(BasicBlock* block) {
61 return BlockProcessResult::kContinue;
62 }
63 void PostPhiProcessing() {}

65 template <typename NodeT>
66 ProcessResult Process(NodeT* node, const ProcessingState& state) {
67 if constexpr (NodeT::kProperties.is_call() ||
68 NodeT::kProperties.needs_register_snapshot()) {
69 int node_stack_args = node->MaxCallStackArgs();
70 if constexpr (NodeT::kProperties.needs_register_snapshot()) {
71 // Pessimistically assume that we'll push all registers in deferred
72 // calls.
73 node_stack_args +=
74 kAllocatableGeneralRegisterCount + kAllocatableDoubleRegisterCount;
75 }
76 max_call_stack_args_ = std::max(max_call_stack_args_, node_stack_args);
77 }
78 if constexpr (NodeT::kProperties.can_eager_deopt()) {
79 UpdateMaxDeoptedStackSize(node->eager_deopt_info());
80 }
81 if constexpr (NodeT::kProperties.can_lazy_deopt()) {
82 UpdateMaxDeoptedStackSize(node->lazy_deopt_info());
83 }
84 return ProcessResult::kContinue;
85 }
86
87 private:
88 void UpdateMaxDeoptedStackSize(DeoptInfo* deopt_info) {
89 const DeoptFrame* deopt_frame = &deopt_info->top_frame();
90 int frame_size = 0;
91 if (deopt_frame->type() == DeoptFrame::FrameType::kInterpretedFrame) {
92 if (&deopt_frame->as_interpreted().unit() == last_seen_unit_) return;
93 last_seen_unit_ = &deopt_frame->as_interpreted().unit();
94 frame_size = deopt_frame->as_interpreted().unit().max_arguments() *
95 kSystemPointerSize;
96 }
97
98 do {
99 frame_size += ConservativeFrameSize(deopt_frame);
100 deopt_frame = deopt_frame->parent();
101 } while (deopt_frame != nullptr);
102 max_deopted_stack_size_ = std::max(frame_size, max_deopted_stack_size_);
103 }
104 int ConservativeFrameSize(const DeoptFrame* deopt_frame) {
105 switch (deopt_frame->type()) {
106 case DeoptFrame::FrameType::kInterpretedFrame: {
107 auto info = UnoptimizedFrameInfo::Conservative(
108 deopt_frame->as_interpreted().unit().parameter_count(),
109 deopt_frame->as_interpreted().unit().register_count());
110 return info.frame_size_in_bytes();
111 }
112 case DeoptFrame::FrameType::kConstructInvokeStubFrame: {
113 return FastConstructStubFrameInfo::Conservative().frame_size_in_bytes();
114 }
115 case DeoptFrame::FrameType::kInlinedArgumentsFrame: {
116 return std::max(
117 0,
118 static_cast<int>(
119 deopt_frame->as_inlined_arguments().arguments().size() -
120 deopt_frame->as_inlined_arguments().unit().parameter_count()) *
121 kSystemPointerSize);
122 }
123 case DeoptFrame::FrameType::kBuiltinContinuationFrame: {
124 // PC + FP + Closure + Params + Context
125 const RegisterConfiguration* config = RegisterConfiguration::Default();
126 auto info = BuiltinContinuationFrameInfo::Conservative(
127 deopt_frame->as_builtin_continuation().parameters().length(),
128 Builtins::CallInterfaceDescriptorFor(
129 deopt_frame->as_builtin_continuation().builtin_id()),
130 config);
131 return info.frame_size_in_bytes();
132 }
133 }
134 }
135
136 int max_call_stack_args_ = 0;
137 int max_deopted_stack_size_ = 0;
138 // Optimize UpdateMaxDeoptedStackSize to not re-calculate if it sees the same
139 // compilation unit multiple times in a row.
140 const MaglevCompilationUnit* last_seen_unit_ = nullptr;
141};

160 template <typename NodeT>
161 ProcessResult Process(NodeT* node, const ProcessingState& state) {
162 node->set_id(next_node_id_++);
163 LoopUsedNodes* loop_used_nodes = GetCurrentLoopUsedNodes();
164 if (loop_used_nodes && node->properties().is_call() &&
165 loop_used_nodes->header->has_state()) {
166 if (loop_used_nodes->first_call == kInvalidNodeId) {
167 loop_used_nodes->first_call = node->id();
168 }
169 loop_used_nodes->last_call = node->id();
170 }
171 MarkInputUses(node, state);
172 return ProcessResult::kContinue;

175 template <typename NodeT>
176 void MarkInputUses(NodeT* node, const ProcessingState& state) {
177 LoopUsedNodes* loop_used_nodes = GetCurrentLoopUsedNodes();
178 // Mark input uses in the same order as inputs are assigned in the register
179 // allocator (see StraightForwardRegisterAllocator::AssignInputs).
180 node->ForAllInputsInRegallocAssignmentOrder(
181 [&](NodeBase::InputAllocationPolicy, Input* input) {
182 MarkUse(input->node(), node->id(), input, loop_used_nodes);
183 });
184 if constexpr (NodeT::kProperties.can_eager_deopt()) {
185 MarkCheckpointNodes(node, node->eager_deopt_info(), loop_used_nodes,
186 state);
187 }
188 if constexpr (NodeT::kProperties.can_lazy_deopt()) {
189 MarkCheckpointNodes(node, node->lazy_deopt_info(), loop_used_nodes,
190 state);
191 }
192 }

199 // Specialize the two unconditional jumps to extend their Phis' inputs' live
200 // ranges.
201
202 void MarkInputUses(JumpLoop* node, const ProcessingState& state) {
203 int predecessor_id = state.block()->predecessor_id();
204 BasicBlock* target = node->target();
205 uint32_t use = node->id();
206
207 DCHECK(!loop_used_nodes_.empty());
208 LoopUsedNodes loop_used_nodes = std::move(loop_used_nodes_.back());
209 loop_used_nodes_.pop_back();
210
211 LoopUsedNodes* outer_loop_used_nodes = GetCurrentLoopUsedNodes();
212
213 if (target->has_phi()) {
214 for (Phi* phi : *target->phis()) {
215 DCHECK(phi->is_used());
216 ValueNode* input = phi->input(predecessor_id).node();
217 MarkUse(input, use, &phi->input(predecessor_id), outer_loop_used_nodes);
218 }
219 }
220
221 DCHECK_EQ(loop_used_nodes.header, target);
222 if (!loop_used_nodes.used_nodes.empty()) {
223 // Try to avoid unnecessary reloads or spills across the back-edge based
224 // on use positions and calls inside the loop.
225 ZonePtrList<ValueNode>& reload_hints =
226 loop_used_nodes.header->reload_hints();
227 ZonePtrList<ValueNode>& spill_hints =
228 loop_used_nodes.header->spill_hints();
229 for (auto p : loop_used_nodes.used_nodes) {
230 // If the node is used before the first call and after the last call,
231 // keep it in a register across the back-edge.
232 if (p.second.first_register_use != kInvalidNodeId &&
233 (loop_used_nodes.first_call == kInvalidNodeId ||
234 (p.second.first_register_use <= loop_used_nodes.first_call &&
235 p.second.last_register_use > loop_used_nodes.last_call))) {
236 reload_hints.Add(p.first, compilation_info_->zone());
237 }
238 // If the node is not used, or used after the first call and before the
239 // last call, keep it spilled across the back-edge.
240 if (p.second.first_register_use == kInvalidNodeId ||
241 (loop_used_nodes.first_call != kInvalidNodeId &&
242 p.second.first_register_use > loop_used_nodes.first_call &&
243 p.second.last_register_use <= loop_used_nodes.last_call)) {
244 spill_hints.Add(p.first, compilation_info_->zone());
245 }
246 }
247
248 // Uses of nodes in this loop may need to propagate to an outer loop, so
249 // that they're lifetime is extended there too.
250 // TODO(leszeks): We only need to extend the lifetime in one outermost
251 // loop, allow nodes to be "moved" between lifetime extensions.
252 base::Vector<Input> used_node_inputs =
253 compilation_info_->zone()->AllocateVector<Input>(
254 loop_used_nodes.used_nodes.size());
255 int i = 0;
256 for (auto& [used_node, info] : loop_used_nodes.used_nodes) {
257 Input* input = new (&used_node_inputs[i++]) Input(used_node);
258 MarkUse(used_node, use, input, outer_loop_used_nodes);
259 }
260 node->set_used_nodes(used_node_inputs);
261 }
262 }
263 void MarkInputUses(Jump* node, const ProcessingState& state) {
264 MarkJumpInputUses(node->id(), node->target(), state);

266 void MarkInputUses(CheckpointedJump* node, const ProcessingState& state) {

269 void MarkJumpInputUses(uint32_t use, BasicBlock* target,
270 const ProcessingState& state) {
271 int i = state.block()->predecessor_id();
272 if (!target->has_phi()) return;
273 LoopUsedNodes* loop_used_nodes = GetCurrentLoopUsedNodes();
274 Phi::List& phis = *target->phis();
275 for (auto it = phis.begin(); it != phis.end();) {
276 Phi* phi = *it;
277 if (!phi->is_used()) {
278 // Skip unused phis -- we're processing phis out of order with the dead
279 // node sweeping processor, so we will still observe unused phis here.
280 // We can eagerly remove them while we're at it so that the dead node
281 // sweeping processor doesn't have to revisit them.
282 it = phis.RemoveAt(it);
283 } else {
284 ValueNode* input = phi->input(i).node();
285 MarkUse(input, use, &phi->input(i), loop_used_nodes);
286 ++it;
287 }
288 }
289 }

310 void MarkUse(ValueNode* node, uint32_t use_id, InputLocation* input,
311 LoopUsedNodes* loop_used_nodes) {
312 DCHECK(!node->Is<Identity>());
313
314 node->record_next_use(use_id, input);
315
316 // If we are in a loop, loop_used_nodes is non-null. In this case, check if
317 // the incoming node is from outside the loop, and make sure to extend its
318 // lifetime to the loop end if yes.
319 if (loop_used_nodes) {
320 // If the node's id is smaller than the smallest id inside the loop, then
321 // it must have been created before the loop. This means that it's alive
322 // on loop entry, and therefore has to be alive across the loop back edge
323 // too.
324 if (node->id() < loop_used_nodes->header->first_id()) {
325 auto [it, info] = loop_used_nodes->used_nodes.emplace(
326 node, NodeUse{kInvalidNodeId, kInvalidNodeId});
327 if (input->operand().IsUnallocated()) {
328 const auto& operand =
329 compiler::UnallocatedOperand::cast(input->operand());
330 if (operand.HasRegisterPolicy() || operand.HasFixedRegisterPolicy() ||
331 operand.HasFixedFPRegisterPolicy()) {
332 if (it->second.first_register_use == kInvalidNodeId) {
333 it->second.first_register_use = use_id;
334 }
335 it->second.last_register_use = use_id;
336 }
337 }
338 }
339 }
