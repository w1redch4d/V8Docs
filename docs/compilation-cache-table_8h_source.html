81// A lookup result from the compilation cache for scripts. There are three
82// possible states:
83//
84// 1. Cache miss: script and toplevel_sfi are both null.
85// 2. Cache hit: script and toplevel_sfi are both non-null. toplevel_sfi is
86// guaranteed to be compiled, and to stay compiled while this lookup result
87// instance is alive.
88// 3. Partial cache hit: script is non-null, but toplevel_sfi is null. The
89// script may contain an uncompiled toplevel SharedFunctionInfo.
90class CompilationCacheScriptLookupResult {
91 public:
92 MaybeHandle<Script> script() const { return script_; }
93 MaybeHandle<SharedFunctionInfo> toplevel_sfi() const { return toplevel_sfi_; }
94 IsCompiledScope is_compiled_scope() const { return is_compiled_scope_; }

127 // Eval code only gets cached after a second probe for the
128 // code object. To do so, on first "put" only a hash identifying the
129 // source is entered into the cache, mapping it to a lifetime count of
130 // the hash. On each call to Age all such lifetimes get reduced, and
131 // removed once they reach zero. If a second put is called while such
132 // a hash is live in the cache, the hash gets replaced by an actual
133 // cache entry. Age also removes stale live entries from the cache.
134 // Such entries are identified by SharedFunctionInfos pointing to
135 // either the recompilation stub, or to "old" code. This avoids memory
136 // leaks due to premature caching of eval strings that are
137 // never needed later.
