38int LiftoffAssembler::NextSpillOffset(ValueKind kind, int top_spill_offset) {
39 int offset = top_spill_offset + SlotSizeForType(kind);
40 if (NeedsAlignment(kind)) {
41 offset = RoundUp(offset, SlotSizeForType(kind));
42 }
43 return offset;
44}

46int LiftoffAssembler::NextSpillOffset(ValueKind kind) {
47 return NextSpillOffset(kind, TopSpillOffset());

50int LiftoffAssembler::TopSpillOffset() const {
51 return cache_state_.stack_state.empty()
52 ? StaticStackFrameSize()

56void LiftoffAssembler::PushRegister(ValueKind kind, LiftoffRegister reg) {
57 DCHECK_EQ(reg_class_for(kind), reg.reg_class());
58 cache_state_.inc_used(reg);
59 cache_state_.stack_state.emplace_back(kind, reg, NextSpillOffset(kind));
60}

62// Assumes that the exception is in {kReturnRegister0}. This is where the
63// exception is stored by the unwinder after a throwing call.
64void LiftoffAssembler::PushException() {
65 LiftoffRegister reg{kReturnRegister0};
66 // This is used after a call, so {kReturnRegister0} is not used yet.
67 DCHECK(cache_state_.is_free(reg));
68 cache_state_.inc_used(reg);
69 cache_state_.stack_state.emplace_back(kRef, reg, NextSpillOffset(kRef));

81void LiftoffAssembler::LoadToFixedRegister(VarState slot, LiftoffRegister reg) {
82 DCHECK(slot.is_const() || slot.is_stack());
83 if (slot.is_const()) {
84 LoadConstant(reg, slot.constant());
85 } else {
86 Fill(reg, slot.offset(), slot.kind());
87 }

90void LiftoffAssembler::PopToFixedRegister(LiftoffRegister reg) {
91 DCHECK(!cache_state_.stack_state.empty());
92 VarState slot = cache_state_.stack_state.back();
93 cache_state_.stack_state.pop_back();
94 if (V8_LIKELY(slot.is_reg())) {
95 cache_state_.dec_used(slot.reg());
96 if (slot.reg() == reg) return;
97 if (cache_state_.is_used(reg)) SpillRegister(reg);
98 Move(reg, slot.reg(), slot.kind());
99 return;
100 }
101 if (cache_state_.is_used(reg)) SpillRegister(reg);
102 LoadToFixedRegister(slot, reg);
103}

112void LiftoffAssembler::LoadSmiAsInt32(LiftoffRegister dst, Register src_addr,
113 int32_t offset) {
114 if constexpr (SmiValuesAre32Bits()) {
115#if V8_TARGET_LITTLE_ENDIAN
116 DCHECK_EQ(kSmiShiftSize + kSmiTagSize, 4 * kBitsPerByte);
117 offset += 4;
118#endif
119 Load(dst, src_addr, no_reg, offset, LoadType::kI32Load);
120 } else {
121 DCHECK(SmiValuesAre31Bits());
122 Load(dst, src_addr, no_reg, offset, LoadType::kI32Load);
123 emit_i32_sari(dst.gp(), dst.gp(), kSmiTagSize);
124 }

222template <void (LiftoffAssembler::*op)(Register, Register, Register)>
223void EmitI64IndependentHalfOperation(LiftoffAssembler* assm,
224 LiftoffRegister dst, LiftoffRegister lhs,
225 LiftoffRegister rhs) {
226 // If {dst.low_gp()} does not overlap with {lhs.high_gp()} or {rhs.high_gp()},
227 // just first compute the lower half, then the upper half.
228 if (dst.low() != lhs.high() && dst.low() != rhs.high()) {
229 (assm->*op)(dst.low_gp(), lhs.low_gp(), rhs.low_gp());
230 (assm->*op)(dst.high_gp(), lhs.high_gp(), rhs.high_gp());
231 return;
232 }
233 // If {dst.high_gp()} does not overlap with {lhs.low_gp()} or {rhs.low_gp()},
234 // we can compute this the other way around.
235 if (dst.high() != lhs.low() && dst.high() != rhs.low()) {
236 (assm->*op)(dst.high_gp(), lhs.high_gp(), rhs.high_gp());
237 (assm->*op)(dst.low_gp(), lhs.low_gp(), rhs.low_gp());
238 return;
239 }
240 // Otherwise, we need a temporary register.
241 Register tmp = assm->GetUnusedRegister(kGpReg, LiftoffRegList{lhs, rhs}).gp();
242 (assm->*op)(tmp, lhs.low_gp(), rhs.low_gp());
243 (assm->*op)(dst.high_gp(), lhs.high_gp(), rhs.high_gp());
244 assm->Move(dst.low_gp(), tmp, kI32);
245}
246
247template <void (LiftoffAssembler::*op)(Register, Register, int32_t)>
248void EmitI64IndependentHalfOperationImm(LiftoffAssembler* assm,
249 LiftoffRegister dst,
250 LiftoffRegister lhs, int64_t imm) {
251 int32_t low_word = static_cast<int32_t>(imm);
252 int32_t high_word = static_cast<int32_t>(imm >> 32);
253 // If {dst.low_gp()} does not overlap with {lhs.high_gp()},
254 // just first compute the lower half, then the upper half.
255 if (dst.low() != lhs.high()) {
256 (assm->*op)(dst.low_gp(), lhs.low_gp(), low_word);
257 (assm->*op)(dst.high_gp(), lhs.high_gp(), high_word);
258 return;
259 }
260 // If {dst.high_gp()} does not overlap with {lhs.low_gp()},
261 // we can compute this the other way around.
262 if (dst.high() != lhs.low()) {
263 (assm->*op)(dst.high_gp(), lhs.high_gp(), high_word);
264 (assm->*op)(dst.low_gp(), lhs.low_gp(), low_word);
265 return;
266 }
267 // Otherwise, we need a temporary register.
268 Register tmp = assm->GetUnusedRegister(kGpReg, LiftoffRegList{lhs}).gp();
269 (assm->*op)(tmp, lhs.low_gp(), low_word);
270 (assm->*op)(dst.high_gp(), lhs.high_gp(), high_word);
271 assm->Move(dst.low_gp(), tmp, kI32);
272}
