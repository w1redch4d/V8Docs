22// Abstraction over the stack. Supports handling of:
23// - native stack;
24// - ASAN/MSAN;
25// - SafeStack: https://releases.llvm.org/10.0.0/tools/clang/docs/SafeStack.html
26//
27// Stacks grow down, so throughout this class "start" refers the highest
28// address of the stack, and top/marker the lowest.
29//
30// TODO(chromium:1056170): Consider adding a component that keeps track
31// of relevant GC stack regions where interesting pointers can be found.

113 // Stack segments that may contain pointers and should be scanned.
114 struct Segment {
115 // The start and top of the stack. It must be sp <= top <= start.
116 // The top pointer is generally a marker that signals the end of the
117 // interesting stack region in which on-heap pointers can be found.
118 const void* start = nullptr;
119 const void* top = nullptr;
120
121#ifdef V8_USE_ADDRESS_SANITIZER
122 // The start of ASAN's fake stack.
123 const void* asan_fake_stack = nullptr;
124#endif // V8_USE_ADDRESS_SANITIZER
125
126#ifdef V8_USE_SAFE_STACK
127 // Start and top for the unsafe stack that is used in clang with
128 // -fsanitizer=safe-stack.
129 // It must be unsafe_sp <= unsafe_stack_top <= unsafe_stack_start.
130 // Notice that the terms "start" and "top" have here a different meaning in
131 // the terminology used in this feature's documentation.
132 const void* unsafe_stack_start = nullptr;
133 const void* unsafe_stack_top = nullptr;
134#endif // V8_USE_SAFE_STACK
135
136 Segment() = default;
137 Segment(const void* stack_start, const void* stack_top)
138 : start(stack_start), top(stack_top) {
139#ifdef V8_USE_ADDRESS_SANITIZER
140 asan_fake_stack = __asan_get_current_fake_stack();
141#endif // V8_USE_ADDRESS_SANITIZER
142#ifdef V8_USE_SAFE_STACK
143 unsafe_stack_start = __builtin___get_unsafe_stack_top();
144 unsafe_stack_top = __builtin___get_unsafe_stack_ptr();
145#endif // V8_USE_SAFE_STACK
146 }
147 };

157 template <typename Callback>
158 static void SetMarkerAndCallbackImpl(Stack* stack, void* argument,
159 const void* stack_end) {
160 Segment previous_segment = stack->current_segment_;
161 stack->current_segment_.top = stack_end;
162#ifdef V8_USE_SAFE_STACK
163 stack->current_segment_.unsafe_stack_top =
164 __builtin___get_unsafe_stack_ptr();
165#endif // V8_USE_SAFE_STACK
166 Callback* callback = static_cast<Callback*>(argument);
167 (*callback)();
168 stack->current_segment_ = previous_segment;
169 }
170
171 template <typename Callback>
172 static void SetMarkerForBackgroundThreadAndCallbackImpl(
173 Stack* stack, void* argument, const void* stack_end) {
174 DCHECK(IsOnCurrentStack(stack_end));
175 auto [thread, callback] =
176 *static_cast<std::pair<ThreadId, Callback*>*>(argument);
177 auto& background_stacks = stack->background_stacks_;
178 Segment previous_segment;
179 {
180 v8::base::SpinningMutexGuard guard(&stack->lock_);
181 if (auto it = background_stacks.find(thread);
182 it != background_stacks.end()) {
183 previous_segment = it->second;
184 DCHECK_NOT_NULL(previous_segment.top);
185 } else {
186 DCHECK_NULL(previous_segment.top);
187 }
188 // This implicitly uses the current values (if applicable) for:
189 // - asan_fake_start
190 // - unsafe stack start
191 // - unsafe stack top
192 background_stacks[thread] =
193 Segment(v8::base::Stack::GetStackStart(), stack_end);
194 }
195 (*callback)();
196 {
197 v8::base::SpinningMutexGuard guard(&stack->lock_);
198 if (previous_segment.top)
199 background_stacks[thread] = previous_segment;
200 else
201 background_stacks.erase(thread);
202 }
203 }
204
205 Segment current_segment_;
206
207 mutable v8::base::SpinningMutex lock_;
208 std::map<ThreadId, Segment> background_stacks_;
209};
