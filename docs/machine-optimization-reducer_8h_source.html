49// ******************************** OVERVIEW ********************************
50//
51// The MachineOptimizationAssembler performs basic optimizations on low-level
52// operations that can be performed on-the-fly, without requiring type analysis
53// or analyzing uses. It largely corresponds to MachineOperatorReducer in
54// sea-of-nodes Turbofan.
55//
56// These peephole optimizations are typically very local: they based on the
57// immediate inputs of an operation, we try to constant-fold or strength-reduce
58// the operation.
59//
60// Typical examples include:
61//
62// * Reducing `a == a` to `1`
63//
64// * Reducing `a + 0` to `a`
65//
66// * Reducing `a * 2^k` to `a << k`
67//

78// Represents an operation of the form `(source & mask) == masked_value`.
79// where each bit set in masked_value also has to be set in mask.
80struct BitfieldCheck {
81 V<Word> const source;
82 uint32_t const mask;
83 uint32_t const masked_value;
84 bool const truncate_from_64_bit;
85
86 BitfieldCheck(V<Word> source, uint32_t mask, uint32_t masked_value,
87 bool truncate_from_64_bit)
88 : source(source),
89 mask(mask),
90 masked_value(masked_value),
91 truncate_from_64_bit(truncate_from_64_bit) {
92 CHECK_EQ(masked_value & ~mask, 0);
93 }
94
95 static std::optional<BitfieldCheck> Detect(const OperationMatcher& matcher,
96 const Graph& graph,
97 V<Word> index) {
98 // There are two patterns to check for here:
99 // 1. Single-bit checks: `(val >> shift) & 1`, where:
100 // - the shift may be omitted, and/or
101 // - the result may be truncated from 64 to 32
102 // 2. Equality checks: `(val & mask) == expected`, where:
103 // - val may be truncated from 64 to 32 before masking (see
104 // ReduceWordEqualForConstantRhs)
105 const Operation& op = graph.Get(index);
106 if (const ComparisonOp* equal = op.TryCast<Opmask::kWord32Equal>()) {
107 if (const WordBinopOp* left_and =
108 graph.Get(equal->left<Word32>())
109 .TryCast<Opmask::kWord32BitwiseAnd>()) {
110 uint32_t mask;
111 uint32_t masked_value;
112 if (matcher.MatchIntegralWord32Constant(left_and->right(), &mask) &&
113 matcher.MatchIntegralWord32Constant(equal->right<Word32>(),
114 &masked_value)) {
115 if ((masked_value & ~mask) != 0) return std::nullopt;
116 if (const ChangeOp* truncate =
117 graph.Get(left_and->left())
118 .TryCast<Opmask::kTruncateWord64ToWord32>()) {
119 return BitfieldCheck{truncate->input<Word64>(), mask, masked_value,
120 true};
121 } else {
122 return BitfieldCheck{left_and->left(), mask, masked_value, false};
123 }
124 }
125 }
126 } else if (const ChangeOp* truncate =
127 op.TryCast<Opmask::kTruncateWord64ToWord32>()) {
128 return TryDetectShiftAndMaskOneBit<Word64>(matcher,
129 truncate->input<Word64>());
130 } else {
131 return TryDetectShiftAndMaskOneBit<Word32>(matcher, index);
132 }
133 return std::nullopt;
134 }
135
136 std::optional<BitfieldCheck> TryCombine(const BitfieldCheck& other) {
137 if (source != other.source ||
138 truncate_from_64_bit != other.truncate_from_64_bit) {
139 return std::nullopt;
140 }
141 uint32_t overlapping_bits = mask & other.mask;
142 // It would be kind of strange to have any overlapping bits, but they can be
143 // allowed as long as they don't require opposite values in the same
144 // positions.
145 if ((masked_value & overlapping_bits) !=
146 (other.masked_value & overlapping_bits)) {
147 return std::nullopt;
148 }
149 return BitfieldCheck{source, mask | other.mask,
150 masked_value | other.masked_value,
151 truncate_from_64_bit};
152 }
153
154 private:
155 template <typename WordType>
156 static std::optional<BitfieldCheck> TryDetectShiftAndMaskOneBit(
157 const OperationMatcher& matcher, V<Word> index) {
158 constexpr WordRepresentation Rep = V<WordType>::rep;
159 // Look for the pattern `(val >> shift) & 1`. The shift may be omitted.
160 V<WordType> value;
161 uint64_t constant;
162 if (matcher.MatchBitwiseAndWithConstant(index, &value, &constant, Rep) &&
163 constant == 1) {
164 V<WordType> input;
165 if (int shift_amount;
166 matcher.MatchConstantRightShift(value, &input, Rep, &shift_amount) &&
167 shift_amount >= 0 && shift_amount < 32) {
168 uint32_t mask = 1 << shift_amount;
169 return BitfieldCheck{input, mask, mask,
170 Rep == WordRepresentation::Word64()};
171 }
172 return BitfieldCheck{value, 1, 1, Rep == WordRepresentation::Word64()};
173 }
174 return std::nullopt;
175 }
176};

198 V<Untagged> REDUCE(Change)(V<Untagged> input, ChangeOp::Kind kind,
199 ChangeOp::Assumption assumption,
200 RegisterRepresentation from,
201 RegisterRepresentation to) {
202 if (ShouldSkipOptimizationStep()) {
203 return Next::ReduceChange(input, kind, assumption, from, to);
204 }
205 using Kind = ChangeOp::Kind;
206 if (from == WordRepresentation::Word32()) {
207 input = TryRemoveWord32ToWord64Conversion(V<Word64>::Cast(input));
208 }
209 if (uint64_t value;
210 from.IsWord() && matcher_.MatchIntegralWordConstant(
211 input, WordRepresentation(from), &value)) {
212 using Rep = RegisterRepresentation;
213 switch (multi(kind, from, to)) {
214 case multi(Kind::kSignExtend, Rep::Word32(), Rep::Word64()):
215 return __ Word64Constant(int64_t{static_cast<int32_t>(value)});
216 case multi(Kind::kZeroExtend, Rep::Word32(), Rep::Word64()):
217 case multi(Kind::kBitcast, Rep::Word32(), Rep::Word64()):
218 return __ Word64Constant(uint64_t{static_cast<uint32_t>(value)});
219 case multi(Kind::kBitcast, Rep::Word32(), Rep::Float32()):
220 return __ Float32Constant(
221 i::Float32::FromBits(static_cast<uint32_t>(value)));
222 case multi(Kind::kBitcast, Rep::Word64(), Rep::Float64()):
223 return __ Float64Constant(i::Float64::FromBits(value));
224 case multi(Kind::kSignedToFloat, Rep::Word32(), Rep::Float64()):
225 return __ Float64Constant(
226 static_cast<double>(static_cast<int32_t>(value)));
227 case multi(Kind::kSignedToFloat, Rep::Word64(), Rep::Float64()):
228 return __ Float64Constant(
229 static_cast<double>(static_cast<int64_t>(value)));
230 case multi(Kind::kUnsignedToFloat, Rep::Word32(), Rep::Float64()):
231 return __ Float64Constant(
232 static_cast<double>(static_cast<uint32_t>(value)));
233 case multi(Kind::kTruncate, Rep::Word64(), Rep::Word32()):
234 return __ Word32Constant(static_cast<uint32_t>(value));
235 default:
236 break;
237 }
238 }
239 if (i::Float32 value; from == RegisterRepresentation::Float32() &&
240 matcher_.MatchFloat32Constant(input, &value)) {
241 if (kind == Kind::kFloatConversion &&
242 to == RegisterRepresentation::Float64()) {
243 return __ Float64Constant(value.get_scalar());
244 }
245 if (kind == Kind::kBitcast && to == WordRepresentation::Word32()) {
246 return __ Word32Constant(value.get_bits());
247 }
248 }
249 if (i::Float64 value; from == RegisterRepresentation::Float64() &&
250 matcher_.MatchFloat64Constant(input, &value)) {
251 if (kind == Kind::kFloatConversion &&
252 to == RegisterRepresentation::Float32()) {
253 return __ Float32Constant(DoubleToFloat32_NoInline(value.get_scalar()));
254 }
255 if (kind == Kind::kBitcast && to == WordRepresentation::Word64()) {
256 return __ Word64Constant(base::bit_cast<uint64_t>(value));
257 }
258 if (kind == Kind::kSignedFloatTruncateOverflowToMin) {
259 double truncated = std::trunc(value.get_scalar());
260 if (to == WordRepresentation::Word64()) {
261 int64_t result = std::numeric_limits<int64_t>::min();
262 if (truncated >= std::numeric_limits<int64_t>::min() &&
263 truncated <= kMaxDoubleRepresentableInt64) {
264 result = static_cast<int64_t>(truncated);
265 }
266 return __ Word64Constant(result);
267 }
268 if (to == WordRepresentation::Word32()) {
269 int32_t result = std::numeric_limits<int32_t>::min();
270 if (truncated >= std::numeric_limits<int32_t>::min() &&
271 truncated <= std::numeric_limits<int32_t>::max()) {
272 result = static_cast<int32_t>(truncated);
273 }
274 return __ Word32Constant(result);
275 }
276 }
277 if (kind == Kind::kJSFloatTruncate &&
278 to == WordRepresentation::Word32()) {
279 return __ Word32Constant(DoubleToInt32_NoInline(value.get_scalar()));
280 }
281 if (kind == Kind::kExtractHighHalf) {
282 DCHECK_EQ(to, RegisterRepresentation::Word32());
283 return __ Word32Constant(static_cast<uint32_t>(value.get_bits() >> 32));
284 }
285 if (kind == Kind::kExtractLowHalf) {
286 DCHECK_EQ(to, RegisterRepresentation::Word32());
287 return __ Word32Constant(static_cast<uint32_t>(value.get_bits()));
288 }
289 }
290 if (float value; from == RegisterRepresentation::Float32() &&
291 matcher_.MatchFloat32Constant(input, &value)) {
292 if (kind == Kind::kFloatConversion &&
293 to == RegisterRepresentation::Float64()) {
294 return __ Float64Constant(value);
295 }
296 }
297
298 const Operation& input_op = matcher_.Get(input);
299 if (const ChangeOp* change_op = input_op.TryCast<ChangeOp>()) {
300 if (change_op->from == to && change_op->to == from &&
301 change_op->IsReversibleBy(kind, signalling_nan_possible)) {
302 return change_op->input();
303 }
304 }
305 return Next::ReduceChange(input, kind, assumption, from, to);
306 }

322 V<Any> REDUCE(TaggedBitcast)(V<Any> input, RegisterRepresentation from,
323 RegisterRepresentation to,
324 TaggedBitcastOp::Kind kind) {
325 if (ShouldSkipOptimizationStep()) {
326 return Next::ReduceTaggedBitcast(input, from, to, kind);
327 }
328 // A Tagged -> Untagged -> Tagged sequence can be short-cut.
329 // An Untagged -> Tagged -> Untagged sequence however cannot be removed,
330 // because the GC might have modified the pointer.
331 if (auto* input_bitcast = matcher_.TryCast<TaggedBitcastOp>(input)) {
332 if (all_of(input_bitcast->to, from) ==
333 RegisterRepresentation::WordPtr() &&
334 all_of(input_bitcast->from, to) == RegisterRepresentation::Tagged()) {
335 return input_bitcast->input();
336 }
337 }
338 // An Untagged -> Smi -> Untagged sequence can be short-cut.
339 if (auto* input_bitcast = matcher_.TryCast<TaggedBitcastOp>(input);
340 input_bitcast && to.IsWord() &&
341 (kind == TaggedBitcastOp::Kind::kSmi ||
342 input_bitcast->kind == TaggedBitcastOp::Kind::kSmi)) {
343 if (input_bitcast->from == to) return input_bitcast->input();
344 if (input_bitcast->from == RegisterRepresentation::Word32()) {
345 DCHECK_EQ(to, RegisterRepresentation::Word64());
346 return __ BitcastWord32ToWord64(input_bitcast->input());
347 }
348 DCHECK(input_bitcast->from == RegisterRepresentation::Word64() &&
349 to == RegisterRepresentation::Word32());
350 return __ TruncateWord64ToWord32(input_bitcast->input());
351 }
352 // Try to constant-fold TaggedBitcast from Word Constant to Word.
353 if (to.IsWord()) {
354 if (const ConstantOp* cst = matcher_.TryCast<ConstantOp>(input)) {
355 if (cst->kind == ConstantOp::Kind::kWord32 ||
356 cst->kind == ConstantOp::Kind::kWord64) {
357 if (to == RegisterRepresentation::Word64()) {
358 return __ Word64Constant(cst->integral());
359 } else {
360 DCHECK_EQ(to, RegisterRepresentation::Word32());
361 return __ Word32Constant(static_cast<uint32_t>(cst->integral()));
362 }
363 }
364 }
365 }
366 if (const ConstantOp* cst = matcher_.TryCast<ConstantOp>(input)) {
367 // Try to constant-fold Word constant -> Tagged (Smi).
368 if (cst->IsIntegral() && to == RegisterRepresentation::Tagged()) {
369 if (Smi::IsValid(cst->integral())) {
370 return __ SmiConstant(
371 i::Tagged<Smi>(static_cast<intptr_t>(cst->integral())));
372 }
373 }
374 // Try to constant-fold Smi -> Untagged.
375 if (cst->kind == ConstantOp::Kind::kSmi) {
376 if (to == RegisterRepresentation::Word32()) {
377 return __ Word32Constant(static_cast<uint32_t>(cst->smi().ptr()));
378 } else if (to == RegisterRepresentation::Word64()) {
379 return __ Word64Constant(static_cast<uint64_t>(cst->smi().ptr()));
380 }
381 }
382 }
383 return Next::ReduceTaggedBitcast(input, from, to, kind);
384 }

541 return __ Word32Constant(base::bits::CountPopulation(w32_k));
542 case WordUnaryOp::Kind::kSignExtend8:
543 return __ Word32Constant(int32_t{static_cast<int8_t>(w32_k)});
544 case WordUnaryOp::Kind::kSignExtend16:
545 return __ Word32Constant(int32_t{static_cast<int16_t>(w32_k)});
546 }
547 } else if (uint64_t w64_k;
548 rep == WordRepresentation::Word64() &&
549 matcher_.MatchIntegralWord64Constant(input, &w64_k)) {
550 switch (kind) {
551 case WordUnaryOp::Kind::kReverseBytes:
552 return __ Word64Constant(base::bits::ReverseBytes(w64_k));
553 case WordUnaryOp::Kind::kCountLeadingZeros:
554 return __ Word64Constant(
555 uint64_t{base::bits::CountLeadingZeros(w64_k)});
556 case WordUnaryOp::Kind::kCountTrailingZeros:
557 return __ Word64Constant(
558 uint64_t{base::bits::CountTrailingZeros(w64_k)});
559 case WordUnaryOp::Kind::kPopCount:
560 return __ Word64Constant(
561 uint64_t{base::bits::CountPopulation(w64_k)});
562 case WordUnaryOp::Kind::kSignExtend8:
563 return __ Word64Constant(int64_t{static_cast<int8_t>(w64_k)});
564 case WordUnaryOp::Kind::kSignExtend16:
565 return __ Word64Constant(int64_t{static_cast<int16_t>(w64_k)});
566 }
567 }
568 return Next::ReduceWordUnary(input, kind, rep);
569 }

571 V<Float> REDUCE(FloatBinop)(V<Float> lhs, V<Float> rhs,
572 FloatBinopOp::Kind kind,
573 FloatRepresentation rep) {
574 if (ShouldSkipOptimizationStep()) {
575 return Next::ReduceFloatBinop(lhs, rhs, kind, rep);
576 }
577
578 using Kind = FloatBinopOp::Kind;
579
580 // Place constant on the right for commutative operators.
581 if (FloatBinopOp::IsCommutative(kind) && matcher_.Is<ConstantOp>(lhs) &&
582 !matcher_.Is<ConstantOp>(rhs)) {
583 return ReduceFloatBinop(rhs, lhs, kind, rep);
584 }
585
586 // constant folding
587 if (float k1, k2; rep == FloatRepresentation::Float32() &&
588 matcher_.MatchFloat32Constant(lhs, &k1) &&
589 matcher_.MatchFloat32Constant(rhs, &k2)) {
590 switch (kind) {
591 case Kind::kAdd:
592 return __ Float32Constant(k1 + k2);
593 case Kind::kMul:
594 return __ Float32Constant(k1 * k2);
595 case Kind::kSub:
596 return __ Float32Constant(k1 - k2);
597 case Kind::kMin:
598 return __ Float32Constant(JSMin(k1, k2));
599 case Kind::kMax:
600 return __ Float32Constant(JSMax(k1, k2));
601 case Kind::kDiv:
602 return __ Float32Constant(k1 / k2);
603 case Kind::kPower:
604 return __ Float32Constant(internal::math::pow(k1, k2));
605 case Kind::kAtan2:
606 return __ Float32Constant(base::ieee754::atan2(k1, k2));
607 case Kind::kMod:
608 UNREACHABLE();
609 }
610 }
611 if (double k1, k2; rep == FloatRepresentation::Float64() &&
612 matcher_.MatchFloat64Constant(lhs, &k1) &&
613 matcher_.MatchFloat64Constant(rhs, &k2)) {
614 switch (kind) {
615 case Kind::kAdd:
616 return __ Float64Constant(k1 + k2);
617 case Kind::kMul:
618 return __ Float64Constant(k1 * k2);
619 case Kind::kSub:
620 return __ Float64Constant(k1 - k2);
621 case Kind::kMin:
622 return __ Float64Constant(JSMin(k1, k2));
623 case Kind::kMax:
624 return __ Float64Constant(JSMax(k1, k2));
625 case Kind::kDiv:
626 return __ Float64Constant(k1 / k2);
627 case Kind::kMod:
628 return __ Float64Constant(Modulo(k1, k2));
629 case Kind::kPower:
630 return __ Float64Constant(math::pow(k1, k2));
631 case Kind::kAtan2:
632 return __ Float64Constant(base::ieee754::atan2(k1, k2));
633 }
634 }
635
636 // lhs <op> NaN => NaN
637 if (matcher_.MatchNaN(rhs) ||
638 (matcher_.MatchNaN(lhs) && kind != Kind::kPower)) {
639 // Return a quiet NaN since Wasm operations could have signalling NaN as
640 // input but not as output.
641 return __ FloatConstant(std::numeric_limits<double>::quiet_NaN(), rep);
642 }
643
644 if (matcher_.Is<ConstantOp>(rhs)) {
645 if (kind == Kind::kMul) {
646 // lhs * 1 => lhs
647 if (!signalling_nan_possible && matcher_.MatchFloat(rhs, 1.0)) {
648 return lhs;
649 }
650 // lhs * 2 => lhs + lhs
651 if (matcher_.MatchFloat(rhs, 2.0)) {
652 return __ FloatAdd(lhs, lhs, rep);
653 }
654 // lhs * -1 => -lhs
655 if (!signalling_nan_possible && matcher_.MatchFloat(rhs, -1.0)) {
656 return __ FloatNegate(lhs, rep);
657 }
658 }
659
660 if (kind == Kind::kDiv) {
661 // lhs / 1 => lhs
662 if (!signalling_nan_possible && matcher_.MatchFloat(rhs, 1.0)) {
663 return lhs;
664 }
665 // lhs / -1 => -lhs
666 if (!signalling_nan_possible && matcher_.MatchFloat(rhs, -1.0)) {
667 return __ FloatNegate(lhs, rep);
668 }
669 // All reciprocals of non-denormal powers of two can be represented
670 // exactly, so division by power of two can be reduced to
671 // multiplication by reciprocal, with the same result.
672 // x / k => x * (1 / k)
673 if (rep == FloatRepresentation::Float32()) {
674 if (float k;
675 matcher_.MatchFloat32Constant(rhs, &k) && std::isnormal(k) &&
676 k != 0 && std::isfinite(k) &&
677 base::bits::IsPowerOfTwo(base::Double(k).Significand())) {
678 return __ FloatMul(lhs, __ FloatConstant(1.0 / k, rep), rep);
679 }
680 } else {
681 DCHECK_EQ(rep, FloatRepresentation::Float64());
682 if (double k;
683 matcher_.MatchFloat64Constant(rhs, &k) && std::isnormal(k) &&
684 k != 0 && std::isfinite(k) &&
685 base::bits::IsPowerOfTwo(base::Double(k).Significand())) {
686 return __ FloatMul(lhs, __ FloatConstant(1.0 / k, rep), rep);
687 }
688 }
689 }
690
691 if (kind == Kind::kMod) {
692 // x % 0 => NaN
693 if (matcher_.MatchFloat(rhs, 0.0)) {
694 return __ FloatConstant(std::numeric_limits<double>::quiet_NaN(),
695 rep);
696 }
697 }
698
699 if (kind == Kind::kSub) {
700 // lhs - +0.0 => lhs
701 if (!signalling_nan_possible && matcher_.MatchFloat(rhs, +0.0)) {
702 return lhs;
703 }
704 }
705
706 if (kind == Kind::kPower) {
707 if (matcher_.MatchFloat(rhs, 0.0) || matcher_.MatchFloat(rhs, -0.0)) {
708 // lhs ** 0 ==> 1
709 return __ FloatConstant(1.0, rep);
710 }
711 if (matcher_.MatchFloat(rhs, 2.0)) {
712 // lhs ** 2 ==> lhs * lhs
713 return __ FloatMul(lhs, lhs, rep);
714 }
715 if (matcher_.MatchFloat(rhs, 0.5)) {
716 // lhs ** 0.5 ==> sqrt(lhs)
717 // (unless if lhs is -infinity)
718 Variable result = __ NewLoopInvariantVariable(rep);
719 IF (UNLIKELY(__ FloatLessThanOrEqual(
720 lhs, __ FloatConstant(-V8_INFINITY, rep), rep))) {
721 __ SetVariable(result, __ FloatConstant(V8_INFINITY, rep));
722 } ELSE {
723 __ SetVariable(
724 result,
725 __ FloatSqrt(__ FloatAdd(lhs, __ FloatConstant(0, rep), rep),
726 rep));
727 }
728
729 return __ GetVariable(result);
730 }
731 }
732 }
733
734 if (!signalling_nan_possible && kind == Kind::kSub &&
735 matcher_.MatchFloat(lhs, -0.0)) {
736 // -0.0 - round_down(-0.0 - y) => round_up(y)
737 if (V<Float> a, b, c;
738 FloatUnaryOp::IsSupported(FloatUnaryOp::Kind::kRoundUp, rep) &&
739 matcher_.MatchFloatRoundDown(rhs, &a, rep) &&
740 matcher_.MatchFloatSub(a, &b, &c, rep) &&
741 matcher_.MatchFloat(b, -0.0)) {
742 return __ FloatRoundUp(c, rep);
743 }
744 // -0.0 - rhs => -rhs
745 return __ FloatNegate(rhs, rep);
746 }
747
748 return Next::ReduceFloatBinop(lhs, rhs, kind, rep);
749 }
750
751 V<Word> REDUCE(WordBinop)(V<Word> left, V<Word> right, WordBinopOp::Kind kind,
752 WordRepresentation rep) {
753 if (ShouldSkipOptimizationStep()) {
754 return Next::ReduceWordBinop(left, right, kind, rep);
755 }
756
757 using Kind = WordBinopOp::Kind;
758
759 DCHECK_EQ(rep, any_of(WordRepresentation::Word32(),
760 WordRepresentation::Word64()));
761 bool is_64 = rep == WordRepresentation::Word64();
762
763 if (!is_64) {
764 left = TryRemoveWord32ToWord64Conversion(left);
765 right = TryRemoveWord32ToWord64Conversion(right);
766 }
767
768 // Place constant on the right for commutative operators.
769 if (WordBinopOp::IsCommutative(kind) && matcher_.Is<ConstantOp>(left) &&
770 !matcher_.Is<ConstantOp>(right)) {
771 return ReduceWordBinop(right, left, kind, rep);
772 }
773 // constant folding
774 if (uint64_t k1, k2; matcher_.MatchIntegralWordConstant(left, rep, &k1) &&
775 matcher_.MatchIntegralWordConstant(right, rep, &k2)) {
776 switch (kind) {
777 case Kind::kAdd:
778 return __ WordConstant(k1 + k2, rep);
779 case Kind::kMul:
780 return __ WordConstant(k1 * k2, rep);
781 case Kind::kBitwiseAnd:
782 return __ WordConstant(k1 & k2, rep);
783 case Kind::kBitwiseOr:
784 return __ WordConstant(k1 | k2, rep);
785 case Kind::kBitwiseXor:
786 return __ WordConstant(k1 ^ k2, rep);
787 case Kind::kSub:
788 return __ WordConstant(k1 - k2, rep);
789 case Kind::kSignedMulOverflownBits:
790 return __ WordConstant(
791 is_64 ? base::bits::SignedMulHigh64(static_cast<int64_t>(k1),
792 static_cast<int64_t>(k2))
793 : base::bits::SignedMulHigh32(static_cast<int32_t>(k1),
794 static_cast<int32_t>(k2)),
795 rep);
796 case Kind::kUnsignedMulOverflownBits:
797 return __ WordConstant(
798 is_64 ? base::bits::UnsignedMulHigh64(k1, k2)
799 : base::bits::UnsignedMulHigh32(static_cast<uint32_t>(k1),
800 static_cast<uint32_t>(k2)),
801 rep);
802 case Kind::kSignedDiv:
803 return __ WordConstant(
804 is_64 ? base::bits::SignedDiv64(k1, k2)
805 : base::bits::SignedDiv32(static_cast<int32_t>(k1),
806 static_cast<int32_t>(k2)),
807 rep);
808 case Kind::kUnsignedDiv:
809 return __ WordConstant(
810 is_64 ? base::bits::UnsignedDiv64(k1, k2)
811 : base::bits::UnsignedDiv32(static_cast<uint32_t>(k1),
812 static_cast<uint32_t>(k2)),
813 rep);
814 case Kind::kSignedMod:
815 return __ WordConstant(
816 is_64 ? base::bits::SignedMod64(k1, k2)
817 : base::bits::SignedMod32(static_cast<int32_t>(k1),
818 static_cast<int32_t>(k2)),
819 rep);
820 case Kind::kUnsignedMod:
821 return __ WordConstant(
822 is_64 ? base::bits::UnsignedMod64(k1, k2)
823 : base::bits::UnsignedMod32(static_cast<uint32_t>(k1),
824 static_cast<uint32_t>(k2)),
825 rep);
826 }
827 }
828
829 if (kind == WordBinopOp::Kind::kBitwiseAnd &&
830 rep == WordRepresentation::Word32()) {
831 if (auto right_bitfield = detail::BitfieldCheck::Detect(
832 matcher_, __ output_graph(), right)) {
833 if (auto left_bitfield = detail::BitfieldCheck::Detect(
834 matcher_, __ output_graph(), left)) {
835 if (auto combined_bitfield =
836 left_bitfield->TryCombine(*right_bitfield)) {
837 V<Word> source = combined_bitfield->source;
838 if (combined_bitfield->truncate_from_64_bit) {
839 source = __ TruncateWord64ToWord32(V<Word64>::Cast(source));
840 }
841 return __ Word32Equal(__ Word32BitwiseAnd(V<Word32>::Cast(source),
842 combined_bitfield->mask),
843 combined_bitfield->masked_value);
844 }
845 }
846 }
847 }
848
849 if (uint64_t right_value;
850 matcher_.MatchIntegralWordConstant(right, rep, &right_value)) {
851 // TODO(jkummerow): computing {right_value_signed} could probably be
852 // handled by the 4th argument to {MatchIntegralWordConstant}.
853 int64_t right_value_signed =
854 is_64 ? static_cast<int64_t>(right_value)
855 : int64_t{static_cast<int32_t>(right_value)};
856 // (a <op> k1) <op> k2 => a <op> (k1 <op> k2)
857 if (V<Word> a, k1; WordBinopOp::IsAssociative(kind) &&
858 matcher_.MatchWordBinop(left, &a, &k1, kind, rep) &&
859 matcher_.Is<ConstantOp>(k1)) {
860 V<Word> k2 = right;
861 // This optimization allows to do constant folding of `k1` and `k2`.
862 // However, if (a <op> k1) has to be calculated anyways, then constant
863 // folding does not save any calculations during runtime, and it may
864 // increase register pressure because it extends the lifetime of `a`.
865 // Therefore we do the optimization only when `left = (a <op k1)` has no
866 // other uses.
867 if (matcher_.Get(left).saturated_use_count.IsZero()) {
868 return ReduceWordBinop(a, ReduceWordBinop(k1, k2, kind, rep), kind,
869 rep);
870 }
871 }
872 switch (kind) {
873 case Kind::kSub:
874 // left - k => left + -k
875 return ReduceWordBinop(left, __ WordConstant(-right_value, rep),
876 Kind::kAdd, rep);
877 case Kind::kAdd:
878 // left + 0 => left
879 if (right_value == 0) {
880 return left;
881 }
882 break;
883 case Kind::kBitwiseXor:
884 // left ^ 0 => left
885 if (right_value == 0) {
886 return left;
887 }
888 // left ^ 1 => left == 0 if left is 0 or 1
889 if (right_value == 1 && IsBit(left)) {
890 return __ Word32Equal(V<Word32>::Cast(left), 0);
891 }
892 // (x ^ -1) ^ -1 => x
893 {
894 V<Word> x, y;
895 int64_t k;
896 if (right_value_signed == -1 &&
897 matcher_.MatchBitwiseAnd(left, &x, &y, rep) &&
898 matcher_.MatchIntegralWordConstant(y, rep, &k) && k == -1) {
899 return x;
900 }
901 }
902 break;
903 case Kind::kBitwiseOr:
904 // left | 0 => left
905 if (right_value == 0) {
906 return left;
907 }
908 // left | -1 => -1
909 if (right_value_signed == -1) {
910 return right;
911 }
912 // (x & K1) | K2 => x | K2 if K2 has ones for every zero bit in K1.
913 // This case can be constructed by UpdateWord and UpdateWord32 in CSA.
914 {
915 V<Word> x, y;
916 uint64_t k1;
917 uint64_t k2 = right_value;
918 if (matcher_.MatchBitwiseAnd(left, &x, &y, rep) &&
919 matcher_.MatchIntegralWordConstant(y, rep, &k1) &&
920 (k1 | k2) == rep.MaxUnsignedValue()) {
921 return __ WordBitwiseOr(x, right, rep);
922 }
923 }
924 break;
925 case Kind::kMul:
926 // left * 0 => 0
927 if (right_value == 0) {
928 return __ WordConstant(0, rep);
929 }
930 // left * 1 => left
931 if (right_value == 1) {
932 return left;
933 }
934 // left * -1 => 0 - left
935 if (right_value_signed == -1) {
936 return __ WordSub(__ WordConstant(0, rep), left, rep);
937 }
938 // left * 2^k => left << k
939 if (base::bits::IsPowerOfTwo(right_value)) {
940 return __ ShiftLeft(left, base::bits::WhichPowerOfTwo(right_value),
941 rep);
942 }
943 break;
944 case Kind::kBitwiseAnd:
945 // left & -1 => left
946 if (right_value_signed == -1) {
947 return left;
948 }
949 // x & 0 => 0
950 if (right_value == 0) {
951 return right;
952 }
953
954 if (right_value == 1) {
955 // (x + x) & 1 => 0
956 V<Word> left_ignore_extensions =
957 IsWord32ConvertedToWord64(left)
958 ? UndoWord32ToWord64Conversion(left)
959 : left;
960 if (V<Word> a, b;
961 matcher_.MatchWordAdd(left_ignore_extensions, &a, &b,
962 WordRepresentation::Word32()) &&
963 a == b) {
964 return __ WordConstant(0, rep);
965 }
966
967 // CMP & 1 => CMP
968 if (IsBit(left_ignore_extensions)) {
969 return left;
970 }
971
972 static_assert(kSmiTagMask == 1);
973 // HeapObject & 1 => 1 ("& 1" is a Smi-check)
974 if (TryMatchHeapObject(left)) {
975 return __ WordConstant(1, rep);
976 }
977 }
978
979 // asm.js often benefits from these transformations, to optimize out
980 // unnecessary memory access alignment masks. Conventions used in
981 // the comments below:
982 // x, y: arbitrary values
983 // K, L, M: arbitrary constants
984 // (-1 << K) == mask: the right-hand side of the bitwise AND.
985 if (IsNegativePowerOfTwo(right_value_signed)) {
986 uint64_t mask = right_value;
987 int K = base::bits::CountTrailingZeros64(mask);
988 V<Word> x, y;
989 {
990 int L;
991 // (x << L) & (-1 << K)
992 // => x << L iff L >= K
993 if (matcher_.MatchConstantLeftShift(left, &x, rep, &L) &&
994 L >= K) {
995 return left;
996 }
997 }
998
999 if (matcher_.MatchWordAdd(left, &x, &y, rep)) {
1000 uint64_t L; // L == (M << K) iff (L & mask) == L.
1001
1002 // (x + (M << K)) & (-1 << K)
1003 // => (x & (-1 << K)) + (M << K)
1004 if (matcher_.MatchIntegralWordConstant(y, rep, &L) &&
1005 (L & mask) == L) {
1006 return __ WordAdd(__ WordBitwiseAnd(x, right, rep),
1007 __ WordConstant(L, rep), rep);
1008 }
1009
1010 // (x1 * (M << K) + y) & (-1 << K)
1011 // => x1 * (M << K) + (y & (-1 << K))
1012 V<Word> x1, x2, y1, y2;
1013 if (matcher_.MatchWordMul(x, &x1, &x2, rep) &&
1014 matcher_.MatchIntegralWordConstant(x2, rep, &L) &&
1015 (L & mask) == L) {
1016 return __ WordAdd(x, __ WordBitwiseAnd(y, right, rep), rep);
1017 }
1018 // Same as above with swapped order:
1019 // (x + y1 * (M << K)) & (-1 << K)
1020 // => (x & (-1 << K)) + y1 * (M << K)
1021 if (matcher_.MatchWordMul(y, &y1, &y2, rep) &&
1022 matcher_.MatchIntegralWordConstant(y2, rep, &L) &&
1023 (L & mask) == L) {
1024 return __ WordAdd(__ WordBitwiseAnd(x, right, rep), y, rep);
1025 }
1026
1027 // ((x1 << K) + y) & (-1 << K)
1028 // => (x1 << K) + (y & (-1 << K))
1029 int K2;
1030 if (matcher_.MatchConstantLeftShift(x, &x1, rep, &K2) &&
1031 K2 == K) {
1032 return __ WordAdd(x, __ WordBitwiseAnd(y, right, rep), rep);
1033 }
1034 // Same as above with swapped order:
1035 // (x + (y1 << K)) & (-1 << K)
1036 // => (x & (-1 << K)) + (y1 << K)
1037 if (matcher_.MatchConstantLeftShift(y, &y1, rep, &K2) &&
1038 K2 == K) {
1039 return __ WordAdd(__ WordBitwiseAnd(x, right, rep), y, rep);
1040 }
1041 } else if (matcher_.MatchWordMul(left, &x, &y, rep)) {
1042 // (x * (M << K)) & (-1 << K) => x * (M << K)
1043 uint64_t L; // L == (M << K) iff (L & mask) == L.
1044 if (matcher_.MatchIntegralWordConstant(y, rep, &L) &&
1045 (L & mask) == L) {
1046 return left;
1047 }
1048 }
1049 }
1050 break;
1051 case WordBinopOp::Kind::kSignedDiv:
1052 return ReduceSignedDiv(left, right_value_signed, rep);
1053 case WordBinopOp::Kind::kUnsignedDiv:
1054 return ReduceUnsignedDiv(left, right_value, rep);
1055 case WordBinopOp::Kind::kSignedMod:
1056 // left % 0 => 0
1057 // left % 1 => 0
1058 // left % -1 => 0
1059 if (right_value_signed == any_of(0, 1, -1)) {
1060 return __ WordConstant(0, rep);
1061 }
1062 if (right_value_signed != rep.MinSignedValue()) {
1063 right_value_signed = Abs(right_value_signed);
1064 }
1065 // left % 2^n => ((left + m) & (2^n - 1)) - m
1066 // where m = (left >> bits-1) >>> bits-n
1067 // This is a branch-free version of the following:
1068 // left >= 0 ? left & (2^n - 1)
1069 // : ((left + (2^n - 1)) & (2^n - 1)) - (2^n - 1)
1070 // Adding and subtracting (2^n - 1) before and after the bitwise-and
1071 // keeps the result congruent modulo 2^n, but shifts the resulting
1072 // value range to become -(2^n - 1) ... 0.
1073 if (base::bits::IsPowerOfTwo(right_value_signed)) {
1074 uint32_t bits = rep.bit_width();
1075 uint32_t n = base::bits::WhichPowerOfTwo(right_value_signed);
1076 V<Word> m = __ ShiftRightLogical(
1077 __ ShiftRightArithmetic(left, bits - 1, rep), bits - n, rep);
1078 return __ WordSub(
1079 __ WordBitwiseAnd(__ WordAdd(left, m, rep),
1080 __ WordConstant(right_value_signed - 1, rep),
1081 rep),
1082 m, rep);
1083 }
1084 // The `IntDiv` with a constant right-hand side will be turned into a
1085 // multiplication, avoiding the expensive integer division.
1086 return __ WordSub(
1087 left, __ WordMul(__ IntDiv(left, right, rep), right, rep), rep);
1088 case WordBinopOp::Kind::kUnsignedMod:
1089 // left % 0 => 0
1090 // left % 1 => 0
1091 if (right_value == 0 || right_value == 1) {
1092 return __ WordConstant(0, rep);
1093 }
1094 // x % 2^n => x & (2^n - 1)
1095 if (base::bits::IsPowerOfTwo(right_value)) {
1096 return __ WordBitwiseAnd(
1097 left, __ WordConstant(right_value - 1, rep), rep);
1098 }
1099 // The `UintDiv` with a constant right-hand side will be turned into a
1100 // multiplication, avoiding the expensive integer division.
1101 return __ WordSub(
1102 left, __ WordMul(right, __ UintDiv(left, right, rep), rep), rep);
1103 case WordBinopOp::Kind::kSignedMulOverflownBits:
1104 case WordBinopOp::Kind::kUnsignedMulOverflownBits:
1105 break;
1106 }
1107 }
1108
1109 if (kind == Kind::kAdd) {
1110 V<Word> x, y, zero;
1111 // (0 - x) + y => y - x
1112 if (matcher_.MatchWordSub(left, &zero, &x, rep) &&
1113 matcher_.MatchZero(zero)) {
1114 y = right;
1115 return __ WordSub(y, x, rep);
1116 }
1117 // x + (0 - y) => x - y
1118 if (matcher_.MatchWordSub(right, &zero, &y, rep) &&
1119 matcher_.MatchZero(zero)) {
1120 x = left;
1121 return __ WordSub(x, y, rep);
1122 }
1123 }
1124
1125 // 0 / right => 0
1126 // 0 % right => 0
1127 if (matcher_.MatchZero(left) &&
1128 kind == any_of(Kind::kSignedDiv, Kind::kUnsignedDiv, Kind::kUnsignedMod,
1129 Kind::kSignedMod)) {
1130 return __ WordConstant(0, rep);
1131 }
1132
1133 if (left == right) {
1134 V<Word> x = left;
1135 switch (kind) {
1136 // x & x => x
1137 // x | x => x
1138 case WordBinopOp::Kind::kBitwiseAnd:
1139 case WordBinopOp::Kind::kBitwiseOr:
1140 return x;
1141 // x ^ x => 0
1142 // x - x => 0
1143 // x % x => 0
1144 case WordBinopOp::Kind::kBitwiseXor:
1145 case WordBinopOp::Kind::kSub:
1146 case WordBinopOp::Kind::kSignedMod:
1147 case WordBinopOp::Kind::kUnsignedMod:
1148 return __ WordConstant(0, rep);
1149 // x / x => x != 0
1150 case WordBinopOp::Kind::kSignedDiv:
1151 case WordBinopOp::Kind::kUnsignedDiv: {
1152 V<Word> zero = __ WordConstant(0, rep);
1153 V<Word32> result = __ Word32Equal(__ Equal(left, zero, rep), 0);
1154 return __ ZeroExtendWord32ToRep(result, rep);
1155 }
1156 case WordBinopOp::Kind::kAdd:
1157 case WordBinopOp::Kind::kMul:
1158 case WordBinopOp::Kind::kSignedMulOverflownBits:
1159 case WordBinopOp::Kind::kUnsignedMulOverflownBits:
1160 break;
1161 }
1162 }
1163
1164 if (std::optional<V<Word>> ror = TryReduceToRor(left, right, kind, rep)) {
1165 return *ror;
1166 }
1167
1168 return Next::ReduceWordBinop(left, right, kind, rep);
1169 }

1171 bool TryMatchHeapObject(V<Any> idx, int depth = 0) {
1172 constexpr int kMaxDepth = 2;
1173 if (depth == kMaxDepth) return false;
1174
1175 if (matcher_.MatchHeapConstant(idx)) return true;
1176 if (matcher_.Is<AllocateOp>(idx)) return true;
1177 if (matcher_.Is<Opmask::kTaggedBitcastHeapObject>(idx)) return true;
1178
1179 // A Phi whose inputs are all HeapObject is itself a HeapObject.
1180 if (const PhiOp* phi = matcher_.TryCast<Opmask::kTaggedPhi>(idx)) {
1181 return base::all_of(phi->inputs(), [depth, this](V<Any> input) {
1182 return TryMatchHeapObject(input, depth + 1);
1183 });
1184 }
1185
1186 // For anything else, assume that it's not a heap object.
1187 return false;
1188 }

1190 std::optional<V<Word>> TryReduceToRor(V<Word> left, V<Word> right,
1191 WordBinopOp::Kind kind,
1192 WordRepresentation rep) {
1193 // Recognize rotation, we are matcher_.Matching and transforming as follows
1194 // (assuming kWord32, kWord64 is handled correspondingly):
1195 // x << y | x >>> (32 - y) => x ror (32 - y)
1196 // x << (32 - y) | x >>> y => x ror y
1197 // x << y ^ x >>> (32 - y) => x ror (32 - y) if 1 <= y < 32
1198 // x << (32 - y) ^ x >>> y => x ror y if 1 <= y < 32
1199 // (As well as the commuted forms.)
1200 // Note the side condition for XOR: the optimization doesn't hold for
1201 // an effective rotation amount of 0.
1202
1203 if (!(kind == any_of(WordBinopOp::Kind::kBitwiseOr,
1204 WordBinopOp::Kind::kBitwiseXor))) {
1205 return {};
1206 }
1207
1208 const ShiftOp* high = matcher_.TryCast<ShiftOp>(left);
1209 if (!high) return {};
1210 const ShiftOp* low = matcher_.TryCast<ShiftOp>(right);
1211 if (!low) return {};
1212
1213 if (low->kind == ShiftOp::Kind::kShiftLeft) {
1214 std::swap(low, high);
1215 }
1216 if (high->kind != ShiftOp::Kind::kShiftLeft ||
1217 low->kind != ShiftOp::Kind::kShiftRightLogical) {
1218 return {};
1219 }
1220 V<Word> x = high->left();
1221 if (low->left() != x) return {};
1222 V<Word> amount, a, b;
1223 uint64_t k;
1224 if (matcher_.MatchWordSub(high->right(), &a, &b, rep) &&
1225 matcher_.MatchIntegralWordConstant(a, rep, &k) && b == low->right() &&
1226 k == rep.bit_width()) {
1227 amount = b;
1228 } else if (matcher_.MatchWordSub(low->right(), &a, &b, rep) &&
1229 a == high->right() &&
1230 matcher_.MatchIntegralWordConstant(b, rep, &k) &&
1231 k == rep.bit_width()) {
1232 amount = low->right();
1233 } else if (uint64_t k1, k2;
1234 matcher_.MatchIntegralWordConstant(high->right(), rep, &k1) &&
1235 matcher_.MatchIntegralWordConstant(low->right(), rep, &k2) &&
1236 k1 + k2 == rep.bit_width() && k1 >= 0 && k2 >= 0) {
1237 if (k1 == 0 || k2 == 0) {
1238 if (kind == WordBinopOp::Kind::kBitwiseXor) {
1239 return __ WordConstant(0, rep);
1240 } else {
1241 DCHECK_EQ(kind, WordBinopOp::Kind::kBitwiseOr);
1242 return x;
1243 }
1244 }
1245 return __ RotateRight(x, low->right(), rep);
1246 } else {
1247 return {};
1248 }
1249 if (kind == WordBinopOp::Kind::kBitwiseOr) {
1250 return __ RotateRight(x, amount, rep);
1251 } else {
1252 DCHECK_EQ(kind, WordBinopOp::Kind::kBitwiseXor);
1253 // Can't guarantee that rotation amount is not 0.
1254 return {};
1255 }
1256 }

1258 V<Tuple<Word, Word32>> REDUCE(OverflowCheckedBinop)(
1259 V<Word> left, V<Word> right, OverflowCheckedBinopOp::Kind kind,
1260 WordRepresentation rep) {
1261 if (ShouldSkipOptimizationStep()) {
1262 return Next::ReduceOverflowCheckedBinop(left, right, kind, rep);
1263 }
1264 using Kind = OverflowCheckedBinopOp::Kind;
1265 if (OverflowCheckedBinopOp::IsCommutative(kind) &&
1266 matcher_.Is<ConstantOp>(left) && !matcher_.Is<ConstantOp>(right)) {
1267 return ReduceOverflowCheckedBinop(right, left, kind, rep);
1268 }
1269 if (rep == WordRepresentation::Word32()) {
1270 left = TryRemoveWord32ToWord64Conversion(left);
1271 right = TryRemoveWord32ToWord64Conversion(right);
1272 }
1273 // constant folding
1274 if (rep == WordRepresentation::Word32()) {
1275 if (int32_t k1, k2; matcher_.MatchIntegralWord32Constant(left, &k1) &&
1276 matcher_.MatchIntegralWord32Constant(right, &k2)) {
1277 bool overflow;
1278 int32_t res;
1279 switch (kind) {
1280 case OverflowCheckedBinopOp::Kind::kSignedAdd:
1281 overflow = base::bits::SignedAddOverflow32(k1, k2, &res);
1282 break;
1283 case OverflowCheckedBinopOp::Kind::kSignedMul:
1284 overflow = base::bits::SignedMulOverflow32(k1, k2, &res);
1285 break;
1286 case OverflowCheckedBinopOp::Kind::kSignedSub:
1287 overflow = base::bits::SignedSubOverflow32(k1, k2, &res);
1288 break;
1289 }
1290 return __ Tuple(__ Word32Constant(res), __ Word32Constant(overflow));
1291 }
1292 } else {
1293 DCHECK_EQ(rep, WordRepresentation::Word64());
1294 if (int64_t k1, k2; matcher_.MatchIntegralWord64Constant(left, &k1) &&
1295 matcher_.MatchIntegralWord64Constant(right, &k2)) {
1296 bool overflow;
1297 int64_t res;
1298 switch (kind) {
1299 case OverflowCheckedBinopOp::Kind::kSignedAdd:
1300 overflow = base::bits::SignedAddOverflow64(k1, k2, &res);
1301 break;
1302 case OverflowCheckedBinopOp::Kind::kSignedMul:
1303 overflow = base::bits::SignedMulOverflow64(k1, k2, &res);
1304 break;
1305 case OverflowCheckedBinopOp::Kind::kSignedSub:
1306 overflow = base::bits::SignedSubOverflow64(k1, k2, &res);
1307 break;
1308 }
1309 return __ Tuple(__ Word64Constant(res), __ Word32Constant(overflow));
1310 }
1311 }
1312
1313 // left + 0 => (left, false)
1314 // left - 0 => (left, false)
1315 if (kind == any_of(Kind::kSignedAdd, Kind::kSignedSub) &&
1316 matcher_.MatchZero(right)) {
1317 return __ Tuple(left, __ Word32Constant(0));
1318 }
1319
1320 if (kind == Kind::kSignedMul) {
1321 if (int64_t k; matcher_.MatchIntegralWordConstant(right, rep, &k)) {
1322 // left * 0 => (0, false)
1323 if (k == 0) {
1324 return __ Tuple(__ WordConstant(0, rep), __ Word32Constant(false));
1325 }
1326 // left * 1 => (left, false)
1327 if (k == 1) {
1328 return __ Tuple(left, __ Word32Constant(false));
1329 }
1330 // left * -1 => 0 - left
1331 if (k == -1) {
1332 return __ IntSubCheckOverflow(__ WordConstant(0, rep), left, rep);
1333 }
1334 // left * 2 => left + left
1335 if (k == 2) {
1336 return __ IntAddCheckOverflow(left, left, rep);
1337 }
1338 }
1339 }
1340
1341 // UntagSmi(x) + UntagSmi(x) => (x, false)
1342 // (where UntagSmi(x) = x >> 1 with a ShiftOutZeros shift)
1343 if (kind == Kind::kSignedAdd && left == right) {
1344 uint16_t amount;
1345 if (V<Word32> x; matcher_.MatchConstantShiftRightArithmeticShiftOutZeros(
1346 left, &x, WordRepresentation::Word32(), &amount) &&
1347 amount == 1) {
1348 return __ Tuple(x, __ Word32Constant(0));
1349 }
1350 }
1351
1352 return Next::ReduceOverflowCheckedBinop(left, right, kind, rep);
1353 }

1355 V<Word32> REDUCE(Comparison)(V<Any> left, V<Any> right,
1356 ComparisonOp::Kind kind,
1357 RegisterRepresentation rep) {
1358 if (ShouldSkipOptimizationStep()) {
1359 return Next::ReduceComparison(left, right, kind, rep);
1360 }
1361 if (kind == ComparisonOp::Kind::kEqual) {
1362 return ReduceCompareEqual(left, right, rep);
1363 }
1364 if (rep == WordRepresentation::Word32()) {
1365 left = TryRemoveWord32ToWord64Conversion(V<Word>::Cast(left));
1366 right = TryRemoveWord32ToWord64Conversion(V<Word>::Cast(right));
1367 }
1368 using Kind = ComparisonOp::Kind;
1369 if (left == right &&
1370 !(rep == any_of(RegisterRepresentation::Float32(),
1371 RegisterRepresentation::Float64())) &&
1372 kind == any_of(Kind::kSignedLessThanOrEqual,
1373 Kind::kUnsignedLessThanOrEqual)) {
1374 switch (kind) {
1375 case Kind::kEqual:
1376 UNREACHABLE();
1377 case Kind::kUnsignedLessThanOrEqual:
1378 case Kind::kSignedLessThanOrEqual:
1379 return __ Word32Constant(1);
1380 case Kind::kUnsignedLessThan:
1381 case Kind::kSignedLessThan:
1382 return __ Word32Constant(0);
1383 }
1384 }
1385 // constant folding
1386 if (matcher_.Is<ConstantOp>(right) && matcher_.Is<ConstantOp>(left)) {
1387 switch (rep.value()) {
1388 case RegisterRepresentation::Word32():
1389 case RegisterRepresentation::Word64(): {
1390 if (kind ==
1391 any_of(Kind::kSignedLessThan, Kind::kSignedLessThanOrEqual)) {
1392 if (int64_t k1, k2; matcher_.MatchIntegralWordConstant(
1393 left, WordRepresentation(rep), &k1) &&
1394 matcher_.MatchIntegralWordConstant(
1395 right, WordRepresentation(rep), &k2)) {
1396 switch (kind) {
1397 case ComparisonOp::Kind::kSignedLessThan:
1398 return __ Word32Constant(k1 < k2);
1399 case ComparisonOp::Kind::kSignedLessThanOrEqual:
1400 return __ Word32Constant(k1 <= k2);
1401 case ComparisonOp::Kind::kEqual:
1402 case ComparisonOp::Kind::kUnsignedLessThan:
1403 case ComparisonOp::Kind::kUnsignedLessThanOrEqual:
1404 UNREACHABLE();
1405 }
1406 }
1407 } else {
1408 if (uint64_t k1, k2; matcher_.MatchIntegralWordConstant(
1409 left, WordRepresentation(rep), &k1) &&
1410 matcher_.MatchIntegralWordConstant(
1411 right, WordRepresentation(rep), &k2)) {
1412 switch (kind) {
1413 case ComparisonOp::Kind::kUnsignedLessThan:
1414 return __ Word32Constant(k1 < k2);
1415 case ComparisonOp::Kind::kUnsignedLessThanOrEqual:
1416 return __ Word32Constant(k1 <= k2);
1417 case ComparisonOp::Kind::kEqual:
1418 case ComparisonOp::Kind::kSignedLessThan:
1419 case ComparisonOp::Kind::kSignedLessThanOrEqual:
1420 UNREACHABLE();
1421 }
1422 }
1423 }
1424 break;
1425 }
1426 case RegisterRepresentation::Float32(): {
1427 if (float k1, k2; matcher_.MatchFloat32Constant(left, &k1) &&
1428 matcher_.MatchFloat32Constant(right, &k2)) {
1429 switch (kind) {
1430 case ComparisonOp::Kind::kSignedLessThan:
1431 return __ Word32Constant(k1 < k2);
1432 case ComparisonOp::Kind::kSignedLessThanOrEqual:
1433 return __ Word32Constant(k1 <= k2);
1434 case ComparisonOp::Kind::kEqual:
1435 case ComparisonOp::Kind::kUnsignedLessThan:
1436 case ComparisonOp::Kind::kUnsignedLessThanOrEqual:
1437 UNREACHABLE();
1438 }
1439 }
1440 break;
1441 }
1442 case RegisterRepresentation::Float64(): {
1443 if (double k1, k2; matcher_.MatchFloat64Constant(left, &k1) &&
1444 matcher_.MatchFloat64Constant(right, &k2)) {
1445 switch (kind) {
1446 case ComparisonOp::Kind::kSignedLessThan:
1447 return __ Word32Constant(k1 < k2);
1448 case ComparisonOp::Kind::kSignedLessThanOrEqual:
1449 return __ Word32Constant(k1 <= k2);
1450 case ComparisonOp::Kind::kEqual:
1451 case ComparisonOp::Kind::kUnsignedLessThan:
1452 case ComparisonOp::Kind::kUnsignedLessThanOrEqual:
1453 UNREACHABLE();
1454 }
1455 }
1456 break;
1457 }
1458 default:
1459 UNREACHABLE();
1460 }
1461 }
1462 if (rep == RegisterRepresentation::Float64() &&
1463 IsFloat32ConvertedToFloat64(left) &&
1464 IsFloat32ConvertedToFloat64(right)) {
1465 return __ Comparison(
1466 UndoFloat32ToFloat64Conversion(V<Float64>::Cast(left)),
1467 UndoFloat32ToFloat64Conversion(V<Float64>::Cast(right)), kind,
1468 RegisterRepresentation::Float32());
1469 }
1470 if (rep.IsWord()) {
1471 WordRepresentation rep_w{rep};
1472 if (kind == Kind::kUnsignedLessThanOrEqual) {
1473 // 0 <= x => true
1474 if (uint64_t k;
1475 matcher_.MatchIntegralWordConstant(left, rep_w, &k) && k == 0) {
1476 return __ Word32Constant(1);
1477 }
1478 // x <= MaxUint => true
1479 if (uint64_t k; matcher_.MatchIntegralWordConstant(right, rep_w, &k) &&
1480 k == rep.MaxUnsignedValue()) {
1481 return __ Word32Constant(1);
1482 }
1483 // x <= 0 => x == 0
1484 if (uint64_t k;
1485 matcher_.MatchIntegralWordConstant(right, rep_w, &k) && k == 0) {
1486 return __ Equal(left, __ WordConstant(0, rep_w), rep_w);
1487 }
1488 }
1489 if (kind == Kind::kUnsignedLessThan) {
1490 // x < 0 => false
1491 if (uint64_t k;
1492 matcher_.MatchIntegralWordConstant(right, rep_w, &k) && k == 0) {
1493 return __ Word32Constant(0);
1494 }
1495 // MaxUint < x => true
1496 if (uint64_t k; matcher_.MatchIntegralWordConstant(left, rep_w, &k) &&
1497 k == rep.MaxUnsignedValue()) {
1498 return __ Word32Constant(0);
1499 }
1500 }
1501 {
1502 // (x >> k) </<= (y >> k) => x </<= y if shifts reversible
1503 V<Word> x, y;
1504 uint16_t k1, k2;
1505 if (matcher_.MatchConstantShiftRightArithmeticShiftOutZeros(
1506 left, &x, rep_w, &k1) &&
1507 matcher_.MatchConstantShiftRightArithmeticShiftOutZeros(
1508 right, &y, rep_w, &k2) &&
1509 k1 == k2) {
1510 return __ Comparison(x, y, kind, rep_w);
1511 }
1512 }
1513 {
1514 // (x >> k1) </<= k2 => x </<= (k2 << k1) if shifts reversible
1515 // Only perform the transformation if the shift is not used yet, to
1516 // avoid keeping both the shift and x alive.
1517 V<Word> x;
1518 uint16_t k1;
1519 int64_t k2;
1520 if (matcher_.MatchConstantShiftRightArithmeticShiftOutZeros(
1521 left, &x, rep_w, &k1) &&
1522 matcher_.MatchIntegralWordConstant(right, rep_w, &k2) &&
1523 CountLeadingSignBits(k2, rep_w) > k1) {
1524 if (matcher_.Get(left).saturated_use_count.IsZero()) {
1525 return __ Comparison(
1526 x, __ WordConstant(base::bits::Unsigned(k2) << k1, rep_w), kind,
1527 rep_w);
1528 } else if constexpr (reducer_list_contains<
1529 ReducerList, ValueNumberingReducer>::value) {
1530 // If the shift has uses, we only apply the transformation if the
1531 // result would be GVNed away.
1532 V<Word> rhs =
1533 __ WordConstant(base::bits::Unsigned(k2) << k1, rep_w);
1534 static_assert(ComparisonOp::input_count == 2);
1535 static_assert(sizeof(ComparisonOp) == 8);
1536 base::SmallVector<OperationStorageSlot, 32> storage;
1537 ComparisonOp* cmp =
1538 CreateOperation<ComparisonOp>(storage, x, rhs, kind, rep_w);
1539 if (__ WillGVNOp(*cmp)) {
1540 return __ Comparison(x, rhs, kind, rep_w);
1541 }
1542 }
1543 }
1544 // k2 </<= (x >> k1) => (k2 << k1) </<= x if shifts reversible
1545 // Only perform the transformation if the shift is not used yet, to
1546 // avoid keeping both the shift and x alive.
1547 if (matcher_.MatchConstantShiftRightArithmeticShiftOutZeros(
1548 right, &x, rep_w, &k1) &&
1549 matcher_.MatchIntegralWordConstant(left, rep_w, &k2) &&
1550 CountLeadingSignBits(k2, rep_w) > k1) {
1551 if (matcher_.Get(right).saturated_use_count.IsZero()) {
1552 return __ Comparison(
1553 __ WordConstant(base::bits::Unsigned(k2) << k1, rep_w), x, kind,
1554 rep_w);
1555 } else if constexpr (reducer_list_contains<
1556 ReducerList, ValueNumberingReducer>::value) {
1557 // If the shift has uses, we only apply the transformation if the
1558 // result would be GVNed away.
1559 V<Word> lhs =
1560 __ WordConstant(base::bits::Unsigned(k2) << k1, rep_w);
1561 static_assert(ComparisonOp::input_count == 2);
1562 static_assert(sizeof(ComparisonOp) == 8);
1563 base::SmallVector<OperationStorageSlot, 32> storage;
1564 ComparisonOp* cmp =
1565 CreateOperation<ComparisonOp>(storage, lhs, x, kind, rep_w);
1566 if (__ WillGVNOp(*cmp)) {
1567 return __ Comparison(lhs, x, kind, rep_w);
1568 }
1569 }
1570 }
1571 }
1572 // Map 64bit to 32bit comparisons.
1573 if (rep_w == WordRepresentation::Word64()) {
1574 std::optional<bool> left_sign_extended;
1575 std::optional<bool> right_sign_extended;
1576 if (IsWord32ConvertedToWord64(left, &left_sign_extended) &&
1577 IsWord32ConvertedToWord64(right, &right_sign_extended)) {
1578 if (left_sign_extended != true && right_sign_extended != true) {
1579 // Both sides were zero-extended, so the resulting comparison always
1580 // behaves unsigned even if it was a signed 64bit comparison.
1581 auto SetSigned = [](Kind kind, bool is_signed) {
1582 switch (kind) {
1583 case Kind::kSignedLessThan:
1584 case Kind::kUnsignedLessThan:
1585 return is_signed ? Kind::kSignedLessThan
1586 : Kind::kUnsignedLessThan;
1587 case Kind::kSignedLessThanOrEqual:
1588 case Kind::kUnsignedLessThanOrEqual:
1589 return is_signed ? Kind::kSignedLessThanOrEqual
1590 : Kind::kUnsignedLessThanOrEqual;
1591 case Kind::kEqual:
1592 UNREACHABLE();
1593 }
1594 };
1595 return __ Comparison(
1596 UndoWord32ToWord64Conversion(V<Word64>::Cast(left)),
1597 UndoWord32ToWord64Conversion(V<Word64>::Cast(right)),
1598 SetSigned(kind, false), WordRepresentation::Word32());
1599 } else if (left_sign_extended != false &&
1600 right_sign_extended != false) {
1601 // Both sides were sign-extended, this preserves both signed and
1602 // unsigned comparisons.
1603 return __ Comparison(
1604 UndoWord32ToWord64Conversion(V<Word64>::Cast(left)),
1605 UndoWord32ToWord64Conversion(V<Word64>::Cast(right)), kind,
1606 WordRepresentation::Word32());
1607 }
1608 }
1609 }
1610 }
1611 return Next::ReduceComparison(left, right, kind, rep);
1612 }
1613
1614 V<Word> REDUCE(Shift)(V<Word> left, V<Word32> right, ShiftOp::Kind kind,
1615 WordRepresentation rep) {
1616 if (ShouldSkipOptimizationStep()) {
1617 return Next::ReduceShift(left, right, kind, rep);
1618 }
1619
1620 if (rep == WordRepresentation::Word32()) {
1621 left = TryRemoveWord32ToWord64Conversion(left);
1622 }
1623
1624 using Kind = ShiftOp::Kind;
1625 uint64_t c_unsigned;
1626 int64_t c_signed;
1627 if (matcher_.MatchIntegralWordConstant(left, rep, &c_unsigned, &c_signed)) {
1628 if (uint32_t amount;
1629 matcher_.MatchIntegralWord32Constant(right, &amount)) {
1630 amount = amount & (rep.bit_width() - 1);
1631 switch (kind) {
1632 case Kind::kShiftRightArithmeticShiftOutZeros:
1633 if (base::bits::CountTrailingZeros(c_signed) < amount) {
1634 // This assumes that we never hoist operations to before their
1635 // original place in the control flow.
1636 __ Unreachable();
1637 return V<Word>::Invalid();
1638 }
1639 [[fallthrough]];
1640 case Kind::kShiftRightArithmetic:
1641 switch (rep.value()) {
1642 case WordRepresentation::Word32():
1643 return __ Word32Constant(static_cast<int32_t>(c_signed) >>
1644 amount);
1645 case WordRepresentation::Word64():
1646 return __ Word64Constant(c_signed >> amount);
1647 }
1648 case Kind::kShiftRightLogical:
1649 switch (rep.value()) {
1650 case WordRepresentation::Word32():
1651 return __ Word32Constant(static_cast<uint32_t>(c_unsigned) >>
1652 amount);
1653 case WordRepresentation::Word64():
1654 return __ Word64Constant(c_unsigned >> amount);
1655 }
1656 case Kind::kShiftLeft:
1657 return __ WordConstant(c_unsigned << amount, rep);
1658 case Kind::kRotateRight:
1659 switch (rep.value()) {
1660 case WordRepresentation::Word32():
1661 return __ Word32Constant(base::bits::RotateRight32(
1662 static_cast<uint32_t>(c_unsigned), amount));
1663 case WordRepresentation::Word64():
1664 return __ Word64Constant(
1665 base::bits::RotateRight64(c_unsigned, amount));
1666 }
1667 case Kind::kRotateLeft:
1668 switch (rep.value()) {
1669 case WordRepresentation::Word32():
1670 return __ Word32Constant(base::bits::RotateLeft32(
1671 static_cast<uint32_t>(c_unsigned), amount));
1672 case WordRepresentation::Word64():
1673 return __ Word64Constant(
1674 base::bits::RotateLeft64(c_unsigned, amount));
1675 }
1676 }
1677 }
1678 }
1679 if (int32_t amount; matcher_.MatchIntegralWord32Constant(right, &amount) &&
1680 0 <= amount && amount < rep.bit_width()) {
1681 if (amount == 0) {
1682 return left;
1683 }
1684 if (kind == Kind::kShiftLeft) {
1685 // If x >> K only shifted out zeros:
1686 // (x >> K) << L => x if K == L
1687 // (x >> K) << L => x >> (K-L) if K > L
1688 // (x >> K) << L => x << (L-K) if K < L
1689 // Since this is used for Smi untagging, we currently only need it for
1690 // signed shifts.
1691 int k;
1692 V<Word> x;
1693 if (matcher_.MatchConstantShift(
1694 left, &x, Kind::kShiftRightArithmeticShiftOutZeros, rep, &k)) {
1695 int32_t l = amount;
1696 if (k == l) {
1697 return x;
1698 } else if (k > l) {
1699 return __ ShiftRightArithmeticShiftOutZeros(
1700 x, __ Word32Constant(k - l), rep);
1701 } else if (k < l) {
1702 return __ ShiftLeft(x, __ Word32Constant(l - k), rep);
1703 }
1704 }
1705 // (x >>> K) << K => x & ~(2^K - 1)
1706 // (x >> K) << K => x & ~(2^K - 1)
1707 if (matcher_.MatchConstantRightShift(left, &x, rep, &k) &&
1708 k == amount) {
1709 return __ WordBitwiseAnd(
1710 x, __ WordConstant(rep.MaxUnsignedValue() << k, rep), rep);
1711 }
1712 }
1713 if (kind == any_of(Kind::kShiftRightArithmetic,
1714 Kind::kShiftRightArithmeticShiftOutZeros)) {
1715 V<Word> x;
1716 int left_shift_amount;
1717 // (x << k) >> k
1718 if (matcher_.MatchConstantShift(left, &x, ShiftOp::Kind::kShiftLeft,
1719 rep, &left_shift_amount) &&
1720 amount == left_shift_amount) {
1721 // x << (bit_width - 1) >> (bit_width - 1) => 0 - x if x is 0 or 1
1722 if (amount == rep.bit_width() - 1 && IsBit(x)) {

2228 V<Word32> ReduceCompareEqual(V<Any> left, V<Any> right,
2229 RegisterRepresentation rep) {
2230 if (left == right && !rep.IsFloat()) {
2231 return __ Word32Constant(1);
2232 }
2233 if (rep == WordRepresentation::Word32()) {
2234 left = TryRemoveWord32ToWord64Conversion(V<Word>::Cast(left));
2235 right = TryRemoveWord32ToWord64Conversion(V<Word>::Cast(right));
2236 }
2237 if (matcher_.Is<ConstantOp>(left) && !matcher_.Is<ConstantOp>(right)) {
2238 return ReduceCompareEqual(right, left, rep);
2239 }
2240 if (matcher_.Is<ConstantOp>(right)) {
2241 if (matcher_.Is<ConstantOp>(left)) {
2242 // k1 == k2 => k
2243 switch (rep.value()) {
2244 case RegisterRepresentation::Word32():
2245 case RegisterRepresentation::Word64(): {
2246 if (uint64_t k1, k2; matcher_.MatchIntegralWordConstant(
2247 left, WordRepresentation(rep), &k1) &&
2248 matcher_.MatchIntegralWordConstant(
2249 right, WordRepresentation(rep), &k2)) {
2250 return __ Word32Constant(k1 == k2);
2251 }
2252 break;
2253 }
2254 case RegisterRepresentation::Float32(): {
2255 if (float k1, k2; matcher_.MatchFloat32Constant(left, &k1) &&
2256 matcher_.MatchFloat32Constant(right, &k2)) {
2257 return __ Word32Constant(k1 == k2);
2258 }
2259 break;
2260 }
2261 case RegisterRepresentation::Float64(): {
2262 if (double k1, k2; matcher_.MatchFloat64Constant(left, &k1) &&
2263 matcher_.MatchFloat64Constant(right, &k2)) {
2264 return __ Word32Constant(k1 == k2);
2265 }
2266 break;
2267 }
2268 case RegisterRepresentation::Tagged(): {
2269 if (Handle<HeapObject> o1, o2;
2270 matcher_.MatchHeapConstant(left, &o1) &&
2271 matcher_.MatchHeapConstant(right, &o2)) {
2272 UnparkedScopeIfNeeded unparked(broker);
2273 if (IsString(*o1) && IsString(*o2)) {
2274 // If handles refer to the same object, we can eliminate the
2275 // check.
2276 if (o1.address() == o2.address()) return __ Word32Constant(1);
2277 // But if they are different, we cannot eliminate the
2278 // comparison, because the objects might be different now, but
2279 // if they contain the same content, they might be internalized
2280 // to the same object eventually.
2281 break;
2282 }
2283 return __ Word32Constant(o1.address() == o2.address());
2284 }
2285 break;

2291 if (rep.IsWord()) {
2292 WordRepresentation rep_w{rep};
2293 // x - y == 0 => x == y
2294 if (V<Word> x, y; matcher_.MatchWordSub(left, &x, &y, rep_w) &&
2295 matcher_.MatchZero(right)) {
2296 return ReduceCompareEqual(x, y, rep);

2298 {
2299 // ((x >> shift_amount) & mask) == k
2300 // => (x & (mask << shift_amount)) == (k << shift_amount)
2301 V<Word> shift, x, mask_op;
2302 int shift_amount;
2303 uint64_t mask, k;
2304 if (matcher_.MatchBitwiseAnd(left, &shift, &mask_op, rep_w) &&
2305 matcher_.MatchConstantRightShift(shift, &x, rep_w,
2306 &shift_amount) &&
2307 matcher_.MatchIntegralWordConstant(mask_op, rep_w, &mask) &&
2308 matcher_.MatchIntegralWordConstant(right, rep_w, &k) &&
2309 mask <= rep.MaxUnsignedValue() >> shift_amount &&
2310 k <= rep.MaxUnsignedValue() >> shift_amount) {
2311 return ReduceCompareEqual(
2312 __ WordBitwiseAnd(
