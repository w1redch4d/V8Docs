47Tagged<Object> ThrowRedeclarationError(Isolate* isolate, Handle<String> name,
48 RedeclarationType redeclaration_type) {
49 HandleScope scope(isolate);
50 if (redeclaration_type == RedeclarationType::kSyntaxError) {
51 THROW_NEW_ERROR_RETURN_FAILURE(
52 isolate, NewSyntaxError(MessageTemplate::kVarRedeclaration, name));
53 } else {
54 THROW_NEW_ERROR_RETURN_FAILURE(
55 isolate, NewTypeError(MessageTemplate::kVarRedeclaration, name));
56 }

59// May throw a RedeclarationError.
60Tagged<Object> DeclareGlobal(Isolate* isolate,
61 DirectHandle<JSGlobalObject> global,
62 Handle<String> name, Handle<Object> value,
63 PropertyAttributes attr, bool is_var,
64 RedeclarationType redeclaration_type) {
65 DirectHandle<ScriptContextTable> script_contexts(
66 global->native_context()->script_context_table(), isolate);
67 VariableLookupResult lookup;
68 if (script_contexts->Lookup(name, &lookup) &&
69 IsLexicalVariableMode(lookup.mode)) {
70 // ES#sec-globaldeclarationinstantiation 6.a:
71 // If envRec.HasLexicalDeclaration(name) is true, throw a SyntaxError
72 // exception.
73 return ThrowRedeclarationError(isolate, name,
74 RedeclarationType::kSyntaxError);
75 }
76
77 // Do the lookup own properties only, see ES5 erratum.
78 LookupIterator::Configuration lookup_config(
79 LookupIterator::Configuration::OWN_SKIP_INTERCEPTOR);
80 if (!is_var) {
81 // For function declarations, use the interceptor on the declaration. For
82 // non-functions, use it only on initialization.
83 lookup_config = LookupIterator::Configuration::OWN;
84 }
85 LookupIterator it(isolate, global, name, global, lookup_config);
86 Maybe<PropertyAttributes> maybe = JSReceiver::GetPropertyAttributes(&it);
87 if (maybe.IsNothing()) return ReadOnlyRoots(isolate).exception();
88
89 if (it.IsFound()) {
90 PropertyAttributes old_attributes = maybe.FromJust();
91 // The name was declared before; check for conflicting re-declarations.
92
93 // Skip var re-declarations.
94 if (is_var) return ReadOnlyRoots(isolate).undefined_value();
95
96 if ((old_attributes & DONT_DELETE) != 0) {
97 // Only allow reconfiguring globals to functions in user code (no
98 // natives, which are marked as read-only).
99 DCHECK_EQ(attr & READ_ONLY, 0);
100
101 // Check whether we can reconfigure the existing property into a
102 // function.
103 if (old_attributes & READ_ONLY || old_attributes & DONT_ENUM ||
104 (it.state() == LookupIterator::ACCESSOR)) {
105 // ECMA-262 section 15.1.11 GlobalDeclarationInstantiation 5.d:
106 // If hasRestrictedGlobal is true, throw a SyntaxError exception.
107 // ECMA-262 section 18.2.1.3 EvalDeclarationInstantiation 8.a.iv.1.b:
108 // If fnDefinable is false, throw a TypeError exception.
109 return ThrowRedeclarationError(isolate, name, redeclaration_type);
110 }
111 // If the existing property is not configurable, keep its attributes. Do
112 attr = old_attributes;
113 }
114
115 // If the current state is ACCESSOR, this could mean it's an AccessorInfo
116 // type property. We are not allowed to call into such setters during global
117 // function declaration since this would break e.g., onload. Meaning
118 // 'function onload() {}' would invalidly register that function as the
119 // onload callback. To avoid this situation, we first delete the property
120 // before readding it as a regular data property below.
121 if (it.state() == LookupIterator::ACCESSOR) it.Delete();
122 }
123
124 if (!is_var) it.Restart();
125
126 // Define or redefine own property.
127 RETURN_FAILURE_ON_EXCEPTION(
128 isolate, JSObject::DefineOwnPropertyIgnoreAttributes(&it, value, attr));
129
130 return ReadOnlyRoots(isolate).undefined_value();
131}

180 HandleScope scope(isolate);
181 DCHECK_EQ(2, args.length());
182
183 DirectHandle<FixedArray> declarations = args.at<FixedArray>(0);
184 DirectHandle<JSFunction> closure = args.at<JSFunction>(1);
185
186 DirectHandle<JSGlobalObject> global(isolate->global_object());
187 DirectHandle<Context> context(isolate->context(), isolate);
188
189 DirectHandle<ClosureFeedbackCellArray> closure_feedback_cell_array(
190 closure->has_feedback_vector()
191 ? closure->feedback_vector()->closure_feedback_cell_array()
192 : closure->closure_feedback_cell_array(),
193 isolate);
194
195 // Traverse the name/value pairs and set the properties.
196 int length = declarations->length();
197 FOR_WITH_HANDLE_SCOPE(isolate, int, i = 0, i, i < length, i++, {
198 Handle<Object> decl(declarations->get(i), isolate);
199 Handle<String> name;
200 Handle<Object> value;
201 bool is_var = IsString(*decl);
202
203 if (is_var) {
204 name = Cast<String>(decl);
205 value = isolate->factory()->undefined_value();
206 } else {
207 DirectHandle<SharedFunctionInfo> sfi = Cast<SharedFunctionInfo>(decl);
208 name = handle(sfi->Name(), isolate);
209 int index = Smi::ToInt(declarations->get(++i));
210 DirectHandle<FeedbackCell> feedback_cell(
211 closure_feedback_cell_array->get(index), isolate);
212 value = Factory::JSFunctionBuilder(isolate, sfi, context)
213 .set_feedback_cell(feedback_cell)
214 .Build();
215 }
216
217 // Compute the property attributes. According to ECMA-262,
218 // the property must be non-configurable except in eval.
219 Tagged<Script> script = Cast<Script>(closure->shared()->script());
220 PropertyAttributes attr =
221 script->compilation_type() == Script::CompilationType::kEval
222 ? NONE
223 : DONT_DELETE;
224
225 // ES#sec-globaldeclarationinstantiation 5.d:
226 // If hasRestrictedGlobal is true, throw a SyntaxError exception.
227 Tagged<Object> result =
228 DeclareGlobal(isolate, global, name, value, attr, is_var,
229 RedeclarationType::kSyntaxError);
230 if (IsException(result)) return result;
231 });
232
233 return ReadOnlyRoots(isolate).undefined_value();
234}

346Tagged<Object> DeclareEvalHelper(Isolate* isolate, Handle<String> name,
347 Handle<Object> value) {
348 // Declarations are always made in a function, native, eval, or script
349 // context, or a declaration block scope. Since this is called from eval, the
350 // context passed is the context of the caller, which may be some nested
351 // context and not the declaration context.
352 Handle<Context> context(isolate->context()->declaration_context(), isolate);
353
354 // For debug-evaluate we always use sloppy eval, in which case context could
355 // also be a module context. As module contexts re-use the extension slot
356 // we need to check for this.
357 const bool is_debug_evaluate_in_module =
358 isolate->context()->IsDebugEvaluateContext() &&
359 context->IsModuleContext();
360
361 DCHECK(context->IsFunctionContext() || IsNativeContext(*context) ||
362 context->IsScriptContext() || context->IsEvalContext() ||
363 (context->IsBlockContext() &&
364 context->scope_info()->is_declaration_scope()) ||
365 is_debug_evaluate_in_module);
366
367 bool is_var = IsUndefined(*value, isolate);
368 DCHECK_IMPLIES(!is_var, IsJSFunction(*value));
369
370 int index;
371 PropertyAttributes attributes;
372 InitializationFlag init_flag;
373 VariableMode mode;
374
375 DirectHandle<Object> holder =
376 Context::Lookup(context, name, DONT_FOLLOW_CHAINS, &index, &attributes,
377 &init_flag, &mode);
378 DCHECK(holder.is_null() || !IsSourceTextModule(*holder));
379 DCHECK(!isolate->has_exception());
380
381 DirectHandle<JSObject> object;
382
383 if (attributes != ABSENT && IsJSGlobalObject(*holder)) {
384 // ES#sec-evaldeclarationinstantiation 8.a.iv.1.b:
385 // If fnDefinable is false, throw a TypeError exception.
386 return DeclareGlobal(isolate, Cast<JSGlobalObject>(holder), name, value,
387 NONE, is_var, RedeclarationType::kTypeError);
388 }
389 if (context->has_extension() && IsJSGlobalObject(context->extension())) {
390 DirectHandle<JSGlobalObject> global(
391 Cast<JSGlobalObject>(context->extension()), isolate);
392 return DeclareGlobal(isolate, global, name, value, NONE, is_var,
393 RedeclarationType::kTypeError);
394 } else if (context->IsScriptContext()) {
395 DCHECK(IsJSGlobalObject(context->global_object()));
396 DirectHandle<JSGlobalObject> global(
397 Cast<JSGlobalObject>(context->global_object()), isolate);
398 return DeclareGlobal(isolate, global, name, value, NONE, is_var,
399 RedeclarationType::kTypeError);
400 }
401
402 if (attributes != ABSENT) {
403 DCHECK_EQ(NONE, attributes);
404
405 // Skip var re-declarations.
406 if (is_var) return ReadOnlyRoots(isolate).undefined_value();
407
408 if (index != Context::kNotFound) {
409 DCHECK(holder.is_identical_to(context));
410 context->set(index, *value);
411 return ReadOnlyRoots(isolate).undefined_value();
412 }
413
414 object = Cast<JSObject>(holder);
415
416 } else if (context->has_extension() && !is_debug_evaluate_in_module) {
417 object = direct_handle(context->extension_object(), isolate);
418 DCHECK(IsJSContextExtensionObject(*object));
419 } else if (context->scope_info()->HasContextExtensionSlot() &&
420 !is_debug_evaluate_in_module) {
421 // Sloppy varblock and function contexts might not have an extension object
422 // yet. Sloppy eval will never have an extension object, as vars are hoisted
423 // out, and lets are known statically.
424 DCHECK((context->IsBlockContext() &&
425 context->scope_info()->is_declaration_scope()) ||
426 context->IsFunctionContext());
427 DCHECK(context->scope_info()->SloppyEvalCanExtendVars());
428 object =
429 isolate->factory()->NewJSObject(isolate->context_extension_function());
430 context->set_extension(*object);
431 {
432 Tagged<ScopeInfo> scope_info = context->scope_info();
433 if (!scope_info->SomeContextHasExtension()) {
434 scope_info->mark_some_context_has_extension();
435 DependentCode::DeoptimizeDependencyGroups(
436 isolate, scope_info, DependentCode::kEmptyContextExtensionGroup);
437 }
438 }
439 } else {
440 THROW_NEW_ERROR_RETURN_FAILURE(
441 isolate,
442 NewEvalError(MessageTemplate::kVarNotAllowedInEvalScope, name));
443 }
444
445 RETURN_FAILURE_ON_EXCEPTION(isolate, JSObject::SetOwnPropertyIgnoreAttributes(
446 object, name, value, NONE));
447
448 return ReadOnlyRoots(isolate).undefined_value();
449}

471// Find the arguments of the JavaScript function invocation that called
472// into C++ code. Collect these in a newly allocated array of handles.
473DirectHandleVector<Object> GetCallerArguments(Isolate* isolate) {
474 // Find frame containing arguments passed to the caller.
475 JavaScriptStackFrameIterator it(isolate);
476 JavaScriptFrame* frame = it.frame();
477 std::vector<Tagged<SharedFunctionInfo>> functions;
478 frame->GetFunctions(&functions);
479 if (functions.size() > 1) {
480 int inlined_jsframe_index = static_cast<int>(functions.size()) - 1;
481 TranslatedState translated_values(frame);
482 translated_values.Prepare(frame->fp());
483
484 int argument_count = 0;
485 TranslatedFrame* translated_frame =
486 translated_values.GetArgumentsInfoFromJSFrameIndex(
487 inlined_jsframe_index, &argument_count);
488 TranslatedFrame::iterator iter = translated_frame->begin();
489
490 // Skip the function.
491 iter++;
492
493 // Skip the receiver.
494 iter++;
495 argument_count--;
496
497 DirectHandleVector<Object> param_data(isolate, argument_count);
498 bool should_deoptimize = false;
499 for (int i = 0; i < argument_count; i++) {
500 // If we materialize any object, we should deoptimize the frame because we
501 // might alias an object that was eliminated by escape analysis.
502 should_deoptimize = should_deoptimize || iter->IsMaterializedObject();
503 DirectHandle<Object> value = iter->GetValue();
504 param_data[i] = value;
505 iter++;
506 }
507
508 if (should_deoptimize) {
509 translated_values.StoreMaterializedValuesAndDeopt(frame);
510 }
511
512 return param_data;
513 } else {
514 int args_count = frame->GetActualArgumentCount();
515 DirectHandleVector<Object> param_data(isolate, args_count);
516 for (int i = 0; i < args_count; i++) {
517 DirectHandle<Object> val =
518 DirectHandle<Object>(frame->GetParameter(i), isolate);
519 param_data[i] = val;
520 }
521 return param_data;
522 }

525template <typename T>
526Handle<JSObject> NewSloppyArguments(Isolate* isolate,
527 DirectHandle<JSFunction> callee,
528 T parameters, int argument_count) {
529 CHECK(!IsDerivedConstructor(callee->shared()->kind()));
530 DCHECK(callee->shared()->has_simple_parameters());
531 Handle<JSObject> result =
532 isolate->factory()->NewArgumentsObject(callee, argument_count);
533
534 // Allocate the elements if needed.
535 int parameter_count =
536 callee->shared()->internal_formal_parameter_count_without_receiver();
537 if (argument_count > 0) {
538 if (parameter_count > 0) {
539 int mapped_count = std::min(argument_count, parameter_count);
540
541 // Store the context and the arguments array at the beginning of the
542 // parameter map.
543 DirectHandle<Context> context(isolate->context(), isolate);
544 DirectHandle<FixedArray> arguments = isolate->factory()->NewFixedArray(
545 argument_count, AllocationType::kYoung);
546
547 DirectHandle<SloppyArgumentsElements> parameter_map =
548 isolate->factory()->NewSloppyArgumentsElements(
549 mapped_count, context, arguments, AllocationType::kYoung);
550
551 result->set_map(isolate,
552 isolate->native_context()->fast_aliased_arguments_map());
553 result->set_elements(*parameter_map);
554
555 // Loop over the actual parameters backwards.
556 int index = argument_count - 1;
557 while (index >= mapped_count) {
558 // These go directly in the arguments array and have no
559 // corresponding slot in the parameter map.
560 arguments->set(index, parameters[index]);
561 --index;
562 }
563
564 DirectHandle<ScopeInfo> scope_info(callee->shared()->scope_info(),
565 isolate);
566
567 // First mark all mappable slots as unmapped and copy the values into the
568 // arguments object.
569 for (int i = 0; i < mapped_count; i++) {
570 arguments->set(i, parameters[i]);
571 parameter_map->set_mapped_entries(
572 i, *isolate->factory()->the_hole_value());
573 }
574
575 // Walk all context slots to find context allocated parameters. Mark each
576 // found parameter as mapped.
577 ReadOnlyRoots roots{isolate};
578 for (int i = 0; i < scope_info->ContextLocalCount(); i++) {
579 if (!scope_info->ContextLocalIsParameter(i)) continue;
580 int parameter = scope_info->ContextLocalParameterNumber(i);
581 if (parameter >= mapped_count) continue;
582 arguments->set_the_hole(roots, parameter);
583 Tagged<Smi> slot = Smi::FromInt(scope_info->ContextHeaderLength() + i);
584 parameter_map->set_mapped_entries(parameter, slot);
585 }
586 } else {
587 // If there is no aliasing, the arguments object elements are not
588 // special in any way.
589 DirectHandle<FixedArray> elements = isolate->factory()->NewFixedArray(
590 argument_count, AllocationType::kYoung);
591 result->set_elements(*elements);
592 for (int i = 0; i < argument_count; ++i) {
593 elements->set(i, parameters[i]);
594 }
595 }
596 }
597 return result;
598}

638RUNTIME_FUNCTION(Runtime_NewStrictArguments) {
639 HandleScope scope(isolate);
640 DCHECK_EQ(1, args.length());
641 DirectHandle<JSFunction> callee = args.at<JSFunction>(0);
642 // This generic runtime function can also be used when the caller has been
643 // inlined, we use the slow but accurate {GetCallerArguments}.

646 DirectHandle<JSObject> result =

648 if (argument_count) {
649 DirectHandle<FixedArray> array =

793MaybeHandle<Object> LoadLookupSlot(Isolate* isolate, Handle<String> name,
794 ShouldThrow should_throw,
795 Handle<Object>* receiver_return = nullptr) {
796 int index;
797 PropertyAttributes attributes;
798 InitializationFlag flag;
799 VariableMode mode;
800 Handle<Context> context(isolate->context(), isolate);
801 Handle<Object> holder = Context::Lookup(context, name, FOLLOW_CHAINS, &index,
802 &attributes, &flag, &mode);
803 if (isolate->has_exception()) return MaybeHandle<Object>();
804
805 if (!holder.is_null() && IsSourceTextModule(*holder)) {
806 Handle<Object> receiver = isolate->factory()->undefined_value();
807 if (receiver_return) *receiver_return = receiver;
808 return SourceTextModule::LoadVariable(
809 isolate, Cast<SourceTextModule>(holder), index);
810 }
811 if (index != Context::kNotFound) {
812 DCHECK(IsContext(*holder));
813 DirectHandle<Context> holder_context = Cast<Context>(holder);
814 // If the "property" we were looking for is a local variable, the
815 // receiver is the global object; see ECMA-262, 3rd., 10.1.6 and 10.2.3.
816 Handle<Object> receiver = isolate->factory()->undefined_value();
817 Handle<Object> value = handle(holder_context->get(index), isolate);
818 // Check for uninitialized bindings.
819 if (flag == kNeedsInitialization && IsTheHole(*value, isolate)) {
820 THROW_NEW_ERROR(isolate,
821 NewReferenceError(MessageTemplate::kNotDefined, name));
822 }
823 DCHECK(!IsTheHole(*value, isolate));
824 if (receiver_return) *receiver_return = receiver;
825 if (v8_flags.script_context_mutable_heap_number &&
826 holder_context->IsScriptContext()) {
827 return handle(*Context::LoadScriptContextElement(holder_context, index,
828 value, isolate),
829 isolate);
830 }
831 return value;
832 }
833
834 // Otherwise, if the slot was found the holder is a context extension
835 // object, subject of a with, or a global object. We read the named
836 // property from it.
837 if (!holder.is_null()) {
838 // No need to unhole the value here. This is taken care of by the
839 // GetProperty function.
840 Handle<Object> value;
841 ASSIGN_RETURN_ON_EXCEPTION(
842 isolate, value,
843 Object::GetProperty(isolate, Cast<JSAny>(holder), name));
844 if (receiver_return) {
845 *receiver_return =
846 (IsJSGlobalObject(*holder) || IsJSContextExtensionObject(*holder))
847 ? Cast<Object>(isolate->factory()->undefined_value())
848 : holder;
849 }
850 return value;
851 }
852
853 if (should_throw == kThrowOnError) {
854 // The property doesn't exist - throw exception.
855 THROW_NEW_ERROR(isolate,
856 NewReferenceError(MessageTemplate::kNotDefined, name));
857 }
858
859 // The property doesn't exist - return undefined.
860 if (receiver_return) *receiver_return = isolate->factory()->undefined_value();
861 return isolate->factory()->undefined_value();
862}

919MaybeHandle<Object> StoreLookupSlot(
920 Isolate* isolate, Handle<Context> context, Handle<String> name,
921 Handle<Object> value, LanguageMode language_mode,
922 ContextLookupFlags context_lookup_flags = FOLLOW_CHAINS) {
923 int index;
924 PropertyAttributes attributes;
925 InitializationFlag flag;
926 VariableMode mode;
927 bool is_sloppy_function_name;
928 Handle<Object> holder =
929 Context::Lookup(context, name, context_lookup_flags, &index, &attributes,
930 &flag, &mode, &is_sloppy_function_name);
931 if (holder.is_null()) {
932 // In case of JSProxy, an exception might have been thrown.
933 if (isolate->has_exception()) return MaybeHandle<Object>();
934 } else if (IsSourceTextModule(*holder)) {
935 if ((attributes & READ_ONLY) == 0) {
936 SourceTextModule::StoreVariable(Cast<SourceTextModule>(holder), index,
937 value);
938 } else {
939 THROW_NEW_ERROR(isolate,
940 NewTypeError(MessageTemplate::kConstAssign, name));
941 }
942 return value;
943 }
944 // The property was found in a context slot.
945 if (index != Context::kNotFound) {
946 auto holder_context = Cast<Context>(holder);
947 if (flag == kNeedsInitialization &&
948 IsTheHole(holder_context->get(index), isolate)) {
949 THROW_NEW_ERROR(isolate,
950 NewReferenceError(MessageTemplate::kNotDefined, name));
951 }
952 if ((attributes & READ_ONLY) == 0) {
953 if ((v8_flags.script_context_mutable_heap_number ||
954 v8_flags.const_tracking_let) &&
955 holder_context->IsScriptContext()) {
956 Context::StoreScriptContextAndUpdateSlotProperty(holder_context, index,
957 value, isolate);
958 } else {
959 Cast<Context>(holder)->set(index, *value);
960 }
961 } else if (!is_sloppy_function_name || is_strict(language_mode)) {
962 THROW_NEW_ERROR(isolate,
963 NewTypeError(MessageTemplate::kConstAssign, name));
964 }
965 return value;
966 }
967
968 // Slow case: The property is not in a context slot. It is either in a
969 // context extension object, a property of the subject of a with, or a
970 // property of the global object.
971 DirectHandle<JSReceiver> object;
972 if (attributes != ABSENT) {
973 // The property exists on the holder.
974 object = Cast<JSReceiver>(holder);
975 } else if (is_strict(language_mode)) {
976 // If absent in strict mode: throw.
977 THROW_NEW_ERROR(isolate,
978 NewReferenceError(MessageTemplate::kNotDefined, name));
979 } else {
980 // If absent in sloppy mode: add the property to the global object.
981 object = direct_handle(context->global_object(), isolate);
982 }
983
984 ASSIGN_RETURN_ON_EXCEPTION(isolate, value,
985 Object::SetProperty(isolate, object, name, value));
986 return value;
987}
