35// If {to} is nullable, it means that null passes the check.
36// {from} may change in compiler optimization passes as the object's type gets
37// narrowed.
38// TODO(12166): Add modules if we have cross-module inlining.
39struct WasmTypeCheckConfig {
40 wasm::ValueType from;
41 const wasm::ValueType to;
42};
43
44V8_INLINE std::ostream& operator<<(std::ostream& os,
45 WasmTypeCheckConfig const& p) {
46 return os << p.from.name() << " -> " << p.to.name();
47}
48
49V8_INLINE size_t hash_value(WasmTypeCheckConfig const& p) {
50 return base::hash_combine(p.from.raw_bit_field(), p.to.raw_bit_field());
51}
52
53V8_INLINE bool operator==(const WasmTypeCheckConfig& p1,
54 const WasmTypeCheckConfig& p2) {
55 return p1.from == p2.from && p1.to == p2.to;
56}
