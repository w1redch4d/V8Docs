38 DisallowGarbageCollection no_alloc;

504MaybeHandle<Map> Map::CopyWithField(Isolate* isolate, DirectHandle<Map> map,
505 DirectHandle<Name> name,
506 Handle<FieldType> type,
507 PropertyAttributes attributes,
508 PropertyConstness constness,
509 Representation representation,
510 TransitionFlag flag) {
511 DCHECK(map->instance_descriptors(isolate)
512 ->Search(*name, map->NumberOfOwnDescriptors())
513 .is_not_found());
514
515 // Ensure the descriptor array does not get too big.
516 if (map->NumberOfOwnDescriptors() >= kMaxNumberOfDescriptors) {
517 return MaybeHandle<Map>();
518 }
519
520 // Compute the new index for new field.
521 int index = map->NextFreePropertyIndex();
522
523 if (map->instance_type() == JS_CONTEXT_EXTENSION_OBJECT_TYPE) {
524 constness = PropertyConstness::kMutable;
525 representation = Representation::Tagged();
526 type = FieldType::Any(isolate);
527 } else {
528 Map::GeneralizeIfCanHaveTransitionableFastElementsKind(
529 isolate, map->instance_type(), &representation, &type);
530 }
531
532 MaybeObjectHandle wrapped_type = WrapFieldType(type);
533
534 Descriptor d = Descriptor::DataField(name, index, attributes, constness,
535 representation, wrapped_type);
536 Handle<Map> new_map = Map::CopyAddDescriptor(isolate, map, &d, flag);
537 new_map->AccountAddedPropertyField();
538 return new_map;
539}

570
571bool Map::InstancesNeedRewriting(Tagged<Map> target,
572 int target_number_of_fields,
573 int target_inobject, int target_unused,
574 int* old_number_of_fields,
575 ConcurrencyMode cmode) const {
576 // If fields were added (or removed), rewrite the instance.
577 *old_number_of_fields = NumberOfFields(cmode);
578 DCHECK(target_number_of_fields >= *old_number_of_fields);
579 if (target_number_of_fields != *old_number_of_fields) return true;
580
581 // If smi descriptors were replaced by double descriptors, rewrite.
582 Tagged<DescriptorArray> old_desc = IsConcurrent(cmode)
583 ? instance_descriptors(kAcquireLoad)
584 : instance_descriptors();
585 Tagged<DescriptorArray> new_desc =
586 IsConcurrent(cmode) ? target->instance_descriptors(kAcquireLoad)
587 : target->instance_descriptors();
588 for (InternalIndex i : IterateOwnDescriptors()) {
589 if (new_desc->GetDetails(i).representation().IsDouble() !=
590 old_desc->GetDetails(i).representation().IsDouble()) {
591 return true;
592 }
593 }
594
595 // If no fields were added, and no inobject properties were removed, setting
596 // the map is sufficient.
597 if (target_inobject == GetInObjectProperties()) return false;
598 // In-object slack tracking may have reduced the object size of the new map.
599 // In that case, succeed if all existing fields were inobject, and they still
600 // fit within the new inobject size.
601 DCHECK(target_inobject < GetInObjectProperties());
602 if (target_number_of_fields <= target_inobject) {
603 DCHECK(target_number_of_fields + target_unused == target_inobject);
604 return false;
605 }
606 // Otherwise, properties will need to be moved to the backing store.
607 return true;
608}

612 ? instance_descriptors(kAcquireLoad)
613 : instance_descriptors();

615 for (InternalIndex i : IterateOwnDescriptors()) {
616 if (descriptors->GetDetails(i).location() == PropertyLocation::kField)
617 result++;
618 }

643 if (is_deprecated()) return;
644 DisallowGarbageCollection no_gc;
645 ReadOnlyRoots roots(isolate);
646 TransitionsAccessor transitions(isolate, *this);
647 transitions.ForEachTransition(

660 }
661 DependentCode::DeoptimizeDependencyGroups(isolate, *this,
662 DependentCode::kTransitionGroup);
663 NotifyLeafMapLayoutChange(isolate);
664}

666// Installs |new_descriptors| over the current instance_descriptors to ensure
667// proper sharing of descriptor arrays.
668void Map::ReplaceDescriptors(Isolate* isolate,
669 Tagged<DescriptorArray> new_descriptors) {
670 PtrComprCageBase cage_base(isolate);
671 // Don't overwrite the empty descriptor array or initial map's descriptors.
672 if (NumberOfOwnDescriptors() == 0 ||
673 IsUndefined(GetBackPointer(cage_base), isolate)) {
674 return;
675 }
676
677 Tagged<DescriptorArray> to_replace = instance_descriptors(cage_base);
678 // Replace descriptors by new_descriptors in all maps that share it. The old
679 // descriptors will not be trimmed in the mark-compactor, we need to mark
680 // all its elements.
681 Tagged<Map> current = *this;
682#ifndef V8_DISABLE_WRITE_BARRIERS
683 WriteBarrier::ForDescriptorArray(to_replace,
684 to_replace->number_of_descriptors());
685#endif
686 while (current->instance_descriptors(cage_base) == to_replace) {
687 Tagged<Map> next;
688 if (!current->TryGetBackPointer(cage_base, &next)) {
689 break; // Stop overwriting at initial map.
690 }
691 current->SetEnumLength(kInvalidEnumCacheSentinel);
692 current->UpdateDescriptors(isolate, new_descriptors,
693 current->NumberOfOwnDescriptors());
694 current = next;
695 }
696 set_owns_descriptors(false);
697}

700 DisallowGarbageCollection no_gc;
701 Tagged<Map> result = *this;
702 while (true) {

773Tagged<Map> Map::TryReplayPropertyTransitions(Isolate* isolate,
774 Tagged<Map> old_map,
775 ConcurrencyMode cmode) {
776 DisallowGarbageCollection no_gc;
777
778 const int root_nof = NumberOfOwnDescriptors();
779 const int old_nof = old_map->NumberOfOwnDescriptors();
780 // TODO(jgruber,chromium:1239009): The main thread should use non-atomic
781 // reads, but this currently leads to odd behavior (see the linked bug).
782 // Investigate and fix this properly. Also below and in called functions.
783 Tagged<DescriptorArray> old_descriptors =
784 old_map->instance_descriptors(isolate, kAcquireLoad);
785
786 Tagged<Map> new_map = *this;
787 for (InternalIndex i : InternalIndex::Range(root_nof, old_nof)) {
788 PropertyDetails old_details = old_descriptors->GetDetails(i);
789 Tagged<Map> transition =
790 TransitionsAccessor(isolate, new_map, IsConcurrent(cmode))
791 .SearchTransition(old_descriptors->GetKey(i), old_details.kind(),
792 old_details.attributes());
793 if (transition.is_null()) return Map();
794 new_map = transition;
795 Tagged<DescriptorArray> new_descriptors =
796 new_map->instance_descriptors(isolate, kAcquireLoad);
797
798 PropertyDetails new_details = new_descriptors->GetDetails(i);
799 DCHECK_EQ(old_details.kind(), new_details.kind());
800 DCHECK_EQ(old_details.attributes(), new_details.attributes());
801 if (!IsGeneralizableTo(old_details.constness(), new_details.constness())) {
802 return Map();
803 }
804 DCHECK(IsGeneralizableTo(old_details.location(), new_details.location()));
805 if (!old_details.representation().fits_into(new_details.representation())) {
806 return Map();
807 }
808 if (new_details.location() == PropertyLocation::kField) {
809 if (new_details.kind() == PropertyKind::kData) {
810 Tagged<FieldType> new_type = new_descriptors->GetFieldType(i);
811 DCHECK_EQ(PropertyKind::kData, old_details.kind());
812 DCHECK_EQ(PropertyLocation::kField, old_details.location());
813 Tagged<FieldType> old_type = old_descriptors->GetFieldType(i);
814 if (!FieldType::NowIs(old_type, new_type)) {
815 return Map();
816 }
817 } else {
818 DCHECK_EQ(PropertyKind::kAccessor, new_details.kind());
819#ifdef DEBUG
820 Tagged<FieldType> new_type = new_descriptors->GetFieldType(i);
821 DCHECK(IsAny(new_type));
822#endif
823 UNREACHABLE();
824 }
825 } else {
826 DCHECK_EQ(PropertyLocation::kDescriptor, new_details.location());
827 if (old_details.location() == PropertyLocation::kField ||
828 old_descriptors->GetStrongValue(i) !=
829 new_descriptors->GetStrongValue(i)) {
830 return Map();
831 }
832 }
833 }
834 if (new_map->NumberOfOwnDescriptors() != old_nof) return Map();
835 return new_map;
836}

861void Map::EnsureDescriptorSlack(Isolate* isolate, DirectHandle<Map> map,
862 int slack) {
863 // Only supports adding slack to owned descriptors.
864 CHECK(map->owns_descriptors());
865
866 DirectHandle<DescriptorArray> descriptors(map->instance_descriptors(isolate),
867 isolate);
868 int old_size = map->NumberOfOwnDescriptors();
869 if (slack <= descriptors->number_of_slack_descriptors()) return;
870
871 DirectHandle<DescriptorArray> new_descriptors =
872 DescriptorArray::CopyUpTo(isolate, descriptors, old_size, slack);
873
874 DisallowGarbageCollection no_gc;
875 if (old_size == 0) {
876 map->UpdateDescriptors(isolate, *new_descriptors,
877 map->NumberOfOwnDescriptors());
878 return;
879 }
880
881 // If the source descriptors had an enum cache we copy it. This ensures
882 // that the maps to which we push the new descriptor array back can rely
883 // on a cache always being available once it is set. If the map has more
884 // enumerated descriptors than available in the original cache, the cache
885 // will be lazily replaced by the extended cache when needed.
886 new_descriptors->CopyEnumCacheFrom(*descriptors);
887
888 // Replace descriptors by new_descriptors in all maps that share it. The old
889 // descriptors will not be trimmed in the mark-compactor, we need to mark
890 // all its elements.
891#ifndef V8_DISABLE_WRITE_BARRIERS
892 WriteBarrier::ForDescriptorArray(*descriptors,
893 descriptors->number_of_descriptors());
894#endif
895
896 // Update the descriptors from {map} (inclusive) until the initial map
897 // (exclusive). In the case that {map} is the initial map, update it.
898 map->UpdateDescriptors(isolate, *new_descriptors,
899 map->NumberOfOwnDescriptors());
900 Tagged<Object> next = map->GetBackPointer();
901 if (IsUndefined(next, isolate)) return;
902
903 Tagged<Map> current = Cast<Map>(next);
904 while (current->instance_descriptors(isolate) == *descriptors) {
905 next = current->GetBackPointer();
906 if (IsUndefined(next, isolate)) break;
907 current->UpdateDescriptors(isolate, *new_descriptors,
908 current->NumberOfOwnDescriptors());
909 current = Cast<Map>(next);
910 }
911}

993Tagged<Map> Map::FindElementsKindTransitionedMap(Isolate* isolate,
994 MapHandlesSpan candidates,
995 ConcurrencyMode cmode) {
996 DisallowGarbageCollection no_gc;
997
998 if (IsDetached(isolate)) return Map();
999
1000 ElementsKind kind = elements_kind();
1001 bool is_packed = IsFastPackedElementsKind(kind);
1002
1003 Tagged<Map> transition;
1004 if (IsTransitionableFastElementsKind(kind)) {
1005 // Check the state of the root map.
1006 Tagged<Map> root_map = FindRootMap(isolate);
1007 if (!EquivalentToForElementsKindTransition(root_map, cmode)) return Map();
1008 root_map = root_map->LookupElementsTransitionMap(isolate, kind, cmode);
1009 DCHECK(!root_map.is_null());
1010 // Starting from the next existing elements kind transition try to
1011 // replay the property transitions that does not involve instance rewriting
1012 // (ElementsTransitionAndStoreStub does not support that).
1013 for (root_map = root_map->ElementsTransitionMap(isolate, cmode);
1014 !root_map.is_null() && root_map->has_fast_elements();
1015 root_map = root_map->ElementsTransitionMap(isolate, cmode)) {
1016 // If root_map's elements kind doesn't match any of the elements kind in
1017 // the candidates there is no need to do any additional work.
1018 if (!HasElementsKind(candidates, root_map->elements_kind())) continue;
1019 Tagged<Map> current =
1020 root_map->TryReplayPropertyTransitions(isolate, *this, cmode);
1021 if (current.is_null()) continue;
1022 if (InstancesNeedRewriting(current, cmode)) continue;
1023
1024 const bool current_is_packed =
1025 IsFastPackedElementsKind(current->elements_kind());
1026 if (ContainsMap(candidates, current) &&
1027 (is_packed || !current_is_packed)) {
1028 transition = current;
1029 is_packed = is_packed && current_is_packed;
1030 }
1031 }
1032 }
1033 return transition;
1034}

1036static Tagged<Map> FindClosestElementsTransition(Isolate* isolate,
1037 Tagged<Map> map,
1038 ElementsKind to_kind,
1039 ConcurrencyMode cmode) {
1040 DisallowGarbageCollection no_gc;
1041 // Ensure we are requested to search elements kind transition "near the root".
1042 DCHECK_EQ(map->FindRootMap(isolate)->NumberOfOwnDescriptors(),
1043 map->NumberOfOwnDescriptors());
1044 Tagged<Map> current_map = map;
1045
1046 ElementsKind kind = map->elements_kind();
1047 while (kind != to_kind) {
1048 Tagged<Map> next_map = current_map->ElementsTransitionMap(isolate, cmode);
1049 if (next_map.is_null()) return current_map;
1050 kind = next_map->elements_kind();
1051 current_map = next_map;
1052 }
1053
1054 DCHECK_EQ(to_kind, current_map->elements_kind());
1055 return current_map;

1079Handle<Map> Map::TransitionElementsTo(Isolate* isolate, Handle<Map> map,
1080 ElementsKind to_kind) {
1081 ElementsKind from_kind = map->elements_kind();
1082 if (from_kind == to_kind) return map;
1083
1084 Tagged<Context> native_context = isolate->context()->native_context();
1085 if (from_kind == FAST_SLOPPY_ARGUMENTS_ELEMENTS) {
1086 if (*map == native_context->fast_aliased_arguments_map()) {
1087 DCHECK_EQ(SLOW_SLOPPY_ARGUMENTS_ELEMENTS, to_kind);
1088 return handle(native_context->slow_aliased_arguments_map(), isolate);
1089 }
1090 } else if (from_kind == SLOW_SLOPPY_ARGUMENTS_ELEMENTS) {
1091 if (*map == native_context->slow_aliased_arguments_map()) {
1092 DCHECK_EQ(FAST_SLOPPY_ARGUMENTS_ELEMENTS, to_kind);
1093 return handle(native_context->fast_aliased_arguments_map(), isolate);
1094 }
1095 } else if (IsFastElementsKind(from_kind) && IsFastElementsKind(to_kind)) {
1096 // Reuse map transitions for JSArrays.
1097 DisallowGarbageCollection no_gc;
1098 if (native_context->GetInitialJSArrayMap(from_kind) == *map) {
1099 Tagged<Object> maybe_transitioned_map =
1100 native_context->get(Context::ArrayMapIndex(to_kind));
1101 if (IsMap(maybe_transitioned_map)) {
1102 return handle(Cast<Map>(maybe_transitioned_map), isolate);
1103 }
1104 }
1105 }
1106
1107 DCHECK(!IsUndefined(*map, isolate));
1108 // Check if we can go back in the elements kind transition chain.
1109 if (IsHoleyElementsKind(from_kind) &&
1110 to_kind == GetPackedElementsKind(from_kind) &&
1111 IsMap(map->GetBackPointer()) &&
1112 Cast<Map>(map->GetBackPointer())->elements_kind() == to_kind) {
1113 return handle(Cast<Map>(map->GetBackPointer()), isolate);
1114 }
1115
1116 bool allow_store_transition = IsTransitionElementsKind(from_kind);
1117 // Only store fast element maps in ascending generality.
1118 if (IsFastElementsKind(to_kind)) {
1119 allow_store_transition =
1120 allow_store_transition && IsTransitionableFastElementsKind(from_kind) &&
1121 IsMoreGeneralElementsKindTransition(from_kind, to_kind);
1122 }
1123
1124 if (!allow_store_transition) {
1125 return Map::CopyAsElementsKind(isolate, map, to_kind, OMIT_TRANSITION);
1126 }
1127
1128 return MapUpdater{isolate, map}.ReconfigureElementsKind(to_kind);
1129}

1131static Handle<Map> AddMissingElementsTransitions(Isolate* isolate,
1132 Handle<Map> map,
1133 ElementsKind to_kind) {
1134 DCHECK(IsTransitionElementsKind(map->elements_kind()));
1135
1136 Handle<Map> current_map = map;
1137
1138 ElementsKind kind = map->elements_kind();
1139 TransitionFlag flag;
1140 if (map->IsDetached(isolate)) {
1141 flag = OMIT_TRANSITION;
1142 } else {
1143 flag = INSERT_TRANSITION;
1144 if (IsFastElementsKind(kind)) {
1145 while (kind != to_kind && !IsTerminalElementsKind(kind)) {
1146 kind = GetNextTransitionElementsKind(kind);
1147 current_map = Map::CopyAsElementsKind(isolate, current_map, kind, flag);
1148 }
1149 }
1150 }
1151
1152 // In case we are exiting the fast elements kind system, just add the map in
1153 // the end.
1154 if (kind != to_kind) {
1155 current_map = Map::CopyAsElementsKind(isolate, current_map, to_kind, flag);
1156 }
1157
1158 DCHECK(current_map->elements_kind() == to_kind);
1159 return current_map;

1220bool Map::ShouldCheckForReadOnlyElementsInPrototypeChain(Isolate* isolate) {
1221 // If this map has TypedArray elements kind, we won't look at the prototype
1222 // chain, so we can return early.
1223 if (IsTypedArrayElementsKind(elements_kind())) return false;
1224
1225 for (PrototypeIterator iter(isolate, *this); !iter.IsAtEnd();
1226 iter.Advance()) {
1227 // Be conservative, don't look into any JSReceivers that may have custom
1228 // elements. For example, into JSProxies, String wrappers (which have have
1229 // non-configurable, non-writable elements), API objects, etc.
1230 if (IsCustomElementsReceiverMap(iter.GetCurrent()->map())) return true;
1231
1232 Tagged<JSObject> current = iter.GetCurrent<JSObject>();
1233 ElementsKind elements_kind = current->GetElementsKind(isolate);
1234 // If this prototype has TypedArray elements kind, we won't look any further
1235 // in the prototype chain, so we can return early.
1236 if (IsTypedArrayElementsKind(elements_kind)) return false;
1237 if (IsFrozenElementsKind(elements_kind)) return true;
1238
1239 if (IsDictionaryElementsKind(elements_kind) &&
1240 current->element_dictionary(isolate)->requires_slow_elements()) {
1241 return true;
1242 }
1243
1244 if (IsSlowArgumentsElementsKind(elements_kind)) {
1245 Tagged<SloppyArgumentsElements> elements =
1246 Cast<SloppyArgumentsElements>(current->elements(isolate));
1247 Tagged<Object> arguments = elements->arguments();
1248 if (Cast<NumberDictionary>(arguments)->requires_slow_elements()) {
1249 return true;
1250 }
1251 }
1252 }
1253
1254 return false;
1255}

1257Handle<Map> Map::RawCopy(Isolate* isolate, DirectHandle<Map> src_handle,
1258 int instance_size, int inobject_properties) {
1259 Handle<Map> result = isolate->factory()->NewMap(
1260 src_handle, src_handle->instance_type(), instance_size,
1261 TERMINAL_FAST_ELEMENTS_KIND, inobject_properties);
1262
1263 // We have to set the bitfields before any potential GCs could happen because
1264 // heap verification might fail otherwise.
1265 {
1266 DisallowGarbageCollection no_gc;
1267 Tagged<Map> src = *src_handle;
1268 Tagged<Map> raw = *result;
1269 raw->set_constructor_or_back_pointer(src->GetConstructorRaw());
1270 raw->set_bit_field(src->bit_field());
1271 raw->set_bit_field2(src->bit_field2());
1272 int new_bit_field3 = src->bit_field3();
1273 new_bit_field3 = Bits3::OwnsDescriptorsBit::update(new_bit_field3, true);
1274 new_bit_field3 =
1275 Bits3::NumberOfOwnDescriptorsBits::update(new_bit_field3, 0);
1276 new_bit_field3 = Bits3::EnumLengthBits::update(new_bit_field3,
1277 kInvalidEnumCacheSentinel);
1278 new_bit_field3 = Bits3::IsDeprecatedBit::update(new_bit_field3, false);
1279 new_bit_field3 =
1280 Bits3::IsInRetainedMapListBit::update(new_bit_field3, false);
1281 if (!src->is_dictionary_map()) {
1282 new_bit_field3 = Bits3::IsUnstableBit::update(new_bit_field3, false);
1283 }
1284 // Same as bit_field comment above.
1285 raw->set_bit_field3(new_bit_field3);
1286 raw->clear_padding();

1293Handle<Map> Map::Normalize(Isolate* isolate, DirectHandle<Map> fast_map,
1294 ElementsKind new_elements_kind,
1295 DirectHandle<JSPrototype> new_prototype,
1296 PropertyNormalizationMode mode, bool use_cache,
1297 const char* reason) {
1298 DCHECK(!fast_map->is_dictionary_map());
1299
1300 Tagged<Map> meta_map = fast_map->map();
1301 if (fast_map->is_prototype_map()) {
1302 use_cache = false;
1303 }
1304 DirectHandle<NormalizedMapCache> cache;
1305 if (use_cache) {
1306 Tagged<Object> normalized_map_cache =
1307 meta_map->native_context()->normalized_map_cache();
1308 use_cache = !IsUndefined(normalized_map_cache, isolate);
1309 if (use_cache) {
1310 cache = Cast<NormalizedMapCache>(
1311 direct_handle(normalized_map_cache, isolate));
1312 }
1313 }
1314
1315 Handle<Map> new_map;
1316 if (use_cache && cache
1317 ->Get(isolate, fast_map, new_elements_kind,
1318 new_prototype.is_null() ? fast_map->prototype()
1319 : *new_prototype,
1320 mode)
1321 .ToHandle(&new_map)) {
1322#ifdef VERIFY_HEAP
1323 if (v8_flags.verify_heap) new_map->DictionaryMapVerify(isolate);
1324#endif
1325#ifdef ENABLE_SLOW_DCHECKS
1326 if (v8_flags.enable_slow_asserts) {
1327 // The cached map should match newly created normalized map bit-by-bit,
1328 // except for the code cache, which can contain some ICs which can be
1329 // applied to the shared map, dependent code and weak cell cache.
1330 DirectHandle<Map> fresh = Map::CopyNormalized(isolate, fast_map, mode);
1331 fresh->set_elements_kind(new_elements_kind);
1332 if (!new_prototype.is_null()) {
1333 Map::SetPrototype(isolate, fresh, new_prototype);
1334 }
1335
1336 static_assert(Map::kPrototypeValidityCellOffset ==
1337 Map::kDependentCodeOffset + kTaggedSize);
1338 DCHECK_EQ(0, memcmp(reinterpret_cast<void*>(fresh->address()),
1339 reinterpret_cast<void*>(new_map->address()),
1340 Map::kBitField3Offset));
1341 // The IsInRetainedMapListBit might be different if the {new_map}
1342 // that we got from the {cache} was already embedded into optimized
1343 // code somewhere.
1344 // The IsMigrationTargetBit might be different if the {new_map} from
1345 // {cache} has already been marked as a migration target.
1346 constexpr int ignored_bit_field3_bits =
1347 Bits3::IsInRetainedMapListBit::kMask |
1348 Bits3::IsMigrationTargetBit::kMask;
1349 DCHECK_EQ(fresh->bit_field3() & ~ignored_bit_field3_bits,
1350 new_map->bit_field3() & ~ignored_bit_field3_bits);
1351 int offset = Map::kBitField3Offset + kInt32Size;
1352 DCHECK_EQ(0, memcmp(reinterpret_cast<void*>(fresh->address() + offset),
1353 reinterpret_cast<void*>(new_map->address() + offset),
1354 Map::kDependentCodeOffset - offset));
1355 offset = Map::kPrototypeValidityCellOffset + kTaggedSize;
1356 if (new_map->is_prototype_map()) {
1357 // For prototype maps, the PrototypeInfo is not copied.
1358 static_assert(Map::kTransitionsOrPrototypeInfoOffset ==
1359 Map::kPrototypeValidityCellOffset + kTaggedSize);
1360 offset = kTransitionsOrPrototypeInfoOffset + kTaggedSize;
1361 DCHECK_EQ(fresh->raw_transitions(), Smi::zero());
1362 }
1363 DCHECK_EQ(0, memcmp(reinterpret_cast<void*>(fresh->address() + offset),
1364 reinterpret_cast<void*>(new_map->address() + offset),
1365 Map::kSize - offset));
1366 }
1367#endif
1368 if (v8_flags.log_maps) {
1369 LOG(isolate, MapEvent("NormalizeCached", fast_map, new_map, reason));
1370 }
1371 } else {
1372 new_map = Map::CopyNormalized(isolate, fast_map, mode);
1373 new_map->set_elements_kind(new_elements_kind);
1374 if (!new_prototype.is_null()) {
1375 Map::SetPrototype(isolate, new_map, new_prototype);
1376 DCHECK(new_map->is_dictionary_map() && !new_map->is_deprecated());
1377 }
1378 if (use_cache) {
1379 cache->Set(isolate, fast_map, new_map);
1380 }
1381 if (v8_flags.log_maps) {
1382 LOG(isolate, MapEvent("Normalize", fast_map, new_map, reason));
1383 }
1384 }
1385 fast_map->NotifyLeafMapLayoutChange(isolate);
1386 return new_map;
1387}

1506 DirectHandle<DescriptorArray> descriptors,
1507 Descriptor* descriptor) {
1508 // Sanity check. This path is only to be taken if the map owns its descriptor
1509 // array, implying that its NumberOfOwnDescriptors equals the number of

1511 DCHECK_EQ(map->NumberOfOwnDescriptors(),

1536 descriptors->Append(descriptor);
1537 result->InitializeDescriptors(isolate, *descriptors);
1538 }
1539
1540 DCHECK(result->NumberOfOwnDescriptors() == map->NumberOfOwnDescriptors() + 1);

1577 Isolate* isolate, DirectHandle<Map> map,
1578 DirectHandle<DescriptorArray> descriptors, TransitionFlag flag,
1579 MaybeDirectHandle<Name> maybe_name, const char* reason,
1580 TransitionKindFlag transition_kind) {
1581 DCHECK(descriptors->IsSortedNoDuplicates());
1582
1583 Handle<Map> result = CopyDropDescriptors(isolate, map);
1584 bool is_connected = false;
1585
1586 // Properly mark the {result} if the {name} is an "interesting symbol".
1587 DirectHandle<Name> name;
1588 if (maybe_name.ToHandle(&name) && name->IsInteresting(isolate)) {
1589 result->set_may_have_interesting_properties(true);
1590 }
1591
1592 if (map->is_prototype_map()) {
1593 result->InitializeDescriptors(isolate, *descriptors);
1594 } else {

1598
1599 DCHECK(!maybe_name.is_null());
1600 ConnectTransition(isolate, map, result, name, transition_kind);
1601 is_connected = true;
1602 } else if ((transition_kind == PROTOTYPE_TRANSITION &&
1603 v8_flags.move_prototype_transitions_first) ||
1604 isolate->bootstrapper()->IsActive()) {
1605 // Prototype transitions are always between root maps. UpdatePrototype
1606 // uses the MapUpdater and instance migration. Thus, field generalization
1607 // is allowed to happen lazily.
1608 DCHECK_IMPLIES(transition_kind == PROTOTYPE_TRANSITION,
1609 IsUndefined(map->GetBackPointer()));
1610 result->InitializeDescriptors(isolate, *descriptors);
1611 } else {
1612 DCHECK_IMPLIES(transition_kind == PROTOTYPE_TRANSITION,
1613 !v8_flags.move_prototype_transitions_first);
1614 descriptors->GeneralizeAllFields(transition_kind == PROTOTYPE_TRANSITION);
1615 result->InitializeDescriptors(isolate, *descriptors);
1616 }
1617 }
1618 if (v8_flags.log_maps && !is_connected) {
1619 LOG(isolate,
1620 MapEvent("ReplaceDescriptors", map, result, reason,
1621 maybe_name.is_null() ? DirectHandle<HeapObject>() : name));
1622 }
1623 return result;
1624}

1626// Creates transition tree starting from |split_map| and adding all descriptors
1627// starting from descriptor with index |split_map|.NumberOfOwnDescriptors().
1628// The way how it is done is tricky because of GC and special descriptors
1629// marking logic.
1630Handle<Map> Map::AddMissingTransitions(
1631 Isolate* isolate, DirectHandle<Map> split_map,
1632 DirectHandle<DescriptorArray> descriptors) {
1633 DCHECK(descriptors->IsSortedNoDuplicates());
1634 int split_nof = split_map->NumberOfOwnDescriptors();
1635 int nof_descriptors = descriptors->number_of_descriptors();
1636 CHECK_LT(split_nof, nof_descriptors);
1637
1638 // Start with creating last map which will own full descriptors array.
1639 // This is necessary to guarantee that GC will mark the whole descriptor
1640 // array if any of the allocations happening below fail.
1641 // Number of unused properties is temporarily incorrect and the layout
1642 // descriptor could unnecessarily be in slow mode but we will fix after
1643 // all the other intermediate maps are created.
1644 // Also the last map might have interesting symbols, we temporarily set
1645 // the flag and clear it right before the descriptors are installed. This
1646 // makes heap verification happy and ensures the flag ends up accurate.
1647 Handle<Map> last_map = CopyDropDescriptors(isolate, split_map);
1648 last_map->InitializeDescriptors(isolate, *descriptors);
1649 last_map->SetInObjectUnusedPropertyFields(0);
1650 last_map->set_may_have_interesting_properties(true);
1651
1652 // During creation of intermediate maps we violate descriptors sharing
1653 // invariant since the last map is not yet connected to the transition tree
1654 // we create here. But it is safe because GC never trims map's descriptors
1655 // if there are no dead transitions from that map and this is exactly the
1656 // case for all the intermediate maps we create here.
1657 DirectHandle<Map> map = split_map;
1658 for (InternalIndex i : InternalIndex::Range(split_nof, nof_descriptors - 1)) {
1659 DirectHandle<Map> new_map = CopyDropDescriptors(isolate, map);
1660 // Force connection of these maps to prevent split_map being a root map to
1661 // be treated as detached.
1662 InstallDescriptors(isolate, map, new_map, i, descriptors,
1663 /* force_connect */ true);
1664 DCHECK_EQ(*new_map->GetBackPointer(), *map);
1665 map = new_map;
1666 }
1667 map->NotifyLeafMapLayoutChange(isolate);
1668 last_map->set_may_have_interesting_properties(false);
1669 InstallDescriptors(isolate, map, last_map, InternalIndex(nof_descriptors - 1),
1670 descriptors);
1671 return last_map;
1672}

1674// Since this method is used to rewrite an existing transition tree, it can
1675// always insert transitions without checking.
1676void Map::InstallDescriptors(Isolate* isolate, DirectHandle<Map> parent,
1677 DirectHandle<Map> child,
1678 InternalIndex new_descriptor,
1679 DirectHandle<DescriptorArray> descriptors,
1680 bool force_connect) {
1681 DCHECK(descriptors->IsSortedNoDuplicates());

1683 child->SetInstanceDescriptors(isolate, *descriptors,

1687 if (details.location() == PropertyLocation::kField) {

1690
1691 DirectHandle<Name> name(descriptors->GetKey(new_descriptor), isolate);
1692 if (parent->may_have_interesting_properties() ||

1694 child->set_may_have_interesting_properties(true);
1695 }
1696 ConnectTransition(isolate, parent, child, name, SIMPLE_PROPERTY_TRANSITION,
1697 force_connect);
1698}

1835Handle<Map> Map::Create(Isolate* isolate, int inobject_properties) {
1836 Handle<Map> copy_handle =
1837 Copy(isolate,
1838 direct_handle(isolate->object_function()->initial_map(), isolate),
1839 "MapCreate");
1840 DisallowGarbageCollection no_gc;
1841 Tagged<Map> copy = *copy_handle;
1842
1843 // Check that we do not overflow the instance size when adding the extra
1844 // inobject properties. If the instance size overflows, we allocate as many
1845 // properties as we can as inobject properties.
1846 if (inobject_properties > JSObject::kMaxInObjectProperties) {
1847 inobject_properties = JSObject::kMaxInObjectProperties;
1848 }
1849
1850 int new_instance_size =
1851 JSObject::kHeaderSize + kTaggedSize * inobject_properties;
1852
