23// Returns true if op in node_group have same kind.
24bool IsSameOpAndKind(const Operation& op0, const Operation& op1) {
25#define CASE(operation) \
26 case Opcode::k##operation: { \
27 using Op = operation##Op; \
28 return op0.Cast<Op>().kind == op1.Cast<Op>().kind; \
29 }
30 if (op0.opcode != op1.opcode) {
31 return false;
32 }
33 switch (op0.opcode) {
34 CASE(Simd128Unary)
35 CASE(Simd128Binop)
36 CASE(Simd128Shift)
37 CASE(Simd128Ternary)
38 CASE(Simd128Splat)
39 default:
40 return true;
41 }
42#undef CASE
43}

57// This class is the wrapper for StoreOp/LoadOp, which is helpful to calcualte
58// the relative offset between two StoreOp/LoadOp.
59template <typename Op,
60 typename = std::enable_if_t<
61 std::is_same_v<Op, StoreOp> || std::is_same_v<Op, LoadOp> ||
62 std::is_same_v<Op, Simd128LoadTransformOp>>>
63class StoreLoadInfo {
64 public:
65 StoreLoadInfo(const Graph* graph, const Op* op)
66 : op_(op), offset_(op->offset) {
67 base_ = &graph->Get(op->base());
68 if constexpr (std::is_same_v<Op, Simd128LoadTransformOp>) {
69 DCHECK_EQ(offset_, 0);
70 const WordBinopOp* add_op = base_->TryCast<WordBinopOp>();
71 if (!add_op || add_op->kind != WordBinopOp::Kind::kAdd ||
72 add_op->rep != WordRepresentation::Word64()) {
73 SetInvalid();
74 return;
75 }
76 base_ = &graph->Get(add_op->left());
77 const ConstantOp* const_op =
78 graph->Get(add_op->right()).TryCast<ConstantOp>();
79 if (!const_op) {
80 SetInvalid();
81 return;
82 }
83 // const_op->word64() won't be greater than uint32::max under 32-bits wasm
84 // memory.
85 DCHECK_EQ(const_op->word64(), const_op->word32());
86 offset_ = const_op->word32();
87 }
88 const ChangeOp* change = nullptr;
89 if constexpr (std::is_same_v<Op, Simd128LoadTransformOp>) {
90 change = graph->Get(op->index()).template TryCast<ChangeOp>();
91 } else {
92 if (!op->index().has_value()) return;
93 change = graph->Get(op->index().value()).template TryCast<ChangeOp>();
94 }
95 if (change == nullptr) {
96 SetInvalid();
97 return;
98 }
99 DCHECK_EQ(change->kind, ChangeOp::Kind::kZeroExtend);
100 const Operation* change_input = &graph->Get(change->input());
101 if (const ConstantOp* const_op = change_input->TryCast<ConstantOp>()) {
102 DCHECK_EQ(const_op->kind, ConstantOp::Kind::kWord32);
103 int new_offset;
104 if (base::bits::SignedAddOverflow32(static_cast<int>(const_op->word32()),
105 offset_, &new_offset)) {
106 // offset is overflow
107 SetInvalid();
108 return;
109 }
110 offset_ = new_offset;
111 return;
112 }
113 index_ = change_input;
114 }
115
116 std::optional<int> operator-(const StoreLoadInfo<Op>& rhs) const {
117 DCHECK(IsValid() && rhs.IsValid());
118 bool calculatable = base_ == rhs.base_ && index_ == rhs.index_;
119
120 if constexpr (std::is_same_v<Op, Simd128LoadTransformOp>) {
121 calculatable &= (op_->load_kind == rhs.op_->load_kind &&
122 op_->transform_kind == rhs.op_->transform_kind);
123 } else {
124 calculatable &= (op_->kind == rhs.op_->kind);
125 }
126
127 if constexpr (std::is_same_v<Op, StoreOp>) {
128 // TODO(v8:12716) If one store has a full write barrier and the other has
129 // no write barrier, consider combine them with a full write barrier.
130 calculatable &= (op_->write_barrier == rhs.op_->write_barrier);
131 }
132
133 if (calculatable) {
134 return offset_ - rhs.offset_;
135 }
136 return {};
137 }
138
139 bool IsValid() const { return op_ != nullptr; }
140
141 const Operation* index() const { return index_; }
142 int offset() const { return offset_; }
143 const Op* op() const { return op_; }
144
145 private:
146 void SetInvalid() { op_ = nullptr; }
147
148 const Op* op_;
149 const Operation* base_ = nullptr;
150 const Operation* index_ = nullptr;
151 int offset_;
152};

166// Return whether the stride of node_group equal to a specific value
167template <class Op, class Info>
168bool LoadStrideEqualTo(const Graph& graph, const NodeGroup& node_group,
169 int stride) {
170 base::SmallVector<Info, 2> load_infos;
171 for (OpIndex op_idx : node_group) {
172 const Operation& op = graph.Get(op_idx);
173 const Op& load_op = op.Cast<Op>();
174 Info info(&graph, &load_op);
175 if (!info.IsValid()) {
176 return false;
177 }
178 load_infos.push_back(info);
179 }
180 return load_infos[1] - load_infos[0] == stride;
181}

257bool SLPTree::HasInputDependencies(const NodeGroup& node_group) {
258 DCHECK_EQ(node_group.size(), 2);
259 if (node_group[0] == node_group[1]) return false;
260 OpIndex start, end;
261 if (node_group[0] < node_group[1]) {
262 start = node_group[0];
263 end = node_group[1];
264 } else {
265 start = node_group[1];
266 end = node_group[0];
267 }
268 // Do BFS from the end node and see if there is a path to the start node.
269 ZoneQueue<OpIndex> to_visit(phase_zone_);
270 to_visit.push(end);
271 while (!to_visit.empty()) {
272 OpIndex to_visit_node = to_visit.front();
273 Operation& op = graph_.Get(to_visit_node);
274 to_visit.pop();
275 for (OpIndex input : op.inputs()) {
276 if (input == start) {
277 return true;
278 } else if (input > start) {
279 // We should ensure that there is no back edge.
280 DCHECK_LT(input, to_visit_node);
281 to_visit.push(input);
282 }
283 }
284 }
285 return false;
286}

299PackNode* SLPTree::NewForcePackNode(const NodeGroup& node_group,
300 ForcePackNode::ForcePackType type,
301 const Graph& graph) {
302 // Currently we only support force packing two nodes.
303 DCHECK_EQ(node_group.size(), 2);
304 // We should guarantee that the one node in the NodeGroup does not rely on the
305 // result of the other. Because it is costly to force pack such candidates.
306 // For example, we have four nodes {A, B, C, D} which are connected by input
307 // edges: A <-- B <-- C <-- D. If {B} and {D} are already packed into a
308 // PackNode and we want to force pack {A} and {C}, we need to duplicate {B}
309 // and the result will be {A, B, C}, {B, D}. This increase the cost of
310 // ForcePack so currently we do not support it.
311 if (HasInputDependencies(node_group)) {
312 TRACE("ForcePackNode %s(#%d, #%d) failed due to input dependencies.\n",
313 GetSimdOpcodeName(graph_.Get(node_group[0])).c_str(),
314 node_group[0].id(), node_group[1].id());
315 return nullptr;
316 }
317
318 TRACE("ForcePackNode %s(#%d, #%d)\n",
319 GetSimdOpcodeName(graph_.Get(node_group[0])).c_str(),
320 node_group[0].id(), node_group[1].id());
321 ForcePackNode* pnode =
322 phase_zone_->New<ForcePackNode>(phase_zone_, node_group, type);
323 for (OpIndex node : node_group) {
324 node_to_packnode_[node] = pnode;
325 }
326
327 return pnode;
328}

378PackNode* SLPTree::NewCommutativePackNodeAndRecurs(const NodeGroup& node_group,
379 unsigned depth) {
380 PackNode* pnode = NewPackNode(node_group);
381
382 const Simd128BinopOp& op0 = graph_.Get(node_group[0]).Cast<Simd128BinopOp>();
383 const Simd128BinopOp& op1 = graph_.Get(node_group[1]).Cast<Simd128BinopOp>();
384
385 bool same_kind =
386 (op0.left() == op1.left()) ||
387 IsSameOpAndKind(graph_.Get(op0.left()), graph_.Get(op1.left()));
388 bool need_swap = Simd128BinopOp::IsCommutative(op0.kind) && !same_kind;
389 if (need_swap) {
390 TRACE("Change the order of binop operands\n");
391 }
392 for (int i = 0; i < 2; ++i) {
393 // Swap the left and right input if necessary
394 unsigned node1_input_index = need_swap ? 1 - i : i;
395 NodeGroup operands(graph_.Get(node_group[0]).input(i),
396 graph_.Get(node_group[1]).input(node1_input_index));
397
398 PackNode* child = BuildTreeRec(operands, depth + 1);
399 if (child) {
400 pnode->SetOperand(i, child);
401 } else {
402 return nullptr;
403 }
404 }
405 return pnode;
406}

441ShufflePackNode* SLPTree::Try256ShuffleMatchLoad8x8U(
442 const NodeGroup& node_group, const uint8_t* shuffle0,
443 const uint8_t* shuffle1) {
444 uint8_t shuffle_copy0[kSimd128Size];
445 uint8_t shuffle_copy1[kSimd128Size];
446
447 V<Simd128> op_idx0 = node_group[0];
448 V<Simd128> op_idx1 = node_group[1];
449 const Simd128ShuffleOp& op0 = graph_.Get(op_idx0).Cast<Simd128ShuffleOp>();
450 const Simd128ShuffleOp& op1 = graph_.Get(op_idx1).Cast<Simd128ShuffleOp>();
451
452 if (op0.left() == op0.right() || op1.left() == op1.right()) {
453 // Here shuffle couldn't be swizzle
454 return nullptr;
455 }
456
457 CopyChars(shuffle_copy0, shuffle0, kSimd128Size);
458 CopyChars(shuffle_copy1, shuffle1, kSimd128Size);
459
460 bool need_swap, is_swizzle;
461
462#define CANONICALIZE_SHUFFLE(n) \
463 wasm::SimdShuffle::CanonicalizeShuffle(false, shuffle_copy##n, &need_swap, \
464 &is_swizzle); \
465 if (is_swizzle) { \
466 /* Here shuffle couldn't be swizzle*/ \
467 return nullptr; \
468 } \
469 V<Simd128> shuffle##n##_left_idx = need_swap ? op##n.right() : op##n.left(); \
470 V<Simd128> shuffle##n##_right_idx = need_swap ? op##n.left() : op##n.right();
471
472 CANONICALIZE_SHUFFLE(0);
473 CANONICALIZE_SHUFFLE(1);
474
475#undef CANONICALIZE_SHUFFLE
476 if (shuffle0_left_idx != shuffle1_left_idx) {
477 // Not the same left
478 return nullptr;
479 }
480
481 const Simd128LoadTransformOp* load_transform =
482 graph_.Get(shuffle0_left_idx).TryCast<Simd128LoadTransformOp>();
483
484 if (!load_transform) {
485 // shuffle left is not Simd128LoadTransformOp
486 return nullptr;
487 }
488
489 Simd128ConstantOp* shuffle0_const =
490 graph_.Get(shuffle0_right_idx).TryCast<Simd128ConstantOp>();
491 Simd128ConstantOp* shuffle1_const =
492 graph_.Get(shuffle1_right_idx).TryCast<Simd128ConstantOp>();
493
494 if (!shuffle0_const || !shuffle1_const || !shuffle0_const->IsZero() ||
495 !shuffle1_const->IsZero()) {
496 // Shuffle right is not zero
497 return nullptr;
498 }
499
500 if (load_transform->transform_kind ==
501 Simd128LoadTransformOp::TransformKind::k64Zero) {
502 /*
503 should look like this:
504 shuffle0 = 0,x,x,x, 1,x,x,x 2,x,x,x 3,x,x,x
505 shuffle1 = 4,x,x,x, 5,x,x,x 6,x,x,x 7,x,x,x
506 x >= 16
507 */
508
509 for (int i = 0; i < kSimd128Size / 4; ++i) {
510 if (shuffle_copy0[i * 4] != i || shuffle_copy1[i * 4] != i + 4) {
511 // not match
512 return nullptr;
513 }
514
515 if (shuffle_copy0[i * 4 + 1] < kSimd128Size ||
516 shuffle_copy0[i * 4 + 2] < kSimd128Size ||
517 shuffle_copy0[i * 4 + 3] < kSimd128Size ||
518 shuffle_copy1[i * 4 + 1] < kSimd128Size ||
519 shuffle_copy1[i * 4 + 2] < kSimd128Size ||
520 shuffle_copy1[i * 4 + 3] < kSimd128Size) {
521 // not match
522 return nullptr;
523 }
524 }
525 TRACE("match load extend 8x8->32x8\n");
526 return NewShufflePackNode(
527 node_group, ShufflePackNode::SpecificInfo::Kind::kS256Load8x8U);
528 }
529 return nullptr;
530}

626// (local.get 7))))
627// (f32.convert_i32_u
628// (i8x16.extract_lane_u 3
629// (local.get 7))))
630std::optional<SLPTree::ExtendIntToF32x4Info>
631SLPTree::TryGetExtendIntToF32x4Info(OpIndex index) {
632 OpIndex current = index;
633 LaneExtendInfo lane_extend_info[4];
634
635 // Get information for lane 1 to lane 3
636 for (int lane_index = 3; lane_index > 0; lane_index--) {
637 const Simd128ReplaceLaneOp* replace_lane =
638 graph_.Get(current)
639 .TryCast<turboshaft::Opmask::kSimd128ReplaceLaneF32x4>();
640 if (!replace_lane) {
641 TRACE("Mismatch in replace lane\n");
642 return {};
643 }
644 const ChangeOp* change =
645 graph_.Get(replace_lane->new_lane()).TryCast<ChangeOp>();
646 if (!change) {
647 TRACE("Mismatch in type convert\n");
648 return {};
649 }
650 const Simd128ExtractLaneOp* extract_lane =
651 graph_.Get(change->input()).TryCast<Simd128ExtractLaneOp>();
652 if (!extract_lane) {
653 TRACE("Mismatch in extract lane\n");
654 return {};
655 }
656 lane_extend_info[lane_index].replace_lane_index = replace_lane->lane;
657 lane_extend_info[lane_index].change_kind = change->kind;
658 lane_extend_info[lane_index].extract_from = extract_lane->input();
659 lane_extend_info[lane_index].extract_kind = extract_lane->kind;
660 lane_extend_info[lane_index].extract_lane_index = extract_lane->lane;
661
662 current = replace_lane->into();
663 }
664
665 // Get information for lane 0(splat)
666 const Simd128SplatOp* splat = graph_.Get(current).TryCast<Simd128SplatOp>();
667 if (!splat) {
668 TRACE("Mismatch in splat\n");
669 return {};
670 }
671 const ChangeOp* change = graph_.Get(splat->input()).TryCast<ChangeOp>();
672 if (!change) {
673 TRACE("Mismatch in splat type convert\n");
674 return {};
675 }
676 const Simd128ExtractLaneOp* extract_lane =
677 graph_.Get(change->input()).TryCast<Simd128ExtractLaneOp>();
678 if (!extract_lane) {
679 TRACE("Mismatch in splat extract lane\n");
680 return {};
681 }
682 lane_extend_info[0].replace_lane_index = 0;
683 lane_extend_info[0].change_kind = change->kind;
684 lane_extend_info[0].extract_from = extract_lane->input();
685 lane_extend_info[0].extract_kind = extract_lane->kind;
686 lane_extend_info[0].extract_lane_index = extract_lane->lane;
687
688 // Pattern matching for f32x4.convert_i32x4(i32x4.extract_lane)
689 for (int i = 0; i < 4; i++) {
690 if (lane_extend_info[i].replace_lane_index != i) {
691 return {};
692 }
693 if (lane_extend_info[i].change_kind != lane_extend_info[0].change_kind ||
694 (lane_extend_info[i].change_kind != ChangeOp::Kind::kSignedToFloat &&
695 lane_extend_info[i].change_kind != ChangeOp::Kind::kUnsignedToFloat)) {
696 return {};
697 }
698 if (lane_extend_info[i].extract_from != lane_extend_info[0].extract_from) {
699 return {};
700 }
701 if (lane_extend_info[i].extract_kind != lane_extend_info[0].extract_kind ||
702 (lane_extend_info[i].extract_kind !=
703 Simd128ExtractLaneOp::Kind::kI8x16S &&
704 lane_extend_info[i].extract_kind !=
705 Simd128ExtractLaneOp::Kind::kI8x16U &&
706 lane_extend_info[i].extract_kind !=
707 Simd128ExtractLaneOp::Kind::kI16x8S &&
708 lane_extend_info[i].extract_kind !=
709 Simd128ExtractLaneOp::Kind::kI16x8U)) {
710 return {};
711 }
712 if (lane_extend_info[i].extract_lane_index !=
713 lane_extend_info[0].extract_lane_index + i) {
714 return {};
715 }
716 }
717
718 ExtendIntToF32x4Info info;
719 info.extend_from = lane_extend_info[0].extract_from;
720 info.start_lane = lane_extend_info[0].extract_lane_index;
721 if (lane_extend_info[0].extract_kind == Simd128ExtractLaneOp::Kind::kI8x16S ||
722 lane_extend_info[0].extract_kind == Simd128ExtractLaneOp::Kind::kI8x16U) {
723 info.lane_size = 1;
724 } else {
725 info.lane_size = 2;
726 }
727 info.is_sign_extract =
728 lane_extend_info[0].extract_kind == Simd128ExtractLaneOp::Kind::kI8x16S ||
729 lane_extend_info[0].extract_kind == Simd128ExtractLaneOp::Kind::kI16x8S;
730 info.is_sign_convert =
731 lane_extend_info[0].change_kind == ChangeOp::Kind::kSignedToFloat;
732
733 return info;
734}

736bool SLPTree::TryMatchExtendIntToF32x4(const NodeGroup& node_group,
737 ExtendIntToF32x4Info* info) {
738 OpIndex node0 = node_group[0];
739 OpIndex node1 = node_group[1];
740 std::optional<ExtendIntToF32x4Info> info0 = TryGetExtendIntToF32x4Info(node0);
741 std::optional<ExtendIntToF32x4Info> info1 = TryGetExtendIntToF32x4Info(node1);
742 if (!info0.has_value() || !info1.has_value()) {
743 return false;
744 }
745
746 if (info0.value().extend_from != info1.value().extend_from ||
747 info0.value().is_sign_extract != info1.value().is_sign_extract ||
748 info0.value().lane_size != info1.value().lane_size ||
749 info0.value().is_sign_convert != info1.value().is_sign_convert) {
750 return false;
751 }
752
753 uint32_t min_lane_index =
754 std::min(info0.value().start_lane, info1.value().start_lane);
755 if (std::abs(info0.value().start_lane - info1.value().start_lane) != 4) {
756 return false;
757 }
758 if (info0.value().lane_size == 1) {
759 if (min_lane_index != 0 && min_lane_index != 8) {
760 return false;
761 }
762 } else {
763 DCHECK_EQ(info0.value().lane_size, 2);
764 if (min_lane_index != 0) {
765 return false;
766 }
767 }
768
769 *info = info0.value();
770 info->start_lane = min_lane_index;
771 return true;
772}

774bool CannotSwapProtectedLoads(OpEffects first, OpEffects second) {
775 EffectDimensions produces = first.produces;
776 // The control flow effects produces by Loads are due to trap handler. We can
777 // ignore this kind of effect when swapping two Loads that both have trap
778 // handler.
779 produces.control_flow = false;
780 return produces.bits() & (second.consumes.bits());
781}
782
783bool IsProtectedLoad(Operation& op) {
784 if (op.opcode == Opcode::kLoad) {
785 return op.Cast<LoadOp>().kind.with_trap_handler;
786 } else if (op.opcode == Opcode::kSimd128LoadTransform) {
787 return op.Cast<Simd128LoadTransformOp>().load_kind.with_trap_handler;
788 }
789 return false;
790}

792bool SLPTree::IsSideEffectFree(OpIndex first, OpIndex second) {
793 DCHECK_LE(first.offset(), second.offset());
794 if (first == second) return true;
795 OpEffects effects = graph().Get(second).Effects();
796 OpIndex prev_node = graph().PreviousIndex(second);
797 while (prev_node != first) {
798 OpEffects prev_effects = graph().Get(prev_node).Effects();
799 if ((IsProtectedLoad(graph().Get(second)) &&
800 IsProtectedLoad(graph().Get(prev_node)))
801 ? CannotSwapProtectedLoads(prev_effects, effects)
802 : CannotSwapOperations(prev_effects, effects)) {
803 TRACE("break side effect %d, %d\n", prev_node.id(), second.id());
804 return false;
805 }
806 prev_node = graph().PreviousIndex(prev_node);
807 }
808 return true;
809}

811bool IsSignExtensionOp(Operation& op) {
812 if (const Simd128UnaryOp* unop = op.TryCast<Simd128UnaryOp>()) {
813 return unop->kind >= Simd128UnaryOp::Kind::kFirstSignExtensionOp &&
814 unop->kind <= Simd128UnaryOp::Kind::kLastSignExtensionOp;
815 } else if (const Simd128BinopOp* binop = op.TryCast<Simd128BinopOp>()) {
816 return binop->kind >= Simd128BinopOp::Kind::kFirstSignExtensionOp &&
817 binop->kind <= Simd128BinopOp::Kind::kLastSignExtensionOp;
818 }
819 return false;

822bool SLPTree::CanBePacked(const NodeGroup& node_group) {
823 OpIndex node0 = node_group[0];
824 OpIndex node1 = node_group[1];
825 Operation& op0 = graph_.Get(node0);
826 Operation& op1 = graph_.Get(node1);
827
828 if (op0.opcode != op1.opcode) {
829 TRACE("Different opcode\n");
830 return false;
831 }
832
833 if (graph().BlockIndexOf(node0) != graph().BlockIndexOf(node1)) {
834 TRACE("Can't pack operations of different basic block\n");
835 return false;
836 }
837
838 auto is_sign_ext = IsSignExtensionOp(op0) && IsSignExtensionOp(op1);
839
840 if (!is_sign_ext && !IsSameOpAndKind(op0, op1)) {
841 TRACE("(%s, %s) have different op\n", GetSimdOpcodeName(op0).c_str(),
842 GetSimdOpcodeName(op1).c_str());
843 return false;
844 }
845
846 if (node0.offset() <= node1.offset() ? !IsSideEffectFree(node0, node1)
847 : !IsSideEffectFree(node1, node0)) {
848 TRACE("Break side effect\n");
849 return false;
850 }
851 return true;
852}

854bool SLPTree::IsEqual(const OpIndex node0, const OpIndex node1) {
855 if (node0 == node1) return true;
856 if (const ConstantOp* const0 = graph_.Get(node0).TryCast<ConstantOp>()) {
857 if (const ConstantOp* const1 = graph_.Get(node1).TryCast<ConstantOp>()) {
858 return *const0 == *const1;
859 }
860 }
861 return false;
862}

894
895PackNode* SLPTree::BuildTreeRec(const NodeGroup& node_group,
896 unsigned recursion_depth) {
897 DCHECK_EQ(node_group.size(), 2);
898
899 OpIndex node0 = node_group[0];
900 OpIndex node1 = node_group[1];
901 Operation& op0 = graph_.Get(node0);
902 Operation& op1 = graph_.Get(node1);
903
904 if (recursion_depth == RecursionMaxDepth) {
905 TRACE("Failed due to max recursion depth!\n");
906 return nullptr;
907 }
908
909 if (!CanBePacked(node_group)) {
910 return nullptr;
911 }
912
913 // Check if this is a duplicate of another entry.
914 bool is_intersected = false;
915 // For revisited node_group, we only need to match from node0.
916 if (PackNode* pnode = GetPackNode(node0)) {
917 const Operation& op = graph_.Get(node0);
918 if (pnode->IsSame(node_group)) {
919 TRACE("Perfect diamond merge at #%d,%s\n", node0.id(),
920 GetSimdOpcodeName(op).c_str());
921 return pnode;
922 }
923
924 // TODO(yolanda): Support other intersect PackNode e.g. overlapped loads.
925 if (!pnode->IsForcePackNode() || recursion_depth < 1) {
926 TRACE("Unsupported partial overlap at #%d,%s!\n", node0.id(),
927 GetSimdOpcodeName(op).c_str());
928 return nullptr;
929 }
930
931 // Match intersect packnodes from current tree.
932 if (auto intersect_packnodes = GetIntersectPackNodes(node0)) {
933 for (auto intersect_pnode : *intersect_packnodes) {
934 if (intersect_pnode->IsSame(node_group)) {
935 TRACE("Perfect diamond merge at intersect pack node #%d,%s, #%d\n",
936 node0.id(), GetSimdOpcodeName(op).c_str(), node1.id());
937 return intersect_pnode;
938 }
939 }
940 }
941
942 // Match intersect packnodes from analyzer
943 if (auto intersect_packnodes = analyzer_->GetIntersectPackNodes(node0)) {
944 for (auto intersect_pnode : *intersect_packnodes) {
945 if (intersect_pnode->IsSame(node_group)) {
946 TRACE("Perfect diamond merge at intersect pack node #%d,%s, #%d\n",
947 node0.id(), GetSimdOpcodeName(op).c_str(), node1.id());
948 return intersect_pnode;
949 }
950 }
951 }
952
953 is_intersected = true;
954 TRACE("Partial overlap at #%d,%s!\n", node0.id(),
955 GetSimdOpcodeName(op).c_str());
956 }
957
958 // Catch overlapped PackNode on the other nodes.
959 if (!is_intersected) {
960 for (int i = 1; i < static_cast<int>(node_group.size()); i++) {
961 const OpIndex op_idx = node_group[i];
962 const Operation& op = graph_.Get(op_idx);
963 if (auto pnode = GetPackNode(op_idx)) {
964 if (!pnode->IsForcePackNode()) {
965 TRACE("Unsupported partial overlap at #%d,%s!\n", op_idx.id(),
966 GetSimdOpcodeName(op).c_str());
967 return nullptr;
968 }
969 } else if (!GetIntersectPackNodes(op_idx) &&
970 !analyzer_->GetIntersectPackNodes(op_idx)) {
971 continue;
972 }
973
974 is_intersected = true;
975 TRACE("Partial overlap at #%d,%s!\n", op_idx.id(),
976 GetSimdOpcodeName(op).c_str());
977 break;
978 }
979 }
980
981 if (is_intersected) {
982 TRACE("Create IntersectPackNode due to partial overlap!\n");
983 PackNode* pnode = NewIntersectPackNode(node_group);
984 return pnode;
985 }
986
987 int value_in_count = op0.input_count;
988
989 switch (op0.opcode) {
990 case Opcode::kSimd128Constant: {
991 PackNode* p = NewPackNode(node_group);
992 return p;
993 }
994
995 case Opcode::kSimd128LoadTransform: {
996 const Simd128LoadTransformOp& transform_op0 =
997 op0.Cast<Simd128LoadTransformOp>();
998 const Simd128LoadTransformOp& transform_op1 =
999 op1.Cast<Simd128LoadTransformOp>();
1000 StoreLoadInfo<Simd128LoadTransformOp> info0(&graph_, &transform_op0);
1001 StoreLoadInfo<Simd128LoadTransformOp> info1(&graph_, &transform_op1);
1002 auto stride = info1 - info0;
1003 if (IsLoadSplat(transform_op0)) {
1004 TRACE("Simd128LoadTransform: LoadSplat\n");
1005 if (IsSplat(node_group) ||
1006 (stride.has_value() && stride.value() == 0)) {
1007 return NewPackNode(node_group);
1008 }
1009 return NewForcePackNode(node_group, ForcePackNode::kGeneral, graph_);
1010 } else if (IsLoadExtend(transform_op0)) {
1011 TRACE("Simd128LoadTransform: LoadExtend\n");
1012 if (stride.has_value()) {
1013 const int value = stride.value();
1014 if (value == kSimd128Size / 2) {
1015 return NewPackNode(node_group);
1016 } else if (value == 0) {
1017 return NewForcePackNode(node_group, ForcePackNode::kSplat, graph_);
1018 }
1019 }
1020 return NewForcePackNode(node_group, ForcePackNode::kGeneral, graph_);
1021 } else {
1022 TRACE("Load Transfrom k64Zero/k32Zero!\n");
1023 DCHECK(transform_op0.transform_kind ==
1024 Simd128LoadTransformOp::TransformKind::k32Zero ||
1025 transform_op0.transform_kind ==
1026 Simd128LoadTransformOp::TransformKind::k64Zero);
1027 if (stride.has_value() && stride.value() == 0) {
1028 return NewForcePackNode(node_group, ForcePackNode::kSplat, graph_);
1029 }
1030 return NewForcePackNode(node_group, ForcePackNode::kGeneral, graph_);
1031 }
1032 }
1033
1034 case Opcode::kLoad: {
1035 TRACE("Load leaf node\n");
1036 const LoadOp& load0 = op0.Cast<LoadOp>();
1037 const LoadOp& load1 = op1.Cast<LoadOp>();
1038 if (load0.loaded_rep != MemoryRepresentation::Simd128() ||
1039 load1.loaded_rep != MemoryRepresentation::Simd128()) {
1040 TRACE("Failed due to non-simd load representation!\n");
1041 return nullptr;
1042 }
1043 StoreLoadInfo<LoadOp> info0(&graph_, &load0);
1044 StoreLoadInfo<LoadOp> info1(&graph_, &load1);
1045 auto stride = info1 - info0;
1046 if (stride.has_value()) {
1047 if (const int value = stride.value(); value == kSimd128Size) {
1048 // TODO(jiepan) Sort load
1049 return NewPackNode(node_group);
1050 } else if (value == 0) {
1051 return NewForcePackNode(node_group, ForcePackNode::kSplat, graph_);
1052 }
1053 }
1054 return NewForcePackNode(node_group, ForcePackNode::kGeneral, graph_);
1055 }
1056 case Opcode::kStore: {
1057 TRACE("Added a vector of stores.\n");
1058 // input: base, value, [index]
1059 PackNode* pnode = NewPackNodeAndRecurs(node_group, 1, 1, recursion_depth);
1060 return pnode;
1061 }
1062 case Opcode::kPhi: {
1063 TRACE("Added a vector of phi nodes.\n");
1064 const PhiOp& phi = graph().Get(node0).Cast<PhiOp>();
1065 if (phi.rep != RegisterRepresentation::Simd128() ||
1066 op0.input_count != op1.input_count) {
1067 TRACE("Failed due to invalid phi\n");
1068 return nullptr;
1069 }
1070 PackNode* pnode =
1071 NewPackNodeAndRecurs(node_group, 0, value_in_count, recursion_depth);
1072 return pnode;
1073 }
1074 case Opcode::kSimd128Unary: {
1075#define UNARY_CASE(op_128, not_used) case Simd128UnaryOp::Kind::k##op_128:
1076#define UNARY_SIGN_EXTENSION_CASE(op_low, not_used1, op_high) \
1077 case Simd128UnaryOp::Kind::k##op_low: { \
1078 if (const Simd128UnaryOp* unop1 = \
1079 op1.TryCast<Opmask::kSimd128##op_high>(); \
1080 unop1 && op0.Cast<Simd128UnaryOp>().input() == unop1->input()) { \
1081 return NewPackNode(node_group); \
1082 } \
1083 [[fallthrough]]; \
1084 } \
1085 case Simd128UnaryOp::Kind::k##op_high: { \
1086 if (op1.Cast<Simd128UnaryOp>().kind == op0.Cast<Simd128UnaryOp>().kind) { \
1087 auto force_pack_type = \
1088 node0 == node1 ? ForcePackNode::kSplat : ForcePackNode::kGeneral; \
1089 return NewForcePackNode(node_group, force_pack_type, graph_); \
1090 } else { \
1091 return nullptr; \
1092 } \
1093 }
1094 switch (op0.Cast<Simd128UnaryOp>().kind) {
1095 SIMD256_UNARY_SIGN_EXTENSION_OP(UNARY_SIGN_EXTENSION_CASE)
1096 SIMD256_UNARY_SIMPLE_OP(UNARY_CASE) {
1097 TRACE("Added a vector of Unary\n");
1098 PackNode* pnode = NewPackNodeAndRecurs(node_group, 0, value_in_count,
1099 recursion_depth);
1100 return pnode;
1101 }
1102 default: {
1103 TRACE("Unsupported Simd128Unary: %s\n",
1104 GetSimdOpcodeName(op0).c_str());
1105 return nullptr;
1106 }
1107 }
1108#undef UNARY_CASE
1109#undef UNARY_SIGN_EXTENSION_CASE
1110 }
1111 case Opcode::kSimd128Binop: {
1112#define BINOP_CASE(op_128, not_used) case Simd128BinopOp::Kind::k##op_128:
1113#define BINOP_SIGN_EXTENSION_CASE(op_low, not_used1, op_high) \
1114 case Simd128BinopOp::Kind::k##op_low: { \
1115 if (const Simd128BinopOp* binop1 = \
1116 op1.TryCast<Opmask::kSimd128##op_high>(); \
1117 binop1 && op0.Cast<Simd128BinopOp>().left() == binop1->left() && \
1118 op0.Cast<Simd128BinopOp>().right() == binop1->right()) { \
1119 return NewPackNode(node_group); \
1120 } \
1121 [[fallthrough]]; \
1122 } \
1123 case Simd128BinopOp::Kind::k##op_high: { \
1124 if (op1.Cast<Simd128BinopOp>().kind == op0.Cast<Simd128BinopOp>().kind) { \
1125 auto force_pack_type = \
1126 node0 == node1 ? ForcePackNode::kSplat : ForcePackNode::kGeneral; \
1127 return NewForcePackNode(node_group, force_pack_type, graph_); \
1128 } else { \
1129 return nullptr; \
1130 } \
1131 }
1132 switch (op0.Cast<Simd128BinopOp>().kind) {
1133 SIMD256_BINOP_SIGN_EXTENSION_OP(BINOP_SIGN_EXTENSION_CASE)
1134 SIMD256_BINOP_SIMPLE_OP(BINOP_CASE) {
1135 TRACE("Added a vector of Binop\n");
1136 PackNode* pnode =
1137 NewCommutativePackNodeAndRecurs(node_group, recursion_depth);
1138 return pnode;
1139 }
1140 default: {
1141 TRACE("Unsupported Simd128Binop: %s\n",
1142 GetSimdOpcodeName(op0).c_str());
1143 return nullptr;
1144 }
1145 }
1146#undef BINOP_CASE
1147#undef BINOP_SIGN_EXTENSION_CASE
1148 }
1149 case Opcode::kSimd128Shift: {
1150 Simd128ShiftOp& shift_op0 = op0.Cast<Simd128ShiftOp>();
1151 Simd128ShiftOp& shift_op1 = op1.Cast<Simd128ShiftOp>();
1152 if (IsEqual(shift_op0.shift(), shift_op1.shift())) {
1153 switch (op0.Cast<Simd128ShiftOp>().kind) {
1154#define SHIFT_CASE(op_128, not_used) case Simd128ShiftOp::Kind::k##op_128:
1155 SIMD256_SHIFT_OP(SHIFT_CASE) {
1156 TRACE("Added a vector of Shift op.\n");
1157 // We've already checked that the "shift by" input of both shifts is
1158 // the same, and we'll only pack the 1st input of the shifts
1159 // together anyways (since on both Simd128 and Simd256, the "shift
1160 // by" input of shifts is a Word32). Thus we only need to check the
1161 // 1st input of the shift when recursing.
1162 constexpr int kShiftValueInCount = 1;
1163 PackNode* pnode = NewPackNodeAndRecurs(
1164 node_group, 0, kShiftValueInCount, recursion_depth);
1165 return pnode;
1166 }
1167#undef SHIFT_CASE
1168 default: {
1169 TRACE("Unsupported Simd128ShiftOp: %s\n",
1170 GetSimdOpcodeName(op0).c_str());
1171 return nullptr;
1172 }
1173 }
1174 }
1175 TRACE("Failed due to SimdShiftOp kind or shift scalar is different!\n");
1176 return nullptr;
1177 }
1178 case Opcode::kSimd128Ternary: {
1179#define TERNARY_CASE(op_128, not_used) case Simd128TernaryOp::Kind::k##op_128:
1180 switch (op0.Cast<Simd128TernaryOp>().kind) {
1181 SIMD256_TERNARY_OP(TERNARY_CASE) {
1182 TRACE("Added a vector of Ternary\n");
1183 PackNode* pnode = NewPackNodeAndRecurs(node_group, 0, value_in_count,
1184 recursion_depth);
1185 return pnode;
1186 }
1187#undef TERNARY_CASE
1188 default: {
1189 TRACE("Unsupported Simd128Ternary: %s\n",
1190 GetSimdOpcodeName(op0).c_str());
1191 return nullptr;
1192 }
1193 }
1194 }
1195
1196 case Opcode::kSimd128Splat: {
1197 if (op0.input(0) != op1.input(0)) {
1198 TRACE("Failed due to different splat input!\n");
1199 return nullptr;
1200 }
1201 PackNode* pnode = NewPackNode(node_group);
1202 return pnode;
1203 }
1204
1205 case Opcode::kSimd128Shuffle: {
1206 // We pack shuffles only if it can match specific patterns. We should
1207 // avoid packing general shuffles because it will cause regression.
1208 const auto& shuffle0 = op0.Cast<Simd128ShuffleOp>().shuffle;
1209 const auto& shuffle1 = op1.Cast<Simd128ShuffleOp>().shuffle;
1210
1211 if (CompareCharsEqual(shuffle0, shuffle1, kSimd128Size)) {
1212 if (IsSplat(node_group)) {
1213 // Check if the shuffle can be replaced by a loadsplat.
1214 // Take load32splat as an example:
1215 // 1. Param0 # be used as load base
1216 // 2. Param1 # be used as load index
1217 // 3. Param2 # be used as store base
1218 // 4. Param3 # be used as store index
1219 // 5. Load128(base, index, offset=0)
1220 // 6. AnyOp
1221 // 7. Shuffle32x4 (1,2, [2,2,2,2])
1222 // 8. Store128(3,4,7, offset=0)
1223 // 9. Store128(3,4,7, offset=16)
1224 //
1225 // We can replace the load128 and shuffle with a loadsplat32:
1226 // 1. Param0 # be used as load base
1227 // 2. Param1 # be used as load index
1228 // 3. Param2 # be used as store base
1229 // 4. Param3 # be used as store index
1230 // 5. Load32Splat256(base, index, offset=4)
1231 // 6. Store256(3,4,7,offset=0)
1232 int index;
1233 if (wasm::SimdShuffle::TryMatchSplat<4>(shuffle0, &index) &&
1234 graph_.Get(op0.input(index >> 2)).opcode == Opcode::kLoad) {
1235 ShufflePackNode* pnode = NewShufflePackNode(
1236 node_group,
1237 ShufflePackNode::SpecificInfo::Kind::kS256Load32Transform);
1238 pnode->info().set_splat_index(index);
1239 return pnode;
1240 } else if (wasm::SimdShuffle::TryMatchSplat<2>(shuffle0, &index) &&
1241 graph_.Get(op0.input(index >> 1)).opcode ==
1242 Opcode::kLoad) {
1243 ShufflePackNode* pnode = NewShufflePackNode(
1244 node_group,
1245 ShufflePackNode::SpecificInfo::Kind::kS256Load64Transform);
1246 pnode->info().set_splat_index(index);
1247 return pnode;
1248 }
1249 } else {
1250#ifdef V8_TARGET_ARCH_X64
1251 if (ShufflePackNode* pnode =
1252 X64TryMatch256Shuffle(node_group, shuffle0, shuffle1)) {
1253 // Manually invoke recur build tree for shuffle node
1254 for (int i = 0; i < value_in_count; ++i) {
1255 NodeGroup operands(graph_.Get(node_group[0]).input(i),
1256 graph_.Get(node_group[1]).input(i));
1257
1258 PackNode* child = BuildTreeRec(operands, recursion_depth + 1);
1259 if (child) {
1260 pnode->SetOperand(i, child);
1261 } else {
1262 return nullptr;
1263 }
1264 }
1265 return pnode;
1266 }
1267#endif // V8_TARGET_ARCH_X64
1268 return nullptr;
1269 }
1270
1271 TRACE("Unsupported Simd128Shuffle\n");
1272 return nullptr;
1273
1274 } else {
1275 return Try256ShuffleMatchLoad8x8U(node_group, shuffle0, shuffle1);
1276 }
1277 }
1278
1279 case Opcode::kSimd128ReplaceLane: {
1280 ExtendIntToF32x4Info info;
1281 if (TryMatchExtendIntToF32x4(node_group, &info)) {
1282 TRACE("Match extend i8x4/i16x4 to f32x4\n");
1283 PackNode* p = NewBundlePackNode(
1284 node_group, info.extend_from, info.start_lane, info.lane_size,
1285 info.is_sign_extract, info.is_sign_convert);
1286 return p;
1287 }
1288 if (recursion_depth < 1) {
1289 TRACE("Do not force pack at root #%d:%s\n", node0.id(),
1290 GetSimdOpcodeName(op0).c_str());
1291 return nullptr;
1292 }
1293 return NewForcePackNode(
1294 node_group,
1295 node0 == node1 ? ForcePackNode::kSplat : ForcePackNode::kGeneral,
1296 graph_);
1297 }
1298
1299 default:
1300 TRACE("Default branch #%d:%s\n", node0.id(),
1301 GetSimdOpcodeName(op0).c_str());
1302 break;
1303 }
1304 return nullptr;
1305}

1340void WasmRevecAnalyzer::ProcessBlock(const Block& block) {
1341 StoreInfoSet simd128_stores(phase_zone_);
1342 for (const Operation& op : base::Reversed(graph_.operations(block))) {
1343 if (const StoreOp* store_op = op.TryCast<StoreOp>()) {
1344 if (store_op->stored_rep == MemoryRepresentation::Simd128()) {
1345 StoreLoadInfo<StoreOp> info(&graph_, store_op);
1346 if (info.IsValid()) {
1347 simd128_stores.insert(info);
1348 }
1349 }
1350 }
1351 // Try to find reduce op which can be used as revec seeds.
1352 if (IsSupportedReduceSeed(op)) {
1353 const Simd128BinopOp& binop = op.Cast<Simd128BinopOp>();
1354 V<Simd128> left_index = binop.left();
1355 V<Simd128> right_index = binop.right();
1356 const Operation& left_op = graph_.Get(left_index);
1357 const Operation& right_op = graph_.Get(right_index);
1358
1359 if (left_index != right_index && left_op.opcode == right_op.opcode &&
1360 IsSameOpAndKind(left_op, right_op)) {
1361 reduce_seeds_.push_back({left_index, right_index});
1362 }
1363 }
1364 }
1365
1366 if (simd128_stores.size() >= 2) {
1367 for (auto it = std::next(simd128_stores.cbegin()),
1368 end = simd128_stores.cend();
1369 it != end;) {
1370 const StoreLoadInfo<StoreOp>& info0 = *std::prev(it);
1371 const StoreLoadInfo<StoreOp>& info1 = *it;
1372 auto diff = info1 - info0;
1373
1374 if (diff.has_value()) {
1375 const int value = diff.value();
1376 DCHECK_GE(value, 0);
1377 if (value == kSimd128Size) {
1378 store_seeds_.push_back(
1379 {graph_.Index(*info0.op()), graph_.Index(*info1.op())});
1380 if (std::distance(it, end) < 2) {
1381 break;
1382 }
1383 std::advance(it, 2);
1384 continue;
1385 }
1386 }
1387 it++;
1388 }
1389 }
1390}

1392void WasmRevecAnalyzer::Run() {
1393 for (Block& block : base::Reversed(graph_.blocks())) {
1394 ProcessBlock(block);
1395 }
1396
1397 if (store_seeds_.empty() && reduce_seeds_.empty()) {
1398 TRACE("Empty seed\n");
1399 return;
1400 }
1401
1402 if (v8_flags.trace_wasm_revectorize) {
1403 PrintF("store seeds:\n");
1404 for (auto pair : store_seeds_) {
1405 PrintF("{\n");
1406 PrintF("#%u ", pair.first.id());
1407 graph_.Get(pair.first).Print();
1408 PrintF("#%u ", pair.second.id());
1409 graph_.Get(pair.second).Print();
1410 PrintF("}\n");
1411 }
1412
1413 PrintF("reduce seeds:\n");
1414 for (auto pair : reduce_seeds_) {
1415 PrintF("{ ");
1416 PrintF("#%u, ", pair.first.id());
1417 PrintF("#%u ", pair.second.id());
1418 PrintF("}\n");
1419 }
1420 }
1421
1422 ZoneVector<std::pair<OpIndex, OpIndex>> all_seeds(
1423 store_seeds_.begin(), store_seeds_.end(), phase_zone_);
1424 all_seeds.insert(all_seeds.end(), reduce_seeds_.begin(), reduce_seeds_.end());
1425
1426 for (auto pair : all_seeds) {
1427 NodeGroup roots(pair.first, pair.second);
1428 SLPTree slp_tree(graph_, this, phase_zone_);
1429 PackNode* root = slp_tree.BuildTree(roots);
1430 if (!root) {
1431 TRACE("Build tree failed!\n");
1432 continue;
1433 }
1434
1435 slp_tree.Print("After build tree");
1436 MergeSLPTree(slp_tree);
1437 }
1438
1439 // Early exist when no revectorizable node found.
1440 if (revectorizable_node_.empty()) return;
1441
1442 // Build SIMD usemap
1443 use_map_ = phase_zone_->New<SimdUseMap>(graph_, phase_zone_);
1444 if (!DecideVectorize()) {
1445 revectorizable_node_.clear();
1446 revectorizable_intersect_node_.clear();
1447 } else {
1448 should_reduce_ = true;
1449 Print("Decide to vectorize");
1450 }
1451}

1453bool WasmRevecAnalyzer::DecideVectorize() {
1454 TRACE("Enter %s\n", __func__);
1455 int save = 0, cost = 0;
1456 ForEach(
1457 [&](PackNode const* pnode) {
1458 const NodeGroup& nodes = pnode->nodes();
1459 // An additional store is emitted in case of OOB trap at the higher
1460 // 128-bit address. Thus no save if the store at lower address is
1461 // executed first. Return directly as we dont need to check external use
1462 // for stores.
1463 if (graph_.Get(nodes[0]).opcode == Opcode::kStore) {
1464 if (nodes[0] > nodes[1]) save++;
1465 return;
1466 }
1467
1468 if (pnode->IsForcePackNode()) {
1469 cost++;
1470 return;
1471 }
1472
1473 // Splat nodes will not cause a saving as it simply extends itself.
1474 if (!IsSplat(nodes)) {
1475 save++;
1476 }
1477
1478#ifdef V8_TARGET_ARCH_X64
1479 // On x64 platform, we dont emit extract for lane 0 as the source ymm
1480 // register is alias to the corresponding xmm register in lower 128-bit.
1481 for (int i = 1; i < static_cast<int>(nodes.size()); i++) {
1482 if (nodes[i] == nodes[0]) continue;
1483#else
1484 for (int i = 0; i < static_cast<int>(nodes.size()); i++) {
1485 if (i > 0 && nodes[i] == nodes[0]) continue;
1486#endif // V8_TARGET_ARCH_X64
1487
1488 for (auto use : use_map_->uses(nodes[i])) {
1489 if (!GetPackNode(use) || GetPackNode(use)->is_force_packing()) {
1490 TRACE("External use edge: (%d:%s) -> (%d:%s)\n", use.id(),
1491 OpcodeName(graph_.Get(use).opcode), nodes[i].id(),
1492 OpcodeName(graph_.Get(nodes[i]).opcode));
1493 ++cost;
1494
1495 // We only need one Extract node and all other uses can share.
1496 break;
1497 }
1498 }
1499 }
1500 },
1501 revectorizable_node_);
1502
1503 ForEach(
1504 [&](PackNode const* pnode) {
1505 // We always generate SimdPack128To256Op for IntersectPackNode.
1506 cost++;
1507 return;
1508 },
1509 revectorizable_intersect_node_);
1510
1511 TRACE("Save: %d, cost: %d\n", save, cost);
1512 return save > cost;
1513}
