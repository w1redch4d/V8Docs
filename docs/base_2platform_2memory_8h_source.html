79// Aligned allocation. Memory must be freed with `AlignedFree()` as not all
80// platforms support using general free for aligned allocations.
81inline void* AlignedAlloc(size_t size, size_t alignment) {
82 DCHECK_LE(alignof(void*), alignment);
83 DCHECK(base::bits::IsPowerOfTwo(alignment));
84#if V8_OS_WIN
85 return _aligned_malloc(size, alignment);
86#elif V8_LIBC_BIONIC
87 // posix_memalign is not exposed in some Android versions, so we fall back to
88 // memalign. See http://code.google.com/p/android/issues/detail?id=35391.
89 return memalign(alignment, size);
90#elif V8_OS_ZOS
91 return __aligned_malloc(size, alignment);
92#else // POSIX
93 void* ptr;
94 if (posix_memalign(&ptr, alignment, size)) ptr = nullptr;
95 return ptr;
96#endif // POSIX
97}

99inline void AlignedFree(void* ptr) {
100#if V8_OS_WIN
101 _aligned_free(ptr);
102#elif V8_OS_ZOS
103 __aligned_free(ptr);
104#else
105 // Using regular Free() is not correct in general. For most platforms,
106 // including V8_LIBC_BIONIC, it is though.
107 base::Free(ptr);
108#endif
109}

113// Note that the use of additional bytes that deviate from the original
114// `Malloc()` request returned by `MallocUsableSize()` is not UBSan-safe. Use
115// `AllocateAtLeast()` for a safe version.
116inline size_t MallocUsableSize(void* ptr) {
117#if V8_OS_WIN
118 // |_msize| cannot handle a null pointer.
119 if (!ptr) return 0;
120 return _msize(ptr);
121#elif V8_OS_DARWIN
122 return malloc_size(ptr);
123#else // POSIX.
124 return malloc_usable_size(ptr);
125#endif // POSIX.
126}

130// Mimics C++23 `allocation_result`.
131template <class Pointer>
132struct AllocationResult {
133 Pointer ptr = nullptr;
134 size_t count = 0;
135};
136
137// Allocates at least `n * sizeof(T)` uninitialized storage but may allocate
138// more which is indicated by the return value. Mimics C++23
139// `allocate_at_least()`.
140template <typename T>
141V8_NODISCARD AllocationResult<T*> AllocateAtLeast(size_t n) {
142 const size_t min_wanted_size = n * sizeof(T);
143 auto* memory = static_cast<T*>(Malloc(min_wanted_size));
144#if !V8_HAS_MALLOC_USABLE_SIZE
145 return {memory, min_wanted_size};
146#else // V8_HAS_MALLOC_USABLE_SIZE
147 const size_t usable_size = MallocUsableSize(memory);
148#if V8_USE_UNDEFINED_BEHAVIOR_SANITIZER
149 if (memory == nullptr)
150 return {nullptr, 0};
151 // UBSan (specifically, -fsanitize=bounds) assumes that any access outside
152 // of the requested size for malloc is UB and will trap in ud2 instructions.
153 // This can be worked around by using `Realloc()` on the specific memory
154 // region.
155 if (usable_size != min_wanted_size) {
156 memory = static_cast<T*>(Realloc(memory, usable_size));
157 }
158#endif // V8_USE_UNDEFINED_BEHAVIOR_SANITIZER
159 return {memory, usable_size};
160#endif // V8_HAS_MALLOC_USABLE_SIZE
161}
