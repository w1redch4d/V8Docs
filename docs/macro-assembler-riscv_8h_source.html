30// Reserved Register Usage Summary.
31//
32// Registers t5, t6, and t3 are reserved for use by the MacroAssembler.
33//
34// The programmer should know that the MacroAssembler may clobber these three,
35// but won't touch other registers except in special cases.
36//
37// TODO(RISCV): Cannot find info about this ABI. We chose t6 for now.
38// Per the RISC-V ABI, register t6 must be used for indirect function call
39// via 'jalr t6' or 'jr t6' instructions. This is relied upon by gcc when
40// trying to update gp register for position-independent-code. Whenever
41// RISC-V generated code calls C code, it must be via t6 register.

44// Flags used for the li macro-assembler function.
45enum LiFlags {
46 // If the constant value can be represented in just 16 bits, then
47 // optimize the li to use a single instruction, rather than lui/ori/slli
48 // sequence. A number of other optimizations that emits less than
49 // maximum number of instructions exists.
50 OPTIMIZE_SIZE = 0,
51 // Always use 8 instructions (lui/addi/slliw sequence), even if the
52 // constant
53 // could be loaded with just one, so that this value is patchable later.
54 CONSTANT_SIZE = 1,
55 // For address loads 8 instruction are required. Used to mark
56 // constant load that will be used as address without relocation
57 // information. It ensures predictable code size, so specific sites
58 // in code are patchable.
59 ADDRESS_LOAD = 2
60};

257 inline void GenPCRelativeJump(Register rd, int32_t imm32) {
258 BlockTrampolinePoolScope block_trampoline_pool(this);
259 DCHECK(is_int32(imm32 + 0x800));
260 int32_t Hi20 = ((imm32 + 0x800) >> 12);
261 int32_t Lo12 = imm32 << 20 >> 20;
262 auipc(rd, Hi20); // Read PC + Hi20 into scratch.
263 jr(rd, Lo12); // jump PC + Hi20 + Lo12
264 }

266 inline void GenPCRelativeJumpAndLink(Register rd, int32_t imm32) {
267 BlockTrampolinePoolScope block_trampoline_pool(this);
268 DCHECK(is_int32(imm32 + 0x800));
269 int32_t Hi20 = ((imm32 + 0x800) >> 12);
270 int32_t Lo12 = imm32 << 20 >> 20;
271 auipc(rd, Hi20); // Read PC + Hi20 into scratch.
272 jalr(rd, Lo12); // jump PC + Hi20 + Lo12
273 }

456 // These PushAll/PopAll respect the order of the registers in the stack from
457 // low index to high.
458 void PushAll(RegList registers) {
459 if (registers.is_empty()) return;
460 ASM_CODE_COMMENT(this);
461 // TODO(victorgomes): pushes/pops registers in the opposite order
462 // as expected by Maglev frame. Consider massaging Maglev to accept this
463 // order instead.
464 int16_t num_to_push = registers.Count();
465 int16_t stack_offset = num_to_push * kSystemPointerSize;
466
467 SubWord(sp, sp, Operand(stack_offset));
468 for (int16_t i = 0; i < kNumRegisters; i++) {
469 if ((registers.bits() & (1 << i)) != 0) {
470 stack_offset -= kSystemPointerSize;
471 StoreWord(Register::from_code(i), MemOperand(sp, stack_offset));
472 }
473 }
474 }
475
476 void PopAll(RegList registers) {
477 if (registers.is_empty()) return;
478 int16_t stack_offset = 0;
479 for (int16_t i = kNumRegisters - 1; i >= 0; i--) {
480 if ((registers.bits() & (1 << i)) != 0) {
481 LoadWord(Register::from_code(i), MemOperand(sp, stack_offset));
482 stack_offset += kSystemPointerSize;
483 }
484 }
485 addi(sp, sp, stack_offset);
486 }

1900void MacroAssembler::GenerateSwitchTable(Register index, size_t case_count,
1901 Func GetLabelFunction) {
1902 // Ensure that dd-ed labels following this instruction use 8 bytes aligned
1903 // addresses.
1904 BlockTrampolinePoolFor(static_cast<int>(case_count) * 2 +
1905 kSwitchTablePrologueSize);
1906 UseScratchRegisterScope temps(this);
1907 Register scratch = temps.Acquire();
1908 Register scratch2 = temps.Acquire();
1909
1910 Align(8);
1911 // Load the address from the jump table at index and jump to it
1912 auipc(scratch, 0); // Load the current PC into scratch
1913 slli(scratch2, index,
1914 kSystemPointerSizeLog2); // scratch2 = offset of indexth entry
1915 add(scratch2, scratch2,
1916 scratch); // scratch2 = (saved PC) + (offset of indexth entry)
1917 LoadWord(scratch2,
1918 MemOperand(scratch2,
1919 6 * kInstrSize)); // Add the size of these 6 instructions
1920 // to the offset, then load
1921 jr(scratch2); // Jump to the address loaded from the table
1922 nop(); // For 16-byte alignment
1923 for (size_t index = 0; index < case_count; ++index) {
1924 dd(GetLabelFunction(index));
1925 }
1926}
