156// Avoids emitting calls to the {Builtin::kAbort} builtin when emitting
157// debug code during the lifetime of this scope object.
158class V8_NODISCARD HardAbortScope {
159 public:
160 explicit HardAbortScope(MacroAssemblerBase* assembler)
161 : assembler_(assembler), old_value_(assembler->should_abort_hard()) {
162 assembler_->set_abort_hard(true);
163 }
164 ~HardAbortScope() { assembler_->set_abort_hard(old_value_); }
165
166 private:
167 MacroAssemblerBase* assembler_;
168 bool old_value_;
169};
