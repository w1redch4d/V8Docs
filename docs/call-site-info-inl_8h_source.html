39Tagged<HeapObject> CallSiteInfo::code_object(IsolateForSandbox isolate) const {
40 DCHECK(!IsTrustedPointerFieldEmpty(kCodeObjectOffset));
41 // The field can contain either a Code or a BytecodeArray, so we need to use
42 // the kUnknownIndirectPointerTag. Since we can then no longer rely on the
43 // type-checking mechanism of trusted pointers we need to perform manual type
44 // checks afterwards.
45 Tagged<HeapObject> code_object =
46 ReadTrustedPointerField<kUnknownIndirectPointerTag>(kCodeObjectOffset,
47 isolate);
48 CHECK(IsCode(code_object) || IsBytecodeArray(code_object));
49 return code_object;
50}

52void CallSiteInfo::set_code_object(Tagged<HeapObject> code,
53 WriteBarrierMode mode) {
54 DCHECK(IsCode(code) || IsBytecodeArray(code) || IsUndefined(code));
55 if (IsCode(code) || IsBytecodeArray(code)) {
56 WriteTrustedPointerField<kUnknownIndirectPointerTag>(
57 kCodeObjectOffset, Cast<ExposedTrustedObject>(code));
58 CONDITIONAL_TRUSTED_POINTER_WRITE_BARRIER(
59 *this, kCodeObjectOffset, kUnknownIndirectPointerTag, code, mode);
60 } else {
61 DCHECK(IsUndefined(code));
62 ClearTrustedPointerField(kCodeObjectOffset);
63 }
64}
