56 DCHECK(IsValidSweepingSpace(identity));

108 private:
109 // Major sweeping jobs don't sweep new space.

116 MajorSweeperJob(Isolate* isolate, Sweeper* sweeper)

119 sweeper_->major_sweeping_state_.concurrent_sweepers()),

122 DCHECK_LE(concurrent_sweepers.size(), kMaxTasks);

128 MajorSweeperJob& operator=(const MajorSweeperJob&) = delete;

132 }

135 static constexpr int kPagePerTask = 2;
136 return std::min<size_t>(

138 worker_count +

141 }
142
143 private:
144 void RunImpl(JobDelegate* delegate, bool is_joining_thread) {
145 // In case multi-cage pointer compression mode is enabled ensure that
146 // current thread's cage base values are properly initialized.
147 PtrComprCageAccessScope ptr_compr_cage_access_scope(
148 sweeper_->heap_->isolate());
149

152 DCHECK_LT(offset, concurrent_sweepers.size());
153 ConcurrentMajorSweeper& concurrent_sweeper = concurrent_sweepers[offset];
154 TRACE_GC_EPOCH_WITH_FLOW(
155 tracer_, sweeper_->GetTracingScope(OLD_SPACE, is_joining_thread),
156 is_joining_thread ? ThreadKind::kMain : ThreadKind::kBackground,
157 trace_id_, TRACE_EVENT_FLAG_FLOW_IN);
158 for (int i = 0; i < kNumberOfMajorSweepingSpaces; i++) {
159 const AllocationSpace space_id = static_cast<AllocationSpace>(
160 FIRST_SWEEPABLE_SPACE + 1 +
161 ((offset + i) % kNumberOfMajorSweepingSpaces));
162 DCHECK_LE(FIRST_SWEEPABLE_SPACE, space_id);
163 DCHECK_LE(space_id, LAST_SWEEPABLE_SPACE);
164 DCHECK_NE(NEW_SPACE, space_id);
165 if (!concurrent_sweeper.ConcurrentSweepSpace(space_id, delegate)) return;
166 }
167 }
168
169 Sweeper* const sweeper_;
170 std::vector<ConcurrentMajorSweeper>& concurrent_sweepers;
171 GCTracer* const tracer_;
172 const uint64_t trace_id_;
173};

286template <Sweeper::SweepingScope scope>
287void Sweeper::SweepingState<scope>::StartConcurrentSweeping() {
288 DCHECK(!HasValidJob());
289 DCHECK(in_progress_);
290 if (v8_flags.concurrent_sweeping &&
291 !sweeper_->heap_->delay_sweeper_tasks_for_testing_) {
292 auto job =
293 std::make_unique<SweeperJob>(sweeper_->heap_->isolate(), sweeper_);
294 GCTracer::Scope::ScopeId scope_id =
295 scope == SweepingScope::kMinor
296 ? GCTracer::Scope::MINOR_MS_SWEEP_START_JOBS
297 : GCTracer::Scope::MC_SWEEP_START_JOBS;
298 TRACE_GC_WITH_FLOW(sweeper_->heap_->tracer(), scope_id,
299 background_trace_id(), TRACE_EVENT_FLAG_FLOW_OUT);
300 DCHECK_IMPLIES(v8_flags.minor_ms, concurrent_sweepers_.empty());
301 int max_concurrent_sweeper_count =
302 std::min(SweeperJob::kMaxTasks,
303 V8::GetCurrentPlatform()->NumberOfWorkerThreads() + 1);
304 if (concurrent_sweepers_.empty()) {
305 for (int i = 0; i < max_concurrent_sweeper_count; ++i) {
306 concurrent_sweepers_.emplace_back(sweeper_);
307 }
308 }
309 DCHECK_EQ(max_concurrent_sweeper_count, concurrent_sweepers_.size());
310 job_handle_ = V8::GetCurrentPlatform()->PostJob(TaskPriority::kUserVisible,
311 std::move(job));
312 }

348bool Sweeper::LocalSweeper::ParallelSweepSpace(AllocationSpace identity,
349 SweepingMode sweeping_mode,
350 uint32_t max_pages) {
351 uint32_t pages_swept = 0;
352 bool found_usable_pages = false;
353 PageMetadata* page = nullptr;
354 while ((page = sweeper_->GetSweepingPageSafe(identity)) != nullptr) {
355 ParallelSweepPage(page, identity, sweeping_mode);
356 if (!page->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE)) {
357 found_usable_pages = true;
358#if DEBUG
359 } else {
360 // All remaining pages are also marked with NEVER_ALLOCATE_ON_PAGE.
361 base::MutexGuard guard(&sweeper_->mutex_);
362 int space_index = GetSweepSpaceIndex(identity);
363 Sweeper::SweepingList& sweeping_list =
364 sweeper_->sweeping_list_[space_index];
365 DCHECK(std::all_of(sweeping_list.begin(), sweeping_list.end(),
366 [](const PageMetadata* p) {
367 return p->Chunk()->IsFlagSet(
368 MemoryChunk::NEVER_ALLOCATE_ON_PAGE);
369 }));
370#endif // DEBUG
371 }
372 if (++pages_swept >= max_pages) break;
373 }
374 return found_usable_pages;
375}

377void Sweeper::LocalSweeper::ParallelSweepPage(PageMetadata* page,
378 AllocationSpace identity,
379 SweepingMode sweeping_mode) {
380 DCHECK(IsValidSweepingSpace(identity));
381
382 // The Scavenger may add already swept pages back.
383 if (page->SweepingDone()) return;
384
385 {
386 base::SpinningMutexGuard guard(page->mutex());
387 DCHECK(!page->SweepingDone());
388 DCHECK_EQ(PageMetadata::ConcurrentSweepingState::kPendingSweeping,
389 page->concurrent_sweeping_state());
390 page->set_concurrent_sweeping_state(
391 PageMetadata::ConcurrentSweepingState::kInProgress);
392 const FreeSpaceTreatmentMode free_space_treatment_mode =
393 heap::ShouldZapGarbage() ? FreeSpaceTreatmentMode::kZapFreeSpace
394 : FreeSpaceTreatmentMode::kIgnoreFreeSpace;
395 DCHECK_IMPLIES(identity == NEW_SPACE,
396 !sweeper_->minor_sweeping_state_.should_reduce_memory());
397 sweeper_->RawSweep(
398 page, free_space_treatment_mode, sweeping_mode,
399 identity == NEW_SPACE
400 ? false
401 : sweeper_->major_sweeping_state_.should_reduce_memory());
402 sweeper_->AddSweptPage(page, identity);
403 DCHECK(page->SweepingDone());
404 }

577// Atomically zap the specified area.
578V8_INLINE void AtomicZapBlock(Address addr, size_t size_in_bytes) {
579 static_assert(sizeof(Tagged_t) == kTaggedSize);
580 static constexpr Tagged_t kZapTagged = static_cast<Tagged_t>(kZapValue);
581 DCHECK(IsAligned(addr, kTaggedSize));

583 const size_t size_in_tagged = size_in_bytes / kTaggedSize;
584 Tagged_t* current_addr = reinterpret_cast<Tagged_t*>(addr);
585 for (size_t i = 0; i < size_in_tagged; ++i) {
586 base::AsAtomicPtr(current_addr++)
587 ->store(kZapTagged, std::memory_order_relaxed);
588 }

670V8_INLINE bool ComparePagesForSweepingOrder(const PageMetadata* a,
671 const PageMetadata* b) {
672 // Prioritize pages that can be allocated on.
673 if (a->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE) !=
674 b->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE))
675 return a->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE);
676 // We sort in descending order of live bytes, i.e., ascending order of
677 // free bytes, because GetSweepingPageSafe returns pages in reverse order.
678 // This works automatically for black allocated pages, since we set live bytes
679 // for them to the area size.
680 return a->live_bytes() > b->live_bytes();
681}

684void Sweeper::StartMajorSweeping() {
685 DCHECK_EQ(GarbageCollector::MARK_COMPACTOR,
686 heap_->tracer()->GetCurrentCollector());
687 DCHECK(!minor_sweeping_in_progress());
688 major_sweeping_state_.StartSweeping();
689 ForAllSweepingSpaces([this](AllocationSpace space) {
690 // Sorting is done in order to make compaction more efficient: by sweeping
691 // pages with the most free bytes first, we make it more likely that when
692 // evacuating a page, already swept pages will have enough free bytes to
693 // hold the objects to move (and therefore, we won't need to wait for more
694 // pages to be swept in order to move those objects).
695 int space_index = GetSweepSpaceIndex(space);
696 DCHECK_IMPLIES(space == NEW_SPACE, sweeping_list_[space_index].empty());
697 std::sort(sweeping_list_[space_index].begin(),
698 sweeping_list_[space_index].end(), ComparePagesForSweepingOrder);
699 });
700}

707 std::sort(sweeping_list_[new_space_index].begin(),

908V8_INLINE size_t Sweeper::FreeAndProcessFreedMemory(
909 Address free_start, Address free_end, PageMetadata* page, Space* space,
910 FreeSpaceTreatmentMode free_space_treatment_mode,
911 bool should_reduce_memory) {
912 CHECK_GT(free_end, free_start);
913 size_t freed_bytes = 0;
914 size_t size = static_cast<size_t>(free_end - free_start);
915 if (free_space_treatment_mode == FreeSpaceTreatmentMode::kZapFreeSpace) {
916 CodePageMemoryModificationScopeForDebugging memory_modification_scope(page);
917 AtomicZapBlock(free_start, size);
918 }
919 freed_bytes = reinterpret_cast<PagedSpaceBase*>(space)->FreeDuringSweep(
920 free_start, size);
921 if (should_reduce_memory) {
922 ZeroOrDiscardUnusedMemory(page, free_start, size);
923 }
924
925 if (v8_flags.sticky_mark_bits) {
926 // Clear the bitmap, since fillers or slack may still be marked from black
927 // allocation.
928 page->marking_bitmap()->ClearRange<AccessMode::NON_ATOMIC>(
929 MarkingBitmap::AddressToIndex(free_start),
930 MarkingBitmap::AddressToIndex(free_end));
931 }
932
933 return freed_bytes;
934}

941 const Address discard_end = RoundDown(end, page_size);

950void Sweeper::ZeroOrDiscardUnusedMemory(PageMetadata* page, Address addr,
951 size_t size) {
952 if (size < FreeSpace::kSize) {
953 return;
954 }
955
956 const Address unused_start = addr + FreeSpace::kSize;
957 DCHECK(page->ContainsLimit(unused_start));
958 const Address unused_end = addr + size;
959 DCHECK(page->ContainsLimit(unused_end));
960
961 std::optional<RwxMemoryWriteScope> scope;
962 if (page->Chunk()->executable()) {
963 scope.emplace("For zeroing unused memory.");
964 }
965 const std::optional<base::AddressRegion> discard_area =
966 ComputeDiscardMemoryArea(unused_start, unused_end);
967
968#if !defined(V8_OS_WIN)
969 constexpr bool kDiscardEmptyPages = true;
970#else
971 // Discarding memory on Windows does not decommit the memory and does not
972 // contribute to reduce the memory footprint. On the other hand, these
973 // calls become expensive the more memory is allocated in the system and
974 // can result in hangs. Thus, it is better to not discard on Windows.
975 constexpr bool kDiscardEmptyPages = false;
976#endif // !defined(V8_OS_WIN)
977
978 if (kDiscardEmptyPages && discard_area) {
979 {
980 v8::PageAllocator* page_allocator =
981 heap_->memory_allocator()->page_allocator(page->owner_identity());
982 DiscardSealedMemoryScope discard_scope("Discard unused memory");
983 CHECK(page_allocator->DiscardSystemPages(
984 reinterpret_cast<void*>(discard_area->begin()),
985 discard_area->size()));
986 }
987
988 if (v8_flags.zero_unused_memory) {
989 // Now zero unused memory right before and after the discarded OS pages to
990 // help with OS page compression.
991 memset(reinterpret_cast<void*>(unused_start), 0,
992 discard_area->begin() - unused_start);
993 memset(reinterpret_cast<void*>(discard_area->end()), 0,
994 unused_end - discard_area->end());
995 }
996 } else if (v8_flags.zero_unused_memory) {
997 // Unused memory does not span a full OS page. Simply clear all of the
998 // unused memory. This helps with OS page compression.
999 memset(reinterpret_cast<void*>(unused_start), 0, unused_end - unused_start);
1000 }
1001}

1003V8_INLINE void Sweeper::CleanupRememberedSetEntriesForFreedMemory(
1004 Address free_start, Address free_end, PageMetadata* page,
1005 bool record_free_ranges, TypedSlotSet::FreeRangesMap* free_ranges_map,
1006 SweepingMode sweeping_mode) {
1007 DCHECK_LE(free_start, free_end);
1008 if (sweeping_mode == SweepingMode::kEagerDuringGC) {
1009 // New space and in consequence the old-to-new remembered set is always
1010 // empty after a full GC, so we do not need to remove from it after the full
1011 // GC. However, we wouldn't even be allowed to do that, since the main
1012 // thread then owns the old-to-new remembered set. Removing from it from a
1013 // sweeper thread would race with the main thread.
1014 RememberedSet<OLD_TO_NEW>::RemoveRange(page, free_start, free_end,
1015 SlotSet::KEEP_EMPTY_BUCKETS);
1016 RememberedSet<OLD_TO_NEW_BACKGROUND>::RemoveRange(
1017 page, free_start, free_end, SlotSet::KEEP_EMPTY_BUCKETS);
1018
1019 // While we only add old-to-old slots on live objects, we can still end up
1020 // with old-to-old slots in free memory with e.g. right-trimming of objects.
1021 RememberedSet<OLD_TO_OLD>::RemoveRange(page, free_start, free_end,
1022 SlotSet::KEEP_EMPTY_BUCKETS);
1023 RememberedSet<TRUSTED_TO_TRUSTED>::RemoveRange(page, free_start, free_end,
1024 SlotSet::KEEP_EMPTY_BUCKETS);
1025 } else {
1026 DCHECK_NULL(page->slot_set<OLD_TO_OLD>());
1027 DCHECK_NULL(page->slot_set<TRUSTED_TO_TRUSTED>());
1028 }
1029
1030 // Old-to-shared isn't reset after a full GC, so needs to be cleaned both
1031 // during and after a full GC.
1032 RememberedSet<OLD_TO_SHARED>::RemoveRange(page, free_start, free_end,
1033 SlotSet::KEEP_EMPTY_BUCKETS);
1034 RememberedSet<TRUSTED_TO_SHARED_TRUSTED>::RemoveRange(
1035 page, free_start, free_end, SlotSet::KEEP_EMPTY_BUCKETS);
1036
1037 if (record_free_ranges) {
1038 MemoryChunk* chunk = page->Chunk();
1039 free_ranges_map->insert(std::pair<uint32_t, uint32_t>(
1040 static_cast<uint32_t>(chunk->Offset(free_start)),
1041 static_cast<uint32_t>(chunk->Offset(free_end))));
1042 }
1043}

1045void Sweeper::CleanupTypedSlotsInFreeMemory(
1046 PageMetadata* page, const TypedSlotSet::FreeRangesMap& free_ranges_map,
1047 SweepingMode sweeping_mode) {
1048 // No support for typed trusted-to-shared-trusted pointers.
1049 DCHECK_NULL(page->typed_slot_set<TRUSTED_TO_SHARED_TRUSTED>());
1050
1051 if (sweeping_mode == SweepingMode::kEagerDuringGC) {
1052 page->ClearTypedSlotsInFreeMemory<OLD_TO_NEW>(free_ranges_map);
1053
1054 // Typed old-to-old slot sets are only ever recorded in live code objects.
1055 // Also code objects are never right-trimmed, so there cannot be any slots
1056 // in a free range.
1057 page->AssertNoTypedSlotsInFreeMemory<OLD_TO_OLD>(free_ranges_map);
1058 page->ClearTypedSlotsInFreeMemory<OLD_TO_SHARED>(free_ranges_map);
1059 return;
1060 }
1061
1062 DCHECK_EQ(sweeping_mode, SweepingMode::kLazyOrConcurrent);
1063
1064 // After a full GC there are no old-to-new typed slots. The main thread
1065 // could create new slots but not in a free range.
1066 page->AssertNoTypedSlotsInFreeMemory<OLD_TO_NEW>(free_ranges_map);
1067 DCHECK_NULL(page->typed_slot_set<OLD_TO_OLD>());
1068 page->ClearTypedSlotsInFreeMemory<OLD_TO_SHARED>(free_ranges_map);
1069}

1082void Sweeper::RawSweep(PageMetadata* p,
1083 FreeSpaceTreatmentMode free_space_treatment_mode,
1084 SweepingMode sweeping_mode, bool should_reduce_memory) {
1085 DCHECK_NOT_NULL(p);
1086 Space* space = p->owner();
1087 DCHECK_NOT_NULL(space);
1088 DCHECK(space->identity() == OLD_SPACE || space->identity() == CODE_SPACE ||
1089 space->identity() == SHARED_SPACE ||
1090 space->identity() == TRUSTED_SPACE ||
1091 space->identity() == SHARED_TRUSTED_SPACE ||
1092 (space->identity() == NEW_SPACE && v8_flags.minor_ms));
1093 DCHECK(!p->Chunk()->IsEvacuationCandidate());
1094 DCHECK(!p->SweepingDone());
1095 DCHECK(!p->Chunk()->IsFlagSet(MemoryChunk::BLACK_ALLOCATED));
1096 DCHECK_IMPLIES(space->identity() == NEW_SPACE,
1097 !heap_->incremental_marking()->IsMinorMarking());
1098 DCHECK_IMPLIES(space->identity() != NEW_SPACE,
1099 !heap_->incremental_marking()->IsMajorMarking());
1100
1101 // Phase 1: Prepare the page for sweeping.
1102
1103 std::optional<ActiveSystemPages> active_system_pages_after_sweeping;
1104 if (should_reduce_memory) {
1105 // Only decrement counter when we discard unused system pages.
1106 active_system_pages_after_sweeping = ActiveSystemPages();
1107 active_system_pages_after_sweeping->Init(
1108 sizeof(MemoryChunk), MemoryAllocator::GetCommitPageSizeBits(),
1109 PageMetadata::kPageSize);
1110 }
1111
1112 // Phase 2: Free the non-live memory and clean-up the regular remembered set
1113 // entires.
1114
1115 // Liveness and freeing statistics.
1116 size_t live_bytes = 0;
1117
1118 // Promoted pages have no interesting remebered sets yet.
1119 bool record_free_ranges = (p->typed_slot_set<OLD_TO_NEW>() != nullptr ||
1120 p->typed_slot_set<OLD_TO_OLD>() != nullptr ||
1121 p->typed_slot_set<OLD_TO_SHARED>() != nullptr) ||
1122 DEBUG_BOOL;
1123
1124 // The free ranges map is used for filtering typed slots.
1125 TypedSlotSet::FreeRangesMap free_ranges_map;
1126
1127 // Iterate over the page using the live objects and free the memory before
1128 // the given live object.
1129 Address free_start = p->area_start();
1130
1131 for (auto [object, size] : LiveObjectRange(p)) {
1132 DCHECK(marking_state_->IsMarked(object));
1133 Address free_end = object.address();
1134 if (free_end != free_start) {
1135 FreeAndProcessFreedMemory(free_start, free_end, p, space,
1136 free_space_treatment_mode,
1137 should_reduce_memory);
1138 CleanupRememberedSetEntriesForFreedMemory(
1139 free_start, free_end, p, record_free_ranges, &free_ranges_map,
1140 sweeping_mode);
1141 }
1142 live_bytes += size;
1143 free_start = free_end + size;
1144
1145 if (active_system_pages_after_sweeping) {
1146 MemoryChunk* chunk = p->Chunk();
1147 active_system_pages_after_sweeping->Add(
1148 chunk->Offset(free_end), chunk->Offset(free_start),
1149 MemoryAllocator::GetCommitPageSizeBits());
1150 }
1151 }
1152
1153 // If there is free memory after the last live object also free that.
1154 Address free_end = p->area_end();
1155 if (free_end != free_start) {
1156 FreeAndProcessFreedMemory(free_start, free_end, p, space,
1157 free_space_treatment_mode, should_reduce_memory);
1158 CleanupRememberedSetEntriesForFreedMemory(free_start, free_end, p,
1159 record_free_ranges,
1160 &free_ranges_map, sweeping_mode);
1161 }
1162
1163 // Phase 3: Post process the page.
1164 p->ReleaseSlotSet(SURVIVOR_TO_EXTERNAL_POINTER);
1165 CleanupTypedSlotsInFreeMemory(p, free_ranges_map, sweeping_mode);
1166 ClearMarkBitsAndHandleLivenessStatistics(p, live_bytes);
1167
1168 if (active_system_pages_after_sweeping) {
1169 // Decrement accounted memory for discarded memory.
1170 PagedSpaceBase* paged_space = static_cast<PagedSpaceBase*>(p->owner());
1171 paged_space->ReduceActiveSystemPages(p,
1172 *active_system_pages_after_sweeping);
1173 }
1174}

1228void Sweeper::EnsurePageIsSwept(PageMetadata* page) {
1229 DCHECK(heap_->IsMainThread());
1230
1231 auto concurrent_sweeping_state = page->concurrent_sweeping_state();
1232 DCHECK_IMPLIES(!sweeping_in_progress(),
1233 concurrent_sweeping_state ==
1234 PageMetadata::ConcurrentSweepingState::kDone);

1237 DCHECK(page->SweepingDone());

1250 PageMetadata::ConcurrentSweepingState::kPendingSweeping) &&
1251 TryRemoveSweepingPageSafe(space, page)) {

1253 main_thread_local_sweeper_.ParallelSweepPage(

1475 DCHECK_EQ(kNewSpace, page->owner_identity());
