38// The {BackingStore} data structure stores all the low-level details about the
39// backing store of an array buffer or Wasm memory, including its base address
40// and length, whether it is shared, provided by the embedder, has guard
41// regions, etc. Instances of this classes *own* the underlying memory
42// when they are created through one of the {Allocate()} methods below,
43// and the destructor frees the memory (and page allocation if necessary).
44class V8_EXPORT_PRIVATE BackingStore : public BackingStoreBase {
45 public:
46 ~BackingStore();

81 // Accessors.
82 // Internally, we treat nullptr as the empty buffer value. However,
83 // externally, we should use the EmptyBackingStoreBuffer() constant for that
84 // purpose as the buffer pointer should always point into the sandbox. As

149 // Returns the size of the external memory owned by this backing store.
150 // It is used for triggering GCs based on the external memory pressure.
151 size_t PerIsolateAccountingLength() {
152 if (is_shared_) {
153 // TODO(titzer): SharedArrayBuffers and shared WasmMemorys cause problems
154 // with accounting for per-isolate external memory. In particular, sharing
155 // the same array buffer or memory multiple times, which happens in stress
156 // tests, can cause overcounting, leading to GC thrashing. Fix with global
157 // accounting?
158 return 0;
159 }
160 if (empty_deleter_) {
161 // The backing store has an empty deleter. Even if the backing store is
162 // freed after GC, it would not free the memory block.
163 return 0;
164 }
165 return byte_length();
166 }
