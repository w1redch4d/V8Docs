45std::optional<Builtin> TryGetBuiltinId(const ConstantOp* target,
46 JSHeapBroker* broker) {
47 if (!target) return std::nullopt;
48 if (target->kind != ConstantOp::Kind::kHeapObject) return std::nullopt;
49 // TODO(nicohartmann@): For builtin compilation we don't have a broker. We
50 // could try to access the heap directly instead.
51 if (broker == nullptr) return std::nullopt;
52 UnparkedScopeIfNeeded scope(broker);
53 AllowHandleDereference allow_handle_dereference;
54 HeapObjectRef ref = MakeRef(broker, target->handle());
55 if (ref.IsCode()) {
56 CodeRef code = ref.AsCode();
57 if (code.object()->is_builtin()) {
58 return code.object()->builtin_id();
59 }
60 }
61 return std::nullopt;
62}

64bool CallOp::IsStackCheck(const Graph& graph, JSHeapBroker* broker,
65 StackCheckKind kind) const {
66 auto builtin_id =
67 TryGetBuiltinId(graph.Get(callee()).TryCast<ConstantOp>(), broker);
68 if (!builtin_id.has_value()) return false;
69 if (*builtin_id != Builtin::kCEntry_Return1_ArgvOnStack_NoBuiltinExit) {
70 return false;
71 }
72 DCHECK_GE(input_count, 4);
73 Runtime::FunctionId builtin = GetBuiltinForStackCheckKind(kind);
74 auto is_this_builtin = [&](int input_index) {
75 if (const ConstantOp* real_callee =
76 graph.Get(input(input_index)).TryCast<ConstantOp>();
77 real_callee != nullptr &&
78 real_callee->kind == ConstantOp::Kind::kExternal &&
79 real_callee->external_reference() ==
80 ExternalReference::Create(builtin)) {
81 return true;
82 }
83 return false;
84 };
85 // The function called by `CEntry_Return1_ArgvOnStack_NoBuiltinExit` is the
86 // 3rd or the 4th argument of the CallOp (depending on the stack check kind),
87 // so we check both of them.
88 return is_this_builtin(2) || is_this_builtin(3);
89}

100bool ValidOpInputRep(
101 const Graph& graph, OpIndex input,
102 std::initializer_list<RegisterRepresentation> expected_reps,
103 std::optional<size_t> projection_index) {
104 base::Vector<const RegisterRepresentation> input_reps =
105 graph.Get(input).outputs_rep();
106 RegisterRepresentation input_rep;
107 if (projection_index) {
108 if (*projection_index < input_reps.size()) {
109 input_rep = input_reps[*projection_index];
110 } else {
111 std::cerr << "Turboshaft operation has input #" << input
112 << " with wrong arity.\n";
113 std::cerr << "Input has results " << PrintCollection(input_reps)
114 << ", but expected at least " << (*projection_index + 1)
115 << " results.\n";
116 return false;
117 }
118 } else if (input_reps.size() == 1) {
119 input_rep = input_reps[0];
120 } else {
121 std::cerr << "Turboshaft operation has input #" << input
122 << " with wrong arity.\n";
123 std::cerr << "Expected a single output but found " << input_reps.size()
124 << ".\n";
125 return false;
126 }
127 for (RegisterRepresentation expected_rep : expected_reps) {
128 if (input_rep.AllowImplicitRepresentationChangeTo(
129 expected_rep, graph.IsCreatedFromTurbofan())) {
130 return true;
131 }
132 }
133 std::cerr << "Turboshaft operation has input #" << input
134 << " with wrong representation.\n";
135 std::cerr << "Expected " << (expected_reps.size() > 1 ? "one of " : "")
136 << PrintCollection(expected_reps).WithoutBrackets() << " but found "
137 << input_rep << ".\n";
138 std::cout << "Input: " << graph.Get(input) << "\n";
139 return false;
140}

795void FrameStateOp::Validate(const Graph& graph) const {
796 if (inlined) {
797 DCHECK(Get(graph, parent_frame_state()).Is<FrameStateOp>());
798 }
799 FrameStateData::Iterator it = data->iterator(state_values());
800 while (it.has_more()) {
801 switch (it.current_instr()) {
802 case FrameStateData::Instr::kInput: {
803 MachineType type;
804 OpIndex input;
805 it.ConsumeInput(&type, &input);
806 RegisterRepresentation rep =
807 RegisterRepresentation::FromMachineRepresentation(
808 type.representation());
809 if (rep == RegisterRepresentation::Tagged()) {
810 // The deoptimizer can handle compressed values.
811 rep = RegisterRepresentation::Compressed();
812 }
813 DCHECK(ValidOpInputRep(graph, input, rep));
814 break;
815 }
816 case FrameStateData::Instr::kUnusedRegister:
817 it.ConsumeUnusedRegister();
818 break;
819 case FrameStateData::Instr::kDematerializedObject: {
820 uint32_t id;
821 uint32_t field_count;
822 it.ConsumeDematerializedObject(&id, &field_count);
823 break;
824 }
825 case FrameStateData::Instr::kDematerializedObjectReference: {
826 uint32_t id;
827 it.ConsumeDematerializedObjectReference(&id);
828 break;
829 }
830 case FrameStateData::Instr::kDematerializedStringConcat: {
831 uint32_t id;
832 it.ConsumeDematerializedStringConcat(&id);
833 break;
834 }
835 case FrameStateData::Instr::kDematerializedStringConcatReference: {
836 uint32_t id;
837 it.ConsumeDematerializedStringConcatReference(&id);
838 break;
839 }
840 case FrameStateData::Instr::kArgumentsElements: {
841 CreateArgumentsType type;
842 it.ConsumeArgumentsElements(&type);
843 break;
844 }
845 case FrameStateData::Instr::kArgumentsLength: {
846 it.ConsumeArgumentsLength();
847 break;
848 }
849 case FrameStateData::Instr::kRestLength: {
850 it.ConsumeRestLength();
851 break;
852 }
853 }

1998 if (strategy == HashingStrategy::kMakeSnapshotStable) {
1999 // Destructure here to cause a compilation error in case `options` is
2000 // changed.
2001 auto [cases_value, default_case_value, default_hint_value] = options();
2002 DCHECK_NOT_NULL(default_case_value);
2003 size_t hash = HashWithOptions(index_for_bound_block(default_case_value),
2004 default_hint_value);

2006 hash = fast_hash_combine(hash, c.value,
2007 index_for_bound_block(c.destination), c.hint);
2008 }
2009 return hash;
2010 } else {

2035bool IsUnlikelySuccessor(const Block* block, const Block* successor,
2036 const Graph& graph) {
2037 DCHECK(base::contains(successor->Predecessors(), block));
2038 const Operation& terminator = block->LastOperation(graph);
2039 switch (terminator.opcode) {
2040 case Opcode::kCheckException: {
2041 const CheckExceptionOp& check_exception =
2042 terminator.Cast<CheckExceptionOp>();
2043 return successor == check_exception.catch_block;
2044 }
2045 case Opcode::kGoto:
2046 return false;
2047 case Opcode::kBranch: {
2048 const BranchOp& branch = terminator.Cast<BranchOp>();
2049 return (branch.hint == BranchHint::kTrue &&
2050 successor == branch.if_false) ||
2051 (branch.hint == BranchHint::kFalse && successor == branch.if_true);
2052 }
2053 case Opcode::kSwitch: {
2054 const SwitchOp& swtch = terminator.Cast<SwitchOp>();
2055 if (successor == swtch.default_case) {
2056 return swtch.default_hint == BranchHint::kFalse;
2057 }
2058 auto it = std::find_if(swtch.cases.begin(), swtch.cases.end(),
2059 [successor](const SwitchOp::Case& c) {
2060 return c.destination == successor;
2061 });
2062 DCHECK_NE(it, swtch.cases.end());
2063 return it->hint == BranchHint::kFalse;
2064 }
2065 case Opcode::kDeoptimize:
2066 case Opcode::kTailCall:
2067 case Opcode::kUnreachable:
2068 case Opcode::kReturn:
2069 UNREACHABLE();
2070
2071#define NON_TERMINATOR_CASE(op) case Opcode::k##op:
2072 TURBOSHAFT_OPERATION_LIST_NOT_BLOCK_TERMINATOR(NON_TERMINATOR_CASE)
2073 UNREACHABLE();
2074#undef NON_TERMINATOR_CASE
2075 }
2076}
