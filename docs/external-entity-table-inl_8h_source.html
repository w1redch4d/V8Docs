45template <typename Entry, size_t size>
46void ExternalEntityTable<Entry, size>::Initialize() {
47 Base::Initialize();
48
49 if (!ExternalEntityTable::kUseContiguousMemory) return;
50
51 // Allocate the read-only segment of the table. This segment is always
52 // located at offset 0, and contains the null entry (pointing at
53 // kNullAddress) at index 0. It may later be temporarily marked read-write,
54 // see UnsealedReadOnlySegmentScope.
55 Address first_segment = this->vas_->AllocatePages(
56 this->vas_->base(), kSegmentSize, kSegmentSize, PagePermissions::kRead);
57 if (first_segment != this->vas_->base()) {
58 V8::FatalProcessOutOfMemory(
59 nullptr,
60 "ExternalEntityTable::InitializeTable (first segment allocation)");
61 }
62 DCHECK_EQ(first_segment - this->vas_->base(), kInternalReadOnlySegmentOffset);
63}

95template <typename Entry, size_t size>
96void ExternalEntityTable<Entry, size>::AttachSpaceToReadOnlySegment(
97 Space* space) {
98 CHECK(ExternalEntityTable::kUseContiguousMemory);
99 DCHECK(this->is_initialized());
100 DCHECK(space->BelongsTo(this));
101
102 DCHECK(!space->is_internal_read_only_space());
103 space->is_internal_read_only_space_ = true;
104
105 UnsealReadOnlySegmentScope unseal_scope(this);
106
107 // Physically attach the segment.
108 FreelistHead freelist;
109 {
110 base::SpinningMutexGuard guard(&space->mutex_);
111 DCHECK_EQ(space->segments_.size(), 0);
112 Segment segment = Segment::At(kInternalReadOnlySegmentOffset);
113 DCHECK_EQ(segment.first_entry(), kInternalNullEntryIndex);
114
115 // For the internal read-only segment, index 0 is reserved for the `null`
116 // entry, so start the freelist at offset 1.
117 freelist = Base::InitializeFreeList(segment, 1);
118
119 Extend(space, segment, freelist);
120 }
121
122 DCHECK(!freelist.is_empty());
123 DCHECK_EQ(freelist.next(), kInternalNullEntryIndex + 1);
124 DCHECK(space->Contains(freelist.next()));
125}

155template <typename Entry, size_t size>
156uint32_t ExternalEntityTable<Entry, size>::AllocateEntry(Space* space) {
157 DCHECK(this->is_initialized());
158 DCHECK(space->BelongsTo(this));
159
160 // We currently don't want entry allocation to trigger garbage collection as
161 // this may cause seemingly harmless pointer field assignments to trigger
162 // garbage collection. This is especially true for lazily-initialized
163 // external pointer slots which will typically only allocate the external
164 // pointer table entry when the pointer is first set to a non-null value.
165 DisallowGarbageCollection no_gc;
166
167 FreelistHead freelist;
168 for (;;) {
169 // This is essentially DCLP (see
170 // https://preshing.com/20130930/double-checked-locking-is-fixed-in-cpp11/)
171 // and so requires an acquire load as well as a release store in Grow() to
172 // prevent reordering of memory accesses, which could for example cause one
173 // thread to read a freelist entry before it has been properly initialized.
174 freelist = space->freelist_head_.load(std::memory_order_acquire);
175 if (V8_UNLIKELY(freelist.is_empty())) {
176 // Freelist is empty. Need to take the lock, then attempt to allocate a
177 // new segment if no other thread has done it in the meantime.
178 base::SpinningMutexGuard guard(&space->mutex_);
179
180 // Reload freelist head in case another thread already grew the table.
181 freelist = space->freelist_head_.load(std::memory_order_relaxed);
182
183 if (freelist.is_empty()) {
184 // Freelist is (still) empty so extend this space by another segment.
185 freelist = Extend(space);
186 // Extend() adds one segment to the space and so to its freelist.
187 DCHECK_EQ(freelist.length(), kEntriesPerSegment);
188 }
189 }
190
191 if (V8_LIKELY(TryAllocateEntryFromFreelist(space, freelist))) {
192 break;
193 }
194 }
195
196 uint32_t allocated_entry = freelist.next();
197 DCHECK(space->Contains(allocated_entry));
198 DCHECK_IMPLIES(!space->is_internal_read_only_space(), allocated_entry != 0);
199 return allocated_entry;
200}

202template <typename Entry, size_t size>
203uint32_t ExternalEntityTable<Entry, size>::AllocateEntryBelow(
204 Space* space, uint32_t threshold_index) {
205 DCHECK(this->is_initialized());
206
207 FreelistHead freelist;
208 bool success = false;
209 while (!success) {
210 freelist = space->freelist_head_.load(std::memory_order_acquire);
211 // Check that the next free entry is below the threshold.
212 if (freelist.is_empty() || freelist.next() >= threshold_index) return 0;
213
214 success = TryAllocateEntryFromFreelist(space, freelist);
215 }
216
217 uint32_t allocated_entry = freelist.next();
218 DCHECK(space->Contains(allocated_entry));
219 DCHECK_NE(allocated_entry, 0);
220 DCHECK_LT(allocated_entry, threshold_index);
221 return allocated_entry;
222}

224template <typename Entry, size_t size>
225bool ExternalEntityTable<Entry, size>::TryAllocateEntryFromFreelist(
226 Space* space, FreelistHead freelist) {
227 DCHECK(!freelist.is_empty());
228 DCHECK(space->Contains(freelist.next()));
229
230 Entry& freelist_entry = this->at(freelist.next());
231 uint32_t next_freelist_entry = freelist_entry.GetNextFreelistEntryIndex();
232 FreelistHead new_freelist(next_freelist_entry, freelist.length() - 1);
233 bool success = space->freelist_head_.compare_exchange_strong(
234 freelist, new_freelist, std::memory_order_relaxed);
235
236 // When the CAS succeeded, the entry must've been a freelist entry.
237 // Otherwise, this is not guaranteed as another thread may have allocated
238 // and overwritten the same entry in the meantime.
239 if (success) {
240 DCHECK_IMPLIES(freelist.length() > 1, !new_freelist.is_empty());
241 DCHECK_IMPLIES(freelist.length() == 1, new_freelist.is_empty());
242 }
243 return success;
244}

246template <typename Entry, size_t size>
247typename ExternalEntityTable<Entry, size>::FreelistHead
248ExternalEntityTable<Entry, size>::Extend(Space* space) {
249 // Freelist should be empty when calling this method.
250 DCHECK_EQ(space->freelist_length(), 0);
251 // The caller must lock the space's mutex before extending it.
252 space->mutex_.AssertHeld();
253 // The read-only space must never be extended with a newly-allocated segment.
254 DCHECK(!space->is_internal_read_only_space());
255
256 // Allocate the new segment.
257 auto [segment, freelist_head] = this->AllocateAndInitializeSegment();
258 Extend(space, segment, freelist_head);
259 return freelist_head;
260}

262template <typename Entry, size_t size>
263void ExternalEntityTable<Entry, size>::Extend(Space* space, Segment segment,
264 FreelistHead freelist) {
265 // Freelist should be empty when calling this method.
266 DCHECK_EQ(space->freelist_length(), 0);
267 // The caller must lock the space's mutex before extending it.
268 space->mutex_.AssertHeld();
269
270 space->segments_.insert(segment);
271 CHECK_IMPLIES(!ExternalEntityTable::kUseContiguousMemory,
272 segment.number() != 0);
273 DCHECK_EQ(space->is_internal_read_only_space(), segment.number() == 0);
274 DCHECK_EQ(space->is_internal_read_only_space(),
275 segment.offset() == kInternalReadOnlySegmentOffset);
276
277 if (V8_UNLIKELY(space->is_internal_read_only_space())) {
278 // For the internal read-only segment, index 0 is reserved for the `null`
279 // entry. The underlying memory has been nulled by allocation, and is
280 // therefore already initialized.
281#ifdef DEBUG
282 uint32_t first = segment.first_entry();
283 CHECK_EQ(first, kInternalNullEntryIndex);
284 static constexpr uint8_t kNullBytes[kEntrySize] = {0};
285 CHECK_EQ(memcmp(&this->at(first), kNullBytes, kEntrySize), 0);
286#endif // DEBUG
287 }
288
289 // This must be a release store to prevent reordering of of earlier stores to
290 // the freelist (for example during initialization of the segment) from being
291 // reordered past this store. See AllocateEntry() for more details.
292 space->freelist_head_.store(freelist, std::memory_order_release);
293}

300template <typename Entry, size_t size>
301template <typename Callback>
302uint32_t ExternalEntityTable<Entry, size>::GenericSweep(Space* space,
303 Callback callback) {
304 DCHECK(space->BelongsTo(this));
305
306 // Lock the space. Technically this is not necessary since no other thread can
307 // allocate entries at this point, but some of the methods we call on the
308 // space assert that the lock is held.
309 base::SpinningMutexGuard guard(&space->mutex_);
310
311 // There must not be any entry allocations while the table is being swept as
312 // that would not be safe. Set the freelist to this special marker value to
313 // easily catch any violation of this requirement.
314 space->freelist_head_.store(kEntryAllocationIsForbiddenMarker,
315 std::memory_order_relaxed);
316
317 // Here we can iterate over the segments collection without taking a lock
318 // because no other thread can currently allocate entries in this space.
319 uint32_t current_freelist_head = 0;
320 uint32_t current_freelist_length = 0;
321 std::vector<Segment> segments_to_deallocate;
322
323 for (auto segment : base::Reversed(space->segments_)) {
324 // Remember the state of the freelist before this segment in case this
325 // segment turns out to be completely empty and we deallocate it.
326 uint32_t previous_freelist_head = current_freelist_head;
327 uint32_t previous_freelist_length = current_freelist_length;
328
329 // Process every entry in this segment, again going top to bottom.
330 for (WriteIterator it = this->iter_at(segment.last_entry());
331 it.index() >= segment.first_entry(); --it) {
332 if (!it->IsMarked()) {
333 it->MakeFreelistEntry(current_freelist_head);
334 current_freelist_head = it.index();
335 current_freelist_length++;
336 } else {
337 callback(*it);
338 it->Unmark();
339 }
340 }
341
342 // If a segment is completely empty, free it.
343 uint32_t free_entries = current_freelist_length - previous_freelist_length;
344 bool segment_is_empty = free_entries == kEntriesPerSegment;
345 if (segment_is_empty) {
346 segments_to_deallocate.push_back(segment);
347 // Restore the state of the freelist before this segment.
348 current_freelist_head = previous_freelist_head;
349 current_freelist_length = previous_freelist_length;
350 }
351 }
352
353 // We cannot remove the segments while iterating over the segments set, so
354 // defer that until now.
355 for (auto segment : segments_to_deallocate) {
356 // Segment zero is reserved.
357 DCHECK_NE(segment.number(), 0);
358 this->FreeTableSegment(segment);
359 space->segments_.erase(segment);
360 }
361
362 FreelistHead new_freelist(current_freelist_head, current_freelist_length);
363 space->freelist_head_.store(new_freelist, std::memory_order_release);
364 DCHECK_EQ(space->freelist_length(), current_freelist_length);
365
366 uint32_t num_live_entries = space->capacity() - current_freelist_length;
367 return num_live_entries;
368}
