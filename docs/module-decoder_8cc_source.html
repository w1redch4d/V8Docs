75ModuleResult DecodeWasmModule(
76 WasmEnabledFeatures enabled_features,
77 base::Vector<const uint8_t> wire_bytes, bool validate_functions,
78 ModuleOrigin origin, Counters* counters,
79 std::shared_ptr<metrics::Recorder> metrics_recorder,
80 v8::metrics::Recorder::ContextId context_id, DecodingMethod decoding_method,
81 WasmDetectedFeatures* detected_features) {
82 if (counters) {
83 auto size_counter =
84 SELECT_WASM_COUNTER(counters, origin, wasm, module_size_bytes);
85 static_assert(kV8MaxWasmModuleSize < kMaxInt);
86 size_counter->AddSample(static_cast<int>(wire_bytes.size()));
87 }
88
89 v8::metrics::WasmModuleDecoded metrics_event;
90 base::ElapsedTimer timer;
91 timer.Start();
92 ModuleResult result =
93 DecodeWasmModule(enabled_features, wire_bytes, validate_functions, origin,
94 detected_features);
95 if (counters && result.ok()) {
96 auto counter =
97 SELECT_WASM_COUNTER(counters, origin, wasm_functions_per, module);
98 counter->AddSample(
99 static_cast<int>(result.value()->num_declared_functions));
100 }
101
102 // Record event metrics.
103 metrics_event.wall_clock_duration_in_us = timer.Elapsed().InMicroseconds();
104 timer.Stop();
105 metrics_event.success = result.ok();
106 metrics_event.async = decoding_method == DecodingMethod::kAsync ||
107 decoding_method == DecodingMethod::kAsyncStream;
108 metrics_event.streamed = decoding_method == DecodingMethod::kSyncStream ||
109 decoding_method == DecodingMethod::kAsyncStream;
110 if (result.ok()) {
111 metrics_event.function_count = result.value()->num_declared_functions;
112 }
113 metrics_event.module_size_in_bytes = wire_bytes.size();
114 metrics_recorder->DelayMainThreadEvent(metrics_event, context_id);
115
116 return result;
117}

225AsmJsOffsetsResult DecodeAsmJsOffsets(
226 base::Vector<const uint8_t> encoded_offsets) {
227 std::vector<AsmJsOffsetFunctionEntries> functions;
228
229 Decoder decoder(encoded_offsets);
230 uint32_t functions_count = decoder.consume_u32v("functions count");
231 // Consistency check.
232 DCHECK_GE(encoded_offsets.size(), functions_count);
233 functions.reserve(functions_count);
234
235 for (uint32_t i = 0; i < functions_count; ++i) {
236 uint32_t size = decoder.consume_u32v("table size");
237 if (size == 0) {
238 functions.emplace_back();
239 continue;
240 }
241 DCHECK(decoder.checkAvailable(size));
242 const uint8_t* table_end = decoder.pc() + size;
243 uint32_t locals_size = decoder.consume_u32v("locals size");
244 int function_start_position = decoder.consume_u32v("function start pos");
245 int function_end_position = function_start_position;
246 int last_byte_offset = locals_size;
247 int last_asm_position = function_start_position;
248 std::vector<AsmJsOffsetEntry> func_asm_offsets;
249 func_asm_offsets.reserve(size / 4); // conservative estimation
250 // Add an entry for the stack check, associated with position 0.
251 func_asm_offsets.push_back(
252 {0, function_start_position, function_start_position});
253 while (decoder.pc() < table_end) {
254 DCHECK(decoder.ok());
255 last_byte_offset += decoder.consume_u32v("byte offset delta");
256 int call_position =
257 last_asm_position + decoder.consume_i32v("call position delta");
258 int to_number_position =
259 call_position + decoder.consume_i32v("to_number position delta");
260 last_asm_position = to_number_position;
261 if (decoder.pc() == table_end) {
262 // The last entry is the function end marker.
263 DCHECK_EQ(call_position, to_number_position);
264 function_end_position = call_position;
265 } else {
266 func_asm_offsets.push_back(
267 {last_byte_offset, call_position, to_number_position});
268 }
269 }
270 DCHECK_EQ(decoder.pc(), table_end);
271 functions.emplace_back(AsmJsOffsetFunctionEntries{
272 function_start_position, function_end_position,
273 std::move(func_asm_offsets)});
274 }
275 DCHECK(decoder.ok());
276 DCHECK(!decoder.more());
277
278 return decoder.toResult(AsmJsOffsets{std::move(functions)});
279}

281std::vector<CustomSectionOffset> DecodeCustomSections(
282 base::Vector<const uint8_t> bytes) {
283 Decoder decoder(bytes);
284 decoder.consume_bytes(4, "wasm magic");
285 decoder.consume_bytes(4, "wasm version");
286
287 std::vector<CustomSectionOffset> result;
288
289 while (decoder.more()) {
290 uint8_t section_code = decoder.consume_u8("section code");
291 uint32_t section_length = decoder.consume_u32v("section length");
292 uint32_t section_start = decoder.pc_offset();
293 if (section_code != 0) {
294 // Skip known sections.
295 decoder.consume_bytes(section_length, "section bytes");
296 continue;
297 }
298 uint32_t name_length = decoder.consume_u32v("name length");
299 uint32_t name_offset = decoder.pc_offset();
300 decoder.consume_bytes(name_length, "section name");
301 uint32_t payload_offset = decoder.pc_offset();
302 if (section_length < (payload_offset - section_start)) {
303 decoder.error("invalid section length");
304 break;
305 }
306 uint32_t payload_length = section_length - (payload_offset - section_start);
307 decoder.consume_bytes(payload_length);
308 if (decoder.failed()) break;
309 result.push_back({{section_start, section_length},
310 {name_offset, name_length},
311 {payload_offset, payload_length}});
312 }
313
314 return result;
315}

319bool FindNameSection(Decoder* decoder) {
320 static constexpr int kModuleHeaderSize = 8;
321 decoder->consume_bytes(kModuleHeaderSize, "module header");
322
323 WasmSectionIterator section_iter(decoder, ITracer::NoTrace);
324
325 while (decoder->ok() && section_iter.more() &&
326 section_iter.section_code() != kNameSectionCode) {
327 section_iter.advance(true);
328 }
329 if (!section_iter.more()) return false;
330
331 // Reset the decoder to not read beyond the name section end.
332 decoder->Reset(section_iter.payload(), decoder->pc_offset());
333 return true;
334}

336enum class EmptyNames : bool { kAllow, kSkip };
337
338void DecodeNameMapInternal(NameMap& target, Decoder& decoder,
339 EmptyNames empty_names = EmptyNames::kSkip) {
340 uint32_t count = decoder.consume_u32v("names count");
341 for (uint32_t i = 0; i < count; i++) {
342 uint32_t index = decoder.consume_u32v("index");
343 WireBytesRef name =
344 consume_string(&decoder, unibrow::Utf8Variant::kLossyUtf8, "name");
345 if (!decoder.ok()) break;
346 if (index > NameMap::kMaxKey) continue;
347 if (empty_names == EmptyNames::kSkip && name.is_empty()) continue;
348 if (!validate_utf8(&decoder, name)) continue;
349 target.Put(index, name);
350 }
351 target.FinishInitialization();
352}
353
354void DecodeNameMap(NameMap& target, Decoder& decoder,
355 uint32_t subsection_payload_length,
356 EmptyNames empty_names = EmptyNames::kSkip) {
357 if (target.is_set()) {
358 decoder.consume_bytes(subsection_payload_length);
359 return;
360 }
361 DecodeNameMapInternal(target, decoder, empty_names);
362}
363
364void DecodeIndirectNameMap(IndirectNameMap& target, Decoder& decoder,
365 uint32_t subsection_payload_length) {
366 if (target.is_set()) {
367 decoder.consume_bytes(subsection_payload_length);
368 return;
369 }
370 uint32_t outer_count = decoder.consume_u32v("outer count");
371 for (uint32_t i = 0; i < outer_count; ++i) {
372 uint32_t outer_index = decoder.consume_u32v("outer index");
373 if (outer_index > IndirectNameMap::kMaxKey) continue;
374 NameMap names;
375 DecodeNameMapInternal(names, decoder);
376 target.Put(outer_index, std::move(names));
377 if (!decoder.ok()) break;
378 }
379 target.FinishInitialization();
380}

384void DecodeFunctionNames(base::Vector<const uint8_t> wire_bytes,
385 NameMap& names) {
386 Decoder decoder(wire_bytes);
387 if (FindNameSection(&decoder)) {
388 while (decoder.ok() && decoder.more()) {
389 uint8_t name_type = decoder.consume_u8("name type");
390 if (name_type & 0x80) break; // no varuint7
391
392 uint32_t name_payload_len = decoder.consume_u32v("name payload length");
393 if (!decoder.checkAvailable(name_payload_len)) break;
394
395 if (name_type != NameSectionKindCode::kFunctionCode) {
396 decoder.consume_bytes(name_payload_len, "name subsection payload");
397 continue;
398 }
399 // We need to allow empty function names for spec-conformant stack traces.
400 DecodeNameMapInternal(names, decoder, EmptyNames::kAllow);
401 // The spec allows only one occurrence of each subsection. We could be
402 // more permissive and allow repeated subsections; in that case we'd
403 // have to delay calling {target.FinishInitialization()} on the function
404 // names map until we've seen them all.
405 // For now, we stop decoding after finding the first function names
406 // subsection.
407 return;
408 }
409 }

413// A task that validates multiple functions in parallel, storing the earliest
414// validation error in {this} decoder.
415class ValidateFunctionsTask : public JobTask {
416 public:
417 explicit ValidateFunctionsTask(
418 base::Vector<const uint8_t> wire_bytes, const WasmModule* module,
419 WasmEnabledFeatures enabled_features, std::function<bool(int)> filter,
420 WasmError* error_out,
421 std::atomic<WasmDetectedFeatures>* detected_features)
422 : wire_bytes_(wire_bytes),
423 module_(module),
424 enabled_features_(enabled_features),
425 filter_(std::move(filter)),
426 next_function_(module->num_imported_functions),
427 after_last_function_(next_function_ + module->num_declared_functions),
428 error_out_(error_out),
429 detected_features_(detected_features) {
430 DCHECK(!error_out->has_error());
431 }
432
433 void Run(JobDelegate* delegate) override {
434 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.wasm.detailed"),
435 "wasm.ValidateFunctionsTask");
436
437 WasmDetectedFeatures detected_features;
438 Zone zone(GetWasmEngine()->allocator(), ZONE_NAME);
439 do {
440 // Get the index of the next function to validate.
441 // {fetch_add} might overrun {after_last_function_} by a bit. Since the
442 // number of functions is limited to a value much smaller than the
443 // integer range, this is near impossible to happen.
444 static_assert(kV8MaxWasmTotalFunctions < kMaxInt / 2);
445 int func_index;
446 do {
447 func_index = next_function_.fetch_add(1, std::memory_order_relaxed);
448 if (V8_UNLIKELY(func_index >= after_last_function_)) {
449 UpdateDetectedFeatures(detected_features);
450 return;
451 }
452 DCHECK_LE(0, func_index);
453 } while ((filter_ && !filter_(func_index)) ||
454 module_->function_was_validated(func_index));
455
456 zone.Reset();
457 if (!ValidateFunction(func_index, &zone, &detected_features)) {
458 // No need to validate any more functions.
459 next_function_.store(after_last_function_, std::memory_order_relaxed);
460 return;
461 }
462 } while (!delegate->ShouldYield());
463 UpdateDetectedFeatures(detected_features);
464 }
465
466 size_t GetMaxConcurrency(size_t /* worker_count */) const override {
467 int next_func = next_function_.load(std::memory_order_relaxed);
468 return std::max(0, after_last_function_ - next_func);
469 }
470
471 private:
472 bool ValidateFunction(int func_index, Zone* zone,
473 WasmDetectedFeatures* detected_features) {
474 const WasmFunction& function = module_->functions[func_index];
475 DCHECK_LT(0, function.code.offset());
476 bool is_shared = module_->type(function.sig_index).is_shared;
477 FunctionBody body{function.sig, function.code.offset(),
478 wire_bytes_.begin() + function.code.offset(),
479 wire_bytes_.begin() + function.code.end_offset(),
480 is_shared};
481 DecodeResult validation_result = ValidateFunctionBody(
482 zone, enabled_features_, module_, detected_features, body);
483 if (V8_UNLIKELY(validation_result.failed())) {
484 SetError(func_index, std::move(validation_result).error());
485 return false;
486 }
487 module_->set_function_validated(func_index);
488 return true;
489 }
490
491 // Set the error from the argument if it's earlier than the error we already
492 // have (or if we have none yet). Thread-safe.
493 void SetError(int func_index, WasmError error) {
494 base::SpinningMutexGuard mutex_guard{&set_error_mutex_};
495 if (error_out_->has_error() && error_out_->offset() <= error.offset()) {
496 return;
497 }
498 *error_out_ = GetWasmErrorWithName(wire_bytes_, func_index, module_, error);
499 }

501 void UpdateDetectedFeatures(WasmDetectedFeatures detected_features) {
502 WasmDetectedFeatures old_features =
503 detected_features_->load(std::memory_order_relaxed);
504 while (!detected_features_->compare_exchange_weak(
505 old_features, old_features | detected_features,
506 std::memory_order_relaxed)) {
507 // Retry with updated {old_features}.
508 }
509 }
510
511 const base::Vector<const uint8_t> wire_bytes_;
512 const WasmModule* const module_;
513 const WasmEnabledFeatures enabled_features_;
514 const std::function<bool(int)> filter_;
515 std::atomic<int> next_function_;
516 const int after_last_function_;
517 base::SpinningMutex set_error_mutex_;
518 WasmError* const error_out_;
519 std::atomic<WasmDetectedFeatures>* const detected_features_;
520};

533 class NeverYieldDelegate final : public JobDelegate {
534 public:
535 bool ShouldYield() override { return false; }
536
537 bool IsJoiningThread() const override { UNIMPLEMENTED(); }
538 void NotifyConcurrencyIncrease() override { UNIMPLEMENTED(); }
539 uint8_t GetTaskId() override { UNIMPLEMENTED(); }

541
542 // Create a {ValidateFunctionsTask} to validate all functions. The earliest
543 // error found will be set on this decoder.
544 WasmError validation_error;
545 std::atomic<WasmDetectedFeatures> detected_features;
546 std::unique_ptr<JobTask> validate_job =
547 std::make_unique<ValidateFunctionsTask>(
548 wire_bytes, module, enabled_features, std::move(filter),
549 &validation_error, &detected_features);
550
551 if (v8_flags.single_threaded) {
552 // In single-threaded mode, run the {ValidateFunctionsTask} synchronously.
553 NeverYieldDelegate delegate;
554 validate_job->Run(&delegate);
555 } else {
556 // Spawn the task and join it.
557 std::unique_ptr<JobHandle> job_handle = V8::GetCurrentPlatform()->CreateJob(
558 TaskPriority::kUserVisible, std::move(validate_job));
559 job_handle->Join();
560 }
561
562 *detected_features_out |= detected_features.load(std::memory_order_relaxed);
563 return validation_error;
564}

582DecodedNameSection::DecodedNameSection(base::Vector<const uint8_t> wire_bytes,
583 WireBytesRef name_section) {
584 if (name_section.is_empty()) return; // No name section.
585 Decoder decoder(wire_bytes.begin() + name_section.offset(),
586 wire_bytes.begin() + name_section.end_offset(),
587 name_section.offset());
588 while (decoder.ok() && decoder.more()) {
589 uint8_t name_type = decoder.consume_u8("name type");
590 if (name_type & 0x80) break; // no varuint7
591
592 uint32_t name_payload_len = decoder.consume_u32v("name payload length");
593 if (!decoder.checkAvailable(name_payload_len)) break;
594
595 switch (name_type) {
596 case kModuleCode:
597 case kFunctionCode:
598 // Already handled elsewhere.
599 decoder.consume_bytes(name_payload_len);
600 break;
601 case kLocalCode:
602 static_assert(kV8MaxWasmTotalFunctions <= IndirectNameMap::kMaxKey);
603 static_assert(kV8MaxWasmFunctionLocals <= NameMap::kMaxKey);
604 DecodeIndirectNameMap(local_names_, decoder, name_payload_len);
605 break;
606 case kLabelCode:
607 static_assert(kV8MaxWasmTotalFunctions <= IndirectNameMap::kMaxKey);
608 static_assert(kV8MaxWasmFunctionSize <= NameMap::kMaxKey);
609 DecodeIndirectNameMap(label_names_, decoder, name_payload_len);
610 break;
611 case kTypeCode:
612 static_assert(kV8MaxWasmTypes <= NameMap::kMaxKey);
613 DecodeNameMap(type_names_, decoder, name_payload_len);
614 break;
615 case kTableCode:
616 static_assert(kV8MaxWasmTables <= NameMap::kMaxKey);
617 DecodeNameMap(table_names_, decoder, name_payload_len);
618 break;
619 case kMemoryCode:
620 static_assert(kV8MaxWasmMemories <= NameMap::kMaxKey);
621 DecodeNameMap(memory_names_, decoder, name_payload_len);
622 break;
623 case kGlobalCode:
624 static_assert(kV8MaxWasmGlobals <= NameMap::kMaxKey);
625 DecodeNameMap(global_names_, decoder, name_payload_len);
626 break;
627 case kElementSegmentCode:
628 static_assert(kV8MaxWasmTableInitEntries <= NameMap::kMaxKey);
629 DecodeNameMap(element_segment_names_, decoder, name_payload_len);
630 break;
631 case kDataSegmentCode:
632 static_assert(kV8MaxWasmDataSegments <= NameMap::kMaxKey);
633 DecodeNameMap(data_segment_names_, decoder, name_payload_len);
634 break;
635 case kFieldCode:
636 static_assert(kV8MaxWasmTypes <= IndirectNameMap::kMaxKey);
637 static_assert(kV8MaxWasmStructFields <= NameMap::kMaxKey);
638 DecodeIndirectNameMap(field_names_, decoder, name_payload_len);
639 break;
640 case kTagCode:
