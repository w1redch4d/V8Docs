64inline Tagged<Object> DeoptimizationLiteralArray::get(
65 PtrComprCageBase cage_base, int index) const {
66 Tagged<MaybeObject> maybe = TrustedWeakFixedArray::get(index);
67
68 // Slots in the DeoptimizationLiteralArray should only be cleared when there
69 // is no possible code path that could need that slot. This works because the
70 // weakly-held deoptimization literals are basically local variables that
71 // TurboFan has decided not to keep on the stack. Thus, if the deoptimization
72 // literal goes away, then whatever code needed it should be unreachable. The
73 // exception is currently running InstructionStream: in that case, the
74 // deoptimization literals array might be the only thing keeping the target
75 // object alive. Thus, when an InstructionStream is running, we strongly mark
76 // all of its deoptimization literals.
77 CHECK(!maybe.IsCleared());
78
79 return maybe.GetHeapObjectOrSmi();
80}

87inline void DeoptimizationLiteralArray::set(int index, Tagged<Object> value) {
88 Tagged<MaybeObject> maybe = value;
89 if (IsBytecodeArray(value)) {
90 // The BytecodeArray lives in trusted space, so we cannot reference it from
91 // a fixed array. However, we can use the BytecodeArray's wrapper object,
92 // which exists for exactly this purpose.
93 maybe = Cast<BytecodeArray>(value)->wrapper();
94#ifdef V8_ENABLE_SANDBOX
95 } else if (IsRegExpData(value)) {
96 // Store the RegExpData wrapper if the sandbox is enabled, as data lives in
97 // trusted space. We can't store a tagged value to a trusted space object
98 // inside the sandbox, we'd need to go through the trusted pointer table.
99 // Otherwise we can store the RegExpData object directly.
100 maybe = Cast<RegExpData>(value)->wrapper();
101#endif
102 } else if (Code::IsWeakObjectInDeoptimizationLiteralArray(value)) {
103 maybe = MakeWeak(maybe);
104 }
105 TrustedWeakFixedArray::set(index, maybe);
106}
