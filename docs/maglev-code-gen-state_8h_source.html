92 uint32_t stack_check_offset() {
93 int32_t parameter_slots =
94 compilation_info_->toplevel_compilation_unit()->parameter_count();
95 uint32_t stack_slots = tagged_slots_ + untagged_slots_;
96 DCHECK(is_int32(stack_slots));
97 int32_t optimized_frame_height = parameter_slots * kSystemPointerSize +
98 StandardFrameConstants::kFixedFrameSize +
99 stack_slots * kSystemPointerSize;
100 DCHECK(is_int32(max_deopted_stack_size_));
101 int32_t signed_max_unoptimized_frame_height =
102 static_cast<int32_t>(max_deopted_stack_size_);
103
104 // The offset is either the delta between the optimized frames and the
105 // interpreted frame, or the maximal number of bytes pushed to the stack
106 // while preparing for function calls, whichever is bigger.
107 uint32_t frame_height_delta = static_cast<uint32_t>(std::max(
108 signed_max_unoptimized_frame_height - optimized_frame_height, 0));
109 uint32_t max_pushed_argument_bytes =
110 static_cast<uint32_t>(max_call_stack_args_ * kSystemPointerSize);
111 return std::max(frame_height_delta, max_pushed_argument_bytes);
112 }
