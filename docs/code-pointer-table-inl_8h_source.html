17void CodePointerTableEntry::MakeCodePointerEntry(Address code,
18 Address entrypoint,
19 CodeEntrypointTag tag,
20 bool mark_as_alive) {
21 DCHECK_EQ(code & kMarkingBit, 0);
22 DCHECK_EQ(entrypoint >> kCodeEntrypointTagShift, 0);
23 DCHECK_NE(tag, kFreeCodePointerTableEntryTag);
24
25 if (mark_as_alive) code |= kMarkingBit;
26 entrypoint_.store(entrypoint ^ tag, std::memory_order_relaxed);
27 code_.store(code, std::memory_order_relaxed);
28}

44Address CodePointerTableEntry::GetCodeObject() const {
45 DCHECK(!IsFreelistEntry());
46 // We reuse the heap object tag bit as marking bit, so we need to explicitly
47 // set it here when accessing the pointer.
48 return code_.load(std::memory_order_relaxed) | kMarkingBit;
49}

51void CodePointerTableEntry::SetCodeObject(Address new_value) {
52 DCHECK(!IsFreelistEntry());
53 // SetContent shouldn't change the marking state of the entry. Currently this
54 // is always automatically the case, but if this ever fails, we might need to
55 // manually copy the marking bit.
56 DCHECK_EQ(code_ & kMarkingBit, new_value & kMarkingBit);
57 code_.store(new_value, std::memory_order_relaxed);

60void CodePointerTableEntry::MakeFreelistEntry(uint32_t next_entry_index) {
61 Address value = kFreeEntryTag | next_entry_index;

75void CodePointerTableEntry::Mark() {
76 Address old_value = code_.load(std::memory_order_relaxed);
77 Address new_value = old_value | kMarkingBit;
78
79 // We don't need to perform the CAS in a loop since it can only fail if a new
80 // value has been written into the entry. This, however, will also have set
81 // the marking bit.
82 bool success = code_.compare_exchange_strong(old_value, new_value,
83 std::memory_order_relaxed);
84 DCHECK(success || (old_value & kMarkingBit) == kMarkingBit);
85 USE(success);
86}

88void CodePointerTableEntry::Unmark() {
89 Address value = code_.load(std::memory_order_relaxed);
90 value &= ~kMarkingBit;
91 code_.store(value, std::memory_order_relaxed);
92}

94bool CodePointerTableEntry::IsMarked() const {

105Address CodePointerTable::GetCodeObject(CodePointerHandle handle) const {
106 uint32_t index = HandleToIndex(handle);
107 // Due to the fact that we use the heap object tag as marking bit, this table
108 // (in contrast to the trusted pointer table) does not return Smi::zero() for
109 // the 0th entry. That entry must therefore not be accessed here.
110 DCHECK_NE(index, 0);
111 return at(index).GetCodeObject();
112}

131 DCHECK(space->BelongsTo(this));
132 uint32_t index = AllocateEntry(space);

139void CodePointerTable::Mark(Space* space, CodePointerHandle handle) {
140 DCHECK(space->BelongsTo(this));
141 // The null entry is immortal and immutable, so no need to mark it as alive.
142 if (handle == kNullCodePointerHandle) return;
143
144 uint32_t index = HandleToIndex(handle);
145 DCHECK(space->Contains(index));
146
147 CFIMetadataWriteScope write_scope("CodePointerTable write");
148 at(index).Mark();
149}

151template <typename Callback>
152void CodePointerTable::IterateActiveEntriesIn(Space* space, Callback callback) {
153 IterateEntriesIn(space, [&](uint32_t index) {
154 if (!at(index).IsFreelistEntry()) {
155 callback(IndexToHandle(index), at(index).GetCodeObject());
156 }
157 });

160uint32_t CodePointerTable::HandleToIndex(CodePointerHandle handle) const {
161 uint32_t index = handle >> kCodePointerHandleShift;
162 DCHECK_EQ(handle,
163 (index << kCodePointerHandleShift) | kCodePointerHandleMarker);
164 return index;
165}

167CodePointerHandle CodePointerTable::IndexToHandle(uint32_t index) const {
168 CodePointerHandle handle = index << kCodePointerHandleShift;
169 DCHECK_EQ(index, handle >> kCodePointerHandleShift);
170 return handle | kCodePointerHandleMarker;
171}
