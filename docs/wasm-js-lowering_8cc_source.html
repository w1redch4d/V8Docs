22Reduction WasmJSLowering::Reduce(Node* node) {
23 switch (node->opcode()) {
24 case IrOpcode::kTrapIf:
25 case IrOpcode::kTrapUnless: {
26 Node* effect = NodeProperties::GetEffectInput(node);
27 Node* control = NodeProperties::GetControlInput(node);
28 Node* trap_condition = NodeProperties::GetValueInput(node, 0);
29 auto ool_trap = gasm_.MakeDeferredLabel();
30 gasm_.InitializeEffectControl(effect, control);
31 if (node->opcode() == IrOpcode::kTrapIf) {
32 gasm_.GotoIf(trap_condition, &ool_trap);
33 } else {
34 DCHECK_EQ(node->opcode(), IrOpcode::kTrapUnless);
35 gasm_.GotoIfNot(trap_condition, &ool_trap);
36 }
37 effect = gasm_.effect();
38 control = gasm_.control();
39 Node* goto_node = control;
40
41 // Generate out of line code.
42 gasm_.InitializeEffectControl(nullptr, nullptr);
43 gasm_.Bind(&ool_trap);
44 TrapId trap_id = TrapIdOf(node->op());
45 Builtin trap = static_cast<Builtin>(trap_id);
46
47 // Create new FrameState with the correct source position (the position
48 // of the trap location).
49 Node* frame_state = NodeProperties::GetValueInput(node, 1);
50 const FrameStateInfo& info = FrameState(frame_state).frame_state_info();
51 SourcePosition position = source_position_table_->GetSourcePosition(node);
52 Node* new_frame_state = mcgraph_->graph()->CloneNode(frame_state);
53 BytecodeOffset bailout_id(position.ScriptOffset());
54 const Operator* frame_state_op = mcgraph_->common()->FrameState(
55 bailout_id, info.state_combine(), info.function_info());
56 NodeProperties::ChangeOp(new_frame_state, frame_state_op);
57
58 gasm_.CallBuiltinWithFrameState(trap, Operator::kNoProperties,
59 new_frame_state);
60 Node* terminate = mcgraph_->graph()->NewNode(
61 mcgraph_->common()->Throw(), gasm_.effect(), gasm_.control());
62 MergeControlToEnd(mcgraph_->graph(), mcgraph_->common(), terminate);
63
64 // Replace the trap node with the conditional branch.
65 gasm_.InitializeEffectControl(effect, control);
66 ReplaceWithValue(node, goto_node, gasm_.effect(), gasm_.control());
67 node->Kill();
68 return Replace(goto_node);
69 }
70 default:
71 return NoChange();
72 }
73}
