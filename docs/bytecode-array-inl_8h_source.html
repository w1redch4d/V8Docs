82interpreter::Register BytecodeArray::incoming_new_target_or_generator_register()
83 const {
84 int32_t register_operand =
85 ReadField<int32_t>(kIncomingNewTargetOrGeneratorRegisterOffset);
86 if (register_operand == 0) {
87 return interpreter::Register::invalid_value();
88 } else {
89 return interpreter::Register::FromOperand(register_operand);

93void BytecodeArray::set_incoming_new_target_or_generator_register(
94 interpreter::Register incoming_new_target_or_generator_register) {
95 if (!incoming_new_target_or_generator_register.is_valid()) {
96 WriteField<int32_t>(kIncomingNewTargetOrGeneratorRegisterOffset, 0);
97 } else {
98 DCHECK(incoming_new_target_or_generator_register.index() <
99 register_count());
100 DCHECK_NE(0, incoming_new_target_or_generator_register.ToOperand());
101 WriteField<int32_t>(kIncomingNewTargetOrGeneratorRegisterOffset,
102 incoming_new_target_or_generator_register.ToOperand());
103 }

120DEF_GETTER(BytecodeArray, SourcePositionTable, Tagged<TrustedByteArray>) {
121 // WARNING: This function may be called from a background thread, hence
122 // changes to how it accesses the heap can easily lead to bugs.
123 Tagged<Object> maybe_table = raw_source_position_table(kAcquireLoad);
124 if (IsTrustedByteArray(maybe_table))
125 return Cast<TrustedByteArray>(maybe_table);
126 DCHECK_EQ(maybe_table, Smi::zero());
127 return GetIsolateFromWritableObject(*this)
128 ->heap()
129 ->empty_trusted_byte_array();
130}

162DEF_GETTER(BytecodeArray, SizeIncludingMetadata, int) {
163 int size = BytecodeArraySize();
164 Tagged<Object> maybe_constant_pool = raw_constant_pool(cage_base);
165 if (IsTrustedFixedArray(maybe_constant_pool)) {
166 size += Cast<TrustedFixedArray>(maybe_constant_pool)->Size();
167 } else {
168 DCHECK_EQ(maybe_constant_pool, Smi::zero());
169 }
170 Tagged<Object> maybe_handler_table = raw_handler_table(cage_base);
171 if (IsTrustedByteArray(maybe_handler_table)) {
172 size += Cast<TrustedByteArray>(maybe_handler_table)->AllocatedSize();
173 } else {
174 DCHECK_EQ(maybe_handler_table, Smi::zero());
175 }
176 Tagged<Object> maybe_table = raw_source_position_table(kAcquireLoad);
177 if (IsByteArray(maybe_table)) {
178 size += Cast<ByteArray>(maybe_table)->AllocatedSize();
179 }
180 return size;
181}
