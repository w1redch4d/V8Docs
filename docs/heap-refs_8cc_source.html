36// There are several kinds of ObjectData values.
37//
38// kSmi: The underlying V8 object is a Smi and the data is an instance of the
39// base class (ObjectData), i.e. it's basically just the handle. Because the
40// object is a Smi, it's safe to access the handle in order to extract the
41// number value, and AsSmi() does exactly that.
42//
43// kBackgroundSerializedHeapObject: The underlying V8 object is a HeapObject
44// and the data is an instance of the corresponding (most-specific) subclass,
45// e.g. JSFunctionData, which provides serialized information about the
46// object. Allows serialization from the background thread.
47//
48// kUnserializedHeapObject: The underlying V8 object is a HeapObject and the
49// data is an instance of the base class (ObjectData), i.e. it basically
50// carries no information other than the handle.
51//
52// kNeverSerializedHeapObject: The underlying V8 object is a (potentially
53// mutable) HeapObject and the data is an instance of ObjectData. Its handle
54// must be persistent so that the GC can update it at a safepoint. Via this
55// handle, the object can be accessed concurrently to the main thread.
56//
57// kUnserializedReadOnlyHeapObject: The underlying V8 object is a read-only
58// HeapObject and the data is an instance of ObjectData. For
59// ReadOnlyHeapObjects, it is OK to access heap even from off-thread, so
60// these objects need not be serialized.
61enum ObjectDataKind {
62 kSmi,
63 kBackgroundSerializedHeapObject,
64 kUnserializedHeapObject,
65 kNeverSerializedHeapObject,
66 kUnserializedReadOnlyHeapObject
67};

220bool PropertyCellData::Cache(JSHeapBroker* broker) {
221 if (serialized()) return true;
222
223 TraceScope tracer(broker, this, "PropertyCellData::Serialize");
224 auto cell = Cast<PropertyCell>(object());
225
226 // While this code runs on a background thread, the property cell might
227 // undergo state transitions via calls to PropertyCell::Transition. These
228 // transitions follow a certain protocol on which we rely here to ensure that
229 // we only report success when we can guarantee consistent data. A key
230 // property is that after transitioning from cell type A to B (A != B), there
231 // will never be a transition back to A, unless A is kConstant and the new
232 // value is the hole (i.e. the property cell was invalidated, which is a final
233 // state).
234
235 PropertyDetails property_details = cell->property_details(kAcquireLoad);
236
237 Handle<Object> value =
238 broker->CanonicalPersistentHandle(cell->value(kAcquireLoad));
239 if (broker->ObjectMayBeUninitialized(value)) {
240 DCHECK(!broker->IsMainThread());
241 return false;
242 }
243
244 {
245 PropertyDetails property_details_again =
246 cell->property_details(kAcquireLoad);
247 if (property_details != property_details_again) {
248 DCHECK(!broker->IsMainThread());
249 return false;
250 }
251 }
252
253 if (property_details.cell_type() == PropertyCellType::kInTransition) {
254 DCHECK(!broker->IsMainThread());
255 return false;
256 }
257
258 ObjectData* value_data = broker->TryGetOrCreateData(value);
259 if (value_data == nullptr) {
260 DCHECK(!broker->IsMainThread());
261 return false;
262 }
263
264 PropertyCell::CheckDataIsCompatible(property_details, *value);
265
266 DCHECK(!serialized());
267 property_details_ = property_details;
268 value_ = value_data;
269 DCHECK(serialized());
270 return true;
271}

300std::optional<Tagged<Object>> GetOwnFastConstantDataPropertyFromHeap(
301 JSHeapBroker* broker, JSObjectRef holder, Representation representation,
302 FieldIndex field_index) {
303 std::optional<Tagged<Object>> constant;
304 {
305 DisallowGarbageCollection no_gc;
306 PtrComprCageBase cage_base = broker->cage_base();
307
308 // This check to ensure the live map is the same as the cached map to
309 // to protect us against reads outside the bounds of the heap. This could
310 // happen if the Ref was created in a prior GC epoch, and the object
311 // shrunk in size. It might end up at the edge of a heap boundary. If
312 // we see that the map is the same in this GC epoch, we are safe.
313 Tagged<Map> map = holder.object()->map(cage_base, kAcquireLoad);
314 if (*holder.map(broker).object() != map) {
315 TRACE_BROKER_MISSING(broker, "Map changed for " << holder);
316 return {};
317 }
318
319 if (field_index.is_inobject()) {
320 constant =
321 holder.object()->RawInobjectPropertyAt(cage_base, map, field_index);
322 if (!constant.has_value()) {
323 TRACE_BROKER_MISSING(
324 broker, "Constant field in " << holder << " is unsafe to read");
325 return {};
326 }
327 } else {
328 Tagged<Object> raw_properties_or_hash =
329 holder.object()->raw_properties_or_hash(cage_base, kRelaxedLoad);
330 // Ensure that the object is safe to inspect.
331 if (broker->ObjectMayBeUninitialized(raw_properties_or_hash)) {
332 return {};
333 }
334 if (!IsPropertyArray(raw_properties_or_hash, cage_base)) {
335 TRACE_BROKER_MISSING(
336 broker,
337 "Expected PropertyArray for backing store in " << holder << ".");
338 return {};
339 }
340 Tagged<PropertyArray> properties =
341 Cast<PropertyArray>(raw_properties_or_hash);
342 const int array_index = field_index.outobject_array_index();
343 if (array_index < properties->length(kAcquireLoad)) {
344 constant = properties->get(array_index);
345 } else {
346 TRACE_BROKER_MISSING(
347 broker, "Backing store for " << holder << " not long enough.");
348 return {};
349 }
350 }
351 // We might read the uninitialized sentinel, if we race with the main
352 // thread adding a new property to the object (having set the map, but not
353 // yet initialised the property value). Since this is a tight race, it won't
354 // happen very often, so we can just abort the load.
355 // TODO(leszeks): We could instead sleep/yield and spin the load, since the
356 // timing on this is tight enough that we wouldn't delay the compiler thread
357 // by much.
358 if (IsUninitialized(constant.value())) {
359 TRACE_BROKER_MISSING(broker, "Read uninitialized property.");
360 return {};
361 }
362
363 // {constant} needs to pass the gc predicate before we can introspect on it.
364 if (broker->ObjectMayBeUninitialized(constant.value())) return {};
365
366 // Ensure that {constant} matches the {representation} we expect for the
367 // field.
368 if (!Object::FitsRepresentation(*constant, representation, false)) {
369 const char* repString = IsSmi(*constant) ? "Smi"
370 : IsHeapNumber(*constant) ? "HeapNumber"
371 : "HeapObject";
372 TRACE_BROKER_MISSING(broker, "Mismatched representation for "
373 << holder << ". Expected "
374 << representation << ", but object is a "
375 << repString);
376 return {};
377 }
378 }
379 return constant;
380}

588int InstanceSizeWithMinSlack(JSHeapBroker* broker, MapRef map) {
589 // This operation is split into two phases (1. map collection, 2. map
590 // processing). This is to avoid having to take two locks
591 // (full_transition_array_access and map_updater_access) at once and thus
592 // having to deal with related deadlock issues.
593 ZoneVector<IndirectHandle<Map>> maps(broker->zone());
594 maps.push_back(map.object());
595
596 {
597 DisallowGarbageCollection no_gc;
598
599 // Has to be an initial map.
600 DCHECK(IsUndefined(map.object()->GetBackPointer(), broker->isolate()));
601
602 static constexpr bool kConcurrentAccess = true;
603 TransitionsAccessor(broker->isolate(), *map.object(), kConcurrentAccess)
604 .TraverseTransitionTree([&](Tagged<Map> m) {
605 maps.push_back(broker->CanonicalPersistentHandle(m));
606 });
607 }
608
609 // The lock is needed for UnusedPropertyFields and InstanceSizeFromSlack.
610 JSHeapBroker::MapUpdaterGuardIfNeeded mumd_scope(broker);
611
612 int slack = std::numeric_limits<int>::max();
613 for (DirectHandle<Map> m : maps) {
614 slack = std::min(slack, m->UnusedPropertyFields());
615 }
616
617 return map.object()->InstanceSizeFromSlack(slack);
618}

622// IMPORTANT: Keep this sync'd with JSFunctionData::IsConsistentWithHeapState.
623void JSFunctionData::Cache(JSHeapBroker* broker) {
624 DCHECK(!serialized_);
625
626 TraceScope tracer(broker, this, "JSFunctionData::Cache");
627 DirectHandle<JSFunction> function = Cast<JSFunction>(object());
628
629 // This function may run on the background thread and thus must be individual
630 // fields in a thread-safe manner. Consistency between fields is *not*
631 // guaranteed here, instead we verify it in `IsConsistentWithHeapState`,
632 // called during job finalization. Relaxed loads are thus okay: we're
633 // guaranteed to see an initialized JSFunction object, and after
634 // initialization fields remain in a valid state.
635
636 ContextRef context =
637 MakeRefAssumeMemoryFence(broker, function->context(kRelaxedLoad));
638 context_ = context.data();
639
640 SharedFunctionInfoRef shared =
641 MakeRefAssumeMemoryFence(broker, function->shared(kRelaxedLoad));
642 shared_ = shared.data();
643
644 if (function->has_prototype_slot()) {
645 prototype_or_initial_map_ = broker->GetOrCreateData(
646 function->prototype_or_initial_map(kAcquireLoad), kAssumeMemoryFence);
647
648 has_initial_map_ = prototype_or_initial_map_->IsMap();
649 if (has_initial_map_) {
650 // MapData is not used for initial_map_ because some
651 // AlwaysSharedSpaceJSObject subclass constructors (e.g. SharedArray) have
652 // initial maps in RO space, which can be accessed directly.
653 initial_map_ = prototype_or_initial_map_;
654
655 MapRef initial_map_ref = TryMakeRef<Map>(broker, initial_map_).value();
656 if (initial_map_ref.IsInobjectSlackTrackingInProgress()) {
657 initial_map_instance_size_with_min_slack_ =
658 InstanceSizeWithMinSlack(broker, initial_map_ref);
659 } else {
660 initial_map_instance_size_with_min_slack_ =
661 initial_map_ref.instance_size();
662 }
663 CHECK_GT(initial_map_instance_size_with_min_slack_, 0);
664 }
665
666 if (has_initial_map_) {
667 has_instance_prototype_ = true;
668 instance_prototype_ =
669 MakeRefAssumeMemoryFence(
670 broker, Cast<Map>(initial_map_->object())->prototype())
671 .data();
672 } else if (prototype_or_initial_map_->IsHeapObject() &&
673 !IsTheHole(
674 *Cast<HeapObject>(prototype_or_initial_map_->object()))) {
675 has_instance_prototype_ = true;
676 instance_prototype_ = prototype_or_initial_map_;
677 }
678 }
679
680 PrototypeRequiresRuntimeLookup_ = function->PrototypeRequiresRuntimeLookup();

1059ObjectData* JSHeapBroker::TryGetOrCreateData(IndirectHandle<Object> object,
1060 GetOrCreateDataFlags flags) {
1061 RefsMap::Entry* entry = refs_->Lookup(object.address());
1062 if (entry != nullptr) return entry->value;
1063
1064 if (mode() == JSHeapBroker::kDisabled) {
1065 entry = refs_->LookupOrInsert(object.address());
1066 ObjectData** storage = &entry->value;
1067 if (*storage == nullptr) {
1068 entry->value = zone()->New<ObjectData>(
1069 this, storage, object,
1070 IsSmi(*object) ? kSmi : kUnserializedHeapObject);
1071 }
1072 return *storage;
1073 }
1074
1075 CHECK(mode() == JSHeapBroker::kSerializing ||
1076 mode() == JSHeapBroker::kSerialized);
1077
1078 ObjectData* object_data;
1079 if (IsSmi(*object)) {
1080 entry = refs_->LookupOrInsert(object.address());
1081 return zone()->New<ObjectData>(this, &entry->value, object, kSmi);
1082 }
1083
1084 DCHECK(!IsSmi(*object));
1085
1086 const bool crash_on_error = (flags & kCrashOnError) != 0;
1087
1088 if ((flags & kAssumeMemoryFence) == 0 &&
1089 ObjectMayBeUninitialized(Cast<HeapObject>(*object))) {
1090 TRACE_BROKER_MISSING(this, "Object may be uninitialized " << *object);
1091 CHECK_WITH_MSG(!crash_on_error, "Ref construction failed");
1092 return nullptr;
1093 }
1094
1095 if (ReadOnlyHeap::SandboxSafeContains(Cast<HeapObject>(*object))) {
1096 entry = refs_->LookupOrInsert(object.address());
1097 return zone()->New<ObjectData>(this, &entry->value, object,
1098 kUnserializedReadOnlyHeapObject);
1099 }
1100
1101 InstanceType instance_type =
1102 Cast<HeapObject>(*object)->map()->instance_type();
1103#define CREATE_DATA(Name) \
1104 if (i::InstanceTypeChecker::Is##Name(instance_type)) { \
1105 entry = refs_->LookupOrInsert(object.address()); \
1106 object_data = zone()->New<ref_traits<Name>::data_type>( \
1107 this, &entry->value, Cast<Name>(object), \
1108 ObjectDataKindFor(ref_traits<Name>::ref_serialization_kind)); \
1109 /* NOLINTNEXTLINE(readability/braces) */ \
1110 } else
1111 HEAP_BROKER_OBJECT_LIST(CREATE_DATA)
1112#undef CREATE_DATA
1113 {
1114 UNREACHABLE();
1115 }
1116
1117 // Ensure that the original instance type matches the one of the serialized
1118 // object (if the object was serialized). In particular, this is important
1119 // for Maps: in GetMapInstanceType we have special handling for maps and will
1120 // report MAP_TYPE for objects whose map pointer points back to itself. With
1121 // heap corruption, a non-map object can be made to point to itself though,
1122 // in which case we may later treat a non-MapData object as a MapData object.
1123 // See also crbug.com/326700497 for more details.
1124 if (!object_data->should_access_heap()) {
1125 SBXCHECK_EQ(
1126 instance_type,
1127 static_cast<HeapObjectData*>(object_data)->GetMapInstanceType());
1128 }
1129
1130 // At this point the entry pointer is not guaranteed to be valid as
1131 // the refs_ hash hable could be resized by one of the constructors above.
1132 DCHECK_EQ(object_data, refs_->Lookup(object.address())->value);
1133 return object_data;
1134}

1215bool MapRef::PrototypesElementsDoNotHaveAccessorsOrThrow(
1216 JSHeapBroker* broker, ZoneVector<MapRef>* prototype_maps) {
1217 DCHECK_NOT_NULL(prototype_maps);
1218 MapRef prototype_map = prototype(broker).map(broker);
1219 while (prototype_map.oddball_type(broker) != OddballType::kNull) {
1220 // For the purposes of depending on prototypes' elements behavior when
1221 // doing keyed property sets, non-extensible and sealed fast elements are
1222 // okay because they behave like fast elements for stores into holes on
1223 // the receiver. In such cases, the value is stored on the receiver's
1224 // elements and does not trigger any setters and does not throw.
1225 //
1226 // Note that frozen elements are _not_ okay because of the "override
1227 // mistake":
1228 //
1229 // Object.prototype[1] = "x";
1230 // Object.freeze(Object.prototype);
1231 // ([])[1] = "y"; <-- throws in strict mode, nop in sloppy mode
1232 if (!prototype_map.IsJSObjectMap() || !prototype_map.is_stable() ||
1233 !IsFastOrNonextensibleOrSealedElementsKind(
1234 prototype_map.elements_kind())) {
1235 return false;
1236 }
1237 prototype_maps->push_back(prototype_map);
1238 prototype_map = prototype_map.prototype(broker).map(broker);
1239 }
1240 return true;
1241}

1310 CHECK(index.is_inobject());
1311 Handle<Object> value;
1312 {
1313 DisallowGarbageCollection no_gc;
1314 PtrComprCageBase cage_base = broker->cage_base();
1315 Tagged<Map> current_map = object()->map(cage_base, kAcquireLoad);
1316
1317 // If the map changed in some prior GC epoch, our {index} could be
1318 // outside the valid bounds of the cached map.
1319 if (*map(broker).object() != current_map) {
1320 TRACE_BROKER_MISSING(broker, "Map change detected in " << *this);
1321 return {};
1322 }
1323
1324 std::optional<Tagged<Object>> maybe_value =
1325 object()->RawInobjectPropertyAt(cage_base, current_map, index);
1326 if (!maybe_value.has_value()) {
1327 TRACE_BROKER_MISSING(broker,
1328 "Unable to safely read property in " << *this);
1329 return {};
1330 }
1331 value = broker->CanonicalPersistentHandle(maybe_value.value());
1332 }
1333 return TryMakeRef(broker, value);
1334}

2034std::optional<Tagged<Object>> JSObjectRef::GetOwnConstantElementFromHeap(
2035 JSHeapBroker* broker, Tagged<FixedArrayBase> elements,
2036 ElementsKind elements_kind, uint32_t index) const {
2037 DCHECK_LE(index, JSObject::kMaxElementIndex);
2038
2039 DirectHandle<JSObject> holder = object();
2040
2041 // This block is carefully constructed to avoid Ref creation and access since
2042 // this method may be called after the broker has retired.
2043 // The relaxed `length` read is safe to use in this case since:
2044 // - TryGetOwnConstantElement (below) only detects a constant for JSArray
2045 // holders if the array is frozen.
2046 // - Frozen arrays can't change length.
2047 // - We've already seen the corresponding map (when this JSObjectRef was
2048 // created);
2049 // - The release-load of that map ensures we read the newest value
2050 // of `length` below.
2051 if (i::IsJSArray(*holder)) {
2052 Tagged<Object> array_length_obj =
2053 Cast<JSArray>(*holder)->length(broker->isolate(), kRelaxedLoad);
2054 if (!i::IsSmi(array_length_obj)) {
2055 // Can't safely read into HeapNumber objects without atomic semantics
2056 // (relaxed would be sufficient due to the guarantees above).
2057 return {};
2058 }
2059 uint32_t array_length;
2060 if (!Object::ToArrayLength(array_length_obj, &array_length)) {
2061 return {};
2062 }
2063 // See also ElementsAccessorBase::GetMaxIndex.
2064 if (index >= array_length) return {};
2065 }
2066
2067 Tagged<Object> maybe_element;
2068 auto result = ConcurrentLookupIterator::TryGetOwnConstantElement(
2069 &maybe_element, broker->isolate(), broker->local_isolate(), *holder,
2070 elements, elements_kind, index);
2071
2072 if (result == ConcurrentLookupIterator::kGaveUp) {
2073 TRACE_BROKER_MISSING(broker, "JSObject::GetOwnConstantElement on "
2074 << *this << " at index " << index);
2075 return {};
2076 } else if (result == ConcurrentLookupIterator::kNotPresent) {
2077 return {};
2078 }
2079
2080 DCHECK_EQ(result, ConcurrentLookupIterator::kPresent);
2081 return maybe_element;
2082}
