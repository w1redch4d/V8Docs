22static constexpr size_t kDefaultDeadlineCheckInterval = 750u;

24template <size_t kDeadlineCheckInterval = kDefaultDeadlineCheckInterval,
25 typename WorklistLocal, typename Callback>
26bool DrainWorklistWithYielding(
27 JobDelegate* job_delegate, ConcurrentMarkingState& marking_state,
28 heap::base::IncrementalMarkingSchedule& incremental_marking_schedule,
29 WorklistLocal& worklist_local, Callback callback) {
30 return DrainWorklistWithPredicate<kDeadlineCheckInterval>(
31 [&incremental_marking_schedule, &marking_state, job_delegate]() {
32 incremental_marking_schedule.AddConcurrentlyMarkedBytes(
33 marking_state.RecentlyMarkedBytes());
34 return job_delegate->ShouldYield();
35 },
36 worklist_local, callback);
37}

39size_t WorkSizeForConcurrentMarking(MarkingWorklists& marking_worklists) {
40 return marking_worklists.marking_worklist()->Size() +
41 marking_worklists.write_barrier_worklist()->Size() +
42 marking_worklists.previously_not_fully_constructed_worklist()->Size();
43}
44
45// Checks whether worklists' global pools hold any segment a concurrent marker
46// can steal. This is called before the concurrent marker holds any Locals, so
47// no need to check local segments.
48bool HasWorkForConcurrentMarking(MarkingWorklists& marking_worklists) {
49 return !marking_worklists.marking_worklist()->IsEmpty() ||
50 !marking_worklists.write_barrier_worklist()->IsEmpty() ||
51 !marking_worklists.previously_not_fully_constructed_worklist()
52 ->IsEmpty();
53}

73void ConcurrentMarkingTask::Run(JobDelegate* job_delegate) {
74 StatsCollector::EnabledConcurrentScope stats_scope(
75 concurrent_marker_.heap().stats_collector(),
76 StatsCollector::kConcurrentMark);
77 if (!HasWorkForConcurrentMarking(concurrent_marker_.marking_worklists()))
78 return;
79 ConcurrentMarkingState concurrent_marking_state(
80 concurrent_marker_.heap(), concurrent_marker_.marking_worklists(),
81 concurrent_marker_.heap().compactor().compaction_worklists());
82 std::unique_ptr<Visitor> concurrent_marking_visitor =
83 concurrent_marker_.CreateConcurrentMarkingVisitor(
84 concurrent_marking_state);
85 ProcessWorklists(job_delegate, concurrent_marking_state,
86 *concurrent_marking_visitor);
87 concurrent_marker_.incremental_marking_schedule().AddConcurrentlyMarkedBytes(
88 concurrent_marking_state.RecentlyMarkedBytes());
89 concurrent_marking_state.Publish();
90}

98void ConcurrentMarkingTask::ProcessWorklists(
99 JobDelegate* job_delegate, ConcurrentMarkingState& concurrent_marking_state,
100 Visitor& concurrent_marking_visitor) {
101 do {
102 if (!DrainWorklistWithYielding(
103 job_delegate, concurrent_marking_state,
104 concurrent_marker_.incremental_marking_schedule(),
105 concurrent_marking_state
106 .previously_not_fully_constructed_worklist(),
107 [&concurrent_marking_state,
108 &concurrent_marking_visitor](HeapObjectHeader* header) {
109 BasePage::FromPayload(header)->SynchronizedLoad();
110 concurrent_marking_state.AccountMarkedBytes(*header);
111 DynamicallyTraceMarkedObject<AccessMode::kAtomic>(
112 concurrent_marking_visitor, *header);
113 })) {
114 return;
115 }
116
117 if (!DrainWorklistWithYielding(
118 job_delegate, concurrent_marking_state,
119 concurrent_marker_.incremental_marking_schedule(),
120 concurrent_marking_state.marking_worklist(),
121 [&concurrent_marking_state, &concurrent_marking_visitor](
122 const MarkingWorklists::MarkingItem& item) {
123 BasePage::FromPayload(item.base_object_payload)
124 ->SynchronizedLoad();
125 const HeapObjectHeader& header =
126 HeapObjectHeader::FromObject(item.base_object_payload);
127 DCHECK(!header.IsInConstruction<AccessMode::kAtomic>());
128 DCHECK(header.IsMarked<AccessMode::kAtomic>());
129 concurrent_marking_state.AccountMarkedBytes(header);
130 item.callback(&concurrent_marking_visitor,
131 item.base_object_payload);
132 })) {
133 return;
134 }
135
136 if (!DrainWorklistWithYielding(
137 job_delegate, concurrent_marking_state,
138 concurrent_marker_.incremental_marking_schedule(),
139 concurrent_marking_state.write_barrier_worklist(),
140 [&concurrent_marking_state,
141 &concurrent_marking_visitor](HeapObjectHeader* header) {
142 BasePage::FromPayload(header)->SynchronizedLoad();
143 concurrent_marking_state.AccountMarkedBytes(*header);
144 DynamicallyTraceMarkedObject<AccessMode::kAtomic>(
145 concurrent_marking_visitor, *header);
146 })) {
147 return;
148 }
149
150 {
151 StatsCollector::DisabledConcurrentScope stats_scope(
152 concurrent_marker_.heap().stats_collector(),
153 StatsCollector::kConcurrentMarkProcessEphemerons);
154 if (!DrainWorklistWithYielding(
155 job_delegate, concurrent_marking_state,
156 concurrent_marker_.incremental_marking_schedule(),
157 concurrent_marking_state
158 .ephemeron_pairs_for_processing_worklist(),
159 [&concurrent_marking_state, &concurrent_marking_visitor](
160 const MarkingWorklists::EphemeronPairItem& item) {
161 concurrent_marking_state.ProcessEphemeron(
162 item.key, item.value, item.value_desc,
163 concurrent_marking_visitor);
164 })) {
165 return;
166 }
167 }
168 } while (
169 !concurrent_marking_state.marking_worklist().IsLocalAndGlobalEmpty());
170}

232void ConcurrentMarkerBase::IncreaseMarkingPriorityIfNeeded() {
233 if (!concurrent_marking_handle_->UpdatePriorityEnabled()) return;
234 if (concurrent_marking_priority_increased_) return;
235 // If concurrent tasks aren't executed, it might delay GC finalization.
236 // As long as GC is active so is the write barrier, which incurs a
237 // performance cost. Marking is estimated to take overall
238 // |MarkingSchedulingOracle::kEstimatedMarkingTimeMs|. If
239 // concurrent marking tasks have not reported any progress (i.e. the
240 // concurrently marked bytes count as not changed) in over
241 // |kMarkingScheduleRatioBeforeConcurrentPriorityIncrease| of
242 // that expected duration, we increase the concurrent task priority
243 // for the duration of the current GC. This is meant to prevent the
244 // GC from exceeding it's expected end time.
245 size_t current_concurrently_marked_bytes_ =
246 incremental_marking_schedule_.GetConcurrentlyMarkedBytes();
247 if (current_concurrently_marked_bytes_ > last_concurrently_marked_bytes_) {
248 last_concurrently_marked_bytes_ = current_concurrently_marked_bytes_;
249 last_concurrently_marked_bytes_update_ = v8::base::TimeTicks::Now();
250 } else if ((v8::base::TimeTicks::Now() -
251 last_concurrently_marked_bytes_update_)
252 .InMilliseconds() >
253 kMarkingScheduleRatioBeforeConcurrentPriorityIncrease *
254 heap::base::IncrementalMarkingSchedule::kEstimatedMarkingTime
255 .InMillisecondsF()) {
256 concurrent_marking_handle_->UpdatePriority(
257 cppgc::TaskPriority::kUserBlocking);
258 concurrent_marking_priority_increased_ = true;
259 }
260}
