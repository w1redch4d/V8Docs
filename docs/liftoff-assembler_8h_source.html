446 // Returns the register which holds the value of stack slot {index}. If the
447 // value is not stored in a register yet, a register is allocated for it. The
448 // register is then assigned to the stack slot. The value stack height is not
449 // modified. The top of the stack is index 0, i.e. {PopToRegister()} and
450 // {PeekToRegister(0)} should result in the same register.
451 // When the value is finally popped, the use counter of its register has to be
452 // decremented. This can be done by popping the value with {DropValues}.
453 LiftoffRegister PeekToRegister(int index, LiftoffRegList pinned) {
454 DCHECK_LT(index, cache_state_.stack_state.size());
455 VarState& slot = cache_state_.stack_state.end()[-1 - index];
456 if (V8_LIKELY(slot.is_reg())) return slot.reg();
457 LiftoffRegister reg = LoadToRegister(slot, pinned);
458 cache_state_.inc_used(reg);
459 slot.MakeRegister(reg);
460 return reg;
461 }
