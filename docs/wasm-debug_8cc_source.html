43Address FindNewPC(WasmFrame* frame, WasmCode* wasm_code, int byte_offset,
44 ReturnLocation return_location) {
45 base::Vector<const uint8_t> new_pos_table = wasm_code->source_positions();
46
47 DCHECK_LE(0, byte_offset);
48
49 // Find the size of the call instruction by computing the distance from the
50 // source position entry to the return address.
51 WasmCode* old_code = frame->wasm_code();
52 int pc_offset = static_cast<int>(frame->pc() - old_code->instruction_start());
53 base::Vector<const uint8_t> old_pos_table = old_code->source_positions();
54 SourcePositionTableIterator old_it(old_pos_table);
55 int call_offset = -1;
56 while (!old_it.done() && old_it.code_offset() < pc_offset) {
57 call_offset = old_it.code_offset();
58 old_it.Advance();
59 }
60 DCHECK_LE(0, call_offset);
61 int call_instruction_size = pc_offset - call_offset;
62
63 // If {return_location == kAfterBreakpoint} we search for the first code
64 // offset which is marked as instruction (i.e. not the breakpoint).
65 // If {return_location == kAfterWasmCall} we return the last code offset
66 // associated with the byte offset.
67 SourcePositionTableIterator it(new_pos_table);
68 while (!it.done() && it.source_position().ScriptOffset() != byte_offset) {
69 it.Advance();
70 }
71 if (return_location == kAfterBreakpoint) {
72 while (!it.is_statement()) it.Advance();
73 DCHECK_EQ(byte_offset, it.source_position().ScriptOffset());
74 return wasm_code->instruction_start() + it.code_offset() +
75 call_instruction_size;
76 }
77
78 DCHECK_EQ(kAfterWasmCall, return_location);
79 int code_offset;
80 do {
81 code_offset = it.code_offset();
82 it.Advance();
83 } while (!it.done() && it.source_position().ScriptOffset() == byte_offset);
84 return wasm_code->instruction_start() + code_offset + call_instruction_size;
85}

194#if !V8_ENABLE_DRUMBRAKE

201#endif // !V8_ENABLE_DRUMBRAKE

203 if (static_cast<int>(wasm_frame->function_index()) != func_index) return 0;
204 return DeadBreakpoint(wasm_frame, breakpoints);

207 WasmCode* RecompileLiftoffWithBreakpoints(int func_index,
208 base::Vector<const int> offsets,
209 int dead_breakpoint) {
210 mutex_.AssertHeld(); // Mutex is held externally.
211
212 ForDebugging for_debugging = offsets.size() == 1 && offsets[0] == 0
213 ? kForStepping
214 : kWithBreakpoints;
215
216 // Check the cache first.
217 for (auto begin = cached_debugging_code_.begin(), it = begin,
218 end = cached_debugging_code_.end();
219 it != end; ++it) {
220 if (it->func_index == func_index &&
221 it->breakpoint_offsets.as_vector() == offsets &&
222 it->dead_breakpoint == dead_breakpoint) {
223 // Rotate the cache entry to the front (for LRU).
224 for (; it != begin; --it) std::iter_swap(it, it - 1);
225 if (for_debugging == kWithBreakpoints) {
226 // Re-install the code, in case it was replaced in the meantime.
227 native_module_->ReinstallDebugCode(it->code);
228 }
229 return it->code;
230 }
231 }
232
233 // Recompile the function with Liftoff, setting the new breakpoints.
234 // Not thread-safe. The caller is responsible for locking {mutex_}.
235 CompilationEnv env = CompilationEnv::ForModule(native_module_);
236 const WasmFunction* function = &env.module->functions[func_index];
237 base::Vector<const uint8_t> wire_bytes = native_module_->wire_bytes();
238 bool is_shared = env.module->type(function->sig_index).is_shared;
239 FunctionBody body{function->sig, function->code.offset(),
240 wire_bytes.begin() + function->code.offset(),
241 wire_bytes.begin() + function->code.end_offset(),
242 is_shared};
243 std::unique_ptr<DebugSideTable> debug_sidetable;
244
245 // Debug side tables for stepping are generated lazily.
246 bool generate_debug_sidetable = for_debugging == kWithBreakpoints;
247 // If lazy validation is on, we might need to lazily validate here.
248 if (V8_UNLIKELY(!env.module->function_was_validated(func_index))) {
249 WasmDetectedFeatures unused_detected_features;
250 Zone validation_zone(wasm::GetWasmEngine()->allocator(), ZONE_NAME);
251 DecodeResult validation_result =
252 ValidateFunctionBody(&validation_zone, env.enabled_features,
253 env.module, &unused_detected_features, body);
254 // Handling illegal modules here is tricky. As lazy validation is off by
255 // default anyway and this is for debugging only, we just crash for now.
256 CHECK_WITH_MSG(validation_result.ok(),
257 validation_result.error().message().c_str());
258 env.module->set_function_validated(func_index);
259 }
260 WasmCompilationResult result = ExecuteLiftoffCompilation(
261 &env, body,
262 LiftoffOptions{}
263 .set_func_index(func_index)
264 .set_for_debugging(for_debugging)
265 .set_breakpoints(offsets)
266 .set_dead_breakpoint(dead_breakpoint)
267 .set_debug_sidetable(generate_debug_sidetable ? &debug_sidetable
268 : nullptr));
269 // Liftoff compilation failure is a FATAL error. We rely on complete Liftoff
270 // support for debugging.
271 if (!result.succeeded()) FATAL("Liftoff compilation failed");
272 DCHECK_EQ(generate_debug_sidetable, debug_sidetable != nullptr);
273
274 DCHECK_NULL(result.assumptions);
275 WasmCode* new_code =
276 native_module_->PublishCode(native_module_->AddCompiledCode(result));
277
278 DCHECK(new_code->is_inspectable());
279 if (generate_debug_sidetable) {
280 base::SpinningMutexGuard lock(&debug_side_tables_mutex_);
281 DCHECK_EQ(0, debug_side_tables_.count(new_code));
282 debug_side_tables_.emplace(new_code, std::move(debug_sidetable));
283 }
284
285 // Insert new code into the cache. Insert before existing elements for LRU.
286 cached_debugging_code_.insert(
287 cached_debugging_code_.begin(),
288 CachedDebuggingCode{func_index, base::OwnedCopyOf(offsets),
289 dead_breakpoint, new_code});
290 // Increase the ref count (for the cache entry).
291 new_code->IncRef();
292 // Remove exceeding element.
293 if (cached_debugging_code_.size() > kMaxCachedDebuggingCode) {
294 // Put the code in the surrounding CodeRefScope to delay deletion until
295 // after the mutex is released.
296 WasmCodeRefScope::AddRef(cached_debugging_code_.back().code);
297 cached_debugging_code_.back().code->DecRefOnLiveCode();
298 cached_debugging_code_.pop_back();
299 }
300 DCHECK_GE(kMaxCachedDebuggingCode, cached_debugging_code_.size());
301
302 return new_code;
303 }

305 void SetBreakpoint(int func_index, int offset, Isolate* isolate) {
306 // TODO(paolosev@microsoft.com) - Add support for breakpoints in Wasm
307 // interpreter.
308 if (v8_flags.wasm_jitless) return;
309
310 // Put the code ref scope outside of the mutex, so we don't unnecessarily
311 // hold the mutex while freeing code.
312 WasmCodeRefScope wasm_code_ref_scope;
313
314 // Hold the mutex while modifying breakpoints, to ensure consistency when
315 // multiple isolates set/remove breakpoints at the same time.
316 base::SpinningMutexGuard guard(&mutex_);
317
318 // offset == 0 indicates flooding and should not happen here.
319 DCHECK_NE(0, offset);
320
321 // Get the set of previously set breakpoints, to check later whether a new
322 // breakpoint was actually added.
323 std::vector<int> all_breakpoints = FindAllBreakpoints(func_index);
324
325 auto& isolate_data = per_isolate_data_[isolate];
326 std::vector<int>& breakpoints =
327 isolate_data.breakpoints_per_function[func_index];
328 auto insertion_point =
329 std::lower_bound(breakpoints.begin(), breakpoints.end(), offset);
330 if (insertion_point != breakpoints.end() && *insertion_point == offset) {
331 // The breakpoint is already set for this isolate.
332 return;
333 }
334 breakpoints.insert(insertion_point, offset);
335
336 DCHECK(std::is_sorted(all_breakpoints.begin(), all_breakpoints.end()));
337 // Find the insertion position within {all_breakpoints}.
338 insertion_point = std::lower_bound(all_breakpoints.begin(),
339 all_breakpoints.end(), offset);
340 bool breakpoint_exists =
341 insertion_point != all_breakpoints.end() && *insertion_point == offset;
342 // If the breakpoint was already set before, then we can just reuse the old
343 // code. Otherwise, recompile it. In any case, rewrite this isolate's stack
344 // to make sure that it uses up-to-date code containing the breakpoint.
345 WasmCode* new_code;
346 if (breakpoint_exists) {
347 new_code = native_module_->GetCode(func_index);
348 } else {
349 all_breakpoints.insert(insertion_point, offset);
350 int dead_breakpoint =
351 DeadBreakpoint(func_index, base::VectorOf(all_breakpoints), isolate);
352 new_code = RecompileLiftoffWithBreakpoints(
353 func_index, base::VectorOf(all_breakpoints), dead_breakpoint);
354 }
355 UpdateReturnAddresses(isolate, new_code, isolate_data.stepping_frame);
356 }

358 std::vector<int> FindAllBreakpoints(int func_index) {
359 mutex_.AssertHeld(); // Mutex must be held externally.
360 std::set<int> breakpoints;
361 for (auto& data : per_isolate_data_) {
362 auto it = data.second.breakpoints_per_function.find(func_index);
363 if (it == data.second.breakpoints_per_function.end()) continue;
364 for (int offset : it->second) breakpoints.insert(offset);
365 }
366 return {breakpoints.begin(), breakpoints.end()};
367 }

446 void RemoveBreakpoint(int func_index, int position, Isolate* isolate) {
447 // Put the code ref scope outside of the mutex, so we don't unnecessarily
448 // hold the mutex while freeing code.
449 WasmCodeRefScope wasm_code_ref_scope;
450
451 // Hold the mutex while modifying breakpoints, to ensure consistency when
452 // multiple isolates set/remove breakpoints at the same time.
453 base::SpinningMutexGuard guard(&mutex_);
454
455 const auto& function = native_module_->module()->functions[func_index];
456 int offset = position - function.code.offset();
457
458 auto& isolate_data = per_isolate_data_[isolate];
459 std::vector<int>& breakpoints =
460 isolate_data.breakpoints_per_function[func_index];
461 DCHECK_LT(0, offset);
462 auto insertion_point =
463 std::lower_bound(breakpoints.begin(), breakpoints.end(), offset);
464 if (insertion_point == breakpoints.end()) return;
465 if (*insertion_point != offset) return;
466 breakpoints.erase(insertion_point);
467
468 std::vector<int> remaining = FindAllBreakpoints(func_index);
469 // If the breakpoint is still set in another isolate, don't remove it.
470 DCHECK(std::is_sorted(remaining.begin(), remaining.end()));
471 if (std::binary_search(remaining.begin(), remaining.end(), offset)) return;
472 int dead_breakpoint =
473 DeadBreakpoint(func_index, base::VectorOf(remaining), isolate);
474 UpdateBreakpoints(func_index, base::VectorOf(remaining), isolate,
475 isolate_data.stepping_frame, dead_breakpoint);
476 }

582 const DebugSideTable* GetDebugSideTable(WasmCode* code) {
583 DCHECK(code->is_inspectable());
584 {
585 // Only hold the mutex temporarily. We can't hold it while generating the
586 // debug side table, because compilation takes the {NativeModule} lock.
587 base::SpinningMutexGuard guard(&debug_side_tables_mutex_);
588 auto it = debug_side_tables_.find(code);
589 if (it != debug_side_tables_.end()) return it->second.get();
590 }
591
592 // Otherwise create the debug side table now.
593 std::unique_ptr<DebugSideTable> debug_side_table =
594 GenerateLiftoffDebugSideTable(code);
595 DebugSideTable* ret = debug_side_table.get();
596
597 // Check cache again, maybe another thread concurrently generated a debug
598 // side table already.
599 {
600 base::SpinningMutexGuard guard(&debug_side_tables_mutex_);
601 auto& slot = debug_side_tables_[code];
602 if (slot != nullptr) return slot.get();
603 slot = std::move(debug_side_table);
604 }
605
606 // Print the code together with the debug table, if requested.
607 code->MaybePrint();
608 return ret;
609 }

611 // Get the value of a local (including parameters) or stack value. Stack
612 // values follow the locals in the same index space.
613 WasmValue GetValue(const DebugSideTable* debug_side_table,
614 const DebugSideTable::Entry* debug_side_table_entry,
615 int index, Address stack_frame_base,
616 Address debug_break_fp, Isolate* isolate) const {
617 const DebugSideTable::Entry::Value* value =
618 debug_side_table->FindValue(debug_side_table_entry, index);
619 if (value->is_constant()) {
620 DCHECK(value->type == kWasmI32 || value->type == kWasmI64);
621 return value->type == kWasmI32 ? WasmValue(value->i32_const)
622 : WasmValue(int64_t{value->i32_const});
623 }
624
625 if (value->is_register()) {
626 auto reg = LiftoffRegister::from_liftoff_code(value->reg_code);
627 auto gp_addr = [debug_break_fp](Register reg) {
628 return debug_break_fp +
629 WasmDebugBreakFrameConstants::GetPushedGpRegisterOffset(
630 reg.code());
631 };
632 if (reg.is_gp_pair()) {
633 DCHECK_EQ(kWasmI64, value->type);
634 uint32_t low_word = ReadUnalignedValue<uint32_t>(gp_addr(reg.low_gp()));
635 uint32_t high_word =
636 ReadUnalignedValue<uint32_t>(gp_addr(reg.high_gp()));
637 return WasmValue((uint64_t{high_word} << 32) | low_word);
638 }
639 if (reg.is_gp()) {
640 if (value->type == kWasmI32) {
641 return WasmValue(ReadUnalignedValue<uint32_t>(gp_addr(reg.gp())));
642 } else if (value->type == kWasmI64) {
643 return WasmValue(ReadUnalignedValue<uint64_t>(gp_addr(reg.gp())));
644 } else if (value->type.is_reference()) {
645 Handle<Object> obj(
646 Tagged<Object>(ReadUnalignedValue<Address>(gp_addr(reg.gp()))),
647 isolate);
648 return WasmValue(obj, value->type, value->module);
649 } else {
650 UNREACHABLE();
651 }
652 }
653 DCHECK(reg.is_fp() || reg.is_fp_pair());
654 // ifdef here to workaround unreachable code for is_fp_pair.
655#ifdef V8_TARGET_ARCH_ARM
656 int code = reg.is_fp_pair() ? reg.low_fp().code() : reg.fp().code();
657#else
658 int code = reg.fp().code();
659#endif
660 Address spilled_addr =
661 debug_break_fp +
662 WasmDebugBreakFrameConstants::GetPushedFpRegisterOffset(code);
663 if (value->type == kWasmF32) {
664 return WasmValue(ReadUnalignedValue<float>(spilled_addr));
665 } else if (value->type == kWasmF64) {
666 return WasmValue(ReadUnalignedValue<double>(spilled_addr));
667 } else if (value->type == kWasmS128) {
668 return WasmValue(Simd128(ReadUnalignedValue<int8x16>(spilled_addr)));
669 } else {
670 // All other cases should have been handled above.
671 UNREACHABLE();
672 }
673 }
674
675 // Otherwise load the value from the stack.
676 Address stack_address = stack_frame_base - value->stack_offset;
677 switch (value->type.kind()) {
678 case kI32:
679 return WasmValue(ReadUnalignedValue<int32_t>(stack_address));
680 case kI64:
681 return WasmValue(ReadUnalignedValue<int64_t>(stack_address));
682 case kF32:
683 return WasmValue(ReadUnalignedValue<float>(stack_address));
684 case kF64:
685 return WasmValue(ReadUnalignedValue<double>(stack_address));
686 case kS128:
687 return WasmValue(Simd128(ReadUnalignedValue<int8x16>(stack_address)));
688 case kRef:
689 case kRefNull:
690 case kRtt: {
691 Handle<Object> obj(
692 Tagged<Object>(ReadUnalignedValue<Address>(stack_address)),
693 isolate);
694 return WasmValue(obj, value->type, value->module);
695 }
696 case kI8:
697 case kI16:
698 case kF16:
699 case kVoid:
700 case kTop:
701 case kBottom:
702 UNREACHABLE();
703 }
704 }

706 // After installing a Liftoff code object with a different set of breakpoints,
707 // update return addresses on the stack so that execution resumes in the new
708 // code. The frame layout itself should be independent of breakpoints.
709 void UpdateReturnAddresses(Isolate* isolate, WasmCode* new_code,
710 StackFrameId stepping_frame) {
711 // The first return location is after the breakpoint, others are after wasm
712 // calls.
713 ReturnLocation return_location = kAfterBreakpoint;
714 for (DebuggableStackFrameIterator it(isolate); !it.done();
715 it.Advance(), return_location = kAfterWasmCall) {
716 // We still need the flooded function for stepping.
717 if (it.frame()->id() == stepping_frame) continue;
718#if !V8_ENABLE_DRUMBRAKE
719 if (!it.is_wasm()) continue;
720#else // !V8_ENABLE_DRUMBRAKE
721 // TODO(paolosev@microsoft.com) - Implement for Wasm interpreter.
722 if (!it.is_wasm() || it.is_wasm_interpreter_entry()) continue;
723#endif // !V8_ENABLE_DRUMBRAKE
724 WasmFrame* frame = WasmFrame::cast(it.frame());
725 if (frame->native_module() != new_code->native_module()) continue;
726 if (frame->function_index() != new_code->index()) continue;
727 if (!frame->wasm_code()->is_liftoff()) continue;
728 UpdateReturnAddress(frame, new_code, return_location);
729 }
730 }

756 DisallowGarbageCollection no_gc;
757 int position = frame->position();

888// Return the next breakable position at or after {offset_in_func} in function
889// {func_index}, or 0 if there is none.
890// Note that 0 is never a breakable position in wasm, since the first uint8_t
891// contains the locals count for the function.
892int FindNextBreakablePosition(wasm::NativeModule* native_module, int func_index,
893 int offset_in_func) {
894 Zone zone{wasm::GetWasmEngine()->allocator(), ZONE_NAME};
895 wasm::BodyLocalDecls locals;
896 const uint8_t* module_start = native_module->wire_bytes().begin();
897 const wasm::WasmFunction& func =
898 native_module->module()->functions[func_index];
899 wasm::BytecodeIterator iterator(module_start + func.code.offset(),
900 module_start + func.code.end_offset(),
901 &locals, &zone);
902 DCHECK_LT(0, locals.encoded_size);
903 if (offset_in_func < 0) return 0;
904 for (; iterator.has_next(); iterator.next()) {
905 if (iterator.pc_offset() < static_cast<uint32_t>(offset_in_func)) continue;
906 if (!wasm::WasmOpcodes::IsBreakable(iterator.current())) continue;
907 return static_cast<int>(iterator.pc_offset());
908 }
909 return 0;
910}

1004int FindBreakpointInfoInsertPos(Isolate* isolate,
1005 DirectHandle<FixedArray> breakpoint_infos,
1006 int position) {
1007 // Find insert location via binary search, taking care of undefined values on
1008 // the right. {position} is either {kOnEntryBreakpointPosition} (which is -1),
1009 // or positive.
1010 DCHECK(position == WasmScript::kOnEntryBreakpointPosition || position > 0);
1011
1012 int left = 0; // inclusive
1013 int right = breakpoint_infos->length(); // exclusive
1014 while (right - left > 1) {
1015 int mid = left + (right - left) / 2;
1016 Tagged<Object> mid_obj = breakpoint_infos->get(mid);
1017 if (GetBreakpointPos(isolate, mid_obj) <= position) {
1018 left = mid;
1019 } else {
1020 right = mid;
1021 }
1022 }
1023
1024 int left_pos = GetBreakpointPos(isolate, breakpoint_infos->get(left));
1025 return left_pos < position ? left + 1 : left;
1026}

1030// static
1031bool WasmScript::ClearBreakPoint(DirectHandle<Script> script, int position,
1032 DirectHandle<BreakPoint> break_point) {
1033 if (!script->has_wasm_breakpoint_infos()) return false;
1034
1035 Isolate* isolate = script->GetIsolate();
1036 DirectHandle<FixedArray> breakpoint_infos(script->wasm_breakpoint_infos(),
1037 isolate);
1038
1039 int pos = FindBreakpointInfoInsertPos(isolate, breakpoint_infos, position);
1040
1041 // Does a BreakPointInfo object already exist for this position?
1042 if (pos == breakpoint_infos->length()) return false;
1043
1044 DirectHandle<BreakPointInfo> info(
1045 Cast<BreakPointInfo>(breakpoint_infos->get(pos)), isolate);
1046 BreakPointInfo::ClearBreakPoint(isolate, info, break_point);
1047
1048 // Check if there are no more breakpoints at this location.
1049 if (info->GetBreakPointCount(isolate) == 0) {
1050 // Update array by moving breakpoints up one position.
1051 for (int i = pos; i < breakpoint_infos->length() - 1; i++) {
1052 Tagged<Object> entry = breakpoint_infos->get(i + 1);
1053 breakpoint_infos->set(i, entry);
1054 if (IsUndefined(entry, isolate)) break;
1055 }
1056 // Make sure last array element is empty as a result.
1057 breakpoint_infos->set(breakpoint_infos->length() - 1,
1058 ReadOnlyRoots{isolate}.undefined_value(),
1059 SKIP_WRITE_BARRIER);
1060 }
1061
1062 if (break_point->id() == v8::internal::Debug::kInstrumentationId) {
1063 // Special handling for instrumentation breakpoints.
1064 SetBreakOnEntryFlag(*script, false);
1065 } else {
1066 // Remove the breakpoint from DebugInfo and recompile.
1067 wasm::NativeModule* native_module = script->wasm_native_module();
1068 const wasm::WasmModule* module = native_module->module();
1069 int func_index = GetContainingWasmFunction(module, position);
1070 native_module->GetDebugInfo()->RemoveBreakpoint(func_index, position,
1071 isolate);
1072 }
1073
1074 return true;
1075}

1114// static
1115void WasmScript::AddBreakpointToInfo(DirectHandle<Script> script, int position,
1116 DirectHandle<BreakPoint> break_point) {
1117 Isolate* isolate = script->GetIsolate();
1118 DirectHandle<FixedArray> breakpoint_infos;
1119 if (script->has_wasm_breakpoint_infos()) {
1120 breakpoint_infos = direct_handle(script->wasm_breakpoint_infos(), isolate);
1121 } else {
1122 breakpoint_infos =
1123 isolate->factory()->NewFixedArray(4, AllocationType::kOld);
1124 script->set_wasm_breakpoint_infos(*breakpoint_infos);
1125 }
1126
1127 int insert_pos =
1128 FindBreakpointInfoInsertPos(isolate, breakpoint_infos, position);
1129
1130 // If a BreakPointInfo object already exists for this position, add the new
1131 // breakpoint object and return.
1132 if (insert_pos < breakpoint_infos->length() &&
1133 GetBreakpointPos(isolate, breakpoint_infos->get(insert_pos)) ==
1134 position) {
1135 DirectHandle<BreakPointInfo> old_info(
1136 Cast<BreakPointInfo>(breakpoint_infos->get(insert_pos)), isolate);
1137 BreakPointInfo::SetBreakPoint(isolate, old_info, break_point);
1138 return;
1139 }
1140
1141 // Enlarge break positions array if necessary.
1142 bool need_realloc = !IsUndefined(
1143 breakpoint_infos->get(breakpoint_infos->length() - 1), isolate);
1144 DirectHandle<FixedArray> new_breakpoint_infos = breakpoint_infos;
1145 if (need_realloc) {
1146 new_breakpoint_infos = isolate->factory()->NewFixedArray(
1147 2 * breakpoint_infos->length(), AllocationType::kOld);
1148 script->set_wasm_breakpoint_infos(*new_breakpoint_infos);
1149 // Copy over the entries [0, insert_pos).
1150 for (int i = 0; i < insert_pos; ++i)
1151 new_breakpoint_infos->set(i, breakpoint_infos->get(i));
1152 }
1153
1154 // Move elements [insert_pos, ...] up by one.
1155 for (int i = breakpoint_infos->length() - 1; i >= insert_pos; --i) {
1156 Tagged<Object> entry = breakpoint_infos->get(i);
1157 if (IsUndefined(entry, isolate)) continue;
1158 new_breakpoint_infos->set(i + 1, entry);
1159 }
1160
1161 // Generate new BreakpointInfo.
1162 DirectHandle<BreakPointInfo> breakpoint_info =
1163 isolate->factory()->NewBreakPointInfo(position);
1164 BreakPointInfo::SetBreakPoint(isolate, breakpoint_info, break_point);
1165
1166 // Now insert new position at insert_pos.
1167 new_breakpoint_infos->set(insert_pos, *breakpoint_info);
1168}

1170// static
1171bool WasmScript::GetPossibleBreakpoints(
1172 wasm::NativeModule* native_module, const v8::debug::Location& start,
1173 const v8::debug::Location& end,
1174 std::vector<v8::debug::BreakLocation>* locations) {
1175 DisallowGarbageCollection no_gc;
1176
1177 const wasm::WasmModule* module = native_module->module();
1178 const std::vector<wasm::WasmFunction>& functions = module->functions;
1179
1180 if (start.GetLineNumber() != 0 || start.GetColumnNumber() < 0 ||
1181 (!end.IsEmpty() &&
1182 (end.GetLineNumber() != 0 || end.GetColumnNumber() < 0 ||
1183 end.GetColumnNumber() < start.GetColumnNumber())))
1184 return false;
1185
1186 // start_func_index, start_offset and end_func_index is inclusive.
1187 // end_offset is exclusive.
1188 // start_offset and end_offset are module-relative byte offsets.
1189 // We set strict to false because offsets may be between functions.
1190 int start_func_index =
1191 GetNearestWasmFunction(module, start.GetColumnNumber());
1192 if (start_func_index < 0) return false;
1193 uint32_t start_offset = start.GetColumnNumber();
1194 int end_func_index;
1195 uint32_t end_offset;
1196
1197 if (end.IsEmpty()) {

1232 locations->emplace_back(0, total_offset, debug::kCommonBreakLocation);
1233 }

1264 StackFrameId frame_id) {

1267 DirectHandle<FixedArray> breakpoint_infos(script->wasm_breakpoint_infos(),
1268 isolate);
1269 int insert_pos =
1270 FindBreakpointInfoInsertPos(isolate, breakpoint_infos, position);
1271 if (insert_pos >= breakpoint_infos->length()) return {};
1272
1273 DirectHandle<Object> maybe_breakpoint_info(breakpoint_infos->get(insert_pos),
1274 isolate);
1275 if (IsUndefined(*maybe_breakpoint_info, isolate)) return {};
1276 auto breakpoint_info = Cast<BreakPointInfo>(maybe_breakpoint_info);
1277 if (breakpoint_info->source_position() != position) return {};
1278
1279 DirectHandle<Object> break_points(breakpoint_info->break_points(), isolate);
1280 if (!IsFixedArray(*break_points)) {
1281 if (!CheckBreakPoint(isolate, Cast<BreakPoint>(break_points), frame_id)) {
1282 // A breakpoint that doesn't break mutes traps. (Rule enables the
1283 // "Never Pause Here" feature.)
1284 isolate->debug()->SetMutedWasmLocation(script, position);
1285 return {};
1286 }
1287 // If breakpoint does fire, clear any prior muting behavior.
1288 isolate->debug()->ClearMutedLocation();
1289 Handle<FixedArray> break_points_hit = isolate->factory()->NewFixedArray(1);
1290 break_points_hit->set(0, *break_points);
1291 return break_points_hit;
1292 }
1293
1294 auto array = Cast<FixedArray>(break_points);
1295 Handle<FixedArray> break_points_hit =
1296 isolate->factory()->NewFixedArray(array->length());
1297 int break_points_hit_count = 0;
1298 for (int i = 0; i < array->length(); ++i) {
1299 DirectHandle<BreakPoint> break_point(Cast<BreakPoint>(array->get(i)),
1300 isolate);
1301 if (CheckBreakPoint(isolate, break_point, frame_id)) {
1302 break_points_hit->set(break_points_hit_count++, *break_point);
1303 }
1304 }
1305 if (break_points_hit_count == 0) return {};
1306 break_points_hit->RightTrim(isolate, break_points_hit_count);
1307 return break_points_hit;
1308}
