78Address RelocInfo::target_address_address() {
79 DCHECK(HasTargetAddressAddress());
80 // Read the address of the word containing the target_address in an
81 // instruction stream.
82 // The only architecture-independent user of this function is the serializer.
83 // The serializer uses it to find out how many raw bytes of instruction to
84 // output before the next target.
85 // For an instruction like LUI/ORI where the target bits are mixed into the
86 // instruction bits, the size of the target will be zero, indicating that the
87 // serializer should not step forward in memory after a target is resolved
88 // and written. In this case the target_address_address function should
89 // return the end of the instructions to be patched, allowing the
90 // deserializer to deserialize the instructions as raw bytes and put them in
91 // place, ready to be patched with the target. After jump optimization,
92 // that is the address of the instruction that follows J/JAL/JR/JALR
93 // instruction.
94#ifdef V8_TARGET_ARCH_RISCV64
95 return pc_ + Assembler::kInstructionsFor64BitConstant * kInstrSize;
96#elif defined(V8_TARGET_ARCH_RISCV32)
97 return pc_ + Assembler::kInstructionsFor32BitConstant * kInstrSize;
98#endif
99}

165Handle<HeapObject> Assembler::embedded_object_handle_at(Address pc) {
166 DEBUG_PRINTF("\tembedded_object_handle_at: pc: 0x%" PRIxPTR " \n", pc);
167 disasm::NameConverter converter;
168 disasm::Disassembler disasm(converter);
169 base::EmbeddedVector<char, 128> disasm_buffer;
170
171 disasm.InstructionDecode(disasm_buffer, reinterpret_cast<uint8_t*>(pc));
172 DEBUG_PRINTF("%s\n", disasm_buffer.begin());
173 disasm.InstructionDecode(disasm_buffer,
174 reinterpret_cast<uint8_t*>(pc + kInstrSize));
175 DEBUG_PRINTF("%s\n", disasm_buffer.begin());
176#if V8_TARGET_ARCH_RISCV64
177 Instr instr1 = Assembler::instr_at(pc);
178 Instr instr2 = Assembler::instr_at(pc + kInstrSize);
179 DCHECK(IsAuipc(instr1));
180 DCHECK(IsLd(instr2));
181 int32_t embedded_target_offset = BrachlongOffset(instr1, instr2);
182 DEBUG_PRINTF("\tembedded_target_offset %d\n", embedded_target_offset);
183 static_assert(sizeof(EmbeddedObjectIndex) == sizeof(intptr_t));
184 DEBUG_PRINTF("\t EmbeddedObjectIndex %lu\n",
185 Memory<EmbeddedObjectIndex>(pc + embedded_target_offset));
186 return GetEmbeddedObject(
187 Memory<EmbeddedObjectIndex>(pc + embedded_target_offset));
188#else
189 DCHECK(IsLui(Assembler::instr_at(pc)));
190 DCHECK(IsAddi(Assembler::instr_at(pc + kInstrSize)));
191 auto target = target_address_at(pc, kNullAddress);
192 DEBUG_PRINTF("\ttarget %d\n", target);
193 return Handle<HeapObject>(reinterpret_cast<Address*>(target));
194#endif
195}

263void WritableRelocInfo::set_target_object(Tagged<HeapObject> target,
264 ICacheFlushMode icache_flush_mode) {
265 DCHECK(IsCodeTarget(rmode_) || IsEmbeddedObjectMode(rmode_));
266 if (IsCompressedEmbeddedObject(rmode_)) {
267 DCHECK(COMPRESS_POINTERS_BOOL);
268 // We must not compress pointers to objects outside of the main pointer
269 // compression cage as we wouldn't be able to decompress them with the
270 // correct cage base.
271 DCHECK_IMPLIES(V8_ENABLE_SANDBOX_BOOL, !HeapLayout::InTrustedSpace(target));
272 DCHECK_IMPLIES(V8_EXTERNAL_CODE_SPACE_BOOL,
273 !HeapLayout::InCodeSpace(target));
274 Assembler::set_target_compressed_address_at(
275 pc_, constant_pool_,
276 V8HeapCompressionScheme::CompressObject(target.ptr()), &jit_allocation_,
277 icache_flush_mode);
278 } else {
279 DCHECK(IsFullEmbeddedObject(rmode_));
280 Assembler::set_target_address_at(pc_, constant_pool_, target.ptr(),
281 &jit_allocation_, icache_flush_mode);
282 }

351int32_t Assembler::target_constant32_at(Address pc) {
352 Instruction* instr0 = Instruction::At((unsigned char*)pc);
353 Instruction* instr1 = Instruction::At((unsigned char*)(pc + 1 * kInstrSize));
354
355 // Interpret instructions for address generated by li: See listing in
356 // Assembler::set_target_address_at() just below.
357 if (IsLui(*reinterpret_cast<Instr*>(instr0)) &&
358 IsAddi(*reinterpret_cast<Instr*>(instr1))) {
359 // Assemble the 32bit value.
360 int32_t constant32 = (int32_t)(instr0->Imm20UValue() << kImm20Shift) +
361 (int32_t)instr1->Imm12Value();
362 return constant32;
363 }
364 // We should never get here, force a bad address if we do.
365 UNREACHABLE();
366}

368void Assembler::set_target_constant32_at(Address pc, uint32_t target,
369 WritableJitAllocation* jit_allocation,
370 ICacheFlushMode icache_flush_mode) {
371 Instruction* instr0 = Instruction::At((unsigned char*)pc);
372 Instruction* instr1 = Instruction::At((unsigned char*)(pc + 1 * kInstrSize));
373#ifdef DEBUG
374 // Check we have the result from a li macro-instruction.
375 DCHECK(IsLui(*reinterpret_cast<Instr*>(instr0)) &&
376 IsAddi(*reinterpret_cast<Instr*>(instr1)));
377#endif
378 int32_t high_20 = ((target + 0x800) >> 12); // 20 bits
379 int32_t low_12 = target & 0xfff; // 12 bits
380 instr_at_put(pc, SetHi20Offset(high_20, instr0->InstructionBits()),
381 jit_allocation);
382 instr_at_put(pc + 1 * kInstrSize,
383 SetLo12Offset(low_12, instr1->InstructionBits()),
384 jit_allocation);
385 if (icache_flush_mode != SKIP_ICACHE_FLUSH) {
386 FlushInstructionCache(pc, 2 * kInstrSize);
387 }
388 DCHECK_EQ(static_cast<uint32_t>(target_constant32_at(pc)), target);
389}

409[[nodiscard]] static inline Instr SetHi20Offset(int32_t hi20, Instr instr) {
410 DCHECK(Assembler::IsAuipc(instr) | Assembler::IsLui(instr));
411 DCHECK(is_int20(hi20));
412 instr = (instr & ~kImm31_12Mask) | ((hi20 & kImm19_0Mask) << 12);
413 return instr;
414}
415
416[[nodiscard]] static inline Instr SetLo12Offset(int32_t lo12, Instr instr) {
417 DCHECK(Assembler::IsJalr(instr) | Assembler::IsAddi(instr));
418 DCHECK(is_int12(lo12));
419 instr &= ~kImm12Mask;
420 int32_t imm12 = lo12 << kImm12Shift;
421 DCHECK(Assembler::IsJalr(instr | (imm12 & kImm12Mask)) ||
422 Assembler::IsAddi(instr | (imm12 & kImm12Mask)));
423 return instr | (imm12 & kImm12Mask);
424}
