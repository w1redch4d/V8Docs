32// Sweeping for full GC may be interleaved with sweeping for minor
33// gc. The below scopes should use TRACE_GC_EPOCH to associate them
34// with the right cycle.
35#define TRACE_GC(tracer, scope_id) \
36 DCHECK_NE(GCTracer::Scope::MC_SWEEP, scope_id); \
37 DCHECK_NE(GCTracer::Scope::MC_BACKGROUND_SWEEPING, scope_id); \
38 GCTracer::Scope UNIQUE_IDENTIFIER(gc_tracer_scope)( \
39 tracer, GCTracer::Scope::ScopeId(scope_id), ThreadKind::kMain); \
40 TRACE_EVENT0(TRACE_GC_CATEGORIES, \
41 GCTracer::Scope::Name(GCTracer::Scope::ScopeId(scope_id)))
42
43#define TRACE_GC_ARG1(tracer, scope_id, arg0_name, arg0_value) \
44 DCHECK_NE(GCTracer::Scope::MC_SWEEP, scope_id); \
45 DCHECK_NE(GCTracer::Scope::MC_BACKGROUND_SWEEPING, scope_id); \
46 GCTracer::Scope UNIQUE_IDENTIFIER(gc_tracer_scope)( \
47 tracer, GCTracer::Scope::ScopeId(scope_id), ThreadKind::kMain); \
48 TRACE_EVENT1(TRACE_GC_CATEGORIES, \
49 GCTracer::Scope::Name(GCTracer::Scope::ScopeId(scope_id)), \
50 arg0_name, arg0_value)
51
52#define TRACE_GC_WITH_FLOW(tracer, scope_id, bind_id, flow_flags) \
53 DCHECK_NE(GCTracer::Scope::MC_SWEEP, scope_id); \
54 DCHECK_NE(GCTracer::Scope::MC_BACKGROUND_SWEEPING, scope_id); \
55 GCTracer::Scope UNIQUE_IDENTIFIER(gc_tracer_scope)( \
56 tracer, GCTracer::Scope::ScopeId(scope_id), ThreadKind::kMain); \
57 TRACE_EVENT_WITH_FLOW0( \
58 TRACE_GC_CATEGORIES, \
59 GCTracer::Scope::Name(GCTracer::Scope::ScopeId(scope_id)), bind_id, \
60 flow_flags)
61
62#define TRACE_GC1(tracer, scope_id, thread_kind) \
63 GCTracer::Scope UNIQUE_IDENTIFIER(gc_tracer_scope)( \
64 tracer, GCTracer::Scope::ScopeId(scope_id), thread_kind); \
65 TRACE_EVENT0(TRACE_GC_CATEGORIES, \
66 GCTracer::Scope::Name(GCTracer::Scope::ScopeId(scope_id)))
67
68#define TRACE_GC1_WITH_FLOW(tracer, scope_id, thread_kind, bind_id, \
69 flow_flags) \
70 GCTracer::Scope UNIQUE_IDENTIFIER(gc_tracer_scope)( \
71 tracer, GCTracer::Scope::ScopeId(scope_id), thread_kind); \
72 TRACE_EVENT_WITH_FLOW0( \
73 TRACE_GC_CATEGORIES, \
74 GCTracer::Scope::Name(GCTracer::Scope::ScopeId(scope_id)), bind_id, \
75 flow_flags)
76
77#define TRACE_GC_EPOCH(tracer, scope_id, thread_kind) \
78 GCTracer::Scope UNIQUE_IDENTIFIER(gc_tracer_scope)( \
79 tracer, GCTracer::Scope::ScopeId(scope_id), thread_kind); \
80 TRACE_EVENT1(TRACE_GC_CATEGORIES, \
81 GCTracer::Scope::Name(GCTracer::Scope::ScopeId(scope_id)), \
82 "epoch", tracer->CurrentEpoch(scope_id))
83
84#define TRACE_GC_EPOCH_WITH_FLOW(tracer, scope_id, thread_kind, bind_id, \
85 flow_flags) \
86 GCTracer::Scope UNIQUE_IDENTIFIER(gc_tracer_scope)( \
87 tracer, GCTracer::Scope::ScopeId(scope_id), thread_kind); \
88 TRACE_EVENT_WITH_FLOW1( \
89 TRACE_GC_CATEGORIES, \
90 GCTracer::Scope::Name(GCTracer::Scope::ScopeId(scope_id)), bind_id, \
91 flow_flags, "epoch", tracer->CurrentEpoch(scope_id))

158 class Event {
159 public:
160 enum class Type {
161 SCAVENGER = 0,
162 MARK_COMPACTOR = 1,
163 INCREMENTAL_MARK_COMPACTOR = 2,
164 MINOR_MARK_SWEEPER = 3,
165 INCREMENTAL_MINOR_MARK_SWEEPER = 4,
166 START = 5,
167 };
168
169 // Returns true if the event corresponds to a young generation GC.
170 V8_INLINE static constexpr bool IsYoungGenerationEvent(Type type);
171
172 // The state diagram for a GC cycle:
173 // (NOT_RUNNING) -----(StartCycle)----->
174 // MARKING --(StartAtomicPause)-->
175 // ATOMIC ---(StopAtomicPause)-->
176 // SWEEPING ------(StopCycle)-----> NOT_RUNNING
177 enum class State { NOT_RUNNING, MARKING, ATOMIC, SWEEPING };
178
179 Event(Type type, State state, GarbageCollectionReason gc_reason,
180 const char* collector_reason, Priority priority);
181
182 // Type of the event.
183 Type type;
184
185 // State of the cycle corresponding to the event.
186 State state;
187
188 GarbageCollectionReason gc_reason;
189 const char* collector_reason;
190
191 // The Isolate's priority during the current GC cycle. The priority is set
192 // when the cycle starts. If the priority changes before the cycle is
193 // finished, the priority will be reset to denote a mixed priority.
194 std::optional<Priority> priority;
195
196 // Timestamp set in the constructor.
197 base::TimeTicks start_time;
198
199 // Timestamp set in the destructor.
200 base::TimeTicks end_time;
201
202 // Memory reduction flag set.
203 bool reduce_memory = false;
204
205 // Size of objects in heap set in constructor.
206 size_t start_object_size = 0;
207
208 // Size of objects in heap set in destructor.
209 size_t end_object_size = 0;
210
211 // Size of memory allocated from OS set in constructor.
212 size_t start_memory_size = 0;
213
214 // Size of memory allocated from OS set in destructor.
215 size_t end_memory_size = 0;
216
217 // Total amount of space either wasted or contained in one of free lists
218 // before the current GC.
219 size_t start_holes_size = 0;
220
221 // Total amount of space either wasted or contained in one of free lists
222 // after the current GC.
223 size_t end_holes_size = 0;
224
225 // Size of young objects in constructor.
226 size_t young_object_size = 0;
227
228 // Size of survived young objects in destructor.
229 size_t survived_young_object_size = 0;
230
231 // Bytes marked incrementally for INCREMENTAL_MARK_COMPACTOR
232 size_t incremental_marking_bytes = 0;
233
234 // Approximate number of threads that contributed in garbage collection.
235 size_t concurrency_estimate = 1;
236
237 // Duration (in ms) of incremental marking steps for
238 // INCREMENTAL_MARK_COMPACTOR.
239 base::TimeDelta incremental_marking_duration;
240
241 base::TimeTicks incremental_marking_start_time;
242
243 // Start/end of atomic/safepoint pause.
244 base::TimeTicks start_atomic_pause_time;
245 base::TimeTicks end_atomic_pause_time;
246
247 // Amounts of time spent in different scopes during GC.
248 base::TimeDelta scopes[Scope::NUMBER_OF_SCOPES];
249
250 // Holds details for incremental marking scopes.
251 IncrementalInfos incremental_scopes[Scope::NUMBER_OF_INCREMENTAL_SCOPES];
252 };
