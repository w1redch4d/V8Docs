317 static constexpr RegisterRepresentation FromCTypeInfo(
318 CTypeInfo t, CFunctionInfo::Int64Representation int64_repr) {
319 if (t.GetType() == CTypeInfo::Type::kVoid ||
320 t.GetType() == CTypeInfo::Type::kPointer) {
321 return RegisterRepresentation::Tagged();
322 } else if (t.GetType() == CTypeInfo::Type::kInt64 ||
323 t.GetType() == CTypeInfo::Type::kUint64) {
324 if (int64_repr == CFunctionInfo::Int64Representation::kBigInt) {

330 } else {
331 return RegisterRepresentation::FromMachineType(

361constexpr bool RegisterRepresentation::AllowImplicitRepresentationChangeTo(
362 RegisterRepresentation dst_rep, bool graph_created_from_turbofan) const {
363 if (*this == dst_rep) {
364 return true;
365 }
366 switch (dst_rep.value()) {
367 case RegisterRepresentation::Word32():
368 // We allow implicit tagged -> untagged conversions.
369 // Even without pointer compression, we use `Word32And` for Smi-checks on
370 // tagged values.
371 if (*this == any_of(RegisterRepresentation::Tagged(),
372 RegisterRepresentation::Compressed())) {
373 return true;
374 }
375 if (graph_created_from_turbofan &&
376 *this == RegisterRepresentation::Word64()) {
377 // TODO(12783): Remove this once Turboshaft graphs are not constructed
378 // via Turbofan any more. Unfortunately Turbofan has many implicit
379 // truncations which are hard to fix. Still, for wasm it is required
380 // that truncations in Turboshaft are explicit.
381 return true;
382 }
383 break;
384 case RegisterRepresentation::Word64():
385 // We allow implicit tagged -> untagged conversions.
386 if (kTaggedSize == kInt64Size &&
387 *this == RegisterRepresentation::Tagged()) {
388 return true;
389 }
390 break;
391 case RegisterRepresentation::Tagged():
392 // We allow implicit untagged -> tagged conversions. This is only safe for
393 // Smi values.
394 if (*this == RegisterRepresentation::WordPtr()) {
395 return true;
396 }
397 break;
398 case RegisterRepresentation::Compressed():
399 // Compression is a no-op.
400 if (*this == any_of(RegisterRepresentation::Tagged(),
401 RegisterRepresentation::WordPtr(),
402 RegisterRepresentation::Word32())) {
403 return true;
404 }
405 break;
406 default:
407 break;
408 }
409 return false;
410}
