30void IsolateSafepoint::EnterLocalSafepointScope() {
31 // Safepoints need to be initiated on some main thread.
32 DCHECK_NULL(LocalHeap::Current());
33 DCHECK(AllowGarbageCollection::IsAllowed());
34
35 LockMutex(isolate()->main_thread_local_heap());
36 if (++active_safepoint_scopes_ > 1) return;
37
38 // Local safepoint can only be initiated on the isolate's main thread.
39 DCHECK_EQ(ThreadId::Current(), isolate()->thread_id());
40
41 TimedHistogramScope timer(isolate()->counters()->gc_time_to_safepoint());
42 TRACE_GC(heap_->tracer(), GCTracer::Scope::TIME_TO_SAFEPOINT);
43
44 barrier_.Arm();
45 size_t running = SetSafepointRequestedFlags(IncludeMainThread::kNo);
46 barrier_.WaitUntilRunningThreadsInSafepoint(running);
47}

102void IsolateSafepoint::InitiateGlobalSafepointScopeRaw(
103 Isolate* initiator, PerClientSafepointData* client_data) {
104 CHECK_EQ(++active_safepoint_scopes_, 1);
105 barrier_.Arm();
106
107 size_t running =
108 SetSafepointRequestedFlags(ShouldIncludeMainThread(initiator));
109 client_data->set_locked_and_running(running);
110
111 if (isolate() != initiator) {
112 // An isolate might be waiting in the event loop. Post a task in order to
113 // wake it up.
114 isolate()->heap()->GetForegroundTaskRunner()->PostTask(
115 std::make_unique<GlobalSafepointInterruptTask>(heap_));
116
117 // Request an interrupt in case of long-running code.
118 isolate()->stack_guard()->RequestGlobalSafepoint();
119 }
120}

128size_t IsolateSafepoint::SetSafepointRequestedFlags(
129 IncludeMainThread include_main_thread) {
130 size_t running = 0;
131
132 // There needs to be at least one LocalHeap for the main thread.
133 DCHECK_NOT_NULL(local_heaps_head_);
134
135 for (LocalHeap* local_heap = local_heaps_head_; local_heap;
136 local_heap = local_heap->next_) {
137 if (local_heap->is_main_thread() &&
138 include_main_thread == IncludeMainThread::kNo) {
139 continue;
140 }
141
142 const LocalHeap::ThreadState old_state =
143 local_heap->state_.SetSafepointRequested();
144
145 if (old_state.IsRunning()) running++;
146 CHECK_IMPLIES(old_state.IsCollectionRequested(),
147 local_heap->is_main_thread());
148 CHECK(!old_state.IsSafepointRequested());
149 }
150
151 return running;
152}

183void IsolateSafepoint::ClearSafepointRequestedFlags(
184 IncludeMainThread include_main_thread) {
185 for (LocalHeap* local_heap = local_heaps_head_; local_heap;
186 local_heap = local_heap->next_) {
187 if (local_heap->is_main_thread() &&
188 include_main_thread == IncludeMainThread::kNo) {
189 continue;
190 }
191
192 const LocalHeap::ThreadState old_state =
193 local_heap->state_.ClearSafepointRequested();
194
195 CHECK(old_state.IsParked());
196 CHECK(old_state.IsSafepointRequested());
197 CHECK_IMPLIES(old_state.IsCollectionRequested(),
198 local_heap->is_main_thread());
199 }

340void GlobalSafepoint::EnterGlobalSafepointScope(Isolate* initiator) {
341 // Safepoints need to be initiated on some main thread.
342 DCHECK_NULL(LocalHeap::Current());
343
344 if (!clients_mutex_.TryLock()) {
345 IgnoreLocalGCRequests ignore_gc_requests(initiator->heap());
346 initiator->main_thread_local_heap()->ExecuteWhileParked(
347 [this]() { clients_mutex_.Lock(); });
348 }
349
350 if (++active_safepoint_scopes_ > 1) return;
351
352 TimedHistogramScope timer(
353 initiator->counters()->gc_time_to_global_safepoint());
354 TRACE_GC(initiator->heap()->tracer(),
355 GCTracer::Scope::TIME_TO_GLOBAL_SAFEPOINT);
356
357 std::vector<PerClientSafepointData> clients;
358
359 // Try to initiate safepoint for all clients. Fail immediately when the
360 // local_heaps_mutex_ can't be locked without blocking.
361 IterateSharedSpaceAndClientIsolates([&clients, initiator](Isolate* client) {
362 clients.emplace_back(client);
363 client->heap()->safepoint()->TryInitiateGlobalSafepointScope(
364 initiator, &clients.back());
365 });
366
367 // Iterate all clients again to initiate the safepoint for all of them - even
368 // if that means blocking.
369 for (PerClientSafepointData& client : clients) {
370 if (client.is_locked()) continue;
371 client.safepoint()->InitiateGlobalSafepointScope(initiator, &client);
372 }
373
374#if DEBUG
375 for (const PerClientSafepointData& client : clients) {
376 DCHECK_EQ(client.isolate()->shared_space_isolate(), shared_space_isolate_);
377 }
378#endif // DEBUG
379
380 // Now that safepoints were initiated for all clients, wait until all threads
381 // of all clients reached a safepoint.
382 for (const PerClientSafepointData& client : clients) {
383 DCHECK(client.is_locked());
384 client.safepoint()->WaitUntilRunningThreadsInSafepoint(&client);
385 }
386}

388void GlobalSafepoint::LeaveGlobalSafepointScope(Isolate* initiator) {
389 clients_mutex_.AssertHeld();
390 DCHECK_GT(active_safepoint_scopes_, 0);
391
392 if (--active_safepoint_scopes_ == 0) {
393 IterateSharedSpaceAndClientIsolates([initiator](Isolate* client) {
394 Heap* client_heap = client->heap();
395 client_heap->safepoint()->LeaveGlobalSafepointScope(initiator);
