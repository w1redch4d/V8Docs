29template <bool is_element>
30void LookupIterator::Start() {
31 // GetRoot might allocate if lookup_start_object_ is a string.
32 MaybeDirectHandle<JSReceiver> maybe_holder =
33 GetRoot(isolate_, lookup_start_object_, index_, configuration_);
34 if (!maybe_holder.ToHandle(&holder_)) {
35 // This is an attempt to perform an own property lookup on a non-JSReceiver
36 // that doesn't have any properties.
37 DCHECK(!IsJSReceiver(*lookup_start_object_));
38 DCHECK(!check_prototype_chain());
39 has_property_ = false;
40 state_ = NOT_FOUND;
41 return;
42 }
43
44 {
45 DisallowGarbageCollection no_gc;
46
47 has_property_ = false;
48 state_ = NOT_FOUND;
49
50 Tagged<JSReceiver> holder = *holder_;
51 Tagged<Map> map = holder->map(isolate_);
52
53 state_ = LookupInHolder<is_element>(map, holder);
54 if (IsFound()) return;
55
56 NextInternal<is_element>(map, holder);
57 }
58}

63void LookupIterator::Next() {
64 DCHECK_NE(JSPROXY, state_);
65 DCHECK_NE(TRANSITION, state_);
66 DCHECK_NE(NOT_FOUND, state_);
67 DisallowGarbageCollection no_gc;
68 has_property_ = false;
69
70 Tagged<JSReceiver> holder = *holder_;
71 Tagged<Map> map = holder->map(isolate_);
72
73 if (IsSpecialReceiverMap(map)) {
74 state_ = IsElement() ? LookupInSpecialHolder<true>(map, holder)
75 : LookupInSpecialHolder<false>(map, holder);
76 if (IsFound()) return;
77 }
78
79 IsElement() ? NextInternal<true>(map, holder)
80 : NextInternal<false>(map, holder);
81}

83template <bool is_element>
84void LookupIterator::NextInternal(Tagged<Map> map, Tagged<JSReceiver> holder) {
85 do {
86 Tagged<JSReceiver> maybe_holder = NextHolder(map);
87 if (maybe_holder.is_null()) {
88 if (interceptor_state_ == InterceptorState::kSkipNonMasking) {
89 RestartLookupForNonMaskingInterceptors<is_element>();
90 return;
91 }
92 state_ = NOT_FOUND;
93 if (holder != *holder_) holder_ = direct_handle(holder, isolate_);
94 return;
95 }
96 holder = maybe_holder;
97 map = holder->map(isolate_);
98 state_ = LookupInHolder<is_element>(map, holder);
99 } while (!IsFound());
100
101 holder_ = direct_handle(holder, isolate_);
102}

115void LookupIterator::RecheckTypedArrayBounds() {
116 DCHECK(IsJSTypedArray(*holder_, isolate_));
117 DCHECK_EQ(state_, TYPED_ARRAY_INDEX_NOT_FOUND);
118
119 if (!IsElement(*holder_)) {
120 // This happens when the index is not an allowed index.
121 return;
122 }
123
124 Tagged<JSObject> js_object = Cast<JSObject>(*holder_);
125 ElementsAccessor* accessor = js_object->GetElementsAccessor(isolate_);
126 Tagged<FixedArrayBase> backing_store = js_object->elements(isolate_);
127 number_ =
128 accessor->GetEntryForIndex(isolate_, js_object, backing_store, index_);
129
130 if (number_.is_not_found()) {
131 // The state is already TYPED_ARRAY_INDEX_NOT_FOUND.
132 return;
133 }
134 property_details_ = accessor->GetDetails(js_object, number_);
135#ifdef DEBUG
136 Tagged<Map> map = holder_->map(isolate_);
137 DCHECK(!map->has_frozen_elements());
138 DCHECK(!map->has_sealed_elements());
139#endif // DEBUG
140 has_property_ = true;
141 DCHECK_EQ(property_details_.kind(), v8::internal::PropertyKind::kData);
142 state_ = DATA;
143}

145// static
146MaybeDirectHandle<JSReceiver> LookupIterator::GetRootForNonJSReceiver(
147 Isolate* isolate, DirectHandle<JSPrimitive> lookup_start_object,
148 size_t index, Configuration configuration) {
149 // Strings are the only non-JSReceiver objects with properties (only elements
150 // and 'length') directly on the wrapper. Hence we can skip generating
151 // the wrapper for all other cases.
152 bool own_property_lookup = (configuration & kPrototypeChain) == 0;
153 if (IsString(*lookup_start_object, isolate)) {
154 if (own_property_lookup ||
155 index <
156 static_cast<size_t>(Cast<String>(*lookup_start_object)->length())) {
157 // TODO(verwaest): Speed this up. Perhaps use a cached wrapper on the
158 // native context, ensuring that we don't leak it into JS?
159 DirectHandle<JSFunction> constructor = isolate->string_function();
160 DirectHandle<JSObject> result =
161 isolate->factory()->NewJSObject(constructor);
162 Cast<JSPrimitiveWrapper>(result)->set_value(*lookup_start_object);
163 return result;
164 }
165 } else if (own_property_lookup) {
166 // Signal that the lookup will not find anything.
167 return {};
168 }
169 DirectHandle<HeapObject> root(
170 Object::GetPrototypeChainRootMap(*lookup_start_object, isolate)
171 ->prototype(isolate),
172 isolate);
173 if (IsNull(*root, isolate)) {
174 isolate->PushStackTraceAndDie(
175 reinterpret_cast<void*>((*lookup_start_object).ptr()));
176 }
177 return Cast<JSReceiver>(root);
178}

200void LookupIterator::InternalUpdateProtector(
201 Isolate* isolate, DirectHandle<JSAny> receiver_generic,
202 DirectHandle<Name> name) {
203 if (isolate->bootstrapper()->IsActive()) return;
204 if (!IsJSObject(*receiver_generic)) return;
205 auto receiver = Cast<JSObject>(receiver_generic);
206
207 ReadOnlyRoots roots(isolate);
208 if (*name == roots.constructor_string()) {
209 // Setting the constructor property could change an instance's @@species
210 if (IsJSArray(*receiver, isolate)) {
211 if (!Protectors::IsArraySpeciesLookupChainIntact(isolate)) return;
212 isolate->CountUsage(
213 v8::Isolate::UseCounterFeature::kArrayInstanceConstructorModified);
214 Protectors::InvalidateArraySpeciesLookupChain(isolate);
215 return;
216 } else if (IsJSPromise(*receiver, isolate)) {
217 if (!Protectors::IsPromiseSpeciesLookupChainIntact(isolate)) return;
218 Protectors::InvalidatePromiseSpeciesLookupChain(isolate);
219 return;
220 } else if (IsJSRegExp(*receiver, isolate)) {
221 if (!Protectors::IsRegExpSpeciesLookupChainIntact(isolate)) return;
222 Protectors::InvalidateRegExpSpeciesLookupChain(isolate);
223 return;
224 } else if (IsJSTypedArray(*receiver, isolate)) {
225 if (!Protectors::IsTypedArraySpeciesLookupChainIntact(isolate)) return;
226 Protectors::InvalidateTypedArraySpeciesLookupChain(isolate);
227 return;
228 }
229 if (receiver->map(isolate)->is_prototype_map()) {
230 DisallowGarbageCollection no_gc;
231 // Setting the constructor of any prototype with the @@species protector
232 // (of any realm) also needs to invalidate the protector.
233 if (isolate->IsInCreationContext(
234 Cast<JSObject>(*receiver),
235 Context::INITIAL_ARRAY_PROTOTYPE_INDEX)) {
236 if (!Protectors::IsArraySpeciesLookupChainIntact(isolate)) return;
237 isolate->CountUsage(
238 v8::Isolate::UseCounterFeature::kArrayPrototypeConstructorModified);
239 Protectors::InvalidateArraySpeciesLookupChain(isolate);
240 } else if (IsJSPromisePrototype(*receiver)) {
241 if (!Protectors::IsPromiseSpeciesLookupChainIntact(isolate)) return;
242 Protectors::InvalidatePromiseSpeciesLookupChain(isolate);
243 } else if (IsJSRegExpPrototype(*receiver)) {
244 if (!Protectors::IsRegExpSpeciesLookupChainIntact(isolate)) return;
245 Protectors::InvalidateRegExpSpeciesLookupChain(isolate);
246 } else if (IsJSTypedArrayPrototype(*receiver)) {
247 if (!Protectors::IsTypedArraySpeciesLookupChainIntact(isolate)) return;
248 Protectors::InvalidateTypedArraySpeciesLookupChain(isolate);
249 }
250 }
251 } else if (*name == roots.next_string()) {
252 if (IsJSArrayIterator(*receiver) || IsJSArrayIteratorPrototype(*receiver)) {
253 // Setting the next property of %ArrayIteratorPrototype% also needs to
254 // invalidate the array iterator protector.
255 if (!Protectors::IsArrayIteratorLookupChainIntact(isolate)) return;
256 Protectors::InvalidateArrayIteratorLookupChain(isolate);
257 } else if (IsJSMapIterator(*receiver) ||
258 IsJSMapIteratorPrototype(*receiver)) {
259 if (!Protectors::IsMapIteratorLookupChainIntact(isolate)) return;
260 Protectors::InvalidateMapIteratorLookupChain(isolate);
261 } else if (IsJSSetIterator(*receiver) ||
262 IsJSSetIteratorPrototype(*receiver)) {
263 if (!Protectors::IsSetIteratorLookupChainIntact(isolate)) return;
264 Protectors::InvalidateSetIteratorLookupChain(isolate);
265 } else if (IsJSStringIterator(*receiver) ||
266 IsJSStringIteratorPrototype(*receiver)) {
267 // Setting the next property of %StringIteratorPrototype% invalidates the
268 // string iterator protector.
269 if (!Protectors::IsStringIteratorLookupChainIntact(isolate)) return;
270 Protectors::InvalidateStringIteratorLookupChain(isolate);
271 }
272 } else if (*name == roots.species_symbol()) {
273 // Setting the Symbol.species property of any Array, Promise or TypedArray
274 // constructor invalidates the @@species protector
275 if (IsJSArrayConstructor(*receiver)) {
276 if (!Protectors::IsArraySpeciesLookupChainIntact(isolate)) return;
277 isolate->CountUsage(
278 v8::Isolate::UseCounterFeature::kArraySpeciesModified);
279 Protectors::InvalidateArraySpeciesLookupChain(isolate);
280 } else if (IsJSPromiseConstructor(*receiver)) {
281 if (!Protectors::IsPromiseSpeciesLookupChainIntact(isolate)) return;
282 Protectors::InvalidatePromiseSpeciesLookupChain(isolate);
283 } else if (IsJSRegExpConstructor(*receiver)) {
284 if (!Protectors::IsRegExpSpeciesLookupChainIntact(isolate)) return;
285 Protectors::InvalidateRegExpSpeciesLookupChain(isolate);
286 } else if (IsTypedArrayConstructor(*receiver)) {
287 if (!Protectors::IsTypedArraySpeciesLookupChainIntact(isolate)) return;
288 Protectors::InvalidateTypedArraySpeciesLookupChain(isolate);
289 }
290 } else if (*name == roots.is_concat_spreadable_symbol()) {
291 if (!Protectors::IsIsConcatSpreadableLookupChainIntact(isolate)) return;
292 Protectors::InvalidateIsConcatSpreadableLookupChain(isolate);
293 } else if (*name == roots.iterator_symbol()) {
294 if (IsJSArray(*receiver, isolate)) {
295 if (!Protectors::IsArrayIteratorLookupChainIntact(isolate)) return;
296 Protectors::InvalidateArrayIteratorLookupChain(isolate);
297 } else if (IsJSSet(*receiver, isolate) || IsJSSetIterator(*receiver) ||
298 IsJSSetIteratorPrototype(*receiver) ||
299 IsJSSetPrototype(*receiver)) {
300 if (Protectors::IsSetIteratorLookupChainIntact(isolate)) {
301 Protectors::InvalidateSetIteratorLookupChain(isolate);
302 }
303 } else if (IsJSMapIterator(*receiver) ||
304 IsJSMapIteratorPrototype(*receiver)) {
305 if (Protectors::IsMapIteratorLookupChainIntact(isolate)) {
306 Protectors::InvalidateMapIteratorLookupChain(isolate);
307 }
308 } else if (IsJSIteratorPrototype(*receiver)) {
309 if (Protectors::IsMapIteratorLookupChainIntact(isolate)) {
310 Protectors::InvalidateMapIteratorLookupChain(isolate);
311 }
312 if (Protectors::IsSetIteratorLookupChainIntact(isolate)) {
313 Protectors::InvalidateSetIteratorLookupChain(isolate);
314 }
315 } else if (isolate->IsInCreationContext(
316 *receiver, Context::INITIAL_STRING_PROTOTYPE_INDEX)) {
317 // Setting the Symbol.iterator property of String.prototype invalidates
318 // the string iterator protector. Symbol.iterator can also be set on a
319 // String wrapper, but not on a primitive string. We only support
320 // protector for primitive strings.
321 if (!Protectors::IsStringIteratorLookupChainIntact(isolate)) return;
322 Protectors::InvalidateStringIteratorLookupChain(isolate);
323 }
324 } else if (*name == roots.resolve_string()) {
325 if (!Protectors::IsPromiseResolveLookupChainIntact(isolate)) return;
326 // Setting the "resolve" property on any %Promise% intrinsic object
327 // invalidates the Promise.resolve protector.
328 if (IsJSPromiseConstructor(*receiver)) {
329 Protectors::InvalidatePromiseResolveLookupChain(isolate);
330 }
331 } else if (*name == roots.then_string()) {
332 if (!Protectors::IsPromiseThenLookupChainIntact(isolate)) return;
333 // Setting the "then" property on any JSPromise instance or on the
334 // initial %PromisePrototype% invalidates the Promise#then protector.
335 // Also setting the "then" property on the initial %ObjectPrototype%
336 // invalidates the Promise#then protector, since we use this protector
337 // to guard the fast-path in AsyncGeneratorResolve, where we can skip
338 // the ResolvePromise step and go directly to FulfillPromise if we
339 // know that the Object.prototype doesn't contain a "then" method.
340 if (IsJSPromise(*receiver, isolate) || IsJSObjectPrototype(*receiver) ||
341 IsJSPromisePrototype(*receiver)) {
342 Protectors::InvalidatePromiseThenLookupChain(isolate);
343 }
344 } else if (*name == roots.match_all_symbol() ||
345 *name == roots.replace_symbol() || *name == roots.split_symbol()) {
346 if (!Protectors::IsNumberStringNotRegexpLikeIntact(isolate)) return;
347 // We need to protect the prototype chains of `Number.prototype` and
348 // `String.prototype`: that `Symbol.{matchAll|replace|split}` is not added
349 // as a property on any object on these prototype chains. We detect
350 // `Number.prototype` and `String.prototype` by checking for a prototype
351 // that is a JSPrimitiveWrapper. This is a safe approximation. Using
352 // JSPrimitiveWrapper as prototype should be sufficiently rare.
353 if (receiver->map()->is_prototype_map() &&
354 (IsJSPrimitiveWrapper(*receiver) || IsJSObjectPrototype(*receiver))) {
355 Protectors::InvalidateNumberStringNotRegexpLike(isolate);
356 }
357 } else if (*name == roots.to_primitive_symbol()) {
358 if (!Protectors::IsStringWrapperToPrimitiveIntact(isolate)) return;
359 if (isolate->IsInCreationContext(*receiver,
360 Context::INITIAL_STRING_PROTOTYPE_INDEX) ||
361 isolate->IsInCreationContext(*receiver,
362 Context::INITIAL_OBJECT_PROTOTYPE_INDEX) ||
363 IsStringWrapper(*receiver)) {
364 Protectors::InvalidateStringWrapperToPrimitive(isolate);
365 }
366 } else if (*name == roots.valueOf_string()) {
367 if (!Protectors::IsStringWrapperToPrimitiveIntact(isolate)) return;
368 if (isolate->IsInCreationContext(*receiver,
369 Context::INITIAL_STRING_PROTOTYPE_INDEX) ||
370 IsStringWrapper(*receiver)) {
371 Protectors::InvalidateStringWrapperToPrimitive(isolate);
372 }
373 }
374}

376void LookupIterator::PrepareForDataProperty(DirectHandle<Object> value) {
377 DCHECK(state_ == DATA || state_ == ACCESSOR);
378 DCHECK(HolderIsReceiverOrHiddenPrototype());
379 DCHECK(!IsWasmObject(*receiver_, isolate_));
380
381 DirectHandle<JSReceiver> holder = GetHolder<JSReceiver>();
382 // We are not interested in tracking constness of a JSProxy's direct
383 // properties.
384 DCHECK_IMPLIES(IsJSProxy(*holder, isolate_), name()->IsPrivate());
385 if (IsJSProxy(*holder, isolate_)) return;
386
387 if (IsElement(*holder)) {
388 DirectHandle<JSObject> holder_obj = Cast<JSObject>(holder);
389 ElementsKind kind = holder_obj->GetElementsKind(isolate_);
390 ElementsKind to = Object::OptimalElementsKind(*value, isolate_);
391 if (IsHoleyElementsKind(kind)) to = GetHoleyElementsKind(to);
392 to = GetMoreGeneralElementsKind(kind, to);
393
394 if (kind != to) {
395 JSObject::TransitionElementsKind(holder_obj, to);
396 }
397
398 // Copy the backing store if it is copy-on-write.
399 if (IsSmiOrObjectElementsKind(to) || IsSealedElementsKind(to) ||
400 IsNonextensibleElementsKind(to)) {
401 JSObject::EnsureWritableFastElements(holder_obj);
402 }
403 return;
404 }
405
406 if (IsJSGlobalObject(*holder, isolate_)) {
407 DirectHandle<GlobalDictionary> dictionary(
408 Cast<JSGlobalObject>(*holder)->global_dictionary(isolate_,
409 kAcquireLoad),
410 isolate());
411 DirectHandle<PropertyCell> cell(
412 dictionary->CellAt(isolate_, dictionary_entry()), isolate());
413 property_details_ = cell->property_details();
414 PropertyCell::PrepareForAndSetValue(
415 isolate(), dictionary, dictionary_entry(), value, property_details_);
416 return;
417 }
418
419 PropertyConstness new_constness = PropertyConstness::kConst;
420 if (constness() == PropertyConstness::kConst) {
421 DCHECK_EQ(PropertyKind::kData, property_details_.kind());
422 // Check that current value matches new value otherwise we should make
423 // the property mutable.
424 if (holder->HasFastProperties(isolate_)) {
425 if (!CanStayConst(*value)) new_constness = PropertyConstness::kMutable;
426 } else if (V8_DICT_PROPERTY_CONST_TRACKING_BOOL) {
427 if (!DictCanStayConst(*value)) {
428 property_details_ =
429 property_details_.CopyWithConstness(PropertyConstness::kMutable);
430
431 // We won't reach the map updating code after Map::Update below, because
432 // that's only for the case that the existing map is a fast mode map.
433 // Therefore, we need to perform the necessary updates to the property
434 // details and the prototype validity cell directly.
435 if constexpr (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
436 Tagged<SwissNameDictionary> dict =
437 holder->property_dictionary_swiss();
438 dict->DetailsAtPut(dictionary_entry(), property_details_);
439 } else {
440 Tagged<NameDictionary> dict = holder->property_dictionary();
441 dict->DetailsAtPut(dictionary_entry(), property_details_);
442 }
443
444 Tagged<Map> old_map = holder->map(isolate_);
445 if (old_map->is_prototype_map()) {
446 JSObject::InvalidatePrototypeChains(old_map);
447 }
448 }
449 return;
450 }
451 }
452
453 if (!holder->HasFastProperties(isolate_)) return;
454
455 auto holder_obj = Cast<JSObject>(holder);
456 Handle<Map> old_map(holder->map(isolate_), isolate_);
457
458 Handle<Map> new_map = Map::Update(isolate_, old_map);
459 if (!new_map->is_dictionary_map()) { // fast -> fast
460 new_map = Map::PrepareForDataProperty(
461 isolate(), new_map, descriptor_number(), new_constness, value);
462
463 if (old_map.is_identical_to(new_map)) {
464 // Update the property details if the representation was None.
465 if (constness() != new_constness || representation().IsNone()) {
466 property_details_ = new_map->instance_descriptors(isolate_)->GetDetails(
467 descriptor_number());
468 }
469 return;
470 }
471 }
472 // We should only get here if the new_map is different from the old map,
473 // otherwise we would have falled through to the is_identical_to check above.
474 DCHECK_NE(*old_map, *new_map);
475
476 JSObject::MigrateToMap(isolate_, holder_obj, new_map);
477 ReloadPropertyInformation<false>();
478
479 // If we transitioned from fast to slow and the property changed from kConst
480 // to kMutable, then this change in the constness is indicated by neither the
481 // old or the new map. We need to update the constness ourselves.
482 DCHECK(!old_map->is_dictionary_map());
483 if (V8_DICT_PROPERTY_CONST_TRACKING_BOOL && new_map->is_dictionary_map() &&
484 new_constness == PropertyConstness::kMutable) { // fast -> slow
485 property_details_ =
486 property_details_.CopyWithConstness(PropertyConstness::kMutable);
487
488 if constexpr (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
489 Tagged<SwissNameDictionary> dict =
490 holder_obj->property_dictionary_swiss();
491 dict->DetailsAtPut(dictionary_entry(), property_details_);
492 } else {
493 Tagged<NameDictionary> dict = holder_obj->property_dictionary();
494 dict->DetailsAtPut(dictionary_entry(), property_details_);
495 }
496
497 DCHECK_IMPLIES(new_map->is_prototype_map(),
498 !new_map->IsPrototypeValidityCellValid());
499 }
500}

502void LookupIterator::ReconfigureDataProperty(DirectHandle<Object> value,
503 PropertyAttributes attributes) {
504 DCHECK(state_ == DATA || state_ == ACCESSOR);
505 DCHECK(HolderIsReceiverOrHiddenPrototype());
506
507 DirectHandle<JSReceiver> holder = GetHolder<JSReceiver>();
508 if (V8_UNLIKELY(IsWasmObject(*holder))) UNREACHABLE();
509
510 // Property details can never change for private properties.
511 if (IsJSProxy(*holder, isolate_)) {
512 DCHECK(name()->IsPrivate());
513 return;
514 }
515
516 DirectHandle<JSObject> holder_obj = Cast<JSObject>(holder);
517 if (IsElement(*holder)) {
518 DCHECK(!holder_obj->HasTypedArrayOrRabGsabTypedArrayElements(isolate_));
519 DCHECK(attributes != NONE || !holder_obj->HasFastElements(isolate_));
520 DirectHandle<FixedArrayBase> elements(holder_obj->elements(isolate_),
521 isolate());
522 holder_obj->GetElementsAccessor(isolate_)->Reconfigure(
523 holder_obj, elements, number_, value, attributes);
524 ReloadPropertyInformation<true>();
525 } else if (holder_obj->HasFastProperties(isolate_)) {
526 Handle<Map> old_map(holder_obj->map(isolate_), isolate_);
527 // Force mutable to avoid changing constant value by reconfiguring
528 // kData -> kAccessor -> kData.
529 Handle<Map> new_map = MapUpdater::ReconfigureExistingProperty(
530 isolate_, old_map, descriptor_number(), i::PropertyKind::kData,
531 attributes, PropertyConstness::kMutable);
532 if (!new_map->is_dictionary_map()) {
533 // Make sure that the data property has a compatible representation.
534 // TODO(leszeks): Do this as part of ReconfigureExistingProperty.
535 new_map =
536 Map::PrepareForDataProperty(isolate(), new_map, descriptor_number(),
537 PropertyConstness::kMutable, value);
538 }
539 JSObject::MigrateToMap(isolate_, holder_obj, new_map);
540 ReloadPropertyInformation<false>();
541 }
542
543 if (!IsElement(*holder) && !holder_obj->HasFastProperties(isolate_)) {
544 if (holder_obj->map(isolate_)->is_prototype_map() &&
545 (((property_details_.attributes() & READ_ONLY) == 0 &&
546 (attributes & READ_ONLY) != 0) ||
547 (property_details_.attributes() & DONT_ENUM) !=
548 (attributes & DONT_ENUM))) {
549 // Invalidate prototype validity cell when a property is reconfigured
550 // from writable to read-only as this may invalidate transitioning store
551 // IC handlers.
552 // Invalidate prototype validity cell when a property changes
553 // enumerability to clear the prototype chain enum cache.
554 JSObject::InvalidatePrototypeChains(holder->map(isolate_));
555 }
556 if (IsJSGlobalObject(*holder_obj, isolate_)) {
557 PropertyDetails details(PropertyKind::kData, attributes,
558 PropertyCellType::kMutable);
559 DirectHandle<GlobalDictionary> dictionary(
560 Cast<JSGlobalObject>(*holder_obj)
561 ->global_dictionary(isolate_, kAcquireLoad),
562 isolate());
563
564 DirectHandle<PropertyCell> cell = PropertyCell::PrepareForAndSetValue(
565 isolate(), dictionary, dictionary_entry(), value, details);
566 property_details_ = cell->property_details();
567 DCHECK_EQ(cell->value(), *value);
568 } else {
569 PropertyDetails details(PropertyKind::kData, attributes,
570 PropertyConstness::kMutable);
571 if constexpr (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
572 DirectHandle<SwissNameDictionary> dictionary(
573 holder_obj->property_dictionary_swiss(isolate_), isolate());
574 dictionary->ValueAtPut(dictionary_entry(), *value);
575 dictionary->DetailsAtPut(dictionary_entry(), details);
576 DCHECK_EQ(details.AsSmi(),
577 dictionary->DetailsAt(dictionary_entry()).AsSmi());
578 property_details_ = details;
579 } else {
580 DirectHandle<NameDictionary> dictionary(
581 holder_obj->property_dictionary(isolate_), isolate());
582 PropertyDetails original_details =
583 dictionary->DetailsAt(dictionary_entry());
584 int enumeration_index = original_details.dictionary_index();
585 DCHECK_GT(enumeration_index, 0);
586 details = details.set_index(enumeration_index);
587 dictionary->SetEntry(dictionary_entry(), *name(), *value, details);
588 property_details_ = details;
589 }
590 }

603// Can only be called when the receiver is a JSObject, or when the name is a
604// private field, otherwise JSProxy has to be handled via a trap.
605// Adding properties to primitive values is not observable.
606void LookupIterator::PrepareTransitionToDataProperty(
607 DirectHandle<JSReceiver> receiver, DirectHandle<Object> value,
608 PropertyAttributes attributes, StoreOrigin store_origin) {
609 DCHECK_IMPLIES(IsJSProxy(*receiver, isolate_), name()->IsPrivate());
610 DCHECK_IMPLIES(!receiver.is_identical_to(GetStoreTarget<JSReceiver>()),
611 name()->IsPrivateName());
612 DCHECK(!IsAlwaysSharedSpaceJSObject(*receiver));
613 if (state_ == TRANSITION) return;
614
615 if (!IsElement() && name()->IsPrivate()) {
616 attributes = static_cast<PropertyAttributes>(attributes | DONT_ENUM);
617 }
618
619 DCHECK(state_ != LookupIterator::ACCESSOR ||
620 IsAccessorInfo(*GetAccessors(), isolate_));
621 DCHECK_NE(TYPED_ARRAY_INDEX_NOT_FOUND, state_);
622 DCHECK(state_ == NOT_FOUND || !HolderIsReceiverOrHiddenPrototype());
623
624 DirectHandle<Map> map(receiver->map(isolate_), isolate_);
625
626 // Dictionary maps can always have additional data properties.
627 if (map->is_dictionary_map()) {
628 state_ = TRANSITION;
629 if (IsJSGlobalObjectMap(*map)) {
630 DCHECK(!IsTheHole(*value, isolate_));
631 // Don't set enumeration index (it will be set during value store).
632 property_details_ =
633 PropertyDetails(PropertyKind::kData, attributes,
634 PropertyCell::InitialType(isolate_, *value));
635 transition_ = isolate_->factory()->NewPropertyCell(
636 name(), property_details_, value);
637 has_property_ = true;
638 } else {
639 // Don't set enumeration index (it will be set during value store).
640 property_details_ =
641 PropertyDetails(PropertyKind::kData, attributes,
642 PropertyDetails::kConstIfDictConstnessTracking);
643 transition_ = map;
644 }
645 return;
646 }
647
648 DirectHandle<Map> transition =
649 Map::TransitionToDataProperty(isolate_, map, name_, value, attributes,
650 PropertyConstness::kConst, store_origin);
651 state_ = TRANSITION;
652 transition_ = transition;
653
654 if (transition->is_dictionary_map()) {
655 DCHECK(!IsJSGlobalObjectMap(*transition));
656 // Don't set enumeration index (it will be set during value store).
657 property_details_ =
658 PropertyDetails(PropertyKind::kData, attributes,
659 PropertyDetails::kConstIfDictConstnessTracking);
660 } else {
661 property_details_ = transition->GetLastDescriptorDetails(isolate_);
662 has_property_ = true;
663 }
664}

666void LookupIterator::ApplyTransitionToDataProperty(
667 DirectHandle<JSReceiver> receiver) {
668 DCHECK_EQ(TRANSITION, state_);
669
670 DCHECK_IMPLIES(!receiver.is_identical_to(GetStoreTarget<JSReceiver>()),
671 name()->IsPrivateName());
672 holder_ = receiver;
673 if (IsJSGlobalObject(*receiver, isolate_)) {
674 JSObject::InvalidatePrototypeChains(receiver->map(isolate_));
675
676 // Install a property cell.
677 auto global = Cast<JSGlobalObject>(receiver);
678 DCHECK(!global->HasFastProperties());
679 DirectHandle<GlobalDictionary> dictionary(
680 global->global_dictionary(isolate_, kAcquireLoad), isolate_);
681
682 dictionary =
683 GlobalDictionary::Add(isolate_, dictionary, name(), transition_cell(),
684 property_details_, &number_);
685 global->set_global_dictionary(*dictionary, kReleaseStore);
686
687 // Reload details containing proper enumeration index value.
688 property_details_ = transition_cell()->property_details();
689 has_property_ = true;
690 state_ = DATA;
691 return;
692 }
693 DirectHandle<Map> transition = transition_map();
694 bool simple_transition =
695 transition->GetBackPointer(isolate_) == receiver->map(isolate_);
696
697 if (configuration_ == DEFAULT && !transition->is_dictionary_map() &&
698 !transition->IsPrototypeValidityCellValid()) {
699 // Only LookupIterator instances with DEFAULT (full prototype chain)
700 // configuration can produce valid transition handler maps.
701 DirectHandle<UnionOf<Smi, Cell>> validity_cell =
702 Map::GetOrCreatePrototypeChainValidityCell(transition, isolate());
703 transition->set_prototype_validity_cell(*validity_cell, kRelaxedStore);
704 }
705
706 if (!IsJSProxy(*receiver, isolate_)) {
707 JSObject::MigrateToMap(isolate_, Cast<JSObject>(receiver), transition);
708 }
709
710 if (simple_transition) {
711 number_ = transition->LastAdded();
712 property_details_ = transition->GetLastDescriptorDetails(isolate_);
713 state_ = DATA;
714 } else if (receiver->map(isolate_)->is_dictionary_map()) {
715 if (receiver->map(isolate_)->is_prototype_map() &&
716 IsJSObject(*receiver, isolate_)) {
717 JSObject::InvalidatePrototypeChains(receiver->map(isolate_));
718 }
719 if constexpr (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
720 DirectHandle<SwissNameDictionary> dictionary(
721 receiver->property_dictionary_swiss(isolate_), isolate_);
722
723 dictionary =
724 SwissNameDictionary::Add(isolate(), dictionary, name(),
725 isolate_->factory()->uninitialized_value(),
726 property_details_, &number_);
727 receiver->SetProperties(*dictionary);
728 } else {
729 DirectHandle<NameDictionary> dictionary(
730 receiver->property_dictionary(isolate_), isolate_);
731
732 dictionary =
733 NameDictionary::Add(isolate(), dictionary, name(),
734 isolate_->factory()->uninitialized_value(),
735 property_details_, &number_);
736 receiver->SetProperties(*dictionary);
737 // TODO(pthier): Add flags to swiss dictionaries.
738 if (name()->IsInteresting(isolate())) {
739 dictionary->set_may_have_interesting_properties(true);
740 }
741 // Reload details containing proper enumeration index value.
742 property_details_ = dictionary->DetailsAt(number_);
743 }
744 has_property_ = true;
745 state_ = DATA;
746
747 } else {
748 ReloadPropertyInformation<false>();
749 }

782void LookupIterator::TransitionToAccessorProperty(
783 DirectHandle<Object> getter, DirectHandle<Object> setter,
784 PropertyAttributes attributes) {
785 DCHECK(!IsNull(*getter, isolate_) || !IsNull(*setter, isolate_));
786 // Can only be called when the receiver is a JSObject. JSProxy has to be
787 // handled via a trap. Adding properties to primitive values is not
788 // observable.
789 DirectHandle<JSObject> receiver = GetStoreTarget<JSObject>();
790 if (!IsElement() && name()->IsPrivate()) {
791 attributes = static_cast<PropertyAttributes>(attributes | DONT_ENUM);
792 }
793
794 if (!IsElement(*receiver) && !receiver->map(isolate_)->is_dictionary_map()) {
795 Handle<Map> old_map(receiver->map(isolate_), isolate_);
796
797 if (!holder_.is_identical_to(receiver)) {
798 holder_ = receiver;
799 state_ = NOT_FOUND;
800 } else if (state_ == INTERCEPTOR) {
801 LookupInRegularHolder<false>(*old_map, *holder_);
802 }
803 // The case of IsFound() && number_.is_not_found() can occur for
804 // interceptors.
805 DCHECK_IMPLIES(!IsFound(), number_.is_not_found());
806
807 DirectHandle<Map> new_map = Map::TransitionToAccessorProperty(
808 isolate_, old_map, name_, number_, getter, setter, attributes);
809 bool simple_transition =
810 new_map->GetBackPointer(isolate_) == receiver->map(isolate_);
811 JSObject::MigrateToMap(isolate_, receiver, new_map);
812
813 if (simple_transition) {
814 number_ = new_map->LastAdded();
815 property_details_ = new_map->GetLastDescriptorDetails(isolate_);
816 state_ = ACCESSOR;
817 return;
818 }
819
820 ReloadPropertyInformation<false>();
821 if (!new_map->is_dictionary_map()) return;
822 }
823
824 DirectHandle<AccessorPair> pair;
825 if (state() == ACCESSOR && IsAccessorPair(*GetAccessors(), isolate_)) {
826 pair = Cast<AccessorPair>(GetAccessors());
827 // If the component and attributes are identical, nothing has to be done.
828 if (pair->Equals(*getter, *setter)) {
829 if (property_details().attributes() == attributes) {
830 if (!IsElement(*receiver)) JSObject::ReoptimizeIfPrototype(receiver);
831 return;
832 }
833 } else {
834 pair = AccessorPair::Copy(isolate(), pair);
835 pair->SetComponents(*getter, *setter);
836 }
837 } else {
838 pair = factory()->NewAccessorPair();
839 pair->SetComponents(*getter, *setter);
840 }
841
842 TransitionToAccessorPair(pair, attributes);
843
844#if VERIFY_HEAP
845 if (v8_flags.verify_heap) {
846 receiver->JSObjectVerify(isolate());
847 }
848#endif
849}

851void LookupIterator::TransitionToAccessorPair(DirectHandle<Object> pair,
852 PropertyAttributes attributes) {
853 DirectHandle<JSObject> receiver = GetStoreTarget<JSObject>();
854 holder_ = receiver;
855
856 PropertyDetails details(PropertyKind::kAccessor, attributes,
857 PropertyCellType::kMutable);
858
859 if (IsElement(*receiver)) {
860 // TODO(verwaest): Move code into the element accessor.
861 isolate_->CountUsage(v8::Isolate::kIndexAccessor);
862 DirectHandle<NumberDictionary> dictionary =
863 JSObject::NormalizeElements(receiver);
864
865 dictionary = NumberDictionary::Set(isolate_, dictionary, array_index(),
866 pair, receiver, details);
867 receiver->RequireSlowElements(*dictionary);
868
869 if (receiver->HasSlowArgumentsElements(isolate_)) {
870 Tagged<SloppyArgumentsElements> parameter_map =
871 Cast<SloppyArgumentsElements>(receiver->elements(isolate_));
872 uint32_t length = parameter_map->length();
873 if (number_.is_found() && number_.as_uint32() < length) {
874 parameter_map->set_mapped_entries(
875 number_.as_int(), ReadOnlyRoots(isolate_).the_hole_value());
876 }
877 parameter_map->set_arguments(*dictionary);
878 } else {
879 receiver->set_elements(*dictionary);
880 }
881
882 ReloadPropertyInformation<true>();
883 } else {
884 PropertyNormalizationMode mode = CLEAR_INOBJECT_PROPERTIES;
885 if (receiver->map(isolate_)->is_prototype_map()) {
886 JSObject::InvalidatePrototypeChains(receiver->map(isolate_));
887 mode = KEEP_INOBJECT_PROPERTIES;
888 }
889
890 // Normalize object to make this operation simple.
891 JSObject::NormalizeProperties(isolate_, receiver, mode, 0,
892 "TransitionToAccessorPair");
893
894 JSObject::SetNormalizedProperty(receiver, name_, pair, details);
895 JSObject::ReoptimizeIfPrototype(receiver);
896
897 ReloadPropertyInformation<false>();
898 }

957bool LookupIterator::CanStayConst(Tagged<Object> value) const {
958 DCHECK(!holder_.is_null());
959 DCHECK(!IsElement(*holder_));
960 DCHECK(holder_->HasFastProperties(isolate_));
961 DCHECK_EQ(PropertyLocation::kField, property_details_.location());
962 DCHECK_EQ(PropertyConstness::kConst, property_details_.constness());
963 if (IsUninitialized(value, isolate())) {
964 // Storing uninitialized value means that we are preparing for a computed
965 // property value in an object literal. The initializing store will follow
966 // and it will properly update constness based on the actual value.
967 return true;
968 }
969 DirectHandle<JSObject> holder = GetHolder<JSObject>();
970 FieldIndex field_index =
971 FieldIndex::ForDetails(holder->map(isolate_), property_details_);
972 if (property_details_.representation().IsDouble()) {
973 if (!IsNumber(value, isolate_)) return false;
974 uint64_t bits;
975 Tagged<Object> current_value =
976 holder->RawFastPropertyAt(isolate_, field_index);
977 DCHECK(IsHeapNumber(current_value, isolate_));
978 bits = Cast<HeapNumber>(current_value)->value_as_bits();
979 // Use bit representation of double to check for hole double, since
980 // manipulating the signaling NaN used for the hole in C++, e.g. with
981 // base::bit_cast or value(), will change its value on ia32 (the x87
982 // stack is used to return values and stores to the stack silently clear the
983 // signalling bit).
984 // Only allow initializing stores to double to stay constant.
985 return bits == kHoleNanInt64;
986 }
987
988 Tagged<Object> current_value =
989 holder->RawFastPropertyAt(isolate_, field_index);
990 return IsUninitialized(current_value, isolate());
991}

1275 Tagged<Map> const map, Tagged<JSReceiver> const holder) {

1282#if V8_ENABLE_WEBASSEMBLY

1453 if (!lookup_start_object_.is_identical_to(receiver_) &&

1478 // We have found a cached property! Modify the iterator accordingly.

1485std::optional<Tagged<Object>> ConcurrentLookupIterator::TryGetOwnCowElement(
1486 Isolate* isolate, Tagged<FixedArray> array_elements,
1487 ElementsKind elements_kind, int array_length, size_t index) {
1488 DisallowGarbageCollection no_gc;
1489
1490 CHECK_EQ(array_elements->map(), ReadOnlyRoots(isolate).fixed_cow_array_map());
1491 DCHECK(IsFastElementsKind(elements_kind) &&
1492 IsSmiOrObjectElementsKind(elements_kind));
1493 USE(elements_kind);
1494 DCHECK_GE(array_length, 0);
1495
1496 // ________________________________________
1497 // ( Check against both JSArray::length and )
1498 // ( FixedArray::length. )
1499 // ----------------------------------------
1500 // o ^__^
1501 // o (oo)\_______
1502 // (__)\ )\/\
1503 // ||----w |
1504 // || ||
1505 // The former is the source of truth, but due to concurrent reads it may not
1506 // match the given `array_elements`.
1507 if (index >= static_cast<size_t>(array_length)) return {};
1508 if (index >= static_cast<size_t>(array_elements->length())) return {};
1509
1510 Tagged<Object> result = array_elements->get(static_cast<int>(index));
1511
1512 // ______________________________________
1513 // ( Filter out holes irrespective of the )
1514 // ( elements kind. )
1515 // --------------------------------------
1516 // o ^__^
1517 // o (..)\_______
1518 // (__)\ )\/\
1519 // ||----w |
1520 // || ||
1521 // The elements kind may not be consistent with the given elements backing
1522 // store.
1523 if (result == ReadOnlyRoots(isolate).the_hole_value()) return {};
1524
1525 return result;
1526}

1529ConcurrentLookupIterator::Result
1530ConcurrentLookupIterator::TryGetOwnConstantElement(
1531 Tagged<Object>* result_out, Isolate* isolate, LocalIsolate* local_isolate,
1532 Tagged<JSObject> holder, Tagged<FixedArrayBase> elements,
1533 ElementsKind elements_kind, size_t index) {
1534 DisallowGarbageCollection no_gc;
1535
1536 DCHECK_LE(index, JSObject::kMaxElementIndex);
1537
1538 // Own 'constant' elements (PropertyAttributes READ_ONLY|DONT_DELETE) occur in
1539 // three main cases:
1540 //
1541 // 1. Frozen elements: guaranteed constant.
1542 // 2. Dictionary elements: may be constant.
1543 // 3. String wrapper elements: guaranteed constant.
1544
1545 // Interesting field reads below:
1546 //
1547 // - elements.length (immutable on FixedArrays).
1548 // - elements[i] (immutable if constant; be careful around dictionaries).
1549 // - holder.AsJSPrimitiveWrapper.value.AsString.length (immutable).
1550 // - holder.AsJSPrimitiveWrapper.value.AsString[i] (immutable).
1551 // - single_character_string_table()->get().
1552
1553 if (IsFrozenElementsKind(elements_kind)) {
1554 if (!IsFixedArray(elements)) return kGaveUp;
1555 Tagged<FixedArray> elements_fixed_array = Cast<FixedArray>(elements);
1556 if (index >= static_cast<uint32_t>(elements_fixed_array->length())) {
1557 return kGaveUp;
1558 }
1559 Tagged<Object> result = elements_fixed_array->get(static_cast<int>(index));
1560 if (IsHoleyElementsKindForRead(elements_kind) &&
1561 result == ReadOnlyRoots(isolate).the_hole_value()) {
1562 return kNotPresent;
1563 }
1564 *result_out = result;
1565 return kPresent;
1566 } else if (IsDictionaryElementsKind(elements_kind)) {
1567 if (!IsNumberDictionary(elements)) return kGaveUp;
1568 // TODO(jgruber, v8:7790): Add support. Dictionary elements require racy
1569 // NumberDictionary lookups. This should be okay in general (slot iteration
1570 // depends only on the dict's capacity), but 1. we'd need to update
1571 // NumberDictionary methods to do atomic reads, and 2. the dictionary
1572 // elements case isn't very important for callers of this function.
1573 return kGaveUp;
1574 } else if (IsStringWrapperElementsKind(elements_kind)) {
1575 // In this case we don't care about the actual `elements`. All in-bounds
1576 // reads are redirected to the wrapped String.
1577
1578 Tagged<JSPrimitiveWrapper> js_value = Cast<JSPrimitiveWrapper>(holder);
1579 Tagged<String> wrapped_string = Cast<String>(js_value->value());
1580 return ConcurrentLookupIterator::TryGetOwnChar(
1581 reinterpret_cast<Tagged<String>*>(result_out), isolate, local_isolate,
1582 wrapped_string, index);
1583 } else {
1584 DCHECK(!IsFrozenElementsKind(elements_kind));
1585 DCHECK(!IsDictionaryElementsKind(elements_kind));
1586 DCHECK(!IsStringWrapperElementsKind(elements_kind));
1587 return kGaveUp;
1588 }
1589
1590 UNREACHABLE();
1591}
