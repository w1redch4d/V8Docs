25template <class IsolateT>
26Handle<ObjectBoilerplateDescription> ObjectBoilerplateDescription::New(
27 IsolateT* isolate, int boilerplate, int all_properties, int index_keys,
28 bool has_seen_proto, AllocationType allocation) {
29 DCHECK_GE(boilerplate, 0);
30 DCHECK_GE(all_properties, index_keys);
31 DCHECK_GE(index_keys, 0);
32
33 int capacity = boilerplate * kElementsPerEntry;
34 CHECK_LE(static_cast<unsigned>(capacity), kMaxCapacity);
35
36 int backing_store_size =
37 all_properties - index_keys - (has_seen_proto ? 1 : 0);
38 DCHECK_GE(backing_store_size, 0);
39
40 // Note we explicitly do NOT canonicalize to the
41 // empty_object_boilerplate_description here since `flags` may be modified
42 // even on empty descriptions.
43
44 std::optional<DisallowGarbageCollection> no_gc;
45 auto result = Cast<ObjectBoilerplateDescription>(
46 Allocate(isolate, capacity, &no_gc, allocation));
47 result->set_flags(0);
48 result->set_backing_store_size(backing_store_size);
49 MemsetTagged((*result)->RawFieldOfFirstElement(),
50 ReadOnlyRoots{isolate}.undefined_value(), capacity);
51 return result;
52}
