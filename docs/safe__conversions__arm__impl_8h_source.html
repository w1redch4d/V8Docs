23// Fast saturation to a destination type.
24template <typename Dst, typename Src>
25struct SaturateFastAsmOp {
26 static constexpr bool is_supported =
27 kEnableAsmCode && std::is_signed_v<Src> && std::is_integral_v<Dst> &&
28 std::is_integral_v<Src> &&
29 IntegerBitsPlusSign<Src>::value <= IntegerBitsPlusSign<int32_t>::value &&
30 IntegerBitsPlusSign<Dst>::value <= IntegerBitsPlusSign<int32_t>::value &&
31 !IsTypeInRangeForNumericType<Dst, Src>::value;
32
33 __attribute__((always_inline)) static Dst Do(Src value) {
34 int32_t src = value;
35 typename std::conditional<std::is_signed_v<Dst>, int32_t, uint32_t>::type
36 result;
37 if (std::is_signed_v<Dst>) {
38 asm("ssat %[dst], %[shift], %[src]"
39 : [dst] "=r"(result)
40 : [src] "r"(src), [shift] "n"(IntegerBitsPlusSign<Dst>::value <= 32
41 ? IntegerBitsPlusSign<Dst>::value
42 : 32));
43 } else {
44 asm("usat %[dst], %[shift], %[src]"
45 : [dst] "=r"(result)
46 : [src] "r"(src), [shift] "n"(IntegerBitsPlusSign<Dst>::value < 32
47 ? IntegerBitsPlusSign<Dst>::value
48 : 31));
49 }
50 return static_cast<Dst>(result);
51 }
52};
