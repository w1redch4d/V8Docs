18inline Tagged<ClearedWeakValue> ClearedValue(PtrComprCageBase cage_base) {
19 // Construct cleared weak ref value.
20 Address value;
21#ifdef V8_COMPRESS_POINTERS
22 // This is necessary to make pointer decompression computation also
23 // suitable for cleared weak references.
24 value = V8HeapCompressionScheme::DecompressTagged(
25 cage_base, kClearedWeakHeapObjectLower32);
26#else
27 value = kClearedWeakHeapObjectLower32;
28#endif
29 // The rest of the code will check only the lower 32-bits.
30 DCHECK_EQ(kClearedWeakHeapObjectLower32, static_cast<uint32_t>(value));
31 return Tagged<ClearedWeakValue>(value);
32}

45template <typename THeapObjectSlot>
46void UpdateHeapObjectReferenceSlot(THeapObjectSlot slot,
47 Tagged<HeapObject> value) {
48 static_assert(std::is_same<THeapObjectSlot, FullHeapObjectSlot>::value ||
49 std::is_same<THeapObjectSlot, HeapObjectSlot>::value,
50 "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
51 Address old_value = (*slot).ptr();
52 DCHECK(!HAS_SMI_TAG(old_value));
53 Address new_value = value.ptr();
54 DCHECK(Internals::HasHeapObjectTag(new_value));
55
56#ifdef DEBUG
57 bool weak_before = HAS_WEAK_HEAP_OBJECT_TAG(old_value);
58#endif
59
60 slot.store(Cast<HeapObjectReference>(
61 Tagged<MaybeObject>(new_value | (old_value & kWeakHeapObjectMask))));
62
63#ifdef DEBUG
64 bool weak_after = HAS_WEAK_HEAP_OBJECT_TAG((*slot).ptr());
65 DCHECK_EQ(weak_before, weak_after);
66#endif
67}
