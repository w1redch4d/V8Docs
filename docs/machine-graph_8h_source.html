21// Implements a facade on a Graph, enhancing the graph with machine-specific
22// notions, including a builder for common and machine operators, as well
23// as caching primitive constants.

25 public:
26 MachineGraph(Graph* graph, CommonOperatorBuilder* common,
27 MachineOperatorBuilder* machine)

30 machine_(machine),
31 cache_(zone()),
32 call_counts_(zone()) {}
