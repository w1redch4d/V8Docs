173uint32_t Name::EnsureRawHash() {
174 // Fast case: has hash code already been computed?
175 uint32_t field = raw_hash_field(kAcquireLoad);
176 if (IsHashFieldComputed(field)) return field;
177 // The computed hash might be stored in the forwarding table.
178 if (V8_UNLIKELY(IsForwardingIndex(field))) {
179 return GetRawHashFromForwardingTable(field);
180 }
181 // Slow case: compute hash code and set it. Has to be a string.
182 return Cast<String>(this)->ComputeAndSetRawHash();
183}

212void Name::set_raw_hash_field_if_empty(uint32_t hash) {
213 uint32_t field_value = kEmptyHashField;
214 bool result = raw_hash_field_.compare_exchange_strong(field_value, hash);
215 USE(result);
216 // CAS can only fail if the string is shared or we use the forwarding table
217 // for all strings and the hash was already set (by another thread) or it is
218 // a forwarding index (that overwrites the previous hash).
219 // In all cases we don't want overwrite the old value, so we don't handle the
220 // failure case.
221 DCHECK_IMPLIES(!result, (Cast<String>(this)->IsShared() ||
222 v8_flags.always_use_string_forwarding_table) &&
223 (field_value == hash || IsForwardingIndex(hash)));
224}

226uint32_t Name::hash() const {
227 uint32_t field = raw_hash_field(kAcquireLoad);
228 if (V8_UNLIKELY(!IsHashFieldComputed(field))) {
229 DCHECK(IsForwardingIndex(field));
230 return HashBits::decode(GetRawHashFromForwardingTable(field));
231 }
232 return HashBits::decode(field);
233}

235bool Name::TryGetHash(uint32_t* hash) const {
236 uint32_t field = raw_hash_field(kAcquireLoad);
237 if (IsHashFieldComputed(field)) {
238 *hash = HashBits::decode(field);
239 return true;
240 }
241 if (V8_UNLIKELY(IsForwardingIndex(field))) {
242 *hash = HashBits::decode(GetRawHashFromForwardingTable(field));
243 return true;
244 }
245 return false;
246}
