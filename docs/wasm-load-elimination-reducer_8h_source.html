143 void Invalidate(const StructSetOp& set) {
144 // This is like LateLoadElimination's {InvalidateAtOffset}, but based
145 // on Wasm types instead of tracked JS maps.
146 int offset = field_offset(set.type, set.field_index);
147 auto offset_keys = offset_keys_.find(offset);
148 if (offset_keys == offset_keys_.end()) return;
149 for (auto it = offset_keys->second.begin();
150 it != offset_keys->second.end();) {
151 Key key = *it;
152 DCHECK_EQ(offset, key.data().mem.offset);
153 OpIndex base = key.data().mem.base;
154
155 // If the base is guaranteed non-aliasing, we don't need to clear any
156 // other entries. Any existing entry for this base will be overwritten
157 // by {Insert(set)}.
158 if (non_aliasing_objects_.Get(base)) {
159 ++it;
160 continue;
161 }
162
163 if (TypesUnrelated(set.type_index, key.data().mem.type_index)) {
164 ++it;
165 continue;
166 }
167
168 it = offset_keys->second.RemoveAt(it);
169 Set(key, OpIndex::Invalid());
170 }
171 }

173 // Invalidates all Keys that are not known as non-aliasing.
174 enum class EntriesWithOffsets { kInvalidate, kKeep };
175 template <EntriesWithOffsets offsets = EntriesWithOffsets::kInvalidate>
176 void InvalidateMaybeAliasing() {
177 // We find current active keys through {base_keys_} so that we can bail out
178 // for whole buckets non-aliasing buckets (if we had gone through
179 // {offset_keys_} instead, then for each key we would've had to check
180 // whether it was non-aliasing or not).
181 for (auto& base_keys : base_keys_) {
182 OpIndex base = base_keys.first;
183 if (non_aliasing_objects_.Get(base)) continue;
184 if constexpr (offsets == EntriesWithOffsets::kInvalidate) {
185 for (auto it = base_keys.second.with_offsets.begin();
186 it != base_keys.second.with_offsets.end();) {
187 Key key = *it;
188 if (key.data().mem.mutability == false) {
189 ++it;
190 continue;
191 }
192 // It's important to remove with RemoveAt before Setting the key to
193 // invalid, otherwise OnKeyChange will remove {key} from {base_keys},
194 // which will invalidate {it}.
195 it = base_keys.second.with_offsets.RemoveAt(it);
196 Set(key, OpIndex::Invalid());
197 }
198 }
199 }

276 void Insert(OpIndex base, int32_t offset, wasm::ModuleTypeIndex type_index,
277 uint8_t size, bool mutability, OpIndex value) {
278 DCHECK_EQ(base, ResolveBase(base));
279
280 WasmMemoryAddress mem{base, offset, type_index, size, mutability};
281 auto existing_key = all_keys_.find(mem);
282 if (existing_key != all_keys_.end()) {
283 if (mutability) {
284 Set(existing_key->second, value);
285 } else {
286 SetNoNotify(existing_key->second, value);
287 }
288 return;
289 }
290
291 // Creating a new key.
292 Key key = NewKey({mem});
293 all_keys_.insert({mem, key});
294 if (mutability) {
295 Set(key, value);
296 } else {
297 // Call `SetNoNotify` to avoid calls to `OnNewKey` and `OnValueChanged`.
298 SetNoNotify(key, value);
299 }
300 }

303 OpIndex ResolveBase(OpIndex base) {
304 while (true) {
305 if (replacements_[base] != OpIndex::Invalid()) {
306 base = replacements_[base];
307 continue;
308 }
309 Operation& op = graph_.Get(base);
310 if (AssertNotNullOp* check = op.TryCast<AssertNotNullOp>()) {
311 base = check->object();
312 continue;
313 }
314 if (WasmTypeCastOp* cast = op.TryCast<WasmTypeCastOp>()) {
315 base = cast->object();
316 continue;
317 }
318 break; // Terminate if nothing happened.
319 }
320 return base;
321 }

323 void AddKeyInBaseOffsetMaps(Key key) {
324 // Inserting in {base_keys_}.
325 OpIndex base = key.data().mem.base;
326 auto base_keys = base_keys_.find(base);
327 if (base_keys != base_keys_.end()) {
328 base_keys->second.with_offsets.PushFront(key);
329 } else {
330 BaseData data;
331 data.with_offsets.PushFront(key);
332 base_keys_.insert({base, std::move(data)});
333 }
334
335 // Inserting in {offset_keys_}.
336 int offset = key.data().mem.offset;
337 auto offset_keys = offset_keys_.find(offset);
338 if (offset_keys != offset_keys_.end()) {
339 offset_keys->second.PushFront(key);
340 } else {
341 v8::base::DoublyThreadedList<Key, OffsetListTraits> list;
342 list.PushFront(key);
343 offset_keys_.insert({offset, std::move(list)});
344 }
345 }
346
347 void RemoveKeyFromBaseOffsetMaps(Key key) {
348 // Removing from {base_keys_}.
349 v8::base::DoublyThreadedList<Key, BaseListTraits>::Remove(key);
350 v8::base::DoublyThreadedList<Key, OffsetListTraits>::Remove(key);
351 }

385 WasmLoadEliminationAnalyzer(PipelineData* data, Graph& graph,
386 Zone* phase_zone)
387 : graph_(graph),
388 phase_zone_(phase_zone),
389 replacements_(graph.op_id_count(), phase_zone, &graph),
390 non_aliasing_objects_(phase_zone),
391 memory_(data, phase_zone, non_aliasing_objects_, replacements_, graph_),
392 block_to_snapshot_mapping_(graph.block_count(), phase_zone),
393 predecessor_alias_snapshots_(phase_zone),
394 predecessor_memory_snapshots_(phase_zone) {}
395
396 void Run() {
397 LoopFinder loop_finder(phase_zone_, &graph_);
398 AnalyzerIterator iterator(phase_zone_, graph_, loop_finder);
399
400 bool compute_start_snapshot = true;
401 while (iterator.HasNext()) {
402 const Block* block = iterator.Next();
403
404 ProcessBlock(*block, compute_start_snapshot);
405 compute_start_snapshot = true;
406
407 // Consider re-processing for loops.
408 if (const GotoOp* last = block->LastOperation(graph_).TryCast<GotoOp>()) {
409 if (last->destination->IsLoop() &&
410 last->destination->LastPredecessor() == block) {
411 const Block* loop_header = last->destination;
412 // {block} is the backedge of a loop. We recompute the loop header's
413 // initial snapshots, and if they differ from its original snapshot,
414 // then we revisit the loop.
415 if (BeginBlock<true>(loop_header)) {
416 // We set the snapshot of the loop's 1st predecessor to the newly
417 // computed snapshot. It's not quite correct, but this predecessor
418 // is guaranteed to end with a Goto, and we are now visiting the
419 // loop, which means that we don't really care about this
420 // predecessor anymore.
421 // The reason for saving this snapshot is to prevent infinite
422 // looping, since the next time we reach this point, the backedge
423 // snapshot could still invalidate things from the forward edge
424 // snapshot. By restricting the forward edge snapshot, we prevent
425 // this.
426 const Block* loop_1st_pred =
427 loop_header->LastPredecessor()->NeighboringPredecessor();
428 FinishBlock(loop_1st_pred);
429 // And we start a new fresh snapshot from this predecessor.
430 auto pred_snapshots =
431 block_to_snapshot_mapping_[loop_1st_pred->index()];
432 non_aliasing_objects_.StartNewSnapshot(
433 pred_snapshots->alias_snapshot);
434 memory_.StartNewSnapshot(pred_snapshots->memory_snapshot);
435
436 iterator.MarkLoopForRevisit();
437 compute_start_snapshot = false;
438 } else {
439 SealAndDiscard();
440 }
441 }
442 }
443 }
444 }

558void WasmLoadEliminationAnalyzer::ProcessBlock(const Block& block,
559 bool compute_start_snapshot) {
560 if (compute_start_snapshot) {
561 BeginBlock(&block);
562 }
563 if (block.IsLoop() && BackedgeHasSnapshot(block)) {
564 // Update the associated snapshot for the forward edge with the merged
565 // snapshot information from the forward- and backward edge.
566 // This will make sure that when evaluating whether a loop needs to be
567 // revisited, the inner loop compares the merged state with the backedge
568 // preventing us from exponential revisits for loops where the backedge
569 // invalidates loads which are eliminatable on the forward edge.
570 StoreLoopSnapshotInForwardPredecessor(block);
571 }
572
573 for (OpIndex op_idx : graph_.OperationIndices(block)) {
574 Operation& op = graph_.Get(op_idx);
575 if (ShouldSkipOptimizationStep()) continue;
576 if (ShouldSkipOperation(op)) continue;
577 switch (op.opcode) {
578 case Opcode::kStructGet:
579 ProcessStructGet(op_idx, op.Cast<StructGetOp>());
580 break;
581 case Opcode::kStructSet:
582 ProcessStructSet(op_idx, op.Cast<StructSetOp>());
583 break;
584 case Opcode::kArrayLength:
585 ProcessArrayLength(op_idx, op.Cast<ArrayLengthOp>());
586 break;
587 case Opcode::kWasmAllocateArray:
588 ProcessWasmAllocateArray(op_idx, op.Cast<WasmAllocateArrayOp>());
589 break;
590 case Opcode::kStringAsWtf16:
591 ProcessStringAsWtf16(op_idx, op.Cast<StringAsWtf16Op>());
592 break;
593 case Opcode::kStringPrepareForGetCodeUnit:
594 ProcessStringPrepareForGetCodeUnit(
595 op_idx, op.Cast<StringPrepareForGetCodeUnitOp>());
596 break;
597 case Opcode::kAnyConvertExtern:
598 ProcessAnyConvertExtern(op_idx, op.Cast<AnyConvertExternOp>());
599 break;
600 case Opcode::kAssertNotNull:
601 // TODO(14108): We'll probably want to handle WasmTypeCast as
602 // a "load-like" instruction too, to eliminate repeated casts.
603 ProcessAssertNotNull(op_idx, op.Cast<AssertNotNullOp>());
604 break;
605 case Opcode::kArraySet:
606 break;
607 case Opcode::kAllocate:
608 // Create new non-alias.
609 ProcessAllocate(op_idx, op.Cast<AllocateOp>());
610 break;
611 case Opcode::kCall:
612 // Invalidate state (+ maybe invalidate aliases).
613 ProcessCall(op_idx, op.Cast<CallOp>());
614 break;
615 case Opcode::kPhi:
616 // Invalidate aliases.
617 ProcessPhi(op_idx, op.Cast<PhiOp>());
618 break;
619 case Opcode::kLoad:
620 // Atomic loads have the "can_write" bit set, because they make
621 // writes on other threads visible. At any rate, we have to
622 // explicitly skip them here.
623 case Opcode::kStore:
624 // We rely on having no raw "Store" operations operating on Wasm
625 // objects at this point in the pipeline.
626 // TODO(jkummerow): Is there any way to DCHECK that?
627 case Opcode::kAssumeMap:
628 case Opcode::kCatchBlockBegin:
629 case Opcode::kRetain:
630 case Opcode::kDidntThrow:
631 case Opcode::kCheckException:
632 case Opcode::kAtomicRMW:
633 case Opcode::kAtomicWord32Pair:
634 case Opcode::kMemoryBarrier:
635 case Opcode::kJSStackCheck:
636 case Opcode::kWasmStackCheck:
637 case Opcode::kSimd128LaneMemory:
638 case Opcode::kGlobalSet:
639 case Opcode::kParameter:
640 // We explicitly break for those operations that have can_write effects
641 // but don't actually write, or cannot interfere with load elimination.
642 break;
643
644 case Opcode::kWordBinop:
645 // A WordBinop should never invalidate aliases (since the only time when
646 // it should take a non-aliasing object as input is for Smi checks).
647 DcheckWordBinop(op_idx, op.Cast<WordBinopOp>());
648 break;
649
650 case Opcode::kFrameState:
651 case Opcode::kDeoptimizeIf:
652 case Opcode::kComparison:
653 case Opcode::kTrapIf:
654 // We explicitly break for these opcodes so that we don't call
655 // InvalidateAllNonAliasingInputs on their inputs, since they don't
656 // really create aliases. (and also, they don't write so it's
657 // fine to break)
658 DCHECK(!op.Effects().can_write());
659 break;
660
661 case Opcode::kDeoptimize:
662 case Opcode::kReturn:
663 // We explicitly break for these opcodes so that we don't call
664 // InvalidateAllNonAliasingInputs on their inputs, since they are block
665 // terminators without successors, meaning that it's not useful for the
666 // rest of the analysis to invalidate anything here.
667 DCHECK(op.IsBlockTerminator() && SuccessorBlocks(op).empty());
668 break;
669
670 default:
671 // Operations that `can_write` should invalidate the state. All such
672 // operations should be already handled above, which means that we don't
673 // need a `if (can_write) { Invalidate(); }` here.
674 CHECK(!op.Effects().can_write());
675
676 // Even if the operation doesn't write, it could create an alias to its
677 // input by returning it. This happens for instance in Phis and in
678 // Change (although ChangeOp is already handled earlier by calling
679 // ProcessChange). We are conservative here by calling
680 // InvalidateAllNonAliasingInputs for all operations even though only
681 // few can actually create aliases to fresh allocations, the reason
682 // being that missing such a case would be a security issue, and it
683 // should be rare for fresh allocations to be used outside of
684 // Call/Store/Load/Change anyways.
685 InvalidateAllNonAliasingInputs(op);
686
687 break;
688 }
689 }
690
691 FinishBlock(&block);
692}

694// Returns true if replacing a load with a RegisterRepresentation
695// {expected_reg_rep} and size {in_memory_size} with an
696// operation with RegisterRepresentation {actual} is valid. For instance,
697// replacing an operation that returns a Float64 by one that returns a Word64 is
698// not valid. Similarly, replacing a Tagged with an untagged value is probably
699// not valid because of the GC.
700
701bool RepIsCompatible(RegisterRepresentation actual,
702 RegisterRepresentation expected_reg_repr,
703 uint8_t in_memory_size) {
704 if (in_memory_size !=
705 MemoryRepresentation::FromRegisterRepresentation(actual, true)
706 .SizeInBytes()) {
707 // The replacement was truncated when being stored or should be truncated
708 // (or sign-extended) during the load. Since we don't have enough
709 // truncations operators in Turboshaft (eg, we don't have Int32 to Int8
710 // truncation), we just prevent load elimination in this case.
711
712 // TODO(jkummerow): support eliminating repeated loads of the same i8/i16
713 // field.
714 return false;
715 }
716
717 return expected_reg_repr == actual;
718}

974template <bool for_loop_revisit>
975bool WasmLoadEliminationAnalyzer::BeginBlock(const Block* block) {
976 DCHECK_IMPLIES(
977 for_loop_revisit,
978 block->IsLoop() &&
979 block_to_snapshot_mapping_[block->LastPredecessor()->index()]
980 .has_value());
981
982 // Collect the snapshots of all predecessors.
983 {
984 predecessor_alias_snapshots_.clear();
985 predecessor_memory_snapshots_.clear();
986 for (const Block* p : block->PredecessorsIterable()) {
987 auto pred_snapshots = block_to_snapshot_mapping_[p->index()];
988 // When we visit the loop for the first time, the loop header hasn't
989 // been visited yet, so we ignore it.
990 DCHECK_IMPLIES(!pred_snapshots.has_value(),
991 block->IsLoop() && block->LastPredecessor() == p);
992 if (!pred_snapshots.has_value()) {
993 DCHECK(!for_loop_revisit);
994 continue;
995 }
996 // Note that the backedge snapshot of an inner loop in kFirstVisit will
997 // also be taken into account if we are in the kSecondVisit of an outer
998 // loop. The data in the backedge snapshot could be out-dated, but if it
999 // is, then it's fine: if the backedge of the outer-loop was more
1000 // restrictive than its forward incoming edge, then the forward incoming
1001 // edge of the inner loop should reflect this restriction.
1002 predecessor_alias_snapshots_.push_back(pred_snapshots->alias_snapshot);
1003 predecessor_memory_snapshots_.push_back(pred_snapshots->memory_snapshot);
1004 }
1005 }
1006
1007 // Note that predecessors are in reverse order, which means that the backedge
1008 // is at offset 0.
1009 constexpr int kBackedgeOffset = 0;
1010 constexpr int kForwardEdgeOffset = 1;
1011
1012 bool loop_needs_revisit = false;
1013 // Start a new snapshot for this block by merging information from
1014 // predecessors.
1015 auto merge_aliases = [&](AliasKey key,
1016 base::Vector<const bool> predecessors) -> bool {
1017 if (for_loop_revisit && predecessors[kForwardEdgeOffset] &&
1018 !predecessors[kBackedgeOffset]) {
1019 // The backedge doesn't think that {key} is no-alias, but the loop
1020 // header previously thought it was --> need to revisit.
1021 loop_needs_revisit = true;
1022 }
1023 return base::all_of(predecessors);
1024 };
1025 non_aliasing_objects_.StartNewSnapshot(
1026 base::VectorOf(predecessor_alias_snapshots_), merge_aliases);
1027
1028 // Merging for {memory_} means setting values to Invalid unless all
1029 // predecessors have the same value.
1030 // TODO(dmercadier): we could insert of Phis during the pass to merge existing
1031 // information. This is a bit hard, because we are currently in an analyzer
1032 // rather than a reducer. Still, we could "prepare" the insertion now and then
1033 // really insert them during the Reduce phase of the CopyingPhase.
1034 auto merge_memory = [&](MemoryKey key,
1035 base::Vector<const OpIndex> predecessors) -> OpIndex {
1036 if (for_loop_revisit && predecessors[kForwardEdgeOffset].valid() &&
1037 predecessors[kBackedgeOffset] != predecessors[kForwardEdgeOffset]) {
1038 // {key} had a value in the loop header, but the backedge and the forward
1039 // edge don't agree on its value, which means that the loop invalidated
1040 // some memory data, and thus needs to be revisited.
1041 loop_needs_revisit = true;
1042 }
1043 return base::all_equal(predecessors) ? predecessors[0] : OpIndex::Invalid();
1044 };
1045 memory_.StartNewSnapshot(base::VectorOf(predecessor_memory_snapshots_),
1046 merge_memory);
1047
1048 if (block->IsLoop()) return loop_needs_revisit;
1049 return false;
1050}
