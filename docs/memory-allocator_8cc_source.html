31void DeleteMemoryChunk(MutablePageMetadata* metadata) {
32 DCHECK(metadata->reserved_memory()->IsReserved());
33 DCHECK(!metadata->Chunk()->InReadOnlySpace());
34 // The Metadata contains a VirtualMemory reservation and the destructor will
35 // release the MemoryChunk.
36 DiscardSealedMemoryScope discard_scope("Deleting a memory chunk");
37 if (metadata->IsLargePage()) {
38 delete reinterpret_cast<LargePageMetadata*>(metadata);
39 } else {
40 delete reinterpret_cast<PageMetadata*>(metadata);
41 }
42}

132Address MemoryAllocator::AllocateAlignedMemory(
133 size_t chunk_size, size_t area_size, size_t alignment,
134 AllocationSpace space, Executability executable, void* hint,
135 VirtualMemory* controller) {
136 DCHECK_EQ(space == CODE_SPACE || space == CODE_LO_SPACE,
137 executable == EXECUTABLE);
138 v8::PageAllocator* page_allocator = this->page_allocator(space);
139 DCHECK_LT(area_size, chunk_size);
140
141 PageAllocator::Permission permissions =
142 executable == EXECUTABLE
143 ? MutablePageMetadata::GetCodeModificationPermission()
144 : PageAllocator::kReadWrite;
145 VirtualMemory reservation(page_allocator, chunk_size, hint, alignment,
146 permissions);
147 if (!reservation.IsReserved()) return HandleAllocationFailure(executable);
148
149 // We cannot use the last chunk in the address space because we would
150 // overflow when comparing top and limit if this chunk is used for a
151 // linear allocation area.
152 if ((reservation.address() + static_cast<Address>(chunk_size)) == 0u) {
153 CHECK(!reserved_chunk_at_virtual_memory_limit_);
154 reserved_chunk_at_virtual_memory_limit_ = std::move(reservation);
155 CHECK(reserved_chunk_at_virtual_memory_limit_);
156
157 // Retry reserve virtual memory.
158 reservation =
159 VirtualMemory(page_allocator, chunk_size, hint, alignment, permissions);
160 if (!reservation.IsReserved()) return HandleAllocationFailure(executable);
161 }
162
163 Address base = reservation.address();
164
165 if (executable == EXECUTABLE) {
166 ThreadIsolation::RegisterJitPage(base, chunk_size);
167 }
168
169 UpdateAllocatedSpaceLimits(base, base + chunk_size, executable);
170
171 *controller = std::move(reservation);
172 return base;
173}

175Address MemoryAllocator::HandleAllocationFailure(Executability executable) {
176 Heap* heap = isolate_->heap();
177 if (!heap->deserialization_complete()) {
178 heap->FatalProcessOutOfMemory(
179 executable == EXECUTABLE
180 ? "Executable MemoryChunk allocation failed during deserialization."
181 : "MemoryChunk allocation failed during deserialization.");
182 }
183 return kNullAddress;

186size_t MemoryAllocator::ComputeChunkSize(size_t area_size,
187 AllocationSpace space) {
188 //
189 // +----------------------------+<- base aligned at MemoryChunk::kAlignment
190 // | Header |
191 // +----------------------------+<- area_start_ (base + area_start_)
192 // | Area |
193 // +----------------------------+<- area_end_ (area_start + area_size)
194 // | Committed but not used |
195 // +----------------------------+<- base + chunk_size
196 //
197
198 return ::RoundUp(
199 MemoryChunkLayout::ObjectStartOffsetInMemoryChunk(space) + area_size,
200 GetCommitPageSize());
201}

203std::optional<MemoryAllocator::MemoryChunkAllocationResult>
204MemoryAllocator::AllocateUninitializedChunkAt(BaseSpace* space,
205 size_t area_size,
206 Executability executable,
207 Address hint,
208 PageSize page_size) {
209#ifndef V8_COMPRESS_POINTERS
210 // When pointer compression is enabled, spaces are expected to be at a
211 // predictable address (see mkgrokdump) so we don't supply a hint and rely on
212 // the deterministic behaviour of the BoundedPageAllocator.
213 if (hint == kNullAddress) {
214 hint = reinterpret_cast<Address>(
215 AlignedAddress(isolate_->heap()->GetRandomMmapAddr(),
216 MemoryChunk::GetAlignmentForAllocation()));
217 }
218#endif
219
220 VirtualMemory reservation;
221 size_t chunk_size = ComputeChunkSize(area_size, space->identity());
222 DCHECK_EQ(chunk_size % GetCommitPageSize(), 0);
223
224 Address base = AllocateAlignedMemory(
225 chunk_size, area_size, MemoryChunk::GetAlignmentForAllocation(),
226 space->identity(), executable, reinterpret_cast<void*>(hint),
227 &reservation);
228 if (base == kNullAddress) return {};
229
230 size_ += reservation.size();
231
232 // Update executable memory size.
233 if (executable == EXECUTABLE) {
234 size_executable_ += reservation.size();
235 }
236
237 if (heap::ShouldZapGarbage()) {
238 if (executable == EXECUTABLE) {
239 CodePageMemoryModificationScopeForDebugging memory_write_scope(
240 isolate_->heap(), &reservation,
241 base::AddressRegion(base, chunk_size));
242 heap::ZapBlock(base, chunk_size, kZapValue);
243 } else {
244 DCHECK_EQ(executable, NOT_EXECUTABLE);
245 // Zap both page header and object area at once. No guard page in-between.
246 heap::ZapBlock(base, chunk_size, kZapValue);
247 }
248 }
249
250 LOG(isolate_,
251 NewEvent("MemoryChunk", reinterpret_cast<void*>(base), chunk_size));
252
253 Address area_start = base + MemoryChunkLayout::ObjectStartOffsetInMemoryChunk(
254 space->identity());
255 Address area_end = area_start + area_size;
256
257 return MemoryChunkAllocationResult{
258 reinterpret_cast<void*>(base), nullptr, chunk_size, area_start, area_end,
259 std::move(reservation),
260 };
261}

263void MemoryAllocator::PartialFreeMemory(MemoryChunkMetadata* chunk,
264 Address start_free,
265 size_t bytes_to_free,
266 Address new_area_end) {
267 VirtualMemory* reservation = chunk->reserved_memory();
268 DCHECK(reservation->IsReserved());
269 chunk->set_size(chunk->size() - bytes_to_free);
270 chunk->set_area_end(new_area_end);
271 if (chunk->Chunk()->IsFlagSet(MemoryChunk::IS_EXECUTABLE)) {
272 // Add guard page at the end.
273 size_t page_size = GetCommitPageSize();
274 DCHECK_EQ(0, chunk->area_end() % static_cast<Address>(page_size));
275 DCHECK_EQ(chunk->ChunkAddress() + chunk->size(), chunk->area_end());
276
277 if ((V8_HEAP_USE_PTHREAD_JIT_WRITE_PROTECT ||
278 V8_HEAP_USE_BECORE_JIT_WRITE_PROTECT) &&
279 !isolate_->jitless()) {
280 DCHECK(isolate_->RequiresCodeRange());
281 DiscardSealedMemoryScope discard_scope("Partially free memory.");
282 reservation->DiscardSystemPages(chunk->area_end(), page_size);
283 } else {
284 CHECK(reservation->SetPermissions(chunk->area_end(), page_size,
285 PageAllocator::kNoAccess));
286 }
287 }
288 // On e.g. Windows, a reservation may be larger than a page and releasing
289 // partially starting at |start_free| will also release the potentially
290 // unused part behind the current page.
291 const size_t released_bytes = reservation->Release(start_free);
292 DCHECK_GE(size_, released_bytes);
293 size_ -= released_bytes;
294}

305 Executability executable) {
306 MemoryChunk* chunk = chunk_metadata->Chunk();
307 DCHECK(!chunk->IsFlagSet(MemoryChunk::UNREGISTERED));
308 VirtualMemory* reservation = chunk_metadata->reserved_memory();
309 const size_t size =

337void MemoryAllocator::FreeReadOnlyPage(ReadOnlyPageMetadata* chunk) {
338 DCHECK(!chunk->Chunk()->IsFlagSet(MemoryChunk::PRE_FREED));
339 LOG(isolate_, DeleteEvent("MemoryChunk", chunk));
340
341 UnregisterSharedMemoryChunk(chunk);
342
343 v8::PageAllocator* allocator = page_allocator(RO_SPACE);
344 VirtualMemory* reservation = chunk->reserved_memory();
345 if (reservation->IsReserved()) {
346 reservation->Free();
347 } else {
348 // Only read-only pages can have a non-initialized reservation object. This
349 // happens when the pages are remapped to multiple locations and where the
350 // reservation would therefore be invalid.
351 FreeMemoryRegion(allocator, chunk->ChunkAddress(),
352 RoundUp(chunk->size(), allocator->AllocatePageSize()));
353 }
354
355 delete chunk;
356}

379void MemoryAllocator::Free(MemoryAllocator::FreeMode mode,
380 MutablePageMetadata* chunk_metadata) {
381 MemoryChunk* chunk = chunk_metadata->Chunk();
382 RecordMemoryChunkDestroyed(chunk);
383
384 switch (mode) {
385 case FreeMode::kImmediately:
386 PreFreeMemory(chunk_metadata);
387 PerformFreeMemory(chunk_metadata);
388 break;
389 case FreeMode::kPostpone:
390 PreFreeMemory(chunk_metadata);
391 // Record page to be freed later.
392 queued_pages_to_be_freed_.push_back(chunk_metadata);
393 break;
394 case FreeMode::kPool:
395 DCHECK_EQ(chunk_metadata->size(),
396 static_cast<size_t>(MutablePageMetadata::kPageSize));
397 DCHECK_EQ(chunk->executable(), NOT_EXECUTABLE);
398 PreFreeMemory(chunk_metadata);
399 // The chunks added to this queue will be cached until memory reducing GC.
400 pool()->Add(chunk_metadata);
401 break;
402 }
403}

405PageMetadata* MemoryAllocator::AllocatePage(
406 MemoryAllocator::AllocationMode alloc_mode, Space* space,
407 Executability executable) {
408 const size_t size =
409 MemoryChunkLayout::AllocatableMemoryInMemoryChunk(space->identity());
410 std::optional<MemoryChunkAllocationResult> chunk_info;
411 if (alloc_mode == AllocationMode::kUsePool) {
412 DCHECK_EQ(executable, NOT_EXECUTABLE);
413 chunk_info = AllocateUninitializedPageFromPool(space);
414 }
415
416 if (!chunk_info) {
417 chunk_info =
418 AllocateUninitializedChunk(space, size, executable, PageSize::kRegular);
419 }
420
421 if (!chunk_info) return nullptr;
422
423 PageMetadata* metadata;
424 if (chunk_info->optional_metadata) {
425 metadata = new (chunk_info->optional_metadata) PageMetadata(
426 isolate_->heap(), space, chunk_info->size, chunk_info->area_start,
427 chunk_info->area_end, std::move(chunk_info->reservation));
428 } else {
429 metadata = new PageMetadata(isolate_->heap(), space, chunk_info->size,
430 chunk_info->area_start, chunk_info->area_end,
431 std::move(chunk_info->reservation));
432 }
433 MemoryChunk* chunk;
434 MemoryChunk::MainThreadFlags flags = metadata->InitialFlags(executable);
435 if (v8_flags.black_allocated_pages && space->identity() != NEW_SPACE &&
436 space->identity() != NEW_LO_SPACE &&
437 isolate_->heap()->incremental_marking()->black_allocation()) {
438 // Disable the write barrier for objects pointing to this page. We don't
439 // need to trigger the barrier for pointers to old black-allocated pages,
440 // since those are never considered for evacuation. However, we have to
441 // keep the old->shared remembered set across multiple GCs, so those
442 // pointers still need to be recorded.
443 if (!IsAnySharedSpace(space->identity())) {
444 flags &= ~MemoryChunk::POINTERS_TO_HERE_ARE_INTERESTING;
445 }
446 // And mark the page as black allocated.
447 flags |= MemoryChunk::BLACK_ALLOCATED;
448 }
449 if (executable) {
450 RwxMemoryWriteScope scope("Initialize a new MemoryChunk.");
451 chunk = new (chunk_info->chunk) MemoryChunk(flags, metadata);
452 } else {
453 chunk = new (chunk_info->chunk) MemoryChunk(flags, metadata);
454 }
455
456#ifdef DEBUG
457 if (chunk->executable()) RegisterExecutableMemoryChunk(metadata);
458#endif // DEBUG
459
460 space->InitializePage(metadata);
461 RecordMemoryChunkCreated(chunk);
462 return metadata;
463}

530std::optional<MemoryAllocator::MemoryChunkAllocationResult>
531MemoryAllocator::AllocateUninitializedPageFromPool(Space* space) {
532 MemoryChunkMetadata* chunk_metadata = pool()->TryGetPooled();
533 if (chunk_metadata == nullptr) return {};
534 const int size = MutablePageMetadata::kPageSize;
535 const Address start = chunk_metadata->ChunkAddress();
536 const Address area_start =
537 start +
538 MemoryChunkLayout::ObjectStartOffsetInMemoryChunk(space->identity());
539 const Address area_end = start + size;
540 // Pooled pages are always regular data pages.
541 DCHECK_NE(CODE_SPACE, space->identity());
542 DCHECK_NE(TRUSTED_SPACE, space->identity());
543 VirtualMemory reservation(data_page_allocator(), start, size);
544 if (heap::ShouldZapGarbage()) {
545 heap::ZapBlock(start, size, kZapValue);
546 }
547
548 size_ += size;
549 return MemoryChunkAllocationResult{
550 chunk_metadata->Chunk(), chunk_metadata, size, area_start, area_end,
551 std::move(reservation),
552 };
553}

563bool MemoryAllocator::SetPermissionsOnExecutableMemoryChunk(VirtualMemory* vm,
564 Address start,
565 size_t chunk_size) {
566 // All addresses and sizes must be aligned to the commit page size.
567 DCHECK(IsAligned(start, GetCommitPageSize()));
568 DCHECK_EQ(0, chunk_size % GetCommitPageSize());

582const MemoryChunk* MemoryAllocator::LookupChunkContainingAddress(
583 Address addr) const {
584 // All threads should be either parked or in a safepoint whenever this method
585 // is called, thus pages cannot be allocated or freed at the same time and a
586 // mutex is not required here.
587 // As the address may not correspond to a valid heap object, the chunk we
588 // obtain below is not necessarily a valid chunk.
589 MemoryChunk* chunk = MemoryChunk::FromAddress(addr);
590 // Check if it corresponds to a known normal or large page.
591 if (auto it = normal_pages_.find(chunk); it != normal_pages_.end()) {
592 // The chunk is a normal page.
593 // auto* normal_page = PageMetadata::cast(chunk);
594 DCHECK_LE((*it)->address(), addr);
595 if (chunk->Metadata()->Contains(addr)) return chunk;
596 } else if (auto it = large_pages_.upper_bound(chunk);
597 it != large_pages_.begin()) {
598 // The chunk could be inside a large page.
599 DCHECK_IMPLIES(it != large_pages_.end(), addr < (*it)->address());
600 auto* large_page_chunk = *std::next(it, -1);
601 DCHECK_NOT_NULL(large_page_chunk);
602 DCHECK_LE(large_page_chunk->address(), addr);
603 if (large_page_chunk->Metadata()->Contains(addr)) return large_page_chunk;
604 }
605 // Not found in any page.
606 return nullptr;
607}
