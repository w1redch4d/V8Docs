72bool EphemeronUpdater(Ephemeron slot_in, Ephemeron* slot_out) {
73 Tagged<HeapObject> key = slot_in.key;
74 Tagged<HeapObject> value = slot_in.value;
75 Tagged<HeapObject> forwarded_key = ForwardingAddress(key);
76 Tagged<HeapObject> forwarded_value = ForwardingAddress(value);
77
78 if (!forwarded_key.is_null() && !forwarded_value.is_null()) {
79 *slot_out = Ephemeron{forwarded_key, forwarded_value};

82

106template <typename TSlot>
107void UpdateWeakReferencesHelper(
108 WeakObjects::WeakObjectWorklist<TSlot>& weak_references) {
109 weak_references.Update([](TSlot slot_in, TSlot* slot_out) -> bool {
110 Tagged<HeapObject> heap_obj = slot_in.heap_object;
111 Tagged<HeapObject> forwarded = ForwardingAddress(heap_obj);
112
113 if (!forwarded.is_null()) {
114 ptrdiff_t distance_to_slot =
115 slot_in.slot.address() - slot_in.heap_object.ptr();
116 Address new_slot = forwarded.ptr() + distance_to_slot;
117 slot_out->heap_object = forwarded;
118 slot_out->slot = typename TSlot::SlotType(new_slot);
119 return true;
120 }
121
122 return false;
123 });
124}
125} // anonymous namespace
