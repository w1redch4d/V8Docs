63// Reads a length-prefixed string, checking that it is within bounds. Returns
64// the offset of the string, and the length as an out parameter.
65inline WireBytesRef consume_string(Decoder* decoder,
66 unibrow::Utf8Variant grammar,
67 const char* name, ITracer* tracer) {
68 if (tracer) {
69 tracer->Description(name);
70 tracer->Description(" ");
71 }
72 uint32_t length = decoder->consume_u32v("length", tracer);
73 if (tracer) {
74 tracer->Description(": ");
75 tracer->Description(length);
76 tracer->NextLine();
77 }
78 uint32_t offset = decoder->pc_offset();
79 const uint8_t* string_start = decoder->pc();
80 // Consume bytes before validation to guarantee that the string is not oob.
81 if (length > 0) {
82 if (tracer) {
83 tracer->Bytes(decoder->pc(), length);
84 tracer->Description(name);
85 tracer->Description(": ");
86 tracer->Description(reinterpret_cast<const char*>(decoder->pc()), length);
87 tracer->NextLine();
88 }
89 decoder->consume_bytes(length, name);
90 if (decoder->ok()) {
91 switch (grammar) {
92 case unibrow::Utf8Variant::kLossyUtf8:
93 break;
94 case unibrow::Utf8Variant::kUtf8:
95 if (!unibrow::Utf8::ValidateEncoding(string_start, length)) {
96 decoder->errorf(string_start, "%s: no valid UTF-8 string", name);
97 }
98 break;
99 case unibrow::Utf8Variant::kWtf8:
100 if (!unibrow::Wtf8::ValidateEncoding(string_start, length)) {
101 decoder->errorf(string_start, "%s: no valid WTF-8 string", name);
102 }
103 break;
104 case unibrow::Utf8Variant::kUtf8NoTrap:
105 UNREACHABLE();
106 }
107 }
108 }
109 return {offset, decoder->failed() ? 0 : length};
110}

123inline SectionCode IdentifyUnknownSectionInternal(Decoder* decoder,
124 ITracer* tracer) {
125 WireBytesRef string = consume_utf8_string(decoder, "section name", tracer);
126 if (decoder->failed()) {
127 return kUnknownSectionCode;
128 }
129 const uint8_t* section_name_start =
130 decoder->start() + decoder->GetBufferRelativeOffset(string.offset());
131
132 TRACE(" +%d section name : \"%.*s\"\n",
133 static_cast<int>(section_name_start - decoder->start()),
134 string.length() < 20 ? string.length() : 20, section_name_start);
135
136 using SpecialSectionPair = std::pair<base::Vector<const char>, SectionCode>;
137 static constexpr SpecialSectionPair kSpecialSections[]{
138 {base::StaticCharVector(kNameString), kNameSectionCode},
139 {base::StaticCharVector(kSourceMappingURLString),
140 kSourceMappingURLSectionCode},
141 {base::StaticCharVector(kInstTraceString), kInstTraceSectionCode},
142 {base::StaticCharVector(kCompilationHintsString),
143 kCompilationHintsSectionCode},
144 {base::StaticCharVector(kBranchHintsString), kBranchHintsSectionCode},
145 {base::StaticCharVector(kDebugInfoString), kDebugInfoSectionCode},
146 {base::StaticCharVector(kExternalDebugInfoString),
147 kExternalDebugInfoSectionCode},
148 {base::StaticCharVector(kBuildIdString), kBuildIdSectionCode}};
149
150 auto name_vec = base::Vector<const char>::cast(
151 base::VectorOf(section_name_start, string.length()));
152 for (auto& special_section : kSpecialSections) {
153 if (name_vec == special_section.first) return special_section.second;
154 }
155
156 return kUnknownSectionCode;
157}

194 // Advances to the next section, checking that decoding the current section
195 // stopped at {section_end_}.
196 void advance(bool move_to_section_end = false) {
197 if (move_to_section_end && decoder_->pc() < section_end_) {
198 decoder_->consume_bytes(
199 static_cast<uint32_t>(section_end_ - decoder_->pc()));
200 }
201 if (decoder_->pc() != section_end_) {
202 const char* msg = decoder_->pc() < section_end_ ? "shorter" : "longer";
203 decoder_->errorf(decoder_->pc(),
204 "section was %s than expected size "
205 "(%u bytes expected, %zu decoded)",
206 msg, section_length(),
207 static_cast<size_t>(decoder_->pc() - section_start_));
208 }
209 next();
210 }

212 private:
213 Decoder* decoder_;
214 ITracer* tracer_;
215 SectionCode section_code_;
216 const uint8_t* section_start_;
217 const uint8_t* payload_start_;
218 const uint8_t* section_end_;
219
220 // Reads the section code/name at the current position and sets up
221 // the embedder fields.
222 void next() {
223 if (!decoder_->more()) {
224 section_code_ = kUnknownSectionCode;
225 return;
226 }
227 section_start_ = decoder_->pc();
228 // Empty line before next section.
229 if (tracer_) tracer_->NextLine();
230 uint8_t section_code = decoder_->consume_u8("section kind", tracer_);
231 if (tracer_) {
232 tracer_->Description(": ");
233 tracer_->Description(SectionName(static_cast<SectionCode>(section_code)));
234 tracer_->NextLine();
235 }
236 // Read and check the section size.
237 uint32_t section_length = decoder_->consume_u32v("section length", tracer_);
238 if (tracer_) {
239 tracer_->Description(section_length);
240 tracer_->NextLine();
241 }
242 payload_start_ = decoder_->pc();
243 section_end_ = payload_start_ + section_length;
244 if (section_length > decoder_->available_bytes()) {
245 decoder_->errorf(
246 section_start_,
247 "section (code %u, \"%s\") extends past end of the module "
248 "(length %u, remaining bytes %u)",
249 section_code, SectionName(static_cast<SectionCode>(section_code)),
250 section_length, decoder_->available_bytes());
251 section_end_ = payload_start_;
252 }
253
254 if (section_code == kUnknownSectionCode) {
255 // Check for the known "name", "sourceMappingURL", or "compilationHints"
256 // section.
257 // To identify the unknown section we set the end of the decoder bytes to
258 // the end of the custom section, so that we do not read the section name
259 // beyond the end of the section.
260 const uint8_t* module_end = decoder_->end();
261 decoder_->set_end(section_end_);
262 section_code = IdentifyUnknownSectionInternal(decoder_, tracer_);
263 if (decoder_->ok()) decoder_->set_end(module_end);
264 // As a side effect, the above function will forward the decoder to after
265 // the identifier string.
266 payload_start_ = decoder_->pc();
267 } else if (!IsValidSectionCode(section_code)) {
268 decoder_->errorf(decoder_->pc(), "unknown section code #0x%02x",
269 section_code);
270 }
271 section_code_ = decoder_->failed() ? kUnknownSectionCode
272 : static_cast<SectionCode>(section_code);
273
274 if (section_code_ == kUnknownSectionCode && section_end_ > decoder_->pc()) {
275 // Skip to the end of the unknown section.
276 uint32_t remaining = static_cast<uint32_t>(section_end_ - decoder_->pc());
277 decoder_->consume_bytes(remaining, "section payload", tracer_);
278 }
279 }

282inline void DumpModule(const base::Vector<const uint8_t> module_bytes,
283 bool ok) {
284 std::string path;
285 if (v8_flags.dump_wasm_module_path) {
286 path = v8_flags.dump_wasm_module_path;
287 if (path.size() && !base::OS::isDirectorySeparator(path[path.size() - 1])) {
288 path += base::OS::DirectorySeparator();
289 }
290 }
291 // File are named `<hash>.{ok,failed}.wasm`.
292 // Limit the hash to 8 characters (32 bits).
293 uint32_t hash = static_cast<uint32_t>(GetWireBytesHash(module_bytes));
294 base::EmbeddedVector<char, 32> buf;
295 SNPrintF(buf, "%08x.%s.wasm", hash, ok ? "ok" : "failed");
296 path += buf.begin();
297 size_t rv = 0;
298 if (FILE* file = base::OS::FOpen(path.c_str(), "wb")) {
299 rv = fwrite(module_bytes.begin(), module_bytes.length(), 1, file);
300 base::Fclose(file);
301 }
302 if (rv != 1) {
303 OFStream os(stderr);
304 os << "Error while dumping wasm file to " << path << std::endl;
305 }

328 if (failed()) return;

330

332 uint32_t magic_word = consume_u32("wasm magic", tracer_);
333 if (tracer_) tracer_->NextLine();
334#define BYTES(x) (x & 0xFF), (x >> 8) & 0xFF, (x >> 16) & 0xFF, (x >> 24) & 0xFF

356 bool CheckSectionOrder(SectionCode section_code) {
357 // Check the order of ordered sections.
358 if (section_code >= kFirstSectionInModule &&
359 section_code < kFirstUnorderedSection) {
360 if (section_code < next_ordered_section_) {
361 errorf(pc(), "unexpected section <%s>", SectionName(section_code));
362 return false;
363 }
364 next_ordered_section_ = section_code + 1;
365 return true;
366 }
367
368 // Ignore ordering problems in unknown / custom sections. Even allow them to
369 // appear multiple times. As optional sections we use them on a "best
370 // effort" basis.
371 if (section_code == kUnknownSectionCode) return true;
372 if (section_code > kLastKnownModuleSection) return true;
373
374 // The rest is standardized unordered sections; they are checked more
375 // thoroughly..
376 DCHECK_LE(kFirstUnorderedSection, section_code);
377 DCHECK_GE(kLastKnownModuleSection, section_code);
378
379 // Check that unordered sections don't appear multiple times.
380 if (has_seen_unordered_section(section_code)) {
381 errorf(pc(), "Multiple %s sections not allowed",
382 SectionName(section_code));
383 return false;
384 }
385 set_seen_unordered_section(section_code);
386
387 // Define a helper to ensure that sections <= {before} appear before the
388 // current unordered section, and everything >= {after} appears after it.
389 auto check_order = [this, section_code](SectionCode before,
390 SectionCode after) -> bool {
391 DCHECK_LT(before, after);
392 if (next_ordered_section_ > after) {
393 errorf(pc(), "The %s section must appear before the %s section",
394 SectionName(section_code), SectionName(after));
395 return false;
396 }
397 if (next_ordered_section_ <= before) next_ordered_section_ = before + 1;
398 return true;
399 };
400
401 // Now check the ordering constraints of specific unordered sections.
402 switch (section_code) {
403 case kDataCountSectionCode:
404 return check_order(kElementSectionCode, kCodeSectionCode);
405 case kTagSectionCode:
406 return check_order(kMemorySectionCode, kGlobalSectionCode);
407 case kStringRefSectionCode:
408 // TODO(12868): If there's a tag section, assert that we're after the
409 // tag section.
410 return check_order(kMemorySectionCode, kGlobalSectionCode);
411 case kInstTraceSectionCode:
412 // Custom section following code.metadata tool convention containing
413 // offsets specifying where trace marks should be emitted.
414 // Be lenient with placement of instruction trace section. All except
415 // first occurrence after function section and before code section are
416 // ignored.
417 return true;
418 default:
419 return true;
420 }
421 }

423 void DecodeSection(SectionCode section_code,
424 base::Vector<const uint8_t> bytes, uint32_t offset) {
425 if (failed()) return;
426 Reset(bytes, offset);
427 TRACE("Section: %s\n", SectionName(section_code));
428 TRACE("Decode Section %p - %p\n", bytes.begin(), bytes.end());
429
430 if (!CheckSectionOrder(section_code)) return;
431
432 switch (section_code) {
433 case kUnknownSectionCode:
434 break;
435 case kTypeSectionCode:
436 DecodeTypeSection();
437 break;
438 case kImportSectionCode:
439 DecodeImportSection();
440 break;
441 case kFunctionSectionCode:
442 DecodeFunctionSection();
443 break;
444 case kTableSectionCode:
445 DecodeTableSection();
446 break;
447 case kMemorySectionCode:
448 DecodeMemorySection();
449 break;
450 case kGlobalSectionCode:
451 DecodeGlobalSection();
452 break;
453 case kExportSectionCode:
454 DecodeExportSection();
455 break;
456 case kStartSectionCode:
457 DecodeStartSection();
458 break;
459 case kCodeSectionCode:
460 DecodeCodeSection();
461 break;
462 case kElementSectionCode:
463 DecodeElementSection();
464 break;
465 case kDataSectionCode:
466 DecodeDataSection();
467 break;
468 case kNameSectionCode:
469 DecodeNameSection();
470 break;
471 case kSourceMappingURLSectionCode:
472 DecodeSourceMappingURLSection();
473 break;
474 case kDebugInfoSectionCode:
475 module_->debug_symbols[WasmDebugSymbols::Type::EmbeddedDWARF] = {
476 WasmDebugSymbols::Type::EmbeddedDWARF, {}};
477 consume_bytes(static_cast<uint32_t>(end_ - start_), ".debug_info");
478 break;
479 case kExternalDebugInfoSectionCode:
480 DecodeExternalDebugInfoSection();
481 break;
482 case kBuildIdSectionCode:
483 DecodeBuildIdSection();
484 break;
485 case kInstTraceSectionCode:
486 if (enabled_features_.has_instruction_tracing()) {
487 DecodeInstTraceSection();
488 } else {
489 // Ignore this section when feature is disabled. It is an optional
490 // custom section anyways.
491 consume_bytes(static_cast<uint32_t>(end_ - start_), nullptr);
492 }
493 break;
494 case kCompilationHintsSectionCode:
495 // TODO(jkummerow): We're missing tracing support for well-known
496 // custom sections. This confuses `wami --full-hexdump` e.g.
497 // for the modules created by
498 // mjsunit/wasm/compilation-hints-streaming-compilation.js.
499 if (enabled_features_.has_compilation_hints()) {
500 DecodeCompilationHintsSection();
501 } else {
502 // Ignore this section when feature was disabled. It is an optional
503 // custom section anyways.
504 consume_bytes(static_cast<uint32_t>(end_ - start_), nullptr);
505 }
506 break;
507 case kBranchHintsSectionCode:
508 if (enabled_features_.has_branch_hinting()) {
509 DecodeBranchHintsSection();
510 } else {
511 // Ignore this section when feature was disabled. It is an optional
512 // custom section anyways.
513 consume_bytes(static_cast<uint32_t>(end_ - start_), nullptr);
514 }
515 break;
516 case kDataCountSectionCode:
517 DecodeDataCountSection();
518 break;
519 case kTagSectionCode:
520 DecodeTagSection();
521 break;
522 case kStringRefSectionCode:
523 if (enabled_features_.has_stringref()) {
524 DecodeStringRefSection();
525 } else {
526 errorf(pc(),
527 "unexpected section <%s> (enable with "
528 "--experimental-wasm-stringref)",
529 SectionName(section_code));
530 }
531 break;
532 default:
533 errorf(pc(), "unexpected section <%s>", SectionName(section_code));
534 return;
535 }
536
537 if (pc() != bytes.end()) {
538 const char* msg = pc() < bytes.end() ? "shorter" : "longer";
539 errorf(pc(),
540 "section was %s than expected size "
541 "(%zu bytes expected, %zu decoded)",
542 msg, bytes.size(), static_cast<size_t>(pc() - bytes.begin()));
543 }
544 }

557 TypeDefinition consume_base_type_definition() {
558 const bool is_final = true;
559 bool shared = false;
560 uint8_t kind = consume_u8(" kind", tracer_);
561 if (tracer_) tracer_->Description(": ");
562 if (kind == kSharedFlagCode) {
563 if (!v8_flags.experimental_wasm_shared) {
564 errorf(pc() - 1,
565 "unknown type form: %d, enable with --experimental-wasm-shared",
566 kind);
567 return {};
568 }
569 shared = true;
570 module_->has_shared_part = true;
571 kind = consume_u8("shared ", tracer_);
572 }
573 if (tracer_) tracer_->Description(TypeKindName(kind));
574 switch (kind) {
575 case kWasmFunctionTypeCode: {
576 const FunctionSig* sig = consume_sig(&module_->signature_zone, shared);
577 if (sig == nullptr) {
578 CHECK(!ok());
579 return {};
580 }
581 return {sig, kNoSuperType, is_final, shared};
582 }
583 case kWasmStructTypeCode: {
584 module_->is_wasm_gc = true;
585 const StructType* type =
586 consume_struct(&module_->signature_zone, shared);
587 if (type == nullptr) {
588 CHECK(!ok());
589 return {};
590 }
591 return {type, kNoSuperType, is_final, shared};
592 }
593 case kWasmArrayTypeCode: {
594 module_->is_wasm_gc = true;
595 const ArrayType* type = consume_array(&module_->signature_zone, shared);
596 if (type == nullptr) {
597 CHECK(!ok());
598 return {};
599 }
600 return {type, kNoSuperType, is_final, shared};
601 }
602 default:
603 if (tracer_) tracer_->NextLine();
604 errorf(pc() - 1, "unknown type form: %d", kind);
605 return {};
606 }

643 void DecodeTypeSection() {
644 TypeCanonicalizer* type_canon = GetTypeCanonicalizer();
645 uint32_t types_count = consume_count("types count", kV8MaxWasmTypes);
646
647 for (uint32_t i = 0; ok() && i < types_count; ++i) {
648 TRACE("DecodeType[%d] module+%d\n", i, static_cast<int>(pc_ - start_));
649 uint8_t kind = read_u8<Decoder::FullValidationTag>(pc(), "type kind");
650 size_t initial_size = module_->types.size();
651 if (kind == kWasmRecursiveTypeGroupCode) {
652 module_->is_wasm_gc = true;
653 uint32_t rec_group_offset = pc_offset();
654 consume_bytes(1, "rec. group definition", tracer_);
655 if (tracer_) tracer_->NextLine();
656 uint32_t group_size =
657 consume_count("recursive group size", kV8MaxWasmTypes);
658 if (tracer_) tracer_->RecGroupOffset(rec_group_offset, group_size);
659 if (initial_size + group_size > kV8MaxWasmTypes) {
660 errorf(pc(), "Type definition count exceeds maximum %zu",
661 kV8MaxWasmTypes);
662 return;
663 }
664 // We need to resize types before decoding the type definitions in this
665 // group, so that the correct type size is visible to type definitions.
666 module_->types.resize(initial_size + group_size);
667 module_->isorecursive_canonical_type_ids.resize(initial_size +
668 group_size);
669 for (uint32_t j = 0; j < group_size; j++) {
670 if (tracer_) tracer_->TypeOffset(pc_offset());
671 TypeDefinition type = consume_subtype_definition(initial_size + j);
672 module_->types[initial_size + j] = type;
673 }
674 if (failed()) return;
675 type_canon->AddRecursiveGroup(module_.get(), group_size);
676 if (tracer_) {
677 tracer_->Description("end of rec. group");
678 tracer_->NextLine();
679 }
680 } else {
681 if (tracer_) tracer_->TypeOffset(pc_offset());
682 if (initial_size + 1 > kV8MaxWasmTypes) {
683 errorf(pc(), "Type definition count exceeds maximum %zu",
684 kV8MaxWasmTypes);
685 return;
686 }
687 // Similarly to above, we need to resize types for a group of size 1.
688 module_->types.resize(initial_size + 1);
689 module_->isorecursive_canonical_type_ids.resize(initial_size + 1);
690 TypeDefinition type = consume_subtype_definition(initial_size);
691 if (ok()) {
692 module_->types[initial_size] = type;
693 type_canon->AddRecursiveSingletonGroup(module_.get());
694 }
695 }
696 }
697
698 // Check validity of explicitly defined supertypes and propagate subtyping
699 // depth.
700 const WasmModule* module = module_.get();
701 for (uint32_t i = 0; ok() && i < module_->types.size(); ++i) {
702 ModuleTypeIndex explicit_super = module_->supertype(ModuleTypeIndex{i});
703 if (!explicit_super.valid()) continue;
704 DCHECK_LT(explicit_super.index, i); // Checked during decoding.
705 uint32_t depth = module->type(explicit_super).subtyping_depth + 1;
706 module_->types[i].subtyping_depth = depth;
707 DCHECK_GE(depth, 0);
708 if (depth > kV8MaxRttSubtypingDepth) {
709 errorf("type %u: subtyping depth is greater than allowed", i);
710 continue;
711 }
712 // This check is technically redundant; we include for the improved error
713 // message.
714 if (module->type(explicit_super).is_final) {
715 errorf("type %u extends final type %u", i, explicit_super.index);
716 continue;
717 }
718 if (!ValidSubtypeDefinition(ModuleTypeIndex{i}, explicit_super, module,
719 module)) {
720 errorf("type %u has invalid explicit supertype %u", i,
721 explicit_super.index);
722 continue;
723 }
724 }

727 void DecodeImportSection() {
728 uint32_t import_table_count =
729 consume_count("imports count", kV8MaxWasmImports);
730 module_->import_table.reserve(import_table_count);
731 for (uint32_t i = 0; ok() && i < import_table_count; ++i) {
732 TRACE("DecodeImportTable[%d] module+%d\n", i,
733 static_cast<int>(pc_ - start_));
734 if (tracer_) tracer_->ImportOffset(pc_offset());
735
736 const uint8_t* pos = pc_;
737 WireBytesRef module_name =
738 consume_utf8_string(this, "module name", tracer_);
739 WireBytesRef field_name =
740 consume_utf8_string(this, "field name", tracer_);
741 ImportExportKindCode kind =
742 static_cast<ImportExportKindCode>(consume_u8("kind", tracer_));
743 if (tracer_) {
744 tracer_->Description(": ");
745 tracer_->Description(ExternalKindName(kind));
746 }
747 module_->import_table.push_back(WasmImport{
748 .module_name = module_name, .field_name = field_name, .kind = kind});
749 WasmImport* import = &module_->import_table.back();
750 switch (kind) {
751 case kExternalFunction: {
752 // ===== Imported function ===========================================
753 import->index = static_cast<uint32_t>(module_->functions.size());
754 module_->num_imported_functions++;
755 module_->functions.push_back(WasmFunction{
756 .func_index = import->index,
757 .imported = true,
758 });
759 WasmFunction* function = &module_->functions.back();
760 function->sig_index =
761 consume_sig_index(module_.get(), &function->sig);
762 break;
763 }
764 case kExternalTable: {
765 // ===== Imported table ==============================================
766 import->index = static_cast<uint32_t>(module_->tables.size());
767 const uint8_t* type_position = pc();
768 ValueType type = consume_value_type();
769 if (!type.is_object_reference()) {
770 errorf(type_position, "Invalid table type %s", type.name().c_str());
771 break;
772 }
773 module_->num_imported_tables++;
774 module_->tables.push_back(WasmTable{
775 .type = type,
776 .imported = true,
777 });
778 WasmTable* table = &module_->tables.back();
779 consume_table_flags(table);
780 DCHECK_IMPLIES(table->shared,
781 v8_flags.experimental_wasm_shared || !ok());
782 if (table->shared && v8_flags.experimental_wasm_shared) {
783 module_->has_shared_part = true;
784 if (!IsShared(type, module_.get())) {
785 errorf(type_position,
786 "Shared table %i must have shared element type, actual "
787 "type %s",
788 i, type.name().c_str());
789 break;
790 }
791 }
792 // Note that we should not throw an error if the declared maximum size
793 // is oob. We will instead fail when growing at runtime.
794 uint64_t kNoMaximum = kMaxUInt64;
795 consume_resizable_limits(
796 "table", "elements", wasm::max_table_size(), &table->initial_size,
797 table->has_maximum_size, kNoMaximum, &table->maximum_size,
798 table->is_table64() ? k64BitLimits : k32BitLimits);
799 break;
800 }
801 case kExternalMemory: {
802 // ===== Imported memory =============================================
803 static_assert(kV8MaxWasmMemories <= kMaxUInt32);
804 if (module_->memories.size() >= kV8MaxWasmMemories - 1) {
805 errorf("At most %u imported memories are supported",
806 kV8MaxWasmMemories);
807 break;
808 }
809 uint32_t mem_index = static_cast<uint32_t>(module_->memories.size());
810 import->index = mem_index;
811 module_->memories.emplace_back();
812 WasmMemory* external_memory = &module_->memories.back();
813 external_memory->imported = true;
814 external_memory->index = mem_index;
815
816 consume_memory_flags(external_memory);
817 uint32_t max_pages = external_memory->is_memory64()
818 ? kSpecMaxMemory64Pages
819 : kSpecMaxMemory32Pages;
820 consume_resizable_limits(
821 "memory", "pages", max_pages, &external_memory->initial_pages,
822 external_memory->has_maximum_pages, max_pages,
823 &external_memory->maximum_pages,
824 external_memory->is_memory64() ? k64BitLimits : k32BitLimits);
825 break;
826 }
827 case kExternalGlobal: {
828 // ===== Imported global =============================================
829 import->index = static_cast<uint32_t>(module_->globals.size());
830 ValueType type = consume_value_type();
831 auto [mutability, shared] = consume_global_flags();
832 if (V8_UNLIKELY(failed())) break;
833 if (V8_UNLIKELY(shared && !IsShared(type, module_.get()))) {
834 error("shared imported global must have shared type");
835 break;
836 }
837 module_->globals.push_back(
838 WasmGlobal{.type = type,
839 .mutability = mutability,
840 .index = 0, // set later in CalculateGlobalOffsets
841 .shared = shared,
842 .imported = true});
843 module_->num_imported_globals++;
844 DCHECK_EQ(module_->globals.size(), module_->num_imported_globals);
845 if (shared) module_->has_shared_part = true;
846 if (mutability) module_->num_imported_mutable_globals++;
847 if (tracer_) tracer_->NextLine();
848 break;
849 }
850 case kExternalTag: {
851 // ===== Imported tag ================================================
852 import->index = static_cast<uint32_t>(module_->tags.size());
853 module_->num_imported_tags++;
854 const WasmTagSig* tag_sig = nullptr;
855 consume_exception_attribute(); // Attribute ignored for now.
856 ModuleTypeIndex sig_index =
857 consume_tag_sig_index(module_.get(), &tag_sig);
858 module_->tags.emplace_back(tag_sig, sig_index);
859 break;
860 }
861 default:
862 errorf(pos, "unknown import kind 0x%02x", kind);
863 break;
864 }
865 }
866 if (module_->memories.size() > 1) {
867 detected_features_->add_multi_memory();
868 if (v8_flags.wasm_jitless) {
869 error("Multiple memories not supported in Wasm jitless mode");
870 }
871 }
872 UpdateComputedMemoryInformation();
873 module_->type_feedback.well_known_imports.Initialize(
874 module_->num_imported_functions);
875 if (tracer_) tracer_->ImportsDone(module_.get());
876 }

879 uint32_t functions_count =
880 consume_count("functions count", v8_flags.max_wasm_functions);
881 DCHECK_EQ(module_->functions.size(), module_->num_imported_functions);
882 uint32_t total_function_count =
883 module_->num_imported_functions + functions_count;

895 0xff);
896 }
897
898 for (uint32_t func_index = module_->num_imported_functions;
899 func_index < total_function_count; ++func_index) {
900 WasmFunction* function = &module_->functions[func_index];
901 function->func_index = func_index;
902 if (tracer_) tracer_->FunctionName(func_index);
903 function->sig_index = consume_sig_index(module_.get(), &function->sig);
904 if (!ok()) return;
905 }

908 void DecodeTableSection() {
909 static_assert(kV8MaxWasmTables <= kMaxUInt32);
910 uint32_t table_count = consume_count("table count", kV8MaxWasmTables);
911
912 for (uint32_t i = 0; ok() && i < table_count; i++) {
913 if (tracer_) tracer_->TableOffset(pc_offset());
914 module_->tables.emplace_back();
915 WasmTable* table = &module_->tables.back();
916 const uint8_t* type_position = pc();
917
918 bool has_initializer = false;
919 if (read_u8<Decoder::FullValidationTag>(
920 pc(), "table-with-initializer byte") == 0x40) {
921 consume_bytes(1, "with-initializer ", tracer_);
922 has_initializer = true;
923 type_position++;
924 uint8_t reserved = consume_u8("reserved-byte", tracer_);
925 if (reserved != 0) {
926 error(type_position, "Reserved byte must be 0x00");
927 break;
928 }
929 type_position++;
930 }
931
932 ValueType table_type = consume_value_type();
933 if (!table_type.is_object_reference()) {
934 error(type_position, "Only reference types can be used as table types");
935 break;
936 }
937 if (!has_initializer && !table_type.is_defaultable()) {
938 errorf(type_position,
939 "Table of non-defaultable table %s needs initial value",
940 table_type.name().c_str());
941 break;
942 }
943 table->type = table_type;
944
945 consume_table_flags(table);
946 DCHECK_IMPLIES(table->shared, v8_flags.experimental_wasm_shared || !ok());
947 if (table->shared && v8_flags.experimental_wasm_shared) {
948 module_->has_shared_part = true;
949 if (!IsShared(table_type, module_.get())) {
950 errorf(
951 type_position,
952 "Shared table %i must have shared element type, actual type %s",
953 i + module_->num_imported_tables, table_type.name().c_str());
954 break;
955 }
956 }
957 // Note that we should not throw an error if the declared maximum size is
958 // oob. We will instead fail when growing at runtime.
959 uint64_t kNoMaximum = kMaxUInt64;
960 consume_resizable_limits(
961 "table", "elements", wasm::max_table_size(), &table->initial_size,
962 table->has_maximum_size, kNoMaximum, &table->maximum_size,
963 table->is_table64() ? k64BitLimits : k32BitLimits);
964
965 if (has_initializer) {
966 table->initial_value =
967 consume_init_expr(module_.get(), table_type, table->shared);
968 }
969 }

1018 uint32_t imported_globals = static_cast<uint32_t>(module_->globals.size());
1019 // It is important to not resize the globals vector from the beginning,

1048 void DecodeExportSection() {
1049 uint32_t export_table_count =
1050 consume_count("exports count", kV8MaxWasmExports);
1051 module_->export_table.reserve(export_table_count);
1052 for (uint32_t i = 0; ok() && i < export_table_count; ++i) {
1053 TRACE("DecodeExportTable[%d] module+%d\n", i,
1054 static_cast<int>(pc_ - start_));
1055 if (tracer_) {
1056 tracer_->Description("export #");
1057 tracer_->Description(i);
1058 tracer_->NextLine();
1059 }
1060
1061 WireBytesRef name = consume_utf8_string(this, "field name", tracer_);
1062
1063 const uint8_t* kind_pos = pc();
1064 ImportExportKindCode kind =
1065 static_cast<ImportExportKindCode>(consume_u8("kind", tracer_));
1066
1067 module_->export_table.push_back(WasmExport{.name = name, .kind = kind});
1068 WasmExport* exp = &module_->export_table.back();
1069
1070 if (tracer_) {
1071 tracer_->Description(": ");
1072 tracer_->Description(ExternalKindName(exp->kind));
1073 tracer_->Description(" ");
1074 }
1075 switch (kind) {
1076 case kExternalFunction: {
1077 WasmFunction* func = nullptr;
1078 exp->index = consume_func_index(module_.get(), &func);
1079
1080 if (failed()) break;
1081 DCHECK_NOT_NULL(func);
1082
1083 module_->num_exported_functions++;
1084 func->exported = true;
1085 // Exported functions are considered "declared".
1086 func->declared = true;
1087 break;
1088 }
1089 case kExternalTable: {
1090 WasmTable* table = nullptr;
1091 exp->index = consume_table_index(module_.get(), &table);
1092 if (table) table->exported = true;
1093 break;
1094 }
1095 case kExternalMemory: {
1096 const uint8_t* index_pos = pc();
1097 exp->index = consume_u32v("memory index", tracer_);
1098 size_t num_memories = module_->memories.size();
1099 if (exp->index >= module_->memories.size()) {
1100 errorf(index_pos,
1101 "invalid exported memory index %u (having %zu memor%s)",
1102 exp->index, num_memories, num_memories == 1 ? "y" : "ies");
1103 break;
1104 }
1105 module_->memories[exp->index].exported = true;
1106 break;
1107 }
1108 case kExternalGlobal: {
1109 WasmGlobal* global = nullptr;
1110 exp->index = consume_global_index(module_.get(), &global);
1111 if (global) {
1112 global->exported = true;
1113 }
1114 break;
1115 }
1116 case kExternalTag: {
1117 WasmTag* tag = nullptr;
1118 exp->index = consume_tag_index(module_.get(), &tag);
1119 break;
1120 }
1121 default:
1122 errorf(kind_pos, "invalid export kind 0x%02x", exp->kind);
1123 break;
1124 }
1125 if (tracer_) tracer_->NextLine();
1126 }
1127 // Check for duplicate exports (except for asm.js).
1128 if (ok() && module_->origin == kWasmOrigin &&
1129 module_->export_table.size() > 1) {
1130 std::vector<WasmExport> sorted_exports(module_->export_table);
1131
1132 auto cmp_less = [this](const WasmExport& a, const WasmExport& b) {
1133 // Return true if a < b.
1134 if (a.name.length() != b.name.length()) {
1135 return a.name.length() < b.name.length();
1136 }
1137 const uint8_t* left =
1138 start() + GetBufferRelativeOffset(a.name.offset());
1139 const uint8_t* right =
1140 start() + GetBufferRelativeOffset(b.name.offset());
1141 return memcmp(left, right, a.name.length()) < 0;
1142 };
1143 std::stable_sort(sorted_exports.begin(), sorted_exports.end(), cmp_less);
1144
1145 auto it = sorted_exports.begin();
1146 WasmExport* last = &*it++;
1147 for (auto end = sorted_exports.end(); it != end; last = &*it++) {
1148 DCHECK(!cmp_less(*it, *last)); // Vector must be sorted.
1149 if (!cmp_less(*last, *it)) {
1150 const uint8_t* pc =
1151 start() + GetBufferRelativeOffset(it->name.offset());
1152 TruncatedUserString<> name(pc, it->name.length());
1153 errorf(pc, "Duplicate export name '%.*s' for %s %d and %s %d",
1154 name.length(), name.start(), ExternalKindName(last->kind),
1155 last->index, ExternalKindName(it->kind), it->index);
1156 break;
1157 }
1158 }
1159 }
1160 }

1195 void DecodeCodeSection() {
1196 // Make sure global offset were calculated before they get accessed during
1197 // function compilation.
1198 CalculateGlobalOffsets(module_.get());
1199 uint32_t code_section_start = pc_offset();
1200 uint32_t functions_count = consume_u32v("functions count", tracer_);
1201 if (tracer_) {
1202 tracer_->Description(functions_count);
1203 tracer_->NextLine();
1204 }
1205 CheckFunctionsCount(functions_count, code_section_start);
1206
1207 auto inst_traces_it = this->inst_traces_.begin();
1208 std::vector<std::pair<uint32_t, uint32_t>> inst_traces;
1209
1210 for (uint32_t i = 0; ok() && i < functions_count; ++i) {
1211 int function_index = module_->num_imported_functions + i;
1212 if (tracer_) {
1213 tracer_->Description("function #");
1214 tracer_->FunctionName(function_index);
1215 tracer_->NextLine();
1216 }
1217 const uint8_t* pos = pc();
1218 uint32_t size = consume_u32v("body size", tracer_);
1219 if (tracer_) {
1220 tracer_->Description(size);
1221 tracer_->NextLine();
1222 }
1223 if (size > kV8MaxWasmFunctionSize) {
1224 errorf(pos, "size %u > maximum function size %zu", size,
1225 kV8MaxWasmFunctionSize);
1226 return;
1227 }
1228 uint32_t offset = pc_offset();
1229 consume_bytes(size, "function body");
1230 if (failed()) break;
1231 DecodeFunctionBody(function_index, size, offset);
1232
1233 // Now that the function has been decoded, we can compute module offsets.
1234 for (; inst_traces_it != this->inst_traces_.end() &&
1235 std::get<0>(*inst_traces_it) == i;
1236 ++inst_traces_it) {
1237 uint32_t trace_offset = offset + std::get<1>(*inst_traces_it);
1238 uint32_t mark_id = std::get<2>(*inst_traces_it);
1239 std::pair<uint32_t, uint32_t> trace_mark = {trace_offset, mark_id};
1240 inst_traces.push_back(trace_mark);
1241 }
1242 }
1243 // If we have actually decoded traces and they were all decoded without
1244 // error, then we can move them to the module. If any errors are found, it
1245 // is safe to throw away all traces.
1246 if (V8_UNLIKELY(!inst_traces.empty() &&
1247 inst_traces_it == this->inst_traces_.end())) {
1248 // This adds an invalid entry at the end of the traces. An invalid entry
1249 // is defined as having an module offset of 0 and a markid of 0.
1250 inst_traces.push_back({0, 0});
1251 this->module_->inst_traces = std::move(inst_traces);
1252 }
1253 DCHECK_GE(pc_offset(), code_section_start);
1254 module_->code = {code_section_start, pc_offset() - code_section_start};
1255 }

1261 CalculateGlobalOffsets(module_.get());
1262 module_->code = section_bytes;
1263 }

1274 void DecodeFunctionBody(uint32_t func_index, uint32_t length,
1275 uint32_t offset) {
1276 WasmFunction* function = &module_->functions[func_index];
1277 function->code = {offset, length};
1278 constexpr uint32_t kSmallFunctionThreshold = 50;

1415 void DecodeInstTraceSection() {
1416 TRACE("DecodeInstTrace module+%d\n", static_cast<int>(pc_ - start_));
1417 if (!has_seen_unordered_section(kInstTraceSectionCode)) {
1418 set_seen_unordered_section(kInstTraceSectionCode);
1419
1420 // Use an inner decoder so that errors don't fail the outer decoder.
1421 Decoder inner(start_, pc_, end_, buffer_offset_);
1422
1423 std::vector<std::tuple<uint32_t, uint32_t, uint32_t>> inst_traces;
1424
1425 uint32_t func_count = inner.consume_u32v("number of functions");
1426 // Keep track of the previous function index to validate the ordering.
1427 int64_t last_func_idx = -1;
1428 for (uint32_t i = 0; i < func_count; i++) {
1429 uint32_t func_idx = inner.consume_u32v("function index");
1430 if (int64_t{func_idx} <= last_func_idx) {
1431 inner.errorf("Invalid function index: %d", func_idx);
1432 break;
1433 }
1434 last_func_idx = func_idx;
1435
1436 uint32_t num_traces = inner.consume_u32v("number of trace marks");
1437 TRACE("DecodeInstTrace[%d] module+%d\n", func_idx,
1438 static_cast<int>(inner.pc() - inner.start()));
1439 // Keep track of the previous offset to validate the ordering.
1440 int64_t last_func_off = -1;
1441 for (uint32_t j = 0; j < num_traces; ++j) {
1442 uint32_t func_off = inner.consume_u32v("function offset");
1443
1444 uint32_t mark_size = inner.consume_u32v("mark size");
1445 uint32_t trace_mark_id = 0;
1446 // Build the mark id from the individual bytes.
1447 for (uint32_t k = 0; k < mark_size; k++) {
1448 trace_mark_id |= inner.consume_u8("trace mark id") << k * 8;
1449 }
1450 if (int64_t{func_off} <= last_func_off) {
1451 inner.errorf("Invalid branch offset: %d", func_off);
1452 break;
1453 }
1454 last_func_off = func_off;
1455 TRACE("DecodeInstTrace[%d][%d] module+%d\n", func_idx, func_off,
1456 static_cast<int>(inner.pc() - inner.start()));
1457 // Store the function index, function offset, and mark id into a
1458 // temporary 3-tuple. This will later be translated to a module
1459 // offset and mark id.
1460 std::tuple<uint32_t, uint32_t, uint32_t> mark_tuple = {
1461 func_idx, func_off, trace_mark_id};
1462 inst_traces.push_back(mark_tuple);
1463 }
1464 }
1465 // Extra unexpected bytes are an error.
1466 if (inner.more()) {
1467 inner.errorf("Unexpected extra bytes: %d\n",
1468 static_cast<int>(inner.pc() - inner.start()));
1469 }
1470 // If everything went well, accept the traces for the module.
1471 if (inner.ok()) {
1472 this->inst_traces_ = std::move(inst_traces);
1473 }
1474 }
1475
1476 // Skip the whole instruction trace section in the outer decoder.
1477 consume_bytes(static_cast<uint32_t>(end_ - start_), nullptr);
1478 }

1480 void DecodeCompilationHintsSection() {
1481 TRACE("DecodeCompilationHints module+%d\n", static_cast<int>(pc_ - start_));
1482
1483 // TODO(frgossen): Find a way to report compilation hint errors as warnings.
1484 // All except first occurrence after function section and before code
1485 // section are ignored.
1486 const bool before_function_section =
1487 next_ordered_section_ <= kFunctionSectionCode;
1488 const bool after_code_section = next_ordered_section_ > kCodeSectionCode;
1489 if (before_function_section || after_code_section ||
1490 has_seen_unordered_section(kCompilationHintsSectionCode)) {
1491 return;
1492 }
1493 set_seen_unordered_section(kCompilationHintsSectionCode);
1494
1495 // TODO(frgossen) Propagate errors to outer decoder in experimental phase.
1496 // We should use an inner decoder later and propagate its errors as
1497 // warnings.
1498 Decoder& decoder = *this;
1499 // Decoder decoder(start_, pc_, end_, buffer_offset_);
1500
1501 // Ensure exactly one compilation hint per function.
1502 uint32_t hint_count = decoder.consume_u32v("compilation hint count");
1503 if (hint_count != module_->num_declared_functions) {
1504 decoder.errorf(decoder.pc(), "Expected %u compilation hints (%u found)",
1505 module_->num_declared_functions, hint_count);
1506 }
1507
1508 // Decode sequence of compilation hints.
1509 if (decoder.ok()) {
1510 module_->compilation_hints.reserve(hint_count);
1511 }
1512 for (uint32_t i = 0; decoder.ok() && i < hint_count; i++) {
1513 TRACE("DecodeCompilationHints[%d] module+%d\n", i,
1514 static_cast<int>(pc_ - start_));
1515
1516 // Compilation hints are encoded in one byte each.
1517 // +-------+----------+---------------+----------+
1518 // | 2 bit | 2 bit | 2 bit | 2 bit |
1519 // | ... | Top tier | Baseline tier | Strategy |
1520 // +-------+----------+---------------+----------+
1521 uint8_t hint_byte = decoder.consume_u8("compilation hint");
1522 if (!decoder.ok()) break;
1523
1524 // Validate the hint_byte.
1525 // For the compilation strategy, all 2-bit values are valid. For the tier,
1526 // only 0x0, 0x1, and 0x2 are allowed.
1527 static_assert(
1528 static_cast<int>(WasmCompilationHintTier::kDefault) == 0 &&
1529 static_cast<int>(WasmCompilationHintTier::kBaseline) == 1 &&
1530 static_cast<int>(WasmCompilationHintTier::kOptimized) == 2,
1531 "The check below assumes that 0x03 is the only invalid 2-bit number "
1532 "for a compilation tier");
1533 if (((hint_byte >> 2) & 0x03) == 0x03 ||
1534 ((hint_byte >> 4) & 0x03) == 0x03) {
1535 decoder.errorf(decoder.pc(),
1536 "Invalid compilation hint %#04x (invalid tier 0x03)",
1537 hint_byte);
1538 break;
1539 }
1540
1541 // Decode compilation hint.
1542 WasmCompilationHint hint;
1543 hint.strategy =
1544 static_cast<WasmCompilationHintStrategy>(hint_byte & 0x03);
1545 hint.baseline_tier =
1546 static_cast<WasmCompilationHintTier>((hint_byte >> 2) & 0x03);
1547 hint.top_tier =
1548 static_cast<WasmCompilationHintTier>((hint_byte >> 4) & 0x03);
1549
1550 // Ensure that the top tier never downgrades a compilation result. If
1551 // baseline and top tier are the same compilation will be invoked only
1552 // once.
1553 if (hint.top_tier < hint.baseline_tier &&
1554 hint.top_tier != WasmCompilationHintTier::kDefault) {
1555 decoder.errorf(decoder.pc(),
1556 "Invalid compilation hint %#04x (forbidden downgrade)",
1557 hint_byte);
1558 }
1559
1560 // Happily accept compilation hint.
1561 if (decoder.ok()) {
1562 module_->compilation_hints.push_back(std::move(hint));
1563 }
1564 }
1565
1566 // If section was invalid reset compilation hints.
1567 if (decoder.failed()) {
1568 module_->compilation_hints.clear();
1569 }
1570
1571 // @TODO(frgossen) Skip the whole compilation hints section in the outer
1572 // decoder if inner decoder was used.
1573 // consume_bytes(static_cast<uint32_t>(end_ - start_), nullptr);
1574 }

1576 void DecodeBranchHintsSection() {
1577 TRACE("DecodeBranchHints module+%d\n", static_cast<int>(pc_ - start_));
1578 if (!has_seen_unordered_section(kBranchHintsSectionCode)) {
1579 set_seen_unordered_section(kBranchHintsSectionCode);
1580 // Use an inner decoder so that errors don't fail the outer decoder.
1581 Decoder inner(start_, pc_, end_, buffer_offset_);
1582 BranchHintInfo branch_hints;
1583
1584 uint32_t func_count = inner.consume_u32v("number of functions");
1585 // Keep track of the previous function index to validate the ordering
1586 int64_t last_func_idx = -1;
1587 for (uint32_t i = 0; i < func_count; i++) {
1588 uint32_t func_idx = inner.consume_u32v("function index");
1589 if (int64_t{func_idx} <= last_func_idx) {
1590 inner.errorf("Invalid function index: %d", func_idx);
1591 break;
1592 }
1593 last_func_idx = func_idx;
1594 uint32_t num_hints = inner.consume_u32v("number of hints");
1595 BranchHintMap func_branch_hints;
1596 TRACE("DecodeBranchHints[%d] module+%d\n", func_idx,
1597 static_cast<int>(inner.pc() - inner.start()));
1598 // Keep track of the previous branch offset to validate the ordering
1599 int64_t last_br_off = -1;
1600 for (uint32_t j = 0; j < num_hints; ++j) {
1601 uint32_t br_off = inner.consume_u32v("branch instruction offset");
1602 if (int64_t{br_off} <= last_br_off) {
1603 inner.errorf("Invalid branch offset: %d", br_off);
1604 break;
1605 }
1606 last_br_off = br_off;
1607 uint32_t data_size = inner.consume_u32v("data size");
1608 if (data_size != 1) {
1609 inner.errorf("Invalid data size: %#x. Expected 1.", data_size);
1610 break;
1611 }
1612 uint32_t br_dir = inner.consume_u8("branch direction");
1613 TRACE("DecodeBranchHints[%d][%d] module+%d\n", func_idx, br_off,
1614 static_cast<int>(inner.pc() - inner.start()));
1615 WasmBranchHint hint;
1616 switch (br_dir) {
1617 case 0:
1618 hint = WasmBranchHint::kUnlikely;
1619 break;
1620 case 1:
1621 hint = WasmBranchHint::kLikely;
1622 break;
1623 default:
1624 hint = WasmBranchHint::kNoHint;
1625 inner.errorf(inner.pc(), "Invalid branch hint %#x", br_dir);
1626 break;
1627 }
1628 if (!inner.ok()) {
1629 break;
1630 }
1631 func_branch_hints.insert(br_off, hint);
1632 }
1633 if (!inner.ok()) {
1634 break;
1635 }
1636 branch_hints.emplace(func_idx, std::move(func_branch_hints));
1637 }
1638 // Extra unexpected bytes are an error.
1639 if (inner.more()) {
1640 inner.errorf("Unexpected extra bytes: %d\n",
1641 static_cast<int>(inner.pc() - inner.start()));
1642 }
1643 // If everything went well, accept the hints for the module.
1644 if (inner.ok()) {
1645 module_->branch_hints = std::move(branch_hints);
1646 }
1647 }
1648 // Skip the whole branch hints section in the outer decoder.
1649 consume_bytes(static_cast<uint32_t>(end_ - start_), nullptr);
1650 }

1726 // Decodes an entire module.
1727 ModuleResult DecodeModule(bool validate_functions) {
1728 // Keep a reference to the wire bytes, in case this decoder gets reset on
1729 // error.
1730 base::Vector<const uint8_t> wire_bytes(start_, end_ - start_);
1731 size_t max_size = max_module_size();
1732 if (wire_bytes.size() > max_size) {
1733 return ModuleResult{WasmError{0, "size > maximum module size (%zu): %zu",
1734 max_size, wire_bytes.size()}};
1735 }
1736
1737 DecodeModuleHeader(wire_bytes);
1738 if (failed()) return toResult(nullptr);
1739
1740 static constexpr uint32_t kWasmHeaderSize = 8;
1741 Decoder section_iterator_decoder(start_ + kWasmHeaderSize, end_,
1742 kWasmHeaderSize);
1743 WasmSectionIterator section_iter(&section_iterator_decoder, tracer_);
1744
1745 while (ok()) {
1746 if (section_iter.section_code() != SectionCode::kUnknownSectionCode) {
1747 uint32_t offset = static_cast<uint32_t>(section_iter.payload().begin() -
1748 wire_bytes.begin());
1749 DecodeSection(section_iter.section_code(), section_iter.payload(),
1750 offset);
1751 if (!ok()) break;
1752 }
1753 if (!section_iter.more()) break;
1754 section_iter.advance(true);
1755 }
1756
1757 // Check for module structure errors before validating function bodies, to
1758 // produce consistent error message independent of whether validation
1759 // happens here or later.
1760 if (section_iterator_decoder.failed()) {
1761 return section_iterator_decoder.toResult(nullptr);
1762 }
1763
1764 ModuleResult result = FinishDecoding();
1765 if (!result.failed() && validate_functions) {
1766 std::function<bool(int)> kNoFilter;
1767 if (WasmError validation_error =
1768 ValidateFunctions(module_.get(), enabled_features_, wire_bytes,
1769 kNoFilter, detected_features_)) {
1770 result = ModuleResult{validation_error};
1771 }
1772 }
1773
1774 if (v8_flags.dump_wasm_module) DumpModule(wire_bytes, result.ok());
1775
1776 return result;
1777 }

1847 // Calculate individual global offsets and total size of globals table. This
1848 // function should be called after all globals have been defined, which is
1849 // after the import section and the global section, but before the global
1850 // offsets are accessed, e.g. by the function compilers. The moment when this
1851 // function should be called is not well-defined, as the global section may
1852 // not exist. Therefore this function is called multiple times.
1853 void CalculateGlobalOffsets(WasmModule* module) {
1854 if (module->globals.empty() || module->untagged_globals_buffer_size != 0 ||
1855 module->tagged_globals_buffer_size != 0) {
1856 // This function has already been executed before, so we don't have to
1857 // execute it again.
1858 return;
1859 }
1860 uint32_t untagged_offset = 0;
1861 uint32_t tagged_offset = 0;

2131 ConstantExpression consume_init_expr(WasmModule* module, ValueType expected,
2132 bool is_shared) {
2133 // The error message mimics the one generated by the {WasmFullDecoder}.
2134#define TYPE_CHECK(found) \
2135 if (V8_UNLIKELY(!IsSubtypeOf(found, expected, module))) { \
2136 errorf(pc() + 1, \
2137 "type error in constant expression[0] (expected %s, got %s)", \
2138 expected.name().c_str(), found.name().c_str()); \
2139 return {}; \
2140 }
2141
2142 if (tracer_) tracer_->NextLineIfNonEmpty();
2143 // To avoid initializing a {WasmFullDecoder} for the most common
2144 // expressions, we replicate their decoding and validation here. The
2145 // manually handled cases correspond to {ConstantExpression}'s kinds.
2146 // We need to make sure to check that the expression ends in {kExprEnd};
2147 // otherwise, it is just the first operand of a composite expression, and we
2148 // fall back to the default case.
2149 if (!more()) {
2150 error("Beyond end of code");
2151 return {};
2152 }
2153 switch (static_cast<WasmOpcode>(*pc())) {
2154 case kExprI32Const: {
2155 auto [value, length] =
2156 read_i32v<FullValidationTag>(pc() + 1, "i32.const");
2157 if (V8_UNLIKELY(failed())) return {};
2158 if (V8_LIKELY(lookahead(1 + length, kExprEnd))) {
2159 TYPE_CHECK(kWasmI32)
2160 if (tracer_) {
2161 tracer_->InitializerExpression(pc_, pc_ + length + 2, kWasmI32);
2162 }
2163 consume_bytes(length + 2);
2164 return ConstantExpression::I32Const(value);
2165 }
2166 break;
2167 }
2168 case kExprRefFunc: {
2169 auto [index, length] =
2170 read_u32v<FullValidationTag>(pc() + 1, "ref.func");
2171 if (V8_UNLIKELY(failed())) return {};
2172 if (V8_LIKELY(lookahead(1 + length, kExprEnd))) {
2173 if (V8_UNLIKELY(index >= module->functions.size())) {
2174 errorf(pc() + 1, "function index %u out of bounds", index);
2175 return {};
2176 }
2177 ValueType type = ValueType::Ref(module->functions[index].sig_index);
2178 TYPE_CHECK(type)
2179 if (V8_UNLIKELY(is_shared && !IsShared(type, module))) {
2180 error(pc(), "ref.func does not have a shared type");
2181 return {};
2182 }
2183 module->functions[index].declared = true;
2184 if (tracer_) {
2185 tracer_->InitializerExpression(pc_, pc_ + length + 2, type);
2186 }
2187 consume_bytes(length + 2);
2188 return ConstantExpression::RefFunc(index);
2189 }
2190 break;
2191 }
2192 case kExprRefNull: {
2193 auto [type, length] =
2194 value_type_reader::read_heap_type<FullValidationTag>(
2195 this, pc() + 1, enabled_features_);
2196 value_type_reader::ValidateHeapType<FullValidationTag>(this, pc_,
2197 module, type);
2198 if (V8_UNLIKELY(failed())) return {};
2199 if (V8_LIKELY(lookahead(1 + length, kExprEnd))) {
2200 TYPE_CHECK(ValueType::RefNull(type))
2201 if (V8_UNLIKELY(is_shared &&
2202 !IsShared(ValueType::RefNull(type), module))) {
2203 error(pc(), "ref.null does not have a shared type");
2204 return {};
2205 }
2206 if (tracer_) {
2207 tracer_->InitializerExpression(pc_, pc_ + length + 2,
2208 ValueType::RefNull(type));
2209 }
2210 consume_bytes(length + 2);
2211 return ConstantExpression::RefNull(type.representation());
2212 }
2213 break;
2214 }
2215 default:
2216 break;
2217 }
2218#undef TYPE_CHECK
2219
2220 auto sig = FixedSizeSignature<ValueType>::Returns(expected);
2221 FunctionBody body(&sig, this->pc_offset(), pc_, end_, is_shared);
2222 WasmDetectedFeatures detected;
2223 ConstantExpression result;
2224 {
2225 // We need a scope for the decoder because its destructor resets some Zone
2226 // elements, which has to be done before we reset the Zone afterwards.
2227 WasmFullDecoder<Decoder::FullValidationTag, ConstantExpressionInterface,
2228 kConstantExpression>
2229 decoder(&init_expr_zone_, module, enabled_features_, &detected, body,
2230 module);
2231
2232 uint32_t offset = this->pc_offset();
2233
2234 decoder.DecodeFunctionBody();
2235
2236 if (tracer_) {
2237 // In case of error, decoder.end() is set to the position right before
2238 // the byte(s) that caused the error. For debugging purposes, we should
2239 // print these bytes, but we don't know how many of them there are, so
2240 // for now we have to guess. For more accurate behavior, we'd have to
2241 // pass {num_invalid_bytes} to every {decoder->DecodeError()} call.
2242 static constexpr size_t kInvalidBytesGuess = 4;
2243 const uint8_t* end =
2244 decoder.ok() ? decoder.end()
2245 : std::min(decoder.end() + kInvalidBytesGuess, end_);
2246 tracer_->InitializerExpression(pc_, end, expected);
2247 }
2248 this->pc_ = decoder.end();
2249
2250 if (decoder.failed()) {
2251 error(decoder.error().offset(), decoder.error().message().c_str());
2252 return {};
2253 }
2254
2255 if (!decoder.interface().end_found()) {
2256 error("constant expression is missing 'end'");
2257 return {};
2258 }
2259
2260 result = ConstantExpression::WireBytes(
2261 offset, static_cast<uint32_t>(decoder.end() - decoder.start()));
2262 }
2263
2264 // We reset the zone here; its memory is not used anymore, and we do not
2265 // want memory from all constant expressions to add up.
2266 init_expr_zone_.Reset();
2267
2268 return result;
2269 }

2425 WasmElemSegment consume_element_segment_header() {
2426 const uint8_t* pos = pc();
2427
2428 // The mask for the bit in the flag which indicates if the segment is
2429 // active or not (0 is active).
2430 constexpr uint8_t kNonActiveMask = 1 << 0;
2431 // The mask for the bit in the flag which indicates:
2432 // - for active tables, if the segment has an explicit table index field.
2433 // - for non-active tables, whether the table is declarative (vs. passive).
2434 constexpr uint8_t kHasTableIndexOrIsDeclarativeMask = 1 << 1;
2435 // The mask for the bit in the flag which indicates if the functions of this
2436 // segment are defined as function indices (0) or constant expressions (1).
2437 constexpr uint8_t kExpressionsAsElementsMask = 1 << 2;
2438 // The mask for the bit which denotes whether this segment is shared.
2439 constexpr uint8_t kSharedFlag = 1 << 3;
2440 constexpr uint8_t kFullMask = kNonActiveMask |
2441 kHasTableIndexOrIsDeclarativeMask |
2442 kExpressionsAsElementsMask | kSharedFlag;
2443
2444 uint32_t flag = consume_u32v("flag", tracer_);
2445 if ((flag & kFullMask) != flag) {
2446 errorf(pos, "illegal flag value %u", flag);
2447 return {};
2448 }
2449
2450 bool is_shared = flag & kSharedFlag;
2451 if (is_shared && !v8_flags.experimental_wasm_shared) {
2452 errorf(pos,
2453 "illegal flag value %u, enable with --experimental-wasm-shared",
2454 flag);
2455 return {};
2456 }
2457 if (is_shared) module_->has_shared_part = true;
2458
2459 const WasmElemSegment::Status status =
2460 (flag & kNonActiveMask) ? (flag & kHasTableIndexOrIsDeclarativeMask)
2461 ? WasmElemSegment::kStatusDeclarative
2462 : WasmElemSegment::kStatusPassive
2463 : WasmElemSegment::kStatusActive;
2464 const bool is_active = status == WasmElemSegment::kStatusActive;
2465 if (tracer_) {
2466 tracer_->Description(": ");
2467 tracer_->Description(status == WasmElemSegment::kStatusActive ? "active"
2468 : status == WasmElemSegment::kStatusPassive
2469 ? "passive,"
2470 : "declarative,");
2471 }
2472
2473 WasmElemSegment::ElementType element_type =
2474 flag & kExpressionsAsElementsMask
2475 ? WasmElemSegment::kExpressionElements
2476 : WasmElemSegment::kFunctionIndexElements;
2477
2478 const bool has_table_index =
2479 is_active && (flag & kHasTableIndexOrIsDeclarativeMask);
2480 uint32_t table_index = 0;
2481 if (has_table_index) {
2482 table_index = consume_u32v(", table index", tracer_);
2483 if (tracer_) tracer_->Description(table_index);
2484 }
2485 if (V8_UNLIKELY(is_active && table_index >= module_->tables.size())) {
2486 // If `has_table_index`, we have an explicit table index. Otherwise, we
2487 // always have the implicit table index 0.
2488 errorf(pos, "out of bounds%s table index %u",
2489 has_table_index ? "" : " implicit", table_index);
