36// Allocate a new ArrayBuffer and initialize it with empty properties and
37// elements.
38// TODO(bmeurer,v8:4153): Rename this and maybe fix up the implementation a bit.
39TNode<JSArrayBuffer> TypedArrayBuiltinsAssembler::AllocateEmptyOnHeapBuffer(
40 TNode<Context> context) {
41 TNode<NativeContext> native_context = LoadNativeContext(context);
42 TNode<Map> map =
43 CAST(LoadContextElement(native_context, Context::ARRAY_BUFFER_MAP_INDEX));
44 TNode<FixedArray> empty_fixed_array = EmptyFixedArrayConstant();
45
46 TNode<JSArrayBuffer> buffer = UncheckedCast<JSArrayBuffer>(
47 Allocate(JSArrayBuffer::kSizeWithEmbedderFields));
48 StoreMapNoWriteBarrier(buffer, map);
49 StoreObjectFieldNoWriteBarrier(buffer, JSArray::kPropertiesOrHashOffset,
50 empty_fixed_array);
51 StoreObjectFieldNoWriteBarrier(buffer, JSArray::kElementsOffset,
52 empty_fixed_array);
53 // Setup the ArrayBuffer.
54 // - Set BitField to 0.
55 // - Set IsExternal and IsDetachable bits of BitFieldSlot.
56 // - Set the byte_length field to zero.
57 // - Set backing_store to null/Tagged<Smi>(0).
58 // - Set extension to null.
59 // - Set all embedder fields to Tagged<Smi>(0).
60 if (FIELD_SIZE(JSArrayBuffer::kOptionalPaddingOffset) != 0) {
61 DCHECK_EQ(4, FIELD_SIZE(JSArrayBuffer::kOptionalPaddingOffset));
62 StoreObjectFieldNoWriteBarrier(
63 buffer, JSArrayBuffer::kOptionalPaddingOffset, Int32Constant(0));
64 }
65 int32_t bitfield_value = (1 << JSArrayBuffer::IsExternalBit::kShift) |
66 (1 << JSArrayBuffer::IsDetachableBit::kShift);
67 StoreObjectFieldNoWriteBarrier(buffer, JSArrayBuffer::kBitFieldOffset,
68 Int32Constant(bitfield_value));
69
70 StoreObjectFieldNoWriteBarrier(buffer, JSArrayBuffer::kDetachKeyOffset,
71 UndefinedConstant());
72 StoreBoundedSizeToObject(buffer, JSArrayBuffer::kRawByteLengthOffset,
73 UintPtrConstant(0));
74 StoreBoundedSizeToObject(buffer, JSArrayBuffer::kRawMaxByteLengthOffset,
75 UintPtrConstant(0));
76 StoreSandboxedPointerToObject(buffer, JSArrayBuffer::kBackingStoreOffset,
77 EmptyBackingStoreBufferConstant());
78#ifdef V8_COMPRESS_POINTERS
79 // When pointer compression is enabled, the extension slot contains a
80 // (lazily-initialized) external pointer handle.
81 StoreObjectFieldNoWriteBarrier(buffer, JSArrayBuffer::kExtensionOffset,
82 ExternalPointerHandleNullConstant());
83#else
84 StoreObjectFieldNoWriteBarrier(buffer, JSArrayBuffer::kExtensionOffset,
85 IntPtrConstant(0));
86#endif
87 InitializeJSAPIObjectWithEmbedderSlotsCppHeapWrapperPtr(buffer);
88 for (int offset = JSArrayBuffer::kHeaderSize;
89 offset < JSArrayBuffer::kSizeWithEmbedderFields; offset += kTaggedSize) {
90 // TODO(v8:10391, saelo): Handle external pointers in EmbedderDataSlot
91 StoreObjectFieldNoWriteBarrier(buffer, offset, SmiConstant(0));
92 }
93 return buffer;
94}

399void TypedArrayBuiltinsAssembler::DispatchTypedArrayByElementsKind(
400 TNode<Word32T> elements_kind, const TypedArraySwitchCase& case_function) {
401 Label next(this), if_unknown_type(this, Label::kDeferred);
402
403 int32_t elements_kinds[] = {
404#define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) TYPE##_ELEMENTS,
405 TYPED_ARRAYS(TYPED_ARRAY_CASE) RAB_GSAB_TYPED_ARRAYS(TYPED_ARRAY_CASE)
406#undef TYPED_ARRAY_CASE
407 };
408
409#define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) Label if_##type##array(this);
410 TYPED_ARRAYS(TYPED_ARRAY_CASE)
411 RAB_GSAB_TYPED_ARRAYS(TYPED_ARRAY_CASE)
412#undef TYPED_ARRAY_CASE
413
414 Label* elements_kind_labels[] = {
415#define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) &if_##type##array,
416 TYPED_ARRAYS(TYPED_ARRAY_CASE) RAB_GSAB_TYPED_ARRAYS(TYPED_ARRAY_CASE)
417#undef TYPED_ARRAY_CASE
418 };
419 static_assert(arraysize(elements_kinds) == arraysize(elements_kind_labels));
420
421 Switch(elements_kind, &if_unknown_type, elements_kinds, elements_kind_labels,
422 arraysize(elements_kinds));
423
424#define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) \
425 BIND(&if_##type##array); \
426 { \
427 case_function(TYPE##_ELEMENTS, sizeof(ctype), \
428 Context::TYPE##_ARRAY_FUN_INDEX); \
429 Goto(&next); \
430 }
431 TYPED_ARRAYS(TYPED_ARRAY_CASE)
432#undef TYPED_ARRAY_CASE
433
434#define TYPED_ARRAY_CASE(Type, type, TYPE, ctype, NON_RAB_GSAB_TYPE) \
435 BIND(&if_##type##array); \
436 { \
437 case_function(TYPE##_ELEMENTS, sizeof(ctype), \
438 Context::NON_RAB_GSAB_TYPE##_ARRAY_FUN_INDEX); \
439 Goto(&next); \
440 }
441 RAB_GSAB_TYPED_ARRAYS_WITH_NON_RAB_GSAB_ELEMENTS_KIND(TYPED_ARRAY_CASE)
442#undef TYPED_ARRAY_CASE
443
444 BIND(&if_unknown_type);
445 Unreachable();
446
447 BIND(&next);
448}

482void TypedArrayBuiltinsAssembler::StoreJSTypedArrayElementFromNumeric(
483 TNode<Context> context, TNode<JSTypedArray> typed_array,
484 TNode<UintPtrT> index, TNode<Numeric> value, ElementsKind elements_kind) {
485 TNode<RawPtrT> data_ptr = LoadJSTypedArrayDataPtr(typed_array);
486 switch (elements_kind) {
487 case UINT8_ELEMENTS:
488 case UINT8_CLAMPED_ELEMENTS:
489 case INT8_ELEMENTS:
490 case UINT16_ELEMENTS:
491 case INT16_ELEMENTS:
492 StoreElement(data_ptr, elements_kind, index, SmiToInt32(CAST(value)));
493 break;
494 case UINT32_ELEMENTS:
495 case INT32_ELEMENTS:
496 StoreElement(data_ptr, elements_kind, index,
497 TruncateTaggedToWord32(context, value));
498 break;
499 case FLOAT16_ELEMENTS:
500 StoreElement(data_ptr, elements_kind, index,
501 TruncateFloat64ToFloat16(LoadHeapNumberValue(CAST(value))));
502 break;
503 case FLOAT32_ELEMENTS:
504 StoreElement(data_ptr, elements_kind, index,
505 TruncateFloat64ToFloat32(LoadHeapNumberValue(CAST(value))));
506 break;
507 case FLOAT64_ELEMENTS:
508 StoreElement(data_ptr, elements_kind, index,
509 LoadHeapNumberValue(CAST(value)));
510 break;
511 case BIGUINT64_ELEMENTS:
512 case BIGINT64_ELEMENTS:
513 StoreElement(data_ptr, elements_kind, index,
514 UncheckedCast<BigInt>(value));
515 break;
516 default:
517 UNREACHABLE();
518 }
519}

521template <typename TValue>
522void TypedArrayBuiltinsAssembler::StoreJSTypedArrayElementFromPreparedValue(
523 TNode<Context> context, TNode<JSTypedArray> typed_array,
524 TNode<UintPtrT> index, TNode<TValue> prepared_value,
525 ElementsKind elements_kind, Label* if_detached_or_out_of_bounds) {
526 static_assert(std::is_same_v<TValue, Word32T> ||
527 std::is_same_v<TValue, Float16RawBitsT> ||
528 std::is_same_v<TValue, Float32T> ||
529 std::is_same_v<TValue, Float64T> ||
530 std::is_same_v<TValue, BigInt>,
531 "Only Word32T, Float16T, Float32T, Float64T or BigInt values "
532 "are allowed");
533 // ToNumber/ToBigInt (or other functions called by the upper level) may
534 // execute JavaScript code, which could detach the TypedArray's buffer or make
535 // the TypedArray out of bounds.
536 TNode<UintPtrT> length = LoadJSTypedArrayLengthAndCheckDetached(
537 typed_array, if_detached_or_out_of_bounds);
538 GotoIf(UintPtrGreaterThanOrEqual(index, length),
539 if_detached_or_out_of_bounds);
540
541 TNode<RawPtrT> data_ptr = LoadJSTypedArrayDataPtr(typed_array);
542 StoreElement(data_ptr, elements_kind, index, prepared_value);
543}

609 // FIRST_FIXED_TYPED_ARRAY_ELEMENTS_KIND.
610 static_assert(LAST_FIXED_TYPED_ARRAY_ELEMENTS_KIND + 1 ==
611 FIRST_RAB_GSAB_FIXED_TYPED_ARRAY_ELEMENTS_KIND);
612 size_t const kTypedElementsKindCount =
613 LAST_RAB_GSAB_FIXED_TYPED_ARRAY_ELEMENTS_KIND -
614 FIRST_FIXED_TYPED_ARRAY_ELEMENTS_KIND + 1;
615#define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) \
616 Label return_##type##array(this); \
617 BIND(&return_##type##array); \
618 Return(StringConstant(#Type "Array"));
619 TYPED_ARRAYS(TYPED_ARRAY_CASE)
620#undef TYPED_ARRAY_CASE
621
622 // clang-format off
623 Label* elements_kind_labels[kTypedElementsKindCount] = {
624 // The TYPED_ARRAYS macro is invoked twice because while the RAB/GSAB-backed
625 // TAs have distinct ElementsKinds internally, they have the same "class"
626 // name for toString output.
627#define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) &return_##type##array,
628 TYPED_ARRAYS(TYPED_ARRAY_CASE)
629 TYPED_ARRAYS(TYPED_ARRAY_CASE)
630#undef TYPED_ARRAY_CASE
631 };
632
633 int32_t elements_kinds[kTypedElementsKindCount] = {
634#define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) \
635 TYPE##_ELEMENTS - FIRST_FIXED_TYPED_ARRAY_ELEMENTS_KIND,
636 // The use of FIRST_FIXED_TYPED_ARRAY_ELEMENTS_KIND below is not a typo! This
637 // computes an index into elements_kind_labels, and all TypedArray
638 // ElementsKind values are contiguous.
639#define RAB_GSAB_TYPED_ARRAY_CASE(Type, type, TYPE, ctype) \
640 TYPE##_ELEMENTS - FIRST_FIXED_TYPED_ARRAY_ELEMENTS_KIND,
641 TYPED_ARRAYS(TYPED_ARRAY_CASE)
642 RAB_GSAB_TYPED_ARRAYS(RAB_GSAB_TYPED_ARRAY_CASE)
643#undef TYPED_ARRAY_CASE
644#undef RAB_GSAB_TYPED_ARRAY_CASE
645 };
646 // clang-format on
647
648 // We offset the dispatch by FIRST_FIXED_TYPED_ARRAY_ELEMENTS_KIND, so that
649 // this can be turned into a non-sparse table switch for ideal performance.
650 BIND(&if_receiverisheapobject);
651 TNode<HeapObject> receiver_heap_object = CAST(receiver);
652 TNode<Int32T> elements_kind =
653 Int32Sub(LoadElementsKind(receiver_heap_object),
654 Int32Constant(FIRST_FIXED_TYPED_ARRAY_ELEMENTS_KIND));
655 Switch(elements_kind, &return_undefined, elements_kinds, elements_kind_labels,
656 kTypedElementsKindCount);

658 BIND(&return_undefined);
