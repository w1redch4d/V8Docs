25FieldIndex FieldIndex::ForSmiLoadHandler(Tagged<Map> map, int32_t handler) {
26 DCHECK_EQ(LoadHandler::KindBits::decode(handler), LoadHandler::Kind::kField);
27
28 bool is_inobject = LoadHandler::IsInobjectBits::decode(handler);
29 int inobject_properties = map->GetInObjectProperties();
30 int first_inobject_offset;
31 if (is_inobject) {
32 first_inobject_offset = map->GetInObjectPropertyOffset(0);
33 } else {
34 first_inobject_offset = OFFSET_OF_DATA_START(FixedArray);
35 }
36 return FieldIndex(
37 is_inobject, LoadHandler::FieldIndexBits::decode(handler) * kTaggedSize,
38 LoadHandler::IsDoubleBits::decode(handler) ? kDouble : kTagged,
39 inobject_properties, first_inobject_offset);
40}

42FieldIndex FieldIndex::ForPropertyIndex(Tagged<Map> map, int property_index,
43 Representation representation) {
44 DCHECK(map->instance_type() >= FIRST_NONSTRING_TYPE);
45 int inobject_properties = map->GetInObjectProperties();
46 bool is_inobject = property_index < inobject_properties;
47 int first_inobject_offset;
48 int offset;
49 if (is_inobject) {
50 first_inobject_offset = map->GetInObjectPropertyOffset(0);
51 offset = map->GetInObjectPropertyOffset(property_index);
52 } else {
53 first_inobject_offset = OFFSET_OF_DATA_START(FixedArray);
54 property_index -= inobject_properties;
55 offset = PropertyArray::OffsetOfElementAt(property_index);
56 }
57 Encoding encoding = FieldEncoding(representation);
58 return FieldIndex(is_inobject, offset, encoding, inobject_properties,
59 first_inobject_offset);
60}

62// Returns the index format accepted by the LoadFieldByIndex instruction.
63// (In-object: zero-based from (object start + JSObject::kHeaderSize),
64// out-of-object: zero-based from OFFSET_OF_DATA_START(FixedArray).)
65int FieldIndex::GetLoadByFieldIndex() const {
66 // For efficiency, the LoadByFieldIndex instruction takes an index that is
67 // optimized for quick access. If the property is inline, the index is
68 // positive. If it's out-of-line, the encoded index is -raw_index - 1 to
69 // disambiguate the zero out-of-line index from the zero inobject case.
70 // The index itself is shifted up by one bit, the lower-most bit
71 // signifying if the field is a mutable double box (1) or not (0).
72 int result = index();
73 if (is_inobject()) {
74 result -= JSObject::kHeaderSize / kTaggedSize;
75 } else {
76 result -= OFFSET_OF_DATA_START(FixedArray) / kTaggedSize;
77 result = -result - 1;
78 }
79 result = static_cast<uint32_t>(result) << 1;
80 return is_double() ? (result | 1) : result;
81}
