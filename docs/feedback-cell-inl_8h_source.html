28void FeedbackCell::clear_padding() {
29 if (FeedbackCell::kAlignedSize == FeedbackCell::kUnalignedSize) return;
30 DCHECK_GE(FeedbackCell::kAlignedSize, FeedbackCell::kUnalignedSize);
31 memset(reinterpret_cast<uint8_t*>(address() + FeedbackCell::kUnalignedSize),
32 0, FeedbackCell::kAlignedSize - FeedbackCell::kUnalignedSize);
33}

35void FeedbackCell::reset_feedback_vector(
36 std::optional<std::function<void(Tagged<HeapObject> object, ObjectSlot slot,
37 Tagged<HeapObject> target)>>
38 gc_notify_updated_slot) {
39 clear_interrupt_budget();
40 if (IsUndefined(value()) || IsClosureFeedbackCellArray(value())) return;
41
42 CHECK(IsFeedbackVector(value()));
43 Tagged<ClosureFeedbackCellArray> closure_feedback_cell_array =
44 Cast<FeedbackVector>(value())->closure_feedback_cell_array();
45 set_value(closure_feedback_cell_array, kReleaseStore);
46 if (gc_notify_updated_slot) {
47 (*gc_notify_updated_slot)(*this, RawField(FeedbackCell::kValueOffset),
48 closure_feedback_cell_array);
49 }
50}

85FeedbackCell::ClosureCountTransition FeedbackCell::IncrementClosureCount(
86 Isolate* isolate) {
87 ReadOnlyRoots r(isolate);
88 if (map() == r.no_closures_cell_map()) {
89 set_map(isolate, r.one_closure_cell_map());
90 return kNoneToOne;
91 } else if (map() == r.one_closure_cell_map()) {
92 set_map(isolate, r.many_closures_cell_map());
93 return kOneToMany;
94 } else {
95 DCHECK(map() == r.many_closures_cell_map());
96 return kMany;
97 }
98}
