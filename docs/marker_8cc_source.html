38bool EnterIncrementalMarkingIfNeeded(MarkingConfig config, HeapBase& heap) {
39 if (config.marking_type == MarkingConfig::MarkingType::kIncremental ||
40 config.marking_type ==
41 MarkingConfig::MarkingType::kIncrementalAndConcurrent) {
42 WriteBarrier::FlagUpdater::Enter();
43 heap.set_incremental_marking_in_progress(true);
44 return true;
45 }
46 return false;
47}
48
49bool ExitIncrementalMarkingIfNeeded(MarkingConfig config, HeapBase& heap) {
50 if (config.marking_type == MarkingConfig::MarkingType::kIncremental ||
51 config.marking_type ==
52 MarkingConfig::MarkingType::kIncrementalAndConcurrent) {
53 WriteBarrier::FlagUpdater::Exit();
54 heap.set_incremental_marking_in_progress(false);
55 return true;
56 }
57 return false;
58}

60static constexpr size_t kDefaultDeadlineCheckInterval = 150u;

62template <size_t kDeadlineCheckInterval = kDefaultDeadlineCheckInterval,
63 typename WorklistLocal, typename Callback>
64bool DrainWorklistWithBytesAndTimeDeadline(BasicMarkingState& marking_state,
65 size_t marked_bytes_deadline,
66 v8::base::TimeTicks time_deadline,
67 WorklistLocal& worklist_local,
68 Callback callback) {
69 return DrainWorklistWithPredicate<kDeadlineCheckInterval>(
70 [&marking_state, marked_bytes_deadline, time_deadline]() {
71 return (marked_bytes_deadline <= marking_state.marked_bytes()) ||
72 (time_deadline <= v8::base::TimeTicks::Now());
73 },
74 worklist_local, callback);
75}

110// static
111MarkerBase::IncrementalMarkingTask::Handle
112MarkerBase::IncrementalMarkingTask::Post(cppgc::TaskRunner* runner,
113 MarkerBase* marker) {
114 // Incremental GC is possible only via the GCInvoker, so getting here
115 // guarantees that either non-nestable tasks or conservative stack
116 // scanning are supported. This is required so that the incremental
117 // task can safely finalize GC if needed.
118 DCHECK_IMPLIES(marker->heap().stack_support() !=
119 HeapBase::StackSupport::kSupportsConservativeStackScan,
120 runner->NonNestableTasksEnabled());
121
122 const bool non_nestable_tasks_enabled = runner->NonNestableTasksEnabled();
123
124 auto task = std::make_unique<IncrementalMarkingTask>(
125 marker, non_nestable_tasks_enabled ? StackState::kNoHeapPointers
126 : StackState::kMayContainHeapPointers);
127 auto handle = task->handle_;
128 if (non_nestable_tasks_enabled) {
129 runner->PostNonNestableTask(std::move(task));
130 } else {
131 runner->PostTask(std::move(task));
132 }
133 return handle;
134}

136void MarkerBase::IncrementalMarkingTask::Run() {
137 if (handle_.IsCanceled()) return;
138
139 StatsCollector::EnabledScope stats_scope(marker_->heap().stats_collector(),
140 StatsCollector::kIncrementalMark);
141
142 if (marker_->IncrementalMarkingStep(stack_state_)) {
143 // Incremental marking is done so should finalize GC.
144 marker_->heap().FinalizeIncrementalGarbageCollectionIfNeeded(stack_state_);
145 }
146}

165MarkerBase::~MarkerBase() {
166 // The fixed point iteration may have found not-fully-constructed objects.
167 // Such objects should have already been found through the stack scan though
168 // and should thus already be marked.
169 if (!marking_worklists_.not_fully_constructed_worklist()->IsEmpty()) {
170#if DEBUG
171 DCHECK_NE(StackState::kNoHeapPointers, config_.stack_state);
172 std::unordered_set<HeapObjectHeader*> objects =
173 mutator_marking_state_.not_fully_constructed_worklist().Extract();
174 for (HeapObjectHeader* object : objects) DCHECK(object->IsMarked());
175#else
176 marking_worklists_.not_fully_constructed_worklist()->Clear();
177#endif
178 }
179
180 // |discovered_ephemeron_pairs_worklist_| may still hold ephemeron pairs with
181 // dead keys.
182 if (!marking_worklists_.discovered_ephemeron_pairs_worklist()->IsEmpty()) {
183#if DEBUG
184 MarkingWorklists::EphemeronPairItem item;
185 while (mutator_marking_state_.discovered_ephemeron_pairs_worklist().Pop(
186 &item)) {
187 DCHECK(!HeapObjectHeader::FromObject(item.key).IsMarked());
188 }
189#else
190 marking_worklists_.discovered_ephemeron_pairs_worklist()->Clear();
191#endif
192 }
193
194 marking_worklists_.weak_containers_worklist()->Clear();
195}

210void MarkerBase::StartMarking() {
211 DCHECK(!is_marking_);
212 StatsCollector::EnabledScope stats_scope(
213 heap().stats_collector(),
214 config_.marking_type == MarkingConfig::MarkingType::kAtomic
215 ? StatsCollector::kAtomicMark
216 : StatsCollector::kIncrementalMark);
217
218 heap().stats_collector()->NotifyMarkingStarted(
219 config_.collection_type, config_.marking_type, config_.is_forced_gc);
220
221 is_marking_ = true;
222 if (EnterIncrementalMarkingIfNeeded(config_, heap())) {
223 StatsCollector::EnabledScope inner_stats_scope(
224 heap().stats_collector(), StatsCollector::kMarkIncrementalStart);
225
226 // Performing incremental or concurrent marking.
227 schedule_->NotifyIncrementalMarkingStart();
228 // Scanning the stack is expensive so we only do it at the atomic pause.
229 VisitLocalRoots(StackState::kNoHeapPointers);
230 ScheduleIncrementalMarkingTask();
231 if (config_.marking_type ==
232 MarkingConfig::MarkingType::kIncrementalAndConcurrent) {
233 mutator_marking_state_.Publish();
234 concurrent_marker_->Start();
235 }
236 incremental_marking_allocation_observer_ =
237 std::make_unique<IncrementalMarkingAllocationObserver>(*this);
238 heap().stats_collector()->RegisterObserver(
239 incremental_marking_allocation_observer_.get());
240 }
241}

243void MarkerBase::HandleNotFullyConstructedObjects() {

245 mutator_marking_state_.FlushNotFullyConstructedObjects();
246 } else {
247 MarkNotFullyConstructedObjects();
248 }

251void MarkerBase::EnterAtomicPause(StackState stack_state) {
252 StatsCollector::EnabledScope top_stats_scope(heap().stats_collector(),
253 StatsCollector::kAtomicMark);
254 StatsCollector::EnabledScope stats_scope(heap().stats_collector(),
255 StatsCollector::kMarkAtomicPrologue);
256
257 const MarkingConfig::MarkingType old_marking_type = config_.marking_type;
258
259 if (ExitIncrementalMarkingIfNeeded(config_, heap())) {
260 // Cancel remaining incremental tasks. Concurrent marking jobs are left to
261 // run in parallel with the atomic pause until the mutator thread runs out
262 // of work.
263 incremental_marking_handle_.Cancel();
264 heap().stats_collector()->UnregisterObserver(
265 incremental_marking_allocation_observer_.get());
266 incremental_marking_allocation_observer_.reset();
267 }
268 config_.stack_state = stack_state;
269 config_.marking_type = MarkingConfig::MarkingType::kAtomic;
270 mutator_marking_state_.set_in_atomic_pause();
271
272 {
273 // VisitLocalRoots() also resets the LABs.
274 VisitLocalRoots(config_.stack_state);
275 HandleNotFullyConstructedObjects();
276 }
277 if (old_marking_type ==
278 MarkingConfig::MarkingType::kIncrementalAndConcurrent) {
279 // Start parallel marking.
280 mutator_marking_state_.Publish();
281 if (concurrent_marker_->IsActive()) {
282 concurrent_marker_->NotifyIncrementalMutatorStepCompleted();
283 } else {
284 concurrent_marker_->Start();
285 }
286 }
287}

289void MarkerBase::ReEnableConcurrentMarking() {
290 CHECK(is_marking_);
291
292 if (config_.marking_type == MarkingConfig::MarkingType::kAtomic) {
293 return;
294 }
295
296 CHECK_EQ(config_.marking_type, MarkingConfig::MarkingType::kIncremental);
297 config_.marking_type = MarkingConfig::MarkingType::kIncrementalAndConcurrent;
298 mutator_marking_state_.Publish();
299 CHECK(!concurrent_marker_->IsActive());
300 concurrent_marker_->Start();
301 CHECK(concurrent_marker_->IsActive());
302}

304void MarkerBase::LeaveAtomicPause() {
305 {
306 StatsCollector::EnabledScope top_stats_scope(heap().stats_collector(),
307 StatsCollector::kAtomicMark);
308 StatsCollector::EnabledScope stats_scope(
309 heap().stats_collector(), StatsCollector::kMarkAtomicEpilogue);
310 DCHECK(!incremental_marking_handle_);
311 heap().stats_collector()->NotifyMarkingCompleted(
312 // GetOverallMarkedBytes also includes concurrently marked bytes.
313 schedule_->GetOverallMarkedBytes());
314 is_marking_ = false;
315 }
316 {
317 // Weakness callbacks are forbidden from allocating objects.
318 cppgc::subtle::DisallowGarbageCollectionScope disallow_gc_scope(heap_);
319 ProcessWeakness();
320 }
321 heap().SetStackStateOfPrevGC(config_.stack_state);
322}

324void MarkerBase::EnterProcessGlobalAtomicPause() { VisitCrossThreadRoots(); }

326void MarkerBase::FinishMarking(StackState stack_state) {
327 DCHECK(is_marking_);
328 EnterAtomicPause(stack_state);
329 EnterProcessGlobalAtomicPause();
330 {
331 StatsCollector::EnabledScope stats_scope(heap().stats_collector(),
332 StatsCollector::kAtomicMark);
333 CHECK(AdvanceMarkingWithLimits(v8::base::TimeDelta::Max(), SIZE_MAX));
334 if (JoinConcurrentMarkingIfNeeded()) {
335 CHECK(AdvanceMarkingWithLimits(v8::base::TimeDelta::Max(), SIZE_MAX));
336 }
337 mutator_marking_state_.Publish();
338 }
339 LeaveAtomicPause();
340}

342class WeakCallbackJobTask final : public cppgc::JobTask {
343 public:
344 WeakCallbackJobTask(MarkerBase* marker,
345 MarkingWorklists::WeakCallbackWorklist* callback_worklist,
346 LivenessBroker& broker)
347 : marker_(marker),
348 callback_worklist_(callback_worklist),
349 broker_(broker) {}
350
351 void Run(JobDelegate* delegate) override {
352 StatsCollector::EnabledConcurrentScope stats_scope(
353 marker_->heap().stats_collector(),
354 StatsCollector::kConcurrentWeakCallback);
355 MarkingWorklists::WeakCallbackWorklist::Local local(*callback_worklist_);
356 MarkingWorklists::WeakCallbackItem item;
357 while (local.Pop(&item)) {
358 item.callback(broker_, item.parameter);
359 }
360 }
361
362 size_t GetMaxConcurrency(size_t worker_count) const override {
363 return std::min(static_cast<size_t>(1),
364 callback_worklist_->Size() + worker_count);
365 }
366
367 private:
368 MarkerBase* marker_;
369 MarkingWorklists::WeakCallbackWorklist* callback_worklist_;
370 LivenessBroker& broker_;
371};

373void MarkerBase::ProcessWeakness() {
374 DCHECK_EQ(MarkingConfig::MarkingType::kAtomic, config_.marking_type);
375
376 StatsCollector::EnabledScope stats_scope(heap().stats_collector(),
377 StatsCollector::kAtomicWeak);
378
379 RootMarkingVisitor root_marking_visitor(mutator_marking_state_);
380
381 // Processing cross-thread roots requires taking the global process lock.
382 // Process these weak roots first to minimize the time the lock is held.
383 g_process_mutex.Get().AssertHeld();
384 CHECK(visited_cross_thread_persistents_in_atomic_pause_);
385 heap().GetWeakCrossThreadPersistentRegion().Iterate(root_marking_visitor);
386 g_process_mutex.Pointer()->Unlock();
387
388 // Launch the parallel job before anything else to provide the maximum time
389 // slice for processing.
390 LivenessBroker broker = LivenessBrokerFactory::Create();
391 std::unique_ptr<cppgc::JobHandle> job_handle{nullptr};
392 if (heap().marking_support() ==
393 cppgc::Heap::MarkingType::kIncrementalAndConcurrent) {
394 job_handle = platform_->PostJob(
395 cppgc::TaskPriority::kUserBlocking,
396 std::make_unique<WeakCallbackJobTask>(
397 this, marking_worklists_.parallel_weak_callback_worklist(),
398 broker));
399 }
400
401 // Process same-thread roots.
402 heap().GetWeakPersistentRegion().Iterate(root_marking_visitor);
403
404 // Call weak callbacks on objects that may now be pointing to dead objects.
405#if defined(CPPGC_YOUNG_GENERATION)
406 if (heap().generational_gc_supported()) {
407 auto& remembered_set = heap().remembered_set();
408 if (config_.collection_type == CollectionType::kMinor) {
409 // Custom callbacks assume that untraced pointers point to not yet freed
410 // objects. They must make sure that upon callback completion no
411 // UntracedMember points to a freed object. This may not hold true if a
412 // custom callback for an old object operates with a reference to a young
413 // object that was freed on a minor collection cycle. To maintain the
414 // invariant that UntracedMembers always point to valid objects, execute
415 // custom callbacks for old objects on each minor collection cycle.
416 remembered_set.ExecuteCustomCallbacks(broker);
417 } else {
418 // For major GCs, just release all the remembered weak callbacks.
419 remembered_set.ReleaseCustomCallbacks();
420 }
421 }
422#endif // defined(CPPGC_YOUNG_GENERATION)
423
424 {
425 // First, process weak container callbacks.
426 StatsCollector::EnabledScope stats_scope(
427 heap().stats_collector(),
428 StatsCollector::kWeakContainerCallbacksProcessing);
429 MarkingWorklists::WeakCallbackItem item;
430 MarkingWorklists::WeakCallbackWorklist::Local& collections_local =
431 mutator_marking_state_.weak_container_callback_worklist();
432 while (collections_local.Pop(&item)) {
433 item.callback(broker, item.parameter);
434 }
435 }
436 {
437 // Then, process custom weak callbacks.
438 StatsCollector::EnabledScope stats_scope(
439 heap().stats_collector(), StatsCollector::kCustomCallbacksProcessing);
440 MarkingWorklists::WeakCallbackItem item;
441 MarkingWorklists::WeakCustomCallbackWorklist::Local& custom_callbacks =
442 mutator_marking_state_.weak_custom_callback_worklist();
443 while (custom_callbacks.Pop(&item)) {
444 item.callback(broker, item.parameter);
445#if defined(CPPGC_YOUNG_GENERATION)
446 if (heap().generational_gc_supported())
447 heap().remembered_set().AddWeakCallback(item);
448#endif // defined(CPPGC_YOUNG_GENERATION)
449 }
450 }
451
452 if (job_handle) {
453 job_handle->Join();
454 } else {
455 MarkingWorklists::WeakCallbackItem item;
456 MarkingWorklists::WeakCallbackWorklist::Local& local =
457 mutator_marking_state_.parallel_weak_callback_worklist();
458 while (local.Pop(&item)) {
459 item.callback(broker, item.parameter);
460 }
461 }
462
463 // Weak callbacks should not add any new objects for marking.
464 DCHECK(marking_worklists_.marking_worklist()->IsEmpty());
465}

467void MarkerBase::VisitLocalRoots(StackState stack_state) {
468 StatsCollector::EnabledScope stats_scope(heap().stats_collector(),
469 StatsCollector::kMarkVisitRoots);
470
471 // Reset LABs before scanning roots. LABs are cleared to allow
472 // ObjectStartBitmap handling without considering LABs.
473 heap().object_allocator().ResetLinearAllocationBuffers();
474
475 {
476 StatsCollector::DisabledScope inner_stats_scope(
477 heap().stats_collector(), StatsCollector::kMarkVisitPersistents);
478 RootMarkingVisitor root_marking_visitor(mutator_marking_state_);
479 heap().GetStrongPersistentRegion().Iterate(root_marking_visitor);
480 }
481
482 if (stack_state != StackState::kNoHeapPointers) {
483 StatsCollector::DisabledScope stack_stats_scope(
484 heap().stats_collector(), StatsCollector::kMarkVisitStack);
485 heap().stack()->SetMarkerIfNeededAndCallback([this]() {
486 heap().stack()->IteratePointersUntilMarker(&stack_visitor());
487 });
488 }
489
490#if defined(CPPGC_YOUNG_GENERATION)
491 if (config_.collection_type == CollectionType::kMinor) {
492 StatsCollector::EnabledScope stats_scope(
493 heap().stats_collector(), StatsCollector::kMarkVisitRememberedSets);
494 heap().remembered_set().Visit(visitor(), conservative_visitor(),
495 mutator_marking_state_);
496 }
497#endif // defined(CPPGC_YOUNG_GENERATION)
498}

500void MarkerBase::VisitCrossThreadRoots() {
501 StatsCollector::DisabledScope inner_stats_scope(
502 heap().stats_collector(),
503 StatsCollector::kMarkVisitCrossThreadPersistents);
504 CHECK_EQ(config_.marking_type, MarkingConfig::MarkingType::kAtomic);
505 CHECK(!visited_cross_thread_persistents_in_atomic_pause_);
506 // Lock guards against changes to {Weak}CrossThreadPersistent handles, that
507 // may conflict with marking. E.g., a WeakCrossThreadPersistent may be
508 // converted into a CrossThreadPersistent which requires that the handle
509 // is either cleared or the object is retained.
510 g_process_mutex.Pointer()->Lock();
511 RootMarkingVisitor root_marking_visitor(mutator_marking_state_);
512 heap().GetStrongCrossThreadPersistentRegion().Iterate(root_marking_visitor);
513 visited_cross_thread_persistents_in_atomic_pause_ = true;
514}

566bool MarkerBase::AdvanceMarkingWithLimits(v8::base::TimeDelta max_duration,
567 size_t marked_bytes_limit) {
568 bool is_done = false;
569 if (!main_marking_disabled_for_testing_) {
570 if (marked_bytes_limit == 0) {
571 marked_bytes_limit = mutator_marking_state_.marked_bytes() +
572 GetNextIncrementalStepDuration(*schedule_, heap_);
573 }
574 StatsCollector::EnabledScope deadline_scope(
575 heap().stats_collector(),
576 StatsCollector::kMarkTransitiveClosureWithDeadline, "deadline_ms",
577 max_duration.InMillisecondsF());
578 const auto deadline = v8::base::TimeTicks::Now() + max_duration;
579 is_done = ProcessWorklistsWithDeadline(marked_bytes_limit, deadline);
580 schedule_->UpdateMutatorThreadMarkedBytes(
581 mutator_marking_state_.marked_bytes());
582 }
583 mutator_marking_state_.Publish();
584 if (!is_done) {
585 // If marking is atomic, |is_done| should always be true.
586 DCHECK_NE(MarkingConfig::MarkingType::kAtomic, config_.marking_type);
587 ScheduleIncrementalMarkingTask();
588 if (config_.marking_type ==
589 MarkingConfig::MarkingType::kIncrementalAndConcurrent) {
590 concurrent_marker_->NotifyIncrementalMutatorStepCompleted();
591 }
592 }
593 return is_done;
594}

596bool MarkerBase::ProcessWorklistsWithDeadline(
597 size_t marked_bytes_deadline, v8::base::TimeTicks time_deadline) {
598 StatsCollector::EnabledScope stats_scope(
599 heap().stats_collector(), StatsCollector::kMarkTransitiveClosure);
600 bool saved_did_discover_new_ephemeron_pairs;
601 do {
602 mutator_marking_state_.ResetDidDiscoverNewEphemeronPairs();
603 if ((config_.marking_type == MarkingConfig::MarkingType::kAtomic) ||
604 schedule_->ShouldFlushEphemeronPairs()) {
605 mutator_marking_state_.FlushDiscoveredEphemeronPairs();
606 }
607
608 // Bailout objects may be complicated to trace and thus might take longer
609 // than other objects. Therefore we reduce the interval between deadline
610 // checks to guarantee the deadline is not exceeded.
611 {
612 StatsCollector::DisabledScope inner_scope(
613 heap().stats_collector(), StatsCollector::kMarkProcessBailOutObjects);
614 if (!DrainWorklistWithBytesAndTimeDeadline<kDefaultDeadlineCheckInterval /
615 5>(
616 mutator_marking_state_, SIZE_MAX, time_deadline,
617 mutator_marking_state_.concurrent_marking_bailout_worklist(),
618 [this](
619 const MarkingWorklists::ConcurrentMarkingBailoutItem& item) {
620 mutator_marking_state_.AccountMarkedBytes(
621 BasePage::FromPayload(const_cast<void*>(item.parameter)),
622 item.bailedout_size);
623 item.callback(&visitor(), item.parameter);
624 })) {
625 return false;
626 }
627 }
628
629 {
630 StatsCollector::DisabledScope inner_scope(
631 heap().stats_collector(),
632 StatsCollector::kMarkProcessNotFullyconstructedWorklist);
633 if (!DrainWorklistWithBytesAndTimeDeadline(
634 mutator_marking_state_, marked_bytes_deadline, time_deadline,
635 mutator_marking_state_
636 .previously_not_fully_constructed_worklist(),
637 [this](HeapObjectHeader* header) {
638 mutator_marking_state_.AccountMarkedBytes(*header);
639 DynamicallyTraceMarkedObject<AccessMode::kNonAtomic>(visitor(),
640 *header);
641 })) {
642 return false;
643 }
644 }
645
646 {
647 StatsCollector::DisabledScope inner_scope(
648 heap().stats_collector(),
649 StatsCollector::kMarkProcessMarkingWorklist);
650 if (!DrainWorklistWithBytesAndTimeDeadline(
651 mutator_marking_state_, marked_bytes_deadline, time_deadline,
652 mutator_marking_state_.marking_worklist(),
653 [this](const MarkingWorklists::MarkingItem& item) {
654 const HeapObjectHeader& header =
655 HeapObjectHeader::FromObject(item.base_object_payload);
656 DCHECK(!header.IsInConstruction<AccessMode::kNonAtomic>());
657 DCHECK(header.IsMarked<AccessMode::kAtomic>());
658 mutator_marking_state_.AccountMarkedBytes(header);
659 item.callback(&visitor(), item.base_object_payload);
660 })) {
661 return false;
662 }
663 }
664
665 {
666 StatsCollector::DisabledScope inner_scope(
667 heap().stats_collector(),
668 StatsCollector::kMarkProcessWriteBarrierWorklist);
669 if (!DrainWorklistWithBytesAndTimeDeadline(
670 mutator_marking_state_, marked_bytes_deadline, time_deadline,
671 mutator_marking_state_.write_barrier_worklist(),
672 [this](HeapObjectHeader* header) {
673 mutator_marking_state_.AccountMarkedBytes(*header);
674 DynamicallyTraceMarkedObject<AccessMode::kNonAtomic>(visitor(),
675 *header);
676 })) {
677 return false;
678 }
679 if (!DrainWorklistWithBytesAndTimeDeadline(
680 mutator_marking_state_, marked_bytes_deadline, time_deadline,
681 mutator_marking_state_.retrace_marked_objects_worklist(),
682 [this](HeapObjectHeader* header) {
683 // Retracing does not increment marked bytes as the object has
684 // already been processed before.
685 DynamicallyTraceMarkedObject<AccessMode::kNonAtomic>(visitor(),
686 *header);
687 })) {
688 return false;
689 }
690 }
691
692 saved_did_discover_new_ephemeron_pairs =
693 mutator_marking_state_.DidDiscoverNewEphemeronPairs();
694 {
695 StatsCollector::DisabledScope inner_stats_scope(
696 heap().stats_collector(), StatsCollector::kMarkProcessEphemerons);
697 if (!DrainWorklistWithBytesAndTimeDeadline(
698 mutator_marking_state_, marked_bytes_deadline, time_deadline,
699 mutator_marking_state_.ephemeron_pairs_for_processing_worklist(),
700 [this](const MarkingWorklists::EphemeronPairItem& item) {
701 mutator_marking_state_.ProcessEphemeron(
702 item.key, item.value, item.value_desc, visitor());
703 })) {
704 return false;
705 }
706 }
707 } while (!mutator_marking_state_.marking_worklist().IsLocalAndGlobalEmpty() ||
708 saved_did_discover_new_ephemeron_pairs);
709 return true;
710}

712void MarkerBase::MarkNotFullyConstructedObjects() {
713 // Parallel marking may still be running which is why atomic extraction is
714 // required.
715 std::unordered_set<HeapObjectHeader*> objects =
716 mutator_marking_state_.not_fully_constructed_worklist()
717 .Extract<AccessMode::kAtomic>();
718 if (objects.empty()) {
719 return;
720 }
721 StatsCollector::DisabledScope stats_scope(
722 heap().stats_collector(),
723 StatsCollector::kMarkVisitNotFullyConstructedObjects);
724 for (HeapObjectHeader* object : objects) {
725 DCHECK(object);
726 // TraceConservativelyIfNeeded delegates to either in-construction or
727 // fully constructed handling. Both handlers have their own marked bytes
728 // accounting and markbit handling (bailout).
729 conservative_visitor().TraceConservativelyIfNeeded(*object);
730 }
