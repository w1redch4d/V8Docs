87inline Address StackFrame::maybe_unauthenticated_pc() const {
88 if (!InFastCCall() && !is_profiler_entry_frame() && !is_stack_exit_frame()) {
89 // Here the pc_address() is on the stack and properly authenticated.
90 return pc();
91 } else {
92 // For fast C calls pc_address() points into IsolateData and the pc in there
93 // is unauthenticated. For the profiler, the pc_address of the first visited
94 // frame is also not written by a call instruction.
95 // For wasm stacks, the exit frame's pc is stored in the jump buffer
96 // unsigned.
97 return unauthenticated_pc(pc_address());
98 }
99}

425// static
426inline bool StackFrameIteratorForProfiler::IsValidFrameType(
427 StackFrame::Type type) {
428#if V8_ENABLE_WEBASSEMBLY
429 DCHECK_NE(type, StackFrame::C_WASM_ENTRY);
430#endif // V8_ENABLE_WEBASSEMBLY
431 return StackFrame::IsJavaScript(type) || type == StackFrame::EXIT ||
432 type == StackFrame::BUILTIN_EXIT ||
433 type == StackFrame::API_ACCESSOR_EXIT ||
434 type == StackFrame::API_CALLBACK_EXIT ||
435#if V8_ENABLE_WEBASSEMBLY
436 type == StackFrame::WASM || type == StackFrame::WASM_TO_JS ||
437 type == StackFrame::JS_TO_WASM ||
438 type == StackFrame::WASM_SEGMENT_START ||
439#if V8_ENABLE_DRUMBRAKE
440 type == StackFrame::WASM_INTERPRETER_ENTRY ||
441#endif // V8_ENABLE_DRUMBRAKE
442#endif // V8_ENABLE_WEBASSEMBLY
443 false;
444}
