90void InitializeProcess(PageAllocator* page_allocator,
91 size_t desired_heap_size) {
92#if defined(V8_USE_ADDRESS_SANITIZER) && defined(V8_HOST_ARCH_64_BIT)
93 // Retrieve asan's internal shadow memory granularity and check that Oilpan's
94 // object alignment/sizes are multiple of this granularity. This is needed to
95 // perform poisoness checks.
96 size_t shadow_scale;
97 __asan_get_shadow_mapping(&shadow_scale, nullptr);
98 DCHECK(shadow_scale);
99 const size_t poisoning_granularity = 1 << shadow_scale;
100 CHECK_EQ(0u, internal::kAllocationGranularity % poisoning_granularity);
101#endif
102
103 auto& allocator = internal::CreateAllocatorIfNeeded(page_allocator);
104
105 CHECK(!internal::g_page_allocator);
106 internal::GlobalGCInfoTable::Initialize(allocator);
107#if defined(CPPGC_CAGED_HEAP)
108 internal::CagedHeap::InitializeIfNeeded(allocator, desired_heap_size);
109#endif // defined(CPPGC_CAGED_HEAP)
110 internal::g_page_allocator = &allocator;
111}
