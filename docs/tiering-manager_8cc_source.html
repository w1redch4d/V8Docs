34#define OPTIMIZATION_REASON_LIST(V) \
35 V(DoNotOptimize, "do not optimize") \
36 V(HotAndStable, "hot and stable")
37
38enum class OptimizationReason : uint8_t {
39#define OPTIMIZATION_REASON_CONSTANTS(Constant, message) k##Constant,
40 OPTIMIZATION_REASON_LIST(OPTIMIZATION_REASON_CONSTANTS)
41#undef OPTIMIZATION_REASON_CONSTANTS
42};
43
44char const* OptimizationReasonToString(OptimizationReason reason) {
45 static char const* reasons[] = {
46#define OPTIMIZATION_REASON_TEXTS(Constant, message) message,
47 OPTIMIZATION_REASON_LIST(OPTIMIZATION_REASON_TEXTS)
48#undef OPTIMIZATION_REASON_TEXTS
49 };
50 size_t const index = static_cast<size_t>(reason);
51 DCHECK_LT(index, arraysize(reasons));
52 return reasons[index];
53}
54
55#undef OPTIMIZATION_REASON_LIST
56
57std::ostream& operator<<(std::ostream& os, OptimizationReason reason) {
58 return os << OptimizationReasonToString(reason);
59}

61class OptimizationDecision {
62 public:
63 static constexpr OptimizationDecision Maglev() {
64 // TODO(v8:7700): Consider using another reason here.
65 return {OptimizationReason::kHotAndStable, CodeKind::MAGLEV,
66 ConcurrencyMode::kConcurrent};
67 }
68 static constexpr OptimizationDecision TurbofanHotAndStable() {
69 return {OptimizationReason::kHotAndStable, CodeKind::TURBOFAN_JS,
70 ConcurrencyMode::kConcurrent};
71 }
72 static constexpr OptimizationDecision DoNotOptimize() {
73 return {OptimizationReason::kDoNotOptimize,
74 // These values don't matter but we have to pass something.
75 CodeKind::TURBOFAN_JS, ConcurrencyMode::kConcurrent};
76 }
77
78 constexpr bool should_optimize() const {
79 return optimization_reason != OptimizationReason::kDoNotOptimize;
80 }
81
82 OptimizationReason optimization_reason;
83 CodeKind code_kind;
84 ConcurrencyMode concurrency_mode;
85
86 private:
87 OptimizationDecision() = default;
88 constexpr OptimizationDecision(OptimizationReason optimization_reason,
89 CodeKind code_kind,
90 ConcurrencyMode concurrency_mode)
91 : optimization_reason(optimization_reason),
92 code_kind(code_kind),
93 concurrency_mode(concurrency_mode) {}
94};
95// Since we pass by value:
96static_assert(sizeof(OptimizationDecision) <= kInt32Size);

156// Returns true when |function| should be enqueued for sparkplug compilation for
157// the first time.
158bool FirstTimeTierUpToSparkplug(Isolate* isolate, Tagged<JSFunction> function) {
159 return !function->has_feedback_vector() ||
160 // We request sparkplug even in the presence of a fbv, if we are
161 // running ignition and haven't enqueued the function for sparkplug
162 // batch compilation yet. This ensures we tier-up to sparkplug when the
163 // feedback vector is allocated eagerly (e.g. for logging function
164 // events; see JSFunction::InitializeFeedbackCell()).
165 (function->ActiveTierIsIgnition(isolate) &&
166 CanCompileWithBaseline(isolate, function->shared()) &&
167 function->shared()->cached_tiering_decision() ==
168 CachedTieringDecision::kPending);
169}

171bool TiersUpToMaglev(CodeKind code_kind) {
172 return V8_LIKELY(maglev::IsMaglevEnabled()) &&
173 CodeKindIsUnoptimizedJSFunction(code_kind);
174}

176bool TiersUpToMaglev(std::optional<CodeKind> code_kind) {
177 return code_kind.has_value() && TiersUpToMaglev(code_kind.value());
178}

180int InterruptBudgetFor(Isolate* isolate, std::optional<CodeKind> code_kind,
181 Tagged<JSFunction> function,
182 CachedTieringDecision cached_tiering_decision,
183 int bytecode_length) {
184 const std::optional<CodeKind> existing_request =
185 function->GetRequestedOptimizationIfAny(isolate);
186 if (existing_request == CodeKind::TURBOFAN_JS ||
187 (code_kind.has_value() && code_kind.value() == CodeKind::TURBOFAN_JS)) {
188 return v8_flags.invocation_count_for_osr * bytecode_length;
189 }
190 if (maglev::IsMaglevOsrEnabled() && existing_request == CodeKind::MAGLEV) {
191 return v8_flags.invocation_count_for_maglev_osr * bytecode_length;
192 }
193 if (TiersUpToMaglev(code_kind) &&
194 !function->IsTieringRequestedOrInProgress(isolate)) {
195 if (v8_flags.profile_guided_optimization) {
196 switch (cached_tiering_decision) {
197 case CachedTieringDecision::kDelayMaglev:
198 return (std::max(v8_flags.invocation_count_for_maglev,
199 v8_flags.minimum_invocations_after_ic_update) +
200 v8_flags.invocation_count_for_maglev_with_delay) *
201 bytecode_length;
202 case CachedTieringDecision::kEarlyMaglev:
203 case CachedTieringDecision::kEarlyTurbofan:
204 return v8_flags.invocation_count_for_early_optimization *
205 bytecode_length;
206 case CachedTieringDecision::kPending:
207 case CachedTieringDecision::kEarlySparkplug:
208 case CachedTieringDecision::kNormal:
209 return v8_flags.invocation_count_for_maglev * bytecode_length;
210 }
211 }
212 return v8_flags.invocation_count_for_maglev * bytecode_length;
213 }
214 return v8_flags.invocation_count_for_turbofan * bytecode_length;
215}

219// static
220int TieringManager::InterruptBudgetFor(
221 Isolate* isolate, Tagged<JSFunction> function,
222 std::optional<CodeKind> override_active_tier) {
223 DCHECK(function->shared()->is_compiled());
224 const int bytecode_length =
225 function->shared()->GetBytecodeArray(isolate)->length();
226
227 if (FirstTimeTierUpToSparkplug(isolate, function)) {
228 return bytecode_length * v8_flags.invocation_count_for_feedback_allocation;
229 }
230
231 DCHECK(function->has_feedback_vector());
232 if (bytecode_length > v8_flags.max_optimized_bytecode_size) {
233 // Decrease times of interrupt budget underflow, the reason of not setting
234 // to INT_MAX is the interrupt budget may overflow when doing add
235 // operation for forward jump.
236 return INT_MAX / 2;
237 }
238 return ::i::InterruptBudgetFor(
239 isolate,
240 override_active_tier ? override_active_tier
241 : function->GetActiveTier(isolate),
242 function, function->shared()->cached_tiering_decision(), bytecode_length);
243}

247void TrySetOsrUrgency(Isolate* isolate, Tagged<JSFunction> function,
248 int osr_urgency) {
249 Tagged<SharedFunctionInfo> shared = function->shared();
250 if (V8_UNLIKELY(!v8_flags.use_osr)) return;
251 if (V8_UNLIKELY(shared->optimization_disabled())) return;
252
253 // We've passed all checks - bump the OSR urgency.
254
255 Tagged<FeedbackVector> fv = function->feedback_vector();
256 if (V8_UNLIKELY(v8_flags.trace_osr)) {
257 CodeTracer::Scope scope(isolate->GetCodeTracer());
258 PrintF(scope.file(),
259 "[OSR - setting osr urgency. function: %s, old urgency: %d, new "
260 "urgency: %d]\n",
261 function->DebugNameCStr().get(), fv->osr_urgency(), osr_urgency);
262 }
263
264 DCHECK_GE(osr_urgency, fv->osr_urgency()); // Never lower urgency here.
265 fv->set_osr_urgency(osr_urgency);
266}
267
268void TryIncrementOsrUrgency(Isolate* isolate, Tagged<JSFunction> function) {
269 int old_urgency = function->feedback_vector()->osr_urgency();
270 int new_urgency = std::min(old_urgency + 1, FeedbackVector::kMaxOsrUrgency);
271 TrySetOsrUrgency(isolate, function, new_urgency);
272}

274void TryRequestOsrAtNextOpportunity(Isolate* isolate,
275 Tagged<JSFunction> function) {
276 TrySetOsrUrgency(isolate, function, FeedbackVector::kMaxOsrUrgency);
277}

286void TieringManager::MaybeOptimizeFrame(Tagged<JSFunction> function,
287 CodeKind current_code_kind) {
288 const bool tiering_in_progress = function->tiering_in_progress();
289 const bool osr_in_progress =
290 function->feedback_vector()->osr_tiering_in_progress();
291 // Attenzione! Update this constant in case the condition below changes.
292 static_assert(kTieringStateInProgressBlocksTierup);
293 if (V8_UNLIKELY(tiering_in_progress) || V8_UNLIKELY(osr_in_progress)) {
294 if (v8_flags.concurrent_recompilation_front_running &&
295 ((tiering_in_progress && function->ActiveTierIsMaglev(isolate_)) ||
296 (osr_in_progress &&
297 function->feedback_vector()->maybe_has_optimized_osr_code()))) {
298 // TODO(olivf): In the case of Maglev we tried a queue with two
299 // priorities, but it seems not actually beneficial. More
300 // investigation is needed.
301 isolate_->IncreaseConcurrentOptimizationPriority(CodeKind::TURBOFAN_JS,
302 function->shared());
303 }
304 // Note: This effectively disables further tiering actions (e.g. OSR, or
305 // tiering up into Maglev) for the function while it is being compiled.
306 TraceInOptimizationQueue(function, current_code_kind);
307 return;
308 }
309
310 if (V8_UNLIKELY(v8_flags.testing_d8_test_runner) &&
311 ManualOptimizationTable::IsMarkedForManualOptimization(isolate_,
312 function)) {
313 TraceHeuristicOptimizationDisallowed(function);
314 return;
315 }
316
317 // TODO(v8:7700): Consider splitting this up for Maglev/Turbofan.
318 if (V8_UNLIKELY(function->shared()->optimization_disabled())) return;
319
320 if (V8_UNLIKELY(v8_flags.always_osr)) {
321 TryRequestOsrAtNextOpportunity(isolate_, function);
322 // Continue below and do a normal optimized compile as well.
323 }
324
325 const bool maglev_osr = maglev::IsMaglevOsrEnabled();
326 const CodeKinds available_kinds = function->GetAvailableCodeKinds(isolate_);
327 const bool waiting_for_tierup =
328 (current_code_kind < CodeKind::TURBOFAN_JS &&
329 (available_kinds & CodeKindFlag::TURBOFAN_JS)) ||
330 (maglev_osr && current_code_kind < CodeKind::MAGLEV &&
331 (available_kinds & CodeKindFlag::MAGLEV));
332 // Baseline OSR uses a separate mechanism and must not be considered here,
333 // therefore we limit to kOptimizedJSFunctionCodeKindsMask.
334 if (function->IsOptimizationRequested(isolate_) || waiting_for_tierup) {
335 if (V8_UNLIKELY(maglev_osr && current_code_kind == CodeKind::MAGLEV &&
336 (!v8_flags.osr_from_maglev ||
337 isolate_->EfficiencyModeEnabledForTiering() ||
338 isolate_->BatterySaverModeEnabled()))) {
339 return;
340 }
341
342 // OSR kicks in only once we've previously decided to tier up, but we are
343 // still in a lower-tier frame (this implies a long-running loop).
344 TryIncrementOsrUrgency(isolate_, function);
345
346 // Return unconditionally and don't run through the optimization decision
347 // again; we've already decided to tier up previously.
348 return;
349 }
350
351 const std::optional<CodeKind> existing_request =
352 function->GetRequestedOptimizationIfAny(isolate_);
353 DCHECK(existing_request != CodeKind::TURBOFAN_JS);
354 DCHECK(!function->HasAvailableCodeKind(isolate_, CodeKind::TURBOFAN_JS));
355 OptimizationDecision d =
356 ShouldOptimize(function->feedback_vector(), current_code_kind);
357 // We might be stuck in a baseline frame that wants to tier up to Maglev, but
358 // is in a loop, and can't OSR, because Maglev doesn't have OSR. Allow it to
359 // skip over Maglev by re-checking ShouldOptimize as if we were in Maglev.
360 if (V8_UNLIKELY(!isolate_->EfficiencyModeEnabledForTiering() && !maglev_osr &&
361 d.should_optimize() && d.code_kind == CodeKind::MAGLEV)) {
362 bool is_marked_for_maglev_optimization =
363 existing_request == CodeKind::MAGLEV ||
364 (available_kinds & CodeKindFlag::MAGLEV);
365 if (is_marked_for_maglev_optimization) {
366 d = ShouldOptimize(function->feedback_vector(), CodeKind::MAGLEV);
367 }
368 }
369
370 if (V8_UNLIKELY(isolate_->EfficiencyModeEnabledForTiering() &&
371 d.code_kind != CodeKind::TURBOFAN_JS)) {
372 d.concurrency_mode = ConcurrencyMode::kSynchronous;
373 }
374
375 if (d.should_optimize()) Optimize(function, d);
376}

378OptimizationDecision TieringManager::ShouldOptimize(
379 Tagged<FeedbackVector> feedback_vector, CodeKind current_code_kind) {
380 Tagged<SharedFunctionInfo> shared = feedback_vector->shared_function_info();
381 if (current_code_kind == CodeKind::TURBOFAN_JS) {
382 return OptimizationDecision::DoNotOptimize();
383 }
384
385 if (TiersUpToMaglev(current_code_kind) &&
386 shared->PassesFilter(v8_flags.maglev_filter) &&
387 !shared->maglev_compilation_failed()) {
388 if (v8_flags.profile_guided_optimization &&
389 shared->cached_tiering_decision() ==
390 CachedTieringDecision::kEarlyTurbofan) {
391 return OptimizationDecision::TurbofanHotAndStable();
392 }
393 return OptimizationDecision::Maglev();
394 }
395
396 if (V8_UNLIKELY(!v8_flags.turbofan ||
397 !shared->PassesFilter(v8_flags.turbo_filter) ||
398 (v8_flags.efficiency_mode_disable_turbofan &&
399 isolate_->EfficiencyModeEnabledForTiering()) ||
400 isolate_->BatterySaverModeEnabled())) {
401 return OptimizationDecision::DoNotOptimize();
402 }
403
404 if (isolate_->EfficiencyModeEnabledForTiering() &&
405 v8_flags.efficiency_mode_delay_turbofan &&
406 feedback_vector->invocation_count() <
407 v8_flags.efficiency_mode_delay_turbofan) {
408 return OptimizationDecision::DoNotOptimize();
409 }
410
411 Tagged<BytecodeArray> bytecode = shared->GetBytecodeArray(isolate_);
412 if (bytecode->length() > v8_flags.max_optimized_bytecode_size) {
413 return OptimizationDecision::DoNotOptimize();
414 }
415
416 return OptimizationDecision::TurbofanHotAndStable();
417}

421bool ShouldResetInterruptBudgetByICChange(
422 CachedTieringDecision cached_tiering_decision) {
423 switch (cached_tiering_decision) {
424 case CachedTieringDecision::kEarlyMaglev:
425 case CachedTieringDecision::kEarlyTurbofan:
426 return false;
427 case CachedTieringDecision::kPending:
428 case CachedTieringDecision::kEarlySparkplug:

431 return true;
432 }

437void TieringManager::NotifyICChanged(Tagged<FeedbackVector> vector) {
438 CodeKind code_kind = vector->shared_function_info()->HasBaselineCode()
439 ? CodeKind::BASELINE
440 : CodeKind::INTERPRETED_FUNCTION;
441
442#ifndef V8_ENABLE_LEAPTIERING
443 if (vector->has_optimized_code()) {
444 code_kind = vector->optimized_code(isolate_)->kind();
445 }
446#endif // !V8_ENABLE_LEAPTIERING
447
448 if (code_kind == CodeKind::INTERPRETED_FUNCTION &&
449 CanCompileWithBaseline(isolate_, vector->shared_function_info()) &&
450 vector->shared_function_info()->cached_tiering_decision() ==
451 CachedTieringDecision::kPending) {
452 // Don't delay tier-up if we haven't tiered up to baseline yet, but will --
453 // baseline code is feedback independent.
454 return;
455 }
456
457 OptimizationDecision decision = ShouldOptimize(vector, code_kind);
458 if (decision.should_optimize()) {
459 Tagged<SharedFunctionInfo> shared = vector->shared_function_info();
460 int bytecode_length = shared->GetBytecodeArray(isolate_)->length();
461 Tagged<FeedbackCell> cell = vector->parent_feedback_cell();
462 int invocations = v8_flags.minimum_invocations_after_ic_update;
463 int bytecodes = std::min(bytecode_length, (kMaxInt >> 1) / invocations);
464 int new_budget = invocations * bytecodes;
465 int current_budget = cell->interrupt_budget();
466 if (v8_flags.profile_guided_optimization &&
467 shared->cached_tiering_decision() <=
468 CachedTieringDecision::kEarlySparkplug) {
469 DCHECK_LT(v8_flags.invocation_count_for_early_optimization,
470 FeedbackVector::kInvocationCountBeforeStableDeoptSentinel);
471 if (vector->invocation_count_before_stable() <
472 v8_flags.invocation_count_for_early_optimization) {
473 // Record how many invocation count were consumed before the last IC
474 // change.
475 int new_invocation_count_before_stable;
476 if (vector->interrupt_budget_reset_by_ic_change()) {
477 // Initial interrupt budget is
478 // v8_flags.minimum_invocations_after_ic_update * bytecodes
479 int new_consumed_budget = new_budget - current_budget;
480 new_invocation_count_before_stable =
481 vector->invocation_count_before_stable(kRelaxedLoad) +
482 std::ceil(static_cast<float>(new_consumed_budget) / bytecodes);
483 } else {
484 // Initial interrupt budget is
485 // v8_flags.invocation_count_for_{maglev|turbofan} * bytecodes
486 int total_consumed_budget =
487 (maglev::IsMaglevEnabled()
488 ? v8_flags.invocation_count_for_maglev
489 : v8_flags.invocation_count_for_turbofan) *
490 bytecodes -
491 current_budget;
492 new_invocation_count_before_stable =
493 std::ceil(static_cast<float>(total_consumed_budget) / bytecodes);
494 }
495 if (new_invocation_count_before_stable >=
496 v8_flags.invocation_count_for_early_optimization) {
497 vector->set_invocation_count_before_stable(
498 v8_flags.invocation_count_for_early_optimization, kRelaxedStore);
499 shared->set_cached_tiering_decision(CachedTieringDecision::kNormal);
500 } else {
501 vector->set_invocation_count_before_stable(
502 new_invocation_count_before_stable, kRelaxedStore);
503 }
504 } else {
505 shared->set_cached_tiering_decision(CachedTieringDecision::kNormal);
506 }
507 }
508 if (!v8_flags.profile_guided_optimization ||
509 ShouldResetInterruptBudgetByICChange(
510 shared->cached_tiering_decision())) {
511 if (new_budget > current_budget) {
512 if (v8_flags.trace_opt_verbose) {
513 PrintF("[delaying optimization of %s, IC changed]\n",
514 shared->DebugNameCStr().get());
515 }
516 vector->set_interrupt_budget_reset_by_ic_change(true);
517 cell->set_interrupt_budget(new_budget);
518 }
519 }
520 }

528void TieringManager::OnInterruptTick(DirectHandle<JSFunction> function,
529 CodeKind code_kind) {
530 IsCompiledScope is_compiled_scope(
531 function->shared()->is_compiled_scope(isolate_));
532
533 // Remember whether the function had a vector at this point. This is
534 // relevant later since the configuration 'Ignition without a vector' can be
535 // considered a tier on its own. We begin tiering up to tiers higher than
536 // Sparkplug only when reaching this point *with* a feedback vector.
537 const bool had_feedback_vector = function->has_feedback_vector();
538 const bool first_time_tiered_up_to_sparkplug =
539 FirstTimeTierUpToSparkplug(isolate_, *function);
540 // We don't want to trigger GC in the middle of OSR, so do not build a
541 // baseline code for such case.
542 const bool maybe_had_optimized_osr_code =
543 had_feedback_vector &&
544 function->feedback_vector()->maybe_has_optimized_osr_code();
545 const bool compile_sparkplug =
546 CanCompileWithBaseline(isolate_, function->shared()) &&
547 function->ActiveTierIsIgnition(isolate_) && !maybe_had_optimized_osr_code;
548
549 // Ensure that the feedback vector has been allocated.
550 if (!had_feedback_vector) {
551 if (compile_sparkplug && function->shared()->cached_tiering_decision() ==
552 CachedTieringDecision::kPending) {
553 // Mark the function as compiled with sparkplug before the feedback
554 // vector is created to initialize the interrupt budget for the next
555 // tier.
556 function->shared()->set_cached_tiering_decision(
557 CachedTieringDecision::kEarlySparkplug);
558 }
559 JSFunction::CreateAndAttachFeedbackVector(isolate_, function,
560 &is_compiled_scope);
561 DCHECK(is_compiled_scope.is_compiled());
562 // Also initialize the invocation count here. This is only really needed
563 // for OSR. When we OSR functions with lazy feedback allocation we want to
564 // have a non zero invocation count so we can inline functions.
565 function->feedback_vector()->set_invocation_count(1, kRelaxedStore);
566 }
567
568 DCHECK(function->has_feedback_vector());
569 DCHECK(function->shared()->is_compiled());
570 DCHECK(function->shared()->HasBytecodeArray());
571
572 // TODO(jgruber): Consider integrating this into a linear tiering system
573 // controlled by TieringState in which the order is always
574 // Ignition-Sparkplug-Turbofan, and only a single tierup is requested at
575 // once.
576 // It's unclear whether this is possible and/or makes sense - for example,
577 // batching compilation can introduce arbitrary latency between the SP
578 // compile request and fulfillment, which doesn't work with strictly linear
579 // tiering.
580 if (compile_sparkplug) {
581#ifdef V8_ENABLE_SPARKPLUG
582 if (v8_flags.baseline_batch_compilation) {
583 isolate_->baseline_batch_compiler()->EnqueueFunction(function);
584 } else {
585 IsCompiledScope is_compiled_scope(
586 function->shared()->is_compiled_scope(isolate_));
587 Compiler::CompileBaseline(isolate_, function, Compiler::CLEAR_EXCEPTION,
588 &is_compiled_scope);
589 }
590#else
591 UNREACHABLE();
592#endif // V8_ENABLE_SPARKPLUG
593 }
594
595 // We only tier up beyond sparkplug if we already had a feedback vector.
596 if (first_time_tiered_up_to_sparkplug) {
597 // If we didn't have a feedback vector, the interrupt budget has already
598 // been set by JSFunction::CreateAndAttachFeedbackVector, so no need to
599 // set it again.
600 if (had_feedback_vector) {
601 if (function->shared()->cached_tiering_decision() ==
602 CachedTieringDecision::kPending) {
603 function->shared()->set_cached_tiering_decision(
604 CachedTieringDecision::kEarlySparkplug);
605 }
606 function->SetInterruptBudget(isolate_, BudgetModification::kRaise);
607 }
608 return;
609 }
610
611 // Don't tier up if Turbofan is disabled.
612 // TODO(jgruber): Update this for a multi-tier world.
613 if (V8_UNLIKELY(!isolate_->use_optimizer())) {
614 function->SetInterruptBudget(isolate_, BudgetModification::kRaise);
615 return;
616 }
617
618 // --- We've decided to proceed for now. ---
619
620 DisallowGarbageCollection no_gc;
621 OnInterruptTickScope scope;
622 Tagged<JSFunction> function_obj = *function;
623
624 MaybeOptimizeFrame(function_obj, code_kind);
625
626 // Make sure to set the interrupt budget after maybe starting an optimization,
627 // so that the interrupt budget size takes into account tiering state.
628 DCHECK(had_feedback_vector);
629 function->SetInterruptBudget(isolate_, BudgetModification::kRaise);
630}
