128 bool AddListener(LogEventListener* listener) {
129 base::RecursiveMutexGuard guard(&mutex_);
130 auto position = std::find(listeners_.begin(), listeners_.end(), listener);
131 if (position != listeners_.end()) return false;
132 // Add the listener to the end and update the element
133 listeners_.push_back(listener);
134 return true;
135 }
136
137 bool RemoveListener(LogEventListener* listener) {
138 base::RecursiveMutexGuard guard(&mutex_);
139 auto position = std::find(listeners_.begin(), listeners_.end(), listener);
140 if (position == listeners_.end()) return false;
141 listeners_.erase(position);
142 return true;
143 }
144
145 bool is_listening_to_code_events() {
146 base::RecursiveMutexGuard guard(&mutex_);
147 for (auto listener : listeners_) {
148 if (listener->is_listening_to_code_events()) return true;
149 }
150 return false;
151 }
