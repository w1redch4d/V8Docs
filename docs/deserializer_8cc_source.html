50// A SlotAccessor for a slot in a HeapObject, which abstracts the slot
51// operations done by the deserializer in a way which is GC-safe. In particular,
52// rather than an absolute slot address, this accessor holds a Handle to the
53// HeapObject, which is updated if the HeapObject moves.
54class SlotAccessorForHeapObject {
55 public:
56 static SlotAccessorForHeapObject ForSlotIndex(Handle<HeapObject> object,
57 int index) {
58 return SlotAccessorForHeapObject(object, index * kTaggedSize);
59 }
60 static SlotAccessorForHeapObject ForSlotOffset(Handle<HeapObject> object,
61 int offset) {
62 return SlotAccessorForHeapObject(object, offset);
63 }
64
65 MaybeObjectSlot slot() const { return object_->RawMaybeWeakField(offset_); }
66 ExternalPointerSlot external_pointer_slot(ExternalPointerTag tag) const {
67 return object_->RawExternalPointerField(offset_, tag);
68 }
69 Handle<HeapObject> object() const { return object_; }
70 int offset() const { return offset_; }
71
72 // Writes the given value to this slot, with an offset (e.g. for repeat
73 // writes). Returns the number of slots written (which is one).
74 int Write(Tagged<MaybeObject> value, int slot_offset, WriteBarrierMode mode) {
75 MaybeObjectSlot current_slot = slot() + slot_offset;
76 current_slot.Relaxed_Store(value);
77#ifdef V8_STATIC_ROOTS_BOOL
78 if (mode != SKIP_WRITE_BARRIER && FastInReadOnlySpaceOrSmallSmi(value)) {
79 // TODO(jgruber): Remove this once WriteBarrier::ForValue() contains the
80 // same check.
81 mode = SKIP_WRITE_BARRIER;
82 }
83#endif // V8_STATIC_ROOTS_BOOL
84 WriteBarrier::ForValue(*object_, current_slot, value, mode);
85 return 1;
86 }
87 int Write(Tagged<HeapObject> value, HeapObjectReferenceType ref_type,
88 int slot_offset, WriteBarrierMode mode) {
89 return Write(Tagged<HeapObjectReference>(value, ref_type), slot_offset,
90 mode);
91 }
92 int Write(DirectHandle<HeapObject> value, HeapObjectReferenceType ref_type,
93 int slot_offset, WriteBarrierMode mode) {
94 return Write(*value, ref_type, slot_offset, mode);
95 }
96
97 int WriteIndirectPointerTo(Tagged<HeapObject> value, WriteBarrierMode mode) {
98 // Only ExposedTrustedObjects can be referenced via indirect pointers, so
99 // we must have one of these objects here. See the comments in
100 // trusted-object.h for more details.
101 DCHECK(IsExposedTrustedObject(value));
102 Tagged<ExposedTrustedObject> object = Cast<ExposedTrustedObject>(value);
103
104 InstanceType instance_type = value->map()->instance_type();
105 IndirectPointerTag tag = IndirectPointerTagFromInstanceType(instance_type);
106 IndirectPointerSlot dest = object_->RawIndirectPointerField(offset_, tag);
107 dest.store(object);
108
109 WriteBarrier::ForIndirectPointer(*object_, dest, value, mode);
110 return 1;
111 }
112
113 int WriteProtectedPointerTo(Tagged<TrustedObject> value,
114 WriteBarrierMode mode) {
115 DCHECK(IsTrustedObject(*object_));
116 Tagged<TrustedObject> host = Cast<TrustedObject>(*object_);
117 ProtectedPointerSlot dest = host->RawProtectedPointerField(offset_);
118 dest.store(value);
119 WriteBarrier::ForProtectedPointer(host, dest, value, mode);
120 return 1;
121 }
122
123 private:
124 SlotAccessorForHeapObject(Handle<HeapObject> object, int offset)
125 : object_(object), offset_(offset) {}
126
127 const Handle<HeapObject> object_;
128 const int offset_;
129};

245template <typename IsolateT>
246int Deserializer<IsolateT>::WriteExternalPointer(Tagged<HeapObject> host,
247 ExternalPointerSlot dest,
248 Address value,
249 ExternalPointerTag tag) {
250 DCHECK(!next_reference_is_weak_ && !next_reference_is_indirect_pointer_ &&
251 !next_reference_is_protected_pointer);
252
253 #ifdef V8_ENABLE_SANDBOX
254 ExternalPointerTable::ManagedResource* managed_resource = nullptr;
255 ExternalPointerTable* owning_table = nullptr;
256 ExternalPointerHandle original_handle = kNullExternalPointerHandle;
257 if (IsManagedExternalPointerType(tag)) {
258 // This can currently only happen during snapshot stress mode as we cannot
259 // normally serialized managed resources. In snapshot stress mode, the new
260 // isolate will be destroyed and the old isolate (really, the old isolate's
261 // external pointer table) therefore effectively retains ownership of the
262 // resource. As such, we need to save and restore the relevant fields of
263 // the external resource. Once the external pointer table itself destroys
264 // the managed resource when freeing the corresponding table entry, this
265 // workaround can be removed again.
266 DCHECK(v8_flags.stress_snapshot);
267 managed_resource =
268 reinterpret_cast<ExternalPointerTable::ManagedResource*>(value);
269 owning_table = managed_resource->owning_table_;
270 original_handle = managed_resource->ept_entry_;
271 managed_resource->owning_table_ = nullptr;
272 managed_resource->ept_entry_ = kNullExternalPointerHandle;
273 }
274#endif // V8_ENABLE_SANDBOX
275
276 dest.init(main_thread_isolate(), host, value, tag);
277
278#ifdef V8_ENABLE_SANDBOX
279 if (managed_resource) {
280 managed_resource->owning_table_ = owning_table;
281 managed_resource->ept_entry_ = original_handle;
282 }
283#endif // V8_ENABLE_SANDBOX
284
285 // ExternalPointers can only be written into HeapObject fields, therefore they
286 // cover (kExternalPointerSlotSize / kTaggedSize) slots.
287 return (kExternalPointerSlotSize / kTaggedSize);
288}

501// Should be called only on the main thread (not thread safe).
502template <>
503void Deserializer<Isolate>::PostProcessNewJSReceiver(
504 Tagged<Map> map, DirectHandle<JSReceiver> obj, InstanceType instance_type,
505 SnapshotSpace space) {
506 DCHECK_EQ(map->instance_type(), instance_type);
507
508 if (InstanceTypeChecker::IsJSDataView(instance_type) ||
509 InstanceTypeChecker::IsJSRabGsabDataView(instance_type)) {
510 auto data_view = Cast<JSDataViewOrRabGsabDataView>(*obj);
511 auto buffer = Cast<JSArrayBuffer>(data_view->buffer());
512 if (buffer->was_detached()) {
513 // Directly set the data pointer to point to the EmptyBackingStoreBuffer.
514 // Otherwise, we might end up setting it to EmptyBackingStoreBuffer() +
515 // byte_offset() which would result in an invalid pointer.
516 data_view->set_data_pointer(main_thread_isolate(),
517 EmptyBackingStoreBuffer());
518 } else {
519 void* backing_store = buffer->backing_store();
520 data_view->set_data_pointer(
521 main_thread_isolate(),
522 reinterpret_cast<uint8_t*>(backing_store) + data_view->byte_offset());
523 }
524 } else if (InstanceTypeChecker::IsJSTypedArray(instance_type)) {
525 auto typed_array = Cast<JSTypedArray>(*obj);
526 // Note: ByteArray objects must not be deferred s.t. they are
527 // available here for is_on_heap(). See also: CanBeDeferred.
528 // Fixup typed array pointers.
529 if (typed_array->is_on_heap()) {
530 typed_array->AddExternalPointerCompensationForDeserialization(
531 main_thread_isolate());
532 } else {
533 // Serializer writes backing store ref as a DataPtr() value.
534 uint32_t store_index =
535 typed_array->GetExternalBackingStoreRefForDeserialization();
536 auto backing_store = backing_stores_[store_index];
537 void* start = backing_store ? backing_store->buffer_start() : nullptr;
538 if (!start) start = EmptyBackingStoreBuffer();
539 typed_array->SetOffHeapDataPtr(main_thread_isolate(), start,
540 typed_array->byte_offset());
541 }
542 } else if (InstanceTypeChecker::IsJSArrayBuffer(instance_type)) {
543 auto buffer = Cast<JSArrayBuffer>(*obj);
544 uint32_t store_index = buffer->GetBackingStoreRefForDeserialization();
545 buffer->init_extension();
546 if (store_index == kEmptyBackingStoreRefSentinel) {
547 buffer->set_backing_store(main_thread_isolate(),
548 EmptyBackingStoreBuffer());
549 } else {
550 auto bs = backing_store(store_index);
551 SharedFlag shared =
552 bs && bs->is_shared() ? SharedFlag::kShared : SharedFlag::kNotShared;
553 DCHECK_IMPLIES(bs,
554 buffer->is_resizable_by_js() == bs->is_resizable_by_js());
555 ResizableFlag resizable = bs && bs->is_resizable_by_js()
556 ? ResizableFlag::kResizable
557 : ResizableFlag::kNotResizable;
558 buffer->Setup(shared, resizable, bs, main_thread_isolate());
559 }
560 }

570template <typename IsolateT>
571void Deserializer<IsolateT>::PostProcessNewObject(DirectHandle<Map> map,
572 Handle<HeapObject> obj,
573 SnapshotSpace space) {
574 DisallowGarbageCollection no_gc;
575 Tagged<Map> raw_map = *map;
576 DCHECK_EQ(raw_map, obj->map(isolate_));
577 InstanceType instance_type = raw_map->instance_type();
578 Tagged<HeapObject> raw_obj = *obj;
579 DCHECK_IMPLIES(deserializing_user_code(), should_rehash());
580 if (should_rehash()) {
581 if (InstanceTypeChecker::IsString(instance_type)) {
582 // Uninitialize hash field as we need to recompute the hash.
583 Tagged<String> string = Cast<String>(raw_obj);
584 string->set_raw_hash_field(String::kEmptyHashField);
585 // Rehash strings before read-only space is sealed. Strings outside
586 // read-only space are rehashed lazily. (e.g. when rehashing dictionaries)
587 if (space == SnapshotSpace::kReadOnlyHeap) {
588 PushObjectToRehash(obj);
589 }
590 } else if (raw_obj->NeedsRehashing(instance_type)) {
591 PushObjectToRehash(obj);
592 }
593
594 if (deserializing_user_code()) {
595 if (InstanceTypeChecker::IsInternalizedString(instance_type)) {
596 // Canonicalize the internalized string. If it already exists in the
597 // string table, set the string to point to the existing one and patch
598 // the deserialized string handle to point to the existing one.
599 // TODO(leszeks): This handle patching is ugly, consider adding an
600 // explicit internalized string bytecode. Also, the new thin string
601 // should be dead, try immediately freeing it.
602 Handle<String> string = Cast<String>(obj);
603
604 StringTableInsertionKey key(
605 isolate(), string,
606 DeserializingUserCodeOption::kIsDeserializingUserCode);
607 Tagged<String> result =
608 *isolate()->string_table()->LookupKey(isolate(), &key);
609
610 if (result != raw_obj) {
611 Cast<String>(raw_obj)->MakeThin(isolate(), result);
612 // Mutate the given object handle so that the backreference entry is
613 // also updated.
614 obj.PatchValue(result);
615 }
616 return;
617 } else if (InstanceTypeChecker::IsScript(instance_type)) {
618 new_scripts_.push_back(Cast<Script>(obj));
619 } else if (InstanceTypeChecker::IsAllocationSite(instance_type)) {
620 // We should link new allocation sites, but we can't do this immediately
621 // because |AllocationSite::HasWeakNext()| internally accesses
622 // |Heap::roots_| that may not have been initialized yet. So defer this
623 // to |ObjectDeserializer::CommitPostProcessedObjects()|.
624 new_allocation_sites_.push_back(Cast<AllocationSite>(obj));
625 } else {
626 // We dont defer ByteArray because JSTypedArray needs the base_pointer
627 // ByteArray immediately if it's on heap.
628 DCHECK(CanBeDeferred(*obj, SlotType::kAnySlot) ||
629 InstanceTypeChecker::IsByteArray(instance_type));
630 }
631 }
632 }
633
634 if (InstanceTypeChecker::IsInstructionStream(instance_type)) {
635 // We flush all code pages after deserializing the startup snapshot.
636 // Hence we only remember each individual code object when deserializing
637 // user code.
638 if (deserializing_user_code()) {
639 new_code_objects_.push_back(Cast<InstructionStream>(obj));
640 }
641 } else if (InstanceTypeChecker::IsCode(instance_type)) {
642 Tagged<Code> code = Cast<Code>(raw_obj);
643 if (!code->has_instruction_stream()) {
644 code->SetInstructionStartForOffHeapBuiltin(
645 main_thread_isolate(), EmbeddedData::FromBlob(main_thread_isolate())
646 .InstructionStartOf(code->builtin_id()));
647 } else {
648 code->UpdateInstructionStart(main_thread_isolate(),
649 code->instruction_stream());
650 }
651 } else if (InstanceTypeChecker::IsSharedFunctionInfo(instance_type)) {
652 Tagged<SharedFunctionInfo> sfi = Cast<SharedFunctionInfo>(raw_obj);
653 // Reset the id to avoid collisions - it must be unique in this isolate.
654 sfi->set_unique_id(isolate()->GetAndIncNextUniqueSfiId());
655 } else if (InstanceTypeChecker::IsMap(instance_type)) {
656 if (v8_flags.log_maps) {
657 // Keep track of all seen Maps to log them later since they might be only
658 // partially initialized at this point.
659 new_maps_.push_back(Cast<Map>(obj));
660 }
661 } else if (InstanceTypeChecker::IsAccessorInfo(instance_type)) {
662#ifdef USE_SIMULATOR
663 accessor_infos_.push_back(Cast<AccessorInfo>(obj));
664#endif
665 } else if (InstanceTypeChecker::IsFunctionTemplateInfo(instance_type)) {
666#ifdef USE_SIMULATOR
667 function_template_infos_.push_back(Cast<FunctionTemplateInfo>(obj));
668#endif
669 } else if (InstanceTypeChecker::IsExternalString(instance_type)) {
670 PostProcessExternalString(Cast<ExternalString>(raw_obj),
671 main_thread_isolate());
672 } else if (InstanceTypeChecker::IsJSReceiver(instance_type)) {
673 // PostProcessNewJSReceiver may trigger GC.
674 no_gc.Release();
675 return PostProcessNewJSReceiver(raw_map, Cast<JSReceiver>(obj),
676 instance_type, space);
677 } else if (InstanceTypeChecker::IsDescriptorArray(instance_type)) {
678 DCHECK(InstanceTypeChecker::IsStrongDescriptorArray(instance_type));
679 auto descriptors = Cast<DescriptorArray>(obj);
680 new_descriptor_arrays_.Push(*descriptors);
681 } else if (InstanceTypeChecker::IsNativeContext(instance_type)) {
682 Cast<NativeContext>(raw_obj)->init_microtask_queue(main_thread_isolate(),
683 nullptr);
684 } else if (InstanceTypeChecker::IsScript(instance_type)) {
685 LogScriptEvents(Cast<Script>(*obj));
686 }

747template <typename IsolateT>
748Handle<HeapObject> Deserializer<IsolateT>::ReadObject(SnapshotSpace space) {
749 const int size_in_tagged = source_.GetUint30();
750 const int size_in_bytes = size_in_tagged * kTaggedSize;
751
752 // The map can't be a forward ref. If you want the map to be a forward ref,
753 // then you're probably serializing the meta-map, in which case you want to
754 // use the kNewContextlessMetaMap/kNewContextfulMetaMap bytecode.
755 DCHECK_NE(source()->Peek(), kRegisterPendingForwardRef);
756 DirectHandle<Map> map = Cast<Map>(ReadObject());
757
758 AllocationType allocation = SpaceToAllocation(space);
759
760 // When sharing a string table, all in-place internalizable and internalized
761 // strings internalized strings are allocated in the shared heap.
762 //
763 // TODO(12007): When shipping, add a new SharedOld SnapshotSpace.
764 if (v8_flags.shared_string_table) {
765 InstanceType instance_type = map->instance_type();
766 if (InstanceTypeChecker::IsInternalizedString(instance_type) ||
767 String::IsInPlaceInternalizable(instance_type)) {
768 allocation = isolate()
769 ->factory()
770 ->RefineAllocationTypeForInPlaceInternalizableString(
771 allocation, *map);
772 }
773 }
774
775 // Filling an object's fields can cause GCs and heap walks, so this object has
776 // to be in a 'sufficiently initialised' state by the time the next allocation
777 // can happen. For this to be the case, the object is carefully deserialized
778 // as follows:
779 // * The space for the object is allocated.
780 // * The map is set on the object so that the GC knows what type the object
781 // has.
782 // * The rest of the object is filled with a fixed Smi value
783 // - This is a Smi so that tagged fields become initialized to a valid
784 // tagged value.
785 // - It's a fixed value, "Smi::uninitialized_deserialization_value()", so
786 // that we can DCHECK for it when reading objects that are assumed to be
787 // partially initialized objects.
788 // * The fields of the object are deserialized in order, under the
789 // assumption that objects are laid out in such a way that any fields
790 // required for object iteration (e.g. length fields) are deserialized
791 // before fields with objects.
792 // - We ensure this is the case by DCHECKing on object allocation that the
793 // previously allocated object has a valid size (see `Allocate`).
794 Tagged<HeapObject> raw_obj =
795 Allocate(allocation, size_in_bytes, HeapObject::RequiredAlignment(*map));
796 raw_obj->set_map_after_allocation(isolate_, *map);
797 MemsetTagged(raw_obj->RawField(kTaggedSize),
798 Smi::uninitialized_deserialization_value(), size_in_tagged - 1);
799 DCHECK(raw_obj->CheckRequiredAlignment(isolate()));
800
801 // Make sure BytecodeArrays have a valid age, so that the marker doesn't
802 // break when making them older.
803 if (IsSharedFunctionInfo(raw_obj, isolate())) {
804 Cast<SharedFunctionInfo>(raw_obj)->set_age(0);
805 } else if (IsEphemeronHashTable(raw_obj)) {
806 // Make sure EphemeronHashTables have valid HeapObject keys, so that the
807 // marker does not break when marking EphemeronHashTable, see
808 // MarkingVisitorBase::VisitEphemeronHashTable.
809 Tagged<EphemeronHashTable> table = Cast<EphemeronHashTable>(raw_obj);
810 MemsetTagged(Cast<HeapObject>(table)->RawField(table->kElementsStartOffset),
811 ReadOnlyRoots(isolate()).undefined_value(),
812 (size_in_bytes - table->kElementsStartOffset) / kTaggedSize);
813 }
814
815#ifdef DEBUG
816 PtrComprCageBase cage_base(isolate());
817 // We want to make sure that all embedder pointers are initialized to null.
818 if (IsJSObject(raw_obj, cage_base) &&
819 Cast<JSObject>(raw_obj)->MayHaveEmbedderFields()) {
820 Tagged<JSObject> js_obj = Cast<JSObject>(raw_obj);
821 for (int i = 0; i < js_obj->GetEmbedderFieldCount(); ++i) {
822 void* pointer;
823 CHECK(EmbedderDataSlot(js_obj, i).ToAlignedPointer(main_thread_isolate(),
824 &pointer));
825 CHECK_NULL(pointer);
826 }
827 } else if (IsEmbedderDataArray(raw_obj, cage_base)) {
828 Tagged<EmbedderDataArray> array = Cast<EmbedderDataArray>(raw_obj);
829 EmbedderDataSlot start(array, 0);
830 EmbedderDataSlot end(array, array->length());
831 for (EmbedderDataSlot slot = start; slot < end; ++slot) {
832 void* pointer;
833 CHECK(slot.ToAlignedPointer(main_thread_isolate(), &pointer));
834 CHECK_NULL(pointer);
835 }
836 }
837#endif
838
839 Handle<HeapObject> obj = handle(raw_obj, isolate());
840 back_refs_.push_back(obj);
841 if (v8_flags.trace_deserialization) {
842 PrintF(" %*s(set obj backref %u)\n", depth_, "",
843 static_cast<int>(back_refs_.size() - 1));
844 }
845
846 ReadData(obj, 1, size_in_tagged);
847 PostProcessNewObject(map, obj, space);
848
849#ifdef DEBUG
850 if (IsInstructionStream(*obj, cage_base)) {
851 DCHECK(space == SnapshotSpace::kCode ||
852 space == SnapshotSpace::kReadOnlyHeap);
853 } else {
854 DCHECK_NE(space, SnapshotSpace::kCode);
855 }
856 if (IsTrustedObject(*obj)) {
857 DCHECK_EQ(space, SnapshotSpace::kTrusted);
858 } else {
859 DCHECK_NE(space, SnapshotSpace::kTrusted);
860 }
861#endif // DEBUG
862
863 return obj;
864}

866template <typename IsolateT>
867Handle<HeapObject> Deserializer<IsolateT>::ReadMetaMap(SnapshotSpace space) {
868 const int size_in_bytes = Map::kSize;
869 const int size_in_tagged = size_in_bytes / kTaggedSize;
870
871 Tagged<HeapObject> raw_obj =
872 Allocate(SpaceToAllocation(space), size_in_bytes, kTaggedAligned);
873 raw_obj->set_map_after_allocation(isolate_, UncheckedCast<Map>(raw_obj));
874 MemsetTagged(raw_obj->RawField(kTaggedSize),
875 Smi::uninitialized_deserialization_value(), size_in_tagged - 1);

878 Handle<HeapObject> obj = handle(raw_obj, isolate());
879 back_refs_.push_back(obj);
880 if (v8_flags.trace_deserialization) {

883 }
884
885 // Set the instance-type manually, to allow backrefs to read it.
886 UncheckedCast<Map>(*obj)->set_instance_type(MAP_TYPE);
887
888 ReadData(obj, 1, size_in_tagged);
889 PostProcessNewObject(Cast<Map>(obj), obj, space);
890
891 return obj;

920// given number of cases matches the number of expected cases for that bytecode.
921template <int byte_code_count, int expected>
922constexpr uint8_t VerifyBytecodeCount(uint8_t bytecode) {
923 static_assert(byte_code_count == expected);
924 return bytecode;
925}

929// Helper macro (and its implementation detail) for specifying a range of cases.
930// Use as "case CASE_RANGE(byte_code, num_bytecodes):"
931#define CASE_RANGE(byte_code, num_bytecodes) \
932 CASE_R##num_bytecodes( \
933 (VerifyBytecodeCount<byte_code##Count, num_bytecodes>(byte_code)))
934#define CASE_R1(byte_code) byte_code
935#define CASE_R2(byte_code) CASE_R1(byte_code) : case CASE_R1(byte_code + 1)
936#define CASE_R3(byte_code) CASE_R2(byte_code) : case CASE_R1(byte_code + 2)
937#define CASE_R4(byte_code) CASE_R2(byte_code) : case CASE_R2(byte_code + 2)
938#define CASE_R8(byte_code) CASE_R4(byte_code) : case CASE_R4(byte_code + 4)
939#define CASE_R16(byte_code) CASE_R8(byte_code) : case CASE_R8(byte_code + 8)
940#define CASE_R32(byte_code) CASE_R16(byte_code) : case CASE_R16(byte_code + 16)
941
942// This generates a case range for all the spaces.
943// clang-format off
944#define CASE_RANGE_ALL_SPACES(bytecode) \
945 SpaceEncoder<bytecode>::Encode(SnapshotSpace::kOld): \
946 case SpaceEncoder<bytecode>::Encode(SnapshotSpace::kCode): \
947 case SpaceEncoder<bytecode>::Encode(SnapshotSpace::kReadOnlyHeap): \
948 case SpaceEncoder<bytecode>::Encode(SnapshotSpace::kTrusted)
949// clang-format on
950
951template <typename IsolateT>
952void Deserializer<IsolateT>::ReadData(Handle<HeapObject> object,
953 int start_slot_index,
954 int end_slot_index) {
955 int current = start_slot_index;
956 while (current < end_slot_index) {
957 uint8_t data = source_.Get();
958 current += ReadSingleBytecodeData(
959 data, SlotAccessorForHeapObject::ForSlotIndex(object, current));
960 }
961 CHECK_EQ(current, end_slot_index);
962}

965void Deserializer<IsolateT>::ReadData(FullMaybeObjectSlot start,
966 FullMaybeObjectSlot end) {
967 FullMaybeObjectSlot current = start;
968 while (current < end) {
969 uint8_t data = source_.Get();
970 current += ReadSingleBytecodeData(data, SlotAccessorForRootSlots(current));
971 }

975template <typename IsolateT>
976template <typename SlotAccessor>
977int Deserializer<IsolateT>::ReadSingleBytecodeData(uint8_t data,
978 SlotAccessor slot_accessor) {
979 if (v8_flags.trace_deserialization) {
980 PrintF("%02x ", data);
981 }
982 switch (data) {
983 case CASE_RANGE_ALL_SPACES(kNewObject):
984 return ReadNewObject(data, slot_accessor);
985 case kBackref:
986 return ReadBackref(data, slot_accessor);
987 case kReadOnlyHeapRef:
988 return ReadReadOnlyHeapRef(data, slot_accessor);
989 case kRootArray:
990 return ReadRootArray(data, slot_accessor);
991 case kStartupObjectCache:
992 return ReadStartupObjectCache(data, slot_accessor);
993 case kSharedHeapObjectCache:
994 return ReadSharedHeapObjectCache(data, slot_accessor);
995 case kNewContextlessMetaMap:
996 case kNewContextfulMetaMap:
997 return ReadNewMetaMap(data, slot_accessor);
998 case kSandboxedExternalReference:
999 case kExternalReference:
1000 return ReadExternalReference(data, slot_accessor);
1001 case kSandboxedRawExternalReference:
1002 return ReadRawExternalReference(data, slot_accessor);
1003 case kAttachedReference:
1004 return ReadAttachedReference(data, slot_accessor);
1005 case kNop:
1006 return 0;
1007 case kRegisterPendingForwardRef:
1008 return ReadRegisterPendingForwardRef(data, slot_accessor);
1009 case kResolvePendingForwardRef:
1010 return ReadResolvePendingForwardRef(data, slot_accessor);
1011 case kSynchronize:
1012 // If we get here then that indicates that you have a mismatch between
1013 // the number of GC roots when serializing and deserializing.
1014 UNREACHABLE();
1015 case kVariableRawData:
1016 return ReadVariableRawData(data, slot_accessor);
1017 case kVariableRepeatRoot:
1018 return ReadVariableRepeatRoot(data, slot_accessor);
1019 case kOffHeapBackingStore:
1020 case kOffHeapResizableBackingStore:
1021 return ReadOffHeapBackingStore(data, slot_accessor);
1022 case kSandboxedApiReference:
1023 case kApiReference:
1024 return ReadApiReference(data, slot_accessor);
1025 case kClearedWeakReference:
1026 return ReadClearedWeakReference(data, slot_accessor);
1027 case kWeakPrefix:
1028 return ReadWeakPrefix(data, slot_accessor);
1029 case kIndirectPointerPrefix:
1030 return ReadIndirectPointerPrefix(data, slot_accessor);
1031 case kInitializeSelfIndirectPointer:
1032 return ReadInitializeSelfIndirectPointer(data, slot_accessor);
1033 case kAllocateJSDispatchEntry:
1034 return ReadAllocateJSDispatchEntry(data, slot_accessor);
1035 case kProtectedPointerPrefix:
1036 return ReadProtectedPointerPrefix(data, slot_accessor);
1037 case CASE_RANGE(kRootArrayConstants, 32):
1038 return ReadRootArrayConstants(data, slot_accessor);
1039 case CASE_RANGE(kHotObject, 8):
1040 return ReadHotObject(data, slot_accessor);
1041 case CASE_RANGE(kFixedRawData, 32):
1042 return ReadFixedRawData(data, slot_accessor);
1043 case CASE_RANGE(kFixedRepeatRoot, 16):
1044 return ReadFixedRepeatRoot(data, slot_accessor);
1045
1046#ifdef DEBUG
1047#define UNUSED_CASE(byte_code) \
1048 case byte_code: \
1049 UNREACHABLE();
1050 UNUSED_SERIALIZER_BYTE_CODES(UNUSED_CASE)
1051#endif
1052#undef UNUSED_CASE
1053 }
1054
1055 // The above switch, including UNUSED_SERIALIZER_BYTE_CODES, covers all
1056 // possible bytecodes; but, clang doesn't realize this, so we have an explicit
1057 // UNREACHABLE here too.
1058 UNREACHABLE();
1059}
