17void LocalDeclEncoder::Prepend(Zone* zone, const uint8_t** start,
18 const uint8_t** end) const {
19 size_t size = (*end - *start);
20 uint8_t* buffer =
21 zone->AllocateArray<uint8_t, LocalDeclEncoderBuffer>(Size() + size);
22 size_t pos = Emit(buffer);
23 if (size > 0) {
24 memcpy(buffer + pos, *start, size);
25 }
26 pos += size;
27 *start = buffer;
28 *end = buffer + pos;
29}

31size_t LocalDeclEncoder::Emit(uint8_t* buffer) const {
32 uint8_t* pos = buffer;
33 LEBHelper::write_u32v(&pos, static_cast<uint32_t>(local_decls.size()));
34 for (auto& local_decl : local_decls) {
35 uint32_t locals_count = local_decl.first;
36 ValueType locals_type = local_decl.second;
37 LEBHelper::write_u32v(&pos, locals_count);
38 *pos = locals_type.value_type_code();
39 ++pos;
40 if (locals_type.is_rtt()) {
41 LEBHelper::write_u32v(&pos, locals_type.ref_index().index);
42 }
43 if (locals_type.encoding_needs_shared()) {
44 *pos = kSharedFlagCode;
45 ++pos;
46 }
47 if (locals_type.encoding_needs_heap_type()) {
48 LEBHelper::write_i32v(&pos, locals_type.heap_type().code());
49 }
50 }
51 DCHECK_EQ(Size(), pos - buffer);
52 return static_cast<size_t>(pos - buffer);
53}

55uint32_t LocalDeclEncoder::AddLocals(uint32_t count, ValueType type) {
56 uint32_t result =
57 static_cast<uint32_t>(total + (sig ? sig->parameter_count() : 0));
58 total += count;
59 if (!local_decls.empty() && local_decls.back().second == type) {
60 count += local_decls.back().first;
61 local_decls.pop_back();
62 }
63 local_decls.push_back(std::pair<uint32_t, ValueType>(count, type));
64 return result;
65}

67// Size = (size of locals count) +
68// (for each local pair <reps, type>, (size of reps) + (size of type))
69size_t LocalDeclEncoder::Size() const {
70 size_t size = LEBHelper::sizeof_u32v(local_decls.size());
71 for (auto p : local_decls) {
72 size +=
73 LEBHelper::sizeof_u32v(p.first) + // number of locals
74 1 + // Opcode
75 (p.second.encoding_needs_shared() ? 1 : 0) +
76 (p.second.encoding_needs_heap_type()
77 ? LEBHelper::sizeof_i32v(p.second.heap_type().code())
78 : 0) +
79 (p.second.is_rtt() ? LEBHelper::sizeof_u32v(p.second.ref_index().index)
80 : 0);
81 }
82 return size;
83}
