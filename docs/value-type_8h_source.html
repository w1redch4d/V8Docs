97// Represents a WebAssembly heap type, as per the typed-funcref and gc
98// proposals.
99// The underlying Representation enumeration encodes heap types as follows:
100// a number t < kV8MaxWasmTypes represents the type defined in the module at
101// index t. Numbers directly beyond that represent the generic heap types. The
102// next number represents the bottom heap type (internal use).
103class HeapTypeBase {
104 public:
105 enum Representation : uint32_t {
106 kFunc = kV8MaxWasmTypes, // shorthand: c
107 kEq, // shorthand: q
108 kI31, // shorthand: j
109 kStruct, // shorthand: o
110 kArray, // shorthand: g
111 kAny, //
112 kExtern, // shorthand: a.
113 kExternString, // Internal type for optimization purposes.
114 // Subtype of extern.
115 // Used by the js-builtin-strings proposal.
116 kExn, //
117 kString, // shorthand: w.
118 kStringViewWtf8, // shorthand: x.
119 kStringViewWtf16, // shorthand: y.
120 kStringViewIter, // shorthand: z.
121 kNone, //
122 kNoFunc, //
123 kNoExtern, //
124 kNoExn, //
125 kFuncShared,
126 kEqShared,
127 kI31Shared,
128 kStructShared,
129 kArrayShared,
130 kAnyShared,
131 kExternShared,
132 kExternStringShared,
133 kExnShared,
134 kStringShared,
135 kStringViewWtf8Shared,
136 kStringViewWtf16Shared,
137 kStringViewIterShared,
138 kNoneShared,
139 kNoFuncShared,
140 kNoExternShared,
141 kNoExnShared,
142 // This value is an internal type (not part of the Wasm spec) that
143 // is the common supertype across all type hierarchies. It should never
144 // appear in validated Wasm programs, but is used to signify that we don't
145 // have any information about a particular value and to prevent bugs in our
146 // typed optimizations, see crbug.com/361652141. Note: kTop is the neutral
147 // element wrt. to intersection (whereas kBottom is for union), and kBottom
148 // is indicating unreachable code, which might be used for subsequent
149 // optimizations, e.g., DCE.
150 kTop,
151 // This value is used to represent failures in the parsing of heap types and
152 // does not correspond to a Wasm heap type. It has to be last in this list.
153 kBottom
154 };

417 // Returns the code that represents this heap type in the wasm binary format.
418 constexpr int32_t code() const {
419 // Type codes represent the first byte of the LEB128 encoding. To get the
420 // int32 represented by a code, we need to sign-extend it from 7 to 32 bits.
421 int32_t mask = 0xFFFFFF80;
422 switch (representation()) {
423 case kFunc:
424 case kFuncShared:
425 return mask | kFuncRefCode;
426 case kEq:
427 case kEqShared:
428 return mask | kEqRefCode;
429 case kI31:
430 case kI31Shared:
431 return mask | kI31RefCode;
432 case kStruct:
433 case kStructShared:
434 return mask | kStructRefCode;
435 case kArray:
436 case kArrayShared:
437 return mask | kArrayRefCode;
438 case kExtern:
439 case kExternShared:
440 return mask | kExternRefCode;
441 case kAny:
442 case kAnyShared:
443 return mask | kAnyRefCode;
444 case kExn:
445 case kExnShared:
446 return mask | kExnRefCode;
447 case kString:
448 case kStringShared:
449 return mask | kStringRefCode;
450 case kStringViewWtf8:
451 case kStringViewWtf8Shared:
452 return mask | kStringViewWtf8Code;
453 case kStringViewWtf16:
454 case kStringViewWtf16Shared:
455 return mask | kStringViewWtf16Code;
456 case kStringViewIter:
457 case kStringViewIterShared:
458 return mask | kStringViewIterCode;
459 case kNone:
460 case kNoneShared:
461 return mask | kNoneCode;

463 case kNoExternShared:
464 return mask | kNoExternCode;
465 case kNoFunc:
466 case kNoFuncShared:
467 return mask | kNoFuncCode;
468 case kNoExn:
469 case kNoExnShared:
470 return mask | kNoExnCode;
471 default:
472 DCHECK(is_index());
473 return static_cast<int32_t>(representation());

729 // format.
730 // For compatibility with the reftypes and exception-handling proposals, this
731 // function prioritizes shorthand encodings
732 // (e.g., {Ref(HeapType::kFunc, kNullable).value_type_code()} will return
733 // kFuncrefCode and not kRefNullCode).
734 constexpr ValueTypeCode value_type_code() const {
735 switch (kind()) {
736 case kRefNull:
737 switch (heap_representation()) {
738 case HeapType::kFunc:
739 return kFuncRefCode;
740 case HeapType::kEq:
741 return kEqRefCode;
742 case HeapType::kExtern:
743 return kExternRefCode;
744 case HeapType::kAny:
745 return kAnyRefCode;
746 case HeapType::kExn:
747 return kExnRefCode;
748 case HeapType::kI31:
749 return kI31RefCode;
750 case HeapType::kStruct:
751 return kStructRefCode;
752 case HeapType::kArray:
753 return kArrayRefCode;
754 case HeapType::kString:
755 return kStringRefCode;
756 case HeapType::kNone:
757 return kNoneCode;
758 case HeapType::kNoExtern:
759 return kNoExternCode;
760 case HeapType::kNoFunc:
761 return kNoFuncCode;
762 default:
763 return kRefNullCode;
764 }
765 case kRef:
766 switch (heap_representation()) {
767 // String views are non-nullable references.
768 case HeapType::kStringViewWtf8:
769 return kStringViewWtf8Code;
770 case HeapType::kStringViewWtf16:
771 return kStringViewWtf16Code;
772 case HeapType::kStringViewIter:
773 return kStringViewIterCode;
774 // Currently, no other non-nullable shorthands exist.
775 default:
776 return kRefCode;
777 }
778#define NUMERIC_TYPE_CASE(kind, ...) \
779 case k##kind: \
780 return k##kind##Code;
781 FOREACH_NUMERIC_VALUE_TYPE(NUMERIC_TYPE_CASE)
782#undef NUMERIC_TYPE_CASE
783 case kVoid:

786 // compiler-internal type only.
787 case kRtt:
788 case kTop:
789 case kBottom:
790 UNREACHABLE();
