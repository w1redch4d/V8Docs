28// The fast double-to-unsigned-int conversion routine does not guarantee
29// rounding towards zero, or any reasonable value if the argument is larger
30// than what fits in an unsigned 32-bit integer.
31inline unsigned int FastD2UI(double x) {
32 // There is no unsigned version of lrint, so there is no fast path
33 // in this function as there is in FastD2I. Using lrint doesn't work
34 // for values of 2^31 and above.
35
36 // Convert "small enough" doubles to uint32_t by fixing the 32
37 // least significant non-fractional bits in the low 32 bits of the
38 // double, and reading them from there.
39 const double k2Pow52 = 4503599627370496.0;
40 bool negative = x < 0;
41 if (negative) {
42 x = -x;
43 }
44 if (x < k2Pow52) {
45 x += k2Pow52;
46 uint32_t result;
47#ifndef V8_TARGET_BIG_ENDIAN
48 void* mantissa_ptr = reinterpret_cast<void*>(&x);
49#else
50 void* mantissa_ptr =
51 reinterpret_cast<void*>(reinterpret_cast<Address>(&x) + kInt32Size);
52#endif
53 // Copy least significant 32 bits of mantissa.
54 memcpy(&result, mantissa_ptr, sizeof(result));
55 return negative ? ~result + 1 : result;
56 }
57 // Large number (outside uint32 range), Infinity or NaN.
58 return 0x80000000u; // Return integer indefinite.
59}

61// Adopted from https://gist.github.com/rygorous/2156668
62inline uint16_t DoubleToFloat16(double value) {
63 uint64_t in = base::bit_cast<uint64_t>(value);
64 uint16_t out = 0;
65
66 // Take the absolute value of the input.
67 uint64_t sign = in & kFP64SignMask;
68 in ^= sign;
69
70 if (in >= kFP16InfinityAndNaNInfimum) {
71 // Result is infinity or NaN.
72 out = (in > kFP64Infinity) ? kFP16qNaN // NaN->qNaN
73 : kFP16Infinity; // Inf->Inf
74 } else {
75 // Result is a (de)normalized number or zero.
76
77 if (in < kFP16DenormalThreshold) {
78 // Result is a denormal or zero. Use the magic value and FP addition to
79 // align 10 mantissa bits at the bottom of the float. Depends on FP
80 // addition being round-to-nearest-even.
81 double temp = base::bit_cast<double>(in) +
82 base::bit_cast<double>(kFP64To16DenormalMagic);
83 out = base::bit_cast<uint64_t>(temp) - kFP64To16DenormalMagic;
84 } else {
85 // Result is not a denormal.
86
87 // Remember if the result mantissa will be odd before rounding.
88 uint64_t mant_odd = (in >> (kFP64MantissaBits - kFP16MantissaBits)) & 1;
89
90 // Update the exponent and round to nearest even.
91 //
92 // Rounding to nearest even is handled in two parts. First, adding
93 // kFP64To16RebiasExponentAndRound has the effect of rebiasing the
94 // exponent and that if any of the lower 41 bits of the mantissa are set,
95 // the 11th mantissa bit from the front becomes set. Second, adding
96 // mant_odd ensures ties are rounded to even.
97 in += kFP64To16RebiasExponentAndRound;
98 in += mant_odd;
99
100 out = in >> (kFP64MantissaBits - kFP16MantissaBits);
101 }
102 }
103
104 out |= sign >> 48;
105 return out;
106}

108inline float DoubleToFloat32(double x) {
109 using limits = std::numeric_limits<float>;
110 if (x > limits::max()) {
111 // kRoundingThreshold is the maximum double that rounds down to
112 // the maximum representable float. Its mantissa bits are:
113 // 1111111111111111111111101111111111111111111111111111
114 // [<--- float range --->]
115 // Note the zero-bit right after the float mantissa range, which
116 // determines the rounding-down.
117 static const double kRoundingThreshold = 3.4028235677973362e+38;
118 if (x <= kRoundingThreshold) return limits::max();
119 return limits::infinity();
120 }
121 if (x < limits::lowest()) {
122 // Same as above, mirrored to negative numbers.
123 static const double kRoundingThreshold = -3.4028235677973362e+38;
124 if (x >= kRoundingThreshold) return limits::lowest();
125 return -limits::infinity();
126 }
127 return static_cast<float>(x);
128}

130// #sec-tointegerorinfinity
131inline double DoubleToInteger(double x) {
132 // ToIntegerOrInfinity normalizes -0 to +0. Special case 0 for performance.
133 if (std::isnan(x) || x == 0.0) return 0;
134 if (!std::isfinite(x)) return x;
135 // Add 0.0 in the truncation case to ensure this doesn't return -0.
136 return ((x > 0) ? std::floor(x) : std::ceil(x)) + 0.0;
137}

139// Implements most of https://tc39.github.io/ecma262/#sec-toint32.
140int32_t DoubleToInt32(double x) {
141 if ((std::isfinite(x)) && (x <= INT_MAX) && (x >= INT_MIN)) {
142 // All doubles within these limits are trivially convertable to an int.
143 return static_cast<int32_t>(x);
144 }
145 base::Double d(x);
146 int exponent = d.Exponent();
147 uint64_t bits;
148 if (exponent < 0) {
149 if (exponent <= -base::Double::kSignificandSize) return 0;
150 bits = d.Significand() >> -exponent;
151 } else {
152 if (exponent > 31) return 0;
153 // Masking to a 32-bit value ensures that the result of the
154 // static_cast<int64_t> below is not the minimal int64_t value,
155 // which would overflow on multiplication with d.Sign().
156 bits = (d.Significand() << exponent) & 0xFFFFFFFFul;
157 }
158 return static_cast<int32_t>(d.Sign() * static_cast<int64_t>(bits));
159}

161// Implements https://heycam.github.io/webidl/#abstract-opdef-converttoint for
162// the general case (step 1 and steps 8 to 12). Support for Clamp and
163// EnforceRange will come in the future.
164inline int64_t DoubleToWebIDLInt64(double x) {
165 if ((std::isfinite(x)) && (x <= kMaxSafeInteger) && (x >= kMinSafeInteger)) {
166 // All doubles within these limits are trivially convertable to an int.
167 return static_cast<int64_t>(x);
168 }
169 base::Double d(x);
170 int exponent = d.Exponent();
171 uint64_t bits;
172 if (exponent < 0) {
173 if (exponent <= -base::Double::kSignificandSize) return 0;
174 bits = d.Significand() >> -exponent;
175 } else {
176 if (exponent > 63) return 0;
177 bits = (d.Significand() << exponent);
178 int64_t bits_int64 = static_cast<int64_t>(bits);
179 if (bits_int64 == std::numeric_limits<int64_t>::min()) {
180 return bits_int64;
181 }
182 }
183 return static_cast<int64_t>(d.Sign() * static_cast<int64_t>(bits));
184}

212bool DoubleToUint32IfEqualToSelf(double value, uint32_t* uint32_value) {
213 const double k2Pow52 = 4503599627370496.0;
214 const uint32_t kValidTopBits = 0x43300000;
215 const uint64_t kBottomBitMask = 0x0000'0000'FFFF'FFFF;
216
217 // Add 2^52 to the double, to place valid uint32 values in the low-significant
218 // bits of the exponent, by effectively setting the (implicit) top bit of the
219 // significand. Note that this addition also normalises 0.0 and -0.0.
220 double shifted_value = value + k2Pow52;
221
222 // At this point, a valid uint32 valued double will be represented as:
223 //
224 // sign = 0
225 // exponent = 52
226 // significand = 1. 00...00 <value>
227 // implicit^ ^^^^^^^ 32 bits
228 // ^^^^^^^^^^^^^^^ 52 bits
229 //
230 // Therefore, we can first check the top 32 bits to make sure that the sign,
231 // exponent and remaining significand bits are valid, and only then check the
232 // value in the bottom 32 bits.
233
234 uint64_t result = base::bit_cast<uint64_t>(shifted_value);
235 if ((result >> 32) == kValidTopBits) {
236 *uint32_value = result & kBottomBitMask;
237 return FastUI2D(result & kBottomBitMask) == value;
238 }
239 return false;
240}

256 return value;

262 if (value < max) return static_cast<uint32_t>(value);

293bool TryNumberToSize(Tagged<Object> number, size_t* result) {
294 // Do not create handles in this function! Don't use SealHandleScope because
295 // the function can be used concurrently.
296 if (IsSmi(number)) {
297 int value = Smi::ToInt(number);
298 DCHECK(static_cast<unsigned>(Smi::kMaxValue) <=
299 std::numeric_limits<size_t>::max());
300 if (value >= 0) {
301 *result = static_cast<size_t>(value);
302 return true;
303 }
304 return false;
305 } else {
306 double value = Cast<HeapNumber>(number)->value();
307 // If value is compared directly to the limit, the limit will be
308 // casted to a double and could end up as limit + 1,
309 // because a double might not have enough mantissa bits for it.
310 // So we might as well cast the limit first, and use < instead of <=.
311 double maxSize = static_cast<double>(std::numeric_limits<size_t>::max());
312 if (value >= 0 && value < maxSize) {
313 *result = static_cast<size_t>(value);
314 return true;
315 } else {
316 return false;
317 }
318 }
319}
