41template <DeoptFrameVisitMode mode, typename Function>
42void DeepForEachInputSingleFrameImpl(
43 const_if_default<mode, DeoptFrame>& frame, InputLocation*& input_location,
44 Function&& f,
45 std::function<bool(interpreter::Register)> is_result_register) {
46 switch (frame.type()) {
47 case DeoptFrame::FrameType::kInterpretedFrame:
48 f(frame.as_interpreted().closure(), input_location);
49 frame.as_interpreted().frame_state()->ForEachValue(
50 frame.as_interpreted().unit(),
51 [&](ValueNodeT<mode> node, interpreter::Register reg) {
52 // Skip over the result location for lazy deopts, since it is
53 // irrelevant for lazy deopts (unoptimized code will recreate the
54 // result).
55 if (is_result_register(reg)) return;
56 f(node, input_location);
57 });
58 break;
59 case DeoptFrame::FrameType::kInlinedArgumentsFrame: {
60 // The inlined arguments frame can never be the top frame.
61 f(frame.as_inlined_arguments().closure(), input_location);
62 for (ValueNodeT<mode> node : frame.as_inlined_arguments().arguments()) {
63 f(node, input_location);
64 }
65 break;
66 }
67 case DeoptFrame::FrameType::kConstructInvokeStubFrame: {
68 f(frame.as_construct_stub().receiver(), input_location);
69 f(frame.as_construct_stub().context(), input_location);
70 break;
71 }
72 case DeoptFrame::FrameType::kBuiltinContinuationFrame:
73 for (ValueNodeT<mode> node :
74 frame.as_builtin_continuation().parameters()) {
75 f(node, input_location);
76 }
77 f(frame.as_builtin_continuation().context(), input_location);
78 break;
79 }
80}

82template <DeoptFrameVisitMode mode, typename Function>
83void DeepForVirtualObject(VirtualObject* vobject,
84 InputLocation*& input_location,
85 const VirtualObject::List& virtual_objects,
86 Function&& f) {
87 if (vobject->type() != VirtualObject::kDefault) return;
88 for (uint32_t i = 0; i < vobject->slot_count(); i++) {
89 ValueNode* value = vobject->get_by_index(i);
90 if (IsConstantNode(value->opcode())) {
91 // No location assigned to constants.
92 continue;
93 }
94 if constexpr (mode == DeoptFrameVisitMode::kRemoveIdentities) {
95 if (value->Is<Identity>()) {
96 value = value->input(0).node();
97 vobject->set_by_index(i, value);
98 }
99 }
100 // Special nodes.
101 switch (value->opcode()) {
102 case Opcode::kArgumentsElements:
103 case Opcode::kArgumentsLength:
104 case Opcode::kRestLength:
105 // No location assigned to these opcodes.
106 break;
107 case Opcode::kVirtualObject:
108 UNREACHABLE();
109 case Opcode::kInlinedAllocation: {
110 InlinedAllocation* alloc = value->Cast<InlinedAllocation>();
111 VirtualObject* inner_vobject = virtual_objects.FindAllocatedWith(alloc);
112 CHECK_NOT_NULL(inner_vobject);
113 // Check if it has escaped.
114 if (alloc->HasBeenAnalysed() && alloc->HasBeenElided()) {
115 input_location++; // Reserved for the inlined allocation.
116 DeepForVirtualObject<mode>(inner_vobject, input_location,
117 virtual_objects, f);
118 } else {
119 f(alloc, input_location);
120 input_location +=
121 inner_vobject->InputLocationSizeNeeded(virtual_objects) + 1;
122 }
123 break;
124 }
125 default:
126 f(value, input_location);
127 input_location++;
128 break;
129 }
130 }
131}

133template <DeoptFrameVisitMode mode, typename Function>
134void DeepForEachInputAndVirtualObject(
135 const_if_default<mode, DeoptFrame>& frame, InputLocation*& input_location,
136 const VirtualObject::List& virtual_objects, Function&& f,
137 std::function<bool(interpreter::Register)> is_result_register =
138 [](interpreter::Register) { return false; }) {
139 auto update_node = [&f, &virtual_objects](ValueNodeT<mode> node,
140 InputLocation*& input_location) {
141 DCHECK(!node->template Is<VirtualObject>());
142 if constexpr (mode == DeoptFrameVisitMode::kRemoveIdentities) {
143 if (node->template Is<Identity>()) {
144 node = node->input(0).node();
145 }
146 }
147 if (auto alloc = node->template TryCast<InlinedAllocation>()) {
148 VirtualObject* vobject = virtual_objects.FindAllocatedWith(alloc);
149 CHECK_NOT_NULL(vobject);
150 if (alloc->HasBeenAnalysed() && alloc->HasBeenElided()) {
151 input_location++; // Reserved for the inlined allocation.
152 return DeepForVirtualObject<mode>(vobject, input_location,
153 virtual_objects, f);
154 } else {
155 f(alloc, input_location);
156 input_location += vobject->InputLocationSizeNeeded(virtual_objects) + 1;
157 }
158 } else {
159 f(node, input_location);
160 input_location++;
161 }
162 };
163 DeepForEachInputSingleFrameImpl<mode>(frame, input_location, update_node,
164 is_result_register);
165}

167template <DeoptFrameVisitMode mode, typename Function>
168void DeepForEachInputImpl(const_if_default<mode, DeoptFrame>& frame,
169 InputLocation*& input_location,
170 const VirtualObject::List& virtual_objects,
171 Function&& f) {
172 if (frame.parent()) {
173 DeepForEachInputImpl<mode>(*frame.parent(), input_location, virtual_objects,
174 f);
175 }
176 DeepForEachInputAndVirtualObject<mode>(frame, input_location, virtual_objects,
177 f);
178}

180template <DeoptFrameVisitMode mode, typename Function>
181void DeepForEachInputForEager(
182 const_if_default<mode, EagerDeoptInfo>* deopt_info, Function&& f) {
183 InputLocation* input_location = deopt_info->input_locations();
184 const VirtualObject::List& virtual_objects =
185 GetVirtualObjects(deopt_info->top_frame());
186 DeepForEachInputImpl<mode>(deopt_info->top_frame(), input_location,
187 virtual_objects, std::forward<Function>(f));
188}
189
190template <DeoptFrameVisitMode mode, typename Function>
191void DeepForEachInputForLazy(const_if_default<mode, LazyDeoptInfo>* deopt_info,
192 Function&& f) {
193 InputLocation* input_location = deopt_info->input_locations();
194 auto& top_frame = deopt_info->top_frame();
195 const VirtualObject::List& virtual_objects = GetVirtualObjects(top_frame);
196 if (top_frame.parent()) {
197 DeepForEachInputImpl<mode>(*top_frame.parent(), input_location,
198 virtual_objects, f);
199 }
200 DeepForEachInputAndVirtualObject<mode>(
201 top_frame, input_location, virtual_objects, f,
202 [deopt_info](interpreter::Register reg) {
203 return deopt_info->IsResultRegister(reg);
204 });
205}

207template <typename Function>
208void DeepForEachInput(const EagerDeoptInfo* deopt_info, Function&& f) {
209 return DeepForEachInputForEager<DeoptFrameVisitMode::kDefault>(deopt_info, f);
