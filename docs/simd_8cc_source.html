63// Searches for |search_element| in |array| using a simple non-vectorized linear
64// search. This is used as a fall-back when SIMD are not available, and to
65// process the end of arrays than SIMD cannot process.
66template <typename T>
67inline uintptr_t slow_search(T* array, uintptr_t array_len, uintptr_t index,
68 T search_element) {
69 for (; index < array_len; index++) {
70 if (array[index] == search_element) {
71 return index;
72 }
73 }
74 return -1;
75}

77#ifdef NEON64
78// extract_first_nonzero_index returns the first non-zero index in |v|. |v| is a
79// Neon vector that can be either 32x4 (the return is then 0, 1, 2 or 3) or 64x2
80// (the return is then 0 or 1). This is more or less equivalent to doing a
81// movemask followed by a tzcnt on Intel.
82//
83// The input |v| should be a vector of -1 or 0 (for instance {0, 0},
84// {0, -1, 0, -1}, {0, -1, 0, 0}), where -1 represents a match (and 0 a
85// non-match), that was obtained by doing a vceqq. This function extract the
86// index of the first non-zero item of the vector. To do so, we "and" the vector
87// with {4, 3, 2, 1} (each number is "4 - the index of the item it's in"), which
88// produces a vector of "indices or 0". Then, we extract the maximum of this
89// vector, which is the index of the 1st match. An example:
90//
91// v = {-1, 0, 0, -1}
92// mask = {4, 3, 2, 1}
93// v & mask = {4, 0, 0, 1}
94// max(v & mask) = 4
95// index of the first match = 4-max = 4-4 = 0
96//
97
98// With MSVC, uint32x4_t and uint64x2_t typedef to a union, where first member
99// is uint64_t[2], and not uint32_t[4].
100// C++ standard dictates that a union can only be initialized through its first
101// member, which forces us to have uint64_t[2] for definition.
102#if defined(_MSC_VER) && !defined(__clang__)
103#define PACK32x4(w, x, y, z) \
104 { ((w) + (uint64_t(x) << 32)), ((y) + (uint64_t(z) << 32)) }
105#else
106#define PACK32x4(w, x, y, z) \
107 { (w), (x), (y), (z) }
108#endif // MSVC workaround
109
110V8_ALLOW_UNUSED inline int extract_first_nonzero_index_uint32x4_t(
111 uint32x4_t v) {
112 uint32x4_t mask = PACK32x4(4, 3, 2, 1);
113 mask = vandq_u32(mask, v);
114 return 4 - vmaxvq_u32(mask);
115}
116
117inline int extract_first_nonzero_index_uint64x2_t(uint64x2_t v) {
118 uint32x4_t mask =
119 PACK32x4(2, 0, 1, 0); // Could also be {2,2,1,1} or {0,2,0,1}
120 mask = vandq_u32(mask, vreinterpretq_u32_u64(v));
121 return 2 - vmaxvq_u32(mask);
122}
123
124inline int32_t reinterpret_vmaxvq_u64(uint64x2_t v) {
125 return vmaxvq_u32(vreinterpretq_u32_u64(v));
126}

129#define VECTORIZED_LOOP_Neon(type_load, type_eq, set1, cmp, movemask) \
130 { \
131 constexpr int elems_in_vector = sizeof(type_load) / sizeof(T); \
132 type_load search_element_vec = set1(search_element); \
133 \
134 for (; index + elems_in_vector <= array_len; index += elems_in_vector) { \
135 type_load vector = *reinterpret_cast<type_load*>(&array[index]); \
136 type_eq eq = cmp(vector, search_element_vec); \
137 if (movemask(eq)) { \
138 return index + extract_first_nonzero_index_##type_eq(eq); \
139 } \
140 } \
141 }
142
143#define VECTORIZED_LOOP_x86(type_load, type_eq, set1, cmp, movemask, extract) \
144 { \
145 constexpr int elems_in_vector = sizeof(type_load) / sizeof(T); \
146 type_load search_element_vec = set1(search_element); \
147 \
148 for (; index + elems_in_vector <= array_len; index += elems_in_vector) { \
149 type_load vector = *reinterpret_cast<type_load*>(&array[index]); \
150 type_eq eq = cmp(vector, search_element_vec); \
151 int eq_mask = movemask(eq); \
152 if (eq_mask) { \
153 return index + extract(eq_mask); \
154 } \
155 } \
156 }

158#ifdef __SSE3__
159__m128i _mm_cmpeq_epi64_nosse4_2(__m128i a, __m128i b) {
160 __m128i res = _mm_cmpeq_epi32(a, b);
161 // For each 64-bit value swap results of lower 32 bits comparison with
162 // the results of upper 32 bits comparison.
163 __m128i res_swapped = _mm_shuffle_epi32(res, _MM_SHUFFLE(2, 3, 0, 1));
164 // Report match only when both upper and lower parts of 64-bit values match.
165 return _mm_and_si128(res, res_swapped);
166}
167#endif // __SSE3__

169// Uses SIMD to vectorize the search loop. This function should only be called
170// for large-ish arrays. Note that nothing will break if |array_len| is less
171// than vectorization_threshold: things will just be slower than necessary.
172template <typename T>
173inline uintptr_t fast_search_noavx(T* array, uintptr_t array_len,
174 uintptr_t index, T search_element) {
175 static constexpr bool is_uint32 =
176 sizeof(T) == sizeof(uint32_t) && std::is_integral_v<T>;
177 static constexpr bool is_uint64 =
178 sizeof(T) == sizeof(uint64_t) && std::is_integral_v<T>;
179 static constexpr bool is_double =
180 sizeof(T) == sizeof(double) && std::is_floating_point_v<T>;
181
182 static_assert(is_uint32 || is_uint64 || is_double);
183
184#if !(defined(__SSE3__) || defined(NEON64))
185 // No SIMD available.
186 return slow_search(array, array_len, index, search_element);
187#endif
188
189#ifdef __SSE3__
190 const int target_align = 16;
191#elif defined(NEON64)
192 const int target_align = 16;
193#else
194 const int target_align = 4;
195 UNREACHABLE();
196#endif
197
198 // Scalar loop to reach desired alignment
199 for (;
200 index < array_len &&
201 (reinterpret_cast<std::uintptr_t>(&(array[index])) % target_align) != 0;
202 index++) {
203 if (array[index] == search_element) {
204 return index;
205 }
206 }
207
208 // Inserting one of the vectorized loop
209#ifdef __SSE3__
210 if constexpr (is_uint32) {
211#define MOVEMASK(x) _mm_movemask_ps(_mm_castsi128_ps(x))
212#define EXTRACT(x) base::bits::CountTrailingZeros32(x)
213 VECTORIZED_LOOP_x86(__m128i, __m128i, _mm_set1_epi32, _mm_cmpeq_epi32,
214 MOVEMASK, EXTRACT)
215#undef MOVEMASK
216#undef EXTRACT
217 } else if constexpr (is_uint64) {
218#define MOVEMASK(x) _mm_movemask_ps(_mm_castsi128_ps(x))
219// _mm_cmpeq_epi64_nosse4_2() might produce only the following non-zero
220// patterns:
221// 0b0011 -> 0 (the first value matches),
222// 0b1100 -> 1 (the second value matches),
223// 0b1111 -> 0 (both first and second value match).
224// Thus it's enough to check only the least significant bit.
225#define EXTRACT(x) (((x) & 1) ? 0 : 1)
226 VECTORIZED_LOOP_x86(__m128i, __m128i, _mm_set1_epi64x,
227 _mm_cmpeq_epi64_nosse4_2, MOVEMASK, EXTRACT)
228#undef MOVEMASK
229#undef EXTRACT
230 } else if constexpr (is_double) {
231#define EXTRACT(x) base::bits::CountTrailingZeros32(x)
232 VECTORIZED_LOOP_x86(__m128d, __m128d, _mm_set1_pd, _mm_cmpeq_pd,
233 _mm_movemask_pd, EXTRACT)
234#undef EXTRACT
235 }
236#elif defined(NEON64)
237 if constexpr (is_uint32) {
238 VECTORIZED_LOOP_Neon(uint32x4_t, uint32x4_t, vdupq_n_u32, vceqq_u32,
239 vmaxvq_u32)
240 } else if constexpr (is_uint64) {
241 VECTORIZED_LOOP_Neon(uint64x2_t, uint64x2_t, vdupq_n_u64, vceqq_u64,
242 reinterpret_vmaxvq_u64)
243 } else if constexpr (is_double) {
244 VECTORIZED_LOOP_Neon(float64x2_t, uint64x2_t, vdupq_n_f64, vceqq_f64,
245 reinterpret_vmaxvq_u64)
246 }
247#else
248 UNREACHABLE();
249#endif
250
251 // The vectorized loop stops when there are not enough items left in the array
252 // to fill a vector register. The slow_search function will take care of
253 // iterating through the few remaining items.
254 return slow_search(array, array_len, index, search_element);
255}

273template <typename T>
274TARGET_AVX2 inline uintptr_t fast_search_avx(T* array, uintptr_t array_len,
275 uintptr_t index,
276 T search_element) {
277 static constexpr bool is_uint32 =
278 sizeof(T) == sizeof(uint32_t) && std::is_integral_v<T>;
279 static constexpr bool is_uint64 =
280 sizeof(T) == sizeof(uint64_t) && std::is_integral_v<T>;
281 static constexpr bool is_double =
282 sizeof(T) == sizeof(double) && std::is_floating_point_v<T>;
283
284 static_assert(is_uint32 || is_uint64 || is_double);
285
286 const int target_align = 32;
287 // Scalar loop to reach desired alignment
288 for (;
289 index < array_len &&
290 (reinterpret_cast<std::uintptr_t>(&(array[index])) % target_align) != 0;
291 index++) {
292 if (array[index] == search_element) {
293 return index;
294 }
295 }
296
297 // Generating vectorized loop
298 if constexpr (is_uint32) {
299#define MOVEMASK(x) _mm256_movemask_ps(_mm256_castsi256_ps(x))
300#define EXTRACT(x) base::bits::CountTrailingZeros32(x)
301 VECTORIZED_LOOP_x86(__m256i, __m256i, _mm256_set1_epi32, _mm256_cmpeq_epi32,
302 MOVEMASK, EXTRACT)
303#undef MOVEMASK
304#undef EXTRACT
305 } else if constexpr (is_uint64) {
306#define MOVEMASK(x) _mm256_movemask_pd(_mm256_castsi256_pd(x))
307#define EXTRACT(x) base::bits::CountTrailingZeros32(x)
308 VECTORIZED_LOOP_x86(__m256i, __m256i, _mm256_set1_epi64x,
309 _mm256_cmpeq_epi64, MOVEMASK, EXTRACT)
310#undef MOVEMASK
311#undef EXTRACT
312 } else if constexpr (is_double) {
313#define CMP(a, b) _mm256_cmp_pd(a, b, _CMP_EQ_OQ)
314#define EXTRACT(x) base::bits::CountTrailingZeros32(x)
315 VECTORIZED_LOOP_x86(__m256d, __m256d, _mm256_set1_pd, CMP,
316 _mm256_movemask_pd, EXTRACT)
317#undef CMP
318#undef EXTRACT
319 }
320
321 // The vectorized loop stops when there are not enough items left in the array
322 // to fill a vector register. The slow_search function will take care of
323 // iterating through the few remaining items.
324 return slow_search(array, array_len, index, search_element);
325}

357enum class ArrayIndexOfIncludesKind { DOUBLE, OBJECTORSMI };
358
359// ArrayIndexOfIncludes only handles cases that can be efficiently
360// vectorized:
361//
362// * Searching for a Smi in a Smi array
363//
364// * Searching for a Smi or Double in a Double array
365//
366// * Searching for an object in an object array.
367//
368// Other cases should be dealt with either with the CSA builtin or with the
369// inlined optimized code.
370template <ArrayIndexOfIncludesKind kind>
371Address ArrayIndexOfIncludes(Address array_start, uintptr_t array_len,
372 uintptr_t from_index, Address search_element) {
373 if (array_len == 0) {
374 return Smi::FromInt(-1).ptr();
375 }
376
377 if constexpr (kind == ArrayIndexOfIncludesKind::DOUBLE) {
378 Tagged<FixedDoubleArray> fixed_array =
379 Cast<FixedDoubleArray>(Tagged<Object>(array_start));
380 UnalignedDoubleMember* unaligned_array = fixed_array->begin();
381 // TODO(leszeks): This reinterpret cast is a bit sketchy because the values
382 // are unaligned doubles. Ideally we'd fix the search method to support
383 // UnalignedDoubleMember.
384 static_assert(sizeof(UnalignedDoubleMember) == sizeof(double));
385 double* array = reinterpret_cast<double*>(unaligned_array);
386
387 double search_num;
388 if (IsSmi(Tagged<Object>(search_element))) {
389 search_num = Tagged<Object>(search_element).ToSmi().value();
390 } else {
391 DCHECK(IsHeapNumber(Tagged<Object>(search_element)));
392 search_num = Cast<HeapNumber>(Tagged<Object>(search_element))->value();
393 }
394
395 DCHECK(!std::isnan(search_num));
396
397 if (reinterpret_cast<uintptr_t>(array) % sizeof(double) != 0) {
398 // Slow scalar search for unaligned double array.
399 for (; from_index < array_len; from_index++) {
400 if (fixed_array->is_the_hole(static_cast<int>(from_index))) {
401 // |search_num| cannot be NaN, so there is no need to check against
402 // holes.
403 continue;
404 }
405 if (fixed_array->get_scalar(static_cast<int>(from_index)) ==
406 search_num) {
407 return from_index;
408 }
409 }
410 return Smi::FromInt(-1).ptr();
411 }
412
413 return search<double>(array, array_len, from_index, search_num);
414 }
415
416 if constexpr (kind == ArrayIndexOfIncludesKind::OBJECTORSMI) {
417 Tagged<FixedArray> fixed_array =
418 Cast<FixedArray>(Tagged<Object>(array_start));
419 Tagged_t* array = static_cast<Tagged_t*>(
420 fixed_array->RawFieldOfFirstElement().ToVoidPtr());
421
422 DCHECK(!IsHeapNumber(Tagged<Object>(search_element)));
423 DCHECK(!IsBigInt(Tagged<Object>(search_element)));
424 DCHECK(!IsString(Tagged<Object>(search_element)));
425
426 return search<Tagged_t>(array, array_len, from_index,
427 static_cast<Tagged_t>(search_element));
428 }
429}
