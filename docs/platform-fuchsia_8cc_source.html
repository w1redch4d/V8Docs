29// If VmexResource is unavailable or does not return a valid handle then
30// this will be observed as failures in vmo_replace_as_executable() calls.
31void SetVmexResource() {
32 DCHECK_EQ(g_vmex_resource, ZX_HANDLE_INVALID);
33
34 auto vmex_resource_client =
35 component::Connect<fuchsia_kernel::VmexResource>();
36 if (vmex_resource_client.is_error()) {
37 return;
38 }
39
40 fidl::SyncClient sync_vmex_resource_client(
41 std::move(vmex_resource_client.value()));
42 auto result = sync_vmex_resource_client->Get();
43 if (result.is_error()) {
44 return;
45 }
46
47 g_vmex_resource = result->resource().release();
48}

50zx_vm_option_t GetProtectionFromMemoryPermission(OS::MemoryPermission access) {
51 switch (access) {
52 case OS::MemoryPermission::kNoAccess:
53 case OS::MemoryPermission::kNoAccessWillJitLater:
54 return 0; // no permissions
55 case OS::MemoryPermission::kRead:
56 return ZX_VM_PERM_READ;
57 case OS::MemoryPermission::kReadWrite:
58 return ZX_VM_PERM_READ | ZX_VM_PERM_WRITE;
59 case OS::MemoryPermission::kReadWriteExecute:
60 return ZX_VM_PERM_READ | ZX_VM_PERM_WRITE | ZX_VM_PERM_EXECUTE;
61 case OS::MemoryPermission::kReadExecute:
62 return ZX_VM_PERM_READ | ZX_VM_PERM_EXECUTE;
63 }
64 UNREACHABLE();
65}

67// Determine ZX_VM_ALIGN_X constant corresponding to the specified alignment.
68// Returns 0 if there is none.
69zx_vm_option_t GetAlignmentOptionFromAlignment(size_t alignment) {
70 // The alignment must be one of the ZX_VM_ALIGN_X constants.
71 // See zircon/system/public/zircon/types.h.
72 static_assert(
73 ZX_VM_ALIGN_1KB == (10 << ZX_VM_ALIGN_BASE),
74 "Fuchsia's ZX_VM_ALIGN_1KB constant doesn't match expected value");
75 static_assert(
76 ZX_VM_ALIGN_4GB == (32 << ZX_VM_ALIGN_BASE),
77 "Fuchsia's ZX_VM_ALIGN_4GB constant doesn't match expected value");
78 zx_vm_option_t alignment_log2 = 0;
79 for (int shift = 10; shift <= 32; shift++) {
80 if (alignment == (size_t{1} << shift)) {
81 alignment_log2 = shift;
82 break;
83 }
84 }
85 return alignment_log2 << ZX_VM_ALIGN_BASE;
86}

88enum class PlacementMode {
89 // Attempt to place the object at the provided address, otherwise elsewhere.
90 kUseHint,
91 // Place the object anywhere it fits.
92 kAnywhere,
93 // Place the object at the provided address, otherwise fail.
94 kFixed
95};

97void* MapVmo(const zx::vmar& vmar, void* vmar_base, size_t page_size,
98 void* address, const zx::vmo& vmo, uint64_t offset,
99 PlacementMode placement, size_t size, size_t alignment,
100 OS::MemoryPermission access) {
101 DCHECK_EQ(0, size % page_size);
102 DCHECK_EQ(0, reinterpret_cast<uintptr_t>(address) % page_size);
103 DCHECK_IMPLIES(placement != PlacementMode::kAnywhere, address != nullptr);
104
105 zx_vm_option_t options = GetProtectionFromMemoryPermission(access);
106
107 zx_vm_option_t alignment_option = GetAlignmentOptionFromAlignment(alignment);
108 CHECK_NE(0, alignment_option); // Invalid alignment specified
109 options |= alignment_option;
110
111 size_t vmar_offset = 0;
112 if (placement != PlacementMode::kAnywhere) {
113 // Try placing the mapping at the specified address.
114 uintptr_t target_addr = reinterpret_cast<uintptr_t>(address);
115 uintptr_t base = reinterpret_cast<uintptr_t>(vmar_base);
116 DCHECK_GE(target_addr, base);
117 vmar_offset = target_addr - base;
118 options |= ZX_VM_SPECIFIC;
119 }
120
121 zx_vaddr_t result;
122 zx_status_t status = vmar.map(options, vmar_offset, vmo, 0, size, &result);
123
124 if (status != ZX_OK && placement == PlacementMode::kUseHint) {
125 // If a placement hint was specified but couldn't be used (for example,
126 // because the offset overlapped another mapping), then retry again without
127 // a vmar_offset to let the kernel pick another location.
128 options &= ~(ZX_VM_SPECIFIC);
129 status = vmar.map(options, 0, vmo, 0, size, &result);
130 }
131
132 if (status != ZX_OK) {
133 return nullptr;
134 }
135
136 return reinterpret_cast<void*>(result);
137}

139void* CreateAndMapVmo(const zx::vmar& vmar, void* vmar_base, size_t page_size,
140 void* address, PlacementMode placement, size_t size,
141 size_t alignment, OS::MemoryPermission access) {
142 zx::vmo vmo;
143 if (zx::vmo::create(size, 0, &vmo) != ZX_OK) {
144 return nullptr;
145 }
146 static const char kVirtualMemoryName[] = "v8-virtualmem";
147 vmo.set_property(ZX_PROP_NAME, kVirtualMemoryName,
148 strlen(kVirtualMemoryName));
149
150 // Always call zx_vmo_replace_as_executable() in case the memory will need
151 // to be marked as executable in the future.
152 // TOOD(https://crbug.com/v8/8899): Only call this when we know that the
153 // region will need to be marked as executable in the future.
154 zx::unowned_resource vmex(g_vmex_resource);
155 if (vmo.replace_as_executable(*vmex, &vmo) != ZX_OK) {
156 return nullptr;
157 }
158
159 return MapVmo(vmar, vmar_base, page_size, address, vmo, 0, placement, size,
160 alignment, access);
161}

163bool UnmapVmo(const zx::vmar& vmar, size_t page_size, void* address,
164 size_t size) {
165 DCHECK_EQ(0, reinterpret_cast<uintptr_t>(address) % page_size);

170bool SetPermissionsInternal(const zx::vmar& vmar, size_t page_size,
171 void* address, size_t size,
172 OS::MemoryPermission access) {
173 DCHECK_EQ(0, reinterpret_cast<uintptr_t>(address) % page_size);
174 DCHECK_EQ(0, size % page_size);
175 uint32_t prot = GetProtectionFromMemoryPermission(access);
176 zx_status_t status =
177 vmar.protect(prot, reinterpret_cast<uintptr_t>(address), size);
178
179 // Any failure that's not OOM likely indicates a bug in the caller (e.g.
180 // using an invalid mapping) so attempt to catch that here to facilitate
181 // debugging of these failures. According to the documentation,
182 // zx_vmar_protect cannot return ZX_ERR_NO_MEMORY, so any error here is
183 // unexpected.
184 CHECK_EQ(status, ZX_OK);
185 return status == ZX_OK;
186}

188bool DiscardSystemPagesInternal(const zx::vmar& vmar, size_t page_size,
189 void* address, size_t size) {
190 DCHECK_EQ(0, reinterpret_cast<uintptr_t>(address) % page_size);
191 DCHECK_EQ(0, size % page_size);
192 uint64_t address_int = reinterpret_cast<uint64_t>(address);
193 return vmar.op_range(ZX_VMO_OP_DECOMMIT, address_int, size, nullptr, 0) ==
194 ZX_OK;
195}

197zx_status_t CreateAddressSpaceReservationInternal(
198 const zx::vmar& vmar, void* vmar_base, size_t page_size, void* address,
199 PlacementMode placement, size_t size, size_t alignment,
200 OS::MemoryPermission max_permission, zx::vmar* child,
201 zx_vaddr_t* child_addr) {
202 DCHECK_EQ(0, size % page_size);
203 DCHECK_EQ(0, alignment % page_size);
204 DCHECK_EQ(0, reinterpret_cast<uintptr_t>(address) % alignment);
205 DCHECK_IMPLIES(placement != PlacementMode::kAnywhere, address != nullptr);
206
207 // TODO(v8) determine these based on max_permission.
208 zx_vm_option_t options = ZX_VM_CAN_MAP_READ | ZX_VM_CAN_MAP_WRITE |
209 ZX_VM_CAN_MAP_EXECUTE | ZX_VM_CAN_MAP_SPECIFIC;
210
211 zx_vm_option_t alignment_option = GetAlignmentOptionFromAlignment(alignment);
212 CHECK_NE(0, alignment_option); // Invalid alignment specified
213 options |= alignment_option;
214
215 size_t vmar_offset = 0;
216 if (placement != PlacementMode::kAnywhere) {
217 // Try placing the mapping at the specified address.
218 uintptr_t target_addr = reinterpret_cast<uintptr_t>(address);
219 uintptr_t base = reinterpret_cast<uintptr_t>(vmar_base);
220 DCHECK_GE(target_addr, base);
221 vmar_offset = target_addr - base;
222 options |= ZX_VM_SPECIFIC;
223 }
224
225 zx_status_t status =
226 vmar.allocate(options, vmar_offset, size, child, child_addr);
227 if (status != ZX_OK && placement == PlacementMode::kUseHint) {
228 // If a placement hint was specified but couldn't be used (for example,
229 // because the offset overlapped another mapping), then retry again without
230 // a vmar_offset to let the kernel pick another location.
231 options &= ~(ZX_VM_SPECIFIC);
232 status = vmar.allocate(options, 0, size, child, child_addr);
233 }
234
235 return status;
236}

380int OS::GetUserTime(uint32_t* secs, uint32_t* usecs) {
381 const auto kNanosPerMicrosecond = 1000ULL;
382 const auto kMicrosPerSecond = 1000000ULL;
383
384 zx_info_thread_stats_t info = {};
385 if (zx::thread::self()->get_info(ZX_INFO_THREAD_STATS, &info, sizeof(info),
386 nullptr, nullptr) != ZX_OK) {
387 return -1;
388 }
389
390 // First convert to microseconds, rounding up.
391 const uint64_t micros_since_thread_started =
392 (info.total_runtime + kNanosPerMicrosecond - 1ULL) / kNanosPerMicrosecond;
393
394 *secs = static_cast<uint32_t>(micros_since_thread_started / kMicrosPerSecond);
395 *usecs =
396 static_cast<uint32_t>(micros_since_thread_started % kMicrosPerSecond);
397 return 0;
398}
