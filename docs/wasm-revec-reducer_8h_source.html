216class NodeGroup {
217 public:
218 // Current only support merge 2 Simd128 into Simd256
219 static constexpr int kSize = kSimd256Size / kSimd128Size;
220 NodeGroup(OpIndex a, OpIndex b) {
221 indexes_[0] = a;
222 indexes_[1] = b;
223 }
224 size_t size() const { return kSize; }
225 OpIndex operator[](int i) const { return indexes_[i]; }
226
227 bool operator==(const NodeGroup& other) const {
228 return indexes_[0] == other.indexes_[0] && indexes_[1] == other.indexes_[1];
229 }
230 bool operator!=(const NodeGroup& other) const {
231 return indexes_[0] != other.indexes_[0] || indexes_[1] != other.indexes_[1];
232 }
233
234 const OpIndex* begin() const { return indexes_; }
235 const OpIndex* end() const { return indexes_ + kSize; }
236
237 private:
238 OpIndex indexes_[kSize];
239};

245// A PackNode consists of a fixed number of isomorphic simd128 nodes which can
246// execute in parallel and convert to a 256-bit simd node later. The nodes in a
247// PackNode must satisfy that they can be scheduled in the same basic block and
248// are mutually independent.
249class PackNode : public NON_EXPORTED_BASE(ZoneObject) {
250 public:
251 enum NodeType {
252 kDefault, // Nodes are naturally packed without special attributes.
253 kForcePackNode, // Nodes do not satisfy some packing rule, but can be
254 // forcely coalesced with a Pack128To256 operation. E.g.
255 // inconsecutive loads. In x64, we can use the vinsertf128
256 // instruction to forcely coalescing two 128-bit values.
257 kShufflePackNode, // Nodes are Simd128Shuffle operations with specific
258 // info.
259 kBundlePackNode, // Nodes representing a i8x16/i16x8 to f32x4 conversion.
260 kIntersectPackNode, // One or more nodes already packed by an existing
261 // PackNode.
262 };
263
264 explicit PackNode(Zone* zone, const NodeGroup& node_group,
265 NodeType node_type = kDefault)
266 : nodes_(node_group),
267 revectorized_node_(),
268 operands_(zone),
269 node_type_(node_type) {}
270 const NodeGroup& nodes() const { return nodes_; }
271 bool IsSame(const NodeGroup& node_group) const {
272 return nodes_ == node_group;
273 }
274 bool IsSame(const PackNode& other) const { return nodes_ == other.nodes_; }
275 V<Simd256> RevectorizedNode() const { return revectorized_node_; }
276 void SetRevectorizedNode(V<Simd256> node) { revectorized_node_ = node; }
277
278 bool IsDefaultPackNode() const { return node_type_ == kDefault; }
279 bool IsForcePackNode() const { return node_type_ == kForcePackNode; }
280 bool IsShufflePackNode() const { return node_type_ == kShufflePackNode; }
281 bool IsBundlePackNode() const { return node_type_ == kBundlePackNode; }
282 // We will force-pack nodes for both ForcePackNode and IntersectPackNode.
283 bool is_force_packing() const {
284 return node_type_ == kForcePackNode || node_type_ == kIntersectPackNode;
285 }
286
287 ForcePackNode* AsForcePackNode() {
288 DCHECK(IsForcePackNode());
289 return reinterpret_cast<ForcePackNode*>(this);
290 }
291 ShufflePackNode* AsShufflePackNode() {
292 DCHECK(IsShufflePackNode());
293 return reinterpret_cast<ShufflePackNode*>(this);
294 }
295 BundlePackNode* AsBundlePackNode() {
296 DCHECK(IsBundlePackNode());
297 return reinterpret_cast<BundlePackNode*>(this);
298 }
299
300 PackNode* GetOperand(int index) const {
301 DCHECK_LT(index, operands_.size());
302 DCHECK(operands_[index]);
303 return operands_[index];
304 }
305
306 void SetOperand(int index, PackNode* pnode) {
307 DCHECK_GE(index, 0);
308 if (operands_.size() < static_cast<size_t>(index + 1)) {
309 operands_.resize(index + 1);
310 }
311 operands_[index] = pnode;
312 }
313
314 ZoneVector<PackNode*>::size_type GetOperandsSize() const {
315 return operands_.size();
316 }
317
318 void Print(Graph* graph) const;
319
320 private:
321 friend class ForcePackNode;
322 NodeGroup nodes_;
323 V<Simd256> revectorized_node_;
324 ZoneVector<PackNode*> operands_;
325 NodeType node_type_;
326};

665 OpIndex GetExtractOpIfNeeded(const PackNode* pnode, OpIndex ig_index,
666 OpIndex og_index) {
667 const auto lane = base::checked_cast<uint8_t>(
668 std::find(pnode->nodes().begin(), pnode->nodes().end(), ig_index) -
669 pnode->nodes().begin());
670
671 // Force PackNode has a dedicated use in SimdPack128To256Op.
672 if (pnode->is_force_packing()) {
673 SimdPack128To256Op& op = __ output_graph()
674 .Get(pnode -> RevectorizedNode())
675 .template Cast<SimdPack128To256Op>();
676 return lane == 0 ? op.left() : op.right();
677 }
678
679 for (auto use : analyzer_.uses(ig_index)) {
680 // Extract128 is needed for the additional Simd128 store before
681 // Simd256 store in case of OOB trap at the higher 128-bit
682 // address.
683 auto use_pnode = analyzer_.GetPackNode(use);
684 if (use_pnode != nullptr && !use_pnode->is_force_packing()) {
685 DCHECK_GE(use_pnode->nodes().size(), 2);
686 if (__ input_graph().Get(use).opcode != Opcode::kStore ||
687 use_pnode->nodes()[0] != use ||
688 use_pnode->nodes()[0] > use_pnode->nodes()[1])
689 continue;
690 }
691
692 return __ Simd256Extract128Lane(og_index, lane);
693 }
694
695 return OpIndex::Invalid();
696 }

786 (analyzer_.GetStartOperation(pnode, ig_index, store))
787 .template Cast<StoreOp>();

828 return analyzer_.GetReducedInput(pnode, old_block_id);

830 RegisterRepresentation::Simd256());

983 V<Simd128> REDUCE_INPUT_GRAPH(Simd128Shuffle)(V<Simd128> ig_index,
984 const Simd128ShuffleOp& op) {
985 PackNode* p = analyzer_.GetPackNode(ig_index);
986 if (!p) {
987 return Adapter::ReduceInputGraphSimd128Shuffle(ig_index, op);
988 }
989
990 ShufflePackNode* pnode = p->AsShufflePackNode();
991 V<Simd256> og_index = pnode->RevectorizedNode();
992 // Skip revectorized node.
993 if (!og_index.valid()) {
994 const ShufflePackNode::SpecificInfo::Kind kind = pnode->info().kind();
995 switch (kind) {
996 case ShufflePackNode::SpecificInfo::Kind::kS256Load32Transform:
997 case ShufflePackNode::SpecificInfo::Kind::kS256Load64Transform: {
998 const bool is_32 =
999 kind == ShufflePackNode::SpecificInfo::Kind::kS256Load32Transform;
1000
1001 const OpIndex load_index =
1002 op.input(pnode->info().splat_index() >> (is_32 ? 2 : 1));
1003 const LoadOp& load =
1004 __ input_graph().Get(load_index).template Cast<LoadOp>();
1005
1006 const int bytes_per_lane = is_32 ? 4 : 8;
1007 const int splat_index = pnode->info().splat_index() * bytes_per_lane;
1008 const int offset = splat_index + load.offset;
1009
1010 V<WordPtr> base = __ WordPtrAdd(__ MapToNewGraph(load.base()),
1011 __ IntPtrConstant(offset));
1012
1013 V<WordPtr> index = load.index().has_value()
1014 ? __ MapToNewGraph(load.index().value())
1015 : __ IntPtrConstant(0);
1016
1017 const Simd256LoadTransformOp::TransformKind transform_kind =
1018 is_32 ? Simd256LoadTransformOp::TransformKind::k32Splat
1019 : Simd256LoadTransformOp::TransformKind::k64Splat;
1020 og_index = __ Simd256LoadTransform(base, index, load.kind,
1021 transform_kind, 0);
1022 pnode->SetRevectorizedNode(og_index);
1023 break;
1024 }
1025 case ShufflePackNode::SpecificInfo::Kind::kS256Load8x8U: {
1026 const Simd128ShuffleOp& op0 = __ input_graph()
1027 .Get(pnode -> nodes()[0])
1028 .template Cast<Simd128ShuffleOp>();
1029
1030 V<Simd128> load_transform_idx =
1031 __ input_graph()
1032 .Get(op0.left())
1033 .template Is<Simd128LoadTransformOp>()
1034 ? op0.left()
1035 : op0.right();
1036 const Simd128LoadTransformOp& load_transform =
1037 __ input_graph()
1038 .Get(load_transform_idx)
1039 .template Cast<Simd128LoadTransformOp>();
1040 DCHECK_EQ(load_transform.transform_kind,
1041 Simd128LoadTransformOp::TransformKind::k64Zero);
1042 V<WordPtr> base = __ MapToNewGraph(load_transform.base());
1043 V<WordPtr> index = __ MapToNewGraph(load_transform.index());
1044 og_index = __ Simd256LoadTransform(
1045 base, index, load_transform.load_kind,
1046 Simd256LoadTransformOp::TransformKind::k8x8U,
1047 load_transform.offset);
1048 pnode->SetRevectorizedNode(og_index);
1049 break;
1050 }
1051#ifdef V8_TARGET_ARCH_X64
1052 case ShufflePackNode::SpecificInfo::Kind::kShufd: {
1053 V<Simd256> og_left = analyzer_.GetReducedInput(pnode, 0);
1054 DCHECK_EQ(og_left, analyzer_.GetReducedInput(pnode, 1));
1055 og_index = __ Simd256Shufd(og_left, pnode->info().shufd_control());
1056 pnode->SetRevectorizedNode(og_index);
1057 break;
1058 }
1059 case ShufflePackNode::SpecificInfo::Kind::kShufps: {
1060 V<Simd256> og_left = analyzer_.GetReducedInput(pnode, 0);
1061 V<Simd256> og_right = analyzer_.GetReducedInput(pnode, 1);
1062 og_index = __ Simd256Shufps(og_left, og_right,
1063 pnode->info().shufps_control());
1064 pnode->SetRevectorizedNode(og_index);
1065 break;
1066 }
1067 case ShufflePackNode::SpecificInfo::Kind::kS32x8UnpackLow: {
1068 V<Simd256> og_left = analyzer_.GetReducedInput(pnode, 0);
1069 V<Simd256> og_right = analyzer_.GetReducedInput(pnode, 1);
1070 og_index = __ Simd256Unpack(og_left, og_right,
1071 Simd256UnpackOp::Kind::k32x8Low);
1072 pnode->SetRevectorizedNode(og_index);
1073 break;
1074 }
1075 case ShufflePackNode::SpecificInfo::Kind::kS32x8UnpackHigh: {
1076 V<Simd256> og_left = analyzer_.GetReducedInput(pnode, 0);
1077 V<Simd256> og_right = analyzer_.GetReducedInput(pnode, 1);
1078 og_index = __ Simd256Unpack(og_left, og_right,
1079 Simd256UnpackOp::Kind::k32x8High);
1080 pnode->SetRevectorizedNode(og_index);
1081 break;
1082 }
1083#endif // V8_TARGET_ARCH_X64
1084 default:
1085 UNREACHABLE();
1086 }
1087 }
1088 return GetExtractOpIfNeeded(pnode, ig_index, og_index);
1089 }

1101 const BundlePackNode* bundle_pnode = pnode->AsBundlePackNode();
1102 V<Simd128> base_index = __ MapToNewGraph(bundle_pnode->base());
1103 V<Simd128> i16x8_index = base_index;
1104 V<Simd256> i32x8_index;

1106 if (bundle_pnode->lane_size() == 1) {

1146 void ReduceInputsOfOp(OpIndex cur_index, OpIndex op_index) {
1147 // Reduce all the operations of op_index's input tree, which should be
1148 // bigger than the cur_index. The traversal is done in a DFS manner
1149 // to make sure all inputs are emitted before the use.
1150 const Block* current_input_block = Asm().current_input_block();
1151 std::stack<OpIndex> inputs;
1152 ZoneUnorderedSet<OpIndex> visited(Asm().phase_zone());
1153 inputs.push(op_index);
1154
1155 while (!inputs.empty()) {
1156 OpIndex idx = inputs.top();
1157 if (visited.find(idx) != visited.end()) {
1158 inputs.pop();
1159 continue;
1160 }
1161
1162 const Operation& op = __ input_graph().Get(idx);
1163 bool has_unvisited_inputs = false;
1164 for (OpIndex input : op.inputs()) {
1165 if (input > cur_index && visited.find(input) == visited.end()) {
1166 inputs.push(input);
1167 has_unvisited_inputs = true;
1168 }
1169 }
1170
1171 if (!has_unvisited_inputs) {
1172 inputs.pop();
1173 visited.insert(idx);
1174
1175 // op_index will be reduced later.
1176 if (idx == op_index) continue;
1177
1178 DCHECK(!Asm().input_graph().Get(idx).template Is<PhiOp>());
1179 Asm().template VisitOpAndUpdateMapping<false>(idx, current_input_block);
1180 }
1181 }

1184 template <typename Op, typename Continuation>
1185 void ReduceForceOrIntersectPackNode(PackNode* pnode, const OpIndex ig_index,
1186 OpIndex* og_index) {
1187 std::array<OpIndex, 2> v;
1188 DCHECK_EQ(pnode->nodes().size(), 2);
1189 // The operation order in pnode is determined by the store or reduce
1190 // seed when build the SLPTree. It is not quaranteed to align with
1191 // the visiting order in each basic block from input graph. E.g. we
1192 // can have a block including {a1, a2, b1, b2} operations, and the
1193 // SLPTree can be pnode1: (a2, a1), pnode2: (b1, b2) if a2 is input
1194 // of b1, and a1 is input of b2.
1195 for (int i = 0; i < static_cast<int>(pnode->nodes().size()); i++) {
1196 OpIndex cur_index = pnode->nodes()[i];
1197 if ((*og_index).valid() && cur_index == ig_index) {
1198 v[i] = *og_index;
1199 } else {
1200 // The current index maybe already reduced by the IntersectPackNode.
1201 v[i] = __ template MapToNewGraph<true>(cur_index);
1202 }
1203
1204 if (v[i].valid()) continue;
1205
1206 if (cur_index != ig_index) {
1207 ReduceInputsOfOp(ig_index, cur_index);
1208 }
1209 const Op& op = Asm().input_graph().Get(cur_index).template Cast<Op>();
1210 v[i] = Continuation{this}.ReduceInputGraph(cur_index, op);
1211
1212 if (cur_index == ig_index) {
1213 *og_index = v[i];
1214 } else {
1215 // We have to create the mapping as cur_index may exist in other
1216 // IntersectPackNode and reduce again.
1217 __ CreateOldToNewMapping(cur_index, v[i]);
1218 }
1219 }
1220
1221 OpIndex revec_index = __ SimdPack128To256(v[0], v[1]);
1222 pnode->SetRevectorizedNode(revec_index);
1223 }

1225 template <typename Op, typename Continuation>
1226 OpIndex ReduceInputGraphOperation(OpIndex ig_index, const Op& op) {
1227 OpIndex og_index;
1228 // Reduce ForcePackNode
1229 if (PackNode* p = analyzer_.GetPackNode(ig_index);
1230 p && p->IsForcePackNode()) {
1231 // Handle force packing nodes.
1232 ForcePackNode* pnode = p->AsForcePackNode();
1233 if (!pnode->RevectorizedNode().valid()) {
1234 switch (pnode->force_pack_type()) {
1235 case ForcePackNode::kSplat: {
1236 // The og_index maybe already reduced by the IntersectPackNode.
1237 OpIndex reduced_index = __ template MapToNewGraph<true>(ig_index);
1238 if (!reduced_index.valid()) {
1239 og_index = reduced_index =
1240 Continuation{this}.ReduceInputGraph(ig_index, op);
1241 }
1242 OpIndex revec_index =
1243 __ SimdPack128To256(reduced_index, reduced_index);
1244 pnode->SetRevectorizedNode(revec_index);
1245 break;
1246 }
1247 case ForcePackNode::kGeneral: {
1248 ReduceForceOrIntersectPackNode<Op, Continuation>(pnode, ig_index,
1249 &og_index);
1250 break;
1251 }
1252 }
1253 }
1254 }
1255
1256 // Reduce IntersectPackNode
1257 if (auto intersect_packnodes = analyzer_.GetIntersectPackNodes(ig_index)) {
1258 for (PackNode* pnode : *intersect_packnodes) {
1259 if (!(pnode->RevectorizedNode()).valid()) {
1260 ReduceForceOrIntersectPackNode<Op, Continuation>(pnode, ig_index,
1261 &og_index);
1262 }
1263 }
1264 }
1265
1266 if (og_index.valid()) {
1267 return og_index;
1268 }
1269
1270 if (__ template MapToNewGraph<true>(ig_index).valid()) {
1271 // The op is already emitted during emitting force pack right node input
1272 // trees.
1273 return OpIndex::Invalid();
1274 }
1275
1276 return Continuation{this}.ReduceInputGraph(ig_index, op);
