22// Adds RISC-V-specific methods for generating InstructionOperands.
23template <typename Adapter>
24class RiscvOperandGeneratorT final : public OperandGeneratorT<Adapter> {
25 public:
26 OPERAND_GENERATOR_T_BOILERPLATE(Adapter)
27
28 explicit RiscvOperandGeneratorT<Adapter>(
29 InstructionSelectorT<Adapter>* selector)
30 : super(selector) {}
31
32 InstructionOperand UseOperand(typename Adapter::node_t node,
33 InstructionCode opcode) {
34 if (CanBeImmediate(node, opcode)) {
35 return UseImmediate(node);
36 }
37 return UseRegister(node);
38 }
39
40 // Use the zero register if the node has the immediate value zero, otherwise
41 // assign a register.
42 InstructionOperand UseRegisterOrImmediateZero(typename Adapter::node_t node) {
43 if (this->is_constant(node)) {
44 auto constant = selector()->constant_view(node);
45 if ((IsIntegerConstant(constant) &&
46 GetIntegerConstantValue(constant) == 0) ||
47 constant.is_float_zero()) {
48 return UseImmediate(node);
49 }
50 }
51 return UseRegister(node);
52 }
53
54 bool IsIntegerConstant(typename Adapter::node_t node) const {
55 return selector()->is_integer_constant(node);
56 }
57
58 int64_t GetIntegerConstantValue(Node* node);
59
60 bool IsIntegerConstant(typename Adapter::ConstantView constant) {
61 return constant.is_int32() || constant.is_int64();
62 }
63
64 int64_t GetIntegerConstantValue(typename Adapter::ConstantView constant) {
65 if (constant.is_int32()) {
66 return constant.int32_value();
67 }
68 DCHECK(constant.is_int64());
69 return constant.int64_value();
70 }
71
72 std::optional<int64_t> GetOptionalIntegerConstant(
73 InstructionSelectorT<TurboshaftAdapter>* selector,
74 turboshaft::OpIndex operation) {
75 if (!this->is_constant(operation)) return {};
76 auto constant = selector->constant_view(operation);
77 if (!this->IsIntegerConstant(constant)) return {};
78 return this->GetIntegerConstantValue(constant);
79 }
80
81 bool IsFloatConstant(Node* node) {
82 return (node->opcode() == IrOpcode::kFloat32Constant) ||
83 (node->opcode() == IrOpcode::kFloat64Constant);
84 }
85
86 double GetFloatConstantValue(Node* node) {
87 if (node->opcode() == IrOpcode::kFloat32Constant) {
88 return OpParameter<float>(node->op());
89 }
90 DCHECK_EQ(IrOpcode::kFloat64Constant, node->opcode());
91 return OpParameter<double>(node->op());
92 }
93 bool CanBeZero(typename Adapter::node_t node) {
94 if (this->is_constant(node)) {
95 auto constant = selector()->constant_view(node);
96 if ((IsIntegerConstant(constant) &&
97 GetIntegerConstantValue(constant) == 0) ||
98 constant.is_float_zero()) {
99 return true;
100 }
101 }
102 return false;
103 }
104
105 bool CanBeImmediate(node_t node, InstructionCode mode) {
106 if (!this->is_constant(node)) return false;
107 auto constant = this->constant_view(node);
108 if (constant.is_compressed_heap_object()) {
109 if (!COMPRESS_POINTERS_BOOL) return false;
110 // For builtin code we need static roots
111 if (selector()->isolate()->bootstrapper() && !V8_STATIC_ROOTS_BOOL) {
112 return false;
113 }
114 const RootsTable& roots_table = selector()->isolate()->roots_table();
115 RootIndex root_index;
116 Handle<HeapObject> value = constant.heap_object_value();
117 if (roots_table.IsRootHandle(value, &root_index)) {
118 if (!RootsTable::IsReadOnly(root_index)) return false;
119 return CanBeImmediate(MacroAssemblerBase::ReadOnlyRootPtr(
120 root_index, selector()->isolate()),
121 mode);
122 }
123 return false;
124 }
125
126 return IsIntegerConstant(constant) &&
127 CanBeImmediate(GetIntegerConstantValue(constant), mode);
128 }
129
130 bool CanBeImmediate(int64_t value, InstructionCode opcode);
131
132 private:
133 bool ImmediateFitsAddrMode1Instruction(int32_t imm) const {
134 TRACE("UNIMPLEMENTED instr_sel: %s at line %d\n", __FUNCTION__, __LINE__);
135 return false;
136 }
137};

253template <typename Adapter>
254bool TryMatchImmediate(InstructionSelectorT<Adapter>* selector,
255 InstructionCode* opcode_return,
256 typename Adapter::node_t node,
257 size_t* input_count_return, InstructionOperand* inputs) {
258 RiscvOperandGeneratorT<Adapter> g(selector);
259 if (g.CanBeImmediate(node, *opcode_return)) {
260 *opcode_return |= AddressingModeField::encode(kMode_MRI);
261 inputs[0] = g.UseImmediate(node);
262 *input_count_return = 1;
263 return true;
264 }
265 return false;
266}

268// Shared routine for multiple binary operations.
269template <typename Adapter, typename Matcher>
270static void VisitBinop(InstructionSelectorT<TurboshaftAdapter>* selector,
271 typename TurboshaftAdapter::node_t node,
272 InstructionCode opcode, bool has_reverse_opcode,
273 InstructionCode reverse_opcode,
274 FlagsContinuationT<TurboshaftAdapter>* cont) {
275 using namespace turboshaft; // NOLINT(build/namespaces)
276 RiscvOperandGeneratorT<TurboshaftAdapter> g(selector);
277 InstructionOperand inputs[2];
278 size_t input_count = 0;
279 InstructionOperand outputs[1];
280 size_t output_count = 0;
281
282 const Operation& binop = selector->Get(node);
283 OpIndex left_node = binop.input(0);
284 OpIndex right_node = binop.input(1);
285
286 if (TryMatchImmediate(selector, &opcode, right_node, &input_count,
287 &inputs[1])) {
288 inputs[0] = g.UseRegisterOrImmediateZero(left_node);
289 input_count++;
290 } else if (has_reverse_opcode &&
291 TryMatchImmediate(selector, &reverse_opcode, left_node,
292 &input_count, &inputs[1])) {
293 inputs[0] = g.UseRegisterOrImmediateZero(right_node);
294 opcode = reverse_opcode;
295 input_count++;
296 } else {
297 inputs[input_count++] = g.UseRegister(left_node);
298 inputs[input_count++] = g.UseOperand(right_node, opcode);
299 }
300
301 if (cont->IsDeoptimize()) {
302 // If we can deoptimize as a result of the binop, we need to make sure that
303 // the deopt inputs are not overwritten by the binop result. One way
304 // to achieve that is to declare the output register as same-as-first.
305 outputs[output_count++] = g.DefineSameAsFirst(node);
306 } else {
307 outputs[output_count++] = g.DefineAsRegister(node);
308 }
309
310 DCHECK_NE(0u, input_count);
311 DCHECK_EQ(1u, output_count);
312 DCHECK_GE(arraysize(inputs), input_count);
313 DCHECK_GE(arraysize(outputs), output_count);
314
315 selector->EmitWithContinuation(opcode, output_count, outputs, input_count,
316 inputs, cont);
317}

329template <typename Adapter, typename Matcher>
330static void VisitBinop(InstructionSelectorT<Adapter>* selector, Node* node,
331 InstructionCode opcode, bool has_reverse_opcode,
332 InstructionCode reverse_opcode,
333 FlagsContinuationT<Adapter>* cont) {
334 RiscvOperandGeneratorT<Adapter> g(selector);
335 Int32BinopMatcher m(node);
336 InstructionOperand inputs[2];
337 size_t input_count = 0;
338 InstructionOperand outputs[1];
339 size_t output_count = 0;
340
341 if (TryMatchImmediate(selector, &opcode, m.right().node(), &input_count,
342 &inputs[1])) {
343 inputs[0] = g.UseRegisterOrImmediateZero(m.left().node());
344 input_count++;
345 } else if (has_reverse_opcode &&
346 TryMatchImmediate(selector, &reverse_opcode, m.left().node(),
347 &input_count, &inputs[1])) {
348 inputs[0] = g.UseRegisterOrImmediateZero(m.right().node());
349 opcode = reverse_opcode;
350 input_count++;
351 } else {
352 inputs[input_count++] = g.UseRegister(m.left().node());
353 inputs[input_count++] = g.UseOperand(m.right().node(), opcode);
354 }
355
356 if (cont->IsDeoptimize()) {
357 // If we can deoptimize as a result of the binop, we need to make sure that
358 // the deopt inputs are not overwritten by the binop result. One way
359 // to achieve that is to declare the output register as same-as-first.
360 outputs[output_count++] = g.DefineSameAsFirst(node);
361 } else {
362 outputs[output_count++] = g.DefineAsRegister(node);
363 }
364
365 DCHECK_NE(0u, input_count);
366 DCHECK_EQ(1u, output_count);
367 DCHECK_GE(arraysize(inputs), input_count);
368 DCHECK_GE(arraysize(outputs), output_count);
369
370 selector->EmitWithContinuation(opcode, output_count, outputs, input_count,
371 inputs, cont);
372}

613template <typename Adapter>
614static Instruction* VisitCompare(InstructionSelectorT<Adapter>* selector,
615 InstructionCode opcode,
616 InstructionOperand left,
617 InstructionOperand right,
618 FlagsContinuationT<Adapter>* cont) {
619#ifdef V8_COMPRESS_POINTERS

621 RiscvOperandGeneratorT<Adapter> g(selector);
622 InstructionOperand inputs[] = {left, right};

624 InstructionOperand temps[1] = {g.TempRegister()};
625 return selector->EmitWithContinuation(opcode, 0, nullptr,
626 arraysize(inputs), inputs,
627 arraysize(temps), temps, cont);

630 return selector->EmitWithContinuation(opcode, 0, nullptr,
631 arraysize(inputs), inputs,
632 arraysize(temps), temps, cont);
633 }
634 }
635#endif
636 return selector->EmitWithContinuation(opcode, left, right, cont);

647// Shared routine for multiple float32 compare operations.
648template <typename Adapter>
649void VisitFloat32Compare(InstructionSelectorT<Adapter>* selector,
650 typename Adapter::node_t node,
651 FlagsContinuationT<Adapter>* cont) {
652 RiscvOperandGeneratorT<Adapter> g(selector);
653 if constexpr (Adapter::IsTurboshaft) {
654 using namespace turboshaft; // NOLINT(build/namespaces)
655 const ComparisonOp& op = selector->Get(node).template Cast<ComparisonOp>();
656 OpIndex left = op.left();
657 OpIndex right = op.right();
658 if (selector->MatchZero(right)) {
659 VisitCompare(selector, kRiscvCmpS, g.UseRegister(left),
660 g.UseImmediate(right), cont);
661 } else if (selector->MatchZero(left)) {
662 cont->Commute();
663 VisitCompare(selector, kRiscvCmpS, g.UseRegister(right),
664 g.UseImmediate(left), cont);
665 } else {
666 VisitCompare(selector, kRiscvCmpS, g.UseRegister(left),
667 g.UseRegister(right), cont);
668 }
669 } else {
670 Float32BinopMatcher m(node);
671 InstructionOperand lhs, rhs;
672
673 lhs = m.left().IsZero() ? g.UseImmediate(m.left().node())
674 : g.UseRegister(m.left().node());
675 rhs = m.right().IsZero() ? g.UseImmediate(m.right().node())
676 : g.UseRegister(m.right().node());
677 VisitCompare(selector, kRiscvCmpS, lhs, rhs, cont);
678 }
679}

682template <typename Adapter>
683void VisitFloat64Compare(InstructionSelectorT<Adapter>* selector,
684 typename Adapter::node_t node,
685 FlagsContinuationT<Adapter>* cont) {
686 if constexpr (Adapter::IsTurboshaft) {

688 using namespace turboshaft; // NOLINT(build/namespaces)

696 } else if (selector->MatchZero(lhs)) {
697 VisitCompare(selector, kRiscvCmpD, g.UseImmediate(lhs),
698 g.UseRegister(rhs), cont);
699 } else {
700 VisitCompare(selector, kRiscvCmpD, g.UseRegister(lhs), g.UseRegister(rhs),
701 cont);
702 }
703 } else {
704 RiscvOperandGeneratorT<Adapter> g(selector);
705 Float64BinopMatcher m(node);
706 InstructionOperand lhs, rhs;
707
708 lhs = m.left().IsZero() ? g.UseImmediate(m.left().node())
709 : g.UseRegister(m.left().node());
710 rhs = m.right().IsZero() ? g.UseImmediate(m.right().node())
711 : g.UseRegister(m.right().node());
712 VisitCompare(selector, kRiscvCmpD, lhs, rhs, cont);

716// Shared routine for multiple word compare operations.
717template <typename Adapter>
718Instruction* VisitWordCompare(InstructionSelectorT<Adapter>* selector,
719 typename Adapter::node_t node,
720 InstructionCode opcode,
721 FlagsContinuationT<Adapter>* cont,
722 bool commutative) {
723 RiscvOperandGeneratorT<Adapter> g(selector);
724 DCHECK_EQ(selector->value_input_count(node), 2);
725 auto left = selector->input_at(node, 0);
726 auto right = selector->input_at(node, 1);
727 // If one of the two inputs is an immediate, make sure it's on the right.
728 if (!g.CanBeImmediate(right, opcode) && g.CanBeImmediate(left, opcode)) {
729 cont->Commute();
730 std::swap(left, right);
731 }
732 // Match immediates on right side of comparison.
733 if (g.CanBeImmediate(right, opcode)) {
734#if V8_TARGET_ARCH_RISCV64
735 if (opcode == kRiscvTst64 || opcode == kRiscvTst32) {
736#elif V8_TARGET_ARCH_RISCV32
737 if (opcode == kRiscvTst32) {
738#endif
739 if constexpr (Adapter::IsTurbofan) {
740 if (selector->opcode(left) ==
741 Adapter::opcode_t::kTruncateInt64ToInt32) {
742 return VisitCompare(selector, opcode,
743 g.UseRegister(selector->input_at(left, 0)),
744 g.UseImmediate(right), cont);
745 } else {
746 return VisitCompare(selector, opcode, g.UseRegister(left),
747 g.UseImmediate(right), cont);
748 }
749 } else {
750 return VisitCompare(selector, opcode, g.UseRegister(left),
751 g.UseImmediate(right), cont);
752 }
753 } else {
754 switch (cont->condition()) {
755 case kEqual:
756 case kNotEqual:
757 if (cont->IsSet()) {
758 return VisitCompare(selector, opcode, g.UseRegister(left),
759 g.UseImmediate(right), cont);
760 } else {
761 if (g.CanBeZero(right)) {
762 return VisitWordCompareZero(
763 selector, g.UseRegisterOrImmediateZero(left), cont);
764 } else {
765 return VisitCompare(selector, opcode, g.UseRegister(left),
766 g.UseRegister(right), cont);
767 }
768 }
769 break;
770 case kSignedLessThan:
771 case kSignedGreaterThanOrEqual:
772 case kUnsignedLessThan:
773 case kUnsignedGreaterThanOrEqual: {
774 if (g.CanBeZero(right)) {
775 return VisitWordCompareZero(
776 selector, g.UseRegisterOrImmediateZero(left), cont);
777 } else {
778 return VisitCompare(selector, opcode, g.UseRegister(left),
779 g.UseImmediate(right), cont);
780 }
781 } break;
782 default:
783 if (g.CanBeZero(right)) {
784 return VisitWordCompareZero(
785 selector, g.UseRegisterOrImmediateZero(left), cont);
786 } else {
787 return VisitCompare(selector, opcode, g.UseRegister(left),
788 g.UseRegister(right), cont);
789 }
790 }
791 }
792 } else {
793 return VisitCompare(selector, opcode, g.UseRegister(left),
794 g.UseRegister(right), cont);
795 }
796}

798template <typename Adapter>
799void InstructionSelectorT<Adapter>::VisitSwitch(node_t node,
800 const SwitchInfo& sw) {
801 RiscvOperandGeneratorT<Adapter> g(this);
802 InstructionOperand value_operand = g.UseRegister(this->input_at(node, 0));
803
804 // Emit either ArchTableSwitch or ArchBinarySearchSwitch.
805 if (enable_switch_jump_table_ ==
806 InstructionSelector::kEnableSwitchJumpTable) {
807 static const size_t kMaxTableSwitchValueRange = 2 << 16;
808 size_t table_space_cost = 10 + 2 * sw.value_range();
809 size_t table_time_cost = 3;
810 size_t lookup_space_cost = 2 + 2 * sw.case_count();
811 size_t lookup_time_cost = sw.case_count();
812 if (sw.case_count() > 0 &&
813 table_space_cost + 3 * table_time_cost <=
814 lookup_space_cost + 3 * lookup_time_cost &&
815 sw.min_value() > std::numeric_limits<int32_t>::min() &&
816 sw.value_range() <= kMaxTableSwitchValueRange) {
817 InstructionOperand index_operand = value_operand;
818 if (sw.min_value()) {
819 index_operand = g.TempRegister();
820 Emit(kRiscvSub32, index_operand, value_operand,
821 g.TempImmediate(sw.min_value()));
822 }
823 // Generate a table lookup.
824 return EmitTableSwitch(sw, index_operand);
825 }
826 }
827
828 // Generate a tree of conditional jumps.
829 return EmitBinarySearchSwitch(sw, value_operand);
830}

959 node_t node) {
960 RiscvOperandGeneratorT<Adapter> g(this);

964 // Skip any alignment holes in nodes.

966 DCHECK(!call_descriptor->IsCFunctionCall());

970 MarkAsFloat64(output.node);

972 MarkAsSimd128(output.node);

974 int offset = call_descriptor->GetOffsetToReturns();
975 int reverse_slot = -output.location.GetLocation() - offset;
976 Emit(kRiscvPeek, g.DefineAsRegister(output.node),
977 g.UseImmediate(reverse_slot));
978 }
979 }

1154template <typename Adapter>
1155void InstructionSelectorT<Adapter>::VisitWord32Shl(node_t node) {
1156 if constexpr (Adapter::IsTurboshaft) {
1157 // todo(RISCV): Optimize it
1158 VisitRRO(this, kRiscvShl32, node);
1159 } else {
1160 Int32BinopMatcher m(node);
1161 if (m.left().IsWord32And() && CanCover(node, m.left().node()) &&
1162 m.right().IsInRange(1, 31)) {
1163 RiscvOperandGeneratorT<Adapter> g(this);
1164 Int32BinopMatcher mleft(m.left().node());
1165 // Match Word32Shl(Word32And(x, mask), imm) to Shl where the mask is
1166 // contiguous, and the shift immediate non-zero.
1167 if (mleft.right().HasResolvedValue()) {
1168 uint32_t mask = mleft.right().ResolvedValue();
1169 uint32_t mask_width = base::bits::CountPopulation(mask);
1170 uint32_t mask_msb = base::bits::CountLeadingZeros32(mask);
1171 if ((mask_width != 0) && (mask_msb + mask_width == 32)) {
1172 uint32_t shift = m.right().ResolvedValue();
1173 DCHECK_EQ(0u, base::bits::CountTrailingZeros32(mask));
1174 DCHECK_NE(0u, shift);
1175 if ((shift + mask_width) >= 32) {
1176 // If the mask is contiguous and reaches or extends beyond the top
1177 // bit, only the shift is needed.
1178 Emit(kRiscvShl32, g.DefineAsRegister(node),
1179 g.UseRegister(mleft.left().node()),
1180 g.UseImmediate(m.right().node()));
1181 return;
1182 }
1183 }
1184 }
1185 }
1186 VisitRRO(this, kRiscvShl32, node);
1187 }

1496template <typename Adapter>
1497void InstructionSelectorT<Adapter>::VisitS128Const(node_t node) {
1498 RiscvOperandGeneratorT<Adapter> g(this);
1499 static const int kUint32Immediates = kSimd128Size / sizeof(uint32_t);
1500 uint32_t val[kUint32Immediates];
1501 if constexpr (Adapter::IsTurboshaft) {
1502 const turboshaft::Simd128ConstantOp& constant =
1503 this->Get(node).template Cast<turboshaft::Simd128ConstantOp>();
1504 memcpy(val, constant.value, kSimd128Size);
1505 } else {
1506 memcpy(val, S128ImmediateParameterOf(node->op()).data(), kSimd128Size);
1507 }
1508 // If all bytes are zeros or ones, avoid emitting code for generic constants
1509 bool all_zeros = !(val[0] || val[1] || val[2] || val[3]);
1510 bool all_ones = val[0] == UINT32_MAX && val[1] == UINT32_MAX &&
1511 val[2] == UINT32_MAX && val[3] == UINT32_MAX;
1512 InstructionOperand dst = g.DefineAsRegister(node);
1513 if (all_zeros) {
1514 Emit(kRiscvS128Zero, dst);
1515 } else if (all_ones) {
1516 Emit(kRiscvS128AllOnes, dst);
1517 } else {
1518 Emit(kRiscvS128Const, dst, g.UseImmediate(val[0]), g.UseImmediate(val[1]),
1519 g.UseImmediate(val[2]), g.UseImmediate(val[3]));
1520 }
1521}
