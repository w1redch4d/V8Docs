18// Liftoff Frames.
19//
20// slot Frame
21// +--------------------+---------------------------
22// n+4 | optional padding slot to keep the stack 16 byte aligned.
23// n+3 | parameter n |
24// ... | ... |
25// 4 | parameter 1 | or parameter 2
26// 3 | parameter 0 | or parameter 1
27// 2 | (result address) | or parameter 0
28// -----+--------------------+---------------------------
29// 1 | return addr (ra) |
30// 0 | previous frame (fp)|
31// -----+--------------------+ <-- frame ptr (fp)
32// -1 | StackFrame::WASM |
33// -2 | instance |
34// -3 | feedback vector|
35// -----+--------------------+---------------------------
36// -4 | slot 0 | ^
37// -5 | slot 1 | |
38// | | Frame slots
39// | | |
40// | | v
41// | optional padding slot to keep the stack 16 byte aligned.
42// -----+--------------------+ <-- stack ptr (sp)
43//

45inline MemOperand GetMemOp(LiftoffAssembler* assm, Register addr,
46 Register offset, uintptr_t offset_imm,
47 bool i64_offset = false, unsigned shift_amount = 0) {
48 if (offset != no_reg) {
49 if (!i64_offset) {
50 // extract bit[0:31] without sign extend
51 assm->ExtractBits(kScratchReg2, offset, 0, 32, false);
52 offset = kScratchReg2;
53 }
54 if (shift_amount != 0) {
55 assm->CalcScaledAddress(kScratchReg2, addr, offset, shift_amount);
56 } else {
57 assm->Add64(kScratchReg2, offset, addr);
58 }
59 addr = kScratchReg2;
60 }
61 if (is_int31(offset_imm)) {
62 int32_t offset_imm32 = static_cast<int32_t>(offset_imm);
63 return MemOperand(addr, offset_imm32);
64 } else {
65 assm->li(kScratchReg, Operand(offset_imm));
66 assm->Add64(kScratchReg2, addr, kScratchReg);
67 return MemOperand(kScratchReg2, 0);
68 }
69}

71inline void Load(LiftoffAssembler* assm, LiftoffRegister dst, MemOperand src,
72 ValueKind kind) {
73 switch (kind) {
74 case kI32:
75 assm->Lw(dst.gp(), src);
76 break;
77 case kI64:
78 case kRef:
79 case kRefNull:
80 case kRtt:
81 assm->Ld(dst.gp(), src);
82 break;
83 case kF32:
84 assm->LoadFloat(dst.fp(), src);
85 break;
86 case kF64:
87 assm->LoadDouble(dst.fp(), src);
88 break;
89 case kS128:{
90 assm->VU.set(kScratchReg, E8, m1);
91 Register src_reg = src.offset() == 0 ? src.rm() : kScratchReg;
92 if (src.offset() != 0) {
93 assm->AddWord(src_reg, src.rm(), src.offset());
94 }
95 assm->vl(dst.fp().toV(), src_reg, 0, E8);
96 break;
97 }
98 default:
99 UNREACHABLE();
100 }

167inline void StoreToMemory(LiftoffAssembler* assm, MemOperand dst,
168 const LiftoffAssembler::VarState& src) {
169 UseScratchRegisterScope temps(assm);
170 if (src.is_const()) {
171 Register src_reg = no_reg;
172 if (src.i32_const() == 0) {
173 src_reg = zero_reg;
174 } else {
175 src_reg = temps.Acquire();
176 assm->li(src_reg, src.i32_const());
177 }
178 assm->StoreWord(src_reg, dst);
179 } else if (src.is_reg()) {
180 switch (src.kind()) {
181 case kI32:
182 return assm->Sw(src.reg().gp(), dst);
183 case kI64:
184 case kRef:
185 case kRefNull:
186 case kRtt:
187 return assm->Sd(src.reg().gp(), dst);
188 case kF32:
189 return assm->StoreFloat(src.reg().fp(), dst);
190 case kF64:
191 return assm->StoreDouble(src.reg().fp(), dst);
192 case kS128: {
193 assm->VU.set(kScratchReg, E8, m1);
194 Register dst_reg = temps.Acquire();
195 assm->Add64(dst_reg, dst.rm(), dst.offset());
196 assm->vs(src.reg().fp().toV(), dst_reg, 0, VSew::E8);
197 return;
198 }
199 default:
200 UNREACHABLE();
201 }
202 } else {
203 DCHECK(src.is_stack());
204 Register temp = temps.Acquire();
205 switch (src.kind()) {
206 case kI32:
207 assm->Lw(temp, GetStackSlot(src.offset()));
208 assm->Sw(temp, dst);
209 return;
210 case kI64:
211 case kRef:
212 case kRefNull:
213 assm->Ld(temp, GetStackSlot(src.offset()));
214 assm->Sd(temp, dst);
215 return;
216 case kF32:
217 assm->LoadFloat(kScratchDoubleReg, GetStackSlot(src.offset()));
218 assm->StoreFloat(kScratchDoubleReg, dst);
219 return;
220 case kF64:
221 assm->LoadDouble(kScratchDoubleReg, GetStackSlot(src.offset()));
222 assm->StoreDouble(kScratchDoubleReg, dst);
223 return;
224 case kS128: {
225 assm->VU.set(kScratchReg, E8, m1);
226 Register src_reg = temp;
227 assm->Add64(src_reg, sp, src.offset());
228 assm->vl(kScratchDoubleReg.toV(), src_reg, 0, VSew::E8);
229 Register dst_reg = temp;
230 assm->Add64(dst_reg, dst.rm(), dst.offset());
231 assm->vs(kScratchDoubleReg.toV(), dst_reg, 0, VSew::E8);
232 return;
233 }
234 default:
235 UNREACHABLE();
236 }
237 }

301void LiftoffAssembler::StoreTaggedPointer(Register dst_addr,
302 Register offset_reg,
303 int32_t offset_imm, Register src,
304 LiftoffRegList pinned,
305 uint32_t* protected_store_pc,
306 SkipWriteBarrier skip_write_barrier) {
307 UseScratchRegisterScope temps(this);
308 Operand offset_op =
309 offset_reg.is_valid() ? Operand(offset_reg) : Operand(offset_imm);
310 // For the write barrier (below), we cannot have both an offset register and
311 // an immediate offset. Add them to a 32-bit offset initially, but in a 64-bit
312 // register, because that's needed in the MemOperand below.
313 if (offset_reg.is_valid() && offset_imm) {
314 Register effective_offset = temps.Acquire();
315 AddWord(effective_offset, offset_reg, Operand(offset_imm));
316 offset_op = Operand(effective_offset);
317 }
318 auto trapper = [protected_store_pc](int offset) {
319 if (protected_store_pc) *protected_store_pc = static_cast<uint32_t>(offset);
320 };
321 if (offset_op.is_reg()) {
322 AddWord(kScratchReg, dst_addr, offset_op.rm());
323 StoreTaggedField(src, MemOperand(kScratchReg, 0), trapper);
324 } else {
325 StoreTaggedField(src, MemOperand(dst_addr, offset_imm), trapper);
326 }
327 if (protected_store_pc) {
328 DCHECK(InstructionAt(*protected_store_pc)->IsStore());
329 }
330
331 if (skip_write_barrier || v8_flags.disable_write_barriers) return;
332
333 Label exit;
334 CheckPageFlag(dst_addr, MemoryChunk::kPointersFromHereAreInterestingMask,
335 kZero, &exit);
336 JumpIfSmi(src, &exit);
337 CheckPageFlag(src, MemoryChunk::kPointersToHereAreInterestingMask, eq, &exit);
338 CallRecordWriteStubSaveRegisters(dst_addr, offset_op, SaveFPRegsMode::kSave,
339 StubCallMode::kCallWasmRuntimeStub);
340 bind(&exit);
341}

343void LiftoffAssembler::Load(LiftoffRegister dst, Register src_addr,
344 Register offset_reg, uintptr_t offset_imm,
345 LoadType type, uint32_t* protected_load_pc,
346 bool is_load_mem, bool i64_offset,
347 bool needs_shift) {
348 unsigned shift_amount = needs_shift ? type.size_log_2() : 0;
349 MemOperand src_op = liftoff::GetMemOp(this, src_addr, offset_reg, offset_imm,
350 i64_offset, shift_amount);
351 Assembler::BlockPoolsScope blocked_pools_scope_(this, 4 * kInstrSize);
352 auto trapper = [protected_load_pc](int offset) {
353 if (protected_load_pc) *protected_load_pc = static_cast<uint32_t>(offset);
354 };
355 switch (type.value()) {
356 case LoadType::kI32Load8U:
357 case LoadType::kI64Load8U:
358 Lbu(dst.gp(), src_op, trapper);
359 break;
360 case LoadType::kI32Load8S:
361 case LoadType::kI64Load8S:
362 Lb(dst.gp(), src_op, trapper);
363 break;
364 case LoadType::kI32Load16U:
365 case LoadType::kI64Load16U:
366 Lhu(dst.gp(), src_op, trapper);
367 break;
368 case LoadType::kI32Load16S:
369 case LoadType::kI64Load16S:
370 Lh(dst.gp(), src_op, trapper);
371 break;
372 case LoadType::kI64Load32U:
373 Lwu(dst.gp(), src_op, trapper);
374 break;
375 case LoadType::kI32Load:
376 case LoadType::kI64Load32S:
377 Lw(dst.gp(), src_op, trapper);
378 break;
379 case LoadType::kI64Load:
380 Ld(dst.gp(), src_op, trapper);
381 break;
382 case LoadType::kF32Load:
383 LoadFloat(dst.fp(), src_op, trapper);
384 break;
385 case LoadType::kF64Load:
386 LoadDouble(dst.fp(), src_op, trapper);
387 break;
388 case LoadType::kS128Load: {
389 VU.set(kScratchReg, E8, m1);
390 Register src_reg = src_op.offset() == 0 ? src_op.rm() : kScratchReg;
391 if (src_op.offset() != 0) {
392 MacroAssembler::AddWord(src_reg, src_op.rm(), src_op.offset());
393 }
394 trapper(pc_offset());
395 vl(dst.fp().toV(), src_reg, 0, E8);
396 break;
397 }
398 case LoadType::kF32LoadF16:
399 UNIMPLEMENTED();
400 break;
401 default:
402 UNREACHABLE();
403 }
404 if (protected_load_pc) {
405 DCHECK(InstructionAt(*protected_load_pc)->IsLoad());
406 }
407
408#if defined(V8_TARGET_BIG_ENDIAN)
409 if (is_load_mem) {
410 pinned.set(src_op.rm());
411 liftoff::ChangeEndiannessLoad(this, dst, type, pinned);
412 }
413#endif
414}

417 uintptr_t offset_imm, LiftoffRegister src,
418 StoreType type, LiftoffRegList pinned,

472 }
473 default:
474 UNREACHABLE();
475 }
476 if (protected_store_pc) {
477 DCHECK(InstructionAt(*protected_store_pc)->IsStore());
478 }

484inline Register CalculateActualAddress(LiftoffAssembler* lasm,
485 UseScratchRegisterScope& temps,
486 Register addr_reg, Register offset_reg,
487 uintptr_t offset_imm) {
488 DCHECK_NE(addr_reg, no_reg);
489 if (offset_reg == no_reg && offset_imm == 0) return addr_reg;
490 Register result = temps.Acquire();
491 if (offset_reg == no_reg) {
492 __ AddWord(result, addr_reg, Operand(offset_imm));
493 } else {
494 __ AddWord(result, addr_reg, Operand(offset_reg));
495 if (offset_imm != 0) __ AddWord(result, result, Operand(offset_imm));
496 }
497 return result;

502inline void AtomicBinop(LiftoffAssembler* lasm, Register dst_addr,
503 Register offset_reg, uintptr_t offset_imm,
504 LiftoffRegister value, LiftoffRegister result,
505 StoreType type, Binop op) {
506 LiftoffRegList pinned{dst_addr, value, result};
507 if (offset_reg != no_reg) pinned.set(offset_reg);
508 Register store_result = pinned.set(__ GetUnusedRegister(kGpReg, pinned)).gp();
509
510 // Make sure that {result} is unique.
511 Register result_reg = result.gp();
512 if (result_reg == value.gp() || result_reg == dst_addr ||
513 result_reg == offset_reg) {
514 result_reg = __ GetUnusedRegister(kGpReg, pinned).gp();
515 }
516
517 UseScratchRegisterScope temps(lasm);
518 Register actual_addr = liftoff::CalculateActualAddress(
519 lasm, temps, dst_addr, offset_reg, offset_imm);
520
521 // Allocate an additional {temp} register to hold the result that should be
522 // stored to memory. Note that {temp} and {store_result} are not allowed to be
523 // the same register.
524 Register temp = temps.Acquire();
525
526 Label retry;
527 __ bind(&retry);
528 switch (type.value()) {
529 case StoreType::kI64Store8:
530 case StoreType::kI32Store8:
531 __ lbu(result_reg, actual_addr, 0);
532 __ sync();
533 break;
534 case StoreType::kI64Store16:
535 case StoreType::kI32Store16:
536 __ lhu(result_reg, actual_addr, 0);
537 __ sync();
538 break;
539 case StoreType::kI64Store32:
540 __ lr_w(true, false, result_reg, actual_addr);
541 __ ZeroExtendWord(result_reg, result_reg);
542 break;
543 case StoreType::kI32Store:
544 __ lr_w(true, false, result_reg, actual_addr);
545 break;
546 case StoreType::kI64Store:
547 __ lr_d(true, false, result_reg, actual_addr);
548 break;
549 default:
550 UNREACHABLE();
551 }
552
553 switch (op) {
554 case Binop::kAdd:
555 __ add(temp, result_reg, value.gp());
556 break;
557 case Binop::kSub:
558 __ sub(temp, result_reg, value.gp());
559 break;
560 case Binop::kAnd:
561 __ and_(temp, result_reg, value.gp());
562 break;
563 case Binop::kOr:
564 __ or_(temp, result_reg, value.gp());
565 break;
566 case Binop::kXor:
567 __ xor_(temp, result_reg, value.gp());
568 break;
569 case Binop::kExchange:
570 __ mv(temp, value.gp());
571 break;
572 }
573 switch (type.value()) {
574 case StoreType::kI64Store8:
575 case StoreType::kI32Store8:
576 __ sync();
577 __ sb(temp, actual_addr, 0);
578 __ sync();
579 __ mv(store_result, zero_reg);
580 break;
581 case StoreType::kI64Store16:
582 case StoreType::kI32Store16:
583 __ sync();
584 __ sh(temp, actual_addr, 0);
585 __ sync();
586 __ mv(store_result, zero_reg);
587 break;
588 case StoreType::kI64Store32:
589 case StoreType::kI32Store:
590 __ sc_w(false, true, store_result, actual_addr, temp);
591 break;
592 case StoreType::kI64Store:
593 __ sc_d(false, true, store_result, actual_addr, temp);
594 break;
595 default:
596 UNREACHABLE();
597 }
598
599 __ bnez(store_result, &retry);
600 if (result_reg != result.gp()) {
601 __ mv(result.gp(), result_reg);
602 }
603}

643void LiftoffAssembler::AtomicStore(Register dst_addr, Register offset_reg,
644 uintptr_t offset_imm, LiftoffRegister src,
645 StoreType type, LiftoffRegList pinned,
646 bool i64_offset) {
647 UseScratchRegisterScope temps(this);
648 Register dst_reg = liftoff::CalculateActualAddress(this, temps, dst_addr,
649 offset_reg, offset_imm);
650 switch (type.value()) {
651 case StoreType::kI64Store8:
652 case StoreType::kI32Store8:
653 sync();
654 sb(src.gp(), dst_reg, 0);
655 return;
656 case StoreType::kI64Store16:
657 case StoreType::kI32Store16:
658 sync();
659 sh(src.gp(), dst_reg, 0);
660 return;
661 case StoreType::kI64Store32:
662 case StoreType::kI32Store:
663 sync();
664 sw(src.gp(), dst_reg, 0);
665 return;

725 store_conditional) \
726 do { \
727 Label compareExchange; \
728 Label exit; \
729 sync(); \
730 bind(&compareExchange); \
731 load_linked(result.gp(), MemOperand(temp0, 0)); \
732 BranchShort(&exit, ne, expected.gp(), Operand(result.gp())); \
733 mv(temp2, new_value.gp()); \
734 store_conditional(temp2, MemOperand(temp0, 0)); \
735 BranchShort(&compareExchange, ne, temp2, Operand(zero_reg)); \
736 bind(&exit); \
737 sync(); \
738 } while (0)

740#define ASSEMBLE_ATOMIC_COMPARE_EXCHANGE_INTEGER_EXT( \
741 load_linked, store_conditional, size, aligned) \
742 do { \
743 Label compareExchange; \
744 Label exit; \
745 andi(temp1, temp0, aligned); \
746 Sub64(temp0, temp0, Operand(temp1)); \
747 Sll32(temp1, temp1, 3); \
748 sync(); \
749 bind(&compareExchange); \
750 load_linked(temp2, MemOperand(temp0, 0)); \
751 ExtractBits(result.gp(), temp2, temp1, size, false); \
752 ExtractBits(temp2, expected.gp(), zero_reg, size, false); \
753 BranchShort(&exit, ne, temp2, Operand(result.gp())); \
754 InsertBits(temp2, new_value.gp(), temp1, size); \
755 store_conditional(temp2, MemOperand(temp0, 0)); \
756 BranchShort(&compareExchange, ne, temp2, Operand(zero_reg)); \
757 bind(&exit); \
758 sync(); \
759 } while (0)

761void LiftoffAssembler::AtomicCompareExchange(
762 Register dst_addr, Register offset_reg, uintptr_t offset_imm,
763 LiftoffRegister expected, LiftoffRegister new_value, LiftoffRegister result,
764 StoreType type, bool i64_offset) {
765 LiftoffRegList pinned{dst_addr, expected, new_value, result};
766 if (offset_reg != no_reg) pinned.set(offset_reg);
767
768 Register temp0 = pinned.set(GetUnusedRegister(kGpReg, pinned)).gp();
769 Register temp1 = pinned.set(GetUnusedRegister(kGpReg, pinned)).gp();
770 Register temp2 = pinned.set(GetUnusedRegister(kGpReg, pinned)).gp();
771 MemOperand dst_op = liftoff::GetMemOp(this, dst_addr, offset_reg, offset_imm);
772 Add64(temp0, dst_op.rm(), dst_op.offset());
773 switch (type.value()) {
774 case StoreType::kI64Store8:
775 ASSEMBLE_ATOMIC_COMPARE_EXCHANGE_INTEGER_EXT(Ll, Sc, 8, 7);
776 break;
777 case StoreType::kI32Store8:
778 ASSEMBLE_ATOMIC_COMPARE_EXCHANGE_INTEGER_EXT(Ll, Sc, 8, 3);
779 break;
780 case StoreType::kI64Store16:
781 ASSEMBLE_ATOMIC_COMPARE_EXCHANGE_INTEGER_EXT(Ll, Sc, 16, 7);
782 break;
783 case StoreType::kI32Store16:
784 ASSEMBLE_ATOMIC_COMPARE_EXCHANGE_INTEGER_EXT(Ll, Sc, 16, 3);
785 break;
786 case StoreType::kI64Store32:
787 ASSEMBLE_ATOMIC_COMPARE_EXCHANGE_INTEGER_EXT(Lld, Scd, 32, 7);
788 break;
789 case StoreType::kI32Store:
790 ASSEMBLE_ATOMIC_COMPARE_EXCHANGE_INTEGER(Ll, Sc);
791 break;
792 case StoreType::kI64Store:
793 ASSEMBLE_ATOMIC_COMPARE_EXCHANGE_INTEGER(Lld, Scd);
794 break;
795 default:
796 UNREACHABLE();
797 }
798}

824void LiftoffAssembler::MoveStackValue(uint32_t dst_offset, uint32_t src_offset,
825 ValueKind kind) {
826 DCHECK_NE(dst_offset, src_offset);
827
828 MemOperand src = liftoff::GetStackSlot(src_offset);
829 MemOperand dst = liftoff::GetStackSlot(dst_offset);
830 switch (kind) {
831 case kI32:
832 Lw(kScratchReg, src);
833 Sw(kScratchReg, dst);
834 break;
835 case kI64:
836 case kRef:
837 case kRefNull:
838 case kRtt:
839 Ld(kScratchReg, src);
840 Sd(kScratchReg, dst);
841 break;
842 case kF32:
843 LoadFloat(kScratchDoubleReg, src);
844 StoreFloat(kScratchDoubleReg, dst);
845 break;
846 case kF64:
847 MacroAssembler::LoadDouble(kScratchDoubleReg, src);
848 MacroAssembler::StoreDouble(kScratchDoubleReg, dst);
849 break;
850 case kS128: {
851 VU.set(kScratchReg, E8, m1);
852 Register src_reg = src.offset() == 0 ? src.rm() : kScratchReg;
853 if (src.offset() != 0) {
854 MacroAssembler::Add64(src_reg, src.rm(), src.offset());
855 }
856 vl(kSimd128ScratchReg, src_reg, 0, E8);
857 Register dst_reg = dst.offset() == 0 ? dst.rm() : kScratchReg;
858 if (dst.offset() != 0) {
859 Add64(kScratchReg, dst.rm(), dst.offset());
860 }
861 vs(kSimd128ScratchReg, dst_reg, 0, VSew::E8);
862 break;
863 }
864 case kVoid:
865 case kI8:
866 case kI16:
867 case kTop:
868 case kBottom:
869 case kF16:
870 UNREACHABLE();
871 }
872}

986void LiftoffAssembler::FillStackSlotsWithZero(int start, int size) {
987 DCHECK_LT(0, size);
988 RecordUsedSpillOffset(start + size);
989
990 if (size <= 12 * kStackSlotSize) {
991 // Special straight-line code for up to 12 slots. Generates one
992 // instruction per slot (<= 12 instructions total).
993 uint32_t remainder = size;
994 for (; remainder >= kStackSlotSize; remainder -= kStackSlotSize) {
995 Sd(zero_reg, liftoff::GetStackSlot(start + remainder));
996 }
997 DCHECK(remainder == 4 || remainder == 0);
998 if (remainder) {
999 Sw(zero_reg, liftoff::GetStackSlot(start + remainder));
1000 }
1001 } else {
1002 // General case for bigger counts (12 instructions).
1003 // Use a0 for start address (inclusive), a1 for end address (exclusive).
1004 Push(a1, a0);
1005 Add64(a0, fp, Operand(-start - size));
1006 Add64(a1, fp, Operand(-start));
1007
1008 Label loop;
1009 bind(&loop);
1010 Sd(zero_reg, MemOperand(a0));
1011 addi(a0, a0, kSystemPointerSize);
1012 BranchShort(&loop, ne, a0, Operand(a1));
1013
1014 Pop(a1, a0);
1015 }

1293bool LiftoffAssembler::emit_type_conversion(WasmOpcode opcode,
1294 LiftoffRegister dst,
1295 LiftoffRegister src, Label* trap) {
1296 switch (opcode) {
1297 case kExprI32ConvertI64:
1298 // According to WebAssembly spec, if I64 value does not fit the range of
1299 // I32, the value is undefined. Therefore, We use sign extension to
1300 // implement I64 to I32 truncation
1301 MacroAssembler::SignExtendWord(dst.gp(), src.gp());
1302 return true;
1303 case kExprI32SConvertF32:
1304 case kExprI32UConvertF32:
1305 case kExprI32SConvertF64:
1306 case kExprI32UConvertF64:
1307 case kExprI64SConvertF32:
1308 case kExprI64UConvertF32:
1309 case kExprI64SConvertF64:
1310 case kExprI64UConvertF64:
1311 case kExprF32ConvertF64: {
1312 // real conversion, if src is out-of-bound of target integer types,
1313 // kScratchReg is set to 0
1314 switch (opcode) {
1315 case kExprI32SConvertF32:
1316 Trunc_w_s(dst.gp(), src.fp(), kScratchReg);
1317 break;
1318 case kExprI32UConvertF32:
1319 Trunc_uw_s(dst.gp(), src.fp(), kScratchReg);
1320 break;
1321 case kExprI32SConvertF64:
1322 Trunc_w_d(dst.gp(), src.fp(), kScratchReg);
1323 break;
1324 case kExprI32UConvertF64:
1325 Trunc_uw_d(dst.gp(), src.fp(), kScratchReg);
1326 break;
1327 case kExprI64SConvertF32:
1328 Trunc_l_s(dst.gp(), src.fp(), kScratchReg);
1329 break;
1330 case kExprI64UConvertF32:
1331 Trunc_ul_s(dst.gp(), src.fp(), kScratchReg);
1332 break;
1333 case kExprI64SConvertF64:
1334 Trunc_l_d(dst.gp(), src.fp(), kScratchReg);
1335 break;
1336 case kExprI64UConvertF64:
1337 Trunc_ul_d(dst.gp(), src.fp(), kScratchReg);
1338 break;
1339 case kExprF32ConvertF64:
1340 fcvt_s_d(dst.fp(), src.fp());
1341 break;
1342 default:
1343 UNREACHABLE();
1344 }
1345
1346 // Checking if trap.
1347 if (trap != nullptr) {
1348 MacroAssembler::Branch(trap, eq, kScratchReg, Operand(zero_reg));
1349 }
1350
1351 return true;
1352 }
1353 case kExprI32ReinterpretF32:
1354 MacroAssembler::ExtractLowWordFromF64(dst.gp(), src.fp());
1355 return true;
1356 case kExprI64SConvertI32:
1357 MacroAssembler::SignExtendWord(dst.gp(), src.gp());
1358 return true;
1359 case kExprI64UConvertI32:
1360 MacroAssembler::ZeroExtendWord(dst.gp(), src.gp());
1361 return true;
1362 case kExprI64ReinterpretF64:
1363 fmv_x_d(dst.gp(), src.fp());
1364 return true;
1365 case kExprF32SConvertI32: {
1366 MacroAssembler::Cvt_s_w(dst.fp(), src.gp());
1367 return true;
1368 }
1369 case kExprF32UConvertI32:
1370 MacroAssembler::Cvt_s_uw(dst.fp(), src.gp());
1371 return true;
1372 case kExprF32ReinterpretI32:
1373 fmv_w_x(dst.fp(), src.gp());
1374 return true;
1375 case kExprF64SConvertI32: {
1376 MacroAssembler::Cvt_d_w(dst.fp(), src.gp());
1377 return true;
1378 }
1379 case kExprF64UConvertI32:
1380 MacroAssembler::Cvt_d_uw(dst.fp(), src.gp());
1381 return true;
1382 case kExprF64ConvertF32:
1383 fcvt_d_s(dst.fp(), src.fp());
1384 return true;
1385 case kExprF64ReinterpretI64:
1386 fmv_d_x(dst.fp(), src.gp());
1387 return true;
1388 case kExprI32SConvertSatF32: {
1389 fcvt_w_s(dst.gp(), src.fp(), RTZ);
1390 Clear_if_nan_s(dst.gp(), src.fp());
1391 return true;
1392 }
1393 case kExprI32UConvertSatF32: {
1394 fcvt_wu_s(dst.gp(), src.fp(), RTZ);
1395 Clear_if_nan_s(dst.gp(), src.fp());
1396 return true;
1397 }
1398 case kExprI32SConvertSatF64: {
1399 fcvt_w_d(dst.gp(), src.fp(), RTZ);
1400 Clear_if_nan_d(dst.gp(), src.fp());
1401 return true;
1402 }
1403 case kExprI32UConvertSatF64: {
1404 fcvt_wu_d(dst.gp(), src.fp(), RTZ);
1405 Clear_if_nan_d(dst.gp(), src.fp());
1406 return true;
1407 }
1408 case kExprI64SConvertSatF32: {
1409 fcvt_l_s(dst.gp(), src.fp(), RTZ);
1410 Clear_if_nan_s(dst.gp(), src.fp());
1411 return true;
1412 }
1413 case kExprI64UConvertSatF32: {
1414 fcvt_lu_s(dst.gp(), src.fp(), RTZ);
1415 Clear_if_nan_s(dst.gp(), src.fp());
1416 return true;
1417 }
1418 case kExprI64SConvertSatF64: {
1419 fcvt_l_d(dst.gp(), src.fp(), RTZ);
1420 Clear_if_nan_d(dst.gp(), src.fp());
1421 return true;
1422 }
1423 case kExprI64UConvertSatF64: {
1424 fcvt_lu_d(dst.gp(), src.fp(), RTZ);
1425 Clear_if_nan_d(dst.gp(), src.fp());
1426 return true;
1427 }
1428 default:
1429 return false;
1430 }
1431}

1479 Register rhs,
1480 const FreezeCacheState& frozen) {
1481 if (rhs == no_reg) {

1486 MacroAssembler::Branch(label, cond, scratch0, Operand(zero_reg));

1496 (kind == kRefNull));
1497 MacroAssembler::CompareTaggedAndBranch(label, cond, lhs, Operand(rhs));
1498 }
1499 }

1556void LiftoffAssembler::LoadTransform(LiftoffRegister dst, Register src_addr,
1557 Register offset_reg, uintptr_t offset_imm,
1558 LoadType type,
1559 LoadTransformationKind transform,
1560 uint32_t* protected_load_pc) {
1561 UseScratchRegisterScope temps(this);
1562 Register scratch = temps.Acquire();
1563 MemOperand src_op = liftoff::GetMemOp(this, src_addr, offset_reg, offset_imm);
1564 VRegister dst_v = dst.fp().toV();
1565 auto trapper = [protected_load_pc](int offset) {
1566 if (protected_load_pc) *protected_load_pc = static_cast<uint32_t>(offset);
1567 };
1568 MachineType memtype = type.mem_type();
1569 if (transform == LoadTransformationKind::kExtend) {
1570 Ld(scratch, src_op, trapper);
1571 if (memtype == MachineType::Int8()) {
1572 VU.set(kScratchReg, E64, m1);
1573 vmv_vx(kSimd128ScratchReg, scratch);
1574 VU.set(kScratchReg, E16, m1);
1575 vsext_vf2(dst_v, kSimd128ScratchReg);
1576 } else if (memtype == MachineType::Uint8()) {
1577 VU.set(kScratchReg, E64, m1);
1578 vmv_vx(kSimd128ScratchReg, scratch);
1579 VU.set(kScratchReg, E16, m1);
1580 vzext_vf2(dst_v, kSimd128ScratchReg);
1581 } else if (memtype == MachineType::Int16()) {
1582 VU.set(kScratchReg, E64, m1);
1583 vmv_vx(kSimd128ScratchReg, scratch);
1584 VU.set(kScratchReg, E32, m1);
1585 vsext_vf2(dst_v, kSimd128ScratchReg);
1586 } else if (memtype == MachineType::Uint16()) {
1587 VU.set(kScratchReg, E64, m1);
1588 vmv_vx(kSimd128ScratchReg, scratch);
1589 VU.set(kScratchReg, E32, m1);
1590 vzext_vf2(dst_v, kSimd128ScratchReg);
1591 } else if (memtype == MachineType::Int32()) {
1592 VU.set(kScratchReg, E64, m1);
1593 vmv_vx(kSimd128ScratchReg, scratch);
1594 vsext_vf2(dst_v, kSimd128ScratchReg);
1595 } else if (memtype == MachineType::Uint32()) {
1596 VU.set(kScratchReg, E64, m1);
1597 vmv_vx(kSimd128ScratchReg, scratch);
1598 vzext_vf2(dst_v, kSimd128ScratchReg);
1599 }
1600 } else if (transform == LoadTransformationKind::kZeroExtend) {
1601 vxor_vv(dst_v, dst_v, dst_v);
1602 if (memtype == MachineType::Int32()) {
1603 VU.set(kScratchReg, E32, m1);
1604 Lwu(scratch, src_op, trapper);
1605 vmv_sx(dst_v, scratch);
1606 } else {
1607 DCHECK_EQ(MachineType::Int64(), memtype);
1608 VU.set(kScratchReg, E64, m1);
1609 Ld(scratch, src_op, trapper);
1610 vmv_sx(dst_v, scratch);
1611 }
1612 } else {
1613 DCHECK_EQ(LoadTransformationKind::kSplat, transform);
1614 if (memtype == MachineType::Int8()) {
1615 VU.set(kScratchReg, E8, m1);
1616 Lb(scratch, src_op, trapper);
1617 vmv_vx(dst_v, scratch);
1618 } else if (memtype == MachineType::Int16()) {
1619 VU.set(kScratchReg, E16, m1);
1620 Lh(scratch, src_op, trapper);
1621 vmv_vx(dst_v, scratch);
1622 } else if (memtype == MachineType::Int32()) {
1623 VU.set(kScratchReg, E32, m1);
1624 Lw(scratch, src_op, trapper);
1625 vmv_vx(dst_v, scratch);
1626 } else if (memtype == MachineType::Int64()) {
1627 VU.set(kScratchReg, E64, m1);
1628 Ld(scratch, src_op, trapper);
1629 vmv_vx(dst_v, scratch);
1630 }
1631 }
1632 if (protected_load_pc) {
1633 DCHECK(InstructionAt(*protected_load_pc)->IsLoad());
1634 }
1635}

1637void LiftoffAssembler::LoadLane(LiftoffRegister dst, LiftoffRegister src,
1638 Register addr, Register offset_reg,
1639 uintptr_t offset_imm, LoadType type,
1640 uint8_t laneidx, uint32_t* protected_load_pc,
1641 bool i64_offset) {
1642 MemOperand src_op =
1643 liftoff::GetMemOp(this, addr, offset_reg, offset_imm, i64_offset);
1644 MachineType mem_type = type.mem_type();
1645 UseScratchRegisterScope temps(this);
1646 Register scratch = temps.Acquire();
1647 auto trapper = [protected_load_pc](int offset) {
1648 if (protected_load_pc) *protected_load_pc = static_cast<uint32_t>(offset);
1649 };
1650 if (mem_type == MachineType::Int8()) {
1651 Lbu(scratch, src_op, trapper);
1652 VU.set(kScratchReg, E64, m1);
1653 li(kScratchReg, 0x1 << laneidx);
1654 vmv_sx(v0, kScratchReg);
1655 VU.set(kScratchReg, E8, m1);

1658 Lhu(scratch, src_op, trapper);
1659 VU.set(kScratchReg, E16, m1);
1660 li(kScratchReg, 0x1 << laneidx);
1661 vmv_sx(v0, kScratchReg);
1662 vmerge_vx(dst.fp().toV(), scratch, dst.fp().toV());
1663 } else if (mem_type == MachineType::Int32()) {
1664 Lwu(scratch, src_op, trapper);
1665 VU.set(kScratchReg, E32, m1);
1666 li(kScratchReg, 0x1 << laneidx);
1667 vmv_sx(v0, kScratchReg);
1668 vmerge_vx(dst.fp().toV(), scratch, dst.fp().toV());
1669 } else if (mem_type == MachineType::Int64()) {
1670 Ld(scratch, src_op, trapper);
1671 VU.set(kScratchReg, E64, m1);
1672 li(kScratchReg, 0x1 << laneidx);
1673 vmv_sx(v0, kScratchReg);
1674 vmerge_vx(dst.fp().toV(), scratch, dst.fp().toV());
1675 } else {
1676 UNREACHABLE();
1677 }
1678 if (protected_load_pc) {
1679 DCHECK(InstructionAt(*protected_load_pc)->IsLoad());

1827void LiftoffAssembler::CallCWithStackBuffer(
1828 const std::initializer_list<VarState> args, const LiftoffRegister* rets,
1829 ValueKind return_kind, ValueKind out_argument_kind, int stack_bytes,
1830 ExternalReference ext_ref) {
1831 AddWord(sp, sp, Operand(-stack_bytes));
1832
1833 int arg_offset = 0;
1834 for (const VarState& arg : args) {
1835 liftoff::StoreToMemory(this, MemOperand{sp, arg_offset}, arg);
1836 arg_offset += value_kind_size(arg.kind());
1837 }
1838 DCHECK_LE(arg_offset, stack_bytes);
1839
1840 // Pass a pointer to the buffer with the arguments to the C function.
1841 // On RISC-V, the first argument is passed in {a0}.
1842 constexpr Register kFirstArgReg = a0;
1843 mv(kFirstArgReg, sp);
1844
1845 // Now call the C function.
1846 constexpr int kNumCCallArgs = 1;
1847 PrepareCallCFunction(kNumCCallArgs, kScratchReg);
1848 CallCFunction(ext_ref, kNumCCallArgs);
1849
1850 // Move return value to the right register.
1851 const LiftoffRegister* next_result_reg = rets;
1852 if (return_kind != kVoid) {
1853 constexpr Register kReturnReg = a0;
1854 if (kReturnReg != next_result_reg->gp()) {
1855 Move(*next_result_reg, LiftoffRegister(kReturnReg), return_kind);
1856 }
1857 ++next_result_reg;
1858 }
1859
1860 // Load potential output value from the buffer on the stack.
1861 if (out_argument_kind != kVoid) {
1862 liftoff::Load(this, *next_result_reg, MemOperand(sp, 0), out_argument_kind);
1863 }
1864
1865 AddWord(sp, sp, Operand(stack_bytes));
1866}

1868void LiftoffAssembler::CallC(const std::initializer_list<VarState> args_list,
1869 ExternalReference ext_ref) {
1870 const int num_args = static_cast<int>(args_list.size());
1871 const VarState* const args = args_list.begin();
1872 // Note: If we ever need more than eight arguments we would need to load the
1873 // stack arguments to registers (via LoadToRegister) in pairs of two, then use
1874 // Stp with MemOperand{sp, -2 * kSystemPointerSize, PreIndex} to push them to
1875 // the stack.
1876 // Execute the parallel register move for register parameters.
1877 DCHECK_GE(arraysize(kCArgRegs), num_args);
1878 ParallelMove parallel_move{this};
1879 for (int reg_arg = 0; reg_arg < num_args; ++reg_arg) {
1880 parallel_move.LoadIntoRegister(LiftoffRegister{kCArgRegs[reg_arg]},
1881 args[reg_arg]);
1882 }
1883 parallel_move.Execute();
1884 // Now call the C function.
1885 PrepareCallCFunction(num_args, kScratchReg);
1886 CallCFunction(ext_ref, num_args);
1887}

1889void LiftoffStackSlots::Construct(int param_slots) {
1890 DCHECK_LT(0, slots_.size());
1891 SortInPushOrder();
1892 int last_stack_slot = param_slots;
1893 for (auto& slot : slots_) {
1894 const int stack_slot = slot.dst_slot_;
1895 int stack_decrement = (last_stack_slot - stack_slot) * kSystemPointerSize;
1896 DCHECK_LT(0, stack_decrement);
1897 last_stack_slot = stack_slot;
1898 const LiftoffAssembler::VarState& src = slot.src_;
1899 switch (src.loc()) {
1900 case LiftoffAssembler::VarState::kStack:
1901 if (src.kind() != kS128) {
1902 asm_->AllocateStackSpace(stack_decrement - kSystemPointerSize);
1903 asm_->Ld(kScratchReg, liftoff::GetStackSlot(slot.src_offset_));
1904 asm_->push(kScratchReg);
1905 } else {
1906 asm_->AllocateStackSpace(stack_decrement - kSimd128Size);
1907 asm_->Ld(kScratchReg, liftoff::GetStackSlot(slot.src_offset_ - 8));
1908 asm_->push(kScratchReg);
1909 asm_->Ld(kScratchReg, liftoff::GetStackSlot(slot.src_offset_));
1910 asm_->push(kScratchReg);
1911 }
1912 break;
1913 case LiftoffAssembler::VarState::kRegister: {
1914 int pushed_bytes = SlotSizeInBytes(slot);
1915 asm_->AllocateStackSpace(stack_decrement - pushed_bytes);
1916 liftoff::push(asm_, src.reg(), src.kind());
1917 break;
1918 }
1919 case LiftoffAssembler::VarState::kIntConst: {
1920 asm_->AllocateStackSpace(stack_decrement - kSystemPointerSize);
1921 asm_->li(kScratchReg, Operand(src.i32_const()));
1922 asm_->push(kScratchReg);
1923 break;
1924 }
1925 }
1926 }
