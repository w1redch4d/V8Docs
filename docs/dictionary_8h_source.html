441// The comparator is passed two indices |a| and |b|, and it returns < 0 when the
442// property at index |a| comes before the property at index |b| in the
443// enumeration order.
444template <typename Dictionary>
445struct EnumIndexComparator {
446 explicit EnumIndexComparator(Tagged<Dictionary> dict) : dict(dict) {}
447 bool operator()(Tagged_t a, Tagged_t b) {
448 PropertyDetails da(dict->DetailsAt(
449 InternalIndex(Tagged<Smi>(static_cast<Address>(a)).value())));
450 PropertyDetails db(dict->DetailsAt(
451 InternalIndex(Tagged<Smi>(static_cast<Address>(b)).value())));
452 return da.dictionary_index() < db.dictionary_index();
453 }
454 Tagged<Dictionary> dict;
455};
