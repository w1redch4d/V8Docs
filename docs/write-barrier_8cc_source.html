29template <MarkerBase::WriteBarrierType type>
30void ProcessMarkValue(HeapObjectHeader& header, MarkerBase* marker,
31 const void* value) {
32 DCHECK(marker->heap().is_incremental_marking_in_progress());
33 DCHECK(header.IsMarked<AccessMode::kAtomic>());
34 DCHECK(marker);
35
36 if (V8_UNLIKELY(header.IsInConstruction<AccessMode::kNonAtomic>())) {
37 // In construction objects are traced only if they are unmarked. If marking
38 // reaches this object again when it is fully constructed, it will re-mark
39 // it and tracing it as a previously not fully constructed object would know
40 // to bail out.
41 header.Unmark<AccessMode::kAtomic>();
42 marker->WriteBarrierForInConstructionObject(header);
43 return;
44 }
45
46 marker->WriteBarrierForObject<type>(header);
47}

59// static
60void WriteBarrier::DijkstraMarkingBarrierSlow(const void* value) {
61 const BasePage* page = BasePage::FromPayload(value);
62 const auto& heap = page->heap();
63
64 // GetWriteBarrierType() checks marking state.
65 DCHECK(heap.marker());
66 // No write barriers should be executed from atomic pause marking.
67 DCHECK(!heap.in_atomic_pause());
68
69 auto& header =
70 const_cast<HeapObjectHeader&>(page->ObjectHeaderFromInnerAddress(value));
71 if (!header.TryMarkAtomic()) return;
72
73 ProcessMarkValue<MarkerBase::WriteBarrierType::kDijkstra>(
74 header, heap.marker(), value);
75}

77// static
78void WriteBarrier::DijkstraMarkingBarrierRangeSlow(
79 HeapHandle& heap_handle, const void* first_element, size_t element_size,
80 size_t number_of_elements, TraceCallback trace_callback) {
81 auto& heap_base = HeapBase::From(heap_handle);
82
83 // GetWriteBarrierType() checks marking state.
84 DCHECK(heap_base.marker());
85 // No write barriers should be executed from atomic pause marking.
86 DCHECK(!heap_base.in_atomic_pause());
87
88 cppgc::subtle::DisallowGarbageCollectionScope disallow_gc_scope(heap_base);
89 const char* array = static_cast<const char*>(first_element);
90 while (number_of_elements-- > 0) {
91 trace_callback(&heap_base.marker()->Visitor(), array);
92 array += element_size;
93 }
94}

105void WriteBarrier::SteeleMarkingBarrierSlow(const void* value) {
106 const BasePage* page = BasePage::FromPayload(value);
107 const auto& heap = page->heap();

109 // GetWriteBarrierType() checks marking state.

123// static
124void WriteBarrier::GenerationalBarrierSlow(const CagedHeapLocalData& local_data,
125 const AgeTable& age_table,
126 const void* slot,
127 uintptr_t value_offset,
128 HeapHandle* heap_handle) {
129 DCHECK(slot);
130 DCHECK(heap_handle);
131 DCHECK_GT(api_constants::kCagedHeapMaxReservationSize, value_offset);
132 // A write during atomic pause (e.g. pre-finalizer) may trigger the slow path
133 // of the barrier. This is a result of the order of bailouts where not marking
134 // results in applying the generational barrier.
135 auto& heap = HeapBase::From(*heap_handle);
136 if (heap.in_atomic_pause()) return;
137
138 if (value_offset > 0 && age_table.GetAge(value_offset) == AgeTable::Age::kOld)
139 return;
140
141 // Record slot.
142 heap.remembered_set().AddSlot((const_cast<void*>(slot)));
143}

199void YoungGenerationEnabler::Enable() {
200 auto& instance = Instance();
201 v8::base::SpinningMutexGuard _(&instance.mutex_);
202 if (++instance.is_enabled_ == 1) {
203 // Enter the flag so that the check in the write barrier will always trigger
204 // when young generation is enabled.
205 WriteBarrier::FlagUpdater::Enter();
206 }
207}
208
209void YoungGenerationEnabler::Disable() {
210 auto& instance = Instance();
211 v8::base::SpinningMutexGuard _(&instance.mutex_);
212 DCHECK_LT(0, instance.is_enabled_);
213 if (--instance.is_enabled_ == 0) {
214 WriteBarrier::FlagUpdater::Exit();
215 }
216}

228// static
229template <WriteBarrierSlotType SlotType>
230void WriteBarrier::CombinedWriteBarrierSlow(const void* slot) {
231 DCHECK_NOT_NULL(slot);
232
233 const void* value = nullptr;
234#if defined(CPPGC_POINTER_COMPRESSION)
235 if constexpr (SlotType == WriteBarrierSlotType::kCompressed) {
236 value = CompressedPointer::Decompress(
237 *static_cast<const CompressedPointer::IntegralType*>(slot));
238 } else {
239 value = *reinterpret_cast<const void* const*>(slot);
240 }
241#else
242 static_assert(SlotType == WriteBarrierSlotType::kUncompressed);
243 value = *reinterpret_cast<const void* const*>(slot);
244#endif
245
246 WriteBarrier::Params params;
247 const WriteBarrier::Type type =
248 WriteBarrier::GetWriteBarrierType(slot, value, params);
249 switch (type) {
250 case WriteBarrier::Type::kGenerational:
251 WriteBarrier::GenerationalBarrier<
252 WriteBarrier::GenerationalBarrierType::kPreciseSlot>(params, slot);
253 break;
254 case WriteBarrier::Type::kMarking:
255 WriteBarrier::DijkstraMarkingBarrier(params, value);
256 break;
257 case WriteBarrier::Type::kNone:
258 // The fast checks are approximate and may trigger spuriously if any heap
259 // has marking in progress. `GetWriteBarrierType()` above is exact which
260 // is the reason we could still observe a bailout here.
261 break;
262 }
263}
