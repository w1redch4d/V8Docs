67 struct CandidateInfo {
68 Node* node;
69 uint32_t inlinee_index;
70 int call_count;
71 int wire_byte_size;
72
73 int64_t score() const {
74 // Note that the zero-point is arbitrary. Functions with negative score
75 // can still get inlined.
76
77 // Note(mliedtke): Adding information about "this call has constant
78 // arguments" didn't seem to provide measurable gains at the current
79 // state, still this would be an interesting measure to retry at a later
80 // point potentially together with other metrics.
81 const int count_factor = 2;
82 const int size_factor = 3;
83 return int64_t{call_count} * count_factor -
84 int64_t{wire_byte_size} * size_factor;
85 }
86 };
87
88 struct LexicographicOrdering {
89 // Returns if c1 should be prioritized less than c2.
90 bool operator()(CandidateInfo& c1, CandidateInfo& c2) {
91 return c1.score() < c2.score();
92 }
93 };
