24// The std library doesn't provide a binary max_exponent for integers, however
25// we can compute an analog using std::numeric_limits<>::digits.
26template <typename NumericType>
27struct MaxExponent {
28 static const int value = std::is_floating_point_v<NumericType>
29 ? std::numeric_limits<NumericType>::max_exponent
30 : std::numeric_limits<NumericType>::digits + 1;
31};

62// This performs a fast negation, returning a signed value. It works on unsigned
63// arguments, but probably doesn't do what you want for any unsigned value
64// larger than max / 2 + 1 (i.e. signed min cast to unsigned).
65template <typename T>
66constexpr typename std::make_signed<T>::type ConditionalNegate(
67 T x, bool is_negative) {
68 static_assert(std::is_integral_v<T>, "Type must be integral");
69 using SignedT = typename std::make_signed<T>::type;
70 using UnsignedT = typename std::make_unsigned<T>::type;
71 return static_cast<SignedT>((static_cast<UnsignedT>(x) ^
72 static_cast<UnsignedT>(-SignedT(is_negative))) +
73 is_negative);
74}

76// This performs a safe, absolute value via unsigned overflow.
77template <typename T>
78constexpr typename std::make_unsigned<T>::type SafeUnsignedAbs(T value) {
79 static_assert(std::is_integral_v<T>, "Type must be integral");
80 using UnsignedT = typename std::make_unsigned<T>::type;
81 return IsValueNegative(value)
82 ? static_cast<UnsignedT>(0u - static_cast<UnsignedT>(value))
83 : static_cast<UnsignedT>(value);
84}

121// A range for a given nunmeric Src type is contained for a given numeric Dst
122// type if both numeric_limits<Src>::max() <= numeric_limits<Dst>::max() and
123// numeric_limits<Src>::lowest() >= numeric_limits<Dst>::lowest() are true.
124// We implement this as template specializations rather than simple static
125// comparisons to ensure type correctness in our comparisons.
126enum NumericRangeRepresentation {
127 NUMERIC_RANGE_NOT_CONTAINED,
128 NUMERIC_RANGE_CONTAINED
129};

175// This class wraps the range constraints as separate booleans so the compiler
176// can identify constants and eliminate unused code paths.
177class RangeCheck {
178 public:
179 constexpr RangeCheck(bool is_in_lower_bound, bool is_in_upper_bound)
180 : is_underflow_(!is_in_lower_bound), is_overflow_(!is_in_upper_bound) {}
181 constexpr RangeCheck() : is_underflow_(false), is_overflow_(false) {}
182 constexpr bool IsValid() const { return !is_overflow_ && !is_underflow_; }
183 constexpr bool IsInvalid() const { return is_overflow_ && is_underflow_; }
184 constexpr bool IsOverflow() const { return is_overflow_ && !is_underflow_; }
185 constexpr bool IsUnderflow() const { return !is_overflow_ && is_underflow_; }
186 constexpr bool IsOverflowFlagSet() const { return is_overflow_; }
187 constexpr bool IsUnderflowFlagSet() const { return is_underflow_; }
188 constexpr bool operator==(const RangeCheck rhs) const {
189 return is_underflow_ == rhs.is_underflow_ &&
190 is_overflow_ == rhs.is_overflow_;
191 }
192 constexpr bool operator!=(const RangeCheck rhs) const {
193 return !(*this == rhs);
194 }
195
196 private:
197 // Do not change the order of these member variables. The integral conversion
198 // optimization depends on this exact order.
199 const bool is_underflow_;
200 const bool is_overflow_;
201};

203// The following helper template addresses a corner case in range checks for
204// conversion from a floating-point type to an integral type of smaller range
205// but larger precision (e.g. float -> unsigned). The problem is as follows:
206// 1. Integral maximum is always one less than a power of two, so it must be
207// truncated to fit the mantissa of the floating point. The direction of
208// rounding is implementation defined, but by default it's always IEEE
209// floats, which round to nearest and thus result in a value of larger
210// magnitude than the integral value.
211// Example: float f = UINT_MAX; // f is 4294967296f but UINT_MAX
212// // is 4294967295u.
213// 2. If the floating point value is equal to the promoted integral maximum
214// value, a range check will erroneously pass.
215// Example: (4294967296f <= 4294967295u) // This is true due to a precision
216// // loss in rounding up to float.
217// 3. When the floating point value is then converted to an integral, the
218// resulting value is out of range for the target integral type and
219// thus is implementation defined.
220// Example: unsigned u = (float)INT_MAX; // u will typically overflow to 0.
221// To fix this bug we manually truncate the maximum value when the destination
222// type is an integral of larger precision than the source floating-point type,
223// such that the resulting maximum is represented exactly as a floating point.
224template <typename Dst, typename Src, template <typename> class Bounds>
225struct NarrowingRange {
226 using SrcLimits = std::numeric_limits<Src>;
227 using DstLimits = typename std::numeric_limits<Dst>;
228
229 // Computes the mask required to make an accurate comparison between types.
230 static const int kShift =
231 (MaxExponent<Src>::value > MaxExponent<Dst>::value &&
232 SrcLimits::digits < DstLimits::digits)
233 ? (DstLimits::digits - SrcLimits::digits)
234 : 0;
235
236 template <typename T>
237 requires(std::integral<T>)
238 // Masks out the integer bits that are beyond the precision of the
239 // intermediate type used for comparison.
240 static constexpr T Adjust(T value) {
241 static_assert(std::is_same_v<T, Dst>, "");
242 static_assert(kShift < DstLimits::digits, "");
243 using UnsignedDst = typename std::make_unsigned_t<T>;
244 return static_cast<T>(ConditionalNegate(
245 SafeUnsignedAbs(value) & ~((UnsignedDst{1} << kShift) - UnsignedDst{1}),
246 IsValueNegative(value)));
247 }
248
249 template <typename T>
250 requires(std::floating_point<T>)
251 static constexpr T Adjust(T value) {
252 static_assert(std::is_same_v<T, Dst>, "");
253 static_assert(kShift == 0, "");
254 return value;
255 }
256
257 static constexpr Dst max() { return Adjust(Bounds<Dst>::max()); }
258 static constexpr Dst lowest() { return Adjust(Bounds<Dst>::lowest()); }
259};

261template <typename Dst, typename Src, template <typename> class Bounds,
262 IntegerRepresentation DstSign = std::is_signed_v<Dst>
263 ? INTEGER_REPRESENTATION_SIGNED
264 : INTEGER_REPRESENTATION_UNSIGNED,
265 IntegerRepresentation SrcSign = std::is_signed_v<Src>
266 ? INTEGER_REPRESENTATION_SIGNED
267 : INTEGER_REPRESENTATION_UNSIGNED,
268 NumericRangeRepresentation DstRange =
269 StaticDstRangeRelationToSrcRange<Dst, Src>::value>
270struct DstRangeRelationToSrcRangeImpl;

272// The following templates are for ranges that must be verified at runtime. We
273// split it into checks based on signedness to avoid confusing casts and
274// compiler warnings on signed an unsigned comparisons.
275
276// Same sign narrowing: The range is contained for normal limits.
277template <typename Dst, typename Src, template <typename> class Bounds,
278 IntegerRepresentation DstSign, IntegerRepresentation SrcSign>
279struct DstRangeRelationToSrcRangeImpl<Dst, Src, Bounds, DstSign, SrcSign,
280 NUMERIC_RANGE_CONTAINED> {
281 static constexpr RangeCheck Check(Src value) {
282 using SrcLimits = std::numeric_limits<Src>;
283 using DstLimits = NarrowingRange<Dst, Src, Bounds>;
284 return RangeCheck(
285 static_cast<Dst>(SrcLimits::lowest()) >= DstLimits::lowest() ||
286 static_cast<Dst>(value) >= DstLimits::lowest(),
287 static_cast<Dst>(SrcLimits::max()) <= DstLimits::max() ||
288 static_cast<Dst>(value) <= DstLimits::max());
289 }
290};
291
292// Signed to signed narrowing: Both the upper and lower boundaries may be
293// exceeded for standard limits.
294template <typename Dst, typename Src, template <typename> class Bounds>
295struct DstRangeRelationToSrcRangeImpl<
296 Dst, Src, Bounds, INTEGER_REPRESENTATION_SIGNED,
297 INTEGER_REPRESENTATION_SIGNED, NUMERIC_RANGE_NOT_CONTAINED> {
298 static constexpr RangeCheck Check(Src value) {
299 using DstLimits = NarrowingRange<Dst, Src, Bounds>;
300 return RangeCheck(value >= DstLimits::lowest(), value <= DstLimits::max());
301 }
302};
303
304// Unsigned to unsigned narrowing: Only the upper bound can be exceeded for
305// standard limits.
306template <typename Dst, typename Src, template <typename> class Bounds>
307struct DstRangeRelationToSrcRangeImpl<
308 Dst, Src, Bounds, INTEGER_REPRESENTATION_UNSIGNED,
309 INTEGER_REPRESENTATION_UNSIGNED, NUMERIC_RANGE_NOT_CONTAINED> {
310 static constexpr RangeCheck Check(Src value) {
311 using DstLimits = NarrowingRange<Dst, Src, Bounds>;
312 return RangeCheck(
313 DstLimits::lowest() == Dst(0) || value >= DstLimits::lowest(),
314 value <= DstLimits::max());
315 }
316};
317
318// Unsigned to signed: Only the upper bound can be exceeded for standard limits.
319template <typename Dst, typename Src, template <typename> class Bounds>
320struct DstRangeRelationToSrcRangeImpl<
321 Dst, Src, Bounds, INTEGER_REPRESENTATION_SIGNED,
322 INTEGER_REPRESENTATION_UNSIGNED, NUMERIC_RANGE_NOT_CONTAINED> {
323 static constexpr RangeCheck Check(Src value) {
324 using DstLimits = NarrowingRange<Dst, Src, Bounds>;
325 using Promotion = decltype(Src() + Dst());
326 return RangeCheck(DstLimits::lowest() <= Dst(0) ||
327 static_cast<Promotion>(value) >=
328 static_cast<Promotion>(DstLimits::lowest()),
329 static_cast<Promotion>(value) <=
330 static_cast<Promotion>(DstLimits::max()));
331 }
332};
333
334// Signed to unsigned: The upper boundary may be exceeded for a narrower Dst,
335// and any negative value exceeds the lower boundary for standard limits.
336template <typename Dst, typename Src, template <typename> class Bounds>
337struct DstRangeRelationToSrcRangeImpl<
338 Dst, Src, Bounds, INTEGER_REPRESENTATION_UNSIGNED,
339 INTEGER_REPRESENTATION_SIGNED, NUMERIC_RANGE_NOT_CONTAINED> {
340 static constexpr RangeCheck Check(Src value) {
341 using SrcLimits = std::numeric_limits<Src>;
342 using DstLimits = NarrowingRange<Dst, Src, Bounds>;
343 using Promotion = decltype(Src() + Dst());
344 bool ge_zero = false;
345 // Converting floating-point to integer will discard fractional part, so
346 // values in (-1.0, -0.0) will truncate to 0 and fit in Dst.
347 if (std::is_floating_point_v<Src>) {
348 ge_zero = value > Src(-1);
349 } else {
350 ge_zero = value >= Src(0);
351 }
352 return RangeCheck(
353 ge_zero && (DstLimits::lowest() == 0 ||
354 static_cast<Dst>(value) >= DstLimits::lowest()),
355 static_cast<Promotion>(SrcLimits::max()) <=
356 static_cast<Promotion>(DstLimits::max()) ||
357 static_cast<Promotion>(value) <=
358 static_cast<Promotion>(DstLimits::max()));
359 }
360};

369template <typename Dst, template <typename> class Bounds = std::numeric_limits,

371constexpr RangeCheck DstRangeRelationToSrcRange(Src value) {

373 static_assert(std::is_arithmetic_v<Dst>, "Result must be numeric.");
374 static_assert(Bounds<Dst>::lowest() < Bounds<Dst>::max(), "");

501// We can statically check if operations on the provided types can wrap, so we
502// can skip the checked operations if they're not needed. So, for an integer we
503// care if the destination type preserves the sign and is twice the width of
504// the source.
505template <typename T, typename Lhs, typename Rhs = Lhs>
506struct IsIntegerArithmeticSafe {
507 static const bool value =
508 !std::is_floating_point_v<T> && !std::is_floating_point_v<Lhs> &&
509 !std::is_floating_point_v<Rhs> &&
510 std::is_signed_v<T> >= std::is_signed_v<Lhs> &&
511 IntegerBitsPlusSign<T>::value >= (2 * IntegerBitsPlusSign<Lhs>::value) &&
512 std::is_signed_v<T> >= std::is_signed_v<Rhs> &&
513 IntegerBitsPlusSign<T>::value >= (2 * IntegerBitsPlusSign<Rhs>::value);
514};

745// These perform the actual math operations on the CheckedNumerics.
746// Binary arithmetic operations.
747template <template <typename, typename> class C, typename L, typename R>
748constexpr bool SafeCompare(const L lhs, const R rhs) {
749 static_assert(std::is_arithmetic_v<L> && std::is_arithmetic_v<R>,
750 "Types must be numeric.");
751 using Promotion = BigEnoughPromotion<L, R>;
752 using BigType = typename Promotion::type;
753 return Promotion::is_contained
754 // Force to a larger type for speed if both are contained.
755 ? C<BigType, BigType>::Test(
756 static_cast<BigType>(static_cast<L>(lhs)),
757 static_cast<BigType>(static_cast<R>(rhs)))
758 // Let the template functions figure it out for mixed types.
759 : C<L, R>::Test(lhs, rhs);
760}
