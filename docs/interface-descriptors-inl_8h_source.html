95template <typename DerivedDescriptor>
96void StaticCallInterfaceDescriptor<DerivedDescriptor>::Initialize(
97 CallInterfaceDescriptorData* data) {
98 // Static local copy of the Registers array, for platform-specific
99 // initialization
100 static constexpr auto registers = DerivedDescriptor::registers();
101 static constexpr auto double_registers =
102 DerivedDescriptor::double_registers();
103 static constexpr auto return_registers =
104 DerivedDescriptor::return_registers();
105 static constexpr auto return_double_registers =
106 DerivedDescriptor::return_double_registers();
107
108 // The passed pointer should be a modifiable pointer to our own data.
109 DCHECK_EQ(data, this->data());
110 DCHECK(!data->IsInitialized());
111
112 if (DerivedDescriptor::kRestrictAllocatableRegisters) {
113 data->RestrictAllocatableRegisters(registers.data(), registers.size());
114 } else {
115 DCHECK(!DerivedDescriptor::kCalleeSaveRegisters);
116 }
117
118 // Make sure the defined arrays are big enough. The arrays can be filled up
119 // with `no_reg` and `no_dreg` to pass this DCHECK.
120 DCHECK_GE(registers.size(), GetRegisterParameterCount());
121 DCHECK_GE(double_registers.size(), GetRegisterParameterCount());
122 DCHECK_GE(return_registers.size(), DerivedDescriptor::kReturnCount);
123 DCHECK_GE(return_double_registers.size(), DerivedDescriptor::kReturnCount);
124 data->InitializeRegisters(
125 DerivedDescriptor::flags(), DerivedDescriptor::kEntrypointTag,
126 DerivedDescriptor::kReturnCount, DerivedDescriptor::GetParameterCount(),
127 DerivedDescriptor::kStackArgumentOrder,
128 DerivedDescriptor::GetRegisterParameterCount(), registers.data(),
129 double_registers.data(), return_registers.data(),
130 return_double_registers.data());
131
132 // InitializeTypes is customizable by the DerivedDescriptor subclass.
133 DerivedDescriptor::InitializeTypes(data);
134
135 DCHECK(data->IsInitialized());
136 DCHECK(this->CheckFloatingPointParameters(data));
137#if DEBUG
138 DerivedDescriptor::Verify(data);
139#endif
140}

167// Helper trait for statically checking if a type is a std::array<Register,N>.
168template <typename T>
169struct IsRegisterArray : public std::false_type {};
170template <size_t N>
171struct IsRegisterArray<std::array<Register, N>> : public std::true_type {};
172template <>
173struct IsRegisterArray<EmptyRegisterArray> : public std::true_type {};
174
175// Helper for finding the index of the first invalid register in a register
176// array.
177template <size_t N, size_t Index>
178struct FirstInvalidRegisterHelper {
179 static constexpr int Call(std::array<Register, N> regs) {
180 if (!std::get<Index>(regs).is_valid()) {
181 // All registers after the first invalid one have to also be invalid (this
182 // DCHECK will be checked recursively).
183 DCHECK_EQ((FirstInvalidRegisterHelper<N, Index + 1>::Call(regs)),
184 Index + 1);
185 return Index;
186 }
187 return FirstInvalidRegisterHelper<N, Index + 1>::Call(regs);
188 }
189};
190template <size_t N>
191struct FirstInvalidRegisterHelper<N, N> {
192 static constexpr int Call(std::array<Register, N> regs) { return N; }
193};
194template <size_t N, size_t Index = 0>
195constexpr size_t FirstInvalidRegister(std::array<Register, N> regs) {
196 return FirstInvalidRegisterHelper<N, 0>::Call(regs);
197}
198constexpr size_t FirstInvalidRegister(EmptyRegisterArray regs) { return 0; }
199
200} // namespace detail

202// static
203template <typename DerivedDescriptor>
204constexpr int
205StaticCallInterfaceDescriptor<DerivedDescriptor>::GetRegisterParameterCount() {
206 static_assert(
207 detail::IsRegisterArray<decltype(DerivedDescriptor::registers())>::value,
208 "DerivedDescriptor subclass should define a registers() function "
209 "returning a std::array<Register>");
210
211 // The register parameter count is the minimum of:
212 // 1. The number of named parameters in the descriptor, and
213 // 2. The number of valid registers the descriptor provides with its
214 // registers() function, e.g. for {rax, rbx, no_reg} this number is 2.
215 // 3. The maximum number of register parameters allowed (
216 // kMaxBuiltinRegisterParams for most builtins,
217 // kMaxTFSBuiltinRegisterParams for TFS builtins, customizable by the
218 // subclass otherwise).
219 return std::min<int>({DerivedDescriptor::GetParameterCount(),
220 static_cast<int>(detail::FirstInvalidRegister(
221 DerivedDescriptor::registers())),
222 DerivedDescriptor::kMaxRegisterParams});
223}

249// static
250template <typename DerivedDescriptor>
251constexpr MachineType
252StaticCallInterfaceDescriptor<DerivedDescriptor>::GetParameterType(int i) {
253 if constexpr (!DerivedDescriptor::kCustomMachineTypes) {
254 // If there are no custom machine types, all results and parameters are
255 // tagged.
256 return MachineType::AnyTagged();
257 } else {
258 // All varags are tagged.
259 if (DerivedDescriptor::AllowVarArgs() &&
260 i >= DerivedDescriptor::GetParameterCount()) {
261 return MachineType::AnyTagged();
262 }
263 DCHECK_LT(i, DerivedDescriptor::GetParameterCount());
264 return DerivedDescriptor::kMachineTypes
265 [DerivedDescriptor::GetReturnCount() + i];
266 }
267}

302// static
303constexpr RegList WriteBarrierDescriptor::ComputeSavedRegisters(
304 Register object, Register slot_address) {
305 DCHECK(!AreAliased(object, slot_address));
306 RegList saved_registers;
307#if V8_TARGET_ARCH_X64
308 // Only push clobbered registers.
309 if (object != ObjectRegister()) saved_registers.set(ObjectRegister());
310 if (slot_address != no_reg && slot_address != SlotAddressRegister()) {
311 saved_registers.set(SlotAddressRegister());
312 }
313#elif V8_TARGET_ARCH_ARM64 || V8_TARGET_ARCH_ARM || V8_TARGET_ARCH_LOONG64 || \
314 V8_TARGET_ARCH_MIPS64
315 if (object != ObjectRegister()) saved_registers.set(ObjectRegister());
316 // The slot address is always clobbered.
317 saved_registers.set(SlotAddressRegister());
318#else
319 // TODO(cbruni): Enable callee-saved registers for other platforms.
320 // This is a temporary workaround to prepare code for callee-saved registers.
321 constexpr auto allocated_registers = registers();
322 for (size_t i = 0; i < allocated_registers.size(); ++i) {
323 saved_registers.set(allocated_registers[i]);
324 }
325#endif
326 return saved_registers;
327}
