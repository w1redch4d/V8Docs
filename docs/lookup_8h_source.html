72 enum State {
73 // The property was not found by the iterator (this is a terminal state,
74 // iteration should not continue after hitting a not-found state).
75 NOT_FOUND,
76 // Typed arrays have special handling for "canonical numeric index string"
77 // (https://tc39.es/ecma262/#sec-canonicalnumericindexstring), where not
78 // finding such an index (either because of OOB, or because it's not a valid
79 // integer index) immediately returns undefined, without walking the
80 // prototype (https://tc39.es/ecma262/#sec-typedarray-get).
81 TYPED_ARRAY_INDEX_NOT_FOUND,
82 // The next lookup requires an access check -- we can continue iteration on
83 // a successful check, otherwise we should abort.
84 ACCESS_CHECK,
85 // Interceptors are API-level hooks for optionally handling a lookup in
86 // embedder code -- if their handling returns false, then we should continue
87 // the iteration, though we should be conscious that an interceptor may have
88 // side effects despite returning false and might invalidate the lookup
89 // iterator state.
90 INTERCEPTOR,
91 // Proxies are user-space hooks for handling lookups in JS code.
92 // https://tc39.es/ecma262/#proxy-exotic-object
93 JSPROXY,
94 // Accessors are hooks for property getters/setters -- these can be
95 // user-space accessors (AccessorPair), or API accessors (AccessorInfo).
96 ACCESSOR,
97 // Data properties are stored as data fields on an object (either properties
98 // or elements).
99 DATA,
100 // WasmGC objects are opaque in JS, and appear to have no properties.
101 WASM_OBJECT,
102
103 // A LookupIterator in the transition state is in the middle of performing
104 // a data transition (that is, as part of a data property write, updating
105 // the receiver and its map to allow the write).
106 //
107 // This state is not expected to be observed while performing a lookup.
108 TRANSITION,
109
110 // Set state_ to BEFORE_PROPERTY to ensure that the next lookup will be a
111 // PROPERTY lookup.
112 BEFORE_PROPERTY = INTERCEPTOR
113 };

379// Similar to the LookupIterator, but for concurrent accesses from a background
380// thread.
381//
382// Note: This is a work in progress, intended to bundle code related to
383// concurrent lookups here. In its current state, the class is obviously not an
384// 'iterator'. Still, keeping the name for now, with the intent to clarify
385// names and implementation once we've gotten some experience with more
386// involved logic.
387// TODO(jgruber, v8:7790): Consider using a LookupIterator-style interface.
388// TODO(jgruber, v8:7790): Consider merging back into the LookupIterator once
389// functionality and constraints are better known.
390class ConcurrentLookupIterator final : public AllStatic {
391 public:
392 // Tri-state to distinguish between 'not-present' and 'who-knows' failures.
393 enum Result {
394 kPresent, // The value was found.
395 kNotPresent, // No value exists.
396 kGaveUp, // The operation can't be completed.
397 };
398
399 // Implements the own data property lookup for the specialized case of
400 // fixed_cow_array backing stores (these are only in use for array literal
401 // boilerplates). The contract is that the elements, elements kind, and array
402 // length passed to this function should all be read from the same JSArray
403 // instance; but due to concurrency it's possible that they may not be
404 // consistent among themselves (e.g. the elements kind may not match the
405 // given elements backing store). We are thus extra-careful to handle
406 // exceptional situations.
407 V8_EXPORT_PRIVATE static std::optional<Tagged<Object>> TryGetOwnCowElement(
408 Isolate* isolate, Tagged<FixedArray> array_elements,
409 ElementsKind elements_kind, int array_length, size_t index);
410
411 // As above, the contract is that the elements and elements kind should be
412 // read from the same holder, but this function is implemented defensively to
413 // tolerate concurrency issues.
414 V8_EXPORT_PRIVATE static Result TryGetOwnConstantElement(
415 Tagged<Object>* result_out, Isolate* isolate, LocalIsolate* local_isolate,
416 Tagged<JSObject> holder, Tagged<FixedArrayBase> elements,
417 ElementsKind elements_kind, size_t index);
418
419 // Implements the own data property lookup for the specialized case of
420 // strings.
421 V8_EXPORT_PRIVATE static Result TryGetOwnChar(Tagged<String>* result_out,
422 Isolate* isolate,
423 LocalIsolate* local_isolate,
424 Tagged<String> string,
425 size_t index);
426
427 // This method reimplements the following sequence in a concurrent setting:
428 //
429 // LookupIterator it(holder, isolate, name, LookupIterator::OWN);
430 // it.TryLookupCachedProperty();
431 // if (it.state() == LookupIterator::DATA) it.GetPropertyCell();
432 V8_EXPORT_PRIVATE static std::optional<Tagged<PropertyCell>>
433 TryGetPropertyCell(Isolate* isolate, LocalIsolate* local_isolate,
434 DirectHandle<JSGlobalObject> holder,
435 DirectHandle<Name> name);
436};
