21// We use ValueType instances constructed from canonical type indices, so we
22// can't let them get bigger than what we have storage space for.
23// TODO(jkummerow): Raise this limit. Possible options:
24// - increase the size of ValueType::HeapTypeField, using currently-unused bits.
25// - change the encoding of ValueType: one bit says whether it's a ref type,
26// the other bits then encode the index or the kind of non-ref type.
27// - refactor the TypeCanonicalizer's internals to no longer use ValueTypes
28// and related infrastructure, and use a wider encoding of canonicalized
29// type indices only here.
30// - wait for 32-bit platforms to no longer be relevant, and increase the
31// size of ValueType to 64 bits.
32// None of this seems urgent, as we have no evidence of the current limit
33// being an actual limitation in practice.
34static constexpr size_t kMaxCanonicalTypes = kV8MaxWasmTypes;
35// We don't want any valid modules to fail canonicalization.
36static_assert(kMaxCanonicalTypes >= kV8MaxWasmTypes);
37// We want the invalid index to fail any range checks.
38static_assert(kInvalidCanonicalIndex > kMaxCanonicalTypes);
39// Ensure that ValueType can hold all canonical type indexes.
40static_assert(kMaxCanonicalTypes <= (1 << ValueType::kHeapTypeBits));

42// A singleton class, responsible for isorecursive canonicalization of wasm
43// types.
44// A recursive group is a subsequence of types explicitly marked in the type
45// section of a wasm module. Identical recursive groups have to be canonicalized
46// to a single canonical group. Respective types in two identical groups are
47// considered identical for all purposes.
48// Two groups are considered identical if they have the same shape, and all
49// type indices referenced in the same position in both groups reference:
50// - identical types, if those do not belong to the rec. group,
51// - types in the same relative position in the group, if those belong to the
52// rec. group.

182 // Support for hashing of recursion groups, where type indexes have to be
183 // hashed relative to the recursion group.
184 struct CanonicalHashing {
185 base::Hasher hasher;
186 const RecursionGroupRange recgroup;
187
188 explicit CanonicalHashing(RecursionGroupRange recgroup)
189 : recgroup{recgroup} {}
190
191 void Add(CanonicalType type) {
192 // Add three pieces of information in a single number, for faster hashing:
193 // Whether the type is final, whether the supertype index is relative
194 // within the recgroup, and the supertype index itself.
195 // For relative supertypes add {kMaxCanonicalTypes} to map those to a
196 // separate index space (note that collisions in hashing are OK though).
197 uint32_t is_relative = recgroup.Contains(type.supertype) ? 1 : 0;
198 uint32_t supertype_index =
199 type.supertype.index - is_relative * recgroup.first.index;
200 static_assert(kMaxCanonicalTypes <= kMaxUInt32 >> 2);
201 uint32_t metadata =
202 (supertype_index << 2) | (is_relative << 1) | (type.is_final ? 1 : 0);
203 hasher.Add(metadata);
204 switch (type.kind) {
205 case CanonicalType::kFunction:
206 Add(*type.function_sig);
207 break;
208 case CanonicalType::kStruct:
209 Add(*type.struct_type);
210 break;
211 case CanonicalType::kArray:
212 Add(*type.array_type);
213 break;
214 }
215 }
216
217 void Add(CanonicalValueType value_type) {
218 if (value_type.has_index() && recgroup.Contains(value_type.ref_index())) {
219 // For relative indexed types add the kind and the relative index to the
220 // hash. Shift the relative index by {kMaxCanonicalTypes} to map it to a
221 // different index space (note that collisions in hashing are OK
222 // though).
223 static_assert(kMaxCanonicalTypes <= kMaxUInt32 / 2);
224 hasher.Add(value_type.kind());
225 hasher.Add((value_type.ref_index().index - recgroup.first.index) +
226 kMaxCanonicalTypes);
227 } else {
228 hasher.Add(value_type);
229 }
230 }
231
232 void Add(const CanonicalSig& sig) {
233 hasher.Add(sig.parameter_count());
234 for (CanonicalValueType type : sig.all()) Add(type);
235 }
236
237 void Add(const CanonicalStructType& struct_type) {
238 hasher.AddRange(struct_type.mutabilities());
239 for (const ValueTypeBase& field : struct_type.fields()) {
240 Add(CanonicalValueType{field});
241 }
242 }
243
244 void Add(const CanonicalArrayType& array_type) {
245 hasher.Add(array_type.mutability());
246 Add(array_type.element_type());
247 }
248
249 size_t hash() const { return hasher.hash(); }
250 };

252 // Support for equality checking of recursion groups, where type indexes have
253 // to be compared relative to their respective recursion group.
254 struct CanonicalEquality {
255 // Recursion group bounds for LHS and RHS.
256 const RecursionGroupRange recgroup1;
257 const RecursionGroupRange recgroup2;
258
259 CanonicalEquality(RecursionGroupRange recgroup1,
260 RecursionGroupRange recgroup2)
261 : recgroup1{recgroup1}, recgroup2{recgroup2} {}
262
263 bool EqualTypeIndex(CanonicalTypeIndex index1,
264 CanonicalTypeIndex index2) const {
265 const bool relative_index = recgroup1.Contains(index1);
266 if (relative_index != recgroup2.Contains(index2)) return false;
267 if (relative_index) {
268 // Compare relative type indexes within the respective recgroups.
269 uint32_t rel_type1 = index1.index - recgroup1.first.index;
270 uint32_t rel_type2 = index2.index - recgroup2.first.index;
271 if (rel_type1 != rel_type2) return false;
272 } else if (index1 != index2) {
273 return false;
274 }
275 return true;
276 }
277
278 bool EqualType(const CanonicalType& type1,
279 const CanonicalType& type2) const {
280 if (!EqualTypeIndex(type1.supertype, type2.supertype)) return false;
281 if (type1.is_final != type2.is_final) return false;
282 if (type1.is_shared != type2.is_shared) return false;
283 switch (type1.kind) {
284 case CanonicalType::kFunction:
285 return type2.kind == CanonicalType::kFunction &&
286 EqualSig(*type1.function_sig, *type2.function_sig);
287 case CanonicalType::kStruct:
288 return type2.kind == CanonicalType::kStruct &&
289 EqualStructType(*type1.struct_type, *type2.struct_type);
290 case CanonicalType::kArray:
291 return type2.kind == CanonicalType::kArray &&
292 EqualArrayType(*type1.array_type, *type2.array_type);
293 }
294 }
295
296 bool EqualTypes(base::Vector<const CanonicalType> types1,
297 base::Vector<const CanonicalType> types2) const {
298 return std::equal(types1.begin(), types1.end(), types2.begin(),
299 types2.end(),
300 std::bind_front(&CanonicalEquality::EqualType, this));
301 }
302
303 bool EqualValueType(CanonicalValueType type1,
304 CanonicalValueType type2) const {
305 if (type1.kind() != type2.kind()) return false;
306 const bool indexed = type1.has_index();
307 if (indexed != type2.has_index()) return false;
308 if (indexed) return EqualTypeIndex(type1.ref_index(), type2.ref_index());
309 const bool is_ref = type1.is_object_reference();
310 DCHECK_EQ(is_ref, type2.is_object_reference());
311 if (is_ref &&
312 type1.heap_representation() != type2.heap_representation()) {
313 return false;
314 }
315 return true;
316 }
317
318 bool EqualSig(const CanonicalSig& sig1, const CanonicalSig& sig2) const {
319 if (sig1.parameter_count() != sig2.parameter_count()) return false;
320 return std::equal(
321 sig1.all().begin(), sig1.all().end(), sig2.all().begin(),
322 sig2.all().end(),
323 std::bind_front(&CanonicalEquality::EqualValueType, this));
324 }
325
326 bool EqualStructType(const CanonicalStructType& type1,
327 const CanonicalStructType& type2) const {
328 return
329 // Compare fields, including a check that the size is the same.
330 std::equal(
331 type1.fields().begin(), type1.fields().end(),
332 type2.fields().begin(), type2.fields().end(),
333 std::bind_front(&CanonicalEquality::EqualValueType, this)) &&
334 // Compare mutabilities, skipping the check for the size.
335 std::equal(type1.mutabilities().begin(), type1.mutabilities().end(),
336 type2.mutabilities().begin());
337 }
338
339 bool EqualArrayType(const CanonicalArrayType& type1,
340 const CanonicalArrayType& type2) const {
341 return type1.mutability() == type2.mutability() &&
342 EqualValueType(type1.element_type(), type2.element_type());
343 }
344 };

346 struct CanonicalGroup {
347 CanonicalGroup(Zone* zone, size_t size, CanonicalTypeIndex first)
348 : types(zone->AllocateVector<CanonicalType>(size)), first(first) {
349 // size >= 2; otherwise a `CanonicalSingletonGroup` should have been used.
350 DCHECK_LE(2, size);
351 }
352
353 bool operator==(const CanonicalGroup& other) const {
354 CanonicalTypeIndex last{first.index +
355 static_cast<uint32_t>(types.size() - 1)};
356 CanonicalTypeIndex other_last{
357 other.first.index + static_cast<uint32_t>(other.types.size() - 1)};
358 CanonicalEquality equality{{first, last}, {other.first, other_last}};
359 return equality.EqualTypes(types, other.types);
360 }
361
362 size_t hash_value() const {
363 CanonicalTypeIndex last{first.index +
364 static_cast<uint32_t>(types.size()) - 1};
365 CanonicalHashing hasher{{first, last}};
366 for (CanonicalType t : types) {
367 hasher.Add(t);
368 }
369 return hasher.hash();
370 }
371
372 // The storage of this vector is the TypeCanonicalizer's zone_.
373 const base::Vector<CanonicalType> types;
374 const CanonicalTypeIndex first;
375 };

381 }
382
383 size_t hash_value() const {
384 CanonicalHashing hasher{{index, index}};
385 hasher.Add(type);
386 return hasher.hash();
387 }
388
389 CanonicalType type;
390 CanonicalTypeIndex index;
391 };
