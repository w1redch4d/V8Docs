96bool EmbedderDataSlot::ToAlignedPointer(IsolateForSandbox isolate,
97 void** out_pointer) const {
98 // We don't care about atomicity of access here because embedder slots
99 // are accessed this way only from the main thread via API during "mutator"
100 // phase which is propely synched with GC (concurrent marker may still look
101 // at the tagged part of the embedder slot but read-only access is ok).
102#ifdef V8_ENABLE_SANDBOX
103 // The raw part must always contain a valid external pointer table index.
104 *out_pointer = reinterpret_cast<void*>(
105 ReadExternalPointerField<kEmbedderDataSlotPayloadTag>(
106 address() + kExternalPointerOffset, isolate));
107 return true;
108#else
109 Address raw_value;
110 if (COMPRESS_POINTERS_BOOL) {
111 // TODO(ishell, v8:8875): When pointer compression is enabled 8-byte size
112 // fields (external pointers, doubles and BigInt data) are only kTaggedSize
113 // aligned so we have to use unaligned pointer friendly way of accessing
114 // them in order to avoid undefined behavior in C++ code.
115 raw_value = base::ReadUnalignedValue<Address>(address());
116 } else {
117 raw_value = *location();
118 }
119 *out_pointer = reinterpret_cast<void*>(raw_value);
120 return HAS_SMI_TAG(raw_value);
121#endif // V8_ENABLE_SANDBOX
122}

124bool EmbedderDataSlot::store_aligned_pointer(IsolateForSandbox isolate,
125 Tagged<HeapObject> host,
126 void* ptr) {
127 Address value = reinterpret_cast<Address>(ptr);
128 if (!HAS_SMI_TAG(value)) return false;
129#ifdef V8_ENABLE_SANDBOX
130 // When the sandbox is enabled, the external pointer handles in
131 // EmbedderDataSlots are lazily initialized: initially they contain the null
132 // external pointer handle (see EmbedderDataSlot::Initialize), and only once
133 // an external pointer is stored in them are they properly initialized.
134 // TODO(saelo): here we currently have to use the accessor on the host object
135 // as we may need a write barrier. This is a bit awkward. Maybe we should
136 // introduce helper methods on the ExternalPointerSlot class that allow us to
137 // determine whether the slot needs to be initialized, in which case a write
138 // barrier can be performed here.
139 size_t offset = address() - host.address() + kExternalPointerOffset;
140 host->WriteLazilyInitializedExternalPointerField<kEmbedderDataSlotPayloadTag>(
141 offset, isolate, value);
142 ObjectSlot(address() + kTaggedPayloadOffset).Relaxed_Store(Smi::zero());
143 return true;
144#else
145 gc_safe_store(isolate, value);
146 return true;
147#endif // V8_ENABLE_SANDBOX
148}

150EmbedderDataSlot::RawData EmbedderDataSlot::load_raw(
151 IsolateForSandbox isolate, const DisallowGarbageCollection& no_gc) const {
152 // We don't care about atomicity of access here because embedder slots
153 // are accessed this way only by serializer from the main thread when
154 // GC is not active (concurrent marker may still look at the tagged part
155 // of the embedder slot but read-only access is ok).
156#ifdef V8_COMPRESS_POINTERS
157 // TODO(ishell, v8:8875): When pointer compression is enabled 8-byte size
158 // fields (external pointers, doubles and BigInt data) are only kTaggedSize
159 // aligned so we have to use unaligned pointer friendly way of accessing them
160 // in order to avoid undefined behavior in C++ code.
161 return base::ReadUnalignedValue<EmbedderDataSlot::RawData>(address());
162#else
163 return *location();
164#endif
165}

173void EmbedderDataSlot::gc_safe_store(IsolateForSandbox isolate, Address value) {
174#ifdef V8_COMPRESS_POINTERS
175 static_assert(kSmiShiftSize == 0);
176 static_assert(SmiValuesAre31Bits());
177 static_assert(kTaggedSize == kInt32Size);
178
179 // We have to do two 32-bit stores here because
180 // 1) tagged part modifications must be atomic to be properly synchronized
181 // with the concurrent marker.
182 // 2) atomicity of full pointer store is not guaranteed for embedder slots
183 // since the address of the slot may not be kSystemPointerSize aligned
184 // (only kTaggedSize alignment is guaranteed).
185 // TODO(ishell, v8:8875): revisit this once the allocation alignment
186 // inconsistency is fixed.
187 Address lo = static_cast<intptr_t>(static_cast<int32_t>(value));
188 ObjectSlot(address() + kTaggedPayloadOffset).Relaxed_Store(Tagged<Smi>(lo));
189 Tagged_t hi = static_cast<Tagged_t>(value >> 32);
190 // The raw part of the payload does not contain a valid tagged value, so we
191 // need to use a raw store operation for it here.
192 AsAtomicTagged::Relaxed_Store(
193 reinterpret_cast<AtomicTagged_t*>(address() + kRawPayloadOffset), hi);
194#else
195 ObjectSlot(address() + kTaggedPayloadOffset)
196 .Relaxed_Store(Tagged<Smi>(value));
197#endif
198}

200bool EmbedderDataSlot::MustClearDuringSerialization(
201 const DisallowGarbageCollection& no_gc) {
202 // Serialization must avoid writing external pointer handles. If we were to
203 // accidentally write an external pointer handle, that ends up deserializing
204 // as a dangling pointer. For consistency it would be nice to avoid writing
205 // external pointers also in the wide-pointer case, but as we can't
206 // distinguish between Smi values and pointers we just leave them be.
207#ifdef V8_ENABLE_SANDBOX
208 auto* location = reinterpret_cast<ExternalPointerHandle*>(
209 address() + kExternalPointerOffset);
210 return base::AsAtomic32::Relaxed_Load(location) != kNullExternalPointerHandle;
211#else // !V8_ENABLE_SANDBOX
212 return false;
213#endif // !V8_ENABLE_SANDBOX
214}
