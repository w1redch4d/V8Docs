36Reduction TypedOptimization::Reduce(Node* node) {
37 switch (node->opcode()) {
38 case IrOpcode::kConvertReceiver:
39 return ReduceConvertReceiver(node);
40 case IrOpcode::kMaybeGrowFastElements:
41 return ReduceMaybeGrowFastElements(node);
42 case IrOpcode::kCheckHeapObject:
43 return ReduceCheckHeapObject(node);
44 case IrOpcode::kCheckBounds:
45 return ReduceCheckBounds(node);
46 case IrOpcode::kCheckNotTaggedHole:
47 return ReduceCheckNotTaggedHole(node);
48 case IrOpcode::kCheckMaps:
49 return ReduceCheckMaps(node);
50 case IrOpcode::kCheckNumber:
51 return ReduceCheckNumber(node);
52 case IrOpcode::kCheckNumberFitsInt32:
53 return ReduceCheckNumberFitsInt32(node);
54 case IrOpcode::kCheckString:
55 return ReduceCheckString(node);
56 case IrOpcode::kCheckStringOrStringWrapper:
57 return ReduceCheckStringOrStringWrapper(node);
58 case IrOpcode::kCheckEqualsInternalizedString:
59 return ReduceCheckEqualsInternalizedString(node);
60 case IrOpcode::kCheckEqualsSymbol:
61 return ReduceCheckEqualsSymbol(node);
62 case IrOpcode::kLoadField:
63 return ReduceLoadField(node);
64 case IrOpcode::kNumberCeil:
65 case IrOpcode::kNumberRound:
66 case IrOpcode::kNumberTrunc:
67 return ReduceNumberRoundop(node);
68 case IrOpcode::kNumberFloor:
69 return ReduceNumberFloor(node);
70 case IrOpcode::kNumberSilenceNaN:
71 return ReduceNumberSilenceNaN(node);
72 case IrOpcode::kNumberToUint8Clamped:
73 return ReduceNumberToUint8Clamped(node);
74 case IrOpcode::kPhi:
75 return ReducePhi(node);
76 case IrOpcode::kReferenceEqual:
77 return ReduceReferenceEqual(node);
78 case IrOpcode::kStringEqual:
79 case IrOpcode::kStringLessThan:
80 case IrOpcode::kStringLessThanOrEqual:
81 return ReduceStringComparison(node);
82 case IrOpcode::kStringLength:
83 return ReduceStringLength(node);
84 case IrOpcode::kSameValue:
85 return ReduceSameValue(node);
86 case IrOpcode::kSelect:
87 return ReduceSelect(node);
88 case IrOpcode::kTypeOf:
89 return ReduceTypeOf(node);
90 case IrOpcode::kToBoolean:
91 return ReduceToBoolean(node);
92 case IrOpcode::kSpeculativeToNumber:
93 return ReduceSpeculativeToNumber(node);
94 case IrOpcode::kSpeculativeNumberAdd:
95 return ReduceSpeculativeNumberAdd(node);
96 case IrOpcode::kSpeculativeNumberSubtract:
97 case IrOpcode::kSpeculativeNumberMultiply:
98 case IrOpcode::kSpeculativeNumberPow:
99 case IrOpcode::kSpeculativeNumberDivide:
100 case IrOpcode::kSpeculativeNumberModulus:
101 return ReduceSpeculativeNumberBinop(node);
102 case IrOpcode::kSpeculativeNumberEqual:
103 case IrOpcode::kSpeculativeNumberLessThan:
104 case IrOpcode::kSpeculativeNumberLessThanOrEqual:
105 return ReduceSpeculativeNumberComparison(node);
106 case IrOpcode::kTransitionElementsKindOrCheckMap:
107 return ReduceTransitionElementsKindOrCheckMap(node);
108 default:
109 break;
110 }
111 return NoChange();
112}

116OptionalMapRef GetStableMapFromObjectType(JSHeapBroker* broker,
117 Type object_type) {
118 if (object_type.IsHeapConstant()) {
119 HeapObjectRef object = object_type.AsHeapConstant()->Ref();

121 if (object_map.is_stable()) return object_map;
122 }
123 return {};

126Node* ResolveSameValueRenames(Node* node) {
127 while (true) {
128 switch (node->opcode()) {
129 case IrOpcode::kCheckHeapObject:
130 case IrOpcode::kCheckNumber:
131 case IrOpcode::kCheckNumberFitsInt32:
132 case IrOpcode::kCheckSmi:
133 case IrOpcode::kFinishRegion:
134 case IrOpcode::kTypeGuard:
135 if (node->IsDead()) {

137 } else {
138 node = node->InputAt(0);

140 }
141 default:
142 return node;
143 }

173Reduction TypedOptimization::ReduceMaybeGrowFastElements(Node* node) {
174 Node* const elements = NodeProperties::GetValueInput(node, 1);
175 Node* const index = NodeProperties::GetValueInput(node, 2);
176 Node* const length = NodeProperties::GetValueInput(node, 3);
177 Node* const effect = NodeProperties::GetEffectInput(node);
178 Node* const control = NodeProperties::GetControlInput(node);
179
180 Type const index_type = NodeProperties::GetType(index);
181 Type const length_type = NodeProperties::GetType(length);
182 CHECK(index_type.Is(Type::Unsigned31()));
183 CHECK(length_type.Is(Type::Unsigned31()));
184
185 if (!index_type.IsNone() && !length_type.IsNone() &&
186 index_type.Max() < length_type.Min()) {
187 if (v8_flags.turbo_typer_hardening) {
188 Node* check_bounds = graph()->NewNode(
189 simplified()->CheckBounds(FeedbackSource{},
190 CheckBoundsFlag::kAbortOnOutOfBounds),
191 index, length, effect, control);
192 ReplaceWithValue(node, elements, check_bounds);
193 return Replace(check_bounds);
194 } else {
195 RelaxEffectsAndControls(node);
196 return Replace(elements);
197 }
198 }
199
200 return NoChange();
201}

204 CheckBoundsParameters const& p = CheckBoundsParametersOf(node->op());
205 Node* const input = NodeProperties::GetValueInput(node, 0);
206 Type const input_type = NodeProperties::GetType(input);

212 simplified()->CheckBounds(

214 p.flags().without(CheckBoundsFlag::kConvertStringAndMinusZero)));
215 return Changed(node);
216 }
217 return NoChange();

231// The CheckMaps(o, ...map...) can be eliminated if map is stable,
232// o has type Constant(object) and map == object->map, and either
233// (1) map cannot transition further, or
234// (2) we can add a code dependency on the stability of map
235// (to guard the Constant type information).
236bool CheckMapsHelper(OptionalMapRef object_map, ZoneRefSet<Map> maps,
237 CompilationDependencies* dependencies) {
238 if (object_map.has_value()) {
239 for (MapRef map : maps) {
240 if (map.equals(*object_map)) {
241 if (object_map->CanTransition()) {
242 dependencies->DependOnStableMap(*object_map);
243 }
244 return true;
245 }
246 }
247 }
248 return false;
249}

252Reduction TypedOptimization::ReduceCheckMaps(Node* node) {
253 Node* const object = NodeProperties::GetValueInput(node, 0);
254 Type const object_type = NodeProperties::GetType(object);
255 Node* const effect = NodeProperties::GetEffectInput(node);
256 OptionalMapRef object_map = GetStableMapFromObjectType(broker(), object_type);
257 CheckMapsParameters p = CheckMapsParametersOf(node->op());
258 if (CheckMapsHelper(object_map, p.maps(), dependencies())) {
259 return Replace(effect);
260 }
261 return NoChange();
262}

341Reduction TypedOptimization::ReduceLoadField(Node* node) {
342 Node* const object = NodeProperties::GetValueInput(node, 0);
343 Type const object_type = NodeProperties::GetType(object);
344 FieldAccess const& access = FieldAccessOf(node->op());
345 if (access.base_is_tagged == kTaggedBase &&
346 access.offset == HeapObject::kMapOffset) {
347 // We can replace LoadField[Map](o) with map if is stable, and
348 // o has type Constant(object) and map == object->map, and either
349 // (1) map cannot transition further, or
350 // (2) deoptimization is enabled and we can add a code dependency on the
351 // stability of map (to guard the Constant type information).
352 OptionalMapRef object_map =
353 GetStableMapFromObjectType(broker(), object_type);
354 if (object_map.has_value()) {
355 dependencies()->DependOnStableMap(*object_map);
356 Node* const value = jsgraph()->ConstantNoHole(*object_map, broker());
357 ReplaceWithValue(node, value);
358 return Replace(value);
359 }
360 }
361 return NoChange();
362}

364Reduction TypedOptimization::ReduceNumberFloor(Node* node) {
365 Node* const input = NodeProperties::GetValueInput(node, 0);
366 Type const input_type = NodeProperties::GetType(input);
367 if (input_type.Is(type_cache_->kIntegerOrMinusZeroOrNaN)) {
368 return Replace(input);
369 }
370 if (input_type.Is(Type::PlainNumber()) &&
371 (input->opcode() == IrOpcode::kNumberDivide ||
372 input->opcode() == IrOpcode::kSpeculativeNumberDivide)) {
373 Node* const lhs = NodeProperties::GetValueInput(input, 0);
374 Type const lhs_type = NodeProperties::GetType(lhs);
375 Node* const rhs = NodeProperties::GetValueInput(input, 1);
376 Type const rhs_type = NodeProperties::GetType(rhs);
377 if (lhs_type.IsNone() || rhs_type.IsNone()) return NoChange();
378 if (lhs_type.Is(Type::Unsigned32()) && rhs_type.Is(Type::Unsigned32())) {
379 // We can replace
380 //
381 // NumberFloor(NumberDivide(lhs: unsigned32,
382 // rhs: unsigned32)): plain-number
383 //
384 // with
385 //
386 // Unsigned32Divide(lhs, rhs)
387 //
388 // and have the new node typed to [0...lhs.Max],
389 // as the truncated result must be lower than {lhs}'s maximum
390 // value (note that {rhs} cannot be less than 1 due to the
391 // plain-number type constraint on the {node}).
392 node = graph()->NewNode(simplified()->Unsigned32Divide(), lhs, rhs);
393 return Replace(node);
394 }
395 }
396 return NoChange();
397}

426Reduction TypedOptimization::ReducePhi(Node* node) {
427 // Try to narrow the type of the Phi {node}, which might be more precise now
428 // after lowering based on types, i.e. a SpeculativeNumberAdd has a more
429 // precise type than the JSAdd that was in the graph when the Typer was run.
430 DCHECK_EQ(IrOpcode::kPhi, node->opcode());
431 // Prevent new types from being propagated through loop-related Phis for now.
432 // This is to avoid slow convergence of type narrowing when we learn very
433 // precise information about loop variables.
434 if (NodeProperties::GetControlInput(node, 0)->opcode() == IrOpcode::kLoop) {
435 return NoChange();
436 }
437 int arity = node->op()->ValueInputCount();
438 Type type = NodeProperties::GetType(node->InputAt(0));
439 for (int i = 1; i < arity; ++i) {
440 type = Type::Union(type, NodeProperties::GetType(node->InputAt(i)),
441 graph()->zone());
442 }
443 Type const node_type = NodeProperties::GetType(node);
444 if (!node_type.Is(type)) {
445 type = Type::Intersect(node_type, type, graph()->zone());
446 NodeProperties::SetType(node, type);
447 return Changed(node);
448 }
449 return NoChange();
450}

452Reduction TypedOptimization::ReduceReferenceEqual(Node* node) {
453 DCHECK_EQ(IrOpcode::kReferenceEqual, node->opcode());
454 Node* const lhs = NodeProperties::GetValueInput(node, 0);
455 Node* const rhs = NodeProperties::GetValueInput(node, 1);
456 Type const lhs_type = NodeProperties::GetType(lhs);
457 Type const rhs_type = NodeProperties::GetType(rhs);
458 if (!lhs_type.Maybe(rhs_type)) {
459 Node* replacement = jsgraph()->FalseConstant();
460 // Make sure we do not widen the type.
461 if (NodeProperties::GetType(replacement)
462 .Is(NodeProperties::GetType(node))) {
463 return Replace(jsgraph()->FalseConstant());
464 }
465 }
466 if (rhs_type.Is(Type::Boolean()) && rhs_type.IsHeapConstant() &&
467 lhs_type.Is(Type::Boolean())) {
468 std::optional<bool> maybe_result =
469 rhs_type.AsHeapConstant()->Ref().TryGetBooleanValue(broker());
470 if (maybe_result.has_value()) {
471 if (maybe_result.value()) {
472 return Replace(node->InputAt(0));
473 } else {
474 node->TrimInputCount(1);
475 NodeProperties::ChangeOp(node, simplified()->BooleanNot());
476 return Changed(node);
477 }
478 }
479 }
480 return NoChange();
481}

483const Operator* TypedOptimization::NumberComparisonFor(const Operator* op) {
484 switch (op->opcode()) {
485 case IrOpcode::kStringEqual:
486 return simplified()->NumberEqual();
487 case IrOpcode::kStringLessThan:
488 return simplified()->NumberLessThan();
489 case IrOpcode::kStringLessThanOrEqual:

497Reduction TypedOptimization::
498 TryReduceStringComparisonOfStringFromSingleCharCodeToConstant(
499 Node* comparison, StringRef string, bool inverted) {
500 switch (comparison->opcode()) {
501 case IrOpcode::kStringEqual:
502 if (string.length() != 1) {
503 // String.fromCharCode(x) always has length 1.
504 return Replace(jsgraph()->BooleanConstant(false));
505 }
506 break;
507 case IrOpcode::kStringLessThan:
508 [[fallthrough]];
509 case IrOpcode::kStringLessThanOrEqual:
510 if (string.length() == 0) {
511 // String.fromCharCode(x) <= "" is always false,
512 // "" < String.fromCharCode(x) is always true.
513 return Replace(jsgraph()->BooleanConstant(inverted));
514 }
515 break;
516 default:
517 UNREACHABLE();
518 }
519 return NoChange();
520}

522// Try to reduces a string comparison of the form
523// String.fromCharCode(x) {comparison} {constant} if inverted is false,
524// and {constant} {comparison} String.fromCharCode(x) if inverted is true.
525Reduction
526TypedOptimization::TryReduceStringComparisonOfStringFromSingleCharCode(
527 Node* comparison, Node* from_char_code, Type constant_type, bool inverted) {
528 DCHECK_EQ(IrOpcode::kStringFromSingleCharCode, from_char_code->opcode());
529
530 if (!constant_type.IsHeapConstant()) return NoChange();
531 ObjectRef constant = constant_type.AsHeapConstant()->Ref();
532
533 if (!constant.IsString()) return NoChange();
534 StringRef string = constant.AsString();
535
536 // Check if comparison can be resolved statically.
537 Reduction red = TryReduceStringComparisonOfStringFromSingleCharCodeToConstant(
538 comparison, string, inverted);
539 if (red.Changed()) return red;
540
541 const Operator* comparison_op = NumberComparisonFor(comparison->op());
542 Node* from_char_code_repl = NodeProperties::GetValueInput(from_char_code, 0);
543 Type from_char_code_repl_type = NodeProperties::GetType(from_char_code_repl);
544 if (!from_char_code_repl_type.Is(type_cache_->kUint16)) {
545 // Convert to signed int32 to satisfy type of {NumberBitwiseAnd}.
546 from_char_code_repl =
547 graph()->NewNode(simplified()->NumberToInt32(), from_char_code_repl);
548 from_char_code_repl = graph()->NewNode(
549 simplified()->NumberBitwiseAnd(), from_char_code_repl,
550 jsgraph()->ConstantNoHole(std::numeric_limits<uint16_t>::max()));
551 }
552 if (!string.GetFirstChar(broker()).has_value()) return NoChange();
553 Node* constant_repl =
554 jsgraph()->ConstantNoHole(string.GetFirstChar(broker()).value());
555
556 Node* number_comparison = nullptr;
557 if (inverted) {
558 // "x..." <= String.fromCharCode(z) is true if x < z.
559 if (string.length() > 1 &&
560 comparison->opcode() == IrOpcode::kStringLessThanOrEqual) {
561 comparison_op = simplified()->NumberLessThan();
562 }
563 number_comparison =
564 graph()->NewNode(comparison_op, constant_repl, from_char_code_repl);
565 } else {
566 // String.fromCharCode(z) < "x..." is true if z <= x.
567 if (string.length() > 1 &&
568 comparison->opcode() == IrOpcode::kStringLessThan) {
569 comparison_op = simplified()->NumberLessThanOrEqual();
570 }
571 number_comparison =
572 graph()->NewNode(comparison_op, from_char_code_repl, constant_repl);
573 }
574 ReplaceWithValue(comparison, number_comparison);
575 return Replace(number_comparison);
576}

578Reduction TypedOptimization::ReduceStringComparison(Node* node) {
579 DCHECK(IrOpcode::kStringEqual == node->opcode() ||
580 IrOpcode::kStringLessThan == node->opcode() ||
581 IrOpcode::kStringLessThanOrEqual == node->opcode());
582 Node* const lhs = NodeProperties::GetValueInput(node, 0);
583 Node* const rhs = NodeProperties::GetValueInput(node, 1);
584 Type lhs_type = NodeProperties::GetType(lhs);
585 Type rhs_type = NodeProperties::GetType(rhs);
586 if (lhs->opcode() == IrOpcode::kStringFromSingleCharCode) {
587 if (rhs->opcode() == IrOpcode::kStringFromSingleCharCode) {
588 Node* left = NodeProperties::GetValueInput(lhs, 0);
589 Node* right = NodeProperties::GetValueInput(rhs, 0);
590 Type left_type = NodeProperties::GetType(left);
591 Type right_type = NodeProperties::GetType(right);
592 if (!left_type.Is(type_cache_->kUint16)) {
593 // Convert to signed int32 to satisfy type of {NumberBitwiseAnd}.
594 left = graph()->NewNode(simplified()->NumberToInt32(), left);
595 left = graph()->NewNode(
596 simplified()->NumberBitwiseAnd(), left,
597 jsgraph()->ConstantNoHole(std::numeric_limits<uint16_t>::max()));
598 }
599 if (!right_type.Is(type_cache_->kUint16)) {
600 // Convert to signed int32 to satisfy type of {NumberBitwiseAnd}.
601 right = graph()->NewNode(simplified()->NumberToInt32(), right);
602 right = graph()->NewNode(
603 simplified()->NumberBitwiseAnd(), right,
604 jsgraph()->ConstantNoHole(std::numeric_limits<uint16_t>::max()));
605 }
606 Node* equal =
607 graph()->NewNode(NumberComparisonFor(node->op()), left, right);
608 ReplaceWithValue(node, equal);
609 return Replace(equal);
610 } else {
611 return TryReduceStringComparisonOfStringFromSingleCharCode(
612 node, lhs, rhs_type, false);
613 }
614 } else if (rhs->opcode() == IrOpcode::kStringFromSingleCharCode) {
615 return TryReduceStringComparisonOfStringFromSingleCharCode(node, rhs,
616 lhs_type, true);
617 }
618 return NoChange();
619}

621Reduction TypedOptimization::ReduceStringLength(Node* node) {
622 DCHECK_EQ(IrOpcode::kStringLength, node->opcode());
623 Node* const input = NodeProperties::GetValueInput(node, 0);
624 switch (input->opcode()) {
625 case IrOpcode::kHeapConstant: {
626 // Constant-fold the String::length of the {input}.
627 HeapObjectMatcher m(input);
628 if (m.Ref(broker()).IsString()) {
629 uint32_t const length = m.Ref(broker()).AsString().length();
630 Node* value = jsgraph()->ConstantNoHole(length);
631 return Replace(value);
632 }
633 break;
634 }
635 case IrOpcode::kStringConcat: {
636 // The first value input to the {input} is the resulting length.
637 return Replace(input->InputAt(0));
638 }
639 case IrOpcode::kStringFromSingleCharCode: {
640 // Note that this isn't valid for StringFromCodePointAt, since it the
641 // string it returns can be 1 or 2 characters long.
642 return Replace(jsgraph()->ConstantNoHole(1));
643 }
644 default:
645 break;
646 }
647 return NoChange();
648}

650Reduction TypedOptimization::ReduceSameValue(Node* node) {
651 DCHECK_EQ(IrOpcode::kSameValue, node->opcode());
652 Node* const lhs = NodeProperties::GetValueInput(node, 0);
653 Node* const rhs = NodeProperties::GetValueInput(node, 1);
654 Type const lhs_type = NodeProperties::GetType(lhs);
655 Type const rhs_type = NodeProperties::GetType(rhs);
656 if (ResolveSameValueRenames(lhs) == ResolveSameValueRenames(rhs)) {
657 if (NodeProperties::GetType(node).IsNone()) {
658 return NoChange();
659 }
660 // SameValue(x,x) => #true
661 return Replace(jsgraph()->TrueConstant());
662 } else if (lhs_type.Is(Type::Unique()) && rhs_type.Is(Type::Unique())) {
663 // SameValue(x:unique,y:unique) => ReferenceEqual(x,y)
664 NodeProperties::ChangeOp(node, simplified()->ReferenceEqual());
665 return Changed(node);
666 } else if (lhs_type.Is(Type::String()) && rhs_type.Is(Type::String())) {
667 // SameValue(x:string,y:string) => StringEqual(x,y)
668 NodeProperties::ChangeOp(node, simplified()->StringEqual());
669 return Changed(node);
670 } else if (lhs_type.Is(Type::MinusZero())) {
671 // SameValue(x:minus-zero,y) => ObjectIsMinusZero(y)
672 node->RemoveInput(0);
673 NodeProperties::ChangeOp(node, simplified()->ObjectIsMinusZero());
674 return Changed(node);
675 } else if (rhs_type.Is(Type::MinusZero())) {
676 // SameValue(x,y:minus-zero) => ObjectIsMinusZero(x)
677 node->RemoveInput(1);
678 NodeProperties::ChangeOp(node, simplified()->ObjectIsMinusZero());
679 return Changed(node);
680 } else if (lhs_type.Is(Type::NaN())) {
681 // SameValue(x:nan,y) => ObjectIsNaN(y)
682 node->RemoveInput(0);
683 NodeProperties::ChangeOp(node, simplified()->ObjectIsNaN());
684 return Changed(node);
685 } else if (rhs_type.Is(Type::NaN())) {
686 // SameValue(x,y:nan) => ObjectIsNaN(x)
687 node->RemoveInput(1);
688 NodeProperties::ChangeOp(node, simplified()->ObjectIsNaN());
689 return Changed(node);
690 } else if (lhs_type.Is(Type::PlainNumber()) &&
691 rhs_type.Is(Type::PlainNumber())) {
692 // SameValue(x:plain-number,y:plain-number) => NumberEqual(x,y)
693 NodeProperties::ChangeOp(node, simplified()->NumberEqual());
694 return Changed(node);
695 }
696 return NoChange();
697}

699Reduction TypedOptimization::ReduceSelect(Node* node) {
700 DCHECK_EQ(IrOpcode::kSelect, node->opcode());
701 Node* const condition = NodeProperties::GetValueInput(node, 0);
702 Type const condition_type = NodeProperties::GetType(condition);
703 Node* const vtrue = NodeProperties::GetValueInput(node, 1);
704 Type const vtrue_type = NodeProperties::GetType(vtrue);
705 Node* const vfalse = NodeProperties::GetValueInput(node, 2);
706 Type const vfalse_type = NodeProperties::GetType(vfalse);
707 if (condition_type.Is(true_type_)) {
708 // Select(condition:true, vtrue, vfalse) => vtrue
709 return Replace(vtrue);
710 }
711 if (condition_type.Is(false_type_)) {
712 // Select(condition:false, vtrue, vfalse) => vfalse
713 return Replace(vfalse);
714 }
715 if (vtrue_type.Is(true_type_) && vfalse_type.Is(false_type_)) {
716 // Select(condition, vtrue:true, vfalse:false) => condition
717 return Replace(condition);
718 }
719 if (vtrue_type.Is(false_type_) && vfalse_type.Is(true_type_)) {
720 // Select(condition, vtrue:false, vfalse:true) => BooleanNot(condition)
721 node->TrimInputCount(1);
722 NodeProperties::ChangeOp(node, simplified()->BooleanNot());
723 return Changed(node);
724 }
725 // Try to narrow the type of the Select {node}, which might be more precise
726 // now after lowering based on types.
727 Type type = Type::Union(vtrue_type, vfalse_type, graph()->zone());
728 Type const node_type = NodeProperties::GetType(node);
729 if (!node_type.Is(type)) {
730 type = Type::Intersect(node_type, type, graph()->zone());
731 NodeProperties::SetType(node, type);
732 return Changed(node);
733 }
734 return NoChange();
735}

780Reduction TypedOptimization::ReduceToBoolean(Node* node) {
781 Node* const input = node->InputAt(0);
782 Type const input_type = NodeProperties::GetType(input);
783 if (input_type.Is(Type::Boolean())) {
784 // ToBoolean(x:boolean) => x
785 return Replace(input);
786 } else if (input_type.Is(Type::OrderedNumber())) {
787 // SToBoolean(x:ordered-number) => BooleanNot(NumberEqual(x,#0))
788 node->ReplaceInput(0, graph()->NewNode(simplified()->NumberEqual(), input,
789 jsgraph()->ZeroConstant()));
790 node->TrimInputCount(1);
791 NodeProperties::ChangeOp(node, simplified()->BooleanNot());
792 return Changed(node);
793 } else if (input_type.Is(Type::Number())) {
794 // ToBoolean(x:number) => NumberToBoolean(x)
795 node->TrimInputCount(1);
796 NodeProperties::ChangeOp(node, simplified()->NumberToBoolean());
797 return Changed(node);
798 } else if (input_type.Is(Type::DetectableReceiverOrNull())) {
799 // ToBoolean(x:detectable receiver \/ null)
800 // => BooleanNot(ReferenceEqual(x,#null))
801 node->ReplaceInput(0, graph()->NewNode(simplified()->ReferenceEqual(),
802 input, jsgraph()->NullConstant()));
803 node->TrimInputCount(1);
804 NodeProperties::ChangeOp(node, simplified()->BooleanNot());
805 return Changed(node);
806 } else if (input_type.Is(Type::ReceiverOrNullOrUndefined())) {
807 // ToBoolean(x:receiver \/ null \/ undefined)
808 // => BooleanNot(ObjectIsUndetectable(x))
809 node->ReplaceInput(
810 0, graph()->NewNode(simplified()->ObjectIsUndetectable(), input));
811 node->TrimInputCount(1);
812 NodeProperties::ChangeOp(node, simplified()->BooleanNot());
813 return Changed(node);
814 } else if (input_type.Is(Type::String())) {
815 // ToBoolean(x:string) => BooleanNot(ReferenceEqual(x,""))
816 node->ReplaceInput(0,
817 graph()->NewNode(simplified()->ReferenceEqual(), input,
818 jsgraph()->EmptyStringConstant()));
819 node->TrimInputCount(1);
820 NodeProperties::ChangeOp(node, simplified()->BooleanNot());
821 return Changed(node);
822 }
823 return NoChange();
824}

833const Operator* NumberOpFromSpeculativeNumberOp(
834 SimplifiedOperatorBuilder* simplified, const Operator* op) {
835 switch (op->opcode()) {
836 case IrOpcode::kSpeculativeNumberEqual:
837 return simplified->NumberEqual();
838 case IrOpcode::kSpeculativeNumberLessThan:
839 return simplified->NumberLessThan();
840 case IrOpcode::kSpeculativeNumberLessThanOrEqual:
841 return simplified->NumberLessThanOrEqual();
842 case IrOpcode::kSpeculativeNumberAdd:
843 // Handled by ReduceSpeculativeNumberAdd.
844 UNREACHABLE();
845 case IrOpcode::kSpeculativeNumberSubtract:
846 return simplified->NumberSubtract();
847 case IrOpcode::kSpeculativeNumberMultiply:
848 return simplified->NumberMultiply();
849 case IrOpcode::kSpeculativeNumberPow:
850 return simplified->NumberPow();
851 case IrOpcode::kSpeculativeNumberDivide:
852 return simplified->NumberDivide();
853 case IrOpcode::kSpeculativeNumberModulus:
854 return simplified->NumberModulus();
855 default:
856 break;
857 }
858 UNREACHABLE();

863Reduction TypedOptimization::ReduceSpeculativeNumberAdd(Node* node) {
864 Node* const lhs = NodeProperties::GetValueInput(node, 0);
865 Node* const rhs = NodeProperties::GetValueInput(node, 1);
866 Type const lhs_type = NodeProperties::GetType(lhs);
867 Type const rhs_type = NodeProperties::GetType(rhs);
868 NumberOperationHint hint = NumberOperationHintOf(node->op());
869 if ((hint == NumberOperationHint::kNumber ||
870 hint == NumberOperationHint::kNumberOrOddball) &&
871 BothAre(lhs_type, rhs_type, Type::PlainPrimitive()) &&
872 NeitherCanBe(lhs_type, rhs_type, Type::StringOrReceiver())) {
873 // SpeculativeNumberAdd(x:-string, y:-string) =>
874 // NumberAdd(ToNumber(x), ToNumber(y))
875 Node* const toNum_lhs = ConvertPlainPrimitiveToNumber(lhs);
876 Node* const toNum_rhs = ConvertPlainPrimitiveToNumber(rhs);
877 Node* const value =
878 graph()->NewNode(simplified()->NumberAdd(), toNum_lhs, toNum_rhs);
879 ReplaceWithValue(node, value);
880 return Replace(value);
881 }
882 return NoChange();
883}

885Reduction TypedOptimization::ReduceJSToNumberInput(Node* input) {
886 // Try constant-folding of JSToNumber with constant inputs.
887 Type input_type = NodeProperties::GetType(input);
888
889 if (input_type.Is(Type::String())) {
890 HeapObjectMatcher m(input);
891 if (m.HasResolvedValue() && m.Ref(broker()).IsString()) {
892 StringRef input_value = m.Ref(broker()).AsString();
893 std::optional<double> number = input_value.ToNumber(broker());
894 if (!number.has_value()) return NoChange();
895 return Replace(jsgraph()->ConstantNoHole(number.value()));
896 }
897 }
898 if (input_type.IsHeapConstant()) {
899 HeapObjectRef input_value = input_type.AsHeapConstant()->Ref();
900 double value;
901 if (input_value.OddballToNumber(broker()).To(&value)) {
902 return Replace(jsgraph()->ConstantNoHole(value));
903 }
904 }
905 if (input_type.Is(Type::Number())) {
906 // JSToNumber(x:number) => x
907 return Changed(input);
908 }
909 if (input_type.Is(Type::Undefined())) {
910 // JSToNumber(undefined) => #NaN
911 return Replace(jsgraph()->NaNConstant());
912 }
913 if (input_type.Is(Type::Null())) {
914 // JSToNumber(null) => #0
915 return Replace(jsgraph()->ZeroConstant());
916 }
917 return NoChange();
918}
