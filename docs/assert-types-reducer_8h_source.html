45 template <typename Op, typename Continuation>
46 OpIndex ReduceInputGraphOperation(OpIndex ig_index, const Op& operation) {
47 OpIndex og_index = Continuation{this}.ReduceInputGraph(ig_index, operation);
48 if constexpr (std::is_same_v<Op, LoadRootRegisterOp>) {
49 // LoadRootRegister is a bit special and should never be materialized,
50 // hence we cannot assert its type.
51 return og_index;
52 }
53 if (std::is_same_v<Op, ConstantOp>) {
54 // Constants are constant by definition, so asserting their types doesn't
55 // seem super useful. Additionally, they can appear before Parameters in
56 // the graph, which leads to issues because asserting their types requires
57 // inserting a Call in the graph, which can overwrite the value of
58 // Parameters.
59 return og_index;
60 }
61 if (!og_index.valid()) return og_index;
62 if (!CanBeTyped(operation)) return og_index;
63 // Unfortunately, we cannot insert assertions after block terminators, so we
64 // skip them here.
65 if (operation.IsBlockTerminator()) return og_index;
66
67 auto reps = operation.outputs_rep();
68 DCHECK_GT(reps.size(), 0);
69 if (reps.size() == 1) {
70 Type type = __ GetInputGraphType(ig_index);
71 InsertTypeAssert(reps[0], og_index, type);
72 }
73 return og_index;
74 }

76 void InsertTypeAssert(RegisterRepresentation rep, OpIndex value,
77 const Type& type) {
78 DCHECK(!type.IsInvalid());
79 if (type.IsNone()) {
80 __ Unreachable();
81 return;
82 }
83
84 if (type.IsAny()) {
85 // Ignore any typed for now.
86 return;
87 }
88
89 auto GenerateBuiltinCall =
90 [this](Builtin builtin, OpIndex original_value,
91 base::SmallVector<OpIndex, 6> actual_value_indices,
92 const Type& type) {
93 i::Tagged<Smi> op_id = Smi::FromInt(original_value.id());
94 // Add expected type and operation id.
95 Handle<TurboshaftType> expected_type = type.AllocateOnHeap(factory());
96 actual_value_indices.push_back(__ HeapConstant(expected_type));
97 actual_value_indices.push_back(__ SmiConstant(op_id));
98 actual_value_indices.push_back(__ SmiConstant(NoContextConstant()));
99 __ CallBuiltin(
100 builtin, OpIndex::Invalid(),
101 {actual_value_indices.data(), actual_value_indices.size()},
102 CanThrow::kNo, isolate_);
103#ifdef DEBUG
104 // Used for debugging
105 if (v8_flags.turboshaft_trace_typing) {
106 PrintF("Inserted assert for %3d:%-40s (%s)\n", original_value.id(),
107 __ output_graph().Get(original_value).ToString().c_str(),
108 type.ToString().c_str());
109 }
110#endif
111 };
112
113 switch (rep.value()) {
114 case RegisterRepresentation::Word32(): {
115 DCHECK(type.IsWord32());
116 base::SmallVector<OpIndex, 6> actual_value_indices = {value};
117 GenerateBuiltinCall(Builtin::kCheckTurboshaftWord32Type, value,
118 std::move(actual_value_indices), type);
119 break;
120 }
121 case RegisterRepresentation::Word64(): {
122 DCHECK(type.IsWord64());
123 OpIndex value_high =
124 __ TruncateWord64ToWord32(__ Word64ShiftRightLogical(value, 32));
125 OpIndex value_low = __ TruncateWord64ToWord32(value);
126 base::SmallVector<OpIndex, 6> actual_value_indices = {value_high,
127 value_low};
128 GenerateBuiltinCall(Builtin::kCheckTurboshaftWord64Type, value,
129 std::move(actual_value_indices), type);
130 break;
131 }
132 case RegisterRepresentation::Float32(): {
133 DCHECK(type.IsFloat32());
134 base::SmallVector<OpIndex, 6> actual_value_indices = {value};
135 GenerateBuiltinCall(Builtin::kCheckTurboshaftFloat32Type, value,
136 std::move(actual_value_indices), type);
137 break;
138 }
139 case RegisterRepresentation::Float64(): {
140 DCHECK(type.IsFloat64());
141 base::SmallVector<OpIndex, 6> actual_value_indices = {value};
142 GenerateBuiltinCall(Builtin::kCheckTurboshaftFloat64Type, value,
143 std::move(actual_value_indices), type);
144 break;
145 }
146 case RegisterRepresentation::Tagged():
147 case RegisterRepresentation::Compressed():
148 case RegisterRepresentation::Simd128():
149 case RegisterRepresentation::Simd256():
150 // TODO(nicohartmann@): Handle remaining cases.
151 break;
152 }
153 }
154
155 private:
156 Factory* factory() { return isolate_->factory(); }
157 Isolate* isolate_ = __ data() -> isolate();
158};
