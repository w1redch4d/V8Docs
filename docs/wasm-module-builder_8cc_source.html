87void WriteInitializerExpressionWithoutEnd(ZoneBuffer* buffer,
88 const WasmInitExpr& init) {
89 switch (init.kind()) {
90 case WasmInitExpr::kI32Const:
91 buffer->write_u8(kExprI32Const);
92 buffer->write_i32v(init.immediate().i32_const);
93 break;
94 case WasmInitExpr::kI64Const:
95 buffer->write_u8(kExprI64Const);
96 buffer->write_i64v(init.immediate().i64_const);
97 break;
98 case WasmInitExpr::kF32Const:
99 buffer->write_u8(kExprF32Const);
100 buffer->write_f32(init.immediate().f32_const);
101 break;
102 case WasmInitExpr::kF64Const:
103 buffer->write_u8(kExprF64Const);
104 buffer->write_f64(init.immediate().f64_const);
105 break;
106 case WasmInitExpr::kS128Const:
107 buffer->write_u8(kSimdPrefix);
108 buffer->write_u8(kExprS128Const & 0xFF);
109 buffer->write(init.immediate().s128_const.data(), kSimd128Size);
110 break;
111 case WasmInitExpr::kI32Add:
112 case WasmInitExpr::kI32Sub:
113 case WasmInitExpr::kI32Mul:
114 case WasmInitExpr::kI64Add:
115 case WasmInitExpr::kI64Sub:
116 case WasmInitExpr::kI64Mul:
117 WriteInitializerExpressionWithoutEnd(buffer, (*init.operands())[0]);
118 WriteInitializerExpressionWithoutEnd(buffer, (*init.operands())[1]);
119 buffer->write_u8(FromInitExprOperator(init.kind()));
120 break;
121 case WasmInitExpr::kGlobalGet:
122 buffer->write_u8(kExprGlobalGet);
123 buffer->write_u32v(init.immediate().index);
124 break;
125 case WasmInitExpr::kRefNullConst:
126 buffer->write_u8(kExprRefNull);
127 buffer->write_i32v(HeapType(init.immediate().heap_type).code());
128 break;
129 case WasmInitExpr::kRefFuncConst:
130 buffer->write_u8(kExprRefFunc);
131 buffer->write_u32v(init.immediate().index);
132 break;
133 case WasmInitExpr::kStructNew:
134 case WasmInitExpr::kStructNewDefault:
135 case WasmInitExpr::kArrayNew:
136 case WasmInitExpr::kArrayNewDefault: {
137 if (init.operands() != nullptr) {
138 for (const WasmInitExpr& operand : *init.operands()) {
139 WriteInitializerExpressionWithoutEnd(buffer, operand);
140 }
141 }
142 WasmOpcode opcode = FromInitExprOperator(init.kind());
143 DCHECK_EQ(opcode >> 8, kGCPrefix);
144 DCHECK_EQ(opcode & 0x80, 0);
145 buffer->write_u8(kGCPrefix);
146 buffer->write_u8(static_cast<uint8_t>(opcode));
147 buffer->write_u32v(init.immediate().index);
148 break;
149 }
150 case WasmInitExpr::kArrayNewFixed: {
151 static_assert((kExprArrayNewFixed >> 8) == kGCPrefix);
152 static_assert((kExprArrayNewFixed & 0x80) == 0);
153 for (const WasmInitExpr& operand : *init.operands()) {
154 WriteInitializerExpressionWithoutEnd(buffer, operand);
155 }
156 buffer->write_u8(kGCPrefix);
157 buffer->write_u8(static_cast<uint8_t>(kExprArrayNewFixed));
158 buffer->write_u32v(init.immediate().index);
159 buffer->write_u32v(static_cast<uint32_t>(init.operands()->size()));
160 break;
161 }
162 case WasmInitExpr::kRefI31:
163 case WasmInitExpr::kAnyConvertExtern:
164 case WasmInitExpr::kExternConvertAny: {
165 WriteInitializerExpressionWithoutEnd(buffer, (*init.operands())[0]);
166 WasmOpcode opcode = FromInitExprOperator(init.kind());
167 DCHECK_EQ(opcode >> 8, kGCPrefix);
168 DCHECK_EQ(opcode & 0x80, 0);
169 buffer->write_u8(kGCPrefix);
170 buffer->write_u8(opcode);
171 break;
172 }
173 case WasmInitExpr::kStringConst:
174 buffer->write_u8(kGCPrefix);
175 buffer->write_u32v(kExprStringConst & 0xFF);
176 buffer->write_u32v(init.immediate().index);
177 break;
178 }
179}

342void WasmFunctionBuilder::AddAsmWasmOffset(size_t call_position,
343 size_t to_number_position) {
344 // We only want to emit one mapping per byte offset.
345 DCHECK(asm_offsets_.size() == 0 || body_.size() > last_asm_byte_offset_);
346
347 DCHECK_LE(body_.size(), kMaxUInt32);
348 uint32_t byte_offset = static_cast<uint32_t>(body_.size());
349 asm_offsets_.write_u32v(byte_offset - last_asm_byte_offset_);
350 last_asm_byte_offset_ = byte_offset;
351
352 DCHECK_GE(std::numeric_limits<uint32_t>::max(), call_position);
353 uint32_t call_position_u32 = static_cast<uint32_t>(call_position);
354 asm_offsets_.write_i32v(call_position_u32 - last_asm_source_position_);
355
356 DCHECK_GE(std::numeric_limits<uint32_t>::max(), to_number_position);
357 uint32_t to_number_position_u32 = static_cast<uint32_t>(to_number_position);
358 asm_offsets_.write_i32v(to_number_position_u32 - call_position_u32);
359 last_asm_source_position_ = to_number_position_u32;
360}

392void WasmFunctionBuilder::WriteBody(ZoneBuffer* buffer) const {
393 size_t locals_size = locals_.Size();
394 buffer->write_size(locals_size + body_.size());
395 buffer->EnsureSpace(locals_size);
396 uint8_t** ptr = buffer->pos_ptr();
397 locals_.Emit(*ptr);
398 (*ptr) += locals_size; // UGLY: manual bump of position pointer
399 if (body_.size() > 0) {
400 size_t base = buffer->offset();
401 buffer->write(body_.begin(), body_.size());
402 for (DirectCallIndex call : direct_calls_) {
403 buffer->patch_u32v(
404 base + call.offset,
405 call.direct_index +
406 static_cast<uint32_t>(builder_->function_imports_.size()));
407 }
408 }
409}

411void WasmFunctionBuilder::WriteAsmWasmOffsetTable(ZoneBuffer* buffer) const {
412 if (asm_func_start_source_position_ == 0 && asm_offsets_.size() == 0) {
413 buffer->write_size(0);
414 return;
415 }
416 size_t locals_enc_size = LEBHelper::sizeof_u32v(locals_.Size());
417 size_t func_start_size =
418 LEBHelper::sizeof_u32v(asm_func_start_source_position_);
419 buffer->write_size(asm_offsets_.size() + locals_enc_size + func_start_size);
420 // Offset of the recorded byte offsets.
421 DCHECK_GE(kMaxUInt32, locals_.Size());
422 buffer->write_u32v(static_cast<uint32_t>(locals_.Size()));
423 // Start position of the function.
424 buffer->write_u32v(asm_func_start_source_position_);
425 buffer->write(asm_offsets_.begin(), asm_offsets_.size());
426}

657void WasmModuleBuilder::WriteTo(ZoneBuffer* buffer) const {
658 // == Emit magic =============================================================
659 buffer->write_u32(kWasmMagic);
660 buffer->write_u32(kWasmVersion);
661
662 // == Emit types =============================================================
663 if (!types_.empty()) {
664 size_t start = EmitSection(kTypeSectionCode, buffer);
665 // Every recursion group occupies one type entry.
666 size_t type_count = types_.size() + recursive_groups_.size();
667 // Types inside recursion groups occupy no additional type entry.
668 for (auto [first_index, size] : recursive_groups_) {
669 type_count -= size;
670 }
671
672 buffer->write_size(type_count);
673
674 const RecGroup* next_rec_group =
675 recursive_groups_.empty() ? nullptr : recursive_groups_.data();
676
677 for (uint32_t i = 0; i < types_.size(); i++) {
678 // Note: while loop, because recgroups can be empty.
679 while (next_rec_group && i == next_rec_group->start_index) {
680 buffer->write_u8(kWasmRecursiveTypeGroupCode);
681 buffer->write_u32v(next_rec_group->size);
682 next_rec_group = next_rec_group == &recursive_groups_.back()
683 ? nullptr
684 : next_rec_group + 1;
685 }
686
687 const TypeDefinition& type = types_[i];
688
689 if (type.supertype.valid()) {
690 buffer->write_u8(type.is_final ? kWasmSubtypeFinalCode
691 : kWasmSubtypeCode);
692 buffer->write_u8(1);
693 buffer->write_u32v(type.supertype);
694 } else if (!type.is_final) {
695 buffer->write_u8(kWasmSubtypeCode);
696 buffer->write_u8(0);
697 }
698 switch (type.kind) {
699 case TypeDefinition::kFunction: {
700 const FunctionSig* sig = type.function_sig;
701 buffer->write_u8(kWasmFunctionTypeCode);
702 buffer->write_size(sig->parameter_count());
703 for (auto param : sig->parameters()) {
704 WriteValueType(buffer, param);
705 }
706 buffer->write_size(sig->return_count());
707 for (auto ret : sig->returns()) {
708 WriteValueType(buffer, ret);
709 }
710 break;
711 }
712 case TypeDefinition::kStruct: {
713 const StructType* struct_type = type.struct_type;
714 buffer->write_u8(kWasmStructTypeCode);
715 buffer->write_size(struct_type->field_count());
716 for (uint32_t j = 0; j < struct_type->field_count(); j++) {
717 WriteValueType(buffer, struct_type->field(j));
718 buffer->write_u8(struct_type->mutability(j) ? 1 : 0);
719 }
720 break;
721 }
722 case TypeDefinition::kArray: {
723 const ArrayType* array_type = type.array_type;
724 buffer->write_u8(kWasmArrayTypeCode);
725 WriteValueType(buffer, array_type->element_type());
726 buffer->write_u8(array_type->mutability() ? 1 : 0);
727 break;
728 }
729 }
730 }
731
732 // Handle empty recursion groups defined after all types.
733 while (next_rec_group) {
734 DCHECK_EQ(types_.size(), next_rec_group->start_index);
735 DCHECK_EQ(0, next_rec_group->size);
736 buffer->write_u8(kWasmRecursiveTypeGroupCode);
737 buffer->write_u32v(0);
738 if (next_rec_group == &recursive_groups_.back()) break;
739 ++next_rec_group;
740 }
741
742 FixupSection(buffer, start);
743 }
744
745 // == Emit imports ===========================================================
746 if (global_imports_.size() + function_imports_.size() > 0) {
747 size_t start = EmitSection(kImportSectionCode, buffer);
748 buffer->write_size(global_imports_.size() + function_imports_.size());
749 for (auto import : global_imports_) {
750 buffer->write_string(import.module); // module name
751 buffer->write_string(import.name); // field name
752 buffer->write_u8(kExternalGlobal);
753 buffer->write_u8(import.type_code);
754 buffer->write_u8(import.mutability ? 1 : 0);
755 }
756 for (auto import : function_imports_) {
757 buffer->write_string(import.module); // module name
758 buffer->write_string(import.name); // field name
759 buffer->write_u8(kExternalFunction);
760 buffer->write_u32v(import.sig_index);
761 }
762 FixupSection(buffer, start);
763 }
764
765 // == Emit function signatures ===============================================
766 uint32_t num_function_names = 0;
767 if (!functions_.empty()) {
768 size_t start = EmitSection(kFunctionSectionCode, buffer);
769 buffer->write_size(functions_.size());
770 for (auto* function : functions_) {
771 function->WriteSignature(buffer);
772 if (!function->name_.empty()) ++num_function_names;
773 }
774 FixupSection(buffer, start);
775 }
776
777 // == Emit tables ============================================================
778 if (!tables_.empty()) {
779 size_t start = EmitSection(kTableSectionCode, buffer);
780 buffer->write_size(tables_.size());
781 for (const WasmTable& table : tables_) {
782 if (table.init) {
783 buffer->write_u8(0x40); // table-with-initializer
784 buffer->write_u8(0x00); // reserved byte
785 }
786 WriteValueType(buffer, table.type);
787 uint8_t limits_byte = (table.is_table64() ? 4 : 0) |
788 (table.is_shared ? 2 : 0) |
789 (table.has_maximum ? 1 : 0);
790 buffer->write_u8(limits_byte);
791 auto WriteValToBuffer = [&](uint32_t val) {
792 table.is_table64() ? buffer->write_u64v(val) : buffer->write_u32v(val);
793 };
794 WriteValToBuffer(table.min_size);
795 if (table.has_maximum) {
796 WriteValToBuffer(table.max_size);
797 }
798 if (table.init) {
799 WriteInitializerExpression(buffer, *table.init);
800 }
801 }
802 FixupSection(buffer, start);
803 }
804
805 // == Emit memory declaration ================================================
806 if (!memories_.empty()) {
807 size_t start = EmitSection(kMemorySectionCode, buffer);
808 buffer->write_size(memories_.size());
809 for (const WasmMemory& memory : memories_) {
810 uint8_t limits_byte = (memory.is_memory64() ? 4 : 0) |
811 (memory.is_shared ? 2 : 0) |
812 (memory.has_max_pages ? 1 : 0);
813 buffer->write_u8(limits_byte);
814 auto WriteValToBuffer = [&](uint32_t val) {
815 memory.is_memory64() ? buffer->write_u64v(val)
816 : buffer->write_u32v(val);
817 };
818 WriteValToBuffer(memory.min_pages);
819 if (memory.has_max_pages) {
820 WriteValToBuffer(memory.max_pages);
821 }
822 }
823 FixupSection(buffer, start);
824 }
825
826 // == Emit event section =====================================================
827 if (!tags_.empty()) {
828 size_t start = EmitSection(kTagSectionCode, buffer);
829 buffer->write_size(tags_.size());
830 for (ModuleTypeIndex type : tags_) {
831 buffer->write_u32v(kExceptionAttribute);
832 buffer->write_u32v(type);
833 }
834 FixupSection(buffer, start);
835 }
836
837 // == Emit globals ===========================================================
838 if (!globals_.empty()) {
839 size_t start = EmitSection(kGlobalSectionCode, buffer);
840 buffer->write_size(globals_.size());
841
842 for (const WasmGlobal& global : globals_) {
843 WriteValueType(buffer, global.type);
844 buffer->write_u8(global.mutability ? 1 : 0);
845 WriteInitializerExpression(buffer, global.init);
846 }
847 FixupSection(buffer, start);
848 }
849
850 // == Emit exports ===========================================================
851 if (!exports_.empty()) {
852 size_t start = EmitSection(kExportSectionCode, buffer);
853 buffer->write_size(exports_.size());
854 for (auto ex : exports_) {
855 buffer->write_string(ex.name);
856 buffer->write_u8(ex.kind);
857 switch (ex.kind) {
858 case kExternalFunction:
859 buffer->write_size(ex.index + function_imports_.size());
860 break;
861 case kExternalGlobal:
862 buffer->write_size(ex.index + global_imports_.size());
863 break;
864 case kExternalMemory:
865 case kExternalTable:
866 // The WasmModuleBuilder doesn't support importing tables or memories
867 // yet, so there is no index offset to add.
868 buffer->write_size(ex.index);
869 break;
870 case kExternalTag:
871 UNREACHABLE();
872 }
873 }
874 FixupSection(buffer, start);
875 }
876
877 // == Emit start function index ==============================================
878 if (start_function_index_ >= 0) {
879 size_t start = EmitSection(kStartSectionCode, buffer);
880 buffer->write_size(start_function_index_ + function_imports_.size());
881 FixupSection(buffer, start);
882 }
883
884 // == Emit element segments ==================================================
885 if (!element_segments_.empty()) {
886 size_t start = EmitSection(kElementSectionCode, buffer);
887 buffer->write_size(element_segments_.size());
888 for (const WasmElemSegment& segment : element_segments_) {
889 bool is_active = segment.status == WasmElemSegment::kStatusActive;
890 // We pick the most general syntax, i.e., we always explicitly emit the
891 // table index and the type, and use the expressions-as-elements syntax.
892 // The initial byte is one of 0x05, 0x06, and 0x07.
893 uint8_t kind_mask =
894 segment.status == WasmElemSegment::kStatusActive
895 ? 0b10
896 : segment.status == WasmElemSegment::kStatusDeclarative ? 0b11
897 : 0b01;
898 uint8_t expressions_as_elements_mask = 0b100;
899 buffer->write_u8(kind_mask | expressions_as_elements_mask);
900 if (is_active) {
901 buffer->write_u32v(segment.table_index);
902 WriteInitializerExpression(buffer, segment.offset);
903 }
904 WriteValueType(buffer, segment.type);
905 buffer->write_size(segment.entries.size());
906 for (const WasmElemSegment::Entry entry : segment.entries) {
907 uint8_t opcode =
908 entry.kind == WasmElemSegment::Entry::kGlobalGetEntry
909 ? kExprGlobalGet
910 : entry.kind == WasmElemSegment::Entry::kRefFuncEntry
911 ? kExprRefFunc
912 : kExprRefNull;
913 bool needs_function_offset =
914 segment.indexing_mode ==
915 WasmElemSegment::kRelativeToDeclaredFunctions &&
916 entry.kind == WasmElemSegment::Entry::kRefFuncEntry;
917 uint32_t index =
918 entry.index + (needs_function_offset
919 ? static_cast<uint32_t>(function_imports_.size())
920 : 0);
921 buffer->write_u8(opcode);
922 buffer->write_u32v(index);
923 buffer->write_u8(kExprEnd);
924 }
925 }
926 FixupSection(buffer, start);
927 }
928
929 // == Emit data segment count section ========================================
930 if (std::any_of(
931 data_segments_.begin(), data_segments_.end(),
932 [](const WasmDataSegment& segment) { return !segment.is_active; })) {
933 buffer->write_u8(kDataCountSectionCode);
934 buffer->write_u32v(1); // section length
935 buffer->write_u32v(static_cast<uint32_t>(data_segments_.size()));
936 }
937
938 // == Emit compilation hints section =========================================
939 bool emit_compilation_hints = false;
940 for (auto* fn : functions_) {
941 if (fn->hint_ != kNoCompilationHint) {
942 emit_compilation_hints = true;
943 break;
944 }
945 }
946 if (emit_compilation_hints) {
947 // Emit the section code.
948 buffer->write_u8(kUnknownSectionCode);
949 // Emit a placeholder for section length.
950 size_t start = buffer->reserve_u32v();
951 // Emit custom section name.
952 buffer->write_string(base::CStrVector("compilationHints"));
953 // Emit hint count.
954 buffer->write_size(functions_.size());
955 // Emit hint bytes.
956 for (auto* fn : functions_) {
957 uint8_t hint_byte =
958 fn->hint_ != kNoCompilationHint ? fn->hint_ : kDefaultCompilationHint;
959 buffer->write_u8(hint_byte);
960 }
961 FixupSection(buffer, start);
962 }
963
964 // == Emit code ==============================================================
965 if (!functions_.empty()) {
966 size_t start = EmitSection(kCodeSectionCode, buffer);
967 buffer->write_size(functions_.size());
968 for (auto* function : functions_) {
969 function->WriteBody(buffer);
970 }
971 FixupSection(buffer, start);
972 }
973
974 // == Emit data segments =====================================================
975 if (!data_segments_.empty()) {
976 size_t start = EmitSection(kDataSectionCode, buffer);
977 buffer->write_size(data_segments_.size());
978
979 for (auto segment : data_segments_) {
980 if (segment.is_active) {
981 buffer->write_u8(0); // linear memory segment
982 buffer->write_u8(kExprI32Const); // constant expression for dest
983 buffer->write_u32v(segment.dest);
984 buffer->write_u8(kExprEnd);
985 } else {
986 buffer->write_u8(kPassive);
987 }
988 buffer->write_u32v(static_cast<uint32_t>(segment.data.size()));
989 buffer->write(segment.data.data(), segment.data.size());
990 }
991 FixupSection(buffer, start);
992 }
993
994 // == Emit names =============================================================
995 if (num_function_names > 0 || !function_imports_.empty()) {
996 // Emit the section code.
997 buffer->write_u8(kUnknownSectionCode);
998 // Emit a placeholder for the length.
999 size_t start = buffer->reserve_u32v();
1000 // Emit the section string.
1001 buffer->write_string(base::CStrVector("name"));
1002 // Emit a subsection for the function names.
1003 buffer->write_u8(NameSectionKindCode::kFunctionCode);
1004 // Emit a placeholder for the subsection length.
1005 size_t functions_start = buffer->reserve_u32v();
1006 // Emit the function names.
1007 // Imports are always named.
1008 uint32_t num_imports = static_cast<uint32_t>(function_imports_.size());
1009 buffer->write_size(num_imports + num_function_names);
1010 uint32_t function_index = 0;
1011 for (; function_index < num_imports; ++function_index) {
1012 const WasmFunctionImport* import = &function_imports_[function_index];
1013 DCHECK(!import->name.empty());
1014 buffer->write_u32v(function_index);
1015 buffer->write_string(import->name);
1016 }
1017 if (num_function_names > 0) {
1018 for (auto* function : functions_) {
1019 DCHECK_EQ(function_index,
1020 function->func_index() + function_imports_.size());
1021 if (!function->name_.empty()) {
1022 buffer->write_u32v(function_index);
1023 buffer->write_string(function->name_);
1024 }
1025 ++function_index;
1026 }
1027 }
1028 FixupSection(buffer, functions_start);
1029 FixupSection(buffer, start);
1030 }
1031}
