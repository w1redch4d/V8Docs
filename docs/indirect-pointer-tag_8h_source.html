14// Defines the list of valid indirect pointer tags.
15//
16// When accessing a trusted/indirect pointer, an IndirectPointerTag must be
17// provided which indicates the expected instance type of the pointed-to
18// object. When the sandbox is enabled, this tag is used to ensure type-safe
19// access to objects referenced via trusted pointers: if the provided tag
20// doesn't match the tag of the object in the trusted pointer table, an
21// inaccessible pointer will be returned.
22//
23// We use the shifted instance type as tag and an AND-based type-checking
24// mechanism in the TrustedPointerTable, similar to the one used by the
25// ExternalPointerTable: the entry in the table is ORed with the tag and then
26// ANDed with the inverse of the tag upon access. This has the benefit that the
27// type check and the removal of the marking bit can be folded into a single
28// bitwise operations. However, it is not technically guaranteed that simply
29// using the instance type as tag works for this scheme as the bits of one
30// instance type may happen to be a superset of those of another instance type,
31// thereby causing the type check to incorrectly pass. As such, the chance of
32// getting "incompabitle" tags increases when adding more tags here so we may
33// at some point want to consider manually assigning tag values that are
34// guaranteed to work (similar for how we do it for ExternalPointerTags).
35
36constexpr int kIndirectPointerTagShift = 48;
37constexpr uint64_t kIndirectPointerTagMask = 0x7fff000000000000;
38constexpr uint64_t kTrustedPointerTableMarkBit = 0x8000000000000000;
39// We use a reserved bit for the free entry tag so that the
40// kUnknownIndirectPointerTag cannot untag free entries. Due to that, not all
41// tags in the kAllTagsForAndBasedTypeChecking are usable here (which is
42// ensured by static asserts below, see VALIDATE_INDIRECT_POINTER_TAG).
43// However, in practice this would probably be fine since the payload is a
44// table index, and so would likely always crash when treated as a pointer. As
45// such, if there is ever need for more tags, this can be reconsidered.
46// Note that we use a bit in the 2nd most significant byte here due to top byte
47// ignore (TBI), which allows dereferencing pointers even if bits in the most
48// significant byte are set.
49constexpr uint64_t kTrustedPointerTableFreeEntryBit = 0x0080000000000000;
50constexpr uint64_t kIndirectPointerTagMaskWithoutFreeEntryBit =
51 0x7f7f000000000000;
52

71// Shared trusted pointers are owned by the shared Isolate and stored in the
72// shared trusted pointer table associated with that Isolate, where they can
73// be accessed from multiple threads at the same time. The objects referenced
74// in this way must therefore always be thread-safe.
75// TODO(358918874): Consider having explicitly shared types (e.g.
76// `ExposedSharedTrustedObject`) and enforcing that shared tags are only ever
77// used with shared types.
78#define SHARED_TRUSTED_POINTER_TAG_LIST(V) \
79 V(kFirstSharedTrustedTag, 1) \
80 V(kLastSharedTrustedTag, 1)
81// Leave some space in the tag range here for future shared tags.

84// pointer table and can only be accessed when this Isolate is active.
85#define PER_ISOLATE_INDIRECT_POINTER_TAG_LIST(V) \

102#define MAKE_TAG(i) \
103 (kAllTagsForAndBasedTypeChecking[i] << kIndirectPointerTagShift)

105// TODO(saelo): consider renaming this to something like TypeTag or
106// InstanceTypeTag since that better captures what this represents.
107enum IndirectPointerTag : uint64_t {
108 // The null tag. Usually used to express the lack of a valid tag, for example
109 // in non-sandbox builds.
110 kIndirectPointerNullTag = 0,
111
112 // This tag can be used when an indirect pointer field can legitimately refer
113 // to objects of different types.
114 // NOTE: this tag effectively disables the built-in type-checking mechanism.
115 // As such, in virtually all cases the caller needs to perform runtime-type
116 // checks (i.e. IsXyzObject(obj))` afterwards which need to be able to
117 // correctly handle unexpected types. The last point is worth stressing
118 // further. As an example, the following code is NOT correct:
119 //
120 // auto obj = LoadTrustedPointerField<kUnknownIndirectPointerTag>(...);
121 // if (IsFoo(obj)) {
122 // Cast<Foo>(obj)->foo();
123 // } else if (IsBar(obj)) {
124 // Cast<Bar>(obj)->bar();
125 // } else {
126 // // Potential type confusion here!
127 // Cast<Baz>(obj)->baz();
128 // }
129 //
130 // This is because an attacker can swap trusted pointers and thereby cause an
131 // object of a different/unexpected type to be returned. Instead, in this
132 // case a CHECK can for example be used to make the code correct:
133 //
134 // // ...
135 // } else {
136 // // Must be a Baz object
137 // CHECK(IsBaz(obj));
138 // Cast<Baz>(obj)->baz();
139 // }
140 //
141 kUnknownIndirectPointerTag = kIndirectPointerTagMaskWithoutFreeEntryBit,
142
143 // Tag used internally by the trusted pointer table to mark free entries.
144 // See also the comment above kTrustedPointerTableFreeEntryBit for why this
145 // uses a dedicated bit.
146 kFreeTrustedPointerTableEntryTag = kTrustedPointerTableFreeEntryBit,
147
148// "Regular" tags. One per supported instance type.
149#define INDIRECT_POINTER_TAG_ENUM_DECL(name, tag_id) name = MAKE_TAG(tag_id),
150 INDIRECT_POINTER_TAG_LIST(INDIRECT_POINTER_TAG_ENUM_DECL)
151#undef INDIRECT_POINTER_TAG_ENUM_DECL
152};
153
154#define VALIDATE_INDIRECT_POINTER_TAG(name, tag_id) \
155 static_assert((name & kIndirectPointerTagMask) == name); \
156 static_assert((name & kIndirectPointerTagMaskWithoutFreeEntryBit) == name);
157INDIRECT_POINTER_TAG_LIST(VALIDATE_INDIRECT_POINTER_TAG)
158#undef VALIDATE_INDIRECT_POINTER_TAG
159static_assert((kFreeTrustedPointerTableEntryTag & kIndirectPointerTagMask) ==
160 kFreeTrustedPointerTableEntryTag);
161static_assert((kFreeTrustedPointerTableEntryTag &
162 kIndirectPointerTagMaskWithoutFreeEntryBit) == 0);

164// True if the external pointer must be accessed from the shared isolate's
165// external pointer table.
166V8_INLINE static constexpr bool IsSharedTrustedPointerType(
167 IndirectPointerTag tag) {
168 static_assert(IndirectPointerTag::kFirstSharedTrustedTag <=
169 IndirectPointerTag::kLastSharedTrustedTag);
170 return tag >= IndirectPointerTag::kFirstSharedTrustedTag &&
171 tag <= IndirectPointerTag::kLastSharedTrustedTag;
172}
173
174V8_INLINE static constexpr bool IsPerIsolateTrustedPointerType(
175 IndirectPointerTag tag) {
176 static_assert(IndirectPointerTag::kFirstPerIsolateTrustedTag <=
177 IndirectPointerTag::kLastPerIsolateTrustedTag);
178 return tag >= IndirectPointerTag::kFirstPerIsolateTrustedTag &&
179 tag <= IndirectPointerTag::kLastPerIsolateTrustedTag;
180}

182V8_INLINE constexpr bool IsValidIndirectPointerTag(IndirectPointerTag tag) {
183 return IsPerIsolateTrustedPointerType(tag) || IsSharedTrustedPointerType(tag);
184}

186// Migrating objects into trusted space is typically performed in multiple
187// steps, where all references to the object from inside the sandbox are first
188// changed to indirect pointers before actually moving the object out of the
189// sandbox. As we have CHECKs that trusted pointer table entries point outside
190// of the sandbox, we need this helper function to disable that CHECK for
191// objects that are in the process of being migrated into trusted space.
192V8_INLINE constexpr bool IsTrustedSpaceMigrationInProgressForObjectsWithTag(
193 IndirectPointerTag tag) {
194 return false;
195}

197// The null tag is also considered an invalid tag since no indirect pointer
198// field should be using this tag.
199static_assert(!IsValidIndirectPointerTag(kIndirectPointerNullTag));

201V8_INLINE IndirectPointerTag
202IndirectPointerTagFromInstanceType(InstanceType instance_type) {
203 switch (instance_type) {
204 case CODE_TYPE:
205 return kCodeIndirectPointerTag;
206 case BYTECODE_ARRAY_TYPE:
207 return kBytecodeArrayIndirectPointerTag;
208 case INTERPRETER_DATA_TYPE:
209 return kInterpreterDataIndirectPointerTag;
210 case UNCOMPILED_DATA_WITHOUT_PREPARSE_DATA_TYPE:
211 case UNCOMPILED_DATA_WITH_PREPARSE_DATA_TYPE:
212 case UNCOMPILED_DATA_WITHOUT_PREPARSE_DATA_WITH_JOB_TYPE:
213 case UNCOMPILED_DATA_WITH_PREPARSE_DATA_AND_JOB_TYPE:
214 // TODO(saelo): Consider adding support for inheritance hierarchies in
215 // our tag checking mechanism.
216 return kUncompiledDataIndirectPointerTag;
217 case ATOM_REG_EXP_DATA_TYPE:
218 case IR_REG_EXP_DATA_TYPE:
219 // TODO(saelo): Consider adding support for inheritance hierarchies in
220 // our tag checking mechanism.
221 return kRegExpDataIndirectPointerTag;
222#if V8_ENABLE_WEBASSEMBLY
223 case WASM_DISPATCH_TABLE_TYPE:
224 return kWasmDispatchTableIndirectPointerTag;
225 case WASM_TRUSTED_INSTANCE_DATA_TYPE:
226 return kWasmTrustedInstanceDataIndirectPointerTag;
227 case WASM_INTERNAL_FUNCTION_TYPE:
228 return kWasmInternalFunctionIndirectPointerTag;
229 case WASM_FUNCTION_DATA_TYPE:
230 case WASM_EXPORTED_FUNCTION_DATA_TYPE:
231 case WASM_JS_FUNCTION_DATA_TYPE:
232 case WASM_CAPI_FUNCTION_DATA_TYPE:
233 // TODO(saelo): Consider adding support for inheritance hierarchies in
234 // our tag checking mechanism.
235 return kWasmFunctionDataIndirectPointerTag;
236#endif // V8_ENABLE_WEBASSEMBLY
237 default:
238 UNREACHABLE();
239 }

242V8_INLINE InstanceType
243InstanceTypeFromIndirectPointerTag(IndirectPointerTag tag) {
244 DCHECK(IsValidIndirectPointerTag(tag));
245 switch (tag) {
246#define CASE(name, instance_type, tag_id) \
247 case MAKE_TAG(tag_id): \
248 return instance_type; \
249 break;
250#undef CASE
251 default:
252 UNREACHABLE();
253 }
254}
