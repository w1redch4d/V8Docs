24// interface as AstNodeFactory, so ParserBase doesn't need to care which one is

844// Preparsing checks a JavaScript program and emits preparse-data that helps
845// a later parsing to be faster.
846// See preparse-data-format.h for the data format.
847
848// The PreParser checks that the syntax follows the grammar for JavaScript,
849// and collects some information about the program along the way.
850// The grammar check is only performed in order to understand the program
851// sufficiently to deduce some information about it, that can be used
852// to speed up later parsing. Finding errors is not the goal of pre-parsing,
853// rather it is to speed up properly written and correct programs.
854// That means that contextual checks (like a label being declared where
855// it is used) are generally omitted.

857 friend class ParserBase<PreParser>;

1022 Variable* DeclareVariableName(const AstRawString* name, VariableMode mode,
1023 Scope* scope, bool* was_added,
1024 int position = kNoSourcePosition,
1025 VariableKind kind = NORMAL_VARIABLE) {
1026 DCHECK(!IsPrivateMethodOrAccessorVariableMode(mode));
1027 Variable* var = scope->DeclareVariableName(name, mode, was_added, kind);
1028 if (var == nullptr) {
1029 ReportUnidentifiableError();
1030 if (!IsLexicalVariableMode(mode)) scope = scope->GetDeclarationScope();
1031 var = scope->LookupLocal(name);
1032 } else if (var->scope() != scope) {
1033 DCHECK_NE(kNoSourcePosition, position);
1034 DCHECK_EQ(VariableMode::kVar, mode);
1035 Declaration* nested_declaration =
1036 factory()->ast_node_factory()->NewNestedVariableDeclaration(scope,
1037 position);
1038 nested_declaration->set_var(var);
1039 var->scope()->declarations()->Add(nested_declaration);
1040 }
1041 return var;
1042 }
