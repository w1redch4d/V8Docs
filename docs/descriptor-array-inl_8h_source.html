53InternalIndex DescriptorArray::Search(Tagged<Name> name, int valid_descriptors,
54 bool concurrent_search) {
55 DCHECK(IsUniqueName(name));
56 SLOW_DCHECK_IMPLIES(!concurrent_search, IsSortedNoDuplicates());
57
58 if (valid_descriptors == 0) {
59 return InternalIndex::NotFound();
60 }
61
62 // Do linear search for small arrays, and for searches in the background
63 // thread.
64 const int kMaxElementsForLinearSearch = 8;
65 if (valid_descriptors <= kMaxElementsForLinearSearch || concurrent_search) {
66 return LinearSearch(name, valid_descriptors);
67 }
68
69 return BinarySearch(name, valid_descriptors);
70}

72InternalIndex DescriptorArray::BinarySearch(Tagged<Name> name,
73 int valid_descriptors) {
74 // We have to binary search all descriptors, not just valid ones, since the
75 // binary search ordering is across all descriptors.
76 int end = number_of_descriptors();
77 uint32_t hash = name->hash();
78
79 // Find the first descriptor whose key's hash is greater-than-or-equal-to the
80 // search hash.
81 int number = *std::ranges::lower_bound(std::views::iota(0, end), hash,
82 std::less<>(), [&](int i) {
83 Tagged<Name> entry = GetSortedKey(i);
84 return entry->hash();
85 });
86
87 // There may have been hash collisions, so search for the name from the first
88 // index until the first non-matching hash.
89 for (; number < end; ++number) {
90 InternalIndex index(GetSortedKeyIndex(number));
91 Tagged<Name> entry = GetKey(index);
92 if (entry == name) {
93 // If we found the entry, but it's outside the owned descriptors of the
94 // caller, return not found.
95 if (index.as_int() >= valid_descriptors) {
96 return InternalIndex::NotFound();
97 }
98 return index;
99 }
100 if (entry->hash() != hash) {
101 return InternalIndex::NotFound();
102 }
103 }
104
105 return InternalIndex::NotFound();
106}

108InternalIndex DescriptorArray::LinearSearch(Tagged<Name> name,
109 int valid_descriptors) {
110 DCHECK_LE(valid_descriptors, number_of_descriptors());

112 InternalIndex index(i);

127 for (int desc_index = field_index; desc_index < valid_descriptors;

145InternalIndex DescriptorArray::SearchWithCache(Isolate* isolate,
146 Tagged<Name> name,
147 Tagged<Map> map) {
148 DCHECK(IsUniqueName(name));
149 int number_of_own_descriptors = map->NumberOfOwnDescriptors();
150 if (number_of_own_descriptors == 0) return InternalIndex::NotFound();
151
152 DescriptorLookupCache* cache = isolate->descriptor_lookup_cache();
153 int number = cache->Lookup(map, name);
154
155 if (number == DescriptorLookupCache::kAbsent) {
156 InternalIndex result = Search(name, number_of_own_descriptors);
157 number = result.is_found() ? result.as_int() : DescriptorArray::kNotFound;
158 cache->Update(map, name, number);
159 }
160 if (number == DescriptorArray::kNotFound) return InternalIndex::NotFound();
161 return InternalIndex(number);
162}

296void DescriptorArray::Append(Descriptor* desc) {
297 DisallowGarbageCollection no_gc;
298 int descriptor_number = number_of_descriptors();
299 DCHECK_LE(descriptor_number + 1, number_of_all_descriptors());
300 set_number_of_descriptors(descriptor_number + 1);
301 Set(InternalIndex(descriptor_number), desc);
302
303 uint32_t desc_hash = desc->GetKey()->hash();
304 // Hash value can't be zero, see String::ComputeAndSetHash()
305 uint32_t collision_hash = 0;
306
307 int insertion;
308
309 for (insertion = descriptor_number; insertion > 0; --insertion) {
310 Tagged<Name> key = GetSortedKey(insertion - 1);
311 collision_hash = key->hash();
312 if (collision_hash <= desc_hash) break;
313 SetSortedKey(insertion, GetSortedKeyIndex(insertion - 1));
314 }
315
316 SetSortedKey(insertion, descriptor_number);
317
318 if (V8_LIKELY(collision_hash != desc_hash)) return;
319
320 CheckNameCollisionDuringInsertion(desc, desc_hash, insertion);
321}

330bool DescriptorArrayMarkingState::TryUpdateIndicesToMark(
331 unsigned gc_epoch, Tagged<DescriptorArray> array,
332 DescriptorIndex index_to_mark) {
333 const auto current_epoch = gc_epoch & Epoch::kMask;
334 while (true) {
335 const RawGCStateType raw_gc_state = array->raw_gc_state(kRelaxedLoad);
336 const auto epoch_from_state = Epoch::decode(raw_gc_state);
337 RawGCStateType new_raw_gc_state = 0;
338 if (current_epoch != epoch_from_state) {
339 // If the epochs do not match, then either the raw_gc_state is zero
340 // (freshly allocated descriptor array) or the epoch from value lags
341 // by 1.
342 DCHECK_IMPLIES(raw_gc_state != 0,
343 Epoch::decode(epoch_from_state + 1) == current_epoch);
344 new_raw_gc_state = NewState(current_epoch, 0, index_to_mark);
345 } else {
346 const DescriptorIndex already_marked = Marked::decode(raw_gc_state);
347 const DescriptorIndex delta = Delta::decode(raw_gc_state);
348 if ((already_marked + delta) >= index_to_mark) {
349 return false;
350 }
351 new_raw_gc_state = NewState(current_epoch, already_marked,
352 index_to_mark - already_marked);
353 }
354 if (SwapState(array, raw_gc_state, new_raw_gc_state)) {
355 return true;
356 }
357 }
358}

360// static
361std::pair<DescriptorArrayMarkingState::DescriptorIndex,
362 DescriptorArrayMarkingState::DescriptorIndex>
363DescriptorArrayMarkingState::AcquireDescriptorRangeToMark(
364 unsigned gc_epoch, Tagged<DescriptorArray> array) {
365 const auto current_epoch = gc_epoch & Epoch::kMask;
366 while (true) {
367 const RawGCStateType raw_gc_state = array->raw_gc_state(kRelaxedLoad);
368 const DescriptorIndex marked = Marked::decode(raw_gc_state);
369 const DescriptorIndex delta = Delta::decode(raw_gc_state);
370 // We may encounter an array here that was merely pushed to the marker. In
371 // such a case, we process all descriptors (if we succeed). The cases to
372 // check are:
373 // 1. Epoch mismatch: Happens when descriptors survive a GC cycle.
374 // 2. Epoch matches but marked/delta is 0: Can happen when descriptors are
375 // newly allocated in the current cycle.
376 if (current_epoch != Epoch::decode(raw_gc_state) || (marked + delta) == 0) {
377 // In case number of descriptors is 0 and we reach the array through roots
378 // marking, mark also slack to get a proper transition from 0 marked to X
379 // marked. Otherwise, we would need to treat the state [0,0[ for marked
380 // and delta as valid state which leads to double-accounting through the
381 // marking barrier (when nof>1 in the barrier).
382 const int16_t number_of_descriptors =
383 array->number_of_descriptors() ? array->number_of_descriptors()
384 : array->number_of_all_descriptors();
385 DCHECK_GT(number_of_descriptors, 0);
386 if (SwapState(array, raw_gc_state,
387 NewState(current_epoch, number_of_descriptors, 0))) {
388 return {0, number_of_descriptors};
389 }
390 continue;
391 }
392
393 // The delta is 0, so everything has been processed. Return the marked
394 // indices.
395 if (delta == 0) {
396 return {marked, marked};
397 }
398
399 if (SwapState(array, raw_gc_state,
400 NewState(current_epoch, marked + delta, 0))) {
401 return {marked, marked + delta};
402 }
403 }
404}
