19// V8 objects are defined as:
20//
21// V8_OBJECT class Foo : public Base {
22// ...
23// } V8_OBJECT_END;
24//
25// These macros are to enable packing down to 4-byte alignment (i.e. int32
26// alignment, since we have int32 fields), and to add warnings which ensure that
27// there is no unwanted within-object padding.
28#if V8_CC_GNU
29#define V8_OBJECT_PUSH \
30 _Pragma("pack(push)") _Pragma("pack(4)") _Pragma("GCC diagnostic push") \
31 _Pragma("GCC diagnostic error \"-Wpadded\"")
32#define V8_OBJECT_POP _Pragma("pack(pop)") _Pragma("GCC diagnostic pop")
33#elif V8_CC_MSVC
34#define V8_OBJECT_PUSH \
35 __pragma(pack(push)) __pragma(pack(4)) __pragma(warning(push)) \
36 __pragma(warning(default : 4820))
37#define V8_OBJECT_POP __pragma(pack(pop)) __pragma(warning(pop))
38#else
39#error Unsupported compiler
40#endif
41
42#define V8_OBJECT V8_OBJECT_PUSH
43// Compilers wants the pragmas to be a new statement, but we prefer to have
44// V8_OBJECT_END look like part of the definition. Insert a semicolon before the
45// pragma to make the compilers happy, and use static_assert(true) to swallow
46// the next semicolon.
47#define V8_OBJECT_END \
48 ; \
49 V8_OBJECT_POP static_assert(true)
50
51#define V8_OBJECT_INNER_CLASS V8_OBJECT_POP
52#define V8_OBJECT_INNER_CLASS_END \
53 ; \
54 V8_OBJECT_PUSH static_assert(true)
