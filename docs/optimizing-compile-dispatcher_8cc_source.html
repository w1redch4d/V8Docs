25class OptimizingCompileDispatcher::CompileTask : public v8::JobTask {
26 public:
27 explicit CompileTask(Isolate* isolate,
28 OptimizingCompileDispatcher* dispatcher)
29 : isolate_(isolate),
30 worker_thread_runtime_call_stats_(
31 isolate->counters()->worker_thread_runtime_call_stats()),
32 dispatcher_(dispatcher) {}
33
34 void Run(JobDelegate* delegate) override {
35 LocalIsolate local_isolate(isolate_, ThreadKind::kBackground);
36 DCHECK(local_isolate.heap()->IsParked());
37
38 {
39 RCS_SCOPE(&local_isolate,
40 RuntimeCallCounterId::kOptimizeBackgroundDispatcherJob);
41
42 TimerEventScope<TimerEventRecompileConcurrent> timer(isolate_);
43 while (!delegate->ShouldYield()) {
44 TurbofanCompilationJob* job = dispatcher_->NextInput(&local_isolate);
45 if (!job) break;
46 TRACE_EVENT_WITH_FLOW0(
47 TRACE_DISABLED_BY_DEFAULT("v8.compile"), "V8.OptimizeBackground",
48 job->trace_id(),
49 TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
50
51 if (dispatcher_->recompilation_delay_ != 0) {
52 base::OS::Sleep(base::TimeDelta::FromMilliseconds(
53 dispatcher_->recompilation_delay_));
54 }
55
56 dispatcher_->CompileNext(job, &local_isolate);
57 }
58 }
59 }
60
61 size_t GetMaxConcurrency(size_t worker_count) const override {
62 size_t num_tasks = dispatcher_->input_queue_.Length() + worker_count;
63 size_t max_threads = v8_flags.concurrent_turbofan_max_threads;
64 if (max_threads > 0) {
65 return std::min(max_threads, num_tasks);
66 }
67 return num_tasks;
68 }
69
70 private:
71 Isolate* isolate_;
72 WorkerThreadRuntimeCallStats* worker_thread_runtime_call_stats_;
73 OptimizingCompileDispatcher* dispatcher_;
74};

90void OptimizingCompileDispatcher::CompileNext(TurbofanCompilationJob* job,
91 LocalIsolate* local_isolate) {
92 if (!job) return;
93
94 // The function may have already been optimized by OSR. Simply continue.
95 CompilationJob::Status status =
96 job->ExecuteJob(local_isolate->runtime_call_stats(), local_isolate);
97 USE(status); // Prevent an unused-variable error.
98
99 {
100 // The function may have already been optimized by OSR. Simply continue.
101 // Use a mutex to make sure that functions marked for install
102 // are always also queued.
103 base::SpinningMutexGuard access_output_queue_(&output_queue_mutex_);
104 output_queue_.push_back(job);
105 }
106
107 if (finalize()) isolate_->stack_guard()->RequestInstallCode();
108}

110void OptimizingCompileDispatcher::FlushOutputQueue() {
111 for (;;) {
112 std::unique_ptr<TurbofanCompilationJob> job;
113 {
114 base::SpinningMutexGuard access_output_queue_(&output_queue_mutex_);
115 if (output_queue_.empty()) return;
116 job.reset(output_queue_.front());
117 output_queue_.pop_front();
118 }

120 Compiler::DisposeTurbofanCompilationJob(isolate_, job.get());
121 }

139void OptimizingCompileDispatcher::AwaitCompileTasks() {
140 {
141 AllowGarbageCollection allow_before_parking;
142 isolate_->main_thread_local_isolate()->ExecuteMainThreadWhileParked(
143 [this]() { job_handle_->Join(); });
144 }
145 // Join kills the job handle, so drop it and post a new one.
146 job_handle_ = V8::GetCurrentPlatform()->PostJob(
147 kTaskPriority, std::make_unique<CompileTask>(isolate_, this));

149#ifdef DEBUG
150 CHECK_EQ(input_queue_.Length(), 0);
151#endif // DEBUG

171void OptimizingCompileDispatcher::Stop() {
172 HandleScope handle_scope(isolate_);
173 FlushQueues(BlockingBehavior::kBlock);
174 // At this point the optimizing compiler thread's event loop has stopped.
175 // There is no need for a mutex when reading input_queue_length_.

179void OptimizingCompileDispatcher::InstallOptimizedFunctions() {
180 HandleScope handle_scope(isolate_);
181
182 for (;;) {
183 std::unique_ptr<TurbofanCompilationJob> job;
184 {
185 base::SpinningMutexGuard access_output_queue_(&output_queue_mutex_);
186 if (output_queue_.empty()) return;
187 job.reset(output_queue_.front());
188 output_queue_.pop_front();
189 }
190 OptimizedCompilationInfo* info = job->compilation_info();
191 DirectHandle<JSFunction> function(*info->closure(), isolate_);
192
193 // If another racing task has already finished compiling and installing the
194 // requested code kind on the function, throw out the current job.
195 if (!info->is_osr() &&
196 function->HasAvailableCodeKind(isolate_, info->code_kind())) {
197 if (v8_flags.trace_concurrent_recompilation) {
198 PrintF(" ** Aborting compilation for ");
199 ShortPrint(*function);
200 PrintF(" as it has already been optimized.\n");
201 }
202 Compiler::DisposeTurbofanCompilationJob(isolate_, job.get());
203 continue;
204 }
205 // Discard code compiled for a discarded native context without
206 // finalization.
207 if (function->native_context()->global_object()->IsDetached()) {
208 Compiler::DisposeTurbofanCompilationJob(isolate_, job.get());
209 continue;
210 }
211
212 Compiler::FinalizeTurbofanCompilationJob(job.get(), isolate_);
213 }

216int OptimizingCompileDispatcher::InstallGeneratedBuiltins(int installed_count) {
217 // Builtin generation needs to be deterministic, meaning heap allocations must
218 // happen in a deterministic order. To ensure determinism with concurrent
219 // compilation, only finalize contiguous builtins in ascending order of their
220 // finalization order, which is set at job creation time.
221
222 CHECK(isolate_->IsGeneratingEmbeddedBuiltins());
223
224 base::SpinningMutexGuard access_output_queue_(&output_queue_mutex_);
225
226 std::sort(output_queue_.begin(), output_queue_.end(),
227 [](const TurbofanCompilationJob* job1,
228 const TurbofanCompilationJob* job2) {
229 return job1->FinalizeOrder() < job2->FinalizeOrder();
230 });
231
232 while (!output_queue_.empty()) {
233 int current = output_queue_.front()->FinalizeOrder();
234 CHECK_EQ(installed_count, current);
235 std::unique_ptr<TurbofanCompilationJob> job(output_queue_.front());
236 output_queue_.pop_front();
237 CHECK_EQ(CompilationJob::SUCCEEDED, job->FinalizeJob(isolate_));
238 installed_count = current + 1;
239 }
240 return installed_count;
241}

260void OptimizingCompileDispatcherQueue::Prioritize(
261 Tagged<SharedFunctionInfo> function) {
262 base::SpinningMutexGuard access(&mutex_);
263 if (length_ > 1) {
264 for (int i = length_ - 1; i > 1; --i) {
265 if (*queue_[QueueIndex(i)]->compilation_info()->shared_info() ==
266 function) {
267 std::swap(queue_[QueueIndex(i)], queue_[QueueIndex(0)]);
268 return;
269 }
270 }
271 }
272}
