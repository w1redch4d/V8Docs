76Address RelocInfo::target_address_address() {
77 DCHECK(HasTargetAddressAddress());
78 // Read the address of the word containing the target_address in an
79 // instruction stream.
80 // The only architecture-independent user of this function is the serializer.
81 // The serializer uses it to find out how many raw bytes of instruction to
82 // output before the next target.
83 // For an instruction like LUI/ORI where the target bits are mixed into the
84 // instruction bits, the size of the target will be zero, indicating that the
85 // serializer should not step forward in memory after a target is resolved
86 // and written. In this case the target_address_address function should
87 // return the end of the instructions to be patched, allowing the
88 // deserializer to deserialize the instructions as raw bytes and put them in
89 // place, ready to be patched with the target. After jump optimization,
90 // that is the address of the instruction that follows J/JAL/JR/JALR
91 // instruction.
92 return pc_ + Assembler::kInstructionsFor64BitConstant * kInstrSize;
93}

104void Assembler::set_target_internal_reference_encoded_at(
105 Address pc, Address target, WritableJitAllocation& jit_allocation) {
106 // Encoded internal references are j/jal instructions.
107 Instr instr = Assembler::instr_at(pc + 0 * kInstrSize);
108
109 uint64_t imm28 = target & static_cast<uint64_t>(kImm28Mask);
110
111 instr &= ~kImm26Mask;
112 uint64_t imm26 = imm28 >> 2;
113 DCHECK(is_uint26(imm26));
114
115 instr_at_put(pc, instr | (imm26 & kImm26Mask), &jit_allocation);
116 // Currently used only by deserializer, and all code will be flushed
117 // after complete deserialization, no need to flush on each reference.
118}

175Address RelocInfo::target_internal_reference() {
176 if (rmode_ == INTERNAL_REFERENCE) {
177 return Memory<Address>(pc_);
178 } else {
179 // Encoded internal references are j/jal instructions.
180 DCHECK(rmode_ == INTERNAL_REFERENCE_ENCODED);
181 Instr instr = Assembler::instr_at(pc_ + 0 * kInstrSize);
182 instr &= kImm26Mask;
183 uint64_t imm28 = instr << 2;
184 uint64_t segment = pc_ & ~static_cast<uint64_t>(kImm28Mask);
185 return static_cast<Address>(segment | imm28);
186 }
187}

206uint32_t Assembler::uint32_constant_at(Address pc, Address constant_pool) {
207 Instr instr0 = instr_at(pc);
208 Instr instr1 = instr_at(pc + 1 * kInstrSize);
209
210 DCHECK((GetOpcodeField(instr0) == LUI) && (GetOpcodeField(instr1) == ORI));
211
212 // Assemble the 32 bit value.
213 uint32_t upper16 = GetImmediate16(instr0) << 16;
214 uint32_t lower16 = GetImmediate16(instr1);
215 uint32_t addr = upper16 | lower16;
216
217 return addr;
218}

220void Assembler::set_uint32_constant_at(Address pc, Address constant_pool,
221 uint32_t new_constant,
222 WritableJitAllocation* jit_allocation,
223 ICacheFlushMode icache_flush_mode) {
224 Instr instr1 = instr_at(pc + kInstrSize);
225 uint32_t rt_code = GetRt(instr1);
226
227#ifdef DEBUG
228 // Check we have the result from a li macro-instruction.
229 Instr instr0 = instr_at(pc);
230 DCHECK((GetOpcodeField(instr0) == LUI) && (GetOpcodeField(instr1) == ORI) &&
231 (GetRt(instr0) == rt_code));
232#endif
233
234 // Must use 2 instructions to insure patchable 32-bit value.
235 // lui rt, upper-16.
236 // ori rt, rt, lower-16.
237 Instr new_instr0 =
238 LUI | (rt_code << kRtShift) | ((new_constant >> 16) & kImm16Mask);
239 Instr new_instr1 = ORI | (rt_code << kRtShift) | (rt_code << kRsShift) |
240 (new_constant & kImm16Mask);
241 instr_at_put(pc, new_instr0, jit_allocation);
242 instr_at_put(pc + kInstrSize, new_instr1, jit_allocation);
243
244 if (icache_flush_mode != SKIP_ICACHE_FLUSH) {
245 FlushInstructionCache(pc, 2 * kInstrSize);
246 }

258void Assembler::CheckForEmitInForbiddenSlot() {
259 if (!is_buffer_growth_blocked()) {
260 CheckBuffer();
261 }
262 if (IsPrevInstrCompactBranch()) {
263 // Nop instruction to precede a CTI in forbidden slot:
264 Instr nop = SPECIAL | SLL;
265 *reinterpret_cast<Instr*>(pc_) = nop;
266 pc_ += kInstrSize;
267
268 ClearCompactBranchState();
269 }
270}

272void Assembler::EmitHelper(Instr x, CompactBranchType is_compact_branch) {
273 if (IsPrevInstrCompactBranch()) {
274 if (Instruction::IsForbiddenAfterBranchInstr(x)) {
275 // Nop instruction to precede a CTI in forbidden slot:
276 Instr nop = SPECIAL | SLL;
277 *reinterpret_cast<Instr*>(pc_) = nop;
278 pc_ += kInstrSize;
279 }
280 ClearCompactBranchState();
281 }
282 *reinterpret_cast<Instr*>(pc_) = x;
283 pc_ += kInstrSize;
284 if (is_compact_branch == CompactBranchType::COMPACT_BRANCH) {
285 EmittedCompactBranchInstruction();
286 }
287 CheckTrampolinePoolQuick();
288}

302 *reinterpret_cast<uint8_t*>(pc_) = x;
303 pc_ += sizeof(x);
304 if (reinterpret_cast<intptr_t>(pc_) % kInstrSize == 0) {
305 CheckTrampolinePoolQuick();
306 }

309void Assembler::emit(Instr x, CompactBranchType is_compact_branch) {
310 if (!is_buffer_growth_blocked()) {
311 CheckBuffer();
312 }
313 EmitHelper(x, is_compact_branch);
314}
