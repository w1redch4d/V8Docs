67i::Handle<i::String> GetBigIntStringPresentationHandle(
68 i::Isolate* i_isolate, i::DirectHandle<i::BigInt> i_bigint) {
69 // For large BigInts computing the decimal string representation
70 // can take a long time, so we go with hexadecimal in that case.
71 int radix = (i_bigint->Words64Count() > 100 * 1000) ? 16 : 10;
72 i::Handle<i::String> string_value =
73 i::BigInt::ToString(i_isolate, i_bigint, radix, i::kDontThrow)
74 .ToHandleChecked();
75 if (radix == 16) {
76 if (i_bigint->IsNegative()) {
77 string_value =
78 i_isolate->factory()
79 ->NewConsString(
80 i_isolate->factory()->NewStringFromAsciiChecked("-0x"),
81 i_isolate->factory()->NewProperSubString(
82 string_value, 1, string_value->length() - 1))
83 .ToHandleChecked();
84 } else {
85 string_value =
86 i_isolate->factory()
87 ->NewConsString(
88 i_isolate->factory()->NewStringFromAsciiChecked("0x"),
89 string_value)
90 .ToHandleChecked();
91 }
92 }
93 return string_value;
94}

138 auto receiver = Utils::OpenHandle(*function);
139 auto i_isolate = receiver->GetIsolate();
140 ENTER_V8_NO_SCRIPT_NO_EXCEPTION(i_isolate);
141 if (IsJSBoundFunction(*receiver)) {
142 return Utils::ToLocal(
143 i::JSBoundFunction::ToString(i::Cast<i::JSBoundFunction>(receiver)));
144 }
145 if (IsJSFunction(*receiver)) {
146 auto js_function = i::Cast<i::JSFunction>(receiver);
147#if V8_ENABLE_WEBASSEMBLY

149 i::DirectHandle<i::WasmExportedFunctionData> function_data(
150 js_function->shared()->wasm_exported_function_data(), i_isolate);
151 int func_index = function_data->function_index();
152 i::DirectHandle<i::WasmTrustedInstanceData> instance_data(
153 function_data->instance_data(), i_isolate);
154 if (instance_data->module()->origin == i::wasm::kWasmOrigin) {
155 // For asm.js functions, we can still print the source
156 // code (hopefully), so don't bother with them here.
157 auto debug_name =
158 i::GetWasmFunctionDebugName(i_isolate, instance_data, func_index);
159 i::IncrementalStringBuilder builder(i_isolate);
160 builder.AppendCStringLiteral("function ");
161 builder.AppendString(debug_name);
162 builder.AppendCStringLiteral("() { [native code] }");
163 return Utils::ToLocal(builder.Finish().ToHandleChecked());
164 }
165 }
166#endif // V8_ENABLE_WEBASSEMBLY
167 return Utils::ToLocal(i::JSFunction::ToString(js_function));
168 }
169 return Utils::ToLocal(
170 receiver->GetIsolate()->factory()->function_native_code_string());
171}

198using FlagFilter = std::function<bool(i::IsStaticFlag)>;
199using VariableModeFilter = std::function<bool(i::VariableMode)>;
200using ContextLocalIterator = std::function<void(
201 i::VariableMode, i::Handle<i::String>, i::Handle<i::Object>)>;
202
203void ForEachContextLocal(i::Isolate* isolate,
204 i::DirectHandle<i::Context> context,
205 const VariableModeFilter& var_mode_filter,
206 const FlagFilter& flag_filter,
207 const ContextLocalIterator& context_local_it) {
208 DCHECK_NO_SCRIPT_NO_EXCEPTION(isolate);
209 i::Handle<i::ScopeInfo> scope_info(context->scope_info(), isolate);
210 for (auto it : i::ScopeInfo::IterateLocalNames(scope_info)) {
211 i::Handle<i::String> name(it->name(), isolate);
212 i::VariableMode mode = scope_info->ContextLocalMode(it->index());
213 if (!var_mode_filter(mode)) {
214 continue;
215 }
216 i::IsStaticFlag flag = scope_info->ContextLocalIsStaticFlag(it->index());
217 if (!flag_filter(flag)) {
218 continue;
219 }
220 int context_index = scope_info->ContextHeaderLength() + it->index();
221 i::Handle<i::Object> slot_value(context->get(context_index), isolate);
222 context_local_it(mode, name, slot_value);
223 }
224}

228bool GetPrivateMembers(Local<Context> context, Local<Object> object, int filter,
229 LocalVector<Value>* names_out,
230 LocalVector<Value>* values_out) {
231 i::Isolate* isolate = reinterpret_cast<i::Isolate*>(context->GetIsolate());
232 API_RCS_SCOPE(isolate, debug, GetPrivateMembers);
233 ENTER_V8_NO_SCRIPT_NO_EXCEPTION(isolate);
234
235 bool include_methods =
236 filter & static_cast<int>(PrivateMemberFilter::kPrivateMethods);
237 bool include_fields =
238 filter & static_cast<int>(PrivateMemberFilter::kPrivateFields);
239 bool include_accessors =
240 filter & static_cast<int>(PrivateMemberFilter::kPrivateAccessors);
241 bool include_methods_or_accessors = include_methods || include_accessors;
242
243 auto var_mode_filter =
244 include_methods
245 ? (include_accessors ? i::IsPrivateMethodOrAccessorVariableMode
246 : i::IsPrivateMethodVariableMode)
247 : i::IsPrivateAccessorVariableMode;
248 auto constexpr instance_filter = [](i::IsStaticFlag flag) {
249 return flag == i::IsStaticFlag::kNotStatic;
250 };
251 auto constexpr static_filter = [](i::IsStaticFlag flag) {
252 return flag == i::IsStaticFlag::kStatic;
253 };
254
255 i::DirectHandle<i::JSReceiver> receiver = Utils::OpenDirectHandle(*object);
256
257 i::PropertyFilter key_filter =
258 static_cast<i::PropertyFilter>(i::PropertyFilter::PRIVATE_NAMES_ONLY);
259 i::DirectHandle<i::FixedArray> keys;
260 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
261 isolate, keys,
262 i::KeyAccumulator::GetKeys(isolate, receiver,
263 i::KeyCollectionMode::kOwnOnly, key_filter,
264 i::GetKeysConversion::kConvertToString),
265 false);
266
267 // Estimate number of private fields and private instance methods/accessors.
268 int private_entries_count = 0;
269 auto count_private_entry =
270 [&](i::VariableMode mode, i::DirectHandle<i::String>,
271 i::DirectHandle<i::Object>) { private_entries_count++; };
272 for (int i = 0; i < keys->length(); ++i) {
273 // Exclude the private brand symbols.
274 i::DirectHandle<i::Symbol> key(i::Cast<i::Symbol>(keys->get(i)), isolate);
275 if (key->is_private_brand()) {
276 if (include_methods_or_accessors) {
277 i::DirectHandle<i::Object> value;
278 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
279 isolate, value, i::Object::GetProperty(isolate, receiver, key),
280 false);
281
282 i::DirectHandle<i::Context> value_context(i::Cast<i::Context>(*value),
283 isolate);
284 ForEachContextLocal(isolate, value_context, var_mode_filter,
285 instance_filter, count_private_entry);
286 }
287 } else if (include_fields) {
288 private_entries_count++;
289 }
290 }
291
292 // Estimate number of static private methods/accessors for classes.
293 bool has_static_private_methods_or_accessors = false;
294 if (include_methods_or_accessors) {
295 if (IsJSFunction(*receiver)) {
296 i::DirectHandle<i::JSFunction> func(i::Cast<i::JSFunction>(*receiver),
297 isolate);
298 i::DirectHandle<i::SharedFunctionInfo> shared(func->shared(), isolate);
299 if (shared->is_class_constructor() &&
300 shared->has_static_private_methods_or_accessors()) {
301 has_static_private_methods_or_accessors = true;
302 i::DirectHandle<i::Context> func_context(func->context(), isolate);
303 ForEachContextLocal(isolate, func_context, var_mode_filter,
304 static_filter, count_private_entry);
305 }
306 }
307 }
308
309 DCHECK(names_out->empty());
310 names_out->reserve(private_entries_count);
311 DCHECK(values_out->empty());
312 values_out->reserve(private_entries_count);
313
314 auto add_private_entry = [&](i::VariableMode mode,
315 i::DirectHandle<i::String> name,
316 i::DirectHandle<i::Object> value) {
317 DCHECK_IMPLIES(mode == i::VariableMode::kPrivateMethod,
318 IsJSFunction(*value));
319 DCHECK_IMPLIES(mode != i::VariableMode::kPrivateMethod,
320 IsAccessorPair(*value));
321 names_out->push_back(Utils::ToLocal(name));
322 values_out->push_back(Utils::ToLocal(value));
323 };
324 if (has_static_private_methods_or_accessors) {
325 i::DirectHandle<i::Context> receiver_context(
326 i::Cast<i::JSFunction>(*receiver)->context(), isolate);
327 ForEachContextLocal(isolate, receiver_context, var_mode_filter,
328 static_filter, add_private_entry);
329 }
330
331 for (int i = 0; i < keys->length(); ++i) {
332 i::DirectHandle<i::Object> obj_key(keys->get(i), isolate);
333 i::DirectHandle<i::Symbol> key(i::Cast<i::Symbol>(*obj_key), isolate);
334 CHECK(key->is_private_name());
335 i::DirectHandle<i::Object> value;
336 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
337 isolate, value, i::Object::GetProperty(isolate, receiver, key), false);
338 if (key->is_private_brand()) {
339 if (include_methods_or_accessors) {
340 DCHECK(IsContext(*value));
341 i::DirectHandle<i::Context> value_context(i::Cast<i::Context>(*value),
342 isolate);
343 ForEachContextLocal(isolate, value_context, var_mode_filter,
344 instance_filter, add_private_entry);
345 }
346 } else if (include_fields) { // Private fields
347 i::DirectHandle<i::String> name(
348 i::Cast<i::String>(i::Cast<i::Symbol>(*key)->description()), isolate);
349 names_out->push_back(Utils::ToLocal(name));
350 values_out->push_back(Utils::ToLocal(value));
351 }
352 }
353
354 DCHECK_EQ(names_out->size(), values_out->size());
355 DCHECK_LE(names_out->size(), private_entries_count);
356 return true;
357}

612bool Script::GetPossibleBreakpoints(
613 const Location& start, const Location& end, bool restrict_to_function,
614 std::vector<BreakLocation>* locations) const {
615 CHECK(!start.IsEmpty());
616 i::Handle<i::Script> script = Utils::OpenHandle(this);
617#if V8_ENABLE_WEBASSEMBLY
618 if (script->type() == i::Script::Type::kWasm) {
619 i::wasm::NativeModule* native_module = script->wasm_native_module();
620 return i::WasmScript::GetPossibleBreakpoints(native_module, start, end,
621 locations);
622 }
623#endif // V8_ENABLE_WEBASSEMBLY
624
625 i::Isolate* isolate = script->GetIsolate();
626
627 int start_offset, end_offset;
628 if (!GetSourceOffset(start, GetSourceOffsetMode::kClamp).To(&start_offset)) {
629 return false;
630 }
631 if (end.IsEmpty()) {
632 end_offset = std::numeric_limits<int>::max();
633 } else if (!GetSourceOffset(end, GetSourceOffsetMode::kClamp)
634 .To(&end_offset)) {
635 return false;
636 }
637 if (start_offset >= end_offset) return true;
638
639 std::vector<i::BreakLocation> v8_locations;
640 if (!isolate->debug()->GetPossibleBreakpoints(
641 script, start_offset, end_offset, restrict_to_function,
642 &v8_locations)) {
643 return false;
644 }
645
646 std::sort(v8_locations.begin(), v8_locations.end(), CompareBreakLocation);
647 for (const auto& v8_location : v8_locations) {
648 Location location = GetSourceLocation(v8_location.position());
649 locations->emplace_back(location.GetLineNumber(),
650 location.GetColumnNumber(), v8_location.type());
651 }
652 return true;
653}

655Maybe<int> Script::GetSourceOffset(const Location& location,
656 GetSourceOffsetMode mode) const {
657 i::DirectHandle<i::Script> script = Utils::OpenDirectHandle(this);
658#if V8_ENABLE_WEBASSEMBLY
659 if (script->type() == i::Script::Type::kWasm) {
660 return location.GetLineNumber() == 0 ? Just(location.GetColumnNumber())
661 : Nothing<int>();
662 }
663#endif // V8_ENABLE_WEBASSEMBLY
664
665 int line = location.GetLineNumber();
666 int column = location.GetColumnNumber();
667 if (!script->HasSourceURLComment()) {
668 // Line/column number for inline <script>s with sourceURL annotation
669 // are supposed to be related to the <script> tag, otherwise they
670 // are relative to the parent file. Keep this in sync with the logic
671 // in GetSourceLocation() below.
672 line -= script->line_offset();
673 if (line == 0) column -= script->column_offset();
674 }
675
676 i::Script::InitLineEnds(script->GetIsolate(), script);
677 auto line_ends = i::Cast<i::FixedArray>(
678 i::direct_handle(script->line_ends(), script->GetIsolate()));
679 if (line < 0) {
680 if (mode == GetSourceOffsetMode::kClamp) {
681 return Just(0);
682 }
683 return Nothing<int>();
684 }
685 if (line >= line_ends->length()) {
686 if (mode == GetSourceOffsetMode::kClamp) {
687 return Just(GetSmiValue(line_ends, line_ends->length() - 1));
688 }
689 return Nothing<int>();
690 }
691 if (column < 0) {
692 if (mode != GetSourceOffsetMode::kClamp) {
693 return Nothing<int>();
694 }
695 column = 0;
696 }
697 int offset = column;
698 if (line > 0) {
699 int prev_line_end_offset = GetSmiValue(line_ends, line - 1);
700 offset += prev_line_end_offset + 1;
701 }
702 int line_end_offset = GetSmiValue(line_ends, line);
703 if (offset > line_end_offset) {
704 // Be permissive with columns that don't exist,
705 // as long as they are clearly within the range
706 // of the script.
707 if (line < line_ends->length() - 1 || mode == GetSourceOffsetMode::kClamp) {
708 return Just(line_end_offset);
709 }
710 return Nothing<int>();
711 }
712 return Just(offset);
713}
