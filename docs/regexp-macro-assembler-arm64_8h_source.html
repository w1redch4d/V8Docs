266 RegisterState GetRegisterState(int register_index) {
267 DCHECK_LE(0, register_index);
268 if (register_index >= kNumCachedRegisters) {
269 return STACKED;
270 } else {
271 if ((register_index % 2) == 0) {
272 return CACHED_LSW;
273 } else {
274 return CACHED_MSW;
275 }
276 }
277 }
