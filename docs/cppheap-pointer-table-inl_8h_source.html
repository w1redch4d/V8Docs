16void CppHeapPointerTableEntry::MakePointerEntry(Address value,
17 CppHeapPointerTag tag,
18 bool mark_as_alive) {
19 // Top bits must be zero, otherwise we'd loose information when shifting.
20 DCHECK_EQ(0, value >> (kBitsPerSystemPointer - kCppHeapPointerPayloadShift));
21 DCHECK_NE(tag, CppHeapPointerTag::kFreeEntryTag);
22 DCHECK_NE(tag, CppHeapPointerTag::kEvacuationEntryTag);
23
24 Payload new_payload(value, tag);
25 DCHECK(!new_payload.HasMarkBitSet());
26 if (V8_UNLIKELY(mark_as_alive)) {
27 new_payload.SetMarkBit();
28 }
29 payload_.store(new_payload, std::memory_order_relaxed);
30}

73void CppHeapPointerTableEntry::Mark() {
74 auto old_payload = payload_.load(std::memory_order_relaxed);
75 DCHECK(old_payload.ContainsPointer());
76
77 auto new_payload = old_payload;
78 new_payload.SetMarkBit();
79
80 // We don't need to perform the CAS in a loop: if the new value is not equal
81 // to the old value, then the mutator must've just written a new value into
82 // the entry. This in turn must've set the marking bit already (see e.g.
83 // SetPointer()), so we don't need to do it again.
84 bool success = payload_.compare_exchange_strong(old_payload, new_payload,
85 std::memory_order_relaxed);
86 DCHECK(success || old_payload.HasMarkBitSet());
87 USE(success);
88}

100void CppHeapPointerTableEntry::Evacuate(CppHeapPointerTableEntry& dest) {
101 auto payload = payload_.load(std::memory_order_relaxed);
102 // We expect to only evacuate entries containing external pointers.
103 DCHECK(payload.ContainsPointer());
104 // Currently, evacuation only happens during table compaction. In that case,
105 // the marking bit must be unset as the entry has already been visited by the
106 // sweeper (which clears the marking bit). If this ever changes, we'll need
107 // to let the caller specify what to do with the marking bit during
108 // evacuation.
109 DCHECK(!payload.HasMarkBitSet());
110
111 dest.payload_.store(payload, std::memory_order_relaxed);
112
113 // The destination entry takes ownership of the pointer.
114 MakeZappedEntry();
115}

142void CppHeapPointerTable::Mark(Space* space, CppHeapPointerHandle handle,
143 Address handle_location) {
144 DCHECK(space->BelongsTo(this));
145
146 // The handle_location must always contain the given handle. Except if the
147 // slot is lazily-initialized. In that case, the handle may transition from
148 // the null handle to a valid handle. However, in that case the
149 // newly-allocated entry will already have been marked as alive during
150 // allocation, and so we don't need to do anything here.
151#ifdef DEBUG
152 CppHeapPointerHandle current_handle = base::AsAtomic32::Acquire_Load(
153 reinterpret_cast<CppHeapPointerHandle*>(handle_location));
154 DCHECK(handle == kNullCppHeapPointerHandle || handle == current_handle);
155#endif
156
157 // If the handle is null, it doesn't have an EPT entry; no mark is needed.
158 if (handle == kNullCppHeapPointerHandle) return;
159
160 uint32_t index = HandleToIndex(handle);
161 DCHECK(space->Contains(index));
162
163 // If the table is being compacted and the entry is inside the evacuation
164 // area, then allocate and set up an evacuation entry for it.
165 MaybeCreateEvacuationEntry(space, index, handle_location);
166
167 // Even if the entry is marked for evacuation, it still needs to be marked as
168 // alive as it may be visited during sweeping before being evacuation.
169 at(index).Mark();
170}
