15template <typename ObjectVisitor>
16void RelocInfo::Visit(Tagged<InstructionStream> host, ObjectVisitor* visitor) {
17 Mode mode = rmode();
18 if (IsEmbeddedObjectMode(mode)) {
19 visitor->VisitEmbeddedPointer(host, this);
20 } else if (IsCodeTargetMode(mode)) {
21 visitor->VisitCodeTarget(host, this);
22 } else if (IsExternalReference(mode)) {
23 visitor->VisitExternalReference(host, this);
24 } else if (IsInternalReference(mode) || IsInternalReferenceEncoded(mode)) {
25 visitor->VisitInternalReference(host, this);
26 } else if (IsBuiltinEntryMode(mode)) {
27 visitor->VisitOffHeapTarget(host, this);
28 } else if (IsJSDispatchHandle(mode)) {
29#ifdef V8_ENABLE_LEAPTIERING
30 // This would need to pass the RelocInfo if dispatch entries were allowed
31 // to move and we needed to update this slot.
32 static_assert(!JSDispatchTable::kSupportsCompaction);
33 visitor->VisitJSDispatchTableEntry(host, js_dispatch_handle());
34#else
35 UNREACHABLE();
36#endif
37 }
38}

50template <typename RelocInfoT>
51RelocIteratorBase<RelocInfoT>::RelocIteratorBase(RelocInfoT reloc_info,
52 const uint8_t* pos,
53 const uint8_t* end,
54 int mode_mask)
55 : pos_(pos), end_(end), rinfo_(reloc_info), mode_mask_(mode_mask) {
56 DCHECK_EQ(reloc_info.rmode(), RelocInfo::NO_INFO);
57 DCHECK_EQ(reloc_info.data(), 0);
58 // Relocation info is read backwards.
59 DCHECK_GE(pos_, end_);
60 if (mode_mask_ == 0) pos_ = end_;
61 next();
62}
