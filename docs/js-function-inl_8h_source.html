78void JSFunction::UpdateOptimizedCode(Isolate* isolate, Tagged<Code> code,
79 WriteBarrierMode mode) {
80 DisallowGarbageCollection no_gc;
81 DCHECK(code->is_optimized_code());
82#ifdef V8_ENABLE_LEAPTIERING
83 if (code->is_context_specialized()) {
84 // We can only set context-specialized code for single-closure cells.
85 if (raw_feedback_cell()->map() !=
86 ReadOnlyRoots(isolate).one_closure_cell_map()) {
87 return;
88 }
89 }
90 // Required for being able to deoptimize this code.
91 code->set_js_dispatch_handle(dispatch_handle());
92#endif // V8_ENABLE_LEAPTIERING
93 UpdateCodeImpl(code, mode, false);
94}

96void JSFunction::UpdateCodeImpl(Tagged<Code> value, WriteBarrierMode mode,
97 bool keep_tiering_request) {
98 DisallowGarbageCollection no_gc;
99
100#ifdef V8_ENABLE_LEAPTIERING
101 JSDispatchHandle handle = dispatch_handle();
102 if (handle == kNullJSDispatchHandle) {
103 handle = raw_feedback_cell()->dispatch_handle();
104 DCHECK_NE(handle, kNullJSDispatchHandle);
105 set_dispatch_handle(handle, mode);
106 }
107 if (keep_tiering_request) {
108 UpdateDispatchEntryKeepTieringRequest(value, mode);
109 } else {
110 UpdateDispatchEntry(value, mode);
111 }
112
113 if (V8_UNLIKELY(v8_flags.log_function_events)) {
114 IsolateGroup::current()->js_dispatch_table()->SetTieringRequest(
115 dispatch_handle(), TieringBuiltin::kFunctionLogNextExecution,
116 GetIsolate());
117 }
118#else
119 WriteCodePointerField(kCodeOffset, value);
120 CONDITIONAL_CODE_POINTER_WRITE_BARRIER(*this, kCodeOffset, value, mode);
121
122 if (V8_UNLIKELY(v8_flags.log_function_events && has_feedback_vector())) {
123 feedback_vector()->set_log_next_execution(true);
124 }
125#endif // V8_ENABLE_LEAPTIERING
126}

286bool JSFunction::IsOptimizationRequested(Isolate* isolate) const {
287#ifdef V8_ENABLE_LEAPTIERING
288 JSDispatchTable* jdt = IsolateGroup::current()->js_dispatch_table();
289 Address entrypoint = jdt->GetEntrypoint(dispatch_handle());
290 const EmbeddedData& embedded_data = EmbeddedData::FromBlob(isolate);
291#define CASE(name, ...) \
292 if (entrypoint == embedded_data.InstructionStartOf(Builtin::k##name)) { \
293 DCHECK(jdt->IsTieringRequested(dispatch_handle(), TieringBuiltin::k##name, \
294 isolate)); \
295 return TieringBuiltin::k##name != \
296 TieringBuiltin::kFunctionLogNextExecution; \
297 }
298 BUILTIN_LIST_BASE_TIERING(CASE)
299#undef CASE
300 return {};
301#else
302 return IsRequestMaglev(tiering_state()) || IsRequestTurbofan(tiering_state());
303#endif
304}

306std::optional<CodeKind> JSFunction::GetRequestedOptimizationIfAny(
307 Isolate* isolate, ConcurrencyMode mode) const {
308#ifdef V8_ENABLE_LEAPTIERING
309 JSDispatchTable* jdt = IsolateGroup::current()->js_dispatch_table();
310 Address entrypoint = jdt->GetEntrypoint(dispatch_handle());
311 const EmbeddedData& embedded_data = EmbeddedData::FromBlob(isolate);
312 auto builtin = ([&]() -> std::optional<TieringBuiltin> {
313#define CASE(name, ...) \
314 if (entrypoint == embedded_data.InstructionStartOf(Builtin::k##name)) { \
315 DCHECK(jdt->IsTieringRequested(dispatch_handle(), TieringBuiltin::k##name, \
316 isolate)); \
317 return TieringBuiltin::k##name; \
318 }
319 BUILTIN_LIST_BASE_TIERING(CASE)
320#undef CASE
321 DCHECK(!jdt->IsTieringRequested(dispatch_handle()));
322 return {};
323 })();
324 if (V8_LIKELY(!builtin)) return {};
325 switch (*builtin) {
326 case TieringBuiltin::kOptimizeMaglevEager:
327 if (mode == ConcurrencyMode::kSynchronous) return CodeKind::MAGLEV;
328 break;
329 case TieringBuiltin::kStartMaglevOptimizeJob:
330 if (mode == ConcurrencyMode::kConcurrent) return CodeKind::MAGLEV;
331 break;
332 case TieringBuiltin::kOptimizeTurbofanEager:
333 if (mode == ConcurrencyMode::kSynchronous) return CodeKind::TURBOFAN_JS;
334 break;
335 case TieringBuiltin::kStartTurbofanOptimizeJob:
336 if (mode == ConcurrencyMode::kConcurrent) return CodeKind::TURBOFAN_JS;
337 break;
338 case TieringBuiltin::kMarkLazyDeoptimized:
339 case TieringBuiltin::kMarkReoptimizeLazyDeoptimized:
340 case TieringBuiltin::kFunctionLogNextExecution:
341 break;
342 }
343#else
344 switch (mode) {
345 case ConcurrencyMode::kConcurrent:
346 if (IsRequestTurbofan_Concurrent(tiering_state())) {
347 return CodeKind::TURBOFAN_JS;
348 }
349 if (IsRequestMaglev_Concurrent(tiering_state())) {
350 return CodeKind::MAGLEV;
351 }
352 break;
353 case ConcurrencyMode::kSynchronous:
354 if (IsRequestTurbofan_Synchronous(tiering_state())) {
355 return CodeKind::TURBOFAN_JS;
356 }
357 if (IsRequestMaglev_Synchronous(tiering_state())) {
358 return CodeKind::MAGLEV;
359 }
360 break;
361 }
362#endif // !V8_ENABLE_LEAPTIERING
363 return {};
364}

513bool JSFunction::NeedsResetDueToFlushedBytecode(IsolateForSandbox isolate) {
514 // Do a raw read for shared and code fields here since this function may be
515 // called on a concurrent thread. JSFunction itself should be fully
516 // initialized here but the SharedFunctionInfo, Code objects may not be
517 // initialized. We read using acquire loads to defend against that.
518 // TODO(v8) the branches for !IsSharedFunctionInfo() and !IsCode() are
519 // probably dead code by now. Investigate removing them or replacing them
520 // with CHECKs.
521 Tagged<Object> maybe_shared =
522 ACQUIRE_READ_FIELD(*this, kSharedFunctionInfoOffset);
523 if (!IsSharedFunctionInfo(maybe_shared)) return false;
524
525 Tagged<Object> maybe_code = raw_code(isolate, kAcquireLoad);
526 if (!IsCode(maybe_code)) return false;
527 Tagged<Code> code = Cast<Code>(maybe_code);
528
529 Tagged<SharedFunctionInfo> shared = Cast<SharedFunctionInfo>(maybe_shared);
530 return !shared->is_compiled() && code->builtin_id() != Builtin::kCompileLazy;
531}

538void JSFunction::ResetIfCodeFlushed(
539 Isolate* isolate,
540 std::optional<std::function<void(Tagged<HeapObject> object, ObjectSlot slot,
541 Tagged<HeapObject> target)>>
542 gc_notify_updated_slot) {
543 const bool kBytecodeCanFlush =
544 v8_flags.flush_bytecode || v8_flags.stress_snapshot;
545 const bool kBaselineCodeCanFlush =
546 v8_flags.flush_baseline_code || v8_flags.stress_snapshot;
547 if (!kBytecodeCanFlush && !kBaselineCodeCanFlush) return;
548
549 DCHECK_IMPLIES(NeedsResetDueToFlushedBytecode(isolate), kBytecodeCanFlush);
550 if (kBytecodeCanFlush && NeedsResetDueToFlushedBytecode(isolate)) {
551 // Bytecode was flushed and function is now uncompiled, reset JSFunction
552 // by setting code to CompileLazy and clearing the feedback vector.
553 ResetTieringRequests(isolate);
554 UpdateCode(*BUILTIN_CODE(isolate, CompileLazy));
555 raw_feedback_cell()->reset_feedback_vector(gc_notify_updated_slot);
556 return;
557 }
558
559 DCHECK_IMPLIES(NeedsResetDueToFlushedBaselineCode(isolate),
560 kBaselineCodeCanFlush);
561 if (kBaselineCodeCanFlush && NeedsResetDueToFlushedBaselineCode(isolate)) {
562 // Flush baseline code from the closure if required
563 ResetTieringRequests(isolate);
564 UpdateCode(*BUILTIN_CODE(isolate, InterpreterEntryTrampoline));
565 }
566}
