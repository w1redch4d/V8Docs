61// This method may transition the elements kind of the JSArray once, to make
62// sure that all elements provided as arguments in the specified range can be
63// added without further elements kinds transitions.
64void MatchArrayElementsKindToArguments(Isolate* isolate,
65 DirectHandle<JSArray> array,
66 BuiltinArguments* args,
67 int first_arg_index, int num_arguments) {
68 int args_length = args->length();
69 if (first_arg_index >= args_length) return;
70
71 ElementsKind origin_kind = array->GetElementsKind();
72
73 // We do not need to transition for PACKED/HOLEY_ELEMENTS.
74 if (IsObjectElementsKind(origin_kind)) return;
75
76 ElementsKind target_kind = origin_kind;
77 {
78 DisallowGarbageCollection no_gc;
79 int last_arg_index = std::min(first_arg_index + num_arguments, args_length);
80 for (int i = first_arg_index; i < last_arg_index; i++) {
81 Tagged<Object> arg = (*args)[i];
82 if (IsHeapObject(arg)) {
83 if (IsHeapNumber(arg)) {
84 target_kind = PACKED_DOUBLE_ELEMENTS;
85 } else {
86 target_kind = PACKED_ELEMENTS;
87 break;
88 }
89 }
90 }
91 }
92 if (target_kind != origin_kind) {
93 // Use a short-lived HandleScope to avoid creating several copies of the
94 // elements handle which would cause issues when left-trimming later-on.
95 HandleScope scope(isolate);
96 JSObject::TransitionElementsKind(array, target_kind);
97 }
98}

103V8_WARN_UNUSED_RESULT
104inline bool EnsureJSArrayWithWritableFastElements(Isolate* isolate,
105 DirectHandle<Object> receiver,
106 BuiltinArguments* args,
107 int first_arg_index,
108 int num_arguments) {
109 if (!IsJSArray(*receiver)) return false;
110 DirectHandle<JSArray> array = Cast<JSArray>(receiver);
111 ElementsKind origin_kind = array->GetElementsKind();
112 if (IsDictionaryElementsKind(origin_kind)) return false;
113 if (!array->map()->is_extensible()) return false;
114 if (args == nullptr) return true;
115
116 // If there may be elements accessors in the prototype chain, the fast path
117 // cannot be used if there arguments to add to the array.
118 if (!IsJSArrayFastElementMovingAllowed(isolate, *array)) return false;
119
120 // Adding elements to the array prototype would break code that makes sure
121 // it has no elements. Handle that elsewhere.
122 if (isolate->IsInitialArrayPrototype(*array)) return false;
123
124 // Need to ensure that the arguments passed in args can be contained in
125 // the array.
126 MatchArrayElementsKindToArguments(isolate, array, args, first_arg_index,
127 num_arguments);
128 return true;
129}

131// If |index| is Undefined, returns init_if_undefined.
132// If |index| is negative, returns length + index.
133// If |index| is positive, returns index.
134// Returned value is guaranteed to be in the interval of [0, length].
135V8_WARN_UNUSED_RESULT Maybe<double> GetRelativeIndex(Isolate* isolate,
136 double length,
137 DirectHandle<Object> index,
138 double init_if_undefined) {
139 double relative_index = init_if_undefined;
140 if (!IsUndefined(*index)) {
141 DirectHandle<Object> relative_index_obj;
142 ASSIGN_RETURN_ON_EXCEPTION_VALUE(isolate, relative_index_obj,
143 Object::ToInteger(isolate, index),
144 Nothing<double>());
145 relative_index = Object::NumberValue(*relative_index_obj);
146 }
147
148 if (relative_index < 0) {
149 return Just(std::max(length + relative_index, 0.0));
150 }
151
152 return Just(std::min(relative_index, length));
153}

215V8_WARN_UNUSED_RESULT Maybe<bool> TryFastArrayFill(
216 Isolate* isolate, BuiltinArguments* args, DirectHandle<JSReceiver> receiver,
217 DirectHandle<Object> value, double start_index, double end_index) {
218 // If indices are too large, use generic path since they are stored as
219 // properties, not in the element backing store.
220 if (end_index > kMaxUInt32) return Just(false);
221 if (!IsJSObject(*receiver)) return Just(false);
222
223 if (!EnsureJSArrayWithWritableFastElements(isolate, receiver, args, 1, 1)) {
224 return Just(false);
225 }
226
227 DirectHandle<JSArray> array = Cast<JSArray>(receiver);
228
229 // If no argument was provided, we fill the array with 'undefined'.
230 // EnsureJSArrayWith... does not handle that case so we do it here.
231 // TODO(szuend): Pass target elements kind to EnsureJSArrayWith... when
232 // it gets refactored.
233 if (args->length() == 1 && array->GetElementsKind() != PACKED_ELEMENTS) {
234 // Use a short-lived HandleScope to avoid creating several copies of the
235 // elements handle which would cause issues when left-trimming later-on.
236 HandleScope scope(isolate);
237 JSObject::TransitionElementsKind(array, PACKED_ELEMENTS);
238 }
239
240 DCHECK_LE(start_index, kMaxUInt32);
241 DCHECK_LE(end_index, kMaxUInt32);
242
243 uint32_t start, end;
244 CHECK(DoubleToUint32IfEqualToSelf(start_index, &start));
245 CHECK(DoubleToUint32IfEqualToSelf(end_index, &end));
246
247 ElementsAccessor* accessor = array->GetElementsAccessor();
248 RETURN_ON_EXCEPTION_VALUE(isolate, accessor->Fill(array, value, start, end),
249 Nothing<bool>());
250
251 // It's possible the JSArray's 'length' property was assigned to after the
252 // length was loaded due to user code during argument coercion of the start
253 // and end parameters. The spec algorithm does a Set, meaning the length would
254 // grow as needed during the fill.
255 //
256 // ElementAccessor::Fill is able to grow the backing store as needed, but we
257 // need to ensure the JSArray's length is correctly set in case the user
258 // assigned a smaller value.
259 if (Object::NumberValue(array->length()) < end) {
260 CHECK(accessor->SetLength(array, end).FromJust());
261 }
262
263 return Just(true);
264}

267BUILTIN(ArrayPrototypeFill) {
268 HandleScope scope(isolate);
269
270 // 1. Let O be ? ToObject(this value).
271 DirectHandle<JSReceiver> receiver;
272 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
273 isolate, receiver, Object::ToObject(isolate, args.receiver()));
274
275 // 2. Let len be ? ToLength(? Get(O, "length")).
276 double length;
277 MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
278 isolate, length, GetLengthProperty(isolate, receiver));
279
280 // 3. Let relativeStart be ? ToInteger(start).
281 // 4. If relativeStart < 0, let k be max((len + relativeStart), 0);
282 // else let k be min(relativeStart, len).
283 DirectHandle<Object> start = args.atOrUndefined(isolate, 2);
284
285 double start_index;
286 MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
287 isolate, start_index, GetRelativeIndex(isolate, length, start, 0));
288
289 // 5. If end is undefined, let relativeEnd be len;
290 // else let relativeEnd be ? ToInteger(end).
291 // 6. If relativeEnd < 0, let final be max((len + relativeEnd), 0);
292 // else let final be min(relativeEnd, len).
293 DirectHandle<Object> end = args.atOrUndefined(isolate, 3);
294
295 double end_index;
296 MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
297 isolate, end_index, GetRelativeIndex(isolate, length, end, length));
298
299 if (start_index >= end_index) return *receiver;
300
301 // Ensure indexes are within array bounds
302 DCHECK_LE(0, start_index);
303 DCHECK_LE(start_index, end_index);
304 DCHECK_LE(end_index, length);
305
306 DirectHandle<Object> value = args.atOrUndefined(isolate, 1);
307
308 bool success;
309 MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
310 isolate, success,
311 TryFastArrayFill(isolate, &args, receiver, value, start_index,
312 end_index));
313 if (success) return *receiver;
314 return GenericArrayFill(isolate, receiver, value, start_index, end_index);
315}

318V8_WARN_UNUSED_RESULT Tagged<Object> GenericArrayPush(Isolate* isolate,
319 BuiltinArguments* args) {
320 // 1. Let O be ? ToObject(this value).
321 DirectHandle<JSReceiver> receiver;
322 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
323 isolate, receiver, Object::ToObject(isolate, args->receiver()));
324
325 // 2. Let len be ? ToLength(? Get(O, "length")).
326 DirectHandle<Object> raw_length_number;
327 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
328 isolate, raw_length_number,
329 Object::GetLengthFromArrayLike(isolate, receiver));
330
331 // 3. Let args be a List whose elements are, in left to right order,
332 // the arguments that were passed to this function invocation.
333 // 4. Let arg_count be the number of elements in args.
334 int arg_count = args->length() - 1;
335
336 // 5. If len + arg_count > 2^53-1, throw a TypeError exception.
337 double length = Object::NumberValue(*raw_length_number);
338 if (arg_count > kMaxSafeInteger - length) {
339 THROW_NEW_ERROR_RETURN_FAILURE(
340 isolate, NewTypeError(MessageTemplate::kPushPastSafeLength,
341 isolate->factory()->NewNumberFromInt(arg_count),
342 raw_length_number));
343 }
344
345 // 6. Repeat, while args is not empty.
346 for (int i = 0; i < arg_count; ++i) {
347 // a. Remove the first element from args and let E be the value of the
348 // element.
349 Handle<Object> element = args->at(i + 1);
350
351 // b. Perform ? Set(O, ! ToString(len), E, true).
352 if (length <= JSObject::kMaxElementIndex) {
353 RETURN_FAILURE_ON_EXCEPTION(
354 isolate, Object::SetElement(isolate, receiver, length, element,
355 ShouldThrow::kThrowOnError));
356 } else {
357 PropertyKey key(isolate, length);
358 LookupIterator it(isolate, receiver, key);
359 MAYBE_RETURN(Object::SetProperty(&it, element, StoreOrigin::kMaybeKeyed,
360 Just(ShouldThrow::kThrowOnError)),
361 ReadOnlyRoots(isolate).exception());
362 }
363
364 // c. Let len be len+1.
365 ++length;
366 }
367
368 // 7. Perform ? Set(O, "length", len, true).
369 DirectHandle<Object> final_length = isolate->factory()->NewNumber(length);
370 RETURN_FAILURE_ON_EXCEPTION(
371 isolate, Object::SetProperty(isolate, receiver,
372 isolate->factory()->length_string(),
373 final_length, StoreOrigin::kMaybeKeyed,
374 Just(ShouldThrow::kThrowOnError)));
375
376 // 8. Return len.
377 return *final_length;
378}

647 DCHECK_LE(to_add, Smi::kMaxValue - Smi::ToInt(array->length()));
648 DCHECK(!JSArray::HasReadOnlyLength(array));
649
650 ElementsAccessor* accessor = array->GetElementsAccessor();
651 uint32_t new_length;
652 MAYBE_ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
653 isolate, new_length, accessor->Unshift(array, &args, to_add));
654 return Smi::FromInt(new_length);
655}

672class ArrayConcatVisitor {
673 public:
674 ArrayConcatVisitor(
675 Isolate* isolate,
676 DirectHandle<UnionOf<JSReceiver, FixedArray, NumberDictionary>> storage,
677 bool fast_elements)
678 : isolate_(isolate),
679 storage_(isolate->global_handles()->Create(*storage)),
680 index_offset_(0u),
681 bit_field_(FastElementsField::encode(fast_elements) |
682 ExceedsLimitField::encode(false) |
683 IsFixedArrayField::encode(IsFixedArray(*storage, isolate)) |
684 HasSimpleElementsField::encode(
685 IsFixedArray(*storage, isolate) ||
686 // Don't take fast path for storages that might have
687 // side effects when storing to them.
688 (!IsCustomElementsReceiverMap(storage->map(isolate)) &&
689 !IsJSTypedArray(*storage, isolate)))) {
690 DCHECK_IMPLIES(this->fast_elements(), is_fixed_array());
691 }
692
693 ~ArrayConcatVisitor() { clear_storage(); }
694
695 V8_WARN_UNUSED_RESULT bool visit(uint32_t i, DirectHandle<Object> elm) {
696 uint32_t index = index_offset_ + i;
697
698 // Note we use >=kMaxArrayLength instead of the more appropriate
699 // >kMaxArrayIndex here due to overflowing arithmetic and
700 // increase_index_offset.
701 if (i >= JSArray::kMaxArrayLength - index_offset_) {
702 set_exceeds_array_limit(true);
703 // Exception hasn't been thrown at this point. Return true to
704 // break out, and caller will throw. !visit would imply that
705 // there is already an exception.
706 return true;
707 }
708
709 if (!is_fixed_array()) {
710 MAYBE_RETURN(JSReceiver::CreateDataProperty(
711 isolate_, direct_handle(Cast<JSReceiver>(storage_)),
712 PropertyKey(isolate_, index), elm, Just(kThrowOnError)),
713 false);
714 return true;
715 }
716
717 if (fast_elements()) {
718 if (index < static_cast<uint32_t>(storage_fixed_array()->length())) {
719 storage_fixed_array()->set(index, *elm);
720 return true;
721 }
722 // Our initial estimate of length was foiled, possibly by
723 // getters on the arrays increasing the length of later arrays
724 // during iteration.
725 // This shouldn't happen in anything but pathological cases.
726 SetDictionaryMode();
727 // Fall-through to dictionary mode.
728 }
729 DCHECK(!fast_elements());
730 DirectHandle<NumberDictionary> dict(Cast<NumberDictionary>(*storage_),
731 isolate_);
732 // The object holding this backing store has just been allocated, so
733 // it cannot yet be used as a prototype.
734 DirectHandle<JSObject> not_a_prototype_holder;
735 DirectHandle<NumberDictionary> result = NumberDictionary::Set(
736 isolate_, dict, index, elm, not_a_prototype_holder);
737 if (!result.is_identical_to(dict)) {
738 // Dictionary needed to grow.
739 clear_storage();
740 set_storage(*result);
741 }
742 return true;
743 }
744
745 uint32_t index_offset() const { return index_offset_; }
746
747 void increase_index_offset(uint32_t delta) {
748 if (JSArray::kMaxArrayLength - index_offset_ < delta) {
749 index_offset_ = JSArray::kMaxArrayLength;
750 } else {
751 index_offset_ += delta;
752 }
753 // If the initial length estimate was off (see special case in visit()),
754 // but the array blowing the limit didn't contain elements beyond the
755 // provided-for index range, go to dictionary mode now.
756 if (fast_elements() &&
757 index_offset_ >
758 static_cast<uint32_t>(Cast<FixedArrayBase>(*storage_)->length())) {
759 SetDictionaryMode();
760 }
761 }
762
763 bool exceeds_array_limit() const {
764 return ExceedsLimitField::decode(bit_field_);
765 }
766
767 Handle<JSArray> ToArray() {
768 DCHECK(is_fixed_array());
769 Handle<JSArray> array = isolate_->factory()->NewJSArray(0);
770 DirectHandle<Number> length =
771 isolate_->factory()->NewNumber(static_cast<double>(index_offset_));
772 DirectHandle<Map> map = JSObject::GetElementsTransitionMap(
773 array, fast_elements() ? HOLEY_ELEMENTS : DICTIONARY_ELEMENTS);
774 {
775 DisallowGarbageCollection no_gc;
776 Tagged<JSArray> raw = *array;
777 raw->set_length(*length);
778 raw->set_elements(*storage_fixed_array());
779 raw->set_map(isolate_, *map, kReleaseStore);
780 }
781 return array;
782 }
783
784 V8_WARN_UNUSED_RESULT MaybeHandle<JSReceiver> ToJSReceiver() {
785 DCHECK(!is_fixed_array());
786 Handle<JSReceiver> result = Cast<JSReceiver>(storage_);
787 DirectHandle<Object> length =
788 isolate_->factory()->NewNumber(static_cast<double>(index_offset_));
789 RETURN_ON_EXCEPTION(
790 isolate_, Object::SetProperty(isolate_, result,
791 isolate_->factory()->length_string(),
792 length, StoreOrigin::kMaybeKeyed,
793 Just(ShouldThrow::kThrowOnError)));
794 return result;
795 }
796 bool has_simple_elements() const {
797 return HasSimpleElementsField::decode(bit_field_);
798 }
799
800 private:
801 // Convert storage to dictionary mode.
802 void SetDictionaryMode() {
803 DCHECK(fast_elements() && is_fixed_array());
804 DirectHandle<FixedArray> current_storage = storage_fixed_array();
805 DirectHandle<NumberDictionary> slow_storage(
806 NumberDictionary::New(isolate_, current_storage->length()));
807 uint32_t current_length = static_cast<uint32_t>(current_storage->length());
808 FOR_WITH_HANDLE_SCOPE(
809 isolate_, uint32_t, i = 0, i, i < current_length, i++, {
810 DirectHandle<Object> element(current_storage->get(i), isolate_);
811 if (!IsTheHole(*element, isolate_)) {
812 // The object holding this backing store has just been allocated, so
813 // it cannot yet be used as a prototype.
814 DirectHandle<JSObject> not_a_prototype_holder;
815 DirectHandle<NumberDictionary> new_storage = NumberDictionary::Set(
816 isolate_, slow_storage, i, element, not_a_prototype_holder);
817 if (!new_storage.is_identical_to(slow_storage)) {
818 slow_storage = loop_scope.CloseAndEscape(new_storage);
819 }
820 }
821 });
822 clear_storage();
823 set_storage(*slow_storage);
824 set_fast_elements(false);
825 }
826
827 inline void clear_storage() { GlobalHandles::Destroy(storage_.location()); }
828
829 inline void set_storage(Tagged<FixedArray> storage) {
830 DCHECK(is_fixed_array());
831 DCHECK(has_simple_elements());
832 storage_ = isolate_->global_handles()->Create(storage);
833 }
834
835 using FastElementsField = base::BitField<bool, 0, 1>;
836 using ExceedsLimitField = base::BitField<bool, 1, 1>;
837 using IsFixedArrayField = base::BitField<bool, 2, 1>;
838 using HasSimpleElementsField = base::BitField<bool, 3, 1>;
839
840 bool fast_elements() const { return FastElementsField::decode(bit_field_); }
841 void set_fast_elements(bool fast) {
842 bit_field_ = FastElementsField::update(bit_field_, fast);
843 }
844 void set_exceeds_array_limit(bool exceeds) {
845 bit_field_ = ExceedsLimitField::update(bit_field_, exceeds);
846 }
847 bool is_fixed_array() const { return IsFixedArrayField::decode(bit_field_); }
848 Handle<FixedArray> storage_fixed_array() {
849 DCHECK(is_fixed_array());
850 DCHECK(has_simple_elements());
851 return Cast<FixedArray>(storage_);
852 }
853
854 Isolate* isolate_;
855 IndirectHandle<UnionOf<JSReceiver, FixedArray, NumberDictionary>>
856 storage_; // Always a global handle.
857 // Index after last seen index. Always less than or equal to
858 // JSArray::kMaxArrayLength.
859 uint32_t index_offset_;
860 uint32_t bit_field_;
861};

863uint32_t EstimateElementCount(Isolate* isolate, DirectHandle<JSArray> array) {
864 DisallowGarbageCollection no_gc;
865 uint32_t length = static_cast<uint32_t>(Object::NumberValue(array->length()));
866 int element_count = 0;
867 switch (array->GetElementsKind()) {
868 case PACKED_SMI_ELEMENTS:
869 case HOLEY_SMI_ELEMENTS:
870 case PACKED_ELEMENTS:
871 case PACKED_FROZEN_ELEMENTS:
872 case PACKED_SEALED_ELEMENTS:
873 case PACKED_NONEXTENSIBLE_ELEMENTS:
874 case HOLEY_FROZEN_ELEMENTS:
875 case HOLEY_SEALED_ELEMENTS:
876 case HOLEY_NONEXTENSIBLE_ELEMENTS:
877 case HOLEY_ELEMENTS: {
878 // Fast elements can't have lengths that are not representable by
879 // a 32-bit signed integer.
880 DCHECK_GE(static_cast<int32_t>(FixedArray::kMaxLength), 0);
881 int fast_length = static_cast<int>(length);
882 Tagged<FixedArray> elements = Cast<FixedArray>(array->elements());
883 for (int i = 0; i < fast_length; i++) {
884 if (!IsTheHole(elements->get(i), isolate)) element_count++;
885 }
886 break;
887 }
888 case PACKED_DOUBLE_ELEMENTS:
889 case HOLEY_DOUBLE_ELEMENTS: {
890 // Fast elements can't have lengths that are not representable by
891 // a 32-bit signed integer.
892 DCHECK_GE(static_cast<int32_t>(FixedDoubleArray::kMaxLength), 0);
893 int fast_length = static_cast<int>(length);
894 if (IsFixedArray(array->elements())) {
895 DCHECK_EQ(Cast<FixedArray>(array->elements())->length(), 0);
896 break;
897 }
898 Tagged<FixedDoubleArray> elements =
899 Cast<FixedDoubleArray>(array->elements());
900 for (int i = 0; i < fast_length; i++) {
901 if (!elements->is_the_hole(i)) element_count++;
902 }
903 break;
904 }
905 case DICTIONARY_ELEMENTS: {
906 Tagged<NumberDictionary> dictionary =
907 Cast<NumberDictionary>(array->elements());
908 ReadOnlyRoots roots(isolate);
909 for (InternalIndex i : dictionary->IterateEntries()) {
910 Tagged<Object> key = dictionary->KeyAt(i);
911 if (dictionary->IsKey(roots, key)) {
912 element_count++;
913 }
914 }
915 break;
916 }
917#define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) case TYPE##_ELEMENTS:
918
919 TYPED_ARRAYS(TYPED_ARRAY_CASE)
920 RAB_GSAB_TYPED_ARRAYS(TYPED_ARRAY_CASE)
921 // External arrays are always dense.
922 return length;
923
924#undef TYPED_ARRAY_CASE
925 case NO_ELEMENTS:
926 return 0;
927 case FAST_SLOPPY_ARGUMENTS_ELEMENTS:
928 case SLOW_SLOPPY_ARGUMENTS_ELEMENTS:
929 case FAST_STRING_WRAPPER_ELEMENTS:
930 case SLOW_STRING_WRAPPER_ELEMENTS:
931 case WASM_ARRAY_ELEMENTS:
932 case SHARED_ARRAY_ELEMENTS:
933 UNREACHABLE();
934 }
935 // As an estimate, we assume that the prototype doesn't contain any
936 // inherited elements.
937 return element_count;
938}

940void CollectElementIndices(Isolate* isolate, DirectHandle<JSObject> object,
941 uint32_t range, std::vector<uint32_t>* indices) {
942 ElementsKind kind = object->GetElementsKind();
943 switch (kind) {
944 case PACKED_SMI_ELEMENTS:
945 case PACKED_ELEMENTS:
946 case PACKED_FROZEN_ELEMENTS:
947 case PACKED_SEALED_ELEMENTS:
948 case PACKED_NONEXTENSIBLE_ELEMENTS:
949 case HOLEY_SMI_ELEMENTS:
950 case HOLEY_FROZEN_ELEMENTS:
951 case HOLEY_SEALED_ELEMENTS:
952 case HOLEY_NONEXTENSIBLE_ELEMENTS:
953 case HOLEY_ELEMENTS: {
954 DisallowGarbageCollection no_gc;
955 Tagged<FixedArray> elements = Cast<FixedArray>(object->elements());
956 uint32_t length = static_cast<uint32_t>(elements->length());
957 if (range < length) length = range;
958 for (uint32_t i = 0; i < length; i++) {
959 if (!IsTheHole(elements->get(i), isolate)) {
960 indices->push_back(i);
961 }
962 }
963 break;
964 }
965 case HOLEY_DOUBLE_ELEMENTS:
966 case PACKED_DOUBLE_ELEMENTS: {
967 if (IsFixedArray(object->elements())) {
968 DCHECK_EQ(object->elements()->length(), 0);
969 break;
970 }
971 DirectHandle<FixedDoubleArray> elements(
972 Cast<FixedDoubleArray>(object->elements()), isolate);
973 uint32_t length = static_cast<uint32_t>(elements->length());
974 if (range < length) length = range;
975 for (uint32_t i = 0; i < length; i++) {
976 if (!elements->is_the_hole(i)) {
977 indices->push_back(i);
978 }
979 }
980 break;
981 }
982 case DICTIONARY_ELEMENTS: {
983 DisallowGarbageCollection no_gc;
984 Tagged<NumberDictionary> dict =
985 Cast<NumberDictionary>(object->elements());
986 uint32_t capacity = dict->Capacity();
987 ReadOnlyRoots roots(isolate);
988 FOR_WITH_HANDLE_SCOPE(isolate, uint32_t, j = 0, j, j < capacity, j++, {
989 Tagged<Object> k = dict->KeyAt(InternalIndex(j));
990 if (!dict->IsKey(roots, k)) continue;
991 DCHECK(IsNumber(k));
992 uint32_t index = static_cast<uint32_t>(Object::NumberValue(k));
993 if (index < range) {
994 indices->push_back(index);
995 }
996 });
997 break;
998 }
999#define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) case TYPE##_ELEMENTS:
1000
1001 TYPED_ARRAYS(TYPED_ARRAY_CASE) RAB_GSAB_TYPED_ARRAYS(TYPED_ARRAY_CASE) {
1002 size_t length = Cast<JSTypedArray>(object)->GetLength();
1003 if (range <= length) {
1004 length = range;
1005 // We will add all indices, so we might as well clear it first
1006 // and avoid duplicates.
1007 indices->clear();
1008 }
1009 // {range} puts a cap on {length}.
1010 DCHECK_LE(length, std::numeric_limits<uint32_t>::max());
1011 for (uint32_t i = 0; i < length; i++) {
1012 indices->push_back(i);
1013 }
1014 if (length == range) return; // All indices accounted for already.
1015 break;
1016 }
1017
1018#undef TYPED_ARRAY_CASE
1019 case FAST_SLOPPY_ARGUMENTS_ELEMENTS:
1020 case SLOW_SLOPPY_ARGUMENTS_ELEMENTS: {
1021 DisallowGarbageCollection no_gc;
1022 DisableGCMole no_gc_mole;
1023 Tagged<FixedArrayBase> elements = object->elements();
1024 Tagged<JSObject> raw_object = *object;
1025 ElementsAccessor* accessor = object->GetElementsAccessor();
1026 for (uint32_t i = 0; i < range; i++) {
1027 if (accessor->HasElement(raw_object, i, elements)) {
1028 indices->push_back(i);
1029 }
1030 }
1031 break;
1032 }
1033 case FAST_STRING_WRAPPER_ELEMENTS:
1034 case SLOW_STRING_WRAPPER_ELEMENTS: {
1035 DCHECK(IsJSPrimitiveWrapper(*object));
1036 auto js_value = Cast<JSPrimitiveWrapper>(object);
1037 DCHECK(IsString(js_value->value()));
1038 DirectHandle<String> string(Cast<String>(js_value->value()), isolate);
1039 uint32_t length = static_cast<uint32_t>(string->length());
1040 uint32_t i = 0;
1041 uint32_t limit = std::min(length, range);
1042 for (; i < limit; i++) {
1043 indices->push_back(i);
1044 }
1045 ElementsAccessor* accessor = object->GetElementsAccessor();
1046 for (; i < range; i++) {
1047 if (accessor->HasElement(*object, i)) {
1048 indices->push_back(i);
1049 }
1050 }
1051 break;
1052 }
1053 case WASM_ARRAY_ELEMENTS:
1054 // TODO(ishell): implement
1055 UNIMPLEMENTED();
1056 case SHARED_ARRAY_ELEMENTS: {
1057 uint32_t length = Cast<JSSharedArray>(object)->elements()->length();
1058 if (range <= length) {
1059 length = range;
1060 indices->clear();
1061 }
1062 for (uint32_t i = 0; i < length; i++) {
1063 // JSSharedArrays are created non-resizable and do not have holes.
1064 SLOW_DCHECK(object->GetElementsAccessor()->HasElement(
1065 *object, i, object->elements()));
1066 indices->push_back(i);
1067 }
1068 if (length == range) return;
1069 break;
1070 }
1071 case NO_ELEMENTS:
1072 break;
1073 }
1074
1075 PrototypeIterator iter(isolate, object);
1076 if (!iter.IsAtEnd()) {
1077 // The prototype will usually have no inherited element indices,
1078 // but we have to check.
1079 // Casting to JSObject is safe because we ran {HasOnlySimpleElements} on
1080 // the receiver before, which checks the prototype chain.
1081 CollectElementIndices(
1082 isolate, PrototypeIterator::GetCurrent<JSObject>(iter), range, indices);
1083 }
1084}

1086bool IterateElementsSlow(Isolate* isolate, DirectHandle<JSReceiver> receiver,
1087 uint32_t length, ArrayConcatVisitor* visitor) {
1088 FOR_WITH_HANDLE_SCOPE(isolate, uint32_t, i = 0, i, i < length, ++i, {
1089 Maybe<bool> maybe = JSReceiver::HasElement(isolate, receiver, i);
1090 if (maybe.IsNothing()) return false;
1091 if (maybe.FromJust()) {
1092 DirectHandle<Object> element_value;
1093 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
1094 isolate, element_value, JSReceiver::GetElement(isolate, receiver, i),
1095 false);
1096 if (!visitor->visit(i, element_value)) return false;
1097 }
1098 });
1099 visitor->increase_index_offset(length);
1100 return true;
1101}
1112bool IterateElements(Isolate* isolate, DirectHandle<JSReceiver> receiver,
1113 ArrayConcatVisitor* visitor) {
1114 uint32_t length = 0;
1115
1116 if (IsJSArray(*receiver)) {
1117 auto array = Cast<JSArray>(receiver);
1118 length = static_cast<uint32_t>(Object::NumberValue(array->length()));
1119 } else {
1120 DirectHandle<Object> val;
1121 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
1122 isolate, val, Object::GetLengthFromArrayLike(isolate, receiver), false);
1123 if (visitor->index_offset() + Object::NumberValue(*val) > kMaxSafeInteger) {
1124 isolate->Throw(*isolate->factory()->NewTypeError(
1125 MessageTemplate::kInvalidArrayLength));
1126 return false;
1127 }
1128 // TODO(caitp): Support larger element indexes (up to 2^53-1).
1129 if (!Object::ToUint32(*val, &length)) {
1130 length = 0;
1131 }
1132 // TODO(cbruni): handle other element kind as well
1133 return IterateElementsSlow(isolate, receiver, length, visitor);
1134 }
1135
1136 if (!visitor->has_simple_elements() ||
1137 !HasOnlySimpleElements(isolate, *receiver)) {
1138 return IterateElementsSlow(isolate, receiver, length, visitor);
1139 }
1140 DirectHandle<JSArray> array = Cast<JSArray>(receiver);
1141
1142 switch (array->GetElementsKind()) {
1143 case PACKED_SMI_ELEMENTS:
1144 case PACKED_ELEMENTS:
1145 case PACKED_FROZEN_ELEMENTS:
1146 case PACKED_SEALED_ELEMENTS:
1147 case PACKED_NONEXTENSIBLE_ELEMENTS:
1148 case HOLEY_SMI_ELEMENTS:
1149 case HOLEY_FROZEN_ELEMENTS:
1150 case HOLEY_SEALED_ELEMENTS:
1151 case HOLEY_NONEXTENSIBLE_ELEMENTS:
1152 case HOLEY_ELEMENTS: {
1153 // Disallow execution so the cached elements won't change mid execution.
1154 DisallowJavascriptExecution no_js(isolate);
1155
1156 // Run through the elements FixedArray and use HasElement and GetElement
1157 // to check the prototype for missing elements.
1158 DirectHandle<FixedArray> elements(Cast<FixedArray>(array->elements()),
1159 isolate);
1160 int fast_length = static_cast<int>(length);
1161 DCHECK(fast_length <= elements->length());
1162 FOR_WITH_HANDLE_SCOPE(isolate, int, j = 0, j, j < fast_length, j++, {
1163 DirectHandle<Object> element_value(elements->get(j), isolate);
1164 if (!IsTheHole(*element_value, isolate)) {
1165 if (!visitor->visit(j, element_value)) return false;
1166 } else {
1167 Maybe<bool> maybe = JSReceiver::HasElement(isolate, array, j);
1168 if (maybe.IsNothing()) return false;
1169 if (maybe.FromJust()) {
1170 // Call GetElement on array, not its prototype, or getters won't
1171 // have the correct receiver.
1172 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
1173 isolate, element_value,
1174 JSReceiver::GetElement(isolate, array, j), false);
1175 if (!visitor->visit(j, element_value)) return false;
1176 }
1177 }
1178 });
1179 break;
1180 }
1181 case HOLEY_DOUBLE_ELEMENTS:
1182 case PACKED_DOUBLE_ELEMENTS: {
1183 // Disallow execution so the cached elements won't change mid execution.
1184 DisallowJavascriptExecution no_js(isolate);
1185
1186 // Empty array is FixedArray but not FixedDoubleArray.
1187 if (length == 0) break;
1188 // Run through the elements FixedArray and use HasElement and GetElement
1189 // to check the prototype for missing elements.
1190 if (IsFixedArray(array->elements())) {
1191 DCHECK_EQ(array->elements()->length(), 0);
1192 break;
1193 }
1194 DirectHandle<FixedDoubleArray> elements(
1195 Cast<FixedDoubleArray>(array->elements()), isolate);
1196 int fast_length = static_cast<int>(length);
1197 DCHECK(fast_length <= elements->length());
1198 FOR_WITH_HANDLE_SCOPE(isolate, int, j = 0, j, j < fast_length, j++, {
1199 if (!elements->is_the_hole(j)) {
1200 double double_value = elements->get_scalar(j);
1201 DirectHandle<Object> element_value =
1202 isolate->factory()->NewNumber(double_value);
1203 if (!visitor->visit(j, element_value)) return false;
1204 } else {
1205 Maybe<bool> maybe = JSReceiver::HasElement(isolate, array, j);
1206 if (maybe.IsNothing()) return false;
1207 if (maybe.FromJust()) {
1208 // Call GetElement on array, not its prototype, or getters won't
1209 // have the correct receiver.
1210 DirectHandle<Object> element_value;
1211 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
1212 isolate, element_value,
1213 JSReceiver::GetElement(isolate, array, j), false);
1214 if (!visitor->visit(j, element_value)) return false;
1215 }
1216 }
1217 });
1218 break;
1219 }
1220
1221 case DICTIONARY_ELEMENTS: {
1222 // Disallow execution so the cached dictionary won't change mid execution.
1223 DisallowJavascriptExecution no_js(isolate);
1224
1225 DirectHandle<NumberDictionary> dict(array->element_dictionary(), isolate);
1226 std::vector<uint32_t> indices;
1227 indices.reserve(dict->Capacity() / 2);
1228
1229 // Collect all indices in the object and the prototypes less
1230 // than length. This might introduce duplicates in the indices list.
1231 CollectElementIndices(isolate, array, length, &indices);
1232 std::sort(indices.begin(), indices.end());
1233 size_t n = indices.size();
1234 FOR_WITH_HANDLE_SCOPE(isolate, size_t, j = 0, j, j < n, (void)0, {
1235 uint32_t index = indices[j];
1236 DirectHandle<Object> element;
1237 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
1238 isolate, element, JSReceiver::GetElement(isolate, array, index),
1239 false);
1240 if (!visitor->visit(index, element)) return false;
1241 // Skip to next different index (i.e., omit duplicates).
1242 do {
1243 j++;
1244 } while (j < n && indices[j] == index);
1245 });
1246 break;
1247 }
1248 case FAST_SLOPPY_ARGUMENTS_ELEMENTS:
1249 case SLOW_SLOPPY_ARGUMENTS_ELEMENTS: {
1250 FOR_WITH_HANDLE_SCOPE(
1251 isolate, uint32_t, index = 0, index, index < length, index++, {
1252 DirectHandle<Object> element;
1253 ASSIGN_RETURN_ON_EXCEPTION_VALUE(
1254 isolate, element, JSReceiver::GetElement(isolate, array, index),
1255 false);
1256 if (!visitor->visit(index, element)) return false;
1257 });
1258 break;
1259 }
1260 case WASM_ARRAY_ELEMENTS:
1261 // TODO(ishell): implement
1262 UNIMPLEMENTED();
1263 case NO_ELEMENTS:
1264 break;
1265 // JSArrays cannot have the following elements kinds:
1266#define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) case TYPE##_ELEMENTS:
1267 TYPED_ARRAYS(TYPED_ARRAY_CASE)
1268 RAB_GSAB_TYPED_ARRAYS(TYPED_ARRAY_CASE)
1269#undef TYPED_ARRAY_CASE
1270 case FAST_STRING_WRAPPER_ELEMENTS:
1271 case SLOW_STRING_WRAPPER_ELEMENTS:
1272 case SHARED_ARRAY_ELEMENTS:
1273 UNREACHABLE();
1274 }
1275 visitor->increase_index_offset(length);
1276 return true;
1277}

1279static Maybe<bool> IsConcatSpreadable(Isolate* isolate,
1280 DirectHandle<Object> obj) {
1281 HandleScope handle_scope(isolate);
1282 DirectHandle<JSReceiver> receiver;
1283 if (!TryCast<JSReceiver>(obj, &receiver)) return Just(false);
1284 if (!Protectors::IsIsConcatSpreadableLookupChainIntact(isolate) ||
1285 receiver->HasProxyInPrototype(isolate)) {
1286 // Slow path if @@isConcatSpreadable has been used.
1287 DirectHandle<Symbol> key(isolate->factory()->is_concat_spreadable_symbol());
1288 DirectHandle<Object> value;
1289 MaybeDirectHandle<Object> maybeValue =
1290 i::Runtime::GetObjectProperty(isolate, receiver, key);
1291 if (!maybeValue.ToHandle(&value)) return Nothing<bool>();

1298Tagged<Object> Slow_ArrayConcat(BuiltinArguments* args,
1299 DirectHandle<Object> species,
1300 Isolate* isolate) {
1301 int argument_count = args->length();
1302
1303 bool is_array_species = *species == isolate->context()->array_function();
1304
1305 // Pass 1: estimate the length and number of elements of the result.
1306 // The actual length can be larger if any of the arguments have getters
1307 // that mutate other arguments (but will otherwise be precise).
1308 // The number of elements is precise if there are no inherited elements.
1309
1310 ElementsKind kind = PACKED_SMI_ELEMENTS;
1311
1312 uint32_t estimate_result_length = 0;
1313 uint32_t estimate_nof = 0;
1314 FOR_WITH_HANDLE_SCOPE(isolate, int, i = 0, i, i < argument_count, i++, {
1315 DirectHandle<Object> obj = args->at(i);
1316 uint32_t length_estimate;
1317 uint32_t element_estimate;
1318 if (IsJSArray(*obj)) {
1319 auto array = Cast<JSArray>(obj);
1320 length_estimate =
1321 static_cast<uint32_t>(Object::NumberValue(array->length()));
1322 if (length_estimate != 0) {
1323 ElementsKind array_kind =
1324 GetPackedElementsKind(array->GetElementsKind());
1325 if (IsAnyNonextensibleElementsKind(array_kind)) {
1326 array_kind = PACKED_ELEMENTS;
1327 }
1328 kind = GetMoreGeneralElementsKind(kind, array_kind);
1329 }
1330 element_estimate = EstimateElementCount(isolate, array);
1331 } else {
1332 if (IsHeapObject(*obj)) {
1333 kind = GetMoreGeneralElementsKind(
1334 kind, IsNumber(*obj) ? PACKED_DOUBLE_ELEMENTS : PACKED_ELEMENTS);
1335 }
1336 length_estimate = 1;
1337 element_estimate = 1;
1338 }
1339 // Avoid overflows by capping at kMaxArrayLength.
1340 if (JSArray::kMaxArrayLength - estimate_result_length < length_estimate) {
1341 estimate_result_length = JSArray::kMaxArrayLength;
1342 } else {
1343 estimate_result_length += length_estimate;
1344 }
1345 if (JSArray::kMaxArrayLength - estimate_nof < element_estimate) {
1346 estimate_nof = JSArray::kMaxArrayLength;
1347 } else {
1348 estimate_nof += element_estimate;
1349 }
1350 });
1351
1352 // If estimated number of elements is more than half of length, a
1353 // fixed array (fast case) is more time and space-efficient than a
1354 // dictionary.
1355 bool fast_case = is_array_species &&
1356 (estimate_nof * 2) >= estimate_result_length &&
1357 Protectors::IsIsConcatSpreadableLookupChainIntact(isolate);
1358
1359 if (fast_case && kind == PACKED_DOUBLE_ELEMENTS) {
1360 DirectHandle<FixedArrayBase> storage =
1361 isolate->factory()->NewFixedDoubleArray(estimate_result_length);
1362 int j = 0;
1363 bool failure = false;
1364 if (estimate_result_length > 0) {
1365 auto double_storage = Cast<FixedDoubleArray>(storage);
1366 for (int i = 0; i < argument_count; i++) {
1367 DirectHandle<Object> obj = args->at(i);
1368 if (IsSmi(*obj)) {
1369 double_storage->set(j, Smi::ToInt(*obj));
1370 j++;
1371 } else if (IsNumber(*obj)) {
1372 double_storage->set(j, Object::NumberValue(*obj));
1373 j++;
1374 } else {
1375 DisallowGarbageCollection no_gc;
1376 Tagged<JSArray> array = Cast<JSArray>(*obj);
1377 uint32_t length =
1378 static_cast<uint32_t>(Object::NumberValue(array->length()));
1379 switch (array->GetElementsKind()) {
1380 case HOLEY_DOUBLE_ELEMENTS:
1381 case PACKED_DOUBLE_ELEMENTS: {
1382 // Empty array is FixedArray but not FixedDoubleArray.
1383 if (length == 0) break;
1384 Tagged<FixedDoubleArray> elements =
1385 Cast<FixedDoubleArray>(array->elements());
1386 for (uint32_t k = 0; k < length; k++) {
1387 if (elements->is_the_hole(k)) {
1388 // TODO(jkummerow/verwaest): We could be a bit more clever
1389 // here: Check if there are no elements/getters on the
1390 // prototype chain, and if so, allow creation of a holey
1391 // result array.
1392 // Same thing below (holey smi case).
1393 failure = true;
1394 break;
1395 }
1396 double double_value = elements->get_scalar(k);
1397 double_storage->set(j, double_value);
1398 j++;
1399 }
1400 break;
1401 }
1402 case HOLEY_SMI_ELEMENTS:
1403 case PACKED_SMI_ELEMENTS: {
1404 Tagged<Object> the_hole = ReadOnlyRoots(isolate).the_hole_value();
1405 Tagged<FixedArray> elements(Cast<FixedArray>(array->elements()));
1406 for (uint32_t k = 0; k < length; k++) {
1407 Tagged<Object> element = elements->get(k);
1408 if (element == the_hole) {
1409 failure = true;
1410 break;
1411 }
1412 int32_t int_value = Smi::ToInt(element);
1413 double_storage->set(j, int_value);
1414 j++;
1415 }
1416 break;
1417 }
1418 case HOLEY_ELEMENTS:
1419 case HOLEY_FROZEN_ELEMENTS:
1420 case HOLEY_SEALED_ELEMENTS:
1421 case HOLEY_NONEXTENSIBLE_ELEMENTS:
1422 case PACKED_ELEMENTS:
1423 case PACKED_FROZEN_ELEMENTS:
1424 case PACKED_SEALED_ELEMENTS:
1425 case PACKED_NONEXTENSIBLE_ELEMENTS:
1426 case DICTIONARY_ELEMENTS:
1427 case NO_ELEMENTS:
1428 DCHECK_EQ(0u, length);
1429 break;
1430 default:
1431 UNREACHABLE();
1432 }
1433 }
1434 if (failure) {
1435#ifdef VERIFY_HEAP
1436 // The allocated storage may contain uninitialized values which will
1437 // cause FixedDoubleArray::FixedDoubleArrayVerify to fail, when the
1438 // heap is verified (see: crbug.com/1415071). To prevent this, we
1439 // initialize the array with holes.
1440 if (v8_flags.verify_heap) {
1441 double_storage->FillWithHoles(0, estimate_result_length);
1442 }
1443#endif // VERIFY_HEAP
1444 break;
1445 }
1446 }
1447 }
1448 if (!failure) {
1449 return *isolate->factory()->NewJSArrayWithElements(storage, kind, j);
1450 }
1451 // In case of failure, fall through.
1452 }
1453
1454 DirectHandle<UnionOf<JSReceiver, FixedArray, NumberDictionary>> storage;
1455 if (fast_case) {
1456 // The backing storage array must have non-existing elements to preserve
1457 // holes across concat operations.
1458 storage =
1459 isolate->factory()->NewFixedArrayWithHoles(estimate_result_length);
1460 } else if (is_array_species) {
1461 storage = NumberDictionary::New(isolate, estimate_nof);
1462 } else {
1463 DCHECK(IsConstructor(*species));
1464 DirectHandle<Object> length(Smi::zero(), isolate);
1465 DirectHandle<JSReceiver> storage_object;
1466 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
1467 isolate, storage_object,
1468 Execution::New(isolate, species, species, {&length, 1}));
1469 storage = storage_object;
1470 }
1471
1472 ArrayConcatVisitor visitor(isolate, storage, fast_case);
1473
1474 for (int i = 0; i < argument_count; i++) {
1475 DirectHandle<Object> obj = args->at(i);
1476 Maybe<bool> spreadable = IsConcatSpreadable(isolate, obj);
1477 MAYBE_RETURN(spreadable, ReadOnlyRoots(isolate).exception());
1478 if (spreadable.FromJust()) {
1479 DirectHandle<JSReceiver> object = Cast<JSReceiver>(obj);
1480 if (!IterateElements(isolate, object, &visitor)) {
1481 return ReadOnlyRoots(isolate).exception();
1482 }
1483 } else {
1484 if (!visitor.visit(0, obj)) return ReadOnlyRoots(isolate).exception();
1485 visitor.increase_index_offset(1);
1486 }
1487 }
1488
1489 if (visitor.exceeds_array_limit()) {
1490 THROW_NEW_ERROR_RETURN_FAILURE(
1491 isolate, NewRangeError(MessageTemplate::kInvalidArrayLength));
1492 }
1493
1494 if (is_array_species) {
1495 return *visitor.ToArray();
1496 } else {
1497 RETURN_RESULT_OR_FAILURE(isolate, visitor.ToJSReceiver());
1498 }
1499}

1515MaybeHandle<JSArray> Fast_ArrayConcat(Isolate* isolate,
1516 BuiltinArguments* args) {
1517 if (!Protectors::IsIsConcatSpreadableLookupChainIntact(isolate)) {
1518 return {};
1519 }
1520 // We shouldn't overflow when adding another len.
1521 const int kHalfOfMaxInt = 1 << (kBitsPerInt - 2);
1522 static_assert(FixedArray::kMaxLength < kHalfOfMaxInt);
1523 static_assert(FixedDoubleArray::kMaxLength < kHalfOfMaxInt);
1524 USE(kHalfOfMaxInt);
1525
1526 int n_arguments = args->length();
1527 int result_len = 0;
1528 {
1529 DisallowGarbageCollection no_gc;
1530 // Iterate through all the arguments performing checks
1531 // and calculating total length.
1532 for (int i = 0; i < n_arguments; i++) {
1533 Tagged<Object> arg = (*args)[i];
1534 if (!IsJSArray(arg)) return {};
1535 if (!HasOnlySimpleReceiverElements(isolate, Cast<JSObject>(arg))) {
1536 return {};
1537 }
1538 // TODO(cbruni): support fast concatenation of DICTIONARY_ELEMENTS.
1539 if (!Cast<JSObject>(arg)->HasFastElements()) {
1540 return {};
1541 }
1542 DirectHandle<JSArray> array(Cast<JSArray>(arg), isolate);
1543 if (!IsSimpleArray(isolate, array)) {
1544 return {};
1545 }
1546 // The Array length is guaranted to be <= kHalfOfMaxInt thus we won't
1547 // overflow.
1548 result_len += Smi::ToInt(array->length());
1549 DCHECK_GE(result_len, 0);
1550 // Throw an Error if we overflow the FixedArray limits
1551 if (FixedDoubleArray::kMaxLength < result_len ||
1552 FixedArray::kMaxLength < result_len) {
1553 AllowGarbageCollection gc;
1554 THROW_NEW_ERROR(isolate,
1555 NewRangeError(MessageTemplate::kInvalidArrayLength));
1556 }
1557 }
1558 }
1559 return ElementsAccessor::Concat(isolate, args, n_arguments, result_len);
1560}
