86 static constexpr intptr_t kExternalResourceIsOneByteTag = 1;
87 static constexpr intptr_t kExternalResourceEncodingMask = 1;
88 static constexpr intptr_t kExternalResourceAddressMask =
89 ~kExternalResourceEncodingMask;
90
91 // Always a pointer to the string that needs to be transitioned.
92 Tagged_t original_string_;
93 // The field either stores the forward string object, or a raw hash.
94 // For strings forwarded to an internalized string (to be converted to a
95 // ThinString during GC), this field always contrains the internalized string
96 // object.
97 // It is guaranteed that only computed hash values (LSB = 0) are stored,
98 // therefore a raw hash is distinguishable from a string object by the
99 // heap object tag.
100 // Raw hashes can be overwritten by forward string objects, whereas
101 // forward string objects will never be overwritten once set.
102 Tagged_t forward_string_or_hash_;
103 // Although this is an external pointer, we are using Address instead of
104 // ExternalPointer_t to not have to deal with the ExternalPointerTable.
105 // This is OK, as the StringForwardingTable is outside of the V8 sandbox.
106 // The LSB is used to indicate whether the external resource is a one-byte
107 // (LSB = 1) or two-byte (LSB = 0) external string resource.
108 Address external_resource_;
109
110 // Possible string transitions and how they affect the fields of the record:
111 // Shared string (not in the table) --> Interalized
112 // forward_string_or_hash_ is set to the internalized string object.
113 // external_resource_ is nullptr.
114 // Shared string (not in the table) --> External
115 // forward_string_or_hash_ is set to the computed hash value of the string.
116 // external_resource_ is set to the address of the external resource.
117 // Shared string (in the table to be internalized) --> External
118 // forward_string_or_hash_ will not be overwritten. It will still contain
119 // the internalized string object from the previous transition.
120 // external_resource_ is set to the address of the external resource.
121 // Shared string (in the table to be made external) --> Internalized
122 // forward_string_or_hash_ (previously contained the computed hash value) is
123 // overwritten with the internalized string object.
124 // external_resource_ is not overwritten (still the external resource).
125
126 friend class StringForwardingTable::Block;
127};

129uint32_t StringForwardingTable::Record::raw_hash(
130 PtrComprCageBase cage_base) const {
131 Tagged<Object> hash_or_string = ForwardStringObjectOrHash(cage_base);
132 uint32_t raw_hash;
133 if (IsHeapObject(hash_or_string)) {
134 raw_hash = Cast<String>(hash_or_string)->RawHash();
135 } else {
136 raw_hash = static_cast<uint32_t>(hash_or_string.ptr());
137 }
138 DCHECK(Name::IsHashFieldComputed(raw_hash));
139 return raw_hash;
140}

142v8::String::ExternalStringResourceBase*
143StringForwardingTable::Record::external_resource(bool* is_one_byte) const {
144 Address address = ExternalResourceAddress();
145 *is_one_byte = (address & kExternalResourceEncodingMask) ==
146 kExternalResourceIsOneByteTag;
147 address &= kExternalResourceAddressMask;
148 return reinterpret_cast<v8::String::ExternalStringResourceBase*>(address);
149}

151void StringForwardingTable::Record::set_raw_hash_if_empty(uint32_t raw_hash) {
152 // Assert that computed hash values don't overlap with heap object tag.
153 static_assert((kHeapObjectTag & Name::kHashNotComputedMask) != 0);
154 DCHECK(Name::IsHashFieldComputed(raw_hash));
155 DCHECK_NE(raw_hash & kHeapObjectTagMask, kHeapObjectTag);
156 AsAtomicTagged::Release_CompareAndSwap(&forward_string_or_hash_,
157 unused_element().value(), raw_hash);
158}

187 DCHECK_NOT_NULL(resource);
188 Address address = reinterpret_cast<Address>(resource);

195bool StringForwardingTable::Record::TryUpdateExternalResource(Address address) {
196 static_assert(kNullAddress == kNullExternalPointer);
197 // Don't set the external resource if another one is already stored. If we
198 // would simply overwrite the resource, the previously stored one would be
199 // leaked.
200 return base::AsAtomicPointer::AcquireRelease_CompareAndSwap(
201 &external_resource_, kNullAddress, address) == kNullAddress;
202}

205 bool is_one_byte;

207 DCHECK_NOT_NULL(resource);

209}

211void StringForwardingTable::Record::DisposeUnusedExternalResource(
212 Isolate* isolate, Tagged<String> original) {
213#ifdef DEBUG
214 Tagged<String> stored_original = original_string(isolate);
215 if (IsThinString(stored_original)) {
216 stored_original = Cast<ThinString>(stored_original)->actual();
217 }
218 DCHECK_EQ(original, stored_original);
219#endif
220 if (!IsExternalString(original)) return;
221 Address original_resource =
222 Cast<ExternalString>(original)->resource_as_address();
223 bool is_one_byte;
224 auto resource = external_resource(&is_one_byte);
225 if (resource != nullptr &&
226 reinterpret_cast<Address>(resource) != original_resource) {
227 resource->Dispose();
228 }

300// static
301uint32_t StringForwardingTable::BlockForIndex(int index,
302 uint32_t* index_in_block) {
303 DCHECK_GE(index, 0);
304 DCHECK_NOT_NULL(index_in_block);
305 // The block is the leftmost set bit of the index, corrected by the size
306 // of the first block.
307 const uint32_t block_index =
308 kBitsPerInt -
309 base::bits::CountLeadingZeros(
310 static_cast<uint32_t>(index + kInitialBlockSize)) -
311 kInitialBlockSizeHighestBit - 1;
312 *index_in_block = IndexInBlock(index, block_index);
313 return block_index;
314}
315
316// static
317uint32_t StringForwardingTable::IndexInBlock(int index, uint32_t block_index) {
318 DCHECK_GE(index, 0);
319 // Clear out the leftmost set bit (the block index) to get the index within
320 // the block.
321 return static_cast<uint32_t>(index + kInitialBlockSize) &
322 ~(1u << (block_index + kInitialBlockSizeHighestBit));
323}
324
325// static
326uint32_t StringForwardingTable::CapacityForBlock(uint32_t block_index) {
327 return 1u << (block_index + kInitialBlockSizeHighestBit);
328}

330template <typename Func>
331void StringForwardingTable::IterateElements(Func&& callback) {
332 if (empty()) return;
333 BlockVector* blocks = blocks_.load(std::memory_order_relaxed);
334 const uint32_t last_block_index = static_cast<uint32_t>(blocks->size() - 1);
335 for (uint32_t block_index = 0; block_index < last_block_index;
336 ++block_index) {
337 Block* block = blocks->LoadBlock(block_index);
338 for (int index = 0; index < block->capacity(); ++index) {
339 Record* rec = block->record(index);
340 callback(rec);
341 }
342 }
343 // Handle last block separately, as it is not filled to capacity.
344 const uint32_t max_index = IndexInBlock(size() - 1, last_block_index) + 1;
345 Block* block = blocks->LoadBlock(last_block_index);
346 for (uint32_t index = 0; index < max_index; ++index) {
347 Record* rec = block->record(index);
348 callback(rec);
349 }
350}
