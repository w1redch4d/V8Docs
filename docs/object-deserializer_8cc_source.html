36MaybeDirectHandle<HeapObject> ObjectDeserializer::Deserialize() {
37 DCHECK(deserializing_user_code());
38 HandleScope scope(isolate());

42 DeserializeDeferredObjects();
43 CHECK(new_code_objects().empty());
44 LinkAllocationSites();

54void ObjectDeserializer::CommitPostProcessedObjects() {
55 for (DirectHandle<Script> script : new_scripts()) {
56 // Assign a new script id to avoid collision.
57 script->set_id(isolate()->GetNextScriptId());
58 LogScriptEvents(*script);
59 // Add script to list.
60 Handle<WeakArrayList> list = isolate()->factory()->script_list();
61 list = WeakArrayList::AddToEnd(isolate(), list,
62 MaybeObjectDirectHandle::Weak(script));
63 isolate()->heap()->SetRootScriptList(*list);
64 }
65}

67void ObjectDeserializer::LinkAllocationSites() {
68 DisallowGarbageCollection no_gc;
69 Heap* heap = isolate()->heap();
70 // Allocation sites are present in the snapshot, and must be linked into
71 // a list at deserialization time.
72 for (DirectHandle<AllocationSite> site : new_allocation_sites()) {
73 if (!site->HasWeakNext()) continue;
74 // TODO(mvstanton): consider treating the heap()->allocation_sites_list()
75 // as a (weak) root. If this root is relocated correctly, this becomes
76 // unnecessary.
77 if (heap->allocation_sites_list() == Smi::zero()) {
78 site->set_weak_next(ReadOnlyRoots(heap).undefined_value());
79 } else {
80 site->set_weak_next(heap->allocation_sites_list());
81 }
82 heap->set_allocation_sites_list(*site);
83 }
84}

91MaybeDirectHandle<SharedFunctionInfo>
92OffThreadObjectDeserializer::DeserializeSharedFunctionInfo(
93 LocalIsolate* isolate, const SerializedCodeData* data,
94 std::vector<IndirectHandle<Script>>* deserialized_scripts) {
95 OffThreadObjectDeserializer d(isolate, data);
96
97 // Attach the empty string as the source.
98 d.AddAttachedObject(isolate->factory()->empty_string());
99
100 DirectHandle<HeapObject> result;
101 if (!d.Deserialize(deserialized_scripts).ToHandle(&result)) {
102 return MaybeDirectHandle<SharedFunctionInfo>();
103 }
104 return Cast<SharedFunctionInfo>(result);
105}

107MaybeDirectHandle<HeapObject> OffThreadObjectDeserializer::Deserialize(
108 std::vector<IndirectHandle<Script>>* deserialized_scripts) {
109 DCHECK(deserializing_user_code());
110 LocalHandleScope scope(isolate());
111 DirectHandle<HeapObject> result;
112 {
113 result = ReadObject();
114 DeserializeDeferredObjects();
115 CHECK(new_code_objects().empty());
116 CHECK(new_allocation_sites().empty());
117 CHECK(new_maps().empty());
118 WeakenDescriptorArrays();
119 }
120
121 Rehash();
122
123 // TODO(leszeks): Figure out a better way of dealing with scripts.
124 CHECK_EQ(new_scripts().size(), 1);
125 for (DirectHandle<Script> script : new_scripts()) {
126 // Assign a new script id to avoid collision.
127 script->set_id(isolate()->GetNextScriptId());
128 LogScriptEvents(*script);
129 deserialized_scripts->push_back(
130 isolate()->heap()->NewPersistentHandle(script));
131 }
132
133 return scope.CloseAndEscape(result);
134}
