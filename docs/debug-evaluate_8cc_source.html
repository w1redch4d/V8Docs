44MaybeHandle<Object> DebugEvaluate::Global(Isolate* isolate,
45 Handle<String> source,
46 debug::EvaluateGlobalMode mode,
47 REPLMode repl_mode) {
48 DirectHandle<SharedFunctionInfo> shared_info;
49 if (!GetFunctionInfo(isolate, source, repl_mode).ToHandle(&shared_info)) {
50 return MaybeHandle<Object>();
51 }
52
53 DirectHandle<NativeContext> context = isolate->native_context();
54 DirectHandle<JSFunction> function =
55 Factory::JSFunctionBuilder{isolate, shared_info, context}.Build();
56
57 DisableBreak disable_break_scope(
58 isolate->debug(),
59 mode == debug::EvaluateGlobalMode::kDisableBreaks ||
60 mode ==
61 debug::EvaluateGlobalMode::kDisableBreaksAndThrowOnSideEffect);
62
63 if (mode == debug::EvaluateGlobalMode::kDisableBreaksAndThrowOnSideEffect) {
64 isolate->debug()->StartSideEffectCheckMode();
65 }
66 // TODO(cbruni, 1244145): Use host-defined options from script context.
67 DirectHandle<FixedArray> host_defined_options(
68 Cast<Script>(function->shared()->script())->host_defined_options(),
69 isolate);
70 MaybeHandle<Object> result = Execution::CallScript(
71 isolate, function,
72 DirectHandle<JSObject>(context->global_proxy(), isolate),
73 host_defined_options);
74 if (mode == debug::EvaluateGlobalMode::kDisableBreaksAndThrowOnSideEffect) {
75 isolate->debug()->StopSideEffectCheckMode();
76 }
77 return result;
78}

80MaybeHandle<Object> DebugEvaluate::Local(Isolate* isolate,
81 StackFrameId frame_id,
82 int inlined_jsframe_index,
83 Handle<String> source,
84 bool throw_on_side_effect) {
85 // Handle the processing of break.
86 DisableBreak disable_break_scope(isolate->debug());
87
88 // Get the frame where the debugging is performed.
89 DebuggableStackFrameIterator it(isolate, frame_id);
90#if V8_ENABLE_WEBASSEMBLY
91 if (it.is_wasm()) {
92#if V8_ENABLE_DRUMBRAKE
93 // TODO(paolosev@microsoft.com) - Not supported by Wasm interpreter.
94 if (it.is_wasm_interpreter_entry()) return {};
95#endif // V8_ENABLE_DRUMBRAKE
96 WasmFrame* frame = WasmFrame::cast(it.frame());
97 Handle<SharedFunctionInfo> outer_info(
98 isolate->native_context()->empty_function()->shared(), isolate);
99 DirectHandle<JSObject> context_extension = GetWasmDebugProxy(frame);
100 DirectHandle<ScopeInfo> scope_info =
101 ScopeInfo::CreateForWithScope(isolate, Handle<ScopeInfo>::null());
102 DirectHandle<Context> context = isolate->factory()->NewWithContext(
103 isolate->native_context(), scope_info, context_extension);
104 return Evaluate(isolate, outer_info, context, context_extension, source,
105 throw_on_side_effect);
106 }
107#endif // V8_ENABLE_WEBASSEMBLY
108
109 CHECK(it.is_javascript());
110 JavaScriptFrame* frame = it.javascript_frame();
111 // This is not a lot different than DebugEvaluate::Global, except that
112 // variables accessible by the function we are evaluating from are
113 // materialized and included on top of the native context. Changes to
114 // the materialized object are written back afterwards.
115 // Note that the native context is taken from the original context chain,
116 // which may not be the current native context of the isolate.
117 ContextBuilder context_builder(isolate, frame, inlined_jsframe_index);
118 if (isolate->has_exception()) return {};
119
120 DirectHandle<Context> context = context_builder.evaluation_context();
121 DirectHandle<JSObject> receiver(context->global_proxy(), isolate);
122 MaybeHandle<Object> maybe_result =
123 Evaluate(isolate, context_builder.outer_info(), context, receiver, source,
124 throw_on_side_effect);
125 if (!maybe_result.is_null()) context_builder.UpdateValues();
126 return maybe_result;
127}

129MaybeHandle<Object> DebugEvaluate::WithTopmostArguments(Isolate* isolate,
130 Handle<String> source) {
131 // Handle the processing of break.
132 DisableBreak disable_break_scope(isolate->debug());
133 Factory* factory = isolate->factory();
134 JavaScriptStackFrameIterator it(isolate);
135
136 // Get context and receiver.
137 DirectHandle<Context> native_context(
138 Cast<Context>(it.frame()->context())->native_context(), isolate);
139
140 // Materialize arguments as property on an extension object.
141 DirectHandle<JSObject> materialized = factory->NewSlowJSObjectWithNullProto();
142 DirectHandle<String> arguments_str = factory->arguments_string();
143 JSObject::SetOwnPropertyIgnoreAttributes(
144 materialized, arguments_str,
145 Accessors::FunctionGetArguments(it.frame(), 0), NONE)
146 .Check();
147
148 // Materialize receiver.
149 Handle<Object> this_value(it.frame()->receiver(), isolate);
150 DCHECK_EQ(it.frame()->IsConstructor(), IsTheHole(*this_value, isolate));
151 if (!IsTheHole(*this_value, isolate)) {
152 DirectHandle<String> this_str = factory->this_string();
153 JSObject::SetOwnPropertyIgnoreAttributes(materialized, this_str, this_value,
154 NONE)
155 .Check();
156 }
157
158 // Use extension object in a debug-evaluate scope.
159 DirectHandle<ScopeInfo> scope_info =
160 ScopeInfo::CreateForWithScope(isolate, Handle<ScopeInfo>::null());
161 scope_info->SetIsDebugEvaluateScope();
162 DirectHandle<Context> evaluation_context = factory->NewDebugEvaluateContext(
163 native_context, scope_info, materialized, DirectHandle<Context>());
164 Handle<SharedFunctionInfo> outer_info(
165 native_context->empty_function()->shared(), isolate);
166 DirectHandle<JSObject> receiver(native_context->global_proxy(), isolate);
167 const bool throw_on_side_effect = false;
168 MaybeHandle<Object> maybe_result =
169 Evaluate(isolate, outer_info, evaluation_context, receiver, source,
170 throw_on_side_effect);
171 return maybe_result;
172}

174// Compile and evaluate source for the given context.
175MaybeHandle<Object> DebugEvaluate::Evaluate(
176 Isolate* isolate, Handle<SharedFunctionInfo> outer_info,
177 DirectHandle<Context> context, DirectHandle<Object> receiver,
178 Handle<String> source, bool throw_on_side_effect) {
179 DirectHandle<JSFunction> eval_fun;
180 ASSIGN_RETURN_ON_EXCEPTION(
181 isolate, eval_fun,
182 Compiler::GetFunctionFromEval(source, outer_info, context,
183 LanguageMode::kSloppy, NO_PARSE_RESTRICTION,
184 kNoSourcePosition, kNoSourcePosition,
185 ParsingWhileDebugging::kYes));
186
187 Handle<Object> result;
188 bool success = false;
189 if (throw_on_side_effect) isolate->debug()->StartSideEffectCheckMode();
190 success = Execution::Call(isolate, eval_fun, receiver, {}).ToHandle(&result);
191 if (throw_on_side_effect) isolate->debug()->StopSideEffectCheckMode();
192 if (!success) DCHECK(isolate->has_exception());
193 return success ? result : MaybeHandle<Object>();
194}

200DebugEvaluate::ContextBuilder::ContextBuilder(Isolate* isolate,
201 JavaScriptFrame* frame,
202 int inlined_jsframe_index)
203 : isolate_(isolate),
204 frame_inspector_(frame, inlined_jsframe_index, isolate),
205 scope_iterator_(isolate, &frame_inspector_,
206 ScopeIterator::ReparseStrategy::kScriptIfNeeded) {
207 Handle<Context> outer_context(frame_inspector_.GetFunction()->context(),
208 isolate);
209 evaluation_context_ = outer_context;
210 Factory* factory = isolate->factory();
211
212 if (scope_iterator_.Done()) return;
213
214 // To evaluate as if we were running eval at the point of the debug break,
215 // we reconstruct the context chain as follows:
216 // - To make stack-allocated variables visible, we materialize them and
217 // use a debug-evaluate context to wrap both the materialized object and
218 // the original context.
219 // - Each scope from the break position up to the function scope is wrapped
220 // in a debug-evaluate context.
221 // - Between the function scope and the native context, we only resolve
222 // variable names that are guaranteed to not be shadowed by stack-allocated
223 // variables. ScopeInfos between the function scope and the native
224 // context have a blocklist attached to implement that.
225 // - The various block lists are calculated by the ScopeIterator during
226 // iteration.
227 // Context::Lookup has special handling for debug-evaluate contexts:
228 // - Look up in the materialized stack variables.
229 // - Look up in the original context.
230 // - Once we have seen a debug-evaluate context we start to take the
231 // block lists into account before moving up the context chain.
232 for (; scope_iterator_.InInnerScope(); scope_iterator_.Next()) {
233 ScopeIterator::ScopeType scope_type = scope_iterator_.Type();
234 if (scope_type == ScopeIterator::ScopeTypeScript) break;
235 ContextChainElement context_chain_element;
236 if (scope_type == ScopeIterator::ScopeTypeLocal ||
237 scope_iterator_.DeclaresLocals(ScopeIterator::Mode::STACK)) {
238 context_chain_element.materialized_object =
239 scope_iterator_.ScopeObject(ScopeIterator::Mode::STACK);
240 }
241 if (scope_iterator_.HasContext()) {
242 context_chain_element.wrapped_context = scope_iterator_.CurrentContext();
243 }
244 context_chain_.push_back(context_chain_element);
245 }
246
247 DirectHandle<ScopeInfo> scope_info =
248 IsNativeContext(*evaluation_context_)
249 ? DirectHandle<ScopeInfo>::null()
250 : direct_handle(evaluation_context_->scope_info(), isolate);
251 for (auto rit = context_chain_.rbegin(); rit != context_chain_.rend();
252 rit++) {
253 ContextChainElement element = *rit;
254 scope_info = ScopeInfo::CreateForWithScope(isolate, scope_info);
255 scope_info->SetIsDebugEvaluateScope();
256
257 // In the case where the "paused function scope" is the script scope
258 // itself, we don't need (and don't have) a blocklist.
259 const bool paused_scope_is_script_scope =
260 scope_iterator_.Done() || scope_iterator_.InInnerScope();
261 if (rit == context_chain_.rbegin() && !paused_scope_is_script_scope) {
262 // The DebugEvaluateContext we create for the closure scope is the only
263 // DebugEvaluateContext with a block list. This means we'll retrieve
264 // the existing block list from the paused function scope
265 // and also associate the temporary scope_info we create here with that
266 // blocklist.
267 DirectHandle<ScopeInfo> function_scope_info(
268 frame_inspector_.GetFunction()->shared()->scope_info(), isolate_);
269 DirectHandle<Object> block_list(
270 isolate_->LocalsBlockListCacheGet(function_scope_info), isolate_);
271 CHECK(IsStringSet(*block_list));
272 isolate_->LocalsBlockListCacheSet(scope_info, Handle<ScopeInfo>::null(),
273 Cast<StringSet>(block_list));
274 }
275
276 evaluation_context_ = factory->NewDebugEvaluateContext(
277 evaluation_context_, scope_info, element.materialized_object,
278 element.wrapped_context);
279 }
280}

282void DebugEvaluate::ContextBuilder::UpdateValues() {
283 scope_iterator_.Restart();
284 for (ContextChainElement& element : context_chain_) {
285 if (!element.materialized_object.is_null()) {
286 DirectHandle<FixedArray> keys =
287 KeyAccumulator::GetKeys(isolate_, element.materialized_object,
288 KeyCollectionMode::kOwnOnly,
289 ENUMERABLE_STRINGS)
290 .ToHandleChecked();
291
292 for (int i = 0; i < keys->length(); i++) {
293 DCHECK(IsString(keys->get(i)));
294 Handle<String> key(Cast<String>(keys->get(i)), isolate_);
295 DirectHandle<Object> value = JSReceiver::GetDataProperty(
296 isolate_, element.materialized_object, key);
297 scope_iterator_.SetVariableValue(key, value);
298 }
299 }
300 scope_iterator_.Next();
301 }

1078// static
1079DebugInfo::SideEffectState DebugEvaluate::FunctionGetSideEffectState(
1080 Isolate* isolate, DirectHandle<SharedFunctionInfo> info) {
1081 if (v8_flags.trace_side_effect_free_debug_evaluate) {
1082 PrintF("[debug-evaluate] Checking function %s for side effect.\n",
1083 info->DebugNameCStr().get());
1084 }
1085
1086 DCHECK(info->is_compiled());
1087 DCHECK(!info->needs_script_context());
1088 if (info->HasBytecodeArray()) {
1089 // Check bytecodes against allowlist.
1090 Handle<BytecodeArray> bytecode_array(info->GetBytecodeArray(isolate),
1091 isolate);
1092 if (v8_flags.trace_side_effect_free_debug_evaluate) {
1093 Print(*bytecode_array);
1094 }
1095 bool requires_runtime_checks = false;
1096 for (interpreter::BytecodeArrayIterator it(bytecode_array); !it.done();
1097 it.Advance()) {
1098 interpreter::Bytecode bytecode = it.current_bytecode();
1099 if (BytecodeHasNoSideEffect(bytecode)) continue;
1100 if (BytecodeRequiresRuntimeCheck(bytecode)) {
1101 requires_runtime_checks = true;
1102 continue;
1103 }
1104
1105 if (v8_flags.trace_side_effect_free_debug_evaluate) {
1106 PrintF("[debug-evaluate] bytecode %s may cause side effect.\n",
1107 interpreter::Bytecodes::ToString(bytecode));
1108 }
1109
1110 // Did not match allowlist.
1111 return DebugInfo::kHasSideEffects;
1112 }
1113 return requires_runtime_checks ? DebugInfo::kRequiresRuntimeChecks
1114 : DebugInfo::kHasNoSideEffect;
1115 } else if (info->IsApiFunction()) {
1116 Tagged<Code> code = info->GetCode(isolate);
1117 if (code->is_builtin()) {
1118 return code->builtin_id() == Builtin::kHandleApiCallOrConstruct
1119 ? DebugInfo::kHasNoSideEffect
1120 : DebugInfo::kHasSideEffects;
1121 }
1122 } else {
1123 // Check built-ins against allowlist.
1124 Builtin builtin =
1125 info->HasBuiltinId() ? info->builtin_id() : Builtin::kNoBuiltinId;
1126 if (!Builtins::IsBuiltinId(builtin)) return DebugInfo::kHasSideEffects;
1127 DebugInfo::SideEffectState state = BuiltinGetSideEffectState(builtin);
1128 return state;
1129 }
1130
1131 return DebugInfo::kHasSideEffects;
1132}

1308void DebugEvaluate::VerifyTransitiveBuiltins(Isolate* isolate) {
1309 // TODO(yangguo): also check runtime calls.
1310 bool failed = false;
1311 bool sanity_check = false;
1312 for (Builtin caller = Builtins::kFirst; caller <= Builtins::kLast; ++caller) {
1313 DebugInfo::SideEffectState state = BuiltinGetSideEffectState(caller);
1314 if (state != DebugInfo::kHasNoSideEffect) continue;
1315 Tagged<Code> code = isolate->builtins()->code(caller);
1316 int mode = RelocInfo::ModeMask(RelocInfo::CODE_TARGET) |
1317 RelocInfo::ModeMask(RelocInfo::RELATIVE_CODE_TARGET);
1318
1319 for (RelocIterator it(code, mode); !it.done(); it.next()) {
1320 RelocInfo* rinfo = it.rinfo();
1321 DCHECK(RelocInfo::IsCodeTargetMode(rinfo->rmode()));
1322 Tagged<Code> lookup_result =
1323 isolate->heap()->FindCodeForInnerPointer(rinfo->target_address());
1324 Builtin callee = lookup_result->builtin_id();
1325 if (BuiltinGetSideEffectState(callee) == DebugInfo::kHasNoSideEffect) {
1326 continue;
1327 }
1328 if (TransitivelyCalledBuiltinHasNoSideEffect(caller, callee)) {
1329 sanity_check = true;
1330 continue;
1331 }
1332 PrintF("Allowlisted builtin %s calls non-allowlisted builtin %s\n",
1333 Builtins::name(caller), Builtins::name(callee));
1334 failed = true;
1335 }

1345 USE(sanity_check);
