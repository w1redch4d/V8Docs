38Address AccessorInfo::getter(i::IsolateForSandbox isolate) const {
39 Address result = maybe_redirected_getter(isolate);
40 if (!USE_SIMULATOR_BOOL) return result;
41 if (result == kNullAddress) return kNullAddress;
42 return ExternalReference::UnwrapRedirection(result);
43}

53void AccessorInfo::set_getter(i::IsolateForSandbox isolate, Address value) {

60void AccessorInfo::init_getter_redirection(i::IsolateForSandbox isolate) {
61 CHECK(USE_SIMULATOR_BOOL);
62 Address value = maybe_redirected_getter(isolate);
63 if (value == kNullAddress) return;
64 value =
65 ExternalReference::Redirect(value, ExternalReference::DIRECT_GETTER_CALL);
66 set_maybe_redirected_getter(isolate, value);
67}
68
69void AccessorInfo::remove_getter_redirection(i::IsolateForSandbox isolate) {
70 CHECK(USE_SIMULATOR_BOOL);
71 Address value = getter(isolate);
72 set_maybe_redirected_getter(isolate, value);
73}

93void AccessorInfo::set_setter_side_effect_type(SideEffectType value) {
94 // We do not support describing setters as having no side effect, since
95 // calling set accessors must go through a store bytecode. Store bytecodes
96 // support checking receivers for temporary objects, but still expect
97 // the receiver to be written to.
98 CHECK_NE(value, SideEffectType::kHasNoSideEffect);
99 set_flags(SetterSideEffectTypeBits::update(flags(), value));
100}
