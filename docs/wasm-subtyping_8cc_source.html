22bool ValidStructSubtypeDefinition(ModuleTypeIndex subtype_index,
23 ModuleTypeIndex supertype_index,
24 const WasmModule* sub_module,
25 const WasmModule* super_module) {
26 const StructType* sub_struct = sub_module->type(subtype_index).struct_type;
27 const StructType* super_struct =
28 super_module->type(supertype_index).struct_type;
29
30 if (sub_struct->field_count() < super_struct->field_count()) {
31 return false;
32 }
33
34 for (uint32_t i = 0; i < super_struct->field_count(); i++) {
35 bool sub_mut = sub_struct->mutability(i);
36 bool super_mut = super_struct->mutability(i);
37 if (sub_mut != super_mut ||
38 (sub_mut &&
39 !EquivalentTypes(sub_struct->field(i), super_struct->field(i),
40 sub_module, super_module)) ||
41 (!sub_mut && !IsSubtypeOf(sub_struct->field(i), super_struct->field(i),
42 sub_module, super_module))) {
43 return false;
44 }
45 }
46 return true;
47}

49bool ValidArraySubtypeDefinition(ModuleTypeIndex subtype_index,
50 ModuleTypeIndex supertype_index,
51 const WasmModule* sub_module,
52 const WasmModule* super_module) {
53 const ArrayType* sub_array = sub_module->type(subtype_index).array_type;
54 const ArrayType* super_array = super_module->type(supertype_index).array_type;
55 bool sub_mut = sub_array->mutability();
56 bool super_mut = super_array->mutability();
57
58 return (sub_mut && super_mut &&
59 EquivalentTypes(sub_array->element_type(),
60 super_array->element_type(), sub_module,
61 super_module)) ||
62 (!sub_mut && !super_mut &&
63 IsSubtypeOf(sub_array->element_type(), super_array->element_type(),
64 sub_module, super_module));
65}

67bool ValidFunctionSubtypeDefinition(ModuleTypeIndex subtype_index,
68 ModuleTypeIndex supertype_index,
69 const WasmModule* sub_module,
70 const WasmModule* super_module) {
71 const FunctionSig* sub_func = sub_module->type(subtype_index).function_sig;
72 const FunctionSig* super_func =
73 super_module->type(supertype_index).function_sig;
74
75 if (sub_func->parameter_count() != super_func->parameter_count() ||
76 sub_func->return_count() != super_func->return_count()) {
77 return false;
78 }
79
80 for (uint32_t i = 0; i < sub_func->parameter_count(); i++) {
81 // Contravariance for params.
82 if (!IsSubtypeOf(super_func->parameters()[i], sub_func->parameters()[i],
83 super_module, sub_module)) {
84 return false;
85 }
86 }
87 for (uint32_t i = 0; i < sub_func->return_count(); i++) {
88 // Covariance for returns.
89 if (!IsSubtypeOf(sub_func->returns()[i], super_func->returns()[i],
90 sub_module, super_module)) {
91 return false;
92 }
93 }
94
95 return true;
96}

170bool ValidSubtypeDefinition(ModuleTypeIndex subtype_index,
171 ModuleTypeIndex supertype_index,
172 const WasmModule* sub_module,
173 const WasmModule* super_module) {
174 const TypeDefinition& subtype = sub_module->type(subtype_index);
175 const TypeDefinition& supertype = super_module->type(supertype_index);
176 if (subtype.kind != supertype.kind) return false;
177 if (supertype.is_final) return false;
178 if (subtype.is_shared != supertype.is_shared) return false;
179 switch (subtype.kind) {
180 case TypeDefinition::kFunction:
181 return ValidFunctionSubtypeDefinition(subtype_index, supertype_index,
182 sub_module, super_module);
183 case TypeDefinition::kStruct:
184 return ValidStructSubtypeDefinition(subtype_index, supertype_index,
185 sub_module, super_module);
186 case TypeDefinition::kArray:
187 return ValidArraySubtypeDefinition(subtype_index, supertype_index,
188 sub_module, super_module);
189 }

253V8_NOINLINE V8_EXPORT_PRIVATE bool IsSubtypeOfImpl(
254 ValueType subtype, ValueType supertype, const WasmModule* sub_module,
255 const WasmModule* super_module) {
256 DCHECK(subtype != supertype || sub_module != super_module);
257
258 // The top type is the super type of all other types.
259 if (supertype.kind() == kTop) return true;
260
261 switch (subtype.kind()) {
262 case kI32:
263 case kI64:
264 case kF16:
265 case kF32:
266 case kF64:
267 case kS128:
268 case kI8:
269 case kI16:
270 case kVoid:
271 case kTop:
272 return subtype == supertype;
273 case kBottom:
274 // The bottom type is a subtype of all types.
275 return true;
276 case kRtt:
277 return supertype.kind() == kRtt &&
278 EquivalentIndices(subtype.ref_index(), supertype.ref_index(),
279 sub_module, super_module);
280 case kRef:
281 case kRefNull:
282 break;
283 }
284
285 DCHECK(subtype.is_object_reference());
286
287 bool compatible_references = subtype.is_nullable()
288 ? supertype.is_nullable()
289 : supertype.is_object_reference();
290 if (!compatible_references) return false;
291
292 DCHECK(supertype.is_object_reference());
293
294 // Now check that sub_heap and super_heap are subtype-related.
295
296 HeapType sub_heap = subtype.heap_type();
297 HeapType super_heap = supertype.heap_type();
298
299 return IsHeapSubtypeOfImpl(sub_heap, super_heap, sub_module, super_module);
300}

302V8_NOINLINE V8_EXPORT_PRIVATE bool IsHeapSubtypeOfImpl(
303 HeapType sub_heap, HeapType super_heap, const WasmModule* sub_module,
304 const WasmModule* super_module) {
305 if (IsShared(sub_heap, sub_module) != IsShared(super_heap, super_module)) {
306 return false;
307 }
308 HeapType::Representation sub_repr_non_shared =
309 sub_heap.representation_non_shared();
310 HeapType::Representation super_repr_non_shared =
311 super_heap.representation_non_shared();
312 switch (sub_repr_non_shared) {
313 case HeapType::kFunc:
314 case HeapType::kAny:
315 case HeapType::kExtern:
316 case HeapType::kExn:
317 case HeapType::kStringViewWtf8:
318 case HeapType::kStringViewWtf16:
319 case HeapType::kStringViewIter:
320 return sub_repr_non_shared == super_repr_non_shared;
321 case HeapType::kEq:
322 case HeapType::kString:
323 return sub_repr_non_shared == super_repr_non_shared ||
324 super_repr_non_shared == HeapType::kAny;
325 case HeapType::kExternString:
326 return super_repr_non_shared == sub_repr_non_shared ||
327 super_repr_non_shared == HeapType::kExtern;
328 case HeapType::kI31:
329 case HeapType::kStruct:
330 case HeapType::kArray:
331 return super_repr_non_shared == sub_repr_non_shared ||
332 super_repr_non_shared == HeapType::kEq ||
333 super_repr_non_shared == HeapType::kAny;
334 case HeapType::kBottom:
335 case HeapType::kTop:
336 UNREACHABLE();
337 case HeapType::kNone:
338 // none is a subtype of every non-func, non-extern and non-exn reference
339 // type under wasm-gc.
340 if (super_heap.is_index()) {
341 return !super_module->has_signature(super_heap.ref_index());
342 }
343 return super_repr_non_shared == HeapType::kAny ||
344 super_repr_non_shared == HeapType::kEq ||
345 super_repr_non_shared == HeapType::kI31 ||
346 super_repr_non_shared == HeapType::kArray ||
347 super_repr_non_shared == HeapType::kStruct ||
348 super_repr_non_shared == HeapType::kString ||
349 super_repr_non_shared == HeapType::kStringViewWtf16 ||
350 super_repr_non_shared == HeapType::kStringViewWtf8 ||
351 super_repr_non_shared == HeapType::kStringViewIter ||
352 super_repr_non_shared == HeapType::kNone;
353 case HeapType::kNoExtern:
354 return super_repr_non_shared == HeapType::kNoExtern ||
355 super_repr_non_shared == HeapType::kExtern ||
356 super_repr_non_shared == HeapType::kExternString;
357 case HeapType::kNoExn:
358 return super_repr_non_shared == HeapType::kExn ||
359 super_repr_non_shared == HeapType::kNoExn;
360 case HeapType::kNoFunc:
361 // nofunc is a subtype of every funcref type under wasm-gc.
362 if (super_heap.is_index()) {
363 return super_module->has_signature(super_heap.ref_index());
364 }
365 return super_repr_non_shared == HeapType::kNoFunc ||
366 super_repr_non_shared == HeapType::kFunc;
367 default:
368 break;
369 }
370
371 DCHECK(sub_heap.is_index());
372 ModuleTypeIndex sub_index = sub_heap.ref_index();
373 DCHECK(sub_module->has_type(sub_index));
374
375 switch (super_repr_non_shared) {
376 case HeapType::kFunc:
377 return sub_module->has_signature(sub_index);
378 case HeapType::kStruct:
379 return sub_module->has_struct(sub_index);
380 case HeapType::kEq:
381 case HeapType::kAny:
382 return !sub_module->has_signature(sub_index);
383 case HeapType::kArray:
384 return sub_module->has_array(sub_index);
385 case HeapType::kI31:
386 case HeapType::kExtern:
387 case HeapType::kExternString:
388 case HeapType::kExn:
389 case HeapType::kString:
390 case HeapType::kStringViewWtf8:
391 case HeapType::kStringViewWtf16:
392 case HeapType::kStringViewIter:
393 case HeapType::kNone:
394 case HeapType::kNoExtern:
395 case HeapType::kNoFunc:
396 case HeapType::kNoExn:
397 return false;
398 case HeapType::kBottom:
399 case HeapType::kTop:
400 UNREACHABLE();
401 default:
402 break;
403 }
404
405 DCHECK(super_heap.is_index());
406 ModuleTypeIndex super_index = super_heap.ref_index();
407 DCHECK(super_module->has_type(super_index));
408 // The {IsSubtypeOf} entry point already has a fast path checking ValueType
409 // equality; here we catch (ref $x) being a subtype of (ref null $x).
410 if (sub_module == super_module && sub_index == super_index) return true;
411 return GetTypeCanonicalizer()->IsCanonicalSubtype(sub_index, super_index,
412 sub_module, super_module);
413}

415V8_NOINLINE bool EquivalentTypes(ValueType type1, ValueType type2,
416 const WasmModule* module1,
417 const WasmModule* module2) {
418 if (type1 == type2 && module1 == module2) return true;
419 if (!type1.has_index() || !type2.has_index()) return type1 == type2;
420 if (type1.kind() != type2.kind()) return false;
421
422 DCHECK(type1 != type2 || module1 != module2);
423 DCHECK(type1.has_index() && module1->has_type(type1.ref_index()) &&
424 type2.has_index() && module2->has_type(type2.ref_index()));
425
426 return EquivalentIndices(type1.ref_index(), type2.ref_index(), module1,
427 module2);
428}

431// Returns the least common ancestor of two type indices, as a type index in
432// {module1}.
433HeapType::Representation CommonAncestor(ModuleTypeIndex type_index1,
434 ModuleTypeIndex type_index2,
435 const WasmModule* module1,
436 const WasmModule* module2) {
437 TypeDefinition type1 = module1->type(type_index1);
438 TypeDefinition type2 = module2->type(type_index2);
439 TypeDefinition::Kind kind1 = type1.kind;
440 TypeDefinition::Kind kind2 = type2.kind;
441 if (type1.is_shared != type2.is_shared) {
442 return HeapType::kTop;
443 }
444 bool both_shared = type1.is_shared;
445 {
446 int depth1 = GetSubtypingDepth(module1, type_index1);
447 int depth2 = GetSubtypingDepth(module2, type_index2);
448 while (depth1 > depth2) {
449 type_index1 = module1->supertype(type_index1);
450 depth1--;
451 }
452 while (depth2 > depth1) {
453 type_index2 = module2->supertype(type_index2);
454 depth2--;
455 }
456 }
457 DCHECK_NE(type_index1, kNoSuperType);
458 DCHECK_NE(type_index2, kNoSuperType);
459 while (type_index1 != kNoSuperType &&
460 !(type_index1 == type_index2 && module1 == module2) &&
461 !EquivalentIndices(type_index1, type_index2, module1, module2)) {
462 type_index1 = module1->supertype(type_index1);
463 type_index2 = module2->supertype(type_index2);
464 }
465 DCHECK_EQ(type_index1 == kNoSuperType, type_index2 == kNoSuperType);
466 if (type_index1 != kNoSuperType) {
467 return static_cast<HeapType::Representation>(type_index1.index);
468 }
469 switch (kind1) {
470 case TypeDefinition::kFunction:
471 switch (kind2) {
472 case TypeDefinition::kFunction:
473 return MaybeShared(HeapType::kFunc, both_shared);
474 case TypeDefinition::kStruct:
475 case TypeDefinition::kArray:
476 return HeapType::kTop;
477 }
478 case TypeDefinition::kStruct:
479 switch (kind2) {
480 case TypeDefinition::kFunction:
481 return HeapType::kTop;
482 case TypeDefinition::kStruct:
483 return MaybeShared(HeapType::kStruct, both_shared);
484 case TypeDefinition::kArray:
485 return MaybeShared(HeapType::kEq, both_shared);
486 }
487 case TypeDefinition::kArray:
488 switch (kind2) {
489 case TypeDefinition::kFunction:
490 return HeapType::kTop;
491 case TypeDefinition::kStruct:
492 return MaybeShared(HeapType::kEq, both_shared);
493 case TypeDefinition::kArray:
494 return MaybeShared(HeapType::kArray, both_shared);
495 }
496 }
497}

499// Returns the least common ancestor of an abstract HeapType {heap1}, and
500// another HeapType {heap2}.
501HeapType::Representation CommonAncestorWithAbstract(HeapType heap1,
502 HeapType heap2,
503 const WasmModule* module2) {
504 DCHECK(heap1.is_abstract());
505 // Passing {module2} with {heap1} below is fine since {heap1} is abstract.
506 bool is_shared = IsShared(heap1, module2);
507 if (is_shared != IsShared(heap2, module2)) {
508 return HeapType::kTop;
509 }
510
511 // TODO(mliedtke): These types should be normalized to the value type kTop and
512 // kBottom and therefore should never appear here. Can we convert these into
513 // assertions?
514 if (heap1.is_top() || heap2.is_top()) return HeapType::kTop;
515 if (heap1.is_bottom()) return heap2.representation();
516 if (heap2.is_bottom()) return heap1.representation();
517
518 HeapType::Representation repr_non_shared2 = heap2.representation_non_shared();
519 switch (heap1.representation_non_shared()) {
520 case HeapType::kFunc: {
521 if (repr_non_shared2 == HeapType::kFunc ||
522 repr_non_shared2 == HeapType::kNoFunc ||
523 (heap2.is_index() && module2->has_signature(heap2.ref_index()))) {
524 return MaybeShared(HeapType::kFunc, is_shared);
525 } else {
526 return HeapType::kTop;
527 }
528 }
529 case HeapType::kAny: {
530 switch (repr_non_shared2) {
531 case HeapType::kI31:
532 case HeapType::kNone:
533 case HeapType::kEq:
534 case HeapType::kStruct:
535 case HeapType::kArray:
536 case HeapType::kAny:
537 case HeapType::kString:
538 return MaybeShared(HeapType::kAny, is_shared);
539 case HeapType::kFunc:
540 case HeapType::kExtern:
541 case HeapType::kExternString:
542 case HeapType::kNoExtern:
543 case HeapType::kNoFunc:
544 case HeapType::kStringViewIter:
545 case HeapType::kStringViewWtf8:
546 case HeapType::kStringViewWtf16:
547 case HeapType::kExn:
548 case HeapType::kNoExn:
549 return HeapType::kTop;
550 default:
551 return module2->has_signature(heap2.ref_index())
552 ? HeapType::kTop
553 : MaybeShared(HeapType::kAny, is_shared);
554 }
555 }
556 case HeapType::kEq: {
557 switch (repr_non_shared2) {
558 case HeapType::kI31:
559 case HeapType::kNone:
560 case HeapType::kEq:
561 case HeapType::kStruct:
562 case HeapType::kArray:
563 return MaybeShared(HeapType::kEq, is_shared);
564 case HeapType::kAny:
565 case HeapType::kString:
566 return MaybeShared(HeapType::kAny, is_shared);
567 case HeapType::kFunc:
568 case HeapType::kExtern:
569 case HeapType::kExternString:
570 case HeapType::kNoExtern:
571 case HeapType::kNoFunc:
572 case HeapType::kStringViewIter:
573 case HeapType::kStringViewWtf8:
574 case HeapType::kStringViewWtf16:
575 case HeapType::kExn:
576 case HeapType::kNoExn:
577 return HeapType::kTop;
578 default:
579 return module2->has_signature(heap2.ref_index())
580 ? HeapType::kTop
581 : MaybeShared(HeapType::kEq, is_shared);
582 }
583 }
584 case HeapType::kI31:
585 switch (repr_non_shared2) {
586 case HeapType::kI31:
587 case HeapType::kNone:
588 return MaybeShared(HeapType::kI31, is_shared);
589 case HeapType::kEq:
590 case HeapType::kStruct:
591 case HeapType::kArray:
592 return MaybeShared(HeapType::kEq, is_shared);
593 case HeapType::kAny:
594 case HeapType::kString:
595 return MaybeShared(HeapType::kAny, is_shared);
596 case HeapType::kFunc:
597 case HeapType::kExtern:
598 case HeapType::kExternString:
599 case HeapType::kNoExtern:
600 case HeapType::kNoFunc:
601 case HeapType::kStringViewIter:
602 case HeapType::kStringViewWtf8:
603 case HeapType::kStringViewWtf16:
604 case HeapType::kExn:
605 case HeapType::kNoExn:
606 return HeapType::kTop;
607 default:
608 return module2->has_signature(heap2.ref_index())
609 ? HeapType::kTop
610 : MaybeShared(HeapType::kEq, is_shared);
611 }
612 case HeapType::kStruct:
613 switch (repr_non_shared2) {
614 case HeapType::kStruct:
615 case HeapType::kNone:
616 return MaybeShared(HeapType::kStruct, is_shared);
617 case HeapType::kArray:
618 case HeapType::kI31:
619 case HeapType::kEq:
620 return MaybeShared(HeapType::kEq, is_shared);
621 case HeapType::kAny:
622 case HeapType::kString:
623 return MaybeShared(HeapType::kAny, is_shared);
624 case HeapType::kFunc:
625 case HeapType::kExtern:
626 case HeapType::kExternString:
627 case HeapType::kNoExtern:
628 case HeapType::kNoFunc:
629 case HeapType::kStringViewIter:
630 case HeapType::kStringViewWtf8:
631 case HeapType::kStringViewWtf16:
632 case HeapType::kExn:
633 case HeapType::kNoExn:
634 return HeapType::kTop;
635 default:
636 return module2->has_struct(heap2.ref_index())
637 ? MaybeShared(HeapType::kStruct, is_shared)
638 : module2->has_array(heap2.ref_index())
639 ? MaybeShared(HeapType::kEq, is_shared)
640 : HeapType::kTop;
641 }
642 case HeapType::kArray:
643 switch (repr_non_shared2) {
644 case HeapType::kArray:
645 case HeapType::kNone:
646 return MaybeShared(HeapType::kArray, is_shared);
647 case HeapType::kStruct:
648 case HeapType::kI31:
649 case HeapType::kEq:
650 return MaybeShared(HeapType::kEq, is_shared);
651 case HeapType::kAny:
652 case HeapType::kString:
653 return MaybeShared(HeapType::kAny, is_shared);
654 case HeapType::kFunc:
655 case HeapType::kExtern:
656 case HeapType::kExternString:
657 case HeapType::kNoExtern:
658 case HeapType::kNoFunc:
659 case HeapType::kStringViewIter:
660 case HeapType::kStringViewWtf8:
661 case HeapType::kStringViewWtf16:
662 case HeapType::kExn:
663 case HeapType::kNoExn:
664 return HeapType::kTop;
665 default:
666 return module2->has_array(heap2.ref_index())
667 ? MaybeShared(HeapType::kArray, is_shared)
668 : module2->has_struct(heap2.ref_index())
669 ? MaybeShared(HeapType::kEq, is_shared)
670 : HeapType::kTop;
671 }
672 case HeapType::kNone:
673 switch (repr_non_shared2) {
674 case HeapType::kArray:
675 case HeapType::kNone:
676 case HeapType::kStruct:
677 case HeapType::kI31:
678 case HeapType::kEq:
679 case HeapType::kAny:
680 case HeapType::kString:
681 return heap2.representation();
682 case HeapType::kExtern:
683 case HeapType::kExternString:
684 case HeapType::kNoExtern:
685 case HeapType::kNoFunc:
686 case HeapType::kFunc:
687 case HeapType::kStringViewIter:
688 case HeapType::kStringViewWtf8:
689 case HeapType::kStringViewWtf16:
690 case HeapType::kExn:
691 case HeapType::kNoExn:
692 return HeapType::kTop;
693 default:
694 return module2->has_signature(heap2.ref_index())
695 ? HeapType::kTop
696 : heap2.representation();
697 }
698 case HeapType::kNoFunc:
699 return (repr_non_shared2 == HeapType::kNoFunc ||
700 repr_non_shared2 == HeapType::kFunc ||
701 (heap2.is_index() && module2->has_signature(heap2.ref_index())))
702 ? heap2.representation()
703 : HeapType::kTop;
704 case HeapType::kNoExtern:
705 return repr_non_shared2 == HeapType::kExtern ||
706 repr_non_shared2 == HeapType::kNoExtern ||
707 repr_non_shared2 == HeapType::kExternString
708 ? heap2.representation()
709 : HeapType::kTop;
710 case HeapType::kExtern:
711 return repr_non_shared2 == HeapType::kExtern ||
712 repr_non_shared2 == HeapType::kNoExtern ||
713 repr_non_shared2 == HeapType::kExternString
714 ? MaybeShared(HeapType::kExtern, is_shared)
715 : HeapType::kTop;
716 case HeapType::kExternString:
717 return repr_non_shared2 == HeapType::kExtern
718 ? MaybeShared(HeapType::kExtern, is_shared)
719 : (repr_non_shared2 == HeapType::kNoExtern ||
720 repr_non_shared2 == HeapType::kExternString)
721 ? MaybeShared(HeapType::kExternString, is_shared)
722 : HeapType::kTop;
723 case HeapType::kNoExn:
724 return repr_non_shared2 == HeapType::kExn ||
725 repr_non_shared2 == HeapType::kNoExn
726 ? heap2.representation()
727 : HeapType::kTop;
728 case HeapType::kExn:
729 return repr_non_shared2 == HeapType::kExn ||
730 repr_non_shared2 == HeapType::kNoExn
731 ? MaybeShared(HeapType::kExn, is_shared)
732 : HeapType::kTop;
733 case HeapType::kString: {
734 switch (repr_non_shared2) {
735 case HeapType::kI31:
736 case HeapType::kEq:
737 case HeapType::kStruct:
738 case HeapType::kArray:
739 case HeapType::kAny:
740 return MaybeShared(HeapType::kAny, is_shared);
741 case HeapType::kNone:
742 case HeapType::kString:
743 return MaybeShared(HeapType::kString, is_shared);
744 case HeapType::kFunc:
745 case HeapType::kExtern:
746 case HeapType::kExternString:
747 case HeapType::kNoExtern:
748 case HeapType::kNoFunc:
749 case HeapType::kStringViewIter:
750 case HeapType::kStringViewWtf8:
751 case HeapType::kStringViewWtf16:
752 case HeapType::kExn:
753 case HeapType::kNoExn:
754 return HeapType::kTop;
755 default:
756 return module2->has_signature(heap2.ref_index())
757 ? HeapType::kTop
758 : MaybeShared(HeapType::kAny, is_shared);
759 }
760 }
761 case HeapType::kStringViewIter:
762 case HeapType::kStringViewWtf16:
763 case HeapType::kStringViewWtf8:
764 return heap1 == heap2 ? heap1.representation() : HeapType::kTop;
765 default:
766 UNREACHABLE();
767 }

771V8_EXPORT_PRIVATE TypeInModule Union(ValueType type1, ValueType type2,
772 const WasmModule* module1,
773 const WasmModule* module2) {
774 if (type1 == kWasmTop || type2 == kWasmTop) return {kWasmTop, module1};
775 if (type1 == kWasmBottom) return {type2, module2};
776 if (type2 == kWasmBottom) return {type1, module1};
777 if (!type1.is_object_reference() || !type2.is_object_reference()) {
778 return {EquivalentTypes(type1, type2, module1, module2) ? type1 : kWasmTop,
779 module1};
780 }
781 Nullability nullability =
782 type1.is_nullable() || type2.is_nullable() ? kNullable : kNonNullable;
783 HeapType heap1 = type1.heap_type();
784 HeapType heap2 = type2.heap_type();
785 if (heap1 == heap2 && module1 == module2) {
786 return {ValueType::RefMaybeNull(heap1, nullability), module1};
787 }
788 HeapType::Representation result_repr;
789 const WasmModule* result_module;
790 if (heap1.is_abstract()) {
791 result_repr = CommonAncestorWithAbstract(heap1, heap2, module2);
792 result_module = module2;
793 } else if (heap2.is_abstract()) {
794 result_repr = CommonAncestorWithAbstract(heap2, heap1, module1);
795 result_module = module1;
796 } else {
797 result_repr =
798 CommonAncestor(heap1.ref_index(), heap2.ref_index(), module1, module2);
799 result_module = module1;
800 }
801 // The type could only be kBottom if the input was kBottom but any kBottom
802 // HeapType should be "normalized" to kWasmBottom ValueType.
803 DCHECK_NE(result_repr, HeapType::kBottom);
804 return {result_repr == HeapType::kTop
805 ? kWasmTop
806 : ValueType::RefMaybeNull(result_repr, nullability),
807 result_module};
808}
809
810TypeInModule Intersection(ValueType type1, ValueType type2,
811 const WasmModule* module1,
812 const WasmModule* module2) {
813 if (type1 == kWasmTop) return {type2, module2};
814 if (type2 == kWasmTop) return {type1, module1};
815 if (!type1.is_object_reference() || !type2.is_object_reference()) {
816 return {
817 EquivalentTypes(type1, type2, module1, module2) ? type1 : kWasmBottom,
818 module1};
819 }
820 Nullability nullability =
821 type1.is_nullable() && type2.is_nullable() ? kNullable : kNonNullable;
822 // non-nullable null type is not a valid type.
823 if (nullability == kNonNullable && (IsNullSentinel(type1.heap_type()) ||
824 IsNullSentinel(type2.heap_type()))) {
825 return {kWasmBottom, module1};
826 }
827 if (IsHeapSubtypeOf(type1.heap_type(), type2.heap_type(), module1, module2)) {
828 return TypeInModule{ValueType::RefMaybeNull(type1.heap_type(), nullability),
829 module1};
830 }
831 if (IsHeapSubtypeOf(type2.heap_type(), type1.heap_type(), module2, module1)) {
832 return TypeInModule{ValueType::RefMaybeNull(type2.heap_type(), nullability),
833 module2};
834 }
835 if (nullability == kNonNullable) {
836 return {kWasmBottom, module1};
837 }
838 // Check for common null representation.
839 ValueType null_type1 = ToNullSentinel({type1, module1});
840 if (null_type1 == ToNullSentinel({type2, module2})) {
841 return {null_type1, module1};
842 }
843 return {kWasmBottom, module1};
844}

846ValueType ToNullSentinel(TypeInModule type) {
847 HeapType::Representation null_heap =
848 NullSentinelImpl(type.type.heap_type(), type.module);
849 DCHECK(
850 IsHeapSubtypeOf(HeapType(null_heap), type.type.heap_type(), type.module));
851 return ValueType::RefNull(null_heap);
852}

854bool IsSameTypeHierarchy(HeapType type1, HeapType type2,
855 const WasmModule* module) {
856 return NullSentinelImpl(type1, module) == NullSentinelImpl(type2, module);
857}
