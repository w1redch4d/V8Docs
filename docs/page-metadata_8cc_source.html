54PageMetadata* PageMetadata::ConvertNewToOld(PageMetadata* old_page) {
55 DCHECK(old_page);
56 MemoryChunk* chunk = old_page->Chunk();
57 DCHECK(chunk->InNewSpace());
58 old_page->ResetAgeInNewSpace();
59 OldSpace* old_space = old_page->heap()->old_space();
60 old_page->set_owner(old_space);
61 chunk->ClearFlagsNonExecutable(MemoryChunk::kAllFlagsMask);
62 DCHECK_NE(old_space->identity(), SHARED_SPACE);
63 chunk->SetOldGenerationPageFlags(
64 old_page->heap()->incremental_marking()->marking_mode(), OLD_SPACE);
65 PageMetadata* new_page = old_space->InitializePage(old_page);
66 old_space->AddPromotedPage(new_page);
67 return new_page;
68}

88// Skips filler starting from the given filler until the end address.
89// Returns the first address after the skipped fillers.
90Address SkipFillers(PtrComprCageBase cage_base, Tagged<HeapObject> filler,
91 Address end) {
92 Address addr = filler.address();

94 filler = HeapObject::FromAddress(addr);
95 CHECK(IsFreeSpaceOrFiller(filler, cage_base));

103size_t PageMetadata::ShrinkToHighWaterMark() {
104 // Shrinking only makes sense outside of the CodeRange, where we don't care
105 // about address space fragmentation.
106 VirtualMemory* reservation = reserved_memory();
107 if (!reservation->IsReserved()) return 0;
108
109 // Shrink pages to high water mark. The water mark points either to a filler
110 // or the area_end.
111 Tagged<HeapObject> filler = HeapObject::FromAddress(HighWaterMark());
112 if (filler.address() == area_end()) return 0;
113 PtrComprCageBase cage_base(heap()->isolate());
114 CHECK(IsFreeSpaceOrFiller(filler, cage_base));
115 // Ensure that no objects were allocated in [filler, area_end) region.
116 DCHECK_EQ(area_end(), SkipFillers(cage_base, filler, area_end()));
117 // Ensure that no objects will be allocated on this page.
118 DCHECK_EQ(0u, AvailableInFreeList());
119
120 // Ensure that slot sets are empty. Otherwise the buckets for the shrunk
121 // area would not be freed when deallocating this page.
122 DCHECK_NULL(slot_set<OLD_TO_NEW>());
123 DCHECK_NULL(slot_set<OLD_TO_NEW_BACKGROUND>());
124 DCHECK_NULL(slot_set<OLD_TO_OLD>());
125
126 size_t unused = RoundDown(static_cast<size_t>(area_end() - filler.address()),
127 MemoryAllocator::GetCommitPageSize());
128 if (unused > 0) {
129 DCHECK_EQ(0u, unused % MemoryAllocator::GetCommitPageSize());
130 if (v8_flags.trace_gc_verbose) {
131 PrintIsolate(heap()->isolate(), "Shrinking page %p: end %p -> %p\n",
132 reinterpret_cast<void*>(this),
133 reinterpret_cast<void*>(area_end()),
134 reinterpret_cast<void*>(area_end() - unused));
135 }
136 heap()->CreateFillerObjectAt(
137 filler.address(),
138 static_cast<int>(area_end() - filler.address() - unused));
139 heap()->memory_allocator()->PartialFreeMemory(
140 this, ChunkAddress() + size() - unused, unused, area_end() - unused);
141 if (filler.address() != area_end()) {
142 CHECK(IsFreeSpaceOrFiller(filler, cage_base));
143 CHECK_EQ(filler.address() + filler->Size(cage_base), area_end());
144 }
145 }
146 return unused;
147}

149void PageMetadata::CreateBlackArea(Address start, Address end) {
150 DCHECK(!v8_flags.black_allocated_pages);
151 DCHECK_NE(NEW_SPACE, owner_identity());
152 DCHECK(v8_flags.sticky_mark_bits ||
153 heap()->incremental_marking()->black_allocation());
154 DCHECK_EQ(PageMetadata::FromAddress(start), this);
155 DCHECK_LT(start, end);
156 DCHECK_EQ(PageMetadata::FromAddress(end - 1), this);
157 marking_bitmap()->SetRange<AccessMode::ATOMIC>(
158 MarkingBitmap::AddressToIndex(start),
159 MarkingBitmap::LimitAddressToIndex(end));
160 IncrementLiveBytesAtomically(static_cast<intptr_t>(end - start));
161 owner()->NotifyBlackAreaCreated(end - start);
162}
