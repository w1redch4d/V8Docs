136bool JSObject::PrototypeHasNoElements(Isolate* isolate,
137 Tagged<JSObject> object) {
138 DisallowGarbageCollection no_gc;
139 Tagged<HeapObject> prototype = Cast<HeapObject>(object->map()->prototype());
140 ReadOnlyRoots roots(isolate);
141 Tagged<HeapObject> null = roots.null_value();
142 Tagged<FixedArrayBase> empty_fixed_array = roots.empty_fixed_array();
143 Tagged<FixedArrayBase> empty_slow_element_dictionary =
144 roots.empty_slow_element_dictionary();
145 while (prototype != null) {
146 Tagged<Map> map = prototype->map();
147 if (IsCustomElementsReceiverMap(map)) return false;
148 Tagged<FixedArrayBase> elements = Cast<JSObject>(prototype)->elements();
149 if (elements != empty_fixed_array &&
150 elements != empty_slow_element_dictionary) {
151 return false;
152 }
153 prototype = Cast<HeapObject>(map->prototype());
154 }
155 return true;
156}

176template <typename TSlot>
177void JSObject::EnsureCanContainElements(DirectHandle<JSObject> object,
178 TSlot objects, uint32_t count,
179 EnsureElementsMode mode) {
180 static_assert(std::is_same<TSlot, FullObjectSlot>::value ||
181 std::is_same<TSlot, ObjectSlot>::value,
182 "Only ObjectSlot and FullObjectSlot are expected here");
183 ElementsKind current_kind = object->GetElementsKind();
184 ElementsKind target_kind = current_kind;
185 {
186 DisallowGarbageCollection no_gc;
187 DCHECK(mode != ALLOW_COPIED_DOUBLE_ELEMENTS);
188 bool is_holey = IsHoleyElementsKind(current_kind);
189 if (current_kind == HOLEY_ELEMENTS) return;
190 Tagged<Object> the_hole = GetReadOnlyRoots().the_hole_value();
191 for (uint32_t i = 0; i < count; ++i, ++objects) {
192 Tagged<Object> current = *objects;
193 if (current == the_hole) {
194 is_holey = true;
195 target_kind = GetHoleyElementsKind(target_kind);
196 } else if (!IsSmi(current)) {
197 if (mode == ALLOW_CONVERTED_DOUBLE_ELEMENTS && IsNumber(current)) {
198 if (IsSmiElementsKind(target_kind)) {
199 if (is_holey) {
200 target_kind = HOLEY_DOUBLE_ELEMENTS;
201 } else {
202 target_kind = PACKED_DOUBLE_ELEMENTS;
203 }
204 }
205 } else if (is_holey) {
206 target_kind = HOLEY_ELEMENTS;
207 break;
208 } else {
209 target_kind = PACKED_ELEMENTS;
210 }
211 }
212 }
213 }
214 if (target_kind != current_kind) {
215 TransitionElementsKind(object, target_kind);
216 }
217}

219void JSObject::EnsureCanContainElements(DirectHandle<JSObject> object,
220 DirectHandle<FixedArrayBase> elements,
221 uint32_t length,
222 EnsureElementsMode mode) {
223 ReadOnlyRoots roots = GetReadOnlyRoots();
224 if (elements->map() != roots.fixed_double_array_map()) {
225 DCHECK(elements->map() == roots.fixed_array_map() ||
226 elements->map() == roots.fixed_cow_array_map());
227 if (mode == ALLOW_COPIED_DOUBLE_ELEMENTS) {
228 mode = DONT_ALLOW_DOUBLE_ELEMENTS;
229 }
230 ObjectSlot objects = Cast<FixedArray>(elements)->RawFieldOfFirstElement();
231 EnsureCanContainElements(object, objects, length, mode);
232 return;
233 }
234
235 DCHECK(mode == ALLOW_COPIED_DOUBLE_ELEMENTS);
236 if (object->GetElementsKind() == HOLEY_SMI_ELEMENTS) {
237 TransitionElementsKind(object, HOLEY_DOUBLE_ELEMENTS);
238 } else if (object->GetElementsKind() == PACKED_SMI_ELEMENTS) {
239 auto double_array = Cast<FixedDoubleArray>(elements);
240 for (uint32_t i = 0; i < length; ++i) {
241 if (double_array->is_the_hole(i)) {
242 TransitionElementsKind(object, HOLEY_DOUBLE_ELEMENTS);
243 return;
244 }
245 }
246 TransitionElementsKind(object, PACKED_DOUBLE_ELEMENTS);
247 }

316int JSObject::GetEmbedderFieldCount(Tagged<Map> map) {
317 int instance_size = map->instance_size();
318 if (instance_size == kVariableSizeSentinel) return 0;
319 // Embedder fields are located after the object header, whereas in-object
320 // properties are located at the end of the object. We don't have to round up
321 // the header size here because division by kEmbedderDataSlotSizeInTaggedSlots
322 // will swallow potential padding in case of (kTaggedSize !=
323 // kSystemPointerSize) anyway.
324 return (((instance_size - GetEmbedderFieldsStartOffset(map)) >>
325 kTaggedSizeLog2) -
326 map->GetInObjectProperties()) /
327 kEmbedderDataSlotSizeInTaggedSlots;
328}

400std::optional<Tagged<Object>> JSObject::RawInobjectPropertyAt(
401 PtrComprCageBase cage_base, Tagged<Map> original_map,
402 FieldIndex index) const {
403 CHECK(index.is_inobject());
404
405 // This method implements a "snapshot" protocol to protect against reading out
406 // of bounds of an object. It's used to access a fast in-object property from
407 // a background thread with no locking. That caller does have the guarantee
408 // that a garbage collection cannot happen during its query. However, it must
409 // contend with the main thread altering the object in heavy ways through
410 // object migration. Specifically, the object can get smaller. Initially, this
411 // may seem benign, because object migration fills the freed-up space with
412 // FillerMap words which, even though they offer wrong values, are at
413 // least tagged values.
414
415 // However, there is an additional danger. Sweeper threads may discover the
416 // filler words and offer that space to the main thread for allocation. Should
417 // a HeapNumber be allocated into that space while we're reading a property at
418 // that location (from our out-of-date information), we risk interpreting a
419 // double value as a pointer. This must be prevented.
420 //
421 // We do this by:
422 //
423 // a) Reading the map first
424 // b) Reading the property with acquire semantics (but do not inspect it!)
425 // c) Re-read the map with acquire semantics.
426 //
427 // Only if the maps match can the property be inspected. It may have a "wrong"
428 // value, but it will be within the bounds of the objects instance size as
429 // given by the map and it will be a valid Smi or object pointer.
430 Tagged<Object> maybe_tagged_object =
431 TaggedField<Object>::Acquire_Load(cage_base, *this, index.offset());
432 if (original_map != map(cage_base, kAcquireLoad)) return {};
433 return maybe_tagged_object;
434}

473void JSObject::WriteToField(InternalIndex descriptor, PropertyDetails details,
474 Tagged<Object> value) {
475 DCHECK_EQ(PropertyLocation::kField, details.location());
476 DCHECK_EQ(PropertyKind::kData, details.kind());
477 DisallowGarbageCollection no_gc;
478 FieldIndex index = FieldIndex::ForDetails(map(), details);
479 if (details.representation().IsDouble()) {
480 // Manipulating the signaling NaN used for the hole and uninitialized
481 // double field sentinel in C++, e.g. with base::bit_cast or
482 // value()/set_value(), will change its value on ia32 (the x87 stack is used
483 // to return values and stores to the stack silently clear the signalling
484 // bit).
485 uint64_t bits;
486 if (IsSmi(value)) {
487 bits = base::bit_cast<uint64_t>(static_cast<double>(Smi::ToInt(value)));
488 } else if (IsUninitialized(value)) {
489 bits = kHoleNanInt64;
490 } else {
491 DCHECK(IsHeapNumber(value));
492 bits = Cast<HeapNumber>(value)->value_as_bits();
493 }
494 auto box = Cast<HeapNumber>(RawFastPropertyAt(index));
495 box->set_value_as_bits(bits);
496 } else {
497 FastPropertyAtPut(index, value);
498 }
499}

563void JSObject::InitializeBody(Tagged<Map> map, int start_offset,
564 bool is_slack_tracking_in_progress,
565 MapWord filler_map,
566 Tagged<Object> undefined_filler) {
567 int size = map->instance_size();
568 int offset = start_offset;
569
570 // embedder data slots need to be initialized separately
571 if (MayHaveEmbedderFields(map)) {
572 int embedder_field_start = GetEmbedderFieldsStartOffset(map);
573 int embedder_field_count = GetEmbedderFieldCount(map);
574
575 // fill start with references to the undefined value object
576 DCHECK_LE(offset, embedder_field_start);
577 while (offset < embedder_field_start) {
578 WRITE_FIELD(*this, offset, undefined_filler);
579 offset += kTaggedSize;
580 }
581
582 // initialize embedder data slots
583 DCHECK_EQ(offset, embedder_field_start);
584 for (int i = 0; i < embedder_field_count; i++) {
585 // TODO(v8): consider initializing embedded data slots with Smi::zero().
586 EmbedderDataSlot(Tagged<JSObject>(*this), i).Initialize(undefined_filler);
587 offset += kEmbedderDataSlotSize;
588 }
589 } else {
590 DCHECK_EQ(0, GetEmbedderFieldCount(map));
591 }
592
593 DCHECK_LE(offset, size);
594 if (is_slack_tracking_in_progress) {
595 int end_of_pre_allocated_offset =
596 size - (map->UnusedPropertyFields() * kTaggedSize);
597 DCHECK_LE(kHeaderSize, end_of_pre_allocated_offset);
598 DCHECK_LE(offset, end_of_pre_allocated_offset);
599 // fill pre allocated slots with references to the undefined value object
600 while (offset < end_of_pre_allocated_offset) {
601 WRITE_FIELD(*this, offset, undefined_filler);
602 offset += kTaggedSize;
603 }
604 // fill the remainder with one word filler objects (ie just a map word)
605 while (offset < size) {
606 Tagged<Object> fm = Tagged<Object>(filler_map.ptr());
607 WRITE_FIELD(*this, offset, fm);

1031// If the fast-case backing storage takes up much more memory than a dictionary
1032// backing storage would, the object should have slow elements.
1033// static
1034static inline bool ShouldConvertToSlowElements(uint32_t used_elements,
1035 uint32_t new_capacity) {
1036 uint32_t size_threshold = NumberDictionary::kPreferFastElementsSizeFactor *
1037 NumberDictionary::ComputeCapacity(used_elements) *
1038 NumberDictionary::kEntrySize;
1039 return size_threshold <= new_capacity;
1040}
1041
1042static inline bool ShouldConvertToSlowElements(Tagged<JSObject> object,
1043 uint32_t capacity,
1044 uint32_t index,
1045 uint32_t* new_capacity) {
1046 static_assert(JSObject::kMaxUncheckedOldFastElementsLength <=
1047 JSObject::kMaxUncheckedFastElementsLength);
1048 if (index < capacity) {
1049 *new_capacity = capacity;
1050 return false;
1051 }
1052 if (index - capacity >= JSObject::kMaxGap) return true;
1053 *new_capacity = JSObject::NewElementsCapacity(index + 1);
1054 DCHECK_LT(index, *new_capacity);
1055 if (*new_capacity <= JSObject::kMaxUncheckedOldFastElementsLength ||
1056 (*new_capacity <= JSObject::kMaxUncheckedFastElementsLength &&
1057 HeapLayout::InYoungGeneration(object))) {
1058 return false;
1059 }
1060 return ShouldConvertToSlowElements(object->GetFastElementsUsage(),
1061 *new_capacity);
1062}
