119template <typename ConcreteVisitor>
120size_t HeapVisitor<ConcreteVisitor>::Visit(Tagged<Map> map,
121 Tagged<HeapObject> object,
122 MaybeObjectSize maybe_object_size) {
123 if constexpr (ConcreteVisitor::UsePrecomputedObjectSize()) {
124 DCHECK_EQ(maybe_object_size.AssumeSize(), object->SizeFromMap(map));
125 static_assert(!ConcreteVisitor::EnableConcurrentVisitation());
126 } else {
127 DCHECK(maybe_object_size.IsNone());
128 }
129
130 ConcreteVisitor* visitor = static_cast<ConcreteVisitor*>(this);
131 switch (map->visitor_id()) {
132#define CASE(TypeName) \
133 case kVisit##TypeName: \
134 /* If this DCHECK fails, it means that the object type wasn't added \
135 * to the TRUSTED_VISITOR_ID_LIST. \
136 * Note: This would normally be just !IsTrustedObject(obj), however we \
137 * might see trusted objects here before they've been migrated to trusted \
138 * space, hence the second condition. */ \
139 DCHECK(!InstanceTypeChecker::IsTrustedObject(map) || \
140 !HeapLayout::InTrustedSpace(object)); \
141 return visitor->Visit##TypeName( \
142 map, ConcreteVisitor::template Cast<TypeName>(object, heap_), \
143 maybe_object_size);
144 TYPED_VISITOR_ID_LIST(CASE)
145 TYPED_VISITOR_WITH_SLACK_ID_LIST(CASE)
146 TORQUE_VISITOR_ID_LIST(CASE)
147#undef CASE
148#define CASE(TypeName) \
149 case kVisit##TypeName: \
150 DCHECK(InstanceTypeChecker::IsTrustedObject(map)); \
151 /* Trusted objects are protected from modifications by an attacker as \
152 * they are located outside of the sandbox. However, an attacker can \
153 * still craft their own fake trusted objects inside the sandbox. In \
154 * this case, bad things might happen if these objects are then \
155 * processed by e.g. an object visitor as they will typically assume \
156 * that these objects are trustworthy. The following check defends \
157 * against that by ensuring that the object is outside of the sandbox. \
158 * See also crbug.com/c/1505089. */ \
159 SBXCHECK(OutsideSandboxOrInReadonlySpace(object)); \
160 return visitor->Visit##TypeName( \
161 map, ConcreteVisitor::template Cast<TypeName>(object, heap_), \
162 maybe_object_size);
163 TRUSTED_VISITOR_ID_LIST(CASE)
164#undef CASE
165 case kVisitShortcutCandidate:
166 return visitor->VisitShortcutCandidate(
167 map, ConcreteVisitor::template Cast<ConsString>(object, heap_),
168 maybe_object_size);
169 case kVisitJSObjectFast:
170 return visitor->VisitJSObjectFast(
171 map, ConcreteVisitor::template Cast<JSObject>(object, heap_),
172 maybe_object_size);
173 case kVisitJSApiObject:
174 return visitor->VisitJSApiObject(
175 map, ConcreteVisitor::template Cast<JSObject>(object, heap_),
176 maybe_object_size);
177 case kVisitStruct:
178 return visitor->VisitStruct(map, object, maybe_object_size);
179 case kVisitFiller:
180 return visitor->VisitFiller(map, object, maybe_object_size);
181 case kVisitFreeSpace:
182 return visitor->VisitFreeSpace(map, i::Cast<FreeSpace>(object),
183 maybe_object_size);
184 case kDataOnlyVisitorIdCount:
185 case kVisitorIdCount:
186 UNREACHABLE();
187 }
188 // TODO(chromium:327992715): Remove once we have some clarity why execution
189 // can reach this point.
190 {
191 Isolate* isolate;
192 if (GetIsolateFromHeapObject(object, &isolate)) {
193 isolate->PushParamsAndDie(
194 reinterpret_cast<void*>(object.ptr()),
195 reinterpret_cast<void*>(map.ptr()),
196 reinterpret_cast<void*>(static_cast<intptr_t>(map->visitor_id())));
197 }
198 }
199 UNREACHABLE();
200}

313template <typename ConcreteVisitor>
314template <typename T, typename TBodyDescriptor>
315size_t HeapVisitor<ConcreteVisitor>::VisitJSObjectSubclass(
316 Tagged<Map> map, Tagged<T> object, MaybeObjectSize maybe_object_size) {
317 // JSObject types are subject to slack tracking. At the end of slack tracking
318 // a Map's instance size is adjusted properly. Since this changes the instance
319 // size, we cannot DCHECK that `SizeFromMap()` is consistent with
320 // `TBodyDescriptor::SizeOf()` as that would require taking a snapshot of the
321 // Map.
322
323 ConcreteVisitor* visitor = static_cast<ConcreteVisitor*>(this);
324 visitor->template VisitMapPointerIfNeeded<VisitorId::kVisitJSObject>(object);
325
326 const size_t size = ConcreteVisitor::UsePrecomputedObjectSize()
327 ? maybe_object_size.AssumeSize()
328 : TBodyDescriptor::SizeOf(map, object);
329
330 int visitation_size = static_cast<int>(size);
331
332 if (!ConcreteVisitor::ShouldVisitFullJSObject()) {
333 // It is important to visit only the used field and ignore the slack fields
334 // because the slack fields may be trimmed concurrently and we don't want to
335 // find fillers (slack) during pointer visitation.
336 const int used_size = map->UsedInstanceSize();
337 DCHECK_LE(used_size, size);
338 DCHECK_GE(used_size, JSObject::GetHeaderSize(map));
339
340 visitation_size = used_size;
341 }
342
343 TBodyDescriptor::IterateBody(map, object, visitation_size, visitor);
344
345 return size;
346}

348template <typename ConcreteVisitor>
349template <VisitorId visitor_id, typename T, typename TBodyDescriptor>
350size_t HeapVisitor<ConcreteVisitor>::VisitWithBodyDescriptor(
351 Tagged<Map> map, Tagged<T> object, MaybeObjectSize maybe_object_size) {
352 // If you see the following DCHECK fail, then the size computation of
353 // BodyDescriptor doesn't match the size return via obj.Size(). This is
354 // problematic as the GC requires those sizes to match for accounting reasons.
355 // The fix likely involves adding a padding field in the object definitions.
356 //
357 // We can only perform this check for types that do not support right trimming
358 // when running concurrently. `RefineAllocatedBytesAfterSweeping()` ensures
359 // that we only see sizes that get smaller during marking.
360#ifdef DEBUG
361 if (!SupportsRightTrim<visitor_id>() ||
362 !ConcreteVisitor::EnableConcurrentVisitation()) {
363 DCHECK_EQ(object->SizeFromMap(map), TBodyDescriptor::SizeOf(map, object));
364 }
365#endif // DEBUG
366 DCHECK(!map->IsInobjectSlackTrackingInProgress());
367
368 ConcreteVisitor* visitor = static_cast<ConcreteVisitor*>(this);
369 visitor->template VisitMapPointerIfNeeded<visitor_id>(object);
370 const int size = ConcreteVisitor::UsePrecomputedObjectSize()
371 ? static_cast<int>(maybe_object_size.AssumeSize())
372 : TBodyDescriptor::SizeOf(map, object);
373 TBodyDescriptor::IterateBody(map, object, size, visitor);
374 return size;
375}

439template <typename ConcreteVisitor>
440template <typename T>
441size_t ConcurrentHeapVisitor<ConcreteVisitor>::VisitStringLocked(
442 Tagged<T> object) {
443 ConcreteVisitor* visitor = static_cast<ConcreteVisitor*>(this);
444 SharedObjectLockGuard guard(object);
445 // The object has been locked. At this point shared read access is
446 // guaranteed but we must re-read the map and check whether the string has
447 // transitioned.
448 Tagged<Map> map = object->map();
449 int size;
450 switch (map->visitor_id()) {
451#define UNSAFE_STRING_TRANSITION_TARGET_CASE(VisitorIdType, TypeName) \
452 case kVisit##VisitorIdType: \
453 visitor \
454 ->template VisitMapPointerIfNeeded<VisitorId::kVisit##VisitorIdType>( \
455 object); \
456 size = ObjectTraits<TypeName>::BodyDescriptor::SizeOf(map, object); \
457 ObjectTraits<TypeName>::BodyDescriptor::IterateBody( \
458 map, UncheckedCast<TypeName>(object), size, visitor); \
459 break;
460
461 UNSAFE_STRING_TRANSITION_TARGETS(UNSAFE_STRING_TRANSITION_TARGET_CASE)
462#undef UNSAFE_STRING_TRANSITION_TARGET_CASE
463 default:
464 UNREACHABLE();
465 }
466 return size;
467}
