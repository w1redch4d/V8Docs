31// Function names are generally handled separately from other names; in
32// particular we support decoding function names without decoding any other
33// names, in which case also computing fallback names from imports and exports
34// must happen separately.
35void NamesProvider::ComputeFunctionNamesFromImportsExports() {
36 DCHECK(!has_computed_function_import_names_);
37 has_computed_function_import_names_ = true;
38 // When tracing streaming compilations, we might not yet have wire bytes.
39 if (wire_bytes_.empty()) return;
40 for (const WasmImport& import : module_->import_table) {
41 if (import.kind != kExternalFunction) continue;
42 if (module_->lazily_generated_names.Has(import.index)) continue;
43 ComputeImportName(import, import_export_function_names_);
44 }
45 for (const WasmExport& ex : module_->export_table) {
46 if (ex.kind != kExternalFunction) continue;
47 if (module_->lazily_generated_names.Has(ex.index)) continue;
48 ComputeExportName(ex, import_export_function_names_);
49 }
50}

52void NamesProvider::ComputeNamesFromImportsExports() {
53 DCHECK(!has_computed_import_names_);
54 has_computed_import_names_ = true;
55 // When tracing streaming compilations, we might not yet have wire bytes.
56 if (wire_bytes_.empty()) return;
57 DCHECK(has_decoded_);
58 for (const WasmImport import : module_->import_table) {
59 switch (import.kind) {
60 case kExternalFunction:
61 continue; // Functions are handled separately.
62 case kExternalTable:
63 if (name_section_names_->table_names_.Has(import.index)) continue;
64 ComputeImportName(import, import_export_table_names_);
65 break;
66 case kExternalMemory:
67 if (name_section_names_->memory_names_.Has(import.index)) continue;
68 ComputeImportName(import, import_export_memory_names_);
69 break;
70 case kExternalGlobal:
71 if (name_section_names_->global_names_.Has(import.index)) continue;
72 ComputeImportName(import, import_export_global_names_);
73 break;
74 case kExternalTag:
75 if (name_section_names_->tag_names_.Has(import.index)) continue;
76 ComputeImportName(import, import_export_tag_names_);
77 break;
78 }
79 }
80 for (const WasmExport& ex : module_->export_table) {
81 switch (ex.kind) {
82 case kExternalFunction:
83 continue; // Functions are handled separately.
84 case kExternalTable:
85 if (name_section_names_->table_names_.Has(ex.index)) continue;
86 ComputeExportName(ex, import_export_table_names_);
87 break;
88 case kExternalMemory:
89 if (name_section_names_->memory_names_.Has(ex.index)) continue;
90 ComputeExportName(ex, import_export_memory_names_);
91 break;
92 case kExternalGlobal:
93 if (name_section_names_->global_names_.Has(ex.index)) continue;
94 ComputeExportName(ex, import_export_global_names_);
95 break;
96 case kExternalTag:
97 if (name_section_names_->tag_names_.Has(ex.index)) continue;
98 ComputeExportName(ex, import_export_tag_names_);
99 break;
100 }
101 }

105// Any disallowed characters get replaced with '_'. Reference:
106// https://webassembly.github.io/spec/core/text/values.html#text-id
107static constexpr char kIdentifierChar[] = {
108 '_', '!', '_', '#', '$', '%', '&', '\'', // --
109 '_', '_', '*', '+', '_', '-', '.', '/', // --
110 '0', '1', '2', '3', '4', '5', '6', '7', // --
111 '8', '9', ':', '_', '<', '=', '>', '?', // --
112 '@', 'A', 'B', 'C', 'D', 'E', 'F', 'G', // --
113 'H', 'I', 'J', 'K', 'L', 'M', 'N', 'O', // --
114 'P', 'Q', 'R', 'S', 'T', 'U', 'V', 'W', // --
115 'X', 'Y', 'Z', '_', '\\', '_', '^', '_', // --
116 '`', 'a', 'b', 'c', 'd', 'e', 'f', 'g', // --
117 'h', 'i', 'j', 'k', 'l', 'm', 'n', 'o', // --
118 'p', 'q', 'r', 's', 't', 'u', 'v', 'w', // --
119 'x', 'y', 'z', '_', '|', '_', '~', '_', // --
120};
121
122// To match legacy wasmparser behavior, we emit one '_' per invalid UTF16
123// code unit.
124// We could decide that we don't care much how exactly non-ASCII names are
125// rendered and simplify this to "one '_' per invalid UTF8 byte".
126void SanitizeUnicodeName(StringBuilder& out, const uint8_t* utf8_src,
127 size_t length) {
128 if (length == 0) return; // Illegal nullptrs arise below when length == 0.
129 base::Vector<const uint8_t> utf8_data(utf8_src, length);
130 Utf8Decoder decoder(utf8_data);
131 std::vector<uint16_t> utf16(decoder.utf16_length());
132 decoder.Decode(utf16.data(), utf8_data);
133 for (uint16_t c : utf16) {
134 if (c < 32 || c >= 127) {
135 out << '_';
136 } else {
137 out << kIdentifierChar[c - 32];
138 }
139 }
140}

143void NamesProvider::ComputeImportName(const WasmImport& import,
144 std::map<uint32_t, std::string>& target) {
145 const uint8_t* mod_start = wire_bytes_.begin() + import.module_name.offset();
146 size_t mod_length = import.module_name.length();
147 const uint8_t* field_start = wire_bytes_.begin() + import.field_name.offset();
148 size_t field_length = import.field_name.length();
149 StringBuilder buffer;
150 buffer << '$';
151 SanitizeUnicodeName(buffer, mod_start, mod_length);
152 buffer << '.';
153 SanitizeUnicodeName(buffer, field_start, field_length);
154 target[import.index] = std::string(buffer.start(), buffer.length());
155}

157void NamesProvider::ComputeExportName(const WasmExport& ex,
158 std::map<uint32_t, std::string>& target) {
159 if (target.find(ex.index) != target.end()) return;
160 size_t length = ex.name.length();
161 if (length == 0) return;
162 StringBuilder buffer;
163 buffer << '$';
164 SanitizeUnicodeName(buffer, wire_bytes_.begin() + ex.name.offset(), length);
165 target[ex.index] = std::string(buffer.start(), buffer.length());
166}

181void NamesProvider::PrintFunctionName(StringBuilder& out,
182 uint32_t function_index,
183 FunctionNamesBehavior behavior,
184 IndexAsComment index_as_comment) {
185 // Function names are stored elsewhere, because we need to access them
186 // during (streaming) compilation when the NamesProvider isn't ready yet.
187 WireBytesRef ref = module_->lazily_generated_names.LookupFunctionName(
188 ModuleWireBytes(wire_bytes_), function_index);
189 if (ref.is_set()) {
190 if (behavior == kDevTools) {
191 out << '$';
192 WriteRef(out, ref);
193 MaybeAddComment(out, function_index, index_as_comment);
194 } else {
195 // For kWasmInternal behavior, function names don't get a `$` prefix.
196 WriteRef(out, ref);
197 }
198 return;
199 }
200
201 if (behavior == kWasmInternal) return;
202 {
203 base::SpinningMutexGuard lock(&mutex_);
204 if (!has_computed_function_import_names_) {
205 ComputeFunctionNamesFromImportsExports();
206 }
207 }
208 auto it = import_export_function_names_.find(function_index);
209 if (it != import_export_function_names_.end()) {
210 out << it->second;
211 MaybeAddComment(out, function_index, index_as_comment);
212 } else {
213 out << "$func" << function_index;
214 }
215}

396void NamesProvider::PrintValueType(StringBuilder& out, ValueType type) {
397 switch (type.kind()) {
398 case kRef:
399 case kRefNull:
400 if (type.encoding_needs_heap_type()) {
401 out << (type.kind() == kRef ? "(ref " : "(ref null ");
402 PrintHeapType(out, type.heap_type());
403 out << ')';
404 } else {
405 out << type.heap_type().name() << "ref";
406 }
407 break;
408 case kRtt:
409 out << "(rtt ";
410 PrintTypeName(out, type.ref_index());
411 out << ')';
412 break;
413 default:
414 out << wasm::name(type.kind());
415 }
416}

428size_t NamesProvider::EstimateCurrentMemoryConsumption() const {
429 UPDATE_WHEN_CLASS_CHANGES(NamesProvider, 160);
430 size_t result = sizeof(NamesProvider);
431 if (name_section_names_) {
432 DecodedNameSection* names = name_section_names_.get();
433 result += names->local_names_.EstimateCurrentMemoryConsumption();
434 result += names->label_names_.EstimateCurrentMemoryConsumption();
435 result += names->type_names_.EstimateCurrentMemoryConsumption();
436 result += names->table_names_.EstimateCurrentMemoryConsumption();
437 result += names->memory_names_.EstimateCurrentMemoryConsumption();
438 result += names->global_names_.EstimateCurrentMemoryConsumption();
439 result += names->element_segment_names_.EstimateCurrentMemoryConsumption();
440 result += names->data_segment_names_.EstimateCurrentMemoryConsumption();
441 result += names->field_names_.EstimateCurrentMemoryConsumption();
442 result += names->tag_names_.EstimateCurrentMemoryConsumption();
443 }
444 {
445 base::SpinningMutexGuard lock(&mutex_);
446 result += StringMapSize(import_export_function_names_);
447 result += StringMapSize(import_export_table_names_);
448 result += StringMapSize(import_export_memory_names_);
449 result += StringMapSize(import_export_global_names_);
450 result += StringMapSize(import_export_tag_names_);
451 }
452 if (v8_flags.trace_wasm_offheap_memory) {
453 PrintF("NamesProvider: %zu\n", result);
454 }
455 return result;
456}
