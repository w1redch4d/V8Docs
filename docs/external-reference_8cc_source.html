404// Helper function to verify that all types in a list of types are scalar.
405// This includes primitive types (int, Address) and pointer types. We also
406// allow void.
407template <typename T>
408constexpr bool AllScalar() {
409 return std::is_scalar_v<T> || std::is_void_v<T>;
410}
411
412template <typename T1, typename T2, typename... Rest>
413constexpr bool AllScalar() {
414 return AllScalar<T1>() && AllScalar<T2, Rest...>();
415}
416
417// Checks a function pointer's type for compatibility with the
418// ExternalReference calling mechanism. Specifically, all arguments
419// as well as the result type must pass the AllScalar check above,
420// because we expect each item to fit into one register or stack slot.
421template <typename T>
422struct IsValidExternalReferenceType;
423
424template <typename Result, typename... Args>
425struct IsValidExternalReferenceType<Result (*)(Args...)> {
426 static const bool value = AllScalar<Result, Args...>();
427};
428
429template <typename Result, typename Class, typename... Args>
430struct IsValidExternalReferenceType<Result (Class::*)(Args...)> {
431 static const bool value = AllScalar<Result, Args...>();
432};

481 DisallowGarbageCollection no_gc;
482 Tagged<SharedFunctionInfo> sfi =
483 Cast<SharedFunctionInfo>(Tagged<Object>(raw_sfi));
484 std::optional<Tagged<DebugInfo>> debug_info =
485 isolate->debug()->TryGetDebugInfo(sfi);
486 if (debug_info.has_value() && debug_info.value()->HasCoverageInfo()) {
487 return debug_info.value()->coverage_info().ptr();
488 }
489 return Smi::zero().ptr();
490}
