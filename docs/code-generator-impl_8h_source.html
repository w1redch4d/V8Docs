18// Converts InstructionOperands from a given instruction to
19// architecture-specific
20// registers and operands after they have been assigned by the register
21// allocator.

24 InstructionOperandConverter(CodeGenerator* gen, Instruction* instr)

29 Register InputRegister(size_t index) const {

49 int32_t InputInt32(size_t index) {

52
53 uint32_t InputUint32(size_t index) {
54 return base::bit_cast<uint32_t>(InputInt32(index));

56
57 int64_t InputInt64(size_t index) {

60
61 int8_t InputInt8(size_t index) {

64
65 uint8_t InputUint8(size_t index) {

68
69 int16_t InputInt16(size_t index) {
70 return static_cast<int16_t>(InputInt32(index));
71 }
72
73 uint8_t InputInt3(size_t index) {
74 return static_cast<uint8_t>(InputInt32(index) & 0x7);

89 CodeEntrypointTag InputCodeEntrypointTag(size_t index) {

168 }
169

171 return LocationOperand::cast(op)->GetDoubleRegister();
172 }
173
174 Simd128Register ToSimd128Register(InstructionOperand* op) {
175 LocationOperand* loc_op = LocationOperand::cast(op);
176#ifdef V8_TARGET_ARCH_X64
177 if (loc_op->IsSimd256Register()) {
178 return loc_op->GetSimd256RegisterAsSimd128();
179 }
180#endif
181 return loc_op->GetSimd128Register();
182 }
183
184#if defined(V8_TARGET_ARCH_X64)
185 Simd256Register ToSimd256Register(InstructionOperand* op) {
186 return LocationOperand::cast(op)->GetSimd256Register();
187 }
188#endif
189
190 Constant ToConstant(InstructionOperand* op) const {
191 if (op->IsImmediate()) {
192 return gen_->instructions()->GetImmediate(ImmediateOperand::cast(op));
193 }
194 return gen_->instructions()->GetConstant(
195 ConstantOperand::cast(op)->virtual_register());
196 }
197
198 double ToDouble(InstructionOperand* op) {
199 return ToConstant(op).ToFloat64().value();
200 }
201
202 float ToFloat32(InstructionOperand* op) { return ToConstant(op).ToFloat32(); }
203
204 ExternalReference ToExternalReference(InstructionOperand* op) {
205 return ToConstant(op).ToExternalReference();
206 }
207
208 Handle<Code> ToCode(InstructionOperand* op) {
209 return ToConstant(op).ToCode();
210 }
211
212 const Frame* frame() const { return gen_->frame(); }
213 FrameAccessState* frame_access_state() const {
214 return gen_->frame_access_state();
215 }
216 Isolate* isolate() const { return gen_->isolate(); }
217 Linkage* linkage() const { return gen_->linkage(); }
218
219 protected:
220 CodeGenerator* gen_;
221 Instruction* instr_;
222};

224// Deoptimization exit.
225class DeoptimizationExit : public ZoneObject {
226 public:
227 explicit DeoptimizationExit(SourcePosition pos, BytecodeOffset bailout_id,
228 int translation_id, int pc_offset,
229 DeoptimizeKind kind, DeoptimizeReason reason,
230 NodeId node_id)
231 : deoptimization_id_(kNoDeoptIndex),
232 pos_(pos),
233 bailout_id_(bailout_id),
234 translation_id_(translation_id),
235 pc_offset_(pc_offset),
236 kind_(kind),
237 reason_(reason),
238 node_id_(node_id),
239 immediate_args_(nullptr),
240 emitted_(false) {}
241
242 bool has_deoptimization_id() const {
243 return deoptimization_id_ != kNoDeoptIndex;
244 }
245 int deoptimization_id() const {
246 DCHECK(has_deoptimization_id());
247 return deoptimization_id_;
248 }
249 void set_deoptimization_id(int deoptimization_id) {
250 deoptimization_id_ = deoptimization_id;
251 }
252 SourcePosition pos() const { return pos_; }
253 // The label for the deoptimization call.
254 Label* label() { return &label_; }
255 // The label after the deoptimization check, which will resume execution.
256 Label* continue_label() { return &continue_label_; }
257 BytecodeOffset bailout_id() const { return bailout_id_; }
258 int translation_id() const { return translation_id_; }
259 int pc_offset() const { return pc_offset_; }
260 DeoptimizeKind kind() const { return kind_; }
261 DeoptimizeReason reason() const { return reason_; }
262 NodeId node_id() const { return node_id_; }
263 const ZoneVector<ImmediateOperand*>* immediate_args() const {
264 return immediate_args_;
265 }
266 void set_immediate_args(ZoneVector<ImmediateOperand*>* immediate_args) {
267 immediate_args_ = immediate_args;
268 }
269 // Returns whether the deopt exit has already been emitted. Most deopt exits
270 // are emitted contiguously at the end of the code, but unconditional deopt
271 // exits (kArchDeoptimize) may be inlined where they are encountered.
272 bool emitted() const { return emitted_; }
273 void set_emitted() { emitted_ = true; }
274
275 private:
276 static const int kNoDeoptIndex = kMaxInt16 + 1;
277 int deoptimization_id_;
278 const SourcePosition pos_;
279 Label label_;
280 Label continue_label_;
281 const BytecodeOffset bailout_id_;
282 const int translation_id_;
283 const int pc_offset_;
284 const DeoptimizeKind kind_;
285 const DeoptimizeReason reason_;
286 const NodeId node_id_;
287 ZoneVector<ImmediateOperand*>* immediate_args_;
288 bool emitted_;
289};
