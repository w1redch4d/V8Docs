41// Mutex - a replacement for std::mutex
42//
43// This class is a synchronization primitive that can be used to protect shared
44// data from being simultaneously accessed by multiple threads. A mutex offers
45// exclusive, non-recursive ownership semantics:
46// - A calling thread owns a mutex from the time that it successfully calls
47// either |Lock()| or |TryLock()| until it calls |Unlock()|.
48// - When a thread owns a mutex, all other threads will block (for calls to
49// |Lock()|) or receive a |false| return value (for |TryLock()|) if they
50// attempt to claim ownership of the mutex.
51// A calling thread must not own the mutex prior to calling |Lock()| or
52// |TryLock()|. The behavior of a program is undefined if a mutex is destroyed
53// while still owned by some thread. The Mutex class is non-copyable.
54
55class V8_BASE_EXPORT Mutex final {
56 public:
57 Mutex();
58 Mutex(const Mutex&) = delete;
59 Mutex& operator=(const Mutex&) = delete;
60 ~Mutex();
61
62 // Locks the given mutex. If the mutex is currently unlocked, it becomes
63 // locked and owned by the calling thread, and immediately. If the mutex
64 // is already locked by another thread, suspends the calling thread until
65 // the mutex is unlocked.
66 void Lock();
67
68 // Unlocks the given mutex. The mutex is assumed to be locked and owned by
69 // the calling thread on entrance.
70 void Unlock();
71
72 // Tries to lock the given mutex. Returns whether the mutex was
73 // successfully locked.
74 // Note: Instead of `DCHECK(!mutex.TryLock())` use `mutex.AssertHeld()`.
75 bool TryLock() V8_WARN_UNUSED_RESULT;
76
77 // The implementation-defined native handle type.
78#if V8_OS_POSIX
79 using NativeHandle = pthread_mutex_t;
80#elif V8_OS_WIN
81 using NativeHandle = V8_SRWLOCK;
82#elif V8_OS_STARBOARD
83 using NativeHandle = SbMutex;
84#endif
85
86 NativeHandle& native_handle() {
87 return native_handle_;
88 }
89 const NativeHandle& native_handle() const {
90 return native_handle_;
91 }
92
93 V8_INLINE void AssertHeld() const {
94 // If this access results in a race condition being detected by TSan, this
95 // means that you in fact did *not* hold the mutex.
96 DCHECK_EQ(1, level_);
97 }
98
99 private:
100#ifdef DEBUG
101 // This is being used for Assert* methods. Accesses are only allowed if you
102 // actually hold the mutex, otherwise you would get race conditions.
103 int level_;
104#endif
105
106 V8_INLINE void AssertHeldAndUnmark() {
107#ifdef DEBUG
108 // If this access results in a race condition being detected by TSan, this
109 // means that you in fact did *not* hold the mutex.
110 DCHECK_EQ(1, level_);
111 level_--;
112#endif
113 }
114
115 V8_INLINE void AssertUnheldAndMark() {
116#ifdef DEBUG
117 // This is only invoked *after* actually getting the mutex, so should not
118 // result in race conditions.
119 DCHECK_EQ(0, level_);
120 level_++;
121#endif
122 }
123
124 friend class ConditionVariable;
125
126 NativeHandle native_handle_;
127};

160//

162// (or a non-boolean value that can be interpreted as a boolean).
163// The first argument should be `true` for functions that return `true` on
164// success, or `false` for functions that return `false` on success. The second
165// argument specifies the mutex that is locked on success. If unspecified, this

170// The behavior of this class depends on platform support:
171// 1. When platform supports is available:
172//
173// Simple spinning lock. It will spin in user space a set number of times before
174// going into the kernel to sleep.
175//
176// This is intended to give "the best of both worlds" between a SpinLock and
177// base::Lock:
178// - SpinLock: Inlined fast path, no external function calls, just
179// compare-and-swap. Short waits do not go into the kernel. Good behavior in
180// low contention cases.
181// - base::Lock: Good behavior in case of contention.
182//
183// We don't rely on base::Lock which we could make spin (by calling TryLock() in
184// a loop), as performance is below a custom spinlock as seen on high-level
185// benchmarks. Instead this implements a simple non-recursive mutex on top of:
186// - Linux : futex()
187// - Windows : SRWLock
188// - MacOS : os_unfair_lock
189// - POSIX : pthread_mutex_trylock()
190//
191// The main difference between this and a libc implementation is that it only
192// supports the simplest path: private (to a process), non-recursive mutexes
193// with no priority inheritance, no timed waits.
194//
195// As an interesting side-effect to be used in the allocator, this code does not
196// make any allocations, locks are small with a constexpr constructor and no
197// destructor.
198//
199// 2. Otherwise: This is a simple SpinLock, in the sense that it does not have
200// any awareness of other threads' behavior.
201class V8_BASE_EXPORT V8_LOCKABLE SpinningMutex final {
202 public:
203 SpinningMutex();
204 V8_INLINE void Lock() V8_EXCLUSIVE_LOCK_FUNCTION();
205 void Unlock() V8_UNLOCK_FUNCTION();
206 bool TryLock() V8_EXCLUSIVE_TRYLOCK_FUNCTION(true);
207 void AssertHeld() const {} // Not supported.
208
209 private:
210 V8_NOINLINE void AcquireSpinThenBlock() V8_EXCLUSIVE_LOCK_FUNCTION();
211 void LockSlow() V8_EXCLUSIVE_LOCK_FUNCTION();
212
213 // See below, the latency of YIELD_PROCESSOR can be as high as ~150
214 // cycles. Meanwhile, sleeping costs a few us. Spinning 64 times at 3GHz would
215 // cost 150 * 64 / 3e9 ~= 3.2us.
216 //
217 // This applies to Linux kernels, on x86_64. On ARM we might want to spin
218 // more.
219 static constexpr int kSpinCount = 64;
220
221#if V8_TARGET_OS_LINUX
222 void FutexWait();
223 void FutexWake();
224
225 static constexpr int kUnlocked = 0;
226 static constexpr int kLockedUncontended = 1;
227 static constexpr int kLockedContended = 2;
228
229 std::atomic<int32_t> state_{kUnlocked};
230#elif V8_OS_WIN
231 V8_SRWLOCK lock_;
232#elif V8_OS_DARWIN
233 os_unfair_lock lock_;
234#elif V8_OS_POSIX
235 pthread_mutex_t lock_;
236#elif V8_OS_FUCHSIA
237 sync_mutex lock_;
238#else
239 std::atomic<bool> lock_{false};
240#endif
241};

243V8_INLINE void SpinningMutex::Lock() {
244 // Not marked `[[likely]]`, as:
245 // 1. We don't know how much contention the lock would experience
246 // 2. This may lead to weird-looking code layout when inlined into a caller
247 // with `[[(un)likely]]` attributes.
248 if (TryLock()) {
249 return;
250 }
251
252 return AcquireSpinThenBlock();
253}
254
255#if V8_TARGET_OS_LINUX
256
257V8_INLINE bool SpinningMutex::TryLock() {
258 // Using the weak variant of compare_exchange(), which may fail spuriously. On
259 // some architectures such as ARM, CAS is typically performed as a LDREX/STREX
260 // pair, where the store may fail. In the strong version, there is a loop
261 // inserted by the compiler to retry in these cases.
262 //
263 // Since we are retrying in Lock() anyway, there is no point having two nested
264 // loops.
265 int expected = kUnlocked;
266 return (state_.load(std::memory_order_relaxed) == expected) &&
267 state_.compare_exchange_weak(expected, kLockedUncontended,
268 std::memory_order_acquire,
269 std::memory_order_relaxed);
270}
271
272V8_INLINE void SpinningMutex::Unlock() {
273 if (state_.exchange(kUnlocked, std::memory_order_release) == kLockedContended)
274 [[unlikely]] {
275 // |kLockedContended|: there is a waiter to wake up.
276 //
277 // Here there is a window where the lock is unlocked, since we just set it
278 // to |kUnlocked| above. Meaning that another thread can grab the lock
279 // in-between now and |FutexWake()| waking up a waiter. Aside from
280 // potentially fairness, this is not an issue, as the newly-awaken thread
281 // will check that the lock is still free.
282 //
283 // There is a small pessimization here though: if we have a single waiter,
284 // then when it wakes up, the lock will be set to |kLockedContended|, so
285 // when this waiter releases the lock, it will needlessly call
286 // |FutexWake()|, even though there are no waiters. This is supported by the
287 // kernel, and is what bionic (Android's libc) also does.
288 FutexWake();
289 }
290}

356// RecursiveMutex - a replacement for std::recursive_mutex
357//
358// This class is a synchronization primitive that can be used to protect shared
359// data from being simultaneously accessed by multiple threads. A recursive
360// mutex offers exclusive, recursive ownership semantics:
361// - A calling thread owns a recursive mutex for a period of time that starts
362// when it successfully calls either |Lock()| or |TryLock()|. During this
363// period, the thread may make additional calls to |Lock()| or |TryLock()|.
364// The period of ownership ends when the thread makes a matching number of
365// calls to |Unlock()|.
366// - When a thread owns a recursive mutex, all other threads will block (for
367// calls to |Lock()|) or receive a |false| return value (for |TryLock()|) if
368// they attempt to claim ownership of the recursive mutex.
369// - The maximum number of times that a recursive mutex may be locked is
370// unspecified, but after that number is reached, calls to |Lock()| will
371// probably abort the process and calls to |TryLock()| return false.
372// The behavior of a program is undefined if a recursive mutex is destroyed
373// while still owned by some thread. The RecursiveMutex class is non-copyable.
374
375class V8_BASE_EXPORT RecursiveMutex final {
376 public:
377 RecursiveMutex();
378 RecursiveMutex(const RecursiveMutex&) = delete;
379 RecursiveMutex& operator=(const RecursiveMutex&) = delete;
380 ~RecursiveMutex();
381
382 // Locks the mutex. If another thread has already locked the mutex, a call to
383 // |Lock()| will block execution until the lock is acquired. A thread may call
384 // |Lock()| on a recursive mutex repeatedly. Ownership will only be released
385 // after the thread makes a matching number of calls to |Unlock()|.
386 // The behavior is undefined if the mutex is not unlocked before being
387 // destroyed, i.e. some thread still owns it.
388 void Lock();
389
390 // Unlocks the mutex if its level of ownership is 1 (there was exactly one
391 // more call to |Lock()| than there were calls to unlock() made by this
392 // thread), reduces the level of ownership by 1 otherwise. The mutex must be
393 // locked by the current thread of execution, otherwise, the behavior is
394 // undefined.
395 void Unlock();
396
397 // Tries to lock the given mutex. Returns whether the mutex was
398 // successfully locked.
399 // Note: Instead of `DCHECK(!mutex.TryLock())` use `mutex.AssertHeld()`.
400 bool TryLock() V8_WARN_UNUSED_RESULT;
401
402 V8_INLINE void AssertHeld() const {
403 // If this access results in a race condition being detected by TSan, this
404 // mean that you in fact did *not* hold the mutex.
405 DCHECK_LT(0, level_);
406 }
407
408 private:
409 // The implementation-defined native handle type.
410#if V8_OS_POSIX
411 using NativeHandle = pthread_mutex_t;
412#elif V8_OS_WIN
413 using NativeHandle = V8_CRITICAL_SECTION;
414#elif V8_OS_STARBOARD
415 using NativeHandle = starboard::RecursiveMutex;
416#endif
417
418 NativeHandle native_handle_;
419#ifdef DEBUG
420 // This is being used for Assert* methods. Accesses are only allowed if you
421 // actually hold the mutex, otherwise you would get race conditions.
422 int level_;
423#endif
424};

443// SharedMutex - a replacement for std::shared_mutex
444//
445// This class is a synchronization primitive that can be used to protect shared
446// data from being simultaneously accessed by multiple threads. In contrast to
447// other mutex types which facilitate exclusive access, a shared_mutex has two
448// levels of access:
449// - shared: several threads can share ownership of the same mutex.
450// - exclusive: only one thread can own the mutex.
451// Shared mutexes are usually used in situations when multiple readers can
452// access the same resource at the same time without causing data races, but
453// only one writer can do so.
454// The SharedMutex class is non-copyable.
455
456class V8_BASE_EXPORT SharedMutex final {
457 public:
458 SharedMutex();
459 SharedMutex(const SharedMutex&) = delete;
460 SharedMutex& operator=(const SharedMutex&) = delete;
461 ~SharedMutex();
462
463 // Acquires shared ownership of the {SharedMutex}. If another thread is
464 // holding the mutex in exclusive ownership, a call to {LockShared()} will
465 // block execution until shared ownership can be acquired.
466 // If {LockShared()} is called by a thread that already owns the mutex in any
467 // mode (exclusive or shared), the behavior is undefined and outright fails
468 // with dchecks on.
469 void LockShared();
470
471 // Locks the SharedMutex. If another thread has already locked the mutex, a
472 // call to {LockExclusive()} will block execution until the lock is acquired.
473 // If {LockExclusive()} is called by a thread that already owns the mutex in
474 // any mode (shared or exclusive), the behavior is undefined and outright
475 // fails with dchecks on.
476 void LockExclusive();
477
478 // Releases the {SharedMutex} from shared ownership by the calling thread.
479 // The mutex must be locked by the current thread of execution in shared mode,
480 // otherwise, the behavior is undefined and outright fails with dchecks on.
481 void UnlockShared();
482
483 // Unlocks the {SharedMutex}. It must be locked by the current thread of
484 // execution, otherwise, the behavior is undefined and outright fails with
485 // dchecks on.
486 void UnlockExclusive();
487
488 // Tries to lock the {SharedMutex} in shared mode. Returns immediately. On
489 // successful lock acquisition returns true, otherwise returns false.
490 // This function is allowed to fail spuriously and return false even if the
491 // mutex is not currenly exclusively locked by any other thread.
492 // If it is called by a thread that already owns the mutex in any mode
493 // (shared or exclusive), the behavior is undefined, and outright fails with
494 // dchecks on.
495 bool TryLockShared() V8_WARN_UNUSED_RESULT;
496
497 // Tries to lock the {SharedMutex}. Returns immediately. On successful lock
498 // acquisition returns true, otherwise returns false.
499 // This function is allowed to fail spuriously and return false even if the
500 // mutex is not currently locked by any other thread.
501 // If it is called by a thread that already owns the mutex in any mode
502 // (shared or exclusive), the behavior is undefined, and outright fails with
503 // dchecks on.
504 bool TryLockExclusive() V8_WARN_UNUSED_RESULT;
505
506 private:
507 // The implementation-defined native handle type.
508#if V8_OS_DARWIN
509 // pthread_rwlock_t is broken on MacOS when signals are being sent to the
510 // process (see https://crbug.com/v8/11399).
511 // We thus use std::shared_mutex on MacOS, which does not have this problem.
512 using NativeHandle = absl::Mutex;
513#elif V8_OS_POSIX
514 using NativeHandle = pthread_rwlock_t;
515#elif V8_OS_WIN
516 using NativeHandle = V8_SRWLOCK;
517#elif V8_OS_STARBOARD
518 using NativeHandle = starboard::RWLock;
519#endif
520
521 NativeHandle native_handle_;
522};

524// LockGuard
525//
526// This class is a mutex wrapper that provides a convenient RAII-style mechanism
527// for owning a mutex for the duration of a scoped block.
528// When a LockGuard object is created, it attempts to take ownership of the
529// mutex it is given. When control leaves the scope in which the LockGuard
530// object was created, the LockGuard is destructed and the mutex is released.
531// The LockGuard class is non-copyable.
532
533template <typename Mutex>
534class V8_NODISCARD LockGuard final {
535 public:
536 explicit LockGuard(Mutex* mutex) : mutex_(mutex) {
537 DCHECK_NOT_NULL(mutex_);
538 mutex_->Lock();
539 }
540 explicit LockGuard(Mutex& mutex) : mutex_(&mutex) {
541 // `mutex_` is guaranteed to be non-null here.
542 mutex_->Lock();
543 }
544 LockGuard(const LockGuard&) = delete;
545 LockGuard& operator=(const LockGuard&) = delete;
546 LockGuard(LockGuard&& other) V8_NOEXCEPT : mutex_(other.mutex_) {
547 DCHECK_NOT_NULL(mutex_);
548 other.mutex_ = nullptr;
549 }
550 ~LockGuard() {
551 if (mutex_) {
552 // mutex_ may have been moved away.
553 mutex_->Unlock();
554 }
555 }
556
557 private:
558 Mutex* mutex_;
559};
