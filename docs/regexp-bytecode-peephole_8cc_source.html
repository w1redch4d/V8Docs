288BytecodeSequenceNode& BytecodeSequenceNode::FollowedBy(int bytecode) {
289 DCHECK(0 <= bytecode && bytecode < kRegExpBytecodeCount);
290
291 if (children_.find(bytecode) == children_.end()) {
292 BytecodeSequenceNode* new_node =
293 zone()->New<BytecodeSequenceNode>(bytecode, zone());
294 // If node is not the first in the sequence, set offsets and parent.
295 if (bytecode_ != kDummyBytecode) {
296 new_node->start_offset_ = start_offset_ + RegExpBytecodeLength(bytecode_);
297 new_node->index_in_sequence_ = index_in_sequence_ + 1;
298 new_node->parent_ = this;
299 }
300 children_[bytecode] = new_node;
301 }
302
303 return *children_[bytecode];
304}

314BytecodeSequenceNode& BytecodeSequenceNode::MapArgument(
315 int bytecode_index_in_sequence, int argument_offset,
316 int argument_byte_length, int new_argument_byte_length) {
317 DCHECK(IsSequence());
318 DCHECK_LE(bytecode_index_in_sequence, index_in_sequence_);
319
320 BytecodeSequenceNode& ref_node =
321 GetNodeByIndexInSequence(bytecode_index_in_sequence);
322 DCHECK_LT(argument_offset, RegExpBytecodeLength(ref_node.bytecode_));
323
324 int absolute_offset = ref_node.start_offset_ + argument_offset;
325 if (new_argument_byte_length == 0) {
326 new_argument_byte_length = argument_byte_length;
327 }
328
329 argument_mapping_->push_back(BytecodeArgumentMapping{
330 absolute_offset, argument_byte_length, new_argument_byte_length});
331
332 return *this;
333}

389bool BytecodeSequenceNode::CheckArguments(const uint8_t* bytecode, int pc) {
390 bool is_valid = true;
391 for (auto check_iter = argument_check_->begin();
392 check_iter != argument_check_->end() && is_valid; check_iter++) {
393 auto value =
394 GetArgumentValue(bytecode, pc + check_iter->offset, check_iter->length);
395 if (check_iter->type == BytecodeArgumentCheck::kCheckAddress) {
396 is_valid &= value == pc + check_iter->check_offset;
397 } else if (check_iter->type == BytecodeArgumentCheck::kCheckValue) {
398 auto other_value = GetArgumentValue(
399 bytecode, pc + check_iter->check_offset, check_iter->check_length);
400 is_valid &= value == other_value;
401 } else {
402 UNREACHABLE();
403 }
404 }
405 return is_valid;
406}

472RegExpBytecodePeephole::RegExpBytecodePeephole(
473 Zone* zone, size_t buffer_size,
474 const ZoneUnorderedMap<int, int>& jump_edges)
475 : optimized_bytecode_buffer_(zone),
476 sequences_(zone->New<BytecodeSequenceNode>(
477 BytecodeSequenceNode::kDummyBytecode, zone)),
478 jump_edges_(zone),
479 jump_edges_mapped_(zone),
480 jump_usage_counts_(zone),
481 jump_source_fixups_(zone),
482 jump_destination_fixups_(zone),
483 zone_(zone) {
484 optimized_bytecode_buffer_.reserve(buffer_size);
485 PrepareJumpStructures(jump_edges);
486 DefineStandardSequences();
487 // Sentinel fixups at beginning of bytecode (position -1) so we don't have to
488 // check for end of iterator inside the fixup loop.
489 // In general fixups are deltas of original offsets of jump
490 // sources/destinations (in the old bytecode) to find them in the new
491 // bytecode. All jump targets are fixed after the new bytecode is fully
492 // emitted in the internal buffer.
493 AddSentinelFixups(-1);
494 // Sentinel fixups at end of (old) bytecode so we don't have to check for
495 // end of iterator inside the fixup loop.
496 DCHECK_LE(buffer_size, std::numeric_limits<int>::max());
497 AddSentinelFixups(static_cast<int>(buffer_size));
498}

500void RegExpBytecodePeephole::DefineStandardSequences() {
501 // Commonly used sequences can be found by creating regexp bytecode traces
502 // (--trace-regexp-bytecodes) and using v8/tools/regexp-sequences.py.
503 CreateSequence(BC_LOAD_CURRENT_CHAR)
504 .FollowedBy(BC_CHECK_BIT_IN_TABLE)
505 .FollowedBy(BC_ADVANCE_CP_AND_GOTO)
506 // Sequence is only valid if the jump target of ADVANCE_CP_AND_GOTO is the
507 // first bytecode in this sequence.
508 .IfArgumentEqualsOffset(4, 4, 0)
509 .ReplaceWith(BC_SKIP_UNTIL_BIT_IN_TABLE)
510 .MapArgument(0, 1, 3) // load offset
511 .MapArgument(2, 1, 3, 4) // advance by
512 .MapArgument(1, 8, 16) // bit table
513 .MapArgument(1, 4, 4) // goto when match
514 .MapArgument(0, 4, 4) // goto on failure
515 .IgnoreArgument(2, 4, 4); // loop jump
516
517 CreateSequence(BC_CHECK_CURRENT_POSITION)
518 .FollowedBy(BC_LOAD_CURRENT_CHAR_UNCHECKED)
519 .FollowedBy(BC_CHECK_CHAR)
520 .FollowedBy(BC_ADVANCE_CP_AND_GOTO)
521 // Sequence is only valid if the jump target of ADVANCE_CP_AND_GOTO is the
522 // first bytecode in this sequence.
523 .IfArgumentEqualsOffset(4, 4, 0)
524 .ReplaceWith(BC_SKIP_UNTIL_CHAR_POS_CHECKED)
525 .MapArgument(1, 1, 3) // load offset
526 .MapArgument(3, 1, 3, 2) // advance_by
527 .MapArgument(2, 1, 3, 2) // c
528 .MapArgument(0, 1, 3, 4) // eats at least
529 .MapArgument(2, 4, 4) // goto when match
530 .MapArgument(0, 4, 4) // goto on failure
531 .IgnoreArgument(3, 4, 4); // loop jump
532
533 CreateSequence(BC_CHECK_CURRENT_POSITION)
534 .FollowedBy(BC_LOAD_CURRENT_CHAR_UNCHECKED)
535 .FollowedBy(BC_AND_CHECK_CHAR)
536 .FollowedBy(BC_ADVANCE_CP_AND_GOTO)
537 // Sequence is only valid if the jump target of ADVANCE_CP_AND_GOTO is the
538 // first bytecode in this sequence.
539 .IfArgumentEqualsOffset(4, 4, 0)
540 .ReplaceWith(BC_SKIP_UNTIL_CHAR_AND)
541 .MapArgument(1, 1, 3) // load offset
542 .MapArgument(3, 1, 3, 2) // advance_by
543 .MapArgument(2, 1, 3, 2) // c
544 .MapArgument(2, 4, 4) // mask
545 .MapArgument(0, 1, 3, 4) // eats at least
546 .MapArgument(2, 8, 4) // goto when match
547 .MapArgument(0, 4, 4) // goto on failure
548 .IgnoreArgument(3, 4, 4); // loop jump
549
550 // TODO(pthier): It might make sense for short sequences like this one to only
551 // optimize them if the resulting optimization is not longer than the current
552 // one. This could be the case if there are jumps inside the sequence and we
553 // have to replicate parts of the sequence. A method to mark such sequences
554 // might be useful.
555 CreateSequence(BC_LOAD_CURRENT_CHAR)
556 .FollowedBy(BC_CHECK_CHAR)
557 .FollowedBy(BC_ADVANCE_CP_AND_GOTO)
558 // Sequence is only valid if the jump target of ADVANCE_CP_AND_GOTO is the
559 // first bytecode in this sequence.
560 .IfArgumentEqualsOffset(4, 4, 0)
561 .ReplaceWith(BC_SKIP_UNTIL_CHAR)
562 .MapArgument(0, 1, 3) // load offset
563 .MapArgument(2, 1, 3, 2) // advance by
564 .MapArgument(1, 1, 3, 2) // character
565 .MapArgument(1, 4, 4) // goto when match
566 .MapArgument(0, 4, 4) // goto on failure
567 .IgnoreArgument(2, 4, 4); // loop jump
568
569 CreateSequence(BC_LOAD_CURRENT_CHAR)
570 .FollowedBy(BC_CHECK_CHAR)
571 .FollowedBy(BC_CHECK_CHAR)
572 // Sequence is only valid if the jump targets of both CHECK_CHAR bytecodes
573 // are equal.
574 .IfArgumentEqualsValueAtOffset(4, 4, 1, 4, 4)
575 .FollowedBy(BC_ADVANCE_CP_AND_GOTO)
576 // Sequence is only valid if the jump target of ADVANCE_CP_AND_GOTO is the
577 // first bytecode in this sequence.
578 .IfArgumentEqualsOffset(4, 4, 0)
579 .ReplaceWith(BC_SKIP_UNTIL_CHAR_OR_CHAR)
580 .MapArgument(0, 1, 3) // load offset
581 .MapArgument(3, 1, 3, 4) // advance by
582 .MapArgument(1, 1, 3, 2) // character 1
583 .MapArgument(2, 1, 3, 2) // character 2
584 .MapArgument(1, 4, 4) // goto when match
585 .MapArgument(0, 4, 4) // goto on failure
586 .IgnoreArgument(2, 4, 4) // goto when match 2
587 .IgnoreArgument(3, 4, 4); // loop jump
588
589 CreateSequence(BC_LOAD_CURRENT_CHAR)
590 .FollowedBy(BC_CHECK_GT)
591 // Sequence is only valid if the jump target of CHECK_GT is the first
592 // bytecode AFTER the whole sequence.
593 .IfArgumentEqualsOffset(4, 4, 56)
594 .FollowedBy(BC_CHECK_BIT_IN_TABLE)
595 // Sequence is only valid if the jump target of CHECK_BIT_IN_TABLE is
596 // the ADVANCE_CP_AND_GOTO bytecode at the end of the sequence.
597 .IfArgumentEqualsOffset(4, 4, 48)
598 .FollowedBy(BC_GOTO)
599 // Sequence is only valid if the jump target of GOTO is the same as the
600 // jump target of CHECK_GT (i.e. both jump to the first bytecode AFTER the
601 // whole sequence.
602 .IfArgumentEqualsValueAtOffset(4, 4, 1, 4, 4)
603 .FollowedBy(BC_ADVANCE_CP_AND_GOTO)
604 // Sequence is only valid if the jump target of ADVANCE_CP_AND_GOTO is the
605 // first bytecode in this sequence.
606 .IfArgumentEqualsOffset(4, 4, 0)
607 .ReplaceWith(BC_SKIP_UNTIL_GT_OR_NOT_BIT_IN_TABLE)
608 .MapArgument(0, 1, 3) // load offset
609 .MapArgument(4, 1, 3, 2) // advance by
610 .MapArgument(1, 1, 3, 2) // character
611 .MapArgument(2, 8, 16) // bit table
612 .MapArgument(1, 4, 4) // goto when match
613 .MapArgument(0, 4, 4) // goto on failure
614 .IgnoreArgument(2, 4, 4) // indirect loop jump
615 .IgnoreArgument(3, 4, 4) // jump out of loop
616 .IgnoreArgument(4, 4, 4); // loop jump
617}

619bool RegExpBytecodePeephole::OptimizeBytecode(const uint8_t* bytecode,
620 int length) {
621 int old_pc = 0;
622 bool did_optimize = false;
623
624 while (old_pc < length) {
625 int replaced_len = TryOptimizeSequence(bytecode, length, old_pc);
626 if (replaced_len > 0) {
627 old_pc += replaced_len;
628 did_optimize = true;
629 } else {
630 int bc = bytecode[old_pc];
631 int bc_len = RegExpBytecodeLength(bc);
632 CopyRangeToOutput(bytecode, old_pc, bc_len);
633 old_pc += bc_len;
634 }
635 }
636
637 if (did_optimize) {
638 FixJumps();
639 }
640
641 return did_optimize;
642}

657int RegExpBytecodePeephole::TryOptimizeSequence(const uint8_t* bytecode,
658 int bytecode_length,
659 int start_pc) {
660 BytecodeSequenceNode* seq_node = sequences_;
661 BytecodeSequenceNode* valid_seq_end = nullptr;
662
663 int current_pc = start_pc;
664
665 // Check for the longest valid sequence matching any of the pre-defined
666 // sequences in the Trie data structure.
667 while (current_pc < bytecode_length) {
668 seq_node = seq_node->Find(bytecode[current_pc]);
669 if (seq_node == nullptr) break;
670 if (!seq_node->CheckArguments(bytecode, start_pc)) break;
671
672 if (seq_node->IsSequence()) valid_seq_end = seq_node;
673 current_pc += RegExpBytecodeLength(bytecode[current_pc]);
674 }
675
676 if (valid_seq_end) {
677 EmitOptimization(start_pc, bytecode, *valid_seq_end);
678 return valid_seq_end->SequenceLength();
679 }
680
681 return 0;
682}

684void RegExpBytecodePeephole::EmitOptimization(
685 int start_pc, const uint8_t* bytecode,
686 const BytecodeSequenceNode& last_node) {
687#ifdef DEBUG
688 int optimized_start_pc = pc();
689#endif
690 // Jump sources that are mapped or marked as unused will be deleted at the end
691 // of this method. We don't delete them immediately as we might need the
692 // information when we have to preserve bytecodes at the end.
693 // TODO(pthier): Replace with a stack-allocated data structure.
694 ZoneLinkedList<int> delete_jumps = ZoneLinkedList<int>(zone());
695
696 uint32_t bc = last_node.OptimizedBytecode();
697 EmitValue(bc);
698
699 for (size_t arg = 0; arg < last_node.ArgumentSize(); arg++) {
700 BytecodeArgumentMapping arg_map = last_node.ArgumentMapping(arg);
701 int arg_pos = start_pc + arg_map.offset;
702 // If we map any jump source we mark the old source for deletion and insert
703 // a new jump.
704 auto jump_edge_iter = jump_edges_.find(arg_pos);
705 if (jump_edge_iter != jump_edges_.end()) {
706 int jump_source = jump_edge_iter->first;
707 int jump_destination = jump_edge_iter->second;
708 // Add new jump edge add current position.
709 jump_edges_mapped_.emplace(Length(), jump_destination);
710 // Mark old jump edge for deletion.
711 delete_jumps.push_back(jump_source);
712 // Decrement usage count of jump destination.
713 auto jump_count_iter = jump_usage_counts_.find(jump_destination);
714 DCHECK(jump_count_iter != jump_usage_counts_.end());
715 int& usage_count = jump_count_iter->second;
716 --usage_count;
717 }
718 // TODO(pthier): DCHECK that mapped arguments are never sources of jumps
719 // to destinations inside the sequence.
720 EmitArgument(start_pc, bytecode, arg_map);
721 }
722 DCHECK_EQ(pc(), optimized_start_pc +
723 RegExpBytecodeLength(last_node.OptimizedBytecode()));
724
725 // Remove jumps from arguments we ignore.
726 if (last_node.HasIgnoredArguments()) {
727 for (auto ignored_arg = last_node.ArgumentIgnoredBegin();
728 ignored_arg != last_node.ArgumentIgnoredEnd(); ignored_arg++) {
729 auto jump_edge_iter = jump_edges_.find(start_pc + ignored_arg->offset);
730 if (jump_edge_iter != jump_edges_.end()) {
731 int jump_source = jump_edge_iter->first;
732 int jump_destination = jump_edge_iter->second;
733 // Mark old jump edge for deletion.
734 delete_jumps.push_back(jump_source);
735 // Decrement usage count of jump destination.
736 auto jump_count_iter = jump_usage_counts_.find(jump_destination);
737 DCHECK(jump_count_iter != jump_usage_counts_.end());
738 int& usage_count = jump_count_iter->second;
739 --usage_count;
740 }
741 }
742 }
743
744 int fixup_length = RegExpBytecodeLength(bc) - last_node.SequenceLength();
745
746 // Check if there are any jumps inside the old sequence.
747 // If so we have to keep the bytecodes that are jumped to around.
748 auto jump_destination_candidate = jump_usage_counts_.upper_bound(start_pc);
749 int jump_candidate_destination = jump_destination_candidate->first;
750 int jump_candidate_count = jump_destination_candidate->second;
751 // Jump destinations only jumped to from inside the sequence will be ignored.
752 while (jump_destination_candidate != jump_usage_counts_.end() &&
753 jump_candidate_count == 0) {
754 ++jump_destination_candidate;
755 jump_candidate_destination = jump_destination_candidate->first;
756 jump_candidate_count = jump_destination_candidate->second;
757 }
758
759 int preserve_from = start_pc + last_node.SequenceLength();
760 if (jump_destination_candidate != jump_usage_counts_.end() &&
761 jump_candidate_destination < start_pc + last_node.SequenceLength()) {
762 preserve_from = jump_candidate_destination;
763 // Check if any jump in the sequence we are preserving has a jump
764 // destination inside the optimized sequence before the current position we
765 // want to preserve. If so we have to preserve all bytecodes starting at
766 // this jump destination.
767 for (auto jump_iter = jump_edges_.lower_bound(preserve_from);
768 jump_iter != jump_edges_.end() &&
769 jump_iter->first /* jump source */ <
770 start_pc + last_node.SequenceLength();
771 ++jump_iter) {
772 int jump_destination = jump_iter->second;
773 if (jump_destination > start_pc && jump_destination < preserve_from) {
774 preserve_from = jump_destination;
775 }
776 }
777
778 // We preserve everything to the end of the sequence. This is conservative
779 // since it would be enough to preserve all bytecudes up to an unconditional
780 // jump.
781 int preserve_length = start_pc + last_node.SequenceLength() - preserve_from;
782 fixup_length += preserve_length;
783 // Jumps after the start of the preserved sequence need fixup.
784 AddJumpSourceFixup(fixup_length,
785 start_pc + last_node.SequenceLength() - preserve_length);
786 // All jump targets after the start of the optimized sequence need to be
787 // fixed relative to the length of the optimized sequence including
788 // bytecodes we preserved.
789 AddJumpDestinationFixup(fixup_length, start_pc + 1);
790 // Jumps to the sequence we preserved need absolute fixup as they could
791 // occur before or after the sequence.
792 SetJumpDestinationFixup(pc() - preserve_from, preserve_from);
793 CopyRangeToOutput(bytecode, preserve_from, preserve_length);
794 } else {
795 AddJumpDestinationFixup(fixup_length, start_pc + 1);
796 // Jumps after the end of the old sequence need fixup.
797 AddJumpSourceFixup(fixup_length, start_pc + last_node.SequenceLength());
798 }
799
800 // Delete jumps we definitely don't need anymore
801 for (int del : delete_jumps) {
802 if (del < preserve_from) {
803 jump_edges_.erase(del);
804 }
805 }
806}

847void RegExpBytecodePeephole::FixJumps() {
848 int position_fixup = 0;
849 // Next position where fixup changes.
850 auto next_source_fixup = jump_source_fixups_.lower_bound(0);
851 int next_source_fixup_offset = next_source_fixup->first;
852 int next_source_fixup_value = next_source_fixup->second;
853
854 for (auto jump_edge : jump_edges_) {
855 int jump_source = jump_edge.first;
856 int jump_destination = jump_edge.second;
857 while (jump_source >= next_source_fixup_offset) {
858 position_fixup = next_source_fixup_value;
859 ++next_source_fixup;
860 next_source_fixup_offset = next_source_fixup->first;
861 next_source_fixup_value = next_source_fixup->second;
862 }
863 jump_source += position_fixup;
864
865 FixJump(jump_source, jump_destination);
866 }
867
868 // Mapped jump edges don't need source fixups, as the position already is an
869 // offset in the new bytecode.
870 for (auto jump_edge : jump_edges_mapped_) {
871 int jump_source = jump_edge.first;
872 int jump_destination = jump_edge.second;
873
874 FixJump(jump_source, jump_destination);
875 }
876}

879 int fixed_jump_destination =
880 jump_destination +
881 (--jump_destination_fixups_.upper_bound(jump_destination))->second;
882 DCHECK_LT(fixed_jump_destination, Length());
883#ifdef DEBUG
884 // TODO(pthier): This check could be better if we track the bytecodes
885 // actually used and check if we jump to one of them.
886 uint8_t jump_bc = optimized_bytecode_buffer_[fixed_jump_destination];
887 DCHECK_GT(jump_bc, 0);
888 DCHECK_LT(jump_bc, kRegExpBytecodeCount);
889#endif
890
891 if (jump_destination != fixed_jump_destination) {
892 OverwriteValue<uint32_t>(jump_source, fixed_jump_destination);
893 }

936void RegExpBytecodePeephole::EmitArgument(int start_pc, const uint8_t* bytecode,
937 BytecodeArgumentMapping arg) {
938 int arg_pos = start_pc + arg.offset;
939 switch (arg.length) {
940 case 1:
941 DCHECK_EQ(arg.new_length, arg.length);
942 EmitValue(GetValue<uint8_t>(bytecode, arg_pos));
943 break;
944 case 2:
945 DCHECK_EQ(arg.new_length, arg.length);
946 EmitValue(GetValue<uint16_t>(bytecode, arg_pos));
947 break;
948 case 3: {
949 // Length 3 only occurs in 'packed' arguments where the lowermost byte is
950 // the current bytecode, and the remaining 3 bytes are the packed value.
951 //
952 // We load 4 bytes from position - 1 and shift out the bytecode.
953#ifdef V8_TARGET_BIG_ENDIAN
954 UNIMPLEMENTED();
955 int32_t val = 0;
956#else
957 int32_t val = GetValue<int32_t>(bytecode, arg_pos - 1) >> kBitsPerByte;
958#endif // V8_TARGET_BIG_ENDIAN
959
960 switch (arg.new_length) {
961 case 2:
962 EmitValue<uint16_t>(val);
963 break;
964 case 3: {
965 // Pack with previously emitted value.
966 auto prev_val =
967 GetValue<int32_t>(&(*optimized_bytecode_buffer_.begin()),
968 Length() - sizeof(uint32_t));
969#ifdef V8_TARGET_BIG_ENDIAN
970 UNIMPLEMENTED();
971 USE(prev_val);
972#else
973 DCHECK_EQ(prev_val & 0xFFFFFF00, 0);
974 OverwriteValue<uint32_t>(
975 pc() - sizeof(uint32_t),
976 (static_cast<uint32_t>(val) << 8) | (prev_val & 0xFF));
977#endif // V8_TARGET_BIG_ENDIAN
978 break;
979 }
980 case 4:
981 EmitValue<uint32_t>(val);
982 break;
983 }
984 break;
985 }
986 case 4:
987 DCHECK_EQ(arg.new_length, arg.length);
988 EmitValue(GetValue<uint32_t>(bytecode, arg_pos));
989 break;
990 case 8:
991 DCHECK_EQ(arg.new_length, arg.length);
992 EmitValue(GetValue<uint64_t>(bytecode, arg_pos));
993 break;
994 default:
995 CopyRangeToOutput(bytecode, arg_pos,
996 std::min(arg.length, arg.new_length));
997 if (arg.length < arg.new_length) {
998 SetRange(0x00, arg.new_length - arg.length);
999 }
1000 break;
1001 }
