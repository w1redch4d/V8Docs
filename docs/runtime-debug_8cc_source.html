35RUNTIME_FUNCTION_RETURN_PAIR(Runtime_DebugBreakOnBytecode) {
36 using interpreter::Bytecode;
37 using interpreter::Bytecodes;
38 using interpreter::OperandScale;
39
40 SealHandleScope shs(isolate);
41 DCHECK_EQ(1, args.length());
42 DirectHandle<Object> value = args.at(0);
43 HandleScope scope(isolate);
44
45 // Return value can be changed by debugger. Last set value will be used as
46 // return value.
47 ReturnValueScope result_scope(isolate->debug());
48 isolate->debug()->set_return_value(*value);
49
50 // Get the top-most JavaScript frame.
51 JavaScriptStackFrameIterator it(isolate);
52 if (isolate->debug_execution_mode() == DebugInfo::kBreakpoints) {
53 isolate->debug()->Break(it.frame(),
54 direct_handle(it.frame()->function(), isolate));
55 }
56
57 // If the user requested to restart a frame, there is no need
58 // to get the return value or check the bytecode for side-effects.
59 if (isolate->debug()->IsRestartFrameScheduled()) {
60 Tagged<Object> exception = isolate->TerminateExecution();
61 return MakePair(exception,
62 Smi::FromInt(static_cast<uint8_t>(Bytecode::kIllegal)));
63 }
64
65 // Return the handler from the original bytecode array.
66 DCHECK(it.frame()->is_interpreted());
67 InterpretedFrame* interpreted_frame =
68 reinterpret_cast<InterpretedFrame*>(it.frame());
69
70 bool side_effect_check_failed = false;
71 if (isolate->debug_execution_mode() == DebugInfo::kSideEffects) {
72 side_effect_check_failed =
73 !isolate->debug()->PerformSideEffectCheckAtBytecode(interpreted_frame);
74 }
75
76 // Make sure to only access these objects after the side effect check, as the
77 // check can allocate on failure.
78 Tagged<SharedFunctionInfo> shared = interpreted_frame->function()->shared();
79 Tagged<BytecodeArray> bytecode_array = shared->GetBytecodeArray(isolate);
80 int bytecode_offset = interpreted_frame->GetBytecodeOffset();
81 Bytecode bytecode = Bytecodes::FromByte(bytecode_array->get(bytecode_offset));
82
83 if (Bytecodes::Returns(bytecode)) {
84 // If we are returning (or suspending), reset the bytecode array on the
85 // interpreted stack frame to the non-debug variant so that the interpreter
86 // entry trampoline sees the return/suspend bytecode rather than the
87 // DebugBreak.
88 interpreted_frame->PatchBytecodeArray(bytecode_array);
89 }
90
91 // We do not have to deal with operand scale here. If the bytecode at the
92 // break is prefixed by operand scaling, we would have patched over the
93 // scaling prefix. We now simply dispatch to the handler for the prefix.
94 // We need to deserialize now to ensure we don't hit the debug break again
95 // after deserializing.
96 OperandScale operand_scale = OperandScale::kSingle;
97 isolate->interpreter()->GetBytecodeHandler(bytecode, operand_scale);
98
99 if (side_effect_check_failed) {
100 return MakePair(ReadOnlyRoots(isolate).exception(),
101 Smi::FromInt(static_cast<uint8_t>(bytecode)));
102 }
103 Tagged<Object> interrupt_object = isolate->stack_guard()->HandleInterrupts();
104 if (IsException(interrupt_object, isolate)) {
105 return MakePair(interrupt_object,
106 Smi::FromInt(static_cast<uint8_t>(bytecode)));
107 }
108 return MakePair(isolate->debug()->return_value(),
109 Smi::FromInt(static_cast<uint8_t>(bytecode)));
110}

112RUNTIME_FUNCTION(Runtime_DebugBreakAtEntry) {
113 HandleScope scope(isolate);
114 DCHECK_EQ(1, args.length());
115 DirectHandle<JSFunction> function = args.at<JSFunction>(0);
116
117 DCHECK(function->shared()->BreakAtEntry(isolate));
118
119 // Get the top-most JavaScript frame. This is the debug target function.
120 JavaScriptStackFrameIterator it(isolate);
121 DCHECK_EQ(*function, it.frame()->function());
122 // Check whether the next JS frame is closer than the last API entry.
123 // if yes, then the call to the debug target came from JavaScript. Otherwise,
124 // the call to the debug target came from API. Do not break for the latter.
125 it.Advance();
126 if (!it.done() &&
127 it.frame()->fp() < isolate->thread_local_top()->last_api_entry_) {
128 isolate->debug()->Break(it.frame(), function);
129 }
130
131 return ReadOnlyRoots(isolate).undefined_value();
132}

163template <class IteratorType>
164static Handle<ArrayList> AddIteratorInternalProperties(
165 Isolate* isolate, Handle<ArrayList> result, Handle<IteratorType> iterator) {
166 const char* kind = nullptr;
167 switch (iterator->map()->instance_type()) {
168 case JS_MAP_KEY_ITERATOR_TYPE:
169 kind = "keys";
170 break;
171 case JS_MAP_KEY_VALUE_ITERATOR_TYPE:
172 case JS_SET_KEY_VALUE_ITERATOR_TYPE:
173 kind = "entries";
174 break;
175 case JS_MAP_VALUE_ITERATOR_TYPE:
176 case JS_SET_VALUE_ITERATOR_TYPE:
177 kind = "values";
178 break;
179 default:
180 UNREACHABLE();
181 }
182
183 result = ArrayList::Add(
184 isolate, result,
185 isolate->factory()->NewStringFromAsciiChecked("[[IteratorHasMore]]"),
186 isolate->factory()->ToBoolean(iterator->HasMore()));
187 result = ArrayList::Add(
188 isolate, result,
189 isolate->factory()->NewStringFromAsciiChecked("[[IteratorIndex]]"),
190 direct_handle(iterator->index(), isolate));
191 result = ArrayList::Add(
192 isolate, result,
193 isolate->factory()->NewStringFromAsciiChecked("[[IteratorKind]]"),
194 isolate->factory()->NewStringFromAsciiChecked(kind));
195 return result;
196}

200MaybeHandle<JSArray> Runtime::GetInternalProperties(Isolate* isolate,
201 Handle<Object> object) {
202 Handle<ArrayList> result = ArrayList::New(isolate, 8 * 2);
203 if (IsJSObject(*object)) {
204 PrototypeIterator iter(isolate, Cast<JSObject>(object), kStartAtReceiver);
205 if (iter.HasAccess()) {
206 iter.Advance();
207 DirectHandle<Object> prototype = PrototypeIterator::GetCurrent(iter);
208 if (!iter.IsAtEnd() && iter.HasAccess() && IsJSGlobalProxy(*object)) {
209 // Skip JSGlobalObject as the [[Prototype]].
210 DCHECK(IsJSGlobalObject(*prototype));
211 iter.Advance();
212 prototype = PrototypeIterator::GetCurrent(iter);
213 }
214 if (!IsNull(*prototype, isolate)) {
215 result = ArrayList::Add(
216 isolate, result,
217 isolate->factory()->NewStringFromStaticChars("[[Prototype]]"),
218 prototype);
219 }
220 }
221 }
222 if (IsJSBoundFunction(*object)) {
223 auto function = Cast<JSBoundFunction>(object);
224
225 result = ArrayList::Add(
226 isolate, result,
227 isolate->factory()->NewStringFromAsciiChecked("[[TargetFunction]]"),
228 direct_handle(function->bound_target_function(), isolate));
229 result = ArrayList::Add(
230 isolate, result,
231 isolate->factory()->NewStringFromAsciiChecked("[[BoundThis]]"),
232 direct_handle(function->bound_this(), isolate));
233 result = ArrayList::Add(
234 isolate, result,
235 isolate->factory()->NewStringFromAsciiChecked("[[BoundArgs]]"),
236 isolate->factory()->NewJSArrayWithElements(
237 isolate->factory()->CopyFixedArray(
238 handle(function->bound_arguments(), isolate))));
239 } else if (IsJSMapIterator(*object)) {
240 Handle<JSMapIterator> iterator = Cast<JSMapIterator>(object);
241 result = AddIteratorInternalProperties(isolate, result, iterator);
242 } else if (IsJSSetIterator(*object)) {
243 Handle<JSSetIterator> iterator = Cast<JSSetIterator>(object);
244 result = AddIteratorInternalProperties(isolate, result, iterator);
245 } else if (IsJSGeneratorObject(*object)) {
246 auto generator = Cast<JSGeneratorObject>(object);
247
248 const char* status = "suspended";
249 if (generator->is_closed()) {
250 status = "closed";
251 } else if (generator->is_executing()) {
252 status = "running";
253 } else {
254 DCHECK(generator->is_suspended());
255 }
256
257 result = ArrayList::Add(
258 isolate, result,
259 isolate->factory()->NewStringFromAsciiChecked("[[GeneratorState]]"),
260 isolate->factory()->NewStringFromAsciiChecked(status));
261 result = ArrayList::Add(
262 isolate, result,
263 isolate->factory()->NewStringFromAsciiChecked("[[GeneratorFunction]]"),
264 direct_handle(generator->function(), isolate));
265 result = ArrayList::Add(
266 isolate, result,
267 isolate->factory()->NewStringFromAsciiChecked("[[GeneratorReceiver]]"),
268 direct_handle(generator->receiver(), isolate));
269 } else if (IsJSPromise(*object)) {
270 auto promise = Cast<JSPromise>(object);
271
272 result = ArrayList::Add(
273 isolate, result,
274 isolate->factory()->NewStringFromAsciiChecked("[[PromiseState]]"),
275 isolate->factory()->NewStringFromAsciiChecked(
276 JSPromise::Status(promise->status())));
277 result = ArrayList::Add(
278 isolate, result,
279 isolate->factory()->NewStringFromAsciiChecked("[[PromiseResult]]"),
280 promise->status() == Promise::kPending
281 ? isolate->factory()->undefined_value()
282 : handle(promise->result(), isolate));
283 } else if (IsJSProxy(*object)) {
284 auto js_proxy = Cast<JSProxy>(object);
285
286 result = ArrayList::Add(
287 isolate, result,
288 isolate->factory()->NewStringFromAsciiChecked("[[Handler]]"),
289 direct_handle(js_proxy->handler(), isolate));
290 result = ArrayList::Add(
291 isolate, result,
292 isolate->factory()->NewStringFromAsciiChecked("[[Target]]"),
293 direct_handle(js_proxy->target(), isolate));
294 result = ArrayList::Add(
295 isolate, result,
296 isolate->factory()->NewStringFromAsciiChecked("[[IsRevoked]]"),
297 isolate->factory()->ToBoolean(js_proxy->IsRevoked()));
298 } else if (IsJSPrimitiveWrapper(*object)) {
299 auto js_value = Cast<JSPrimitiveWrapper>(object);
300
301 result = ArrayList::Add(
302 isolate, result,
303 isolate->factory()->NewStringFromAsciiChecked("[[PrimitiveValue]]"),
304 direct_handle(js_value->value(), isolate));
305 } else if (IsJSWeakRef(*object)) {
306 auto js_weak_ref = Cast<JSWeakRef>(object);
307
308 result = ArrayList::Add(
309 isolate, result,
310 isolate->factory()->NewStringFromAsciiChecked("[[WeakRefTarget]]"),
311 direct_handle(js_weak_ref->target(), isolate));
312 } else if (IsJSArrayBuffer(*object)) {
313 DirectHandle<JSArrayBuffer> js_array_buffer = Cast<JSArrayBuffer>(object);
314 if (js_array_buffer->was_detached()) {
315 // Mark a detached JSArrayBuffer and such and don't even try to
316 // create views for it, since the TypedArray constructors will
317 // throw a TypeError when the underlying buffer is detached.
318 result = ArrayList::Add(
319 isolate, result,
320 isolate->factory()->NewStringFromAsciiChecked("[[IsDetached]]"),
321 isolate->factory()->true_value());
322 } else {
323 const size_t byte_length = js_array_buffer->byte_length();
324 static_assert(JSTypedArray::kMaxByteLength ==
325 JSArrayBuffer::kMaxByteLength);
326 using DataView = std::tuple<const char*, ExternalArrayType, size_t>;
327 for (auto [name, type, elem_size] :
328 {DataView{"[[Int8Array]]", kExternalInt8Array, 1},
329 DataView{"[[Uint8Array]]", kExternalUint8Array, 1},
330 DataView{"[[Int16Array]]", kExternalInt16Array, 2},
331 DataView{"[[Int32Array]]", kExternalInt32Array, 4}}) {
332 if ((byte_length % elem_size) != 0) continue;
333 size_t length = byte_length / elem_size;
334 result =
335 ArrayList::Add(isolate, result,
336 isolate->factory()->NewStringFromAsciiChecked(name),
337 isolate->factory()->NewJSTypedArray(
338 type, js_array_buffer, 0, length));
339 }
340 result =
341 ArrayList::Add(isolate, result,
342 isolate->factory()->NewStringFromAsciiChecked(
343 "[[ArrayBufferByteLength]]"),
344 isolate->factory()->NewNumberFromSize(byte_length));
345
346 auto backing_store = js_array_buffer->GetBackingStore();
347 DirectHandle<Object> array_buffer_data;
348 if (backing_store) {
349 array_buffer_data =
350 isolate->factory()->NewNumberFromUint(backing_store->id());
351 } else {
352 array_buffer_data = isolate->factory()->null_value();
353 }
354 result = ArrayList::Add(
355 isolate, result,
356 isolate->factory()->NewStringFromAsciiChecked("[[ArrayBufferData]]"),
357 array_buffer_data);
358
359 DirectHandle<Symbol> memory_symbol =
360 isolate->factory()->array_buffer_wasm_memory_symbol();
361 DirectHandle<Object> memory_object =
362 JSObject::GetDataProperty(isolate, js_array_buffer, memory_symbol);
363 if (!IsUndefined(*memory_object, isolate)) {
364 result = ArrayList::Add(isolate, result,
365 isolate->factory()->NewStringFromAsciiChecked(
366 "[[WebAssemblyMemory]]"),
367 memory_object);
368 }
369 }
370#if V8_ENABLE_WEBASSEMBLY
371 } else if (IsWasmInstanceObject(*object)) {
372 result = AddWasmInstanceObjectInternalProperties(
373 isolate, result, Cast<WasmInstanceObject>(object));
374 } else if (IsWasmModuleObject(*object)) {
375 result = AddWasmModuleObjectInternalProperties(
376 isolate, result, Cast<WasmModuleObject>(object));
377 } else if (IsWasmTableObject(*object)) {
378 result = AddWasmTableObjectInternalProperties(
379 isolate, result, Cast<WasmTableObject>(object));
380#endif // V8_ENABLE_WEBASSEMBLY
381 }
382 return isolate->factory()->NewJSArrayWithElements(
383 ArrayList::ToFixedArray(isolate, result), PACKED_ELEMENTS);
384}

386RUNTIME_FUNCTION(Runtime_GetGeneratorScopeCount) {
387 HandleScope scope(isolate);
388 DCHECK_EQ(1, args.length());
389
390 if (!IsJSGeneratorObject(args[0])) return Smi::zero();
391
392 // Check arguments.
393 Handle<JSGeneratorObject> gen = args.at<JSGeneratorObject>(0);
394
395 // Only inspect suspended generator scopes.
396 if (!gen->is_suspended()) {
397 return Smi::zero();
398 }
399
400 // Count the visible scopes.
401 int n = 0;
402 for (ScopeIterator it(isolate, gen); !it.Done(); it.Next()) {
403 n++;
404 }
405
406 return Smi::FromInt(n);
407}

409RUNTIME_FUNCTION(Runtime_GetGeneratorScopeDetails) {
410 HandleScope scope(isolate);
411 DCHECK_EQ(2, args.length());
412
413 if (!IsJSGeneratorObject(args[0])) {
414 return ReadOnlyRoots(isolate).undefined_value();
415 }
416
417 // Check arguments.
418 Handle<JSGeneratorObject> gen = args.at<JSGeneratorObject>(0);
419 int index = NumberToInt32(args[1]);
420
421 // Only inspect suspended generator scopes.
422 if (!gen->is_suspended()) {
423 return ReadOnlyRoots(isolate).undefined_value();
424 }
425
426 // Find the requested scope.
427 int n = 0;
428 ScopeIterator it(isolate, gen);
429 for (; !it.Done() && n < index; it.Next()) {
430 n++;
431 }
432 if (it.Done()) {
433 return ReadOnlyRoots(isolate).undefined_value();
434 }
435
436 return *it.MaterializeScopeDetails();
437}

440 Handle<String> variable_name,
441 DirectHandle<Object> new_value) {

553int ScriptLinePosition(DirectHandle<Script> script, int line) {
554 if (line < 0) return -1;
555
556#if V8_ENABLE_WEBASSEMBLY
557 if (script->type() == Script::Type::kWasm) {
558 // Wasm positions are relative to the start of the module.
559 return 0;
560 }
561#endif // V8_ENABLE_WEBASSEMBLY
562
563 Script::InitLineEnds(script->GetIsolate(), script);
564
565 Tagged<FixedArray> line_ends_array = Cast<FixedArray>(script->line_ends());
566 const int line_count = line_ends_array->length();
567 DCHECK_LT(0, line_count);
568
569 if (line == 0) return 0;
570 // If line == line_count, we return the first position beyond the last line.
571 if (line > line_count) return -1;
572 return Smi::ToInt(line_ends_array->get(line - 1)) + 1;
573}

575int ScriptLinePositionWithOffset(DirectHandle<Script> script, int line,
576 int offset) {
577 if (line < 0 || offset < 0) return -1;
578
579 if (line == 0 || offset == 0)
580 return ScriptLinePosition(script, line) + offset;
581
582 Script::PositionInfo info;
583 if (!Script::GetPositionInfo(script, offset, &info,
584 Script::OffsetFlag::kNoOffset)) {
585 return -1;
586 }
587
588 const int total_line = info.line + line;
589 return ScriptLinePosition(script, total_line);
590}

592Handle<Object> GetJSPositionInfo(DirectHandle<Script> script, int position,
593 Script::OffsetFlag offset_flag,
594 Isolate* isolate) {
595 Script::PositionInfo info;
596 if (!Script::GetPositionInfo(script, position, &info, offset_flag)) {
597 return isolate->factory()->null_value();
598 }
599
600#if V8_ENABLE_WEBASSEMBLY
601 const bool is_wasm_script = script->type() == Script::Type::kWasm;
602#else
603 const bool is_wasm_script = false;
604#endif // V8_ENABLE_WEBASSEMBLY
605 DirectHandle<String> sourceText =
606 is_wasm_script ? isolate->factory()->empty_string()
607 : isolate->factory()->NewSubString(
608 handle(Cast<String>(script->source()), isolate),
609 info.line_start, info.line_end);
610
611 Handle<JSObject> jsinfo =
612 isolate->factory()->NewJSObject(isolate->object_function());
613
614 JSObject::AddProperty(isolate, jsinfo, isolate->factory()->script_string(),
615 script, NONE);
616 JSObject::AddProperty(isolate, jsinfo, isolate->factory()->position_string(),
617 direct_handle(Smi::FromInt(position), isolate), NONE);
618 JSObject::AddProperty(isolate, jsinfo, isolate->factory()->line_string(),
619 direct_handle(Smi::FromInt(info.line), isolate), NONE);
620 JSObject::AddProperty(isolate, jsinfo, isolate->factory()->column_string(),
621 direct_handle(Smi::FromInt(info.column), isolate),
622 NONE);
623 JSObject::AddProperty(isolate, jsinfo,
624 isolate->factory()->sourceText_string(), sourceText,
625 NONE);
626
627 return jsinfo;
628}

630Handle<Object> ScriptLocationFromLine(Isolate* isolate,
631 DirectHandle<Script> script,
632 DirectHandle<Object> opt_line,
633 DirectHandle<Object> opt_column,
634 int32_t offset) {
635 // Line and column are possibly undefined and we need to handle these cases,
636 // additionally subtracting corresponding offsets.
637
638 int32_t line = 0;
639 if (!IsNullOrUndefined(*opt_line, isolate)) {
640 CHECK(IsNumber(*opt_line));
641 line = NumberToInt32(*opt_line) - script->line_offset();
642 }
643
644 int32_t column = 0;
645 if (!IsNullOrUndefined(*opt_column, isolate)) {
646 CHECK(IsNumber(*opt_column));
647 column = NumberToInt32(*opt_column);
648 if (line == 0) column -= script->column_offset();
649 }
650
651 int line_position = ScriptLinePositionWithOffset(script, line, offset);
652 if (line_position < 0 || column < 0) return isolate->factory()->null_value();
653
654 return GetJSPositionInfo(script, line_position + column,
655 Script::OffsetFlag::kNoOffset, isolate);
656}

741RUNTIME_FUNCTION(Runtime_DebugCollectCoverage) {
742 HandleScope scope(isolate);
743 DCHECK_EQ(0, args.length());
744 // Collect coverage data.
745 std::unique_ptr<Coverage> coverage;
746 if (isolate->is_best_effort_code_coverage()) {
747 coverage = Coverage::CollectBestEffort(isolate);
748 } else {
749 coverage = Coverage::CollectPrecise(isolate);
750 }
751 Factory* factory = isolate->factory();
752 // Turn the returned data structure into JavaScript.
753 // Create an array of scripts.
754 int num_scripts = static_cast<int>(coverage->size());
755 // Prepare property keys.
756 DirectHandle<FixedArray> scripts_array = factory->NewFixedArray(num_scripts);
757 DirectHandle<String> script_string = factory->script_string();
758 for (int i = 0; i < num_scripts; i++) {
759 const auto& script_data = coverage->at(i);
760 HandleScope inner_scope(isolate);
761
762 std::vector<CoverageBlock> ranges;
763 int num_functions = static_cast<int>(script_data.functions.size());
764 for (int j = 0; j < num_functions; j++) {
765 const auto& function_data = script_data.functions[j];
766 ranges.emplace_back(function_data.start, function_data.end,
767 function_data.count);
768 for (size_t k = 0; k < function_data.blocks.size(); k++) {
769 const auto& block_data = function_data.blocks[k];
770 ranges.emplace_back(block_data.start, block_data.end, block_data.count);
771 }
772 }
773
774 int num_ranges = static_cast<int>(ranges.size());
775 DirectHandle<FixedArray> ranges_array = factory->NewFixedArray(num_ranges);
776 for (int j = 0; j < num_ranges; j++) {
777 DirectHandle<JSObject> range_object = MakeRangeObject(isolate, ranges[j]);
778 ranges_array->set(j, *range_object);
779 }
780
781 DirectHandle<JSArray> script_obj =
782 factory->NewJSArrayWithElements(ranges_array, PACKED_ELEMENTS);
783 JSObject::AddProperty(isolate, script_obj, script_string,
784 direct_handle(script_data.script->source(), isolate),
785 NONE);
786 scripts_array->set(i, *script_obj);
787 }
788 return *factory->NewJSArrayWithElements(scripts_array, PACKED_ELEMENTS);
789}
