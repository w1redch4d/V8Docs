17V8_INLINE Address* LocalHandleScope::GetHandle(LocalHeap* local_heap,
18 Address value) {
19 DCHECK(local_heap->IsRunning());
20 if (local_heap->is_main_thread())
21 return LocalHandleScope::GetMainThreadHandle(local_heap, value);
22
23 LocalHandles* handles = local_heap->handles();
24 Address* result = handles->scope_.next;
25 if (result == handles->scope_.limit) {
26 result = handles->AddBlock();
27 }
28 DCHECK_LT(result, handles->scope_.limit);
29 handles->scope_.next++;
30 *result = value;
31 return result;
32}

62template <typename T, template <typename> typename HandleType>
63 requires(std::is_convertible_v<HandleType<T>, DirectHandle<T>>)
64HandleType<T> LocalHandleScope::CloseAndEscape(HandleType<T> handle_value) {
65 HandleScopeData* current;
66 Tagged<T> value = *handle_value;
67 // Throw away all handles in the current scope.
68 if (local_heap_->is_main_thread()) {
69#ifdef V8_ENABLE_CHECKS
70 VerifyMainThreadScope();
71#endif
72 current = local_heap_->heap()->isolate()->handle_scope_data();
73 CloseMainThreadScope(local_heap_, prev_next_, prev_limit_);
74 } else {
75 current = &local_heap_->handles()->scope_;
76 CloseScope(local_heap_, prev_next_, prev_limit_);
77 }
78 // Allocate one handle in the parent scope.
79 DCHECK(current->level > current->sealed_level);
80 HandleType<T> result(value, local_heap_);
81 // Reinitialize the current scope (so that it's ready
82 // to be used or closed again).
83 prev_next_ = current->next;
84 prev_limit_ = current->limit;
85 current->level++;
86 return result;
87}

89void LocalHandleScope::CloseScope(LocalHeap* local_heap, Address* prev_next,
90 Address* prev_limit) {
91 LocalHandles* handles = local_heap->handles();
92 Address* old_limit = handles->scope_.limit;
93
94 handles->scope_.next = prev_next;
95 handles->scope_.limit = prev_limit;
96 handles->scope_.level--;
97
98 if (old_limit != handles->scope_.limit) {
99 handles->RemoveUnusedBlocks();
100 old_limit = handles->scope_.limit;
101 }
102
103#ifdef ENABLE_LOCAL_HANDLE_ZAPPING
104 LocalHandles::ZapRange(handles->scope_.next, old_limit);
105#endif
106
107 MSAN_ALLOCATED_UNINITIALIZED_MEMORY(
108 handles->scope_.next,
109 static_cast<size_t>(reinterpret_cast<Address>(old_limit) -
110 reinterpret_cast<Address>(handles->scope_.next)));
111}
