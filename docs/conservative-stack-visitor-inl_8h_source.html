64template <typename ConcreteVisitor>
65Address ConservativeStackVisitorBase<ConcreteVisitor>::FindBasePtr(
66 Address maybe_inner_ptr, PtrComprCageBase cage_base) const {
67#ifdef V8_COMPRESS_POINTERS
68 DCHECK(IsInterestingCage(cage_base));
69#endif // V8_COMPRESS_POINTERS
70 // Check if the pointer is contained by a normal or large page owned by this
71 // heap. Bail out if it is not.
72 const MemoryChunk* chunk =
73 allocator_->LookupChunkContainingAddress(maybe_inner_ptr);
74 if (chunk == nullptr) {
75 return kNullAddress;
76 }
77 const MemoryChunkMetadata* chunk_metadata = chunk->Metadata();
78 DCHECK(chunk_metadata->Contains(maybe_inner_ptr));
79
80 if (!ConcreteVisitor::FilterPage(chunk)) {
81 return kNullAddress;
82 }
83
84 // If it is contained in a large page, we want to mark the only object on it.
85 if (chunk->IsLargePage()) {
86 // This could be simplified if we could guarantee that there are no free
87 // space or filler objects in large pages. A few cctests violate this now.
88 Tagged<HeapObject> obj(
89 static_cast<const LargePageMetadata*>(chunk_metadata)->GetObject());
90 MapWord map_word = obj->map_word(cage_base, kRelaxedLoad);
91 return (!ConcreteVisitor::FilterLargeObject(obj, map_word) ||
92 InstanceTypeChecker::IsFreeSpaceOrFiller(map_word.ToMap()))
93 ? kNullAddress
94 : obj.address();
95 }
96
97 // Otherwise, we have a pointer inside a normal page.
98 const PageMetadata* page = static_cast<const PageMetadata*>(chunk_metadata);
99 // Try to find the address of a previous valid object on this page.
100 Address base_ptr =
101 MarkingBitmap::FindPreviousValidObject(page, maybe_inner_ptr);
102 // Iterate through the objects in the page forwards, until we find the object
103 // containing maybe_inner_ptr.
104 DCHECK_LE(base_ptr, maybe_inner_ptr);
105 while (true) {
106 Tagged<HeapObject> obj(HeapObject::FromAddress(base_ptr));
107 MapWord map_word = obj->map_word(cage_base, kRelaxedLoad);
108 if (!ConcreteVisitor::FilterNormalObject(obj, map_word)) {
109 return kNullAddress;
110 }
111 const int size = obj->SizeFromMap(map_word.ToMap());
112 DCHECK_LT(0, size);
113 if (maybe_inner_ptr < base_ptr + size) {
114 ConcreteVisitor::HandleObjectFound(obj, size);
115 return IsFreeSpaceOrFiller(obj, cage_base) ? kNullAddress : base_ptr;
116 }
117 base_ptr += ALIGN_TO_ALLOCATION_ALIGNMENT(size);
118 DCHECK_LT(base_ptr, page->area_end());
119 }
120}

122template <typename ConcreteVisitor>
123void ConservativeStackVisitorBase<ConcreteVisitor>::VisitPointer(
124 const void* pointer) {
125 auto address = reinterpret_cast<Address>(const_cast<void*>(pointer));
126 VisitConservativelyIfPointer(address);
127#ifdef V8_COMPRESS_POINTERS
128 V8HeapCompressionScheme::ProcessIntermediatePointers(
129 cage_base_, address,
130 [this](Address ptr) { VisitConservativelyIfPointer(ptr, cage_base_); });
131#ifdef V8_EXTERNAL_CODE_SPACE
132 ExternalCodeCompressionScheme::ProcessIntermediatePointers(
133 code_cage_base_, address, [this](Address ptr) {
134 VisitConservativelyIfPointer(ptr, code_cage_base_);
135 });
136#endif // V8_EXTERNAL_CODE_SPACE
137#ifdef V8_ENABLE_SANDBOX
138 TrustedSpaceCompressionScheme::ProcessIntermediatePointers(
139 trusted_cage_base_, address, [this](Address ptr) {
140 VisitConservativelyIfPointer(ptr, trusted_cage_base_);
141 });
142#endif // V8_ENABLE_SANDBOX
143#endif // V8_COMPRESS_POINTERS
144}

146template <typename ConcreteVisitor>
147void ConservativeStackVisitorBase<
148 ConcreteVisitor>::VisitConservativelyIfPointer(Address address) {
149#ifdef V8_COMPRESS_POINTERS
150 // Only proceed if the address falls in one of the interesting cages,
151 // otherwise bail out.
152 if (V8HeapCompressionScheme::GetPtrComprCageBaseAddress(address) ==
153 cage_base_.address()) {
154 VisitConservativelyIfPointer(address, cage_base_);
155#ifdef V8_EXTERNAL_CODE_SPACE
156 } else if (code_address_region_.contains(address)) {
157 VisitConservativelyIfPointer(address, code_cage_base_);
158#endif // V8_EXTERNAL_CODE_SPACE
159 }
160#else // !V8_COMPRESS_POINTERS
161 VisitConservativelyIfPointer(address, cage_base_);
162#endif // V8_COMPRESS_POINTERS
163}

165template <typename ConcreteVisitor>
166void ConservativeStackVisitorBase<
167 ConcreteVisitor>::VisitConservativelyIfPointer(Address address,
168 PtrComprCageBase cage_base) {
169 // Bail out immediately if the pointer is not in the space managed by the
170 // allocator.
171 if (allocator_->IsOutsideAllocatedSpace(address)) {
172 DCHECK_EQ(nullptr, allocator_->LookupChunkContainingAddress(address));
173 return;
174 }
175 // Proceed with inner-pointer resolution.
176 Address base_ptr = FindBasePtr(address, cage_base);
177 if (base_ptr == kNullAddress) {
178 return;
179 }
180 Tagged<HeapObject> obj = HeapObject::FromAddress(base_ptr);
181 Tagged<Object> root = obj;
182 DCHECK_NOT_NULL(root_visitor_);
183 root_visitor_->VisitRootPointer(Root::kStackRoots, nullptr,
184 FullObjectSlot(&root));
185 // Check that the root visitor did not modify the root slot.
186 DCHECK_EQ(root, obj);
187}
