41inline void JSDisposableStackBase::Add(
42 Isolate* isolate, DirectHandle<JSDisposableStackBase> disposable_stack,
43 DirectHandle<Object> value, DirectHandle<Object> method,
44 DisposeMethodCallType type, DisposeMethodHint hint) {
45 DCHECK(!IsUndefined(disposable_stack->stack()));
46 int length = disposable_stack->length();
47 int stack_type =
48 DisposeCallTypeBit::encode(type) | DisposeHintBit::encode(hint);
49 DirectHandle<Smi> stack_type_handle(Smi::FromInt(stack_type), isolate);
50
51 Handle<FixedArray> array(disposable_stack->stack(), isolate);
52 array = FixedArray::SetAndGrow(isolate, array, length++, value);
53 array = FixedArray::SetAndGrow(isolate, array, length++, method);
54 array = FixedArray::SetAndGrow(isolate, array, length++, stack_type_handle);
55
56 disposable_stack->set_length(length);
57 disposable_stack->set_stack(*array);
58}

62inline MaybeHandle<Object> JSDisposableStackBase::CheckValueAndGetDisposeMethod(
63 Isolate* isolate, DirectHandle<JSAny> value, DisposeMethodHint hint) {
64 Handle<Object> method;
65 if (hint == DisposeMethodHint::kSyncDispose) {
66 // 1. If method is not present, then
67 // a. If V is either null or undefined, then
68 // i. Set V to undefined.
69 // ii. Set method to undefined.
70 // We has already returned from the caller if V is null or undefined, when
71 // hint is `kSyncDispose`.
72 DCHECK(!IsNullOrUndefined(*value));
73
74 // b. Else,
75 // i. If V is not an Object, throw a TypeError exception.
76 if (!IsJSReceiver(*value)) {
77 THROW_NEW_ERROR(isolate,
78 NewTypeError(MessageTemplate::kExpectAnObjectWithUsing));
79 }
80
81 // ii. Set method to ? GetDisposeMethod(V, hint).
82 ASSIGN_RETURN_ON_EXCEPTION(
83 isolate, method,
84 Object::GetProperty(isolate, value,
85 isolate->factory()->dispose_symbol()));
86 // (GetMethod)3. If IsCallable(func) is false, throw a TypeError
87 // exception.
88 if (!IsJSFunction(*method)) {
89 THROW_NEW_ERROR(isolate,
90 NewTypeError(MessageTemplate::kNotCallable,
91 isolate->factory()->dispose_symbol()));
92 }
93
94 // iii. If method is undefined, throw a TypeError exception.
95 // It is already checked in step ii.
96
97 } else if (hint == DisposeMethodHint::kAsyncDispose) {
98 // 1. If method is not present, then
99 // a. If V is either null or undefined, then
100 // i. Set V to undefined.
101 // ii. Set method to undefined.
102 if (IsNullOrUndefined(*value)) {
103 return isolate->factory()->undefined_value();
104 }
105
106 // b. Else,
107 // i. If V is not an Object, throw a TypeError exception.
108 if (!IsJSReceiver(*value)) {
109 THROW_NEW_ERROR(isolate,
110 NewTypeError(MessageTemplate::kExpectAnObjectWithUsing));
111 }
112 // https://tc39.es/proposal-explicit-resource-management/#sec-getdisposemethod
113 // 1. If hint is async-dispose, then
114 // a. Let method be ? GetMethod(V, @@asyncDispose).
115 ASSIGN_RETURN_ON_EXCEPTION(
116 isolate, method,
117 Object::GetProperty(isolate, value,
118 isolate->factory()->async_dispose_symbol()));
119 // b. If method is undefined, then
120 if (IsUndefined(*method)) {
121 // i. Set method to ? GetMethod(V, @@dispose).
122 ASSIGN_RETURN_ON_EXCEPTION(
123 isolate, method,
124 Object::GetProperty(isolate, value,
125 isolate->factory()->dispose_symbol()));
126 // (GetMethod)3. If IsCallable(func) is false, throw a TypeError
127 // exception.
128 if (!IsJSFunction(*method)) {
129 THROW_NEW_ERROR(isolate,
130 NewTypeError(MessageTemplate::kNotCallable,
131 isolate->factory()->dispose_symbol()));
132 }
133 // ii. If method is not undefined, then
134 if (!IsUndefined(*method)) {
135 // 1. Let closure be a new Abstract Closure with no parameters that
136 // captures method and performs the following steps when called:
137 // a. Let O be the this value.
138 // b. Let promiseCapability be ! NewPromiseCapability(%Promise%).
139 // c. Let result be Completion(Call(method, O)).
140 // d. IfAbruptRejectPromise(result, promiseCapability).
141 // e. Perform ? Call(promiseCapability.[[Resolve]], undefined, Â«
142 // undefined Â»).
143 // f. Return promiseCapability.[[Promise]].
144 // 2. NOTE: This function is not observable to user code. It is
145 // used to ensure that a Promise returned from a synchronous
146 // @@dispose method will not be awaited and that any exception
147 // thrown will not be thrown synchronously.
148 // 3. Return CreateBuiltinFunction(closure, 0, "", Â« Â»).
149
150 // (TODO:rezvan): Add `kAsyncFromSyncDispose` to the `DisposeMethodHint`
151 // enum and remove the following allocation of adapter clousre.
152 DirectHandle<Context> async_dispose_from_sync_dispose_context =
153 isolate->factory()->NewBuiltinContext(
154 isolate->native_context(),
155 static_cast<int>(
156 AsyncDisposeFromSyncDisposeContextSlots::kLength));
157 async_dispose_from_sync_dispose_context->set(
158 static_cast<int>(AsyncDisposeFromSyncDisposeContextSlots::kMethod),
159 *method);
160
161 method =
162 Factory::JSFunctionBuilder{
163 isolate,
164 isolate->factory()
165 ->async_dispose_from_sync_dispose_shared_fun(),
166 async_dispose_from_sync_dispose_context}
167 .Build();
168 }
169 }
170 // (GetMethod)3. If IsCallable(func) is false, throw a TypeError
171 // exception.
172 if (!IsJSFunction(*method)) {
173 THROW_NEW_ERROR(isolate,
174 NewTypeError(MessageTemplate::kNotCallable,
175 isolate->factory()->async_dispose_symbol()));
176 }
177 }
178 return method;
179}

181inline void JSDisposableStackBase::HandleErrorInDisposal(
182 Isolate* isolate, DirectHandle<JSDisposableStackBase> disposable_stack,
183 Handle<Object> current_error, DirectHandle<Object> current_error_message) {
184 DCHECK(isolate->is_catchable_by_javascript(*current_error));
185
186 Handle<Object> maybe_error(disposable_stack->error(), isolate);
187
188 // i. If completion is a throw completion, then
189 if (!IsUninitialized(*maybe_error)) {
190 // 1. Set result to result.[[Value]].
191 // 2. Let suppressed be completion.[[Value]].
192 // 3. Let error be a newly created SuppressedError object.
193 // 4. Perform CreateNonEnumerableDataPropertyOrThrow(error, "error",
194 // result).
195 // 5. Perform CreateNonEnumerableDataPropertyOrThrow(error,
196 // "suppressed", suppressed).
197 // 6. Set completion to ThrowCompletion(error).
198 maybe_error = isolate->factory()->NewSuppressedErrorAtDisposal(
199 isolate, current_error, maybe_error);
200 disposable_stack->set_suppressed_error_created(true);
201
202 } else {
203 // ii. Else,
204 // 1. Set completion to result.
205 maybe_error = current_error;
206 }
207
208 disposable_stack->set_error(*maybe_error);
209 disposable_stack->set_error_message(*current_error_message);
210}
