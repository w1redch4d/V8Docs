27GCTracer::Scope::Scope(GCTracer* tracer, ScopeId scope, ThreadKind thread_kind)
28 : tracer_(tracer),
29 scope_(scope),
30 thread_kind_(thread_kind),
31 start_time_(base::TimeTicks::Now()) {
32 DCHECK_IMPLIES(thread_kind_ == ThreadKind::kMain,
33 tracer_->heap_->IsMainThread());
34
35#ifdef V8_RUNTIME_CALL_STATS
36 if (V8_LIKELY(!TracingFlags::is_runtime_stats_enabled())) return;
37 if (thread_kind_ == ThreadKind::kMain) {
38 runtime_stats_ = tracer_->heap_->isolate_->counters()->runtime_call_stats();
39 runtime_stats_->Enter(&timer_, GCTracer::RCSCounterFromScope(scope));
40 } else {
41 runtime_call_stats_scope_.emplace(
42 tracer->worker_thread_runtime_call_stats());
43 runtime_stats_ = runtime_call_stats_scope_->Get();
44 runtime_stats_->Enter(&timer_, GCTracer::RCSCounterFromScope(scope));
45 }
46#endif // defined(V8_RUNTIME_CALL_STATS)
47}

49GCTracer::Scope::~Scope() {
50 const base::TimeDelta duration = base::TimeTicks::Now() - start_time_;
51 tracer_->AddScopeSample(scope_, duration);
52
53 if (thread_kind_ == ThreadKind::kMain) {
54 if (scope_ == ScopeId::MC_INCREMENTAL ||
55 scope_ == ScopeId::MC_INCREMENTAL_START ||
56 scope_ == ScopeId::MC_INCREMENTAL_FINALIZE) {
57 auto* long_task_stats =
58 tracer_->heap_->isolate_->GetCurrentLongTaskStats();
59 long_task_stats->gc_full_incremental_wall_clock_duration_us +=
60 duration.InMicroseconds();
61 }
62 }
63
64#ifdef V8_RUNTIME_CALL_STATS
65 if (V8_LIKELY(runtime_stats_ == nullptr)) return;
66 runtime_stats_->Leave(&timer_);
67#endif // defined(V8_RUNTIME_CALL_STATS)
68}

121void GCTracer::AddScopeSample(Scope::ScopeId id, base::TimeDelta duration) {
122 if (Scope::FIRST_INCREMENTAL_SCOPE <= id &&
123 id <= Scope::LAST_INCREMENTAL_SCOPE) {
124 incremental_scopes_[Scope::IncrementalOffset(id)] += duration;
125 } else if (Scope::FIRST_BACKGROUND_SCOPE <= id &&
126 id <= Scope::LAST_BACKGROUND_SCOPE) {
127 base::SpinningMutexGuard guard(&background_scopes_mutex_);
128 background_scopes_[id] += duration;
129 } else {
130 DCHECK_GT(Scope::NUMBER_OF_SCOPES, id);
131 current_.scopes[id] += duration;
132 }
133}
