154 *this = std::move(other);

157 bits_ = other.bits_;

159 return *this;
160 }
161 static CompileTimeImports FromSerialized(
162 CompileTimeImportFlags::StorageType flags,
163 base::Vector<const char> constants_module) {
164 CompileTimeImports result;
165 result.bits_ = CompileTimeImportFlags::FromIntegral(flags);
166 result.constants_module_.assign(constants_module.begin(),
167 constants_module.end());
168 return result;
169 }
170
171 bool empty() const { return bits_.empty(); }
172 bool has_string_constants(base::Vector<const uint8_t> name) const {
173 return bits_.contains(CompileTimeImport::kStringConstants) &&
174 constants_module_.size() == name.size() &&
175 std::equal(name.begin(), name.end(), constants_module_.begin());
176 }
177 bool contains(CompileTimeImport imp) const { return bits_.contains(imp); }
178
179 int compare(const CompileTimeImports& other) const {
180 if (bits_.ToIntegral() < other.bits_.ToIntegral()) return -1;
181 if (bits_.ToIntegral() > other.bits_.ToIntegral()) return 1;
182 return constants_module_.compare(other.constants_module_);
183 }
184
185 void Add(CompileTimeImport imp) { bits_.Add(imp); }
186
187 std::string& constants_module() { return constants_module_; }
188 const std::string& constants_module() const { return constants_module_; }
189
190 CompileTimeImportFlags flags() const { return bits_; }
191
192 private:
193 CompileTimeImportFlags bits_;
194 std::string constants_module_;
195};
