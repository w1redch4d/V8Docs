21uint32_t StringHasher::AddCharacterCore(uint32_t running_hash, uint16_t c) {
22 running_hash += c;
23 running_hash += (running_hash << 10);
24 running_hash ^= (running_hash >> 6);
25 return running_hash;
26}

28uint32_t StringHasher::GetHashCore(uint32_t running_hash) {
29 running_hash += (running_hash << 3);
30 running_hash ^= (running_hash >> 11);
31 running_hash += (running_hash << 15);
32 int32_t hash = static_cast<int32_t>(running_hash & String::HashBits::kMax);
33 // Ensure that the hash is kZeroHash, if the computed value is 0.
34 int32_t mask = (hash - 1) >> 31;
35 running_hash |= (kZeroHash & mask);
36 return running_hash;
37}

39uint32_t StringHasher::GetTrivialHash(uint32_t length) {
40 DCHECK_GT(length, String::kMaxHashCalcLength);
41 // The hash of a large string is simply computed from the length.
42 // Ensure that the max length is small enough to be encoded without losing
43 // information.
44 static_assert(String::kMaxLength <= String::HashBits::kMax);
45 uint32_t hash = length;
46 return String::CreateHashFieldValue(hash, String::HashFieldType::kHash);
47}

49template <typename char_t>
50uint32_t StringHasher::HashSequentialString(const char_t* chars_raw,
51 uint32_t length, uint64_t seed) {
52 static_assert(std::is_integral<char_t>::value);
53 static_assert(sizeof(char_t) <= 2);
54 using uchar = typename std::make_unsigned<char_t>::type;
55 const uchar* chars = reinterpret_cast<const uchar*>(chars_raw);
56 DCHECK_IMPLIES(length > 0, chars != nullptr);
57 if (length >= 1) {
58 if (IsDecimalDigit(chars[0]) && (length == 1 || chars[0] != '0')) {
59 if (length <= String::kMaxArrayIndexSize) {
60 // Possible array index; try to compute the array index hash.
61 uint32_t index = chars[0] - '0';
62 uint32_t i = 1;
63 do {
64 if (i == length) {
65 return MakeArrayIndexHash(index, length);
66 }
67 } while (TryAddArrayIndexChar(&index, chars[i++]));
68 }
69 // The following block wouldn't do anything on 32-bit platforms,
70 // because kMaxArrayIndexSize == kMaxIntegerIndexSize there, and
71 // if we wanted to compile it everywhere, then {index_big} would
72 // have to be a {size_t}, which the Mac compiler doesn't like to
73 // implicitly cast to uint64_t for the {TryAddIndexChar} call.
74#if V8_HOST_ARCH_64_BIT
75 // No "else" here: if the block above was entered and fell through,
76 // we'll have to take this branch.
77 if (length <= String::kMaxIntegerIndexSize) {
78 // Not an array index, but it could still be an integer index.
79 // Perform a regular hash computation, and additionally check
80 // if there are non-digit characters.
81 String::HashFieldType type = String::HashFieldType::kIntegerIndex;
82 uint32_t running_hash = static_cast<uint32_t>(seed);
83 uint64_t index_big = 0;
84 const uchar* end = &chars[length];
85 while (chars != end) {
86 if (type == String::HashFieldType::kIntegerIndex &&
87 !TryAddIntegerIndexChar(&index_big, *chars)) {
88 type = String::HashFieldType::kHash;
89 }
90 running_hash = AddCharacterCore(running_hash, *chars++);
91 }
92 uint32_t hash =
93 String::CreateHashFieldValue(GetHashCore(running_hash), type);
94 if (Name::ContainsCachedArrayIndex(hash)) {
95 // The hash accidentally looks like a cached index. Fix that by
96 // setting a bit that looks like a longer-than-cacheable string
97 // length.
98 hash |= (String::kMaxCachedArrayIndexLength + 1)
99 << String::ArrayIndexLengthBits::kShift;
100 }
101 DCHECK(!Name::ContainsCachedArrayIndex(hash));
102 return hash;
103 }
104#endif
105 }
106 // No "else" here: if the first character was a decimal digit, we might
107 // still have to take this branch.
108 if (length > String::kMaxHashCalcLength) {
109 return GetTrivialHash(length);
110 }
111 }
112
113 // Non-index hash.
114 uint32_t running_hash = static_cast<uint32_t>(seed);
115 const uchar* end = &chars[length];
116 while (chars != end) {
117 running_hash = AddCharacterCore(running_hash, *chars++);
118 }
119
120 return String::CreateHashFieldValue(GetHashCore(running_hash),
121 String::HashFieldType::kHash);
122}
