216 class ExternalMemoryAccounting {
217 public:
218 static constexpr size_t kExternalAllocationLimitForInterrupt = 128 * KB;
219
220 uint64_t total() const { return total_.load(std::memory_order_relaxed); }
221 uint64_t limit_for_interrupt() const {
222 return limit_for_interrupt_.load(std::memory_order_relaxed);
223 }
224 uint64_t soft_limit() const {
225 return low_since_mark_compact() + kExternalAllocationSoftLimit;
226 }
227 uint64_t low_since_mark_compact() const {
228 return low_since_mark_compact_.load(std::memory_order_relaxed);
229 }
230
231 uint64_t UpdateAmount(int64_t delta) {
232 const uint64_t amount_before =
233 total_.fetch_add(delta, std::memory_order_relaxed);
234 CHECK_GE(static_cast<int64_t>(amount_before), -delta);
235 return amount_before + delta;
236 }
237
238 void UpdateLimitForInterrupt(uint64_t amount) {
239 set_limit_for_interrupt(amount + kExternalAllocationLimitForInterrupt);
240 }
241
242 void UpdateLowSinceMarkCompact(uint64_t amount) {
243 set_low_since_mark_compact(amount);
244 UpdateLimitForInterrupt(amount);
245 }
246
247 uint64_t AllocatedSinceMarkCompact() const {
248 uint64_t total_bytes = total();
249 uint64_t low_since_mark_compact_bytes = low_since_mark_compact();
250
251 if (total_bytes <= low_since_mark_compact_bytes) {
252 return 0;
253 }
254 return total_bytes - low_since_mark_compact_bytes;
255 }
256
257 private:
258 void set_total(uint64_t value) {
259 total_.store(value, std::memory_order_relaxed);
260 }
261
262 void set_limit_for_interrupt(uint64_t value) {
263 limit_for_interrupt_.store(value, std::memory_order_relaxed);
264 }
265
266 void set_low_since_mark_compact(uint64_t value) {
267 low_since_mark_compact_.store(value, std::memory_order_relaxed);
268 }
269
270 // The amount of external memory registered through the API.
271 std::atomic<uint64_t> total_{0};
272
273 // The limit when to trigger memory pressure from the API.
274 std::atomic<uint64_t> limit_for_interrupt_{
275 kExternalAllocationLimitForInterrupt};
276
277 // Caches the amount of external memory registered at the last MC.
278 std::atomic<uint64_t> low_since_mark_compact_{0};
279 };
