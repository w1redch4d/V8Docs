27MaybeHandle<JSSegmenter> JSSegmenter::New(Isolate* isolate,
28 DirectHandle<Map> map,
29 DirectHandle<Object> locales,
30 DirectHandle<Object> input_options) {
31 // 4. Let requestedLocales be ? CanonicalizeLocaleList(locales).
32 Maybe<std::vector<std::string>> maybe_requested_locales =
33 Intl::CanonicalizeLocaleList(isolate, locales);
34 MAYBE_RETURN(maybe_requested_locales, Handle<JSSegmenter>());
35 std::vector<std::string> requested_locales =
36 maybe_requested_locales.FromJust();
37
38 DirectHandle<JSReceiver> options;
39 const char* service = "Intl.Segmenter";
40 // 5. Let options be GetOptionsObject(_options_).
41 ASSIGN_RETURN_ON_EXCEPTION(isolate, options,
42 GetOptionsObject(isolate, input_options, service));
43
44 // 7. Let opt be a new Record.
45 // 8. Let matcher be ? GetOption(options, "localeMatcher", "string",
46 // « "lookup", "best fit" », "best fit").
47 // 9. Set opt.[[localeMatcher]] to matcher.
48 Maybe<Intl::MatcherOption> maybe_locale_matcher =
49 Intl::GetLocaleMatcher(isolate, options, service);
50 MAYBE_RETURN(maybe_locale_matcher, MaybeHandle<JSSegmenter>());
51 Intl::MatcherOption matcher = maybe_locale_matcher.FromJust();
52
53 // 10. Let localeData be %Segmenter%.[[LocaleData]].
54
55 // 11. Let r be ResolveLocale(%Segmenter%.[[AvailableLocales]],
56 // requestedLocales, opt, %Segmenter%.[[RelevantExtensionKeys]]).
57 Maybe<Intl::ResolvedLocale> maybe_resolve_locale =
58 Intl::ResolveLocale(isolate, JSSegmenter::GetAvailableLocales(),
59 requested_locales, matcher, {});
60 if (maybe_resolve_locale.IsNothing()) {
61 THROW_NEW_ERROR(isolate, NewRangeError(MessageTemplate::kIcuError));
62 }
63 Intl::ResolvedLocale r = maybe_resolve_locale.FromJust();
64
65 // 12. Set segmenter.[[Locale]] to the value of r.[[locale]].
66 DirectHandle<String> locale_str =
67 isolate->factory()->NewStringFromAsciiChecked(r.locale.c_str());
68
69 // 13. Let granularity be ? GetOption(options, "granularity", "string", «
70 // "grapheme", "word", "sentence" », "grapheme").
71 Maybe<Granularity> maybe_granularity = GetStringOption<Granularity>(
72 isolate, options, "granularity", service,
73 {"grapheme", "word", "sentence"},
74 {Granularity::GRAPHEME, Granularity::WORD, Granularity::SENTENCE},
75 Granularity::GRAPHEME);
76 MAYBE_RETURN(maybe_granularity, MaybeHandle<JSSegmenter>());
77 Granularity granularity_enum = maybe_granularity.FromJust();
78
79 icu::Locale icu_locale = r.icu_locale;
80 DCHECK(!icu_locale.isBogus());
81
82 UErrorCode status = U_ZERO_ERROR;
83 std::unique_ptr<icu::BreakIterator> icu_break_iterator;
84
85 switch (granularity_enum) {
86 case Granularity::GRAPHEME:
87 icu_break_iterator.reset(
88 icu::BreakIterator::createCharacterInstance(icu_locale, status));
89 break;
90 case Granularity::WORD:
91 icu_break_iterator.reset(
92 icu::BreakIterator::createWordInstance(icu_locale, status));
93 break;
94 case Granularity::SENTENCE:
95 icu_break_iterator.reset(
96 icu::BreakIterator::createSentenceInstance(icu_locale, status));
97 break;
98 }
99
100 DCHECK(U_SUCCESS(status));
101 DCHECK_NOT_NULL(icu_break_iterator.get());
102
103 DirectHandle<Managed<icu::BreakIterator>> managed_break_iterator =
104 Managed<icu::BreakIterator>::From(isolate, 0,
105 std::move(icu_break_iterator));
106
107 // Now all properties are ready, so we can allocate the result object.
108 Handle<JSSegmenter> segmenter =
109 Cast<JSSegmenter>(isolate->factory()->NewFastOrSlowJSObjectFromMap(map));
110 DisallowGarbageCollection no_gc;
111 segmenter->set_flags(0);
112
113 // 12. Set segmenter.[[Locale]] to the value of r.[[Locale]].
114 segmenter->set_locale(*locale_str);
115
116 // 14. Set segmenter.[[SegmenterGranularity]] to granularity.
117 segmenter->set_granularity(granularity_enum);
118
119 segmenter->set_icu_break_iterator(*managed_break_iterator);
120
121 // 15. Return segmenter.
122 return segmenter;
123}

126Handle<JSObject> JSSegmenter::ResolvedOptions(
127 Isolate* isolate, DirectHandle<JSSegmenter> segmenter) {
128 Factory* factory = isolate->factory();
129 // 3. Let options be ! ObjectCreate(%ObjectPrototype%).
130 Handle<JSObject> result = factory->NewJSObject(isolate->object_function());
131 // 4. For each row of Table 1, except the header row, do
132 // a. Let p be the Property value of the current row.
133 // b. Let v be the value of pr's internal slot whose name is the Internal Slot
134 // value of the current row.
135 //
136 // c. If v is not undefined, then
137 // i. Perform ! CreateDataPropertyOrThrow(options, p, v).
138 // Table 1: Resolved Options of Segmenter Instances
139 // Internal Slot Property
140 // [[Locale]] "locale"
141 // [[SegmenterGranularity]] "granularity"
142
143 DirectHandle<String> locale(segmenter->locale(), isolate);
144 JSObject::AddProperty(isolate, result, factory->locale_string(), locale,
145 NONE);
146 JSObject::AddProperty(isolate, result, factory->granularity_string(),
147 segmenter->GranularityAsString(isolate), NONE);
148 // 5. Return options.
149 return result;
150}
