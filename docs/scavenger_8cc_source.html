97 void VisitExternalPointer(Tagged<HeapObject> host,
98 ExternalPointerSlot slot) override {
99#ifdef V8_COMPRESS_POINTERS
100 DCHECK(!slot.tag_range().IsEmpty());
101 DCHECK(!IsSharedExternalPointerType(slot.tag_range()));
102 // TODO(chromium:337580006): Remove when pointer compression always uses
103 // EPT.
104 if (!slot.HasExternalPointerHandle()) return;
105 ExternalPointerHandle handle = slot.Relaxed_LoadHandle();
106 Heap* heap = scavenger_->heap();
107 ExternalPointerTable& table = heap->isolate()->external_pointer_table();
108
109 // For survivor objects, the scavenger marks their EPT entries when they are
110 // copied and then sweeps the young EPT space at the end of collection,
111 // reclaiming unmarked EPT entries. (Exception: if an incremental mark is
112 // in progress, the scavenger neither marks nor sweeps, as it will be the
113 // major GC's responsibility.)
114 //
115 // However when promoting, we just evacuate the entry from new to old space.
116 // Usually the entry will be unmarked, unless an incremental mark is in
117 // progress, or the slot was initialized since the last GC (external pointer
118 // tags have the mark bit set), in which case it may be marked already. In
119 // any case, transfer the color from new to old EPT space.
120 table.Evacuate(heap->young_external_pointer_space(),
121 heap->old_external_pointer_space(), handle, slot.address(),
122 ExternalPointerTable::EvacuateMarkMode::kTransferMark);
123#endif // V8_COMPRESS_POINTERS
124 }

141 V8_INLINE void VisitPointersImpl(Tagged<HeapObject> host, TSlot start,
142 TSlot end) {
143 using THeapObjectSlot = typename TSlot::THeapObjectSlot;
144 // Treat weak references as strong.
145 // TODO(marja): Proper weakness handling in the young generation.
146 for (TSlot slot = start; slot < end; ++slot) {
147 typename TSlot::TObject object = *slot;

155 template <typename THeapObjectSlot>
156 V8_INLINE void HandleSlot(Tagged<HeapObject> host, THeapObjectSlot slot,
157 Tagged<HeapObject> target) {
158 static_assert(
159 std::is_same_v<THeapObjectSlot, FullHeapObjectSlot> ||
160 std::is_same_v<THeapObjectSlot, HeapObjectSlot>,
161 "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
162 scavenger_->SynchronizePageAccess(target);
163
164 if (Heap::InFromPage(target)) {
165 SlotCallbackResult result = scavenger_->ScavengeObject(slot, target);
166 bool success = (*slot).GetHeapObject(&target);
167 USE(success);
168 DCHECK(success);
169
170 if (result == KEEP_SLOT) {
171 SLOW_DCHECK(IsHeapObject(target));
172 MemoryChunk* chunk = MemoryChunk::FromHeapObject(host);
173 MutablePageMetadata* page =
174 MutablePageMetadata::cast(chunk->Metadata());
175
176 // Sweeper is stopped during scavenge, so we can directly
177 // insert into its remembered set here.
178 RememberedSet<OLD_TO_NEW>::Insert<AccessMode::ATOMIC>(
179 page, chunk->Offset(slot.address()));
180 }
181 DCHECK(!MarkCompactCollector::IsOnEvacuationCandidate(target));
182 } else if (record_slots_ &&
183 MarkCompactCollector::IsOnEvacuationCandidate(target)) {
184 // We should never try to record off-heap slots.
185 DCHECK((std::is_same<THeapObjectSlot, HeapObjectSlot>::value));
186 // InstructionStream slots never appear in new space because
187 // Code objects, the only object that can contain code pointers, are
188 // always allocated in the old space.
189 DCHECK_IMPLIES(V8_EXTERNAL_CODE_SPACE_BOOL,
190 !MemoryChunk::FromHeapObject(target)->IsFlagSet(
191 MemoryChunk::IS_EXECUTABLE));
192
193 // We cannot call MarkCompactCollector::RecordSlot because that checks
194 // that the host page is not in young generation, which does not hold
195 // for pending large pages.
196 MemoryChunk* chunk = MemoryChunk::FromHeapObject(host);
197 MutablePageMetadata* page = MutablePageMetadata::cast(chunk->Metadata());
198 RememberedSet<OLD_TO_OLD>::Insert<AccessMode::ATOMIC>(
199 page, chunk->Offset(slot.address()));
200 }
201
202 if (HeapLayout::InWritableSharedSpace(target)) {
203 MemoryChunk* chunk = MemoryChunk::FromHeapObject(host);
204 MutablePageMetadata* page = MutablePageMetadata::cast(chunk->Metadata());
205 RememberedSet<OLD_TO_SHARED>::Insert<AccessMode::ATOMIC>(
206 page, chunk->Offset(slot.address()));
207 }
208 }
209
210 Scavenger* const scavenger_;
211 const bool record_slots_;

255void ScavengerCollector::JobTask::Run(JobDelegate* delegate) {
256 DCHECK_LT(delegate->GetTaskId(), scavengers_->size());
257 // In case multi-cage pointer compression mode is enabled ensure that
258 // current thread's cage base values are properly initialized.
259 PtrComprCageAccessScope ptr_compr_cage_access_scope(
260 collector_->heap_->isolate());
261
262 collector_->estimate_concurrency_.fetch_add(1, std::memory_order_relaxed);
263
264 Scavenger* scavenger = (*scavengers_)[delegate->GetTaskId()].get();
265 if (delegate->IsJoiningThread()) {
266 TRACE_GC_WITH_FLOW(collector_->heap_->tracer(),
267 GCTracer::Scope::SCAVENGER_SCAVENGE_PARALLEL, trace_id_,
268 TRACE_EVENT_FLAG_FLOW_IN);
269 ProcessItems(delegate, scavenger);
270 } else {
271 TRACE_GC_EPOCH_WITH_FLOW(
272 collector_->heap_->tracer(),
273 GCTracer::Scope::SCAVENGER_BACKGROUND_SCAVENGE_PARALLEL,
274 ThreadKind::kBackground, trace_id_, TRACE_EVENT_FLAG_FLOW_IN);
275 ProcessItems(delegate, scavenger);
276 }
277}

279size_t ScavengerCollector::JobTask::GetMaxConcurrency(
280 size_t worker_count) const {
281 // We need to account for local segments held by worker_count in addition to
282 // GlobalPoolSize() of copied_list_, pinned_list_ and promoted_list_.
283 size_t wanted_num_workers =
284 std::max<size_t>(remaining_memory_chunks_.load(std::memory_order_relaxed),
285 worker_count + copied_list_.Size() +
286 pinned_list_.Size() + promoted_list_.Size());
287 if (!collector_->heap_->ShouldUseBackgroundThreads() ||
288 collector_->heap_->ShouldOptimizeForBattery()) {
289 return std::min<size_t>(wanted_num_workers, 1);
290 }
291 return std::min<size_t>(scavengers_->size(), wanted_num_workers);
292}

294void ScavengerCollector::JobTask::ProcessItems(JobDelegate* delegate,
295 Scavenger* scavenger) {
296 double scavenging_time = 0.0;
297 {
298 TimedScope scope(&scavenging_time);
299 scavenger->VisitPinnedObjects();
300 ConcurrentScavengePages(scavenger);
301 scavenger->Process(delegate);
302 }
303 if (V8_UNLIKELY(v8_flags.trace_parallel_scavenge)) {
304 PrintIsolate(collector_->heap_->isolate(),
305 "scavenge[%p]: time=%.2f copied=%zu promoted=%zu\n",
306 static_cast<void*>(this), scavenging_time,
307 scavenger->bytes_copied(), scavenger->bytes_promoted());
308 }

311void ScavengerCollector::JobTask::ConcurrentScavengePages(
312 Scavenger* scavenger) {
313 while (remaining_memory_chunks_.load(std::memory_order_relaxed) > 0) {
314 std::optional<size_t> index = generator_.GetNext();
315 if (!index) {
316 return;
317 }
318 for (size_t i = *index; i < old_to_new_chunks_.size(); ++i) {
319 auto& work_item = old_to_new_chunks_[i];
320 if (!work_item.first.TryAcquire()) {
321 break;
322 }
323 scavenger->ScavengePage(work_item.second);
324 if (remaining_memory_chunks_.fetch_sub(1, std::memory_order_relaxed) <=
325 1) {
326 return;
327 }
328 }
329 }

573 Heap* heap, MemoryChunk* chunk,
574 std::vector<std::pair<Address, size_t>>& pinned_objects_and_sizes) {

576 std::sort(pinned_objects_and_sizes.begin(), pinned_objects_and_sizes.end());

582 start = object + size;

589 ->marking_bitmap()
590 ->IsClean());

593void RestoreAndQuarantinePinnedObjects(SemiSpaceNewSpace& new_space,
594 const PinnedObjects& pinned_objects) {
595 std::unordered_map<MemoryChunk*, std::vector<std::pair<Address, size_t>>,
596 base::hash<const MemoryChunk*>>
597 pages_with_pinned_objects;
598 size_t quarantined_objects_size = 0;
599 // Restore the maps of quarantined objects. We use the iteration over
600 // quarantined objects to split them based on pages. This will be used below
601 // for sweeping the quarantined pages (since there are no markbits).
602 for (const auto& [object_address, map_word] : pinned_objects) {
603 DCHECK(!map_word.IsForwardingAddress());
604 Tagged<HeapObject> object = HeapObject::FromAddress(object_address);
605 DCHECK(HeapLayout::IsSelfForwarded(object));
606 object->set_map_word(map_word.ToMap(), kRelaxedStore);
607 const size_t object_size = object->SizeFromMap(map_word.ToMap());
608 pages_with_pinned_objects[MemoryChunk::FromHeapObject(
609 Cast<HeapObject>(object))]
610 .emplace_back(object_address, object_size);
611 quarantined_objects_size += object_size;
612 }
613 DCHECK_EQ(0, new_space.QuarantinedPageCount());
614 // Sweep quarantined pages to make them iterable.
615 for (auto it : pages_with_pinned_objects) {
616 MemoryChunk* chunk = it.first;
617 std::vector<std::pair<Address, size_t>>& pinned_objects_and_sizes =
618 it.second;
619 DCHECK(chunk->IsFromPage());
620 SweepQuarantinedPage(new_space.heap(), chunk, pinned_objects_and_sizes);
621 new_space.MoveQuarantinedPage(chunk);
622 DCHECK(!chunk->IsFromPage());
623 DCHECK(chunk->IsToPage());
624 }
625 new_space.SetQuarantinedSize(quarantined_objects_size);
626}

630void ScavengerCollector::CollectGarbage() {
631 ScopedFullHeapCrashKey collect_full_heap_dump_if_crash(isolate_);
632
633 SemiSpaceNewSpace* new_space = SemiSpaceNewSpace::From(heap_->new_space());
634 new_space->GarbageCollectionPrologue();
635 new_space->EvacuatePrologue();
636
637 // We also flip the young generation large object space. All large objects
638 // will be in the from space.
639 heap_->new_lo_space()->Flip();
640 heap_->new_lo_space()->ResetPendingObject();
641
642 DCHECK(!heap_->allocator()->new_space_allocator()->IsLabValid());
643
644 DCHECK(surviving_new_large_objects_.empty());
645
646 Scavenger::EmptyChunksList empty_chunks;
647 Scavenger::CopiedList copied_list;
648 Scavenger::PinnedList pinned_list;
649 Scavenger::PromotedList promoted_list;
650 EphemeronRememberedSet::TableList ephemeron_table_list;
651
652 PinnedObjects pinned_objects;
653
654 const int num_scavenge_tasks = NumberOfScavengeTasks();
655 std::vector<std::unique_ptr<Scavenger>> scavengers;
656 {
657 const bool is_logging = isolate_->log_object_relocation();
658 for (int i = 0; i < num_scavenge_tasks; ++i) {
659 scavengers.emplace_back(
660 new Scavenger(this, heap_, is_logging, &empty_chunks, &copied_list,
661 &pinned_list, &promoted_list, &ephemeron_table_list));
662 }
663 Scavenger& main_thread_scavenger = *scavengers[kMainThreadId].get();
664
665 {
666 // Identify weak unmodified handles. Requires an unmodified graph.
667 TRACE_GC(
668 heap_->tracer(),
669 GCTracer::Scope::SCAVENGER_SCAVENGE_WEAK_GLOBAL_HANDLES_IDENTIFY);
670 isolate_->traced_handles()->ComputeWeaknessForYoungObjects();
671 }
672
673 std::vector<std::pair<ParallelWorkItem, MutablePageMetadata*>>
674 old_to_new_chunks;
675 {
676 // Copy roots.
677 TRACE_GC(heap_->tracer(), GCTracer::Scope::SCAVENGER_SCAVENGE_ROOTS);
678
679 // We must collect old-to-new pages before starting Scavenge because pages
680 // could be removed from the old generation for allocation which hides
681 // them from the iteration.
682 OldGenerationMemoryChunkIterator::ForAll(
683 heap_, [&old_to_new_chunks](MutablePageMetadata* chunk) {
684 if (chunk->slot_set<OLD_TO_NEW>() ||
685 chunk->typed_slot_set<OLD_TO_NEW>() ||
686 chunk->slot_set<OLD_TO_NEW_BACKGROUND>()) {
687 old_to_new_chunks.emplace_back(ParallelWorkItem{}, chunk);
688 }
689 });
690
691 if (v8_flags.scavenger_pinning_objects &&
692 heap_->IsGCWithMainThreadStack()) {
693 // Pinning objects must be the first step and must happen before
694 // scavenging any objects. Specifically we must all pin all objects
695 // before visiting other pinned objects. If we scavenge some object X
696 // and move it before all stack-reachable objects are pinned, and we
697 // later find that we need to pin X, it will be too late to undo the
698 // moving.
699 TRACE_GC(heap_->tracer(),
700 GCTracer::Scope::SCAVENGER_SCAVENGE_PIN_OBJECTS);
701 ObjectPinningVisitor pinning_visitor(main_thread_scavenger,
702 pinned_objects);
703 // Scavenger reuses the page's marking bitmap as a temporary object
704 // start bitmap. Stack scanning will incrementally build the map as it
705 // searches through pages.
706 YoungGenerationConservativeStackVisitor stack_visitor(isolate_,
707 &pinning_visitor);
708 // Marker was already set by Heap::CollectGarbage.
709 heap_->stack().IteratePointersUntilMarker(&stack_visitor);
710 if (v8_flags.stress_scavenger_pinning_objects) {
711 TreatConservativelyVisitor handles_visitor(&stack_visitor, heap_);
712 isolate_->handle_scope_implementer()->Iterate(&handles_visitor);
713 }
714 }
715
716 // Scavenger treats all weak roots except for global handles as strong.
717 // That is why we don't set skip_weak = true here and instead visit
718 // global handles separately.
719 base::EnumSet<SkipRoot> options(
720 {SkipRoot::kExternalStringTable, SkipRoot::kGlobalHandles,
721 SkipRoot::kTracedHandles, SkipRoot::kOldGeneration,
722 SkipRoot::kConservativeStack, SkipRoot::kReadOnlyBuiltins});
723 if (V8_UNLIKELY(v8_flags.scavenge_separate_stack_scanning)) {
724 options.Add(SkipRoot::kStack);
725 }
726 RootScavengeVisitor root_scavenge_visitor(main_thread_scavenger);
727
728 heap_->IterateRoots(&root_scavenge_visitor, options);
729 isolate_->global_handles()->IterateYoungStrongAndDependentRoots(
730 &root_scavenge_visitor);
731 isolate_->traced_handles()->IterateYoungRoots(&root_scavenge_visitor);
732 }
733 {
734 // Parallel phase scavenging all copied and promoted objects.
735 TRACE_GC_ARG1(
736 heap_->tracer(), GCTracer::Scope::SCAVENGER_SCAVENGE_PARALLEL_PHASE,
737 "UseBackgroundThreads", heap_->ShouldUseBackgroundThreads());
738
739 auto job = std::make_unique<JobTask>(
740 this, &scavengers, std::move(old_to_new_chunks), copied_list,
741 pinned_list, promoted_list);
742 TRACE_GC_NOTE_WITH_FLOW("Parallel scavenge started", job->trace_id(),
743 TRACE_EVENT_FLAG_FLOW_OUT);
744 V8::GetCurrentPlatform()
745 ->CreateJob(v8::TaskPriority::kUserBlocking, std::move(job))
746 ->Join();
747 DCHECK(copied_list.IsEmpty());
748 DCHECK(pinned_list.IsEmpty());
749 DCHECK(promoted_list.IsEmpty());
750 }
751
752 if (V8_UNLIKELY(v8_flags.scavenge_separate_stack_scanning)) {
753 {
754 RootScavengeVisitor root_scavenge_visitor(main_thread_scavenger);
755 IterateStackAndScavenge(&root_scavenge_visitor, &scavengers,
756 main_thread_scavenger);
757 }
758 DCHECK(copied_list.IsEmpty());
759 DCHECK(pinned_list.IsEmpty());
760 DCHECK(promoted_list.IsEmpty());
761 }
762
763 {
764 // Scavenge weak global handles.
765 TRACE_GC(heap_->tracer(),
766 GCTracer::Scope::SCAVENGER_SCAVENGE_WEAK_GLOBAL_HANDLES_PROCESS);
767 GlobalHandlesWeakRootsUpdatingVisitor visitor;
768 isolate_->global_handles()->ProcessWeakYoungObjects(
769 &visitor, &IsUnscavengedHeapObjectSlot);
770 isolate_->traced_handles()->ProcessYoungObjects(
771 &visitor, &IsUnscavengedHeapObjectSlot);
772 }
773
774 {
775 // Finalize parallel scavenging.
776 TRACE_GC(heap_->tracer(), GCTracer::Scope::SCAVENGER_SCAVENGE_FINALIZE);
777
778 DCHECK(surviving_new_large_objects_.empty());
779
780 for (auto& scavenger : scavengers) {
781 scavenger->Finalize();
782 }
783 scavengers.clear();
784
785#ifdef V8_COMPRESS_POINTERS
786 // Sweep the external pointer table, unless an incremental mark is in
787 // progress, in which case leave sweeping to the end of the
788 // already-scheduled major GC cycle. (If we swept here we'd clear EPT
789 // marks that the major marker was using, which would be an error.)
790 DCHECK(heap_->concurrent_marking()->IsStopped());
791 if (!heap_->incremental_marking()->IsMajorMarking()) {
792 heap_->isolate()->external_pointer_table().Sweep(
793 heap_->young_external_pointer_space(),
794 heap_->isolate()->counters());
795 }
796#endif // V8_COMPRESS_POINTERS
797
798 HandleSurvivingNewLargeObjects();
799
800 heap_->tracer()->SampleConcurrencyEsimate(
801 FetchAndResetConcurrencyEstimate());
802 }
803 }
804
805 {
806 // Update references into new space
807 TRACE_GC(heap_->tracer(), GCTracer::Scope::SCAVENGER_SCAVENGE_UPDATE_REFS);
808 heap_->UpdateYoungReferencesInExternalStringTable(
809 &Heap::UpdateYoungReferenceInExternalStringTableEntry);
810
811 heap_->incremental_marking()->UpdateMarkingWorklistAfterScavenge();
812 heap_->incremental_marking()->UpdateExternalPointerTableAfterScavenge();
813
814 if (V8_UNLIKELY(v8_flags.always_use_string_forwarding_table)) {
815 isolate_->string_forwarding_table()->UpdateAfterYoungEvacuation();
816 }
817 }
818
819 if (v8_flags.concurrent_marking) {
820 // Ensure that concurrent marker does not track pages that are
821 // going to be unmapped.
822 for (PageMetadata* p :
823 PageRange(new_space->from_space().first_page(), nullptr)) {
824 heap_->concurrent_marking()->ClearMemoryChunkData(p);
825 }
826 }
827
828 ProcessWeakReferences(&ephemeron_table_list);
829
830 RestoreAndQuarantinePinnedObjects(*new_space, pinned_objects);
831
832 // Need to free new space LAB that was allocated during scavenge.
833 heap_->allocator()->new_space_allocator()->FreeLinearAllocationArea();
834
835 // Since we promote all surviving large objects immediately, all remaining
836 // large objects must be dead.
837 // TODO(hpayer): Don't free all as soon as we have an intermediate generation.
838 heap_->new_lo_space()->FreeDeadObjects(
839 [](Tagged<HeapObject>) { return true; });
840
841 new_space->GarbageCollectionEpilogue();
842
843 {
844 TRACE_GC(heap_->tracer(), GCTracer::Scope::SCAVENGER_FREE_REMEMBERED_SET);
845 Scavenger::EmptyChunksList::Local empty_chunks_local(empty_chunks);
846 MutablePageMetadata* chunk;
847 while (empty_chunks_local.Pop(&chunk)) {
848 RememberedSet<OLD_TO_NEW>::CheckPossiblyEmptyBuckets(chunk);
849 RememberedSet<OLD_TO_NEW_BACKGROUND>::CheckPossiblyEmptyBuckets(chunk);
850 }
851
852#ifdef DEBUG
853 OldGenerationMemoryChunkIterator::ForAll(
854 heap_, [](MutablePageMetadata* chunk) {
855 if (chunk->slot_set<OLD_TO_NEW>() ||
856 chunk->typed_slot_set<OLD_TO_NEW>() ||
857 chunk->slot_set<OLD_TO_NEW_BACKGROUND>()) {
858 DCHECK(chunk->possibly_empty_buckets()->IsEmpty());
859 }
860 });
861#endif
862 }
863
864 SweepArrayBufferExtensions();
865
866 isolate_->global_handles()->UpdateListOfYoungNodes();
867 isolate_->traced_handles()->UpdateListOfYoungNodes();
868
869 // Update how much has survived scavenge.
870 heap_->IncrementYoungSurvivorsCounter(heap_->SurvivedYoungObjectSize());
871
872 const auto resize_mode = heap_->ShouldResizeNewSpace();
873 switch (resize_mode) {
874 case Heap::ResizeNewSpaceMode::kShrink:
875 heap_->ReduceNewSpaceSize();
876 break;
877 case Heap::ResizeNewSpaceMode::kGrow:
878 heap_->ExpandNewSpaceSize();
879 break;
880 case Heap::ResizeNewSpaceMode::kNone:
881 break;
882 }
883}

885void ScavengerCollector::IterateStackAndScavenge(
886 RootScavengeVisitor* root_scavenge_visitor,
887 std::vector<std::unique_ptr<Scavenger>>* scavengers,
888 Scavenger& main_thread_scavenger) {
889 // Scan the stack, scavenge the newly discovered objects, and report
890 // the survival statistics before and after the stack scanning.
891 // This code is not intended for production.
892 TRACE_GC(heap_->tracer(), GCTracer::Scope::SCAVENGER_SCAVENGE_STACK_ROOTS);
893 const auto sum_survived_bytes = [](size_t count, auto& scavenger) {
894 return count + scavenger->bytes_copied() + scavenger->bytes_promoted();
895 };
896 const size_t survived_bytes_before = std::accumulate(
897 scavengers->begin(), scavengers->end(), 0, sum_survived_bytes);
898 heap_->IterateStackRoots(root_scavenge_visitor);
899 main_thread_scavenger.Process();
900 const size_t survived_bytes_after = std::accumulate(
901 scavengers->begin(), scavengers->end(), 0, sum_survived_bytes);
902 TRACE_EVENT2(TRACE_DISABLED_BY_DEFAULT("v8.gc"),
903 "V8.GCScavengerStackScanning", "survived_bytes_before",
904 survived_bytes_before, "survived_bytes_after",
905 survived_bytes_after);
906 if (v8_flags.trace_gc_verbose && !v8_flags.trace_gc_ignore_scavenger) {
907 isolate_->PrintWithTimestamp(
908 "Scavenge stack scanning: survived_before=%4zuKB, "
909 "survived_after=%4zuKB delta=%.1f%%\n",
910 survived_bytes_before / KB, survived_bytes_after / KB,
911 (survived_bytes_after - survived_bytes_before) * 100.0 /
912 survived_bytes_after);
913 }

925void ScavengerCollector::HandleSurvivingNewLargeObjects() {
926 const bool is_compacting = heap_->incremental_marking()->IsCompacting();
927 MarkingState* marking_state = heap_->marking_state();
928
929 for (SurvivingNewLargeObjectMapEntry update_info :
930 surviving_new_large_objects_) {
931 Tagged<HeapObject> object = update_info.first;
932 Tagged<Map> map = update_info.second;
933 // Order is important here. We have to re-install the map to have access
934 // to meta-data like size during page promotion.
935 object->set_map_word(map, kRelaxedStore);

937 if (is_compacting && marking_state->IsMarked(object) &&
938 MarkCompactCollector::IsOnEvacuationCandidate(map)) {
939 MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
940 MutablePageMetadata* page = MutablePageMetadata::cast(chunk->Metadata());
941 RememberedSet<OLD_TO_OLD>::Insert<AccessMode::ATOMIC>(
942 page, chunk->Offset(object->map_slot().address()));
943 }
944 LargePageMetadata* page = LargePageMetadata::FromHeapObject(object);
945 heap_->lo_space()->PromoteNewLargeObject(page);
946 }
947 surviving_new_large_objects_.clear();
948 heap_->new_lo_space()->set_objects_size(0);
949}

1005void Scavenger::IterateAndScavengePromotedObject(Tagged<HeapObject> target,
1006 Tagged<Map> map, int size) {
1007 // We are not collecting slots on new space objects during mutation thus we
1008 // have to scan for pointers to evacuation candidates when we promote
1009 // objects. But we should not record any slots in non-black objects. Grey
1010 // object's slots would be rescanned. White object might not survive until
1011 // the end of collection it would be a violation of the invariant to record
1012 // its slots.
1013 const bool record_slots =
1014 is_compacting_ && heap()->marking_state()->IsMarked(target);
1015
1016 IterateAndScavengePromotedObjectsVisitor visitor(this, record_slots);
1017
1018 // Iterate all outgoing pointers including map word.
1019 visitor.Visit(map, target, size);
1020
1021 if (IsJSArrayBufferMap(map)) {
1022 DCHECK(!MemoryChunk::FromHeapObject(target)->IsLargePage());
1023 Cast<JSArrayBuffer>(target)->YoungMarkExtensionPromoted();
1024 }

1034void Scavenger::ScavengePage(MutablePageMetadata* page) {
1035 const bool record_old_to_shared_slots = heap_->isolate()->has_shared_space();
1036
1037 MemoryChunk* chunk = page->Chunk();
1038
1039 if (page->slot_set<OLD_TO_NEW, AccessMode::ATOMIC>() != nullptr) {
1040 RememberedSet<OLD_TO_NEW>::IterateAndTrackEmptyBuckets(
1041 page,
1042 [this, chunk, page, record_old_to_shared_slots](MaybeObjectSlot slot) {
1043 SlotCallbackResult result = CheckAndScavengeObject(heap_, slot);
1044 // A new space string might have been promoted into the shared heap
1045 // during GC.
1046 if (result == REMOVE_SLOT && record_old_to_shared_slots) {
1047 CheckOldToNewSlotForSharedUntyped(chunk, page, slot);
1048 }
1049 return result;
1050 },
1051 &local_empty_chunks_);
1052 }
1053
1054 if (chunk->executable()) {
1055 std::vector<std::tuple<Tagged<HeapObject>, SlotType, Address>> slot_updates;
1056
1057 // The code running write access to executable memory poses CFI attack
1058 // surface and needs to be kept to a minimum. So we do the the iteration in
1059 // two rounds. First we iterate the slots and scavenge objects and in the
1060 // second round with write access, we only perform the pointer updates.
1061 const auto typed_slot_count = RememberedSet<OLD_TO_NEW>::IterateTyped(
1062 page, [this, chunk, page, record_old_to_shared_slots, &slot_updates](
1063 SlotType slot_type, Address slot_address) {
1064 Tagged<HeapObject> old_target =
1065 UpdateTypedSlotHelper::GetTargetObject(heap_, slot_type,
1066 slot_address);
1067 Tagged<HeapObject> new_target = old_target;
1068 FullMaybeObjectSlot slot(&new_target);
1069 SlotCallbackResult result = CheckAndScavengeObject(heap(), slot);
1070 if (result == REMOVE_SLOT && record_old_to_shared_slots) {
1071 CheckOldToNewSlotForSharedTyped(chunk, page, slot_type,
1072 slot_address, *slot);
1073 }
1074 if (new_target != old_target) {
1075 slot_updates.emplace_back(new_target, slot_type, slot_address);
1076 }
1077 return result;
1078 });
1079 // Typed slots only exist in code objects. Since code is never young, it is
1080 // safe to release an empty typed slot set as no other scavenge thread will
1081 // attempt to promote to the page and write to the slot set.
1082 if (typed_slot_count == 0) {
1083 page->ReleaseTypedSlotSet(OLD_TO_NEW);
1084 }
1085
1086 WritableJitPage jit_page = ThreadIsolation::LookupWritableJitPage(
1087 page->area_start(), page->area_size());
1088 for (auto& slot_update : slot_updates) {
1089 Tagged<HeapObject> new_target = std::get<0>(slot_update);
1090 SlotType slot_type = std::get<1>(slot_update);
1091 Address slot_address = std::get<2>(slot_update);
1092
1093 WritableJitAllocation jit_allocation =
1094 jit_page.LookupAllocationContaining(slot_address);
1095 UpdateTypedSlotHelper::UpdateTypedSlot(
1096 jit_allocation, heap_, slot_type, slot_address,
1097 [new_target](FullMaybeObjectSlot slot) {
1098 slot.store(new_target);
1099 return KEEP_SLOT;
1100 });
1101 }
1102 } else {
1103 DCHECK_NULL(page->typed_slot_set<OLD_TO_NEW>());
1104 }
1105
1106 if (page->slot_set<OLD_TO_NEW_BACKGROUND, AccessMode::ATOMIC>() != nullptr) {
1107 RememberedSet<OLD_TO_NEW_BACKGROUND>::IterateAndTrackEmptyBuckets(
1108 page,
1109 [this, chunk, page, record_old_to_shared_slots](MaybeObjectSlot slot) {
1110 SlotCallbackResult result = CheckAndScavengeObject(heap_, slot);
1111 // A new space string might have been promoted into the shared heap
1112 // during GC.
1113 if (result == REMOVE_SLOT && record_old_to_shared_slots) {
1114 CheckOldToNewSlotForSharedUntyped(chunk, page, slot);
1115 }
1116 return result;
1117 },
1118 &local_empty_chunks_);
1119 }
1120}

1122void Scavenger::Process(JobDelegate* delegate) {
1123 ScavengeVisitor scavenge_visitor(this);
1124
1125 bool done;
1126 size_t objects = 0;
1127 do {
1128 done = true;
1129 Tagged<HeapObject> object;
1130 while (!ShouldEagerlyProcessPromotedList() &&
1131 local_copied_list_.Pop(&object)) {
1132 scavenge_visitor.Visit(object);
1133 done = false;
1134 if (delegate && ((++objects % kInterruptThreshold) == 0)) {
1135 if (!local_copied_list_.IsLocalEmpty()) {
1136 delegate->NotifyConcurrencyIncrease();
1137 }
1138 }
1139 }
1140
1141 struct PromotedListEntry entry;
1142 while (local_promoted_list_.Pop(&entry)) {
1143 Tagged<HeapObject> target = entry.heap_object;
1144 IterateAndScavengePromotedObject(target, entry.map, entry.size);
1145 done = false;
1146 if (delegate && ((++objects % kInterruptThreshold) == 0)) {
1147 if (!local_promoted_list_.IsGlobalEmpty()) {
1148 delegate->NotifyConcurrencyIncrease();
1149 }
1150 }
1151 }
1152 } while (!done);
1153}

1161// Clear ephemeron entries from EphemeronHashTables in new-space whenever the
1162// entry has a dead new-space key.
1163void ScavengerCollector::ClearYoungEphemerons(
1164 EphemeronRememberedSet::TableList* ephemeron_table_list) {
1165 ephemeron_table_list->Iterate([this](Tagged<EphemeronHashTable> table) {
1166 for (InternalIndex i : table->IterateEntries()) {
1167 // Keys in EphemeronHashTables must be heap objects.
1168 HeapObjectSlot key_slot(
1169 table->RawFieldOfElementAt(EphemeronHashTable::EntryToIndex(i)));
1170 Tagged<HeapObject> key = key_slot.ToHeapObject();
1171 if (IsUnscavengedHeapObject(heap_, key)) {
1172 table->RemoveEntry(i);
1173 } else {
1174 Tagged<HeapObject> forwarded = ForwardingAddress(key);
1175 key_slot.StoreHeapObject(forwarded);
1176 }
1177 }
1178 });
1179 ephemeron_table_list->Clear();
1180}

1182// Clear ephemeron entries from EphemeronHashTables in old-space whenever the
1183// entry has a dead new-space key.
1184void ScavengerCollector::ClearOldEphemerons() {
1185 auto* table_map = heap_->ephemeron_remembered_set_->tables();
1186 for (auto it = table_map->begin(); it != table_map->end();) {
1187 Tagged<EphemeronHashTable> table = it->first;
1188 auto& indices = it->second;

1190 // Keys in EphemeronHashTables must be heap objects.
