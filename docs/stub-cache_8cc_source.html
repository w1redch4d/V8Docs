29// Hash algorithm for the primary table. This algorithm is replicated in
30// the AccessorAssembler. Returns an index into the table that
31// is scaled by 1 << kCacheIndexShift.
32int StubCache::PrimaryOffset(Tagged<Name> name, Tagged<Map> map) {
33 // Compute the hash of the name (use entire hash field).
34 uint32_t field = name->RawHash();
35 DCHECK(Name::IsHashFieldComputed(field));
36 // Using only the low bits in 64-bit mode is unlikely to increase the
37 // risk of collision even if the heap is spread over an area larger than
38 // 4Gb (and not at all if it isn't).
39 uint32_t map_low32bits =
40 static_cast<uint32_t>(map.ptr() ^ (map.ptr() >> kPrimaryTableBits));
41 // Base the offset on a simple combination of name and map.
42 uint32_t key = map_low32bits + field;
43 return key & ((kPrimaryTableSize - 1) << kCacheIndexShift);
44}
45
46// Hash algorithm for the secondary table. This algorithm is replicated in
47// assembler. This hash should be sufficiently different from the primary one
48// in order to avoid collisions for minified code with short names.
49// Returns an index into the table that is scaled by 1 << kCacheIndexShift.
50int StubCache::SecondaryOffset(Tagged<Name> name, Tagged<Map> old_map) {
51 uint32_t name_low32bits = static_cast<uint32_t>(name.ptr());
52 uint32_t map_low32bits = static_cast<uint32_t>(old_map.ptr());
53 uint32_t key = (map_low32bits + name_low32bits);
54 key = key + (key >> kSecondaryTableBits);
55 return key & ((kSecondaryTableSize - 1) << kCacheIndexShift);
56}

83void StubCache::Set(Tagged<Name> name, Tagged<Map> map,
84 Tagged<MaybeObject> handler) {
85 DCHECK(CommonStubCacheChecks(this, name, map, handler));
86
87 // Compute the primary entry.
88 int primary_offset = PrimaryOffset(name, map);
89 Entry* primary = entry(primary_, primary_offset);
90 Tagged<MaybeObject> old_handler(
91 TaggedValue::ToMaybeObject(isolate(), primary->value));
92 // If the primary entry has useful data in it, we retire it to the
93 // secondary cache before overwriting it.
94 // We need SafeEquals here while Builtin Code objects still live in the RO
95 // space inside the sandbox.
96 static_assert(!kAllCodeObjectsLiveInTrustedSpace);
97 if (!old_handler.SafeEquals(isolate()->builtins()->code(Builtin::kIllegal)) &&
98 !primary->map.IsSmi()) {
99 Tagged<Map> old_map =
100 Cast<Map>(StrongTaggedValue::ToObject(isolate(), primary->map));
101 Tagged<Name> old_name =
102 Cast<Name>(StrongTaggedValue::ToObject(isolate(), primary->key));
103 int secondary_offset = SecondaryOffset(old_name, old_map);
104 Entry* secondary = entry(secondary_, secondary_offset);
105 *secondary = *primary;
106 }
107
108 // Update primary cache.
109 primary->key = StrongTaggedValue(name);
110 primary->value = TaggedValue(handler);
111 primary->map = StrongTaggedValue(map);
112 isolate()->counters()->megamorphic_stub_cache_updates()->Increment();
113}

115Tagged<MaybeObject> StubCache::Get(Tagged<Name> name, Tagged<Map> map) {
116 DCHECK(CommonStubCacheChecks(this, name, map, Tagged<MaybeObject>()));
117 int primary_offset = PrimaryOffset(name, map);
118 Entry* primary = entry(primary_, primary_offset);
119 if (primary->key == name && primary->map == map) {
120 return TaggedValue::ToMaybeObject(isolate(), primary->value);
121 }
122 int secondary_offset = SecondaryOffset(name, map);
123 Entry* secondary = entry(secondary_, secondary_offset);
124 if (secondary->key == name && secondary->map == map) {
125 return TaggedValue::ToMaybeObject(isolate(), secondary->value);
126 }
127 return Tagged<MaybeObject>();
128}

130void StubCache::Clear() {
131 Tagged<MaybeObject> empty = isolate_->builtins()->code(Builtin::kIllegal);
132 Tagged<Name> empty_string = ReadOnlyRoots(isolate()).empty_string();
133 for (int i = 0; i < kPrimaryTableSize; i++) {
134 primary_[i].key = StrongTaggedValue(empty_string);
135 primary_[i].map = StrongTaggedValue(Smi::zero());
136 primary_[i].value = TaggedValue(empty);
137 }
138 for (int j = 0; j < kSecondaryTableSize; j++) {
139 secondary_[j].key = StrongTaggedValue(empty_string);
140 secondary_[j].map = StrongTaggedValue(Smi::zero());
141 secondary_[j].value = TaggedValue(empty);
142 }
143}
