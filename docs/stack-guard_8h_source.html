57 // Code locations that check for interrupts might only handle a subset of the
58 // available interrupts, expressed as an `InterruptLevel`. These levels are
59 // also associated with side effects that are allowed for the respective
60 // level. The levels are inclusive, which is specified using the order in the
61 // enum. For example, a site that handles `kAnyEffect` will also handle the
62 // preceding levels.
63 enum class InterruptLevel { kNoGC, kNoHeapWrites, kAnyEffect };
64 static constexpr int kNumberOfInterruptLevels = 3;

66#define INTERRUPT_LIST(V) \
67 V(TERMINATE_EXECUTION, TerminateExecution, 0, InterruptLevel::kNoGC) \
68 V(GC_REQUEST, GC, 1, InterruptLevel::kNoHeapWrites) \
69 V(INSTALL_CODE, InstallCode, 2, InterruptLevel::kAnyEffect) \
70 V(INSTALL_BASELINE_CODE, InstallBaselineCode, 3, InterruptLevel::kAnyEffect) \
71 V(API_INTERRUPT, ApiInterrupt, 4, InterruptLevel::kNoHeapWrites) \
72 V(DEOPT_MARKED_ALLOCATION_SITES, DeoptMarkedAllocationSites, 5, \
73 InterruptLevel::kNoHeapWrites) \
74 V(GROW_SHARED_MEMORY, GrowSharedMemory, 6, InterruptLevel::kAnyEffect) \
75 V(LOG_WASM_CODE, LogWasmCode, 7, InterruptLevel::kAnyEffect) \
76 V(WASM_CODE_GC, WasmCodeGC, 8, InterruptLevel::kNoHeapWrites) \
77 V(INSTALL_MAGLEV_CODE, InstallMaglevCode, 9, InterruptLevel::kAnyEffect) \
78 V(GLOBAL_SAFEPOINT, GlobalSafepoint, 10, InterruptLevel::kNoHeapWrites) \
79 V(START_INCREMENTAL_MARKING, StartIncrementalMarking, 11, \
80 InterruptLevel::kNoHeapWrites)

82#define V(NAME, Name, id, interrupt_level) \
83 inline bool Check##Name() { return CheckInterrupt(NAME); } \
84 inline void Request##Name() { RequestInterrupt(NAME); } \

89 // Flag used to set the interrupt causes.

91#define V(NAME, Name, id, interrupt_level) NAME = (1 << id),

201 // The stack limit has two values: the one with the real_ prefix is the
202 // actual stack limit set for the VM. The one without the real_ prefix has
203 // the same value as the actual stack limit except when there is an
204 // interruption (e.g. debug break or preemption) in which case it is lowered
205 // to make stack checks fail. Both the generated code and the runtime system
206 // check against the one without the real_ prefix.
207 // For simulator builds, we also use a separate C++ stack limit.
208
209 // Actual JavaScript stack limit set for the VM.
210 uintptr_t real_jslimit_ = kIllegalLimit;
211#ifdef USE_SIMULATOR
212 // Actual C++ stack limit set for the VM.
213 uintptr_t real_climit_ = kIllegalLimit;
214#else
215 // Padding to match the missing {real_climit_} field, renamed to make it
216 // explicit that this field is unused in this configuration. But the padding
217 // field is needed:
218 // - To keep the isolate's LinearAllocationArea fields from crossing cache
219 // lines (see Isolate::CheckIsolateLayout).
220 // - To ensure that jslimit_offset() is the same in mksnapshot and in V8:
221 // When cross-compiling V8, mksnapshot's host and target may be different
222 // even if they are the same for V8, which results in a different value for
223 // USE_SIMULATOR. Without this padding, this causes the builtins to use the
224 // wrong jslimit_offset() for stack checks.
225 uintptr_t padding1_;
226#endif
227
228 // jslimit_ and climit_ can be read without any lock.
229 // Writing requires the ExecutionAccess lock, or may be updated with a
230 // strong compare-and-swap (e.g. for stack-switching).
231 base::AtomicWord jslimit_ = kIllegalLimit;
232#ifdef USE_SIMULATOR
233 base::AtomicWord climit_ = kIllegalLimit;
234#else
235 // See {padding1_}.
236 uintptr_t padding2_;
237#endif
238
239 uintptr_t jslimit() {
240 return base::bit_cast<uintptr_t>(base::Relaxed_Load(&jslimit_));
241 }
242 void set_jslimit(uintptr_t limit) {
243 return base::Relaxed_Store(&jslimit_,
244 static_cast<base::AtomicWord>(limit));
245 }
246#ifdef USE_SIMULATOR
247 uintptr_t climit() {
248 return base::bit_cast<uintptr_t>(base::Relaxed_Load(&climit_));
249 }
250 void set_climit(uintptr_t limit) {
251 return base::Relaxed_Store(&climit_,
252 static_cast<base::AtomicWord>(limit));
253 }
254#endif

264 }
