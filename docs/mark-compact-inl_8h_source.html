51// static
52template <typename THeapObjectSlot>
53void MarkCompactCollector::RecordSlot(MemoryChunk* source_chunk,
54 THeapObjectSlot slot,
55 Tagged<HeapObject> target) {
56 MemoryChunk* target_chunk = MemoryChunk::FromHeapObject(target);
57 if (target_chunk->IsEvacuationCandidate()) {
58 MutablePageMetadata* source_page =
59 MutablePageMetadata::cast(source_chunk->Metadata());
60 if (target_chunk->IsFlagSet(MemoryChunk::IS_EXECUTABLE)) {
61 // TODO(377724745): currently needed because flags are untrusted.
62 SBXCHECK(!InsideSandbox(target_chunk->address()));
63 RememberedSet<TRUSTED_TO_CODE>::Insert<AccessMode::ATOMIC>(
64 source_page, source_chunk->Offset(slot.address()));
65 } else if (source_chunk->IsFlagSet(MemoryChunk::IS_TRUSTED) &&
66 target_chunk->IsFlagSet(MemoryChunk::IS_TRUSTED)) {
67 // TODO(377724745): currently needed because flags are untrusted.
68 SBXCHECK(!InsideSandbox(target_chunk->address()));
69 RememberedSet<TRUSTED_TO_TRUSTED>::Insert<AccessMode::ATOMIC>(
70 source_page, source_chunk->Offset(slot.address()));
71 } else if (V8_LIKELY(!target_chunk->InWritableSharedSpace()) ||
72 source_page->heap()->isolate()->is_shared_space_isolate()) {
73 DCHECK_EQ(source_page->heap(), target_chunk->GetHeap());
74 RememberedSet<OLD_TO_OLD>::Insert<AccessMode::ATOMIC>(
75 source_page, source_chunk->Offset(slot.address()));
76 } else {
77 // DCHECK here that we only don't record in case of local->shared
78 // references in a client GC.
79 DCHECK(!source_page->heap()->isolate()->is_shared_space_isolate());
80 DCHECK(target_chunk->GetHeap()->isolate()->is_shared_space_isolate());
81 DCHECK(target_chunk->InWritableSharedSpace());
82 }
83 }

104void RootMarkingVisitor::MarkObjectByPointer(Root root, FullObjectSlot p) {
105 Tagged<Object> object = *p;
106#ifdef V8_ENABLE_DIRECT_HANDLE
107 if (object.ptr() == kTaggedNullAddress) return;
108#endif
109 if (!IsHeapObject(object)) return;
110 Tagged<HeapObject> heap_object = Cast<HeapObject>(object);
111 const auto target_worklist =
112 MarkingHelper::ShouldMarkObject(collector_->heap(), heap_object);
113 if (!target_worklist) {
114 return;
115 }
116 collector_->MarkRootObject(root, heap_object, target_worklist.value());
117}
