84// A simple forward iterator for bytecodes.
85class V8_EXPORT_PRIVATE BytecodeIterator : public NON_EXPORTED_BASE(Decoder) {
86 // Base class for both iterators defined below.
87 class iterator_base {
88 public:
89 iterator_base& operator++() {
90 DCHECK_LT(ptr_, end_);
91 ptr_ += OpcodeLength(ptr_, end_);
92 return *this;
93 }
94 bool operator==(const iterator_base& that) const {
95 return this->ptr_ == that.ptr_;
96 }
97 bool operator!=(const iterator_base& that) const {
98 return this->ptr_ != that.ptr_;
99 }
100
101 protected:
102 const uint8_t* ptr_;
103 const uint8_t* end_;
104 iterator_base(const uint8_t* ptr, const uint8_t* end)
105 : ptr_(ptr), end_(end) {}
106 };
107
108 public:
109 // If one wants to iterate over the bytecode without looking at {pc_offset()}.
110 class opcode_iterator
111 : public iterator_base,
112 public base::iterator<std::input_iterator_tag, WasmOpcode> {
113 public:
114 WasmOpcode operator*() {
115 DCHECK_LT(ptr_, end_);
116 return static_cast<WasmOpcode>(*ptr_);
117 }
118
119 private:
120 friend class BytecodeIterator;
121 opcode_iterator(const uint8_t* ptr, const uint8_t* end)
122 : iterator_base(ptr, end) {}
123 };
124 // If one wants to iterate over the instruction offsets without looking at
125 // opcodes.
126 class offset_iterator
127 : public iterator_base,
128 public base::iterator<std::input_iterator_tag, uint32_t> {
129 public:
130 uint32_t operator*() {
131 DCHECK_LT(ptr_, end_);
132 return static_cast<uint32_t>(ptr_ - start_);
133 }
134
135 private:
136 const uint8_t* start_;
137 friend class BytecodeIterator;
138 offset_iterator(const uint8_t* start, const uint8_t* ptr,
139 const uint8_t* end)
140 : iterator_base(ptr, end), start_(start) {}
141 };
142
143 // Create a new {BytecodeIterator}, starting after the locals declarations.
144 BytecodeIterator(const uint8_t* start, const uint8_t* end);
145
146 // Create a new {BytecodeIterator}, starting with locals declarations.
147 BytecodeIterator(const uint8_t* start, const uint8_t* end,
148 BodyLocalDecls* decls, Zone* zone);
149
150 base::iterator_range<opcode_iterator> opcodes() const {
151 return base::iterator_range<opcode_iterator>(opcode_iterator(pc_, end_),
152 opcode_iterator(end_, end_));
153 }
154
155 base::iterator_range<offset_iterator> offsets() const {
156 return base::iterator_range<offset_iterator>(
157 offset_iterator(start_, pc_, end_),
158 offset_iterator(start_, end_, end_));
159 }
160
161 WasmOpcode current() {
162 return static_cast<WasmOpcode>(
163 read_u8<Decoder::NoValidationTag>(pc_, "expected bytecode"));
164 }
165
166 void next() {
167 if (pc_ < end_) {
168 pc_ += OpcodeLength(pc_, end_);
169 if (pc_ >= end_) pc_ = end_;
170 }
171 }
172
173 bool has_next() const { return pc_ < end_; }
174
175 WasmOpcode prefixed_opcode() {
176 auto [opcode, length] = read_prefixed_opcode<Decoder::NoValidationTag>(pc_);
177 return opcode;
178 }
179
180 const uint8_t* pc() const { return pc_; }
181};
