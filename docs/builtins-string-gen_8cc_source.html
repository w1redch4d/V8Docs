25TNode<RawPtrT> StringBuiltinsAssembler::DirectStringData(
26 TNode<String> string, TNode<Word32T> string_instance_type) {
27 // Compute the effective offset of the first character.
28 TVARIABLE(RawPtrT, var_data);
29 Label if_sequential(this), if_external(this), if_join(this);
30 Branch(Word32Equal(Word32And(string_instance_type,
31 Int32Constant(kStringRepresentationMask)),
32 Int32Constant(kSeqStringTag)),
33 &if_sequential, &if_external);
34
35 BIND(&if_sequential);
36 {
37 static_assert(OFFSET_OF_DATA_START(SeqOneByteString) ==
38 OFFSET_OF_DATA_START(SeqTwoByteString));
39 var_data = RawPtrAdd(ReinterpretCast<RawPtrT>(BitcastTaggedToWord(string)),
40 IntPtrConstant(OFFSET_OF_DATA_START(SeqOneByteString) -
41 kHeapObjectTag));
42 Goto(&if_join);
43 }
44
45 BIND(&if_external);
46 {
47 var_data = LoadExternalStringResourceDataPtr(CAST(string));
48 Goto(&if_join);
49 }
50
51 BIND(&if_join);
52 return var_data.value();
53}

131void StringBuiltinsAssembler::GenerateStringEqual(TNode<String> left,
132 TNode<String> right,
133 TNode<IntPtrT> length) {
134 TVARIABLE(String, var_left, left);
135 TVARIABLE(String, var_right, right);
136 Label if_equal(this), if_notequal(this), if_indirect(this, Label::kDeferred),
137 start(this, {&var_left, &var_right});
138
139 // Callers must handle the case where {lhs} and {rhs} refer to the same
140 // String object.
141 CSA_DCHECK(this, TaggedNotEqual(left, right));
142
143 CSA_DCHECK(this, IntPtrEqual(LoadStringLengthAsWord(left), length));
144 CSA_DCHECK(this, IntPtrEqual(LoadStringLengthAsWord(right), length));
145
146 Goto(&start);
147 BIND(&start);
148 TNode<String> lhs = var_left.value();
149 TNode<String> rhs = var_right.value();
150
151 TNode<Uint16T> lhs_instance_type = LoadInstanceType(lhs);
152 TNode<Uint16T> rhs_instance_type = LoadInstanceType(rhs);
153
154 StringEqual_Core(lhs, lhs_instance_type, rhs, rhs_instance_type, length,
155 &if_equal, &if_notequal, &if_indirect);
156
157 BIND(&if_indirect);
158 {
159 Label restart(this, {&var_left, &var_right});
160 // Try to unwrap indirect strings, restart the above attempt on success.
161 MaybeDerefIndirectStrings(&var_left, lhs_instance_type, &var_right,
162 rhs_instance_type, &restart);
163
164 TailCallRuntime(Runtime::kStringEqual, NoContextConstant(), lhs, rhs);
165
166 BIND(&restart);
167 GotoIf(TaggedEqual(var_left.value(), var_right.value()), &if_equal);
168 Goto(&start);
169 }
170
171 BIND(&if_equal);
172 Return(TrueConstant());
173
174 BIND(&if_notequal);
175 Return(FalseConstant());
176}

178void StringBuiltinsAssembler::StringEqual_Core(
179 TNode<String> lhs, TNode<Word32T> lhs_instance_type, TNode<String> rhs,
180 TNode<Word32T> rhs_instance_type, TNode<IntPtrT> length, Label* if_equal,
181 Label* if_not_equal, Label* if_indirect) {
182 CSA_DCHECK(this, WordEqual(LoadStringLengthAsWord(lhs), length));
183 CSA_DCHECK(this, WordEqual(LoadStringLengthAsWord(rhs), length));
184
185 // Callers must handle the case where {lhs} and {rhs} refer to the same
186 // String object.
187 CSA_DCHECK(this, TaggedNotEqual(lhs, rhs));
188
189 // Combine the instance types into a single 16-bit value, so we can check
190 // both of them at once.
191 TNode<Word32T> both_instance_types = Word32Or(
192 lhs_instance_type, Word32Shl(rhs_instance_type, Int32Constant(8)));
193
194 // Check if both {lhs} and {rhs} are internalized. Since we already know
195 // that they're not the same object, they're not equal in that case.
196 int const kBothInternalizedMask =
197 kIsNotInternalizedMask | (kIsNotInternalizedMask << 8);
198 int const kBothInternalizedTag = kInternalizedTag | (kInternalizedTag << 8);
199 GotoIf(Word32Equal(Word32And(both_instance_types,
200 Int32Constant(kBothInternalizedMask)),
201 Int32Constant(kBothInternalizedTag)),
202 if_not_equal);
203
204 // Check if both {lhs} and {rhs} are direct strings, and that in case of
205 // ExternalStrings the data pointer is cached.
206 static_assert(kUncachedExternalStringTag != 0);
207 static_assert(kIsIndirectStringTag != 0);
208 int const kBothDirectStringMask =
209 kIsIndirectStringMask | kUncachedExternalStringMask |
210 ((kIsIndirectStringMask | kUncachedExternalStringMask) << 8);
211 GotoIfNot(Word32Equal(Word32And(both_instance_types,
212 Int32Constant(kBothDirectStringMask)),
213 Int32Constant(0)),
214 if_indirect);
215
216 Label if_skip_fast_case(this), if_fast_case(this), if_oneonebytestring(this),
217 if_twotwobytestring(this), if_onetwobytestring(this),
218 if_twoonebytestring(this);
219
220 // Dispatch based on the {lhs} and {rhs} string encoding.
221 int const kBothStringEncodingMask =
222 kStringEncodingMask | (kStringEncodingMask << 8);
223 int const kBothExternalStringTag =
224 kExternalStringTag | (kExternalStringTag << 8);
225 int const kOneOneByteStringTag = kOneByteStringTag | (kOneByteStringTag << 8);
226 int const kTwoTwoByteStringTag = kTwoByteStringTag | (kTwoByteStringTag << 8);
227 int const kOneTwoByteStringTag = kOneByteStringTag | (kTwoByteStringTag << 8);
228
229 TNode<Word32T> masked_instance_types =
230 Word32And(both_instance_types, Int32Constant(kBothStringEncodingMask));
231 TNode<Word32T> both_are_one_byte =
232 Word32Equal(masked_instance_types, Int32Constant(kOneOneByteStringTag));
233 TNode<Word32T> both_are_two_byte =
234 Word32Equal(masked_instance_types, Int32Constant(kTwoTwoByteStringTag));
235
236 // If both strings are not external we know that their payload length is
237 // kTagged sized. When they have the same type we can compare in chunks. The
238 // padding bytes are set to zero.
239 GotoIf(Word32And(both_instance_types, Int32Constant(kBothExternalStringTag)),
240 &if_skip_fast_case);
241 TVARIABLE(IntPtrT, byte_length, length);
242 GotoIf(both_are_one_byte, &if_fast_case);
243 byte_length = WordShl(byte_length.value(), IntPtrConstant(1));
244 Branch(both_are_two_byte, &if_fast_case, &if_skip_fast_case);
245 BIND(&if_fast_case);
246 StringEqual_FastLoop(lhs, lhs_instance_type, rhs, rhs_instance_type,
247 byte_length.value(), if_equal, if_not_equal);
248
249 BIND(&if_skip_fast_case);
250 GotoIf(both_are_one_byte, &if_oneonebytestring);
251 GotoIf(both_are_two_byte, &if_twotwobytestring);
252 Branch(
253 Word32Equal(masked_instance_types, Int32Constant(kOneTwoByteStringTag)),
254 &if_onetwobytestring, &if_twoonebytestring);
255
256 BIND(&if_oneonebytestring);
257 StringEqual_Loop(lhs, lhs_instance_type, MachineType::Uint8(), rhs,
258 rhs_instance_type, MachineType::Uint8(), length, if_equal,
259 if_not_equal);
260
261 BIND(&if_twotwobytestring);
262 StringEqual_Loop(lhs, lhs_instance_type, MachineType::Uint16(), rhs,
263 rhs_instance_type, MachineType::Uint16(), length, if_equal,
264 if_not_equal);
265
266 BIND(&if_onetwobytestring);
267 StringEqual_Loop(lhs, lhs_instance_type, MachineType::Uint8(), rhs,
268 rhs_instance_type, MachineType::Uint16(), length, if_equal,
269 if_not_equal);
270
271 BIND(&if_twoonebytestring);
272 StringEqual_Loop(lhs, lhs_instance_type, MachineType::Uint16(), rhs,
273 rhs_instance_type, MachineType::Uint8(), length, if_equal,
274 if_not_equal);
275}

277void StringBuiltinsAssembler::StringEqual_FastLoop(
278 TNode<String> lhs, TNode<Word32T> lhs_instance_type, TNode<String> rhs,
279 TNode<Word32T> rhs_instance_type, TNode<IntPtrT> byte_length,
280 Label* if_equal, Label* if_not_equal) {
281 TNode<RawPtrT> lhs_data = DirectStringData(lhs, lhs_instance_type);
282 TNode<RawPtrT> rhs_data = DirectStringData(rhs, rhs_instance_type);
283
284 const int kChunk = kTaggedSize;
285 static_assert(kObjectAlignment % kChunk == 0);
286 // Round up the byte_length to `ceiling(length / kChunk) * kChunk`
287 TNode<IntPtrT> rounded_up_len = UncheckedCast<IntPtrT>(WordAnd(
288 UncheckedCast<WordT>(IntPtrAdd(byte_length, IntPtrConstant(kChunk - 1))),
289 UncheckedCast<WordT>(IntPtrConstant(~(kChunk - 1)))));
290 TNode<RawPtrT> lhs_end = RawPtrAdd(lhs_data, rounded_up_len);
291
292#ifdef ENABLE_SLOW_DCHECKS
293 // The padding must be zeroed for chunked comparison to be correct. This loop
294 // checks all bytes being 0 from byte_length up to rounded_up_len.
295 // If we ever stop zeroing the padding, GenerateStringRelationalComparison
296 // below will also need to be updated.
297 {
298 TVARIABLE(IntPtrT, var_padding_offset, byte_length);
299 Label loop(this, &var_padding_offset), loop_end(this);
300 Goto(&loop);
301 BIND(&loop);
302 {
303 GotoIf(WordEqual(var_padding_offset.value(), rounded_up_len), &loop_end);
304
305 // Load the next byte
306 TNode<Word32T> lhs_value = UncheckedCast<Word32T>(Load(
307 MachineType::Uint8(), lhs_data,
308 WordShl(var_padding_offset.value(),
309 ElementSizeLog2Of(MachineType::Uint8().representation()))));
310 TNode<Word32T> rhs_value = UncheckedCast<Word32T>(Load(
311 MachineType::Uint8(), rhs_data,
312 WordShl(var_padding_offset.value(),
313 ElementSizeLog2Of(MachineType::Uint8().representation()))));
314
315 // Check the padding is zero.
316 CSA_CHECK(this, Word32Equal(lhs_value, Int32Constant(0)));
317 CSA_CHECK(this, Word32Equal(rhs_value, Int32Constant(0)));
318
319 // Advance to next byte.
320 var_padding_offset =
321 IntPtrAdd(var_padding_offset.value(), IntPtrConstant(1));
322 Goto(&loop);
323 }
324 BIND(&loop_end);
325 }
326#endif // ENABLE_SLOW_DCHECKS
327
328 // Compare strings in chunks of either 4 or 8 bytes, depending on the
329 // alignment of allocations.
330 static_assert(kChunk == ElementSizeInBytes(MachineRepresentation::kWord64) ||
331 kChunk == ElementSizeInBytes(MachineRepresentation::kWord32));
332 TVARIABLE(RawPtrT, rhs_ptr, rhs_data);
333 VariableList vars({&rhs_ptr}, zone());
334
335 if (kChunk == ElementSizeInBytes(MachineRepresentation::kWord64)) {
336 BuildFastLoop<RawPtrT>(
337 vars, lhs_data, lhs_end,
338 [&](TNode<RawPtrT> lhs_ptr) {
339 TNode<Word64T> lhs_value = Load<Uint64T>(lhs_ptr);
340 TNode<Word64T> rhs_value = Load<Uint64T>(rhs_ptr.value());
341 GotoIf(Word64NotEqual(lhs_value, rhs_value), if_not_equal);
342
343 // Advance {rhs_ptr} to next characters. {lhs_ptr} will be
344 // advanced along loop's {var_index}.
345 Increment(&rhs_ptr, kChunk);
346 },
347 kChunk, LoopUnrollingMode::kYes, IndexAdvanceMode::kPost);
348 } else {
349 BuildFastLoop<RawPtrT>(
350 vars, lhs_data, lhs_end,
351 [&](TNode<RawPtrT> lhs_ptr) {
352 TNode<Word32T> lhs_value = Load<Uint32T>(lhs_ptr);
353 TNode<Word32T> rhs_value = Load<Uint32T>(rhs_ptr.value());
354 GotoIf(Word32NotEqual(lhs_value, rhs_value), if_not_equal);
355
356 // Advance {rhs_ptr} to next characters. {lhs_ptr} will be
357 // advanced along loop's {var_index}.
358 Increment(&rhs_ptr, kChunk);
359 },
360 kChunk, LoopUnrollingMode::kYes, IndexAdvanceMode::kPost);
361 }
362 Goto(if_equal);
363}

365void StringBuiltinsAssembler::StringEqual_Loop(
366 TNode<String> lhs, TNode<Word32T> lhs_instance_type, MachineType lhs_type,
367 TNode<String> rhs, TNode<Word32T> rhs_instance_type, MachineType rhs_type,
368 TNode<IntPtrT> length, Label* if_equal, Label* if_not_equal) {
369 Comment("StringEqual_Loop");
370 CSA_DCHECK(this, WordEqual(LoadStringLengthAsWord(lhs), length));
371 CSA_DCHECK(this, WordEqual(LoadStringLengthAsWord(rhs), length));
372
373 // Compute the effective offset of the first character.
374 TNode<RawPtrT> lhs_data = DirectStringData(lhs, lhs_instance_type);
375 TNode<RawPtrT> rhs_data = DirectStringData(rhs, rhs_instance_type);
376 TNode<RawPtrT> lhs_end =
377 RawPtrAdd(lhs_data, WordShl(length, IntPtrConstant(ElementSizeLog2Of(
378 lhs_type.representation()))));
379 TVARIABLE(RawPtrT, rhs_ptr, rhs_data);
380 VariableList vars({&rhs_ptr}, zone());
381
382 // Loop over the {lhs} and {rhs} strings to see if they are equal.
383 BuildFastLoop<RawPtrT>(
384 vars, lhs_data, lhs_end,
385 [&](TNode<RawPtrT> lhs_ptr) {
386 TNode<Word32T> lhs_value =
387 UncheckedCast<Word32T>(Load(lhs_type, lhs_ptr));
388 TNode<Word32T> rhs_value =
389 UncheckedCast<Word32T>(Load(rhs_type, rhs_ptr.value()));
390
391 // Check if the characters match.
392 GotoIf(Word32NotEqual(lhs_value, rhs_value), if_not_equal);
393
394 // Advance {rhs_ptr} to next characters. {lhs_ptr} will be
395 // advanced along loop's {var_index}.
396 Increment(&rhs_ptr, ElementSizeInBytes(rhs_type.representation()));
397 },
398 ElementSizeInBytes(lhs_type.representation()), LoopUnrollingMode::kNo,
399 IndexAdvanceMode::kPost);
400
401 // All characters are checked and no difference was found, so the strings
402 // are equal.
403 Goto(if_equal);
404}

483TNode<String> StringBuiltinsAssembler::StringAdd(
484 TNode<ContextOrEmptyContext> context, TNode<String> left,
485 TNode<String> right) {
486 CSA_DCHECK(this, IsZeroOrContext(context));
487
488 TVARIABLE(String, result);
489 Label check_right(this), runtime(this, Label::kDeferred), cons(this),
490 done(this, &result);
491
492 TNode<Uint32T> left_length = LoadStringLengthAsWord32(left);
493 GotoIfNot(Word32Equal(left_length, Uint32Constant(0)), &check_right);
494 result = right;
495 Goto(&done);
496
497 BIND(&check_right);
498 TNode<Uint32T> right_length = LoadStringLengthAsWord32(right);
499 GotoIfNot(Word32Equal(right_length, Uint32Constant(0)), &cons);
500 result = left;
501 Goto(&done);
502
503 BIND(&cons);
504 {
505 TNode<Uint32T> new_length = Uint32Add(left_length, right_length);
506
507 // If new length is greater than String::kMaxLength, goto runtime to
508 // throw. Note: we also need to invalidate the string length protector, so
509 // can't just throw here directly.
510 GotoIf(Uint32GreaterThan(new_length, Uint32Constant(String::kMaxLength)),
511 &runtime);
512
513 TVARIABLE(String, var_left, left);
514 TVARIABLE(String, var_right, right);
515 Label non_cons(this, {&var_left, &var_right});
516 Label slow(this, Label::kDeferred);
517 GotoIf(Uint32LessThan(new_length, Uint32Constant(ConsString::kMinLength)),
518 &non_cons);
519
520 result =
521 AllocateConsString(new_length, var_left.value(), var_right.value());
522 Goto(&done);
523
524 BIND(&non_cons);
525
526 Comment("Full string concatenate");
527 TNode<Int32T> left_instance_type = LoadInstanceType(var_left.value());
528 TNode<Int32T> right_instance_type = LoadInstanceType(var_right.value());
529 // Compute intersection and difference of instance types.
530
531 TNode<Int32T> ored_instance_types =
532 Word32Or(left_instance_type, right_instance_type);
533 TNode<Word32T> xored_instance_types =
534 Word32Xor(left_instance_type, right_instance_type);
535
536 // Check if both strings have the same encoding and both are sequential.
537 GotoIf(IsSetWord32(xored_instance_types, kStringEncodingMask), &runtime);
538 GotoIf(IsSetWord32(ored_instance_types, kStringRepresentationMask), &slow);
539
540 TNode<IntPtrT> word_left_length = Signed(ChangeUint32ToWord(left_length));
541 TNode<IntPtrT> word_right_length = Signed(ChangeUint32ToWord(right_length));
542
543 Label two_byte(this);
544 GotoIf(Word32Equal(Word32And(ored_instance_types,
545 Int32Constant(kStringEncodingMask)),
546 Int32Constant(kTwoByteStringTag)),
547 &two_byte);
548 // One-byte sequential string case
549 result = AllocateNonEmptySeqOneByteString(new_length);
550 CopyStringCharacters(var_left.value(), result.value(), IntPtrConstant(0),
551 IntPtrConstant(0), word_left_length,
552 String::ONE_BYTE_ENCODING, String::ONE_BYTE_ENCODING);
553 CopyStringCharacters(var_right.value(), result.value(), IntPtrConstant(0),
554 word_left_length, word_right_length,
555 String::ONE_BYTE_ENCODING, String::ONE_BYTE_ENCODING);
556 Goto(&done);
557
558 BIND(&two_byte);
559 {
560 // Two-byte sequential string case
561 result = AllocateNonEmptySeqTwoByteString(new_length);
562 CopyStringCharacters(var_left.value(), result.value(), IntPtrConstant(0),
563 IntPtrConstant(0), word_left_length,
564 String::TWO_BYTE_ENCODING,
565 String::TWO_BYTE_ENCODING);
566 CopyStringCharacters(var_right.value(), result.value(), IntPtrConstant(0),
567 word_left_length, word_right_length,
568 String::TWO_BYTE_ENCODING,
569 String::TWO_BYTE_ENCODING);
570 Goto(&done);
571 }
572
573 BIND(&slow);
574 {
575 // Try to unwrap indirect strings, restart the above attempt on success.
576 MaybeDerefIndirectStrings(&var_left, left_instance_type, &var_right,
577 right_instance_type, &non_cons);
578 Goto(&runtime);
579 }
580 }
581 BIND(&runtime);
582 {
583 result = CAST(CallRuntime(Runtime::kStringAdd, context, left, right));
584 Goto(&done);
585 }
586
587 BIND(&done);
588 return result.value();
589}

689void StringBuiltinsAssembler::GenerateStringRelationalComparison(
690 TNode<String> left, TNode<String> right, StringComparison op) {
691 TVARIABLE(String, var_left, left);
692 TVARIABLE(String, var_right, right);
693
694 Label if_less(this), if_equal(this), if_greater(this);
695 Label restart(this, {&var_left, &var_right});
696 Goto(&restart);
697 BIND(&restart);
698
699 TNode<String> lhs = var_left.value();
700 TNode<String> rhs = var_right.value();
701 // Fast check to see if {lhs} and {rhs} refer to the same String object.
702 GotoIf(TaggedEqual(lhs, rhs), &if_equal);
703
704 // Load instance types of {lhs} and {rhs}.
705 TNode<Uint16T> lhs_instance_type = LoadInstanceType(lhs);
706 TNode<Uint16T> rhs_instance_type = LoadInstanceType(rhs);
707
708 // Combine the instance types into a single 16-bit value, so we can check
709 // both of them at once.
710 TNode<Int32T> both_instance_types = Word32Or(
711 lhs_instance_type, Word32Shl(rhs_instance_type, Int32Constant(8)));
712
713 // Check that both {lhs} and {rhs} are flat one-byte strings.
714 int const kBothSeqOneByteStringMask =
715 kStringEncodingMask | kStringRepresentationMask |
716 ((kStringEncodingMask | kStringRepresentationMask) << 8);
717 int const kBothSeqOneByteStringTag =
718 kOneByteStringTag | kSeqStringTag |
719 ((kOneByteStringTag | kSeqStringTag) << 8);
720 Label if_bothonebyteseqstrings(this), if_notbothonebyteseqstrings(this);
721 Branch(Word32Equal(Word32And(both_instance_types,
722 Int32Constant(kBothSeqOneByteStringMask)),
723 Int32Constant(kBothSeqOneByteStringTag)),
724 &if_bothonebyteseqstrings, &if_notbothonebyteseqstrings);
725
726 BIND(&if_bothonebyteseqstrings);
727 {
728 TNode<IntPtrT> lhs_length = LoadStringLengthAsWord(lhs);
729 TNode<IntPtrT> rhs_length = LoadStringLengthAsWord(rhs);
730
731 TNode<IntPtrT> length = IntPtrMin(lhs_length, rhs_length);
732
733 // Loop over the {lhs} and {rhs} strings to see if they are equal.
734 constexpr int kBeginOffset =
735 OFFSET_OF_DATA_START(SeqOneByteString) - kHeapObjectTag;
736 TNode<IntPtrT> begin = IntPtrConstant(kBeginOffset);
737 TNode<IntPtrT> end = IntPtrAdd(begin, length);
738 TVARIABLE(IntPtrT, var_offset, begin);
739 Label chunk_loop(this, &var_offset), char_loop(this, &var_offset);
740 Label if_done(this);
741
742 // Unrolled first iteration.
743 GotoIf(IntPtrEqual(length, IntPtrConstant(0)), &if_done);
744
745 constexpr int kChunkSize = kTaggedSize;
746 static_assert(
747 kChunkSize == ElementSizeInBytes(MachineRepresentation::kWord64) ||
748 kChunkSize == ElementSizeInBytes(MachineRepresentation::kWord32));
749 if (kChunkSize == ElementSizeInBytes(MachineRepresentation::kWord32)) {
750 TNode<Uint32T> lhs_chunk =
751 Load<Uint32T>(lhs, IntPtrConstant(kBeginOffset));
752 TNode<Uint32T> rhs_chunk =
753 Load<Uint32T>(rhs, IntPtrConstant(kBeginOffset));
754 GotoIf(Word32NotEqual(lhs_chunk, rhs_chunk), &char_loop);
755 } else {
756 TNode<Uint64T> lhs_chunk =
757 Load<Uint64T>(lhs, IntPtrConstant(kBeginOffset));
758 TNode<Uint64T> rhs_chunk =
759 Load<Uint64T>(rhs, IntPtrConstant(kBeginOffset));
760 GotoIf(Word64NotEqual(lhs_chunk, rhs_chunk), &char_loop);
761 }
762
763 var_offset = IntPtrConstant(OFFSET_OF_DATA_START(SeqOneByteString) -
764 kHeapObjectTag + kChunkSize);
765
766 Goto(&chunk_loop);
767
768 // Try skipping over chunks of kChunkSize identical characters.
769 // This depends on padding (between strings' lengths and the actual end
770 // of the heap object) being zeroed out.
771 BIND(&chunk_loop);
772 {
773 GotoIf(IntPtrGreaterThanOrEqual(var_offset.value(), end), &if_done);
774
775 if (kChunkSize == ElementSizeInBytes(MachineRepresentation::kWord32)) {
776 TNode<Uint32T> lhs_chunk = Load<Uint32T>(lhs, var_offset.value());
777 TNode<Uint32T> rhs_chunk = Load<Uint32T>(rhs, var_offset.value());
778 GotoIf(Word32NotEqual(lhs_chunk, rhs_chunk), &char_loop);
779 } else {
780 TNode<Uint64T> lhs_chunk = Load<Uint64T>(lhs, var_offset.value());
781 TNode<Uint64T> rhs_chunk = Load<Uint64T>(rhs, var_offset.value());
782 GotoIf(Word64NotEqual(lhs_chunk, rhs_chunk), &char_loop);
783 }
784
785 var_offset = IntPtrAdd(var_offset.value(), IntPtrConstant(kChunkSize));
786 Goto(&chunk_loop);
787 }
788
789 BIND(&char_loop);
790 {
791 GotoIf(WordEqual(var_offset.value(), end), &if_done);
792
793 TNode<Uint8T> lhs_char = Load<Uint8T>(lhs, var_offset.value());
794 TNode<Uint8T> rhs_char = Load<Uint8T>(rhs, var_offset.value());
795
796 Label if_charsdiffer(this);
797 GotoIf(Word32NotEqual(lhs_char, rhs_char), &if_charsdiffer);
798
799 var_offset = IntPtrAdd(var_offset.value(), IntPtrConstant(1));
800 Goto(&char_loop);
801
802 BIND(&if_charsdiffer);
803 Branch(Uint32LessThan(lhs_char, rhs_char), &if_less, &if_greater);
804 }
805
806 BIND(&if_done);
807 {
808 // All characters up to the min length are equal, decide based on
809 // string length.
810 GotoIf(IntPtrEqual(lhs_length, rhs_length), &if_equal);
811 Branch(IntPtrLessThan(lhs_length, rhs_length), &if_less, &if_greater);
812 }
813 }
814
815 BIND(&if_notbothonebyteseqstrings);
816 {
817 // Try to unwrap indirect strings, restart the above attempt on success.
818 MaybeDerefIndirectStrings(&var_left, lhs_instance_type, &var_right,
819 rhs_instance_type, &restart);
820 // TODO(bmeurer): Add support for two byte string relational comparisons.
821 switch (op) {
822 case StringComparison::kLessThan:
823 TailCallRuntime(Runtime::kStringLessThan, NoContextConstant(), lhs,
824 rhs);
825 break;
826 case StringComparison::kLessThanOrEqual:
827 TailCallRuntime(Runtime::kStringLessThanOrEqual, NoContextConstant(),
828 lhs, rhs);
829 break;
830 case StringComparison::kGreaterThan:
831 TailCallRuntime(Runtime::kStringGreaterThan, NoContextConstant(), lhs,
832 rhs);
833 break;
834 case StringComparison::kGreaterThanOrEqual:
835 TailCallRuntime(Runtime::kStringGreaterThanOrEqual, NoContextConstant(),
836 lhs, rhs);
837 break;
838 case StringComparison::kCompare:
839 TailCallRuntime(Runtime::kStringCompare, NoContextConstant(), lhs, rhs);
840 break;
841 }
842 }
843
844 BIND(&if_less);
845 switch (op) {
846 case StringComparison::kLessThan:
847 case StringComparison::kLessThanOrEqual:
848 Return(TrueConstant());
849 break;
850
851 case StringComparison::kGreaterThan:
852 case StringComparison::kGreaterThanOrEqual:
853 Return(FalseConstant());
854 break;
855
856 case StringComparison::kCompare:
857 Return(SmiConstant(-1));
858 break;
859 }
860
861 BIND(&if_equal);
862 switch (op) {
863 case StringComparison::kLessThan:
864 case StringComparison::kGreaterThan:
865 Return(FalseConstant());
866 break;
867
868 case StringComparison::kLessThanOrEqual:
869 case StringComparison::kGreaterThanOrEqual:
870 Return(TrueConstant());
871 break;
872
873 case StringComparison::kCompare:
874 Return(SmiConstant(0));
875 break;
876 }
877
878 BIND(&if_greater);
879 switch (op) {
880 case StringComparison::kLessThan:
881 case StringComparison::kLessThanOrEqual:
882 Return(FalseConstant());
883 break;
884
885 case StringComparison::kGreaterThan:
886 case StringComparison::kGreaterThanOrEqual:
887 Return(TrueConstant());
888 break;
889
890 case StringComparison::kCompare:
891 Return(SmiConstant(1));
892 break;
893 }
894}

961// NOTE: This needs to be kept in sync with the Turboshaft implementation in

963TF_BUILTIN(StringFromCharCode, StringBuiltinsAssembler) {
964 // TODO(ishell): use constants from Descriptor once the JSFunction linkage
965 // arguments are reordered.
966 auto argc = UncheckedParameter<Int32T>(Descriptor::kJSActualArgumentsCount);
967 auto context = Parameter<Context>(Descriptor::kContext);
968
969 CodeStubArguments arguments(this, argc);
970 TNode<Uint32T> unsigned_argc =
971 Unsigned(TruncateIntPtrToInt32(arguments.GetLengthWithoutReceiver()));
972 // Check if we have exactly one argument (plus the implicit receiver), i.e.
973 // if the parent frame is not an inlined arguments frame.
974 Label if_oneargument(this), if_notoneargument(this);
975 Branch(IntPtrEqual(arguments.GetLengthWithoutReceiver(), IntPtrConstant(1)),
976 &if_oneargument, &if_notoneargument);
977
978 BIND(&if_oneargument);
979 {
980 // Single argument case, perform fast single character string cache lookup
981 // for one-byte code units, or fall back to creating a single character
982 // string on the fly otherwise.
983 TNode<Object> code = arguments.AtIndex(0);
984 TNode<Word32T> code32 = TruncateTaggedToWord32(context, code);
985 TNode<Int32T> code16 =
986 Signed(Word32And(code32, Int32Constant(String::kMaxUtf16CodeUnit)));
987 TNode<String> result = StringFromSingleCharCode(code16);
988 arguments.PopAndReturn(result);
989 }
990
991 TNode<Word32T> code16;
992 BIND(&if_notoneargument);
993 {
994 Label two_byte(this);
995 // Assume that the resulting string contains only one-byte characters.
996 TNode<String> one_byte_result = AllocateSeqOneByteString(unsigned_argc);
997
998 TVARIABLE(IntPtrT, var_max_index, IntPtrConstant(0));
999
1000 // Iterate over the incoming arguments, converting them to 8-bit character
1001 // codes. Stop if any of the conversions generates a code that doesn't fit
1002 // in 8 bits.
1003 CodeStubAssembler::VariableList vars({&var_max_index}, zone());
1004 arguments.ForEach(vars, [&](TNode<Object> arg) {
1005 TNode<Word32T> code32 = TruncateTaggedToWord32(context, arg);
1006 code16 = Word32And(code32, Int32Constant(String::kMaxUtf16CodeUnit));
1007
1008 GotoIf(
1009 Int32GreaterThan(code16, Int32Constant(String::kMaxOneByteCharCode)),
1010 &two_byte);
1011
1012 // The {code16} fits into the SeqOneByteString {one_byte_result}.
1013 TNode<IntPtrT> offset = ElementOffsetFromIndex(
1014 var_max_index.value(), UINT8_ELEMENTS,
1015 OFFSET_OF_DATA_START(SeqOneByteString) - kHeapObjectTag);
1016 StoreNoWriteBarrier(MachineRepresentation::kWord8, one_byte_result,
1017 offset, code16);
1018 var_max_index = IntPtrAdd(var_max_index.value(), IntPtrConstant(1));
1019 });
1020 arguments.PopAndReturn(one_byte_result);
1021
1022 BIND(&two_byte);
1023
1024 // At least one of the characters in the string requires a 16-bit
1025 // representation. Allocate a SeqTwoByteString to hold the resulting
1026 // string.
1027 TNode<String> two_byte_result = AllocateSeqTwoByteString(unsigned_argc);
1028
1029 // Copy the characters that have already been put in the 8-bit string into
1030 // their corresponding positions in the new 16-bit string.
1031 TNode<IntPtrT> zero = IntPtrConstant(0);
1032 CopyStringCharacters(one_byte_result, two_byte_result, zero, zero,
1033 var_max_index.value(), String::ONE_BYTE_ENCODING,
1034 String::TWO_BYTE_ENCODING);
1035
1036 // Write the character that caused the 8-bit to 16-bit fault.
1037 TNode<IntPtrT> max_index_offset = ElementOffsetFromIndex(
1038 var_max_index.value(), UINT16_ELEMENTS,
1039 OFFSET_OF_DATA_START(SeqTwoByteString) - kHeapObjectTag);
1040 StoreNoWriteBarrier(MachineRepresentation::kWord16, two_byte_result,
1041 max_index_offset, code16);
1042 var_max_index = IntPtrAdd(var_max_index.value(), IntPtrConstant(1));
1043
1044 // Resume copying the passed-in arguments from the same place where the
1045 // 8-bit copy stopped, but this time copying over all of the characters
1046 // using a 16-bit representation.
1047 arguments.ForEach(
1048 vars,
1049 [&](TNode<Object> arg) {
1050 TNode<Word32T> code32 = TruncateTaggedToWord32(context, arg);
1051 TNode<Word32T> code16 =
1052 Word32And(code32, Int32Constant(String::kMaxUtf16CodeUnit));
1053
1054 TNode<IntPtrT> offset = ElementOffsetFromIndex(
1055 var_max_index.value(), UINT16_ELEMENTS,
1056 OFFSET_OF_DATA_START(SeqTwoByteString) - kHeapObjectTag);
1057 StoreNoWriteBarrier(MachineRepresentation::kWord16, two_byte_result,
1058 offset, code16);
1059 var_max_index = IntPtrAdd(var_max_index.value(), IntPtrConstant(1));
1060 },
1061 var_max_index.value());
1062
1063 arguments.PopAndReturn(two_byte_result);
1064 }

1069void StringBuiltinsAssembler::MaybeCallFunctionAtSymbol(
1070 const TNode<Context> context, const TNode<Object> object,
1071 const TNode<Object> maybe_string, Handle<Symbol> symbol,
1072 DescriptorIndexNameValue additional_property_to_check,
1073 const NodeFunction0& regexp_call, const NodeFunction1& generic_call) {
1074 Label out(this), no_protector(this), object_is_heapobject(this);
1075 Label get_property_lookup(this);
1076
1077 // The protector guarantees that that the Number and String wrapper
1078 // prototypes do not contain Symbol.{matchAll|replace|split} (aka.
1079 // @@matchAll, @@replace @@split).
1080 GotoIf(IsNumberStringNotRegexpLikeProtectorCellInvalid(), &no_protector);
1081 // Smi is safe thanks to the protector.
1082 GotoIf(TaggedIsSmi(object), &out);
1083 // String is safe thanks to the protector.
1084 GotoIf(IsString(CAST(object)), &out);
1085 // HeapNumber is safe thanks to the protector.
1086 Branch(IsHeapNumber(CAST(object)), &out, &object_is_heapobject);
1087
1088 BIND(&no_protector);
1089 // Smis have to go through the GetProperty lookup in case Number.prototype or
1090 // Object.prototype was modified.
1091 Branch(TaggedIsSmi(object), &get_property_lookup, &object_is_heapobject);
1092
1093 // Take the fast path for RegExps.
1094 // There's two conditions: {object} needs to be a fast regexp, and
1095 // {maybe_string} must be a string (we can't call ToString on the fast path
1096 // since it may mutate {object}).
1097 {
1098 Label stub_call(this), slow_lookup(this);
1099
1100 BIND(&object_is_heapobject);
1101 TNode<HeapObject> heap_object = CAST(object);
1102
1103 GotoIf(TaggedIsSmi(maybe_string), &slow_lookup);
1104 GotoIfNot(IsString(CAST(maybe_string)), &slow_lookup);
1105
1106 // Note we don't run a full (= permissive) check here, because passing the
1107 // check implies calling the fast variants of target builtins, which assume
1108 // we've already made their appropriate fast path checks. This is not the
1109 // case though; e.g.: some of the target builtins access flag getters.
1110 // TODO(jgruber): Handle slow flag accesses on the fast path and make this
1111 // permissive.
1112 RegExpBuiltinsAssembler regexp_asm(state());
1113 regexp_asm.BranchIfFastRegExp(
1114 context, heap_object, LoadMap(heap_object),
1115 PrototypeCheckAssembler::kCheckPrototypePropertyConstness,
1116 additional_property_to_check, &stub_call, &slow_lookup);
1117
1118 BIND(&stub_call);
1119 // TODO(jgruber): Add a no-JS scope once it exists.
1120 regexp_call();
1121
1122 BIND(&slow_lookup);
1123 // Special case null and undefined to skip the property lookup.
1124 Branch(IsNullOrUndefined(heap_object), &out, &get_property_lookup);
1125 }
1126
1127 // Fall back to a slow lookup of {heap_object[symbol]}.
1128 //
1129 // The spec uses GetMethod({heap_object}, {symbol}), which has a few quirks:
1130 // * null values are turned into undefined, and

1647// Return the |word32| codepoint at {index}. Supports SeqStrings and
1648// ExternalStrings.
1649// TODO(v8:9880): Use UintPtrT here.
1650TNode<Int32T> StringBuiltinsAssembler::LoadSurrogatePairAt(
1651 TNode<String> string, TNode<IntPtrT> length, TNode<IntPtrT> index,
1652 UnicodeEncoding encoding) {
1653 Label handle_surrogate_pair(this), return_result(this);
1654 TVARIABLE(Int32T, var_result);
1655 TVARIABLE(Int32T, var_trail);
1656 var_result = StringCharCodeAt(string, Unsigned(index));
1657 var_trail = Int32Constant(0);
1658
1659 GotoIf(Word32NotEqual(Word32And(var_result.value(), Int32Constant(0xFC00)),
1660 Int32Constant(0xD800)),
1661 &return_result);
1662 TNode<IntPtrT> next_index = IntPtrAdd(index, IntPtrConstant(1));
1663
1664 GotoIfNot(IntPtrLessThan(next_index, length), &return_result);
1665 var_trail = StringCharCodeAt(string, Unsigned(next_index));
1666 Branch(Word32Equal(Word32And(var_trail.value(), Int32Constant(0xFC00)),
1667 Int32Constant(0xDC00)),
1668 &handle_surrogate_pair, &return_result);
1669
1670 BIND(&handle_surrogate_pair);
1671 {
1672 TNode<Int32T> lead = var_result.value();
1673 TNode<Int32T> trail = var_trail.value();
1674
1675 // Check that this path is only taken if a surrogate pair is found
1676 CSA_SLOW_DCHECK(this,
1677 Uint32GreaterThanOrEqual(lead, Int32Constant(0xD800)));
1678 CSA_SLOW_DCHECK(this, Uint32LessThan(lead, Int32Constant(0xDC00)));
1679 CSA_SLOW_DCHECK(this,
1680 Uint32GreaterThanOrEqual(trail, Int32Constant(0xDC00)));
1681 CSA_SLOW_DCHECK(this, Uint32LessThan(trail, Int32Constant(0xE000)));
1682
1683 switch (encoding) {
1684 case UnicodeEncoding::UTF16:
1685 var_result = Word32Or(
1686// Need to swap the order for big-endian platforms
1687#if V8_TARGET_BIG_ENDIAN
1688 Word32Shl(lead, Int32Constant(16)), trail);
1689#else
1690 Word32Shl(trail, Int32Constant(16)), lead);
1691#endif
1692 break;
1693
1694 case UnicodeEncoding::UTF32: {
1695 // Convert UTF16 surrogate pair into |word32| code point, encoded as
1696 // UTF32.
1697 TNode<Int32T> surrogate_offset =
1698 Int32Constant(0x10000 - (0xD800 << 10) - 0xDC00);
1699
1700 // (lead << 10) + trail + SURROGATE_OFFSET
1701 var_result = Int32Add(Word32Shl(lead, Int32Constant(10)),
1702 Int32Add(trail, surrogate_offset));
1703 break;
1704 }
1705 }
1706 Goto(&return_result);
1707 }
1708
1709 BIND(&return_result);
1710 return var_result.value();
1711}

1800 TNode<IntPtrT> to_index, TNode<IntPtrT> character_count,
1801 String::Encoding from_encoding, String::Encoding to_encoding) {

1803 // faked sequential strings when handling external subject strings.
1804 bool from_one_byte = from_encoding == String::ONE_BYTE_ENCODING;
1805 bool to_one_byte = to_encoding == String::ONE_BYTE_ENCODING;
1806 Comment("CopyStringCharacters ",
1807 from_one_byte ? "ONE_BYTE_ENCODING" : "TWO_BYTE_ENCODING", " -> ",
1808 to_one_byte ? "ONE_BYTE_ENCODING" : "TWO_BYTE_ENCODING");
1809
1810 ElementsKind from_kind = from_one_byte ? UINT8_ELEMENTS : UINT16_ELEMENTS;
1811 ElementsKind to_kind = to_one_byte ? UINT8_ELEMENTS : UINT16_ELEMENTS;
1812 static_assert(OFFSET_OF_DATA_START(SeqOneByteString) ==
1813 OFFSET_OF_DATA_START(SeqTwoByteString));
1814 int header_size = OFFSET_OF_DATA_START(SeqOneByteString) - kHeapObjectTag;
1815 TNode<IntPtrT> from_offset =
1816 ElementOffsetFromIndex(from_index, from_kind, header_size);
1817 TNode<IntPtrT> to_offset =
1818 ElementOffsetFromIndex(to_index, to_kind, header_size);
1819 TNode<IntPtrT> byte_count =
1820 ElementOffsetFromIndex(character_count, from_kind);
1821 TNode<IntPtrT> limit_offset = IntPtrAdd(from_offset, byte_count);
1822
1823 // Prepare the fast loop.
1824 MachineType type =
1825 from_one_byte ? MachineType::Uint8() : MachineType::Uint16();
1826 MachineRepresentation rep = to_one_byte ? MachineRepresentation::kWord8
1827 : MachineRepresentation::kWord16;
1828 int from_increment = 1 << ElementsKindToShiftSize(from_kind);
1829 int to_increment = 1 << ElementsKindToShiftSize(to_kind);
1830
1831 TVARIABLE(IntPtrT, current_to_offset, to_offset);
1832 VariableList vars({&current_to_offset}, zone());
1833 int to_index_constant = 0, from_index_constant = 0;
1834 bool index_same = (from_encoding == to_encoding) &&
1835 (from_index == to_index ||
1836 (TryToInt32Constant(from_index, &from_index_constant) &&
1837 TryToInt32Constant(to_index, &to_index_constant) &&
1838 from_index_constant == to_index_constant));
1839 BuildFastLoop<IntPtrT>(
1840 vars, from_offset, limit_offset,
1841 [&](TNode<IntPtrT> offset) {
1842 compiler::Node* value = Load(type, from_string, offset);
1843#if DEBUG
1844 // Copying two-byte characters to one-byte is okay if callers have
1845 // checked that this loses no information.
1846 if (v8_flags.debug_code && !from_one_byte && to_one_byte) {
1847 CSA_DCHECK(this, Uint32LessThanOrEqual(UncheckedCast<Uint32T>(value),
1848 Uint32Constant(0xFF)));
1849 }
1850#endif
1851 StoreNoWriteBarrier(rep, to_string,
1852 index_same ? offset : current_to_offset.value(),
1853 value);
1854 if (!index_same) {
1855 Increment(&current_to_offset, to_increment);
1856 }
1857 },
1858 from_increment, LoopUnrollingMode::kYes, IndexAdvanceMode::kPost);
1859}

1861// A wrapper around CopyStringCharacters which determines the correct string
1862// encoding, allocates a corresponding sequential string, and then copies the
1863// given character range using CopyStringCharacters.
1864// |from_string| must be a sequential string.
1865// 0 <= |from_index| <= |from_index| + |character_count| < from_string.length.
1866// |character_count| > 0.
1867template <typename T>
1868TNode<String> StringBuiltinsAssembler::AllocAndCopyStringCharacters(
1869 TNode<T> from, TNode<BoolT> from_is_one_byte, TNode<IntPtrT> from_index,
1870 TNode<IntPtrT> character_count) {
1871 CSA_DCHECK(this, IntPtrGreaterThan(character_count, IntPtrConstant(0)));
1872
1873 Label end(this), one_byte_sequential(this), two_byte_sequential(this);
1874 TVARIABLE(String, var_result);
1875
1876 Branch(from_is_one_byte, &one_byte_sequential, &two_byte_sequential);
1877
1878 // The subject string is a sequential one-byte string.
1879 BIND(&one_byte_sequential);
1880 {
1881 TNode<String> result = AllocateNonEmptySeqOneByteString(
1882 Unsigned(TruncateIntPtrToInt32(character_count)));
1883 CopyStringCharacters<T>(from, result, from_index, IntPtrConstant(0),
1884 character_count, String::ONE_BYTE_ENCODING,
1885 String::ONE_BYTE_ENCODING);
1886 var_result = result;
1887 Goto(&end);
1888 }
1889
1890 // The subject string is a sequential two-byte string.
1891 BIND(&two_byte_sequential);
1892 {
1893 // Check if the to-be-copied range happens to contain only one-byte
1894 // characters, and copy it to a one-byte string if so.
1895 // If the range is long enough, we check 8 characters at a time, to reduce
1896 // the amount of branching.
1897 // For a more readable version of this logic, see {StringFromTwoByteSlice}
1898 // in wasm.tq.
1899 TNode<IntPtrT> start_offset = ElementOffsetFromIndex(
1900 from_index, UINT16_ELEMENTS,
1901 OFFSET_OF_DATA_START(SeqTwoByteString) - kHeapObjectTag);
1902 TNode<IntPtrT> end_offset = IntPtrAdd(
1903 start_offset, ElementOffsetFromIndex(character_count, UINT16_ELEMENTS));
1904 TNode<IntPtrT> eight_char_loop_end = IntPtrSub(
1905 end_offset, ElementOffsetFromIndex(IntPtrConstant(8), UINT16_ELEMENTS));
1906
1907 TVARIABLE(IntPtrT, var_cursor, start_offset);
1908 TNode<RawPtrT> raw_from;
1909 if constexpr (std::is_same_v<T, RawPtrT>) {
1910 raw_from = from;
1911 } else {
1912 raw_from = ReinterpretCast<RawPtrT>(BitcastTaggedToWord(from));
1913 }
1914 Label first_loop(this, &var_cursor), second_loop(this, &var_cursor);
1915 Label twobyte(this);
1916 Branch(IntPtrLessThanOrEqual(start_offset, eight_char_loop_end),
1917 &first_loop, &second_loop);
1918 BIND(&first_loop);
1919 {
1920 TNode<RawPtrT> chunk = RawPtrAdd(raw_from, var_cursor.value());
1921 TNode<Uint32T> c1 = Load<Uint16T>(chunk);
1922 TNode<Uint32T> c2 = Load<Uint16T>(chunk, IntPtrConstant(2));
1923 TNode<Uint32T> bits = Word32Or(c1, c2);
1924 TNode<Uint32T> c3 = Load<Uint16T>(chunk, IntPtrConstant(4));
1925 bits = Word32Or(bits, c3);
1926 TNode<Uint32T> c4 = Load<Uint16T>(chunk, IntPtrConstant(6));
1927 bits = Word32Or(bits, c4);
1928 TNode<Uint32T> c5 = Load<Uint16T>(chunk, IntPtrConstant(8));
1929 bits = Word32Or(bits, c5);
1930 TNode<Uint32T> c6 = Load<Uint16T>(chunk, IntPtrConstant(10));
1931 bits = Word32Or(bits, c6);
1932 TNode<Uint32T> c7 = Load<Uint16T>(chunk, IntPtrConstant(12));
1933 bits = Word32Or(bits, c7);
1934 TNode<Uint32T> c8 = Load<Uint16T>(chunk, IntPtrConstant(14));
1935 bits = Word32Or(bits, c8);
1936 GotoIf(Uint32GreaterThan(bits, Uint32Constant(0xFF)), &twobyte);
1937 Increment(&var_cursor, 8 * sizeof(uint16_t));
1938 Branch(IntPtrLessThanOrEqual(var_cursor.value(), eight_char_loop_end),
1939 &first_loop, &second_loop);
1940 }
1941
1942 BIND(&second_loop);
1943 TVARIABLE(Uint32T, var_bits, Uint32Constant(0));
1944 VariableList vars({&var_bits}, zone());
1945 FastLoopBody<IntPtrT> one_char_loop = [&](TNode<IntPtrT> offset) {
1946 TNode<Uint32T> c = Load<Uint16T>(from, offset);
1947 var_bits = Word32Or(var_bits.value(), c);
1948 };
1949 BuildFastLoop<IntPtrT>(vars, var_cursor, var_cursor.value(), end_offset,
1950 one_char_loop, sizeof(uint16_t),
1951 LoopUnrollingMode::kNo, IndexAdvanceMode::kPost);
1952 GotoIf(Uint32GreaterThan(var_bits.value(), Uint32Constant(0xFF)), &twobyte);
1953 // Fallthrough: only one-byte characters in the to-be-copied range.
1954 {
1955 TNode<String> result = AllocateNonEmptySeqOneByteString(
1956 Unsigned(TruncateIntPtrToInt32(character_count)));
1957 CopyStringCharacters<T>(from, result, from_index, IntPtrConstant(0),
1958 character_count, String::TWO_BYTE_ENCODING,
1959 String::ONE_BYTE_ENCODING);
1960 var_result = result;
1961 Goto(&end);
1962 }
1963
1964 BIND(&twobyte);
1965 {
1966 TNode<String> result = AllocateNonEmptySeqTwoByteString(
1967 Unsigned(TruncateIntPtrToInt32(character_count)));
1968 CopyStringCharacters<T>(from, result, from_index, IntPtrConstant(0),
1969 character_count, String::TWO_BYTE_ENCODING,
1970 String::TWO_BYTE_ENCODING);
1971 var_result = result;
1972 Goto(&end);
1973 }
1974 }
1975
1976 BIND(&end);
1977 return var_result.value();
1978}

2065 }

2089 var_result = string;
2090 Goto(&end);
2091 }

2094 BIND(&runtime);

2097 CAST(CallRuntime(Runtime::kStringSubstring, NoContextConstant(), string,
2098 SmiTag(from), SmiTag(to)));
2099 Goto(&end);
2100 }

2102 BIND(&end);
