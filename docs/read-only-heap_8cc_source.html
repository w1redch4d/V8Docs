41// static
42void ReadOnlyHeap::SetUp(Isolate* isolate,
43 SnapshotData* read_only_snapshot_data,
44 bool can_rehash) {
45 DCHECK_NOT_NULL(isolate);
46 IsolateGroup* group = isolate->isolate_group();
47 if (read_only_snapshot_data != nullptr) {
48 bool read_only_heap_created = false;
49 base::SpinningMutexGuard guard(group->read_only_heap_creation_mutex());
50 ReadOnlyArtifacts* artifacts = group->read_only_artifacts();
51 if (!artifacts) {
52 artifacts = group->InitializeReadOnlyArtifacts();
53 artifacts->InitializeChecksum(read_only_snapshot_data);
54 CreateInitialHeapForBootstrapping(isolate, artifacts);
55 artifacts->read_only_heap()->DeserializeIntoIsolate(
56 isolate, read_only_snapshot_data, can_rehash);
57 artifacts->set_initial_next_unique_sfi_id(isolate->next_unique_sfi_id());
58 read_only_heap_created = true;
59 } else {
60 isolate->SetUpFromReadOnlyArtifacts(artifacts);
61#ifdef V8_COMPRESS_POINTERS
62 isolate->external_pointer_table().SetUpFromReadOnlyArtifacts(
63 isolate->heap()->read_only_external_pointer_space(), artifacts);
64#endif // V8_COMPRESS_POINTERS
65 }
66 artifacts->VerifyChecksum(read_only_snapshot_data, read_only_heap_created);
67 artifacts->read_only_heap()->InitializeIsolateRoots(isolate);
68 } else {
69 // This path should only be taken in mksnapshot, should only be run once
70 // before tearing down the Isolate that holds this ReadOnlyArtifacts and
71 // is not thread-safe.
72 ReadOnlyArtifacts* artifacts = group->read_only_artifacts();
73 CHECK(!artifacts);
74 artifacts = group->InitializeReadOnlyArtifacts();
75 CreateInitialHeapForBootstrapping(isolate, artifacts);
76
77 // Ensure the first read-only page ends up first in the cage.
78 artifacts->read_only_heap()->read_only_space()->EnsurePage();
79 artifacts->VerifyChecksum(read_only_snapshot_data, true);
80 }
81}

93void ReadOnlyHeap::DeserializeIntoIsolate(Isolate* isolate,
94 SnapshotData* read_only_snapshot_data,
95 bool can_rehash) {
96 DCHECK_NOT_NULL(read_only_snapshot_data);
97
98 ReadOnlyDeserializer des(isolate, read_only_snapshot_data, can_rehash);
99 des.DeserializeIntoIsolate();
100 OnCreateRootsComplete(isolate);
101
102 if (V8_UNLIKELY(v8_flags.extensible_ro_snapshot &&
103 isolate->serializer_enabled())) {
104 // If this isolate will be serialized, leave RO space unfinalized and
105 // allocatable s.t. it can be extended (e.g. by future Context::New calls).
106 // We reach this scenario when creating custom snapshots - these initially
107 // create the isolate from the default V8 snapshot, create new customized
108 // contexts, and finally reserialize.
109 } else {
110 InitFromIsolate(isolate);
111 }
112}

114void ReadOnlyHeap::OnCreateRootsComplete(Isolate* isolate) {
115 DCHECK_NOT_NULL(isolate);
116 DCHECK(!roots_init_complete_);
117 InitializeFromIsolateRoots(isolate);
118 roots_init_complete_ = true;
119}

121void ReadOnlyHeap::OnCreateHeapObjectsComplete(Isolate* isolate) {
122 DCHECK_NOT_NULL(isolate);
123
124 // InitFromIsolate mutates MemoryChunk flags which would race with any
125 // concurrently-running sweeper tasks. Ensure that sweeping has been
126 // completed, i.e. no sweeper tasks are currently running.
127 isolate->heap()->EnsureSweepingCompleted(
128 Heap::SweepingForcedFinalizationMode::kV8Only);
129
130 InitFromIsolate(isolate);
131
132#ifdef VERIFY_HEAP
133 if (v8_flags.verify_heap) {
134 HeapVerifier::VerifyReadOnlyHeap(isolate->heap());
135 HeapVerifier::VerifyHeap(isolate->heap());
136 }
137#endif
138}

164void ReadOnlyHeap::InitFromIsolate(Isolate* isolate) {
165 DCHECK(roots_init_complete_);
166 read_only_space_->ShrinkPages();
167 ReadOnlyArtifacts* artifacts =
168 isolate->isolate_group()->read_only_artifacts();
169 read_only_space()->DetachPagesAndAddToArtifacts(artifacts);
170 artifacts->ReinstallReadOnlySpace(isolate);
171
172 read_only_space_ = artifacts->shared_read_only_space();
173
174#ifdef DEBUG
175 artifacts->VerifyHeapAndSpaceRelationships(isolate);
176#endif
177}

179ReadOnlyHeap::ReadOnlyHeap(ReadOnlySpace* ro_space)
180 : read_only_space_(ro_space) {
181#ifdef V8_ENABLE_SANDBOX
182 IsolateGroup::current()->code_pointer_table()->InitializeSpace(
183 &code_pointer_space_);
184#endif // V8_ENABLE_SANDBOX
185#ifdef V8_ENABLE_LEAPTIERING
186 JSDispatchTable* jdt = IsolateGroup::current()->js_dispatch_table();
187 jdt->InitializeSpace(&js_dispatch_table_space_);
188 // To avoid marking trying to write to these read-only cells they are
189 // allocated black. Target code objects in the read-only dispatch table are
190 // read-only code objects.
191 js_dispatch_table_space_.set_allocate_black(true);
192#if V8_STATIC_DISPATCH_HANDLES_BOOL
193 jdt->AttachSpaceToReadOnlySegment(&js_dispatch_table_space_);
194 jdt->PreAllocateEntries(&js_dispatch_table_space_,
195 JSBuiltinDispatchHandleRoot::kCount, true);
196#endif // V8_STATIC_DISPATCH_HANDLES_BOOL
197#endif // V8_ENABLE_LEAPTIERING
198}

201void ReadOnlyHeap::PopulateReadOnlySpaceStatistics(

203 statistics->read_only_space_size_ = 0;
204 statistics->read_only_space_used_size_ = 0;
205 statistics->read_only_space_physical_size_ = 0;
206 ReadOnlyArtifacts* artifacts = IsolateGroup::current()->read_only_artifacts();

240 const ReadOnlySpace* ro_space)
241 : ro_space_(ro_space),

243 page_iterator_(
244 current_page_ == ro_space->pages().end() ? nullptr : *current_page_) {

246
247Tagged<HeapObject> ReadOnlyHeapObjectIterator::Next() {
248 while (current_page_ != ro_space_->pages().end()) {
249 Tagged<HeapObject> obj = page_iterator_.Next();
250 if (!obj.is_null()) return obj;
251
252 ++current_page_;
253 if (current_page_ == ro_space_->pages().end()) return Tagged<HeapObject>();
254 page_iterator_.Reset(*current_page_);
255 }
256
257 DCHECK_EQ(current_page_, ro_space_->pages().end());
258 return Tagged<HeapObject>();
259}

268ReadOnlyPageObjectIterator::ReadOnlyPageObjectIterator(
269 const ReadOnlyPageMetadata* page, Address current_addr,
270 SkipFreeSpaceOrFiller skip_free_space_or_filler)
271 : page_(page),
272 current_addr_(current_addr),
273 skip_free_space_or_filler_(skip_free_space_or_filler) {
274 DCHECK_GE(current_addr, page->GetAreaStart());
275 DCHECK_LT(current_addr, page->GetAreaStart() + page->area_size());
276}
277
278Tagged<HeapObject> ReadOnlyPageObjectIterator::Next() {
279 if (page_ == nullptr) return HeapObject();
280
281 Address end = page_->GetAreaStart() + page_->area_size();
282 for (;;) {
283 DCHECK_LE(current_addr_, end);
284 if (current_addr_ == end) return HeapObject();
285
286 Tagged<HeapObject> object = HeapObject::FromAddress(current_addr_);
287 const int object_size = object->Size();
288 current_addr_ += ALIGN_TO_ALLOCATION_ALIGNMENT(object_size);
289
290 if (skip_free_space_or_filler_ == SkipFreeSpaceOrFiller::kYes &&
291 IsFreeSpaceOrFiller(object)) {
292 continue;
293 }
294
295 DCHECK_OBJECT_SIZE(object_size);
296 return object;
297 }
