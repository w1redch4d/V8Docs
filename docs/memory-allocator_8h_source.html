55 void Add(MutablePageMetadata* chunk) {
56 // This method is called only on the main thread and only during the
57 // atomic pause so a lock is not needed.
58 DCHECK_NOT_NULL(chunk);
59 DCHECK_EQ(chunk->size(), PageMetadata::kPageSize);
60 DCHECK(!chunk->Chunk()->IsLargePage());
61 DCHECK(!chunk->Chunk()->IsTrusted());
62 DCHECK_NE(chunk->Chunk()->executable(), EXECUTABLE);
63 chunk->ReleaseAllAllocatedMemory();
64 pooled_chunks_.push_back(chunk);
65 }

67 MutablePageMetadata* TryGetPooled() {
68 base::SpinningMutexGuard guard(&mutex_);
69 if (pooled_chunks_.empty()) return nullptr;
70 MutablePageMetadata* chunk = pooled_chunks_.back();
71 pooled_chunks_.pop_back();
72 return chunk;
73 }

161 // Returns an indication of whether a pointer is in a space that has
162 // been allocated by this MemoryAllocator. It is conservative, allowing
163 // false negatives (i.e., if a pointer is outside the allocated space, it may
164 // return false) but not false positives (i.e., if a pointer is inside the
165 // allocated space, it will definitely return false).
166 V8_INLINE bool IsOutsideAllocatedSpace(Address address) const {
167 return IsOutsideAllocatedSpace(address, NOT_EXECUTABLE) &&
168 IsOutsideAllocatedSpace(address, EXECUTABLE);

170 V8_INLINE bool IsOutsideAllocatedSpace(Address address,
171 Executability executable) const {
172 switch (executable) {
173 case NOT_EXECUTABLE:
174 return address < lowest_not_executable_ever_allocated_ ||
175 address >= highest_not_executable_ever_allocated_;
176 case EXECUTABLE:
177 return address < lowest_executable_ever_allocated_ ||
178 address >= highest_executable_ever_allocated_;
179 }
180 }

325 void UpdateAllocatedSpaceLimits(Address low, Address high,
326 Executability executable) {
327 // The use of atomic primitives does not guarantee correctness (wrt.
328 // desired semantics) by default. The loop here ensures that we update the
329 // values only if they did not change in between.
330 Address ptr;
331 switch (executable) {
332 case NOT_EXECUTABLE:
333 ptr = lowest_not_executable_ever_allocated_.load(
334 std::memory_order_relaxed);
335 while ((low < ptr) &&
336 !lowest_not_executable_ever_allocated_.compare_exchange_weak(
337 ptr, low, std::memory_order_acq_rel)) {
338 }
339 ptr = highest_not_executable_ever_allocated_.load(
340 std::memory_order_relaxed);
341 while ((high > ptr) &&
342 !highest_not_executable_ever_allocated_.compare_exchange_weak(
343 ptr, high, std::memory_order_acq_rel)) {
344 }
345 break;
346 case EXECUTABLE:
347 ptr = lowest_executable_ever_allocated_.load(std::memory_order_relaxed);
348 while ((low < ptr) &&
349 !lowest_executable_ever_allocated_.compare_exchange_weak(
350 ptr, low, std::memory_order_acq_rel)) {
351 }
352 ptr =
353 highest_executable_ever_allocated_.load(std::memory_order_relaxed);
354 while ((high > ptr) &&
355 !highest_executable_ever_allocated_.compare_exchange_weak(
356 ptr, high, std::memory_order_acq_rel)) {
357 }
358 break;
359 }
360 }
