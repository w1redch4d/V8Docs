57 // LocalSweeper holds local data structures required for sweeping and is used
58 // to initiate sweeping and promoted page iteration on multiple threads. Each
59 // thread should holds its own LocalSweeper. Once sweeping is done, all
60 // LocalSweepers should be finalized on the main thread.
61 //
62 // LocalSweeper is not thread-safe and should not be concurrently by several
63 // threads. The exceptions to this rule are allocations during parallel
64 // evacuation and from concurrent allocators. In practice the data structures
65 // in LocalSweeper are only actively used for new space sweeping. Since
66 // parallel evacuators and concurrent allocators never try to allocate in new
67 // space, they will never contribute to new space sweeping and thus can use
68 // the main thread's local sweeper without risk of data races.

344template <typename ShouldYieldCallback>
345bool Sweeper::LocalSweeper::ContributeAndWaitForPromotedPagesIterationImpl(
346 ShouldYieldCallback should_yield_callback) {
347 if (!sweeper_->sweeping_in_progress()) return true;
348 if (!sweeper_->IsIteratingPromotedPages()) return true;
349 if (!ParallelIteratePromotedPagesImpl(should_yield_callback)) return false;
350 base::MutexGuard guard(
351 &sweeper_->promoted_pages_iteration_notification_mutex_);
352 // Check again that iteration is not yet finished.
353 if (!sweeper_->IsIteratingPromotedPages()) return true;
354 if (should_yield_callback()) {
355 return false;
356 }
357 sweeper_->promoted_pages_iteration_notification_variable_.Wait(
358 &sweeper_->promoted_pages_iteration_notification_mutex_);
359 return true;
360}

362template <typename ShouldYieldCallback>
363bool Sweeper::LocalSweeper::ParallelIteratePromotedPagesImpl(
364 ShouldYieldCallback should_yield_callback) {
365 while (!should_yield_callback()) {
366 MutablePageMetadata* chunk = sweeper_->GetPromotedPageSafe();
367 if (chunk == nullptr) return true;
368 ParallelIteratePromotedPage(chunk);
369 }
370 return false;
371}
