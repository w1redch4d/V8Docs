30struct CodePointerTableEntry {
31 // We write-protect the CodePointerTable on platforms that support it for
32 // forward-edge CFI.
33 static constexpr bool IsWriteProtected = true;
34
35 // Make this entry a code pointer entry for the given code object and
36 // entrypoint.
37 inline void MakeCodePointerEntry(Address code, Address entrypoint,
38 CodeEntrypointTag tag, bool mark_as_alive);
39
40 // Make this entry a freelist entry, containing the index of the next entry
41 // on the freelist.
42 inline void MakeFreelistEntry(uint32_t next_entry_index);
43
44 // Load code entrypoint pointer stored in this entry.
45 // This entry must be a code pointer entry.
46 inline Address GetEntrypoint(CodeEntrypointTag tag) const;
47
48 // Store the given code entrypoint pointer in this entry.
49 // This entry must be a code pointer entry.
50 inline void SetEntrypoint(Address value, CodeEntrypointTag tag);
51
52 // Load the code object pointer stored in this entry.
53 // This entry must be a code pointer entry.
54 inline Address GetCodeObject() const;
55
56 // Store the given code object pointer in this entry.
57 // This entry must be a code pointer entry.
58 inline void SetCodeObject(Address value);
59
60 // Returns true if this entry is a freelist entry.
61 inline bool IsFreelistEntry() const;
62
63 // Get the index of the next entry on the freelist. This method may be
64 // called even when the entry is not a freelist entry. However, the result
65 // is only valid if this is a freelist entry. This behaviour is required
66 // for efficient entry allocation, see TryAllocateEntryFromFreelist.
67 inline uint32_t GetNextFreelistEntryIndex() const;
68
69 // Mark this entry as alive during garbage collection.
70 inline void Mark();
71
72 // Unmark this entry during sweeping.
73 inline void Unmark();
74
75 // Test whether this entry is currently marked as alive.
76 inline bool IsMarked() const;
77
78 private:
79 friend class CodePointerTable;
80
81 // Freelist entries contain the index of the next free entry in their lower 32
82 // bits and are tagged with the kFreeCodePointerTableEntryTag.
83 static constexpr Address kFreeEntryTag = kFreeCodePointerTableEntryTag;
84
85 // The marking bit is stored in the code_ field, see below.
86 static constexpr Address kMarkingBit = 1;
87
88 std::atomic<Address> entrypoint_;
89 // The pointer to the Code object also contains the marking bit: since this is
90 // a tagged pointer to a V8 HeapObject, we know that it will be 4-byte aligned
91 // and that the LSB should always be set. We therefore use the LSB as marking
92 // bit. In this way:
93 // - When loading the pointer, we only need to perform an unconditional OR 1
94 // to get the correctly tagged pointer
95 // - When storing the pointer we don't need to do anything since the tagged
96 // pointer will automatically be marked
97 std::atomic<Address> code_;
98};
