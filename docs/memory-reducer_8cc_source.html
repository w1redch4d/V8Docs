38void MemoryReducer::TimerTask::RunInternal() {
39 Heap* heap = memory_reducer_->heap();
40 const double time_ms = heap->MonotonicallyIncreasingTimeInMs();
41 heap->allocator()->new_space_allocator()->FreeLinearAllocationArea();
42 heap->tracer()->SampleAllocation(base::TimeTicks::Now(),
43 heap->NewSpaceAllocationCounter(),
44 heap->OldGenerationAllocationCounter(),
45 heap->EmbedderAllocationCounter());
46 const bool low_allocation_rate = heap->HasLowAllocationRate();
47 const bool optimize_for_memory = heap->ShouldOptimizeForMemoryUsage();
48 if (v8_flags.trace_memory_reducer) {
49 heap->isolate()->PrintWithTimestamp(
50 "Memory reducer: %s, %s\n",
51 low_allocation_rate ? "low alloc" : "high alloc",
52 optimize_for_memory ? "background" : "foreground");
53 }
54 // The memory reducer will start incremental marking if
55 // 1) mutator is likely idle: js call rate is low and allocation rate is low.
56 // 2) mutator is in background: optimize for memory flag is set.
57 const Event event{
58 kTimer,
59 time_ms,
60 heap->CommittedOldGenerationMemory(),
61 false,
62 low_allocation_rate || optimize_for_memory,
63 heap->incremental_marking()->IsStopped() &&
64 heap->incremental_marking()->CanAndShouldBeStarted(),
65 };
66 memory_reducer_->NotifyTimer(event);
67}

70void MemoryReducer::NotifyTimer(const Event& event) {
71 if (state_.id() != kWait) return;
72 DCHECK_EQ(kTimer, event.type);
73 state_ = Step(state_, event);
74 if (state_.id() == kRun) {
75 DCHECK(heap()->incremental_marking()->IsStopped());
76 DCHECK(v8_flags.incremental_marking);
77 if (v8_flags.trace_memory_reducer) {
78 heap()->isolate()->PrintWithTimestamp("Memory reducer: started GC #%d\n",
79 state_.started_gcs());
80 }
81 GCFlags gc_flags = v8_flags.memory_reducer_favors_memory
82 ? GCFlag::kReduceMemoryFootprint
83 : GCFlag::kNoFlags;
84 heap()->StartIncrementalMarking(gc_flags,
85 GarbageCollectionReason::kMemoryReducer,
86 kGCCallbackFlagCollectAllExternalMemory);
87 } else if (state_.id() == kWait) {
88 // Re-schedule the timer.
89 ScheduleTimer(state_.next_gc_start_ms() - event.time_ms);
90 if (v8_flags.trace_memory_reducer) {
91 heap()->isolate()->PrintWithTimestamp(
92 "Memory reducer: waiting for %.f ms\n",
93 state_.next_gc_start_ms() - event.time_ms);
94 }
95 }
96}

98void MemoryReducer::NotifyMarkCompact(size_t committed_memory_before) {
99 if (!v8_flags.incremental_marking) return;
100 const size_t committed_memory = heap()->CommittedOldGenerationMemory();
101
102 // Trigger one more GC if
103 // - this GC decreased committed memory,
104 // - there is high fragmentation,
105 const MemoryReducer::Event event{
106 MemoryReducer::kMarkCompact,
107 heap()->MonotonicallyIncreasingTimeInMs(),
108 committed_memory,
109 (committed_memory_before > committed_memory + MB) ||
110 heap()->HasHighFragmentation(),
111 false,
112 false};
113 const State old_state = state_;
114 state_ = Step(state_, event);
115 if (old_state.id() != kWait && state_.id() == kWait) {
116 // If we are transitioning to the WAIT state, start the timer.
117 ScheduleTimer(state_.next_gc_start_ms() - event.time_ms);
118 }
119 if (old_state.id() == kRun && v8_flags.trace_memory_reducer) {
120 heap()->isolate()->PrintWithTimestamp(
121 "Memory reducer: finished GC #%d (%s)\n", old_state.started_gcs(),
122 state_.id() == kWait ? "will do more" : "done");
123 }
124}

126void MemoryReducer::NotifyPossibleGarbage() {
127 if (!v8_flags.incremental_marking) return;
128 const MemoryReducer::Event event{MemoryReducer::kPossibleGarbage,
129 heap()->MonotonicallyIncreasingTimeInMs(),
130 0,
131 false,
132 false,
133 false};
134 const Id old_action = state_.id();
135 state_ = Step(state_, event);
136 if (old_action != kWait && state_.id() == kWait) {
137 // If we are transitioning to the WAIT state, start the timer.
138 ScheduleTimer(state_.next_gc_start_ms() - event.time_ms);

142bool MemoryReducer::WatchdogGC(const State& state, const Event& event) {
143 return state.last_gc_time_ms() != 0 &&
144 event.time_ms > state.last_gc_time_ms() + kWatchdogDelayMs;
145}

148// For specification of this function see the comment for MemoryReducer class.
149MemoryReducer::State MemoryReducer::Step(const State& state,
150 const Event& event) {
151 DCHECK(v8_flags.memory_reducer);
152 DCHECK(v8_flags.incremental_marking);
153
154 switch (state.id()) {
155 case kUninit:
156 case kDone:
157 if (event.type == kTimer) {
158 return state;
159 } else if (event.type == kMarkCompact) {
160 if (event.committed_memory <
161 std::max(
162 static_cast<size_t>(state.committed_memory_at_last_run() *
163 kCommittedMemoryFactor),
164 state.committed_memory_at_last_run() + kCommittedMemoryDelta)) {
165 return state;
166 } else {
167 return State::CreateWait(0, event.time_ms + kLongDelayMs,
168 event.time_ms);
169 }
170 } else {
171 DCHECK_EQ(kPossibleGarbage, event.type);
172 return State::CreateWait(
173 0, event.time_ms + v8_flags.gc_memory_reducer_start_delay_ms,
174 state.last_gc_time_ms());
175 }
176 case kWait:
177 CHECK_LE(state.started_gcs(), MaxNumberOfGCs());
178 switch (event.type) {
179 case kPossibleGarbage:
180 return state;
181 case kTimer:
182 if (state.started_gcs() >= MaxNumberOfGCs()) {
183 return State::CreateDone(state.last_gc_time_ms(),
184 event.committed_memory);
185 } else if (event.can_start_incremental_gc &&
186 (event.should_start_incremental_gc ||
187 WatchdogGC(state, event))) {
188 if (state.next_gc_start_ms() <= event.time_ms) {
189 return State::CreateRun(state.started_gcs() + 1);
190 } else {
191 return state;
192 }
193 } else {
194 return State::CreateWait(state.started_gcs(),
195 event.time_ms + kLongDelayMs,
196 state.last_gc_time_ms());
197 }
198 case kMarkCompact:
199 return State::CreateWait(state.started_gcs(),
200 event.time_ms + kLongDelayMs, event.time_ms);
201 }
202 case kRun:
203 CHECK_LE(state.started_gcs(), MaxNumberOfGCs());
204 if (event.type == kMarkCompact) {
205 if (state.started_gcs() < MaxNumberOfGCs() &&
206 (event.next_gc_likely_to_collect_more ||
207 state.started_gcs() == 1)) {
208 return State::CreateWait(state.started_gcs(),
209 event.time_ms + kShortDelayMs,
210 event.time_ms);
211 } else {
212 return State::CreateDone(event.time_ms, event.committed_memory);
213 }
214 } else {
215 return state;
216 }
217 }
218 UNREACHABLE();
219}

221void MemoryReducer::ScheduleTimer(double delay_ms) {
222 DCHECK_LT(0, delay_ms);
223 if (heap()->IsTearingDown()) return;
224 // Leave some room for precision error in task scheduler.
225 const double kSlackMs = 100;
226 taskrunner_->PostDelayedTask(std::make_unique<MemoryReducer::TimerTask>(this),
227 (delay_ms + kSlackMs) / 1000.0);
228}
