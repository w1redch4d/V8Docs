42 bool MatchZero(V<Any> matched) const {
43 const ConstantOp* op = TryCast<ConstantOp>(matched);
44 if (!op) return false;
45 switch (op->kind) {
46 case ConstantOp::Kind::kWord32:
47 case ConstantOp::Kind::kWord64:
48 return op->integral() == 0;
49 case ConstantOp::Kind::kFloat32:
50 return op->float32().get_scalar() == 0;
51 case ConstantOp::Kind::kFloat64:
52 return op->float64().get_scalar() == 0;
53 case ConstantOp::Kind::kSmi:
54 return op->smi().value() == 0;
55 default:
56 return false;
57 }
58 }

143 bool MatchIntegralWordConstant(V<Any> matched, WordRepresentation rep,
144 uint64_t* unsigned_constant,
145 int64_t* signed_constant = nullptr) const {
146 const ConstantOp* op = TryCast<ConstantOp>(matched);
147 if (!op) return false;
148 switch (op->kind) {
149 case ConstantOp::Kind::kWord32:
150 case ConstantOp::Kind::kWord64:
151 case ConstantOp::Kind::kRelocatableWasmCall:
152 case ConstantOp::Kind::kRelocatableWasmStubCall:
153 if (rep.value() == WordRepresentation::Word32()) {
154 if (unsigned_constant) {
155 *unsigned_constant = static_cast<uint32_t>(op->integral());
156 }
157 if (signed_constant) {
158 *signed_constant = static_cast<int32_t>(op->signed_integral());
159 }
160 return true;
161 } else if (rep.value() == WordRepresentation::Word64()) {
162 if (unsigned_constant) {
163 *unsigned_constant = op->integral();
164 }
165 if (signed_constant) {
166 *signed_constant = op->signed_integral();
167 }
168 return true;
169 }
170 return false;
171 default:
172 return false;
173 }
174 UNREACHABLE();
175 }

408 bool MatchConstantShift(V<Any> matched, V<T>* input, ShiftOp::Kind* kind,
409 WordRepresentation* rep, int* amount) const {
410 const ShiftOp* op = TryCast<ShiftOp>(matched);
411 if (uint32_t rhs_constant;
412 op && MatchIntegralWord32Constant(op->right(), &rhs_constant) &&
413 rhs_constant < static_cast<uint64_t>(op->rep.bit_width())) {
414 *input = op->left<T>();
415 *kind = op->kind;
416 *rep = op->rep;
417 *amount = static_cast<int>(rhs_constant);
418 return true;
419 }
420 return false;
