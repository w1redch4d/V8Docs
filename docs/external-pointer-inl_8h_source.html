51template <ExternalPointerTag tag>
52V8_INLINE void InitExternalPointerField(Address host_address,
53 Address field_address,
54 IsolateForSandbox isolate,
55 Address value) {
56#ifdef V8_ENABLE_SANDBOX
57 static_assert(tag != kExternalPointerNullTag);
58 ExternalPointerTable& table = isolate.GetExternalPointerTableFor(tag);
59 ExternalPointerHandle handle = table.AllocateAndInitializeEntry(
60 isolate.GetExternalPointerTableSpaceFor(tag, host_address), value, tag);
61 // Use a Release_Store to ensure that the store of the pointer into the
62 // table is not reordered after the store of the handle. Otherwise, other
63 // threads may access an uninitialized table entry and crash.
64 auto location = reinterpret_cast<ExternalPointerHandle*>(field_address);
65 base::AsAtomic32::Release_Store(location, handle);
66#else
67 WriteExternalPointerField<tag>(field_address, isolate, value);
68#endif // V8_ENABLE_SANDBOX
69}

71template <ExternalPointerTagRange tag_range>
72V8_INLINE Address ReadExternalPointerField(Address field_address,
73 IsolateForSandbox isolate) {
74#ifdef V8_ENABLE_SANDBOX
75 // static_assert(tag != kExternalPointerNullTag); // TODO
76 // Handles may be written to objects from other threads so the handle needs
77 // to be loaded atomically. We assume that the load from the table cannot
78 // be reordered before the load of the handle due to the data dependency
79 // between the two loads and therefore use relaxed memory ordering, but
80 // technically we should use memory_order_consume here.
81 auto location = reinterpret_cast<ExternalPointerHandle*>(field_address);
82 ExternalPointerHandle handle = base::AsAtomic32::Relaxed_Load(location);
83 return isolate.GetExternalPointerTableFor(tag_range).Get(handle, tag_range);
84#else
85 return ReadMaybeUnalignedValue<Address>(field_address);
86#endif // V8_ENABLE_SANDBOX
87}

89template <ExternalPointerTag tag>
90V8_INLINE void WriteExternalPointerField(Address field_address,
91 IsolateForSandbox isolate,
92 Address value) {
93#ifdef V8_ENABLE_SANDBOX
94 static_assert(tag != kExternalPointerNullTag);
95 // See comment above for why this is a Relaxed_Load.
96 auto location = reinterpret_cast<ExternalPointerHandle*>(field_address);
97 ExternalPointerHandle handle = base::AsAtomic32::Relaxed_Load(location);
98 isolate.GetExternalPointerTableFor(tag).Set(handle, value, tag);
99#else
100 WriteMaybeUnalignedValue<Address>(field_address, value);
101#endif // V8_ENABLE_SANDBOX
102}
