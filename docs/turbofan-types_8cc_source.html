27bool RangeType::Limits::IsEmpty() { return this->min > this->max; }

29RangeType::Limits RangeType::Limits::Intersect(Limits lhs, Limits rhs) {
30 DisallowGarbageCollection no_gc;
31 Limits result(lhs);
32 if (lhs.min < rhs.min) result.min = rhs.min;
33 if (lhs.max > rhs.max) result.max = rhs.max;
34 return result;
35}

37RangeType::Limits RangeType::Limits::Union(Limits lhs, Limits rhs) {
38 DisallowGarbageCollection no_gc;
39 if (lhs.IsEmpty()) return rhs;
40 if (rhs.IsEmpty()) return lhs;
41 Limits result(lhs);
42 if (lhs.min > rhs.min) result.min = rhs.min;
43 if (lhs.max < rhs.max) result.max = rhs.max;
44 return result;
45}

47bool Type::Overlap(const RangeType* lhs, const RangeType* rhs) {
48 DisallowGarbageCollection no_gc;

62double Type::Min() const {
63 DCHECK(this->Is(Number()));
64 DCHECK(!this->Is(NaN()));
65 if (this->IsBitset()) return BitsetType::Min(this->AsBitset());
66 if (this->IsUnion()) {
67 double min = +V8_INFINITY;
68 for (int i = 1, n = AsUnion()->Length(); i < n; ++i) {
69 min = std::min(min, AsUnion()->Get(i).Min());
70 }
71 Type bitset = AsUnion()->Get(0);
72 if (!bitset.Is(NaN())) min = std::min(min, bitset.Min());
73 return min;
74 }
75 if (this->IsRange()) return this->AsRange()->Min();
76 DCHECK(this->IsOtherNumberConstant());
77 return this->AsOtherNumberConstant()->Value();
78}

80double Type::Max() const {
81 DCHECK(this->Is(Number()));
82 DCHECK(!this->Is(NaN()));
83 if (this->IsBitset()) return BitsetType::Max(this->AsBitset());
84 if (this->IsUnion()) {
85 double max = -V8_INFINITY;
86 for (int i = 1, n = this->AsUnion()->Length(); i < n; ++i) {
87 max = std::max(max, this->AsUnion()->Get(i).Max());
88 }
89 Type bitset = this->AsUnion()->Get(0);
90 if (!bitset.Is(NaN())) max = std::max(max, bitset.Max());
91 return max;
92 }
93 if (this->IsRange()) return this->AsRange()->Max();
94 DCHECK(this->IsOtherNumberConstant());
95 return this->AsOtherNumberConstant()->Value();
96}

101// The largest bitset subsumed by this type.
102Type::bitset Type::BitsetGlb() const {
103 DisallowGarbageCollection no_gc;
104 // Fast case.
105 if (IsBitset()) {
106 return AsBitset();
107 } else if (IsUnion()) {
108 SLOW_DCHECK(AsUnion()->Wellformed());
109 return AsUnion()->Get(0).BitsetGlb() |
110 AsUnion()->Get(1).BitsetGlb(); // Shortcut.
111 } else if (IsRange()) {
112 bitset glb = BitsetType::Glb(AsRange()->Min(), AsRange()->Max());
113 return glb;
114 } else {
115 return BitsetType::kNone;
116 }
117}

119// The smallest bitset subsuming this type, possibly not a proper one.
120Type::bitset Type::BitsetLub() const {
121 DisallowGarbageCollection no_gc;
122 if (IsBitset()) return AsBitset();
123 if (IsUnion()) {
124 // Take the representation from the first element, which is always
125 // a bitset.
126 bitset lub = AsUnion()->Get(0).BitsetLub();
127 for (int i = 0, n = AsUnion()->Length(); i < n; ++i) {
128 // Other elements only contribute their semantic part.
129 lub |= AsUnion()->Get(i).BitsetLub();
130 }
131 return lub;
132 }
133 if (IsHeapConstant()) return AsHeapConstant()->Lub();
134 if (IsOtherNumberConstant()) {
135 return AsOtherNumberConstant()->Lub();
136 }
137 if (IsRange()) return AsRange()->Lub();
138 if (IsTuple()) return BitsetType::kOtherInternal;
139#if V8_ENABLE_WEBASSEMBLY
140 if (IsWasm()) return static_cast<const WasmType*>(ToTypeBase())->Lub();
141#endif
142 UNREACHABLE();
143}

457Type::bitset BitsetType::ExpandInternals(Type::bitset bits) {
458 DCHECK_IMPLIES(bits & kOtherString, (bits & kString) == kString);
459 DisallowGarbageCollection no_gc;
460 if (!(bits & kPlainNumber)) return bits; // Shortcut.
461 const Boundary* boundaries = Boundaries();
462 for (size_t i = 0; i < BoundariesSize(); ++i) {
463 DCHECK(BitsetType::Is(boundaries[i].internal, boundaries[i].external));
464 if (bits & boundaries[i].internal) bits |= boundaries[i].external;
465 }
466 return bits;
467}

469Type::bitset BitsetType::Lub(double min, double max) {
470 DisallowGarbageCollection no_gc;
471 bitset lub = kNone;
472 const Boundary* mins = Boundaries();
473
474 for (size_t i = 1; i < BoundariesSize(); ++i) {
475 if (min < mins[i].min) {
476 lub |= mins[i - 1].internal;
477 if (max < mins[i].min) return lub;
478 }
479 }
480 return lub | mins[BoundariesSize() - 1].internal;
481}

485Type::bitset BitsetType::Glb(double min, double max) {
486 DisallowGarbageCollection no_gc;
487 bitset glb = kNone;
488 const Boundary* mins = Boundaries();
489
490 // If the range does not touch 0, the bound is empty.
491 if (max < -1 || min > 0) return glb;
492
493 for (size_t i = 1; i + 1 < BoundariesSize(); ++i) {
494 if (min <= mins[i].min) {
495 if (max + 1 < mins[i + 1].min) break;
496 glb |= mins[i].external;
497 }
498 }
499 // OtherNumber also contains float numbers, so it can never be
500 // in the greatest lower bound.
501 return glb & ~(kOtherNumber);
502}

504double BitsetType::Min(bitset bits) {
505 DisallowGarbageCollection no_gc;
506 DCHECK(Is(bits, kNumber));
507 DCHECK(!Is(bits, kNaN));
508 const Boundary* mins = Boundaries();
509 bool mz = bits & kMinusZero;
510 for (size_t i = 0; i < BoundariesSize(); ++i) {
511 if (Is(mins[i].internal, bits)) {
512 return mz ? std::min(0.0, mins[i].min) : mins[i].min;
513 }
514 }
515 DCHECK(mz);
516 return 0;
517}

519double BitsetType::Max(bitset bits) {
520 DisallowGarbageCollection no_gc;
521 DCHECK(Is(bits, kNumber));
522 DCHECK(!Is(bits, kNaN));
523 const Boundary* mins = Boundaries();
524 bool mz = bits & kMinusZero;
525 if (BitsetType::Is(mins[BoundariesSize() - 1].internal, bits)) {
526 return +V8_INFINITY;
527 }
528 for (size_t i = BoundariesSize() - 1; i-- > 0;) {
529 if (Is(mins[i].internal, bits)) {
530 return mz ? std::max(0.0, mins[i + 1].min - 1) : mins[i + 1].min - 1;
531 }
532 }
533 DCHECK(mz);
534 return 0;
535}

555bool Type::SimplyEquals(Type that) const {
556 DisallowGarbageCollection no_gc;
557 if (this->IsHeapConstant()) {
558 return that.IsHeapConstant() &&
559 this->AsHeapConstant()->Value().address() ==
560 that.AsHeapConstant()->Value().address();
561 }
562 if (this->IsOtherNumberConstant()) {
563 return that.IsOtherNumberConstant() &&
564 this->AsOtherNumberConstant()->Value() ==
565 that.AsOtherNumberConstant()->Value();
566 }
567 if (this->IsRange()) {
568 if (that.IsHeapConstant() || that.IsOtherNumberConstant()) return false;
569 }
570 if (this->IsTuple()) {
571 if (!that.IsTuple()) return false;
572 const TupleType* this_tuple = this->AsTuple();
573 const TupleType* that_tuple = that.AsTuple();
574 if (this_tuple->Arity() != that_tuple->Arity()) {
575 return false;
576 }
577 for (int i = 0, n = this_tuple->Arity(); i < n; ++i) {
578 if (!this_tuple->Element(i).Equals(that_tuple->Element(i))) return false;
579 }
580 return true;
581 }
582 UNREACHABLE();
583}

585// Check if [this] <= [that].
586bool Type::SlowIs(Type that) const {
587 DisallowGarbageCollection no_gc;
588
589 // Fast bitset cases
590 if (that.IsBitset()) {
591 return BitsetType::Is(this->BitsetLub(), that.AsBitset());
592 }
593
594 if (this->IsBitset()) {
595 return BitsetType::Is(this->AsBitset(), that.BitsetGlb());
596 }
597
598 // (T1 \/ ... \/ Tn) <= T if (T1 <= T) /\ ... /\ (Tn <= T)
599 if (this->IsUnion()) {
600 for (int i = 0, n = this->AsUnion()->Length(); i < n; ++i) {
601 if (!this->AsUnion()->Get(i).Is(that)) return false;
602 }
603 return true;
604 }
605
606 // T <= (T1 \/ ... \/ Tn) if (T <= T1) \/ ... \/ (T <= Tn)
607 if (that.IsUnion()) {
608 for (int i = 0, n = that.AsUnion()->Length(); i < n; ++i) {
609 if (this->Is(that.AsUnion()->Get(i))) return true;
610 if (i > 1 && this->IsRange()) return false; // Shortcut.
611 }
612 return false;
613 }
614
615 if (that.IsRange()) {
616 return this->IsRange() && Contains(that.AsRange(), this->AsRange());
617 }
618 if (this->IsRange()) return false;
619
620#ifdef V8_ENABLE_WEBASSEMBLY
621 if (this->IsWasm()) {
622 if (!that.IsWasm()) return false;
623 wasm::TypeInModule this_type = this->AsWasm();
624 wasm::TypeInModule that_type = that.AsWasm();
625 return wasm::IsSubtypeOf(this_type.type, that_type.type, this_type.module,
626 that_type.module);
627 }
628#endif
629
630 return this->SimplyEquals(that);
631}

633// Check if [this] and [that] overlap.
634bool Type::Maybe(Type that) const {
635 DisallowGarbageCollection no_gc;
636
637 if (BitsetType::IsNone(this->BitsetLub() & that.BitsetLub())) return false;
638
639 // (T1 \/ ... \/ Tn) overlaps T if (T1 overlaps T) \/ ... \/ (Tn overlaps T)
640 if (this->IsUnion()) {
641 for (int i = 0, n = this->AsUnion()->Length(); i < n; ++i) {
642 if (this->AsUnion()->Get(i).Maybe(that)) return true;
643 }
644 return false;
645 }
646
647 // T overlaps (T1 \/ ... \/ Tn) if (T overlaps T1) \/ ... \/ (T overlaps Tn)
648 if (that.IsUnion()) {
649 for (int i = 0, n = that.AsUnion()->Length(); i < n; ++i) {
650 if (this->Maybe(that.AsUnion()->Get(i))) return true;
651 }
652 return false;
653 }
654
655 if (this->IsBitset() && that.IsBitset()) return true;
656
657 if (this->IsRange()) {
658 if (that.IsRange()) {
659 return Overlap(this->AsRange(), that.AsRange());
660 }
661 if (that.IsBitset()) {
662 bitset number_bits = BitsetType::NumberBits(that.AsBitset());
663 if (number_bits == BitsetType::kNone) {
664 return false;
665 }
666 double min = std::max(BitsetType::Min(number_bits), this->Min());
667 double max = std::min(BitsetType::Max(number_bits), this->Max());
668 return min <= max;
669 }
670 }
671 if (that.IsRange()) {
672 return that.Maybe(*this); // This case is handled above.
673 }
674
675 if (this->IsBitset() || that.IsBitset()) return true;
676
677 return this->SimplyEquals(that);
678}

681Type Type::GetRange() const {
682 DisallowGarbageCollection no_gc;
683 if (this->IsRange()) return *this;
684 if (this->IsUnion() && this->AsUnion()->Get(1).IsRange()) {
685 return this->AsUnion()->Get(1);
686 }
687 return nullptr;

690bool UnionType::Wellformed() const {
691 DisallowGarbageCollection no_gc;
692 // This checks the invariants of the union representation:
693 // 1. There are at least two elements.
694 // 2. The first element is a bitset, no other element is a bitset.
695 // 3. At most one element is a range, and it must be the second one.
696 // 4. No element is itself a union.
697 // 5. No element (except the bitset) is a subtype of any other.
698 // 6. If there is a range, then the bitset type does not contain
699 // plain number bits.
700 DCHECK_LE(2, this->Length()); // (1)
701 DCHECK(this->Get(0).IsBitset()); // (2a)
702
703 for (int i = 0; i < this->Length(); ++i) {
704 if (i != 0) DCHECK(!this->Get(i).IsBitset()); // (2b)
705 if (i != 1) DCHECK(!this->Get(i).IsRange()); // (3)
706 DCHECK(!this->Get(i).IsUnion()); // (4)
707 for (int j = 0; j < this->Length(); ++j) {
708 if (i != j && i != 0) DCHECK(!this->Get(i).Is(this->Get(j))); // (5)
709 }
710 }
711 DCHECK(!this->Get(1).IsRange() ||
712 (BitsetType::NumberBits(this->Get(0).AsBitset()) ==
713 BitsetType::kNone)); // (6)
714 return true;
715}

718// Union and intersection
719
720Type Type::Intersect(Type type1, Type type2, Zone* zone) {
721 // Fast case: bit sets.
722 if (type1.IsBitset() && type2.IsBitset()) {
723 return NewBitset(type1.AsBitset() & type2.AsBitset());
724 }
725
726 // Fast case: top or bottom types.
727 if (type1.IsNone() || type2.IsAny()) return type1; // Shortcut.
728 if (type2.IsNone() || type1.IsAny()) return type2; // Shortcut.
729
730 // Semi-fast case.
731 if (type1.Is(type2)) return type1;
732 if (type2.Is(type1)) return type2;
733
734 // Slow case: create union.
735
736 // Semantic subtyping check - this is needed for consistency with the
737 // semi-fast case above.
738 if (type1.Is(type2)) {
739 type2 = Any();
740 } else if (type2.Is(type1)) {
741 type1 = Any();
742 }
743
744 bitset bits = type1.BitsetGlb() & type2.BitsetGlb();
745 int size1 = type1.IsUnion() ? type1.AsUnion()->Length() : 1;
746 int size2 = type2.IsUnion() ? type2.AsUnion()->Length() : 1;
747 int size;
748 if (base::bits::SignedAddOverflow32(size1, size2, &size)) return Any();
749 if (base::bits::SignedAddOverflow32(size, 2, &size)) return Any();
750 UnionType* result = UnionType::New(size, zone);
751 size = 0;
752
753 // Deal with bitsets.
754 result->Set(size++, NewBitset(bits));
755
756 RangeType::Limits lims = RangeType::Limits::Empty();
757 size = IntersectAux(type1, type2, result, size, &lims, zone);
758
759 // If the range is not empty, then insert it into the union and
760 // remove the number bits from the bitset.
761 if (!lims.IsEmpty()) {
762 size = UpdateRange(Type::Range(lims, zone), result, size, zone);
763
764 // Remove the number bits.
765 bitset number_bits = BitsetType::NumberBits(bits);
766 bits &= ~number_bits;
767 result->Set(0, NewBitset(bits));
768 }
769 return NormalizeUnion(result, size, zone);
770}
771
772int Type::UpdateRange(Type range, UnionType* result, int size, Zone* zone) {
773 if (size == 1) {
774 result->Set(size++, range);
775 } else {
776 // Make space for the range.
777 result->Set(size++, result->Get(1));
778 result->Set(1, range);
779 }
780
781 // Remove any components that just got subsumed.
782 for (int i = 2; i < size;) {
783 if (result->Get(i).Is(range)) {
784 result->Set(i, result->Get(--size));
785 } else {
786 ++i;
787 }
788 }
789 return size;
790}

792RangeType::Limits Type::ToLimits(bitset bits, Zone* zone) {
793 bitset number_bits = BitsetType::NumberBits(bits);
794
795 if (number_bits == BitsetType::kNone) {
796 return RangeType::Limits::Empty();
797 }
798
799 return RangeType::Limits(BitsetType::Min(number_bits),
800 BitsetType::Max(number_bits));
801}

803RangeType::Limits Type::IntersectRangeAndBitset(Type range, Type bitset,
804 Zone* zone) {
805 RangeType::Limits range_lims(range.AsRange());
806 RangeType::Limits bitset_lims = ToLimits(bitset.AsBitset(), zone);
807 return RangeType::Limits::Intersect(range_lims, bitset_lims);
808}

810int Type::IntersectAux(Type lhs, Type rhs, UnionType* result, int size,
811 RangeType::Limits* lims, Zone* zone) {
812 if (lhs.IsUnion()) {
813 for (int i = 0, n = lhs.AsUnion()->Length(); i < n; ++i) {
814 size = IntersectAux(lhs.AsUnion()->Get(i), rhs, result, size, lims, zone);
815 }
816 return size;
817 }
818 if (rhs.IsUnion()) {
819 for (int i = 0, n = rhs.AsUnion()->Length(); i < n; ++i) {
820 size = IntersectAux(lhs, rhs.AsUnion()->Get(i), result, size, lims, zone);
821 }
822 return size;
823 }
824
825 if (BitsetType::IsNone(lhs.BitsetLub() & rhs.BitsetLub())) return size;
826
827 if (lhs.IsRange()) {
828 if (rhs.IsBitset()) {
829 RangeType::Limits lim = IntersectRangeAndBitset(lhs, rhs, zone);
830
831 if (!lim.IsEmpty()) {
832 *lims = RangeType::Limits::Union(lim, *lims);
833 }
834 return size;
835 }
836 if (rhs.IsRange()) {
837 RangeType::Limits lim = RangeType::Limits::Intersect(
838 RangeType::Limits(lhs.AsRange()), RangeType::Limits(rhs.AsRange()));
839 if (!lim.IsEmpty()) {
840 *lims = RangeType::Limits::Union(lim, *lims);
841 }
842 }
843 return size;
844 }
845 if (rhs.IsRange()) {
846 // This case is handled symmetrically above.
847 return IntersectAux(rhs, lhs, result, size, lims, zone);
848 }
849 if (lhs.IsBitset() || rhs.IsBitset()) {
850 return AddToUnion(lhs.IsBitset() ? rhs : lhs, result, size, zone);
851 }
852 if (lhs.SimplyEquals(rhs)) {
853 return AddToUnion(lhs, result, size, zone);
854 }
855 return size;
856}
857
858// Make sure that we produce a well-formed range and bitset:
859// If the range is non-empty, the number bits in the bitset should be
860// clear. Moreover, if we have a canonical range (such as Signed32),
861// we want to produce a bitset rather than a range.
862Type Type::NormalizeRangeAndBitset(Type range, bitset* bits, Zone* zone) {
863 // Fast path: If the bitset does not mention numbers, we can just keep the
864 // range.
865 bitset number_bits = BitsetType::NumberBits(*bits);
866 if (number_bits == 0) {
867 return range;
868 }
869
870 // If the range is semantically contained within the bitset, return None and
871 // leave the bitset untouched.
872 bitset range_lub = range.BitsetLub();
873 if (BitsetType::Is(range_lub, *bits)) {
874 return None();
875 }
876
877 // Slow path: reconcile the bitset range and the range.
878 double bitset_min = BitsetType::Min(number_bits);
879 double bitset_max = BitsetType::Max(number_bits);
880
881 double range_min = range.Min();
882 double range_max = range.Max();
883
884 // Remove the number bits from the bitset, they would just confuse us now.
885 // NOTE: bits contains OtherNumber iff bits contains PlainNumber, in which
886 // case we already returned after the subtype check above.
887 *bits &= ~number_bits;
888
889 if (range_min <= bitset_min && range_max >= bitset_max) {
890 // Bitset is contained within the range, just return the range.
891 return range;
892 }
893
894 if (bitset_min < range_min) {
895 range_min = bitset_min;
896 }
897 if (bitset_max > range_max) {
898 range_max = bitset_max;
899 }
900 return Type::Range(range_min, range_max, zone);
901}

949Type Type::Union(Type type1, Type type2, Zone* zone) {
950 // Fast case: bit sets.
951 if (type1.IsBitset() && type2.IsBitset()) {
952 return NewBitset(type1.AsBitset() | type2.AsBitset());
953 }
954
955 // Fast case: top or bottom types.
956 if (type1.IsAny() || type2.IsNone()) return type1;
957 if (type2.IsAny() || type1.IsNone()) return type2;
958
959 // Semi-fast case.
960 if (type1.Is(type2)) return type2;
961 if (type2.Is(type1)) return type1;
962
963 // Slow case: create union.
964 int size1 = type1.IsUnion() ? type1.AsUnion()->Length() : 1;
965 int size2 = type2.IsUnion() ? type2.AsUnion()->Length() : 1;
966 int size;
967 if (base::bits::SignedAddOverflow32(size1, size2, &size)) return Any();
968 if (base::bits::SignedAddOverflow32(size, 2, &size)) return Any();
969 UnionType* result = UnionType::New(size, zone);
970 size = 0;
971
972 // Compute the new bitset.
973 bitset new_bitset = type1.BitsetGlb() | type2.BitsetGlb();
974
975 // Deal with ranges.
976 Type range = None();
977 Type range1 = type1.GetRange();
978 Type range2 = type2.GetRange();
979 if (range1 != nullptr && range2 != nullptr) {
980 RangeType::Limits lims =
981 RangeType::Limits::Union(RangeType::Limits(range1.AsRange()),
982 RangeType::Limits(range2.AsRange()));
983 Type union_range = Type::Range(lims, zone);
984 range = NormalizeRangeAndBitset(union_range, &new_bitset, zone);
985 } else if (range1 != nullptr) {
986 range = NormalizeRangeAndBitset(range1, &new_bitset, zone);
987 } else if (range2 != nullptr) {
988 range = NormalizeRangeAndBitset(range2, &new_bitset, zone);
989 }
990 Type bits = NewBitset(new_bitset);
991 result->Set(size++, bits);
992 if (!range.IsNone()) result->Set(size++, range);
993
994 size = AddToUnion(type1, result, size, zone);
995 size = AddToUnion(type2, result, size, zone);
996 return NormalizeUnion(result, size, zone);
997}

999// Add [type] to [result] unless [type] is bitset, range, or already subsumed.
1000// Return new size of [result].
1001int Type::AddToUnion(Type type, UnionType* result, int size, Zone* zone) {
1002 if (type.IsBitset() || type.IsRange()) return size;
1003 if (type.IsUnion()) {
1004 for (int i = 0, n = type.AsUnion()->Length(); i < n; ++i) {
1005 size = AddToUnion(type.AsUnion()->Get(i), result, size, zone);
1006 }
1007 return size;
1008 }
1009 for (int i = 0; i < size; ++i) {
1010 if (type.Is(result->Get(i))) return size;
1011 }
1012 result->Set(size++, type);
1013 return size;
1014}

1016Type Type::NormalizeUnion(UnionType* unioned, int size, Zone* zone) {
1017 DCHECK_LE(1, size);
1018 DCHECK(unioned->Get(0).IsBitset());
1019 // If the union has just one element, return it.
1020 if (size == 1) {
1021 return unioned->Get(0);
1022 }
1023 bitset bits = unioned->Get(0).AsBitset();
1024 // If the union only consists of a range, we can get rid of the union.
1025 if (size == 2 && bits == BitsetType::kNone) {
1026 if (unioned->Get(1).IsRange()) {
1027 return Type::Range(unioned->Get(1).AsRange()->Min(),
1028 unioned->Get(1).AsRange()->Max(), zone);
1029 }
1030 }
1031 unioned->Shrink(size);
1032 SLOW_DCHECK(unioned->Wellformed());
1033 return Type(unioned);
1034}
