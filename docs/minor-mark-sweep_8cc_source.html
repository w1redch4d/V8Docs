160int EstimateMaxNumberOfRemeberedSets(Heap* heap) {
161 // old space, lo space, trusted space and trusted lo space can have a maximum
162 // of two remembered sets (OLD_TO_NEW and OLD_TO_NEW_BACKGROUND).
163 // Code space and code lo space can have typed OLD_TO_NEW in addition.
164 return 2 * (heap->old_space()->CountTotalPages() +
165 heap->lo_space()->PageCount() +
166 heap->trusted_space()->CountTotalPages() +
167 heap->trusted_lo_space()->PageCount()) +
168 3 * (heap->code_space()->CountTotalPages() +
169 heap->code_lo_space()->PageCount());
170}

174std::vector<YoungGenerationRememberedSetsMarkingWorklist::MarkingItem>
175YoungGenerationRememberedSetsMarkingWorklist::CollectItems(Heap* heap) {
176 std::vector<MarkingItem> items;
177 int max_remembered_set_count = EstimateMaxNumberOfRemeberedSets(heap);
178 items.reserve(max_remembered_set_count);
179 OldGenerationMemoryChunkIterator::ForAll(
180 heap, [&items](MutablePageMetadata* chunk) {
181 SlotSet* slot_set = chunk->ExtractSlotSet<OLD_TO_NEW>();
182 SlotSet* background_slot_set =
183 chunk->ExtractSlotSet<OLD_TO_NEW_BACKGROUND>();
184 if (slot_set || background_slot_set) {
185 items.emplace_back(chunk, MarkingItem::SlotsType::kRegularSlots,
186 slot_set, background_slot_set);
187 }
188 if (TypedSlotSet* typed_slot_set =
189 chunk->ExtractTypedSlotSet<OLD_TO_NEW>()) {
190 DCHECK(IsAnyCodeSpace(chunk->owner_identity()));
191 items.emplace_back(chunk, MarkingItem::SlotsType::kTypedSlots,
192 typed_slot_set);
193 }
194 });
195 DCHECK_LE(items.size(), max_remembered_set_count);
196 return items;
197}

199void YoungGenerationRememberedSetsMarkingWorklist::MarkingItem::
200 MergeAndDeleteRememberedSets() {
201 DCHECK(IsAcquired());
202 if (slots_type_ == SlotsType::kRegularSlots) {
203 if (slot_set_)
204 RememberedSet<OLD_TO_NEW>::MergeAndDelete(chunk_, std::move(*slot_set_));
205 if (background_slot_set_)
206 RememberedSet<OLD_TO_NEW_BACKGROUND>::MergeAndDelete(
207 chunk_, std::move(*background_slot_set_));
208 } else {
209 DCHECK_EQ(slots_type_, SlotsType::kTypedSlots);
210 DCHECK_NULL(background_slot_set_);
211 if (typed_slot_set_)
212 RememberedSet<OLD_TO_NEW>::MergeAndDeleteTyped(
213 chunk_, std::move(*typed_slot_set_));
214 }
215}

224 DCHECK_EQ(slots_type_, SlotsType::kTypedSlots);

226 if (typed_slot_set_)

339void MinorMarkSweepCollector::StartMarking(bool force_use_background_threads) {
340#if defined(VERIFY_HEAP) && !V8_ENABLE_STICKY_MARK_BITS_BOOL
341 if (v8_flags.verify_heap) {
342 for (PageMetadata* page : *heap_->new_space()) {
343 CHECK(page->marking_bitmap()->IsClean());
344 }
345 }
346#endif // defined(VERIFY_HEAP) && !V8_ENABLE_STICKY_MARK_BITS_BOOL
347
348 // The state for background thread is saved here and maintained for the whole
349 // GC cycle. Both CppHeap and regular V8 heap will refer to this flag.
350 CHECK(!use_background_threads_in_cycle_.has_value());
351 // Once we decided to start concurrent marking we always need to use
352 // background threads, this is because Minor MS doesn't perform incremental
353 // marking. ShouldUseBackgroundThreads() on worker isolates can be updated
354 // concurrently from the main thread outside a task, so we shouldn't invoke it
355 // here again as it could return a different result.
356 use_background_threads_in_cycle_ =
357 force_use_background_threads || heap_->ShouldUseBackgroundThreads();
358
359 auto* cpp_heap = CppHeap::From(heap_->cpp_heap_);
360 // CppHeap's marker must be initialized before the V8 marker to allow
361 // exchanging of worklists.
362 if (cpp_heap && cpp_heap->generational_gc_supported()) {
363 TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_MARK_EMBEDDER_PROLOGUE);
364 cpp_heap->InitializeMarking(CppHeap::CollectionType::kMinor);
365 }
366 DCHECK_NULL(ephemeron_table_list_);
367 ephemeron_table_list_ = std::make_unique<EphemeronRememberedSet::TableList>();
368 DCHECK_NULL(marking_worklists_);
369 marking_worklists_ = std::make_unique<MarkingWorklists>();
370 DCHECK_NULL(main_marking_visitor_);
371 DCHECK_NULL(pretenuring_feedback_);
372 if (v8_flags.sticky_mark_bits) {
373 DCHECK(ExternalPointerRememberedSetsEmpty(heap_->sticky_space()));
374 } else {
375 DCHECK(ExternalPointerRememberedSetsEmpty(
376 heap_->paged_new_space()->paged_space()));
377 }
378 pretenuring_feedback_ =
379 std::make_unique<PretenuringHandler::PretenuringFeedbackMap>(
380 PretenuringHandler::kInitialFeedbackCapacity);
381 main_marking_visitor_ = std::make_unique<YoungGenerationMainMarkingVisitor>(
382 heap_, pretenuring_feedback_.get());
383 DCHECK_NULL(remembered_sets_marking_handler_);
384 remembered_sets_marking_handler_ =
385 std::make_unique<YoungGenerationRememberedSetsMarkingWorklist>(heap_);
386 if (cpp_heap && cpp_heap->generational_gc_supported()) {
387 TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_MARK_EMBEDDER_PROLOGUE);
388 // StartTracing immediately starts marking which requires V8 worklists to
389 // be set up.
390 cpp_heap->StartMarking();
391 }

394void MinorMarkSweepCollector::Finish() {
395 TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_FINISH);
396
397 use_background_threads_in_cycle_.reset();
398
399 {
400 TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_FINISH_ENSURE_CAPACITY);
401 switch (resize_new_space_) {
402 case ResizeNewSpaceMode::kShrink:
403 heap_->ReduceNewSpaceSize();
404 break;
405 case ResizeNewSpaceMode::kGrow:
406 heap_->ExpandNewSpaceSize();
407 break;
408 case ResizeNewSpaceMode::kNone:
409 break;
410 }
411 resize_new_space_ = ResizeNewSpaceMode::kNone;
412
413 if (!v8_flags.sticky_mark_bits &&
414 !heap_->new_space()->EnsureCurrentCapacity()) {
415 heap_->FatalProcessOutOfMemory("NewSpace::EnsureCurrentCapacity");
416 }
417 }
418
419 if (!v8_flags.sticky_mark_bits) {
420 heap_->new_space()->GarbageCollectionEpilogue();
421 }

424void MinorMarkSweepCollector::CollectGarbage() {
425 DCHECK(!heap_->mark_compact_collector()->in_use());
426 DCHECK(heap_->use_new_space());
427 DCHECK(!heap_->array_buffer_sweeper()->sweeping_in_progress());
428 DCHECK(!sweeper()->AreMinorSweeperTasksRunning());
429 if (v8_flags.sticky_mark_bits) {
430 DCHECK(sweeper()->IsSweepingDoneForSpace(OLD_SPACE));
431 } else {
432 DCHECK(sweeper()->IsSweepingDoneForSpace(NEW_SPACE));
433 }
434
435 heap_->new_lo_space()->ResetPendingObject();
436
437 is_in_atomic_pause_.store(true, std::memory_order_relaxed);
438
439 MarkLiveObjects();
440 ClearNonLiveReferences();
441#ifdef VERIFY_HEAP
442 if (v8_flags.verify_heap) {
443 TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_MARK_VERIFY);
444 YoungGenerationMarkingVerifier verifier(heap_);
445 verifier.Run();
446 }
447#endif // VERIFY_HEAP
448
449 if (auto* cpp_heap = CppHeap::From(heap_->cpp_heap_)) {
450 cpp_heap->FinishMarkingAndProcessWeakness();
451 }
452
453 Sweep();
454 Finish();
455
456 auto* isolate = heap_->isolate();
457 isolate->global_handles()->UpdateListOfYoungNodes();
458 isolate->traced_handles()->UpdateListOfYoungNodes();
459
460 isolate->stack_guard()->ClearGC();
461 gc_finalization_requested_.store(false, std::memory_order_relaxed);
462 is_in_atomic_pause_.store(false, std::memory_order_relaxed);
463}

476 void ProcessYoungObjects() {
477 DCHECK(v8_flags.always_use_string_forwarding_table);
478 StringForwardingTable* forwarding_table =
479 isolate_->string_forwarding_table();
480 forwarding_table->IterateElements(
481 [&](StringForwardingTable::Record* record) {
482 ClearNonLiveYoungObjects(record);
483 });
484 }

486 private:

488 Tagged<Object> original = record->OriginalStringObject(isolate_);

490 DCHECK_EQ(original, StringForwardingTable::deleted_element());
491 return;
492 }
493 Tagged<String> original_string = Cast<String>(original);
494 if (!HeapLayout::InYoungGeneration(original_string)) return;
495 if (!marking_state_->IsMarked(original_string)) {
496 DisposeExternalResource(record);
497 record->set_original_string(StringForwardingTable::deleted_element());
498 }
499 }

513void MinorMarkSweepCollector::ClearNonLiveReferences() {
514 TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_CLEAR);
515
516 if (V8_UNLIKELY(v8_flags.always_use_string_forwarding_table)) {
517 TRACE_GC(heap_->tracer(),
518 GCTracer::Scope::MINOR_MS_CLEAR_STRING_FORWARDING_TABLE);
519 // Clear non-live objects in the string fowarding table.
520 YoungStringForwardingTableCleaner forwarding_table_cleaner(heap_);
521 forwarding_table_cleaner.ProcessYoungObjects();
522 }
523
524 Heap::ExternalStringTable& external_string_table =
525 heap_->external_string_table_;
526 if (external_string_table.HasYoung()) {
527 TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_CLEAR_STRING_TABLE);
528 // Internalized strings are always stored in old space, so there is no
529 // need to clean them here.
530 ExternalStringTableCleanerVisitor<
531 ExternalStringTableCleaningMode::kYoungOnly>
532 external_visitor(heap_);
533 external_string_table.IterateYoung(&external_visitor);
534 external_string_table.CleanUpYoung();
535 }
536
537 Isolate* isolate = heap_->isolate();
538 if (isolate->global_handles()->HasYoung() ||
539 isolate->traced_handles()->HasYoung()) {
540 TRACE_GC(heap_->tracer(),
541 GCTracer::Scope::MINOR_MS_CLEAR_WEAK_GLOBAL_HANDLES);
542 isolate->global_handles()->ProcessWeakYoungObjects(
543 nullptr, &IsUnmarkedObjectInYoungGeneration);
544 if (auto* cpp_heap = CppHeap::From(heap_->cpp_heap_);
545 cpp_heap && cpp_heap->generational_gc_supported()) {
546 isolate->traced_handles()->ResetYoungDeadNodes(
547 &IsUnmarkedObjectInYoungGeneration);
548 } else {
549 isolate->traced_handles()->ProcessYoungObjects(
550 nullptr, &IsUnmarkedObjectInYoungGeneration);
551 }
552 }
553
554 // Clear ephemeron entries from EphemeronHashTables in the young generation
555 // whenever the entry has a dead young generation key.
556 //
557 // Worklist is collected during marking.
558 {
559 DCHECK_NOT_NULL(ephemeron_table_list_);
560 EphemeronRememberedSet::TableList::Local local_ephemeron_table_list(
561 *ephemeron_table_list_);
562 Tagged<EphemeronHashTable> table;
563 while (local_ephemeron_table_list.Pop(&table)) {
564 for (InternalIndex i : table->IterateEntries()) {
565 // Keys in EphemeronHashTables must be heap objects.
566 HeapObjectSlot key_slot(
567 table->RawFieldOfElementAt(EphemeronHashTable::EntryToIndex(i)));
568 Tagged<HeapObject> key = key_slot.ToHeapObject();
569 if (HeapLayout::InYoungGeneration(key) &&
570 non_atomic_marking_state_->IsUnmarked(key)) {
571 table->RemoveEntry(i);
572 }
573 }
574 }
575 }
576 ephemeron_table_list_.reset();
577
578 // Clear ephemeron entries from EphemeronHashTables in the old generation
579 // whenever the entry has a dead young generation key.
580 //
581 // Does not need to be iterated as roots but is maintained in the GC to avoid
582 // treating keys as strong. The set is populated from the write barrier and
583 // the sweeper during promoted pages iteration.
584 auto* table_map = heap_->ephemeron_remembered_set()->tables();
585 for (auto it = table_map->begin(); it != table_map->end();) {
586 Tagged<EphemeronHashTable> table = it->first;
587 auto& indices = it->second;
588 for (auto iti = indices.begin(); iti != indices.end();) {
589 // Keys in EphemeronHashTables must be heap objects.
590 HeapObjectSlot key_slot(table->RawFieldOfElementAt(
591 EphemeronHashTable::EntryToIndex(InternalIndex(*iti))));
592 Tagged<HeapObject> key = key_slot.ToHeapObject();
593 // There may be old generation entries left in the remembered set as
594 // MinorMS only promotes pages after clearing non-live references.
595 if (!HeapLayout::InYoungGeneration(key)) {
596 iti = indices.erase(iti);
597 } else if (non_atomic_marking_state_->IsUnmarked(key)) {
598 table->RemoveEntry(InternalIndex(*iti));
599 iti = indices.erase(iti);
600 } else {
601 ++iti;
602 }
603 }
604
605 if (indices.empty()) {
606 it = table_map->erase(it);
607 } else {
608 ++it;
609 }
610 }

742 // finished as it will reset page flags.

791void MinorMarkSweepCollector::TraceFragmentation() {
792 PagedSpaceBase* new_space =
793 v8_flags.sticky_mark_bits ? heap_->sticky_space()
794 : static_cast<PagedSpaceBase*>(
795 heap_->paged_new_space()->paged_space());
796 PtrComprCageBase cage_base(heap_->isolate());
797 const std::array<size_t, 4> free_size_class_limits = {0, 1024, 2048, 4096};
798 size_t free_bytes_of_class[free_size_class_limits.size()] = {0};
799 size_t live_bytes = 0;
800 size_t allocatable_bytes = 0;
801 for (PageMetadata* p : *new_space) {
802 Address free_start = p->area_start();
803 for (auto [object, size] : LiveObjectRange(p)) {
804 Address free_end = object.address();
805 if (free_end != free_start) {
806 size_t free_bytes = free_end - free_start;
807 int free_bytes_index = 0;
808 for (auto free_size_class_limit : free_size_class_limits) {
809 if (free_bytes >= free_size_class_limit) {
810 free_bytes_of_class[free_bytes_index] += free_bytes;
811 }
812 free_bytes_index++;
813 }
814 }
815 live_bytes += size;
816 free_start = free_end + size;
817 }
818 const Address top = heap_->NewSpaceTop();
819 size_t area_end = p->Contains(top) ? top : p->area_end();
820 if (free_start != area_end) {
821 size_t free_bytes = area_end - free_start;
822 int free_bytes_index = 0;
823 for (auto free_size_class_limit : free_size_class_limits) {
824 if (free_bytes >= free_size_class_limit) {
825 free_bytes_of_class[free_bytes_index] += free_bytes;
826 }
827 free_bytes_index++;
828 }
829 }
830 allocatable_bytes += area_end - p->area_start();
831 CHECK_EQ(allocatable_bytes, live_bytes + free_bytes_of_class[0]);
832 }
833 PrintIsolate(heap_->isolate(),
834 "Minor Mark-Sweep Fragmentation: allocatable_bytes=%zu "
835 "live_bytes=%zu "
836 "free_bytes=%zu free_bytes_1K=%zu free_bytes_2K=%zu "
837 "free_bytes_4K=%zu\n",
838 allocatable_bytes, live_bytes, free_bytes_of_class[0],
839 free_bytes_of_class[1], free_bytes_of_class[2],
840 free_bytes_of_class[3]);
841}

852bool ShouldMovePage(PageMetadata* p, intptr_t live_bytes,
853 intptr_t wasted_bytes) {
854 DCHECK(v8_flags.page_promotion);
855 DCHECK(!v8_flags.sticky_mark_bits);
856 Heap* heap = p->heap();
857 DCHECK(!p->Chunk()->NeverEvacuate());
858 const bool should_move_page =
859 ((live_bytes + wasted_bytes) > NewSpacePageEvacuationThreshold() ||
860 (p->AllocatedLabSize() == 0)) &&
861 (heap->new_space()->IsPromotionCandidate(p)) &&
862 heap->CanExpandOldGeneration(live_bytes);
863 if (v8_flags.trace_page_promotions) {
864 PrintIsolate(
865 heap->isolate(),
866 "[Page Promotion] %p: collector=mms, should move: %d"
867 ", live bytes = %zu, wasted bytes = %zu, promotion threshold = %zu"
868 ", allocated labs size = %zu\n",
869 p, should_move_page, live_bytes, wasted_bytes,
870 NewSpacePageEvacuationThreshold(), p->AllocatedLabSize());
871 }
872 if (!should_move_page &&
873 (p->AgeInNewSpace() == v8_flags.minor_ms_max_page_age)) {
874 // Don't allocate on old pages so that recently allocated objects on the
875 // page get a chance to die young. The page will be force promoted on the
876 // next GC because `AllocatedLabSize` will be 0.
877 p->Chunk()->SetFlagNonExecutable(MemoryChunk::NEVER_ALLOCATE_ON_PAGE);
878 }
879 return should_move_page;
880}

910bool MinorMarkSweepCollector::StartSweepNewSpace() {
911 TRACE_GC(heap_->tracer(), GCTracer::Scope::MINOR_MS_SWEEP_NEW);
912 PagedSpaceForNewSpace* paged_space = heap_->paged_new_space()->paged_space();
913 paged_space->ClearAllocatorState();
914
915 int will_be_swept = 0;
916 bool has_promoted_pages = false;
917
918 DCHECK_EQ(Heap::ResizeNewSpaceMode::kNone, resize_new_space_);
919 resize_new_space_ = heap_->ShouldResizeNewSpace();
920 if (resize_new_space_ == Heap::ResizeNewSpaceMode::kShrink) {
921 paged_space->StartShrinking();
922 }
923
924 for (auto it = paged_space->begin(); it != paged_space->end();) {
925 PageMetadata* p = *(it++);
926 DCHECK(p->SweepingDone());
927
928 intptr_t live_bytes_on_page = p->live_bytes();
929 if (live_bytes_on_page == 0) {
930 if (paged_space->ShouldReleaseEmptyPage()) {
931 paged_space->ReleasePage(p);
932 } else {
933 sweeper()->SweepEmptyNewSpacePage(p);
934 }
935 continue;
936 }
937
938 if (ShouldMovePage(p, live_bytes_on_page, p->wasted_memory())) {
939 EvacuateExternalPointerReferences(p);
940 heap_->new_space()->PromotePageToOldSpace(p);
941 has_promoted_pages = true;
942 sweeper()->AddPromotedPage(p);
943 } else {
944 // Page is not promoted. Sweep it instead.
945 sweeper()->AddNewSpacePage(p);
946 will_be_swept++;
947 }
948 }
949
950#ifdef V8_COMPRESS_POINTERS
951 // Now that we have evacuated any external pointers, rebuild EPT free-lists
952 // for the new space.
953 heap_->isolate()->external_pointer_table().SweepAndCompact(
954 heap_->young_external_pointer_space(), heap_->isolate()->counters());
955#endif
956
957 if (v8_flags.gc_verbose) {
958 PrintIsolate(heap_->isolate(),
959 "sweeping: space=%s initialized_for_sweeping=%d",
960 ToString(paged_space->identity()), will_be_swept);
961 }
962
963 return has_promoted_pages;
964}
