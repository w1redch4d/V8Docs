52class GlobalObjectNameResolver final
53 : public v8::HeapProfiler::ObjectNameResolver {
54 public:
55 explicit GlobalObjectNameResolver(V8InspectorSessionImpl* session)
56 : m_offset(0), m_strings(10000), m_session(session) {}
57
58 const char* GetName(v8::Local<v8::Object> object) override {
59 v8::Local<v8::Context> creationContext;
60 if (!object->GetCreationContext(m_session->inspector()->isolate())
61 .ToLocal(&creationContext)) {
62 return "";
63 }
64 InspectedContext* context = m_session->inspector()->getContext(
65 m_session->contextGroupId(),
66 InspectedContext::contextId(creationContext));
67 if (!context) return "";
68 String16 name = context->origin();
69 size_t length = name.length();
70 if (m_offset + length + 1 >= m_strings.size()) return "";
71 for (size_t i = 0; i < length; ++i) {
72 UChar ch = name[i];
73 m_strings[m_offset + i] = ch > 0xFF ? '?' : static_cast<char>(ch);
74 }
75 m_strings[m_offset + length] = '\0';
76 char* result = &*m_strings.begin() + m_offset;
77 m_offset += length + 1;
78 return result;
79 }
80
81 private:
82 size_t m_offset;
83 std::vector<char> m_strings;
84 V8InspectorSessionImpl* m_session;
85};

160class V8HeapProfilerAgentImpl::GCTask : public v8::Task {
161 public:
162 GCTask(v8::Isolate* isolate, std::shared_ptr<AsyncCallbacks> asyncCallbacks)
163 : m_isolate(isolate), m_asyncCallbacks(asyncCallbacks) {}
164
165 void Run() override {
166 std::shared_ptr<AsyncCallbacks> asyncCallbacks = m_asyncCallbacks.lock();
167 if (!asyncCallbacks) return;
168 v8::base::MutexGuard lock(&asyncCallbacks->m_mutex);
169 if (asyncCallbacks->m_canceled) return;
170 v8::debug::ForceGarbageCollection(m_isolate,
171 v8::StackState::kNoHeapPointers);
172 for (auto& callback : asyncCallbacks->m_gcCallbacks) {
173 callback->sendSuccess();
174 }
175 asyncCallbacks->m_gcCallbacks.clear();
176 }
177
178 private:
179 v8::Isolate* m_isolate;
180 std::weak_ptr<AsyncCallbacks> m_asyncCallbacks;

198class V8HeapProfilerAgentImpl::HeapSnapshotTask : public v8::Task {
199 public:
200 HeapSnapshotTask(V8HeapProfilerAgentImpl* agent,
201 std::shared_ptr<AsyncCallbacks> asyncCallbacks,
202 HeapSnapshotProtocolOptions protocolOptions,
203 std::unique_ptr<TakeHeapSnapshotCallback> callback)
204 : m_agent(agent),
205 m_asyncCallbacks(asyncCallbacks),
206 m_protocolOptions(protocolOptions),
207 m_callback(std::move(callback)) {}
208
209 void Run() override { Run(cppgc::EmbedderStackState::kNoHeapPointers); }
210
211 void Run(cppgc::EmbedderStackState stackState) {
212 Response response = Response::Success();
213 {
214 // If the async callbacks object still exists and is not canceled, then
215 // the V8HeapProfilerAgentImpl still exists, so we can safely take a
216 // snapshot.
217 std::shared_ptr<AsyncCallbacks> asyncCallbacks = m_asyncCallbacks.lock();
218 if (!asyncCallbacks) return;
219 v8::base::MutexGuard lock(&asyncCallbacks->m_mutex);
220 if (asyncCallbacks->m_canceled) return;
221
222 auto& heapSnapshotTasks = asyncCallbacks->m_heapSnapshotTasks;
223 auto it =
224 std::find(heapSnapshotTasks.begin(), heapSnapshotTasks.end(), this);
225 if (it == heapSnapshotTasks.end()) {
226 // This task must have already been run. This can happen because the
227 // task was queued with PostNonNestableTask but then got run early by
228 // takePendingHeapSnapshots.
229 return;
230 }
231 heapSnapshotTasks.erase(it);
232
233 response = m_agent->takeHeapSnapshotNow(m_protocolOptions, stackState);
234 }
235
236 // The rest of this function runs without the mutex, because Node expects to
237 // be able to dispose the profiler agent during the callback, which would
238 // deadlock if this function still held the mutex. It's safe to call the
239 // callback without the mutex; the internal implementation of the callback
240 // uses weak pointers to avoid doing anything dangerous if other components
241 // have been disposed (see DomainDispatcher::Callback::sendIfActive).
242 if (response.IsSuccess()) {
243 m_callback->sendSuccess();
244 } else {
245 m_callback->sendFailure(std::move(response));
246 }
247 }
248
249 private:
250 V8HeapProfilerAgentImpl* m_agent;
251 std::weak_ptr<AsyncCallbacks> m_asyncCallbacks;
252 HeapSnapshotProtocolOptions m_protocolOptions;
253 std::unique_ptr<TakeHeapSnapshotCallback> m_callback;
254};

273void V8HeapProfilerAgentImpl::restore() {
274 if (m_state->booleanProperty(HeapProfilerAgentState::heapProfilerEnabled,
275 false))
276 m_frontend.resetProfiles();
277 if (m_state->booleanProperty(
278 HeapProfilerAgentState::heapObjectsTrackingEnabled, false))
279 startTrackingHeapObjectsInternal(m_state->booleanProperty(
280 HeapProfilerAgentState::allocationTrackingEnabled, false));

288 startSampling(

290 flags & v8::HeapProfiler::kSamplingIncludeObjectsCollectedByMajorGC,
291 flags & v8::HeapProfiler::kSamplingIncludeObjectsCollectedByMinorGC);
292 }

347void V8HeapProfilerAgentImpl::takeHeapSnapshot(
348 std::optional<bool> reportProgress,
349 std::optional<bool> treatGlobalObjectsAsRoots,
350 std::optional<bool> captureNumericValue,
351 std::optional<bool> exposeInternals,
352 std::unique_ptr<TakeHeapSnapshotCallback> callback) {
353 HeapSnapshotProtocolOptions protocolOptions(
354 std::move(reportProgress), std::move(treatGlobalObjectsAsRoots),
355 std::move(captureNumericValue), std::move(exposeInternals));
356 std::shared_ptr<v8::TaskRunner> task_runner =
357 v8::debug::GetCurrentPlatform()->GetForegroundTaskRunner(m_isolate);
358
359 // Heap snapshots can be more accurate if we wait until the stack is empty and
360 // run the garbage collector without conservative stack scanning, as done in
361 // V8HeapProfilerAgentImpl::collectGarbage. However, heap snapshots can also
362 // be requested while paused in the debugger, in which case the snapshot must
363 // be taken immediately with conservative stack scanning enabled.
364 if (m_session->inspector()->debugger()->isPaused() ||
365 !task_runner->NonNestableTasksEnabled()) {
366 Response response = takeHeapSnapshotNow(
367 protocolOptions, cppgc::EmbedderStackState::kMayContainHeapPointers);
368 if (response.IsSuccess()) {
369 callback->sendSuccess();
370 } else {
371 callback->sendFailure(std::move(response));
372 }
373 return;
374 }
375
376 std::unique_ptr<HeapSnapshotTask> task = std::make_unique<HeapSnapshotTask>(
377 this, m_asyncCallbacks, protocolOptions, std::move(callback));
378 m_asyncCallbacks->m_heapSnapshotTasks.push_back(task.get());
379 task_runner->PostNonNestableTask(std::move(task));
380}

382Response V8HeapProfilerAgentImpl::takeHeapSnapshotNow(
383 const HeapSnapshotProtocolOptions& protocolOptions,
384 cppgc::EmbedderStackState stackState) {
385 v8::HeapProfiler* profiler = m_isolate->GetHeapProfiler();
386 if (!profiler) return Response::ServerError("Cannot access v8 heap profiler");
387 std::unique_ptr<HeapSnapshotProgress> progress;
388 if (protocolOptions.m_reportProgress)
389 progress.reset(new HeapSnapshotProgress(&m_frontend));
390
391 GlobalObjectNameResolver resolver(m_session);
392 v8::HeapProfiler::HeapSnapshotOptions options;
393 options.global_object_name_resolver = &resolver;
394 options.control = progress.get();
395 options.snapshot_mode =
396 protocolOptions.m_exposeInternals ||
397 // Not treating global objects as roots results into exposing
398 // internals.
399 !protocolOptions.m_treatGlobalObjectsAsRoots
400 ? v8::HeapProfiler::HeapSnapshotMode::kExposeInternals
401 : v8::HeapProfiler::HeapSnapshotMode::kRegular;
402 options.numerics_mode =
403 protocolOptions.m_captureNumericValue
404 ? v8::HeapProfiler::NumericsMode::kExposeNumericValues
405 : v8::HeapProfiler::NumericsMode::kHideNumericValues;
406 options.stack_state = stackState;
407 const v8::HeapSnapshot* snapshot = profiler->TakeHeapSnapshot(options);
408 if (!snapshot) return Response::ServerError("Failed to take heap snapshot");
409 HeapSnapshotOutputStream stream(&m_frontend);
410 snapshot->Serialize(&stream);
411 const_cast<v8::HeapSnapshot*>(snapshot)->Delete();
412 return Response::Success();
413}

415Response V8HeapProfilerAgentImpl::getObjectByHeapObjectId(
416 const String16& heapSnapshotObjectId, std::optional<String16> objectGroup,
417 std::unique_ptr<protocol::Runtime::RemoteObject>* result) {
418 bool ok;
419 int id = heapSnapshotObjectId.toInteger(&ok);
420 if (!ok) return Response::ServerError("Invalid heap snapshot object id");
421
422 v8::HandleScope handles(m_isolate);
423 v8::Local<v8::Object> heapObject = objectByHeapObjectId(m_isolate, id);
424 if (heapObject.IsEmpty())
425 return Response::ServerError("Object is not available");
426
427 if (!m_session->inspector()->client()->isInspectableHeapObject(heapObject))
428 return Response::ServerError("Object is not available");
429
430 v8::Local<v8::Context> creationContext;
431 if (!heapObject->GetCreationContext(m_isolate).ToLocal(&creationContext)) {
432 return Response::ServerError("Object is not available");
433 }
434 *result = m_session->wrapObject(creationContext, heapObject,
435 objectGroup.value_or(""), false);
436 if (!*result) return Response::ServerError("Object is not available");
437 return Response::Success();
438}

494static constexpr v8::base::TimeDelta kDefaultTimerDelay =
495 v8::base::TimeDelta::FromMilliseconds(50);
496
497void V8HeapProfilerAgentImpl::onTimerImpl() {
498 v8::base::TimeTicks start = v8::base::TimeTicks::Now();
499 requestHeapStatsUpdate();
500 v8::base::TimeDelta elapsed = v8::base::TimeTicks::Now() - start;
501 if (m_hasTimer) {
502 // requestHeapStatsUpdate can take a long time on large heaps. To ensure
503 // that there is still some time for the thread to make progress on running
504 // JavaScript or doing other useful work, we'll adjust the timer delay here.
505 const v8::base::TimeDelta minAcceptableDelay =
506 std::max(elapsed * 2, kDefaultTimerDelay);
507 const v8::base::TimeDelta idealDelay =
508 std::max(elapsed * 3, kDefaultTimerDelay);
509 const v8::base::TimeDelta maxAcceptableDelay =
510 std::max(elapsed * 4, kDefaultTimerDelay);
511 if (m_timerDelayInSeconds < minAcceptableDelay.InSecondsF() ||
512 m_timerDelayInSeconds > maxAcceptableDelay.InSecondsF()) {
513 // The existing timer's speed is not very close to ideal, so cancel it and
514 // start a new timer.
515 m_session->inspector()->client()->cancelTimer(
516 reinterpret_cast<void*>(this));
517 m_timerDelayInSeconds = idealDelay.InSecondsF();
518 m_session->inspector()->client()->startRepeatingTimer(
519 m_timerDelayInSeconds, &V8HeapProfilerAgentImpl::onTimer,
520 reinterpret_cast<void*>(this));
521 }
522 }
523}

549Response V8HeapProfilerAgentImpl::startSampling(
550 std::optional<double> samplingInterval,
551 std::optional<bool> includeObjectsCollectedByMajorGC,
552 std::optional<bool> includeObjectsCollectedByMinorGC) {
553 v8::HeapProfiler* profiler = m_isolate->GetHeapProfiler();
554 if (!profiler) return Response::ServerError("Cannot access v8 heap profiler");
555 const unsigned defaultSamplingInterval = 1 << 15;
556 double samplingIntervalValue =
557 samplingInterval.value_or(defaultSamplingInterval);
558 if (samplingIntervalValue <= 0.0) {
559 return Response::ServerError("Invalid sampling interval");
560 }
561 m_state->setDouble(HeapProfilerAgentState::samplingHeapProfilerInterval,
562 samplingIntervalValue);
563 m_state->setBoolean(HeapProfilerAgentState::samplingHeapProfilerEnabled,
564 true);
565 int flags = v8::HeapProfiler::kSamplingForceGC;
566 if (includeObjectsCollectedByMajorGC.value_or(false)) {
567 flags |= v8::HeapProfiler::kSamplingIncludeObjectsCollectedByMajorGC;
568 }
569 if (includeObjectsCollectedByMinorGC.value_or(false)) {
570 flags |= v8::HeapProfiler::kSamplingIncludeObjectsCollectedByMinorGC;
571 }
572 m_state->setInteger(HeapProfilerAgentState::samplingHeapProfilerFlags, flags);
573 profiler->StartSamplingHeapProfiler(
574 static_cast<uint64_t>(samplingIntervalValue), 128,
575 static_cast<v8::HeapProfiler::SamplingFlags>(flags));
576 return Response::Success();
577}

580std::unique_ptr<protocol::HeapProfiler::SamplingHeapProfileNode>
581buildSampingHeapProfileNode(v8::Isolate* isolate,
582 const v8::AllocationProfile::Node* node) {
583 auto children = std::make_unique<
584 protocol::Array<protocol::HeapProfiler::SamplingHeapProfileNode>>();
585 for (const auto* child : node->children)
586 children->emplace_back(buildSampingHeapProfileNode(isolate, child));
587 size_t selfSize = 0;
588 for (const auto& allocation : node->allocations)
589 selfSize += allocation.size * allocation.count;

591 protocol::Runtime::CallFrame::create()
592 .setFunctionName(toProtocolString(isolate, node->name))
593 .setScriptId(String16::fromInteger(node->script_id))

596 .setColumnNumber(node->column_number - 1)

599 protocol::HeapProfiler::SamplingHeapProfileNode::create()

612 if (result.IsSuccess()) {

620Response V8HeapProfilerAgentImpl::getSamplingProfile(
621 std::unique_ptr<protocol::HeapProfiler::SamplingHeapProfile>* profile) {
622 v8::HeapProfiler* profiler = m_isolate->GetHeapProfiler();
623 // Need a scope as v8::AllocationProfile contains Local handles.
624 v8::HandleScope scope(m_isolate);
625 std::unique_ptr<v8::AllocationProfile> v8Profile(
626 profiler->GetAllocationProfile());
627 if (!v8Profile)
628 return Response::ServerError("V8 sampling heap profiler was not started.");
629 v8::AllocationProfile::Node* root = v8Profile->GetRootNode();
630 auto samples = std::make_unique<
631 protocol::Array<protocol::HeapProfiler::SamplingHeapProfileSample>>();
632 for (const auto& sample : v8Profile->GetSamples()) {
633 samples->emplace_back(
634 protocol::HeapProfiler::SamplingHeapProfileSample::create()
635 .setSize(sample.size * sample.count)
636 .setNodeId(sample.node_id)
637 .setOrdinal(static_cast<double>(sample.sample_id))
638 .build());
639 }
640 *profile = protocol::HeapProfiler::SamplingHeapProfile::create()
641 .setHead(buildSampingHeapProfileNode(m_isolate, root))
642 .setSamples(std::move(samples))
643 .build();
644 return Response::Success();
645}
