93 // The module_request value records the order in which modules are

245 // Find any implicitly indirect exports and make them explicit.
246 //
247 // An explicitly indirect export is an export entry arising from an export
248 // statement of the following form:
249 // export {a as c} from "X";
250 // An implicitly indirect export corresponds to
251 // export {b as c};
252 // in the presence of an import statement of the form
253 // import {a as b} from "X";
254 // This function finds such implicitly indirect export entries and rewrites
255 // them by filling in the import name and module request, as well as nulling
256 // out the local name. Effectively, it turns
257 // import {a as b} from "X"; export {b as c};
258 // into:
259 // import {a as b} from "X"; export {a as c} from "X";
260 // (The import entry is never deleted.)
261 void MakeIndirectExportsExplicit(Zone* zone);

268 int AddModuleRequest(const AstRawString* specifier,
269 const ModuleImportPhase import_phase,
270 const ImportAttributes* import_attributes,
271 Scanner::Location specifier_loc, Zone* zone) {
272 DCHECK_NOT_NULL(specifier);
273 int module_requests_count = static_cast<int>(module_requests_.size());
274 auto it = module_requests_
275 .insert(zone->New<AstModuleRequest>(
276 specifier, import_phase, import_attributes,
277 specifier_loc.beg_pos, module_requests_count))
278 .first;
279 return (*it)->index();
280 }
