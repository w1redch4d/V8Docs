35 public:
36 CpuSampler(Isolate* isolate, SamplingEventsProcessor* processor)
37 : sampler::Sampler(reinterpret_cast<v8::Isolate*>(isolate)),
38 processor_(processor),
39 perThreadData_(isolate->FindPerThreadDataForThisThread()) {}

41 void SampleStack(const v8::RegisterState& regs) override {
42 Isolate* isolate = reinterpret_cast<Isolate*>(this->isolate());
43 if (isolate->was_locker_ever_used() &&
44 (!isolate->thread_manager()->IsLockedByThread(
45 perThreadData_->thread_id()) ||
46 perThreadData_->thread_state() != nullptr)) {
47 ProfilerStats::Instance()->AddReason(
48 ProfilerStats::Reason::kIsolateNotLocked);
49 return;
50 }
51#if V8_HEAP_USE_PKU_JIT_WRITE_PROTECT
52 i::RwxMemoryWriteScope::SetDefaultPermissionsForSignalHandler();
53#endif
54 TickSample* sample = processor_->StartTickSample();
55 if (sample == nullptr) {
56 ProfilerStats::Instance()->AddReason(
57 ProfilerStats::Reason::kTickBufferFull);
58 return;
59 }
60 // Every bailout up until here resulted in a dropped sample. From now on,
61 // the sample is created in the buffer.
62 sample->Init(isolate, regs, TickSample::kIncludeCEntryFrame,
63 /* update_stats */ true,
64 /* use_simulator_reg_state */ true, processor_->period());
65 if (is_counting_samples_ && !sample->timestamp.IsNull()) {
66 if (sample->state == JS) ++js_sample_count_;
67 if (sample->state == EXTERNAL) ++external_sample_count_;
68 }
69 processor_->FinishTickSample();
70 }
71
72 private:
73 SamplingEventsProcessor* processor_;
74 Isolate::PerIsolateThreadData* perThreadData_;
75};

151void ProfilerEventsProcessor::AddDeoptStack(Address from, int fp_to_sp_delta) {
152 TickSampleEventRecord record(last_code_event_id_);
153 RegisterState regs;
154 Address fp = isolate_->c_entry_fp(isolate_->thread_local_top());
155 regs.sp = reinterpret_cast<void*>(fp - fp_to_sp_delta);
156 regs.fp = reinterpret_cast<void*>(fp);
157 regs.pc = reinterpret_cast<void*>(from);
158 record.sample.Init(isolate_, regs, TickSample::kSkipCEntryFrame, false,
159 false);
160 ticks_from_vm_buffer_.Enqueue(record);
161}

163void ProfilerEventsProcessor::AddCurrentStack(bool update_stats) {
164 TickSampleEventRecord record(last_code_event_id_);
165 RegisterState regs;
166 StackFrameIterator it(isolate_, isolate_->thread_local_top(),
167 StackFrameIterator::NoHandles{});
168 if (!it.done()) {
169 StackFrame* frame = it.frame();
170 regs.sp = reinterpret_cast<void*>(frame->sp());
171 regs.fp = reinterpret_cast<void*>(frame->fp());
172 regs.pc = reinterpret_cast<void*>(frame->pc());
173 }
174 record.sample.Init(isolate_, regs, TickSample::kSkipCEntryFrame, update_stats,
175 false);
176 ticks_from_vm_buffer_.Enqueue(record);
177}

185void ProfilerEventsProcessor::StopSynchronously() {
186 bool expected = true;
187 if (!running_.compare_exchange_strong(expected, false,
188 std::memory_order_relaxed))
189 return;
190 {
191 base::MutexGuard guard(&running_mutex_);
192 running_cond_.NotifyOne();
193 }
194 Join();
195}

198bool ProfilerEventsProcessor::ProcessCodeEvent() {
199 CodeEventsContainer record;
200 if (events_buffer_.Dequeue(&record)) {
201 if (record.generic.type == CodeEventRecord::Type::kNativeContextMove) {
202 NativeContextMoveEventRecord& nc_record =
203 record.NativeContextMoveEventRecord_;
204 profiles_->UpdateNativeContextAddressForCurrentProfiles(
205 nc_record.from_address, nc_record.to_address);
206 } else {
207 code_observer_->CodeEventHandlerInternal(record);
208 }
209 last_processed_code_event_id_ = record.generic.order;
210 return true;
211 }
212 return false;
213}
214
215void ProfilerEventsProcessor::CodeEventHandler(
216 const CodeEventsContainer& evt_rec) {
217 switch (evt_rec.generic.type) {
218 case CodeEventRecord::Type::kCodeCreation:
219 case CodeEventRecord::Type::kCodeMove:
220 case CodeEventRecord::Type::kCodeDisableOpt:
221 case CodeEventRecord::Type::kCodeDelete:
222 case CodeEventRecord::Type::kNativeContextMove:
223 Enqueue(evt_rec);
224 break;
225 case CodeEventRecord::Type::kCodeDeopt: {
226 const CodeDeoptEventRecord* rec = &evt_rec.CodeDeoptEventRecord_;
227 Address pc = rec->pc;
228 int fp_to_sp_delta = rec->fp_to_sp_delta;
229 Enqueue(evt_rec);
230 AddDeoptStack(pc, fp_to_sp_delta);
231 break;
232 }
233 case CodeEventRecord::Type::kNoEvent:
234 case CodeEventRecord::Type::kReportBuiltin:
235 UNREACHABLE();
236 }
237}

252ProfilerEventsProcessor::SampleProcessingResult
253SamplingEventsProcessor::ProcessOneSample() {
254 TickSampleEventRecord record1;
255 if (ticks_from_vm_buffer_.Peek(&record1) &&
256 (record1.order == last_processed_code_event_id_)) {
257 TickSampleEventRecord record;
258 ticks_from_vm_buffer_.Dequeue(&record);
259 SymbolizeAndAddToProfiles(&record);
260 return OneSampleProcessed;
261 }
262
263 const TickSampleEventRecord* record = ticks_buffer_.Peek();
264 if (record == nullptr) {
265 if (ticks_from_vm_buffer_.IsEmpty()) return NoSamplesInQueue;
266 return FoundSampleForNextCodeEvent;
267 }
268 if (record->order != last_processed_code_event_id_) {
269 return FoundSampleForNextCodeEvent;
270 }
271 SymbolizeAndAddToProfiles(record);
272 ticks_buffer_.Remove();
273 return OneSampleProcessed;
274}

276void SamplingEventsProcessor::Run() {
277 base::MutexGuard guard(&running_mutex_);
278 while (running_.load(std::memory_order_relaxed)) {
279 base::TimeTicks nextSampleTime = base::TimeTicks::Now() + period_;
280 base::TimeTicks now;
281 SampleProcessingResult result;
282 // Keep processing existing events until we need to do next sample
283 // or the ticks buffer is empty.
284 do {
285 result = ProcessOneSample();
286 if (result == FoundSampleForNextCodeEvent) {
287 // All ticks of the current last_processed_code_event_id_ are
288 // processed, proceed to the next code event.
289 ProcessCodeEvent();
290 }
291 now = base::TimeTicks::Now();
292 } while (result != NoSamplesInQueue && now < nextSampleTime);
293
294 if (nextSampleTime > now) {
295#if V8_OS_WIN
296 if (use_precise_sampling_ &&
297 nextSampleTime - now < base::TimeDelta::FromMilliseconds(100)) {
298 if (precise_sleep_timer_.IsInitialized()) {
299 precise_sleep_timer_.Sleep(nextSampleTime - now);
300 } else {
301 // Do not use Sleep on Windows as it is very imprecise, with up to
302 // 16ms jitter, which is unacceptable for short profile intervals.
303 while (base::TimeTicks::Now() < nextSampleTime) {
304 }
305 }
306 } else // NOLINT
307#else
308 USE(use_precise_sampling_);
309#endif // V8_OS_WIN
310 {
311 // Allow another thread to interrupt the delay between samples in the
312 // event of profiler shutdown.
313 while (now < nextSampleTime &&
314 running_cond_.WaitFor(&running_mutex_, nextSampleTime - now)) {
315 // If true was returned, we got interrupted before the timeout
316 // elapsed. If this was not due to a change in running state, a
317 // spurious wakeup occurred (thus we should continue to wait).
318 if (!running_.load(std::memory_order_relaxed)) {
319 break;
320 }
321 now = base::TimeTicks::Now();
322 }
323 }
324 }
325
326 // Schedule next sample.
327 sampler_->DoSample();
328 }
329
330 // Process remaining tick events.
331 do {
332 SampleProcessingResult result;
333 do {
334 result = ProcessOneSample();
335 } while (result == OneSampleProcessed);
336 } while (ProcessCodeEvent());
337}

339void SamplingEventsProcessor::SetSamplingInterval(base::TimeDelta period) {
340 if (period_ == period) return;
341 StopSynchronously();
342
343 period_ = period;
344 running_.store(true, std::memory_order_relaxed);
345
346 CHECK(StartSynchronously());
347}

390void ProfilerCodeObserver::CodeEventHandlerInternal(
391 const CodeEventsContainer& evt_rec) {
392 CodeEventsContainer record = evt_rec;
393 switch (evt_rec.generic.type) {
394#define PROFILER_TYPE_CASE(type, clss) \
395 case CodeEventRecord::Type::type: \
396 record.clss##_.UpdateCodeMap(&code_map_); \
397 break;
398
399 CODE_EVENTS_TYPE_LIST(PROFILER_TYPE_CASE)
400
401#undef PROFILER_TYPE_CASE
402 default:
403 break;
404 }
405}

407void ProfilerCodeObserver::CreateEntriesForRuntimeCallStats() {
408#ifdef V8_RUNTIME_CALL_STATS
409 RuntimeCallStats* rcs = isolate_->counters()->runtime_call_stats();
410 for (int i = 0; i < RuntimeCallStats::kNumberOfCounters; ++i) {
411 RuntimeCallCounter* counter = rcs->GetCounter(i);
412 DCHECK(counter->name());
413 auto entry = code_entries_.Create(LogEventListener::CodeTag::kFunction,
414 counter->name(), "native V8Runtime");
415 code_map_.AddCode(reinterpret_cast<Address>(counter), entry, 1);
416 }
417#endif // V8_RUNTIME_CALL_STATS
418}

420void ProfilerCodeObserver::LogBuiltins() {
421 Builtins* builtins = isolate_->builtins();
422 DCHECK(builtins->is_initialized());
423 for (Builtin builtin = Builtins::kFirst; builtin <= Builtins::kLast;
424 ++builtin) {
425 CodeEventsContainer evt_rec(CodeEventRecord::Type::kReportBuiltin);
426 ReportBuiltinEventRecord* rec = &evt_rec.ReportBuiltinEventRecord_;
427 Tagged<Code> code = builtins->code(builtin);
428 rec->instruction_start = code->instruction_start();
429 rec->instruction_size = code->instruction_size();
430 rec->builtin = builtin;
431 CodeEventHandlerInternal(evt_rec);
432 }

462class CpuProfilersManager {
463 public:
464 void AddProfiler(Isolate* isolate, CpuProfiler* profiler) {
465 base::MutexGuard lock(&mutex_);
466 profilers_.emplace(isolate, profiler);
467 }
468
469 void RemoveProfiler(Isolate* isolate, CpuProfiler* profiler) {
470 base::MutexGuard lock(&mutex_);
471 auto range = profilers_.equal_range(isolate);
472 for (auto it = range.first; it != range.second; ++it) {
473 if (it->second != profiler) continue;
474 profilers_.erase(it);
475 return;
476 }
477 UNREACHABLE();
478 }
479
480 void CallCollectSample(Isolate* isolate) {
481 base::MutexGuard lock(&mutex_);
482 auto range = profilers_.equal_range(isolate);
483 for (auto it = range.first; it != range.second; ++it) {
484 it->second->CollectSample();
485 }
486 }
487
488 size_t GetAllProfilersMemorySize(Isolate* isolate) {
489 base::MutexGuard lock(&mutex_);
490 size_t estimated_memory = 0;
491 auto range = profilers_.equal_range(isolate);
492 for (auto it = range.first; it != range.second; ++it) {
493 estimated_memory += it->second->GetEstimatedMemoryUsage();
494 }
495 return estimated_memory;
496 }
497
498 private:
499 std::unordered_multimap<Isolate*, CpuProfiler*> profilers_;
500 base::Mutex mutex_;
501};

645void CpuProfiler::StartProcessorIfNotStarted() {
646 if (processor_) {
647 processor_->AddCurrentStack();
648 return;
649 }
650
651 if (!profiling_scope_) {
652 DCHECK_EQ(logging_mode_, kLazyLogging);
653 EnableLogging();
654 }
655
656 if (!symbolizer_) {
657 symbolizer_ =
658 std::make_unique<Symbolizer>(code_observer_->instruction_stream_map());
659 }
660
661 base::TimeDelta sampling_interval = ComputeSamplingInterval();
662 processor_.reset(new SamplingEventsProcessor(
663 isolate_, symbolizer_.get(), code_observer_.get(), profiles_.get(),
664 sampling_interval, use_precise_sampling_));
665 is_profiling_ = true;
666
667 // Enable stack sampling.
668 processor_->AddCurrentStack();
669 CHECK(processor_->StartSynchronously());
670}

680CpuProfile* CpuProfiler::StopProfiling(ProfilerId id) {
681 if (!is_profiling_) return nullptr;
682 const bool last_profile = profiles_->IsLastProfileLeft(id);
683 if (last_profile) StopProcessor();
684
685 CpuProfile* profile = profiles_->StopProfiling(id);
686
687 AdjustSamplingInterval();
688
689 DCHECK(profiling_scope_);
690 if (last_profile && logging_mode_ == kLazyLogging) {
691 DisableLogging();
692 }
693
694 return profile;
