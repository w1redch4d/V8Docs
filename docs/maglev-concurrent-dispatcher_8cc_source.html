133CompilationJob::Status MaglevCompilationJob::ExecuteJobImpl(
134 RuntimeCallStats* stats, LocalIsolate* local_isolate) {
135 BeginPhaseKind("V8.MaglevExecuteJob");
136 LocalIsolateScope scope{info(), local_isolate};
137 if (!maglev::MaglevCompiler::Compile(local_isolate, info())) {
138 EndPhaseKind();
139 return CompilationJob::FAILED;
140 }
141 EndPhaseKind();
142 // TODO(v8:7700): Actual return codes.
143 return CompilationJob::SUCCEEDED;

146CompilationJob::Status MaglevCompilationJob::FinalizeJobImpl(Isolate* isolate) {
147 BeginPhaseKind("V8.MaglevFinalizeJob");
148 Handle<Code> code;
149 if (!maglev::MaglevCompiler::GenerateCode(isolate, info()).ToHandle(&code)) {
150 EndPhaseKind();
151 return CompilationJob::FAILED;
152 }
153 // Functions with many inline candidates are sensitive to correct call
154 // frequency feedback and should therefore not be tiered up early.
155 if (v8_flags.profile_guided_optimization &&
156 info()->could_not_inline_all_candidates() &&
157 info()->toplevel_function()->shared()->cached_tiering_decision() !=
158 CachedTieringDecision::kDelayMaglev) {
159 info()->toplevel_function()->shared()->set_cached_tiering_decision(
160 CachedTieringDecision::kNormal);
161 }
162 info()->set_code(code);
163 GlobalHandleVector<Map> maps = CollectRetainedMaps(isolate, code);
164 RegisterWeakObjectsInOptimizedCode(
165 isolate, info()->broker()->target_native_context().object(), code,
166 std::move(maps));
167 EndPhaseKind();
168 return CompilationJob::SUCCEEDED;
169}

205void MaglevCompilationJob::RecordCompilationStats(Isolate* isolate) const {
206 // Don't record samples from machines without high-resolution timers,
207 // as that can cause serious reporting issues. See the thread at
208 // http://g/chrome-metrics-team/NwwJEyL8odU/discussion for more details.
209 if (base::TimeTicks::IsHighResolution()) {
210 Counters* const counters = isolate->counters();
211 counters->maglev_optimize_prepare()->AddSample(
212 static_cast<int>(time_taken_to_prepare_.InMicroseconds()));
213 counters->maglev_optimize_execute()->AddSample(
214 static_cast<int>(time_taken_to_execute_.InMicroseconds()));
215 counters->maglev_optimize_finalize()->AddSample(
216 static_cast<int>(time_taken_to_finalize_.InMicroseconds()));
217 counters->maglev_optimize_total_time()->AddSample(
218 static_cast<int>(ElapsedTime().InMicroseconds()));
219 }
220 if (v8_flags.trace_opt_stats) {
221 static double compilation_time = 0.0;
222 static int compiled_functions = 0;
223 static int code_size = 0;
224
225 compilation_time += (time_taken_to_prepare_.InMillisecondsF() +
226 time_taken_to_execute_.InMillisecondsF() +
227 time_taken_to_finalize_.InMillisecondsF());
228 compiled_functions++;
229 code_size += function()->shared()->SourceSize();
230 PrintF(
231 "[maglev] Compiled: %d functions with %d byte source size in %fms.\n",
232 compiled_functions, code_size, compilation_time);
233 }

236uint64_t MaglevCompilationJob::trace_id() const {
237 // Xor together the this pointer, the info pointer, and the top level
238 // function's Handle address, to try to make the id more unique on platforms
239 // where just the `this` pointer is likely to be reused.
240 return reinterpret_cast<uint64_t>(this) ^
241 reinterpret_cast<uint64_t>(info_.get()) ^
242 info_->toplevel_function().address() ^
243 info_->toplevel_function()->shared()->function_literal_id();

258// The JobTask is posted to V8::GetCurrentPlatform(). It's responsible for
259// processing the incoming queue on a worker thread.
260class MaglevConcurrentDispatcher::JobTask final : public v8::JobTask {
261 public:
262 explicit JobTask(MaglevConcurrentDispatcher* dispatcher)
263 : dispatcher_(dispatcher) {}
264
265 void Run(JobDelegate* delegate) override {
266 if (incoming_queue()->IsEmpty() && destruction_queue()->IsEmpty()) {
267 return;
268 }
269 TRACE_EVENT0(TRACE_DISABLED_BY_DEFAULT("v8.compile"), "V8.MaglevTask");
270 LocalIsolate local_isolate(isolate(), ThreadKind::kBackground);
271 DCHECK(local_isolate.heap()->IsParked());
272
273 std::unique_ptr<MaglevCompilationJob> job_to_destruct;
274 while (!delegate->ShouldYield()) {
275 std::unique_ptr<MaglevCompilationJob> job;
276 if (incoming_queue()->Dequeue(&job)) {
277 DCHECK_NOT_NULL(job);
278 TRACE_EVENT_WITH_FLOW0(
279 TRACE_DISABLED_BY_DEFAULT("v8.compile"), "V8.MaglevBackground",
280 job->trace_id(),
281 TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
282 RCS_SCOPE(&local_isolate,
283 RuntimeCallCounterId::kOptimizeBackgroundMaglev);
284 CompilationJob::Status status =
285 job->ExecuteJob(local_isolate.runtime_call_stats(), &local_isolate);
286 if (status == CompilationJob::SUCCEEDED) {
287 outgoing_queue()->Enqueue(std::move(job));
288 isolate()->stack_guard()->RequestInstallMaglevCode();
289 }
290 } else if (destruction_queue()->Dequeue(&job)) {
291 // Maglev jobs aren't cheap to destruct, so destroy them here in the
292 // background thread rather than on the main thread.
293 DCHECK_NOT_NULL(job);
294 TRACE_EVENT_WITH_FLOW0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
295 "V8.MaglevDestructBackground", job->trace_id(),
296 TRACE_EVENT_FLAG_FLOW_IN);
297 UnparkedScope unparked_scope(&local_isolate);
298 job.reset();
299 } else {
300 break;
301 }
302 }
303 }
304
305 size_t GetMaxConcurrency(size_t worker_count) const override {
306 size_t num_tasks =
307 incoming_queue()->size() + destruction_queue()->size() + worker_count;
308 size_t max_threads = v8_flags.concurrent_maglev_max_threads;
309 if (max_threads > 0) {
310 return std::min(max_threads, num_tasks);
311 }
312 return num_tasks;
313 }
314
315 private:
316 Isolate* isolate() const { return dispatcher_->isolate_; }
317 QueueT* incoming_queue() const { return &dispatcher_->incoming_queue_; }
318 QueueT* outgoing_queue() const { return &dispatcher_->outgoing_queue_; }
319 QueueT* destruction_queue() const { return &dispatcher_->destruction_queue_; }
320
321 MaglevConcurrentDispatcher* const dispatcher_;
322};

324MaglevConcurrentDispatcher::MaglevConcurrentDispatcher(Isolate* isolate)
325 : isolate_(isolate) {
326 bool enable = v8_flags.concurrent_recompilation && maglev::IsMaglevEnabled();
327 if (enable) {
328 bool is_tracing =
329 v8_flags.print_maglev_code || v8_flags.trace_maglev_graph_building ||
330 v8_flags.trace_maglev_inlining || v8_flags.print_maglev_deopt_verbose ||
331 v8_flags.print_maglev_graph || v8_flags.print_maglev_graphs ||
332 v8_flags.trace_maglev_phi_untagging || v8_flags.trace_maglev_regalloc;
333
334 if (is_tracing) {
335 PrintF("Concurrent maglev has been disabled for tracing.\n");
336 enable = false;
337 }
338 }
339 if (enable) {
340 TaskPriority priority = v8_flags.concurrent_maglev_high_priority_threads
341 ? TaskPriority::kUserBlocking
342 : TaskPriority::kUserVisible;
343 job_handle_ = V8::GetCurrentPlatform()->PostJob(
344 priority, std::make_unique<JobTask>(this));
345 DCHECK(is_enabled());
346 } else {
347 DCHECK(!is_enabled());
348 }
349}

366void MaglevConcurrentDispatcher::FinalizeFinishedJobs() {
367 HandleScope handle_scope(isolate_);
368 while (!outgoing_queue_.IsEmpty()) {
369 std::unique_ptr<MaglevCompilationJob> job;
370 outgoing_queue_.Dequeue(&job);
371 TRACE_EVENT_WITH_FLOW0(
372 TRACE_DISABLED_BY_DEFAULT("v8.compile"), "V8.MaglevConcurrentFinalize",
373 job->trace_id(), TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
374 RCS_SCOPE(isolate_,
375 RuntimeCallCounterId::kOptimizeConcurrentFinalizeMaglev);
376 Compiler::FinalizeMaglevCompilationJob(job.get(), isolate_);
377 job->DisposeOnMainThread(isolate_);
378 if (v8_flags.maglev_destroy_on_background) {
379 // Maglev jobs aren't cheap to destruct, so re-enqueue them for
380 // destruction on a background thread.
381 destruction_queue_.Enqueue(std::move(job));
382 job_handle_->NotifyConcurrencyIncrease();
383 } else {
384 TRACE_EVENT_WITH_FLOW0(TRACE_DISABLED_BY_DEFAULT("v8.compile"),
385 "V8.MaglevDestruct", job->trace_id(),
386 TRACE_EVENT_FLAG_FLOW_IN);
387 job.reset();
388 }
389 }

392void MaglevConcurrentDispatcher::AwaitCompileJobs() {
393 // Use Join to wait until there are no more queued or running jobs.
394 {
395 AllowGarbageCollection allow_before_parking;
396 isolate_->main_thread_local_isolate()->ExecuteMainThreadWhileParked(
397 [this]() { job_handle_->Join(); });
398 }
399 // Join kills the job handle, so drop it and post a new one.
400 TaskPriority priority = v8_flags.concurrent_maglev_high_priority_threads
401 ? TaskPriority::kUserBlocking
402 : TaskPriority::kUserVisible;
403 job_handle_ = V8::GetCurrentPlatform()->PostJob(
404 priority, std::make_unique<JobTask>(this));
405 DCHECK(incoming_queue_.IsEmpty());
406}

408void MaglevConcurrentDispatcher::Flush(BlockingBehavior behavior) {
409 while (!incoming_queue_.IsEmpty()) {
410 std::unique_ptr<MaglevCompilationJob> job;
411 if (incoming_queue_.Dequeue(&job)) {
412 Compiler::DisposeMaglevCompilationJob(job.get(), isolate_);
413 }
414 }
415 while (!destruction_queue_.IsEmpty()) {
416 std::unique_ptr<MaglevCompilationJob> job;
417 destruction_queue_.Dequeue(&job);
418 }
419 if (behavior == BlockingBehavior::kBlock && job_handle_->IsValid()) {
420 AwaitCompileJobs();
421 }
422 while (!outgoing_queue_.IsEmpty()) {
423 std::unique_ptr<MaglevCompilationJob> job;
424 if (outgoing_queue_.Dequeue(&job)) {
425 Compiler::DisposeMaglevCompilationJob(job.get(), isolate_);
426 }
427 }
