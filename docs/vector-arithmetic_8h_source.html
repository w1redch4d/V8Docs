22// X += y.
23inline void Add(RWDigits X, digit_t y) {
24 digit_t carry = y;
25 int i = 0;
26 do {
27 X[i] = digit_add2(X[i], carry, &carry);
28 i++;
29 } while (carry != 0);
30}

32// X -= y.
33inline void Subtract(RWDigits X, digit_t y) {
34 digit_t borrow = y;
35 int i = 0;
36 do {
37 X[i] = digit_sub(X[i], borrow, &borrow);
38 i++;
39 } while (borrow != 0);
40}
