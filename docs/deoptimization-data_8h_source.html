57// A deoptimization literal during code generation. For JS this is transformed
58// into a heap object after code generation. For wasm the DeoptimizationLiteral
59// is directly used by the deoptimizer.
60class DeoptimizationLiteral {
61 public:
62 DeoptimizationLiteral()
63 : kind_(DeoptimizationLiteralKind::kInvalid), object_() {}
64 explicit DeoptimizationLiteral(IndirectHandle<Object> object)
65 : kind_(DeoptimizationLiteralKind::kObject), object_(object) {
66 CHECK(!object_.is_null());
67 }
68 explicit DeoptimizationLiteral(Float32 number)
69 : kind_(DeoptimizationLiteralKind::kWasmFloat32), float32_(number) {}
70 explicit DeoptimizationLiteral(Float64 number)
71 : kind_(DeoptimizationLiteralKind::kWasmFloat64), float64_(number) {}
72 explicit DeoptimizationLiteral(double number)
73 : kind_(DeoptimizationLiteralKind::kNumber), number_(number) {}
74 explicit DeoptimizationLiteral(int64_t signed_bigint64)
75 : kind_(DeoptimizationLiteralKind::kSignedBigInt64),
76 int64_(signed_bigint64) {}
77 explicit DeoptimizationLiteral(uint64_t unsigned_bigint64)
78 : kind_(DeoptimizationLiteralKind::kUnsignedBigInt64),
79 uint64_(unsigned_bigint64) {}
80 explicit DeoptimizationLiteral(int32_t int32)
81 : kind_(DeoptimizationLiteralKind::kWasmInt32), int64_(int32) {}
82 explicit DeoptimizationLiteral(Tagged<Smi> smi)
83 : kind_(DeoptimizationLiteralKind::kWasmI31Ref), int64_(smi.value()) {}
84
85 static DeoptimizationLiteral HoleNaN() {
86 DeoptimizationLiteral literal;
87 literal.kind_ = DeoptimizationLiteralKind::kHoleNaN;
88 return literal;
89 }
90
91 IndirectHandle<Object> object() const { return object_; }
92
93 bool operator==(const DeoptimizationLiteral& other) const {
94 if (kind_ != other.kind_) {
95 return false;
96 }
97 switch (kind_) {
98 case DeoptimizationLiteralKind::kObject:
99 return object_.equals(other.object_);
100 case DeoptimizationLiteralKind::kNumber:
101 return base::bit_cast<uint64_t>(number_) ==
102 base::bit_cast<uint64_t>(other.number_);
103 case DeoptimizationLiteralKind::kWasmI31Ref:
104 case DeoptimizationLiteralKind::kWasmInt32:
105 case DeoptimizationLiteralKind::kSignedBigInt64:
106 return int64_ == other.int64_;
107 case DeoptimizationLiteralKind::kUnsignedBigInt64:
108 return uint64_ == other.uint64_;
109 case DeoptimizationLiteralKind::kHoleNaN:
110 return other.kind() == DeoptimizationLiteralKind::kHoleNaN;
111 case DeoptimizationLiteralKind::kInvalid:
112 return true;
113 case DeoptimizationLiteralKind::kWasmFloat32:
114 return float32_.get_bits() == other.float32_.get_bits();
115 case DeoptimizationLiteralKind::kWasmFloat64:
116 return float64_.get_bits() == other.float64_.get_bits();
117 }
118 UNREACHABLE();
119 }
120
121 Handle<Object> Reify(Isolate* isolate) const;
122
123#if V8_ENABLE_WEBASSEMBLY
124 Float64 GetFloat64() const {
125 DCHECK_EQ(kind_, DeoptimizationLiteralKind::kWasmFloat64);
126 return float64_;
127 }
128
129 Float32 GetFloat32() const {
130 DCHECK_EQ(kind_, DeoptimizationLiteralKind::kWasmFloat32);
131 return float32_;
132 }
133
134 int64_t GetInt64() const {
135 DCHECK_EQ(kind_, DeoptimizationLiteralKind::kWasmInt64);
136 return int64_;
137 }
138
139 int32_t GetInt32() const {
140 DCHECK_EQ(kind_, DeoptimizationLiteralKind::kWasmInt32);
141 return static_cast<int32_t>(int64_);
142 }
143
144 Tagged<Smi> GetSmi() const {
145 DCHECK_EQ(kind_, DeoptimizationLiteralKind::kWasmI31Ref);
146 return Smi::FromInt(static_cast<int>(int64_));
147 }
148#endif
149
150 void Validate() const {
151 CHECK_NE(kind_, DeoptimizationLiteralKind::kInvalid);
152 }
153
154 DeoptimizationLiteralKind kind() const {
155 Validate();
156 return kind_;
157 }

159 private:
160 DeoptimizationLiteralKind kind_;

165 Float32 float32_;
166 Float64 float64_;
167 int64_t int64_;
168 uint64_t uint64_;
169 };
