24constexpr const char* WasmOpcodes::OpcodeName(WasmOpcode opcode) {
25 switch (opcode) {
26#define CASE(opcode, binary, sig, name, ...) \
27 case kExpr##opcode: \
28 return name;
29 FOREACH_OPCODE(CASE)
30#undef CASE
31
32 case kNumericPrefix:
33 case kSimdPrefix:
34 case kAtomicPrefix:
35 case kGCPrefix:
36 return "unknown";
37 }
38 // Even though the switch above handles all well-defined enum values,
39 // random modules (e.g. fuzzer generated) can call this function with
40 // random (invalid) opcodes. Handle those here:
41 return "invalid opcode";
42}

127constexpr bool WasmOpcodes::IsRelaxedSimdOpcode(WasmOpcode opcode) {
128 // Relaxed SIMD opcodes have the SIMD prefix (0xfd) shifted by 12 bits, and
129 // nibble 3 must be 0x1. I.e. their encoded opcode is in [0xfd100, 0xfd1ff].
130 static_assert(kSimdPrefix == 0xfd);
131#define CHECK_OPCODE(name, opcode, ...) \
132 static_assert((opcode & 0xfff00) == 0xfd100);
133 FOREACH_RELAXED_SIMD_OPCODE(CHECK_OPCODE)
134#undef CHECK_OPCODE
135
136 return (opcode & 0xfff00) == 0xfd100;
137}

161constexpr uint8_t WasmOpcodes::ExtractPrefix(WasmOpcode opcode) {
162 // See comment on {WasmOpcode} for the encoding.
163 return (opcode > 0xffff) ? opcode >> 12 : opcode >> 8;
164}

253constexpr const FunctionSig* WasmOpcodes::Signature(WasmOpcode opcode) {
254 switch (ExtractPrefix(opcode)) {
255 case 0:
256 DCHECK_GT(impl::kShortSigTable.size(), opcode);
257 return impl::kCachedSigs[impl::kShortSigTable[opcode]];
258 case kSimdPrefix: {
259 // Handle SIMD MVP opcodes (in [0xfd00, 0xfdff]).
260 if (opcode <= 0xfdff) {
261 DCHECK_LE(0xfd00, opcode);
262 return impl::kCachedSigs[impl::kSimdExprSigTable[opcode & 0xff]];
263 }
264 // Handle relaxed SIMD opcodes (in [0xfd100, 0xfd1ff]).
265 if (IsRelaxedSimdOpcode(opcode)) {
266 return impl::kCachedSigs[impl::kRelaxedSimdExprSigTable[opcode & 0xff]];
267 }
268 return nullptr;
269 }
270 case kNumericPrefix:
271 return impl::kCachedSigs[impl::kNumericExprSigTable[opcode & 0xff]];
272 default:
273 UNREACHABLE(); // invalid prefix.
274 }
275}
