155 // 'always' is used to generate unconditional branches.
156 // 'never' is used to not generate a branch (generally as the inverse
157 // branch type of 'always).
158 always,

174inline BranchType InvertBranchType(BranchType type) {
175 if (kBranchTypeFirstCondition <= type && type <= kBranchTypeLastCondition) {
176 return static_cast<BranchType>(
177 NegateCondition(static_cast<Condition>(type)));
178 } else {
179 return static_cast<BranchType>(type ^ 1);
180 }
181}

186// The macro assembler supports moving automatically pre-shifted immediates for
187// arithmetic and logical instructions, and then applying a post shift in the
188// instruction to undo the modification, in order to reduce the code emitted for
189// an operation. For example:
190//
191// Add(x0, x0, 0x1f7de) => movz x16, 0xfbef; add x0, x0, x16, lsl #1.
192//
193// This optimisation can be only partially applied when the stack pointer is an
194// operand or destination, so this enumeration is used to control the shift.
195enum PreShiftImmMode {
196 kNoShift, // Don't pre-shift.
197 kLimitShiftForSP, // Limit pre-shift for add/sub extend use.
198 kAnyShift // Allow any pre-shift.
199};

579 // The 1716 pac and aut instructions encourage people to use x16 and x17
580 // directly, perhaps without realising that this is forbidden. For example:
581 //
582 // UseScratchRegisterScope temps(&masm);
583 // Register temp = temps.AcquireX(); // temp will be x16
584 // __ Mov(x17, ptr);
585 // __ Mov(x16, modifier); // Will override temp!
586 // __ Pacib1716();
587 //
588 // To work around this issue, you must exclude x16 and x17 from the scratch
589 // register list. You may need to replace them with other registers:
590 //
591 // UseScratchRegisterScope temps(&masm);
592 // temps.Exclude(x16, x17);
593 // temps.Include(x10, x11);
594 // __ Mov(x17, ptr);
595 // __ Mov(x16, modifier);
596 // __ Pacib1716();

2390 // Helps resolve branching to labels potentially out of range.
2391 // If the label is not bound, it registers the information necessary to later
2392 // be able to emit a veneer for this branch if necessary.
2393 // If the label is bound, it returns true if the label (or the previous link
2394 // in the label chain) is out of range. In that case the caller is responsible
2395 // for generating appropriate code.
2396 // Otherwise it returns false.
2397 // This function also checks wether veneers need to be emitted.
2398 template <ImmBranchType branch_type>
2399 bool NeedExtraInstructionsOrRegisterBranch(Label* label) {
2400 static_assert((branch_type == CondBranchType) ||
2401 (branch_type == CompareBranchType) ||
2402 (branch_type == TestBranchType));

2405 // There are two situations in which we care about the offset being out of
2406 // range:
2407 // - The label is bound but too far away.
2408 // - The label is not bound but linked, and the previous branch
2409 // instruction in the chain is too far away.
2410 if (label->is_bound() || label->is_linked()) {
2411 need_longer_range = !Instruction::IsValidImmPCOffset(
2412 branch_type, label->pos() - pc_offset());
2413 }
2414 if (!need_longer_range && !label->is_bound()) {
2415 int max_reachable_pc =
2416 pc_offset() + Instruction::ImmBranchRange(branch_type);
2417
2418 // Use the LSB of the max_reachable_pc (always four-byte aligned) to
2419 // encode the branch type. We need only distinguish between TB[N]Z and
2420 // CB[N]Z/conditional branch, as the ranges for the latter are the same.
2421 int branch_type_tag = (branch_type == TestBranchType) ? 1 : 0;
2422
2423 unresolved_branches_.insert(
2424 std::pair<int, Label*>(max_reachable_pc + branch_type_tag, label));
2425 // Also maintain the next pool check.
2426 next_veneer_pool_check_ =
2427 std::min(next_veneer_pool_check_,
2428 max_reachable_pc - kVeneerDistanceCheckMargin);
2429 }
