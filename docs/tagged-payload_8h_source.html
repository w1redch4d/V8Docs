14// Struct providing common utilities for pointer tagging.
15template <typename PayloadTaggingScheme>
16struct TaggedPayload {
17 static_assert(PayloadTaggingScheme::kMarkBit != 0,
18 "Invalid kMarkBit specified in tagging scheme.");
19
20 TaggedPayload(Address pointer, typename PayloadTaggingScheme::TagType tag)
21 : encoded_word_(Tag(pointer, tag)) {}
22
23 Address Untag(typename PayloadTaggingScheme::TagType tag) const {
24 return encoded_word_ & ~(tag | PayloadTaggingScheme::kMarkBit);
25 }
26
27 static Address Tag(Address pointer,
28 typename PayloadTaggingScheme::TagType tag) {
29 return pointer | tag;
30 }
31
32 bool IsTaggedWith(typename PayloadTaggingScheme::TagType tag) const {
33 // We have to explicitly ignore the marking bit (which is part of the
34 // tag) since an unmarked entry with tag kXyzTag is still considered to
35 // be tagged with kXyzTag.
36 uint64_t expected = tag & ~PayloadTaggingScheme::kMarkBit;
37 uint64_t actual = (encoded_word_ & PayloadTaggingScheme::kTagMask) &
38 ~PayloadTaggingScheme::kMarkBit;
39 return expected == actual;
40 }
41
42 void SetMarkBit() { encoded_word_ |= PayloadTaggingScheme::kMarkBit; }
43
44 void ClearMarkBit() { encoded_word_ &= ~PayloadTaggingScheme::kMarkBit; }
45
46 bool HasMarkBitSet() const {
47 return (encoded_word_ & PayloadTaggingScheme::kMarkBit) != 0;
48 }
49
50 uint32_t ExtractFreelistLink() const {
51 return static_cast<uint32_t>(encoded_word_);
52 }
53
54 typename PayloadTaggingScheme::TagType ExtractTag() const {
55 return static_cast<typename PayloadTaggingScheme::TagType>(
56 (encoded_word_ & PayloadTaggingScheme::kTagMask) |
57 PayloadTaggingScheme::kMarkBit);
58 }
59
60 bool ContainsFreelistLink() const {
61 return IsTaggedWith(PayloadTaggingScheme::kFreeEntryTag);
62 }
63
64 bool ContainsEvacuationEntry() const {
65 if constexpr (PayloadTaggingScheme::kSupportsEvacuation) {
66 return IsTaggedWith(PayloadTaggingScheme::kEvacuationEntryTag);
67 } else {
68 return false;
69 }
70 }
71
72 bool IsZapped() const {
73 if constexpr (PayloadTaggingScheme::kSupportsZapping) {
74 return IsTaggedWith(PayloadTaggingScheme::kZappedEntryTag);
75 } else {
76 return false;
77 }
78 }
79
80 Address ExtractEvacuationEntryHandleLocation() const {
81 if constexpr (PayloadTaggingScheme::kSupportsEvacuation) {
82 return Untag(PayloadTaggingScheme::kEvacuationEntryTag);
83 } else {
84 UNREACHABLE();
85 }
86 }
87
88 bool ContainsPointer() const {
89 return !ContainsFreelistLink() && !ContainsEvacuationEntry() && !IsZapped();
90 }
91
92 bool operator==(TaggedPayload other) const {
93 return encoded_word_ == other.encoded_word_;
94 }
95
96 bool operator!=(TaggedPayload other) const {
97 return encoded_word_ != other.encoded_word_;
98 }
99
100 private:
101 Address encoded_word_;
102};
