56void WritableRelocInfo::apply(intptr_t delta) {
57 if (RelocInfo::IsInternalReference(rmode_)) {
58 // absolute code pointer inside code object moves with the code object.
59 int32_t* p = reinterpret_cast<int32_t*>(pc_);
60 jit_allocation_.WriteValue(pc_, *p + delta); // relocate entry
61 } else if (RelocInfo::IsRelativeCodeTarget(rmode_)) {
62 Instruction* branch = Instruction::At(pc_);
63 int32_t branch_offset = branch->GetBranchOffset() - delta;
64 branch->SetBranchOffset(branch_offset, &jit_allocation_);
65 }
66}

216Address Assembler::target_address_at(Address pc, Address constant_pool) {
217 if (is_constant_pool_load(pc)) {
218 // This is a constant pool lookup. Return the value in the constant pool.
219 return Memory<Address>(constant_pool_entry_address(pc, constant_pool));
220 } else if (CpuFeatures::IsSupported(ARMv7) && IsMovW(Memory<int32_t>(pc))) {
221 // This is an movw / movt immediate load. Return the immediate.
222 DCHECK(IsMovW(Memory<int32_t>(pc)) &&
223 IsMovT(Memory<int32_t>(pc + kInstrSize)));
224 Instruction* movw_instr = Instruction::At(pc);
225 Instruction* movt_instr = Instruction::At(pc + kInstrSize);
226 return static_cast<Address>((movt_instr->ImmedMovwMovtValue() << 16) |
227 movw_instr->ImmedMovwMovtValue());
228 } else if (IsMovImmed(Memory<int32_t>(pc))) {
229 // This is an mov / orr immediate load. Return the immediate.
230 DCHECK(IsMovImmed(Memory<int32_t>(pc)) &&
231 IsOrrImmed(Memory<int32_t>(pc + kInstrSize)) &&
232 IsOrrImmed(Memory<int32_t>(pc + 2 * kInstrSize)) &&
233 IsOrrImmed(Memory<int32_t>(pc + 3 * kInstrSize)));
234 Instr mov_instr = instr_at(pc);
235 Instr orr_instr_1 = instr_at(pc + kInstrSize);
236 Instr orr_instr_2 = instr_at(pc + 2 * kInstrSize);
237 Instr orr_instr_3 = instr_at(pc + 3 * kInstrSize);
238 Address ret = static_cast<Address>(
239 DecodeShiftImm(mov_instr) | DecodeShiftImm(orr_instr_1) |
240 DecodeShiftImm(orr_instr_2) | DecodeShiftImm(orr_instr_3));
241 return ret;
242 } else {
243 Instruction* branch = Instruction::At(pc);
244 int32_t delta = branch->GetBranchOffset();
245 return pc + delta + Instruction::kPcLoadDelta;
246 }

249void Assembler::set_target_address_at(Address pc, Address constant_pool,
250 Address target,
251 WritableJitAllocation* jit_allocation,
252 ICacheFlushMode icache_flush_mode) {
253 if (is_constant_pool_load(pc)) {
254 // This is a constant pool lookup. Update the entry in the constant pool.
255 if (jit_allocation) {
256 jit_allocation->WriteValue<Address>(
257 constant_pool_entry_address(pc, constant_pool), target);
258 } else {
259 Memory<Address>(constant_pool_entry_address(pc, constant_pool)) = target;
260 }
261 // Intuitively, we would think it is necessary to always flush the
262 // instruction cache after patching a target address in the code as follows:
263 // FlushInstructionCache(pc, sizeof(target));
264 // However, on ARM, no instruction is actually patched in the case
265 // of embedded constants of the form:
266 // ldr ip, [pp, #...]
267 // since the instruction accessing this address in the constant pool remains
268 // unchanged.
269 } else if (CpuFeatures::IsSupported(ARMv7) && IsMovW(Memory<int32_t>(pc))) {
270 // This is an movw / movt immediate load. Patch the immediate embedded in
271 // the instructions.
272 DCHECK(IsMovW(Memory<int32_t>(pc)));
273 DCHECK(IsMovT(Memory<int32_t>(pc + kInstrSize)));
274 uint32_t* instr_ptr = reinterpret_cast<uint32_t*>(pc);
275 uint32_t immediate = static_cast<uint32_t>(target);
276 if (jit_allocation) {
277 jit_allocation->WriteValue(
278 reinterpret_cast<Address>(&instr_ptr[0]),
279 PatchMovwImmediate(instr_ptr[0], immediate & 0xFFFF));
280 jit_allocation->WriteValue(
281 reinterpret_cast<Address>(&instr_ptr[1]),
282 PatchMovwImmediate(instr_ptr[1], immediate >> 16));
283 } else {
284 instr_ptr[0] = PatchMovwImmediate(instr_ptr[0], immediate & 0xFFFF);
285 instr_ptr[1] = PatchMovwImmediate(instr_ptr[1], immediate >> 16);
286 }
287 DCHECK(IsMovW(Memory<int32_t>(pc)));
288 DCHECK(IsMovT(Memory<int32_t>(pc + kInstrSize)));
289 if (icache_flush_mode != SKIP_ICACHE_FLUSH) {
290 FlushInstructionCache(pc, 2 * kInstrSize);
291 }
292 } else if (IsMovImmed(Memory<int32_t>(pc))) {
293 // This is an mov / orr immediate load. Patch the immediate embedded in
294 // the instructions.
295 DCHECK(IsMovImmed(Memory<int32_t>(pc)) &&
296 IsOrrImmed(Memory<int32_t>(pc + kInstrSize)) &&
297 IsOrrImmed(Memory<int32_t>(pc + 2 * kInstrSize)) &&
298 IsOrrImmed(Memory<int32_t>(pc + 3 * kInstrSize)));
299 uint32_t* instr_ptr = reinterpret_cast<uint32_t*>(pc);
300 uint32_t immediate = static_cast<uint32_t>(target);
301 if (jit_allocation) {
302 jit_allocation->WriteValue(
303 reinterpret_cast<Address>(&instr_ptr[0]),
304 PatchShiftImm(instr_ptr[0], immediate & kImm8Mask));
305 jit_allocation->WriteValue(
306 reinterpret_cast<Address>(&instr_ptr[1]),
307 PatchShiftImm(instr_ptr[1], immediate & (kImm8Mask << 8)));
308 jit_allocation->WriteValue(
309 reinterpret_cast<Address>(&instr_ptr[2]),
310 PatchShiftImm(instr_ptr[2], immediate & (kImm8Mask << 16)));
311 jit_allocation->WriteValue(
312 reinterpret_cast<Address>(&instr_ptr[3]),
313 PatchShiftImm(instr_ptr[3], immediate & (kImm8Mask << 24)));
314 } else {
315 instr_ptr[0] = PatchShiftImm(instr_ptr[0], immediate & kImm8Mask);
316 instr_ptr[1] = PatchShiftImm(instr_ptr[1], immediate & (kImm8Mask << 8));
317 instr_ptr[2] = PatchShiftImm(instr_ptr[2], immediate & (kImm8Mask << 16));
318 instr_ptr[3] = PatchShiftImm(instr_ptr[3], immediate & (kImm8Mask << 24));
319 }
320 DCHECK(IsMovImmed(Memory<int32_t>(pc)) &&
321 IsOrrImmed(Memory<int32_t>(pc + kInstrSize)) &&
322 IsOrrImmed(Memory<int32_t>(pc + 2 * kInstrSize)) &&
323 IsOrrImmed(Memory<int32_t>(pc + 3 * kInstrSize)));
324 if (icache_flush_mode != SKIP_ICACHE_FLUSH) {
325 FlushInstructionCache(pc, 4 * kInstrSize);
326 }
327 } else {
328 intptr_t branch_offset = target - pc - Instruction::kPcLoadDelta;
329 Instruction* branch = Instruction::At(pc);
330 branch->SetBranchOffset(branch_offset, jit_allocation);
331 if (icache_flush_mode != SKIP_ICACHE_FLUSH) {
332 FlushInstructionCache(pc, kInstrSize);
333 }
334 }
335}

354template <typename T>
355bool UseScratchRegisterScope::CanAcquireVfp() const {
356 VfpRegList* available = assembler_->GetScratchVfpRegisterList();
357 DCHECK_NOT_NULL(available);
358 for (int index = 0; index < T::kNumRegisters; index++) {
359 T reg = T::from_code(index);
360 uint64_t mask = reg.ToVfpRegList();
361 if ((*available & mask) == mask) {
362 return true;
363 }
364 }
365 return false;
366}
367
368template <typename T>
369T UseScratchRegisterScope::AcquireVfp() {
370 VfpRegList* available = assembler_->GetScratchVfpRegisterList();
371 DCHECK_NOT_NULL(available);
372 for (int index = 0; index < T::kNumRegisters; index++) {
373 T reg = T::from_code(index);
374 uint64_t mask = reg.ToVfpRegList();
375 if ((*available & mask) == mask) {
376 *available &= ~mask;
377 return reg;
378 }
379 }
380 UNREACHABLE();
381}
