15void StringEscapeAnalyzer::Run() {
16 for (uint32_t processed = graph_.block_count(); processed > 0; --processed) {
17 BlockIndex block_index = static_cast<BlockIndex>(processed - 1);
18
19 const Block& block = graph_.Get(block_index);
20 ProcessBlock(block);
21 }
22
23 // Because of loop phis, some StringConcat could now be escaping even though
24 // they weren't escaping on first use.
25 ReprocessStringConcats();
26
27 // Now that we know for a fact which StringConcat will be elided, we can
28 // compute which FrameStates will need to be reconstructed in the reducer.
29 ComputeFrameStatesToReconstruct();
30}

32void StringEscapeAnalyzer::ProcessBlock(const Block& block) {
33 for (V<Any> index : base::Reversed(graph_.OperationIndices(block))) {
34 const Operation& op = graph_.Get(index);
35 switch (op.opcode) {
36 case Opcode::kFrameState:
37 // FrameState uses are not considered as escaping.
38 max_frame_state_input_count_ =
39 std::max<uint32_t>(max_frame_state_input_count_, op.input_count);
40 for (V<Any> input_idx : op.inputs()) {
41 if (graph_.Get(input_idx).Is<StringConcatOp>()) {
42 // This FrameState has a StringConcat as input, so we might need to
43 // recreate it in the reducer.
44 maybe_to_reconstruct_frame_states_.push_back(
45 V<FrameState>::Cast(index));
46 break;
47 }
48 }
49 break;
50 case Opcode::kStringConcat:
51 // The inputs of a StringConcat are only escaping if the StringConcat
52 // itself is already escaping itself.
53 if (IsEscaping(index)) {
54 MarkAllInputsAsEscaping(op);
55 } else {
56 maybe_non_escaping_string_concats_.push_back(V<String>::Cast(index));
57 }
58 break;
59 case Opcode::kStringLength:
60 // The first input to StringConcat is the length of the result, which
61 // means that StringLength won't prevent eliding StringConcat:
62 // StringLength(StringConcat(len, left, rigth)) == len
63 break;
64 default:
65 // By default, all uses are considered as escaping their inputs.
66 MarkAllInputsAsEscaping(op);
67 }
68 }
69}

79void StringEscapeAnalyzer::RecursivelyMarkAllStringConcatInputsAsEscaping(
80 const StringConcatOp* concat) {
81 base::SmallVector<const StringConcatOp*, 16> to_mark;
82 to_mark.push_back(concat);
83
84 while (!to_mark.empty()) {
85 const StringConcatOp* curr = to_mark.back();
86 to_mark.pop_back();
87
88 for (V<Any> input_index : curr->inputs()) {
89 const Operation& input = graph_.Get(input_index);
90 if (input.Is<StringConcatOp>() && !IsEscaping(input_index)) {
91 MarkAsEscaping(input_index);
92 to_mark.push_back(&input.Cast<StringConcatOp>());
93 }
94 }
95 }
96}

98void StringEscapeAnalyzer::ReprocessStringConcats() {
99 constexpr uint32_t kMaxOpInputCount = std::numeric_limits<
100 decltype(std::declval<Operation>().input_count)>::max();
101 if (maybe_non_escaping_string_concats_.size() + max_frame_state_input_count_ >
102 kMaxOpInputCount) {
103 // There is a risk that in order to elide some StringConcat, we end up
104 // needing more inputs for a FrameState than the maximum number of possible
105 // inputs. Note that this is a bit of an overapproximation, but it should
106 // still happen very rarely since it would require a huge function with
107 // thousand of local variables and/or parameters. When this happens, we mark
108 // all operations as "escaping", so that the reducer doesn't try to elide
109 // anything.
110 for (V<String> index : maybe_non_escaping_string_concats_) {
111 MarkAsEscaping(index);
112 }
113 }
114
115 for (V<String> index : maybe_non_escaping_string_concats_) {
116 if (IsEscaping(index)) {
117 RecursivelyMarkAllStringConcatInputsAsEscaping(
118 &graph_.Get(index).Cast<StringConcatOp>());
119 }
120 }
121}

123void StringEscapeAnalyzer::ComputeFrameStatesToReconstruct() {
124 for (V<FrameState> frame_state_idx : maybe_to_reconstruct_frame_states_) {
125 const FrameStateOp& frame_state =
126 graph_.Get(frame_state_idx).Cast<FrameStateOp>();
127 for (V<Any> input : frame_state.inputs()) {
128 if (graph_.Get(input).Is<StringConcatOp>() && !IsEscaping(input)) {
129 RecursiveMarkAsShouldReconstruct(frame_state_idx);
130 break;
131 }
132 }
133 }
134}
