101 // Offsets from rbp of function parameters and stored registers.
102 static constexpr int kFramePointerOffset = 0;
103 // Above the frame pointer - function parameters and return address.
104 static constexpr int kReturnAddressOffset =
105 kFramePointerOffset + kSystemPointerSize;
106 static constexpr int kFrameAlign = kReturnAddressOffset + kSystemPointerSize;
107 // Below the frame pointer - the stack frame type marker and locals.
108 static constexpr int kFrameTypeOffset =
109 kFramePointerOffset - kSystemPointerSize;
110 static_assert(kFrameTypeOffset ==
111 CommonFrameConstants::kContextOrFrameTypeOffset);
112
113#ifdef V8_TARGET_OS_WIN
114 // Parameters (first four passed as registers, but with room on stack).
115 // In Microsoft 64-bit Calling Convention, there is room on the callers
116 // stack (before the return address) to spill parameter registers. We
117 // use this space to store the register passed parameters.
118 static constexpr int kInputStringOffset = kFrameAlign;
119 // StartIndex is passed as 32 bit int.
120 static constexpr int kStartIndexOffset =
121 kInputStringOffset + kSystemPointerSize;
122 static constexpr int kInputStartOffset =
123 kStartIndexOffset + kSystemPointerSize;
124 static constexpr int kInputEndOffset = kInputStartOffset + kSystemPointerSize;
125 static constexpr int kRegisterOutputOffset =
126 kInputEndOffset + kSystemPointerSize;
127 // For the case of global regular expression, we have room to store at least
128 // one set of capture results. For the case of non-global regexp, we ignore
129 // this value. NumOutputRegisters is passed as 32-bit value. The upper
130 // 32 bit of this 64-bit stack slot may contain garbage.
131 static constexpr int kNumOutputRegistersOffset =
132 kRegisterOutputOffset + kSystemPointerSize;
133 // DirectCall is passed as 32 bit int (values 0 or 1).
134 static constexpr int kDirectCallOffset =
135 kNumOutputRegistersOffset + kSystemPointerSize;
136 static constexpr int kIsolateOffset = kDirectCallOffset + kSystemPointerSize;
137#else
138 // In AMD64 ABI Calling Convention, the first six integer parameters
139 // are passed as registers, and caller must allocate space on the stack
140 // if it wants them stored. We push the parameters after the frame pointer.
141 static constexpr int kInputStringOffset =
142 kFrameTypeOffset - kSystemPointerSize;
143 static constexpr int kStartIndexOffset =
144 kInputStringOffset - kSystemPointerSize;
145 static constexpr int kInputStartOffset =
146 kStartIndexOffset - kSystemPointerSize;
147 static constexpr int kInputEndOffset = kInputStartOffset - kSystemPointerSize;
148 static constexpr int kRegisterOutputOffset =
149 kInputEndOffset - kSystemPointerSize;
150 // For the case of global regular expression, we have room to store at least
151 // one set of capture results. For the case of non-global regexp, we ignore
152 // this value.
153 static constexpr int kNumOutputRegistersOffset =
154 kRegisterOutputOffset - kSystemPointerSize;
155
156 static constexpr int kDirectCallOffset = kFrameAlign;
157 static constexpr int kIsolateOffset = kDirectCallOffset + kSystemPointerSize;
158#endif

160 // We push callee-save registers that we use after the frame pointer (and
161 // after the parameters).

163 static constexpr int kBackupRsiOffset = kFrameTypeOffset - kSystemPointerSize;
164 static constexpr int kBackupRdiOffset = kBackupRsiOffset - kSystemPointerSize;

171 static constexpr int kNumCalleeSaveRegisters = 1;

173#endif
