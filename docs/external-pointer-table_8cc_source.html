27// An iterator over a set of sets of segments that returns a total ordering of
28// segments in highest to lowest address order. This lets us easily build a
29// sorted singly-linked freelist.
30//
31// When given a single set of segments, it's the same as iterating over
32// std::set<Segment> in reverse order.
33//
34// With multiple segment sets, we still produce a total order. Sets are
35// annotated so that we can associate some data with their segments. This is
36// useful when evacuating the young ExternalPointerTable::Space into the old
37// generation in a major collection, as both spaces could have been compacting,
38// with different starts to the evacuation area.
39template <typename Segment, typename Data>
40class SegmentsIterator {
41 using iterator = typename std::set<Segment>::reverse_iterator;
42 using const_iterator = typename std::set<Segment>::const_reverse_iterator;
43
44 public:
45 SegmentsIterator() = default;
46
47 void AddSegments(const std::set<Segment>& segments, Data data) {
48 streams_.emplace_back(segments.rbegin(), segments.rend(), data);
49 }
50
51 std::optional<std::pair<Segment, Data>> Next() {
52 int stream = -1;
53 int min_stream = -1;
54 std::optional<std::pair<Segment, Data>> result;
55 for (auto [iter, end, data] : streams_) {
56 stream++;
57 if (iter != end) {
58 Segment segment = *iter;
59 if (!result || result.value().first < segment) {
60 min_stream = stream;
61 result.emplace(segment, data);
62 }
63 }
64 }
65 if (result) {
66 streams_[min_stream].iter++;
67 return result;
68 }
69 return {};
70 }
71
72 private:
73 struct Stream {
74 iterator iter;
75 const_iterator end;
76 Data data;
77
78 Stream(iterator iter, const_iterator end, Data data)
79 : iter(iter), end(end), data(data) {}
80 };
81
82 std::vector<Stream> streams_;
83};

85uint32_t ExternalPointerTable::EvacuateAndSweepAndCompact(Space* space,
86 Space* from_space,
87 Counters* counters) {
88 DCHECK(space->BelongsTo(this));
89 DCHECK(!space->is_internal_read_only_space());
90
91 DCHECK_IMPLIES(from_space, from_space->BelongsTo(this));
92 DCHECK_IMPLIES(from_space, !from_space->is_internal_read_only_space());
93
94 // Lock the space. Technically this is not necessary since no other thread can
95 // allocate entries at this point, but some of the methods we call on the
96 // space assert that the lock is held.
97 base::SpinningMutexGuard guard(&space->mutex_);
98 // Same for the invalidated fields mutex.
99 base::SpinningMutexGuard invalidated_fields_guard(
100 &space->invalidated_fields_mutex_);
101
102 // There must not be any entry allocations while the table is being swept as
103 // that would not be safe. Set the freelist to this special marker value to
104 // easily catch any violation of this requirement.
105 space->freelist_head_.store(kEntryAllocationIsForbiddenMarker,
106 std::memory_order_relaxed);
107
108 SegmentsIterator<Segment, CompactionResult> segments_iter;
109 Histogram* counter = counters->external_pointer_table_compaction_outcome();
110 CompactionResult space_compaction = FinishCompaction(space, counter);
111 segments_iter.AddSegments(space->segments_, space_compaction);
112
113 // If from_space is present, take its segments and add them to the sweep
114 // iterator. Wait until after the sweep to actually give from_space's
115 // segments to the other space, to avoid invalidating the iterator.
116 std::set<Segment> from_space_segments;
117 if (from_space) {
118 base::SpinningMutexGuard from_space_guard(&from_space->mutex_);
119 base::SpinningMutexGuard from_space_invalidated_fields_guard(
120 &from_space->invalidated_fields_mutex_);
121
122 std::swap(from_space->segments_, from_space_segments);
123 DCHECK(from_space->segments_.empty());
124
125 CompactionResult from_space_compaction =
126 FinishCompaction(from_space, counter);
127 segments_iter.AddSegments(from_space_segments, from_space_compaction);
128
129 FreelistHead empty_freelist;
130 from_space->freelist_head_.store(empty_freelist, std::memory_order_relaxed);
131
132 for (Address field : from_space->invalidated_fields_)
133 space->invalidated_fields_.push_back(field);
134 from_space->ClearInvalidatedFields();
135 }
136
137 // Sweep top to bottom and rebuild the freelist from newly dead and
138 // previously freed entries while also clearing the marking bit on live
139 // entries and resolving evacuation entries table when compacting the table.
140 // This way, the freelist ends up sorted by index which already makes the
141 // table somewhat self-compacting and is required for the compaction
142 // algorithm so that evacuated entries are evacuated to the start of a space.
143 // This method must run either on the mutator thread or while the mutator is
144 // stopped.
145 uint32_t current_freelist_head = 0;
146 uint32_t current_freelist_length = 0;
147 auto AddToFreelist = [&](uint32_t entry_index) {
148 at(entry_index).MakeFreelistEntry(current_freelist_head);
149 current_freelist_head = entry_index;
150 current_freelist_length++;
151 };
152
153 std::vector<Segment> segments_to_deallocate;
154 while (auto current = segments_iter.Next()) {
155 Segment segment = current->first;
156 CompactionResult compaction = current->second;
157
158 bool segment_will_be_evacuated =
159 compaction.success &&
160 segment.first_entry() >= compaction.start_of_evacuation_area;
161
162 // Remember the state of the freelist before this segment in case this
163 // segment turns out to be completely empty and we deallocate it.
164 uint32_t previous_freelist_head = current_freelist_head;
165 uint32_t previous_freelist_length = current_freelist_length;
166
167 // Process every entry in this segment, again going top to bottom.
168 for (uint32_t i = segment.last_entry(); i >= segment.first_entry(); i--) {
169 auto payload = at(i).GetRawPayload();
170 if (payload.ContainsEvacuationEntry()) {
171 // Segments that will be evacuated cannot contain evacuation entries
172 // into which other entries would be evacuated.
173 DCHECK(!segment_will_be_evacuated);
174
175 // An evacuation entry contains the address of the external pointer
176 // field that owns the entry that is to be evacuated.
177 Address handle_location =
178 payload.ExtractEvacuationEntryHandleLocation();
179
180 // The evacuation entry may be invalidated by the Scavenger that has
181 // freed the object.
182 if (handle_location == kNullAddress) {
183 AddToFreelist(i);
184 continue;
185 }
186
187 // The external pointer field may have been invalidated in the meantime
188 // (for example if the host object has been in-place converted to a
189 // different type of object). In that case, the field no longer
190 // contains an external pointer handle and we therefore cannot evacuate
191 // the old entry. This is fine as the entry is guaranteed to be dead.
192 if (space->FieldWasInvalidated(handle_location)) {
193 // In this case, we must, however, free the evacuation entry.
194 // Otherwise, we would be left with effectively a stale evacuation
195 // entry that we'd try to process again during the next GC.
196 AddToFreelist(i);
197 continue;
198 }
199
200 // Resolve the evacuation entry: take the pointer to the handle from the
201 // evacuation entry, copy the entry to its new location, and finally
202 // update the handle to point to the new entry.
203 //
204 // While we now know that the entry being evacuated is free, we don't
205 // add it to (the start of) the freelist because that would immediately
206 // cause new fragmentation when the next entry is allocated. Instead, we
207 // assume that the segments out of which entries are evacuated will all
208 // be decommitted anyway after this loop, which is usually the case
209 // unless compaction was already aborted during marking.
210 ResolveEvacuationEntryDuringSweeping(
211 i, reinterpret_cast<ExternalPointerHandle*>(handle_location),
212 compaction.start_of_evacuation_area);
213
214 // The entry must now contain an external pointer and be unmarked as
215 // the entry that was evacuated must have been processed already (it
216 // is in an evacuated segment, which are processed first as they are
217 // at the end of the space). This will have cleared the marking bit.
218 DCHECK(at(i).HasExternalPointer(kAnyExternalPointerTagRange));
219 DCHECK(!at(i).GetRawPayload().HasMarkBitSet());
220 } else if (!payload.HasMarkBitSet()) {
221 FreeManagedResourceIfPresent(i);
222 AddToFreelist(i);
223 } else {
224 auto new_payload = payload;
225 new_payload.ClearMarkBit();
226 at(i).SetRawPayload(new_payload);
227 }
228
229 // We must have resolved all evacuation entries. Otherwise, we'll try to
230 // process them again during the next GC, which would cause problems.
231 DCHECK(!at(i).HasEvacuationEntry());
232 }
233
234 // If a segment is completely empty, or if all live entries will be
235 // evacuated out of it at the end of this loop, free the segment.
236 // Note: for segments that will be evacuated, we could avoid building up a
237 // freelist, but it's probably not worth the effort.
238 uint32_t free_entries = current_freelist_length - previous_freelist_length;
239 bool segment_is_empty = free_entries == kEntriesPerSegment;
240 if (segment_is_empty || segment_will_be_evacuated) {
241 segments_to_deallocate.push_back(segment);
242 // Restore the state of the freelist before this segment.
243 current_freelist_head = previous_freelist_head;
244 current_freelist_length = previous_freelist_length;
245 }
246 }
247
248 space->segments_.merge(from_space_segments);
249
250 // We cannot deallocate the segments during the above loop, so do it now.
251 for (auto segment : segments_to_deallocate) {
252#ifdef DEBUG
253 // There should not be any live entries in the segments we are freeing.
254 // TODO(saelo): we should be able to assert here that we're not freeing any
255 // entries here. Otherwise, we'd have to FreeManagedResourceIfPresent.
256 // for (uint32_t i = segment.last_entry(); i >= segment.first_entry(); i--)
257 // {
258 // CHECK(!at(i).HasExternalPointer(kAnyExternalPointerTag));
259 //}
260#endif
261 FreeTableSegment(segment);
262 space->segments_.erase(segment);
263 }
264
265 space->ClearInvalidatedFields();
266
267 FreelistHead new_freelist(current_freelist_head, current_freelist_length);
268 space->freelist_head_.store(new_freelist, std::memory_order_release);
269 DCHECK_EQ(space->freelist_length(), current_freelist_length);
270
271 uint32_t num_live_entries = space->capacity() - current_freelist_length;
272 counters->external_pointers_count()->AddSample(num_live_entries);
273 return num_live_entries;
274}

286void ExternalPointerTable::ResolveEvacuationEntryDuringSweeping(
287 uint32_t new_index, ExternalPointerHandle* handle_location,
288 uint32_t start_of_evacuation_area) {
289 // We must have a valid handle here. If this fails, it might mean that an
290 // object with external pointers was in-place converted to another type of
291 // object without informing the external pointer table.
292 ExternalPointerHandle old_handle = *handle_location;
293 CHECK(IsValidHandle(old_handle));
294
295 uint32_t old_index = HandleToIndex(old_handle);
296 ExternalPointerHandle new_handle = IndexToHandle(new_index);
297
298 // The compaction algorithm always moves an entry from the evacuation area to
299 // the front of the table. These DCHECKs verify this invariant.
300 DCHECK_GE(old_index, start_of_evacuation_area);
301 DCHECK_LT(new_index, start_of_evacuation_area);
302 auto& new_entry = at(new_index);
303 at(old_index).Evacuate(new_entry, EvacuateMarkMode::kLeaveUnmarked);
304 *handle_location = new_handle;
305
306 // If this entry references a managed resource, update the resource to
307 // reference the new entry.
308 if (Address addr = at(new_index).ExtractManagedResourceOrNull()) {
309 ManagedResource* resource = reinterpret_cast<ManagedResource*>(addr);
310 DCHECK_EQ(resource->ept_entry_, old_handle);
311 resource->ept_entry_ = new_handle;
312 }
313}

315void ExternalPointerTable::UpdateAllEvacuationEntries(
316 Space* space, std::function<Address(Address)> function) {
317 DCHECK(space->BelongsTo(this));
318 DCHECK(!space->is_internal_read_only_space());
319
320 if (!space->IsCompacting()) return;
321
322 // Lock the space. Technically this is not necessary since no other thread can
323 // allocate entries at this point, but some of the methods we call on the
324 // space assert that the lock is held.
325 base::SpinningMutexGuard guard(&space->mutex_);
326 // Same for the invalidated fields mutex.
327 base::SpinningMutexGuard invalidated_fields_guard(
328 &space->invalidated_fields_mutex_);
329
330 const uint32_t start_of_evacuation_area =
331 space->start_of_evacuation_area_.load(std::memory_order_relaxed);
332
333 // Iterate until the start of evacuation area.
334 for (auto& segment : space->segments_) {
335 if (segment.first_entry() == start_of_evacuation_area) return;
336 for (uint32_t i = segment.first_entry(); i < segment.last_entry() + 1;
337 ++i) {
338 ExternalPointerTableEntry& entry = at(i);
339 ExternalPointerTableEntry::Payload payload = entry.GetRawPayload();
340 if (!payload.ContainsEvacuationEntry()) {
341 continue;
342 }
343 Address new_location =
344 function(payload.ExtractEvacuationEntryHandleLocation());
345 entry.MakeEvacuationEntry(new_location);
346 }
347 }
