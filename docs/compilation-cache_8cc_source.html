53void CompilationCacheRegExp::Age() {
54 static_assert(kGenerations > 1);
55
56 // Age the generations implicitly killing off the oldest.
57 for (int i = kGenerations - 1; i > 0; i--) {
58 tables_[i] = tables_[i - 1];
59 }
60
61 // Set the first generation as unborn.
62 tables_[0] = ReadOnlyRoots(isolate()).undefined_value();
63}

65void CompilationCacheScript::Age() {
66 DisallowGarbageCollection no_gc;
67 if (IsUndefined(table_, isolate())) return;
68 Tagged<CompilationCacheTable> table = Cast<CompilationCacheTable>(table_);
69
70 for (InternalIndex entry : table->IterateEntries()) {
71 Tagged<Object> key;
72 if (!table->ToKey(isolate(), entry, &key)) continue;
73 DCHECK(IsWeakFixedArray(key));
74
75 Tagged<Object> value = table->PrimaryValueAt(entry);
76 if (!IsUndefined(value, isolate())) {
77 Tagged<SharedFunctionInfo> info = Cast<SharedFunctionInfo>(value);
78 // Clear entries after Bytecode was flushed from SFI.
79 if (!info->HasBytecodeArray()) {
80 table->SetPrimaryValueAt(entry,
81 ReadOnlyRoots(isolate()).undefined_value(),
82 SKIP_WRITE_BARRIER);
83 }
84 }
85 }
86}

88void CompilationCacheEval::Age() {
89 DisallowGarbageCollection no_gc;
90 if (IsUndefined(table_, isolate())) return;
91 Tagged<CompilationCacheTable> table = Cast<CompilationCacheTable>(table_);
92
93 for (InternalIndex entry : table->IterateEntries()) {
94 Tagged<Object> key;
95 if (!table->ToKey(isolate(), entry, &key)) continue;
96
97 if (IsNumber(key, isolate())) {
98 // The ageing mechanism for the initial dummy entry in the eval cache.
99 // The 'key' is the hash represented as a Number. The 'value' is a smi
100 // counting down from kHashGenerations. On reaching zero, the entry is
101 // cleared.
102 // Note: The following static assert only establishes an explicit
103 // connection between initialization- and use-sites of the smi value
104 // field.
105 static_assert(CompilationCacheTable::kHashGenerations);
106 const int new_count = Smi::ToInt(table->PrimaryValueAt(entry)) - 1;
107 if (new_count == 0) {
108 table->RemoveEntry(entry);
109 } else {
110 DCHECK_GT(new_count, 0);
111 table->SetPrimaryValueAt(entry, Smi::FromInt(new_count),
112 SKIP_WRITE_BARRIER);
113 }
114 } else {
115 DCHECK(IsFixedArray(key));
116 // The ageing mechanism for eval caches.
117 Tagged<SharedFunctionInfo> info =
118 Cast<SharedFunctionInfo>(table->PrimaryValueAt(entry));
119 // Clear entries after Bytecode was flushed from SFI.
120 if (!info->HasBytecodeArray()) {
121 table->RemoveEntry(entry);
122 }
123 }
124 }
125}

153CompilationCacheScript::LookupResult CompilationCacheScript::Lookup(
154 Handle<String> source, const ScriptDetails& script_details) {
155 LookupResult result;
156 LookupResult::RawObjects raw_result_for_escaping_handle_scope;
157
158 // Probe the script table. Make sure not to leak handles
159 // into the caller's handle scope.
160 {
161 HandleScope scope(isolate());
162 DirectHandle<CompilationCacheTable> table = GetTable();
163 LookupResult probe = CompilationCacheTable::LookupScript(
164 table, source, script_details, isolate());
165 raw_result_for_escaping_handle_scope = probe.GetRawObjects();
166 }
167 result = LookupResult::FromRawObjects(raw_result_for_escaping_handle_scope,
168 isolate());
169
170 // Once outside the manacles of the handle scope, we need to recheck
171 // to see if we actually found a cached script. If so, we return a
172 // handle created in the caller's handle scope.
173 DirectHandle<Script> script;
174 if (result.script().ToHandle(&script)) {
175 DirectHandle<SharedFunctionInfo> sfi;
176 if (result.toplevel_sfi().ToHandle(&sfi)) {
177 isolate()->counters()->compilation_cache_hits()->Increment();
178 LOG(isolate(), CompilationCacheEvent("hit", "script", *sfi));
179 } else {
180 isolate()->counters()->compilation_cache_partial_hits()->Increment();
181 }
182 } else {
183 isolate()->counters()->compilation_cache_misses()->Increment();
184 }
185 return result;
186}

196InfoCellPair CompilationCacheEval::Lookup(
197 Handle<String> source, Handle<SharedFunctionInfo> outer_info,
198 DirectHandle<NativeContext> native_context, LanguageMode language_mode,
199 int position) {
200 HandleScope scope(isolate());
201 // Make sure not to leak the table into the surrounding handle
202 // scope. Otherwise, we risk keeping old tables around even after
203 // having cleared the cache.
204 InfoCellPair result;
205 DirectHandle<CompilationCacheTable> table = GetTable();
206 result = CompilationCacheTable::LookupEval(
207 table, source, outer_info, native_context, language_mode, position);
208 if (result.has_shared()) {
209 isolate()->counters()->compilation_cache_hits()->Increment();
210 } else {
211 isolate()->counters()->compilation_cache_misses()->Increment();
212 }
213 return result;
214}

229MaybeHandle<RegExpData> CompilationCacheRegExp::Lookup(Handle<String> source,
230 JSRegExp::Flags flags) {
231 HandleScope scope(isolate());
232 // Make sure not to leak the table into the surrounding handle
233 // scope. Otherwise, we risk keeping old tables around even after
234 // having cleared the cache.
235 DirectHandle<Object> result = isolate()->factory()->undefined_value();
236 int generation;
237 for (generation = 0; generation < kGenerations; generation++) {
238 DirectHandle<CompilationCacheTable> table = GetTable(generation);
239 result = table->LookupRegExp(source, flags);
240 if (IsRegExpDataWrapper(*result)) break;
241 }
242 if (IsRegExpDataWrapper(*result)) {
243 Handle<RegExpData> data(Cast<RegExpDataWrapper>(result)->data(isolate()),
244 isolate());
245 if (generation != 0) {
246 Put(source, flags, data);
247 }
248 isolate()->counters()->compilation_cache_hits()->Increment();
249 return scope.CloseAndEscape(data);
250 } else {
251 isolate()->counters()->compilation_cache_misses()->Increment();
252 return MaybeHandle<RegExpData>();
253 }
254}

279InfoCellPair CompilationCache::LookupEval(Handle<String> source,
280 Handle<SharedFunctionInfo> outer_info,
281 DirectHandle<Context> context,
282 LanguageMode language_mode,
283 int position) {
284 InfoCellPair result;
285 if (!IsEnabledScriptAndEval()) return result;
286
287 const char* cache_type;
288
289 DirectHandle<NativeContext> native_context;
290 if (TryCast<NativeContext>(context, &native_context)) {
291 result = eval_global_.Lookup(source, outer_info, native_context,
292 language_mode, position);
293 cache_type = "eval-global";
294
295 } else {
296 DCHECK_NE(position, kNoSourcePosition);
297 DirectHandle<NativeContext> native_context(context->native_context(),
298 isolate());
299 result = eval_contextual_.Lookup(source, outer_info, native_context,
300 language_mode, position);
301 cache_type = "eval-contextual";
302 }
303
304 if (result.has_shared()) {
305 LOG(isolate(), CompilationCacheEvent("hit", cache_type, result.shared()));
306 }
307
308 return result;
309}
