38template <class Next>
39class WasmInJSInliningReducer : public Next {
40 public:
41 TURBOSHAFT_REDUCER_BOILERPLATE(WasmInJSInlining)
42
43 V<Any> REDUCE(Call)(V<CallTarget> callee,
44 OptionalV<turboshaft::FrameState> frame_state,
45 base::Vector<const OpIndex> arguments,
46 const TSCallDescriptor* descriptor, OpEffects effects) {
47 if (!descriptor->js_wasm_call_parameters) {
48 // Regular call, nothing to do with Wasm or inlining. Proceed untouched...
49 return Next::ReduceCall(callee, frame_state, arguments, descriptor,
50 effects);
51 }
52
53 // We shouldn't have attached `JSWasmCallParameters` at this call, unless
54 // we have TS Wasm-in-JS inlining enabled.
55 CHECK(v8_flags.turboshaft_wasm_in_js_inlining);
56
57 const wasm::WasmModule* module =
58 descriptor->js_wasm_call_parameters->module();
59 wasm::NativeModule* native_module =
60 descriptor->js_wasm_call_parameters->native_module();
61 uint32_t func_idx = descriptor->js_wasm_call_parameters->function_index();
62
63 V<Any> result =
64 TryInlineWasmCall(module, native_module, func_idx, arguments);
65 if (result.valid()) {
66 return result;
67 } else {
68 // The JS-to-Wasm wrapper was already inlined by the earlier TurboFan
69 // phase, specifically `JSInliner::ReduceJSWasmCall`. However, it did
70 // not toggle the thread-in-Wasm flag, since we don't want to set it
71 // in the inline case above.
72 // For the non-inline case, we need to toggle the flag now.
73 // TODO(dlehmann,353475584): Reuse the code from
74 // `WasmGraphBuilderBase::BuildModifyThreadInWasmFlag`, but that
75 // requires a different assembler stack...
76 OpIndex isolate_root = __ LoadRootRegister();
77 V<WordPtr> thread_in_wasm_flag_address =
78 __ Load(isolate_root, LoadOp::Kind::RawAligned().Immutable(),
79 MemoryRepresentation::UintPtr(),
80 Isolate::thread_in_wasm_flag_address_offset());
81 __ Store(thread_in_wasm_flag_address, __ Word32Constant(1),
82 LoadOp::Kind::RawAligned(), MemoryRepresentation::Int32(),
83 compiler::kNoWriteBarrier);
84
85 result =
86 Next::ReduceCall(callee, frame_state, arguments, descriptor, effects);
87
88 __ Store(thread_in_wasm_flag_address, __ Word32Constant(0),
89 LoadOp::Kind::RawAligned(), MemoryRepresentation::Int32(),
90 compiler::kNoWriteBarrier);
91
92 return result;
93 }
94 }
95
96 private:
97 V<Any> TryInlineWasmCall(const wasm::WasmModule* module,
98 wasm::NativeModule* native_module, uint32_t func_idx,
99 base::Vector<const OpIndex> arguments);
100};

158 void StartFunction(FullDecoder* decoder) {
159 locals_.resize(decoder->num_locals());
160
161 // Initialize the function parameters, which are part of the local space.
162 CHECK_LE(arguments_.size(), locals_.size());
163 std::copy(arguments_.begin(), arguments_.end(), locals_.begin());
164
165 // Initialize the non-parameter locals.
166 uint32_t index = static_cast<uint32_t>(decoder->sig_->parameter_count());
167 CHECK_EQ(index, arguments_.size());
168 while (index < decoder->num_locals()) {
169 ValueType type = decoder->local_type(index);
170 OpIndex op;
171 if (!type.is_defaultable()) {
172 DCHECK(type.is_reference());
173 // TODO(jkummerow): Consider using "the hole" instead, to make any
174 // illegal uses more obvious.
175 op = __ Null(type.AsNullable());
176 } else {
177 op = DefaultValue(type);
178 }
179 while (index < decoder->num_locals() &&
180 decoder->local_type(index) == type) {
181 locals_[index++] = op;
182 }
183 }

1149template <class Next>
1150V<Any> WasmInJSInliningReducer<Next>::TryInlineWasmCall(
1151 const wasm::WasmModule* module, wasm::NativeModule* native_module,
1152 uint32_t func_idx, base::Vector<const OpIndex> arguments) {
1153 const wasm::WasmFunction& func = module->functions[func_idx];
1154
1155 TRACE("Considering wasm function ["
1156 << func_idx << "] "
1157 << JSInliner::WasmFunctionNameForTrace(native_module, func_idx)
1158 << " of module " << module << " for inlining");
1159
1160 if (wasm::is_asmjs_module(module)) {
1161 TRACE("- not inlining: asm.js-in-JS inlining is not supported");
1162 return OpIndex::Invalid();
1163 }
1164
1165 if (func_idx < module->num_imported_functions) {
1166 TRACE("- not inlining: call to an imported function");
1167 return OpIndex::Invalid();
1168 }
1169 DCHECK_LT(func_idx - module->num_imported_functions,
1170 module->num_declared_functions);
1171
1172 // TODO(42204563): Support shared-everything proposal (at some point, or
1173 // possibly never).
1174 bool is_shared = module->type(func.sig_index).is_shared;
1175 if (is_shared) {
1176 TRACE("- not inlining: shared everything is not supported");
1177 return OpIndex::Invalid();
1178 }
1179
1180 base::Vector<const uint8_t> module_bytes = native_module->wire_bytes();
1181 const uint8_t* start = module_bytes.begin() + func.code.offset();
1182 const uint8_t* end = module_bytes.begin() + func.code.end_offset();
1183
1184 wasm::FunctionBody func_body{func.sig, func.code.offset(), start, end,
1185 is_shared};
1186
1187 auto env = wasm::CompilationEnv::ForModule(native_module);
1188 wasm::WasmDetectedFeatures detected{};
1189
1190 // JS-to-Wasm wrapper inlining doesn't support multi-value at the moment,
1191 // so we should never reach here with more than 1 return value.
1192 DCHECK_LE(func.sig->return_count(), 1);
1193 base::Vector<const OpIndex> arguments_without_instance =
1194 arguments.SubVectorFrom(1);
1195 V<WasmTrustedInstanceData> trusted_instance_data =
1196 arguments[wasm::kWasmInstanceDataParameterIndex];
1197
1198 Block* inlinee_body_and_rest = __ NewBlock();
1199 __ Goto(inlinee_body_and_rest);
1200
1201 // First pass: Decode Wasm body to see if we could inline or would bail out.
1202 // Emit into an unreachable block. We are not interested in the operations at
1203 // this point, only in the decoder status afterwards.
1204 Block* unreachable = __ NewBlock();
1205 __ Bind(unreachable);
1206
1207 using Interface = WasmInJsInliningInterface<Assembler<ReducerList>>;
1208 using Decoder =
1209 wasm::WasmFullDecoder<typename Interface::ValidationTag, Interface>;
1210 Decoder can_inline_decoder(Asm().phase_zone(), env.module,
1211 env.enabled_features, &detected, func_body, Asm(),
1212 arguments_without_instance, trusted_instance_data);
1213 DCHECK(env.module->function_was_validated(func_idx));
1214 can_inline_decoder.Decode();
1215
1216 // The function was already validated, so decoding can only fail if we bailed
1217 // out due to an unsupported instruction.
1218 if (!can_inline_decoder.ok()) {
1219 TRACE("- not inlining: " << can_inline_decoder.error().message());
1220 __ Bind(inlinee_body_and_rest);
1221 return OpIndex::Invalid();
1222 }
1223
1224 // Second pass: Actually emit the inlinee instructions now.
1225 __ Bind(inlinee_body_and_rest);
1226 Decoder emitting_decoder(Asm().phase_zone(), env.module, env.enabled_features,
1227 &detected, func_body, Asm(),
1228 arguments_without_instance, trusted_instance_data);
1229 emitting_decoder.Decode();
1230 DCHECK(emitting_decoder.ok());
1231 DCHECK(emitting_decoder.interface().Result().valid());
1232 TRACE("- inlining");
1233 return emitting_decoder.interface().Result();
1234}
