88template <AccessMode mode>
89inline void MarkingBitmap::SetRange(MarkBitIndex start_index,
90 MarkBitIndex end_index) {
91 if (start_index >= end_index) return;
92 end_index--;
93
94 const CellIndex start_cell_index = IndexToCell(start_index);
95 const MarkBit::CellType start_index_mask = IndexInCellMask(start_index);
96 const CellIndex end_cell_index = IndexToCell(end_index);
97 const MarkBit::CellType end_index_mask = IndexInCellMask(end_index);
98
99 if (start_cell_index != end_cell_index) {
100 // Firstly, fill all bits from the start address to the end of the first
101 // cell with 1s.
102 SetBitsInCell<mode>(start_cell_index, ~(start_index_mask - 1));
103 // Then fill all in between cells with 1s.
104 SetCellRangeRelaxed<mode>(start_cell_index + 1, end_cell_index);
105 // Finally, fill all bits until the end address in the last cell with 1s.
106 SetBitsInCell<mode>(end_cell_index, end_index_mask | (end_index_mask - 1));
107 } else {
108 SetBitsInCell<mode>(start_cell_index,
109 end_index_mask | (end_index_mask - start_index_mask));
110 }
111 if (mode == AccessMode::ATOMIC) {
112 // This fence prevents re-ordering of publishing stores with the mark-bit
113 // setting stores.
114 base::SeqCst_MemoryFence();
115 }
116}

119inline void MarkingBitmap::ClearRange(MarkBitIndex start_index,
120 MarkBitIndex end_index) {
121 if (start_index >= end_index) return;
122 end_index--;
123
124 const CellIndex start_cell_index = IndexToCell(start_index);
125 const MarkBit::CellType start_index_mask = IndexInCellMask(start_index);
126 const CellIndex end_cell_index = IndexToCell(end_index);
127 const MarkBit::CellType end_index_mask = IndexInCellMask(end_index);
128
129 if (start_cell_index != end_cell_index) {
130 // Firstly, fill all bits from the start address to the end of the first
131 // cell with 0s.
132 ClearBitsInCell<mode>(start_cell_index, ~(start_index_mask - 1));

134 ClearCellRangeRelaxed<mode>(start_cell_index + 1, end_cell_index);
135 // Finally, set all bits until the end address in the last cell with 0s.
136 ClearBitsInCell<mode>(end_cell_index,
137 end_index_mask | (end_index_mask - 1));
138 } else {
139 ClearBitsInCell<mode>(start_cell_index,
140 end_index_mask | (end_index_mask - start_index_mask));
141 }
142 if (mode == AccessMode::ATOMIC) {
143 // This fence prevents re-ordering of publishing stores with the mark-bit
144 // clearing stores.
145 base::SeqCst_MemoryFence();
146 }

177// static
178inline Address MarkingBitmap::FindPreviousValidObject(const PageMetadata* page,
179 Address maybe_inner_ptr) {
180 DCHECK(page->Contains(maybe_inner_ptr));
181 const auto* bitmap = page->marking_bitmap();
182 const MarkBit::CellType* cells = bitmap->cells();
183
184 // The first actual bit of the bitmap, corresponding to page->area_start(),
185 // is at start_index which is somewhere in (not necessarily at the start of)
186 // start_cell_index.
187 const auto start_index = MarkingBitmap::AddressToIndex(page->area_start());
188 const auto start_cell_index = MarkingBitmap::IndexToCell(start_index);
189 // We assume that all markbits before start_index are clear:
190 // SLOW_DCHECK(bitmap->AllBitsClearInRange(0, start_index));
191 // This has already been checked for the entire bitmap before starting marking
192 // by MarkCompactCollector::VerifyMarkbitsAreClean.
193
194 const auto index = MarkingBitmap::AddressToIndex(maybe_inner_ptr);
195 auto cell_index = MarkingBitmap::IndexToCell(index);
196 const auto index_in_cell = MarkingBitmap::IndexInCell(index);
197 DCHECK_GT(MarkingBitmap::kBitsPerCell, index_in_cell);
198 auto cell = cells[cell_index];
199
200 // Clear the bits corresponding to higher addresses in the cell.
201 cell &= ((~static_cast<MarkBit::CellType>(0)) >>
202 (MarkingBitmap::kBitsPerCell - index_in_cell - 1));
203
204 // Traverse the bitmap backwards, until we find a markbit that is set and
205 // whose previous markbit (if it exists) is unset.
206 // First, iterate backwards to find a cell with any set markbit.
207 while (cell == 0 && cell_index > start_cell_index) cell = cells[--cell_index];
208 if (cell == 0) {
209 DCHECK_EQ(start_cell_index, cell_index);
210 // We have reached the start of the page.
211 return page->area_start();
212 }
213
214 // We have found such a cell.
215 const auto leading_zeros = base::bits::CountLeadingZeros(cell);
216 const auto leftmost_ones =
217 base::bits::CountLeadingZeros(~(cell << leading_zeros));
218 const auto index_of_last_leftmost_one =
219 MarkingBitmap::kBitsPerCell - leading_zeros - leftmost_ones;
220
221 const MemoryChunk* chunk = page->Chunk();
222
223 // If the leftmost sequence of set bits does not reach the start of the cell,
224 // we found it.
225 if (index_of_last_leftmost_one > 0) {
226 return chunk->address() + MarkingBitmap::IndexToAddressOffset(
227 cell_index * MarkingBitmap::kBitsPerCell +
228 index_of_last_leftmost_one);
229 }
230
231 // The leftmost sequence of set bits reaches the start of the cell. We must
232 // keep traversing backwards until we find the first unset markbit.
233 if (cell_index == start_cell_index) {
234 // We have reached the start of the page.
235 return page->area_start();
236 }
237
238 // Iterate backwards to find a cell with any unset markbit.
239 do {
240 cell = cells[--cell_index];
241 } while (~cell == 0 && cell_index > start_cell_index);
242 if (~cell == 0) {
243 DCHECK_EQ(start_cell_index, cell_index);
244 // We have reached the start of the page.
245 return page->area_start();
246 }
247
248 // We have found such a cell.
249 const auto leading_ones = base::bits::CountLeadingZeros(~cell);
250 const auto index_of_last_leading_one =
251 MarkingBitmap::kBitsPerCell - leading_ones;
252 DCHECK_LT(0, index_of_last_leading_one);
253 return chunk->address() + MarkingBitmap::IndexToAddressOffset(
254 cell_index * MarkingBitmap::kBitsPerCell +
255 index_of_last_leading_one);
256}

269std::optional<MarkingHelper::WorklistTarget> MarkingHelper::ShouldMarkObject(
270 Heap* heap, Tagged<HeapObject> object) {
271 const auto* chunk = MemoryChunk::FromHeapObject(object);
272 const auto flags = chunk->GetFlags();
273 if (flags & MemoryChunk::READ_ONLY_HEAP) {
274 return {};
275 }
276 if (v8_flags.black_allocated_pages &&
277 V8_UNLIKELY(flags & MemoryChunk::BLACK_ALLOCATED)) {
278 DCHECK(!(flags & MemoryChunk::kIsInYoungGenerationMask));
279 return {};
280 }
281 if (V8_LIKELY(!(flags & MemoryChunk::IN_WRITABLE_SHARED_SPACE))) {
282 return {MarkingHelper::WorklistTarget::kRegular};
283 }
284 // Object in shared writable space. Only mark it if the Isolate is owning the
285 // shared space.
286 //
287 // TODO(340989496): Speed up check here by keeping the flag on Heap.
288 if (heap->isolate()->is_shared_space_isolate()) {
289 return {MarkingHelper::WorklistTarget::kRegular};
290 }
291 return {};
292}

294// static
295MarkingHelper::LivenessMode MarkingHelper::GetLivenessMode(
296 Heap* heap, Tagged<HeapObject> object) {
297 const auto* chunk = MemoryChunk::FromHeapObject(object);
298 const auto flags = chunk->GetFlags();
299 if (flags & MemoryChunk::READ_ONLY_HEAP) {
300 return MarkingHelper::LivenessMode::kAlwaysLive;
301 }
302 if (v8_flags.black_allocated_pages &&
303 (flags & MemoryChunk::BLACK_ALLOCATED)) {
304 return MarkingHelper::LivenessMode::kAlwaysLive;
305 }
306 if (V8_LIKELY(!(flags & MemoryChunk::IN_WRITABLE_SHARED_SPACE))) {
307 return MarkingHelper::LivenessMode::kMarkbit;
308 }
309 // Object in shared writable space. Only mark it if the Isolate is owning the
310 // shared space.
311 //
312 // TODO(340989496): Speed up check here by keeping the flag on Heap.
313 if (heap->isolate()->is_shared_space_isolate()) {
314 return MarkingHelper::LivenessMode::kMarkbit;
315 }
316 return MarkingHelper::LivenessMode::kAlwaysLive;
317}

339// static
340template <typename MarkingState>
341bool MarkingHelper::TryMarkAndPush(Heap* heap,
342 MarkingWorklists::Local* marking_worklist,
343 MarkingState* marking_state,
344 WorklistTarget target_worklist,
345 Tagged<HeapObject> object) {
346 DCHECK(heap->Contains(object));
347 if (marking_state->TryMark(object)) {
348 if (V8_LIKELY(target_worklist == WorklistTarget::kRegular)) {
349 marking_worklist->Push(object);
350 }
351 return true;
352 }
353 return false;
354}
