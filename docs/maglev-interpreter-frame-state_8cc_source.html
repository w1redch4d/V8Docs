43void KnownNodeAspects::Merge(const KnownNodeAspects& other, Zone* zone) {
44 bool any_merged_map_is_unstable = false;
45 DestructivelyIntersect(node_infos, other.node_infos,
46 [&](NodeInfo& lhs, const NodeInfo& rhs) {
47 lhs.MergeWith(rhs, zone, any_merged_map_is_unstable);
48 return !lhs.no_info_available();
49 });
50
51 if (effect_epoch_ != other.effect_epoch_) {
52 effect_epoch_ = std::max(effect_epoch_, other.effect_epoch_) + 1;
53 }
54 DestructivelyIntersect(
55 available_expressions, other.available_expressions,
56 [&](const AvailableExpression& lhs, const AvailableExpression& rhs) {
57 DCHECK_IMPLIES(lhs.node == rhs.node,
58 lhs.effect_epoch == rhs.effect_epoch);
59 DCHECK_NE(lhs.effect_epoch, kEffectEpochOverflow);
60 DCHECK_EQ(Node::needs_epoch_check(lhs.node->opcode()),
61 lhs.effect_epoch != kEffectEpochForPureInstructions);
62
63 return lhs.node == rhs.node && lhs.effect_epoch >= effect_epoch_;
64 });
65
66 this->any_map_for_any_node_is_unstable = any_merged_map_is_unstable;
67
68 auto merge_loaded_properties =
69 [](ZoneMap<ValueNode*, ValueNode*>& lhs,
70 const ZoneMap<ValueNode*, ValueNode*>& rhs) {
71 // Loaded properties are maps of maps, so just do the destructive
72 // intersection recursively.
73 DestructivelyIntersect(lhs, rhs);
74 return !lhs.empty();
75 };
76 DestructivelyIntersect(loaded_constant_properties,
77 other.loaded_constant_properties,
78 merge_loaded_properties);
79 DestructivelyIntersect(loaded_properties, other.loaded_properties,
80 merge_loaded_properties);
81 DestructivelyIntersect(loaded_context_constants,
82 other.loaded_context_constants);
83 if (may_have_aliasing_contexts() != other.may_have_aliasing_contexts()) {
84 if (may_have_aliasing_contexts() == ContextSlotLoadsAlias::None) {
85 may_have_aliasing_contexts_ = other.may_have_aliasing_contexts_;
86 } else if (other.may_have_aliasing_contexts() !=
87 ContextSlotLoadsAlias::None) {
88 may_have_aliasing_contexts_ = ContextSlotLoadsAlias::Yes;
89 }
90 }
91 DestructivelyIntersect(loaded_context_slots, other.loaded_context_slots);
92}

96template <typename Key>
97bool NextInIgnoreList(typename ZoneSet<Key>::const_iterator& ignore,
98 typename ZoneSet<Key>::const_iterator& ignore_end,
99 const Key& cur) {
100 while (ignore != ignore_end && *ignore < cur) {
101 ++ignore;
102 }
103 return ignore != ignore_end && *ignore == cur;
104}

127KnownNodeAspects::KnownNodeAspects(const KnownNodeAspects& other,
128 bool optimistic_initial_state,
129 LoopEffects* loop_effects, Zone* zone)
130 : any_map_for_any_node_is_unstable(false),
131 loaded_constant_properties(other.loaded_constant_properties),
132 loaded_properties(zone),
133 loaded_context_constants(other.loaded_context_constants),
134 loaded_context_slots(zone),
135 available_expressions(zone),
136 may_have_aliasing_contexts_(
137 KnownNodeAspects::ContextSlotLoadsAlias::None),
138 effect_epoch_(other.effect_epoch_),
139 node_infos(zone) {
140 if (!other.any_map_for_any_node_is_unstable) {
141 node_infos = other.node_infos;
142#ifdef DEBUG
143 for (const auto& it : node_infos) {
144 DCHECK(!it.second.any_map_is_unstable());
145 }
146#endif
147 } else if (optimistic_initial_state &&
148 !loop_effects->unstable_aspects_cleared) {
149 node_infos = other.node_infos;
150 any_map_for_any_node_is_unstable = other.any_map_for_any_node_is_unstable;
151 } else {
152 for (const auto& it : other.node_infos) {
153 node_infos.emplace(it.first,
154 NodeInfo::ClearUnstableMapsOnCopy{it.second});
155 }
156 }
157 if (optimistic_initial_state && !loop_effects->unstable_aspects_cleared) {
158 // IMPORTANT: Whatever we clone here needs to be checked for consistency
159 // in when we try to terminate the loop in `IsCompatibleWithLoopHeader`.
160 if (loop_effects->objects_written.empty() &&
161 loop_effects->keys_cleared.empty()) {
162 loaded_properties = other.loaded_properties;
163 } else {
164 auto cleared_key = loop_effects->keys_cleared.begin();
165 auto cleared_keys_end = loop_effects->keys_cleared.end();
166 auto cleared_obj = loop_effects->objects_written.begin();
167 auto cleared_objs_end = loop_effects->objects_written.end();
168 for (auto loaded_key : other.loaded_properties) {
169 if (NextInIgnoreList(cleared_key, cleared_keys_end, loaded_key.first)) {
170 continue;
171 }
172 auto& props_for_key =
173 loaded_properties.try_emplace(loaded_key.first, zone).first->second;
174 for (auto loaded_obj : loaded_key.second) {
175 if (!NextInIgnoreList(cleared_obj, cleared_objs_end,
176 loaded_obj.first)) {
177 props_for_key.emplace(loaded_obj);
178 }
179 }
180 }
181 }
182 if (loop_effects->context_slot_written.empty()) {
183 loaded_context_slots = other.loaded_context_slots;
184 } else {
185 auto slot_written = loop_effects->context_slot_written.begin();
186 auto slot_written_end = loop_effects->context_slot_written.end();
187 for (auto loaded : other.loaded_context_slots) {
188 if (!NextInIgnoreList(slot_written, slot_written_end, loaded.first)) {
189 loaded_context_slots.emplace(loaded);
190 }
191 }
192 }
193 if (!loaded_context_slots.empty()) {
194 if (loop_effects->may_have_aliasing_contexts) {
195 may_have_aliasing_contexts_ = ContextSlotLoadsAlias::Yes;
196 } else {
197 may_have_aliasing_contexts_ = other.may_have_aliasing_contexts();
198 }
199 }
200 }
201
202 // To account for the back-jump we must not allow effects to be reshuffled
203 // across loop headers.
204 // TODO(olivf): Only do this if the loop contains write effects.
205 increment_effect_epoch();
206 for (const auto& e : other.available_expressions) {
207 if (e.second.effect_epoch >= effect_epoch()) {
208 available_expressions.emplace(e);
209 }
210 }

215// Takes two ordered maps and ensures that every element in `as` is
216// * also present in `bs` and
217// * `Compare(a, b)` holds for each value.
218template <typename As, typename Bs, typename CompareFunction,
219 typename IsEmptyFunction = std::nullptr_t>
220bool AspectIncludes(const As& as, const Bs& bs, const CompareFunction& Compare,
221 const IsEmptyFunction IsEmpty = nullptr) {
222 typename As::const_iterator a = as.begin();
223 typename Bs::const_iterator b = bs.begin();
224 while (a != as.end()) {
225 if constexpr (!std::is_same_v<IsEmptyFunction, std::nullptr_t>) {
226 if (IsEmpty(a->second)) {
227 ++a;
228 continue;
229 }
230 }
231 if (b == bs.end()) return false;
232 while (b->first < a->first) {
233 ++b;
234 if (b == bs.end()) return false;
235 }
236 if (!(a->first == b->first)) return false;
237 if (!Compare(a->second, b->second)) {
238 return false;
239 }
240 ++a;
241 ++b;
242 }
243 return true;
244}

246// Same as above but allows `as` to contain empty collections as values, which
247// do not need to be present in `bs`.
248template <typename As, typename Bs, typename Function>
249bool MaybeEmptyAspectIncludes(const As& as, const Bs& bs,
250 const Function& Compare) {
251 return AspectIncludes<As, Bs, Function>(as, bs, Compare,
252 [](auto x) { return x.empty(); });
253}
254
255template <typename As, typename Bs, typename Function>
256bool MaybeNullAspectIncludes(const As& as, const Bs& bs,
257 const Function& Compare) {
258 return AspectIncludes<As, Bs, Function>(as, bs, Compare,
259 [](auto x) { return x == nullptr; });
260}

263 if (!NodeTypeIs(after.type(), before.type())) {

273 }

289bool KnownNodeAspects::IsCompatibleWithLoopHeader(
290 const KnownNodeAspects& loop_header) const {
291 // Needs to be in sync with `CloneForLoopHeader(zone, true)`.
292
293 // Analysis state can change with loads.
294 if (!loop_header.loaded_context_slots.empty() &&
295 loop_header.may_have_aliasing_contexts() != ContextSlotLoadsAlias::Yes &&
296 loop_header.may_have_aliasing_contexts() !=
297 may_have_aliasing_contexts() &&
298 may_have_aliasing_contexts() != ContextSlotLoadsAlias::None) {
299 if (V8_UNLIKELY(v8_flags.trace_maglev_loop_speeling)) {
300 std::cout << "KNA after loop has incompatible "
301 "loop_header.may_have_aliasing_contexts\n";
302 }
303 return false;
304 }
305
306 bool had_effects = effect_epoch() != loop_header.effect_epoch();
307
308 if (!had_effects) {
309 if (!AspectIncludes(loop_header.node_infos, node_infos, NodeInfoTypeIs,
310 NodeInfoIsEmpty)) {
311 if (V8_UNLIKELY(v8_flags.trace_maglev_loop_speeling)) {
312 std::cout << "KNA after effectless loop has incompatible node_infos\n";
313 }
314 return false;
315 }
316 // In debug builds we do a full comparison to ensure that without an effect
317 // epoch change all unstable properties still hold.
318#ifndef DEBUG
319 return true;
320#endif
321 }
322
323 if (!AspectIncludes(loop_header.node_infos, node_infos, NodeInfoIncludes,
324 NodeInfoIsEmpty)) {
325 if (V8_UNLIKELY(v8_flags.trace_maglev_loop_speeling)) {
326 std::cout << "KNA after loop has incompatible node_infos\n";
327 }
328 DCHECK(had_effects);
329 return false;
330 }
331
332 if (!MaybeEmptyAspectIncludes(
333 loop_header.loaded_properties, loaded_properties,
334 [](auto a, auto b) { return AspectIncludes(a, b, SameValue); })) {
335 if (V8_UNLIKELY(v8_flags.trace_maglev_loop_speeling)) {
336 std::cout << "KNA after loop has incompatible loaded_properties\n";
337 }
338 DCHECK(had_effects);
339 return false;
340 }
341
342 if (!MaybeNullAspectIncludes(loop_header.loaded_context_slots,
343 loaded_context_slots, SameValue)) {
344 if (V8_UNLIKELY(v8_flags.trace_maglev_loop_speeling)) {
345 std::cout << "KNA after loop has incompatible loaded_context_slots\n";
346 }
347 DCHECK(had_effects);
348 return false;
349 }
350
351 return true;
352}

355MergePointInterpreterFrameState* MergePointInterpreterFrameState::New(
356 const MaglevCompilationUnit& info, const InterpreterFrameState& state,
357 int merge_offset, int predecessor_count, BasicBlock* predecessor,
358 const compiler::BytecodeLivenessState* liveness) {
359 MergePointInterpreterFrameState* merge_state =
360 info.zone()->New<MergePointInterpreterFrameState>(
361 info, merge_offset, predecessor_count, 1,
362 info.zone()->AllocateArray<BasicBlock*>(predecessor_count),
363 BasicBlockType::kDefault, liveness);
364 int i = 0;
365 merge_state->frame_state_.ForEachValue(
366 info, [&](ValueNode*& entry, interpreter::Register reg) {
367 entry = state.get(reg);
368 // Initialise the alternatives list and cache the alternative
369 // representations of the node.
370 Alternatives::List* per_predecessor_alternatives =
371 new (&merge_state->per_predecessor_alternatives_[i])
372 Alternatives::List();
373 per_predecessor_alternatives->Add(info.zone()->New<Alternatives>(
374 state.known_node_aspects()->TryGetInfoFor(entry)));
375 i++;
376 });
377 merge_state->predecessors_[0] = predecessor;
378 merge_state->known_node_aspects_ =
379 state.known_node_aspects()->Clone(info.zone());
380 state.virtual_objects().Snapshot();
381 merge_state->set_virtual_objects(state.virtual_objects());
382 return merge_state;
383}

386MergePointInterpreterFrameState* MergePointInterpreterFrameState::NewForLoop(
387 const InterpreterFrameState& start_state, const MaglevCompilationUnit& info,
388 int merge_offset, int predecessor_count,
389 const compiler::BytecodeLivenessState* liveness,
390 const compiler::LoopInfo* loop_info, bool has_been_peeled) {
391 MergePointInterpreterFrameState* state =
392 info.zone()->New<MergePointInterpreterFrameState>(
393 info, merge_offset, predecessor_count, 0,
394 info.zone()->AllocateArray<BasicBlock*>(predecessor_count),
395 BasicBlockType::kLoopHeader, liveness);
396 state->bitfield_ =
397 kIsLoopWithPeeledIterationBit::update(state->bitfield_, has_been_peeled);
398 state->loop_metadata_ = LoopMetadata{loop_info, nullptr};
399 if (loop_info->resumable()) {
400 state->known_node_aspects_ =
401 info.zone()->New<KnownNodeAspects>(info.zone());
402 state->bitfield_ = kIsResumableLoopBit::update(state->bitfield_, true);
403 }
404 auto& assignments = loop_info->assignments();
405 auto& frame_state = state->frame_state_;
406 int i = 0;
407 frame_state.ForEachParameter(
408 info, [&](ValueNode*& entry, interpreter::Register reg) {
409 entry = nullptr;
410 if (assignments.ContainsParameter(reg.ToParameterIndex())) {
411 entry = state->NewLoopPhi(info.zone(), reg);
412 } else if (state->is_resumable_loop()) {
413 // Copy initial values out of the start state.
414 entry = start_state.get(reg);
415 // Initialise the alternatives list for this value.
416 new (&state->per_predecessor_alternatives_[i]) Alternatives::List();
417 DCHECK(entry->Is<InitialValue>());
418 }
419 ++i;
420 });
421 frame_state.context(info) = nullptr;
422 if (state->is_resumable_loop()) {
423 // While contexts are always the same at specific locations, resumable loops
424 // do have different nodes to set the context across resume points. Create a
425 // phi for them.
426 frame_state.context(info) = state->NewLoopPhi(
427 info.zone(), interpreter::Register::current_context());
428 }
429 frame_state.ForEachLocal(
430 info, [&](ValueNode*& entry, interpreter::Register reg) {
431 entry = nullptr;
432 if (assignments.ContainsLocal(reg.index())) {
433 entry = state->NewLoopPhi(info.zone(), reg);
434 }
435 });
436 DCHECK(!frame_state.liveness()->AccumulatorIsLive());
437 return state;
438}

559void MergePointInterpreterFrameState::MergeVirtualObject(
560 MaglevGraphBuilder* builder, const VirtualObject::List unmerged_vos,
561 const KnownNodeAspects& unmerged_aspects, VirtualObject* merged,
562 VirtualObject* unmerged) {
563 if (merged == unmerged) {
564 // No need to merge.
565 return;
566 }
567 // Currently, the graph builder will never change the VO map.
568 DCHECK(unmerged->map().equals(merged->map()));
569 DCHECK_EQ(merged->slot_count(), unmerged->slot_count());
570 DCHECK_EQ(merged->allocation(), unmerged->allocation());
571
572 if (v8_flags.trace_maglev_graph_building) {
573 std::cout << " - Merging VOS: "
574 << PrintNodeLabel(builder->compilation_unit()->graph_labeller(),
575 merged)
576 << "(merged) and "
577 << PrintNodeLabel(builder->compilation_unit()->graph_labeller(),
578 unmerged)
579 << "(unmerged)" << std::endl;
580 }
581
582 VirtualObject* result = builder->CreateVirtualObjectForMerge(
583 unmerged->map(), unmerged->slot_count());
584 for (uint32_t i = 0; i < merged->slot_count(); i++) {
585 std::optional<ValueNode*> merged_value_opt = MergeVirtualObjectValue(
586 builder, unmerged_aspects, merged->get_by_index(i),
587 unmerged->get_by_index(i));
588 if (!merged_value_opt.has_value()) {
589 // Merge failed, we should escape the allocation instead.
590 unmerged->allocation()->ForceEscaping();
591 return;
592 }
593 result->set_by_index(i, merged_value_opt.value());
594 }
595 result->set_allocation(unmerged->allocation());
596 result->Snapshot();
597 unmerged->allocation()->UpdateObject(result);
598 frame_state_.virtual_objects().Add(result);
599}

601void MergePointInterpreterFrameState::MergeVirtualObjects(
602 MaglevGraphBuilder* builder, MaglevCompilationUnit& compilation_unit,
603 const VirtualObject::List unmerged_vos,
604 const KnownNodeAspects& unmerged_aspects) {
605 if (frame_state_.virtual_objects().is_empty()) return;
606 if (unmerged_vos.is_empty()) return;
607
608 frame_state_.virtual_objects().Snapshot();
609
610 PrintVirtualObjects(compilation_unit, unmerged_vos, "VOs before merge:");
611
612 SmallZoneMap<InlinedAllocation*, VirtualObject*, 10> unmerged_map(
613 builder->zone());
614 SmallZoneMap<InlinedAllocation*, VirtualObject*, 10> merged_map(
615 builder->zone());
616
617 // We iterate both list in reversed order of ids collecting the umerged
618 // objects into the map, until we find a common virtual object.
619 VirtualObject::List::WalkUntilCommon(
620 frame_state_.virtual_objects(), unmerged_vos,
621 [&](VirtualObject* vo, VirtualObject::List vos) {
622 // If we have a version in the map, it should be the most up-to-date,
623 // since the list is in reverse order.
624 auto& map = unmerged_vos == vos ? unmerged_map : merged_map;
625 map.emplace(vo->allocation(), vo);
626 });
627
628 // Walk the merged map (values from the merged state) and merge values.
629 for (auto [_, merged] : merged_map) {
630 VirtualObject* unmerged = nullptr;
631 auto it = unmerged_map.find(merged->allocation());
632 if (it != unmerged_map.end()) {
633 unmerged = it->second;
634 unmerged_map.erase(it);
635 } else {
636 unmerged = unmerged_vos.FindAllocatedWith(merged->allocation());
637 }
638 if (unmerged != nullptr) {
639 MergeVirtualObject(builder, unmerged_vos, unmerged_aspects, merged,
640 unmerged);
641 }
642 }
643
644 // Walk the unmerged map (values from the interpreter frame state) and merge
645 // values. If the value was already merged, we would have removed from the
646 // unmerged_map.
647 for (auto [_, unmerged] : unmerged_map) {
648 VirtualObject* merged = nullptr;
649 auto it = merged_map.find(unmerged->allocation());
650 if (it != merged_map.end()) {
651 merged = it->second;
652 } else {
653 merged = frame_state_.virtual_objects().FindAllocatedWith(
654 unmerged->allocation());
655 }
656 if (merged != nullptr) {
657 MergeVirtualObject(builder, unmerged_vos, unmerged_aspects, merged,
658 unmerged);
659 }
660 }
661
662 PrintVirtualObjects(compilation_unit, unmerged_vos, "VOs after merge:");
663}

732void MergePointInterpreterFrameState::MergeLoop(
733 MaglevGraphBuilder* builder, MaglevCompilationUnit& compilation_unit,
734 InterpreterFrameState& loop_end_state, BasicBlock* loop_end_block) {
735 // This should be the last predecessor we try to merge.
736 DCHECK_EQ(predecessors_so_far_, predecessor_count_ - 1);
737 DCHECK(is_unmerged_loop());
738 predecessors_[predecessor_count_ - 1] = loop_end_block;
739
740 backedge_deopt_frame_ =
741 builder->zone()->New<DeoptFrame>(builder->GetLatestCheckpointedFrame());
742
743 if (v8_flags.trace_maglev_graph_building) {
744 std::cout << "Merging loop backedge..." << std::endl;
745 }
746 frame_state_.ForEachValue(
747 compilation_unit, [&](ValueNode* value, interpreter::Register reg) {
748 PrintBeforeMerge(compilation_unit, value, loop_end_state.get(reg), reg,
749 known_node_aspects_);
750 MergeLoopValue(builder, reg, *loop_end_state.known_node_aspects(),
751 value, loop_end_state.get(reg));
752 PrintAfterMerge(compilation_unit, value, known_node_aspects_);
753 });
754 predecessors_so_far_++;
755 DCHECK_EQ(predecessors_so_far_, predecessor_count_);
756
757 // We have to clear the LoopInfo (which is used to record more precise use
758 // hints for Phis) for 2 reasons:
759 //
760 // - Phi::RecordUseReprHint checks if a use is inside the loop defining the
761 // Phi by checking if the LoopInfo of the loop Phi "Contains" the current
762 // bytecode offset, but this will be wrong if the Phi is in a function that
763 // was inlined (because the LoopInfo contains the first and last bytecode
764 // offset of the loop **in its own function**).
765 //
766 // - LoopInfo is obtained from the {header_to_info_} member of
767 // BytecodeAnalysis, but the BytecodeAnalysis is a member of the
768 // MaglevGraphBuilder, and thus gets destructed when the MaglevGraphBuilder
769 // created for inlining is destructed. LoopInfo would then become a stale
770 // pointer.
771 ClearLoopInfo();
772}

774bool MergePointInterpreterFrameState::TryMergeLoop(
775 MaglevGraphBuilder* builder, InterpreterFrameState& loop_end_state,
776 const std::function<BasicBlock*()>& FinishBlock) {
777 // This should be the last predecessor we try to merge.
778 DCHECK_EQ(predecessors_so_far_, predecessor_count_ - 1);
779 DCHECK(is_unmerged_loop());
780
781 backedge_deopt_frame_ =
782 builder->zone()->New<DeoptFrame>(builder->GetLatestCheckpointedFrame());
783
784 auto& compilation_unit = *builder->compilation_unit();
785
786 DCHECK_NOT_NULL(known_node_aspects_);
787 DCHECK(v8_flags.maglev_optimistic_peeled_loops);
788
789 // TODO(olivf): This could be done faster by consulting loop_effects_
790 if (!loop_end_state.known_node_aspects()->IsCompatibleWithLoopHeader(
791 *known_node_aspects_)) {
792 if (v8_flags.trace_maglev_graph_building) {
793 std::cout << "Merging failed, peeling loop instead... " << std::endl;
794 }
795 ClearLoopInfo();
796 return false;
797 }
798
799 bool phis_can_merge = true;
800 frame_state_.ForEachValue(compilation_unit, [&](ValueNode* value,
801 interpreter::Register reg) {
802 if (!value->Is<Phi>()) return;
803 Phi* phi = value->Cast<Phi>();
804 if (!phi->is_loop_phi()) return;
805 if (phi->merge_state() != this) return;
806 NodeType old_type = GetNodeType(builder->broker(), builder->local_isolate(),
807 *known_node_aspects_, phi);
808 if (old_type != NodeType::kUnknown) {
809 NodeType new_type = GetNodeType(
810 builder->broker(), builder->local_isolate(),
811 *loop_end_state.known_node_aspects(), loop_end_state.get(reg));
812 if (!NodeTypeIs(new_type, old_type)) {
813 if (v8_flags.trace_maglev_loop_speeling) {
814 std::cout << "Cannot merge " << new_type << " into " << old_type
815 << " for r" << reg.index() << "\n";
816 }
817 phis_can_merge = false;
818 }
819 }
820 });
821 if (!phis_can_merge) {
822 ClearLoopInfo();
823 return false;
824 }
825
826 BasicBlock* loop_end_block = FinishBlock();
827 int input = predecessor_count_ - 1;
828 loop_end_block->set_predecessor_id(input);
829 predecessors_[input] = loop_end_block;
830
831 if (v8_flags.trace_maglev_graph_building) {
832 std::cout << "Next peeling not needed due to compatible state" << std::endl;
833 }
834
835 frame_state_.ForEachValue(
836 compilation_unit, [&](ValueNode* value, interpreter::Register reg) {
837 PrintBeforeMerge(compilation_unit, value, loop_end_state.get(reg), reg,
838 known_node_aspects_);
839 MergeLoopValue(builder, reg, *loop_end_state.known_node_aspects(),
840 value, loop_end_state.get(reg));
841 PrintAfterMerge(compilation_unit, value, known_node_aspects_);
842 });
843 predecessors_so_far_++;
844 DCHECK_EQ(predecessors_so_far_, predecessor_count_);
845 ClearLoopInfo();
846 return true;
847}

862void MergePointInterpreterFrameState::MergeThrow(
863 MaglevGraphBuilder* builder, const MaglevCompilationUnit* handler_unit,
864 const KnownNodeAspects& known_node_aspects,
865 const VirtualObject::List virtual_objects) {
866 // We don't count total predecessors on exception handlers, but we do want to
867 // special case the first predecessor so we do count predecessors_so_far
868 DCHECK_EQ(predecessor_count_, 0);
869 DCHECK(is_exception_handler());
870
871 DCHECK_EQ(builder->compilation_unit(), handler_unit);
872
873 const InterpreterFrameState& builder_frame =
874 builder->current_interpreter_frame();
875
876 if (v8_flags.trace_maglev_graph_building) {
877 std::cout << "- Merging into exception handler @" << this << std::endl;
878 PrintVirtualObjects(*handler_unit, virtual_objects);
879 }
880
881 if (known_node_aspects_ == nullptr) {
882 DCHECK_EQ(predecessors_so_far_, 0);
883 known_node_aspects_ = known_node_aspects.Clone(builder->zone());
884 virtual_objects.Snapshot();
885 frame_state_.set_virtual_objects(virtual_objects);
886 } else {
887 known_node_aspects_->Merge(known_node_aspects, builder->zone());
888 MergeVirtualObjects(builder, *builder->compilation_unit(), virtual_objects,
889 known_node_aspects);
890 }
891
892 frame_state_.ForEachParameter(
893 *handler_unit, [&](ValueNode*& value, interpreter::Register reg) {
894 PrintBeforeMerge(*handler_unit, value, builder_frame.get(reg), reg,
895 known_node_aspects_);
896 value = MergeValue(builder, reg, known_node_aspects, value,
897 builder_frame.get(reg), nullptr);
898 PrintAfterMerge(*handler_unit, value, known_node_aspects_);
899 });
900 frame_state_.ForEachLocal(
901 *handler_unit, [&](ValueNode*& value, interpreter::Register reg) {
902 PrintBeforeMerge(*handler_unit, value, builder_frame.get(reg), reg,
903 known_node_aspects_);
904 value = MergeValue(builder, reg, known_node_aspects, value,
905 builder_frame.get(reg), nullptr);
906 PrintAfterMerge(*handler_unit, value, known_node_aspects_);
907 });
908
909 // Pick out the context value from the incoming registers.
910 // TODO(leszeks): This should be the same for all incoming states, but we lose
911 // the identity for generator-restored context. If generator value restores
912 // were handled differently, we could avoid emitting a Phi here.
913 ValueNode*& context = frame_state_.context(*handler_unit);
914 PrintBeforeMerge(*handler_unit, context,
915 builder_frame.get(catch_block_context_register_),
916 catch_block_context_register_, known_node_aspects_);
917 context = MergeValue(
918 builder, catch_block_context_register_, known_node_aspects, context,
919 builder_frame.get(catch_block_context_register_), nullptr);
920 PrintAfterMerge(*handler_unit, context, known_node_aspects_);
921
922 predecessors_so_far_++;
923}

1056
1057ValueNode* MergePointInterpreterFrameState::MergeValue(
1058 const MaglevGraphBuilder* builder, interpreter::Register owner,
1059 const KnownNodeAspects& unmerged_aspects, ValueNode* merged,
1060 ValueNode* unmerged, Alternatives::List* per_predecessor_alternatives,
1061 bool optimistic_loop_phis) {
1062 // If the merged node is null, this is a pre-created loop header merge
1063 // frame will null values for anything that isn't a loop Phi.
1064 if (merged == nullptr) {
1065 DCHECK(is_exception_handler() || is_unmerged_loop());
1066 DCHECK_EQ(predecessors_so_far_, 0);
1067 // Initialise the alternatives list and cache the alternative
1068 // representations of the node.
1069 if (per_predecessor_alternatives) {
1070 new (per_predecessor_alternatives) Alternatives::List();
1071 per_predecessor_alternatives->Add(builder->zone()->New<Alternatives>(
1072 unmerged_aspects.TryGetInfoFor(unmerged)));
1073 } else {
1074 DCHECK(is_exception_handler());
1075 }
1076 return unmerged;
1077 }
1078
1079 auto UpdateLoopPhiType = [&](Phi* result, NodeType unmerged_type) {
1080 DCHECK(result->is_loop_phi());
1081 if (predecessors_so_far_ == 0) {
1082 // For loop Phis, `type` is always Unknown until the backedge has been
1083 // bound, so there is no point in updating it here.
1084 result->set_post_loop_type(unmerged_type);
1085 if (optimistic_loop_phis) {
1086 // In the case of optimistic loop headers we try to speculatively use
1087 // the type of the incomming argument as the phi type. We verify if that
1088 // happened to be true before allowing the loop to conclude in
1089 // `TryMergeLoop`. Some types which are known to cause issues are
1090 // generalized here.
1091 NodeType initial_optimistic_type =
1092 (unmerged_type == NodeType::kInternalizedString) ? NodeType::kString
1093 : unmerged_type;
1094 result->set_type(initial_optimistic_type);
1095 }
1096 } else {
1097 if (optimistic_loop_phis) {
1098 if (NodeInfo* node_info = known_node_aspects_->TryGetInfoFor(result)) {
1099 node_info->IntersectType(unmerged_type);
1100 }
1101 result->merge_type(unmerged_type);
1102 }
1103 result->merge_post_loop_type(unmerged_type);
1104 }
1105 };
1106
1107 Phi* result = merged->TryCast<Phi>();
1108 if (result != nullptr && result->merge_state() == this) {
1109 // It's possible that merged == unmerged at this point since loop-phis are
1110 // not dropped if they are only assigned to themselves in the loop.
1111 DCHECK_EQ(result->owner(), owner);
1112 // Don't set inputs on exception phis.
1113 DCHECK_EQ(result->is_exception_phi(), is_exception_handler());
1114 if (is_exception_handler()) {
1115 // If an inlined allocation flows to an exception phi, we should consider
1116 // as an use.
1117 if (unmerged->Is<InlinedAllocation>()) {
1118 unmerged->add_use();
1119 }
1120 return result;
1121 }
1122
1123 NodeType unmerged_type =
1124 GetNodeType(builder->broker(), builder->local_isolate(),
1125 unmerged_aspects, unmerged);
1126 if (result->is_loop_phi()) {
1127 UpdateLoopPhiType(result, unmerged_type);
1128 } else {
1129 result->merge_type(unmerged_type);
1130 }
1131 unmerged = EnsureTagged(builder, unmerged_aspects, unmerged,
1132 predecessors_[predecessors_so_far_]);
1133 result->set_input(predecessors_so_far_, unmerged);
1134
1135 return result;
1136 }
1137
1138 if (merged == unmerged) {
1139 // Cache the alternative representations of the unmerged node.
1140 if (per_predecessor_alternatives) {
1141 DCHECK_EQ(per_predecessor_alternatives->LengthForTest(),
1142 predecessors_so_far_);
1143 per_predecessor_alternatives->Add(builder->zone()->New<Alternatives>(
1144 unmerged_aspects.TryGetInfoFor(unmerged)));
1145 } else {
1146 DCHECK(is_exception_handler());
1147 }
1148 return merged;
1149 }
1150
1151 // We should always statically know what the context is, so we should never
1152 // create Phis for it. The exception is resumable functions and OSR, where the
1153 // context should be statically known but we lose that static information
1154 // across the resume / OSR entry.
1155 DCHECK_IMPLIES(
1156 owner == interpreter::Register::current_context() ||
1157 (is_exception_handler() && owner == catch_block_context_register()),
1158 IsResumableFunction(builder->compilation_unit()
1159 ->info()
1160 ->toplevel_compilation_unit()
1161 ->shared_function_info()
1162 .kind()) ||
1163 builder->compilation_unit()->info()->toplevel_is_osr());
1164
1165 // Up to this point all predecessors had the same value for this interpreter
1166 // frame slot. Now that we find a distinct value, insert a copy of the first
1167 // value for each predecessor seen so far, in addition to the new value.
1168 // TODO(verwaest): Unclear whether we want this for Maglev: Instead of
1169 // letting the register allocator remove phis, we could always merge through
1170 // the frame slot. In that case we only need the inputs for representation
1171 // selection, and hence could remove duplicate inputs. We'd likely need to
1172 // attach the interpreter register to the phi in that case?
1173
1174 // For exception phis, just allocate exception handlers.
1175 if (is_exception_handler()) {
1176 // ... and add an use if inputs are inlined allocation.
1177 if (merged->Is<InlinedAllocation>()) {
1178 merged->add_use();
1179 }
1180 if (unmerged->Is<InlinedAllocation>()) {
1181 unmerged->add_use();
1182 }
1183 return NewExceptionPhi(builder->zone(), owner);
1184 }
1185
1186 result = Node::New<Phi>(builder->zone(), predecessor_count_, this, owner);
1187 if (v8_flags.trace_maglev_graph_building) {
1188 for (uint32_t i = 0; i < predecessor_count_; i++) {
1189 result->initialize_input_null(i);
1190 }
1191 }
1192
1193 NodeType merged_type =
1194 StaticTypeForNode(builder->broker(), builder->local_isolate(), merged);
1195
1196 bool is_tagged = merged->properties().value_representation() ==
1197 ValueRepresentation::kTagged;
1198 NodeType type = merged_type != NodeType::kUnknown
1199 ? merged_type
1200 : AlternativeType(per_predecessor_alternatives->first());
1201 int i = 0;
1202 for (const Alternatives* alt : *per_predecessor_alternatives) {
1203 ValueNode* tagged = is_tagged ? merged : alt->tagged_alternative();
1204 if (tagged == nullptr) {
1205 DCHECK_NOT_NULL(alt);
1206 tagged = NonTaggedToTagged(builder, alt->node_type(), merged,
1207 predecessors_[i]);
1208 }
1209 result->set_input(i, tagged);
1210 type = IntersectType(type, merged_type != NodeType::kUnknown
1211 ? merged_type
1212 : AlternativeType(alt));
1213 i++;
1214 }
1215 DCHECK_EQ(i, predecessors_so_far_);
1216
1217 // Note: it's better to call GetNodeType on {unmerged} before updating it with
1218 // EnsureTagged, since untagged nodes have a higher chance of having a
1219 // StaticType.
1220 NodeType unmerged_type = GetNodeType(
1221 builder->broker(), builder->local_isolate(), unmerged_aspects, unmerged);
1222 unmerged = EnsureTagged(builder, unmerged_aspects, unmerged,
1223 predecessors_[predecessors_so_far_]);
1224 result->set_input(predecessors_so_far_, unmerged);

1227 DCHECK(result->is_unmerged_loop_phi());
1228 UpdateLoopPhiType(result, type);
1229 } else {

1231 }

1237std::optional<ValueNode*>
1238MergePointInterpreterFrameState::MergeVirtualObjectValue(
1239 const MaglevGraphBuilder* builder, const KnownNodeAspects& unmerged_aspects,
1240 ValueNode* merged, ValueNode* unmerged) {
1241 DCHECK_NOT_NULL(merged);
1242 DCHECK_NOT_NULL(unmerged);
1243
1244 Phi* result = merged->TryCast<Phi>();
1245 if (result != nullptr && result->merge_state() == this) {
1246 NodeType unmerged_type =
1247 GetNodeType(builder->broker(), builder->local_isolate(),
1248 unmerged_aspects, unmerged);
1249 unmerged = EnsureTagged(builder, unmerged_aspects, unmerged,
1250 predecessors_[predecessors_so_far_]);
1251 for (uint32_t i = predecessors_so_far_; i < predecessor_count_; i++) {
1252 result->change_input(i, unmerged);
1253 }
1254 DCHECK_GT(predecessors_so_far_, 0);
1255 result->merge_type(unmerged_type);
1256 result->merge_post_loop_type(unmerged_type);
1257 return result;
1258 }
1259
1260 if (merged == unmerged) {
1261 return merged;
1262 }
1263
1264 if (InlinedAllocation* merged_nested_alloc =
1265 merged->TryCast<InlinedAllocation>()) {
1266 if (InlinedAllocation* unmerged_nested_alloc =
1267 unmerged->TryCast<InlinedAllocation>()) {
1268 // If a nested allocation doesn't point to the same object in both
1269 // objects, then we currently give up merging them and escape the
1270 // allocation.
1271 if (merged_nested_alloc != unmerged_nested_alloc) {
1272 return {};
1273 }
1274 }
1275 }
1276
1277 // We don't support exception phis inside a virtual object.
1278 if (is_exception_handler()) {
1279 return {};
1280 }
1281
1282 // We don't have LoopPhis inside a VirtualObject, but this can happen if the
1283 // block is a diamond-merge and a loop entry at the same time. For now, we
1284 // should escape.
1285 if (is_loop()) return {};
1286
1287 result = Node::New<Phi>(builder->zone(), predecessor_count_, this,
1288 interpreter::Register::invalid_value());
1289 if (v8_flags.trace_maglev_graph_building) {
1290 for (uint32_t i = 0; i < predecessor_count_; i++) {
1291 result->initialize_input_null(i);
1292 }
1293 }
1294
1295 NodeType merged_type =
1296 StaticTypeForNode(builder->broker(), builder->local_isolate(), merged);
1297
1298 // We must have seen the same value so far.
1299 DCHECK_NOT_NULL(known_node_aspects_);
1300 for (uint32_t i = 0; i < predecessors_so_far_; i++) {
1301 ValueNode* tagged_merged =
1302 EnsureTagged(builder, *known_node_aspects_, merged, predecessors_[i]);
1303 result->set_input(i, tagged_merged);
1304 }
1305
1306 NodeType unmerged_type = GetNodeType(
1307 builder->broker(), builder->local_isolate(), unmerged_aspects, unmerged);
1308 unmerged = EnsureTagged(builder, unmerged_aspects, unmerged,
1309 predecessors_[predecessors_so_far_]);
1310 for (uint32_t i = predecessors_so_far_; i < predecessor_count_; i++) {
1311 result->set_input(i, unmerged);
1312 }
1313
1314 result->set_type(IntersectType(merged_type, unmerged_type));
1315
1316 phis_.Add(result);
1317 return result;
1318}

1320void MergePointInterpreterFrameState::MergeLoopValue(
1321 MaglevGraphBuilder* builder, interpreter::Register owner,
1322 const KnownNodeAspects& unmerged_aspects, ValueNode* merged,
1323 ValueNode* unmerged) {
1324 Phi* result = merged->TryCast<Phi>();
1325 if (result == nullptr || result->merge_state() != this) {
1326 // Not a loop phi, we don't have to do anything.
1327 return;
1328 }
1329 DCHECK_EQ(result->owner(), owner);
1330 NodeType type = GetNodeType(builder->broker(), builder->local_isolate(),
1331 unmerged_aspects, unmerged);
1332 unmerged = EnsureTagged(builder, unmerged_aspects, unmerged,
1333 predecessors_[predecessors_so_far_]);
1334 result->set_input(predecessor_count_ - 1, unmerged);
1335
1336 result->merge_post_loop_type(type);
1337 // We've just merged the backedge, which means that future uses of this Phi
1338 // will be after the loop, so we can now promote `post_loop_type` to the
1339 // regular `type`.
1340 DCHECK_EQ(predecessors_so_far_, predecessor_count_ - 1);
1341 result->promote_post_loop_type();
1342
1343 if (Phi* unmerged_phi = unmerged->TryCast<Phi>()) {
1344 // Propagating the `uses_repr` from {result} to {unmerged_phi}.
1345 builder->RecordUseReprHint(unmerged_phi, result->get_uses_repr_hints());
1346
1347 // Soundness of the loop phi Smi type relies on the back-edge static types
1348 // sminess.
1349 if (result->uses_require_31_bit_value()) {
1350 unmerged_phi->SetUseRequires31BitValue();
1351 }
1352 }
1353}

1356 Zone* zone, interpreter::Register reg) {
1357 DCHECK_EQ(predecessors_so_far_, 0);
1358 // Create a new loop phi, which for now is empty.
1359 Phi* result = Node::New<Phi>(zone, predecessor_count_, this, reg);
1360
1361 if (v8_flags.trace_maglev_graph_building) {
1362 for (uint32_t i = 0; i < predecessor_count_; i++) {
1363 result->initialize_input_null(i);
1364 }
1365 }
1366 phis_.Add(result);
1367 return result;
1368}

1371 for (Phi* phi : phis_) {
1372 phi->reduce_input_count(num);
1373 if (predecessors_so_far_ == predecessor_count_ - 1 &&
1374 predecessor_count_ > 1 && phi->is_loop_phi()) {

1380bool MergePointInterpreterFrameState::IsUnreachable() const {
1381 DCHECK_EQ(predecessors_so_far_, predecessor_count_);
1382 if (predecessor_count_ > 1) {
1383 return false;
1384 }
1385 // This should actually only support predecessor_count == 1, but we
1386 // currently don't eliminate resumable loop headers (and subsequent code
1387 // until the next resume) that end up being unreachable from JumpLoop.
1388 if (predecessor_count_ == 0) {
1389 DCHECK(is_resumable_loop());
1390 return true;
1391 }
1392 DCHECK_EQ(predecessor_count_, 1);
1393 DCHECK_IMPLIES(is_loop(), predecessor_at(0)->control_node()->Is<JumpLoop>());
1394 return is_loop();
1395}
