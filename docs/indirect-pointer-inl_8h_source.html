18V8_INLINE void InitSelfIndirectPointerField(Address field_address,
19 IsolateForSandbox isolate,
20 Tagged<HeapObject> host,
21 IndirectPointerTag tag) {
22#ifdef V8_ENABLE_SANDBOX
23 DCHECK_NE(tag, kUnknownIndirectPointerTag);
24 // TODO(saelo): in the future, we might want to CHECK here or in
25 // AllocateAndInitializeEntry that the host lives in trusted space.
26
27 IndirectPointerHandle handle;
28 if (tag == kCodeIndirectPointerTag) {
29 CodePointerTable::Space* space =
30 isolate.GetCodePointerTableSpaceFor(field_address);
31 handle =
32 IsolateGroup::current()
33 ->code_pointer_table()
34 ->AllocateAndInitializeEntry(space, host.address(), kNullAddress,
35 kDefaultCodeEntrypointTag);
36 } else {
37 TrustedPointerTable::Space* space =
38 isolate.GetTrustedPointerTableSpaceFor(tag);
39 handle = isolate.GetTrustedPointerTableFor(tag).AllocateAndInitializeEntry(
40 space, host.ptr(), tag);
41 }
42
43 // Use a Release_Store to ensure that the store of the pointer into the table
44 // is not reordered after the store of the handle. Otherwise, other threads
45 // may access an uninitialized table entry and crash.
46 auto location = reinterpret_cast<IndirectPointerHandle*>(field_address);
47 base::AsAtomic32::Release_Store(location, handle);
48#else
49 UNREACHABLE();
50#endif
51}

70template <IndirectPointerTag tag>
71V8_INLINE Tagged<Object> ReadIndirectPointerField(Address field_address,
72 IsolateForSandbox isolate,
73 AcquireLoadTag) {
74#ifdef V8_ENABLE_SANDBOX
75 // Load the indirect pointer handle from the object.
76 // Technically, we could use memory_order_consume here as the loads are
77 // dependent, but that appears to be deprecated in favor of acquire ordering.
78 auto location = reinterpret_cast<IndirectPointerHandle*>(field_address);
79 IndirectPointerHandle handle = base::AsAtomic32::Acquire_Load(location);
80
81 // Resolve the handle. The tag implies the pointer table to use.
82 if constexpr (tag == kUnknownIndirectPointerTag) {
83 // In this case we need to check if the handle is a code pointer handle and
84 // select the appropriate table based on that.
85 if (handle & kCodePointerHandleMarker) {
86 return ResolveCodePointerHandle(handle);
87 } else {
88 // TODO(saelo): once we have type tagging for entries in the trusted
89 // pointer table, we could ASSUME that the top bits of the tag match the
90 // instance type, which might allow the compiler to optimize subsequent
91 // instance type checks.
92 return ResolveTrustedPointerHandle<tag>(handle, isolate);
93 }
94 } else if constexpr (tag == kCodeIndirectPointerTag) {
95 return ResolveCodePointerHandle(handle);
96 } else {
97 return ResolveTrustedPointerHandle<tag>(handle, isolate);
98 }
99#else
100 UNREACHABLE();
101#endif
102}
