19// StringEscapeAnalysisReducer tries to remove string concatenations whose
20// results are unused, or used only in FrameStates or in other string concations
21// that are themselves unused.
22//
23// The analysis (StringEscapeAnalyzer::Run) is pretty simple: we iterate the
24// graph backwards and mark all inputs of all operations as "escaping", except
25// for StringLength and FrameState which don't mark their input as escaping, and
26// for StringConcat, which only marks its inputs as escaping if it is itself
27// marked as escaping.
28
29#include "src/compiler/turboshaft/define-assembler-macros.inc"
30
31class StringEscapeAnalyzer {
32 public:
33 StringEscapeAnalyzer(const Graph& graph, Zone* phase_zone)
34 : graph_(graph),
35 zone_(phase_zone),
36 escaping_operations_and_frame_states_to_reconstruct_(
37 graph.op_id_count(), false, zone_, &graph),
38 maybe_non_escaping_string_concats_(phase_zone),
39 maybe_to_reconstruct_frame_states_(phase_zone) {}
40 void Run();
41
42 bool IsEscaping(OpIndex idx) const {
43 DCHECK(!graph_.Get(idx).Is<FrameStateOp>());
44 return escaping_operations_and_frame_states_to_reconstruct_[idx];
45 }
46
47 bool ShouldReconstructFrameState(V<FrameState> idx) {
48 return escaping_operations_and_frame_states_to_reconstruct_[idx];
49 }

51 private:
52 const Graph& graph_;
53 Zone* zone_;
54
55 void ProcessBlock(const Block& block);
56 void MarkAllInputsAsEscaping(const Operation& op);
57 void RecursivelyMarkAllStringConcatInputsAsEscaping(
58 const StringConcatOp* concat);

62 void MarkAsEscaping(OpIndex index) {

65 }
66
67 void RecursiveMarkAsShouldReconstruct(V<FrameState> idx) {
68 escaping_operations_and_frame_states_to_reconstruct_[idx] = true;
69 const FrameStateOp* frame_state = &graph_.Get(idx).Cast<FrameStateOp>();
70 while (frame_state->inlined) {
71 V<FrameState> parent = frame_state->parent_frame_state();
72 escaping_operations_and_frame_states_to_reconstruct_[parent] = true;
73 frame_state = &graph_.Get(parent).Cast<FrameStateOp>();
74 }
75 }
76
77 // {escaping_operations_and_frame_states_to_recreate_} is used for 2 things:
78 // - For FrameState OpIndex, if the stored value is `true`, then the reducer
79 // should later reconstruct this FrameState (because it either contains an
80 // elided StringConcat, or because it's the parent of such a FrameState).
81 // - For other OpIndex, if the value stored is `true`, then the value is
82 // definitely escaping, and should not be elided (and its inputs shouldn't
83 // be elided either, etc.).
84 // This could easily be split in 2 variables, but it saves space to use a
85 // single variable for this.
86 FixedOpIndexSidetable<bool>
87 escaping_operations_and_frame_states_to_reconstruct_;
88
89 // When we visit a StringConcat for the first time and it's not already in
90 // {escaping_operations_}, we can't know for sure yet that it will never be
91 // escaping, because of loop phis. So, we store it in
92 // {maybe_non_escaping_string_concats_}, which we revisit after having visited
93 // the whole graph, and only after this revisit do we know for sure that
94 // StringConcat that are not in {escaping_operations_} do not indeed escape.
95 ZoneVector<V<String>> maybe_non_escaping_string_concats_;
96
97 ZoneVector<V<FrameState>> maybe_to_reconstruct_frame_states_;
98
99 uint32_t max_frame_state_input_count_ = 0;
100};

102template <class Next>
103class StringEscapeAnalysisReducer : public Next {

105 TURBOSHAFT_REDUCER_BOILERPLATE(StringEscapeAnalysis)

107 // ElidedStringPart is an input of a StringConcat that is getting elided. It
108 // could be either a regular String that appears in the output graph
109 // (kNotElided), or another StringConcat that got elided as well (kElided).
110 struct ElidedStringPart {
111 enum class Kind : uint8_t { kNotElided, kElided };
112 union {
113 V<String> og_index;
114 V<String> ig_index;
115 } data;
116
117 Kind kind;
118
119 static ElidedStringPart Elided(V<String> ig_index) {
120 return ElidedStringPart(Kind::kElided, ig_index);
121 }
122 static ElidedStringPart NotElided(V<String> og_index) {
123 return ElidedStringPart(Kind::kNotElided, og_index);
124 }
125
126 bool is_elided() const { return kind == Kind::kElided; }
127
128 V<String> og_index() const {
129 DCHECK_EQ(kind, Kind::kNotElided);
130 return data.og_index;
131 }
132 V<String> ig_index() const {
133 DCHECK_EQ(kind, Kind::kElided);
134 return data.ig_index;
135 }
136
137 bool operator==(const ElidedStringPart& other) const {
138 if (kind != other.kind) return false;
139 switch (kind) {
140 case Kind::kElided:
141 return ig_index() == other.ig_index();
142 case Kind::kNotElided:
143 return og_index() == other.og_index();
144 }
145 }
146
147 static ElidedStringPart Invalid() {
148 return ElidedStringPart(Kind::kNotElided, V<String>::Invalid());
149 }
150
151 private:
152 ElidedStringPart(Kind kind, V<String> index) : data(index), kind(kind) {}
153 };

162 V<String> REDUCE_INPUT_GRAPH(StringConcat)(V<String> ig_index,
163 const StringConcatOp& op) {
164 LABEL_BLOCK(no_change) {
165 return Next::ReduceInputGraphStringConcat(ig_index, op);
166 }
167 if (!v8_flags.turboshaft_string_concat_escape_analysis) goto no_change;
168 if (analyzer_.IsEscaping(ig_index)) goto no_change;
169
170 // We're eliding this StringConcat.
171 ElidedStringPart left = GetElidedStringInput(op.left());
172 ElidedStringPart right = GetElidedStringInput(op.right());
173 elided_strings_.insert({ig_index, std::pair{left, right}});
174 return V<String>::Invalid();
175 }
176
177 V<FrameState> REDUCE_INPUT_GRAPH(FrameState)(
178 V<FrameState> ig_index, const FrameStateOp& frame_state) {
179 LABEL_BLOCK(no_change) {
180 return Next::ReduceInputGraphFrameState(ig_index, frame_state);
181 }
182 if (!v8_flags.turboshaft_string_concat_escape_analysis) goto no_change;
183
184 if (!analyzer_.ShouldReconstructFrameState(ig_index)) goto no_change;
185
186 return BuildFrameState(frame_state, ig_index);
187 }
188
189 V<Word32> REDUCE_INPUT_GRAPH(StringLength)(V<Word32> ig_index,
190 const StringLengthOp& op) {
191 LABEL_BLOCK(no_change) {
192 return Next::ReduceInputGraphStringLength(ig_index, op);
193 }
194 if (!v8_flags.turboshaft_string_concat_escape_analysis) goto no_change;
195
196 V<String> input_index = op.string();
197 if (const StringConcatOp* input = __ input_graph()
198 .Get(input_index)
199 .template TryCast<StringConcatOp>();
200 input && !analyzer_.IsEscaping(input_index)) {
201 return __ UntagSmi(__ MapToNewGraph(input->length()));
202 } else {
203 goto no_change;
204 }
205 }

243 V<FrameState> BuildFrameState(const FrameStateOp& input_frame_state,
244 OpIndex ig_index) {
245 DCHECK(v8_flags.turboshaft_string_concat_escape_analysis);
246
247 const FrameStateInfo& info = input_frame_state.data->frame_state_info;
248
249 FrameStateData::Builder builder;
250 auto it =
251 input_frame_state.data->iterator(input_frame_state.state_values());
252
253 Deduplicator* deduplicator;
254 if (input_frame_state.inlined) {
255 V<FrameState> parent_ig_index = input_frame_state.parent_frame_state();
256 builder.AddParentFrameState(__ MapToNewGraph(parent_ig_index));
257
258 // The parent FrameState could contain dematerialized objects, and the
259 // current FrameState could reference those. Also, new IDs created for the
260 // current FrameState should not conflict with IDs in the parent frame
261 // state. Thus, we need to initialize the current Deduplicator with the
262 // one from the parent FrameState.
263 DCHECK(analyzer_.ShouldReconstructFrameState(parent_ig_index));
264 deduplicator = deduplicators_[parent_ig_index]->clone(__ phase_zone());
265 } else {
266 deduplicator =
267 __ phase_zone() -> template New<Deduplicator>(__ phase_zone());
268 }
269 deduplicators_[ig_index] = deduplicator;
270
271 // Closure
272 BuildFrameStateInput(&builder, &it, deduplicator);
273
274 // Parameters
275 for (int i = 0; i < info.parameter_count(); i++) {
276 BuildFrameStateInput(&builder, &it, deduplicator);
277 }
278
279 // Context
280 BuildFrameStateInput(&builder, &it, deduplicator);
281
282 // Registers/locals
283 for (int i = 0; i < info.local_count(); i++) {
284 BuildFrameStateInput(&builder, &it, deduplicator);
285 }
286
287 // Accumulator
288 for (int i = 0; i < info.stack_count(); i++) {
289 BuildFrameStateInput(&builder, &it, deduplicator);
290 }
291
292 return __ FrameState(builder.Inputs(), builder.inlined(),
293 builder.AllocateFrameStateData(info, __ graph_zone()));
294 }
295
296 void BuildFrameStateInput(FrameStateData::Builder* builder,
297 FrameStateData::Iterator* it,
298 Deduplicator* deduplicator) {
299 switch (it->current_instr()) {
300 using Instr = FrameStateData::Instr;
301 case Instr::kInput: {
302 MachineType type;
303 OpIndex input;
304 it->ConsumeInput(&type, &input);
305 if (elided_strings_.contains(input)) {
306 DCHECK(type.IsTagged());
307 BuildMaybeElidedString(builder, ElidedStringPart::Elided(input),
308 deduplicator);
309 } else {
310 builder->AddInput(type, __ MapToNewGraph(input));
311 }
312 break;
313 }
314 case Instr::kDematerializedObject: {
315 uint32_t old_id;
316 uint32_t field_count;
317 it->ConsumeDematerializedObject(&old_id, &field_count);
318 builder->AddDematerializedObject(old_id, field_count);
319 for (uint32_t i = 0; i < field_count; ++i) {
320 BuildFrameStateInput(builder, it, deduplicator);
321 }
322 break;
323 }
324 case Instr::kDematerializedObjectReference: {
325 uint32_t old_id;
326 it->ConsumeDematerializedObjectReference(&old_id);
327 builder->AddDematerializedObjectReference(old_id);
328 break;
329 }
330 case Instr::kArgumentsElements: {
331 CreateArgumentsType type;
332 it->ConsumeArgumentsElements(&type);
333 builder->AddArgumentsElements(type);
334 break;
335 }
336 case Instr::kArgumentsLength:
337 it->ConsumeArgumentsLength();
338 builder->AddArgumentsLength();
339 break;
340 case Instr::kRestLength:
341 it->ConsumeRestLength();
342 builder->AddRestLength();
343 break;
344 case Instr::kUnusedRegister:
345 it->ConsumeUnusedRegister();
346 builder->AddUnusedRegister();
347 break;
348 case FrameStateData::Instr::kDematerializedStringConcat:
349 case FrameStateData::Instr::kDematerializedStringConcatReference:
350 // StringConcat should not have been escaped before this point.
351 UNREACHABLE();
352 }
353 }
354
355 void BuildMaybeElidedString(FrameStateData::Builder* builder,
356 ElidedStringPart maybe_elided,
357 Deduplicator* deduplicator) {
358 if (maybe_elided.is_elided()) {
359 typename Deduplicator::DuplicatedId dup_id =
360 deduplicator->GetDuplicatedIdForElidedString(maybe_elided);
361 if (dup_id.duplicated) {
362 // For performance reasons, we de-duplicate repeated StringConcat inputs
363 // in the FrameState. Unlike for elided objects, deduplication has no
364 // impact on correctness.
365 builder->AddDematerializedStringConcatReference(dup_id.id);
366 return;
367 }
368 builder->AddDematerializedStringConcat(dup_id.id);
369 std::pair<ElidedStringPart, ElidedStringPart> inputs =
370 elided_strings_.at(maybe_elided.ig_index());
371 BuildMaybeElidedString(builder, inputs.first, deduplicator);
372 BuildMaybeElidedString(builder, inputs.second, deduplicator);
373 } else {
374 builder->AddInput(MachineType::AnyTagged(), maybe_elided.og_index());
375 }
376 }
377
378 ElidedStringPart GetElidedStringInput(V<String> ig_index) {
379 if (elided_strings_.contains(ig_index)) {
380 return ElidedStringPart::Elided(ig_index);
381 } else {
382 return ElidedStringPart::NotElided(__ MapToNewGraph(ig_index));
383 }
384 }
385
386 StringEscapeAnalyzer analyzer_{Asm().input_graph(), Asm().phase_zone()};
387 // Map from input OpIndex of elided strings to the pair of output OpIndex
388 // that are their left and right sides of the concatenation.
389 ZoneAbslFlatHashMap<V<String>, std::pair<ElidedStringPart, ElidedStringPart>>
390 elided_strings_{Asm().phase_zone()};
391
392 // Mapping from input-graph FrameState to the corresponding deduplicator.
393 SparseOpIndexSideTable<Deduplicator*> deduplicators_{Asm().phase_zone(),
394 &Asm().input_graph()};
395};
