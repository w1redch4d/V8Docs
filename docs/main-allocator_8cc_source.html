140// Perform an allocation step when the step is reached. size_in_bytes is the
141// actual size needed for the object (required for InvokeAllocationObservers).
142// aligned_size_in_bytes is the size of the object including the filler right
143// before it to reach the right alignment (required to DCHECK the start of the
144// object). allocation_size is the size of the actual allocation which needs to
145// be used for the accounting. It can be different from aligned_size_in_bytes in
146// PagedSpace::AllocateRawAligned, where we have to overallocate in order to be
147// able to align the allocation afterwards.
148void MainAllocator::InvokeAllocationObservers(Address soon_object,
149 size_t size_in_bytes,
150 size_t aligned_size_in_bytes,
151 size_t allocation_size) {
152 DCHECK_LE(size_in_bytes, aligned_size_in_bytes);
153 DCHECK_LE(aligned_size_in_bytes, allocation_size);
154 DCHECK(size_in_bytes == aligned_size_in_bytes ||
155 aligned_size_in_bytes == allocation_size);
156
157 if (!SupportsAllocationObserver() ||
158 !isolate_heap()->IsAllocationObserverActive()) {
159 return;
160 }
161
162 if (allocation_size >= allocation_counter().NextBytes()) {
163 // Only the first object in a LAB should reach the next step.
164 DCHECK_EQ(soon_object, allocation_info().start() + aligned_size_in_bytes -
165 size_in_bytes);
166
167 // Right now the LAB only contains that one object.
168 DCHECK_EQ(allocation_info().top() + allocation_size - aligned_size_in_bytes,
169 allocation_info().limit());
170
171 // Ensure that there is a valid object
172 space_heap()->CreateFillerObjectAt(soon_object,
173 static_cast<int>(size_in_bytes));
174
175#if DEBUG
176 // Ensure that allocation_info_ isn't modified during one of the
177 // AllocationObserver::Step methods.
178 LinearAllocationArea saved_allocation_info = allocation_info();
179#endif
180
181 // Run AllocationObserver::Step through the AllocationCounter.
182 allocation_counter().InvokeAllocationObservers(soon_object, size_in_bytes,
183 allocation_size);
184
185 // Ensure that start/top/limit didn't change.
186 DCHECK_EQ(saved_allocation_info.start(), allocation_info().start());
187 DCHECK_EQ(saved_allocation_info.top(), allocation_info().top());
188 DCHECK_EQ(saved_allocation_info.limit(), allocation_info().limit());
189 }
190
191 DCHECK_LT(allocation_info().limit() - allocation_info().start(),
192 allocation_counter().NextBytes());
193}

196 AllocationAlignment alignment,
197 AllocationOrigin origin) {

210AllocationResult MainAllocator::AllocateRawSlowUnaligned(
211 int size_in_bytes, AllocationOrigin origin) {
212 if (!EnsureAllocation(size_in_bytes, kTaggedAligned, origin)) {
213 return AllocationResult::Failure();
214 }
215
216 AllocationResult result = AllocateFastUnaligned(size_in_bytes, origin);
217 DCHECK(!result.IsFailure());
218
219 InvokeAllocationObservers(result.ToAddress(), size_in_bytes, size_in_bytes,
220 size_in_bytes);
221
222 return result;
223}

225AllocationResult MainAllocator::AllocateRawSlowAligned(
226 int size_in_bytes, AllocationAlignment alignment, AllocationOrigin origin) {
227 if (!EnsureAllocation(size_in_bytes, alignment, origin)) {
228 return AllocationResult::Failure();
229 }
230
231 int max_aligned_size = size_in_bytes + Heap::GetMaximumFillToAlign(alignment);
232 int aligned_size_in_bytes;
233
234 AllocationResult result = AllocateFastAligned(
235 size_in_bytes, &aligned_size_in_bytes, alignment, origin);
236 DCHECK_GE(max_aligned_size, aligned_size_in_bytes);
237 DCHECK(!result.IsFailure());
238
239 InvokeAllocationObservers(result.ToAddress(), size_in_bytes,
240 aligned_size_in_bytes, max_aligned_size);
241
242 return result;
243}

356Address MainAllocator::ComputeLimit(Address start, Address end,
357 size_t min_size) const {
358 DCHECK_GE(end - start, min_size);
359
360 // Use the full LAB when allocation observers aren't enabled.
361 if (!SupportsAllocationObserver()) return end;
362
363 // LABs with allocation observers are only used outside GC and on the main
364 // thread.
365 DCHECK(!isolate_heap()->IsInGC());
366 DCHECK(is_main_thread());
367
368 if (!isolate_heap()->IsInlineAllocationEnabled()) {
369 // LABs are disabled, so we fit the requested area exactly.
370 return start + min_size;
371 }
372
373 // When LABs are enabled, pick the largest possible LAB size by default.
374 size_t step_size = end - start;
375
376 if (isolate_heap()->IsAllocationObserverActive()) {
377 // Ensure there are no unaccounted allocations.
378 DCHECK_EQ(allocation_info().start(), allocation_info().top());
379
380 size_t step = allocation_counter().NextBytes();
381 DCHECK_NE(step, 0);
382 // Generated code may allocate inline from the linear allocation area. To
383 // make sure we can observe these allocations, we use a lower limit.
384 size_t rounded_step = static_cast<size_t>(
385 RoundDown(static_cast<int>(step - 1), ObjectAlignment()));
386 step_size = std::min(step_size, rounded_step);
387 }
388
389 if (v8_flags.stress_marking) {
390 step_size = std::min(step_size, static_cast<size_t>(64));
391 }
392
393 DCHECK_LE(start + step_size, end);
394 return start + std::max(step_size, min_size);
395}

458bool SemiSpaceNewSpaceAllocatorPolicy::EnsureAllocation(
459 int size_in_bytes, AllocationAlignment alignment, AllocationOrigin origin) {
460 std::optional<base::SpinningMutexGuard> guard;
461 if (allocator_->in_gc()) guard.emplace(space_->mutex());
462
463 FreeLinearAllocationAreaUnsynchronized();
464
465 std::optional<std::pair<Address, Address>> allocation_result =
466 space_->Allocate(size_in_bytes, alignment);
467 if (!allocation_result) {
468 if (!v8_flags.separate_gc_phases ||
469 !space_->heap()->ShouldExpandYoungGenerationOnSlowAllocation(
470 PageMetadata::kPageSize)) {
471 return false;
472 }
473 allocation_result =
474 space_->AllocateOnNewPageBeyondCapacity(size_in_bytes, alignment);
475 if (!allocation_result) return false;
476 }
477
478 Address start = allocation_result->first;
479 Address end = allocation_result->second;
480
481 int filler_size = Heap::GetFillToAlign(start, alignment);
482 int aligned_size_in_bytes = size_in_bytes + filler_size;
483 DCHECK_LE(start + aligned_size_in_bytes, end);
484
485 Address limit;
486
487 if (allocator_->in_gc()) {
488 // During GC we allow multiple LABs in new space and since Allocate() above
489 // returns the whole remaining page by default, we limit the size of the LAB
490 // here.
491 size_t used = std::max(aligned_size_in_bytes, kLabSizeInGC);
492 limit = std::min(end, start + used);
493 } else {
494 limit = allocator_->ComputeLimit(start, end, aligned_size_in_bytes);
495 }
496 CHECK_LE(limit, end);
497
498 if (limit != end) {
499 space_->Free(limit, end);
500 }
501
502 allocator_->ResetLab(start, limit, limit);
503
504 space_->to_space().AddRangeToActiveSystemPages(allocator_->top(),
505 allocator_->limit());
506 return true;
507}

543 int size_in_bytes, AllocationAlignment alignment, AllocationOrigin origin) {
544 if (space_->paged_space()->last_lab_page_) {
545 space_->paged_space()->last_lab_page_->DecreaseAllocatedLabSize(
546 allocator_->limit() - allocator_->top());

549 // reallocated if the lab can be extended or freed otherwise.

552 if (!paged_space_allocator_policy_->EnsureAllocation(size_in_bytes, alignment,

562 PageMetadata::FromAllocationAreaAddress(allocator_->top());
563 DCHECK_NOT_NULL(space_->paged_space()->last_lab_page_);
564 space_->paged_space()->last_lab_page_->IncreaseAllocatedLabSize(
565 allocator_->limit() - allocator_->top());
566
567 if (space_heap()->incremental_marking()->IsMinorMarking()) {
568 space_heap()->concurrent_marking()->RescheduleJobIfNeeded(
569 GarbageCollector::MINOR_MARK_SWEEPER);
570 }
571
572 return true;

575bool PagedNewSpaceAllocatorPolicy::WaitForSweepingForAllocation(
576 int size_in_bytes, AllocationOrigin origin) {
577 // This method should be called only when there are no more pages for main
578 // thread to sweep.
579 DCHECK(space_heap()->sweeper()->IsSweepingDoneForSpace(NEW_SPACE));
580 if (!v8_flags.concurrent_sweeping || !space_heap()->sweeping_in_progress())
581 return false;
582 Sweeper* sweeper = space_heap()->sweeper();
583 if (!sweeper->AreMinorSweeperTasksRunning() &&
584 !sweeper->ShouldRefillFreelistForSpace(NEW_SPACE)) {
585#if DEBUG
586 for (PageMetadata* p : *space_) {
587 DCHECK(p->SweepingDone());
588 p->ForAllFreeListCategories(
589 [space = space_->paged_space()](FreeListCategory* category) {
590 DCHECK_IMPLIES(!category->is_empty(),
591 category->is_linked(space->free_list()));
592 });
593 }
594#endif // DEBUG
595 // All pages are already swept and relinked to the free list
596 return false;
597 }
598 // When getting here we know that any unswept new space page is currently
599 // being handled by a concurrent sweeping thread. Rather than try to cancel
600 // tasks and restart them, we wait "per page". This should be faster.
601 for (PageMetadata* p : *space_) {
602 if (!p->SweepingDone()) sweeper->WaitForPageToBeSwept(p);
603 }
604 space_->paged_space()->RefillFreeList();
605 DCHECK(!sweeper->ShouldRefillFreelistForSpace(NEW_SPACE));
606 return paged_space_allocator_policy_->TryAllocationFromFreeList(
607 static_cast<size_t>(size_in_bytes), origin);
608}

641bool PagedSpaceAllocatorPolicy::EnsureAllocation(int size_in_bytes,
642 AllocationAlignment alignment,
643 AllocationOrigin origin) {
644 if (allocator_->identity() == NEW_SPACE) {
645 DCHECK(allocator_->is_main_thread());
646 space_heap()->StartMinorMSIncrementalMarkingIfNeeded();
647 }
648 if ((allocator_->identity() != NEW_SPACE) && !allocator_->in_gc()) {
649 // Start incremental marking before the actual allocation, this allows the
650 // allocation function to mark the object black when incremental marking is
651 // running.
652 space_heap()->StartIncrementalMarkingIfAllocationLimitIsReached(
653 allocator_->local_heap(), space_heap()->GCFlagsForIncrementalMarking(),
654 kGCCallbackScheduleIdleGarbageCollection);
655 }
656
657 // We don't know exactly how much filler we need to align until space is
658 // allocated, so assume the worst case.
659 size_in_bytes += Heap::GetMaximumFillToAlign(alignment);
660 if (allocator_->allocation_info().top() + size_in_bytes <=
661 allocator_->allocation_info().limit()) {
662 return true;

667bool PagedSpaceAllocatorPolicy::RefillLab(int size_in_bytes,
668 AllocationOrigin origin) {
669 // Allocation in this space has failed.
670 DCHECK_GE(size_in_bytes, 0);
671
672 if (TryExtendLAB(size_in_bytes)) return true;
673
674 if (TryAllocationFromFreeList(size_in_bytes, origin)) return true;
675
676 // Don't steal pages from the shared space of the main isolate if running as a
677 // client. The issue is that the concurrent marker may be running on the main
678 // isolate and may reach the page and read its flags, which will then end up
679 // in a race, when the page of the compaction space will be merged back to the
680 // main space. For the same reason, don't take swept pages from the main
681 // shared space.
682 const bool running_from_client_isolate_and_allocating_in_shared_space =
683 (allocator_->identity() == SHARED_SPACE) &&
684 !isolate_heap()->isolate()->is_shared_space_isolate();
685 if (running_from_client_isolate_and_allocating_in_shared_space) {
686 // Avoid OOM crash in the GC in order to invoke NearHeapLimitCallback after
687 // GC and give it a chance to increase the heap limit.
688 if (!isolate_heap()->force_oom() &&
689 TryExpandAndAllocate(size_in_bytes, origin)) {
690 return true;
691 }
692 return false;
693 }
694
695 // Sweeping is still in progress. The sweeper doesn't work with black
696 // allocated pages, so it's fine for the compaction space to refill the
697 // freelist from just swept pages.
698 if (space_heap()->sweeping_in_progress()) {
699 // First try to refill the free-list, concurrent sweeper threads
700 // may have freed some objects in the meantime.
701 if (space_heap()->sweeper()->ShouldRefillFreelistForSpace(
702 allocator_->identity())) {
703 space_->RefillFreeList();
704
705 // Retry the free list allocation.
706 if (TryAllocationFromFreeList(static_cast<size_t>(size_in_bytes), origin))
707 return true;
708 }
709
710 static constexpr int kMaxPagesToSweep = 1;
711 if (ContributeToSweeping(kMaxPagesToSweep)) {
712 if (TryAllocationFromFreeList(size_in_bytes, origin)) {
713 return true;
714 }
715 }
716 }
717
718 // If there is not enough memory in the compaction space left, try to steal
719 // a page from the corresponding "regular" page space.
720 // Don't do this though when black allocated pages are enabled and incremental
721 // marking is in progress, because otherwise evacuating into a black allocated
722 // page will cause the marker to miss the object.
723 const bool incremental_marking_with_black_allocated_pages_is_running =
724 v8_flags.black_allocated_pages &&
725 space_heap()->incremental_marking()->IsMajorMarking();
726 if (!incremental_marking_with_black_allocated_pages_is_running &&
727 space_->is_compaction_space()) {
728 DCHECK_NE(NEW_SPACE, allocator_->identity());
729 PagedSpaceBase* main_space =
730 space_heap()->paged_space(allocator_->identity());
731 PageMetadata* page = main_space->RemovePageSafe(size_in_bytes);
732 if (page != nullptr) {
733 // Make sure we don't evacuate into a black allocated page.
734 DCHECK_IMPLIES(v8_flags.black_allocated_pages,
735 !page->Chunk()->IsFlagSet(MemoryChunk::BLACK_ALLOCATED));
736 space_->AddPage(page);
737 if (TryAllocationFromFreeList(static_cast<size_t>(size_in_bytes), origin))
738 return true;
739 }
740 }
741
742 if (allocator_->identity() != NEW_SPACE &&
743 space_heap()->ShouldExpandOldGenerationOnSlowAllocation(
744 allocator_->local_heap(), origin) &&
745 space_heap()->CanExpandOldGeneration(space_->AreaSize())) {
746 if (TryExpandAndAllocate(static_cast<size_t>(size_in_bytes), origin)) {
747 return true;
748 }
749 }
750
751 // Try sweeping all pages.
752 if (ContributeToSweeping()) {
753 if (TryAllocationFromFreeList(size_in_bytes, origin)) {
754 return true;
755 }
756 }
757
758 if (allocator_->identity() != NEW_SPACE && allocator_->in_gc() &&
759 !space_heap()->force_oom()) {
760 // Avoid OOM crash in the GC in order to invoke NearHeapLimitCallback after
761 // GC and give it a chance to increase the heap limit.
762 if (TryExpandAndAllocate(size_in_bytes, origin)) {
763 return true;
764 }
765 }
766 return false;
767}

769bool PagedSpaceAllocatorPolicy::TryExpandAndAllocate(size_t size_in_bytes,
770 AllocationOrigin origin) {
771 // Run in a loop because concurrent threads might allocate from the new free
772 // list entries before this thread gets a chance.
773 while (space_->TryExpand(allocator_->local_heap(), origin)) {
774 if (TryAllocationFromFreeList(static_cast<size_t>(size_in_bytes), origin)) {
775 return true;
776 }
777 }
778 return false;
779}

781bool PagedSpaceAllocatorPolicy::ContributeToSweeping(uint32_t max_pages) {
782 if (!space_heap()->sweeping_in_progress_for_space(allocator_->identity()))
783 return false;
784 if (space_heap()->sweeper()->IsSweepingDoneForSpace(allocator_->identity()))
785 return false;
786
787 const bool is_main_thread =
788 allocator_->is_main_thread() ||
789 (allocator_->in_gc() && isolate_heap()->IsMainThread());
790 const auto sweeping_scope_kind =
791 is_main_thread ? ThreadKind::kMain : ThreadKind::kBackground;
792 const auto sweeping_scope_id = space_heap()->sweeper()->GetTracingScope(
793 allocator_->identity(), is_main_thread);
794
795 TRACE_GC_EPOCH_WITH_FLOW(
796 isolate_heap()->tracer(), sweeping_scope_id, sweeping_scope_kind,
797 isolate_heap()->sweeper()->GetTraceIdForFlowEvent(sweeping_scope_id),
798 TRACE_EVENT_FLAG_FLOW_IN | TRACE_EVENT_FLAG_FLOW_OUT);
799 // Cleanup invalidated old-to-new refs for compaction space in the
800 // final atomic pause.
801 Sweeper::SweepingMode sweeping_mode =
802 allocator_->in_gc_for_space() ? Sweeper::SweepingMode::kEagerDuringGC
803 : Sweeper::SweepingMode::kLazyOrConcurrent;
804
805 if (!space_heap()->sweeper()->ParallelSweepSpace(allocator_->identity(),
806 sweeping_mode, max_pages)) {
807 return false;
808 }
809 space_->RefillFreeList();
810 return true;
811}

826bool PagedSpaceAllocatorPolicy::TryAllocationFromFreeList(
827 size_t size_in_bytes, AllocationOrigin origin) {
828 PagedSpace::ConcurrentAllocationMutex guard(space_);
829 DCHECK(IsAligned(size_in_bytes, kTaggedSize));
830 DCHECK_LE(allocator_->top(), allocator_->limit());
831#ifdef DEBUG
832 if (allocator_->top() != allocator_->limit()) {
833 DCHECK_EQ(PageMetadata::FromAddress(allocator_->top()),
834 PageMetadata::FromAddress(allocator_->limit() - 1));
835 }
836#endif
837 // Don't free list allocate if there is linear space available.
838 DCHECK_LT(static_cast<size_t>(allocator_->limit() - allocator_->top()),
839 size_in_bytes);
840
841 size_t new_node_size = 0;
842 Tagged<FreeSpace> new_node =
843 space_->free_list_->Allocate(size_in_bytes, &new_node_size, origin);
844 if (new_node.is_null()) return false;
845 DCHECK_GE(new_node_size, size_in_bytes);
846
847 // The old-space-step might have finished sweeping and restarted marking.
848 // Verify that it did not turn the page of the new node into an evacuation
849 // candidate.
850 DCHECK(!MarkCompactCollector::IsOnEvacuationCandidate(new_node));
851
852 // Mark the old linear allocation area with a free space map so it can be
853 // skipped when scanning the heap. This also puts it back in the free list
854 // if it is big enough.
855 FreeLinearAllocationAreaUnsynchronized();
856
857 // Memory in the linear allocation area is counted as allocated. We may free
858 // a little of this again immediately - see below.
859 PageMetadata* page = PageMetadata::FromHeapObject(new_node);
860 space_->IncreaseAllocatedBytes(new_node_size, page);
861
862 DCHECK_EQ(allocator_->allocation_info().start(),
863 allocator_->allocation_info().top());
864 Address start = new_node.address();
865 Address end = new_node.address() + new_node_size;
866 Address limit = allocator_->ComputeLimit(start, end, size_in_bytes);
867 DCHECK_LE(limit, end);
868 DCHECK_LE(size_in_bytes, limit - start);
869 if (limit != end) {
870 if (!allocator_->supports_extending_lab()) {
871 space_->Free(limit, end - limit);
872 end = limit;
873 } else {
874 DCHECK(allocator_->is_main_thread());
875 space_heap()->CreateFillerObjectAt(limit, static_cast<int>(end - limit));
876 }
877 }
878 SetLinearAllocationArea(start, limit, end);
879 space_->AddRangeToActiveSystemPages(page, start, limit);
880
881 return true;
882}

884bool PagedSpaceAllocatorPolicy::TryExtendLAB(int size_in_bytes) {
885 if (!allocator_->supports_extending_lab()) return false;
886 Address current_top = allocator_->top();
887 if (current_top == kNullAddress) return false;
888 Address current_limit = allocator_->limit();
889 Address max_limit = allocator_->original_limit_relaxed();
890 if (current_top + size_in_bytes > max_limit) {
891 return false;
892 }
893 allocator_->AdvanceAllocationObservers();
894 Address new_limit =
895 allocator_->ComputeLimit(current_top, max_limit, size_in_bytes);
896 allocator_->ExtendLAB(new_limit);
897 DCHECK(allocator_->is_main_thread());
898 space_heap()->CreateFillerObjectAt(new_limit,
899 static_cast<int>(max_limit - new_limit));
900 PageMetadata* page = PageMetadata::FromAddress(current_top);
901 // No need to create a black allocation area since new space doesn't use
902 // black allocation.
903 DCHECK_EQ(NEW_SPACE, allocator_->identity());
904 space_->AddRangeToActiveSystemPages(page, current_limit, new_limit);
905 return true;
906}

915void PagedSpaceAllocatorPolicy::FreeLinearAllocationAreaUnsynchronized() {
916 if (!allocator_->IsLabValid()) return;
917
918#if DEBUG
919 allocator_->Verify();
920#endif // DEBUG
921
922 Address current_top = allocator_->top();
923 Address current_limit = allocator_->limit();
924
925 Address current_max_limit = allocator_->supports_extending_lab()
926 ? allocator_->original_limit_relaxed()
927 : current_limit;
928 DCHECK_IMPLIES(!allocator_->supports_extending_lab(),
929 current_max_limit == current_limit);
930
931 allocator_->AdvanceAllocationObservers();
932
933 if (!v8_flags.black_allocated_pages) {
934 if (current_top != current_limit &&
935 allocator_->IsBlackAllocationEnabled()) {
936 PageMetadata::FromAddress(current_top)
937 ->DestroyBlackArea(current_top, current_limit);
938 }
939 }
940
941 allocator_->ResetLab(kNullAddress, kNullAddress, kNullAddress);
942 DCHECK_GE(current_limit, current_top);
943
944 DCHECK_IMPLIES(current_limit - current_top >= 2 * kTaggedSize,
945 space_heap()->marking_state()->IsUnmarked(
946 HeapObject::FromAddress(current_top)));
947 space_->Free(current_top, current_max_limit - current_top);
948}
