18void EvacuationVerifier::VerifyHeapObjectImpl(Tagged<HeapObject> heap_object) {
19 if (!ShouldVerifyObject(heap_object)) return;
20 CHECK_IMPLIES(
21 !v8_flags.sticky_mark_bits && HeapLayout::InYoungGeneration(heap_object),
22 Heap::InToPage(heap_object));
23 CHECK(!MarkCompactCollector::IsOnEvacuationCandidate(heap_object));
24}

26bool EvacuationVerifier::ShouldVerifyObject(Tagged<HeapObject> heap_object) {
27 const bool in_shared_heap = HeapLayout::InWritableSharedSpace(heap_object);
28 return heap_->isolate()->is_shared_space_isolate() ? in_shared_heap
29 : !in_shared_heap;
30}

32template <typename TSlot>
33void EvacuationVerifier::VerifyPointersImpl(TSlot start, TSlot end) {
34 for (TSlot current = start; current < end; ++current) {
35 typename TSlot::TObject object = current.load(cage_base());
36#ifdef V8_ENABLE_DIRECT_HANDLE
37 if (object.ptr() == kTaggedNullAddress) continue;
38#endif
39 Tagged<HeapObject> heap_object;
40 if (object.GetHeapObjectIfStrong(&heap_object)) {
41 VerifyHeapObjectImpl(heap_object);
42 }
43 }
44}
