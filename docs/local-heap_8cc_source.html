51LocalHeap::LocalHeap(Heap* heap, ThreadKind kind,
52 std::unique_ptr<PersistentHandles> persistent_handles)
53 : heap_(heap),
54 ptr_compr_cage_access_scope_(heap->isolate()),
55 is_main_thread_(kind == ThreadKind::kMain),
56 state_(ThreadState::Parked()),
57 allocation_failed_(false),
58 nested_parked_scopes_(0),
59 prev_(nullptr),
60 next_(nullptr),
61 handles_(new LocalHandles),
62 persistent_handles_(std::move(persistent_handles)),
63 heap_allocator_(this) {
64 DCHECK_IMPLIES(!is_main_thread(),
65 (v8_flags.concurrent_builtin_generation &&
66 heap->isolate()->IsGeneratingEmbeddedBuiltins()) ||
67 heap_->deserialization_complete());
68 if (!is_main_thread()) {
69 heap_allocator_.Setup();
70 SetUpMarkingBarrier();
71 }
72
73 heap_->safepoint()->AddLocalHeap(this, [this] {
74 if (!is_main_thread()) {
75 saved_marking_barrier_ =
76 WriteBarrier::SetForThread(marking_barrier_.get());
77 if (heap_->incremental_marking()->IsMarking()) {
78 marking_barrier_->Activate(
79 heap_->incremental_marking()->IsCompacting(),
80 heap_->incremental_marking()->marking_mode());
81 }
82
83 SetUpSharedMarking();
84 }
85 });
86
87 if (persistent_handles_) {
88 persistent_handles_->Attach(this);
89 }
90 DCHECK_NULL(LocalHeap::Current());
91 if (!is_main_thread()) {
92 saved_current_isolate_ = Isolate::TryGetCurrent();
93 Isolate::SetCurrent(heap_->isolate());
94 LocalHeap::SetCurrent(this);
95 }
96}

98LocalHeap::~LocalHeap() {
99 // Park thread since removing the local heap could block.
100 EnsureParkedBeforeDestruction();
101
102 heap_->safepoint()->RemoveLocalHeap(this, [this] {
103 FreeLinearAllocationAreas();
104
105 if (!is_main_thread()) {
106 marking_barrier_->PublishIfNeeded();
107 marking_barrier_->PublishSharedIfNeeded();
108 MarkingBarrier* overwritten =
109 WriteBarrier::SetForThread(saved_marking_barrier_);
110 DCHECK_EQ(overwritten, marking_barrier_.get());
111 USE(overwritten);
112 }
113 });
114
115 if (!is_main_thread()) {
116 DCHECK_EQ(Isolate::Current(), heap_->isolate());
117 Isolate::SetCurrent(saved_current_isolate_);
118 DCHECK_EQ(LocalHeap::Current(), this);
119 LocalHeap::SetCurrent(nullptr);
120 }
121
122 DCHECK(gc_epilogue_callbacks_.IsEmpty());
123}

148void LocalHeap::SetUpSharedMarking() {
149#if DEBUG
150 // Ensure the thread is either in the running state or holds the safepoint
151 // lock. This guarantees that the state of incremental marking can't change
152 // concurrently (this requires a safepoint).
153 if (is_main_thread()) {
154 DCHECK(IsRunning());
155 } else {
156 heap()->safepoint()->AssertActive();
157 }
158#endif // DEBUG
159
160 Isolate* isolate = heap_->isolate();
161
162 if (isolate->has_shared_space() && !isolate->is_shared_space_isolate()) {
163 if (isolate->shared_space_isolate()
164 ->heap()
165 ->incremental_marking()
166 ->IsMajorMarking()) {
167 marking_barrier_->ActivateShared();
168 }
169 }

220void LocalHeap::ParkSlowPath() {
221 while (true) {
222 ThreadState current_state = ThreadState::Running();
223 if (state_.CompareExchangeStrong(current_state, ThreadState::Parked()))
224 return;
225
226 // CAS above failed, so state is Running with some additional flag.
227 DCHECK(current_state.IsRunning());
228
229 if (is_main_thread()) {
230 DCHECK(current_state.IsSafepointRequested() ||
231 current_state.IsCollectionRequested());
232
233 if (current_state.IsSafepointRequested()) {
234 ThreadState old_state = state_.SetParked();
235 heap_->safepoint()->NotifyPark();
236 if (old_state.IsCollectionRequested())
237 heap_->collection_barrier_->CancelCollectionAndResumeThreads();
238 return;
239 }
240
241 if (current_state.IsCollectionRequested()) {
242 if (!heap()->ignore_local_gc_requests()) {
243 heap_->CollectGarbageForBackground(this);
244 continue;
245 }
246
247 DCHECK(!current_state.IsSafepointRequested());
248
249 if (state_.CompareExchangeStrong(current_state,
250 current_state.SetParked())) {
251 heap_->collection_barrier_->CancelCollectionAndResumeThreads();
252 return;
253 } else {
254 continue;
255 }
256 }
257 } else {
258 DCHECK(current_state.IsSafepointRequested());
259 DCHECK(!current_state.IsCollectionRequested());
260
261 ThreadState old_state = state_.SetParked();
262 CHECK(old_state.IsRunning());
263 CHECK(old_state.IsSafepointRequested());
264 CHECK(!old_state.IsCollectionRequested());
265
266 heap_->safepoint()->NotifyPark();
267 return;
268 }
269 }
270}

272void LocalHeap::UnparkSlowPath() {
273 while (true) {
274 ThreadState current_state = ThreadState::Parked();
275 if (state_.CompareExchangeStrong(current_state, ThreadState::Running()))
276 return;
277
278 // CAS above failed, so state is Parked with some additional flag.
279 DCHECK(current_state.IsParked());
280
281 if (is_main_thread()) {
282 DCHECK(current_state.IsSafepointRequested() ||
283 current_state.IsCollectionRequested());
284
285 if (current_state.IsSafepointRequested()) {
286 SleepInUnpark();
287 continue;
288 }
289
290 if (current_state.IsCollectionRequested()) {
291 DCHECK(!current_state.IsSafepointRequested());
292
293 if (!state_.CompareExchangeStrong(current_state,
294 current_state.SetRunning()))
295 continue;
296
297 if (!heap()->ignore_local_gc_requests()) {
298 heap_->CollectGarbageForBackground(this);
299 }
300
301 return;
302 }
303 } else {
304 DCHECK(current_state.IsSafepointRequested());
305 DCHECK(!current_state.IsCollectionRequested());
306
307 SleepInUnpark();
308 }
309 }

332void LocalHeap::SafepointSlowPath() {
333 ThreadState current_state = state_.load_relaxed();
334 DCHECK(current_state.IsRunning());
335
336 if (is_main_thread()) {
337 DCHECK(current_state.IsSafepointRequested() ||
338 current_state.IsCollectionRequested());

345 heap_->CollectGarbageForBackground(this);
346 }

349 DCHECK(!current_state.IsCollectionRequested());

355void LocalHeap::SleepInSafepoint() {
356 GCTracer::Scope::ScopeId scope_id;
357 ThreadKind thread_kind;
358
359 if (is_main_thread()) {
360 scope_id = GCTracer::Scope::SAFEPOINT;
361 thread_kind = ThreadKind::kMain;
362 } else {
363 scope_id = GCTracer::Scope::BACKGROUND_SAFEPOINT;
364 thread_kind = ThreadKind::kBackground;
365 }
366
367 TRACE_GC1(heap_->tracer(), scope_id, thread_kind);
368
369 ExecuteWithStackMarker([this]() {
370 // Parking the running thread here is an optimization. We do not need to
371 // wake this thread up to reach the next safepoint.
372 ThreadState old_state = state_.SetParked();
373 CHECK(old_state.IsRunning());
374 CHECK(old_state.IsSafepointRequested());
375 CHECK_IMPLIES(old_state.IsCollectionRequested(), is_main_thread());
376
377 heap_->safepoint()->WaitInSafepoint();
378
379 std::optional<IgnoreLocalGCRequests> ignore_gc_requests;
380 if (is_main_thread()) ignore_gc_requests.emplace(heap());
381 Unpark();
382 });
383}

385#ifdef DEBUG
386bool LocalHeap::IsSafeForConservativeStackScanning() const {
387#if defined(V8_ENABLE_DIRECT_HANDLE) && defined(ENABLE_SLOW_DCHECKS)
388 // There must be no direct handles on the stack below the stack marker.
389 if (DirectHandleBase::NumberOfHandles() > 0) return false;
390#endif // V8_ENABLE_DIRECT_HANDLE && ENABLE_SLOW_DCHECKS
391 // Check if we are inside at least one ParkedScope.
392 if (nested_parked_scopes_ > 0) {
393 // The main thread can avoid the trampoline, if it's not the main thread of
394 // a client isolate.
395 if (is_main_thread() && (heap()->isolate()->is_shared_space_isolate() ||
396 !heap()->isolate()->has_shared_space()))
397 return true;
398 // Otherwise, require that we're inside the trampoline.
399 return is_in_trampoline();
400 }
401 // Otherwise, we are reaching the initial parked state and the stack should
402 // not be interesting.
403 return true;
404}
405#endif // DEBUG
