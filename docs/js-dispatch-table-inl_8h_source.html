20void JSDispatchEntry::MakeJSDispatchEntry(Address object, Address entrypoint,
21 uint16_t parameter_count,
22 bool mark_as_alive) {
23 DCHECK_EQ(object & kHeapObjectTag, 0);
24 DCHECK_EQ((object << kObjectPointerShift) >> kObjectPointerShift, object);
25
26 Address payload =
27 (object << kObjectPointerShift) | (parameter_count & kParameterCountMask);
28 DCHECK(!(payload & kMarkingBit));
29 if (mark_as_alive) payload |= kMarkingBit;
30#ifdef V8_TARGET_ARCH_32_BIT
31 parameter_count_.store(parameter_count, std::memory_order_relaxed);
32 next_free_entry_.store(0, std::memory_order_relaxed);
33#endif
34 encoded_word_.store(payload, std::memory_order_relaxed);
35 entrypoint_.store(entrypoint, std::memory_order_relaxed);
36 DCHECK(!IsFreelistEntry());
37}

44Address JSDispatchEntry::GetCodePointer() const {
45 CHECK(!IsFreelistEntry());
46 // The pointer tag bit (LSB) of the object pointer is used as marking bit,
47 // and so may be 0 or 1 here. As the return value is a tagged pointer, the
48 // bit must be 1 when returned, so we need to set it here.
49 Address payload = encoded_word_.load(std::memory_order_relaxed);
50 return (payload >> kObjectPointerShift) | kHeapObjectTag;
51}

57uint16_t JSDispatchEntry::GetParameterCount() const {
58 // Loading a pointer out of a freed entry will always result in an invalid
59 // pointer (e.g. upper bits set or nullptr). However, here we're just loading
60 // an integer (the parameter count), so we probably want to make sure that
61 // we're not getting that from a freed entry.
62 CHECK(!IsFreelistEntry());
63#ifdef V8_TARGET_ARCH_32_BIT
64 return parameter_count_.load(std::memory_order_relaxed);
65#else
66 static_assert(kParameterCountMask != 0);
67 Address payload = encoded_word_.load(std::memory_order_relaxed);
68 return payload & kParameterCountMask;
69#endif
70}

117bool JSDispatchTable::IsTieringRequested(JSDispatchHandle handle) {
118 uint32_t index = HandleToIndex(handle);
119 DCHECK_GE(index, kEndOfInternalReadOnlySegment);
120 Address entrypoint = at(index).GetEntrypoint();
121 Address code_entrypoint = at(index).GetCode()->instruction_start();
122 return code_entrypoint != entrypoint;
123}

125bool JSDispatchTable::IsTieringRequested(JSDispatchHandle handle,
126 TieringBuiltin builtin,
127 Isolate* isolate) {
128 uint32_t index = HandleToIndex(handle);
129 DCHECK_GE(index, kEndOfInternalReadOnlySegment);
130 Address entrypoint = at(index).GetEntrypoint();
131 Address code_entrypoint = at(index).GetCode()->instruction_start();
132 if (entrypoint == code_entrypoint) return false;
133 return entrypoint == EmbeddedData::FromBlob(isolate).InstructionStartOf(
134 static_cast<Builtin>(builtin));

168void JSDispatchEntry::SetCodeAndEntrypointPointer(Address new_object,
169 Address new_entrypoint) {
170 Address old_payload = encoded_word_.load(std::memory_order_relaxed);
171 Address marking_bit = old_payload & kMarkingBit;
172 Address parameter_count = old_payload & kParameterCountMask;
173 // We want to preserve the marking bit of the entry. Since that happens to
174 // be the tag bit of the pointer, we need to explicitly clear it here.
175 Address object = (new_object << kObjectPointerShift) & ~kMarkingBit;
176 Address new_payload = object | marking_bit | parameter_count;
177 encoded_word_.store(new_payload, std::memory_order_relaxed);
178 entrypoint_.store(new_entrypoint, std::memory_order_relaxed);
179 DCHECK(!IsFreelistEntry());
180}

186void JSDispatchEntry::MakeFreelistEntry(uint32_t next_entry_index) {
187#ifdef V8_TARGET_ARCH_64_BIT
188 Address payload = kFreeEntryTag | next_entry_index;
189 entrypoint_.store(payload, std::memory_order_relaxed);
190#else
191 // Store index + 1 such that we can use 0 for non-free entries.
192 next_free_entry_.store(next_entry_index + 1, std::memory_order_relaxed);
193 entrypoint_.store(kNullAddress, std::memory_order_relaxed);
194#endif
195 encoded_word_.store(kNullAddress, std::memory_order_relaxed);
196 DCHECK(IsFreelistEntry());
197}

199bool JSDispatchEntry::IsFreelistEntry() const {
200#ifdef V8_TARGET_ARCH_64_BIT
201 auto entrypoint = entrypoint_.load(std::memory_order_relaxed);
202 return (entrypoint & kFreeEntryTag) == kFreeEntryTag;
203#else
204 return next_free_entry_.load(std::memory_order_relaxed) != 0;
205#endif
206}

217void JSDispatchEntry::Mark() {
218 Address old_value = encoded_word_.load(std::memory_order_relaxed);
219 Address new_value = old_value | kMarkingBit;
220 // We don't need this cas to succeed. If marking races with
221 // `SetCodeAndEntrypointPointer`, then we are bound to re-set the mark bit in
222 // the write barrier.
223 static_assert(JSDispatchTable::kWriteBarrierSetsEntryMarkBit);
224 encoded_word_.compare_exchange_strong(old_value, new_value,
225 std::memory_order_relaxed);
226}
227
228void JSDispatchEntry::Unmark() {
229 Address value = encoded_word_.load(std::memory_order_relaxed);
230 value &= ~kMarkingBit;
231 encoded_word_.store(value, std::memory_order_relaxed);
232}

262 CFIMetadataWriteScope write_scope("JSDispatchTable write");

309// static
310bool JSDispatchTable::IsCompatibleCode(Tagged<Code> code,
311 uint16_t parameter_count) {
312 if (code->entrypoint_tag() != kJSEntrypointTag) {
313 // Target code doesn't use JS linkage. This cannot be valid.
314 return false;
315 }
316 if (code->parameter_count() == parameter_count) {
317 DCHECK_IMPLIES(code->is_builtin(),
318 parameter_count ==
319 Builtins::GetFormalParameterCount(code->builtin_id()));
320 // Dispatch entry and code have the same signature. This is correct.
321 return true;
322 }
323
324 // Signature mismatch. This is mostly not safe, except for certain varargs
325 // builtins which are able to correctly handle such a mismatch. Examples
326 // include builtins like the InterpreterEntryTrampoline or the JSToWasm and
327 // JSToJS wrappers which determine their actual parameter count at runtime
328 // (see CodeStubAssembler::SetSupportsDynamicParameterCount()), or internal
329 // builtins that end up tailcalling into other code such as CompileLazy.
330 //
331 // Currently, we also allow this for testing code (from our test suites).
332 // TODO(saelo): maybe we should also forbid this just to be sure.
333 if (code->kind() == CodeKind::FOR_TESTING) {
334 return true;
335 }
336 DCHECK(code->is_builtin());
337 DCHECK_EQ(code->parameter_count(), kDontAdaptArgumentsSentinel);
338 switch (code->builtin_id()) {
339 case Builtin::kIllegal:
340 case Builtin::kCompileLazy:
341 case Builtin::kInterpreterEntryTrampoline:
342 case Builtin::kInstantiateAsmJs:
343 case Builtin::kDebugBreakTrampoline:
344#ifdef V8_ENABLE_WEBASSEMBLY
345 case Builtin::kJSToWasmWrapper:
346 case Builtin::kJSToJSWrapper:
347 case Builtin::kJSToJSWrapperInvalidSig:
348 case Builtin::kWasmPromising:
349#if V8_ENABLE_DRUMBRAKE
350 case Builtin::kGenericJSToWasmInterpreterWrapper:
351#endif
352 case Builtin::kWasmStressSwitch:
353#endif
354 return true;
355 default:
356 return false;
357 }
358}
