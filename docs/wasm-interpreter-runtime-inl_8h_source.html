22inline Address WasmInterpreterRuntime::EffectiveAddress(uint64_t index) const {
23 Handle<WasmTrustedInstanceData> trusted_data = wasm_trusted_instance_data();
24 DCHECK_GE(std::numeric_limits<uintptr_t>::max(),
25 trusted_data->memory0_size());
26 DCHECK_GE(trusted_data->memory0_size(), index);
27 // Compute the effective address of the access, making sure to condition
28 // the index even in the in-bounds case.
29 return reinterpret_cast<Address>(trusted_data->memory0_start()) + index;
30}

32inline bool WasmInterpreterRuntime::BoundsCheckMemRange(
33 uint64_t index, uint64_t* size, Address* out_address) const {
34 Handle<WasmTrustedInstanceData> trusted_data = wasm_trusted_instance_data();
35 DCHECK_GE(std::numeric_limits<uintptr_t>::max(),
36 trusted_data->memory0_size());
37 if (!base::ClampToBounds<uint64_t>(index, size,
38 trusted_data->memory0_size())) {
39 return false;
40 }
41 *out_address = EffectiveAddress(index);
42 return true;
43}

50inline Handle<Object> WasmInterpreterRuntime::GetGlobalRef(
51 uint32_t index) const {
52 // This function assumes that it is executed in a HandleScope.
53 const wasm::WasmGlobal& global = module_->globals[index];
54 DCHECK(global.type.is_reference());
55 Tagged<FixedArray> global_buffer; // The buffer of the global.
56 uint32_t global_index = 0; // The index into the buffer.
57 std::tie(global_buffer, global_index) =
58 wasm_trusted_instance_data()->GetGlobalBufferAndIndex(global);
59 return Handle<Object>(global_buffer->get(global_index), isolate_);
60}

151inline bool WasmInterpreterRuntime::WasmStackCheck(
152 const uint8_t* current_bytecode, const uint8_t*& code) {
153 StackLimitCheck stack_check(isolate_);
154
155 current_frame_.current_bytecode_ = current_bytecode;
156 current_thread_->SetCurrentFrame(current_frame_);
157
158 if (stack_check.InterruptRequested()) {
159 if (stack_check.HasOverflowed()) {
160 ClearThreadInWasmScope clear_wasm_flag(isolate_);
161 SealHandleScope shs(isolate_);
162 current_frame_.current_function_ = nullptr;
163 SetTrap(TrapReason::kTrapUnreachable, code);
164 isolate_->StackOverflow();
165 return false;
166 }
167 if (isolate_->stack_guard()->HasTerminationRequest()) {
168 ClearThreadInWasmScope clear_wasm_flag(isolate_);
169 SealHandleScope shs(isolate_);
170 current_frame_.current_function_ = nullptr;
171 SetTrap(TrapReason::kTrapUnreachable, code);
172 isolate_->TerminateExecution();
173 return false;
174 }
175 isolate_->stack_guard()->HandleInterrupts();
176 }
177
178 if (V8_UNLIKELY(v8_flags.drumbrake_fuzzer_timeout &&
179 base::TimeTicks::Now() - fuzzer_start_time_ >
180 base::TimeDelta::FromMilliseconds(
181 v8_flags.drumbrake_fuzzer_timeout_limit_ms))) {
182 ClearThreadInWasmScope clear_wasm_flag(isolate_);
183 SealHandleScope shs(isolate_);
184 current_frame_.current_function_ = nullptr;
185 SetTrap(TrapReason::kTrapUnreachable, code);
186 return false;
187 }
188
189 return true;
190}
