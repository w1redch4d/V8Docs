21Handle<FixedArray> FixedArray::SetAndGrow(Isolate* isolate,
22 Handle<FixedArray> array, int index,
23 DirectHandle<Object> value) {
24 int len = array->length();
25 if (index >= len) {
26 int new_capacity = FixedArray::NewCapacityForIndex(index, len);
27 array = Cast<FixedArray>(FixedArray::Resize(isolate, array, new_capacity));
28 // TODO(jgruber): This is somewhat subtle - other FixedArray methods
29 // use `undefined` as a filler. Make this more explicit.
30 array->FillWithHoles(len, new_capacity);
31 }
32
33 array->set(index, *value);
34 return array;
35}

129Handle<ArrayList> ArrayList::EnsureSpace(Isolate* isolate,
130 Handle<ArrayList> array, int length,
131 AllocationType allocation) {
132 DCHECK_LT(0, length);
133 int old_capacity = array->capacity();
134 if (old_capacity >= length) return array;
135
136 int old_length = array->length();
137 // Ensure calculation matches CodeStubAssembler::ArrayListEnsureSpace.
138 int new_capacity = length + std::max(length / 2, 2);
139 Handle<ArrayList> new_array =
140 ArrayList::New(isolate, new_capacity, allocation);
141 DisallowGarbageCollection no_gc;
142 new_array->set_length(old_length);
143 WriteBarrierMode mode = new_array->GetWriteBarrierMode(no_gc);
144 CopyElements(isolate, *new_array, 0, *array, 0, old_length, mode);
145 return new_array;

183// static
184Handle<WeakArrayList> WeakArrayList::Append(Isolate* isolate,
185 Handle<WeakArrayList> array,
186 MaybeObjectDirectHandle value,
187 AllocationType allocation) {
188 int length = 0;
189 int new_length = 0;
190 {
191 DisallowGarbageCollection no_gc;
192 Tagged<WeakArrayList> raw = *array;
193 length = raw->length();
194
195 if (length < raw->capacity()) {
196 raw->Set(length, *value);
197 raw->set_length(length + 1);
198 return array;
199 }
200
201 // Not enough space in the array left, either grow, shrink or
202 // compact the array.
203 new_length = raw->CountLiveElements() + 1;
204 }
205
206 bool shrink = new_length < length / 4;
207 bool grow = 3 * (length / 4) < new_length;
208
209 if (shrink || grow) {
210 // Grow or shrink array and compact out-of-place.
211 int new_capacity = CapacityForLength(new_length);
212 array = isolate->factory()->CompactWeakArrayList(array, new_capacity,
213 allocation);
214
215 } else {
216 // Perform compaction in the current array.
217 array->Compact(isolate);
218 }
219
220 // Now append value to the array, there should always be enough space now.
221 DCHECK_LT(array->length(), array->capacity());
222
223 {
224 DisallowGarbageCollection no_gc;
225 Tagged<WeakArrayList> raw = *array;
226 // Reload length, allocation might have killed some weak refs.
227 int index = raw->length();
228 raw->Set(index, *value);
229 raw->set_length(index + 1);
230 }
231 return array;
232}

234void WeakArrayList::Compact(Isolate* isolate) {
235 DisallowGarbageCollection no_gc;
236 int length = this->length();
237 int new_length = 0;
238
239 for (int i = 0; i < length; i++) {
240 Tagged<MaybeObject> value = Get(isolate, i);
241
242 if (!value.IsCleared()) {
243 if (new_length != i) {
244 Set(new_length, value);
245 }
246 ++new_length;
247 }
248 }
249
250 set_length(new_length);
251}

289bool WeakArrayList::RemoveOne(MaybeObjectDirectHandle value) {
290 int last_index = length() - 1;
291 // Optimize for the most recently added element to be removed again.
292 for (int i = last_index; i >= 0; --i) {
293 if (Get(i) != *value) continue;
294 // Move the last element into this slot (or no-op, if this is the last
295 // slot).
296 Set(i, Get(last_index));
297 Set(last_index, ClearedValue(GetIsolate()));
298 set_length(last_index);
299 return true;
300 }
301 return false;
302}
