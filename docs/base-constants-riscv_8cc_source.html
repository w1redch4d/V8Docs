40int Registers::Number(const char* name) {
41 // Look through the canonical names.
42 for (int i = 0; i < kNumSimuRegisters; i++) {
43 if (strcmp(names_[i], name) == 0) {
44 return i;
45 }
46 }
47
48 // Look through the alias names.
49 int i = 0;
50 while (aliases_[i].reg != kInvalidRegister) {
51 if (strcmp(aliases_[i].name, name) == 0) {
52 return aliases_[i].reg;
53 }
54 i++;
55 }
56
57 // No register with the reguested name found.
58 return kInvalidRegister;
59}

147bool InstructionBase::IsShortInstruction() const {
148 uint8_t FirstByte = *reinterpret_cast<const uint8_t*>(this);
149 return (FirstByte & 0x03) <= C2;
150}

206template <class T>
207uint32_t InstructionGetters<T>::Rvvzimm() const {
208 if ((this->InstructionBits() &
209 (kBaseOpcodeMask | kFunct3Mask | 0x80000000)) == RO_V_VSETVLI) {
210 uint32_t Bits = this->InstructionBits();
211 uint32_t zimm = Bits & kRvvZimmMask;
212 return zimm >> kRvvZimmShift;
213 } else {
214 DCHECK_EQ(
215 this->InstructionBits() & (kBaseOpcodeMask | kFunct3Mask | 0xC0000000),
216 RO_V_VSETIVLI);
217 uint32_t Bits = this->InstructionBits();
218 uint32_t zimm = Bits & kRvvZimmMask;
219 return (zimm >> kRvvZimmShift) & 0x3FF;
220 }
221}
222
223template <class T>
224uint32_t InstructionGetters<T>::Rvvuimm() const {
225 DCHECK_EQ(
226 this->InstructionBits() & (kBaseOpcodeMask | kFunct3Mask | 0xC0000000),
227 RO_V_VSETIVLI);
228 uint32_t Bits = this->InstructionBits();
229 uint32_t uimm = Bits & kRvvUimmMask;
230 return uimm >> kRvvUimmShift;
231}

233template <class T>
234bool InstructionGetters<T>::IsLoad() {
235 switch (OperandFunct3()) {
236 case RO_LB:
237 case RO_LBU:
238 case RO_LH:
239 case RO_LHU:
240 case RO_LW:
241#ifdef V8_TARGET_ARCH_RISCV64
242 case RO_LD:
243 case RO_LWU:
244#endif
245 return true;
246 case RO_C_LW:
247 case RO_C_LWSP:
248#ifdef V8_TARGET_ARCH_RISCV64
249 case RO_C_LD:
250 case RO_C_LDSP:
251#endif
252 return v8_flags.riscv_c_extension && this->IsShortInstruction();
253 default:
254 return BaseOpcode() == LOAD_FP;
255 }
256}

272 case RO_C_SDSP:
273#endif
274 return v8_flags.riscv_c_extension && this->IsShortInstruction();
275 default:
276 return BaseOpcode() == STORE_FP;

285InstructionBase::Type InstructionBase::InstructionType() const {
286 if (IsIllegalInstruction()) {
287 return kUnsupported;
288 }
289 // RV64C Instruction
290 if (v8_flags.riscv_c_extension && IsShortInstruction()) {
291 switch (InstructionBits() & kRvcOpcodeMask) {
292 case RO_C_ADDI4SPN:
293 return kCIWType;
294 case RO_C_FLD:
295 case RO_C_LW:
296#ifdef V8_TARGET_ARCH_RISCV64
297 case RO_C_LD:
298#endif
299 return kCLType;
300 case RO_C_FSD:
301 case RO_C_SW:
302#ifdef V8_TARGET_ARCH_RISCV64
303 case RO_C_SD:
304#endif
305 return kCSType;
306 case RO_C_NOP_ADDI:
307 case RO_C_LI:
308#ifdef V8_TARGET_ARCH_RISCV64
309 case RO_C_ADDIW:
310#endif
311 case RO_C_LUI_ADD:
312 return kCIType;
313 case RO_C_MISC_ALU:
314 if (Bits(11, 10) != 0b11)
315 return kCBType;
316 else
317 return kCAType;
318 case RO_C_J:
319 return kCJType;
320 case RO_C_BEQZ:
321 case RO_C_BNEZ:
322 return kCBType;
323 case RO_C_SLLI:
324 case RO_C_FLDSP:
325 case RO_C_LWSP:
326#ifdef V8_TARGET_ARCH_RISCV64
327 case RO_C_LDSP:
328#endif
329 return kCIType;
330 case RO_C_JR_MV_ADD:
331 return kCRType;
332 case RO_C_FSDSP:
333 case RO_C_SWSP:
334#ifdef V8_TARGET_ARCH_RISCV64
335 case RO_C_SDSP:
336#endif
337 return kCSSType;
338 default:
339 break;
340 }
341 } else {
342 // RISCV routine
343 switch (InstructionBits() & kBaseOpcodeMask) {
344 case LOAD:
345 return kIType;
346 case LOAD_FP:
347 return kIType;
348 case MISC_MEM:
349 return kIType;
350 case OP_IMM:
351 return kIType;
352 case AUIPC:
353 return kUType;
354 case OP_IMM_32:
355 return kIType;
356 case STORE:
357 return kSType;
358 case STORE_FP:
359 return kSType;
360 case AMO:
361 return kRType;
362 case OP:
363 return kRType;
364 case LUI:
365 return kUType;
366 case OP_32:
367 return kRType;
368 case MADD:
369 case MSUB:
370 case NMSUB:
371 case NMADD:
372 return kR4Type;
373 case OP_FP:
374 return kRType;
375 case BRANCH:
376 return kBType;
377 case JALR:
378 return kIType;
379 case JAL:
380 return kJType;
381 case SYSTEM:
382 return kIType;
383 case OP_V:
384 return kVType;
385 }
386 }
387 return kUnsupported;
388}
