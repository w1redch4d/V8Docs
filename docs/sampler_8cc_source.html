184AtomicGuard::AtomicGuard(AtomicMutex* atomic, bool is_blocking)
185 : atomic_(atomic), is_success_(false) {
186 do {
187 bool expected = false;
188 // We have to use the strong version here for the case where is_blocking
189 // is false, and we will only attempt the exchange once.
190 is_success_ = atomic->compare_exchange_strong(expected, true);
191 } while (is_blocking && !is_success_);
192}
193
194AtomicGuard::~AtomicGuard() {
195 if (!is_success_) return;
196 atomic_->store(false);
197}
198
199bool AtomicGuard::is_success() const { return is_success_; }

213void SamplerManager::AddSampler(Sampler* sampler) {
214 AtomicGuard atomic_guard(&samplers_access_counter_);
215 DCHECK(sampler->IsActive());
216 int thread_id = sampler->platform_data()->vm_tid();
217 auto it = sampler_map_.find(thread_id);
218 if (it == sampler_map_.end()) {
219 SamplerList samplers;
220 samplers.push_back(sampler);
221 sampler_map_.emplace(thread_id, std::move(samplers));
222 } else {
223 SamplerList& samplers = it->second;
224 auto sampler_it = std::find(samplers.begin(), samplers.end(), sampler);
225 if (sampler_it == samplers.end()) samplers.push_back(sampler);
226 }
227}
228
229void SamplerManager::RemoveSampler(Sampler* sampler) {
230 AtomicGuard atomic_guard(&samplers_access_counter_);
231 DCHECK(sampler->IsActive());
232 int thread_id = sampler->platform_data()->vm_tid();
233 auto it = sampler_map_.find(thread_id);
234 DCHECK_NE(it, sampler_map_.end());
235 SamplerList& samplers = it->second;
236 samplers.erase(std::remove(samplers.begin(), samplers.end(), sampler),
237 samplers.end());
238 if (samplers.empty()) {
239 sampler_map_.erase(it);
240 }
241}

243void SamplerManager::DoSample(const v8::RegisterState& state) {
244 AtomicGuard atomic_guard(&samplers_access_counter_, false);
245 // TODO(petermarshall): Add stat counters for the bailouts here.
246 if (!atomic_guard.is_success()) return;
247 int thread_id = base::OS::GetCurrentThreadId();
248 auto it = sampler_map_.find(thread_id);
249 if (it == sampler_map_.end()) return;
250 SamplerList& samplers = it->second;
251
252 for (Sampler* sampler : samplers) {
253 if (!sampler->ShouldRecordSample()) continue;
254 Isolate* isolate = sampler->isolate();
255 // We require a fully initialized and entered isolate.
256 if (isolate == nullptr || !isolate->IsInUse()) continue;
257 sampler->SampleStack(state);
258 }

323#if defined(USE_SIGNALS)
324class SignalHandler {
325 public:
326 static void IncreaseSamplerCount() {
327 base::RecursiveMutexGuard lock_guard(mutex_.Pointer());
328 if (++client_count_ == 1) Install();
329 }
330
331 static void DecreaseSamplerCount() {
332 base::RecursiveMutexGuard lock_guard(mutex_.Pointer());
333 if (--client_count_ == 0) Restore();
334 }
335
336 static bool Installed() {
337 // mutex_ will also be used in Sampler::DoSample to guard the state below.
338 base::RecursiveMutexGuard lock_guard(mutex_.Pointer());
339 return signal_handler_installed_;
340 }
341
342 static v8::base::RecursiveMutex* mutex() { return mutex_.Pointer(); }
343
344 private:
345 static void Install() {
346 struct sigaction sa;
347 sa.sa_sigaction = &HandleProfilerSignal;
348 sigemptyset(&sa.sa_mask);
349#if V8_OS_QNX
350 sa.sa_flags = SA_SIGINFO | SA_ONSTACK;
351#else
352 sa.sa_flags = SA_RESTART | SA_SIGINFO | SA_ONSTACK;
353#endif
354 signal_handler_installed_ =
355 (sigaction(SIGPROF, &sa, &old_signal_handler_) == 0);
356 }
357
358 static void Restore() {
359 if (signal_handler_installed_) {
360 signal_handler_installed_ = false;
361#if V8_OS_AIX || V8_TARGET_ARCH_S390X
362 // On Aix, IBMi & zLinux SIGPROF can sometimes arrive after the
363 // default signal handler is restored, resulting in intermittent test
364 // failure when profiling is enabled (https://crbug.com/v8/12952)
365 base::OS::Sleep(base::TimeDelta::FromMicroseconds(10));
366#endif
367 sigaction(SIGPROF, &old_signal_handler_, nullptr);
368 }
369 }
370
371 static void FillRegisterState(void* context, RegisterState* regs);
372 static void HandleProfilerSignal(int signal, siginfo_t* info, void* context);
373
374 // Protects the process wide state below.
375 static base::LazyRecursiveMutex mutex_;
376 static int client_count_;
377 static bool signal_handler_installed_;
378 static struct sigaction old_signal_handler_;
379};

388void SignalHandler::HandleProfilerSignal(int signal, siginfo_t* info,
389 void* context) {
390 v8::ThreadIsolatedAllocator::SetDefaultPermissionsForSignalHandler();
391 USE(info);

398void SignalHandler::FillRegisterState(void* context, RegisterState* state) {
399 // Extracting the sample from the context is extremely machine dependent.
400 ucontext_t* ucontext = reinterpret_cast<ucontext_t*>(context);
401#if !(V8_OS_OPENBSD || V8_OS_ZOS || \
402 (V8_OS_LINUX && (V8_HOST_ARCH_S390X || V8_HOST_ARCH_PPC64)))
403 mcontext_t& mcontext = ucontext->uc_mcontext;
404#elif V8_OS_ZOS
405 __mcontext_t_* mcontext = reinterpret_cast<__mcontext_t_*>(context);
406#endif
407#if V8_OS_LINUX
408#if V8_HOST_ARCH_IA32
409 state->pc = reinterpret_cast<void*>(mcontext.gregs[REG_EIP]);
410 state->sp = reinterpret_cast<void*>(mcontext.gregs[REG_ESP]);
411 state->fp = reinterpret_cast<void*>(mcontext.gregs[REG_EBP]);
412#elif V8_HOST_ARCH_X64
413 state->pc = reinterpret_cast<void*>(mcontext.gregs[REG_RIP]);
414 state->sp = reinterpret_cast<void*>(mcontext.gregs[REG_RSP]);
415 state->fp = reinterpret_cast<void*>(mcontext.gregs[REG_RBP]);
416#elif V8_HOST_ARCH_ARM
417#if V8_LIBC_GLIBC && !V8_GLIBC_PREREQ(2, 4)
418 // Old GLibc ARM versions used a gregs[] array to access the register
419 // values from mcontext_t.
420 state->pc = reinterpret_cast<void*>(mcontext.gregs[R15]);
421 state->sp = reinterpret_cast<void*>(mcontext.gregs[R13]);
422 state->fp = reinterpret_cast<void*>(mcontext.gregs[R11]);
423 state->lr = reinterpret_cast<void*>(mcontext.gregs[R14]);
424#else
425 state->pc = reinterpret_cast<void*>(mcontext.arm_pc);
426 state->sp = reinterpret_cast<void*>(mcontext.arm_sp);
427 state->fp = reinterpret_cast<void*>(mcontext.arm_fp);
428 state->lr = reinterpret_cast<void*>(mcontext.arm_lr);
429#endif // V8_LIBC_GLIBC && !V8_GLIBC_PREREQ(2, 4)
430#elif V8_HOST_ARCH_ARM64
431 state->pc = reinterpret_cast<void*>(mcontext.pc);
432 state->sp = reinterpret_cast<void*>(mcontext.sp);
433 // FP is an alias for x29.
434 state->fp = reinterpret_cast<void*>(mcontext.regs[29]);
435 // LR is an alias for x30.
436 state->lr = reinterpret_cast<void*>(mcontext.regs[30]);
437#elif V8_HOST_ARCH_MIPS64

441#elif V8_HOST_ARCH_LOONG64

444 state->fp = reinterpret_cast<void*>(mcontext.__gregs[22]);
445#elif V8_HOST_ARCH_PPC64
446#if V8_LIBC_GLIBC
447 state->pc = reinterpret_cast<void*>(ucontext->uc_mcontext.regs->nip);
448 state->sp = reinterpret_cast<void*>(ucontext->uc_mcontext.regs->gpr[PT_R1]);

603 if (profiled_thread == nullptr) return;
604
605 const DWORD kSuspendFailed = static_cast<DWORD>(-1);
606 if (SuspendThread(profiled_thread) == kSuspendFailed) return;

608 // Context used for sampling the register state of the profiled thread.
609 CONTEXT context;
610 memset(&context, 0, sizeof(context));
611 context.ContextFlags = CONTEXT_FULL;

613 v8::RegisterState state;
614#if V8_HOST_ARCH_X64
615 state.pc = reinterpret_cast<void*>(context.Rip);
616 state.sp = reinterpret_cast<void*>(context.Rsp);
617 state.fp = reinterpret_cast<void*>(context.Rbp);
618#elif V8_HOST_ARCH_ARM64

620 state.sp = reinterpret_cast<void*>(context.Sp);
621 state.fp = reinterpret_cast<void*>(context.Fp);
622#else
623 state.pc = reinterpret_cast<void*>(context.Eip);
624 state.sp = reinterpret_cast<void*>(context.Esp);

634void Sampler::DoSample() {
635 zx_handle_t profiled_thread = platform_data()->profiled_thread();
636 if (profiled_thread == ZX_HANDLE_INVALID) return;
637
638 zx_handle_t suspend_token = ZX_HANDLE_INVALID;
639 if (zx_task_suspend_token(profiled_thread, &suspend_token) != ZX_OK) return;
640
641 // Wait for the target thread to become suspended, or to exit.
642 // TODO(wez): There is currently no suspension count for threads, so there
643 // is a risk that some other caller resumes the thread in-between our suspend
644 // and wait calls, causing us to miss the SUSPENDED signal. We apply a 100ms
645 // deadline to protect against hanging the sampler thread in this case.
646 zx_signals_t signals = 0;
647 zx_status_t suspended = zx_object_wait_one(
648 profiled_thread, ZX_THREAD_SUSPENDED | ZX_THREAD_TERMINATED,
649 zx_deadline_after(ZX_MSEC(100)), &signals);
650 if (suspended != ZX_OK || (signals & ZX_THREAD_SUSPENDED) == 0) {
651 zx_handle_close(suspend_token);
652 return;
653 }
654
655 // Fetch a copy of its "general register" states.
656 zx_thread_state_general_regs_t thread_state = {};
657 if (zx_thread_read_state(profiled_thread, ZX_THREAD_STATE_GENERAL_REGS,
658 &thread_state, sizeof(thread_state)) == ZX_OK) {
659 v8::RegisterState state;
660#if V8_HOST_ARCH_X64
661 state.pc = reinterpret_cast<void*>(thread_state.rip);
662 state.sp = reinterpret_cast<void*>(thread_state.rsp);
663 state.fp = reinterpret_cast<void*>(thread_state.rbp);
664#elif V8_HOST_ARCH_ARM64
665 state.pc = reinterpret_cast<void*>(thread_state.pc);
666 state.sp = reinterpret_cast<void*>(thread_state.sp);
667 state.fp = reinterpret_cast<void*>(thread_state.r[29]);
668#endif
669 SampleStack(state);
670 }
671
672 zx_handle_close(suspend_token);
673}
