40void HeapObjectRange::iterator::AdvanceToNextObject() {
41 DCHECK_NE(cur_addr_, kNullAddress);
42 while (cur_addr_ != cur_end_) {
43 DCHECK_LT(cur_addr_, cur_end_);
44 Tagged<HeapObject> obj = HeapObject::FromAddress(cur_addr_);
45 cur_size_ = ALIGN_TO_ALLOCATION_ALIGNMENT(obj->Size(cage_base()));
46 DCHECK_LE(cur_addr_ + cur_size_, cur_end_);
47 if (IsFreeSpaceOrFiller(obj, cage_base())) {
48 cur_addr_ += cur_size_;
49 } else {
50 if (IsInstructionStream(obj, cage_base())) {
51 DCHECK_EQ(PageMetadata::FromHeapObject(obj)->owner_identity(),
52 CODE_SPACE);
53 DCHECK_CODEOBJECT_SIZE(cur_size_);
54 } else {
55 DCHECK_OBJECT_SIZE(cur_size_);
56 }
57 return;
58 }
59 }
60 cur_addr_ = kNullAddress;
61}

85template <bool during_sweep>
86size_t PagedSpaceBase::FreeInternal(Address start, size_t size_in_bytes) {
87 if (size_in_bytes == 0) return 0;
88 size_t wasted;
89 if (executable_) {
90 WritableJitPage jit_page(start, size_in_bytes);
91 WritableFreeSpace free_space = jit_page.FreeRange(start, size_in_bytes);
92 heap()->CreateFillerObjectAtBackground(free_space);
93 wasted = free_list_->Free(
94 free_space, during_sweep ? kDoNotLinkCategory : kLinkCategory);
95 } else {
96 WritableFreeSpace free_space =
97 WritableFreeSpace::ForNonExecutableMemory(start, size_in_bytes);
98 heap()->CreateFillerObjectAtBackground(free_space);
99 wasted = free_list_->Free(
100 free_space, during_sweep ? kDoNotLinkCategory : kLinkCategory);
101 }
102
103 if constexpr (!during_sweep) {
104 PageMetadata* page = PageMetadata::FromAddress(start);
105 accounting_stats_.DecreaseAllocatedBytes(size_in_bytes, page);
106 free_list()->increase_wasted_bytes(wasted);
107 }
108
109 DCHECK_GE(size_in_bytes, wasted);
110 return size_in_bytes - wasted;
111}
