105enum ElementsKind : uint8_t {
106 // The "fast" kind for elements that only contain SMI values. Must be first
107 // to make it possible to efficiently check maps for this kind.
108 PACKED_SMI_ELEMENTS,
109 HOLEY_SMI_ELEMENTS,
110
111 // The "fast" kind for tagged values. Must be second to make it possible to
112 // efficiently check maps for this and the PACKED_SMI_ELEMENTS kind
113 // together at once.
114 PACKED_ELEMENTS,
115 HOLEY_ELEMENTS,
116
117 // The "fast" kind for unwrapped, non-tagged double values.
118 PACKED_DOUBLE_ELEMENTS,
119 HOLEY_DOUBLE_ELEMENTS,
120
121 // The nonextensible kind for elements.
122 PACKED_NONEXTENSIBLE_ELEMENTS,
123 HOLEY_NONEXTENSIBLE_ELEMENTS,
124
125 // The sealed kind for elements.
126 PACKED_SEALED_ELEMENTS,
127 HOLEY_SEALED_ELEMENTS,
128
129 // The frozen kind for elements.
130 PACKED_FROZEN_ELEMENTS,
131 HOLEY_FROZEN_ELEMENTS,
132
133 // SharedArray elements kind. A FAST_SEALED_ELEMENTS variation useful to
134 // code specific paths for SharedArrays.
135 SHARED_ARRAY_ELEMENTS,
136
137 // The "slow" kind.
138 DICTIONARY_ELEMENTS,
139
140 // Elements kind of the "arguments" object (only in sloppy mode).
141 FAST_SLOPPY_ARGUMENTS_ELEMENTS,
142 SLOW_SLOPPY_ARGUMENTS_ELEMENTS,
143
144 // For string wrapper objects ("new String('...')"), the string's characters
145 // are overlaid onto a regular elements backing store.
146 FAST_STRING_WRAPPER_ELEMENTS,
147 SLOW_STRING_WRAPPER_ELEMENTS,
148
149// Fixed typed arrays.
150#define TYPED_ARRAY_ELEMENTS_KIND(Type, type, TYPE, ctype) TYPE##_ELEMENTS,
151 TYPED_ARRAYS(TYPED_ARRAY_ELEMENTS_KIND)
152 RAB_GSAB_TYPED_ARRAYS(TYPED_ARRAY_ELEMENTS_KIND)
153#undef TYPED_ARRAY_ELEMENTS_KIND
154
155 // WasmObject elements kind. The actual elements type is read from the
156 // respective WasmTypeInfo.
157 WASM_ARRAY_ELEMENTS,
158
159 // Sentinel ElementsKind for objects with no elements.
160 NO_ELEMENTS,
161
162 // Derived constants from ElementsKind.
163 FIRST_ELEMENTS_KIND = PACKED_SMI_ELEMENTS,
164 LAST_ELEMENTS_KIND = RAB_GSAB_FLOAT16_ELEMENTS,
165 FIRST_FAST_ELEMENTS_KIND = PACKED_SMI_ELEMENTS,
166 LAST_FAST_ELEMENTS_KIND = HOLEY_DOUBLE_ELEMENTS,
167 FIRST_FIXED_TYPED_ARRAY_ELEMENTS_KIND = UINT8_ELEMENTS,
168 LAST_FIXED_TYPED_ARRAY_ELEMENTS_KIND = FLOAT16_ELEMENTS,
169 FIRST_RAB_GSAB_FIXED_TYPED_ARRAY_ELEMENTS_KIND = RAB_GSAB_UINT8_ELEMENTS,
170 LAST_RAB_GSAB_FIXED_TYPED_ARRAY_ELEMENTS_KIND = RAB_GSAB_FLOAT16_ELEMENTS,
171 TERMINAL_FAST_ELEMENTS_KIND = HOLEY_ELEMENTS,
172 FIRST_ANY_NONEXTENSIBLE_ELEMENTS_KIND = PACKED_NONEXTENSIBLE_ELEMENTS,
173 LAST_ANY_NONEXTENSIBLE_ELEMENTS_KIND = SHARED_ARRAY_ELEMENTS,
174 FIRST_VALID_ATOMICS_TYPED_ARRAY_ELEMENTS_KIND = UINT8_ELEMENTS,
175 LAST_VALID_ATOMICS_TYPED_ARRAY_ELEMENTS_KIND = BIGINT64_ELEMENTS,
176
177// Alias for kSystemPointerSize-sized elements
178#ifdef V8_COMPRESS_POINTERS
179 SYSTEM_POINTER_ELEMENTS = PACKED_DOUBLE_ELEMENTS,
180#else
181 SYSTEM_POINTER_ELEMENTS = PACKED_ELEMENTS,
182#endif
183};

203inline constexpr int ElementsKindToShiftSize(ElementsKind elements_kind) {
204 switch (elements_kind) {
205 case UINT8_ELEMENTS:
206 case INT8_ELEMENTS:
207 case UINT8_CLAMPED_ELEMENTS:
208 case RAB_GSAB_UINT8_ELEMENTS:
209 case RAB_GSAB_INT8_ELEMENTS:
210 case RAB_GSAB_UINT8_CLAMPED_ELEMENTS:
211 return 0;
212 case UINT16_ELEMENTS:
213 case INT16_ELEMENTS:
214 case FLOAT16_ELEMENTS:
215 case RAB_GSAB_FLOAT16_ELEMENTS:
216 case RAB_GSAB_UINT16_ELEMENTS:
217 case RAB_GSAB_INT16_ELEMENTS:
218 return 1;
219 case UINT32_ELEMENTS:
220 case INT32_ELEMENTS:
221 case FLOAT32_ELEMENTS:
222 case RAB_GSAB_UINT32_ELEMENTS:
223 case RAB_GSAB_INT32_ELEMENTS:
224 case RAB_GSAB_FLOAT32_ELEMENTS:
225 return 2;
226 case PACKED_DOUBLE_ELEMENTS:
227 case HOLEY_DOUBLE_ELEMENTS:
228 case FLOAT64_ELEMENTS:
229 case BIGINT64_ELEMENTS:
230 case BIGUINT64_ELEMENTS:
231 case RAB_GSAB_FLOAT64_ELEMENTS:
232 case RAB_GSAB_BIGINT64_ELEMENTS:
233 case RAB_GSAB_BIGUINT64_ELEMENTS:
234 return 3;
235 case PACKED_SMI_ELEMENTS:
236 case PACKED_ELEMENTS:
237 case PACKED_FROZEN_ELEMENTS:
238 case PACKED_SEALED_ELEMENTS:
239 case PACKED_NONEXTENSIBLE_ELEMENTS:
240 case HOLEY_SMI_ELEMENTS:
241 case HOLEY_ELEMENTS:
242 case HOLEY_FROZEN_ELEMENTS:
243 case HOLEY_SEALED_ELEMENTS:
244 case HOLEY_NONEXTENSIBLE_ELEMENTS:
245 case DICTIONARY_ELEMENTS:
246 case FAST_SLOPPY_ARGUMENTS_ELEMENTS:
247 case SLOW_SLOPPY_ARGUMENTS_ELEMENTS:
248 case FAST_STRING_WRAPPER_ELEMENTS:
249 case SLOW_STRING_WRAPPER_ELEMENTS:
250 case SHARED_ARRAY_ELEMENTS:
251 return kTaggedSizeLog2;
252 case WASM_ARRAY_ELEMENTS:
253 case NO_ELEMENTS:
254 CONSTEXPR_UNREACHABLE();
255 }
256 CONSTEXPR_UNREACHABLE();
257}

498inline bool UnionElementsKindUptoPackedness(ElementsKind* a_out,
499 ElementsKind b) {
500 // Assert that the union of two ElementKinds can be computed via std::max.
501 static_assert(PACKED_SMI_ELEMENTS < HOLEY_SMI_ELEMENTS,
502 "ElementsKind union not computable via std::max.");
503 static_assert(PACKED_ELEMENTS < HOLEY_ELEMENTS,
504 "ElementsKind union not computable via std::max.");

506 "ElementsKind union not computable via std::max.");
507 ElementsKind a = *a_out;
508 switch (a) {
509 case HOLEY_SMI_ELEMENTS:
510 case PACKED_SMI_ELEMENTS:
511 if (b == PACKED_SMI_ELEMENTS || b == HOLEY_SMI_ELEMENTS) {
512 *a_out = std::max(a, b);
513 return true;
514 }
515 break;
516 case PACKED_ELEMENTS:
517 case HOLEY_ELEMENTS:
518 if (b == PACKED_ELEMENTS || b == HOLEY_ELEMENTS) {
519 *a_out = std::max(a, b);
520 return true;
521 }
522 break;
523 case PACKED_DOUBLE_ELEMENTS:
524 case HOLEY_DOUBLE_ELEMENTS:
525 if (b == PACKED_DOUBLE_ELEMENTS || b == HOLEY_DOUBLE_ELEMENTS) {
526 *a_out = std::max(a, b);
527 return true;
528 }
529 break;
530 default:
531 break;
532 }
533 return false;
534}
