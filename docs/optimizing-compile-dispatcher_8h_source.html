26// Circular queue of incoming recompilation tasks (including OSR).
27class V8_EXPORT OptimizingCompileDispatcherQueue {
28 public:
29 inline bool IsAvailable() {
30 base::SpinningMutexGuard access(&mutex_);
31 return length_ < capacity_;
32 }
33
34 inline int Length() {
35 base::SpinningMutexGuard access_queue(&mutex_);
36 return length_;
37 }
38
39 explicit OptimizingCompileDispatcherQueue(int capacity)
40 : capacity_(capacity), length_(0), shift_(0) {
41 queue_ = NewArray<TurbofanCompilationJob*>(capacity_);
42 }
43
44 ~OptimizingCompileDispatcherQueue() { DeleteArray(queue_); }
45
46 TurbofanCompilationJob* Dequeue() {
47 base::SpinningMutexGuard access(&mutex_);
48 if (length_ == 0) return nullptr;
49 TurbofanCompilationJob* job = queue_[QueueIndex(0)];
50 DCHECK_NOT_NULL(job);
51 shift_ = QueueIndex(1);
52 length_--;
53 return job;
54 }
55
56 void Enqueue(TurbofanCompilationJob* job) {
57 base::SpinningMutexGuard access(&mutex_);
58 DCHECK_LT(length_, capacity_);
59 queue_[QueueIndex(length_)] = job;
60 length_++;
61 }
62
63 void Flush(Isolate* isolate);
64
65 void Prioritize(Tagged<SharedFunctionInfo> function);
66
67 private:
68 inline int QueueIndex(int i) {
69 int result = (i + shift_) % capacity_;
70 DCHECK_LE(0, result);
71 DCHECK_LT(result, capacity_);
72 return result;
73 }
74
75 TurbofanCompilationJob** queue_;
76 int capacity_;
77 int length_;
78 int shift_;
79 base::SpinningMutex mutex_;
80};
