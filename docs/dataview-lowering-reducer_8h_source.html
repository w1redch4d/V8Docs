19 OpIndex BuildReverseBytes(ExternalArrayType type, OpIndex value) {
20 switch (type) {
21 case kExternalInt8Array:
22 case kExternalUint8Array:
23 case kExternalUint8ClampedArray:
24 return value;
25 case kExternalInt16Array:
26 return __ Word32ShiftRightArithmetic(__ Word32ReverseBytes(value), 16);
27 case kExternalUint16Array:
28 return __ Word32ShiftRightLogical(__ Word32ReverseBytes(value), 16);
29 case kExternalInt32Array:
30 case kExternalUint32Array:
31 return __ Word32ReverseBytes(value);
32 case kExternalFloat32Array: {
33 V<Word32> bytes = __ BitcastFloat32ToWord32(value);
34 V<Word32> reversed = __ Word32ReverseBytes(bytes);
35 return __ BitcastWord32ToFloat32(reversed);
36 }
37 case kExternalFloat64Array: {
38 if constexpr (Is64()) {
39 V<Word64> bytes = __ BitcastFloat64ToWord64(value);
40 V<Word64> reversed = __ Word64ReverseBytes(bytes);
41 return __ BitcastWord64ToFloat64(reversed);
42 } else {
43 V<Word32> reversed_lo =
44 __ Word32ReverseBytes(__ Float64ExtractLowWord32(value));
45 V<Word32> reversed_hi =
46 __ Word32ReverseBytes(__ Float64ExtractHighWord32(value));
47 return __ BitcastWord32PairToFloat64(reversed_lo, reversed_hi);
48 }
49 }
50 case kExternalBigInt64Array:
51 case kExternalBigUint64Array:
52 return __ Word64ReverseBytes(value);
53 case kExternalFloat16Array:
54 UNIMPLEMENTED();
55 }
56 }

58 OpIndex REDUCE(LoadDataViewElement)(V<Object> object, V<WordPtr> storage,
59 V<WordPtr> index,
60 V<Word32> is_little_endian,
61 ExternalArrayType element_type) {
62 const MachineType machine_type =
63 AccessBuilder::ForTypedArrayElement(element_type, true).machine_type;
64 const MemoryRepresentation memory_rep =
65 MemoryRepresentation::FromMachineType(machine_type);
66
67 OpIndex value =
68 __ Load(storage, index,
69 LoadOp::Kind::MaybeUnaligned(memory_rep).NotLoadEliminable(),
70 memory_rep);
71
72 Variable result = Asm().NewLoopInvariantVariable(
73 RegisterRepresentationForArrayType(element_type));
74 IF (is_little_endian) {
75#if V8_TARGET_LITTLE_ENDIAN
76 Asm().SetVariable(result, value);
77#else
78 Asm().SetVariable(result, BuildReverseBytes(element_type, value));
79#endif // V8_TARGET_LITTLE_ENDIAN
80 } ELSE {
81#if V8_TARGET_LITTLE_ENDIAN
82 Asm().SetVariable(result, BuildReverseBytes(element_type, value));
83#else
84 Asm().SetVariable(result, value);
85#endif // V8_TARGET_LITTLE_ENDIAN
86 }
87
88 // We need to keep the {object} (either the JSArrayBuffer or the JSDataView)
89 // alive so that the GC will not release the JSArrayBuffer (if there's any)
90 // as long as we are still operating on it.
91 __ Retain(object);
92 return Asm().GetVariable(result);
93 }
94
95 OpIndex REDUCE(StoreDataViewElement)(V<Object> object, V<WordPtr> storage,
96 V<WordPtr> index, OpIndex value,
97 V<Word32> is_little_endian,
98 ExternalArrayType element_type) {
99 const MachineType machine_type =
100 AccessBuilder::ForTypedArrayElement(element_type, true).machine_type;
101
102 Variable value_to_store = Asm().NewLoopInvariantVariable(
103 RegisterRepresentationForArrayType(element_type));
104 IF (is_little_endian) {
105#if V8_TARGET_LITTLE_ENDIAN
106 Asm().SetVariable(value_to_store, value);
107#else
108 Asm().SetVariable(value_to_store, BuildReverseBytes(element_type, value));
109#endif // V8_TARGET_LITTLE_ENDIAN
110 } ELSE {
111#if V8_TARGET_LITTLE_ENDIAN
112 Asm().SetVariable(value_to_store, BuildReverseBytes(element_type, value));
113#else
114 Asm().SetVariable(value_to_store, value);
115#endif // V8_TARGET_LITTLE_ENDIAN
116 }
117
118 const MemoryRepresentation memory_rep =
119 MemoryRepresentation::FromMachineType(machine_type);
120 __ Store(storage, index, Asm().GetVariable(value_to_store),
121 StoreOp::Kind::MaybeUnaligned(memory_rep).NotLoadEliminable(),
122 memory_rep, WriteBarrierKind::kNoWriteBarrier);
123
124 // We need to keep the {object} (either the JSArrayBuffer or the JSDataView)
125 // alive so that the GC will not release the JSArrayBuffer (if there's any)
126 // as long as we are still operating on it.
127 __ Retain(object);
128 return {};
129 }
130};
