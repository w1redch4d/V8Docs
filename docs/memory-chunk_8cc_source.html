69uint32_t MemoryChunk::MetadataTableIndex(Address chunk_address) {
70 uint32_t index;
71 if (V8HeapCompressionScheme::GetPtrComprCageBaseAddress(chunk_address) ==
72 V8HeapCompressionScheme::base()) {
73 static_assert(kPtrComprCageReservationSize == kPtrComprCageBaseAlignment);
74 Tagged_t offset = V8HeapCompressionScheme::CompressAny(chunk_address);
75 DCHECK_LT(offset >> kPageSizeBits, MemoryChunkConstants::kPagesInMainCage);
76 index = MemoryChunkConstants::kMainCageMetadataOffset +
77 (offset >> kPageSizeBits);
78 } else if (TrustedRange::GetProcessWideTrustedRange()->region().contains(
79 chunk_address)) {
80 Tagged_t offset = TrustedSpaceCompressionScheme::CompressAny(chunk_address);
81 DCHECK_LT(offset >> kPageSizeBits,
82 MemoryChunkConstants::kPagesInTrustedCage);
83 index = MemoryChunkConstants::kTrustedSpaceMetadataOffset +
84 (offset >> kPageSizeBits);
85 } else {
86 CodeRange* code_range = IsolateGroup::current()->GetCodeRange();
87 DCHECK(code_range->region().contains(chunk_address));
88 uint32_t offset = static_cast<uint32_t>(chunk_address - code_range->base());
89 DCHECK_LT(offset >> kPageSizeBits, MemoryChunkConstants::kPagesInCodeCage);
90 index = MemoryChunkConstants::kCodeRangeMetadataOffset +
91 (offset >> kPageSizeBits);
92 }
93 DCHECK_LT(index, MemoryChunkConstants::kMetadataPointerTableSize);
94 return index;
95}

99void MemoryChunk::InitializationMemoryFence() {
100 base::SeqCst_MemoryFence();
101
102#ifdef THREAD_SANITIZER
103 // Since TSAN does not process memory fences, we use the following annotation
104 // to tell TSAN that there is no data race when emitting a
105 // InitializationMemoryFence. Note that the other thread still needs to
106 // perform MutablePageMetadata::synchronized_heap().
107 Metadata()->SynchronizedHeapStore();
108#ifndef V8_ENABLE_SANDBOX
109 base::Release_Store(reinterpret_cast<base::AtomicWord*>(&metadata_),
110 reinterpret_cast<base::AtomicWord>(metadata_));
111#else
112 MemoryChunkMetadata** metadata_pointer_table = MetadataTableAddress();
113 static_assert(sizeof(base::AtomicWord) == sizeof(metadata_pointer_table[0]));
114 static_assert(sizeof(base::Atomic32) == sizeof(metadata_index_));
115 base::Release_Store(reinterpret_cast<base::AtomicWord*>(
116 &metadata_pointer_table[metadata_index_]),
117 reinterpret_cast<base::AtomicWord>(
118 metadata_pointer_table[metadata_index_]));
119 base::Release_Store(reinterpret_cast<base::Atomic32*>(&metadata_index_),
120 metadata_index_);
121#endif
122#endif
123}

127void MemoryChunk::SynchronizedLoad() const {
128#ifndef V8_ENABLE_SANDBOX
129 MemoryChunkMetadata* metadata = reinterpret_cast<MemoryChunkMetadata*>(
130 base::Acquire_Load(reinterpret_cast<base::AtomicWord*>(
131 &(const_cast<MemoryChunk*>(this)->metadata_))));
132#else
133 MemoryChunkMetadata** metadata_pointer_table = MetadataTableAddress();
134 static_assert(sizeof(base::AtomicWord) == sizeof(metadata_pointer_table[0]));
135 static_assert(sizeof(base::Atomic32) == sizeof(metadata_index_));
136 uint32_t metadata_index =
137 base::Acquire_Load(reinterpret_cast<base::Atomic32*>(
138 &(const_cast<MemoryChunk*>(this)->metadata_index_)));
139 MemoryChunkMetadata* metadata = reinterpret_cast<MemoryChunkMetadata*>(
140 base::Acquire_Load(reinterpret_cast<base::AtomicWord*>(
141 &metadata_pointer_table[metadata_index])));
142#endif
143 metadata->SynchronizedHeapLoad();
144}

198MemoryChunk::MainThreadFlags MemoryChunk::OldGenerationPageFlags(
199 MarkingMode marking_mode, AllocationSpace space) {
200 MainThreadFlags flags_to_set = NO_FLAGS;
201
202 if (!v8_flags.sticky_mark_bits || (space != OLD_SPACE)) {
203 flags_to_set |= MemoryChunk::CONTAINS_ONLY_OLD;
204 }
205
206 if (marking_mode == MarkingMode::kMajorMarking) {
207 flags_to_set |= MemoryChunk::POINTERS_TO_HERE_ARE_INTERESTING |
208 MemoryChunk::POINTERS_FROM_HERE_ARE_INTERESTING |
209 MemoryChunk::INCREMENTAL_MARKING |
210 MemoryChunk::IS_MAJOR_GC_IN_PROGRESS;
211 } else if (IsAnySharedSpace(space)) {
212 // We need to track pointers into the SHARED_SPACE for OLD_TO_SHARED.
213 flags_to_set |= MemoryChunk::POINTERS_TO_HERE_ARE_INTERESTING;
214 } else {
215 flags_to_set |= MemoryChunk::POINTERS_FROM_HERE_ARE_INTERESTING;
216 if (marking_mode == MarkingMode::kMinorMarking) {
217 flags_to_set |= MemoryChunk::INCREMENTAL_MARKING;
218 }
219 }
220
221 return flags_to_set;
222}

225MemoryChunk::MainThreadFlags MemoryChunk::YoungGenerationPageFlags(
226 MarkingMode marking_mode) {
227 MainThreadFlags flags = MemoryChunk::POINTERS_TO_HERE_ARE_INTERESTING;
228 if (marking_mode != MarkingMode::kNoMarking) {
229 flags |= MemoryChunk::POINTERS_FROM_HERE_ARE_INTERESTING;
230 flags |= MemoryChunk::INCREMENTAL_MARKING;

232 flags |= MemoryChunk::IS_MAJOR_GC_IN_PROGRESS;

234 }

238void MemoryChunk::SetOldGenerationPageFlags(MarkingMode marking_mode,
239 AllocationSpace space) {
240 MainThreadFlags flags_to_set = OldGenerationPageFlags(marking_mode, space);
241 MainThreadFlags flags_to_clear = NO_FLAGS;
242
243 if (marking_mode != MarkingMode::kMajorMarking) {
244 if (IsAnySharedSpace(space)) {
245 // No need to track OLD_TO_NEW or OLD_TO_SHARED within the shared space.
246 flags_to_clear |= MemoryChunk::POINTERS_FROM_HERE_ARE_INTERESTING |
247 MemoryChunk::INCREMENTAL_MARKING;
248 } else {
249 flags_to_clear |= MemoryChunk::POINTERS_TO_HERE_ARE_INTERESTING;
250 if (marking_mode != MarkingMode::kMinorMarking) {
251 flags_to_clear |= MemoryChunk::INCREMENTAL_MARKING;
252 }
253 }
254 }
255
256 SetFlagsUnlocked(flags_to_set, flags_to_set);
257 ClearFlagsUnlocked(flags_to_clear);
258}

262 MainThreadFlags flags_to_clear = NO_FLAGS;

273#ifdef V8_ENABLE_SANDBOX
274bool MemoryChunk::SandboxSafeInReadOnlySpace() const {
275 // For the sandbox only flags from writable pages can be corrupted so we can
276 // use the flag check as a fast path in this case.
277 // It also helps making TSAN happy, since it doesn't like the way we
278 // initialize the MemoryChunks.
279 // (See MemoryChunkMetadata::SynchronizedHeapLoad).
280 if (!InReadOnlySpace()) {
281 return false;
282 }
283
284 // When the sandbox is enabled, only the ReadOnlyPageMetadata are stored
285 // inline in the MemoryChunk.
286 // ReadOnlyPageMetadata::ChunkAddress() is a special version that boils down
287 // to `metadata_address - kMemoryChunkHeaderSize`.
288 MemoryChunkMetadata** metadata_pointer_table = MetadataTableAddress();
289 MemoryChunkMetadata* metadata = metadata_pointer_table
290 [metadata_index_ & MemoryChunkConstants::kMetadataPointerTableSizeMask];
291 SBXCHECK_EQ(
292 static_cast<const ReadOnlyPageMetadata*>(metadata)->ChunkAddress(),
293 address());
294
295 return true;
296}
297#endif
