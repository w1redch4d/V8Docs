52void WritableRelocInfo::apply(intptr_t delta) {
53 // absolute code pointer inside code object moves with the code object.
54 if (IsInternalReference(rmode_)) {
55 // Jump table entry
56 Address target = Memory<Address>(pc_);
57 jit_allocation_.WriteValue(pc_, target + delta);
58 } else {
59 // mov sequence
60 DCHECK(IsInternalReferenceEncoded(rmode_));
61 Address target = Assembler::target_address_at(pc_, constant_pool_);
62 Assembler::set_target_address_at(pc_, constant_pool_, target + delta,
63 &jit_allocation_, SKIP_ICACHE_FLUSH);
64 }
65}

88Address RelocInfo::target_address_address() {
89 DCHECK(HasTargetAddressAddress());
90
91 if (V8_EMBEDDED_CONSTANT_POOL_BOOL &&
92 Assembler::IsConstantPoolLoadStart(pc_)) {
93 // We return the PC for embedded constant pool since this function is used
94 // by the serializer and expects the address to reside within the code
95 // object.
96 return pc_;
97 }
98
99 // Read the address of the word containing the target_address in an
100 // instruction stream.
101 // The only architecture-independent user of this function is the serializer.
102 // The serializer uses it to find out how many raw bytes of instruction to
103 // output before the next target.
104 // For an instruction like LIS/ORI where the target bits are mixed into the
105 // instruction bits, the size of the target will be zero, indicating that the
106 // serializer should not step forward in memory after a target is resolved
107 // and written.
108 return pc_;
109}

248 if (count == 0) {
249 // Reset

256// Fetch the 32bit value from the FIXED_SEQUENCE lis/ori
257Address Assembler::target_address_at(Address pc, Address constant_pool) {
258 if (V8_EMBEDDED_CONSTANT_POOL_BOOL && constant_pool) {
259 ConstantPoolEntry::Access access;
260 if (IsConstantPoolLoadStart(pc, &access))
261 return Memory<Address>(target_constant_pool_address_at(
262 pc, constant_pool, access, ConstantPoolEntry::INTPTR));
263 }
264
265 Instr instr1 = instr_at(pc);
266 Instr instr2 = instr_at(pc + kInstrSize);
267 // Interpret 2 instructions generated by lis/ori
268 if (IsLis(instr1) && IsOri(instr2)) {
269 Instr instr4 = instr_at(pc + (3 * kInstrSize));
270 Instr instr5 = instr_at(pc + (4 * kInstrSize));
271 // Assemble the 64 bit value.
272 uint64_t hi = (static_cast<uint32_t>((instr1 & kImm16Mask) << 16) |
273 static_cast<uint32_t>(instr2 & kImm16Mask));
274 uint64_t lo = (static_cast<uint32_t>((instr4 & kImm16Mask) << 16) |
275 static_cast<uint32_t>(instr5 & kImm16Mask));
276 return static_cast<Address>((hi << 32) | lo);
277 }
278
279 UNREACHABLE();
280}

282const uint32_t kLoadIntptrOpcode = LD;
283
284// Constant pool load sequence detection:
285// 1) REGULAR access:
286// load <dst>, kConstantPoolRegister + <offset>
287//
288// 2) OVERFLOWED access:
289// addis <scratch>, kConstantPoolRegister, <offset_high>
290// load <dst>, <scratch> + <offset_low>
291bool Assembler::IsConstantPoolLoadStart(Address pc,
292 ConstantPoolEntry::Access* access) {
293 Instr instr = instr_at(pc);
294 uint32_t opcode = instr & kOpcodeMask;
295 if (GetRA(instr) != kConstantPoolRegister) return false;
296 bool overflowed = (opcode == ADDIS);
297#ifdef DEBUG
298 if (overflowed) {
299 opcode = instr_at(pc + kInstrSize) & kOpcodeMask;
300 }
301 DCHECK(opcode == kLoadIntptrOpcode || opcode == LFD);
302#endif
303 if (access) {
304 *access = (overflowed ? ConstantPoolEntry::OVERFLOWED
305 : ConstantPoolEntry::REGULAR);
306 }
307 return true;
308}
309
310bool Assembler::IsConstantPoolLoadEnd(Address pc,
311 ConstantPoolEntry::Access* access) {
312 Instr instr = instr_at(pc);
313 uint32_t opcode = instr & kOpcodeMask;
314 bool overflowed = false;
315 if (!(opcode == kLoadIntptrOpcode || opcode == LFD)) return false;
316 if (GetRA(instr) != kConstantPoolRegister) {
317 instr = instr_at(pc - kInstrSize);
318 opcode = instr & kOpcodeMask;
319 if ((opcode != ADDIS) || GetRA(instr) != kConstantPoolRegister) {
320 return false;
321 }
322 overflowed = true;
323 }
324 if (access) {
325 *access = (overflowed ? ConstantPoolEntry::OVERFLOWED
326 : ConstantPoolEntry::REGULAR);
327 }
328 return true;
329}

331int Assembler::GetConstantPoolOffset(Address pc,
332 ConstantPoolEntry::Access access,
333 ConstantPoolEntry::Type type) {
334 bool overflowed = (access == ConstantPoolEntry::OVERFLOWED);
335#ifdef DEBUG
336 ConstantPoolEntry::Access access_check =
337 static_cast<ConstantPoolEntry::Access>(-1);
338 DCHECK(IsConstantPoolLoadStart(pc, &access_check));
339 DCHECK(access_check == access);
340#endif
341 int offset;
342 if (overflowed) {
343 offset = (instr_at(pc) & kImm16Mask) << 16;
344 offset += SIGN_EXT_IMM16(instr_at(pc + kInstrSize) & kImm16Mask);
345 DCHECK(!is_int16(offset));
346 } else {
347 offset = SIGN_EXT_IMM16((instr_at(pc) & kImm16Mask));
348 }
349 return offset;
350}

352void Assembler::PatchConstantPoolAccessInstruction(
353 int pc_offset, int offset, ConstantPoolEntry::Access access,
354 ConstantPoolEntry::Type type) {
355 Address pc = reinterpret_cast<Address>(buffer_start_) + pc_offset;
356 bool overflowed = (access == ConstantPoolEntry::OVERFLOWED);
357 CHECK(overflowed != is_int16(offset));
358#ifdef DEBUG
359 ConstantPoolEntry::Access access_check =
360 static_cast<ConstantPoolEntry::Access>(-1);
361 DCHECK(IsConstantPoolLoadStart(pc, &access_check));
362 DCHECK(access_check == access);
363#endif
364 if (overflowed) {
365 int hi_word = static_cast<int>(offset >> 16);
366 int lo_word = static_cast<int>(offset & 0xffff);
367 if (lo_word & 0x8000) hi_word++;
368
369 Instr instr1 = instr_at(pc);
370 Instr instr2 = instr_at(pc + kInstrSize);
371 instr1 &= ~kImm16Mask;
372 instr1 |= (hi_word & kImm16Mask);
373 instr2 &= ~kImm16Mask;
374 instr2 |= (lo_word & kImm16Mask);
375 instr_at_put(pc, instr1);
376 instr_at_put(pc + kInstrSize, instr2);
377 } else {
378 Instr instr = instr_at(pc);
379 instr &= ~kImm16Mask;
380 instr |= (offset & kImm16Mask);
381 instr_at_put(pc, instr);
382 }
383}

410// This code assumes the FIXED_SEQUENCE of lis/ori
411void Assembler::set_target_address_at(Address pc, Address constant_pool,
412 Address target,
413 WritableJitAllocation* jit_allocation,
414 ICacheFlushMode icache_flush_mode) {
415 if (V8_EMBEDDED_CONSTANT_POOL_BOOL && constant_pool) {
416 ConstantPoolEntry::Access access;
417 if (IsConstantPoolLoadStart(pc, &access)) {
418 if (jit_allocation) {
419 jit_allocation->WriteUnalignedValue<Address>(
420 target_constant_pool_address_at(pc, constant_pool, access,
421 ConstantPoolEntry::INTPTR),
422 target);
423 } else {
424 Memory<Address>(target_constant_pool_address_at(
425 pc, constant_pool, access, ConstantPoolEntry::INTPTR)) = target;
426 }
427 return;
428 }
429 }
430
431 Instr instr1 = instr_at(pc);
432 Instr instr2 = instr_at(pc + kInstrSize);
433 // Interpret 2 instructions generated by lis/ori
434 if (IsLis(instr1) && IsOri(instr2)) {
435 Instr instr4 = instr_at(pc + (3 * kInstrSize));
436 Instr instr5 = instr_at(pc + (4 * kInstrSize));
437 // Needs to be fixed up when mov changes to handle 64-bit values.
438 uint32_t* p = reinterpret_cast<uint32_t*>(pc);
439 uintptr_t itarget = static_cast<uintptr_t>(target);
440
441 instr5 &= ~kImm16Mask;
442 instr5 |= itarget & kImm16Mask;
443 itarget = itarget >> 16;
444
445 instr4 &= ~kImm16Mask;
446 instr4 |= itarget & kImm16Mask;
447 itarget = itarget >> 16;
448
449 instr2 &= ~kImm16Mask;
450 instr2 |= itarget & kImm16Mask;
451 itarget = itarget >> 16;
452
453 instr1 &= ~kImm16Mask;
454 instr1 |= itarget & kImm16Mask;
455 itarget = itarget >> 16;
456
457 if (jit_allocation) {
458 jit_allocation->WriteUnalignedValue(reinterpret_cast<Address>(&p[0]),
459 instr1);
460 jit_allocation->WriteUnalignedValue(reinterpret_cast<Address>(&p[1]),
461 instr2);
462 jit_allocation->WriteUnalignedValue(reinterpret_cast<Address>(&p[3]),
463 instr4);
464 jit_allocation->WriteUnalignedValue(reinterpret_cast<Address>(&p[4]),
465 instr5);
466 } else {
467 *p = instr1;
468 *(p + 1) = instr2;
469 *(p + 3) = instr4;
470 *(p + 4) = instr5;
471 }
472 if (icache_flush_mode != SKIP_ICACHE_FLUSH) {
473 FlushInstructionCache(p, 5 * kInstrSize);
474 }
475 return;
476 }
477 UNREACHABLE();
478}
