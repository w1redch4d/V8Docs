285 if (IsReal()) {

291size_t SparseInputMask::InputIterator::AdvanceToNextRealOrEnd() {
292 DCHECK_NE(bit_mask_, SparseInputMask::kDenseBitMask);
293
294 size_t count = base::bits::CountTrailingZeros(bit_mask_);
295 bit_mask_ >>= count;
296 DCHECK(IsReal() || IsEnd());
297 return count;
298}
