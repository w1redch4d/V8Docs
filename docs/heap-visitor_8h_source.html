181// The base class for visitors that need to dispatch on object type. The default
182// behavior of all visit functions is to iterate body of the given object using
183// the BodyDescriptor of the object.
184//
185// The visit functions return the size of the object.
186//
187// This class is intended to be used in the following way:
188//
189// class SomeVisitor : public HeapVisitor<SomeVisitor> {
190// ...
191// }

203 requires(!ConcreteVisitor::UsePrecomputedObjectSize());
204

206 int object_size)

209 protected:
210 V8_INLINE size_t Visit(Tagged<Map> map, Tagged<HeapObject> object,
