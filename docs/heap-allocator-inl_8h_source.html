71template <AllocationType type>
72V8_WARN_UNUSED_RESULT V8_INLINE AllocationResult HeapAllocator::AllocateRaw(
73 int size_in_bytes, AllocationOrigin origin, AllocationAlignment alignment) {
74 DCHECK(!heap_->IsInGC());
75 DCHECK(AllowHandleAllocation::IsAllowed());
76 DCHECK(AllowHeapAllocation::IsAllowed());
77 CHECK(AllowHeapAllocationInRelease::IsAllowed());
78 DCHECK(local_heap_->IsRunning());
79#if V8_ENABLE_WEBASSEMBLY
80 trap_handler::AssertThreadNotInWasm();
81#endif
82#if DEBUG
83 local_heap_->VerifyCurrent();
84#endif
85
86 if (v8_flags.single_generation.value() && type == AllocationType::kYoung) {
87 return AllocateRaw(size_in_bytes, AllocationType::kOld, origin, alignment);
88 }
89
90#ifdef V8_ENABLE_ALLOCATION_TIMEOUT
91 if (V8_UNLIKELY(allocation_timeout_.has_value()) &&
92 ReachedAllocationTimeout()) {
93 return AllocationResult::Failure();
94 }
95#endif // V8_ENABLE_ALLOCATION_TIMEOUT
96
97#ifdef DEBUG
98 IncrementObjectCounters();
99#endif // DEBUG
100
101 if (heap_->CanSafepoint()) {
102 local_heap_->Safepoint();
103 }
104
105 const size_t large_object_threshold = heap_->MaxRegularHeapObjectSize(type);
106 const bool large_object =
107 static_cast<size_t>(size_in_bytes) > large_object_threshold;
108
109 Tagged<HeapObject> object;
110 AllocationResult allocation;
111
112 if (V8_UNLIKELY(large_object)) {
113 allocation =
114 AllocateRawLargeInternal(size_in_bytes, type, origin, alignment);
115 } else {
116 switch (type) {
117 case AllocationType::kYoung:
118 allocation =
119 new_space_allocator_->AllocateRaw(size_in_bytes, alignment, origin);
120 break;
121 case AllocationType::kMap:
122 case AllocationType::kOld:
123 allocation =
124 old_space_allocator_->AllocateRaw(size_in_bytes, alignment, origin);
125 DCHECK_IMPLIES(v8_flags.sticky_mark_bits && !allocation.IsFailure(),
126 heap_->marking_state()->IsMarked(allocation.ToObject()));
127 break;
128 case AllocationType::kCode: {
129 DCHECK_EQ(alignment, AllocationAlignment::kTaggedAligned);
130 DCHECK(AllowCodeAllocation::IsAllowed());
131 allocation = code_space_allocator_->AllocateRaw(
132 size_in_bytes, AllocationAlignment::kTaggedAligned, origin);
133 break;
134 }
135 case AllocationType::kReadOnly:
136 DCHECK(read_only_space()->writable());
137 DCHECK_EQ(AllocationOrigin::kRuntime, origin);
138 allocation = read_only_space()->AllocateRaw(size_in_bytes, alignment);
139 break;
140 case AllocationType::kSharedMap:
141 case AllocationType::kSharedOld:
142 allocation = shared_space_allocator_->AllocateRaw(size_in_bytes,
143 alignment, origin);
144 break;
145 case AllocationType::kTrusted:
146 allocation = trusted_space_allocator_->AllocateRaw(size_in_bytes,
147 alignment, origin);
148 break;
149 case AllocationType::kSharedTrusted:
150 allocation = shared_trusted_space_allocator_->AllocateRaw(
151 size_in_bytes, alignment, origin);
152 break;
153 }
154 }
155
156 if (allocation.To(&object)) {
157 if (heap::ShouldZapGarbage() && AllocationType::kCode == type) {
158 heap::ZapCodeBlock(object.address(), size_in_bytes);
159 }
160
161 if (local_heap_->is_main_thread()) {
162 for (auto& tracker : heap_->allocation_trackers_) {
163 tracker->AllocationEvent(object.address(), size_in_bytes);
164 }
165 }
166 }
167
168 return allocation;
169}

171AllocationResult HeapAllocator::AllocateRaw(int size_in_bytes,
172 AllocationType type,
173 AllocationOrigin origin,
174 AllocationAlignment alignment) {
175 switch (type) {
176 case AllocationType::kYoung:
177 return AllocateRaw<AllocationType::kYoung>(size_in_bytes, origin,
178 alignment);
179 case AllocationType::kOld:
180 return AllocateRaw<AllocationType::kOld>(size_in_bytes, origin,
181 alignment);
182 case AllocationType::kCode:
183 return AllocateRaw<AllocationType::kCode>(size_in_bytes, origin,
184 alignment);
185 case AllocationType::kMap:
186 return AllocateRaw<AllocationType::kMap>(size_in_bytes, origin,
187 alignment);
188 case AllocationType::kReadOnly:
189 return AllocateRaw<AllocationType::kReadOnly>(size_in_bytes, origin,
190 alignment);
191 case AllocationType::kSharedMap:
192 return AllocateRaw<AllocationType::kSharedMap>(size_in_bytes, origin,
193 alignment);
194 case AllocationType::kSharedOld:
195 return AllocateRaw<AllocationType::kSharedOld>(size_in_bytes, origin,
196 alignment);

207template <HeapAllocator::AllocationRetryMode mode>
208V8_WARN_UNUSED_RESULT V8_INLINE Tagged<HeapObject>
209HeapAllocator::AllocateRawWith(int size, AllocationType allocation,
210 AllocationOrigin origin,
211 AllocationAlignment alignment) {
212 AllocationResult result;
213 Tagged<HeapObject> object;
214 size = ALIGN_TO_ALLOCATION_ALIGNMENT(size);
215 if (allocation == AllocationType::kYoung) {
216 result = AllocateRaw<AllocationType::kYoung>(size, origin, alignment);
217 if (result.To(&object)) {
218 return object;
219 }
220 } else if (allocation == AllocationType::kOld) {
221 result = AllocateRaw<AllocationType::kOld>(size, origin, alignment);
222 if (result.To(&object)) {
223 return object;
224 }
225 }
226 switch (mode) {
227 case kLightRetry:
228 result = AllocateRawWithLightRetrySlowPath(size, allocation, origin,
229 alignment);
230 break;
231 case kRetryOrFail:
232 result = AllocateRawWithRetryOrFailSlowPath(size, allocation, origin,
233 alignment);
234 break;
235 }
236 if (result.To(&object)) {
237 return object;
238 }
239 return HeapObject();
240}
