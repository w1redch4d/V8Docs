436 Reduction UpdateType(Node* node, Type current) {
437 if (NodeProperties::IsTyped(node)) {
438 // Widen the type of a previously typed node.
439 Type previous = NodeProperties::GetType(node);
440 if (node->opcode() == IrOpcode::kPhi ||
441 node->opcode() == IrOpcode::kInductionVariablePhi) {
442 // Speed up termination in the presence of range types:
443 current = Weaken(node, current, previous);
444 }
445
446 if (V8_UNLIKELY(!previous.Is(current))) {
447 AllowHandleDereference allow;
448 std::ostringstream ostream;
449 node->Print(ostream);
450 FATAL("UpdateType error for node %s", ostream.str().c_str());
451 }
452
453 NodeProperties::SetType(node, current);
454 if (!current.Is(previous)) {
455 // If something changed, revisit all uses.
456 return Changed(node);
457 }
458 return NoChange();
459 } else {
460 // No previous type, simply update the type.
461 NodeProperties::SetType(node, current);
462 return Changed(node);
463 }
464 }

469void Typer::Run(const NodeVector& roots,
470 LoopVariableOptimizer* induction_vars) {
471 if (induction_vars != nullptr) {
472 induction_vars->ChangeToInductionVariablePhis();
473 }
474 Visitor visitor(this, induction_vars);
475 GraphReducer graph_reducer(zone(), graph(), tick_counter_, broker());
476 graph_reducer.AddReducer(&visitor);
477 for (Node* const root : roots) graph_reducer.ReduceNode(root);
478 graph_reducer.ReduceGraph();
479
480 if (induction_vars != nullptr) {
481 // Validate the types computed by TypeInductionVariablePhi.
482 for (auto entry : induction_vars->induction_variables()) {
483 InductionVariable* induction_var = entry.second;
484 if (induction_var->phi()->opcode() == IrOpcode::kInductionVariablePhi) {
485 CHECK(visitor.InductionVariablePhiTypeIsPrefixedPoint(induction_var));
486 }
487 }
488
489 induction_vars->ChangeToPhisAndInsertGuards();
490 }

493void Typer::Decorator::Decorate(Node* node) {
494 if (node->op()->ValueOutputCount() > 0) {
495 // Only eagerly type-decorate nodes with known input types.
496 // Other cases will generally require a proper fixpoint iteration with Run.
497 bool is_typed = NodeProperties::IsTyped(node);
498 if (is_typed || NodeProperties::AllValueInputsAreTyped(node)) {
499 Visitor typing(typer_, nullptr);
500 Type type = typing.TypeNode(node);
501 if (is_typed) {
502 type = Type::Intersect(type, NodeProperties::GetType(node),
503 typer_->zone());
504 }
505 NodeProperties::SetType(node, type);
506 }
507 }

938Type Typer::Visitor::TypeInductionVariablePhi(Node* node) {
939 int arity = NodeProperties::GetControlInput(node)->op()->ControlInputCount();
940 DCHECK_EQ(IrOpcode::kLoop, NodeProperties::GetControlInput(node)->opcode());
941 DCHECK_EQ(2, NodeProperties::GetControlInput(node)->InputCount());
942
943 Type initial_type = Operand(node, 0);
944 Type increment_type = Operand(node, 2);
945
946 // Fallback to normal phi typing in a variety of cases:
947 // - when the induction variable is not initially of type Integer, because we
948 // want to work with ranges in the algorithm below.
949 // - when the increment is zero, because in that case normal phi typing will
950 // generally yield a more precise type.
951 // - when the induction variable can become NaN (through addition/subtraction
952 // of opposing infinities), because the code below can't handle that case.
953 if (initial_type.IsNone() ||
954 increment_type.Is(typer_->cache_->kSingletonZero) ||
955 !initial_type.Is(typer_->cache_->kInteger) ||
956 !increment_type.Is(typer_->cache_->kInteger) ||
957 increment_type.Min() == -V8_INFINITY ||
958 increment_type.Max() == +V8_INFINITY) {
959 // Unfortunately, without baking in the previous type, monotonicity might be
960 // violated because we might not yet have retyped the incrementing operation
961 // even though the increment's type might been already reflected in the
962 // induction variable phi.
963 Type type = NodeProperties::IsTyped(node) ? NodeProperties::GetType(node)
964 : Type::None();
965 for (int i = 0; i < arity; ++i) {
966 type = Type::Union(type, Operand(node, i), zone());
967 }
968 return type;
969 }
970
971 auto res = induction_vars_->induction_variables().find(node->id());
972 DCHECK_NE(res, induction_vars_->induction_variables().end());
973 InductionVariable* induction_var = res->second;
974 InductionVariable::ArithmeticType arithmetic_type = induction_var->Type();
975
976 double min = -V8_INFINITY;
977 double max = V8_INFINITY;
978
979 double increment_min;
980 double increment_max;
981 if (arithmetic_type == InductionVariable::ArithmeticType::kAddition) {
982 increment_min = increment_type.Min();
983 increment_max = increment_type.Max();
984 } else {
985 DCHECK_EQ(arithmetic_type, InductionVariable::ArithmeticType::kSubtraction);
986 increment_min = -increment_type.Max();
987 increment_max = -increment_type.Min();
988 }
989
990 if (increment_min >= 0) {
991 // Increasing sequence.
992 min = initial_type.Min();
993 for (auto bound : induction_var->upper_bounds()) {
994 Type bound_type = TypeOrNone(bound.bound);
995 // If the type is not an integer, just skip the bound.
996 if (!bound_type.Is(typer_->cache_->kInteger)) continue;
997 // If the type is not inhabited, then we can take the initial value.
998 if (bound_type.IsNone()) {
999 max = initial_type.Max();
1000 break;
1001 }
1002 double bound_max = bound_type.Max();
1003 if (bound.kind == InductionVariable::kStrict) {
1004 bound_max -= 1;
1005 }
1006 max = std::min(max, bound_max + increment_max);
1007 }
1008 // The upper bound must be at least the initial value's upper bound.
1009 max = std::max(max, initial_type.Max());
1010 } else if (increment_max <= 0) {
1011 // Decreasing sequence.
1012 max = initial_type.Max();
1013 for (auto bound : induction_var->lower_bounds()) {
1014 Type bound_type = TypeOrNone(bound.bound);
1015 // If the type is not an integer, just skip the bound.
1016 if (!bound_type.Is(typer_->cache_->kInteger)) continue;
1017 // If the type is not inhabited, then we can take the initial value.
1018 if (bound_type.IsNone()) {
1019 min = initial_type.Min();
1020 break;
1021 }
1022 double bound_min = bound_type.Min();
1023 if (bound.kind == InductionVariable::kStrict) {
1024 bound_min += 1;
1025 }
1026 min = std::max(min, bound_min + increment_min);
1027 }
1028 // The lower bound must be at most the initial value's lower bound.
1029 min = std::min(min, initial_type.Min());
1030 } else {
1031 // If the increment can be both positive and negative, the variable can go
1032 // arbitrarily far. Use the maximal range in that case. Note that this may
1033 // be less precise than what ordinary typing would produce.
1034 min = -V8_INFINITY;
1035 max = +V8_INFINITY;
1036 }
1037
1038 if (v8_flags.trace_turbo_loop) {
1039 StdoutStream{} << std::setprecision(10) << "Loop ("
1040 << NodeProperties::GetControlInput(node)->id()
1041 << ") variable bounds in "
1042 << (arithmetic_type ==
1043 InductionVariable::ArithmeticType::kAddition
1044 ? "addition"
1045 : "subtraction")
1046 << " for phi " << node->id() << ": (" << min << ", " << max
1047 << ")\n";
1048 }
1049
1050 return Type::Range(min, max, typer_->zone());
1051}

1053bool Typer::Visitor::InductionVariablePhiTypeIsPrefixedPoint(
1054 InductionVariable* induction_var) {
1055 Node* node = induction_var->phi();
1056 DCHECK_EQ(node->opcode(), IrOpcode::kInductionVariablePhi);
1057 Node* arith = node->InputAt(1);
1058 Type type = NodeProperties::GetType(node);
1059 Type initial_type = Operand(node, 0);
1060 Type arith_type = Operand(node, 1);
1061 Type increment_type = Operand(node, 2);
1062
1063 // Intersect {type} with useful bounds.
1064 for (auto bound : induction_var->upper_bounds()) {
1065 Type bound_type = TypeOrNone(bound.bound);
1066 if (!bound_type.Is(typer_->cache_->kInteger)) continue;
1067 if (!bound_type.IsNone()) {
1068 bound_type = Type::Range(
1069 -V8_INFINITY,
1070 bound_type.Max() - (bound.kind == InductionVariable::kStrict),
1071 zone());
1072 }
1073 type = Type::Intersect(type, bound_type, typer_->zone());
1074 }
1075 for (auto bound : induction_var->lower_bounds()) {
1076 Type bound_type = TypeOrNone(bound.bound);
1077 if (!bound_type.Is(typer_->cache_->kInteger)) continue;
1078 if (!bound_type.IsNone()) {
1079 bound_type = Type::Range(
1080 bound_type.Min() + (bound.kind == InductionVariable::kStrict),
1081 +V8_INFINITY, typer_->zone());
1082 }
1083 type = Type::Intersect(type, bound_type, typer_->zone());
1084 }
1085
1086 if (arith_type.IsNone()) {
1087 type = Type::None();
1088 } else {
1089 // We support a few additional type conversions on the lhs of the arithmetic
1090 // operation. This needs to be kept in sync with the corresponding code in
1091 // {LoopVariableOptimizer::TryGetInductionVariable}.
1092 Node* arith_input = arith->InputAt(0);
1093 switch (arith_input->opcode()) {
1094 case IrOpcode::kSpeculativeToNumber:
1095 type = typer_->operation_typer_.SpeculativeToNumber(type);
1096 break;
1097 case IrOpcode::kJSToNumber:
1098 type = typer_->operation_typer_.ToNumber(type);
1099 break;
1100 case IrOpcode::kJSToNumberConvertBigInt:
1101 type = typer_->operation_typer_.ToNumberConvertBigInt(type);
1102 break;
1103 default:
1104 break;
1105 }
1106
1107 // Apply ordinary typing to the "increment" operation.
1108 // clang-format off
1109 switch (arith->opcode()) {
1110#define CASE(x) \
1111 case IrOpcode::k##x: \
1112 type = Type##x(type, increment_type); \
1113 break;
1114 CASE(JSAdd)
1115 CASE(JSSubtract)
1116 CASE(NumberAdd)
1117 CASE(NumberSubtract)
1118 CASE(SpeculativeNumberAdd)
1119 CASE(SpeculativeNumberSubtract)
1120 CASE(SpeculativeSafeIntegerAdd)
1121 CASE(SpeculativeSafeIntegerSubtract)
1122#undef CASE
1123 default:
1124 UNREACHABLE();
1125 }
1126 // clang-format on
1127 }
1128
1129 type = Type::Union(initial_type, type, typer_->zone());
1130
1131 return type.Is(NodeProperties::GetType(node));
1132}

1309// The ECMAScript specification defines the four relational comparison operators
1310// (<, <=, >=, >) with the help of a single abstract one. It behaves like <
1311// but returns undefined when the inputs cannot be compared.
1312// We implement the typing analogously.
1313Typer::Visitor::ComparisonOutcome Typer::Visitor::JSCompareTyper(Type lhs,
1314 Type rhs,
1315 Typer* t) {
1316 lhs = ToPrimitive(lhs, t);
1317 rhs = ToPrimitive(rhs, t);
1318 if (lhs.Maybe(Type::String()) && rhs.Maybe(Type::String())) {
1319 return ComparisonOutcome(kComparisonTrue) |
1320 ComparisonOutcome(kComparisonFalse);
1321 }
1322 lhs = ToNumeric(lhs, t);
1323 rhs = ToNumeric(rhs, t);
1324 if (lhs.Is(Type::Number()) && rhs.Is(Type::Number())) {
1325 return NumberCompareTyper(lhs, rhs, t);
1326 }
1327 return ComparisonOutcome(kComparisonTrue) |
1328 ComparisonOutcome(kComparisonFalse) |
1329 ComparisonOutcome(kComparisonUndefined);
1330}
1331
1332Typer::Visitor::ComparisonOutcome Typer::Visitor::NumberCompareTyper(Type lhs,
1333 Type rhs,
1334 Typer* t) {
1335 DCHECK(lhs.Is(Type::Number()));
1336 DCHECK(rhs.Is(Type::Number()));
1337
1338 if (lhs.IsNone() || rhs.IsNone()) return {};
1339
1340 // Shortcut for NaNs.
1341 if (lhs.Is(Type::NaN()) || rhs.Is(Type::NaN())) return kComparisonUndefined;
1342
1343 ComparisonOutcome result;
1344 if (lhs.IsHeapConstant() && rhs.Is(lhs)) {
1345 // Types are equal and are inhabited only by a single semantic value.
1346 result = kComparisonFalse;
1347 } else if (lhs.Min() >= rhs.Max()) {
1348 result = kComparisonFalse;
1349 } else if (lhs.Max() < rhs.Min()) {
1350 result = kComparisonTrue;
1351 } else {
1352 return ComparisonOutcome(kComparisonTrue) |
1353 ComparisonOutcome(kComparisonFalse) |
1354 ComparisonOutcome(kComparisonUndefined);
1355 }
1356 // Add the undefined if we could see NaN.
1357 if (lhs.Maybe(Type::NaN()) || rhs.Maybe(Type::NaN())) {
1358 result |= kComparisonUndefined;
1359 }
1360 return result;
1361}

1607// Returns a somewhat larger range if we previously assigned
1608// a (smaller) range to this node. This is used to speed up
1609// the fixpoint calculation in case there appears to be a loop
1610// in the graph. In the current implementation, we are
1611// increasing the limits to the closest power of two.
1612Type Typer::Visitor::Weaken(Node* node, Type current_type, Type previous_type) {
1613 static const double kWeakenMinLimits[] = {0.0,
1614 -1073741824.0,
1615 -2147483648.0,
1616 -4294967296.0,
1617 -8589934592.0,
1618 -17179869184.0,
1619 -34359738368.0,
1620 -68719476736.0,
1621 -137438953472.0,
1622 -274877906944.0,
1623 -549755813888.0,
1624 -1099511627776.0,
1625 -2199023255552.0,
1626 -4398046511104.0,
1627 -8796093022208.0,
1628 -17592186044416.0,
1629 -35184372088832.0,
1630 -70368744177664.0,
1631 -140737488355328.0,
1632 -281474976710656.0,
1633 -562949953421312.0};
1634 static const double kWeakenMaxLimits[] = {0.0,
1635 1073741823.0,
1636 2147483647.0,
1637 4294967295.0,
1638 8589934591.0,
1639 17179869183.0,
1640 34359738367.0,
1641 68719476735.0,
1642 137438953471.0,
1643 274877906943.0,
1644 549755813887.0,
1645 1099511627775.0,
1646 2199023255551.0,
1647 4398046511103.0,
1648 8796093022207.0,
1649 17592186044415.0,
1650 35184372088831.0,
1651 70368744177663.0,
1652 140737488355327.0,
1653 281474976710655.0,
1654 562949953421311.0};
1655 static_assert(arraysize(kWeakenMinLimits) == arraysize(kWeakenMaxLimits));
1656
1657 // If the types have nothing to do with integers, return the types.
1658 Type const integer = typer_->cache_->kInteger;
1659 if (!previous_type.Maybe(integer)) {
1660 return current_type;
1661 }
1662 DCHECK(current_type.Maybe(integer));
1663
1664 Type current_integer = Type::Intersect(current_type, integer, zone());
1665 Type previous_integer = Type::Intersect(previous_type, integer, zone());
1666
1667 // Once we start weakening a node, we should always weaken.
1668 if (!IsWeakened(node->id())) {
1669 // Only weaken if there is range involved; we should converge quickly
1670 // for all other types (the exception is a union of many constants,
1671 // but we currently do not increase the number of constants in unions).
1672 Type previous = previous_integer.GetRange();
1673 Type current = current_integer.GetRange();
1674 if (current.IsInvalid() || previous.IsInvalid()) {
1675 return current_type;
1676 }
1677 // Range is involved => we are weakening.
1678 SetWeakened(node->id());
1679 }
1680
1681 double current_min = current_integer.Min();
1682 double new_min = current_min;
1683 // Find the closest lower entry in the list of allowed
1684 // minima (or negative infinity if there is no such entry).
1685 if (current_min != previous_integer.Min()) {
1686 new_min = -V8_INFINITY;
1687 for (double const min : kWeakenMinLimits) {
1688 if (min <= current_min) {
1689 new_min = min;
1690 break;
1691 }
1692 }
1693 }
1694
1695 double current_max = current_integer.Max();
1696 double new_max = current_max;
1697 // Find the closest greater entry in the list of allowed
1698 // maxima (or infinity if there is no such entry).
1699 if (current_max != previous_integer.Max()) {
1700 new_max = V8_INFINITY;
1701 for (double const max : kWeakenMaxLimits) {
1702 if (max >= current_max) {
1703 new_max = max;
1704 break;
1705 }
1706 }
1707 }
1708
1709 return Type::Union(current_type,
1710 Type::Range(new_min, new_max, typer_->zone()),
1711 typer_->zone());
1712}
