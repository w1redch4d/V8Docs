14// Called by either isolate shutdown or the {ManagedObjectFinalizer} in order
15// to actually delete the shared pointer and decrement the shared refcount.
16template <typename CppType>
17static void Destructor(void* ptr) {
18 auto shared_ptr_ptr = reinterpret_cast<std::shared_ptr<CppType>*>(ptr);
19 delete shared_ptr_ptr;
20}
21} // namespace detail

23// static
24template <class CppType>
25DirectHandle<Managed<CppType>> Managed<CppType>::From(
26 Isolate* isolate, size_t estimated_size,
27 std::shared_ptr<CppType> shared_ptr, AllocationType allocation_type) {
28 static constexpr ExternalPointerTag kTag = TagForManaged<CppType>::value;
29 static_assert(IsManagedExternalPointerType(kTag));
30 auto destructor = new ManagedPtrDestructor(
31 estimated_size, new std::shared_ptr<CppType>{std::move(shared_ptr)},
32 detail::Destructor<CppType>);
33 destructor->external_memory_accounter_.Increase(
34 reinterpret_cast<v8::Isolate*>(isolate), estimated_size);
35 DirectHandle<Managed<CppType>> handle =
36 Cast<Managed<CppType>>(isolate->factory()->NewForeign<kTag>(
37 reinterpret_cast<Address>(destructor), allocation_type));
38 IndirectHandle<Object> global_handle =
39 isolate->global_handles()->Create(*handle);
40 destructor->global_handle_location_ = global_handle.location();
41 GlobalHandles::MakeWeak(destructor->global_handle_location_, destructor,
42 &ManagedObjectFinalizer,
43 v8::WeakCallbackType::kParameter);
44 isolate->RegisterManagedPtrDestructor(destructor);
45 return handle;
46}
