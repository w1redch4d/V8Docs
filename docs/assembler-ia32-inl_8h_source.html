52// The modes possibly affected by apply must be in kApplyMask.
53void WritableRelocInfo::apply(intptr_t delta) {
54 DCHECK_EQ(kApplyMask, (RelocInfo::ModeMask(RelocInfo::CODE_TARGET) |
55 RelocInfo::ModeMask(RelocInfo::INTERNAL_REFERENCE) |
56 RelocInfo::ModeMask(RelocInfo::OFF_HEAP_TARGET) |
57 RelocInfo::ModeMask(RelocInfo::WASM_STUB_CALL)));
58 if (IsCodeTarget(rmode_) || IsOffHeapTarget(rmode_) ||
59 IsWasmStubCall(rmode_)) {
60 base::WriteUnalignedValue(pc_,
61 base::ReadUnalignedValue<int32_t>(pc_) - delta);
62 } else if (IsInternalReference(rmode_)) {
63 // Absolute code pointer inside code object moves with the code object.
64 base::WriteUnalignedValue(pc_,
65 base::ReadUnalignedValue<int32_t>(pc_) + delta);
66 }
67}

195void Assembler::emit(const Immediate& x) {
196 if (x.rmode_ == RelocInfo::INTERNAL_REFERENCE) {
197 Label* label = reinterpret_cast<Label*>(x.immediate());
198 emit_code_relative_offset(label);
199 return;
200 }
201 if (!RelocInfo::IsNoInfo(x.rmode_)) RecordRelocInfo(x.rmode_);
202 if (x.is_heap_number_request()) {
203 RequestHeapNumber(x.heap_number_request());
204 emit(0);
205 return;
206 }
207 emit(x.immediate());
208}

210void Assembler::emit_code_relative_offset(Label* label) {
211 if (label->is_bound()) {
212 int32_t pos;
213 pos = label->pos() + InstructionStream::kHeaderSize - kHeapObjectTag;
214 emit(pos);
215 } else {
216 emit_disp(label, Displacement::CODE_RELATIVE);
217 }
218}

270void Assembler::emit_near_disp(Label* L) {
271 uint8_t disp = 0x00;
272 if (L->is_near_linked()) {
273 int offset = L->near_link_pos() - pc_offset();
274 DCHECK(is_int8(offset));
275 disp = static_cast<uint8_t>(offset & 0xFF);
276 }
277 L->link_to(pc_offset(), Label::kNear);
278 *pc_++ = disp;
279}

287void Operand::set_sib(ScaleFactor scale, Register index, Register base) {
288 DCHECK_EQ(len_, 1);
289 DCHECK_EQ(scale & -4, 0);
290 // Use SIB with no index register only for base esp.
291 DCHECK(index != esp || base == esp);
292 buf_[1] = scale << 6 | index.code() << 3 | base.code();
293 len_ = 2;
294}

296void Operand::set_disp8(int8_t disp) {
297 DCHECK(len_ == 1 || len_ == 2);
298 *reinterpret_cast<int8_t*>(&buf_[len_++]) = disp;
299}
