44template <base::MutexSharedType kIsShared>
45V8_INLINE ParkedSharedMutexGuardIf<kIsShared>::ParkedSharedMutexGuardIf(
46 LocalHeap* local_heap, base::SharedMutex* mutex, bool enable_mutex) {
47 DCHECK(AllowGarbageCollection::IsAllowed());
48 if (!enable_mutex) return;
49 mutex_ = mutex;
50
51 if (kIsShared) {
52 if (!mutex_->TryLockShared()) {
53 local_heap->ExecuteWhileParked([this]() { mutex_->LockShared(); });
54 }
55 } else {
56 if (!mutex_->TryLockExclusive()) {
57 local_heap->ExecuteWhileParked([this]() { mutex_->LockExclusive(); });
