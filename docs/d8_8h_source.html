398 // In analogy to Flag::CheckFlagChange() in src/flags/flag.cc, only allow
399 // repeated flags for identical boolean values. We allow exceptions for flags
400 // with enum-like arguments since their conflicts can also be specified
401 // completely.
402 template <class T,
403 bool kAllowIdenticalAssignment = std::is_same<T, bool>::value>
404 class DisallowReassignment {
405 public:
406 DisallowReassignment(const char* name, T value)
407 : name_(name), value_(value) {}
408
409 operator T() const { return value_; }
410 T get() const { return value_; }
411 DisallowReassignment& operator=(T value) {
412 if (check_d8_flag_contradictions) {
413 if (kAllowIdenticalAssignment) {
414 if (specified_ && value_ != value) {
415 FATAL("Contradictory values for d8 flag --%s", name_);
416 }
417 } else {
418 if (specified_) {
419 FATAL("Repeated specification of d8 flag --%s", name_);
420 }
421 }
422 }
423 value_ = value;
424 specified_ = true;
425 return *this;
426 }
427 void Overwrite(T value) { value_ = value; }
428
429 private:
430 const char* name_;
431 T value_;
432 bool specified_ = false;
433 };
