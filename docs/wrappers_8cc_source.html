160 V<Number> BuildChangeInt32ToNumber(V<Word32> value) {
161 // We expect most integers at runtime to be Smis, so it is important for
162 // wrapper performance that Smi conversion be inlined.
163 if (SmiValuesAre32Bits()) {
164 return BuildChangeInt32ToSmi(value);
165 }
166 DCHECK(SmiValuesAre31Bits());
167
168 // Double value to test if value can be a Smi, and if so, to convert it.
169 V<Tuple<Word32, Word32>> add = __ Int32AddCheckOverflow(value, value);
170 V<Word32> ovf = __ template Projection<1>(add);
171 ScopedVar<Number> result(this, OpIndex::Invalid());
172 IF_NOT (UNLIKELY(ovf)) {
173 // If it didn't overflow, the result is {2 * value} as pointer-sized
174 // value.
175 result = __ BitcastWordPtrToSmi(
176 __ ChangeInt32ToIntPtr(__ template Projection<0>(add)));
177 } ELSE{
178 // Otherwise, call builtin, to convert to a HeapNumber.
179 result = CallBuiltin<WasmInt32ToHeapNumberDescriptor>(
180 Builtin::kWasmInt32ToHeapNumber, Operator::kNoProperties, value);
181 }
182 return result;
183 }

195 V<Object> ToJS(OpIndex ret, CanonicalValueType type, V<Context> context) {
196 switch (type.kind()) {
197 case kI32:
198 return BuildChangeInt32ToNumber(ret);
199 case kI64:
200 return BuildChangeInt64ToBigInt(ret, StubCallMode::kCallBuiltinPointer);
201 case kF32:
202 return BuildChangeFloat32ToNumber(ret);
203 case kF64:
204 return BuildChangeFloat64ToNumber(ret);
205 case kRef:
206 switch (type.heap_representation_non_shared()) {
207 case HeapType::kEq:
208 case HeapType::kI31:
209 case HeapType::kStruct:
210 case HeapType::kArray:
211 case HeapType::kAny:
212 case HeapType::kExtern:
213 case HeapType::kString:
214 case HeapType::kNone:
215 case HeapType::kNoFunc:
216 case HeapType::kNoExtern:
217 return ret;
218 case HeapType::kExn:
219 case HeapType::kNoExn:
220 case HeapType::kBottom:
221 case HeapType::kTop:
222 case HeapType::kStringViewWtf8:
223 case HeapType::kStringViewWtf16:
224 case HeapType::kStringViewIter:
225 UNREACHABLE();
226 case HeapType::kFunc:
227 default:
228 if (type.heap_representation_non_shared() == HeapType::kFunc ||
229 GetTypeCanonicalizer()->IsFunctionSignature(type.ref_index())) {
230 // Function reference. Extract the external function.
231 V<WasmInternalFunction> internal =
232 V<WasmInternalFunction>::Cast(__ LoadTrustedPointerField(
233 ret, LoadOp::Kind::TaggedBase(),
234 kWasmInternalFunctionIndirectPointerTag,
235 WasmFuncRef::kTrustedInternalOffset));
236 ScopedVar<Object> maybe_external(
237 this, __ Load(internal, LoadOp::Kind::TaggedBase(),
238 MemoryRepresentation::TaggedPointer(),
239 WasmInternalFunction::kExternalOffset));
240 IF (__ TaggedEqual(maybe_external, LOAD_ROOT(UndefinedValue))) {
241 maybe_external =
242 CallBuiltin<WasmInternalFunctionCreateExternalDescriptor>(
243 Builtin::kWasmInternalFunctionCreateExternal,
244 Operator::kNoProperties, internal, context);
245 }
246 return maybe_external;
247 } else {
248 return ret;
249 }
250 }
251 case kRefNull:
252 switch (type.heap_representation_non_shared()) {
253 case HeapType::kExtern:
254 case HeapType::kNoExtern:
255 return ret;
256 case HeapType::kNone:
257 case HeapType::kNoFunc:
258 return LOAD_ROOT(NullValue);
259 case HeapType::kExn:
260 case HeapType::kNoExn:
261 UNREACHABLE();
262 case HeapType::kEq:
263 case HeapType::kStruct:
264 case HeapType::kArray:
265 case HeapType::kString:
266 case HeapType::kI31:
267 case HeapType::kAny: {
268 ScopedVar<Object> result(this, OpIndex::Invalid());
269 IF_NOT (__ TaggedEqual(ret, LOAD_ROOT(WasmNull))) {
270 result = ret;
271 } ELSE{
272 result = LOAD_ROOT(NullValue);
273 }
274 return result;
275 }
276 case HeapType::kFunc:
277 default: {
278 if (type.heap_representation_non_shared() == HeapType::kFunc ||
279 GetTypeCanonicalizer()->IsFunctionSignature(type.ref_index())) {
280 ScopedVar<Object> result(this, OpIndex::Invalid());
281 IF (__ TaggedEqual(ret, LOAD_ROOT(WasmNull))) {
282 result = LOAD_ROOT(NullValue);
283 } ELSE{
284 V<WasmInternalFunction> internal =
285 V<WasmInternalFunction>::Cast(__ LoadTrustedPointerField(
286 ret, LoadOp::Kind::TaggedBase(),
287 kWasmInternalFunctionIndirectPointerTag,
288 WasmFuncRef::kTrustedInternalOffset));
289 V<Object> maybe_external =
290 __ Load(internal, LoadOp::Kind::TaggedBase(),
291 MemoryRepresentation::AnyTagged(),
292 WasmInternalFunction::kExternalOffset);
293 IF (__ TaggedEqual(maybe_external, LOAD_ROOT(UndefinedValue))) {
294 V<Object> from_builtin =
295 CallBuiltin<WasmInternalFunctionCreateExternalDescriptor>(
296 Builtin::kWasmInternalFunctionCreateExternal,
297 Operator::kNoProperties, internal, context);
298 result = from_builtin;
299 } ELSE{
300 result = maybe_external;
301 }
302 }
303 return result;
304 } else {
305 ScopedVar<Object> result(this, OpIndex::Invalid());
306 IF (__ TaggedEqual(ret, LOAD_ROOT(WasmNull))) {
307 result = LOAD_ROOT(NullValue);
308 } ELSE{
309 result = ret;
310 }
311 return result;
312 }
313 }
314 }
315 case kRtt:
316 case kI8:
317 case kI16:
318 case kF16:
319 case kS128:
320 case kVoid:
321 case kTop:
322 case kBottom:
323 // If this is reached, then IsJSCompatibleSignature() is too permissive.
324 UNREACHABLE();
325 }
326 }

366 OpIndex BuildCallAndReturn(V<Context> js_context, V<HeapObject> function_data,
367 base::Vector<OpIndex> args, bool do_conversion,
368 bool set_in_wasm_flag) {
369 const int rets_count = static_cast<int>(sig_->return_count());
370 base::SmallVector<OpIndex, 1> rets(rets_count);
371
372 // Set the ThreadInWasm flag before we do the actual call.
373 {
374 std::optional<ModifyThreadInWasmFlagScope>
375 modify_thread_in_wasm_flag_builder;
376 if (set_in_wasm_flag) {
377 modify_thread_in_wasm_flag_builder.emplace(this, Asm());
378 }
379
380 V<WasmInternalFunction> internal =
381 V<WasmInternalFunction>::Cast(__ LoadProtectedPointerField(
382 function_data, LoadOp::Kind::TaggedBase().Immutable(),
383 WasmExportedFunctionData::kProtectedInternalOffset));
384 auto [target, implicit_arg] = BuildFunctionTargetAndImplicitArg(internal);
385 BuildCallWasmFromWrapper(__ phase_zone(), sig_, target, implicit_arg,
386 args, base::VectorOf(rets));
387 }
388
389 V<Object> jsval;
390 if (sig_->return_count() == 0) {
391 jsval = LOAD_ROOT(UndefinedValue);
392 } else if (sig_->return_count() == 1) {
393 jsval = do_conversion ? ToJS(rets[0], sig_->GetReturn(), js_context)
394 : rets[0];
395 } else {
396 int32_t return_count = static_cast<int32_t>(sig_->return_count());
397 V<Smi> size = __ SmiConstant(Smi::FromInt(return_count));
398
399 jsval = BuildCallAllocateJSArray(size, js_context);
400
401 V<FixedArray> fixed_array = __ Load(jsval, LoadOp::Kind::TaggedBase(),
402 MemoryRepresentation::TaggedPointer(),
403 JSObject::kElementsOffset);
404
405 for (int i = 0; i < return_count; ++i) {
406 V<Object> value = ToJS(rets[i], sig_->GetReturn(i), js_context);
407 __ StoreFixedArrayElement(fixed_array, i, value,
408 compiler::kFullWriteBarrier);
409 }
410 }
411 return jsval;
412 }

414 void BuildJSToWasmWrapper(
415 bool do_conversion = true,
416 compiler::turboshaft::OptionalOpIndex frame_state =
417 compiler::turboshaft::OptionalOpIndex::Nullopt(),
418 bool set_in_wasm_flag = true) {
419 const int wasm_param_count = static_cast<int>(sig_->parameter_count());
420
421 __ Bind(__ NewBlock());
422
423 // Create the js_closure and js_context parameters.
424 V<JSFunction> js_closure =
425 __ Parameter(compiler::Linkage::kJSCallClosureParamIndex,
426 RegisterRepresentation::Tagged());
427 V<Context> js_context = __ Parameter(
428 compiler::Linkage::GetJSCallContextParamIndex(wasm_param_count + 1),
429 RegisterRepresentation::Tagged());
430 V<SharedFunctionInfo> shared =
431 __ Load(js_closure, LoadOp::Kind::TaggedBase(),
432 MemoryRepresentation::TaggedPointer(),
433 JSFunction::kSharedFunctionInfoOffset);
434 V<WasmFunctionData> function_data =
435 V<WasmFunctionData>::Cast(__ LoadTrustedPointerField(
436 shared, LoadOp::Kind::TaggedBase(),
437 kWasmFunctionDataIndirectPointerTag,
438 SharedFunctionInfo::kTrustedFunctionDataOffset));
439
440 if (!IsJSCompatibleSignature(sig_)) {
441 // Throw a TypeError. Use the js_context of the calling javascript
442 // function (passed as a parameter), such that the generated code is
443 // js_context independent.
444 CallRuntime(__ phase_zone(), Runtime::kWasmThrowJSTypeError, {},
445 js_context);
446 __ Unreachable();
447 return;
448 }
449
450 const int args_count = wasm_param_count + 1; // +1 for wasm_code.
451
452 // Check whether the signature of the function allows for a fast
453 // transformation (if any params exist that need transformation).
454 // Create a fast transformation path, only if it does.
455 bool include_fast_path =
456 do_conversion && wasm_param_count > 0 && QualifiesForFastTransform();
457
458 // Prepare Param() nodes. Param() nodes can only be created once,
459 // so we need to use the same nodes along all possible transformation paths.
460 base::SmallVector<OpIndex, 16> params(args_count);
461 for (int i = 0; i < wasm_param_count; ++i) {
462 params[i + 1] = __ Parameter(i + 1, RegisterRepresentation::Tagged());
463 }
464
465 Label<Object> done(&Asm());
466 V<Object> jsval;
467 if (include_fast_path) {
468 TSBlock* slow_path = __ NewBlock();
469 // Check if the params received on runtime can be actually transformed
470 // using the fast transformation. When a param that cannot be transformed
471 // fast is encountered, skip checking the rest and fall back to the slow
472 // path.
473 for (int i = 0; i < wasm_param_count; ++i) {
474 CanTransformFast(params[i + 1], sig_->GetParam(i), slow_path);
475 }
476 // Convert JS parameters to wasm numbers using the fast transformation
477 // and build the call.
478 base::SmallVector<OpIndex, 16> args(args_count);
479 for (int i = 0; i < wasm_param_count; ++i) {
480 OpIndex wasm_param = FromJSFast(params[i + 1], sig_->GetParam(i));
481 args[i + 1] = wasm_param;
482 }
483 jsval =
484 BuildCallAndReturn(js_context, function_data, base::VectorOf(args),
485 do_conversion, set_in_wasm_flag);
486 GOTO(done, jsval);
487 __ Bind(slow_path);
488 }
489 // Convert JS parameters to wasm numbers using the default transformation
490 // and build the call.
491 base::SmallVector<OpIndex, 16> args(args_count);
492 for (int i = 0; i < wasm_param_count; ++i) {
493 if (do_conversion) {
494 args[i + 1] =
495 FromJS(params[i + 1], js_context, sig_->GetParam(i), frame_state);
496 } else {
497 OpIndex wasm_param = params[i + 1];
498
499 // For Float32 parameters
500 // we set UseInfo::CheckedNumberOrOddballAsFloat64 in
501 // simplified-lowering and we need to add here a conversion from Float64
502 // to Float32.
503 if (sig_->GetParam(i).kind() == kF32) {
504 wasm_param = __ TruncateFloat64ToFloat32(wasm_param);
505 }
506 args[i + 1] = wasm_param;
507 }
508 }
509
510 jsval = BuildCallAndReturn(js_context, function_data, base::VectorOf(args),
511 do_conversion, set_in_wasm_flag);
512 // If both the default and a fast transformation paths are present,
513 // get the return value based on the path used.
514 if (include_fast_path) {
515 GOTO(done, jsval);
516 BIND(done, result);
517 __ Return(result);
518 } else {
519 __ Return(jsval);
520 }
521 }

523 void BuildWasmToJSWrapper(ImportCallKind kind, int expected_arity,
524 Suspend suspend) {
525 int wasm_count = static_cast<int>(sig_->parameter_count());
526
527 __ Bind(__ NewBlock());
528 base::SmallVector<OpIndex, 16> wasm_params(wasm_count);
529 OpIndex ref = __ Parameter(0, RegisterRepresentation::Tagged());
530 for (int i = 0; i < wasm_count; ++i) {
531 RegisterRepresentation rep = RepresentationFor(sig_->GetParam(i));
532 wasm_params[i] = (__ Parameter(1 + i, rep));
533 }
534
535 V<Context> native_context = __ Load(ref, LoadOp::Kind::TaggedBase(),
536 MemoryRepresentation::TaggedPointer(),
537 WasmImportData::kNativeContextOffset);
538
539 if (kind == ImportCallKind::kRuntimeTypeError) {
540 // =======================================================================
541 // === Runtime TypeError =================================================
542 // =======================================================================
543 CallRuntime(zone_, Runtime::kWasmThrowJSTypeError, {}, native_context);
544 __ Unreachable();
545 return;
546 }
547
548 V<Undefined> undefined_node = LOAD_ROOT(UndefinedValue);
549 int pushed_count = std::max(expected_arity, wasm_count);
550 // 5 extra arguments: receiver, new target, arg count, dispatch handle and
551 // context.
552 bool has_dispatch_handle =
553 kind == ImportCallKind::kUseCallBuiltin
554 ? false
555 : V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE_BOOL;
556 base::SmallVector<OpIndex, 16> args(pushed_count + 4 +
557 (has_dispatch_handle ? 1 : 0));
558 // Position of the first wasm argument in the JS arguments.
559 int pos = kind == ImportCallKind::kUseCallBuiltin ? 3 : 1;
560 pos = AddArgumentNodes(base::VectorOf(args), pos, wasm_params, sig_,
561 native_context);
562 for (int i = wasm_count; i < expected_arity; ++i) {
563 args[pos++] = undefined_node;
564 }
565
566 V<JSFunction> callable_node = __ Load(ref, LoadOp::Kind::TaggedBase(),
567 MemoryRepresentation::TaggedPointer(),
568 WasmImportData::kCallableOffset);
569 auto [old_sp, old_limit] = BuildSwitchToTheCentralStackIfNeeded();
570 BuildModifyThreadInWasmFlag(__ phase_zone(), false);
571 OpIndex call = OpIndex::Invalid();
572 switch (kind) {
573 // =======================================================================
574 // === JS Functions ======================================================
575 // =======================================================================
576 case ImportCallKind::kJSFunctionArityMatch:
577 DCHECK_EQ(expected_arity, wasm_count);
578 [[fallthrough]];
579 case ImportCallKind::kJSFunctionArityMismatch: {
580 auto call_descriptor = compiler::Linkage::GetJSCallDescriptor(
581 __ graph_zone(), false, pushed_count + 1, CallDescriptor::kNoFlags);
582 const TSCallDescriptor* ts_call_descriptor = TSCallDescriptor::Create(
583 call_descriptor, compiler::CanThrow::kYes,
584 compiler::LazyDeoptOnThrow::kNo, __ graph_zone());
585
586 // Determine receiver at runtime.
587 args[0] =
588 BuildReceiverNode(callable_node, native_context, undefined_node);
589 DCHECK_EQ(pos, pushed_count + 1);
590 args[pos++] = undefined_node; // new target
591 args[pos++] =
592 __ Word32Constant(JSParameterCount(wasm_count)); // argument count
593#ifdef V8_JS_LINKAGE_INCLUDES_DISPATCH_HANDLE
594 args[pos++] = __ Word32Constant(kPlaceholderDispatchHandle.value());
595#endif
596 args[pos++] = LoadContextFromJSFunction(callable_node);
597 call = __ Call(callable_node, OpIndex::Invalid(), base::VectorOf(args),
598 ts_call_descriptor);
599 break;
600 }
601 // =======================================================================
602 // === General case of unknown callable ==================================
603 // =======================================================================
604 case ImportCallKind::kUseCallBuiltin: {
605 DCHECK_EQ(expected_arity, wasm_count);
606 OpIndex target = GetBuiltinPointerTarget(Builtin::kCall_ReceiverIsAny);
607 args[0] = callable_node;
608 args[1] =
609 __ Word32Constant(JSParameterCount(wasm_count)); // argument count
610 args[2] = undefined_node; // receiver
611
612 auto call_descriptor = compiler::Linkage::GetStubCallDescriptor(
613 __ graph_zone(), CallTrampolineDescriptor{}, wasm_count + 1,
614 CallDescriptor::kNoFlags, Operator::kNoProperties,
615 StubCallMode::kCallBuiltinPointer);
616 const TSCallDescriptor* ts_call_descriptor = TSCallDescriptor::Create(
617 call_descriptor, compiler::CanThrow::kYes,
618 compiler::LazyDeoptOnThrow::kNo, __ graph_zone());
619
620 // The native_context is sufficient here, because all kind of callables
621 // which depend on the context provide their own context. The context
622 // here is only needed if the target is a constructor to throw a
623 // TypeError, if the target is a native function, or if the target is a
624 // callable JSObject, which can only be constructed by the runtime.
625 args[pos++] = native_context;
626 call = __ Call(target, OpIndex::Invalid(), base::VectorOf(args),
627 ts_call_descriptor);
628 break;
629 }
630 default:
631 UNIMPLEMENTED();
632 }
633 // For asm.js the error location can differ depending on whether an
634 // exception was thrown in imported JS code or an exception was thrown in
635 // the ToNumber builtin that converts the result of the JS code a
636 // WebAssembly value. The source position allows asm.js to determine the
637 // correct error location. Source position 1 encodes the call to ToNumber,
638 // source position 0 encodes the call to the imported JS code.
639 __ output_graph().source_positions()[call] = SourcePosition(0);
640 DCHECK(call.valid());
641
642 if (suspend == kSuspend) {
643 call = BuildSuspend(call, ref, &old_sp, old_limit);
644 }
645
646 // Convert the return value(s) back.
647 OpIndex val;
648 base::SmallVector<OpIndex, 8> wasm_values;
649 if (sig_->return_count() <= 1) {
650 val = sig_->return_count() == 0
651 ? __ Word32Constant(0)
652 : FromJS(call, native_context, sig_->GetReturn());
653 } else {
654 V<FixedArray> fixed_array =
655 BuildMultiReturnFixedArrayFromIterable(call, native_context);
656 wasm_values.resize_no_init(sig_->return_count());
657 for (unsigned i = 0; i < sig_->return_count(); ++i) {
658 wasm_values[i] = FromJS(__ LoadFixedArrayElement(fixed_array, i),
659 native_context, sig_->GetReturn(i));
660 }
661 }
662 BuildModifyThreadInWasmFlag(__ phase_zone(), true);
663 BuildSwitchBackFromCentralStack(old_sp, old_limit);
664 if (sig_->return_count() <= 1) {
665 __ Return(val);
666 } else {
667 __ Return(__ Word32Constant(0), base::VectorOf(wasm_values));
668 }
669 }

671 void BuildCapiCallWrapper() {
672 __ Bind(__ NewBlock());
673 base::SmallVector<OpIndex, 8> incoming_params;
674 // Instance.
675 incoming_params.push_back(
676 __ Parameter(0, RegisterRepresentation::Tagged()));
677 // Wasm parameters.
678 for (int i = 0; i < static_cast<int>(sig_->parameter_count()); ++i) {
679 incoming_params.push_back(
680 __ Parameter(i + 1, RepresentationFor(sig_->GetParam(i))));
681 }
682 // Store arguments on our stack, then align the stack for calling to C.
683 int param_bytes = 0;
684 for (CanonicalValueType type : sig_->parameters()) {
685 param_bytes += type.value_kind_size();
686 }
687 int return_bytes = 0;
688 for (CanonicalValueType type : sig_->returns()) {
689 return_bytes += type.value_kind_size();
690 }
691
692 int stack_slot_bytes = std::max(param_bytes, return_bytes);
693 OpIndex values = stack_slot_bytes == 0
694 ? __ IntPtrConstant(0)
695 : __ StackSlot(stack_slot_bytes, kDoubleAlignment);
696
697 int offset = 0;
698 for (size_t i = 0; i < sig_->parameter_count(); ++i) {
699 CanonicalValueType type = sig_->GetParam(i);
700 // Start from the parameter with index 1 to drop the instance_node.
701 // TODO(jkummerow): When a values is a reference type, we should pass it
702 // in a GC-safe way, not just as a raw pointer.
703 SafeStore(offset, type, values, incoming_params[i + 1]);
704 offset += type.value_kind_size();
705 }
706
707 V<Object> function_node =
708 __ LoadTaggedField(incoming_params[0], WasmImportData::kCallableOffset);
709 V<HeapObject> shared = LoadSharedFunctionInfo(function_node);
710 V<WasmFunctionData> function_data =
711 V<WasmFunctionData>::Cast(__ LoadTrustedPointerField(
712 shared, LoadOp::Kind::TaggedBase(),
713 kWasmFunctionDataIndirectPointerTag,
714 SharedFunctionInfo::kTrustedFunctionDataOffset));
715 V<Object> host_data_foreign = __ LoadTaggedField(
716 function_data, WasmCapiFunctionData::kEmbedderDataOffset);
717
718 BuildModifyThreadInWasmFlag(__ phase_zone(), false);
719 OpIndex isolate_root = __ LoadRootRegister();
720 OpIndex fp_value = __ FramePointer();
721 __ Store(isolate_root, fp_value, StoreOp::Kind::RawAligned(),
722 MemoryRepresentation::UintPtr(), compiler::kNoWriteBarrier,
723 Isolate::c_entry_fp_offset());
724
725 V<WordPtr> call_target =
726 BuildLoadCallTargetFromExportedFunctionData(function_data);
727
728 // Parameters: Address host_data_foreign, Address arguments.
729 auto host_sig =
730 FixedSizeSignature<MachineType>::Returns(MachineType::Pointer())
731 .Params(MachineType::AnyTagged(), MachineType::Pointer());
732 OpIndex return_value =
733 CallC(&host_sig, call_target, {host_data_foreign, values});
734
735 BuildModifyThreadInWasmFlag(__ phase_zone(), true);
736
737 IF_NOT (__ WordPtrEqual(return_value, __ IntPtrConstant(0))) {
738 WasmRethrowExplicitContextDescriptor interface_descriptor;
739 auto call_descriptor = compiler::Linkage::GetStubCallDescriptor(
740 __ graph_zone(), interface_descriptor,
741 interface_descriptor.GetStackParameterCount(),
742 CallDescriptor::kNoFlags, Operator::kNoProperties,
743 StubCallMode::kCallBuiltinPointer);
744 const TSCallDescriptor* ts_call_descriptor = TSCallDescriptor::Create(
745 call_descriptor, compiler::CanThrow::kYes,
746 compiler::LazyDeoptOnThrow::kNo, __ graph_zone());
747 OpIndex rethrow_call_target =
748 GetTargetForBuiltinCall(Builtin::kWasmRethrowExplicitContext);
749 V<Context> context =
750 __ Load(incoming_params[0], LoadOp::Kind::TaggedBase(),
751 MemoryRepresentation::TaggedPointer(),
752 WasmImportData::kNativeContextOffset);
753 __ Call(rethrow_call_target, {return_value, context}, ts_call_descriptor);
754 __ Unreachable();
755 }
756
757 DCHECK_LT(sig_->return_count(), kV8MaxWasmFunctionReturns);
758 size_t return_count = sig_->return_count();
759 if (return_count == 0) {
760 __ Return(__ Word32Constant(0));
761 } else {
762 base::SmallVector<OpIndex, 8> returns(return_count);
763 offset = 0;
764 for (size_t i = 0; i < return_count; ++i) {
765 CanonicalValueType type = sig_->GetReturn(i);
766 OpIndex val = SafeLoad(values, offset, type);
767 returns[i] = val;
768 offset += type.value_kind_size();
769 }
770 __ Return(__ Word32Constant(0), base::VectorOf(returns));
771 }
772 }

838 OpIndex BuildCheckString(OpIndex input, OpIndex js_context,
839 CanonicalValueType type) {
840 auto done = __ NewBlock();
841 auto type_error = __ NewBlock();
842 ScopedVar<Object> result(this, LOAD_ROOT(WasmNull));
843 __ GotoIf(__ IsSmi(input), type_error, BranchHint::kFalse);
844 if (type.is_nullable()) {
845 auto not_null = __ NewBlock();
846 __ GotoIfNot(__ TaggedEqual(input, LOAD_ROOT(NullValue)), not_null);
847 __ Goto(done);
848 __ Bind(not_null);
849 }
850 V<Map> map = LoadMap(input);
851 OpIndex instance_type = LoadInstanceType(map);
852 OpIndex check = __ Uint32LessThan(instance_type,
853 __ Word32Constant(FIRST_NONSTRING_TYPE));
854 result = input;
855 __ GotoIf(check, done, BranchHint::kTrue);
856 __ Goto(type_error);
857 __ Bind(type_error);
858 CallRuntime(__ phase_zone(), Runtime::kWasmThrowJSTypeError, {},
859 js_context);
860 __ Unreachable();
861 __ Bind(done);
862 return result;
863 }

935 OpIndex FromJS(OpIndex input, OpIndex context, CanonicalValueType type,
936 OptionalOpIndex frame_state = {}) {
937 switch (type.kind()) {
938 case kRef:
939 case kRefNull: {
940 switch (type.heap_representation_non_shared()) {
941 // TODO(14034): Add more fast paths?
942 case HeapType::kExtern:
943 if (type.kind() == kRef) {
944 IF (UNLIKELY(__ TaggedEqual(input, LOAD_ROOT(NullValue)))) {
945 CallRuntime(__ phase_zone(), Runtime::kWasmThrowJSTypeError, {},
946 context);
947 __ Unreachable();
948 }
949 }
950 return input;
951 case HeapType::kString:
952 return BuildCheckString(input, context, type);
953 case HeapType::kExn:
954 case HeapType::kNoExn: {
955 UNREACHABLE();
956 }
957 case HeapType::kNoExtern:
958 case HeapType::kNone:
959 case HeapType::kNoFunc:
960 case HeapType::kI31:
961 case HeapType::kAny:
962 case HeapType::kFunc:
963 case HeapType::kStruct:
964 case HeapType::kArray:
965 case HeapType::kEq:
966 default: {
967 // Make sure ValueType fits in a Smi.
968 static_assert(ValueType::kLastUsedBit + 1 <= kSmiValueSize);
969
970 std::initializer_list<const OpIndex> inputs = {
971 input, __ IntPtrConstant(
972 IntToSmi(static_cast<int>(type.raw_bit_field())))};
973 return CallRuntime(__ phase_zone(), Runtime::kWasmJSToWasmObject,
974 inputs, context);
975 }
976 }
977 }
978 case kF32:
979 return __ TruncateFloat64ToFloat32(
980 BuildChangeTaggedToFloat64(input, context, frame_state));
981
982 case kF64:
983 return BuildChangeTaggedToFloat64(input, context, frame_state);
984
985 case kI32:
986 return BuildChangeTaggedToInt32(input, context, frame_state);
987
988 case kI64:
989 // i64 values can only come from BigInt.
990 return BuildChangeBigIntToInt64(input, context, frame_state);
991
992 case kRtt:
993 case kS128:
994 case kI8:
995 case kI16:
996 case kF16:
997 case kTop:
998 case kBottom:
999 case kVoid:
1000 // If this is reached, then IsJSCompatibleSignature() is too permissive.
1001 UNREACHABLE();
1002 }
1003 }

1146 V<WordPtr> BuildSwitchToTheCentralStack(V<WordPtr> old_limit) {
1147 // Set the is_on_central_stack flag.
1148 OpIndex isolate_root = __ LoadRootRegister();
1149 __ Store(isolate_root, __ Word32Constant(1), LoadOp::Kind::RawAligned(),
1150 MemoryRepresentation::Uint8(), compiler::kNoWriteBarrier,
1151 IsolateData::is_on_central_stack_flag_offset());
1152
1153 // Save the old fp and the target sp in the StackMemory's stack switch info.
1154 // We are not on the main stack, so the ActiveContinuation root must exist
1155 // and be of type WasmContinuationObject.
1156 V<WasmContinuationObject> active_continuation =
1157 V<WasmContinuationObject>::Cast(LOAD_ROOT(ActiveContinuation));
1158 V<WordPtr> stack_data = __ LoadExternalPointerFromObject(
1159 active_continuation, WasmContinuationObject::kStackOffset,
1160 kWasmStackMemoryTag);
1161 __ Store(stack_data, __ FramePointer(), StoreOp::Kind::RawAligned(),
1162 MemoryRepresentation::UintPtr(), compiler::kNoWriteBarrier,
1163 StackMemory::stack_switch_source_fp_offset());
1164 V<WordPtr> central_stack_sp = __ Load(
1165 isolate_root, LoadOp::Kind::RawAligned(),
1166 MemoryRepresentation::UintPtr(), Isolate::central_stack_sp_offset());
1167 __ Store(stack_data, central_stack_sp, StoreOp::Kind::RawAligned(),
1168 MemoryRepresentation::UintPtr(), compiler::kNoWriteBarrier,
1169 StackMemory::stack_switch_target_sp_offset());
1170
1171 // Switch the stack limit and the stack pointer.
1172 V<WordPtr> central_stack_limit = __ Load(
1173 isolate_root, LoadOp::Kind::RawAligned(),
1174 MemoryRepresentation::UintPtr(), Isolate::central_stack_limit_offset());
1175 BuildSetNewStackLimit(old_limit, central_stack_limit);
1176 OpIndex old_sp = __ LoadStackPointer();
1177 __ SetStackPointer(central_stack_sp);
1178 return old_sp;
1179 }

1231 V<WordPtr>* old_sp, V<WordPtr> old_limit) {
1232 // If value is a promise, suspend to the js-to-wasm prompt, and resume later
1233 // with the promise's resolved value.
1234 ScopedVar<Object> result(this, value);
1235 ScopedVar<WordPtr> old_sp_var(this, *old_sp);
1236 IF_NOT (__ IsSmi(value)) {
1237 IF (__ HasInstanceType(value, JS_PROMISE_TYPE)) {
1238 OpIndex suspender = LOAD_ROOT(ActiveSuspender);
1239 V<Context> native_context =
1240 __ Load(import_data, LoadOp::Kind::TaggedBase(),
1241 MemoryRepresentation::TaggedPointer(),
1242 WasmImportData::kNativeContextOffset);
1243 IF (__ TaggedEqual(suspender, LOAD_ROOT(UndefinedValue))) {
1244 V<Smi> error = __ SmiConstant(Smi::FromInt(
1245 static_cast<int32_t>(MessageTemplate::kWasmSuspendError)));
1246 CallRuntime(__ phase_zone(), Runtime::kThrowWasmSuspendError, {error},
1247 native_context);
1248 __ Unreachable();
1249 }
1250 if (v8_flags.stress_wasm_stack_switching) {
1251 V<Word32> for_stress_testing = __ TaggedEqual(
1252 __ LoadTaggedField(suspender, WasmSuspenderObject::kResumeOffset),
1253 LOAD_ROOT(UndefinedValue));

1255 V<Smi> error = __ SmiConstant(Smi::FromInt(
1256 static_cast<int32_t>(MessageTemplate::kWasmSuspendJSFrames)));
1257 CallRuntime(__ phase_zone(), Runtime::kThrowWasmSuspendError,
1258 {error}, native_context);
1259 __ Unreachable();
1260 }
1261 }
1262 // If {old_sp} is null, it must be that we were on the central stack
1263 // before entering the wasm-to-js wrapper, which means that there are JS
1264 // frames in the current suspender. JS frames cannot be suspended, so
1265 // trap.
1266 OpIndex has_js_frames = __ WordPtrEqual(__ IntPtrConstant(0), *old_sp);
1267 IF (has_js_frames) {
1268 V<Smi> error = __ SmiConstant(Smi::FromInt(
1269 static_cast<int32_t>(MessageTemplate::kWasmSuspendJSFrames)));
1270 CallRuntime(__ phase_zone(), Runtime::kThrowWasmSuspendError, {error},
1271 native_context);
1272 __ Unreachable();
1273 }
1274 V<Object> on_fulfilled = __ Load(suspender, LoadOp::Kind::TaggedBase(),
1275 MemoryRepresentation::TaggedPointer(),
1276 WasmSuspenderObject::kResumeOffset);
1277 V<Object> on_rejected = __ Load(suspender, LoadOp::Kind::TaggedBase(),
1278 MemoryRepresentation::TaggedPointer(),
1279 WasmSuspenderObject::kRejectOffset);
1280
1281 OpIndex promise_then =
1282 GetBuiltinPointerTarget(Builtin::kPerformPromiseThen);
1283 auto* then_call_desc = GetBuiltinCallDescriptor(
1284 Builtin::kPerformPromiseThen, __ graph_zone());
1285 base::SmallVector<OpIndex, 16> args{value, on_fulfilled, on_rejected,
1286 LOAD_ROOT(UndefinedValue),
1287 native_context};
1288 __ Call(promise_then, OpIndex::Invalid(), base::VectorOf(args),
1289 then_call_desc);
1290
1291 OpIndex suspend = GetTargetForBuiltinCall(Builtin::kWasmSuspend);
1292 auto* suspend_call_descriptor =
1293 GetBuiltinCallDescriptor(Builtin::kWasmSuspend, __ graph_zone());
1294 BuildSwitchBackFromCentralStack(*old_sp, old_limit);
1295 V<Object> resolved =
1296 __ Call<Object>(suspend, {suspender}, suspend_call_descriptor);
1297 old_sp_var = BuildSwitchToTheCentralStack(old_limit);
1298 result = resolved;
1299 }
1300 }
1301 *old_sp = old_sp_var;
1302 return result;
1303 }
