94 std::unordered_set<CodeEntry*, CodeEntry::Hasher, CodeEntry::Equals>*

96 CodeEntry* search_value, CodeEntryStorage& storage) {

108void ProfilerListener::CodeCreateEvent(CodeTag tag,
109 Handle<AbstractCode> abstract_code,
110 Handle<SharedFunctionInfo> shared,
111 Handle<Name> script_name, int line,
112 int column) {
113 CodeEventsContainer evt_rec(CodeEventRecord::Type::kCodeCreation);
114 CodeCreateEventRecord* rec = &evt_rec.CodeCreateEventRecord_;
115 PtrComprCageBase cage_base(isolate_);
116 rec->instruction_start = abstract_code->InstructionStart(cage_base);
117 std::unique_ptr<SourcePositionTable> line_table;
118 std::unordered_map<int, std::vector<CodeEntryAndLineNumber>> inline_stacks;
119 std::unordered_set<CodeEntry*, CodeEntry::Hasher, CodeEntry::Equals>
120 cached_inline_entries;
121 bool is_shared_cross_origin = false;
122 if (IsScript(shared->script(cage_base), cage_base)) {
123 DirectHandle<Script> script(Cast<Script>(shared->script(cage_base)),
124 isolate_);
125 line_table.reset(new SourcePositionTable());
126
127 is_shared_cross_origin = script->origin_options().IsSharedCrossOrigin();
128
129 bool is_baseline = abstract_code->kind(cage_base) == CodeKind::BASELINE;
130 Handle<TrustedByteArray> source_position_table(
131 abstract_code->SourcePositionTable(isolate_, *shared), isolate_);
132 std::unique_ptr<baseline::BytecodeOffsetIterator> baseline_iterator;
133 if (is_baseline) {
134 Handle<BytecodeArray> bytecodes(shared->GetBytecodeArray(isolate_),
135 isolate_);
136 Handle<TrustedByteArray> bytecode_offsets(
137 abstract_code->GetCode()->bytecode_offset_table(), isolate_);
138 baseline_iterator = std::make_unique<baseline::BytecodeOffsetIterator>(
139 bytecode_offsets, bytecodes);
140 }
141 // Add each position to the source position table and store inlining stacks
142 // for inline positions. We store almost the same information in the
143 // profiler as is stored on the code object, except that we transform source
144 // positions to line numbers here, because we only care about attributing
145 // ticks to a given line.
146 for (SourcePositionTableIterator it(source_position_table); !it.done();
147 it.Advance()) {
148 int position = it.source_position().ScriptOffset();
149 int inlining_id = it.source_position().InliningId();
150 int code_offset = it.code_offset();
151 if (is_baseline) {
152 // Use the bytecode offset to calculate pc offset for baseline code.
153 baseline_iterator->AdvanceToBytecodeOffset(code_offset);
154 code_offset =
155 static_cast<int>(baseline_iterator->current_pc_start_offset());
156 }
157
158 if (inlining_id == SourcePosition::kNotInlined) {
159 int line_number = script->GetLineNumber(position) + 1;
160 line_table->SetPosition(code_offset, line_number, inlining_id);
161 } else {
162 DCHECK(!is_baseline);
163 DCHECK(IsCode(*abstract_code, cage_base));
164 std::vector<SourcePositionInfo> stack =
165 it.source_position().InliningStack(isolate_,
166 abstract_code->GetCode());
167 DCHECK(!stack.empty());
168
169 // When we have an inlining id and we are doing cross-script inlining,
170 // then the script of the inlined frames may be different to the script
171 // of |shared|.
172 int line_number = stack.front().line + 1;
173 line_table->SetPosition(code_offset, line_number, inlining_id);
174
175 std::vector<CodeEntryAndLineNumber> inline_stack;
176 for (SourcePositionInfo& pos_info : stack) {
177 if (pos_info.position.ScriptOffset() == kNoSourcePosition) continue;
178 if (pos_info.script.is_null()) continue;
179
180 line_number =
181 pos_info.script->GetLineNumber(pos_info.position.ScriptOffset()) +
182 1;
183
184 const char* resource_name =
185 (IsName(pos_info.script->name()))
186 ? GetName(Cast<Name>(pos_info.script->name()))
187 : CodeEntry::kEmptyResourceName;
188
189 bool inline_is_shared_cross_origin =
190 pos_info.script->origin_options().IsSharedCrossOrigin();
191
192 // We need the start line number and column number of the function for
193 // kLeafNodeLineNumbers mode. Creating a SourcePositionInfo is a handy
194 // way of getting both easily.
195 SourcePositionInfo start_pos_info(
196 isolate_, SourcePosition(pos_info.shared->StartPosition()),
197 pos_info.shared);
198
199 CodeEntry* inline_entry = code_entries_.Create(
200 tag, GetFunctionName(*pos_info.shared), resource_name,
201 start_pos_info.line + 1, start_pos_info.column + 1, nullptr,
202 inline_is_shared_cross_origin);
203 inline_entry->FillFunctionInfo(*pos_info.shared);
204
205 // Create a canonical CodeEntry for each inlined frame and then re-use
206 // them for subsequent inline stacks to avoid a lot of duplication.
207 CodeEntry* cached_entry = GetOrInsertCachedEntry(
208 &cached_inline_entries, inline_entry, code_entries_);
209
210 inline_stack.push_back({cached_entry, line_number});
211 }
212 DCHECK(!inline_stack.empty());
213 inline_stacks.emplace(inlining_id, std::move(inline_stack));
214 }
215 }
216 }
217 rec->entry = code_entries_.Create(
218 tag, GetFunctionName(*shared),
219 GetName(InferScriptName(*script_name, *shared)), line, column,
220 std::move(line_table), is_shared_cross_origin);
221 if (!inline_stacks.empty()) {
222 rec->entry->SetInlineStacks(std::move(cached_inline_entries),
223 std::move(inline_stacks));
224 }
225
226 rec->entry->FillFunctionInfo(*shared);
227 rec->instruction_size = abstract_code->InstructionSize(cage_base);
228 weak_code_registry_.Track(rec->entry, abstract_code);
229 DispatchCodeEvent(evt_rec);
230}

363const char* ProfilerListener::GetName(base::Vector<const char> name) {
364 // TODO(all): Change {StringsStorage} to accept non-null-terminated strings.
365 base::OwnedVector<char> null_terminated =
366 base::OwnedVector<char>::New(name.size() + 1);
367#if defined(__GNUC__) && !defined(__clang__)
368 // Work around a spurious GCC-12 warning (-Werror=array-bounds).
369 if (name.end() < name.begin()) return nullptr;
370#endif
371 std::copy(name.begin(), name.end(), null_terminated.begin());
372 null_terminated[name.size()] = '\0';
373 return GetName(null_terminated.begin());
374}

396void ProfilerListener::AttachDeoptInlinedFrames(DirectHandle<Code> code,
397 CodeDeoptEventRecord* rec) {
398 int deopt_id = rec->deopt_id;
399 SourcePosition last_position = SourcePosition::Unknown();
400 int mask = RelocInfo::ModeMask(RelocInfo::DEOPT_ID) |
401 RelocInfo::ModeMask(RelocInfo::DEOPT_SCRIPT_OFFSET) |
402 RelocInfo::ModeMask(RelocInfo::DEOPT_INLINING_ID);
403
404 rec->deopt_frames = nullptr;
405 rec->deopt_frame_count = 0;
406
407 for (RelocIterator it(*code, mask); !it.done(); it.next()) {
408 RelocInfo* info = it.rinfo();
409 if (info->rmode() == RelocInfo::DEOPT_SCRIPT_OFFSET) {
410 int script_offset = static_cast<int>(info->data());
411 it.next();
412 DCHECK(it.rinfo()->rmode() == RelocInfo::DEOPT_INLINING_ID);
413 int inlining_id = static_cast<int>(it.rinfo()->data());
414 last_position = SourcePosition(script_offset, inlining_id);
415 continue;
416 }
417 if (info->rmode() == RelocInfo::DEOPT_ID) {
418 if (deopt_id != static_cast<int>(info->data())) continue;
419 DCHECK(last_position.IsKnown());
420
421 // SourcePosition::InliningStack allocates a handle for the SFI of each
422 // frame. These don't escape this function, but quickly add up. This
423 // scope limits their lifetime.
424 HandleScope scope(isolate_);
425 std::vector<SourcePositionInfo> stack =
426 last_position.InliningStack(isolate_, *code);
427 CpuProfileDeoptFrame* deopt_frames =
428 new CpuProfileDeoptFrame[stack.size()];
429
430 int deopt_frame_count = 0;
431 for (SourcePositionInfo& pos_info : stack) {
432 if (pos_info.position.ScriptOffset() == kNoSourcePosition) continue;
433 if (pos_info.script.is_null()) continue;
434 int script_id = pos_info.script->id();
435 size_t offset = static_cast<size_t>(pos_info.position.ScriptOffset());
436 deopt_frames[deopt_frame_count++] = {script_id, offset};
437 }
438 rec->deopt_frames = deopt_frames;
439 rec->deopt_frame_count = deopt_frame_count;
440 break;
441 }
442 }
