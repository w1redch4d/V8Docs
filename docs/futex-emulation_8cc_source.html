27// A {FutexWaitList} manages all contexts waiting (synchronously or
28// asynchronously) on any address.
29class FutexWaitList {
30 public:
31 FutexWaitList() = default;
32 FutexWaitList(const FutexWaitList&) = delete;
33 FutexWaitList& operator=(const FutexWaitList&) = delete;

35 void AddNode(FutexWaitListNode* node);

38 static void* ToWaitLocation(Tagged<JSArrayBuffer> array_buffer, size_t addr) {
39 DCHECK_LT(addr, array_buffer->GetByteLength());
40 // Use the cheaper JSArrayBuffer::backing_store() accessor, but DCHECK that
41 // it matches the start of the JSArrayBuffer::GetBackingStore().
42 DCHECK_EQ(array_buffer->backing_store(),
43 array_buffer->GetBackingStore()->buffer_start());
44 return static_cast<uint8_t*>(array_buffer->backing_store()) + addr;
45 }
46
47 // Deletes "node" and returns the next node of its list.
48 static FutexWaitListNode* DeleteAsyncWaiterNode(FutexWaitListNode* node) {
49 DCHECK(node->IsAsync());
50 DCHECK_NOT_NULL(node->async_state_->isolate_for_async_waiters);
51 FutexWaitListNode* next = node->next_;
52 if (node->prev_ != nullptr) {
53 node->prev_->next_ = next;
54 }
55 if (next != nullptr) {
56 next->prev_ = node->prev_;
57 }
58 delete node;
59 return next;
60 }
61
62 static void DeleteNodesForIsolate(Isolate* isolate, FutexWaitListNode** head,
63 FutexWaitListNode** tail) {
64 // For updating head & tail once we've iterated all nodes.
65 FutexWaitListNode* new_head = nullptr;
66 FutexWaitListNode* new_tail = nullptr;
67 for (FutexWaitListNode* node = *head; node;) {
68 if (node->IsAsync() &&
69 node->async_state_->isolate_for_async_waiters == isolate) {
70 node->async_state_->timeout_task_id =
71 CancelableTaskManager::kInvalidTaskId;
72 node = DeleteAsyncWaiterNode(node);
73 } else {
74 if (new_head == nullptr) {
75 new_head = node;
76 }
77 new_tail = node;
78 node = node->next_;
79 }
80 }
81 *head = new_head;
82 *tail = new_tail;
83 }
84
85 // For checking the internal consistency of the FutexWaitList.
86 void Verify() const;
87 // Returns true if |node| is on the linked list starting with |head|.
88 static bool NodeIsOnList(FutexWaitListNode* node, FutexWaitListNode* head);
89
90 base::Mutex* mutex() { return &mutex_; }
91
92 private:
93 friend class FutexEmulation;
94
95 struct HeadAndTail {
96 FutexWaitListNode* head;
97 FutexWaitListNode* tail;
98 };
99
100 // `mutex` protects the composition of the fields below (i.e. no elements may
101 // be added or removed without holding this mutex), as well as the `waiting_`
102 // and `interrupted_` fields for each individual list node that is currently
103 // part of the list. It must be the mutex used together with the `cond_`
104 // condition variable of such nodes.
105 base::Mutex mutex_;
106
107 // Location inside a shared buffer -> linked list of Nodes waiting on that
108 // location.
109 // As long as the map does not grow beyond 16 entries, there is no dynamic
110 // allocation and deallocation happening in wait or wake, which reduces the
111 // time spend in the critical section.
112 base::SmallMap<std::map<void*, HeadAndTail>, 16> location_lists_;
113
114 // Isolate* -> linked list of Nodes which are waiting for their Promises to
115 // be resolved.
116 base::SmallMap<std::map<Isolate*, HeadAndTail>> isolate_promises_to_resolve_;
117};

126bool FutexWaitListNode::CancelTimeoutTask() {
127 DCHECK(IsAsync());
128 if (async_state_->timeout_task_id == CancelableTaskManager::kInvalidTaskId) {
129 return true;
130 }
131 auto* cancelable_task_manager =
132 async_state_->isolate_for_async_waiters->cancelable_task_manager();
133 TryAbortResult return_value =
134 cancelable_task_manager->TryAbort(async_state_->timeout_task_id);
135 async_state_->timeout_task_id = CancelableTaskManager::kInvalidTaskId;
136 return return_value != TryAbortResult::kTaskRunning;
137}

139void FutexWaitListNode::NotifyWake() {
140 DCHECK(!IsAsync());
141 // Lock the FutexEmulation mutex before notifying. We know that the mutex
142 // will have been unlocked if we are currently waiting on the condition
143 // variable. The mutex will not be locked if FutexEmulation::Wait hasn't
144 // locked it yet. In that case, we set the interrupted_
145 // flag to true, which will be tested after the mutex locked by a future wait.
146 FutexWaitList* wait_list = GetWaitList();
147 NoGarbageCollectionMutexGuard lock_guard(wait_list->mutex());
148
149 // if not waiting, this will not have any effect.
150 cond_.NotifyOne();
151 interrupted_ = true;
152}

181void FutexEmulation::NotifyAsyncWaiter(FutexWaitListNode* node) {
182 DCHECK(node->IsAsync());
183 // This function can run in any thread.
184
185 FutexWaitList* wait_list = GetWaitList();
186 wait_list->mutex()->AssertHeld();
187
188 // Nullify the timeout time; this distinguishes timed out waiters from
189 // woken up ones.
190 node->async_state_->timeout_time = base::TimeTicks();
191
192 wait_list->RemoveNode(node);
193
194 // Schedule a task for resolving the Promise. It's still possible that the
195 // timeout task runs before the promise resolving task. In that case, the
196 // timeout task will just ignore the node.
197 auto& isolate_map = wait_list->isolate_promises_to_resolve_;
198 auto it = isolate_map.find(node->async_state_->isolate_for_async_waiters);
199 if (it == isolate_map.end()) {
200 // This Isolate doesn't have other Promises to resolve at the moment.
201 isolate_map.insert(
202 std::make_pair(node->async_state_->isolate_for_async_waiters,
203 FutexWaitList::HeadAndTail{node, node}));
204 auto task = std::make_unique<ResolveAsyncWaiterPromisesTask>(
205 node->async_state_->isolate_for_async_waiters);
206 node->async_state_->task_runner->PostNonNestableTask(std::move(task));
207 } else {
208 // Add this Node into the existing list.
209 node->prev_ = it->second.tail;
210 it->second.tail->next_ = node;
211 it->second.tail = node;
212 }
213}

215void FutexWaitList::AddNode(FutexWaitListNode* node) {
216 DCHECK_NULL(node->prev_);
217 DCHECK_NULL(node->next_);
218 auto [it, inserted] =
219 location_lists_.insert({node->wait_location_, HeadAndTail{node, node}});
220 if (!inserted) {
221 it->second.tail->next_ = node;
222 node->prev_ = it->second.tail;
223 it->second.tail = node;
224 }
225
226 Verify();
227}
228
229void FutexWaitList::RemoveNode(FutexWaitListNode* node) {
230 if (!node->prev_ && !node->next_) {
231 // If the node was the last one on its list, delete the whole list.
232 size_t erased = location_lists_.erase(node->wait_location_);
233 DCHECK_EQ(1, erased);
234 USE(erased);
235 } else if (node->prev_ && node->next_) {
236 // If we have both a successor and a predecessor, skip the lookup in the
237 // list and just update those two nodes directly.
238 node->prev_->next_ = node->next_;
239 node->next_->prev_ = node->prev_;
240 node->prev_ = node->next_ = nullptr;
241 } else {
242 // Otherwise we have to lookup in the list to find the head and tail
243 // pointers.
244 auto it = location_lists_.find(node->wait_location_);
245 DCHECK_NE(location_lists_.end(), it);
246 DCHECK(NodeIsOnList(node, it->second.head));
247
248 if (node->prev_) {
249 DCHECK(!node->next_);
250 node->prev_->next_ = nullptr;
251 DCHECK_EQ(node, it->second.tail);
252 it->second.tail = node->prev_;
253 node->prev_ = nullptr;
254 } else {
255 DCHECK_EQ(node, it->second.head);
256 it->second.head = node->next_;
257 DCHECK(node->next_);
258 node->next_->prev_ = nullptr;
259 node->next_ = nullptr;
260 }
261 }
262
263 Verify();
264}

336template <typename T>
337Tagged<Object> FutexEmulation::Wait(Isolate* isolate, WaitMode mode,
338 Handle<JSArrayBuffer> array_buffer,
339 size_t addr, T value,
340 double rel_timeout_ms) {
341 DCHECK_LT(addr, array_buffer->GetByteLength());
342
343 bool use_timeout = rel_timeout_ms != V8_INFINITY;
344 int64_t rel_timeout_ns = -1;
345
346 if (use_timeout) {
347 // Convert to nanoseconds.
348 double timeout_ns = rel_timeout_ms *
349 base::Time::kNanosecondsPerMicrosecond *
350 base::Time::kMicrosecondsPerMillisecond;
351 if (timeout_ns > static_cast<double>(std::numeric_limits<int64_t>::max())) {
352 // 2**63 nanoseconds is 292 years. Let's just treat anything greater as
353 // infinite.
354 use_timeout = false;
355 } else {
356 rel_timeout_ns = static_cast<int64_t>(timeout_ns);
357 }
358 }
359 return Wait(isolate, mode, array_buffer, addr, value, use_timeout,
360 rel_timeout_ns);
361}

364double WaitTimeoutInMs(double timeout_ns) {
365 return timeout_ns < 0
366 ? V8_INFINITY
367 : timeout_ns / (base::Time::kNanosecondsPerMicrosecond *
368 base::Time::kMicrosecondsPerMillisecond);
369}

387template <typename T>
388Tagged<Object> FutexEmulation::WaitSync(Isolate* isolate,
389 Handle<JSArrayBuffer> array_buffer,
390 size_t addr, T value, bool use_timeout,
391 int64_t rel_timeout_ns,
392 CallType call_type) {
393 VMState<ATOMICS_WAIT> state(isolate);
394 base::TimeDelta rel_timeout =
395 base::TimeDelta::FromNanoseconds(rel_timeout_ns);
396
397 // We have to convert the timeout back to double for the AtomicsWaitCallback.
398 double rel_timeout_ms = WaitTimeoutInMs(static_cast<double>(rel_timeout_ns));
399 AtomicsWaitWakeHandle stop_handle(isolate);
400
401 isolate->RunAtomicsWaitCallback(AtomicsWaitEvent::kStartWait, array_buffer,
402 addr, value, rel_timeout_ms, &stop_handle);
403 if (isolate->has_exception()) return ReadOnlyRoots(isolate).exception();
404
405 DirectHandle<Object> result;
406 AtomicsWaitEvent callback_result = AtomicsWaitEvent::kWokenUp;
407
408 FutexWaitList* wait_list = GetWaitList();
409 FutexWaitListNode* node = isolate->futex_wait_list_node();
410 void* wait_location = FutexWaitList::ToWaitLocation(*array_buffer, addr);
411
412 base::TimeTicks timeout_time;
413 if (use_timeout) {
414 base::TimeTicks current_time = base::TimeTicks::Now();
415 timeout_time = current_time + rel_timeout;
416 }
417
418 // The following is not really a loop; the do-while construct makes it easier
419 // to break out early.
420 // Keep the code in the loop as minimal as possible, because this is all in
421 // the critical section.
422 do {
423 NoGarbageCollectionMutexGuard lock_guard(wait_list->mutex());
424
425 std::atomic<T>* p = reinterpret_cast<std::atomic<T>*>(wait_location);
426 T loaded_value = p->load();
427#if defined(V8_TARGET_BIG_ENDIAN)
428 // If loading a Wasm value, it needs to be reversed on Big Endian platforms.
429 if (call_type == CallType::kIsWasm) {
430 DCHECK(sizeof(T) == kInt32Size || sizeof(T) == kInt64Size);
431 loaded_value = ByteReverse(loaded_value);
432 }
433#endif
434 if (loaded_value != value) {
435 result =
436 direct_handle(Smi::FromInt(WaitReturnValue::kNotEqualValue), isolate);
437 callback_result = AtomicsWaitEvent::kNotEqual;
438 break;
439 }
440
441 node->wait_location_ = wait_location;
442 node->waiting_ = true;
443 wait_list->AddNode(node);
444
445 while (true) {
446 if (V8_UNLIKELY(node->interrupted_)) {
447 // Reset the interrupted flag while still holding the mutex.
448 node->interrupted_ = false;
449
450 // Unlock the mutex here to prevent deadlock from lock ordering between
451 // mutex and mutexes locked by HandleInterrupts.
452 lock_guard.Unlock();
453
454 // Because the mutex is unlocked, we have to be careful about not
455 // dropping an interrupt. The notification can happen in three different
456 // places:
457 // 1) Before Wait is called: the notification will be dropped, but
458 // interrupted_ will be set to 1. This will be checked below.
459 // 2) After interrupted has been checked here, but before mutex is
460 // acquired: interrupted is checked in a loop, with mutex locked.
461 // Because the wakeup signal also acquires mutex, we know it will not
462 // be able to notify until mutex is released below, when waiting on
463 // the condition variable.
464 // 3) After the mutex is released in the call to WaitFor(): this
465 // notification will wake up the condition variable. node->waiting()
466 // will be false, so we'll loop and then check interrupts.
467 Tagged<Object> interrupt_object =
468 isolate->stack_guard()->HandleInterrupts();
469
470 lock_guard.Lock();
471
472 if (IsException(interrupt_object, isolate)) {
473 result = direct_handle(interrupt_object, isolate);
474 callback_result = AtomicsWaitEvent::kTerminatedExecution;
475 break;
476 }
477 }
478
479 if (V8_UNLIKELY(node->interrupted_)) {
480 // An interrupt occurred while the mutex was unlocked. Don't wait yet.
481 continue;
482 }
483
484 if (stop_handle.has_stopped()) {
485 node->waiting_ = false;
486 callback_result = AtomicsWaitEvent::kAPIStopped;
487 }
488
489 if (!node->waiting_) {
490 // We were woken either via the stop_handle or via Wake.
491 result = direct_handle(Smi::FromInt(WaitReturnValue::kOk), isolate);
492 break;
493 }
494
495 // No interrupts, now wait.
496 if (use_timeout) {
497 base::TimeTicks current_time = base::TimeTicks::Now();
498 if (current_time >= timeout_time) {
499 result =
500 direct_handle(Smi::FromInt(WaitReturnValue::kTimedOut), isolate);
501 callback_result = AtomicsWaitEvent::kTimedOut;
502 break;
503 }
504
505 base::TimeDelta time_until_timeout = timeout_time - current_time;
506 DCHECK_GE(time_until_timeout.InMicroseconds(), 0);
507 bool wait_for_result =
508 node->cond_.WaitFor(wait_list->mutex(), time_until_timeout);
509 USE(wait_for_result);
510 } else {
511 node->cond_.Wait(wait_list->mutex());
512 }
513
514 // Spurious wakeup, interrupt or timeout.
515 }
516
517 node->waiting_ = false;
518 wait_list->RemoveNode(node);
519 } while (false);
520 DCHECK(!node->waiting_);
521
522 isolate->RunAtomicsWaitCallback(callback_result, array_buffer, addr, value,
523 rel_timeout_ms, nullptr);
524
525 if (isolate->has_exception() &&
526 callback_result != AtomicsWaitEvent::kTerminatedExecution) {
527 return ReadOnlyRoots(isolate).exception();
528 }
529
530 return *result;
531}

557template <typename T>
558Tagged<Object> FutexEmulation::WaitAsync(
559 Isolate* isolate, DirectHandle<JSArrayBuffer> array_buffer, size_t addr,
560 T value, bool use_timeout, int64_t rel_timeout_ns, CallType call_type) {
561 base::TimeDelta rel_timeout =
562 base::TimeDelta::FromNanoseconds(rel_timeout_ns);
563
564 Factory* factory = isolate->factory();
565 DirectHandle<JSObject> result =
566 factory->NewJSObject(isolate->object_function());
567 DirectHandle<JSObject> promise_capability = factory->NewJSPromise();
568
569 enum class ResultKind { kNotEqual, kTimedOut, kAsync };
570 ResultKind result_kind;
571 void* wait_location = FutexWaitList::ToWaitLocation(*array_buffer, addr);
572 // Get a weak pointer to the backing store, to be stored in the async state of
573 // the node.
574 std::weak_ptr<BackingStore> backing_store{array_buffer->GetBackingStore()};
575 FutexWaitList* wait_list = GetWaitList();
576 {
577 // 16. Perform EnterCriticalSection(WL).
578 NoGarbageCollectionMutexGuard lock_guard(wait_list->mutex());
579
580 // 17. Let w be ! AtomicLoad(typedArray, i).
581 std::atomic<T>* p = static_cast<std::atomic<T>*>(wait_location);
582 T loaded_value = p->load();
583#if defined(V8_TARGET_BIG_ENDIAN)
584 // If loading a Wasm value, it needs to be reversed on Big Endian platforms.
585 if (call_type == CallType::kIsWasm) {
586 DCHECK(sizeof(T) == kInt32Size || sizeof(T) == kInt64Size);
587 loaded_value = ByteReverse(loaded_value);
588 }
589#endif
590 if (loaded_value != value) {
591 result_kind = ResultKind::kNotEqual;
592 } else if (use_timeout && rel_timeout_ns == 0) {
593 result_kind = ResultKind::kTimedOut;
594 } else {
595 result_kind = ResultKind::kAsync;
596
597 FutexWaitListNode* node = new FutexWaitListNode(
598 std::move(backing_store), wait_location, promise_capability, isolate);
599
600 if (use_timeout) {
601 node->async_state_->timeout_time = base::TimeTicks::Now() + rel_timeout;
602 auto task = std::make_unique<AsyncWaiterTimeoutTask>(
603 node->async_state_->isolate_for_async_waiters
604 ->cancelable_task_manager(),
605 node);
606 node->async_state_->timeout_task_id = task->id();
607 node->async_state_->task_runner->PostNonNestableDelayedTask(
608 std::move(task), rel_timeout.InSecondsF());
609 }
610
611 wait_list->AddNode(node);
612 }
613
614 // Leaving the block collapses the following steps:
615 // 18.a. Perform LeaveCriticalSection(WL).
616 // 19.b. Perform LeaveCriticalSection(WL).
617 // 24. Perform LeaveCriticalSection(WL).
618 }
619
620 switch (result_kind) {
621 case ResultKind::kNotEqual:
622 // 18. If v is not equal to w, then
623 // ...
624 // c. Perform ! CreateDataPropertyOrThrow(resultObject, "async", false).
625 // d. Perform ! CreateDataPropertyOrThrow(resultObject, "value",
626 // "not-equal").
627 // e. Return resultObject.
628 CHECK(JSReceiver::CreateDataProperty(
629 isolate, result, factory->async_string(),
630 factory->false_value(), Just(kDontThrow))
631 .FromJust());
632 CHECK(JSReceiver::CreateDataProperty(
633 isolate, result, factory->value_string(),
634 factory->not_equal_string(), Just(kDontThrow))
635 .FromJust());
636 break;
637
638 case ResultKind::kTimedOut:
639 // 19. If t is 0 and mode is async, then
640 // ...
641 // c. Perform ! CreateDataPropertyOrThrow(resultObject, "async", false).
642 // d. Perform ! CreateDataPropertyOrThrow(resultObject, "value",
643 // "timed-out").
644 // e. Return resultObject.
645 CHECK(JSReceiver::CreateDataProperty(
646 isolate, result, factory->async_string(),
647 factory->false_value(), Just(kDontThrow))
648 .FromJust());
649 CHECK(JSReceiver::CreateDataProperty(
650 isolate, result, factory->value_string(),
651 factory->timed_out_string(), Just(kDontThrow))
652 .FromJust());
653 break;
654
655 case ResultKind::kAsync:
656 // Add the Promise into the NativeContext's atomics_waitasync_promises
657 // set, so that the list keeps it alive.
658 DirectHandle<NativeContext> native_context(isolate->native_context());
659 Handle<OrderedHashSet> promises(
660 native_context->atomics_waitasync_promises(), isolate);
661 promises = OrderedHashSet::Add(isolate, promises, promise_capability)
662 .ToHandleChecked();
663 native_context->set_atomics_waitasync_promises(*promises);
664
665 // 26. Perform ! CreateDataPropertyOrThrow(resultObject, "async", true).
666 // 27. Perform ! CreateDataPropertyOrThrow(resultObject, "value",
667 // promiseCapability.[[Promise]]).
668 // 28. Return resultObject.
669 CHECK(JSReceiver::CreateDataProperty(
670 isolate, result, factory->async_string(), factory->true_value(),
671 Just(kDontThrow))
672 .FromJust());
673 CHECK(JSReceiver::CreateDataProperty(isolate, result,
674 factory->value_string(),
675 promise_capability, Just(kDontThrow))
676 .FromJust());
677 break;
678 }
679
680 return *result;
681}

689int FutexEmulation::Wake(void* wait_location, uint32_t num_waiters_to_wake) {
690 int num_waiters_woken = 0;
691 FutexWaitList* wait_list = GetWaitList();
692 NoGarbageCollectionMutexGuard lock_guard(wait_list->mutex());
693
694 auto& location_lists = wait_list->location_lists_;
695 auto it = location_lists.find(wait_location);
696 if (it == location_lists.end()) return num_waiters_woken;
697
698 FutexWaitListNode* node = it->second.head;
699 while (node && num_waiters_to_wake > 0) {
700 if (!node->waiting_) {
701 node = node->next_;
702 continue;
703 }
704 // Relying on wait_location_ here is not enough, since we need to guard
705 // against the case where the BackingStore of the node has been deleted
706 // during an async wait and a new BackingStore recreated in the same memory
707 // area. Note that sync wait always keeps the backing store alive.
708 // It is sufficient to check whether the node's backing store is expired
709 // (and consider this a non-match). If it is not expired, it must be
710 // identical to the backing store from which wait_location was computed by
711 // the caller. In that case, the current context holds the arraybuffer and
712 // backing store alive during this call, so it can not expire while we
713 // execute this code.
714 bool matching_backing_store =
715 !node->IsAsync() || !node->async_state_->backing_store.expired();
716 if (V8_LIKELY(matching_backing_store)) {
717 node->waiting_ = false;
718
719 // Retrieve the next node to iterate before calling NotifyAsyncWaiter,
720 // since NotifyAsyncWaiter will take the node out of the linked list.
721 FutexWaitListNode* next_node = node->next_;
722 if (node->IsAsync()) {
723 NotifyAsyncWaiter(node);
724 } else {
725 // WaitSync will remove the node from the list.
726 node->cond_.NotifyOne();
727 }
728 node = next_node;
729 if (num_waiters_to_wake != kWakeAll) {
730 --num_waiters_to_wake;
731 }
732 num_waiters_woken++;
733 continue;
734 }
735
736 // ---
737 // Code below handles the unlikely case that this node's backing store was
738 // deleted during an async wait and a new one was allocated in its place.
739 // We delete the node if possible (no timeout, or context is gone).
740 // ---
741 bool delete_this_node = false;
742 DCHECK(node->IsAsync());
743 if (node->async_state_->timeout_time.IsNull()) {
744 // Backing store has been deleted and the node is still waiting, and
745 // there's no timeout. It's never going to be woken up, so we can clean it
746 // up now. We don't need to cancel the timeout task, because there is
747 // none.
748
749 // This cleanup code is not very efficient, since it only kicks in when
750 // a new BackingStore has been created in the same memory area where the
751 // deleted BackingStore was.
752 DCHECK(node->IsAsync());
753 DCHECK_EQ(CancelableTaskManager::kInvalidTaskId,
754 node->async_state_->timeout_task_id);
755 delete_this_node = true;
756 }
757 if (node->async_state_->native_context.IsEmpty()) {
758 // The NativeContext related to the async waiter has been deleted.
759 // Ditto, clean up now.
760
761 // Using the CancelableTaskManager here is OK since the Isolate is
762 // guaranteed to be alive - FutexEmulation::IsolateDeinit removes all
763 // FutexWaitListNodes owned by an Isolate which is going to die.
764 if (node->CancelTimeoutTask()) {
765 delete_this_node = true;
766 }
767 // If cancelling the timeout task failed, the timeout task is already
768 // running and will clean up the node.
769 }
770
771 FutexWaitListNode* next_node = node->next_;
772 if (delete_this_node) {
773 wait_list->RemoveNode(node);
774 delete node;
775 }
776 node = next_node;
777 }
778
779 return num_waiters_woken;
780}

782void FutexEmulation::CleanupAsyncWaiterPromise(FutexWaitListNode* node) {
783 DCHECK(node->IsAsync());
784 // This function must run in the main thread of node's Isolate. This function
785 // may allocate memory. To avoid deadlocks, we shouldn't be holding the
786 // FutexEmulationGlobalState::mutex.
787
788 Isolate* isolate = node->async_state_->isolate_for_async_waiters;
789 auto v8_isolate = reinterpret_cast<v8::Isolate*>(isolate);
790
791 if (!node->async_state_->promise.IsEmpty()) {
792 auto promise = Cast<JSPromise>(
793 Utils::OpenDirectHandle(*node->async_state_->promise.Get(v8_isolate)));
794 // Promise keeps the NativeContext alive.
795 DCHECK(!node->async_state_->native_context.IsEmpty());
796 auto native_context = Cast<NativeContext>(Utils::OpenDirectHandle(
797 *node->async_state_->native_context.Get(v8_isolate)));
798
799 // Remove the Promise from the NativeContext's set.
800 Handle<OrderedHashSet> promises(
801 native_context->atomics_waitasync_promises(), isolate);
802 bool was_deleted = OrderedHashSet::Delete(isolate, *promises, *promise);
803 DCHECK(was_deleted);
804 USE(was_deleted);
805 promises = OrderedHashSet::Shrink(isolate, promises);
806 native_context->set_atomics_waitasync_promises(*promises);
807 } else {
808 // NativeContext keeps the Promise alive; if the Promise is dead then
809 // surely NativeContext is too.
810 DCHECK(node->async_state_->native_context.IsEmpty());
811 }
812}

814void FutexEmulation::ResolveAsyncWaiterPromise(FutexWaitListNode* node) {
815 DCHECK(node->IsAsync());
816 // This function must run in the main thread of node's Isolate.
817
818 Isolate* isolate = node->async_state_->isolate_for_async_waiters;
819 v8::Isolate* v8_isolate = reinterpret_cast<v8::Isolate*>(isolate);
820
821 // Try to cancel the timeout task (if one exists). If the timeout task exists,
822 // cancelling it will always succeed. It's not possible for the timeout task
823 // to be running, since it's scheduled to run in the same thread as this task.
824
825 // Using the CancelableTaskManager here is OK since the Isolate is guaranteed
826 // to be alive - FutexEmulation::IsolateDeinit removes all FutexWaitListNodes
827 // owned by an Isolate which is going to die.
828 bool success = node->CancelTimeoutTask();
829 DCHECK(success);
830 USE(success);
831
832 if (!node->async_state_->promise.IsEmpty()) {
833 DCHECK(!node->async_state_->native_context.IsEmpty());
834 Local<v8::Context> native_context =
835 node->async_state_->native_context.Get(v8_isolate);
836 v8::Context::Scope contextScope(native_context);
837 DirectHandle<JSPromise> promise = Cast<JSPromise>(
838 Utils::OpenHandle(*node->async_state_->promise.Get(v8_isolate)));
839 DirectHandle<String> result_string;
840 // When waiters are notified, their timeout_time is reset. Having a
841 // non-zero timeout_time here means the waiter timed out.
842 if (node->async_state_->timeout_time != base::TimeTicks()) {
843 DCHECK(node->waiting_);
844 result_string = isolate->factory()->timed_out_string();
845 } else {
846 DCHECK(!node->waiting_);
847 result_string = isolate->factory()->ok_string();
848 }
849 MaybeHandle<Object> resolve_result =
850 JSPromise::Resolve(promise, result_string);
851 DCHECK(!resolve_result.is_null());
852 USE(resolve_result);
853 }
854}

856void FutexEmulation::ResolveAsyncWaiterPromises(Isolate* isolate) {
857 // This function must run in the main thread of isolate.
858
859 FutexWaitList* wait_list = GetWaitList();
860 FutexWaitListNode* node;
861 {
862 NoGarbageCollectionMutexGuard lock_guard(wait_list->mutex());
863
864 auto& isolate_map = wait_list->isolate_promises_to_resolve_;
865 auto it = isolate_map.find(isolate);
866 DCHECK_NE(isolate_map.end(), it);
867
868 node = it->second.head;
869 isolate_map.erase(it);
870 }
871
872 // The list of nodes starting from "node" are no longer on any list, so it's
873 // ok to iterate them without holding the mutex. We also need to not hold the
874 // mutex while calling CleanupAsyncWaiterPromise, since it may allocate
875 // memory.
876 HandleScope handle_scope(isolate);
877 while (node) {
878 DCHECK(node->IsAsync());
879 DCHECK_EQ(isolate, node->async_state_->isolate_for_async_waiters);
880 DCHECK(!node->waiting_);
881 ResolveAsyncWaiterPromise(node);
882 CleanupAsyncWaiterPromise(node);
883 // We've already tried to cancel the timeout task for the node; since we're
884 // now in the same thread the timeout task is supposed to run, we know the
885 // timeout task will never happen, and it's safe to delete the node here.
886 DCHECK_EQ(CancelableTaskManager::kInvalidTaskId,
887 node->async_state_->timeout_task_id);
888 node = FutexWaitList::DeleteAsyncWaiterNode(node);
889 }
890}

892void FutexEmulation::HandleAsyncWaiterTimeout(FutexWaitListNode* node) {
893 // This function must run in the main thread of node's Isolate.
894 DCHECK(node->IsAsync());
895
896 FutexWaitList* wait_list = GetWaitList();
897
898 {
899 NoGarbageCollectionMutexGuard lock_guard(wait_list->mutex());
900
901 node->async_state_->timeout_task_id = CancelableTaskManager::kInvalidTaskId;
902 if (!node->waiting_) {
903 // If the Node is not waiting, it's already scheduled to have its Promise
904 // resolved. Ignore the timeout.
905 return;
906 }
907 wait_list->RemoveNode(node);
908 }
909
910 // "node" has been taken out of the lists, so it's ok to access it without
911 // holding the mutex. We also need to not hold the mutex while calling
912 // CleanupAsyncWaiterPromise, since it may allocate memory.
913 HandleScope handle_scope(node->async_state_->isolate_for_async_waiters);
914 ResolveAsyncWaiterPromise(node);
915 CleanupAsyncWaiterPromise(node);
916 delete node;
917}

919void FutexEmulation::IsolateDeinit(Isolate* isolate) {
920 FutexWaitList* wait_list = GetWaitList();
921 NoGarbageCollectionMutexGuard lock_guard(wait_list->mutex());
922
923 // Iterate all locations to find nodes belonging to "isolate" and delete them.
924 // The Isolate is going away; don't bother cleaning up the Promises in the
925 // NativeContext. Also we don't need to cancel the timeout tasks, since they
926 // will be cancelled by Isolate::Deinit.
927 {
928 auto& location_lists = wait_list->location_lists_;
929 auto it = location_lists.begin();
930 while (it != location_lists.end()) {
931 FutexWaitListNode*& head = it->second.head;
932 FutexWaitListNode*& tail = it->second.tail;
933 FutexWaitList::DeleteNodesForIsolate(isolate, &head, &tail);
934 // head and tail are either both nullptr or both non-nullptr.
935 DCHECK_EQ(head == nullptr, tail == nullptr);
936 if (head == nullptr) {
937 it = location_lists.erase(it);
938 } else {
939 ++it;
940 }
941 }
942 }
943
944 {
945 auto& isolate_map = wait_list->isolate_promises_to_resolve_;
946 auto it = isolate_map.find(isolate);
947 if (it != isolate_map.end()) {
948 for (FutexWaitListNode* node = it->second.head; node;) {
949 DCHECK(node->IsAsync());
950 DCHECK_EQ(isolate, node->async_state_->isolate_for_async_waiters);
951 node->async_state_->timeout_task_id =
952 CancelableTaskManager::kInvalidTaskId;
953 node = FutexWaitList::DeleteAsyncWaiterNode(node);
954 }
955 isolate_map.erase(it);
956 }
957 }
958
959 wait_list->Verify();
960}
