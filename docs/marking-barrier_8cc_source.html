45void MarkingBarrier::Write(Tagged<HeapObject> host, IndirectPointerSlot slot) {
46#ifdef V8_ENABLE_SANDBOX
47 DCHECK(IsCurrentMarkingBarrier(host));
48 DCHECK(is_activated_ || shared_heap_worklists_.has_value());
49 DCHECK(MemoryChunk::FromHeapObject(host)->IsMarking());
50
51 // An indirect pointer slot can only contain a Smi if it is uninitialized (in
52 // which case the vaue will be Smi::zero()). However, at this point the slot
53 // must have been initialized because it was just written to.
54 Tagged<HeapObject> value = Cast<HeapObject>(slot.load(isolate()));
55
56 // If the host is in shared space, the target must be in the shared trusted
57 // space. No other edges indirect pointers are currently possible in shared
58 // space.
59 DCHECK_IMPLIES(
60 HeapLayout::InWritableSharedSpace(host),
61 MemoryChunk::FromHeapObject(value)->Metadata()->owner()->identity() ==
62 SHARED_TRUSTED_SPACE);
63
64 if (HeapLayout::InReadOnlySpace(value)) return;
65
66 DCHECK(!HeapLayout::InYoungGeneration(value));
67
68 if (V8_UNLIKELY(uses_shared_heap_) && !is_shared_space_isolate_) {
69 if (HeapLayout::InWritableSharedSpace(value)) {
70 // References to the shared trusted space may only originate from the
71 // shared space.
72 CHECK(HeapLayout::InWritableSharedSpace(host));
73 DCHECK(MemoryChunk::FromHeapObject(value)->IsTrusted());
74 MarkValueShared(value);
75 } else {
76 MarkValueLocal(value);
77 }
78 } else {
79 MarkValueLocal(value);
80 }
81
82 // We don't need to record a slot here because the entries in the pointer
83 // tables are not compacted and because the pointers stored in the table
84 // entries are updated after compacting GC.
85 static_assert(!CodePointerTable::kSupportsCompaction);
86 static_assert(!TrustedPointerTable::kSupportsCompaction);
87#else
88 UNREACHABLE();
89#endif
90}

92void MarkingBarrier::WriteWithoutHost(Tagged<HeapObject> value) {
93 DCHECK(is_main_thread_barrier_);
94 DCHECK(is_activated_);
95
96 // Without a shared heap and on the shared space isolate (= main isolate) all
97 // objects are considered local.
98 if (V8_UNLIKELY(uses_shared_heap_) && !is_shared_space_isolate_) {
99 // On client isolates (= worker isolates) shared values can be ignored.
100 if (HeapLayout::InWritableSharedSpace(value)) {
101 return;
102 }
103 }
104 if (HeapLayout::InReadOnlySpace(value)) return;
105 MarkValueLocal(value);
106}

108void MarkingBarrier::Write(Tagged<InstructionStream> host,
109 RelocInfo* reloc_info, Tagged<HeapObject> value) {
110 DCHECK(IsCurrentMarkingBarrier(host));
111 DCHECK(!HeapLayout::InWritableSharedSpace(host));
112 DCHECK(is_activated_ || shared_heap_worklists_.has_value());
113 DCHECK(MemoryChunk::FromHeapObject(host)->IsMarking());
114
115 MarkValue(host, value);
116
117 if (is_compacting_) {
118 DCHECK(is_major());
119 if (is_main_thread_barrier_) {
120 // An optimization to avoid allocating additional typed slots for the
121 // main thread.
122 major_collector_->RecordRelocSlot(host, reloc_info, value);
123 } else {
124 RecordRelocSlot(host, reloc_info, value);
125 }
126 }

144void MarkingBarrier::Write(Tagged<DescriptorArray> descriptor_array,
145 int number_of_own_descriptors) {
146 DCHECK(IsCurrentMarkingBarrier(descriptor_array));
147 DCHECK(HeapLayout::InReadOnlySpace(descriptor_array->map()));
148 DCHECK(MemoryChunk::FromHeapObject(descriptor_array)->IsMarking());
149
150 // Only major GC uses custom liveness.
151 if (is_minor() || IsStrongDescriptorArray(descriptor_array)) {
152 MarkValueLocal(descriptor_array);
153 return;
154 }
155
156 unsigned gc_epoch;
157 MarkingWorklists::Local* worklist;
158 if (V8_UNLIKELY(uses_shared_heap_) &&
159 HeapLayout::InWritableSharedSpace(descriptor_array) &&
160 !is_shared_space_isolate_) {
161 gc_epoch = isolate()
162 ->shared_space_isolate()
163 ->heap()
164 ->mark_compact_collector()
165 ->epoch();
166 DCHECK(shared_heap_worklists_.has_value());
167 worklist = &*shared_heap_worklists_;
168 } else {
169#ifdef DEBUG
170 if (const auto target_worklist =
171 MarkingHelper::ShouldMarkObject(heap_, descriptor_array)) {
172 DCHECK_EQ(target_worklist.value(),
173 MarkingHelper::WorklistTarget::kRegular);
174 } else {
175 DCHECK(HeapLayout::InBlackAllocatedPage(descriptor_array));
176 }
177#endif // DEBUG
178 gc_epoch = major_collector_->epoch();
179 worklist = current_worklists_.get();
180 }
181
182 // The DescriptorArray needs to be marked black here to ensure that slots
183 // are recorded by the Scavenger in case the DescriptorArray is promoted
184 // while incremental marking is running. This is needed as the regular
185 // marking visitor does not re-process any already marked descriptors. If we
186 // don't mark it black here, the Scavenger may promote a DescriptorArray and
187 // any already marked descriptors will not have any slots recorded.
188 if (v8_flags.black_allocated_pages) {
189 // Make sure to only mark the descriptor array for non black allocated
190 // pages. The atomic pause will fix it afterwards.
191 if (MarkingHelper::ShouldMarkObject(heap_, descriptor_array)) {
192 marking_state_.TryMark(descriptor_array);
193 }
194 } else {
195 marking_state_.TryMark(descriptor_array);
196 }
197
198 // `TryUpdateIndicesToMark()` acts as a barrier that publishes the slots'
199 // values corresponding to `number_of_own_descriptors`.
200 if (DescriptorArrayMarkingState::TryUpdateIndicesToMark(
201 gc_epoch, descriptor_array, number_of_own_descriptors)) {
202 worklist->Push(descriptor_array);
203 }
204}

206void MarkingBarrier::RecordRelocSlot(Tagged<InstructionStream> host,
207 RelocInfo* rinfo,
208 Tagged<HeapObject> target) {
209 DCHECK(IsCurrentMarkingBarrier(host));
210 if (!MarkCompactCollector::ShouldRecordRelocSlot(host, rinfo, target)) return;
211
212 MarkCompactCollector::RecordRelocSlotInfo info =
213 MarkCompactCollector::ProcessRelocInfo(host, rinfo, target);
214
215 auto& typed_slots = typed_slots_map_[info.page_metadata];
216 if (!typed_slots) {
217 typed_slots.reset(new TypedSlots());
218 }
219 typed_slots->Insert(info.slot_type, info.offset);
220}

223template <typename Space>
224void SetGenerationPageFlags(Space* space, MarkingMode marking_mode) {
225 if constexpr (std::is_same_v<Space, OldSpace> ||
226 std::is_same_v<Space, SharedSpace> ||
227 std::is_same_v<Space, TrustedSpace> ||
228 std::is_same_v<Space, CodeSpace>) {
229 for (auto* p : *space) {
230 p->SetOldGenerationPageFlags(marking_mode);
231 }
232 } else if constexpr (std::is_same_v<Space, OldLargeObjectSpace> ||
233 std::is_same_v<Space, SharedLargeObjectSpace> ||
234 std::is_same_v<Space, TrustedLargeObjectSpace> ||
235 std::is_same_v<Space, CodeLargeObjectSpace>) {
236 for (auto* p : *space) {
237 DCHECK(p->Chunk()->IsLargePage());
238 p->SetOldGenerationPageFlags(marking_mode);
239 }
240 } else if constexpr (std::is_same_v<Space, NewSpace>) {
241 for (auto* p : *space) {
242 p->SetYoungGenerationPageFlags(marking_mode);
243 }
244 } else {
245 static_assert(std::is_same_v<Space, NewLargeObjectSpace>);
246 for (auto* p : *space) {
247 DCHECK(p->Chunk()->IsLargePage());
248 p->SetYoungGenerationPageFlags(marking_mode);
249 }
250 }
251}

263void ActivateSpaces(Heap* heap, MarkingMode marking_mode) {
264 ActivateSpace(heap->old_space(), marking_mode);
265 ActivateSpace(heap->lo_space(), marking_mode);
266 if (heap->new_space()) {
267 DCHECK(!v8_flags.sticky_mark_bits);
268 ActivateSpace(heap->new_space(), marking_mode);
269 }
270 ActivateSpace(heap->new_lo_space(), marking_mode);
271 {
272 RwxMemoryWriteScope scope("For writing flags.");
273 ActivateSpace(heap->code_space(), marking_mode);
274 ActivateSpace(heap->code_lo_space(), marking_mode);
275 }
276
277 if (marking_mode == MarkingMode::kMajorMarking) {
278 if (heap->shared_space()) {
279 ActivateSpace(heap->shared_space(), marking_mode);
280 }
281 if (heap->shared_lo_space()) {
282 ActivateSpace(heap->shared_lo_space(), marking_mode);
283 }
284 }
285
286 ActivateSpace(heap->trusted_space(), marking_mode);
287 ActivateSpace(heap->trusted_lo_space(), marking_mode);
288}
289
290void DeactivateSpaces(Heap* heap, MarkingMode marking_mode) {
291 DeactivateSpace(heap->old_space());
292 DeactivateSpace(heap->lo_space());
293 if (heap->new_space()) {
294 DCHECK(!v8_flags.sticky_mark_bits);
295 DeactivateSpace(heap->new_space());
296 }
297 DeactivateSpace(heap->new_lo_space());
298 {
299 RwxMemoryWriteScope scope("For writing flags.");
300 DeactivateSpace(heap->code_space());
301 DeactivateSpace(heap->code_lo_space());
302 }
303
304 if (marking_mode == MarkingMode::kMajorMarking) {
305 if (heap->shared_space()) {
306 DeactivateSpace(heap->shared_space());
307 }
308 if (heap->shared_lo_space()) {
309 DeactivateSpace(heap->shared_lo_space());
310 }
311 }
312
313 DeactivateSpace(heap->trusted_space());
314 DeactivateSpace(heap->trusted_lo_space());
315}

319void MarkingBarrier::ActivateAll(Heap* heap, bool is_compacting) {
320 ActivateSpaces(heap, MarkingMode::kMajorMarking);
321
322 heap->safepoint()->IterateLocalHeaps([is_compacting](LocalHeap* local_heap) {
323 local_heap->marking_barrier()->Activate(is_compacting,
324 MarkingMode::kMajorMarking);
325 });
326
327 if (heap->isolate()->is_shared_space_isolate()) {
328 heap->isolate()
329 ->shared_space_isolate()
330 ->global_safepoint()
331 ->IterateClientIsolates([](Isolate* client) {
332 // Force the RecordWrite builtin into the incremental marking code
333 // path.
334 client->heap()->SetIsMarkingFlag(true);
335 client->heap()->safepoint()->IterateLocalHeaps(
336 [](LocalHeap* local_heap) {
337 local_heap->marking_barrier()->ActivateShared();
338 });
339 });
340 }
341}

446void MarkingBarrier::PublishIfNeeded() {
447 if (is_activated_) {
448 current_worklists_->Publish();
449 for (auto& it : typed_slots_map_) {
450 MutablePageMetadata* memory_chunk = it.first;
451 // Access to TypeSlots need to be protected, since LocalHeaps might
452 // publish code in the background thread.
453 base::SpinningMutexGuard guard(memory_chunk->mutex());
454 std::unique_ptr<TypedSlots>& typed_slots = it.second;
455 RememberedSet<OLD_TO_OLD>::MergeTyped(memory_chunk,
456 std::move(typed_slots));
457 }
458 typed_slots_map_.clear();
459 }
