24bool IsSamePage(i::Address ptr1, i::Address ptr2) {
25 const uint32_t kPageSize = 4096;
26 i::Address mask = ~static_cast<i::Address>(kPageSize - 1);
27 return (ptr1 & mask) == (ptr2 & mask);
28}

30// Check if the code at specified address could potentially be a
31// frame setup code.
32bool IsNoFrameRegion(i::Address address) {
33 struct Pattern {
34 int bytes_count;
35 uint8_t bytes[8];
36 int offsets[4];
37 };
38 static Pattern patterns[] = {
39#if V8_HOST_ARCH_IA32
40 // push %ebp
41 // mov %esp,%ebp
42 {3, {0x55, 0x89, 0xE5}, {0, 1, -1}},
43 // pop %ebp
44 // ret N
45 {2, {0x5D, 0xC2}, {0, 1, -1}},
46 // pop %ebp
47 // ret
48 {2, {0x5D, 0xC3}, {0, 1, -1}},
49#elif V8_HOST_ARCH_X64
50 // pushq %rbp
51 // movq %rsp,%rbp
52 {4, {0x55, 0x48, 0x89, 0xE5}, {0, 1, -1}},
53 // popq %rbp
54 // ret N
55 {2, {0x5D, 0xC2}, {0, 1, -1}},
56 // popq %rbp
57 // ret
58 {2, {0x5D, 0xC3}, {0, 1, -1}},
59#endif
60 {0, {}, {}}
61 };
62 uint8_t* pc = reinterpret_cast<uint8_t*>(address);
63 for (Pattern* pattern = patterns; pattern->bytes_count; ++pattern) {
64 for (int* offset_ptr = pattern->offsets; *offset_ptr != -1; ++offset_ptr) {
65 int offset = *offset_ptr;
66 if (!offset || IsSamePage(address, address - offset)) {
67 MSAN_MEMORY_IS_INITIALIZED(pc - offset, pattern->bytes_count);
68 if (!memcmp(pc - offset, pattern->bytes, pattern->bytes_count))
69 return true;
70 } else {
71 // It is not safe to examine bytes on another page as it might not be
72 // allocated thus causing a SEGFAULT.
73 // Check the pattern part that's on the same page and
74 // pessimistically assume it could be the entire pattern match.
75 MSAN_MEMORY_IS_INITIALIZED(pc, pattern->bytes_count - offset);
76 if (!memcmp(pc, pattern->bytes + offset, pattern->bytes_count - offset))
77 return true;
78 }
79 }
80 }
81 return false;
82}

92bool SimulatorHelper::FillRegisters(Isolate* isolate,
93 v8::RegisterState* state) {
94 Simulator* simulator = isolate->thread_local_top()->simulator_;
95 // Check if there is active simulator.
96 if (simulator == nullptr) return false;
97#if V8_TARGET_ARCH_ARM
98 if (!simulator->has_bad_pc()) {
99 state->pc = reinterpret_cast<void*>(simulator->get_pc());
100 }
101 state->sp = reinterpret_cast<void*>(simulator->get_register(Simulator::sp));
102 state->fp = reinterpret_cast<void*>(simulator->get_register(Simulator::r11));
103 state->lr = reinterpret_cast<void*>(simulator->get_register(Simulator::lr));
104#elif V8_TARGET_ARCH_ARM64
105 state->pc = reinterpret_cast<void*>(simulator->pc());
106 state->sp = reinterpret_cast<void*>(simulator->sp());
107 state->fp = reinterpret_cast<void*>(simulator->fp());
108 state->lr = reinterpret_cast<void*>(simulator->lr());
109#elif V8_TARGET_ARCH_MIPS64 || V8_TARGET_ARCH_LOONG64
110 if (!simulator->has_bad_pc()) {
111 state->pc = reinterpret_cast<void*>(simulator->get_pc());
112 }
113 state->sp = reinterpret_cast<void*>(simulator->get_register(Simulator::sp));
114 state->fp = reinterpret_cast<void*>(simulator->get_register(Simulator::fp));
115#elif V8_TARGET_ARCH_PPC64
116 if (!simulator->has_bad_pc()) {
117 state->pc = reinterpret_cast<void*>(simulator->get_pc());
118 }
119 state->sp = reinterpret_cast<void*>(simulator->get_register(Simulator::sp));
120 state->fp = reinterpret_cast<void*>(simulator->get_register(Simulator::fp));
121 state->lr = reinterpret_cast<void*>(simulator->get_lr());
122#elif V8_TARGET_ARCH_S390X
123 if (!simulator->has_bad_pc()) {
124 state->pc = reinterpret_cast<void*>(simulator->get_pc());
125 }
126 state->sp = reinterpret_cast<void*>(simulator->get_register(Simulator::sp));
127 state->fp = reinterpret_cast<void*>(simulator->get_register(Simulator::fp));
128 state->lr = reinterpret_cast<void*>(simulator->get_register(Simulator::ra));
129#elif V8_TARGET_ARCH_RISCV64
130 if (!simulator->has_bad_pc()) {
131 state->pc = reinterpret_cast<void*>(simulator->get_pc());
132 }
133 state->sp = reinterpret_cast<void*>(simulator->get_register(Simulator::sp));
134 state->fp = reinterpret_cast<void*>(simulator->get_register(Simulator::fp));
135 state->lr = reinterpret_cast<void*>(simulator->get_register(Simulator::ra));
136#elif V8_TARGET_ARCH_RISCV32
137 if (!simulator->has_bad_pc()) {
138 state->pc = reinterpret_cast<void*>(simulator->get_pc());
139 }
140 state->sp = reinterpret_cast<void*>(simulator->get_register(Simulator::sp));
141 state->fp = reinterpret_cast<void*>(simulator->get_register(Simulator::fp));
142 state->lr = reinterpret_cast<void*>(simulator->get_register(Simulator::ra));
143#endif
144 if (state->sp == 0 || state->fp == 0) {
145 // It possible that the simulator is interrupted while it is updating
146 // the sp or fp register. ARM64 simulator does this in two steps:
147 // first setting it to zero and then setting it to the new value.
148 // Bailout if sp/fp doesn't contain the new value.
149 //
150 // FIXME: The above doesn't really solve the issue.
151 // If a 64-bit target is executed on a 32-bit host even the final
152 // write is non-atomic, so it might obtain a half of the result.
153 // Moreover as long as the register set code uses memcpy (as of now),
154 // it is not guaranteed to be atomic even when both host and target
155 // are of same bitness.
156 return false;
157 }
158 return true;
159}

164DISABLE_ASAN void TickSample::Init(Isolate* v8_isolate,
165 const RegisterState& reg_state,
166 RecordCEntryFrame record_c_entry_frame,
167 bool update_stats,
168 bool use_simulator_reg_state,
169 base::TimeDelta sampling_interval) {
170 update_stats_ = update_stats;
171 SampleInfo info;
172 RegisterState regs = reg_state;
173 if (!GetStackSample(v8_isolate, &regs, record_c_entry_frame, stack,
174 kMaxFramesCount, &info, &state,
175 use_simulator_reg_state)) {
176 // It is executing JS but failed to collect a stack trace.
177 // Mark the sample as spoiled.
178 pc = nullptr;
179 return;
180 }
181
182 if (state != StateTag::EXTERNAL) {
183 state = info.vm_state;
184 }
185 pc = regs.pc;
186 frames_count = static_cast<unsigned>(info.frames_count);
187 has_external_callback = info.external_callback_entry != nullptr;
188 context = info.context;
189 embedder_context = info.embedder_context;
190 embedder_state = info.embedder_state;
191 if (has_external_callback) {
192 external_callback_entry = info.external_callback_entry;
193 } else if (frames_count) {
194 // sp register may point at an arbitrary place in memory, make
195 // sure sanitizers don't complain about it.
196 ASAN_UNPOISON_MEMORY_REGION(regs.sp, sizeof(void*));
197 MSAN_MEMORY_IS_INITIALIZED(regs.sp, sizeof(void*));
198 // Sample potential return address value for frameless invocation of
199 // stubs (we'll figure out later, if this value makes sense).
200
201 // TODO(petermarshall): This read causes guard page violations on Windows.
202 // Either fix this mechanism for frameless stubs or remove it.
203 // tos =
204 // i::ReadUnalignedValue<void*>(reinterpret_cast<i::Address>(regs.sp));
205 tos = nullptr;
206 } else {
207 tos = nullptr;
208 }
209 sampling_interval_ = sampling_interval;
210 timestamp = base::TimeTicks::Now();
211}

213// IMPORTANT: 'GetStackSample' is sensitive to stack overflows. For this reason
214// we try not to use any function/method marked as V8_EXPORT_PRIVATE with their
215// only use-site in 'GetStackSample': The resulting linker stub needs quite
216// a bit of stack space and has caused stack overflow crashes in the past.
217bool TickSample::GetStackSample(Isolate* v8_isolate, RegisterState* regs,
218 RecordCEntryFrame record_c_entry_frame,
219 void** frames, size_t frames_limit,
220 v8::SampleInfo* sample_info,
221 StateTag* out_state,
222 bool use_simulator_reg_state) {
223 i::Isolate* isolate = reinterpret_cast<i::Isolate*>(v8_isolate);
224 sample_info->frames_count = 0;
225 sample_info->vm_state = isolate->current_vm_state();
226 sample_info->external_callback_entry = nullptr;
227 sample_info->embedder_state = EmbedderStateTag::EMPTY;
228 sample_info->embedder_context = nullptr;
229 sample_info->context = nullptr;
230
231 if (sample_info->vm_state == GC || v8_isolate->heap()->IsInGC()) {
232 // GC can happen any time, not directly caused by its caller. Don't collect
233 // stacks for it. We check for both GC VMState and IsInGC, since we can
234 // observe LOGGING VM states during GC.
235 // TODO(leszeks): We could still consider GC stacks (as long as this isn't a
236 // moving GC), e.g. to surface if one particular function is triggering all
237 // the GCs. However, this is a user-visible change, and we would need to
238 // adjust the symbolizer and devtools to expose this information.
239 return true;
240 }
241
242 EmbedderState* embedder_state = isolate->current_embedder_state();
243 if (embedder_state != nullptr) {
244 sample_info->embedder_context =
245 reinterpret_cast<void*>(embedder_state->native_context_address());
246 sample_info->embedder_state = embedder_state->GetState();
247 }
248
249 Tagged<Context> top_context = isolate->context();
250 if (top_context.ptr() != i::Context::kNoContext &&
251 top_context.ptr() != i::Context::kInvalidContext) {
252 Tagged<NativeContext> top_native_context = top_context->native_context();
253 sample_info->context = reinterpret_cast<void*>(top_native_context.ptr());
254 }
255
256 i::Address js_entry_sp = isolate->js_entry_sp();
257 if (js_entry_sp == 0) return true; // Not executing JS now.
258
259#if defined(USE_SIMULATOR)
260 if (use_simulator_reg_state) {
261 if (!i::SimulatorHelper::FillRegisters(isolate, regs)) {
262 i::ProfilerStats::Instance()->AddReason(
263 i::ProfilerStats::Reason::kSimulatorFillRegistersFailed);
264 return false;
265 }
266 }
267#else
268 USE(use_simulator_reg_state);
269#endif
270 DCHECK(regs->sp);
271
272 // Check whether we interrupted setup/teardown of a stack frame in JS code.
273 // Avoid this check for C++ code, as that would trigger false positives.
274 // TODO(petermarshall): Code range is always null on ia32 so this check for
275 // IsNoFrameRegion will never actually run there.
276 if (regs->pc &&
277 isolate->heap()->code_region().contains(
278 reinterpret_cast<i::Address>(regs->pc)) &&
279 IsNoFrameRegion(reinterpret_cast<i::Address>(regs->pc))) {
280 // The frame is not setup, so it'd be hard to iterate the stack. Bailout.
281 i::ProfilerStats::Instance()->AddReason(
282 i::ProfilerStats::Reason::kNoFrameRegion);
283 return false;
284 }
285
286 i::ExternalCallbackScope* scope = isolate->external_callback_scope();
287 i::Address handler = i::Isolate::handler(isolate->thread_local_top());
288 // If there is a handler on top of the external callback scope then
289 // we have already entered JavaScript again and the external callback
290 // is not the top function.
291 if (scope && scope->JSStackComparableAddress() < handler) {
292 i::Address* external_callback_entry_ptr =
293 scope->callback_entrypoint_address();
294 sample_info->external_callback_entry =
295 external_callback_entry_ptr == nullptr
296 ? nullptr
297 : reinterpret_cast<void*>(*external_callback_entry_ptr);
298 }
299 // 'Fast API calls' are similar to fast C calls (see frames.cc) in that
300 // they don't build an exit frame when entering C from JS. They have the
301 // added speciality of having separate "fast" and "default" callbacks, the
302 // latter being the regular API callback called before the JS function is
303 // optimized. When TurboFan optimizes the JS caller, the fast callback
304 // gets executed instead of the default one, therefore we need to store
305 // its address in the sample.
306 IsolateData* isolate_data = isolate->isolate_data();
307 Address fast_c_fp = isolate_data->fast_c_call_caller_fp();
308 if (fast_c_fp != kNullAddress &&
309 isolate_data->fast_api_call_target() != kNullAddress) {
310 sample_info->external_callback_entry =
311 reinterpret_cast<void*>(isolate_data->fast_api_call_target());
312 if (out_state) {
313 *out_state = StateTag::EXTERNAL;
314 }
315 }
316
317 i::StackFrameIteratorForProfiler it(
318 isolate, reinterpret_cast<i::Address>(regs->pc),
319 reinterpret_cast<i::Address>(regs->fp),
320 reinterpret_cast<i::Address>(regs->sp),
321 reinterpret_cast<i::Address>(regs->lr), js_entry_sp);
322
323 if (it.done()) return true;
324
325 size_t i = 0;
326 if (record_c_entry_frame == kIncludeCEntryFrame &&
327 (it.top_frame_type() == internal::StackFrame::EXIT ||
328 it.top_frame_type() == internal::StackFrame::BUILTIN_EXIT)) {
329 // While BUILTIN_EXIT definitely represents a call to CEntry the EXIT frame
330 // might represent either a call to CEntry or an optimized call to
331 // Api callback. In the latter case the ExternalCallbackScope points to
332 // the same function, so skip adding a frame in that case in order to avoid
333 // double-reporting.
334 void* c_function = reinterpret_cast<void*>(isolate->c_function());
335 if (sample_info->external_callback_entry != c_function) {
336 frames[i] = c_function;
337 i++;
338 }
339 }
340#ifdef V8_RUNTIME_CALL_STATS
341 i::RuntimeCallTimer* timer =
342 isolate->counters()->runtime_call_stats()->current_timer();
343#endif // V8_RUNTIME_CALL_STATS
344 for (; !it.done() && i < frames_limit; it.Advance()) {
345#ifdef V8_RUNTIME_CALL_STATS
346 while (timer && reinterpret_cast<i::Address>(timer) < it.frame()->fp() &&
347 i < frames_limit) {
348 frames[i++] = reinterpret_cast<void*>(timer->counter());
349 timer = timer->parent();
350 }
351 if (i == frames_limit) break;
352#endif // V8_RUNTIME_CALL_STATS
353
354 if (it.frame()->is_interpreted()) {
355 // For interpreted frames use the bytecode array pointer as the pc.
356 i::InterpretedFrame* frame =
357 static_cast<i::InterpretedFrame*>(it.frame());
358 // Since the sampler can interrupt execution at any point the
359 // bytecode_array might be garbage, so don't actually dereference it. We
360 // avoid the frame->GetXXX functions since they call Cast<BytecodeArray>,
361 // which has a heap access in its DCHECK.
362 i::Address bytecode_array = base::Memory<i::Address>(
363 frame->fp() + i::InterpreterFrameConstants::kBytecodeArrayFromFp);
364 i::Address bytecode_offset = base::Memory<i::Address>(
365 frame->fp() + i::InterpreterFrameConstants::kBytecodeOffsetFromFp);
366
367 // If the bytecode array is a heap object and the bytecode offset is a
368 // Smi, use those, otherwise fall back to using the frame's pc.
369 if (HAS_STRONG_HEAP_OBJECT_TAG(bytecode_array) &&
370 HAS_SMI_TAG(bytecode_offset)) {
371 frames[i++] = reinterpret_cast<void*>(
372 bytecode_array + i::Internals::SmiValue(bytecode_offset));
373 continue;
374 }
375 }
376 // For arm64, the PC for the frame sometimes doesn't come from the stack,
377 // but from the link register instead. For this reason, we skip
378 // authenticating it.
379 frames[i++] = reinterpret_cast<void*>(it.frame()->unauthenticated_pc());
380 }
381 sample_info->frames_count = i;
382 return true;
383}

385void TickSample::print() const {
386 PrintF("TickSample: at %p\n", this);
387 PrintF(" - state: %s\n", StateToString(state));
388 PrintF(" - pc: %p\n", pc);
389 PrintF(" - stack: (%u frames)\n", frames_count);
390 for (unsigned i = 0; i < frames_count; i++) {
391 PrintF(" %p\n", stack[i]);
392 }
393 PrintF(" - has_external_callback: %d\n", has_external_callback);
394 PrintF(" - %s: %p\n",
395 has_external_callback ? "external_callback_entry" : "tos", tos);
396 PrintF(" - update_stats: %d\n", update_stats_);
397 PrintF(" - sampling_interval: %" PRId64 "\n",
398 sampling_interval_.InMicroseconds());
399 PrintF("\n");
400}
