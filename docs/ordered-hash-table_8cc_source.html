18template <class Derived, int entrysize>
19MaybeHandle<Derived> OrderedHashTable<Derived, entrysize>::Allocate(
20 Isolate* isolate, int capacity, AllocationType allocation) {
21 // Capacity must be a power of two, since we depend on being able
22 // to divide and multiple by 2 (kLoadFactor) to derive capacity
23 // from number of buckets. If we decide to change kLoadFactor
24 // to something other than 2, capacity should be stored as another
25 // field of this object.
26 capacity =
27 base::bits::RoundUpToPowerOfTwo32(std::max({kInitialCapacity, capacity}));
28 if (capacity > MaxCapacity()) {
29 THROW_NEW_ERROR_RETURN_VALUE(
30 isolate, NewRangeError(MessageTemplate::kTooManyProperties), {});
31 }
32 int num_buckets = capacity / kLoadFactor;
33 Handle<FixedArray> backing_store = isolate->factory()->NewFixedArrayWithMap(
34 Derived::GetMap(isolate->roots_table()),
35 HashTableStartIndex() + num_buckets + (capacity * kEntrySize),
36 allocation);
37 Handle<Derived> table = Cast<Derived>(backing_store);
38 DisallowGarbageCollection no_gc;
39 Tagged<Derived> raw_table = *table;
40 for (int i = 0; i < num_buckets; ++i) {
41 raw_table->set(HashTableStartIndex() + i, Smi::FromInt(kNotFound));
42 }
43 raw_table->SetNumberOfBuckets(num_buckets);
44 raw_table->SetNumberOfElements(0);
45 raw_table->SetNumberOfDeletedElements(0);
46 return table;
47}

69template <class Derived, int entrysize>
70MaybeHandle<Derived>
71OrderedHashTable<Derived, entrysize>::EnsureCapacityForAdding(
72 Isolate* isolate, Handle<Derived> table) {
73 DCHECK(!table->IsObsolete());
74
75 int nof = table->NumberOfElements();
76 int nod = table->NumberOfDeletedElements();
77 int capacity = table->Capacity();
78 if ((nof + nod) < capacity) return table;
79
80 int new_capacity;
81 if (capacity == 0) {
82 // step from empty to minimum proper size
83 new_capacity = kInitialCapacity;
84 } else if (nod >= (capacity >> 1)) {
85 // Don't need to grow if we can simply clear out deleted entries instead.
86 // Note that we can't compact in place, though, so we always allocate
87 // a new table.
88 new_capacity = capacity;
89 } else {
90 new_capacity = capacity << 1;
91 }
92
93 return Derived::Rehash(isolate, table, new_capacity);
94}

109 Isolate* isolate, Handle<Derived> table) {
110 DCHECK(!table->IsObsolete());
111
112 AllocationType allocation_type = HeapLayout::InYoungGeneration(*table)
113 ? AllocationType::kYoung
114 : AllocationType::kOld;
115
116 Handle<Derived> new_table =

139template <class Derived, int entrysize>
140InternalIndex OrderedHashTable<Derived, entrysize>::FindEntry(
141 Isolate* isolate, Tagged<Object> key) {
142 if (NumberOfElements() == 0) {
143 // This is not just an optimization but also ensures that we do the right
144 // thing if Capacity() == 0
145 return InternalIndex::NotFound();
146 }
147
148 int raw_entry;
149 // This special cases for Smi, so that we avoid the HandleScope
150 // creation below.
151 if (IsSmi(key)) {
152 uint32_t hash = ComputeUnseededHash(Smi::ToInt(key));
153 raw_entry = HashToEntryRaw(hash & Smi::kMaxValue);
154 } else {
155 HandleScope scope(isolate);
156 Tagged<Object> hash = Object::GetHash(key);
157 // If the object does not have an identity hash, it was never used as a key
158 if (IsUndefined(hash, isolate)) return InternalIndex::NotFound();
159 raw_entry = HashToEntryRaw(Smi::ToInt(hash));
160 }
161
162 // Walk the chain in the bucket to find the key.
163 while (raw_entry != kNotFound) {
164 Tagged<Object> candidate_key = KeyAt(InternalIndex(raw_entry));
165 if (Object::SameValueZero(candidate_key, key))
166 return InternalIndex(raw_entry);
167 raw_entry = NextChainEntryRaw(raw_entry);
168 }
169
170 return InternalIndex::NotFound();
171}

173MaybeHandle<OrderedHashSet> OrderedHashSet::Add(Isolate* isolate,
174 Handle<OrderedHashSet> table,
175 DirectHandle<Object> key) {
176 int hash;
177 {
178 DisallowGarbageCollection no_gc;
179 Tagged<Object> raw_key = *key;
180 Tagged<OrderedHashSet> raw_table = *table;
181 hash = Object::GetOrCreateHash(raw_key, isolate).value();
182 if (raw_table->NumberOfElements() > 0) {
183 int raw_entry = raw_table->HashToEntryRaw(hash);
184 // Walk the chain of the bucket and try finding the key.
185 while (raw_entry != kNotFound) {
186 Tagged<Object> candidate_key =
187 raw_table->KeyAt(InternalIndex(raw_entry));
188 // Do not add if we have the key already
189 if (Object::SameValueZero(candidate_key, raw_key)) return table;
190 raw_entry = raw_table->NextChainEntryRaw(raw_entry);
191 }
192 }
193 }
194
195 MaybeHandle<OrderedHashSet> table_candidate =
196 OrderedHashSet::EnsureCapacityForAdding(isolate, table);
197 if (!table_candidate.ToHandle(&table)) {
198 CHECK(isolate->has_exception());
199 return table_candidate;
200 }
201 DisallowGarbageCollection no_gc;
202 Tagged<OrderedHashSet> raw_table = *table;
203 // Read the existing bucket values.
204 int bucket = raw_table->HashToBucket(hash);
205 int previous_entry = raw_table->HashToEntryRaw(hash);
206 int nof = raw_table->NumberOfElements();
207 // Insert a new entry at the end,
208 int new_entry = nof + raw_table->NumberOfDeletedElements();
209 int new_index = raw_table->EntryToIndexRaw(new_entry);
210 raw_table->set(new_index, *key);
211 raw_table->set(new_index + kChainOffset, Smi::FromInt(previous_entry));
212 // and point the bucket to the new entry.
213 raw_table->set(HashTableStartIndex() + bucket, Smi::FromInt(new_entry));
214 raw_table->SetNumberOfElements(nof + 1);
215 return table;
216}

218Handle<FixedArray> OrderedHashSet::ConvertToKeysArray(
219 Isolate* isolate, Handle<OrderedHashSet> table, GetKeysConversion convert) {
220 int length = table->NumberOfElements();
221 int nof_buckets = table->NumberOfBuckets();
222 // Convert the dictionary to a linear list.
223 Handle<FixedArray> result = Cast<FixedArray>(table);
224 // From this point on table is no longer a valid OrderedHashSet.
225 result->set_map(isolate, ReadOnlyRoots(isolate).fixed_array_map());
226 int const kMaxStringTableEntries =
227 isolate->heap()->MaxNumberToStringCacheSize();
228 for (int i = 0; i < length; i++) {
229 int index = HashTableStartIndex() + nof_buckets + (i * kEntrySize);
230 Tagged<Object> key = table->get(index);
231 uint32_t index_value;
232 if (convert == GetKeysConversion::kConvertToString) {
233 if (Object::ToArrayIndex(key, &index_value)) {
234 // Avoid trashing the Number2String cache if indices get very large.
235 bool use_cache = i < kMaxStringTableEntries;
236 key = *isolate->factory()->Uint32ToString(index_value, use_cache);
237 } else {
238 CHECK(IsName(key));
239 }
240 } else if (convert == GetKeysConversion::kNoNumbers) {
241 DCHECK(!Object::ToArrayIndex(key, &index_value));
242 }
243 result->set(i, key);
244 }
245 return FixedArray::RightTrimOrEmpty(isolate, result, length);
246}

263template <class Derived, int entrysize>
264MaybeHandle<Derived> OrderedHashTable<Derived, entrysize>::Rehash(
265 Isolate* isolate, Handle<Derived> table, int new_capacity) {
266 DCHECK(!table->IsObsolete());
267
268 MaybeHandle<Derived> new_table_candidate = Derived::Allocate(
269 isolate, new_capacity,
270 HeapLayout::InYoungGeneration(*table) ? AllocationType::kYoung
271 : AllocationType::kOld);
272 Handle<Derived> new_table;
273 if (!new_table_candidate.ToHandle(&new_table)) {
274 return new_table_candidate;
275 }
276 int new_buckets = new_table->NumberOfBuckets();
277 int new_entry = 0;
278 int removed_holes_index = 0;
279
280 DisallowGarbageCollection no_gc;
281
282 for (InternalIndex old_entry : table->IterateEntries()) {
283 int old_entry_raw = old_entry.as_int();
284 Tagged<Object> key = table->KeyAt(old_entry);
285 if (IsHashTableHole(key, isolate)) {
286 table->SetRemovedIndexAt(removed_holes_index++, old_entry_raw);
287 continue;
288 }
289
290 Tagged<Object> hash = Object::GetHash(key);
291 int bucket = Smi::ToInt(hash) & (new_buckets - 1);
292 Tagged<Object> chain_entry = new_table->get(HashTableStartIndex() + bucket);
293 new_table->set(HashTableStartIndex() + bucket, Smi::FromInt(new_entry));
294 int new_index = new_table->EntryToIndexRaw(new_entry);
295 int old_index = table->EntryToIndexRaw(old_entry_raw);
296 for (int i = 0; i < entrysize; ++i) {
297 Tagged<Object> value = table->get(old_index + i);
298 new_table->set(new_index + i, value);
299 }
300 new_table->set(new_index + kChainOffset, chain_entry);
301 ++new_entry;
302 }
303
304 DCHECK_EQ(table->NumberOfDeletedElements(), removed_holes_index);
305
306 new_table->SetNumberOfElements(table->NumberOfElements());
307 if (table->NumberOfBuckets() > 0) {
308 // Don't try to modify the empty canonical table which lives in RO space.
309 table->SetNextTable(*new_table);
310 }
311
312 return new_table_candidate;
313}

348template <class Derived, int entrysize>
349bool OrderedHashTable<Derived, entrysize>::Delete(Isolate* isolate,
350 Tagged<Derived> table,
351 Tagged<Object> key) {
352 DisallowGarbageCollection no_gc;
353 InternalIndex entry = table->FindEntry(isolate, key);
354 if (entry.is_not_found()) return false;
355
356 int nof = table->NumberOfElements();
357 int nod = table->NumberOfDeletedElements();
358 int index = table->EntryToIndex(entry);
359
360 Tagged<Object> hash_table_hole =
361 ReadOnlyRoots(isolate).hash_table_hole_value();
362 for (int i = 0; i < entrysize; ++i) {
363 table->set(index + i, hash_table_hole);
364 }
365
366 table->SetNumberOfElements(nof - 1);
367 table->SetNumberOfDeletedElements(nod + 1);
368
369 return true;
370}

434template <typename IsolateT>
435InternalIndex OrderedNameDictionary::FindEntry(IsolateT* isolate,
436 Tagged<Object> key) {
437 DisallowGarbageCollection no_gc;
438
439 DCHECK(IsUniqueName(key));
440 Tagged<Name> raw_key = Cast<Name>(key);
441
442 if (NumberOfElements() == 0) {
443 // This is not just an optimization but also ensures that we do the right
444 // thing if Capacity() == 0
445 return InternalIndex::NotFound();
446 }
447
448 int raw_entry = HashToEntryRaw(raw_key->hash());
449 while (raw_entry != kNotFound) {
450 InternalIndex entry(raw_entry);
451 Tagged<Object> candidate_key = KeyAt(entry);
452 DCHECK(IsHashTableHole(candidate_key) ||
453 IsUniqueName(Cast<Name>(candidate_key)));
454 if (candidate_key == raw_key) return entry;
455
456 // TODO(gsathya): This is loading the bucket count from the hash
457 // table for every iteration. This should be peeled out of the
458 // loop.
459 raw_entry = NextChainEntryRaw(raw_entry);
460 }
461
462 return InternalIndex::NotFound();
463}

670template <class Derived>
671void SmallOrderedHashTable<Derived>::Initialize(Isolate* isolate,
672 int capacity) {
673 DisallowGarbageCollection no_gc;
674 int num_buckets = capacity / kLoadFactor;
675 int num_chains = capacity;
676
677 SetNumberOfBuckets(num_buckets);
678 SetNumberOfElements(0);
679 SetNumberOfDeletedElements(0);
680 memset(reinterpret_cast<void*>(field_address(PaddingOffset())), 0,
681 PaddingSize());
682
683 Address hashtable_start = GetHashTableStartAddress(capacity);
684 memset(reinterpret_cast<uint8_t*>(hashtable_start), kNotFound,
685 num_buckets + num_chains);
686
687 MemsetTagged(RawField(DataTableStartOffset()),
688 ReadOnlyRoots(isolate).the_hole_value(),
689 capacity * Derived::kEntrySize);
690
691#ifdef DEBUG
692 for (int i = 0; i < num_buckets; ++i) {
693 DCHECK_EQ(kNotFound, GetFirstEntry(i));
694 }
695
696 for (int i = 0; i < num_chains; ++i) {
697 DCHECK_EQ(kNotFound, GetNextEntry(i));
698 }
699
700 for (int i = 0; i < capacity; ++i) {
701 for (int j = 0; j < Derived::kEntrySize; j++) {
702 DCHECK_EQ(ReadOnlyRoots(isolate).the_hole_value(), GetDataEntry(i, j));
703 }
704 }
705#endif // DEBUG
706}

709 Isolate* isolate, Handle<SmallOrderedHashSet> table,
710 DirectHandle<Object> key) {
711 if (table->HasKey(isolate, key)) return table;
712
713 if (table->UsedCapacity() >= table->Capacity()) {
714 MaybeHandle<SmallOrderedHashSet> new_table =
715 SmallOrderedHashSet::Grow(isolate, table);
716 if (!new_table.ToHandle(&table)) {
717 return MaybeHandle<SmallOrderedHashSet>();
718 }
719 }

721 DisallowGarbageCollection no_gc;
722 Tagged<SmallOrderedHashSet> raw_table = *table;
723 int hash = Object::GetOrCreateHash(*key, isolate).value();
724 int nof = raw_table->NumberOfElements();
725
726 // Read the existing bucket values.
727 int bucket = raw_table->HashToBucket(hash);
728 int previous_entry = raw_table->HashToFirstEntry(hash);
729
730 // Insert a new entry at the end,
731 int new_entry = nof + raw_table->NumberOfDeletedElements();
732
733 raw_table->SetDataEntry(new_entry, SmallOrderedHashSet::kKeyIndex, *key);
734 raw_table->SetFirstEntry(bucket, new_entry);
735 raw_table->SetNextEntry(new_entry, previous_entry);
736
737 // and update book keeping.
738 raw_table->SetNumberOfElements(nof + 1);
739
740 return table;

926template <class Derived>
927Handle<Derived> SmallOrderedHashTable<Derived>::Rehash(Isolate* isolate,
928 Handle<Derived> table,
929 int new_capacity) {
930 DCHECK_GE(kMaxCapacity, new_capacity);
931
932 Handle<Derived> new_table = SmallOrderedHashTable<Derived>::Allocate(
933 isolate, new_capacity,
934 HeapLayout::InYoungGeneration(*table) ? AllocationType::kYoung
935 : AllocationType::kOld);
936 int new_entry = 0;
937
938 {
939 DisallowGarbageCollection no_gc;
940 for (InternalIndex old_entry : table->IterateEntries()) {
941 Tagged<Object> key = table->KeyAt(old_entry);
942 if (IsTheHole(key, isolate)) continue;
943
944 int hash = Smi::ToInt(Object::GetHash(key));
945 int bucket = new_table->HashToBucket(hash);
946 int chain = new_table->GetFirstEntry(bucket);
947
948 new_table->SetFirstEntry(bucket, new_entry);
949 new_table->SetNextEntry(new_entry, chain);
950
951 for (int i = 0; i < Derived::kEntrySize; ++i) {
952 Tagged<Object> value = table->GetDataEntry(old_entry.as_int(), i);
953 new_table->SetDataEntry(new_entry, i, value);
954 }
955
956 ++new_entry;
957 }
958
959 new_table->SetNumberOfElements(table->NumberOfElements());
960 }
961 return new_table;
962}

995template <class Derived>
996MaybeHandle<Derived> SmallOrderedHashTable<Derived>::Grow(
997 Isolate* isolate, Handle<Derived> table) {
998 int capacity = table->Capacity();
999 int new_capacity = capacity;
1000
1001 // Don't need to grow if we can simply clear out deleted entries instead.
1002 // TODO(gsathya): Compact in place, instead of allocating a new table.
1003 if (table->NumberOfDeletedElements() < (capacity >> 1)) {
1004 new_capacity = capacity << 1;
1005
1006 // The max capacity of our table is 254. We special case for 256 to
1007 // account for our growth strategy, otherwise we would only fill up
1008 // to 128 entries in our table.
1009 if (new_capacity == kGrowthHack) {
1010 new_capacity = kMaxCapacity;
1011 }
1012
1013 // We need to migrate to a bigger hash table.
1014 if (new_capacity > kMaxCapacity) {
1015 return MaybeHandle<Derived>();
1016 }
1017 }
1018
1019 return Derived::Rehash(isolate, table, new_capacity);
1020}

1440template <class Derived, class TableType>
1441void OrderedHashTableIterator<Derived, TableType>::Transition() {
1442 DisallowGarbageCollection no_gc;
1443 Tagged<TableType> table = Cast<TableType>(this->table());
1444 if (!table->IsObsolete()) return;
1445
1446 int index = Smi::ToInt(this->index());
1447 DCHECK_LE(0, index);
1448 while (table->IsObsolete()) {
1449 Tagged<TableType> next_table = table->NextTable();
1450
1451 if (index > 0) {
1452 int nod = table->NumberOfDeletedElements();
1453
1454 if (nod == TableType::kClearedTableSentinel) {
1455 index = 0;
1456 } else {
1457 int old_index = index;
1458 for (int i = 0; i < nod; ++i) {
1459 int removed_index = table->RemovedIndexAt(i);
1460 if (removed_index >= old_index) break;
1461 --index;
1462 }
1463 }
1464 }
1465
1466 table = next_table;
1467 }
1468
1469 set_table(table);
1470 set_index(Smi::FromInt(index));
1471}

1473template <class Derived, class TableType>
1474bool OrderedHashTableIterator<Derived, TableType>::HasMore() {
1475 DisallowGarbageCollection no_gc;
1476 ReadOnlyRoots ro_roots = GetReadOnlyRoots();
1477
1478 Transition();
1479
1480 Tagged<TableType> table = Cast<TableType>(this->table());
1481 int index = Smi::ToInt(this->index());
1482 int used_capacity = table->UsedCapacity();
1483
1484 while (index < used_capacity &&
1485 IsHashTableHole(table->KeyAt(InternalIndex(index)), ro_roots)) {
1486 index++;
1487 }
1488
1489 set_index(Smi::FromInt(index));
1490
1491 if (index < used_capacity) return true;
1492
1493 set_table(TableType::GetEmpty(ro_roots));
1494 return false;
