36template <template <typename> typename HandleType>
37 requires(std::is_convertible_v<HandleType<String>, DirectHandle<String>>)
38HandleType<String> String::SlowShare(Isolate* isolate,
39 HandleType<String> source) {
40 DCHECK(v8_flags.shared_string_table);
41 HandleType<String> flat =
42 Flatten(isolate, source, AllocationType::kSharedOld);
43
44 // Do not recursively call Share, so directly compute the sharing strategy for
45 // the flat string, which could already be a copy or an existing string from
46 // e.g. a shortcut ConsString.
47 MaybeDirectHandle<Map> new_map;
48 switch (isolate->factory()->ComputeSharingStrategyForString(flat, &new_map)) {
49 case StringTransitionStrategy::kCopy:
50 break;
51 case StringTransitionStrategy::kInPlace:
52 // A relaxed write is sufficient here, because at this point the string
53 // has not yet escaped the current thread.
54 DCHECK(HeapLayout::InAnySharedSpace(*flat));
55 flat->set_map_no_write_barrier(isolate, *new_map.ToHandleChecked());
56 return flat;
57 case StringTransitionStrategy::kAlreadyTransitioned:
58 return flat;
59 }
60
61 uint32_t length = flat->length();
62 if (flat->IsOneByteRepresentation()) {
63 HandleType<SeqOneByteString> copy =
64 isolate->factory()->NewRawSharedOneByteString(length).ToHandleChecked();
65 DisallowGarbageCollection no_gc;
66 WriteToFlat(*flat, copy->GetChars(no_gc), 0, length);
67 return copy;
68 }
69 HandleType<SeqTwoByteString> copy =
70 isolate->factory()->NewRawSharedTwoByteString(length).ToHandleChecked();
71 DisallowGarbageCollection no_gc;
72 WriteToFlat(*flat, copy->GetChars(no_gc), 0, length);
73 return copy;
74}

83template <class StringClass>
84void MigrateExternalStringResource(Isolate* isolate,
85 Tagged<ExternalString> from,
86 Tagged<StringClass> to) {
87 Address to_resource_address = to->resource_as_address();
88 if (to_resource_address == kNullAddress) {
89 Tagged<StringClass> cast_from = Cast<StringClass>(from);
90 // |to| is a just-created internalized copy of |from|. Migrate the resource.
91 to->SetResource(isolate, cast_from->resource());
92 // Zap |from|'s resource pointer to reflect the fact that |from| has
93 // relinquished ownership of its resource.
94 isolate->heap()->UpdateExternalString(
95 from, Cast<ExternalString>(from)->ExternalPayloadSize(), 0);
96 cast_from->SetResource(isolate, nullptr);
97 } else if (to_resource_address != from->resource_as_address()) {
98 // |to| already existed and has its own resource. Finalize |from|.
99 isolate->heap()->FinalizeExternalString(from);
100 }
101}

103void MigrateExternalString(Isolate* isolate, Tagged<String> string,
104 Tagged<String> internalized) {
105 if (IsExternalOneByteString(internalized)) {
106 MigrateExternalStringResource(isolate, Cast<ExternalString>(string),
107 Cast<ExternalOneByteString>(internalized));
108 } else if (IsExternalTwoByteString(internalized)) {
109 MigrateExternalStringResource(isolate, Cast<ExternalString>(string),
110 Cast<ExternalTwoByteString>(internalized));
111 } else {
112 // If the external string is duped into an existing non-external
113 // internalized string, free its resource (it's about to be rewritten
114 // into a ThinString below).
115 isolate->heap()->FinalizeExternalString(string);
116 }

131template <typename IsolateT>
132void String::MakeThin(IsolateT* isolate, Tagged<String> internalized) {
133 DisallowGarbageCollection no_gc;
134 DCHECK_NE(this, internalized);
135 DCHECK(IsInternalizedString(internalized));
136
137 Tagged<Map> initial_map = map(kAcquireLoad);
138 StringShape initial_shape(initial_map);
139
140 DCHECK(!initial_shape.IsThin());
141
142#ifdef DEBUG
143 // Check that shared strings can only transition to ThinStrings on the main
144 // thread when no other thread is active.
145 // The exception is during serialization, as no strings have escaped the
146 // thread yet.
147 if (initial_shape.IsShared() && !isolate->has_active_deserializer()) {
148 isolate->AsIsolate()->global_safepoint()->AssertActive();
149 }
150#endif
151
152 bool may_contain_recorded_slots = initial_shape.IsIndirect();
153 int old_size = SizeFromMap(initial_map);
154 ReadOnlyRoots roots(isolate);
155 Tagged<Map> target_map = internalized->IsOneByteRepresentation()
156 ? roots.thin_one_byte_string_map()
157 : roots.thin_two_byte_string_map();
158 if (initial_shape.IsExternal()) {
159 // Notify GC about the layout change before the transition to avoid
160 // concurrent marking from observing any in-between state (e.g.
161 // ExternalString map where the resource external pointer is overwritten
162 // with a tagged pointer).
163 // ExternalString -> ThinString transitions can only happen on the
164 // main-thread.
165 isolate->AsIsolate()->heap()->NotifyObjectLayoutChange(
166 Tagged(this), no_gc, InvalidateRecordedSlots::kYes,
167 InvalidateExternalPointerSlots::kYes, sizeof(ThinString));
168 MigrateExternalString(isolate->AsIsolate(), this, internalized);
169 }
170
171 // Update actual first and then do release store on the map word. This ensures
172 // that the concurrent marker will read the pointer when visiting a
173 // ThinString.
174 Tagged<ThinString> thin = UncheckedCast<ThinString>(Tagged(this));
175 thin->set_actual(internalized);
176
177 DCHECK_GE(old_size, sizeof(ThinString));
178 int size_delta = old_size - sizeof(ThinString);
179 if (size_delta != 0) {
180 if (!Heap::IsLargeObject(thin)) {
181 isolate->heap()->NotifyObjectSizeChange(
182 thin, old_size, sizeof(ThinString),
183 may_contain_recorded_slots ? ClearRecordedSlots::kYes
184 : ClearRecordedSlots::kNo);
185 } else {
186 // We don't need special handling for the combination IsLargeObject &&
187 // may_contain_recorded_slots, because indirect strings never get that
188 // large.
189 DCHECK(!may_contain_recorded_slots);
190 }
191 }
192
193 if (initial_shape.IsExternal()) {
194 set_map(isolate, target_map, kReleaseStore);
195 } else {
196 set_map_safe_transition(isolate, target_map, kReleaseStore);
197 }
198}

205template <typename T>
206bool String::MarkForExternalizationDuringGC(Isolate* isolate, T* resource) {
207 uint32_t raw_hash = raw_hash_field(kAcquireLoad);
208 if (IsExternalForwardingIndex(raw_hash)) return false;
209 if (IsInternalizedForwardingIndex(raw_hash)) {
210 const int forwarding_index = ForwardingIndexValueBits::decode(raw_hash);
211 if (!isolate->string_forwarding_table()->TryUpdateExternalResource(
212 forwarding_index, resource)) {
213 // The external resource was concurrently updated by another thread.
214 return false;
215 }
216 resource->Unaccount(reinterpret_cast<v8::Isolate*>(isolate));
217 raw_hash = Name::IsExternalForwardingIndexBit::update(raw_hash, true);
218 set_raw_hash_field(raw_hash, kReleaseStore);
219 return true;
220 }
221 // We need to store the hash in the forwarding table, as all non-external
222 // shared strings are in-place internalizable. In case the string gets
223 // internalized, we have to ensure that we can get the hash from the
224 // forwarding table to satisfy the invariant that all internalized strings
225 // have a computed hash value.
226 if (!IsHashFieldComputed(raw_hash)) {
227 raw_hash = EnsureRawHash();
228 }
229 DCHECK(IsHashFieldComputed(raw_hash));
230 resource->Unaccount(reinterpret_cast<v8::Isolate*>(isolate));
231 int forwarding_index =
232 isolate->string_forwarding_table()->AddExternalResourceAndHash(
233 this, resource, raw_hash);
234 set_raw_hash_field(String::CreateExternalForwardingIndex(forwarding_index),
235 kReleaseStore);
236
237 return true;
238}

242template <bool is_one_byte>
243Tagged<Map> ComputeExternalStringMap(Isolate* isolate, Tagged<String> string,
244 int size) {
245 ReadOnlyRoots roots(isolate);
246 StringShape shape(string, isolate);
247 const bool is_internalized = shape.IsInternalized();
248 const bool is_shared = shape.IsShared();
249 if constexpr (is_one_byte) {
250 if (size < static_cast<int>(sizeof(ExternalString))) {
251 if (is_internalized) {
252 return roots.uncached_external_internalized_one_byte_string_map();
253 } else {
254 return is_shared ? roots.shared_uncached_external_one_byte_string_map()
255 : roots.uncached_external_one_byte_string_map();
256 }
257 } else {
258 if (is_internalized) {
259 return roots.external_internalized_one_byte_string_map();
260 } else {
261 return is_shared ? roots.shared_external_one_byte_string_map()
262 : roots.external_one_byte_string_map();
263 }
264 }
265 } else {
266 if (size < static_cast<int>(sizeof(ExternalString))) {
267 if (is_internalized) {
268 return roots.uncached_external_internalized_two_byte_string_map();
269 } else {
270 return is_shared ? roots.shared_uncached_external_two_byte_string_map()
271 : roots.uncached_external_two_byte_string_map();
272 }
273 } else {
274 if (is_internalized) {
275 return roots.external_internalized_two_byte_string_map();
276 } else {
277 return is_shared ? roots.shared_external_two_byte_string_map()
278 : roots.external_two_byte_string_map();
279 }
280 }

286template <typename T>
287void String::MakeExternalDuringGC(Isolate* isolate, T* resource) {
288 isolate->heap()->safepoint()->AssertActive();
289 DCHECK_NE(isolate->heap()->gc_state(), Heap::NOT_IN_GC);
290
291 constexpr bool is_one_byte =
292 std::is_base_of_v<v8::String::ExternalOneByteStringResource, T>;
293 int size = this->Size(); // Byte size of the original string.
294 DCHECK_GE(size, sizeof(UncachedExternalString));
295
296 // Morph the string to an external string by replacing the map and
297 // reinitializing the fields. This won't work if the space the existing
298 // string occupies is too small for a regular external string. Instead, we
299 // resort to an uncached external string instead, omitting the field caching
300 // the address of the backing store. When we encounter uncached external
301 // strings in generated code, we need to bailout to runtime.
302 Tagged<Map> new_map =
303 ComputeExternalStringMap<is_one_byte>(isolate, this, size);
304
305 // Byte size of the external String object.
306 int new_size = this->SizeFromMap(new_map);
307
308 // Shared strings are never indirect.
309 DCHECK(!StringShape(this).IsIndirect());
310
311 if (!isolate->heap()->IsLargeObject(this)) {
312 isolate->heap()->NotifyObjectSizeChange(this, size, new_size,
313 ClearRecordedSlots::kNo);
314 }
315
316 // The external pointer slots must be initialized before the new map is
317 // installed. Otherwise, a GC marking thread may see the new map before the
318 // slots are initialized and attempt to mark the (invalid) external pointers
319 // table entries as alive.
320 static_cast<ExternalString*>(this)
321 ->InitExternalPointerFieldsDuringExternalization(new_map, isolate);
322
323 // We are storing the new map using release store after creating a filler in
324 // the NotifyObjectSizeChange call for the left-over space to avoid races with
325 // the sweeper thread.
326 this->set_map(isolate, new_map, kReleaseStore);
327
328 if constexpr (is_one_byte) {
329 Tagged<ExternalOneByteString> self = Cast<ExternalOneByteString>(this);
330 self->SetResource(isolate, resource);
331 } else {
332 Tagged<ExternalTwoByteString> self = Cast<ExternalTwoByteString>(this);
333 self->SetResource(isolate, resource);
334 }
335 isolate->heap()->RegisterExternalString(this);
336}

344bool String::MakeExternal(Isolate* isolate,
345 v8::String::ExternalStringResource* resource) {
346 // Disallow garbage collection to avoid possible GC vs string access deadlock.
347 DisallowGarbageCollection no_gc;
348
349 // Externalizing twice leaks the external resource, so it's
350 // prohibited by the API.
351 DCHECK(
352 this->SupportsExternalization(v8::String::Encoding::TWO_BYTE_ENCODING));
353 DCHECK(resource->IsCacheable());
354#ifdef ENABLE_SLOW_DCHECKS
355 if (v8_flags.enable_slow_asserts) {
356 // Assert that the resource and the string are equivalent.
357 DCHECK(static_cast<size_t>(this->length()) == resource->length());
358 base::ScopedVector<base::uc16> smart_chars(this->length());
359 String::WriteToFlat(this, smart_chars.begin(), 0, this->length());
360 DCHECK_EQ(0, memcmp(smart_chars.begin(), resource->data(),
361 resource->length() * sizeof(smart_chars[0])));
362 }
363#endif // DEBUG
364 int size = this->Size(); // Byte size of the original string.
365 // Abort if size does not allow in-place conversion.
366 if (size < static_cast<int>(sizeof(UncachedExternalString))) return false;
367 // Read-only strings cannot be made external, since that would mutate the
368 // string.
369 if (HeapLayout::InReadOnlySpace(this)) return false;
370 if (IsShared()) {
371 return MarkForExternalizationDuringGC(isolate, resource);
372 }
373 // For strings in the shared space we need the shared space isolate instead of
374 // the current isolate.
375 if (HeapLayout::InWritableSharedSpace(this)) {
376 resource->Unaccount(reinterpret_cast<v8::Isolate*>(isolate));
377 isolate = isolate->shared_space_isolate();
378 }
379 bool is_internalized = IsInternalizedString(this);
380 bool has_pointers = StringShape(this).IsIndirect();
381
382 base::SharedMutexGuardIf<base::kExclusive> shared_mutex_guard(
383 isolate->internalized_string_access(), is_internalized);
384 // Morph the string to an external string by replacing the map and
385 // reinitializing the fields. This won't work if the space the existing
386 // string occupies is too small for a regular external string. Instead, we
387 // resort to an uncached external string instead, omitting the field caching
388 // the address of the backing store. When we encounter uncached external
389 // strings in generated code, we need to bailout to runtime.
390 constexpr bool is_one_byte = false;
391 Tagged<Map> new_map =
392 ComputeExternalStringMap<is_one_byte>(isolate, this, size);
393
394 // Byte size of the external String object.
395 int new_size = this->SizeFromMap(new_map);
396
397 if (has_pointers) {
398 isolate->heap()->NotifyObjectLayoutChange(
399 this, no_gc, InvalidateRecordedSlots::kYes,
400 InvalidateExternalPointerSlots::kNo, new_size);
401 }
402
403 if (!isolate->heap()->IsLargeObject(this)) {
404 isolate->heap()->NotifyObjectSizeChange(
405 this, size, new_size,
406 has_pointers ? ClearRecordedSlots::kYes : ClearRecordedSlots::kNo);
407 } else {
408 // We don't need special handling for the combination IsLargeObject &&
409 // has_pointers, because indirect strings never get that large.
410 DCHECK(!has_pointers);
411 }
412
413 // The external pointer slots must be initialized before the new map is
414 // installed. Otherwise, a GC marking thread may see the new map before the
415 // slots are initialized and attempt to mark the (invalid) external pointers
416 // table entries as alive.
417 static_cast<ExternalString*>(this)
418 ->InitExternalPointerFieldsDuringExternalization(new_map, isolate);
419
420 // We are storing the new map using release store after creating a filler in
421 // the NotifyObjectSizeChange call for the left-over space to avoid races with
422 // the sweeper thread.
423 this->set_map(isolate, new_map, kReleaseStore);
424
425 Tagged<ExternalTwoByteString> self = Cast<ExternalTwoByteString>(this);
426 self->SetResource(isolate, resource);
427 isolate->heap()->RegisterExternalString(this);
428 // Force regeneration of the hash value.
429 if (is_internalized) self->EnsureHash();
430 return true;
431}

527bool String::SupportsExternalization(v8::String::Encoding encoding) {
528 if (IsThinString(this)) {
529 return i::Cast<i::ThinString>(this)->actual()->SupportsExternalization(
530 encoding);
531 }
532
533 // RO_SPACE strings cannot be externalized.
534 if (HeapLayout::InReadOnlySpace(this)) {
535 return false;
536 }
537
538#if V8_COMPRESS_POINTERS && !V8_ENABLE_SANDBOX
539 // In this configuration, small strings may not be in-place externalizable.
540 if (this->Size() < static_cast<int>(sizeof(UncachedExternalString))) {
541 return false;
542 }
543#else
544 DCHECK_LE(sizeof(UncachedExternalString), this->Size());
545#endif
546
547 StringShape shape(this);
548
549 // Already an external string.
550 if (shape.IsExternal()) {
551 return false;
552 }
553
554 // Only strings in old space can be externalized.
555 if (HeapLayout::InYoungGeneration(Tagged(this))) {
556 return false;
557 }
558
559 // Encoding changes are not supported.
560 static_assert(kStringEncodingMask == 1 << 3);
561 static_assert(v8::String::Encoding::ONE_BYTE_ENCODING == 1 << 3);
562 static_assert(v8::String::Encoding::TWO_BYTE_ENCODING == 0);
563 return shape.encoding_tag() == static_cast<uint32_t>(encoding);
564}

709std::unique_ptr<char[]> String::ToCString(uint32_t offset, uint32_t length,
710 size_t* length_return) {
711 DCHECK_LE(length, this->length());
712 DCHECK_LE(offset, this->length() - length);
713
714 StringCharacterStream stream(this, offset);
715
716 // First, compute the required size of the output buffer.
717 size_t utf8_bytes = 0;
718 uint32_t remaining_chars = length;
719 uint16_t last = unibrow::Utf16::kNoPreviousCharacter;
720 while (stream.HasMore() && remaining_chars-- != 0) {
721 uint16_t character = stream.GetNext();
722 utf8_bytes += unibrow::Utf8::Length(character, last);
723 last = character;
724 }
725 if (length_return) {
726 *length_return = utf8_bytes;
727 }
728
729 // Second, allocate the output buffer.
730 size_t capacity = utf8_bytes + 1;
731 char* result = NewArray<char>(capacity);
732
733 // Third, encode the string into the output buffer.
734 stream.Reset(this, offset);
735 size_t pos = 0;
736 remaining_chars = length;
737 last = unibrow::Utf16::kNoPreviousCharacter;
738 while (stream.HasMore() && remaining_chars-- != 0) {
739 uint16_t character = stream.GetNext();
740 if (character == 0) {
741 character = ' ';
742 }
743
744 // Ensure that there's sufficient space for this character and the null
745 // terminator. This should normally always be the case, unless there is
746 // in-sandbox memory corruption.
747 // Alternatively, we could also over-allocate the output buffer by three
748 // bytes (the maximum we can write OOB) or consider allocating it inside
749 // the sandbox, but it's not clear if that would be worth the effort as the
750 // performance overhead of this check appears to be negligible in practice.
751 SBXCHECK_LE(unibrow::Utf8::Length(character, last) + 1, capacity - pos);
752
753 pos += unibrow::Utf8::Encode(result + pos, character, last);
754
755 last = character;
756 }
757
758 DCHECK_LT(pos, capacity);
759 result[pos++] = 0;
760
761 return std::unique_ptr<char[]>(result);
762}

777// static
778template <typename SinkCharT>
779void String::WriteToFlat(Tagged<String> source, SinkCharT* sink, uint32_t start,
780 uint32_t length,
781 const SharedStringAccessGuardIfNeeded& access_guard) {
782 DisallowGarbageCollection no_gc;
783 if (length == 0) return;
784 while (true) {
785 DCHECK_GT(length, 0);
786 DCHECK_LE(length, source->length());
787 DCHECK_LT(start, source->length());
788 DCHECK_LE(start + length, source->length());
789 switch (StringShape(source).representation_and_encoding_tag()) {
790 case kOneByteStringTag | kExternalStringTag:
791 CopyChars(sink, Cast<ExternalOneByteString>(source)->GetChars() + start,
792 length);
793 return;
794 case kTwoByteStringTag | kExternalStringTag:
795 CopyChars(sink, Cast<ExternalTwoByteString>(source)->GetChars() + start,
796 length);
797 return;
798 case kOneByteStringTag | kSeqStringTag:
799 CopyChars(
800 sink,
801 Cast<SeqOneByteString>(source)->GetChars(no_gc, access_guard) +
802 start,
803 length);
804 return;
805 case kTwoByteStringTag | kSeqStringTag:
806 CopyChars(
807 sink,
808 Cast<SeqTwoByteString>(source)->GetChars(no_gc, access_guard) +
809 start,
810 length);
811 return;
812 case kOneByteStringTag | kConsStringTag:
813 case kTwoByteStringTag | kConsStringTag: {
814 Tagged<ConsString> cons_string = Cast<ConsString>(source);
815 Tagged<String> first = cons_string->first();
816 uint32_t boundary = first->length();
817 // Here we explicity use signed ints as the values can become negative.
818 // The sum of {first_length} and {second_length} is always {length},
819 // but the values can become negative, in which case no characters of
820 // the respective string are needed.
821 int32_t first_length = boundary - start;
822 int32_t second_length = length - first_length;
823 DCHECK_EQ(static_cast<uint32_t>(first_length + second_length), length);
824 if (second_length >= first_length) {
825 DCHECK_GT(second_length, 0);
826 // Right hand side is longer. Recurse over left.
827 if (first_length > 0) {
828 DCHECK_LT(first_length, length);
829 DCHECK_LT(second_length, length);
830
831 WriteToFlat(first, sink, start, first_length, access_guard);
832 if (start == 0 && cons_string->second() == first) {
833 DCHECK_LE(boundary * 2, length);
834 CopyChars(sink + boundary, sink, boundary);
835 return;
836 }
837 sink += first_length;
838 start = 0;
839 length -= first_length;
840 } else {
841 start -= boundary;
842 }
843 source = cons_string->second();
844 } else {
845 DCHECK_GT(first_length, 0);
846 // Left hand side is longer. Recurse over right.
847 if (second_length > 0) {
848 DCHECK_LT(first_length, length);
849 DCHECK_LT(second_length, length);
850
851 uint32_t second_start = first_length;
852 DCHECK_EQ(second_start + second_length, length);
853 Tagged<String> second = cons_string->second();
854 // When repeatedly appending to a string, we get a cons string that
855 // is unbalanced to the left, a list, essentially. We inline the
856 // common case of sequential one-byte right child.
857 if (second_length == 1) {
858 sink[second_start] =
859 static_cast<SinkCharT>(second->Get(0, access_guard));
860 } else if (IsSeqOneByteString(second)) {
861 CopyChars(
862 sink + second_start,
863 Cast<SeqOneByteString>(second)->GetChars(no_gc, access_guard),
864 second_length);
865 } else {
866 WriteToFlat(second, sink + second_start, 0, second_length,
867 access_guard);
868 }
869 length -= second_length;
870 }
871 source = first;
872 }
873 if (length == 0) return;
874 continue;
875 }
876 case kOneByteStringTag | kSlicedStringTag:
877 case kTwoByteStringTag | kSlicedStringTag: {
878 Tagged<SlicedString> slice = Cast<SlicedString>(source);
879 uint32_t offset = slice->offset();
880 source = slice->parent();
881 start += offset;
882 continue;
883 }
884 case kOneByteStringTag | kThinStringTag:
885 case kTwoByteStringTag | kThinStringTag:
886 source = Cast<ThinString>(source)->actual();
887 continue;
888 }
889 UNREACHABLE();
890 }
891 UNREACHABLE();
892}

954// A SmallVector-based stack with a cached top element. The cached top is vital
955// for arm64 performance. This would be more natural within a class, but sadly
956// arm64 performance regresses significantly if so, since that also causes the
957// cached top to be spilled onto the stack.
958using wtf_stack_t = base::SmallVector<Tagged<String>, 32>;
959using wtf_stack_top_t = Tagged<String>;
960
961V8_INLINE void wtf_push(wtf_stack_top_t& top, wtf_stack_t& stack,
962 Tagged<String> value) {
963 if (!top.is_null()) stack.push_back(top);
964 top = value;
965}
966
967V8_INLINE bool wtf_try_pop(wtf_stack_top_t& top, wtf_stack_t& stack,
968 Tagged<String>* value) {
969 if (V8_LIKELY(!top.is_null())) {
970 *value = top;
971 top = {};
972 return true;
973 }
974 if (V8_LIKELY(!stack.empty())) {
975 *value = stack.back();
976 stack.pop_back();
977 return true;
978 }
979 return false;
980}

982// Omits repeated flattening of one string (based on pointer identity) by
983// remembering its first flattened position, and simply copying that region
984// when encountering it again.
985template <typename SinkCharT>
986class WriteToFlat_RepeatOptimizer final {
987 public:
988 V8_INLINE void RecordFirstOccurrence(Tagged<String> s,
989 const SinkCharT* position) {
990 enabled_ = true;
991 auto it = first_occurrence_.find(s.ptr());
992 if (it == first_occurrence_.end()) {
993 first_occurrence_.insert({s.ptr(), position});
994 }
995 }
996
997 V8_INLINE bool TryApply(Tagged<String> s, SinkCharT** current_position) {
998 if (V8_UNLIKELY(enabled_)) {
999 auto it = first_occurrence_.find(s.ptr());
1000 if (it != first_occurrence_.end()) {
1001 const SinkCharT* previous_position = it->second;
1002 if (*current_position != previous_position) {
1003 uint32_t length = s->length();
1004 DCHECK_LE(*current_position, previous_position - length);
1005 previous_position -= length;
1006 (*current_position) -= length;
1007 CopyChars(*current_position, previous_position, length);
1008 return true;
1009 }
1010 }
1011 }
1012 return false;
1013 }
1014
1015 V8_INLINE bool enabled() const { return enabled_; }
1016
1017 private:
1018 // Only enable once we've seen a candidate, to reduce overhead.
1019 bool enabled_ = false;
1020 // Maps a Tagged<String>::ptr() to its first flattened occurrence.
1021 std::unordered_map<Address, const SinkCharT*> first_occurrence_;
1022};

1024template <WriteToFlatImplVariant kVariant, typename SinkCharT>
1025V8_INLINE void WriteToFlat2Impl(SinkCharT*& rdst, wtf_stack_top_t& top,
1026 wtf_stack_t& stack,
1027 WriteToFlat_RepeatOptimizer<SinkCharT>& ropt,
1028 const SharedStringAccessGuardIfNeeded& aguard,
1029 const DisallowGarbageCollection& no_gc) {
1030 Tagged<String> s;
1031 while (V8_LIKELY(wtf_try_pop(top, stack, &s))) {
1032 StringShape shape{s};
1033
1034 if constexpr (kVariant == kWTFGeneric) {
1035 if (V8_UNLIKELY(ropt.TryApply(s, &rdst))) continue;
1036 }
1037
1038 // Descend into the rightmost leaf and push left branches onto the stack.
1039 //
1040 // Alternatively, we could always flatten the shorter side first, where
1041 // substring length is used as a heuristic for substring tree depth, in
1042 // order to minimize stack size. That approach has different trade-offs,
1043 // for example: the stack would have to store both the string and the
1044 // current `rdst` value, and the write sequence may be less cache-friendly.
1045 while (shape.IsCons()) {
1046 auto cons = Cast<ConsString>(s);
1047 auto first = cons->first();
1048 wtf_push(top, stack, first);
1049 s = cons->second();
1050 if (V8_UNLIKELY(s == first)) {
1051 ropt.RecordFirstOccurrence(s, rdst);
1052 }
1053 shape = StringShape{s};
1054 }
1055
1056 if constexpr (kVariant == kWTFSeqOneByte) {
1057 if (!shape.IsSequentialOneByte() || V8_UNLIKELY(ropt.enabled())) {
1058 // Exit the specialized variant. Note the caller MUST follow up with
1059 // the kGeneric variant.
1060 wtf_push(top, stack, s);
1061 return;
1062 }
1063 uint8_t* chars = Cast<SeqOneByteString>(s)->GetChars(no_gc, aguard);
1064 uint32_t length = s->length();
1065 rdst -= length;
1066 CopyChars(rdst, chars, length);
1067 } else {
1068 static_assert(kVariant == kWTFGeneric);
1069 uint32_t length = s->length();
1070 rdst -= length;
1071 WriteNonConsToFlat2(s, shape, rdst, 0, length, aguard, no_gc);
1072 }
1073 }
1074}

1078// static
1079template <typename SinkCharT>
1080void String::WriteToFlat2(SinkCharT* dst, Tagged<ConsString> src,
1081 uint32_t src_index, uint32_t length,
1082 const SharedStringAccessGuardIfNeeded& aguard,
1083 const DisallowGarbageCollection& no_gc) {
1084 DCHECK_NE(length, 0);
1085 DCHECK(!src->IsFlat());
1086 DCHECK_LE(src_index + length, src->length());
1087
1088 // Limitations of the current implementation, which only supports flattening
1089 // the entire string.
1090 DCHECK_EQ(src_index, 0);
1091 DCHECK_EQ(length, src->length());
1092
1093 // The most common form of cons strings are degenerate unbalanced left-heavy
1094 // binary trees (i.e. where `second` is a flat string and `first` another
1095 // cons string). This form is created when building a string by appending
1096 // repeatedly: `str = "a" + "b" + ... + "z";
1097 //
1098 // To optimize for this, we flatten in reverse-DFS order, i.e. right-to-left.
1099 // This way, the stack never grows beyond size 1. Additionally, we elide the
1100 // stack push for the element that will immediately be processed next.
1101 // Finally, the iterative algorithm is split into two physically separate
1102 // loops - the first is optimized for cases when the cons tree contains only
1103 // sequential one-byte strings. The second handles all other cases
1104 // generically.
1105 //
1106 // Note this implementation is highly tuned. Please don't change anything
1107 // without watching benchmark scores.
1108
1109 SinkCharT* rdst = dst + length; // Reverse cursor.
1110 wtf_stack_t stack{src->first()};
1111 wtf_stack_top_t top = src->second();
1112 WriteToFlat_RepeatOptimizer<SinkCharT> ropt;
1113
1114 WriteToFlat2Impl<kWTFSeqOneByte>(rdst, top, stack, ropt, aguard, no_gc);
1115 WriteToFlat2Impl<kWTFGeneric>(rdst, top, stack, ropt, aguard, no_gc);
1116}

1233 uint32_t len = length();

1235 if (len == 0) return true;

1238 // and restart.

1243 } else {

1249 // a fast negative check can be performed.
1250 uint32_t this_hash;
1251 uint32_t other_hash;
1252 if (TryGetHash(&this_hash) && other->TryGetHash(&other_hash)) {
1253#ifdef ENABLE_SLOW_DCHECKS

1255 if (this_hash != other_hash) {
1256 bool found_difference = false;
1257 for (uint32_t i = 0; i < len; i++) {
1258 if (Get(i) != other->Get(i)) {
1259 found_difference = true;
1260 break;
1261 }
1262 }
1263 DCHECK(found_difference);
1264 }
1265 }
1266#endif
1267 if (this_hash != other_hash) return false;
1268 }

1270 // We know the strings are both non-empty. Compare the first chars
1271 // before we try to flatten the strings.
1272 if (this->Get(0, access_guard) != other->Get(0, access_guard)) return false;
1273
1274 if (IsSeqOneByteString(this) && IsSeqOneByteString(other)) {
1275 const uint8_t* str1 =
1276 Cast<SeqOneByteString>(this)->GetChars(no_gc, access_guard);
1277 const uint8_t* str2 =
1278 Cast<SeqOneByteString>(other)->GetChars(no_gc, access_guard);
1279 return CompareCharsEqual(str1, str2, len);
1280 }
1281
1282 StringComparator comparator;
1283 return comparator.Equals(this, other, access_guard);
1284}

1287bool String::SlowEquals(Isolate* isolate, DirectHandle<String> one,
1288 DirectHandle<String> two) {
1289 // Fast check: negative check with lengths.
1290 const uint32_t one_length = one->length();
1291 if (one_length != two->length()) return false;
1292 if (one_length == 0) return true;
1293
1294 // Fast check: if at least one ThinString is involved, dereference it/them
1295 // and restart.
1296 if (IsThinString(*one) || IsThinString(*two)) {
1297 if (IsThinString(*one)) {
1298 one = direct_handle(Cast<ThinString>(*one)->actual(), isolate);
1299 }
1300 if (IsThinString(*two)) {
1301 two = direct_handle(Cast<ThinString>(*two)->actual(), isolate);
1302 }
1303 return String::Equals(isolate, one, two);
1304 }
1305
1306 // Fast check: if hash code is computed for both strings
1307 // a fast negative check can be performed.
1308 uint32_t one_hash;
1309 uint32_t two_hash;
1310 if (one->TryGetHash(&one_hash) && two->TryGetHash(&two_hash)) {
1311#ifdef ENABLE_SLOW_DCHECKS
1312 if (v8_flags.enable_slow_asserts) {
1313 if (one_hash != two_hash) {
1314 bool found_difference = false;
1315 for (uint32_t i = 0; i < one_length; i++) {
1316 if (one->Get(i) != two->Get(i)) {
1317 found_difference = true;
1318 break;
1319 }
1320 }
1321 DCHECK(found_difference);
1322 }
1323 }
1324#endif
1325 if (one_hash != two_hash) return false;
1326 }
1327
1328 // We know the strings are both non-empty. Compare the first chars
1329 // before we try to flatten the strings.
1330 if (one->Get(0) != two->Get(0)) return false;
1331
1332 one = String::Flatten(isolate, one);
1333 two = String::Flatten(isolate, two);
1334
1335 DisallowGarbageCollection no_gc;
1336 String::FlatContent flat1 = one->GetFlatContent(no_gc);
1337 String::FlatContent flat2 = two->GetFlatContent(no_gc);
1338
1339 if (flat1.IsOneByte() && flat2.IsOneByte()) {
1340 return CompareCharsEqual(flat1.ToOneByteVector().begin(),
1341 flat2.ToOneByteVector().begin(), one_length);
1342 } else if (flat1.IsTwoByte() && flat2.IsTwoByte()) {
1343 return CompareCharsEqual(flat1.ToUC16Vector().begin(),
1344 flat2.ToUC16Vector().begin(), one_length);
1345 } else if (flat1.IsOneByte() && flat2.IsTwoByte()) {
1346 return CompareCharsEqual(flat1.ToOneByteVector().begin(),
1347 flat2.ToUC16Vector().begin(), one_length);
1348 } else if (flat1.IsTwoByte() && flat2.IsOneByte()) {
1349 return CompareCharsEqual(flat1.ToUC16Vector().begin(),
1350 flat2.ToOneByteVector().begin(), one_length);
1351 }
1352 UNREACHABLE();
1353}

1355// static
1356ComparisonResult String::Compare(Isolate* isolate, DirectHandle<String> x,
1357 DirectHandle<String> y) {
1358 // A few fast case tests before we flatten.
1359 if (x.is_identical_to(y)) {
1360 return ComparisonResult::kEqual;
1361 } else if (y->length() == 0) {
1362 return x->length() == 0 ? ComparisonResult::kEqual
1363 : ComparisonResult::kGreaterThan;
1364 } else if (x->length() == 0) {
1365 return ComparisonResult::kLessThan;
1366 }
1367
1368 int const d = x->Get(0) - y->Get(0);
1369 if (d < 0) {
1370 return ComparisonResult::kLessThan;
1371 } else if (d > 0) {
1372 return ComparisonResult::kGreaterThan;
1373 }
1374
1375 // Slow case.
1376 x = String::Flatten(isolate, x);
1377 y = String::Flatten(isolate, y);
1378
1379 DisallowGarbageCollection no_gc;
1380 ComparisonResult result = ComparisonResult::kEqual;
1381 uint32_t prefix_length = x->length();
1382 if (y->length() < prefix_length) {
1383 prefix_length = y->length();
1384 result = ComparisonResult::kGreaterThan;
1385 } else if (y->length() > prefix_length) {
1386 result = ComparisonResult::kLessThan;
1387 }
1388 int r;
1389 String::FlatContent x_content = x->GetFlatContent(no_gc);
1390 String::FlatContent y_content = y->GetFlatContent(no_gc);
1391 if (x_content.IsOneByte()) {
1392 base::Vector<const uint8_t> x_chars = x_content.ToOneByteVector();
1393 if (y_content.IsOneByte()) {
1394 base::Vector<const uint8_t> y_chars = y_content.ToOneByteVector();
1395 r = CompareChars(x_chars.begin(), y_chars.begin(), prefix_length);
1396 } else {
1397 base::Vector<const base::uc16> y_chars = y_content.ToUC16Vector();
1398 r = CompareChars(x_chars.begin(), y_chars.begin(), prefix_length);
1399 }
1400 } else {
1401 base::Vector<const base::uc16> x_chars = x_content.ToUC16Vector();
1402 if (y_content.IsOneByte()) {
1403 base::Vector<const uint8_t> y_chars = y_content.ToOneByteVector();
1404 r = CompareChars(x_chars.begin(), y_chars.begin(), prefix_length);
1405 } else {
1406 base::Vector<const base::uc16> y_chars = y_content.ToUC16Vector();
1407 r = CompareChars(x_chars.begin(), y_chars.begin(), prefix_length);
1408 }
1409 }
1410 if (r < 0) {
1411 result = ComparisonResult::kLessThan;
1412 } else if (r > 0) {
1413 result = ComparisonResult::kGreaterThan;
1414 }
1415 return result;
1416}

1498MaybeHandle<String> String::GetSubstitution(Isolate* isolate, Match* match,
1499 Handle<String> replacement,
1500 uint32_t start_index) {
1501 Factory* factory = isolate->factory();
1502
1503 const int replacement_length = replacement->length();
1504 const int captures_length = match->CaptureCount();
1505
1506 replacement = String::Flatten(isolate, replacement);
1507
1508 DirectHandle<String> dollar_string =
1509 factory->LookupSingleCharacterStringFromCode('$');
1510 int next_dollar_ix =
1511 String::IndexOf(isolate, replacement, dollar_string, start_index);
1512 if (next_dollar_ix < 0) {
1513 return replacement;
1514 }
1515
1516 IncrementalStringBuilder builder(isolate);
1517
1518 if (next_dollar_ix > 0) {
1519 builder.AppendString(factory->NewSubString(replacement, 0, next_dollar_ix));
1520 }
1521
1522 while (true) {
1523 const int peek_ix = next_dollar_ix + 1;
1524 if (peek_ix >= replacement_length) {
1525 builder.AppendCharacter('$');
1526 return indirect_handle(builder.Finish(), isolate);
1527 }
1528
1529 int continue_from_ix = -1;
1530 const uint16_t peek = replacement->Get(peek_ix);
1531 switch (peek) {
1532 case '$': // $$
1533 builder.AppendCharacter('$');
1534 continue_from_ix = peek_ix + 1;
1535 break;
1536 case '&': // $& - match
1537 builder.AppendString(match->GetMatch());
1538 continue_from_ix = peek_ix + 1;
1539 break;
1540 case '`': // $` - prefix
1541 builder.AppendString(match->GetPrefix());
1542 continue_from_ix = peek_ix + 1;
1543 break;
1544 case '\'': // $' - suffix
1545 builder.AppendString(match->GetSuffix());
1546 continue_from_ix = peek_ix + 1;
1547 break;
1548 case '0':
1549 case '1':
1550 case '2':
1551 case '3':
1552 case '4':
1553 case '5':
1554 case '6':
1555 case '7':
1556 case '8':
1557 case '9': {
1558 // Valid indices are $1 .. $9, $01 .. $09 and $10 .. $99
1559 int scaled_index = (peek - '0');
1560 int advance = 1;
1561
1562 if (peek_ix + 1 < replacement_length) {
1563 const uint16_t next_peek = replacement->Get(peek_ix + 1);
1564 if (next_peek >= '0' && next_peek <= '9') {
1565 const int new_scaled_index = scaled_index * 10 + (next_peek - '0');
1566 if (new_scaled_index < captures_length) {
1567 scaled_index = new_scaled_index;
1568 advance = 2;
1569 }
1570 }
1571 }
1572
1573 if (scaled_index == 0 || scaled_index >= captures_length) {
1574 builder.AppendCharacter('$');
1575 continue_from_ix = peek_ix;
1576 break;
1577 }
1578
1579 bool capture_exists;
1580 DirectHandle<String> capture;
1581 ASSIGN_RETURN_ON_EXCEPTION(
1582 isolate, capture, match->GetCapture(scaled_index, &capture_exists));
1583 if (capture_exists) builder.AppendString(capture);
1584 continue_from_ix = peek_ix + advance;
1585 break;
1586 }
1587 case '<': { // $<name> - named capture
1588 using CaptureState = String::Match::CaptureState;
1589
1590 if (!match->HasNamedCaptures()) {
1591 builder.AppendCharacter('$');
1592 continue_from_ix = peek_ix;
1593 break;
1594 }
1595
1596 DirectHandle<String> bracket_string =
1597 factory->LookupSingleCharacterStringFromCode('>');
1598 const int closing_bracket_ix =
1599 String::IndexOf(isolate, replacement, bracket_string, peek_ix + 1);
1600
1601 if (closing_bracket_ix == -1) {
1602 // No closing bracket was found, treat '$<' as a string literal.
1603 builder.AppendCharacter('$');
1604 continue_from_ix = peek_ix;
1605 break;
1606 }
1607
1608 DirectHandle<String> capture_name =
1609 factory->NewSubString(replacement, peek_ix + 1, closing_bracket_ix);
1610 DirectHandle<String> capture;
1611 CaptureState capture_state;
1612 ASSIGN_RETURN_ON_EXCEPTION(
1613 isolate, capture,
1614 match->GetNamedCapture(capture_name, &capture_state));
1615
1616 if (capture_state == CaptureState::MATCHED) {
1617 builder.AppendString(capture);
1618 }
1619
1620 continue_from_ix = closing_bracket_ix + 1;
1621 break;
1622 }
1623 default:
1624 builder.AppendCharacter('$');
1625 continue_from_ix = peek_ix;
1626 break;
1627 }
1628
1629 // Go the the next $ in the replacement.
1630 // TODO(jgruber): Single-char lookups could be much more efficient.
1631 DCHECK_NE(continue_from_ix, -1);
1632 next_dollar_ix =
1633 String::IndexOf(isolate, replacement, dollar_string, continue_from_ix);
1634
1635 // Return if there are no more $ characters in the replacement. If we
1636 // haven't reached the end, we need to append the suffix.
1637 if (next_dollar_ix < 0) {
1638 if (continue_from_ix < replacement_length) {
1639 builder.AppendString(factory->NewSubString(
1640 replacement, continue_from_ix, replacement_length));
1641 }
1642 return indirect_handle(builder.Finish(), isolate);
1643 }
1644
1645 // Append substring between the previous and the next $ character.
1646 if (next_dollar_ix > continue_from_ix) {
1647 builder.AppendString(
1648 factory->NewSubString(replacement, continue_from_ix, next_dollar_ix));
1649 }
1650 }
1651
1652 UNREACHABLE();
1653}

1657template <typename schar, typename pchar>
1658int StringMatchBackwards(base::Vector<const schar> subject,
1659 base::Vector<const pchar> pattern, int idx) {
1660 int pattern_length = pattern.length();
1661 DCHECK_GE(pattern_length, 1);
1662 DCHECK(idx + pattern_length <= subject.length());
1663
1664 if (sizeof(schar) == 1 && sizeof(pchar) > 1) {
1665 for (int i = 0; i < pattern_length; i++) {
1666 base::uc16 c = pattern[i];
1667 if (c > String::kMaxOneByteCharCode) {
1668 return -1;
1669 }
1670 }
1671 }
1672
1673 pchar pattern_first_char = pattern[0];
1674 for (int i = idx; i >= 0; i--) {
1675 if (subject[i] != pattern_first_char) continue;
1676 int j = 1;
1677 while (j < pattern_length) {
1678 if (pattern[j] != subject[i + j]) {
1679 break;
1680 }
1681 j++;
1682 }
1683 if (j == pattern_length) {
1684 return i;
1685 }
1686 }
1687 return -1;
1688}

1835uint32_t String::ComputeAndSetRawHash(
1836 const SharedStringAccessGuardIfNeeded& access_guard) {
1837 DisallowGarbageCollection no_gc;
1838 // Should only be called if hash code has not yet been computed.
1839 //
1840 // If in-place internalizable strings are shared, there may be calls to
1841 // ComputeAndSetRawHash in parallel. Since only flat strings are in-place
1842 // internalizable and their contents do not change, the result hash is the
1843 // same. The raw hash field is stored with relaxed ordering.
1844 DCHECK_IMPLIES(!v8_flags.shared_string_table, !HasHashCode());
1845
1846 // Store the hash code in the object.
1847 uint64_t seed = HashSeed(EarlyGetReadOnlyRoots());
1848 size_t start = 0;
1849 Tagged<String> string = this;
1850 StringShape shape(string);
1851 if (shape.IsSliced()) {
1852 Tagged<SlicedString> sliced = Cast<SlicedString>(string);
1853 start = sliced->offset();
1854 string = sliced->parent();
1855 shape = StringShape(string);
1856 }
1857 if (shape.IsCons() && string->IsFlat()) {
1858 string = Cast<ConsString>(string)->first();
1859 shape = StringShape(string);
1860 }
1861 if (shape.IsThin()) {
1862 string = Cast<ThinString>(string)->actual();
1863 shape = StringShape(string);
1864 if (length() == string->length()) {
1865 uint32_t raw_hash = string->RawHash();
1866 DCHECK(IsHashFieldComputed(raw_hash));
1867 set_raw_hash_field(raw_hash);
1868 return raw_hash;
1869 }
1870 }
1871 uint32_t raw_hash_field =
1872 shape.encoding_tag() == kOneByteStringTag
1873 ? HashString<uint8_t>(string, start, length(), seed, access_guard)
1874 : HashString<uint16_t>(string, start, length(), seed, access_guard);
1875 set_raw_hash_field_if_empty(raw_hash_field);
1876 // Check the hash code is there (or a forwarding index if the string was
1877 // internalized/externalized in parallel).
1878 DCHECK(HasHashCode() || HasForwardingIndex(kAcquireLoad));
1879 // Ensure that the hash value of 0 is never computed.
1880 DCHECK_NE(HashBits::decode(raw_hash_field), 0);
1881 return raw_hash_field;
1882}

1932 uint32_t old_length = string->length();

1934
1935 if (IsSeqOneByteString(*string)) {
1936 old_size = SeqOneByteString::SizeFor(old_length);
1937 new_size = SeqOneByteString::SizeFor(new_length);
1938 } else {
1939 DCHECK(IsSeqTwoByteString(*string));
1940 old_size = SeqTwoByteString::SizeFor(old_length);
1941 new_size = SeqTwoByteString::SizeFor(new_length);

1944#if DEBUG

2105 Tagged<ConsString> cons_string = root_;

2107 depth_ = 1;
2108 maximum_depth_ = 1;
2109 frames_[0] = cons_string;
2110 const uint32_t consumed = consumed_;
2111 uint32_t offset = 0;
2112 while (true) {
2113 // Loop until the string is found which contains the target offset.
2114 Tagged<String> string = cons_string->first();

2116 int32_t type;
2117 if (consumed < offset + length) {
2118 // Target offset is in the left branch.
2119 // Keep going if we're still in a ConString.
2120 type = string->map()->instance_type();

2122 cons_string = Cast<ConsString>(string);
2123 PushLeft(cons_string);

2127 AdjustMaximumDepth();

2152 Pop();
