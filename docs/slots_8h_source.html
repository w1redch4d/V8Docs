24template <typename Subclass, typename Data,
25 size_t SlotDataAlignment = sizeof(Data)>
26class SlotBase {
27 public:
28 using TData = Data;
29
30 static constexpr size_t kSlotDataSize = sizeof(Data);
31 static constexpr size_t kSlotDataAlignment = SlotDataAlignment;
32
33 Subclass& operator++() { // Prefix increment.
34 ptr_ += kSlotDataSize;
35 return *static_cast<Subclass*>(this);
36 }
37 Subclass operator++(int) { // Postfix increment.
38 Subclass result = *static_cast<Subclass*>(this);
39 ptr_ += kSlotDataSize;
40 return result;
41 }
42 Subclass& operator--() { // Prefix decrement.
43 ptr_ -= kSlotDataSize;
44 return *static_cast<Subclass*>(this);
45 }
46 Subclass operator--(int) { // Postfix decrement.
47 Subclass result = *static_cast<Subclass*>(this);
48 ptr_ -= kSlotDataSize;
49 return result;
50 }
51
52 bool operator<(const SlotBase& other) const { return ptr_ < other.ptr_; }
53 bool operator<=(const SlotBase& other) const { return ptr_ <= other.ptr_; }
54 bool operator>(const SlotBase& other) const { return ptr_ > other.ptr_; }
55 bool operator>=(const SlotBase& other) const { return ptr_ >= other.ptr_; }
56 bool operator==(const SlotBase& other) const { return ptr_ == other.ptr_; }
57 bool operator!=(const SlotBase& other) const { return ptr_ != other.ptr_; }
58 size_t operator-(const SlotBase& other) const {
59 DCHECK_GE(ptr_, other.ptr_);
60 return static_cast<size_t>((ptr_ - other.ptr_) / kSlotDataSize);
61 }
62 Subclass operator-(int i) const { return Subclass(ptr_ - i * kSlotDataSize); }
63 Subclass operator+(int i) const { return Subclass(ptr_ + i * kSlotDataSize); }
64 friend Subclass operator+(int i, const Subclass& slot) {
65 return Subclass(slot.ptr_ + i * kSlotDataSize);
66 }
67 Subclass& operator+=(int i) {
68 ptr_ += i * kSlotDataSize;
69 return *static_cast<Subclass*>(this);
70 }
71 Subclass operator-(int i) { return Subclass(ptr_ - i * kSlotDataSize); }
72 Subclass& operator-=(int i) {
73 ptr_ -= i * kSlotDataSize;
74 return *static_cast<Subclass*>(this);
75 }
76
77 void* ToVoidPtr() const { return reinterpret_cast<void*>(address()); }
78
79 Address address() const { return ptr_; }
80 // For symmetry with Handle.
81 TData* location() const { return reinterpret_cast<TData*>(ptr_); }
82
83 protected:
84 explicit SlotBase(Address ptr) : ptr_(ptr) {
85 DCHECK(IsAligned(ptr, kSlotDataAlignment));
86 }
87
88 private:
89 // This field usually describes an on-heap address (a slot within an object),
90 // so its type should not be a pointer to another C++ wrapper class.
91 // Type safety is provided by well-defined conversion operations.
92 Address ptr_;
93};

223// TODO(ishell, v8:8875): When pointer compression is enabled the [u]intptr_t
224// and double fields are only kTaggedSize aligned so in order to avoid undefined
225// behavior in C++ code we use this iterator adaptor when using STL algorithms
226// with unaligned pointers.
227// It will be removed once all v8:8875 is fixed and all the full pointer and
228// double values in compressed V8 heap are properly aligned.
229template <typename T>
230class UnalignedSlot : public SlotBase<UnalignedSlot<T>, T, 1> {
231 public:
232 // This class is a stand-in for "T&" that uses custom read/write operations
233 // for the actual memory accesses.
234 class Reference {
235 public:
236 explicit Reference(Address address) : address_(address) {}
237 Reference(const Reference&) V8_NOEXCEPT = default;
238
239 Reference& operator=(const Reference& other) V8_NOEXCEPT {
240 base::WriteUnalignedValue<T>(address_, other.value());
241 return *this;
242 }
243 Reference& operator=(T value) {
244 base::WriteUnalignedValue<T>(address_, value);
245 return *this;
246 }
247
248 // Values of type UnalignedSlot::reference must be implicitly convertible
249 // to UnalignedSlot::value_type.
250 operator T() const { return value(); }
251
252 void swap(Reference& other) {
253 T tmp = value();
254 base::WriteUnalignedValue<T>(address_, other.value());
255 base::WriteUnalignedValue<T>(other.address_, tmp);
256 }
257
258 bool operator<(const Reference& other) const {
259 return value() < other.value();
260 }
261
262 bool operator==(const Reference& other) const {
263 return value() == other.value();
264 }
265
266 private:
267 T value() const { return base::ReadUnalignedValue<T>(address_); }
268
269 Address address_;
270 };

272 // The rest of this class follows C++'s "RandomAccessIterator" requirements.
273 // Most of the heavy lifting is inherited from SlotBase.
274 using difference_type = int;
275 using value_type = T;
276 using reference = Reference;
277 using pointer = T*;
278 using iterator_category = std::random_access_iterator_tag;
279
280 UnalignedSlot() : SlotBase<UnalignedSlot<T>, T, 1>(kNullAddress) {}
281 explicit UnalignedSlot(Address address)
282 : SlotBase<UnalignedSlot<T>, T, 1>(address) {}
283 explicit UnalignedSlot(T* address)
284 : SlotBase<UnalignedSlot<T>, T, 1>(reinterpret_cast<Address>(address)) {}
285
286 Reference operator*() const {
287 return Reference(SlotBase<UnalignedSlot<T>, T, 1>::address());
288 }
289 Reference operator[](difference_type i) const {
290 return Reference(SlotBase<UnalignedSlot<T>, T, 1>::address() +
291 i * sizeof(T));
292 }
293
294 friend void swap(Reference lhs, Reference rhs) { lhs.swap(rhs); }
295
296 friend difference_type operator-(UnalignedSlot a, UnalignedSlot b) {
297 return static_cast<int>(a.address() - b.address()) / sizeof(T);
