108 // The first word contains the pointer to the (executable) entrypoint.
109 std::atomic<Address> entrypoint_;
110
111 // On 64 bit architectures the second word of the entry contains (1) the
112 // pointer to the code object associated with this entry, (2) the marking bit
113 // of the entry in the LSB of the object pointer (which must be unused as the
114 // address must be aligned), and (3) the 16-bit parameter count. The parameter
115 // count is stored in the lower 16 bits and therefore the pointer is shifted
116 // to the left. The final format therefore looks as follows:
117 //
118 // +----------------------+---------------+-------------------+
119 // | Bits 63 ... 17 | Bit 16 | Bits 15 ... 0 |
120 // | HeapObject pointer | Marking bit | Parameter count |
121 // +----------------------+---------------+-------------------+
122 //
123 // On 32 bit architectures only the mark bit is shared with the pointer.
124 //
125 // +----------------------+---------------+
126 // | Bits 32 ... 1 | Bit 0 |
127 // | HeapObject pointer | Marking bit |
128 // +----------------------+---------------+
129 //
130 // TODO(olivf): Find a better format that allows us to write atomically to the
131 // individual parts and unify with 32 bit. For instance we could try to store
132 // the code pointer in some compressd format, such that it fits into 32 bits.
133
134 static constexpr Address kMarkingBit = 1 << kObjectPointerShift;
135 std::atomic<Address> encoded_word_;
136
137#ifdef V8_TARGET_ARCH_32_BIT
138 // TODO(olivf): Investigate if we could shrink the entry size on 32bit
139 // platforms to 12 bytes.
140 std::atomic<uint16_t> parameter_count_;
141 // 16 bits of padding
142 std::atomic<uint32_t> next_free_entry_;
143#endif // V8_TARGET_ARCH_32_BIT
144};
