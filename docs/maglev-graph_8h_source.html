161 compiler::OptionalScopeInfoRef TryGetScopeInfoForContextLoad(
162 ValueNode* context, int offset, compiler::JSHeapBroker* broker) {
163 compiler::OptionalScopeInfoRef cur = TryGetScopeInfo(context, broker);
164 if (offset == Context::OffsetOfElementAt(Context::EXTENSION_INDEX)) {
165 return cur;
166 }
167 CHECK_EQ(offset, Context::OffsetOfElementAt(Context::PREVIOUS_INDEX));
168 if (cur.has_value()) {
169 cur = (*cur).OuterScopeInfo(broker);
170 while (!cur->HasContext() && cur->HasOuterScopeInfo()) {
171 cur = cur->OuterScopeInfo(broker);
172 }
173 if (cur->HasContext()) {
174 return cur;
175 }
176 }
177 return {};
178 }
179
180 // Resolve the scope info of a context value.
181 // An empty result means we don't statically know the context's scope.
182 compiler::OptionalScopeInfoRef TryGetScopeInfo(
183 ValueNode* context, compiler::JSHeapBroker* broker) {
184 auto it = scope_infos_.find(context);
185 if (it != scope_infos_.end()) {
186 return it->second;
187 }
188 compiler::OptionalScopeInfoRef res;
189 if (auto context_const = context->TryCast<Constant>()) {
190 res = context_const->object().AsContext().scope_info(broker);
191 DCHECK(res->HasContext());
192 } else if (auto load = context->TryCast<LoadTaggedFieldForContextSlot>()) {
193 compiler::OptionalScopeInfoRef cur = TryGetScopeInfoForContextLoad(
194 load->input(0).node(), load->offset(), broker);
195 if (cur.has_value()) res = cur;
196 } else if (auto load_script =
197 context->TryCast<LoadTaggedFieldForScriptContextSlot>()) {
198 compiler::OptionalScopeInfoRef cur = TryGetScopeInfoForContextLoad(
199 load_script->input(0).node(), load_script->offset(), broker);
200 if (cur.has_value()) res = cur;
201 } else if (context->Is<InitialValue>()) {
202 // We should only fail to keep track of initial contexts originating from
203 // the OSR prequel.
204 // TODO(olivf): Keep track of contexts when analyzing OSR Prequel.
205 DCHECK(is_osr());
206 } else {
207 // Any context created within a function must be registered in
208 // graph()->scope_infos(). Initial contexts must be registered before
209 // BuildBody. We don't track context in generators (yet) and around eval
210 // the bytecode compiler creates contexts by calling
211 // Runtime::kNewFunctionInfo directly.
212 DCHECK(context->Is<Phi>() || context->Is<GeneratorRestoreRegister>() ||
213 context->Is<RegisterInput>() || context->Is<CallRuntime>());
214 }
215 return scope_infos_[context] = res;
216 }
