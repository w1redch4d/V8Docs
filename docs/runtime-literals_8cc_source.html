34template <class ContextObject>
35class JSObjectWalkVisitor {
36 public:
37 explicit JSObjectWalkVisitor(ContextObject* site_context)
38 : site_context_(site_context) {}
39
40 V8_WARN_UNUSED_RESULT MaybeHandle<JSObject> StructureWalk(
41 Handle<JSObject> object);
42
43 protected:
44 V8_WARN_UNUSED_RESULT inline MaybeHandle<JSObject> VisitElementOrProperty(
45 DirectHandle<JSObject> object, Handle<JSObject> value) {
46 // Dont create allocation sites for nested object literals
47 if (!IsJSArray(*value)) {
48 return StructureWalk(value);
49 }
50
51 Handle<AllocationSite> current_site = site_context()->EnterNewScope();
52 MaybeHandle<JSObject> copy_of_value = StructureWalk(value);
53 site_context()->ExitScope(current_site, value);
54 return copy_of_value;
55 }
56
57 inline ContextObject* site_context() { return site_context_; }
58 inline Isolate* isolate() { return site_context()->isolate(); }
59
60 private:
61 ContextObject* site_context_;
62};

64template <class ContextObject>
65MaybeHandle<JSObject> JSObjectWalkVisitor<ContextObject>::StructureWalk(
66 Handle<JSObject> object) {
67 Isolate* isolate = this->isolate();
68 bool copying = ContextObject::kCopying;
69
70 {
71 StackLimitCheck check(isolate);
72
73 if (check.HasOverflowed()) {
74 isolate->StackOverflow();
75 return MaybeHandle<JSObject>();
76 }
77 }
78
79 if (object->map(isolate)->is_deprecated()) {
80 base::SharedMutexGuard<base::kExclusive> mutex_guard(
81 isolate->boilerplate_migration_access());
82 JSObject::MigrateInstance(isolate, object);
83 }
84
85 Handle<JSObject> copy;
86 if (copying) {
87 // JSFunction objects are not allowed to be in normal boilerplates at all.
88 DCHECK(!IsJSFunction(*object, isolate));
89 Handle<AllocationSite> site_to_pass;
90 if (site_context()->ShouldCreateMemento(object)) {
91 site_to_pass = site_context()->current();
92 }
93 copy = isolate->factory()->CopyJSObjectWithAllocationSite(object,
94 site_to_pass);
95 } else {
96 copy = object;
97 }
98
99 DCHECK(copying || copy.is_identical_to(object));
100
101 HandleScope scope(isolate);
102
103 // Deep copy own properties. Arrays only have 1 property "length".
104 if (!IsJSArray(*copy, isolate)) {
105 if (copy->HasFastProperties(isolate)) {
106 DirectHandle<DescriptorArray> descriptors(
107 copy->map(isolate)->instance_descriptors(isolate), isolate);
108 for (InternalIndex i : copy->map(isolate)->IterateOwnDescriptors()) {
109 PropertyDetails details = descriptors->GetDetails(i);
110 DCHECK_EQ(PropertyLocation::kField, details.location());
111 DCHECK_EQ(PropertyKind::kData, details.kind());
112 FieldIndex index = FieldIndex::ForPropertyIndex(
113 copy->map(isolate), details.field_index(),
114 details.representation());
115 Tagged<Object> raw = copy->RawFastPropertyAt(isolate, index);
116 if (IsJSObject(raw, isolate)) {
117 Handle<JSObject> value(Cast<JSObject>(raw), isolate);
118 ASSIGN_RETURN_ON_EXCEPTION(isolate, value,
119 VisitElementOrProperty(copy, value));
120 if (copying) copy->FastPropertyAtPut(index, *value);
121 } else if (copying && details.representation().IsDouble()) {
122 uint64_t double_value = Cast<HeapNumber>(raw)->value_as_bits();
123 auto value = isolate->factory()->NewHeapNumberFromBits(double_value);
124 copy->FastPropertyAtPut(index, *value);
125 }
126 }
127 } else {
128 if (V8_ENABLE_SWISS_NAME_DICTIONARY_BOOL) {
129 DirectHandle<SwissNameDictionary> dict(
130 copy->property_dictionary_swiss(isolate), isolate);
131 for (InternalIndex i : dict->IterateEntries()) {
132 Tagged<Object> raw = dict->ValueAt(i);
133 if (!IsJSObject(raw, isolate)) continue;
134 DCHECK(IsName(dict->KeyAt(i)));
135 Handle<JSObject> value(Cast<JSObject>(raw), isolate);
136 ASSIGN_RETURN_ON_EXCEPTION(isolate, value,
137 VisitElementOrProperty(copy, value));
138 if (copying) dict->ValueAtPut(i, *value);
139 }
140 } else {
141 DirectHandle<NameDictionary> dict(copy->property_dictionary(isolate),
142 isolate);
143 for (InternalIndex i : dict->IterateEntries()) {
144 Tagged<Object> raw = dict->ValueAt(isolate, i);
145 if (!IsJSObject(raw, isolate)) continue;
146 DCHECK(IsName(dict->KeyAt(isolate, i)));
147 Handle<JSObject> value(Cast<JSObject>(raw), isolate);
148 ASSIGN_RETURN_ON_EXCEPTION(isolate, value,
149 VisitElementOrProperty(copy, value));
150 if (copying) dict->ValueAtPut(i, *value);
151 }
152 }
153 }
154
155 // Assume non-arrays don't end up having elements.
156 if (copy->elements(isolate)->length() == 0) return copy;
157 }
158
159 // Deep copy own elements.
160 switch (copy->GetElementsKind(isolate)) {
161 case PACKED_ELEMENTS:
162 case PACKED_FROZEN_ELEMENTS:
163 case PACKED_SEALED_ELEMENTS:
164 case PACKED_NONEXTENSIBLE_ELEMENTS:
165 case HOLEY_FROZEN_ELEMENTS:
166 case HOLEY_SEALED_ELEMENTS:
167 case HOLEY_NONEXTENSIBLE_ELEMENTS:
168 case HOLEY_ELEMENTS:
169 case SHARED_ARRAY_ELEMENTS: {
170 DirectHandle<FixedArray> elements(
171 Cast<FixedArray>(copy->elements(isolate)), isolate);
172 if (elements->map() == ReadOnlyRoots(isolate).fixed_cow_array_map()) {
173#ifdef DEBUG
174 for (int i = 0; i < elements->length(); i++) {
175 DCHECK(!IsJSObject(elements->get(i)));
176 }
177#endif
178 } else {
179 for (int i = 0; i < elements->length(); i++) {
180 Tagged<Object> raw = elements->get(i);
181 if (!IsJSObject(raw, isolate)) continue;
182 Handle<JSObject> value(Cast<JSObject>(raw), isolate);
183 ASSIGN_RETURN_ON_EXCEPTION(isolate, value,
184 VisitElementOrProperty(copy, value));
185 if (copying) elements->set(i, *value);
186 }
187 }
188 break;
189 }
190 case DICTIONARY_ELEMENTS: {
191 DirectHandle<NumberDictionary> element_dictionary(
192 copy->element_dictionary(isolate), isolate);
193 for (InternalIndex i : element_dictionary->IterateEntries()) {
194 Tagged<Object> raw = element_dictionary->ValueAt(isolate, i);
195 if (!IsJSObject(raw, isolate)) continue;
196 Handle<JSObject> value(Cast<JSObject>(raw), isolate);
197 ASSIGN_RETURN_ON_EXCEPTION(isolate, value,
198 VisitElementOrProperty(copy, value));
199 if (copying) element_dictionary->ValueAtPut(i, *value);
200 }
201 break;
202 }
203 case FAST_SLOPPY_ARGUMENTS_ELEMENTS:
204 case SLOW_SLOPPY_ARGUMENTS_ELEMENTS:
205 UNIMPLEMENTED();
206 break;
207 case FAST_STRING_WRAPPER_ELEMENTS:
208 case SLOW_STRING_WRAPPER_ELEMENTS:
209 case WASM_ARRAY_ELEMENTS:
210 UNREACHABLE();
211
212#define TYPED_ARRAY_CASE(Type, type, TYPE, ctype) case TYPE##_ELEMENTS:
213
214 TYPED_ARRAYS(TYPED_ARRAY_CASE)
215 RAB_GSAB_TYPED_ARRAYS(TYPED_ARRAY_CASE)
216#undef TYPED_ARRAY_CASE
217 // Typed elements cannot be created using an object literal.
218 UNREACHABLE();
219
220 case PACKED_SMI_ELEMENTS:
221 case HOLEY_SMI_ELEMENTS:
222 case PACKED_DOUBLE_ELEMENTS:
223 case HOLEY_DOUBLE_ELEMENTS:
224 case NO_ELEMENTS:
225 // No contained objects, nothing to do.
226 break;
227 }
228
229 return copy;
230}

250// AllocationSiteCreationContext aids in the creation of AllocationSites to
251// accompany object literals.
252class AllocationSiteCreationContext : public AllocationSiteContext {
253 public:
254 explicit AllocationSiteCreationContext(Isolate* isolate)
255 : AllocationSiteContext(isolate) {}
256
257 Handle<AllocationSite> EnterNewScope() {
258 Handle<AllocationSite> scope_site;
259 if (top().is_null()) {
260 // We are creating the top level AllocationSite as opposed to a nested
261 // AllocationSite.
262 InitializeTraversal(isolate()->factory()->NewAllocationSite(true));
263 scope_site = Handle<AllocationSite>(*top(), isolate());
264 if (v8_flags.trace_creation_allocation_sites) {
265 PrintF("*** Creating top level %s AllocationSite %p\n", "Fat",
266 reinterpret_cast<void*>(scope_site->ptr()));
267 }
268 } else {
269 DCHECK(!current().is_null());
270 scope_site = isolate()->factory()->NewAllocationSite(false);
271 if (v8_flags.trace_creation_allocation_sites) {
272 PrintF(
273 "*** Creating nested %s AllocationSite (top, current, new) (%p, "
274 "%p, "
275 "%p)\n",
276 "Slim", reinterpret_cast<void*>(top()->ptr()),
277 reinterpret_cast<void*>(current()->ptr()),
278 reinterpret_cast<void*>(scope_site->ptr()));
279 }
280 current()->set_nested_site(*scope_site);
281 update_current_site(*scope_site);
282 }
283 DCHECK(!scope_site.is_null());
284 return scope_site;
285 }
286 void ExitScope(Handle<AllocationSite> scope_site,
287 DirectHandle<JSObject> object) {
288 if (object.is_null()) return;
289 scope_site->set_boilerplate(*object, kReleaseStore);
290 if (v8_flags.trace_creation_allocation_sites) {
291 bool top_level =
292 !scope_site.is_null() && top().is_identical_to(scope_site);
293 if (top_level) {
294 PrintF("*** Setting AllocationSite %p transition_info %p\n",
295 reinterpret_cast<void*>(scope_site->ptr()),
296 reinterpret_cast<void*>(object->ptr()));
297 } else {
298 PrintF("*** Setting AllocationSite (%p, %p) transition_info %p\n",
299 reinterpret_cast<void*>(top()->ptr()),
300 reinterpret_cast<void*>(scope_site->ptr()),
301 reinterpret_cast<void*>(object->ptr()));
302 }
303 }
304 }
305 static const bool kCopying = false;
306};

368Handle<JSObject> CreateObjectLiteral(
369 Isolate* isolate,
370 DirectHandle<ObjectBoilerplateDescription> object_boilerplate_description,
371 int flags, AllocationType allocation) {
372 DirectHandle<NativeContext> native_context = isolate->native_context();
373 bool use_fast_elements = (flags & ObjectLiteral::kFastElements) != 0;
374 bool has_null_prototype = (flags & ObjectLiteral::kHasNullPrototype) != 0;
375
376 // In case we have function literals, we want the object to be in
377 // slow properties mode for now. We don't go in the map cache because
378 // maps with constant functions can't be shared if the functions are
379 // not the same (which is the common case).
380 int number_of_properties =
381 object_boilerplate_description->backing_store_size();
382
383 // Ignoring number_of_properties for force dictionary map with
384 // __proto__:null.
385 DirectHandle<Map> map =
386 has_null_prototype
387 ? direct_handle(native_context->slow_object_with_null_prototype_map(),
388 isolate)
389 : isolate->factory()->ObjectLiteralMapFromCache(native_context,
390 number_of_properties);
391
392 Handle<JSObject> boilerplate =
393 isolate->factory()->NewFastOrSlowJSObjectFromMap(
394 map, number_of_properties, allocation);
395
396 // Normalize the elements of the boilerplate to save space if needed.
397 if (!use_fast_elements) JSObject::NormalizeElements(boilerplate);
398
399 // Add the constant properties to the boilerplate.
400 int length = object_boilerplate_description->boilerplate_properties_count();
401 // TODO(verwaest): Support tracking representations in the boilerplate.
402 for (int index = 0; index < length; index++) {
403 DirectHandle<Object> key(object_boilerplate_description->name(index),
404 isolate);
405 Handle<Object> value(object_boilerplate_description->value(index), isolate);
406
407 if (IsHeapObject(*value)) {
408 if (IsArrayBoilerplateDescription(Cast<HeapObject>(*value), isolate)) {
409 auto array_boilerplate = Cast<ArrayBoilerplateDescription>(value);
410 value = CreateArrayLiteral(isolate, array_boilerplate, allocation);
411
412 } else if (IsObjectBoilerplateDescription(Cast<HeapObject>(*value),
413 isolate)) {
414 auto object_boilerplate = Cast<ObjectBoilerplateDescription>(value);
415 value = CreateObjectLiteral(isolate, object_boilerplate,
416 object_boilerplate->flags(), allocation);
417 }
418 }
419
420 uint32_t element_index = 0;
421 if (Object::ToArrayIndex(*key, &element_index)) {
422 // Array index (uint32).
423 if (IsUninitialized(*value, isolate)) {
424 value = handle(Smi::zero(), isolate);
425 }
426 JSObject::SetOwnElementIgnoreAttributes(boilerplate, element_index, value,
427 NONE)
428 .Check();
429 } else {
430 DirectHandle<String> name = Cast<String>(key);
431 DCHECK(!name->AsArrayIndex(&element_index));
432 JSObject::SetOwnPropertyIgnoreAttributes(boilerplate, name, value, NONE)
433 .Check();
434 }
435 }
436
437 if (map->is_dictionary_map() && !has_null_prototype) {
438 // TODO(cbruni): avoid making the boilerplate fast again, the clone stub
439 // supports dict-mode objects directly.
440 JSObject::MigrateSlowToFast(
441 boilerplate, boilerplate->map()->UnusedPropertyFields(), "FastLiteral");
442 }
443 return boilerplate;
444}

446Handle<JSObject> CreateArrayLiteral(
447 Isolate* isolate,
448 DirectHandle<ArrayBoilerplateDescription> array_boilerplate_description,
449 AllocationType allocation) {
450 ElementsKind constant_elements_kind =
451 array_boilerplate_description->elements_kind();
452
453 Handle<FixedArrayBase> constant_elements_values(
454 array_boilerplate_description->constant_elements(isolate), isolate);
455
456 // Create the JSArray.
457 Handle<FixedArrayBase> copied_elements_values;
458 if (IsDoubleElementsKind(constant_elements_kind)) {
459 copied_elements_values = isolate->factory()->CopyFixedDoubleArray(
460 Cast<FixedDoubleArray>(constant_elements_values));
461 } else {
462 DCHECK(IsSmiOrObjectElementsKind(constant_elements_kind));
463 const bool is_cow = (constant_elements_values->map() ==
464 ReadOnlyRoots(isolate).fixed_cow_array_map());
465 if (is_cow) {
466 copied_elements_values = constant_elements_values;
467 if (DEBUG_BOOL) {
468 auto fixed_array_values = Cast<FixedArray>(copied_elements_values);
469 for (int i = 0; i < fixed_array_values->length(); i++) {
470 DCHECK(!IsFixedArray(fixed_array_values->get(i)));
471 }
472 }
473 } else {
474 Handle<FixedArray> fixed_array_values =
475 Cast<FixedArray>(constant_elements_values);
476 Handle<FixedArray> fixed_array_values_copy =
477 isolate->factory()->CopyFixedArray(fixed_array_values);
478 copied_elements_values = fixed_array_values_copy;
479 for (int i = 0; i < fixed_array_values->length(); i++) {
480 Tagged<Object> value = fixed_array_values_copy->get(i);
481 Tagged<HeapObject> value_heap_object;
482 if (value.GetHeapObject(isolate, &value_heap_object)) {
483 if (IsArrayBoilerplateDescription(value_heap_object, isolate)) {
484 HandleScope sub_scope(isolate);
485 DirectHandle<ArrayBoilerplateDescription> boilerplate(
486 Cast<ArrayBoilerplateDescription>(value_heap_object), isolate);
487 DirectHandle<JSObject> result =
488 CreateArrayLiteral(isolate, boilerplate, allocation);
489 fixed_array_values_copy->set(i, *result);
490
491 } else if (IsObjectBoilerplateDescription(value_heap_object,
492 isolate)) {
493 HandleScope sub_scope(isolate);
494 DirectHandle<ObjectBoilerplateDescription> boilerplate(
495 Cast<ObjectBoilerplateDescription>(value_heap_object), isolate);
496 DirectHandle<JSObject> result = CreateObjectLiteral(
497 isolate, boilerplate, boilerplate->flags(), allocation);
498 fixed_array_values_copy->set(i, *result);
499 }
500 }
501 }
502 }
503 }
504 return isolate->factory()->NewJSArrayWithElements(
505 copied_elements_values, constant_elements_kind,
506 copied_elements_values->length(), allocation);
507}

519template <typename LiteralHelper>
520MaybeHandle<JSObject> CreateLiteral(Isolate* isolate,
521 Handle<HeapObject> maybe_vector,
522 int literals_index,
523 Handle<HeapObject> description, int flags) {
524 if (!IsFeedbackVector(*maybe_vector)) {
525 DCHECK(IsUndefined(*maybe_vector));
526 return CreateLiteralWithoutAllocationSite<LiteralHelper>(
527 isolate, description, flags);
528 }
529 auto vector = Cast<FeedbackVector>(maybe_vector);
530 FeedbackSlot literals_slot(FeedbackVector::ToSlot(literals_index));
531 CHECK(literals_slot.ToInt() < vector->length());
532 Handle<Object> literal_site(Cast<Object>(vector->Get(literals_slot)),
533 isolate);
534 Handle<AllocationSite> site;
535 Handle<JSObject> boilerplate;
536
537 if (HasBoilerplate(literal_site)) {
538 site = Cast<AllocationSite>(literal_site);
539 boilerplate = Handle<JSObject>(site->boilerplate(), isolate);
540 } else {
541 // Eagerly create AllocationSites for literals that contain an Array.
542 bool needs_initial_allocation_site =
543 (flags & AggregateLiteral::kNeedsInitialAllocationSite) != 0;
544 if (!needs_initial_allocation_site &&
545 IsUninitializedLiteralSite(*literal_site)) {
546 PreInitializeLiteralSite(vector, literals_slot);
547 return CreateLiteralWithoutAllocationSite<LiteralHelper>(
548 isolate, description, flags);
549 } else {
550 boilerplate = LiteralHelper::Create(isolate, description, flags,
551 AllocationType::kOld);
552 }
553 // Install AllocationSite objects.
554 AllocationSiteCreationContext creation_context(isolate);
555 site = creation_context.EnterNewScope();
556 RETURN_ON_EXCEPTION(isolate, DeepWalk(boilerplate, &creation_context));
557 creation_context.ExitScope(site, boilerplate);
558
559 vector->SynchronizedSet(literals_slot, *site);
560 }
561
562 static_assert(static_cast<int>(ObjectLiteral::kDisableMementos) ==
563 static_cast<int>(ArrayLiteral::kDisableMementos));
564 bool enable_mementos = (flags & ObjectLiteral::kDisableMementos) == 0;
565
566 // Copy the existing boilerplate.
567 AllocationSiteUsageContext usage_context(isolate, site, enable_mementos);
568 usage_context.EnterNewScope();
569 MaybeHandle<JSObject> copy = DeepCopy(boilerplate, &usage_context);
570 usage_context.ExitScope(site, boilerplate);
571 return copy;
572}

602RUNTIME_FUNCTION(Runtime_CreateRegExpLiteral) {
603 HandleScope scope(isolate);
604 DCHECK_EQ(4, args.length());
605 Handle<HeapObject> maybe_vector = args.at<HeapObject>(0);
606 int index = args.tagged_index_value_at(1);
607 Handle<String> pattern = args.at<String>(2);
608 int flags = args.smi_value_at(3);
609
610 if (IsUndefined(*maybe_vector)) {
611 // We don't have a vector; don't create a boilerplate, simply construct a
612 // plain JSRegExp instance and return it.
613 RETURN_RESULT_OR_FAILURE(
614 isolate, JSRegExp::New(isolate, pattern, JSRegExp::Flags(flags)));
615 }
616
617 auto vector = Cast<FeedbackVector>(maybe_vector);
618 FeedbackSlot literal_slot(FeedbackVector::ToSlot(index));
619 DirectHandle<Object> literal_site(Cast<Object>(vector->Get(literal_slot)),
620 isolate);
621
622 // This function must not be called when a boilerplate already exists (if it
623 // exists, callers should instead copy the boilerplate into a new JSRegExp
624 // instance).
625 CHECK(!HasBoilerplate(literal_site));
626
627 DirectHandle<JSRegExp> regexp_instance;
628 ASSIGN_RETURN_FAILURE_ON_EXCEPTION(
629 isolate, regexp_instance,
630 JSRegExp::New(isolate, pattern, JSRegExp::Flags(flags)));
631
632 // JSRegExp literal sites are initialized in a two-step process:
633 // Uninitialized-Preinitialized, and Preinitialized-Initialized.
634 if (IsUninitializedLiteralSite(*literal_site)) {
635 PreInitializeLiteralSite(vector, literal_slot);
636 return *regexp_instance;
637 }
638
639 DirectHandle<RegExpData> data(regexp_instance->data(isolate), isolate);
640 DirectHandle<String> source(Cast<String>(regexp_instance->source()), isolate);
641 DirectHandle<RegExpBoilerplateDescription> boilerplate =
642 isolate->factory()->NewRegExpBoilerplateDescription(
643 data, source,
