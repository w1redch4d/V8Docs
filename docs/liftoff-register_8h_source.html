89// Description of LiftoffRegister code encoding.
90// This example uses the ARM architecture, which as of writing has:
91// - 9 GP registers, requiring 4 bits
92// - 13 FP registers, requiring 5 bits
93// - kNeedI64RegPair is true
94// - kNeedS128RegPair is true
95// - thus, kBitsPerRegPair is 2 + 2 * 4 = 10
96// - storage_t is uint16_t
97// The table below illustrates how each RegClass is encoded, with brackets
98// surrounding the bits which encode the register number.
99//
100// +----------------+------------------+
101// | RegClass | Example |
102// +----------------+------------------+
103// | kGpReg (1) | [00 0000 0000] |
104// | kFpReg (2) | [00 0000 1001] |
105// | kGpRegPair (3) | 01 [0000] [0001] |
106// | kFpRegPair (4) | 10 000[0 0010] |
107// +----------------+------------------+
108//
109// gp and fp registers are encoded in the same index space, which means that
110// code has to check for kGpRegPair and kFpRegPair before it can treat the code
111// as a register code.
112// (1) [0 .. kMaxGpRegCode] encodes gp registers
113// (2) [kMaxGpRegCode + 1 .. kMaxGpRegCode + kMaxFpRegCode] encodes fp
114// registers, so in this example, 1001 is really fp register 0.
115// (3) The second top bit is set for kGpRegPair, and the two gp registers are
116// stuffed side by side in code. Note that this is not the second top bit of
117// storage_t, since storage_t is larger than the number of meaningful bits we
118// need for the encoding.
119// (4) The top bit is set for kFpRegPair, and the fp register is stuffed into
120// the bottom part of the code. Unlike (2), this is the fp register code itself
121// (not sharing index space with gp), so in this example, it is fp register 2.
122
123// Maximum code of a gp cache register.
124static constexpr int kMaxGpRegCode = kLiftoffAssemblerGpCacheRegs.last().code();
125// Maximum code of an fp cache register.
126static constexpr int kMaxFpRegCode = kLiftoffAssemblerFpCacheRegs.last().code();
127static constexpr int kAfterMaxLiftoffGpRegCode = kMaxGpRegCode + 1;
128static constexpr int kAfterMaxLiftoffFpRegCode =
129 kAfterMaxLiftoffGpRegCode + kMaxFpRegCode + 1;
130static constexpr int kAfterMaxLiftoffRegCode = kAfterMaxLiftoffFpRegCode;
131static constexpr int kBitsPerLiftoffRegCode =
132 32 - base::bits::CountLeadingZeros<uint32_t>(kAfterMaxLiftoffRegCode - 1);
133static constexpr int kBitsPerGpRegCode =
134 32 - base::bits::CountLeadingZeros<uint32_t>(kMaxGpRegCode);
135static constexpr int kBitsPerFpRegCode =
136 32 - base::bits::CountLeadingZeros<uint32_t>(kMaxFpRegCode);
137// GpRegPair requires 1 extra bit, S128RegPair also needs an extra bit.
138static constexpr int kBitsPerRegPair =
139 (kNeedS128RegPair ? 2 : 1) + 2 * kBitsPerGpRegCode;
140
141static_assert(2 * kBitsPerGpRegCode >= kBitsPerFpRegCode,
142 "encoding for gp pair and fp pair collides");

144class LiftoffRegister {
145 static constexpr int needed_bits =
146 std::max(kNeedI64RegPair || kNeedS128RegPair ? kBitsPerRegPair : 0,
147 kBitsPerLiftoffRegCode);
148 using storage_t = std::conditional<
149 needed_bits <= 8, uint8_t,
150 std::conditional<needed_bits <= 16, uint16_t, uint32_t>::type>::type;
151
152 static_assert(8 * sizeof(storage_t) >= needed_bits,
153 "chosen type is big enough");
154 // Check for smallest required data type being chosen.
155 // Special case for uint8_t as there are no smaller types.
156 static_assert((8 * sizeof(storage_t) < 2 * needed_bits) ||
157 (sizeof(storage_t) == sizeof(uint8_t)),
158 "chosen type is small enough");
159
160 public:
161 constexpr explicit LiftoffRegister(Register reg)
162 : LiftoffRegister(reg.code()) {
163 DCHECK(kLiftoffAssemblerGpCacheRegs.has(reg));
164 DCHECK_EQ(reg, gp());
165 }
166 constexpr explicit LiftoffRegister(DoubleRegister reg)
167 : LiftoffRegister(kAfterMaxLiftoffGpRegCode + reg.code()) {
168 DCHECK(kLiftoffAssemblerFpCacheRegs.has(reg));
169 DCHECK_EQ(reg, fp());
170 }
171
172#if defined(V8_TARGET_ARCH_IA32)
173 // IA32 needs a fixed xmm0 register as a LiftoffRegister, however, xmm0 is not
174 // an allocatable double register (see register-ia32.h). This constructor
175 // allows bypassing the DCHECK that the LiftoffRegister has to be allocatable.
176 static LiftoffRegister from_uncached(DoubleRegister reg) {
177 DCHECK(!kLiftoffAssemblerFpCacheRegs.has(reg));
178 return LiftoffRegister(kAfterMaxLiftoffGpRegCode + reg.code());
179 }
180#endif

182 static LiftoffRegister from_liftoff_code(int code) {
183 LiftoffRegister reg{static_cast<storage_t>(code)};
184 // Check that the code is correct by round-tripping through the
185 // reg-class-specific constructor.
186 DCHECK(
187 (reg.is_gp() && code == LiftoffRegister{reg.gp()}.liftoff_code()) ||
188 (reg.is_fp() && code == LiftoffRegister{reg.fp()}.liftoff_code()) ||
189 (reg.is_gp_pair() &&
190 code == ForPair(reg.low_gp(), reg.high_gp()).liftoff_code()) ||
191 (reg.is_fp_pair() && code == ForFpPair(reg.low_fp()).liftoff_code()));
192 return reg;
193 }

195 static LiftoffRegister from_code(RegClass rc, int code) {
196 switch (rc) {
197 case kGpReg:
198 return LiftoffRegister(Register::from_code(code));
199 case kFpReg:
200 return LiftoffRegister(DoubleRegister::from_code(code));
201 default:
202 UNREACHABLE();
203 }
204 }

206 // Shifts the register code depending on the type before converting to a
207 // LiftoffRegister.
208 static LiftoffRegister from_external_code(RegClass rc, ValueKind kind,
209 int code) {
210 if (kFPAliasing == AliasingKind::kCombine && kind == kF32) {
211 // Liftoff assumes a one-to-one mapping between float registers and
212 // double registers, and so does not distinguish between f32 and f64
213 // registers. The f32 register code must therefore be halved in order
214 // to pass the f64 code to Liftoff.
215 DCHECK_EQ(0, code % 2);
216 return LiftoffRegister::from_code(rc, code >> 1);
217 }
218 if (kNeedS128RegPair && kind == kS128) {
219 // Similarly for double registers and SIMD registers, the SIMD code
220 // needs to be doubled to pass the f64 code to Liftoff.
221 return LiftoffRegister::ForFpPair(DoubleRegister::from_code(code << 1));
222 }
223 return LiftoffRegister::from_code(rc, code);
224 }
225
226 static LiftoffRegister ForPair(Register low, Register high) {
227 DCHECK(kNeedI64RegPair);
228 DCHECK_NE(low, high);
229 storage_t combined_code = low.code() | (high.code() << kBitsPerGpRegCode) |
230 (1 << (2 * kBitsPerGpRegCode));
231 return LiftoffRegister(combined_code);
232 }
233
234 static LiftoffRegister ForFpPair(DoubleRegister low) {
235 DCHECK(kNeedS128RegPair);
236 DCHECK_EQ(0, low.code() % 2);
237 storage_t combined_code = low.code() | 2 << (2 * kBitsPerGpRegCode);
238 return LiftoffRegister(combined_code);
239 }

241 constexpr bool is_pair() const {
242 return (kNeedI64RegPair || kNeedS128RegPair) &&
243 (code_ & (3 << (2 * kBitsPerGpRegCode)));
244 }
245
246 constexpr bool is_gp_pair() const {
247 return kNeedI64RegPair && (code_ & (1 << (2 * kBitsPerGpRegCode))) != 0;
248 }
249 constexpr bool is_fp_pair() const {
250 return kNeedS128RegPair && (code_ & (2 << (2 * kBitsPerGpRegCode))) != 0;
251 }
252 constexpr bool is_gp() const { return code_ < kAfterMaxLiftoffGpRegCode; }
253 constexpr bool is_fp() const {
254 return code_ >= kAfterMaxLiftoffGpRegCode &&
255 code_ < kAfterMaxLiftoffFpRegCode;
256 }

258 LiftoffRegister low() const {
259 // Common case for most archs where only gp pair supported.
260 if (!kNeedS128RegPair) return LiftoffRegister(low_gp());
261 return is_gp_pair() ? LiftoffRegister(low_gp()) : LiftoffRegister(low_fp());
262 }
263
264 LiftoffRegister high() const {
265 // Common case for most archs where only gp pair supported.
266 if (!kNeedS128RegPair) return LiftoffRegister(high_gp());
267 return is_gp_pair() ? LiftoffRegister(high_gp())
268 : LiftoffRegister(high_fp());
269 }
270
271 Register low_gp() const {
272 DCHECK(is_gp_pair());
273 static constexpr storage_t kCodeMask = (1 << kBitsPerGpRegCode) - 1;
274 return Register::from_code(code_ & kCodeMask);
275 }
276
277 Register high_gp() const {
278 DCHECK(is_gp_pair());
279 static constexpr storage_t kCodeMask = (1 << kBitsPerGpRegCode) - 1;
280 return Register::from_code((code_ >> kBitsPerGpRegCode) & kCodeMask);
281 }
282
283 DoubleRegister low_fp() const {
284 DCHECK(is_fp_pair());
285 static constexpr storage_t kCodeMask = (1 << kBitsPerFpRegCode) - 1;
286 return DoubleRegister::from_code(code_ & kCodeMask);
287 }
288
289 DoubleRegister high_fp() const {
290 DCHECK(is_fp_pair());
291 static constexpr storage_t kCodeMask = (1 << kBitsPerFpRegCode) - 1;
292 return DoubleRegister::from_code((code_ & kCodeMask) + 1);
293 }
294
295 constexpr Register gp() const {
296 DCHECK(is_gp());
297 return Register::from_code(code_);
298 }
299
300 constexpr DoubleRegister fp() const {
301 DCHECK(is_fp());
302 return DoubleRegister::from_code(code_ - kAfterMaxLiftoffGpRegCode);
303 }
304
305 constexpr int liftoff_code() const {
306 static_assert(sizeof(int) >= sizeof(storage_t));
307 return static_cast<int>(code_);
308 }
309
310 constexpr RegClass reg_class() const {
311 return is_fp_pair() ? kFpRegPair
312 : is_gp_pair() ? kGpRegPair : is_gp() ? kGpReg : kFpReg;
313 }

315 bool operator==(const LiftoffRegister other) const {
316 DCHECK_EQ(is_gp_pair(), other.is_gp_pair());
317 DCHECK_EQ(is_fp_pair(), other.is_fp_pair());
318 return code_ == other.code_;
319 }
320 bool operator!=(const LiftoffRegister other) const {
321 DCHECK_EQ(is_gp_pair(), other.is_gp_pair());
322 DCHECK_EQ(is_fp_pair(), other.is_fp_pair());
323 return code_ != other.code_;
324 }
325 bool overlaps(const LiftoffRegister other) const {
326 if (is_pair()) return low().overlaps(other) || high().overlaps(other);
327 if (other.is_pair()) return *this == other.low() || *this == other.high();
328 return *this == other;
329 }

350class LiftoffRegList {
351 public:
352 class Iterator;
353
354 static constexpr bool use_u16 = kAfterMaxLiftoffRegCode <= 16;
355 static constexpr bool use_u32 = !use_u16 && kAfterMaxLiftoffRegCode <= 32;
356 using storage_t = std::conditional<
357 use_u16, uint16_t,
358 std::conditional<use_u32, uint32_t, uint64_t>::type>::type;
359
360 static constexpr storage_t kGpMask =
361 storage_t{kLiftoffAssemblerGpCacheRegs.bits()};
362 static constexpr storage_t kFpMask =
363 storage_t{kLiftoffAssemblerFpCacheRegs.bits()}
364 << kAfterMaxLiftoffGpRegCode;
365 // Sets all even numbered fp registers.
366 static constexpr uint64_t kEvenFpSetMask = uint64_t{0x5555555555555555}
367 << kAfterMaxLiftoffGpRegCode;
368 static constexpr uint64_t kOddFpSetMask = uint64_t{0xAAAAAAAAAAAAAAAA}
369 << kAfterMaxLiftoffGpRegCode;
370
371 constexpr LiftoffRegList() = default;
372
373 // Allow to construct LiftoffRegList from a number of
374 // {Register|DoubleRegister|LiftoffRegister}.
375 template <typename... Regs>
376 constexpr explicit LiftoffRegList(Regs... regs)
377 requires(
378 std::conjunction_v<std::disjunction<
379 std::is_same<Register, Regs>, std::is_same<DoubleRegister, Regs>,
380 std::is_same<LiftoffRegister, Regs>>...>)
381 {
382 (..., set(regs));
383 }
384
385 constexpr Register set(Register reg) {
386 return set(LiftoffRegister(reg)).gp();
387 }
388 constexpr DoubleRegister set(DoubleRegister reg) {
389 return set(LiftoffRegister(reg)).fp();
390 }
391
392 constexpr LiftoffRegister set(LiftoffRegister reg) {
393 if (reg.is_pair()) {
394 regs_ |= storage_t{1} << reg.low().liftoff_code();
395 regs_ |= storage_t{1} << reg.high().liftoff_code();
396 } else {
397 regs_ |= storage_t{1} << reg.liftoff_code();
398 }
399 return reg;
400 }
401
402 constexpr LiftoffRegister clear(LiftoffRegister reg) {
403 if (reg.is_pair()) {
404 regs_ &= ~(storage_t{1} << reg.low().liftoff_code());
405 regs_ &= ~(storage_t{1} << reg.high().liftoff_code());
406 } else {
407 regs_ &= ~(storage_t{1} << reg.liftoff_code());
408 }
409 return reg;
410 }
411 constexpr Register clear(Register reg) {
412 return clear(LiftoffRegister{reg}).gp();
413 }
414 constexpr DoubleRegister clear(DoubleRegister reg) {
415 return clear(LiftoffRegister{reg}).fp();
416 }
417
418 bool has(LiftoffRegister reg) const {
419 if (reg.is_pair()) {
420 DCHECK_EQ(has(reg.low()), has(reg.high()));
421 reg = reg.low();
422 }
423 return (regs_ & (storage_t{1} << reg.liftoff_code())) != 0;
424 }
425 bool has(Register reg) const { return has(LiftoffRegister{reg}); }
426 bool has(DoubleRegister reg) const { return has(LiftoffRegister{reg}); }
427
428 constexpr bool is_empty() const { return regs_ == 0; }
429
430 constexpr unsigned GetNumRegsSet() const {
431 return base::bits::CountPopulation(regs_);
432 }

452 constexpr LiftoffRegList GetAdjacentFpRegsSet() const {
453 // And regs_ with a right shifted version of itself, so reg[i] is set only
454 // if reg[i+1] is set. We only care about the even fp registers.
455 storage_t available = (regs_ >> 1) & regs_ & kEvenFpSetMask;
456 return LiftoffRegList(available);
457 }
458
459 constexpr bool HasAdjacentFpRegsSet() const {
460 return !GetAdjacentFpRegsSet().is_empty();
461 }
462
463 // Returns a list where if any part of an adjacent pair of FP regs was set,
464 // both are set in the result. For example, [1, 4] is turned into [0, 1, 4, 5]
465 // because (0, 1) and (4, 5) are adjacent pairs.
466 constexpr LiftoffRegList SpreadSetBitsToAdjacentFpRegs() const {
467 storage_t odd_regs = regs_ & kOddFpSetMask;
468 storage_t even_regs = regs_ & kEvenFpSetMask;
469 return FromBits(regs_ | ((odd_regs >> 1) & kFpMask) |
470 ((even_regs << 1) & kFpMask));
471 }

473 constexpr bool operator==(const LiftoffRegList other) const {
474 return regs_ == other.regs_;
475 }
