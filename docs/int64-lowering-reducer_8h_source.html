49 V<Word> REDUCE(WordBinop)(V<Word> left, V<Word> right, WordBinopOp::Kind kind,
50 WordRepresentation rep) {
51 if (rep == WordRepresentation::Word64()) {
52 V<Word64> left_w64 = V<Word64>::Cast(left);
53 V<Word64> right_w64 = V<Word64>::Cast(right);
54 switch (kind) {
55 case WordBinopOp::Kind::kAdd:
56 return LowerPairBinOp(left_w64, right_w64,
57 Word32PairBinopOp::Kind::kAdd);
58 case WordBinopOp::Kind::kSub:
59 return LowerPairBinOp(left_w64, right_w64,
60 Word32PairBinopOp::Kind::kSub);
61 case WordBinopOp::Kind::kMul:
62 return LowerPairBinOp(left_w64, right_w64,
63 Word32PairBinopOp::Kind::kMul);
64 case WordBinopOp::Kind::kBitwiseAnd:
65 return LowerBitwiseAnd(left_w64, right_w64);
66 case WordBinopOp::Kind::kBitwiseOr:
67 return LowerBitwiseOr(left_w64, right_w64);
68 case WordBinopOp::Kind::kBitwiseXor:
69 return LowerBitwiseXor(left_w64, right_w64);
70 default:
71 FATAL("WordBinopOp kind %d not supported by int64 lowering",
72 static_cast<int>(kind));
73 }
74 }
75 return Next::ReduceWordBinop(left, right, kind, rep);
76 }

101 V<Word32> REDUCE(Comparison)(V<Any> left, V<Any> right,
102 ComparisonOp::Kind kind,
103 RegisterRepresentation rep) {
104 if (rep != WordRepresentation::Word64()) {
105 return Next::ReduceComparison(left, right, kind, rep);
106 }
107
108 auto [left_low, left_high] = Unpack(V<Word64>::Cast(left));
109 auto [right_low, right_high] = Unpack(V<Word64>::Cast(right));
110 V<Word32> high_comparison;
111 V<Word32> low_comparison;
112 switch (kind) {
113 case ComparisonOp::Kind::kEqual:
114 // TODO(wasm): Use explicit comparisons and && here?
115 return __ Word32Equal(
116 __ Word32BitwiseOr(__ Word32BitwiseXor(left_low, right_low),
117 __ Word32BitwiseXor(left_high, right_high)),
118 0);
119 case ComparisonOp::Kind::kSignedLessThan:
120 high_comparison = __ Int32LessThan(left_high, right_high);
121 low_comparison = __ Uint32LessThan(left_low, right_low);
122 break;
123 case ComparisonOp::Kind::kSignedLessThanOrEqual:
124 high_comparison = __ Int32LessThan(left_high, right_high);
125 low_comparison = __ Uint32LessThanOrEqual(left_low, right_low);
126 break;
127 case ComparisonOp::Kind::kUnsignedLessThan:
128 high_comparison = __ Uint32LessThan(left_high, right_high);
129 low_comparison = __ Uint32LessThan(left_low, right_low);
130 break;
131 case ComparisonOp::Kind::kUnsignedLessThanOrEqual:
132 high_comparison = __ Uint32LessThan(left_high, right_high);
133 low_comparison = __ Uint32LessThanOrEqual(left_low, right_low);
134 break;
135 }
136
137 return __ Word32BitwiseOr(
138 high_comparison,
139 __ Word32BitwiseAnd(__ Word32Equal(left_high, right_high),
140 low_comparison));
141 }

244 OpIndex REDUCE(Change)(OpIndex input, ChangeOp::Kind kind,
245 ChangeOp::Assumption assumption,
246 RegisterRepresentation from,
247 RegisterRepresentation to) {
248 auto word32 = RegisterRepresentation::Word32();
249 auto word64 = RegisterRepresentation::Word64();
250 auto float64 = RegisterRepresentation::Float64();
251 using Kind = ChangeOp::Kind;
252 if (from != word64 && to != word64) {
253 return Next::ReduceChange(input, kind, assumption, from, to);
254 }
255
256 if (from == word32 && to == word64) {
257 if (kind == Kind::kZeroExtend) {
258 return __ Tuple(V<Word32>::Cast(input), __ Word32Constant(0));
259 }
260 if (kind == Kind::kSignExtend) {
261 return LowerSignExtend(input);
262 }
263 }
264 if (from == float64 && to == word64) {
265 if (kind == Kind::kBitcast) {
266 return __ Tuple(__ Float64ExtractLowWord32(input),
267 __ Float64ExtractHighWord32(input));
268 }
269 }
270 if (from == word64 && to == float64) {
271 if (kind == Kind::kBitcast) {
272 auto input_w32p = V<Tuple<Word32, Word32>>::Cast(input);
273 return __ BitcastWord32PairToFloat64(
274 __ template Projection<1>(input_w32p),
275 __ template Projection<0>(input_w32p));
276 }
277 }
278 if (from == word64 && to == word32 && kind == Kind::kTruncate) {
279 auto input_w32p = V<Tuple<Word32, Word32>>::Cast(input);
280 return __ template Projection<0>(input_w32p);
281 }
282 std::stringstream str;
283 str << "ChangeOp " << kind << " from " << from << " to " << to
284 << "not supported by int64 lowering";
285 FATAL("%s", str.str().c_str());
286 }

288 std::pair<OptionalV<Word32>, int32_t> IncreaseOffset(OptionalV<Word32> index,
289 int32_t offset,
290 int32_t add_offset,
291 bool tagged_base) {
292 // Note that the offset will just wrap around. Still, we need to always
293 // use an offset that is not std::numeric_limits<int32_t>::min() on tagged
294 // loads.
295 // TODO(dmercadier): Replace LoadOp::OffsetIsValid by taking care of this
296 // special case in the LoadStoreSimplificationReducer instead.
297 int32_t new_offset =
298 static_cast<uint32_t>(offset) + static_cast<uint32_t>(add_offset);
299 OptionalV<Word32> new_index = index;
300 if (!LoadOp::OffsetIsValid(new_offset, tagged_base)) {
301 // We cannot encode the new offset so we use the old offset
302 // instead and use the Index to represent the extra offset.
303 new_offset = offset;
304 if (index.has_value()) {
305 new_index = __ Word32Add(new_index.value(), add_offset);
306 } else {
307 new_index = __ Word32Constant(sizeof(int32_t));
308 }
309 }
310 return {new_index, new_offset};
311 }

313 OpIndex REDUCE(Load)(OpIndex base, OptionalOpIndex index, LoadOp::Kind kind,
314 MemoryRepresentation loaded_rep,
315 RegisterRepresentation result_rep, int32_t offset,
316 uint8_t element_scale) {
317 if (kind.is_atomic) {
318 if (loaded_rep == MemoryRepresentation::Int64() ||
319 loaded_rep == MemoryRepresentation::Uint64()) {
320 // TODO(jkummerow): Support non-zero scales in AtomicWord32PairOp, and
321 // remove the corresponding bailout in MachineOptimizationReducer to
322 // allow generating them.
323 CHECK_EQ(element_scale, 0);
324 return __ AtomicWord32PairLoad(base, index, offset);
325 }
326 if (result_rep == RegisterRepresentation::Word64()) {
327 return __ Tuple(
328 __ Load(base, index, kind, loaded_rep,
329 RegisterRepresentation::Word32(), offset, element_scale),
330 __ Word32Constant(0));
331 }
332 }
333 if (loaded_rep == MemoryRepresentation::Int64() ||
334 loaded_rep == MemoryRepresentation::Uint64()) {
335 auto [high_index, high_offset] =
336 IncreaseOffset(index, offset, sizeof(int32_t), kind.tagged_base);
337 return __ Tuple(
338 Next::ReduceLoad(base, index, kind, MemoryRepresentation::Int32(),
339 RegisterRepresentation::Word32(), offset,
340 element_scale),
341 Next::ReduceLoad(
342 base, high_index, kind, MemoryRepresentation::Int32(),
343 RegisterRepresentation::Word32(), high_offset, element_scale));
344 }
345 return Next::ReduceLoad(base, index, kind, loaded_rep, result_rep, offset,
346 element_scale);

518 V<turboshaft::FrameState> REDUCE(FrameState)(
519 base::Vector<const OpIndex> inputs, bool inlined,
520 const FrameStateData* data) {
521 bool has_int64_input = false;
522
523 for (MachineType type : data->machine_types) {
524 if (RegisterRepresentation::FromMachineType(type) ==
525 RegisterRepresentation::Word64()) {
526 has_int64_input = true;
527 break;
528 }
529 }
530 if (!has_int64_input) {
531 return Next::ReduceFrameState(inputs, inlined, data);
532 }
533 FrameStateData::Builder builder;
534 if (inlined) {
535 builder.AddParentFrameState(V<turboshaft::FrameState>(inputs[0]));
536 }
537 const FrameStateFunctionInfo* function_info =
538 data->frame_state_info.function_info();
539 uint16_t lowered_parameter_count = function_info->parameter_count();
540 int lowered_local_count = function_info->local_count();
541
542 for (size_t i = inlined; i < inputs.size(); ++i) {
543 // In case of inlining the parent FrameState is an additional input,
544 // however, it doesn't have an entry in the machine_types vector, so that
545 // index has to be adapted.
546 size_t machine_type_index = i - inlined;
547 if (RegisterRepresentation::FromMachineType(
548 data->machine_types[machine_type_index]) ==
549 RegisterRepresentation::Word64()) {
550 auto [low, high] = Unpack(V<Word64>::Cast(inputs[i]));
551 builder.AddInput(MachineType::Int32(), low);
552 builder.AddInput(MachineType::Int32(), high);
553 // Note that the first input (after the optional parent FrameState) is
554 // the JSClosure, so the first parameter is at index 1 (+1 in case of
555 // nested inlining).
556 if (i <= inlined + function_info->parameter_count()) {
557 ++lowered_parameter_count;
558 } else {
559 ++lowered_local_count;
560 }
561 } else {
562 // Just copy over the existing input.
563 builder.AddInput(data->machine_types[machine_type_index], inputs[i]);
564 }
565 }
566 Zone* zone = Asm().data()->compilation_zone();
567 auto* function_info_lowered = zone->New<compiler::FrameStateFunctionInfo>(
568 compiler::FrameStateType::kLiftoffFunction, lowered_parameter_count,
569 function_info->max_arguments(), lowered_local_count,
570 function_info->shared_info(), kNullMaybeHandle,
571 function_info->wasm_liftoff_frame_size(),
572 function_info->wasm_function_index());

693 OpIndex LowerRotateRight(V<Word64> left, V<Word32> right) {
694 // This reducer assumes that all rotates are mapped to rotate right.
695 DCHECK(!SupportedOperations::word64_rol());
696 auto [left_low, left_high] = Unpack(left);
697 V<Word32> shift = right;
698 uint32_t constant_shift = 0;
699
700 if (matcher_.MatchIntegralWord32Constant(shift, &constant_shift)) {
701 // Precondition: 0 <= shift < 64.
702 uint32_t shift_value = constant_shift & 0x3F;
703 if (shift_value == 0) {
704 // No-op, return original tuple.
705 return left;
706 }
707 if (shift_value == 32) {
708 // Swap low and high of left.
709 return __ Tuple(left_high, left_low);
710 }
711
712 V<Word32> low_input = left_high;
713 V<Word32> high_input = left_low;
714 if (shift_value < 32) {
715 low_input = left_low;
716 high_input = left_high;
717 }
718
719 uint32_t masked_shift_value = shift_value & 0x1F;
720 V<Word32> masked_shift = __ Word32Constant(masked_shift_value);
721 V<Word32> inv_shift = __ Word32Constant(32 - masked_shift_value);
722
723 V<Word32> low_node = __ Word32BitwiseOr(
724 __ Word32ShiftRightLogical(low_input, masked_shift),
725 __ Word32ShiftLeft(high_input, inv_shift));
726 V<Word32> high_node = __ Word32BitwiseOr(
727 __ Word32ShiftRightLogical(high_input, masked_shift),
728 __ Word32ShiftLeft(low_input, inv_shift));
729 return __ Tuple(low_node, high_node);
730 }
731
732 V<Word32> safe_shift = shift;
733 if (!SupportedOperations::word32_shift_is_safe()) {
734 // safe_shift = shift % 32
735 safe_shift = __ Word32BitwiseAnd(shift, 0x1F);
736 }
737 V<Word32> all_bits_set = __ Word32Constant(-1);
738 V<Word32> inv_mask = __ Word32BitwiseXor(
739 __ Word32ShiftRightLogical(all_bits_set, safe_shift), all_bits_set);
740 V<Word32> bit_mask = __ Word32BitwiseXor(inv_mask, all_bits_set);
741
742 V<Word32> less_than_32 = __ Int32LessThan(shift, 32);
743 // The low word and the high word can be swapped either at the input or
744 // at the output. We swap the inputs so that shift does not have to be
745 // kept for so long in a register.
746 ScopedVar<Word32> var_low(this, left_high);
747 ScopedVar<Word32> var_high(this, left_low);
748 IF (less_than_32) {
749 var_low = left_low;
750 var_high = left_high;
751 }
752
753 V<Word32> rotate_low = __ Word32RotateRight(var_low, safe_shift);
754 V<Word32> rotate_high = __ Word32RotateRight(var_high, safe_shift);
755
756 V<Word32> low_node =
757 __ Word32BitwiseOr(__ Word32BitwiseAnd(rotate_low, bit_mask),
758 __ Word32BitwiseAnd(rotate_high, inv_mask));
759 V<Word32> high_node =
760 __ Word32BitwiseOr(__ Word32BitwiseAnd(rotate_high, bit_mask),
761 __ Word32BitwiseAnd(rotate_low, inv_mask));
762 return __ Tuple(low_node, high_node);
763 }

765 V<Any> LowerCall(V<CallTarget> callee, OptionalV<FrameState> frame_state,
766 base::Vector<const OpIndex> arguments,
767 const TSCallDescriptor* descriptor, OpEffects effects,
768 bool is_tail_call) {
769 // Iterate over the call descriptor to skip lowering if the signature does
770 // not contain an i64.
771 const CallDescriptor* call_descriptor = descriptor->descriptor;
772 size_t param_count = call_descriptor->ParameterCount();
773 size_t i64_params = 0;
774 for (size_t i = 0; i < param_count; ++i) {
775 i64_params += call_descriptor->GetParameterType(i).representation() ==
776 MachineRepresentation::kWord64;
777 }
778 size_t return_count = call_descriptor->ReturnCount();
779 size_t i64_returns = 0;
780 for (size_t i = 0; i < return_count; ++i) {
781 i64_returns += call_descriptor->GetReturnType(i).representation() ==
782 MachineRepresentation::kWord64;
783 }
784 if (i64_params + i64_returns == 0) {
785 // No lowering required.
786 return is_tail_call ? Next::ReduceTailCall(callee, arguments, descriptor)
787 : Next::ReduceCall(callee, frame_state, arguments,
788 descriptor, effects);
789 }
790
791 // Transform the BigIntToI64 call descriptor into BigIntToI32Pair (this is
792 // the only use case currently, it may be extended in the future).
793 // The correct target is already set during graph building.
794 CallDescriptor* maybe_special_replacement =
795 wasm::GetWasmEngine()->call_descriptors()->GetLoweredCallDescriptor(
796 call_descriptor);
797 if (maybe_special_replacement) call_descriptor = maybe_special_replacement;
798 // Create descriptor with 2 i32s for every i64.
799 const CallDescriptor* lowered_descriptor =
800 GetI32WasmCallDescriptor(__ graph_zone(), call_descriptor);
801
802 // Map the arguments by unpacking i64 arguments (which have already been
803 // lowered to Tuple(i32, i32).)
804 base::SmallVector<OpIndex, 16> lowered_args;
805 lowered_args.reserve(param_count + i64_params);
806
807 DCHECK_EQ(param_count, arguments.size());
808 for (size_t i = 0; i < param_count; ++i) {
809 if (call_descriptor->GetParameterType(i).representation() ==
810 MachineRepresentation::kWord64) {
811 auto [low, high] = Unpack(arguments[i]);
812 lowered_args.push_back(low);
813 lowered_args.push_back(high);
814 } else {
815 lowered_args.push_back(arguments[i]);
816 }
817 }
818
819 auto lowered_ts_descriptor =
820 TSCallDescriptor::Create(lowered_descriptor, descriptor->can_throw,
821 LazyDeoptOnThrow::kNo, __ graph_zone());
822 OpIndex call =
823 is_tail_call
824 ? Next::ReduceTailCall(callee, base::VectorOf(lowered_args),
825 lowered_ts_descriptor)
826 : Next::ReduceCall(callee, frame_state,
827 base::VectorOf(lowered_args),
828 lowered_ts_descriptor, effects);
829 if (is_tail_call) {
830 // Tail calls don't return anything to the calling function.
831 return call;
832 }
833 if (i64_returns == 0 || return_count == 0) {
834 return call;
835 } else if (return_count == 1) {
836 // There isn't any projection in the input graph for calls returning
837 // exactly one value. Return a tuple of projections for the int64.
838 DCHECK_EQ(i64_returns, 1);
839 return call;
840 }
841
842 // Wrap the call node with a tuple of projections of the lowered call.
843 // Example for a call returning [int64, int32]:
844 // In: Call(...) -> [int64, int32]
845 // Out: call = Call() -> [int32, int32, int32]
846 // Tuple(
847 // Tuple(Projection(call, 0), Projection(call, 1)),
848 // Projection(call, 2))
849 //
850 // This way projections on the original call node will be automatically
851 // "rewired" to the correct projection of the lowered call.
852 auto word32 = RegisterRepresentation::Word32();
853 base::SmallVector<V<Any>, 16> tuple_inputs;
854 tuple_inputs.reserve(return_count);
855 size_t projection_index = 0; // index of the lowered call results.
856
857 for (size_t i = 0; i < return_count; ++i) {
858 MachineRepresentation machine_rep =
859 call_descriptor->GetReturnType(i).representation();
860 if (machine_rep == MachineRepresentation::kWord64) {
861 tuple_inputs.push_back(
862 __ Tuple(__ Projection(call, projection_index, word32),
863 __ Projection(call, projection_index + 1, word32)));
864 projection_index += 2;
865 } else {
866 tuple_inputs.push_back(__ Projection(
867 call, projection_index++,
868 RegisterRepresentation::FromMachineRepresentation(machine_rep)));
869 }
870 }
871 DCHECK_EQ(projection_index, return_count + i64_returns);
872 return __ Tuple(base::VectorOf(tuple_inputs));
873 }
