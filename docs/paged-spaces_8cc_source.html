54// We have hit the end of the page and should advance to the next block of
55// objects. This happens at the end of the page.
56bool PagedSpaceObjectIterator::AdvanceToNextPage() {
57 if (current_page_ == page_range_.end()) return false;
58 const PageMetadata* cur_page = *(current_page_++);
59 HeapObjectRange heap_objects(cur_page);
60 cur_ = heap_objects.begin();
61 end_ = heap_objects.end();
62 return true;
63}

79PageMetadata* PagedSpaceBase::InitializePage(
80 MutablePageMetadata* mutable_page_metadata) {
81 MemoryChunk* chunk = mutable_page_metadata->Chunk();
82 PageMetadata* page = PageMetadata::cast(mutable_page_metadata);
83 DCHECK_EQ(
84 MemoryChunkLayout::AllocatableMemoryInMemoryChunk(page->owner_identity()),
85 page->area_size());
86 // Make sure that categories are initialized before freeing the area.
87 page->ResetAllocationStatistics();
88 page->AllocateFreeListCategories();

90 page->list_node().Initialize();
91 chunk->InitializationMemoryFence();

95void PagedSpaceBase::TearDown() {
96 while (!memory_chunk_list_.Empty()) {
97 MutablePageMetadata* chunk = memory_chunk_list_.front();
98 memory_chunk_list_.Remove(chunk);
99 heap()->memory_allocator()->Free(MemoryAllocator::FreeMode::kImmediately,
100 chunk);
101 }
102 accounting_stats_.Clear();

105void PagedSpaceBase::MergeCompactionSpace(CompactionSpace* other) {
106 base::SpinningMutexGuard guard(mutex());
107
108 DCHECK_NE(NEW_SPACE, identity());
109 DCHECK_NE(NEW_SPACE, other->identity());
110
111 // Move over pages.
112 for (auto it = other->begin(); it != other->end();) {
113 PageMetadata* p = *(it++);
114
115 // Ensure that pages are initialized before objects on it are discovered by
116 // concurrent markers.
117 p->Chunk()->InitializationMemoryFence();
118
119 // Relinking requires the category to be unlinked.
120 other->RemovePage(p);
121 AddPage(p);
122 DCHECK_IMPLIES(
123 !p->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE),
124 p->AvailableInFreeList() == p->AvailableInFreeListFromAllocatedBytes());
125
126 // TODO(leszeks): Here we should allocation step, but:
127 // 1. Allocation groups are currently not handled properly by the sampling
128 // allocation profiler, and
129 // 2. Observers might try to take the space lock, which isn't reentrant.
130 // We'll have to come up with a better solution for allocation stepping
131 // before shipping, which will likely be using LocalHeap.
132 }
133 const bool is_from_client_heap =
134 (other->destination_heap() ==
135 CompactionSpace::DestinationHeap::kSharedSpaceHeap);
136 DCHECK_IMPLIES(is_from_client_heap, identity() == SHARED_SPACE);
137 for (auto p : other->GetNewPages()) {
138 heap()->NotifyOldGenerationExpansion(
139 heap()->main_thread_local_heap(), identity(), p,
140 is_from_client_heap
141 ? Heap::OldGenerationExpansionNotificationOrigin::kFromClientHeap
142 : Heap::OldGenerationExpansionNotificationOrigin::kFromSameHeap);
143 }
144
145 DCHECK_EQ(0u, other->Size());
146 DCHECK_EQ(0u, other->Capacity());
147}

157void PagedSpaceBase::IncrementCommittedPhysicalMemory(size_t increment_value) {
158 if (!base::OS::HasLazyCommits() || increment_value == 0) return;
159 size_t old_value = committed_physical_memory_.fetch_add(
160 increment_value, std::memory_order_relaxed);
161 USE(old_value);
162 DCHECK_LT(old_value, old_value + increment_value);
163}
164
165void PagedSpaceBase::DecrementCommittedPhysicalMemory(size_t decrement_value) {
166 if (!base::OS::HasLazyCommits() || decrement_value == 0) return;
167 size_t old_value = committed_physical_memory_.fetch_sub(
168 decrement_value, std::memory_order_relaxed);
169 USE(old_value);
170 DCHECK_GT(old_value, old_value - decrement_value);
171}

194void PagedSpaceBase::RefineAllocatedBytesAfterSweeping(PageMetadata* page) {
195 CHECK(page->SweepingDone());
196 // The live_byte on the page was accounted in the space allocated
197 // bytes counter. After sweeping allocated_bytes() contains the
198 // accurate live byte count on the page.
199 size_t old_counter = page->live_bytes();
200 size_t new_counter = page->allocated_bytes();
201 DCHECK_GE(old_counter, new_counter);
202 if (old_counter > new_counter) {
203 size_t counter_diff = old_counter - new_counter;
204 if (identity() == NEW_SPACE) size_at_last_gc_ -= counter_diff;
205 DecreaseAllocatedBytes(counter_diff, page);
206 DCHECK_EQ(new_counter, accounting_stats_.AllocatedOnPage(page));
207 AdjustDifferenceInAllocatedBytes(counter_diff);
208 }
209 if (!v8_flags.sticky_mark_bits) {
210 // With sticky mark-bits the counter is reset on unmarking.
211 page->SetLiveBytes(0);
212 }
213}

223void PagedSpaceBase::AddPageImpl(PageMetadata* page) {
224 DCHECK_NOT_NULL(page);
225 CHECK(page->SweepingDone());
226 page->set_owner(this);
227 DCHECK_IMPLIES(identity() == NEW_SPACE,
228 page->Chunk()->IsFlagSet(MemoryChunk::TO_PAGE));
229 DCHECK_IMPLIES(identity() != NEW_SPACE,
230 !page->Chunk()->IsFlagSet(MemoryChunk::TO_PAGE));
231 memory_chunk_list_.PushBack(page);
232 AccountCommitted(page->size());
233 IncreaseCapacity(page->area_size());
234 IncreaseAllocatedBytes(page->allocated_bytes(), page);
235 ForAll<ExternalBackingStoreType>(
236 [this, page](ExternalBackingStoreType type, int index) {
237 IncrementExternalBackingStoreBytes(
238 type, page->ExternalBackingStoreBytes(type));
239 });
240 IncrementCommittedPhysicalMemory(page->CommittedPhysicalMemory());
241}

243size_t PagedSpaceBase::AddPage(PageMetadata* page) {
244 AddPageImpl(page);
245 return RelinkFreeListCategories(page);
246}

248void PagedSpaceBase::RemovePage(PageMetadata* page) {
249 CHECK(page->SweepingDone());
250 DCHECK_IMPLIES(identity() == NEW_SPACE,
251 page->Chunk()->IsFlagSet(MemoryChunk::TO_PAGE));
252 memory_chunk_list_.Remove(page);
253 UnlinkFreeListCategories(page);
254 // Pages are only removed from new space when they are promoted to old space
255 // during a GC. This happens after sweeping as started and the allocation
256 // counters have been reset.
257 DCHECK_IMPLIES(identity() == NEW_SPACE,
258 heap()->gc_state() != Heap::NOT_IN_GC);
259 if (identity() == NEW_SPACE) {
260 page->ReleaseFreeListCategories();
261 } else {
262 DecreaseAllocatedBytes(page->allocated_bytes(), page);
263 free_list()->decrease_wasted_bytes(page->wasted_memory());
264 }
265 DecreaseCapacity(page->area_size());
266 AccountUncommitted(page->size());
267 ForAll<ExternalBackingStoreType>(
268 [this, page](ExternalBackingStoreType type, int index) {
269 DecrementExternalBackingStoreBytes(
270 type, page->ExternalBackingStoreBytes(type));
271 });
272 DecrementCommittedPhysicalMemory(page->CommittedPhysicalMemory());
273}

299bool PagedSpaceBase::TryExpand(LocalHeap* local_heap, AllocationOrigin origin) {
300 DCHECK_EQ(!local_heap, origin == AllocationOrigin::kGC);
301 const size_t accounted_size =
302 MemoryChunkLayout::AllocatableMemoryInMemoryChunk(identity());
303 if (origin != AllocationOrigin::kGC && identity() != NEW_SPACE) {
304 base::SpinningMutexGuard expansion_guard(heap_->heap_expansion_mutex());
305 if (!heap()->IsOldGenerationExpansionAllowed(accounted_size,
306 expansion_guard)) {
307 return false;
308 }
309 }
310 const MemoryAllocator::AllocationMode allocation_mode =
311 (identity() == NEW_SPACE || identity() == OLD_SPACE)
312 ? MemoryAllocator::AllocationMode::kUsePool
313 : MemoryAllocator::AllocationMode::kRegular;
314 PageMetadata* page = heap()->memory_allocator()->AllocatePage(
315 allocation_mode, this, executable());
316 if (page == nullptr) return false;
317 DCHECK_EQ(page->area_size(), accounted_size);
318 ConcurrentAllocationMutex guard(this);
319 AddPage(page);
320 if (origin != AllocationOrigin::kGC && identity() != NEW_SPACE) {
321 heap()->NotifyOldGenerationExpansion(local_heap, identity(), page);
322 }
323 Free(page->area_start(), page->area_size());
324 NotifyNewPage(page);
325 return true;
326}

345void PagedSpaceBase::ReleasePageImpl(PageMetadata* page,
346 MemoryAllocator::FreeMode free_mode) {
347 DCHECK(page->SweepingDone());
348 DCHECK_EQ(0, page->live_bytes());
349 DCHECK_EQ(page->owner(), this);
350
351 DCHECK_IMPLIES(identity() == NEW_SPACE,
352 page->Chunk()->IsFlagSet(MemoryChunk::TO_PAGE));
353
354 memory_chunk_list_.Remove(page);
355
356 free_list_->EvictFreeListItems(page);
357
358 if (identity() == CODE_SPACE) {
359 heap()->isolate()->RemoveCodeMemoryChunk(page);
360 }
361
362 AccountUncommitted(page->size());
363 DecrementCommittedPhysicalMemory(page->CommittedPhysicalMemory());
364 accounting_stats_.DecreaseCapacity(page->area_size());
365 heap()->memory_allocator()->Free(free_mode, page);
366}

377#ifdef VERIFY_HEAP
378void PagedSpaceBase::Verify(Isolate* isolate,
379 SpaceVerificationVisitor* visitor) const {
380 CHECK_IMPLIES(identity() != NEW_SPACE, size_at_last_gc_ == 0);
381
382 size_t external_space_bytes[static_cast<int>(
383 ExternalBackingStoreType::kNumValues)] = {0};
384 PtrComprCageBase cage_base(isolate);
385 for (const PageMetadata* page : *this) {
386 size_t external_page_bytes[static_cast<int>(
387 ExternalBackingStoreType::kNumValues)] = {0};
388
389 CHECK_EQ(page->owner(), this);
390 CHECK_IMPLIES(identity() != NEW_SPACE, page->AllocatedLabSize() == 0);
391 visitor->VerifyPage(page);
392
393 CHECK(page->SweepingDone());
394 Address end_of_previous_object = page->area_start();
395 Address top = page->area_end();
396
397 for (Tagged<HeapObject> object : HeapObjectRange(page)) {
398 CHECK(end_of_previous_object <= object.address());
399
400 // Invoke verification method for each object.
401 visitor->VerifyObject(object);
402
403 // All the interior pointers should be contained in the heap.
404 int size = object->Size(cage_base);
405 CHECK(object.address() + size <= top);
406 end_of_previous_object = object.address() + size;
407
408 if (IsExternalString(object, cage_base)) {
409 Tagged<ExternalString> external_string = Cast<ExternalString>(object);
410 size_t payload_size = external_string->ExternalPayloadSize();
411 external_page_bytes[static_cast<int>(
412 ExternalBackingStoreType::kExternalString)] += payload_size;
413 }
414 }
415 ForAll<ExternalBackingStoreType>(
416 [page, external_page_bytes, &external_space_bytes](
417 ExternalBackingStoreType type, int index) {
418 CHECK_EQ(external_page_bytes[index],
419 page->ExternalBackingStoreBytes(type));
420 external_space_bytes[index] += external_page_bytes[index];
421 });
422
423 visitor->VerifyPageDone(page);
424 }
425 ForAll<ExternalBackingStoreType>(
426 [this, external_space_bytes](ExternalBackingStoreType type, int index) {
427 if (type == ExternalBackingStoreType::kArrayBuffer) {
428 return;
429 }
430 CHECK_EQ(external_space_bytes[index], ExternalBackingStoreBytes(type));
431 });
432
433 if (!v8_flags.concurrent_array_buffer_sweeping) {
434 if (identity() == OLD_SPACE) {
435 size_t bytes = heap()->array_buffer_sweeper()->old().BytesSlow();
436 CHECK_EQ(bytes, ExternalBackingStoreBytes(
437 ExternalBackingStoreType::kArrayBuffer));
438 } else if (identity() == NEW_SPACE) {
439 CHECK(v8_flags.minor_ms);
440 size_t bytes = heap()->array_buffer_sweeper()->young().BytesSlow();
441 CHECK_EQ(bytes, ExternalBackingStoreBytes(
442 ExternalBackingStoreType::kArrayBuffer));
443 }
444 }
445
446#ifdef DEBUG
447 VerifyCountersAfterSweeping(isolate->heap());
448#endif
449}

451void PagedSpaceBase::VerifyLiveBytes() const {
452 MarkingState* marking_state = heap()->marking_state();
453 PtrComprCageBase cage_base(heap()->isolate());
454 for (const PageMetadata* page : *this) {
455 CHECK(page->SweepingDone());
456 int black_size = 0;
457 for (Tagged<HeapObject> object : HeapObjectRange(page)) {
458 // All the interior pointers should be contained in the heap.
459 if (marking_state->IsMarked(object)) {
460 black_size += object->Size(cage_base);
461 }
462 }
463 CHECK_LE(black_size, page->live_bytes());
464 }

469void PagedSpaceBase::VerifyCountersAfterSweeping(Heap* heap) const {
470 size_t total_capacity = 0;
471 size_t total_allocated = 0;
472 PtrComprCageBase cage_base(heap->isolate());
473 for (const PageMetadata* page : *this) {
474 DCHECK(page->SweepingDone());
475 total_capacity += page->area_size();
476 size_t real_allocated = 0;
477 for (Tagged<HeapObject> object : HeapObjectRange(page)) {
478 if (!IsFreeSpaceOrFiller(object)) {
479 real_allocated +=
480 ALIGN_TO_ALLOCATION_ALIGNMENT(object->Size(cage_base));

484 // The real size can be smaller than the accounted size if array trimming,
485 // object slack tracking happened after sweeping.
486 DCHECK_LE(real_allocated, accounting_stats_.AllocatedOnPage(page));
487 DCHECK_EQ(page->allocated_bytes(), accounting_stats_.AllocatedOnPage(page));
488 }
489 DCHECK_EQ(total_capacity, accounting_stats_.Capacity());
490 DCHECK_EQ(total_allocated, accounting_stats_.Size());
491}

509 Address start, Address end) {
510 MemoryChunk* chunk = page->Chunk();
511 DCHECK_LE(chunk->address(), start);

538size_t PagedSpaceBase::RelinkFreeListCategories(PageMetadata* page) {
539 DCHECK_EQ(this, page->owner());
540 size_t added = 0;
541 page->ForAllFreeListCategories([this, &added](FreeListCategory* category) {
542 added += category->available();

544 });
545 free_list()->increase_wasted_bytes(page->wasted_memory());

548 page->AvailableInFreeList() ==

553void PagedSpaceBase::RefillFreeList() {
554 // Any PagedSpace might invoke RefillFreeList.
555 DCHECK(identity() == OLD_SPACE || identity() == CODE_SPACE ||
556 identity() == SHARED_SPACE || identity() == NEW_SPACE ||
557 identity() == TRUSTED_SPACE);
558 DCHECK_IMPLIES(identity() == NEW_SPACE, heap_->IsMainThread());
559 DCHECK(!is_compaction_space());
560
561 for (PageMetadata* p : heap()->sweeper()->GetAllSweptPagesSafe(this)) {
562 // We regularly sweep NEVER_ALLOCATE_ON_PAGE pages. We drop the freelist
563 // entries here to make them unavailable for allocations.
564 if (p->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE)) {
565 free_list_->EvictFreeListItems(p);
566 }
567
568 ConcurrentAllocationMutex guard(this);
569 DCHECK_EQ(this, p->owner());
570 RefineAllocatedBytesAfterSweeping(p);
571 RelinkFreeListCategories(p);
572 }
573}

582void CompactionSpace::NotifyNewPage(PageMetadata* page) {
583 // Incremental marking can be running on the main thread isolate, so when
584 // allocating a new page for the client's compaction space we can get a black
585 // allocated page. This is fine, since the page is not observed the main
586 // isolate until it's merged.
587 DCHECK_IMPLIES(identity() != SHARED_SPACE ||
588 destination_heap() != DestinationHeap::kSharedSpaceHeap,
589 !page->Chunk()->IsFlagSet(MemoryChunk::BLACK_ALLOCATED));
590 new_pages_.push_back(page);
591}
592
593void CompactionSpace::RefillFreeList() {
594 DCHECK_NE(NEW_SPACE, identity());
595
596 Sweeper* sweeper = heap()->sweeper();
597 size_t added = 0;
598 PageMetadata* p = nullptr;
599 while ((added <= kCompactionMemoryWanted) &&
600 (p = sweeper->GetSweptPageSafe(this))) {
601 // We regularly sweep NEVER_ALLOCATE_ON_PAGE pages. We drop the freelist
602 // entries here to make them unavailable for allocations.
603 if (p->Chunk()->IsFlagSet(MemoryChunk::NEVER_ALLOCATE_ON_PAGE)) {
604 free_list()->EvictFreeListItems(p);
605 }
606
607 // Only during compaction pages can actually change ownership. This is
608 // safe because there exists no other competing action on the page links
609 // during compaction.
610 DCHECK_NE(this, p->owner());
611 PagedSpace* owner = static_cast<PagedSpace*>(p->owner());
612 base::SpinningMutexGuard guard(owner->mutex());
613 owner->RefineAllocatedBytesAfterSweeping(p);
614 owner->RemovePage(p);
615 added += AddPage(p);
616 added += p->wasted_memory();
617 }
618}

620CompactionSpaceCollection::CompactionSpaceCollection(
621 Heap* heap, CompactionSpaceKind compaction_space_kind)
622 : old_space_(heap, OLD_SPACE, Executability::NOT_EXECUTABLE,
623 compaction_space_kind,
624 CompactionSpace::DestinationHeap::kSameHeap),
625 code_space_(heap, CODE_SPACE, Executability::EXECUTABLE,
626 compaction_space_kind,
627 CompactionSpace::DestinationHeap::kSameHeap),
628 trusted_space_(heap, TRUSTED_SPACE, Executability::NOT_EXECUTABLE,
629 compaction_space_kind,
630 CompactionSpace::DestinationHeap::kSameHeap) {
631 if (heap->isolate()->has_shared_space()) {
632 const CompactionSpace::DestinationHeap dest_heap =
633 heap->isolate()->is_shared_space_isolate()
634 ? CompactionSpace::DestinationHeap::kSameHeap
635 : CompactionSpace::DestinationHeap::kSharedSpaceHeap;

645void OldSpace::AddPromotedPage(PageMetadata* page) {
646 if (v8_flags.minor_ms) {
647 // Reset the page's allocated bytes. The page will be swept and the
648 // allocated bytes will be updated to match the live bytes.
649 DCHECK_EQ(page->area_size(), page->allocated_bytes());
650 page->DecreaseAllocatedBytes(page->area_size());
651 }
652 AddPageImpl(page);
653 if (!v8_flags.minor_ms) {
654 RelinkFreeListCategories(page);
655 }
656}
