16// CastTraits<T> is a type trait that defines type checking behaviour for
17// tagged object casting. The expected specialization is:
18//
19// template<>
20// struct CastTraits<SomeObject> {
21// template<typename From>
22// static bool AllowFrom(Tagged<From> value) {
23// return IsSomeObject(value);
24// }
25// };
26//
27// or, likely, just specializations of AllowFrom for Object and HeapObject,
28// under the assumption that the HeapObject implementation is the same for all
29// HeapObjects and the Object implementation has additional overhead in Smi
30// checks.
31//
32// struct CastTraits<Object> {
33// static bool AllowFrom(Tagged<HeapObject> value) {
34// return IsSomeObject(value);
35// }
36// static bool AllowFrom(Tagged<Object> value) {
37// return IsSomeObject(value);
38// }
39// };
40//
41template <typename To>
42struct CastTraits;

131// `GCSafeCast<T>(value)` casts `object` to a tagged object of type `T` and
132// should be used when the cast can be called from within a GC. The case all
133// includes a debug check that `object` is either a tagged object of type `T`,
134// or one of few special cases possible during GC (see GCAwareObjectTypeCheck):
135// 1) `object` was already evacuated and the forwarding address refers to a
136// tagged object of type `T`.
137// 2) During Scavanger, `object` is a large object.
138// 3) During a conservative Scavenger, `object` is a pinned object.
139template <typename T>
140Tagged<T> GCSafeCast(Tagged<Object> object, const Heap* heap) {
141 DCHECK(GCAwareObjectTypeCheck<T>(object, heap));
142 return UncheckedCast<T>(object);
143}

207// `Is<T>(maybe_weak_value)` specialization for possible weak values and strong
208// target `T`, that additionally first checks whether `maybe_weak_value` is
209// actually a strong value (or a Smi, which can't be weak).
210template <typename T, typename U>
211inline bool Is(Tagged<MaybeWeak<U>> value) {
212 // Cast from maybe weak to strong needs to be strong or smi.
213 if constexpr (!is_maybe_weak_v<T>) {
214 if (!value.IsStrongOrSmi()) return false;
215 return CastTraits<T>::AllowFrom(Tagged<U>(value.ptr()));
216 } else {
217 // Dispatches to CastTraits<MaybeWeak<T>> below.
218 return CastTraits<T>::AllowFrom(value);
219 }
220}
221template <typename T, typename... U>
222inline bool Is(Tagged<Union<U...>> value) {
223 using UnionU = Union<U...>;
224 if constexpr (is_subtype_v<UnionU, HeapObject>) {
225 return Is<T>(Tagged<HeapObject>(value));
226 } else if constexpr (is_subtype_v<UnionU, MaybeWeak<HeapObject>>) {
227 return Is<T>(Tagged<MaybeWeak<HeapObject>>(value));
228 } else if constexpr (is_subtype_v<UnionU, Object>) {
229 return Is<T>(Tagged<Object>(value));
230 } else {
231 static_assert(is_subtype_v<UnionU, MaybeWeak<Object>>);
232 return Is<T>(Tagged<MaybeWeak<Object>>(value));
233 }
234}

236// Specialization for maybe weak cast targets, which first converts the incoming
237// value to a strong reference and then checks if the cast to the strong T
238// is allowed. Cleared weak references always return true.
239template <typename T>
240struct CastTraits<MaybeWeak<T>> {
241 template <typename U>
242 static bool AllowFrom(Tagged<U> value) {
243 if constexpr (is_maybe_weak_v<U>) {
244 // Cleared values are always ok.
245 if (value.IsCleared()) return true;
246 // TODO(leszeks): Skip Smi check for values that are known to not be Smi.
247 if (value.IsSmi()) {
248 return CastTraits<T>::AllowFrom(Tagged<Smi>(value.ptr()));
249 }
250 return CastTraits<T>::AllowFrom(MakeStrong(value));
251 } else {
252 return CastTraits<T>::AllowFrom(value);
253 }
254 }
255};
