31void MemoryChunkMetadata::UpdateHighWaterMark(Address mark) {
32 if (mark == kNullAddress) return;
33 // Need to subtract one from the mark because when a chunk is full the
34 // top points to the next address after the chunk, which effectively belongs
35 // to another chunk. See the comment to
36 // PageMetadata::FromAllocationAreaAddress.
37 MemoryChunkMetadata* chunk = MemoryChunkMetadata::FromAddress(mark - 1);
38 intptr_t new_mark = static_cast<intptr_t>(mark - chunk->ChunkAddress());
39 intptr_t old_mark = chunk->high_water_mark_.load(std::memory_order_relaxed);
40 while ((new_mark > old_mark) &&
41 !chunk->high_water_mark_.compare_exchange_weak(
42 old_mark, new_mark, std::memory_order_acq_rel)) {
43 }
44}
