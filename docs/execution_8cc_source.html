25DirectHandle<Object> NormalizeReceiver(Isolate* isolate,
26 DirectHandle<Object> receiver) {
27 // Convert calls on global objects to be calls on the global
28 // receiver instead to avoid having a 'this' pointer which refers
29 // directly to a global object.
30 if (IsJSGlobalObject(*receiver)) {
31 return direct_handle(Cast<JSGlobalObject>(receiver)->global_proxy(),
32 isolate);
33 }
34 return receiver;
35}

181MaybeHandle<Context> NewScriptContext(
182 Isolate* isolate, DirectHandle<JSFunction> function,
183 DirectHandle<FixedArray> host_defined_options) {
184 // TODO(cbruni, 1244145): Use passed in host_defined_options.
185 // Creating a script context is a side effect, so abort if that's not
186 // allowed.
187 if (isolate->should_check_side_effects()) {
188 isolate->Throw(*isolate->factory()->NewEvalError(
189 MessageTemplate::kNoSideEffectDebugEvaluate));
190 return MaybeHandle<Context>();
191 }
192 SaveAndSwitchContext save(isolate, function->context());
193 Tagged<SharedFunctionInfo> sfi = function->shared();
194 Handle<Script> script(Cast<Script>(sfi->script()), isolate);
195 Handle<ScopeInfo> scope_info(sfi->scope_info(), isolate);
196 DirectHandle<NativeContext> native_context(
197 Cast<NativeContext>(function->context()), isolate);
198 DirectHandle<JSGlobalObject> global_object(native_context->global_object(),
199 isolate);
200 Handle<ScriptContextTable> script_context(
201 native_context->script_context_table(), isolate);
202
203 // Find name clashes.
204 for (auto it : ScopeInfo::IterateLocalNames(scope_info)) {
205 Handle<String> name(it->name(), isolate);
206 VariableMode mode = scope_info->ContextLocalMode(it->index());
207 VariableLookupResult lookup;
208 if (script_context->Lookup(name, &lookup)) {
209 if (IsLexicalVariableMode(mode) || IsLexicalVariableMode(lookup.mode)) {
210 DirectHandle<Context> context(script_context->get(lookup.context_index),
211 isolate);
212 // If we are trying to re-declare a REPL-mode let as a let, REPL-mode
213 // const as a const, REPL-mode using as a using and REPL-mode await
214 // using as an await using allow it.
215 if (!((mode == lookup.mode && IsLexicalVariableMode(mode)) &&
216 scope_info->IsReplModeScope() &&
217 context->scope_info()->IsReplModeScope())) {
218 // ES#sec-globaldeclarationinstantiation 5.b:
219 // If envRec.HasLexicalDeclaration(name) is true, throw a SyntaxError
220 // exception.
221 MessageLocation location(script, 0, 1);
222 isolate->ThrowAt(isolate->factory()->NewSyntaxError(
223 MessageTemplate::kVarRedeclaration, name),
224 &location);
225 return MaybeHandle<Context>();
226 }
227 }
228 }
229
230 if (IsLexicalVariableMode(mode)) {
231 LookupIterator it(isolate, global_object, name, global_object,
232 LookupIterator::OWN_SKIP_INTERCEPTOR);
233 Maybe<PropertyAttributes> maybe = JSReceiver::GetPropertyAttributes(&it);
234 // Can't fail since the we looking up own properties on the global object
235 // skipping interceptors.
236 CHECK(!maybe.IsNothing());
237 if ((maybe.FromJust() & DONT_DELETE) != 0) {
238 // ES#sec-globaldeclarationinstantiation 5.a:
239 // If envRec.HasVarDeclaration(name) is true, throw a SyntaxError
240 // exception.
241 // ES#sec-globaldeclarationinstantiation 5.d:
242 // If hasRestrictedGlobal is true, throw a SyntaxError exception.
243 MessageLocation location(script, 0, 1);
244 isolate->ThrowAt(isolate->factory()->NewSyntaxError(
245 MessageTemplate::kVarRedeclaration, name),
246 &location);
247 return MaybeHandle<Context>();
248 }
249
250 JSGlobalObject::InvalidatePropertyCell(global_object, name);
251 }
252 }
253
254 Handle<Context> result =
255 isolate->factory()->NewScriptContext(native_context, scope_info);
256
257 result->Initialize(isolate);
258 // In REPL mode, we are allowed to add/modify let/const/using/await using
259 // variables. We use the previous defined script context for those.
260 const bool ignore_duplicates = scope_info->IsReplModeScope();
261 DirectHandle<ScriptContextTable> new_script_context_table =
262 ScriptContextTable::Add(isolate, script_context, result,
263 ignore_duplicates);
264 native_context->synchronized_set_script_context_table(
265 *new_script_context_table);
266 return result;
267}

269V8_WARN_UNUSED_RESULT MaybeHandle<Object> Invoke(Isolate* isolate,
270 const InvokeParams& params) {
271 RCS_SCOPE(isolate, RuntimeCallCounterId::kInvoke);
272 DCHECK(!IsJSGlobalObject(*params.receiver));
273 DCHECK_LE(params.args.size(), FixedArray::kMaxLength);
274 DCHECK(!isolate->has_exception());
275
276#if V8_ENABLE_WEBASSEMBLY
277 // If we have PKU support for Wasm, ensure that code is currently write
278 // protected for this thread.
279 DCHECK_IMPLIES(wasm::GetWasmCodeManager()->HasMemoryProtectionKeySupport(),
280 !wasm::GetWasmCodeManager()->MemoryProtectionKeyWritable());
281#endif // V8_ENABLE_WEBASSEMBLY
282
283#ifdef USE_SIMULATOR
284 // Simulators use separate stacks for C++ and JS. JS stack overflow checks
285 // are performed whenever a JS function is called. However, it can be the case
286 // that the C++ stack grows faster than the JS stack, resulting in an overflow
287 // there. Add a check here to make that less likely.
288 StackLimitCheck check(isolate);
289 if (check.HasOverflowed()) {
290 isolate->StackOverflow();
291 isolate->ReportPendingMessages(params.message_handling ==
292 Execution::MessageHandling::kReport);
293 return MaybeHandle<Object>();
294 }
295#endif
296
297 // api callbacks can be called directly, unless we want to take the detour
298 // through JS to set up a frame for break-at-entry.
299 if (IsJSFunction(*params.target)) {
300 auto function = Cast<JSFunction>(params.target);
301 if ((!params.is_construct || IsConstructor(*function)) &&
302 function->shared()->IsApiFunction() &&
303 !function->shared()->BreakAtEntry(isolate)) {
304 SaveAndSwitchContext save(isolate, function->context());
305 DCHECK(IsJSGlobalObject(function->context()->global_object()));
306
307 DirectHandle<Object> receiver = params.is_construct
308 ? isolate->factory()->the_hole_value()
309 : params.receiver;
310 DirectHandle<FunctionTemplateInfo> fun_data(
311 function->shared()->api_func_data(), isolate);
312 auto value = Builtins::InvokeApiFunction(
313 isolate, params.is_construct, fun_data, receiver, params.args,
314 Cast<HeapObject>(params.new_target));
315 bool has_exception = value.is_null();
316 DCHECK_EQ(has_exception, isolate->has_exception());
317 if (has_exception) {
318 isolate->ReportPendingMessages(params.message_handling ==
319 Execution::MessageHandling::kReport);
320 return MaybeHandle<Object>();
321 } else {
322 isolate->clear_pending_message();
323 }
324 return value;
325 }
326#ifdef DEBUG
327 if (function->shared()->is_script()) {
328 DCHECK(params.IsScript());
329 DCHECK(IsJSGlobalProxy(*params.receiver));
330 DCHECK_EQ(params.args.size(), 1);
331 DCHECK(IsFixedArray(*params.args[0]));
332 } else {
333 DCHECK(!params.IsScript());
334 }
335#endif
336 // Set up a ScriptContext when running scripts that need it.
337 if (function->shared()->needs_script_context()) {
338 DirectHandle<Context> context;
339 DirectHandle<FixedArray> host_defined_options =
340 const_cast<InvokeParams&>(params).GetAndResetHostDefinedOptions();
341 if (!NewScriptContext(isolate, function, host_defined_options)
342 .ToHandle(&context)) {
343 isolate->ReportPendingMessages(params.message_handling ==
344 Execution::MessageHandling::kReport);
345 return MaybeHandle<Object>();
346 }
347
348 // We mutate the context if we allocate a script context. This is
349 // guaranteed to only happen once in a native context since scripts will
350 // always produce name clashes with themselves.
351 function->set_context(*context);
352 }
353 }
354
355 // Entering JavaScript.
356 VMState<JS> state(isolate);
357 if (!AllowJavascriptExecution::IsAllowed(isolate)) {
358 GRACEFUL_FATAL("Invoke in DisallowJavascriptExecutionScope");
359 }
360 if (!ThrowOnJavascriptExecution::IsAllowed(isolate)) {
361 isolate->ThrowIllegalOperation();
362 isolate->ReportPendingMessages(params.message_handling ==
363 Execution::MessageHandling::kReport);
364 return MaybeHandle<Object>();
365 }
366 if (!DumpOnJavascriptExecution::IsAllowed(isolate)) {
367 V8::GetCurrentPlatform()->DumpWithoutCrashing();
368 return isolate->factory()->undefined_value();
369 }
370 isolate->IncrementJavascriptExecutionCounter();
371
372 if (params.execution_target == Execution::Target::kCallable) {
373 DirectHandle<NativeContext> context = isolate->native_context();
374 if (!IsUndefined(context->script_execution_callback(), isolate)) {
375 v8::Context::AbortScriptExecutionCallback callback =
376 v8::ToCData<v8::Context::AbortScriptExecutionCallback,
377 kApiAbortScriptExecutionCallbackTag>(
378 isolate, context->script_execution_callback());
379 v8::Isolate* api_isolate = reinterpret_cast<v8::Isolate*>(isolate);
380 v8::Local<v8::Context> api_context = v8::Utils::ToLocal(context);
381 callback(api_isolate, api_context);
382 DCHECK(!isolate->has_exception());
383 // Always throw an exception to abort execution, if callback exists.
384 isolate->ThrowIllegalOperation();
385 return MaybeHandle<Object>();
386 }
387 }
388
389 // Placeholder for return value.
390 Tagged<Object> value;
391 DirectHandle<Code> code =
392 JSEntry(isolate, params.execution_target, params.is_construct);
393 {
394 // Save and restore context around invocation.
395 SaveContext save(isolate);
396
397 if (v8_flags.clear_exceptions_on_js_entry) isolate->clear_exception();
398
399 if (params.execution_target == Execution::Target::kCallable) {
400 // clang-format off
401 // {new_target}, {target}, {receiver}, return value: tagged pointers
402 // {argv}: pointer to array of tagged pointers
403 using JSEntryFunction = GeneratedCode<Address(
404 Address root_register_value, Address new_target, Address target,
405 Address receiver, intptr_t argc, Address** argv)>;
406 // clang-format on
407 JSEntryFunction stub_entry =
408 JSEntryFunction::FromAddress(isolate, code->instruction_start());
409
410 Address orig_func = (*params.new_target).ptr();
411 Address func = (*params.target).ptr();
412 Address recv = (*params.receiver).ptr();
413
414 int argc = static_cast<int>(params.args.size());
415#ifdef V8_ENABLE_DIRECT_HANDLE
416 // TODO(42203211): Store the arguments to indirect handles because
417 // generated code still expects them in indirect handles. A fresh handle
418 // scope is introduced for these handles, which is then sealed. When this
419 // is eventually removed, sealing can go back together with saving the
420 // context for both branches.
421 HandleScope scope_for_conversion(isolate);
422 std::vector<IndirectHandle<Object>> args(argc);
423 for (int i = 0; i < argc; ++i)
424 args[i] = indirect_handle(params.args[i], isolate);
425 Address** argv = reinterpret_cast<Address**>(args.data());
426#else
427 Address** argv = const_cast<Address**>(
428 reinterpret_cast<Address* const*>(params.args.data()));
429#endif
430
431 // Block the allocation of handles without explicit handle scopes.
432 SealHandleScope shs(isolate);
433
434 RCS_SCOPE(isolate, RuntimeCallCounterId::kJS_Execution);
435 value = Tagged<Object>(
436 stub_entry.Call(isolate->isolate_data()->isolate_root(), orig_func,
437 func, recv, JSParameterCount(argc), argv));
438 } else {
439 DCHECK_EQ(Execution::Target::kRunMicrotasks, params.execution_target);
440
441 // clang-format off
442 // return value: tagged pointers
443 // {microtask_queue}: pointer to a C++ object
444 using JSEntryFunction = GeneratedCode<Address(
445 Address root_register_value, MicrotaskQueue* microtask_queue)>;
446 // clang-format on
447 JSEntryFunction stub_entry =
448 JSEntryFunction::FromAddress(isolate, code->instruction_start());
449
450 // Block the allocation of handles without explicit handle scopes.
451 SealHandleScope shs(isolate);
452
453 RCS_SCOPE(isolate, RuntimeCallCounterId::kJS_Execution);
454 value = Tagged<Object>(stub_entry.Call(
455 isolate->isolate_data()->isolate_root(), params.microtask_queue));
456 }
457 }
458
459#ifdef VERIFY_HEAP
460 if (v8_flags.verify_heap) {
461 Object::ObjectVerify(value, isolate);
462 }
463#endif
464
465 // Update the pending exception flag and return the value.
466 bool has_exception = IsException(value, isolate);
467 DCHECK_EQ(has_exception, isolate->has_exception());
468 if (has_exception) {
469 isolate->ReportPendingMessages(params.message_handling ==
470 Execution::MessageHandling::kReport);
471 return MaybeHandle<Object>();
472 } else {
473 isolate->clear_pending_message();
474 }
475
476 return Handle<Object>(value, isolate);
477}

479MaybeHandle<Object> InvokeWithTryCatch(Isolate* isolate,
480 const InvokeParams& params) {
481 DCHECK_IMPLIES(v8_flags.strict_termination_checks,
482 !isolate->is_execution_terminating());
483 MaybeHandle<Object> maybe_result;
484 if (params.exception_out != nullptr) {
485 *params.exception_out = {};
486 }
487
488 // Enter a try-block while executing the JavaScript code. To avoid
489 // duplicate error printing it must be non-verbose. Also, to avoid
490 // creating message objects during stack overflow we shouldn't
491 // capture messages.
492 v8::TryCatch catcher(reinterpret_cast<v8::Isolate*>(isolate));
493 catcher.SetVerbose(false);
494 catcher.SetCaptureMessage(false);
495
496 maybe_result = Invoke(isolate, params);
497
498 if (V8_LIKELY(!maybe_result.is_null())) {
499 DCHECK(!isolate->has_exception());
500 return maybe_result;
501 }
502
503 DCHECK(isolate->has_exception());
504 if (isolate->is_execution_terminating()) {
505 return maybe_result;
506 }
507
508 if (params.exception_out != nullptr) {
509 DCHECK(catcher.HasCaught());
510 *params.exception_out = v8::Utils::OpenDirectHandle(*catcher.Exception());
511 }
512
513 return maybe_result;
514}

614#if V8_ENABLE_WEBASSEMBLY
615void Execution::CallWasm(Isolate* isolate, DirectHandle<Code> wrapper_code,
616 WasmCodePointer wasm_call_target,
617 DirectHandle<Object> object_ref, Address packed_args) {
618 using WasmEntryStub = GeneratedCode<Address(
619 Address target, Address object_ref, Address argv, Address c_entry_fp)>;
620 WasmEntryStub stub_entry =
621 WasmEntryStub::FromAddress(isolate, wrapper_code->instruction_start());
622
623 // Save and restore context around invocation and block the
624 // allocation of handles without explicit handle scopes.
625 SaveContext save(isolate);
626 SealHandleScope shs(isolate);
627
628 Address saved_c_entry_fp = *isolate->c_entry_fp_address();
629 Address saved_js_entry_sp = *isolate->js_entry_sp_address();
630 if (saved_js_entry_sp == kNullAddress) {
631 *isolate->js_entry_sp_address() = GetCurrentStackPosition();
632 }
633 StackHandlerMarker stack_handler;
634 stack_handler.next = isolate->thread_local_top()->handler_;
635#ifdef V8_USE_ADDRESS_SANITIZER
636 stack_handler.padding = GetCurrentStackPosition();
637#else
638 stack_handler.padding = 0;
639#endif
640 isolate->thread_local_top()->handler_ =
641 reinterpret_cast<Address>(&stack_handler);
642 trap_handler::SetThreadInWasm();
643
644 {
645 RCS_SCOPE(isolate, RuntimeCallCounterId::kJS_Execution);
646 static_assert(compiler::CWasmEntryParameters::kCodeEntry == 0);
647 static_assert(compiler::CWasmEntryParameters::kObjectRef == 1);
648 static_assert(compiler::CWasmEntryParameters::kArgumentsBuffer == 2);
649 static_assert(compiler::CWasmEntryParameters::kCEntryFp == 3);
650 Address result =
651 stub_entry.Call(wasm_call_target.value(), (*object_ref).ptr(),
652 packed_args, saved_c_entry_fp);
653 if (result != kNullAddress) isolate->set_exception(Tagged<Object>(result));
654 }
