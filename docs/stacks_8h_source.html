23struct JumpBuffer {
24 Address sp;
25 Address fp;
26 Address pc;
27 void* stack_limit;
28 // We track the state below to prevent stack corruptions under the sandbox
29 // security model.
30 // Assuming that the external pointer to the jump buffer has been corrupted
31 // and replaced with a different jump buffer, we check its state before
32 // resuming it to verify that it is not Active or Retired.
33 // The distinction between Suspended and Inactive may not be strictly
34 // necessary since we currently always pass a single JS value in the return
35 // register across stacks (either the Promise, the result of the Promise, or
36 // the result of the export). However adding a state does not cost anything
37 // and is more robust against potential changes in the calling conventions.
38 enum StackState : int32_t {
39 Active, // The (unique) active stack. The jump buffer is invalid in that
40 // state.
41 Suspended, // A stack suspended by WasmSuspend.
42 Inactive, // A parent/ancestor of the active stack. In other words, a
43 // stack that either called or resumed a suspendable stack.
44 Retired // A finished stack. The jump buffer is invalid in that state.
45 };
46 StackState state;
47};

69 return (active_segment_ ? active_segment_->limit_ : limit_) +

73 return active_segment_ ? active_segment_->base()

76 JumpBuffer* jmpbuf() { return &jmpbuf_; }
77 bool Contains(Address addr) {
78 if (!owned_) {
79 return reinterpret_cast<Address>(jslimit()) <= addr && addr < base();
80 }
81 for (auto segment = first_segment_; segment;
82 segment = segment->next_segment_) {
83 if (reinterpret_cast<Address>(segment->limit_) <= addr &&
84 addr < segment->base()) {
85 return true;
86 }
87 if (segment == active_segment_) break;
88 }
89 return false;
90 }
91 int id() { return id_; }
92 bool IsActive() { return jmpbuf_.state == JumpBuffer::Active; }
93 void set_index(size_t index) { index_ = index; }
94 size_t index() { return index_; }
95 size_t allocated_size() {
96 size_t size = 0;
97 auto segment = first_segment_;
98 while (segment) {
99 size += segment->size_;
100 segment = segment->next_segment_;
101 }
102 return size;
103 }
104 void FillWith(uint8_t value) {
105 auto segment = first_segment_;

134 struct StackSwitchInfo {
135 // Source FP and target SP of the frame that switched to the central stack.
136 // The source FP is in the secondary stack, the target SP is in the central
137 // stack.
138 // The stack cannot be suspended while it is on the central stack, so there
139 // can be at most one switch for a given stack.
140 Address source_fp = kNullAddress;
141 Address target_sp = kNullAddress;
142 bool has_value() const { return source_fp != kNullAddress; }
143 };
144 const StackSwitchInfo& stack_switch_info() const {
145 return stack_switch_info_;
146 }
147 void set_stack_switch_info(Address fp, Address sp) {
148 stack_switch_info_ = {fp, sp};
149 }
150 void clear_stack_switch_info() {
151 stack_switch_info_.source_fp = kNullAddress;
152 }
153
154#ifdef DEBUG
155 static constexpr int kJSLimitOffsetKB = 80;
156#else
157 static constexpr int kJSLimitOffsetKB = 40;
158#endif
159
160 friend class StackPool;
161
162 constexpr static uint32_t stack_switch_source_fp_offset() {
163 return OFFSET_OF(StackMemory, stack_switch_info_) +
164 OFFSET_OF(StackMemory::StackSwitchInfo, source_fp);
165 }
166 constexpr static uint32_t stack_switch_target_sp_offset() {
167 return OFFSET_OF(StackMemory, stack_switch_info_) +
168 OFFSET_OF(StackMemory::StackSwitchInfo, target_sp);
169 }
170
171 private:
172 // This constructor allocates a new stack segment.
173 StackMemory();
174
175 // Overload to represent a view of the libc stack.
176 StackMemory(uint8_t* limit, size_t size);
177
178 uint8_t* limit_;
179 size_t size_;
180 bool owned_;
181 JumpBuffer jmpbuf_;
182 // Stable ID.
183 int id_;
184 // Index of this stack in the global Isolate::wasm_stacks() vector. This
185 // allows us to add and remove from the vector in constant time (see
186 // return_switch()).
187 size_t index_;
188 StackSwitchInfo stack_switch_info_;
189 StackSegment* first_segment_ = nullptr;
190 StackSegment* active_segment_ = nullptr;
191};
