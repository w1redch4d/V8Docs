163inline void UpdateComputedInformation(WasmMemory* memory, ModuleOrigin origin) {
164 const uintptr_t platform_max_pages =
165 memory->is_memory64() ? wasm::max_mem64_pages() : wasm::max_mem32_pages();
166 memory->min_memory_size = static_cast<uintptr_t>(std::min<uint64_t>(
167 platform_max_pages, memory->initial_pages)) *
168 kWasmPageSize;
169 memory->max_memory_size = static_cast<uintptr_t>(std::min<uint64_t>(
170 platform_max_pages, memory->maximum_pages)) *
171 kWasmPageSize;
172
173 if (!v8_flags.wasm_bounds_checks) {
174 memory->bounds_checks = kNoBoundsChecks;
175 } else if (v8_flags.wasm_enforce_bounds_checks) {
176 // Explicit bounds checks requested via flag (for testing).
177 memory->bounds_checks = kExplicitBoundsChecks;
178 } else if (origin != kWasmOrigin) {
179 // Asm.js modules can't use trap handling.
180 memory->bounds_checks = kExplicitBoundsChecks;
181 } else if (memory->is_memory64() && !v8_flags.wasm_memory64_trap_handling) {
182 memory->bounds_checks = kExplicitBoundsChecks;
183 } else if (trap_handler::IsTrapHandlerEnabled()) {
184 if constexpr (kSystemPointerSize == 4) UNREACHABLE();
185 memory->bounds_checks = kTrapHandler;
186 } else {
187 // If the trap handler is not enabled, fall back to explicit bounds checks.
188 memory->bounds_checks = kExplicitBoundsChecks;
189 }
190}

321// Uses a map as backing storage when sparsely, or a vector when densely
322// populated. Requires {Value} to implement `bool is_set()` to identify
323// uninitialized objects.
324template <class Value>
325class AdaptiveMap {
326 public:
327 // The technical limitation here is that index+1 must not overflow. Since
328 // we have significantly lower maximums on anything that can be named,
329 // we can have a tighter limit here to reject useless entries early.
330 static constexpr uint32_t kMaxKey = 10'000'000;
331 static_assert(kMaxKey < std::numeric_limits<uint32_t>::max());
332
333 AdaptiveMap() : map_(new MapType()) {}
334
335 explicit AdaptiveMap(const AdaptiveMap&) = delete;
336 AdaptiveMap& operator=(const AdaptiveMap&) = delete;
337
338 AdaptiveMap(AdaptiveMap&& other) V8_NOEXCEPT { *this = std::move(other); }
339
340 AdaptiveMap& operator=(AdaptiveMap&& other) V8_NOEXCEPT {
341 mode_ = other.mode_;
342 vector_.swap(other.vector_);
343 map_.swap(other.map_);
344 return *this;
345 }
346
347 void FinishInitialization();
348
349 bool is_set() const { return mode_ != kInitializing; }
350
351 void Put(uint32_t key, const Value& value) {
352 DCHECK(mode_ == kInitializing);
353 DCHECK_LE(key, kMaxKey);
354 map_->insert(std::make_pair(key, value));
355 }
356
357 void Put(uint32_t key, Value&& value) {
358 DCHECK(mode_ == kInitializing);
359 DCHECK_LE(key, kMaxKey);
360 map_->insert(std::make_pair(key, std::move(value)));
361 }
362
363 const Value* Get(uint32_t key) const {
364 if (mode_ == kDense) {
365 if (key >= vector_.size()) return nullptr;
366 if (!vector_[key].is_set()) return nullptr;
367 return &vector_[key];
368 } else {
369 DCHECK(mode_ == kSparse || mode_ == kInitializing);
370 auto it = map_->find(key);
371 if (it == map_->end()) return nullptr;
372 return &it->second;
373 }
374 }
375
376 bool Has(uint32_t key) const {
377 if (mode_ == kDense) {
378 return key < vector_.size() && vector_[key].is_set();
379 } else {
380 DCHECK(mode_ == kSparse || mode_ == kInitializing);
381 return map_->find(key) != map_->end();
382 }
383 }
384
385 size_t EstimateCurrentMemoryConsumption() const;
386
387 private:
388 static constexpr uint32_t kLoadFactor = 4;
389 using MapType = std::map<uint32_t, Value>;
390 enum Mode { kDense, kSparse, kInitializing };
391
392 Mode mode_{kInitializing};
393 std::vector<Value> vector_;
394 std::unique_ptr<MapType> map_;
395};

519 static CallSiteFeedback CreateMegamorphic() {
520 CallSiteFeedback feedback;
521 feedback.is_megamorphic_ = true;
522 DCHECK(!feedback.is_invalid());
523 DCHECK(!feedback.is_monomorphic());
524 DCHECK(!feedback.is_polymorphic());
525 DCHECK(feedback.is_megamorphic());

529 // Regular constructor: uninitialized/unknown, monomorphic, or
530 // polymorphic.
531 CallSiteFeedback() : index_or_count_(-1), frequency_or_ool_(0) {}
532 CallSiteFeedback(int function_index, int call_count)
533 : index_or_count_(function_index), frequency_or_ool_(call_count) {}
534 CallSiteFeedback(PolymorphicCase* polymorphic_cases, int num_cases)
535 : index_or_count_(-num_cases),
536 frequency_or_ool_(reinterpret_cast<intptr_t>(polymorphic_cases)) {}
537
538 // Copying and assignment: prefer moving, as it's cheaper.
539 // The code below makes sure external polymorphic storage is copied and/or
540 // freed as appropriate.
541 CallSiteFeedback(const CallSiteFeedback& other) V8_NOEXCEPT { *this = other; }
542 CallSiteFeedback(CallSiteFeedback&& other) V8_NOEXCEPT { *this = other; }
543 CallSiteFeedback& operator=(const CallSiteFeedback& other) V8_NOEXCEPT {
544 index_or_count_ = other.index_or_count_;
545 if (other.is_polymorphic()) {
546 int num_cases = other.num_cases();
547 PolymorphicCase* polymorphic = new PolymorphicCase[num_cases];
548 for (int i = 0; i < num_cases; i++) {
549 polymorphic[i].function_index = other.function_index(i);
550 polymorphic[i].absolute_call_frequency = other.call_count(i);
551 }
552 frequency_or_ool_ = reinterpret_cast<intptr_t>(polymorphic);
553 } else {
554 frequency_or_ool_ = other.frequency_or_ool_;
555 }
556 has_non_inlineable_targets_ = other.has_non_inlineable_targets_;
557 is_megamorphic_ = other.is_megamorphic_;
558 return *this;
559 }
560 CallSiteFeedback& operator=(CallSiteFeedback&& other) V8_NOEXCEPT {
561 if (this != &other) {
562 index_or_count_ = other.index_or_count_;
563 frequency_or_ool_ = other.frequency_or_ool_;
564 other.frequency_or_ool_ = 0;
565 }
566 has_non_inlineable_targets_ = other.has_non_inlineable_targets_;
567 is_megamorphic_ = other.is_megamorphic_;
568 return *this;
569 }
570
571 ~CallSiteFeedback() {
572 if (is_polymorphic()) delete[] polymorphic_storage();
573 }
574
575 int num_cases() const {
576 if (is_monomorphic()) return 1;
577 if (is_invalid() || is_megamorphic()) return 0;
578 return -index_or_count_;
579 }
580 int function_index(int i) const {
581 DCHECK(!is_invalid() && !is_megamorphic());
582 if (is_monomorphic()) return index_or_count_;
583 return polymorphic_storage()[i].function_index;
584 }
585 int call_count(int i) const {
586 DCHECK(!is_invalid() && !is_megamorphic());
587 if (is_monomorphic()) return static_cast<int>(frequency_or_ool_);
588 return polymorphic_storage()[i].absolute_call_frequency;
589 }
590 bool has_non_inlineable_targets() const {
591 return has_non_inlineable_targets_;
592 }
593 void set_has_non_inlineable_targets(bool has_non_inlineable_targets) {
594 has_non_inlineable_targets_ = has_non_inlineable_targets;
595 }
596
597 bool is_megamorphic() const { return is_megamorphic_; }
598
599 private:
600 bool is_monomorphic() const { return index_or_count_ >= 0; }
601 bool is_polymorphic() const { return index_or_count_ <= -2; }
602 bool is_invalid() const { return index_or_count_ == -1 && !is_megamorphic_; }
603 const PolymorphicCase* polymorphic_storage() const {
604 DCHECK(is_polymorphic());
605 return reinterpret_cast<PolymorphicCase*>(frequency_or_ool_);
606 }
607
608 int index_or_count_;
609 bool has_non_inlineable_targets_ = false;
610 bool is_megamorphic_ = false;
611 intptr_t frequency_or_ool_;
612};
