47bool Scavenger::MigrateObject(Tagged<Map> map, Tagged<HeapObject> source,
48 Tagged<HeapObject> target, int size,
49 PromotionHeapChoice promotion_heap_choice) {
50 // This CAS can be relaxed because we do not access the object body if the
51 // object was already copied by another thread. We only access the page header
52 // of such objects and this is safe because of the memory fence after page
53 // header initialization.
54 if (!source->relaxed_compare_and_swap_map_word_forwarded(
55 MapWord::FromMap(map), target)) {
56 // Other task migrated the object.
57 return false;
58 }
59
60 // Copy the content of source to target. Note that we do this on purpose
61 // *after* the CAS. This avoids copying of the object in the (unlikely)
62 // failure case. It also helps us to ensure that we do not rely on non-relaxed
63 // memory ordering for the CAS above.
64 target->set_map_word(map, kRelaxedStore);
65 heap()->CopyBlock(target.address() + kTaggedSize,
66 source.address() + kTaggedSize, size - kTaggedSize);
67
68 if (V8_UNLIKELY(is_logging_)) {
69 heap()->OnMoveEvent(source, target, size);
70 }
71
72 if (is_incremental_marking_ &&
73 (promotion_heap_choice != kPromoteIntoSharedHeap || mark_shared_heap_)) {
74 heap()->incremental_marking()->TransferColor(source, target);
75 }
76 PretenuringHandler::UpdateAllocationSite(heap_, map, source, size,
77 &local_pretenuring_feedback_);
78
79 return true;
80}

82template <typename THeapObjectSlot>
83CopyAndForwardResult Scavenger::SemiSpaceCopyObject(
84 Tagged<Map> map, THeapObjectSlot slot, Tagged<HeapObject> object,
85 int object_size, ObjectFields object_fields) {
86 static_assert(std::is_same<THeapObjectSlot, FullHeapObjectSlot>::value ||
87 std::is_same<THeapObjectSlot, HeapObjectSlot>::value,
88 "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
89 DCHECK(heap()->AllowedToBeMigrated(map, object, NEW_SPACE));
90 AllocationAlignment alignment = HeapObject::RequiredAlignment(map);
91 AllocationResult allocation =
92 allocator_.Allocate(NEW_SPACE, object_size, alignment);
93
94 Tagged<HeapObject> target;
95 if (allocation.To(&target)) {
96 DCHECK(heap()->marking_state()->IsUnmarked(target));
97 const bool self_success =
98 MigrateObject(map, object, target, object_size, kPromoteIntoLocalHeap);
99 if (!self_success) {
100 allocator_.FreeLast(NEW_SPACE, target, object_size);
101 MapWord map_word = object->map_word(kRelaxedLoad);
102 UpdateHeapObjectReferenceSlot(slot, map_word.ToForwardingAddress(object));
103 SynchronizePageAccess(*slot);
104 DCHECK(!Heap::InFromPage(*slot));
105 return Heap::InToPage(*slot)
106 ? CopyAndForwardResult::SUCCESS_YOUNG_GENERATION
107 : CopyAndForwardResult::SUCCESS_OLD_GENERATION;
108 }
109 UpdateHeapObjectReferenceSlot(slot, target);
110 if (object_fields == ObjectFields::kMaybePointers) {
111 local_copied_list_.Push(target);
112 }
113 copied_size_ += object_size;
114 return CopyAndForwardResult::SUCCESS_YOUNG_GENERATION;
115 }
116 return CopyAndForwardResult::FAILURE;
117}

119template <typename THeapObjectSlot,
120 Scavenger::PromotionHeapChoice promotion_heap_choice>
121CopyAndForwardResult Scavenger::PromoteObject(Tagged<Map> map,
122 THeapObjectSlot slot,
123 Tagged<HeapObject> object,
124 int object_size,
125 ObjectFields object_fields) {
126 static_assert(std::is_same<THeapObjectSlot, FullHeapObjectSlot>::value ||
127 std::is_same<THeapObjectSlot, HeapObjectSlot>::value,
128 "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
129 DCHECK_GE(object_size, Heap::kMinObjectSizeInTaggedWords * kTaggedSize);
130 AllocationAlignment alignment = HeapObject::RequiredAlignment(map);
131 AllocationResult allocation = allocator_.Allocate(
132 promotion_heap_choice == kPromoteIntoLocalHeap ? OLD_SPACE : SHARED_SPACE,
133 object_size, alignment);
134
135 Tagged<HeapObject> target;
136 if (allocation.To(&target)) {
137 DCHECK(heap()->non_atomic_marking_state()->IsUnmarked(target));
138 const bool self_success =
139 MigrateObject(map, object, target, object_size, promotion_heap_choice);
140 if (!self_success) {
141 allocator_.FreeLast(promotion_heap_choice == kPromoteIntoLocalHeap
142 ? OLD_SPACE
143 : SHARED_SPACE,
144 target, object_size);
145
146 MapWord map_word = object->map_word(kRelaxedLoad);
147 UpdateHeapObjectReferenceSlot(slot, map_word.ToForwardingAddress(object));
148 SynchronizePageAccess(*slot);
149 DCHECK(!Heap::InFromPage(*slot));
150 return Heap::InToPage(*slot)
151 ? CopyAndForwardResult::SUCCESS_YOUNG_GENERATION
152 : CopyAndForwardResult::SUCCESS_OLD_GENERATION;
153 }
154 UpdateHeapObjectReferenceSlot(slot, target);
155
156 // During incremental marking we want to push every object in order to
157 // record slots for map words. Necessary for map space compaction.
158 if (object_fields == ObjectFields::kMaybePointers || is_compacting_) {
159 local_promoted_list_.Push({target, map, object_size});
160 }
161 promoted_size_ += object_size;
162 return CopyAndForwardResult::SUCCESS_OLD_GENERATION;
163 }
164 return CopyAndForwardResult::FAILURE;
165}

174bool Scavenger::HandleLargeObject(Tagged<Map> map, Tagged<HeapObject> object,
175 int object_size, ObjectFields object_fields) {
176 if (NEW_LO_SPACE ==
177 MutablePageMetadata::FromHeapObject(object)->owner_identity()) {
178 DCHECK(MemoryChunk::FromHeapObject(object)->InNewLargeObjectSpace());
179 if (object->relaxed_compare_and_swap_map_word_forwarded(
180 MapWord::FromMap(map), object)) {
181 local_surviving_new_large_objects_.insert({object, map});
182 promoted_size_ += object_size;
183 if (object_fields == ObjectFields::kMaybePointers) {
184 local_promoted_list_.Push({object, map, object_size});
185 }
186 }
187 return true;
188 }
189 return false;
190}

192template <typename THeapObjectSlot,
193 Scavenger::PromotionHeapChoice promotion_heap_choice>
194SlotCallbackResult Scavenger::EvacuateObjectDefault(
195 Tagged<Map> map, THeapObjectSlot slot, Tagged<HeapObject> object,
196 int object_size, ObjectFields object_fields) {
197 static_assert(std::is_same<THeapObjectSlot, FullHeapObjectSlot>::value ||
198 std::is_same<THeapObjectSlot, HeapObjectSlot>::value,
199 "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
200 SLOW_DCHECK(object->SizeFromMap(map) == object_size);
201 CopyAndForwardResult result;
202
203 if (HandleLargeObject(map, object, object_size, object_fields)) {
204 return KEEP_SLOT;
205 }
206
207 SLOW_DCHECK(static_cast<size_t>(object_size) <=
208 MemoryChunkLayout::AllocatableMemoryInDataPage());
209
210 if (!SemiSpaceNewSpace::From(heap()->new_space())
211 ->ShouldBePromoted(object.address())) {
212 // A semi-space copy may fail due to fragmentation. In that case, we
213 // try to promote the object.
214 result = SemiSpaceCopyObject(map, slot, object, object_size, object_fields);
215 if (result != CopyAndForwardResult::FAILURE) {
216 return RememberedSetEntryNeeded(result);
217 }
218 }
219
220 // We may want to promote this object if the object was already semi-space
221 // copied in a previous young generation GC or if the semi-space copy above
222 // failed.
223 result = PromoteObject<THeapObjectSlot, promotion_heap_choice>(
224 map, slot, object, object_size, object_fields);
225 if (result != CopyAndForwardResult::FAILURE) {
226 return RememberedSetEntryNeeded(result);
227 }
228
229 // If promotion failed, we try to copy the object to the other semi-space.
230 result = SemiSpaceCopyObject(map, slot, object, object_size, object_fields);
231 if (result != CopyAndForwardResult::FAILURE) {
232 return RememberedSetEntryNeeded(result);
233 }
234
235 heap()->FatalProcessOutOfMemory("Scavenger: semi-space copy");
236 UNREACHABLE();
237}

239template <typename THeapObjectSlot>
240SlotCallbackResult Scavenger::EvacuateThinString(Tagged<Map> map,
241 THeapObjectSlot slot,
242 Tagged<ThinString> object,
243 int object_size) {
244 static_assert(std::is_same<THeapObjectSlot, FullHeapObjectSlot>::value ||
245 std::is_same<THeapObjectSlot, HeapObjectSlot>::value,
246 "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
247 if (shortcut_strings_) {
248 // The ThinString should die after Scavenge, so avoid writing the proper
249 // forwarding pointer and instead just signal the actual object as forwarded
250 // reference.
251 Tagged<String> actual = object->actual();
252 // ThinStrings always refer to internalized strings, which are always in old
253 // space.
254 DCHECK(!HeapLayout::InYoungGeneration(actual));
255 UpdateHeapObjectReferenceSlot(slot, actual);
256 return REMOVE_SLOT;
257 }
258
259 DCHECK_EQ(ObjectFields::kMaybePointers,
260 Map::ObjectFieldsFrom(map->visitor_id()));
261 return EvacuateObjectDefault(map, slot, object, object_size,
262 ObjectFields::kMaybePointers);
263}

265template <typename THeapObjectSlot>
266SlotCallbackResult Scavenger::EvacuateShortcutCandidate(
267 Tagged<Map> map, THeapObjectSlot slot, Tagged<ConsString> object,
268 int object_size) {
269 static_assert(std::is_same<THeapObjectSlot, FullHeapObjectSlot>::value ||
270 std::is_same<THeapObjectSlot, HeapObjectSlot>::value,
271 "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
272 DCHECK(IsShortcutCandidate(map->instance_type()));
273
274 if (shortcut_strings_ &&
275 object->unchecked_second() == ReadOnlyRoots(heap()).empty_string()) {
276 Tagged<HeapObject> first = Cast<HeapObject>(object->unchecked_first());
277
278 UpdateHeapObjectReferenceSlot(slot, first);
279
280 if (!HeapLayout::InYoungGeneration(first)) {
281 object->set_map_word_forwarded(first, kRelaxedStore);
282 return REMOVE_SLOT;
283 }
284
285 MapWord first_word = first->map_word(kRelaxedLoad);
286 if (first_word.IsForwardingAddress()) {
287 Tagged<HeapObject> target = first_word.ToForwardingAddress(first);
288
289 UpdateHeapObjectReferenceSlot(slot, target);
290 SynchronizePageAccess(target);
291 object->set_map_word_forwarded(target, kRelaxedStore);
292 return HeapLayout::InYoungGeneration(target) ? KEEP_SLOT : REMOVE_SLOT;
293 }
294 Tagged<Map> first_map = first_word.ToMap();
295 SlotCallbackResult result = EvacuateObjectDefault(
296 first_map, slot, first, first->SizeFromMap(first_map),
297 Map::ObjectFieldsFrom(first_map->visitor_id()));
298 object->set_map_word_forwarded(slot.ToHeapObject(), kRelaxedStore);
299 return result;
300 }
301 DCHECK_EQ(ObjectFields::kMaybePointers,
302 Map::ObjectFieldsFrom(map->visitor_id()));
303 return EvacuateObjectDefault(map, slot, object, object_size,
304 ObjectFields::kMaybePointers);
305}

320template <typename THeapObjectSlot>
321SlotCallbackResult Scavenger::EvacuateObject(THeapObjectSlot slot,
322 Tagged<Map> map,
323 Tagged<HeapObject> source) {
324 static_assert(std::is_same<THeapObjectSlot, FullHeapObjectSlot>::value ||
325 std::is_same<THeapObjectSlot, HeapObjectSlot>::value,
326 "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
327 SLOW_DCHECK(Heap::InFromPage(source));
328 SLOW_DCHECK(!MapWord::FromMap(map).IsForwardingAddress());
329 int size = source->SizeFromMap(map);
330 // Cannot use ::cast() below because that would add checks in debug mode
331 // that require re-reading the map.
332 VisitorId visitor_id = map->visitor_id();
333 switch (visitor_id) {
334 case kVisitThinString:
335 // At the moment we don't allow weak pointers to thin strings.
336 DCHECK(!(*slot).IsWeak());
337 return EvacuateThinString(map, slot, UncheckedCast<ThinString>(source),
338 size);
339 case kVisitShortcutCandidate:
340 DCHECK(!(*slot).IsWeak());
341 // At the moment we don't allow weak pointers to cons strings.
342 return EvacuateShortcutCandidate(map, slot,
343 UncheckedCast<ConsString>(source), size);
344 case kVisitSeqOneByteString:
345 case kVisitSeqTwoByteString:
346 DCHECK(String::IsInPlaceInternalizable(map->instance_type()));
347 static_assert(Map::ObjectFieldsFrom(kVisitSeqOneByteString) ==
348 Map::ObjectFieldsFrom(kVisitSeqTwoByteString));
349 return EvacuateInPlaceInternalizableString(
350 map, slot, UncheckedCast<String>(source), size,
351 Map::ObjectFieldsFrom(kVisitSeqOneByteString));
352 default:
353 return EvacuateObjectDefault(map, slot, source, size,
354 Map::ObjectFieldsFrom(visitor_id));
355 }
356}

358template <typename THeapObjectSlot>
359SlotCallbackResult Scavenger::ScavengeObject(THeapObjectSlot p,
360 Tagged<HeapObject> object) {
361 static_assert(std::is_same<THeapObjectSlot, FullHeapObjectSlot>::value ||
362 std::is_same<THeapObjectSlot, HeapObjectSlot>::value,
363 "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
364 DCHECK(Heap::InFromPage(object));
365
366 // Check whether object was already successfully forwarded by the CAS in
367 // MigrateObject. No memory ordering required because we only access the page
368 // header of a relocated object. Page header initialization uses a memory
369 // fence.
370 MapWord first_word = object->map_word(kRelaxedLoad);
371
372 // If the first word is a forwarding address, the object has already been
373 // copied.
374 if (first_word.IsForwardingAddress()) {
375 Tagged<HeapObject> dest = first_word.ToForwardingAddress(object);
376 UpdateHeapObjectReferenceSlot(p, dest);
377 SynchronizePageAccess(dest);
378 // A forwarded object in new space is either in the second (to) semi space,
379 // a large object, or a pinned object on a quarantined page.
380 // Pinned objects have a self forwarding map word. However, since forwarding
381 // addresses are set with relaxed atomics and before the object is actually
382 // copied, it is unfortunately not safe to access `dest` to check whether it
383 // is pinned or not.
384 DCHECK_IMPLIES(HeapLayout::InYoungGeneration(dest),
385 Heap::InToPage(dest) || Heap::IsLargeObject(dest) ||
386 MemoryChunk::FromHeapObject(dest)->IsQuarantined());
387
388 // This load forces us to have memory ordering for the map load above. We
389 // need to have the page header properly initialized.
390 return HeapLayout::InYoungGeneration(dest) ? KEEP_SLOT : REMOVE_SLOT;
391 }
392
393 Tagged<Map> map = first_word.ToMap();
394 // AllocationMementos are unrooted and shouldn't survive a scavenge
395 DCHECK_NE(ReadOnlyRoots(heap()).allocation_memento_map(), map);
396 // Call the slow part of scavenge object.
397 return EvacuateObject(p, map, object);
398}

400template <typename TSlot>
401SlotCallbackResult Scavenger::CheckAndScavengeObject(Heap* heap, TSlot slot) {
402 static_assert(
403 std::is_same<TSlot, FullMaybeObjectSlot>::value ||
404 std::is_same<TSlot, MaybeObjectSlot>::value,
405 "Only FullMaybeObjectSlot and MaybeObjectSlot are expected here");
406 using THeapObjectSlot = typename TSlot::THeapObjectSlot;
407 Tagged<MaybeObject> object = *slot;
408 if (Heap::InFromPage(object)) {
409 Tagged<HeapObject> heap_object = object.GetHeapObject();
410
411 SlotCallbackResult result =
412 ScavengeObject(THeapObjectSlot(slot), heap_object);
413 DCHECK_IMPLIES(result == REMOVE_SLOT,
414 !HeapLayout::InYoungGeneration((*slot).GetHeapObject()));
415 return result;
416 } else if (Heap::InToPage(object)) {
417 // Already updated slot. This can happen when processing of the work list
418 // is interleaved with processing roots.
419 return KEEP_SLOT;
420 }
421 // Slots can point to "to" space if the slot has been recorded multiple
422 // times in the remembered set. We remove the redundant slot now.
423 return REMOVE_SLOT;
424}

523void ScavengeVisitor::VisitExternalPointer(Tagged<HeapObject> host,
524 ExternalPointerSlot slot) {
525#ifdef V8_COMPRESS_POINTERS
526 DCHECK(!slot.tag_range().IsEmpty());
527 DCHECK(!IsSharedExternalPointerType(slot.tag_range()));
528 DCHECK(HeapLayout::InYoungGeneration(host));
529
530 // If an incremental mark is in progress, there is already a whole-heap trace
531 // running that will mark live EPT entries, and the scavenger won't sweep the
532 // young EPT space. So, leave the tracing and sweeping work to the impending
533 // major GC.
534 //
535 // The EPT entry may or may not be marked already by the incremental marker.
536 if (scavenger_->is_incremental_marking_) return;
537
538 // TODO(chromium:337580006): Remove when pointer compression always uses
539 // EPT.
540 if (!slot.HasExternalPointerHandle()) return;
541
542 ExternalPointerHandle handle = slot.Relaxed_LoadHandle();
543 Heap* heap = scavenger_->heap();
544 ExternalPointerTable& table = heap->isolate()->external_pointer_table();
545 table.Mark(heap->young_external_pointer_space(), handle, slot.address());
546#endif // V8_COMPRESS_POINTERS
547}

549size_t ScavengeVisitor::VisitEphemeronHashTable(
550 Tagged<Map> map, Tagged<EphemeronHashTable> table, MaybeObjectSize) {
551 // Register table with the scavenger, so it can take care of the weak keys
552 // later. This allows to only iterate the tables' values, which are treated
553 // as strong independently of whether the key is live.
554 scavenger_->AddEphemeronHashTable(table);
555 for (InternalIndex i : table->IterateEntries()) {
556 ObjectSlot value_slot =
557 table->RawFieldOfElementAt(EphemeronHashTable::EntryToValueIndex(i));
558 VisitPointer(table, value_slot);
559 }
560
561 return table->SizeFromMap(map);
562}
