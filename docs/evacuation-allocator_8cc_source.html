12EvacuationAllocator::EvacuationAllocator(
13 Heap* heap, CompactionSpaceKind compaction_space_kind)
14 : heap_(heap),
15 new_space_(heap->new_space()),
16 compaction_spaces_(heap, compaction_space_kind) {
17 if (new_space_) {
18 DCHECK(!heap_->allocator()->new_space_allocator()->IsLabValid());
19 new_space_allocator_.emplace(heap, new_space_, MainAllocator::kInGC);
20 }
21
22 old_space_allocator_.emplace(heap, compaction_spaces_.Get(OLD_SPACE),
23 MainAllocator::kInGC);
24 code_space_allocator_.emplace(heap, compaction_spaces_.Get(CODE_SPACE),
25 MainAllocator::kInGC);
26 if (heap_->isolate()->has_shared_space()) {
27 shared_space_allocator_.emplace(heap, compaction_spaces_.Get(SHARED_SPACE),
28 MainAllocator::kInGC);
29 }
30 trusted_space_allocator_.emplace(heap, compaction_spaces_.Get(TRUSTED_SPACE),
31 MainAllocator::kInGC);
32}

34void EvacuationAllocator::FreeLast(AllocationSpace space,
35 Tagged<HeapObject> object, int object_size) {
36 DCHECK_IMPLIES(!shared_space_allocator_, space != SHARED_SPACE);
37 object_size = ALIGN_TO_ALLOCATION_ALIGNMENT(object_size);
38 switch (space) {
39 case NEW_SPACE:
40 FreeLastInMainAllocator(new_space_allocator(), object, object_size);
41 return;
42 case OLD_SPACE:
43 FreeLastInMainAllocator(old_space_allocator(), object, object_size);
44 return;
45 case SHARED_SPACE:
46 FreeLastInMainAllocator(shared_space_allocator(), object, object_size);
47 return;
48 default:
49 // Only new and old space supported.
50 UNREACHABLE();
51 }
52}

54void EvacuationAllocator::FreeLastInMainAllocator(MainAllocator* allocator,
55 Tagged<HeapObject> object,
56 int object_size) {
57 if (!allocator->TryFreeLast(object.address(), object_size)) {
58 // We couldn't free the last object so we have to write a proper filler.
59 heap_->CreateFillerObjectAt(object.address(), object_size);
60 }
61}

63void EvacuationAllocator::Finalize() {
64 if (new_space_) {
65 new_space_allocator()->FreeLinearAllocationArea();
66 }
67
68 old_space_allocator()->FreeLinearAllocationArea();
69 heap_->old_space()->MergeCompactionSpace(compaction_spaces_.Get(OLD_SPACE));
70
71 code_space_allocator()->FreeLinearAllocationArea();
72 heap_->code_space()->MergeCompactionSpace(compaction_spaces_.Get(CODE_SPACE));
73
74 if (shared_space_allocator_) {
75 shared_space_allocator_->FreeLinearAllocationArea();
76 heap_->shared_allocation_space()->MergeCompactionSpace(
77 compaction_spaces_.Get(SHARED_SPACE));
78 }
79
80 trusted_space_allocator()->FreeLinearAllocationArea();
81 heap_->trusted_space()->MergeCompactionSpace(
82 compaction_spaces_.Get(TRUSTED_SPACE));
83}
