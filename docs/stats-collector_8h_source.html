236 // Observer for allocated object size. May e.g. be used to implement heap
237 // growing heuristics. Observers may register/unregister observers at any
238 // time when being invoked.
239 class AllocationObserver {
240 public:
241 // Called after observing at least
242 // StatsCollector::kAllocationThresholdBytes changed bytes through
243 // allocation or explicit free. Reports both, negative and positive
244 // increments, to allow observer to decide whether absolute values or only
245 // the deltas is interesting.
246 //
247 // May trigger GC.
248 virtual void AllocatedObjectSizeIncreased(size_t) {}
249 virtual void AllocatedObjectSizeDecreased(size_t) {}
250
251 // Called when the exact size of allocated object size is known. In
252 // practice, this is after marking when marked bytes == allocated bytes.
253 //
254 // Must not trigger GC synchronously.
255 virtual void ResetAllocatedObjectSize(size_t) {}
256
257 // Called upon allocating/releasing chunks of memory (e.g. pages) that can
258 // contain objects.
259 //
260 // Must not trigger GC.
261 virtual void AllocatedSizeIncreased(size_t) {}
262 virtual void AllocatedSizeDecreased(size_t) {}
263 };
264

266 // reasonably interesting sizes.

396template <typename Callback>
397void StatsCollector::ForAllAllocationObservers(Callback callback) {
398 // Iterate using indices to allow push_back() of new observers.
399 for (size_t i = 0; i < allocation_observers_.size(); ++i) {
400 auto* observer = allocation_observers_[i];
401 if (observer) {
402 callback(observer);
403 }
404 }
405 if (allocation_observer_deleted_) {
406 allocation_observers_.erase(
407 std::remove(allocation_observers_.begin(), allocation_observers_.end(),
408 nullptr),
409 allocation_observers_.end());
410 allocation_observer_deleted_ = false;
411 }

490template <StatsCollector::TraceCategory trace_category,
491 StatsCollector::ScopeContext scope_category>
492void StatsCollector::InternalScope<trace_category,
493 scope_category>::IncreaseScopeTime() {
494 DCHECK_NE(GarbageCollectionState::kNotRunning, stats_collector_->gc_state_);
495 // Only record top level scopes.
496 if (static_cast<int>(scope_id_) >=
497 (scope_category == kMutatorThread
498 ? static_cast<int>(kNumHistogramScopeIds)
499 : static_cast<int>(kNumHistogramConcurrentScopeIds)))
500 return;
501 v8::base::TimeDelta time = v8::base::TimeTicks::Now() - start_time_;
502 if (scope_category == StatsCollector::ScopeContext::kMutatorThread) {
503 stats_collector_->current_.scope_data[scope_id_] += time;
504 if (stats_collector_->metric_recorder_)
505 stats_collector_->RecordHistogramSample(scope_id_, time);
506 return;
507 }
508 // scope_category == StatsCollector::ScopeContext::kConcurrentThread
509 using AtomicWord = v8::base::AtomicWord;
510 const int64_t us = time.InMicroseconds();
511 v8::base::Relaxed_AtomicIncrement(
512 &stats_collector_->current_.concurrent_scope_data[scope_id_],
513 static_cast<AtomicWord>(us));
514}
