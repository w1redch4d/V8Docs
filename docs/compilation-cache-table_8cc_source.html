16const int kLiteralEntryLength = 2;
17const int kLiteralInitialLength = 2;
18const int kLiteralContextOffset = 0;
19const int kLiteralLiteralsOffset = 1;

21int SearchLiteralsMapEntry(Tagged<CompilationCacheTable> cache,
22 InternalIndex cache_entry,
23 Tagged<Context> native_context) {
24 DisallowGarbageCollection no_gc;
25 DCHECK(IsNativeContext(native_context));
26 Tagged<Object> obj = cache->EvalFeedbackValueAt(cache_entry);
27
28 // Check that there's no confusion between FixedArray and WeakFixedArray (the
29 // object used to be a FixedArray here).
30 DCHECK(!IsFixedArray(obj));
31 if (IsWeakFixedArray(obj)) {
32 Tagged<WeakFixedArray> literals_map = Cast<WeakFixedArray>(obj);
33 int length = literals_map->length();
34 for (int i = 0; i < length; i += kLiteralEntryLength) {
35 DCHECK(literals_map->get(i + kLiteralContextOffset).IsWeakOrCleared());
36 if (literals_map->get(i + kLiteralContextOffset) ==
37 MakeWeak(native_context)) {
38 return i;
39 }
40 }
41 }
42 return -1;
43}

45void AddToFeedbackCellsMap(DirectHandle<CompilationCacheTable> cache,
46 InternalIndex cache_entry,
47 DirectHandle<Context> native_context,
48 DirectHandle<FeedbackCell> feedback_cell) {
49 Isolate* isolate = native_context->GetIsolate();
50 DCHECK(IsNativeContext(*native_context));
51 static_assert(kLiteralEntryLength == 2);
52 DirectHandle<WeakFixedArray> new_literals_map;
53 int entry;
54
55 Tagged<Object> obj = cache->EvalFeedbackValueAt(cache_entry);
56
57 // Check that there's no confusion between FixedArray and WeakFixedArray (the
58 // object used to be a FixedArray here).
59 DCHECK(!IsFixedArray(obj));
60 if (!IsWeakFixedArray(obj) || Cast<WeakFixedArray>(obj)->length() == 0) {
61 new_literals_map = isolate->factory()->NewWeakFixedArray(
62 kLiteralInitialLength, AllocationType::kOld);
63 entry = 0;
64 } else {
65 DirectHandle<WeakFixedArray> old_literals_map(Cast<WeakFixedArray>(obj),
66 isolate);
67 entry = SearchLiteralsMapEntry(*cache, cache_entry, *native_context);
68 if (entry >= 0) {
69 // Just set the code of the entry.
70 old_literals_map->set(entry + kLiteralLiteralsOffset,
71 MakeWeak(*feedback_cell));
72 return;
73 }
74
75 // Can we reuse an entry?
76 DCHECK_LT(entry, 0);
77 int length = old_literals_map->length();
78 for (int i = 0; i < length; i += kLiteralEntryLength) {
79 if (old_literals_map->get(i + kLiteralContextOffset).IsCleared()) {
80 new_literals_map = old_literals_map;
81 entry = i;
82 break;
83 }
84 }
85
86 if (entry < 0) {
87 // Copy old optimized code map and append one new entry.
88 new_literals_map = isolate->factory()->CopyWeakFixedArrayAndGrow(
89 old_literals_map, kLiteralEntryLength);
90 entry = old_literals_map->length();
91 }
92 }
93
94 new_literals_map->set(entry + kLiteralContextOffset,
95 MakeWeak(*native_context));
96 new_literals_map->set(entry + kLiteralLiteralsOffset,
97 MakeWeak(*feedback_cell));
98
99#ifdef DEBUG
100 for (int i = 0; i < new_literals_map->length(); i += kLiteralEntryLength) {
101 Tagged<MaybeObject> object =
102 new_literals_map->get(i + kLiteralContextOffset);
103 DCHECK(object.IsCleared() ||
104 IsNativeContext(object.GetHeapObjectAssumeWeak()));
105 object = new_literals_map->get(i + kLiteralLiteralsOffset);
106 DCHECK(object.IsCleared() ||
107 IsFeedbackCell(object.GetHeapObjectAssumeWeak()));
108 }
109#endif
110
111 Tagged<Object> old_literals_map = cache->EvalFeedbackValueAt(cache_entry);
112 if (old_literals_map != *new_literals_map) {
113 cache->SetEvalFeedbackValueAt(cache_entry, *new_literals_map);
114 }
115}

117Tagged<FeedbackCell> SearchLiteralsMap(Tagged<CompilationCacheTable> cache,
118 InternalIndex cache_entry,
119 Tagged<Context> native_context) {
120 Tagged<FeedbackCell> result;
121 int entry = SearchLiteralsMapEntry(cache, cache_entry, native_context);
122 if (entry >= 0) {
123 Tagged<WeakFixedArray> literals_map =
124 Cast<WeakFixedArray>(cache->EvalFeedbackValueAt(cache_entry));
125 DCHECK_LE(entry + kLiteralEntryLength, literals_map->length());
126 Tagged<MaybeObject> object =
127 literals_map->get(entry + kLiteralLiteralsOffset);
128
129 if (!object.IsCleared()) {
130 result = Cast<FeedbackCell>(object.GetHeapObjectAssumeWeak());
131 }
132 }
133 DCHECK(result.is_null() || IsFeedbackCell(result));
134 return result;
135}

137// EvalCacheKeys are used as keys in the eval cache.
138class EvalCacheKey : public HashTableKey {
139 public:
140 // This tuple unambiguously identifies calls to eval() or
141 // CreateDynamicFunction() (such as through the Function() constructor).
142 // * source is the string passed into eval(). For dynamic functions, this is
143 // the effective source for the function, some of which is implicitly
144 // generated.
145 // * shared is the shared function info for the function containing the call
146 // to eval(). for dynamic functions, shared is the native context closure.
147 // * When positive, position is the position in the source where eval is
148 // called. When negative, position is the negation of the position in the
149 // dynamic function's effective source where the ')' ends the parameters.
150 EvalCacheKey(Handle<String> source, Handle<SharedFunctionInfo> shared,
151 LanguageMode language_mode, int position)
152 : HashTableKey(CompilationCacheShape::EvalHash(*source, *shared,
153 language_mode, position)),
154 source_(source),
155 shared_(shared),
156 language_mode_(language_mode),
157 position_(position) {}
158
159 bool IsMatch(Tagged<Object> other) override {
160 DisallowGarbageCollection no_gc;
161 if (!IsFixedArray(other)) {
162 DCHECK(IsNumber(other));
163 uint32_t other_hash = static_cast<uint32_t>(Object::NumberValue(other));
164 return Hash() == other_hash;
165 }
166 Tagged<FixedArray> other_array = Cast<FixedArray>(other);
167 DCHECK(IsSharedFunctionInfo(other_array->get(0)));
168 if (*shared_ != other_array->get(0)) return false;
169 int language_unchecked = Smi::ToInt(other_array->get(2));
170 DCHECK(is_valid_language_mode(language_unchecked));
171 LanguageMode language_mode = static_cast<LanguageMode>(language_unchecked);
172 if (language_mode != language_mode_) return false;
173 int position = Smi::ToInt(other_array->get(3));
174 if (position != position_) return false;
175 Tagged<String> source = Cast<String>(other_array->get(1));
176 return source->Equals(*source_);
177 }
178
179 Handle<Object> AsHandle(Isolate* isolate) {
180 Handle<FixedArray> array = isolate->factory()->NewFixedArray(4);
181 array->set(0, *shared_);
182 array->set(1, *source_);
183 array->set(2, Smi::FromEnum(language_mode_));
184 array->set(3, Smi::FromInt(position_));
185 array->set_map(isolate, ReadOnlyRoots(isolate).fixed_cow_array_map());
186 return array;
187 }
188
189 private:
190 Handle<String> source_;
191 Handle<SharedFunctionInfo> shared_;
192 LanguageMode language_mode_;
193 int position_;
194};

234Tagged<Smi> ScriptHash(Tagged<String> source,
235 MaybeDirectHandle<Object> maybe_name, int line_offset,
236 int column_offset,
237 v8::ScriptOriginOptions origin_options,
238 Isolate* isolate) {
239 DisallowGarbageCollection no_gc;
240 size_t hash = base::hash_combine(source->EnsureHash());
241 if (DirectHandle<Object> name;
242 maybe_name.ToHandle(&name) && IsString(*name, isolate)) {
243 hash =
244 base::hash_combine(hash, Cast<String>(*name)->EnsureHash(), line_offset,
245 column_offset, origin_options.Flags());
246 }
247 // The upper bits of the hash are discarded so that the value fits in a Smi.
248 return Smi::From31BitPattern(static_cast<int>(hash & (~(1u << 31))));
249}

253// We only re-use a cached function for some script source code if the
254// script originates from the same place. This is to avoid issues
255// when reporting errors, etc.
256bool ScriptCacheKey::MatchesScript(Tagged<Script> script) {
257 DisallowGarbageCollection no_gc;
258
259 // If the script name isn't set, the boilerplate script should have
260 // an undefined name to have the same origin.
261 Handle<Object> name;
262 if (!name_.ToHandle(&name)) {
263 return IsUndefined(script->name(), isolate_);
264 }
265 // Do the fast bailout checks first.
266 if (line_offset_ != script->line_offset()) return false;
267 if (column_offset_ != script->column_offset()) return false;
268 // Check that both names are strings. If not, no match.
269 if (!IsString(*name, isolate_) || !IsString(script->name(), isolate_))
270 return false;
271 // Are the origin_options same?
272 if (origin_options_.Flags() != script->origin_options().Flags()) {
273 return false;
274 }
275 // Compare the two name strings for equality.
276 if (!Cast<String>(*name)->Equals(Cast<String>(script->name()))) {
277 return false;
278 }
279
280 Handle<FixedArray> wrapped_arguments_handle;
281 if (wrapped_arguments_.ToHandle(&wrapped_arguments_handle)) {
282 if (!script->is_wrapped()) {
283 return false;
284 }
285 Tagged<FixedArray> wrapped_arguments = *wrapped_arguments_handle;
286 Tagged<FixedArray> other_wrapped_arguments = script->wrapped_arguments();
287 int length = wrapped_arguments->length();
288 if (length != other_wrapped_arguments->length()) {
289 return false;
290 }
291 for (int i = 0; i < length; i++) {
292 Tagged<Object> arg = wrapped_arguments->get(i);
293 Tagged<Object> other_arg = other_wrapped_arguments->get(i);
294 DCHECK(IsString(arg));
295 DCHECK(IsString(other_arg));
296 if (!Cast<String>(arg)->Equals(Cast<String>(other_arg))) {
297 return false;
298 }
299 }
300 } else if (script->is_wrapped()) {
301 return false;
302 }
303
304 // Don't compare host options if the script was deserialized because we didn't
305 // serialize host options (see CodeSerializer::SerializeObjectImpl())
306 if (script->deserialized() &&
307 script->host_defined_options() ==
308 ReadOnlyRoots(isolate_).empty_fixed_array()) {
309 return true;
310 }
311 // TODO(cbruni, chromium:1244145): Remove once migrated to the context
312 Handle<Object> maybe_host_defined_options;
313 if (!host_defined_options_.ToHandle(&maybe_host_defined_options)) {
314 maybe_host_defined_options = isolate_->factory()->empty_fixed_array();
315 }
316 Tagged<FixedArray> host_defined_options =
317 Cast<FixedArray>(*maybe_host_defined_options);
318 Tagged<FixedArray> script_options =
319 Cast<FixedArray>(script->host_defined_options());
320 int length = host_defined_options->length();
321 if (length != script_options->length()) return false;
322
323 for (int i = 0; i < length; i++) {
324 // host-defined options is a v8::PrimitiveArray.
325 DCHECK(IsPrimitive(host_defined_options->get(i)));
326 DCHECK(IsPrimitive(script_options->get(i)));
327 if (!Object::StrictEquals(host_defined_options->get(i),
328 script_options->get(i))) {
329 return false;
330 }
331 }
332 return true;
333}

375bool ScriptCacheKey::IsMatch(Tagged<Object> other) {
376 DisallowGarbageCollection no_gc;
377 DCHECK(IsWeakFixedArray(other));
378 Tagged<WeakFixedArray> other_array = Cast<WeakFixedArray>(other);
379 DCHECK_EQ(other_array->length(), kEnd);
380
381 // A hash check can quickly reject many non-matches, even though this step
382 // isn't strictly necessary.
383 uint32_t other_hash =
384 static_cast<uint32_t>(other_array->get(kHash).ToSmi().value());
385 if (other_hash != Hash()) return false;
386
387 Tagged<HeapObject> other_script_object;
388 if (!other_array->get(kWeakScript)
389 .GetHeapObjectIfWeak(&other_script_object)) {
390 return false;
391 }
392 Tagged<Script> other_script = Cast<Script>(other_script_object);
393 Tagged<String> other_source = Cast<String>(other_script->source());
394
395 return other_source->Equals(*source_) && MatchesScript(other_script);
396}

438CompilationCacheScriptLookupResult CompilationCacheTable::LookupScript(
439 DirectHandle<CompilationCacheTable> table, Handle<String> src,
440 const ScriptDetails& script_details, Isolate* isolate) {
441 src = String::Flatten(isolate, src);
442 ScriptCacheKey key(src, &script_details, isolate);
443 InternalIndex entry = table->FindEntry(isolate, &key);
444 if (entry.is_not_found()) return {};
445
446 DisallowGarbageCollection no_gc;
447 Tagged<Object> key_in_table = table->KeyAt(entry);
448 Tagged<Script> script = Cast<Script>(Cast<WeakFixedArray>(key_in_table)
449 ->get(ScriptCacheKey::kWeakScript)
450 .GetHeapObjectAssumeWeak());
451
452 Tagged<Object> obj = table->PrimaryValueAt(entry);
453 Tagged<SharedFunctionInfo> toplevel_sfi;
454 if (!IsUndefined(obj, isolate)) {
455 toplevel_sfi = Cast<SharedFunctionInfo>(obj);
456 DCHECK_EQ(toplevel_sfi->script(), script);
457 }
458
459 return CompilationCacheScriptLookupResult::FromRawObjects(
460 std::make_pair(script, toplevel_sfi), isolate);
461}

463InfoCellPair CompilationCacheTable::LookupEval(
464 DirectHandle<CompilationCacheTable> table, Handle<String> src,
465 Handle<SharedFunctionInfo> outer_info,
466 DirectHandle<NativeContext> native_context, LanguageMode language_mode,
467 int position) {
468 InfoCellPair empty_result;
469 Isolate* isolate = native_context->GetIsolate();
470 src = String::Flatten(isolate, src);
471
472 EvalCacheKey key(src, outer_info, language_mode, position);
473 InternalIndex entry = table->FindEntry(isolate, &key);
474 if (entry.is_not_found()) return empty_result;
475
476 if (!IsFixedArray(table->KeyAt(entry))) return empty_result;
477 Tagged<Object> obj = table->PrimaryValueAt(entry);
478 if (!IsSharedFunctionInfo(obj)) return empty_result;
479
480 static_assert(CompilationCacheShape::kEntrySize == 3);
481 Tagged<FeedbackCell> feedback_cell =
482 SearchLiteralsMap(*table, entry, *native_context);
483 return InfoCellPair(isolate, Cast<SharedFunctionInfo>(obj), feedback_cell);
484}

496Handle<CompilationCacheTable> CompilationCacheTable::EnsureScriptTableCapacity(
497 Isolate* isolate, Handle<CompilationCacheTable> cache) {
498 if (cache->HasSufficientCapacityToAdd(1)) return cache;
499
500 // Before resizing, delete are any entries whose keys contain cleared weak
501 // pointers.
502 {
503 DisallowGarbageCollection no_gc;
504 for (InternalIndex entry : cache->IterateEntries()) {
505 Tagged<Object> key;
506 if (!cache->ToKey(isolate, entry, &key)) continue;
507 if (Cast<WeakFixedArray>(key)
508 ->get(ScriptCacheKey::kWeakScript)
509 .IsCleared()) {
510 DCHECK(IsUndefined(cache->PrimaryValueAt(entry)));
511 cache->RemoveEntry(entry);
512 }
513 }
514 }
515
516 return EnsureCapacity(isolate, cache);
517}

519Handle<CompilationCacheTable> CompilationCacheTable::PutScript(
520 Handle<CompilationCacheTable> cache, Handle<String> src,
521 MaybeHandle<FixedArray> maybe_wrapped_arguments,
522 DirectHandle<SharedFunctionInfo> value, Isolate* isolate) {
523 src = String::Flatten(isolate, src);
524 DirectHandle<Script> script(Cast<Script>(value->script()), isolate);
525 MaybeHandle<Object> script_name;
526 if (IsString(script->name(), isolate)) {
527 script_name = handle(script->name(), isolate);
528 }
529 Handle<FixedArray> host_defined_options(script->host_defined_options(),
530 isolate);
531 ScriptCacheKey key(src, script_name, script->line_offset(),
532 script->column_offset(), script->origin_options(),
533 host_defined_options, maybe_wrapped_arguments, isolate);
534 DirectHandle<Object> k = key.AsHandle(isolate, value);
535
536 // Check whether there is already a matching entry. If so, we must overwrite
537 // it. This allows an entry whose value is undefined to upgrade to contain a
538 // SharedFunctionInfo.
539 InternalIndex entry = cache->FindEntry(isolate, &key);
540 bool found_existing = entry.is_found();
541 if (!found_existing) {
542 cache = EnsureScriptTableCapacity(isolate, cache);
543 entry = cache->FindInsertionEntry(isolate, key.Hash());
544 }
545 // We might be tempted to DCHECK here that the Script in the existing entry
546 // matches the Script in the new key. However, replacing an existing Script
547 // can still happen in some edge cases that aren't common enough to be worth
548 // fixing. Consider the following unlikely sequence of events:
549 // 1. BackgroundMergeTask::SetUpOnMainThread finds a script S1 in the cache.
550 // 2. DevTools is attached and clears the cache.
551 // 3. DevTools is detached; the cache is reenabled.
552 // 4. A new instance of the script, S2, is compiled and placed into the cache.
553 // 5. The merge from step 1 finishes on the main thread, still using S1, and
554 // places S1 into the cache, replacing S2.
555 cache->SetKeyAt(entry, *k);
556 cache->SetPrimaryValueAt(entry, *value);
557 if (!found_existing) {
558 cache->ElementAdded();
559 }
560 return cache;
561}

563Handle<CompilationCacheTable> CompilationCacheTable::PutEval(
564 Handle<CompilationCacheTable> cache, Handle<String> src,
565 Handle<SharedFunctionInfo> outer_info,
566 DirectHandle<SharedFunctionInfo> value,
567 DirectHandle<NativeContext> native_context,
568 DirectHandle<FeedbackCell> feedback_cell, int position) {
569 Isolate* isolate = native_context->GetIsolate();
570 src = String::Flatten(isolate, src);
571 EvalCacheKey key(src, outer_info, value->language_mode(), position);
572
573 // This block handles 'real' insertions, i.e. the initial dummy insert
574 // (below) has already happened earlier.
575 {
576 DirectHandle<Object> k = key.AsHandle(isolate);
577 InternalIndex entry = cache->FindEntry(isolate, &key);
578 if (entry.is_found()) {
579 cache->SetKeyAt(entry, *k);
580 if (cache->PrimaryValueAt(entry) != *value) {
581 cache->SetPrimaryValueAt(entry, *value);
582 // The SFI is changing because the code was aged. Nuke existing feedback
583 // since it can't be reused after this point.
584 cache->SetEvalFeedbackValueAt(entry,
585 ReadOnlyRoots(isolate).the_hole_value());
586 }
587 // AddToFeedbackCellsMap may allocate a new sub-array to live in the
588 // entry, but it won't change the cache array. Therefore EntryToIndex
589 // and entry remains correct.
590 AddToFeedbackCellsMap(cache, entry, native_context, feedback_cell);
591 // Add hash again even on cache hit to avoid unnecessary cache delay in
592 // case of hash collisions.
593 }
594 }
595
596 // Create a dummy entry to mark that this key has already been inserted once.
597 cache = EnsureCapacity(isolate, cache);
598 InternalIndex entry = cache->FindInsertionEntry(isolate, key.Hash());
599 DirectHandle<Object> k =
600 isolate->factory()->NewNumber(static_cast<double>(key.Hash()));
601 cache->SetKeyAt(entry, *k);
602 cache->SetPrimaryValueAt(entry, Smi::FromInt(kHashGenerations));
603 cache->ElementAdded();
604 return cache;
605}

630void CompilationCacheTable::RemoveEntry(InternalIndex entry) {
631 int entry_index = EntryToIndex(entry);
632 Tagged<Object> the_hole_value = GetReadOnlyRoots().the_hole_value();
633 for (int i = 0; i < kEntrySize; i++) {
