182MaybeHandle<Cell> SourceTextModule::ResolveExport(
183 Isolate* isolate, Handle<SourceTextModule> module,
184 DirectHandle<String> module_specifier, Handle<String> export_name,
185 MessageLocation loc, bool must_resolve, Module::ResolveSet* resolve_set) {
186 Handle<Object> object(module->exports()->Lookup(export_name), isolate);
187 if (IsCell(*object)) {
188 // Already resolved (e.g. because it's a local export).
189 return Cast<Cell>(object);
190 }
191
192 // Check for cycle before recursing.
193 {
194 // Attempt insertion with a null string set.
195 auto result = resolve_set->insert({module, nullptr});
196 UnorderedStringSet*& name_set = result.first->second;
197 if (result.second) {
198 // |module| wasn't in the map previously, so allocate a new name set.
199 Zone* zone = resolve_set->zone();
200 name_set = zone->New<UnorderedStringSet>(zone);
201 } else if (name_set->count(export_name)) {
202 // Cycle detected.
203 if (must_resolve) {
204 isolate->ThrowAt(isolate->factory()->NewSyntaxError(
205 MessageTemplate::kCyclicModuleDependency,
206 export_name, module_specifier),
207 &loc);
208 return MaybeHandle<Cell>();
209 }
210 return MaybeHandle<Cell>();
211 }
212 name_set->insert(export_name);
213 }
214
215 if (IsSourceTextModuleInfoEntry(*object)) {
216 // Not yet resolved indirect export.
217 auto entry = Cast<SourceTextModuleInfoEntry>(object);
218 Handle<String> import_name(Cast<String>(entry->import_name()), isolate);
219 Handle<Script> script(module->GetScript(), isolate);
220 MessageLocation new_loc(script, entry->beg_pos(), entry->end_pos());
221
222 Handle<Cell> cell;
223 if (!ResolveImport(isolate, module, import_name, entry->module_request(),
224 new_loc, true, resolve_set)
225 .ToHandle(&cell)) {
226 DCHECK(isolate->has_exception());
227 return MaybeHandle<Cell>();
228 }
229
230 // The export table may have changed but the entry in question should be
231 // unchanged.
232 Handle<ObjectHashTable> exports(module->exports(), isolate);
233 DCHECK(IsSourceTextModuleInfoEntry(exports->Lookup(export_name)));
234
235 exports = ObjectHashTable::Put(exports, export_name, cell);
236 module->set_exports(*exports);
237 return cell;
238 }
239
240 DCHECK(IsTheHole(*object, isolate));
241 return SourceTextModule::ResolveExportUsingStarExports(
242 isolate, module, module_specifier, export_name, loc, must_resolve,
243 resolve_set);
244}

246MaybeHandle<Cell> SourceTextModule::ResolveImport(
247 Isolate* isolate, DirectHandle<SourceTextModule> module,
248 Handle<String> name, int module_request_index, MessageLocation loc,
249 bool must_resolve, Module::ResolveSet* resolve_set) {
250 DirectHandle<ModuleRequest> module_request(
251 Cast<ModuleRequest>(
252 module->info()->module_requests()->get(module_request_index)),
253 isolate);
254 switch (module_request->phase()) {
255 case ModuleImportPhase::kSource: {
256 DCHECK(v8_flags.js_source_phase_imports);
257
258 // https://tc39.es/proposal-source-phase-imports/#sec-source-text-module-record-initialize-environment
259 // InitializeEnvironment
260 // 7.c. Else if in.[[ImportName]] is source, then
261 // 7.c.i. Let moduleSourceObject be ? importedModule.GetModuleSource().
262 // 7.c.ii. Perform ! env.CreateImmutableBinding(in.[[LocalName]], true).
263 // 7.c.iii. Perform ! env.InitializeBinding(in.[[LocalName]],
264 // moduleSourceObject).
265 Handle<Cell> cell = isolate->factory()->NewCell();
266 cell->set_value(module->requested_modules()->get(module_request_index));
267 return cell;
268 }
269 case ModuleImportPhase::kEvaluation: {
270 DCHECK_EQ(module_request->phase(), ModuleImportPhase::kEvaluation);
271 Handle<Module> requested_module(
272 Cast<Module>(module->requested_modules()->get(module_request_index)),
273 isolate);
274 DirectHandle<String> module_specifier(
275 Cast<String>(module_request->specifier()), isolate);
276 MaybeHandle<Cell> result =
277 Module::ResolveExport(isolate, requested_module, module_specifier,
278 name, loc, must_resolve, resolve_set);
279 DCHECK_IMPLIES(isolate->has_exception(), result.is_null());
280 return result;
281 }
282 default:
283 UNREACHABLE();
284 }
285}

287MaybeHandle<Cell> SourceTextModule::ResolveExportUsingStarExports(
288 Isolate* isolate, DirectHandle<SourceTextModule> module,
289 DirectHandle<String> module_specifier, Handle<String> export_name,
290 MessageLocation loc, bool must_resolve, Module::ResolveSet* resolve_set) {
291 if (!export_name->Equals(ReadOnlyRoots(isolate).default_string())) {
292 // Go through all star exports looking for the given name. If multiple star
293 // exports provide the name, make sure they all map it to the same cell.
294 Handle<Cell> unique_cell;
295 DirectHandle<FixedArray> special_exports(module->info()->special_exports(),
296 isolate);
297 for (int i = 0, n = special_exports->length(); i < n; ++i) {
298 i::DirectHandle<i::SourceTextModuleInfoEntry> entry(
299 i::Cast<i::SourceTextModuleInfoEntry>(special_exports->get(i)),
300 isolate);
301 if (!IsUndefined(entry->export_name(), isolate)) {
302 continue; // Indirect export.
303 }
304
305 Handle<Script> script(module->GetScript(), isolate);
306 MessageLocation new_loc(script, entry->beg_pos(), entry->end_pos());
307
308 Handle<Cell> cell;
309 if (ResolveImport(isolate, module, export_name, entry->module_request(),
310 new_loc, false, resolve_set)
311 .ToHandle(&cell)) {
312 if (unique_cell.is_null()) unique_cell = cell;
313 if (*unique_cell != *cell) {
314 isolate->ThrowAt(isolate->factory()->NewSyntaxError(
315 MessageTemplate::kAmbiguousExport,
316 module_specifier, export_name),
317 &loc);
318 return MaybeHandle<Cell>();
319 }
320 } else if (isolate->has_exception()) {
321 return MaybeHandle<Cell>();
322 }
323 }
324
325 if (!unique_cell.is_null()) {
326 // Found a unique star export for this name.
327 Handle<ObjectHashTable> exports(module->exports(), isolate);
328 DCHECK(IsTheHole(exports->Lookup(export_name), isolate));
329 exports = ObjectHashTable::Put(exports, export_name, unique_cell);
330 module->set_exports(*exports);
331 return unique_cell;
332 }
333 }
334
335 // Unresolvable.
336 if (must_resolve) {
337 isolate->ThrowAt(
338 isolate->factory()->NewSyntaxError(MessageTemplate::kUnresolvableExport,
339 module_specifier, export_name),
340 &loc);
341 return MaybeHandle<Cell>();
342 }
343 return MaybeHandle<Cell>();
344}

346bool SourceTextModule::PrepareInstantiate(
347 Isolate* isolate, DirectHandle<SourceTextModule> module,
348 v8::Local<v8::Context> context,
349 v8::Module::ResolveModuleCallback module_callback,
350 v8::Module::ResolveSourceCallback source_callback) {
351 DCHECK_NE(module_callback, nullptr);
352 // Obtain requested modules.
353 DirectHandle<SourceTextModuleInfo> module_info(module->info(), isolate);
354 DirectHandle<FixedArray> module_requests(module_info->module_requests(),
355 isolate);
356 DirectHandle<FixedArray> requested_modules(module->requested_modules(),
357 isolate);
358 for (int i = 0, length = module_requests->length(); i < length; ++i) {
359 DirectHandle<ModuleRequest> module_request(
360 Cast<ModuleRequest>(module_requests->get(i)), isolate);
361 DirectHandle<String> specifier(module_request->specifier(), isolate);
362 DirectHandle<FixedArray> import_attributes(
363 module_request->import_attributes(), isolate);
364 switch (module_request->phase()) {
365 case ModuleImportPhase::kEvaluation: {
366 v8::Local<v8::Module> api_requested_module;
367 if (!module_callback(context, v8::Utils::ToLocal(specifier),
368 v8::Utils::FixedArrayToLocal(import_attributes),
369 v8::Utils::ToLocal(Cast<Module>(module)))
370 .ToLocal(&api_requested_module)) {
371 return false;
372 }
373 DirectHandle<Module> requested_module =
374 Utils::OpenDirectHandle(*api_requested_module);
375 requested_modules->set(i, *requested_module);
376 break;
377 }
378 case ModuleImportPhase::kSource: {
379 DCHECK(v8_flags.js_source_phase_imports);
380#if V8_ENABLE_WEBASSEMBLY
381 v8::Local<v8::Object> api_requested_module_source;
382 if (!source_callback(context, v8::Utils::ToLocal(specifier),
383 v8::Utils::FixedArrayToLocal(import_attributes),
384 v8::Utils::ToLocal(Cast<Module>(module)))
385 .ToLocal(&api_requested_module_source)) {
386 return false;
387 }
388 DirectHandle<JSReceiver> requested_module_source =
389 Utils::OpenDirectHandle(*api_requested_module_source);
390 CHECK(IsWasmModuleObject(*requested_module_source));
391 requested_modules->set(i, *requested_module_source);
392 break;
393#else
394 // Only WebAssembly modules can be requested in the source phase.
395 UNREACHABLE();
396#endif
397 }
398 default:
399 UNREACHABLE();
400 }
401 }
402
403 // Recurse.
404 for (int i = 0, length = requested_modules->length(); i < length; ++i) {
405 DirectHandle<ModuleRequest> module_request(
406 Cast<ModuleRequest>(module_requests->get(i)), isolate);
407 if (module_request->phase() != ModuleImportPhase::kEvaluation) {
408 continue;
409 }
410 DirectHandle<Module> requested_module(
411 Cast<Module>(requested_modules->get(i)), isolate);
412 if (!Module::PrepareInstantiate(isolate, requested_module, context,
413 module_callback, source_callback)) {
414 return false;
415 }
416 }
417
418 // Set up local exports.
419 // TODO(neis): Create regular_exports array here instead of in factory method?
420 for (int i = 0, n = module_info->RegularExportCount(); i < n; ++i) {
421 int cell_index = module_info->RegularExportCellIndex(i);
422 DirectHandle<FixedArray> export_names(
423 module_info->RegularExportExportNames(i), isolate);
424 CreateExport(isolate, module, cell_index, export_names);
425 }
426
427 // Partially set up indirect exports.
428 // For each indirect export, we create the appropriate slot in the export
429 // table and store its SourceTextModuleInfoEntry there. When we later find
430 // the correct Cell in the module that actually provides the value, we replace
431 // the SourceTextModuleInfoEntry by that Cell (see ResolveExport).
432 DirectHandle<FixedArray> special_exports(module_info->special_exports(),
433 isolate);
434 for (int i = 0, n = special_exports->length(); i < n; ++i) {
435 DirectHandle<SourceTextModuleInfoEntry> entry(
436 Cast<SourceTextModuleInfoEntry>(special_exports->get(i)), isolate);
437 DirectHandle<Object> export_name(entry->export_name(), isolate);
438 if (IsUndefined(*export_name, isolate)) continue; // Star export.
439 CreateIndirectExport(isolate, module, Cast<String>(export_name), entry);
440 }
441
442 DCHECK_EQ(module->status(), kPreLinking);
443 return true;
444}

446bool SourceTextModule::RunInitializationCode(
447 Isolate* isolate, DirectHandle<SourceTextModule> module) {
448 DCHECK_EQ(module->status(), kLinking);
449 DirectHandle<JSFunction> function(Cast<JSFunction>(module->code()), isolate);
450 DCHECK_EQ(MODULE_SCOPE, function->shared()->scope_info()->scope_type());
451 DirectHandle<Object> receiver = isolate->factory()->undefined_value();
452
453 DirectHandle<ScopeInfo> scope_info(function->shared()->scope_info(), isolate);
454 DirectHandle<Context> context = isolate->factory()->NewModuleContext(
455 module, isolate->native_context(), scope_info);
456 function->set_context(*context);
457
458 MaybeDirectHandle<Object> maybe_generator =
459 Execution::Call(isolate, function, receiver, {});
460 DirectHandle<Object> generator;
461 if (!maybe_generator.ToHandle(&generator)) {
462 DCHECK(isolate->has_exception());
463 return false;
464 }
465 DCHECK_EQ(*function, Cast<JSGeneratorObject>(generator)->function());
466 module->set_code(Cast<JSGeneratorObject>(*generator));
467 return true;
468}

470// ES#sec-innermoduleevaluation and ES#sec-innermodulelinking
471bool SourceTextModule::MaybeTransitionComponent(
472 Isolate* isolate, DirectHandle<SourceTextModule> module,
473 ZoneForwardList<Handle<SourceTextModule>>* stack, Status new_status) {
474 DCHECK(new_status == kLinked || new_status == kEvaluated);
475
476#ifdef DEBUG
477 if (v8_flags.trace_module_status) {
478 StdoutStream os;
479 os << "Transitioning strongly connected module graph component to "
480 << Module::StatusString(new_status) << " {\n";
481 }
482#endif // DEBUG
483
484 // Below, N/M means step N in InnerModuleEvaluation and step M in
485 // InnerModuleLinking.
486
487 // 14/11. Assert: module occurs exactly once in stack.
488 SLOW_DCHECK(
489 // {module} is on the {stack}.
490 std::count_if(stack->begin(), stack->end(), [&](DirectHandle<Module> m) {
491 return *m == *module;
492 }) == 1);
493
494 // 15/12. Assert: module.[[DFSAncestorIndex]] â‰¤ module.[[DFSIndex]].
495 DCHECK_LE(module->dfs_ancestor_index(), module->dfs_index());
496
497 // 16/13. If module.[[DFSAncestorIndex]] = module.[[DFSIndex]], then
498 if (module->dfs_ancestor_index() == module->dfs_index()) {
499 // This is the root of its strongly connected component.
500 DirectHandle<SourceTextModule> cycle_root = module;
501 DirectHandle<SourceTextModule> ancestor;
502 // This loop handles the loops in both InnerModuleEvaluation and
503 // InnerModuleLinking.
504 //
505 // InnerModuleEvaluation
506 //
507 // a. Let done be false.
508 // b. Repeat, while done is false,
509 // i. Let requiredModule be the last element of stack.
510 // ii. Remove the last element of stack.
511 // iii. Assert: requiredModule is a Cyclic Module Record.
512 // iv. If requiredModule.[[AsyncEvaluation]] is false, set
513 // requiredModule.[[Status]] to EVALUATED.
514 // v. Otherwise, set requiredModule.[[Status]] to EVALUATING-ASYNC.
515 // vi. If requiredModule and module are the same Module Record, set done
516 // to true.
517 // vii. Set requiredModule.[[CycleRoot]] to module.
518 //
519 // InnerModuleLinking
520 //
521 // a. Let done be false.
522 // b. Repeat, while done is false,
523 // i. Let requiredModule be the last element of stack.
524 // ii. Remove the last element of stack.
525 // iii. Assert: requiredModule is a Cyclic Module Record.
526 // iv. Set requiredModule.[[Status]] to LINKED.
527 // v. If requiredModule and module are the same Module Record, set done
528 // to true.
529 do {
530 ancestor = stack->front();
531 stack->pop_front();
532 DCHECK_EQ(ancestor->status(),
533 new_status == kLinked ? kLinking : kEvaluating);
534 if (new_status == kLinked) {
535 if (!SourceTextModule::RunInitializationCode(isolate, ancestor)) {
536 return false;
537 }
538 ancestor->SetStatus(kLinked);
539 } else {
540 DCHECK(IsTheHole(ancestor->cycle_root(), isolate));
541 ancestor->set_cycle_root(*cycle_root);
542 ancestor->SetStatus(ancestor->HasAsyncEvaluationOrdinal()
543 ? kEvaluatingAsync
544 : kEvaluated);
545 }
546 } while (*ancestor != *module);
547 }
548#ifdef DEBUG
549 if (v8_flags.trace_module_status) {
550 StdoutStream os;
551 os << "}\n";
552 }
553#endif // DEBUG
554 return true;
555}

557bool SourceTextModule::FinishInstantiate(
558 Isolate* isolate, Handle<SourceTextModule> module,
559 ZoneForwardList<Handle<SourceTextModule>>* stack, unsigned* dfs_index,
560 Zone* zone) {
561 // Instantiate SharedFunctionInfo and mark module as instantiating for
562 // the recursion.
563 DirectHandle<SharedFunctionInfo> shared(
564 Cast<SharedFunctionInfo>(module->code()), isolate);
565 DirectHandle<JSFunction> function =
566 Factory::JSFunctionBuilder{isolate, shared, isolate->native_context()}
567 .Build();
568 module->set_code(*function);
569 module->SetStatus(kLinking);
570 module->set_dfs_index(*dfs_index);
571 module->set_dfs_ancestor_index(*dfs_index);
572 stack->push_front(module);
573 (*dfs_index)++;
574
575 // Recurse.
576 DirectHandle<FixedArray> module_requests(module->info()->module_requests(),
577 isolate);
578 DirectHandle<FixedArray> requested_modules(module->requested_modules(),
579 isolate);
580 for (int i = 0, length = requested_modules->length(); i < length; ++i) {
581 DirectHandle<ModuleRequest> module_request(
582 Cast<ModuleRequest>(module_requests->get(i)), isolate);
583 if (module_request->phase() != ModuleImportPhase::kEvaluation) {
584 continue;
585 }
586 Handle<Module> requested_module(Cast<Module>(requested_modules->get(i)),
587 isolate);
588 if (!Module::FinishInstantiate(isolate, requested_module, stack, dfs_index,
589 zone)) {
590 return false;
591 }
592
593 DCHECK_NE(requested_module->status(), kEvaluating);
594 DCHECK_GE(requested_module->status(), kLinking);
595 SLOW_DCHECK(
596 // {requested_module} is instantiating iff it's on the {stack}.
597 (requested_module->status() == kLinking) ==
598 std::count_if(
599 stack->begin(), stack->end(),
600 [&](DirectHandle<Module> m) { return *m == *requested_module; }));
601
602 if (requested_module->status() == kLinking) {
603 // SyntheticModules go straight to kLinked so this must be a
604 // SourceTextModule
605 module->set_dfs_ancestor_index(std::min(
606 module->dfs_ancestor_index(),
607 Cast<SourceTextModule>(*requested_module)->dfs_ancestor_index()));
608 }
609 }
610
611 Handle<Script> script(module->GetScript(), isolate);
612 DirectHandle<SourceTextModuleInfo> module_info(module->info(), isolate);
613
614 // Resolve imports.
615 DirectHandle<FixedArray> regular_imports(module_info->regular_imports(),
616 isolate);
617 for (int i = 0, n = regular_imports->length(); i < n; ++i) {
618 DirectHandle<SourceTextModuleInfoEntry> entry(
619 Cast<SourceTextModuleInfoEntry>(regular_imports->get(i)), isolate);
620 Handle<String> name(Cast<String>(entry->import_name()), isolate);
621 MessageLocation loc(script, entry->beg_pos(), entry->end_pos());
622 ResolveSet resolve_set(zone);
623 DirectHandle<Cell> cell;
624 if (!ResolveImport(isolate, module, name, entry->module_request(), loc,
625 true, &resolve_set)
626 .ToHandle(&cell)) {
627 return false;
628 }
629 module->regular_imports()->set(ImportIndex(entry->cell_index()), *cell);
630 }
631
632 // Resolve indirect exports.
633 DirectHandle<FixedArray> special_exports(module_info->special_exports(),
634 isolate);
635 for (int i = 0, n = special_exports->length(); i < n; ++i) {
636 DirectHandle<SourceTextModuleInfoEntry> entry(
637 Cast<SourceTextModuleInfoEntry>(special_exports->get(i)), isolate);
638 Handle<Object> name(entry->export_name(), isolate);
639 if (IsUndefined(*name, isolate)) continue; // Star export.
640 MessageLocation loc(script, entry->beg_pos(), entry->end_pos());
641 ResolveSet resolve_set(zone);
642 if (ResolveExport(isolate, module, {}, Cast<String>(name), loc, true,
643 &resolve_set)
644 .is_null()) {
645 return false;
646 }
647 }
648
649 return MaybeTransitionComponent(isolate, module, stack, kLinked);

652void SourceTextModule::FetchStarExports(Isolate* isolate,
653 Handle<SourceTextModule> module,
654 Zone* zone,
655 UnorderedModuleSet* visited) {
656 DCHECK_GE(module->status(), Module::kLinking);
657
658 if (IsJSModuleNamespace(module->module_namespace())) return; // Shortcut.
659
660 bool cycle = !visited->insert(module).second;
661 if (cycle) return;
662 Handle<ObjectHashTable> exports(module->exports(), isolate);
663 UnorderedStringMap more_exports(zone);
664
665 // TODO(neis): Only allocate more_exports if there are star exports.
666 // Maybe split special_exports into indirect_exports and star_exports.
667
668 ReadOnlyRoots roots(isolate);
669 DirectHandle<FixedArray> special_exports(module->info()->special_exports(),
670 isolate);
671 for (int i = 0, n = special_exports->length(); i < n; ++i) {
672 DirectHandle<SourceTextModuleInfoEntry> entry(
673 Cast<SourceTextModuleInfoEntry>(special_exports->get(i)), isolate);
674 if (!IsUndefined(entry->export_name(), roots)) {
675 continue; // Indirect export.
676 }
677
678 DCHECK_EQ(Cast<ModuleRequest>(module->info()->module_requests()->get(
679 entry->module_request()))
680 ->phase(),
681 ModuleImportPhase::kEvaluation);
682 Handle<Module> requested_module(
683 Cast<Module>(module->requested_modules()->get(entry->module_request())),
684 isolate);
685
686 // Recurse.
687 if (IsSourceTextModule(*requested_module))
688 FetchStarExports(isolate, Cast<SourceTextModule>(requested_module), zone,
689 visited);
690
691 // Collect all of [requested_module]'s exports that must be added to
692 // [module]'s exports (i.e. to [exports]). We record these in
693 // [more_exports]. Ambiguities (conflicting exports) are marked by mapping
694 // the name to undefined instead of a Cell.
695 DirectHandle<ObjectHashTable> requested_exports(requested_module->exports(),
696 isolate);
697 for (InternalIndex index : requested_exports->IterateEntries()) {
698 Tagged<Object> key;
699 if (!requested_exports->ToKey(roots, index, &key)) continue;
700 Handle<String> name(Cast<String>(key), isolate);
701
702 if (name->Equals(roots.default_string())) continue;
703 if (!IsTheHole(exports->Lookup(name), roots)) continue;
704
705 Handle<Cell> cell(Cast<Cell>(requested_exports->ValueAt(index)), isolate);
706 auto insert_result = more_exports.insert(std::make_pair(name, cell));
707 if (!insert_result.second) {
708 auto it = insert_result.first;
709 if (*it->second == *cell || IsUndefined(*it->second, roots)) {
710 // We already recorded this mapping before, or the name is already
711 // known to be ambiguous. In either case, there's nothing to do.
712 } else {
713 DCHECK(IsCell(*it->second));
714 // Different star exports provide different cells for this name, hence
715 // mark the name as ambiguous.
716 it->second = isolate->factory()->undefined_value();
717 }
718 }
719 }
720 }
721
722 // Copy [more_exports] into [exports].
723 for (const auto& elem : more_exports) {
724 if (IsUndefined(*elem.second, isolate)) continue; // Ambiguous export.
725 DCHECK(!elem.first->Equals(ReadOnlyRoots(isolate).default_string()));
726 DCHECK(IsCell(*elem.second));
727 exports = ObjectHashTable::Put(exports, elem.first, elem.second);
728 }
729 module->set_exports(*exports);
730}

732void SourceTextModule::GatherAvailableAncestors(
733 Isolate* isolate, Zone* zone, Handle<SourceTextModule> start,
734 AvailableAncestorsSet* exec_list) {
735 // The spec algorithm is recursive. It is transformed to an equivalent
736 // iterative one here.
737 ZoneStack<Handle<SourceTextModule>> worklist(zone);
738 worklist.push(start);
739

745 for (int i = module->AsyncParentModuleCount(); i-- > 0;) {

748 // a. If execList does not contain m and
749 // m.[[CycleRoot]].[[EvaluationError]] is empty, then
750 if (m->GetCycleRoot(isolate)->status() != kErrored &&
751 exec_list->find(m) == exec_list->end()) {
752 // i. Assert: m.[[Status]] is EVALUATING-ASYNC.
753 // ii. Assert: m.[[EvaluationError]] is empty.

757 DCHECK(m->HasAsyncEvaluationOrdinal());

760 DCHECK(m->HasPendingAsyncDependencies());

763 // m.[[PendingAsyncDependencies]] - 1.
764 m->DecrementPendingAsyncDependencies();
765
766 // vi. If m.[[PendingAsyncDependencies]] = 0, then
767 if (!m->HasPendingAsyncDependencies()) {
768 // 1. Append m to execList.
769 exec_list->insert(m);
770
771 // 2. If m.[[HasTLA]] is false,
772 // perform ! GatherAvailableAncestors(m, execList).
773 if (!m->has_toplevel_await()) worklist.push(m);
774 }
775 }
776 }
777 }
778
779 // 2. Return UNUSED.

977 m->set_async_evaluation_ordinal(kAsyncEvaluateDidFinish);

979 // 2. Set m.[[Status]] to EVALUATED.

1164MaybeHandle<Object> SourceTextModule::InnerModuleEvaluation(
1165 Isolate* isolate, Handle<SourceTextModule> module,
1166 ZoneForwardList<Handle<SourceTextModule>>* stack, unsigned* dfs_index) {
1167 STACK_CHECK(isolate, MaybeHandle<Object>());
1168 int module_status = module->status();
1169 // InnerModuleEvaluation(module, stack, index)
1170
1171 // 2. If module.[[Status]] is either EVALUATING-ASYNC or EVALUATED, then
1172 if (module_status == kEvaluatingAsync || module_status == kEvaluating ||
1173 module_status == kEvaluated) {
1174 // a. If module.[[EvaluationError]] is undefined, return index.
1175 // (We return undefined instead)
1176 //
1177 // 3. If module.[[Status]] is EVALUATING, return index.
1178 // (Out of order)
1179 return isolate->factory()->undefined_value();
1180 } else if (module_status == kErrored) {
1181 // b. Otherwise return module.[[EvaluationError]].
1182 // (We throw on isolate and return a MaybeHandle<Object> instead)
1183 isolate->Throw(module->exception());
1184 return MaybeHandle<Object>();
1185 }
1186
1187 // 4. Assert: module.[[Status]] is LINKED.
1188 CHECK_EQ(module_status, kLinked);
1189
1190 DirectHandle<FixedArray> module_requests;
1191 DirectHandle<FixedArray> requested_modules;
1192
1193 {
1194 DisallowGarbageCollection no_gc;
1195 Tagged<SourceTextModule> raw_module = *module;
1196 // 5. Set module.[[Status]] to EVALUATING.
1197 raw_module->SetStatus(kEvaluating);
1198
1199 // 6. Set module.[[DFSIndex]] to index.
1200 raw_module->set_dfs_index(*dfs_index);
1201
1202 // 7. Set module.[[DFSAncestorIndex]] to index.
1203 raw_module->set_dfs_ancestor_index(*dfs_index);
1204
1205 // 8. Set module.[[PendingAsyncDependencies]] to 0.
1206 DCHECK(!raw_module->HasPendingAsyncDependencies());
1207
1208 // 9. Set index to index + 1.
1209 (*dfs_index)++;
1210
1211 // 10. Append module to stack.
1212 stack->push_front(module);
1213
1214 // Recursion.
1215 module_requests =
1216 direct_handle(raw_module->info()->module_requests(), isolate);
1217 requested_modules = direct_handle(raw_module->requested_modules(), isolate);
1218 }
1219
1220 // 11. For each ModuleRequest Record required of module.[[RequestedModules]],
1221 for (int i = 0, length = requested_modules->length(); i < length; ++i) {
1222 DirectHandle<ModuleRequest> module_request(
1223 Cast<ModuleRequest>(module_requests->get(i)), isolate);
1224 if (module_request->phase() != ModuleImportPhase::kEvaluation) {
1225 continue;
1226 }
1227 // b. If requiredModule.[[Phase]] is evaluation, then
1228 Handle<Module> requested_module(Cast<Module>(requested_modules->get(i)),
1229 isolate);
1230 // c. If requiredModule is a Cyclic Module Record, then
1231 if (IsSourceTextModule(*requested_module)) {
1232 // b. Set index to ? InnerModuleEvaluation(requiredModule, stack, index).
1233 // (Out of order because InnerModuleEvaluation is type-driven.)
1234 Handle<SourceTextModule> required_module(
1235 Cast<SourceTextModule>(*requested_module), isolate);
1236 RETURN_ON_EXCEPTION(
1237 isolate,
1238 InnerModuleEvaluation(isolate, required_module, stack, dfs_index));
1239 int required_module_status = required_module->status();
1240
1241 // i. Assert: requiredModule.[[Status]] is one of EVALUATING,
1242 // EVALUATING-ASYNC, or EVALUATED.
1243 // (We also assert the module cannot be errored, because if it was
1244 // we would have already returned from InnerModuleEvaluation)
1245 CHECK_GE(required_module_status, kEvaluating);
1246 CHECK_NE(required_module_status, kErrored);
1247
1248 // ii. Assert: requiredModule.[[Status]] is EVALUATING if and only if
1249 // requiredModule is in stack.
1250 SLOW_DCHECK((requested_module->status() == kEvaluating) ==
1251 std::count_if(stack->begin(), stack->end(),
1252 [&](DirectHandle<Module> m) {
1253 return *m == *requested_module;
1254 }));
1255
1256 // iii. If requiredModule.[[Status]] is EVALUATING, then
1257 if (required_module_status == kEvaluating) {
1258 // 1. Set module.[[DFSAncestorIndex]] to
1259 // min(module.[[DFSAncestorIndex]],
1260 // requiredModule.[[DFSAncestorIndex]]).
1261 module->set_dfs_ancestor_index(
1262 std::min(module->dfs_ancestor_index(),
1263 required_module->dfs_ancestor_index()));
1264 } else { // iv. Else,
1265 // 1. Set requiredModule to requiredModule.[[CycleRoot]].
1266 required_module = required_module->GetCycleRoot(isolate);
1267 required_module_status = required_module->status();
1268
1269 // 2. Assert: requiredModule.[[Status]] is either EVALUATING-ASYNC or
1270 // EVALUATED.
1271 CHECK_GE(required_module_status, kEvaluatingAsync);
1272
1273 // 3. If requiredModule.[[EvaluationError]] is not EMPTY,
1274 // return ? module.[[EvaluationError]].
1275
1276 // (If there was an exception on the original required module we would
1277 // have already returned. This check handles the case where the
1278 // AsyncCycleRoot has an error. Instead of returning the exception, we
1279 // throw on isolate and return a MaybeHandle<Object>.)
1280 if (required_module_status == kErrored) {
1281 isolate->Throw(required_module->exception());
1282 return MaybeHandle<Object>();
1283 }
1284 }
1285 // v. If requiredModule.[[AsyncEvaluation]] is true, then
1286 if (required_module->HasAsyncEvaluationOrdinal()) {
1287 // 1. Set module.[[PendingAsyncDependencies]] to
1288 // module.[[PendingAsyncDependencies]] + 1.
1289 module->IncrementPendingAsyncDependencies();
1290
1291 // 2. Append module to requiredModule.[[AsyncParentModules]].
1292 AddAsyncParentModule(isolate, required_module, module);
1293 }
1294 } else {
1295 // b. Set index to ? InnerModuleEvaluation(requiredModule, stack, index).
1296 // (Out of order because InnerModuleEvaluation is type-driven.)
1297 RETURN_ON_EXCEPTION(isolate, Module::Evaluate(isolate, requested_module));
1298 }
1299 }
1300
1301 // The spec returns the module index for proper numbering of dependencies.
1302 // However, we pass the module index by pointer instead.
1303 //
1304 // Before async modules v8 returned the value result from calling next
1305 // on the module's implicit iterator. We preserve this behavior for
1306 // synchronous modules, but return undefined for AsyncModules.
1307 Handle<Object> result = isolate->factory()->undefined_value();
1308
1309 // 12. If module.[[PendingAsyncDependencies]] > 0 or module.[[HasTLA]] is
1310 // true, then
1311 if (module->HasPendingAsyncDependencies() || module->has_toplevel_await()) {
1312 // a. Assert: module.[[AsyncEvaluation]] is false and was never previously
1313 // set to true.
1314 DCHECK_EQ(module->async_evaluation_ordinal(), kNotAsyncEvaluated);
1315
1316 // b. Set module.[[AsyncEvaluation]] to true.
1317 // c. NOTE: The order in which module records have their [[AsyncEvaluation]]
1318 // fields transition to true is significant.
1319 module->set_async_evaluation_ordinal(
1320 isolate->NextModuleAsyncEvaluationOrdinal());
1321
1322 // c. If module.[[PendingAsyncDependencies]] = 0, perform
1323 // ExecuteAsyncModule(module).
1324 // The execution may have been terminated and can not be resumed, so just
1325 // raise the exception.
1326 if (!module->HasPendingAsyncDependencies()) {
1327 MAYBE_RETURN(SourceTextModule::ExecuteAsyncModule(isolate, module),
1328 MaybeHandle<Object>());
1329 }
1330 } else { // 13. Else,
1331 // a. Perform ? module.ExecuteModule().
1332 MaybeDirectHandle<Object> exception;
1333 Handle<Object> result;
1334 if (!ExecuteModule(isolate, module, &exception).ToHandle(&result)) {
1335 if (!isolate->is_execution_terminating()) {
1336 isolate->Throw(*exception.ToHandleChecked());
1337 }
1338 return result;
1339 }
1340 }
1341
1342 CHECK(MaybeTransitionComponent(isolate, module, stack, kEvaluated));
1343 return result;
1344}

1405 if (!HasPendingAsyncDependencies() && HasAsyncEvaluationOrdinal()) {
1406 DCHECK(HasAsyncEvaluationOrdinal());

1420 if (IsSourceTextModule(requested_module) &&
1421 visited->insert(handle(requested_module, isolate)).second) {
1422 Tagged<SourceTextModule> source_text_module =
1423 Cast<SourceTextModule>(requested_module);
1424 source_text_module->InnerGetStalledTopLevelAwaitModule(isolate, visited,

1426 }
