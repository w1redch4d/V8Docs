20// Optimizations involving loops which cannot be done at graph building time.
21// Currently mainly loop invariant code motion.
22class LoopOptimizationProcessor {
23 public:
24 explicit LoopOptimizationProcessor(MaglevGraphBuilder* builder)
25 : zone(builder->zone()) {
26 was_deoptimized =
27 builder->compilation_unit()->feedback().was_once_deoptimized();
28 }
29
30 void PreProcessGraph(Graph* graph) {}
31 void PostPhiProcessing() {}
32
33 BlockProcessResult PreProcessBasicBlock(BasicBlock* block) {
34 current_block = block;
35 if (current_block->is_loop()) {
36 loop_effects = current_block->state()->loop_effects();
37 if (loop_effects) return BlockProcessResult::kContinue;
38 } else {
39 // TODO(olivf): Some dominance analysis would allow us to keep loop
40 // effects longer than just the first block of the loop.
41 loop_effects = nullptr;
42 }
43 return BlockProcessResult::kSkip;
44 }
45
46 bool IsLoopPhi(Node* input) {
47 DCHECK(current_block->is_loop());
48 if (auto phi = input->TryCast<Phi>()) {
49 if (phi->is_loop_phi() && phi->merge_state() == current_block->state()) {
50 return true;
51 }
52 }
53 return false;
54 }
55
56 bool CanHoist(Node* candidate) {
57 DCHECK_EQ(candidate->input_count(), 1);
58 DCHECK(current_block->is_loop());
59 ValueNode* input = candidate->input(0).node();
60 DCHECK(!IsLoopPhi(input));
61 // For hoisting an instruction we need:
62 // * A unique loop entry block.
63 // * Inputs live before the loop (i.e., not defined inside the loop).
64 // * No hoisting over checks (done eagerly by clearing loop_effects).
65 // TODO(olivf): We should enforce loops having a unique entry block at graph
66 // building time.
67 if (current_block->predecessor_count() != 2) return false;
68 BasicBlock* loop_entry = current_block->predecessor_at(0);
69 if (loop_entry->successors().size() != 1) {
70 return false;
71 }
72 if (IsConstantNode(input->opcode())) return true;
73 return input->owner() != current_block;
74 }
75
76 ProcessResult Process(LoadTaggedFieldForContextSlot* ltf,
77 const ProcessingState& state) {
78 DCHECK(loop_effects);
79 ValueNode* object = ltf->object_input().node();
80 if (IsLoopPhi(object)) {
81 return ProcessResult::kContinue;
82 }
83 auto key = std::tuple{object, ltf->offset()};
84 if (!loop_effects->may_have_aliasing_contexts &&
85 !loop_effects->unstable_aspects_cleared &&
86 !loop_effects->context_slot_written.count(key) && CanHoist(ltf)) {
87 return ProcessResult::kHoist;
88 }
89 return ProcessResult::kContinue;
90 }
91
92 ProcessResult Process(LoadTaggedFieldForProperty* ltf,
93 const ProcessingState& state) {
94 return ProcessNamedLoad(ltf, ltf->object_input().node(), ltf->name());
95 }
96
97 ProcessResult Process(StringLength* len, const ProcessingState& state) {
98 return ProcessNamedLoad(
99 len, len->object_input().node(),
100 KnownNodeAspects::LoadedPropertyMapKey::StringLength());
101 }
102
103 ProcessResult Process(LoadTypedArrayLength* len,
104 const ProcessingState& state) {
105 return ProcessNamedLoad(
106 len, len->receiver_input().node(),
107 KnownNodeAspects::LoadedPropertyMapKey::TypedArrayLength());
108 }
109
110 ProcessResult ProcessNamedLoad(Node* load, ValueNode* object,
111 KnownNodeAspects::LoadedPropertyMapKey name) {
112 DCHECK(!load->properties().can_deopt());
113 if (!loop_effects) return ProcessResult::kContinue;
114 if (IsLoopPhi(object)) {
115 return ProcessResult::kContinue;
116 }
117 if (!loop_effects->unstable_aspects_cleared &&
118 !loop_effects->keys_cleared.count(name) &&
119 !loop_effects->objects_written.count(object) && CanHoist(load)) {
120 return ProcessResult::kHoist;
121 }
122 return ProcessResult::kContinue;
123 }
124
125 ProcessResult Process(CheckMaps* maps, const ProcessingState& state) {
126 DCHECK(loop_effects);
127 // Hoisting a check out of a loop can cause it to trigger more than actually
128 // needed (i.e., if the loop is executed 0 times). This could lead to
129 // deoptimization loops as there is no feedback to learn here. Thus, we
130 // abort this optimization if the function deoptimized previously. Also, if
131 // hoisting of this check fails we need to abort (and not continue) to
132 // ensure we are not hoisting other instructions over it.
133 if (was_deoptimized) return ProcessResult::kSkipBlock;
134 ValueNode* object = maps->receiver_input().node();
135 if (IsLoopPhi(object)) {
136 return ProcessResult::kSkipBlock;
137 }
138 if (!loop_effects->unstable_aspects_cleared && CanHoist(maps)) {
139 if (auto j = current_block->predecessor_at(0)
140 ->control_node()
141 ->TryCast<CheckpointedJump>()) {
142 maps->SetEagerDeoptInfo(zone, j->eager_deopt_info()->top_frame(),
143 maps->eager_deopt_info()->feedback_to_update());
144 return ProcessResult::kHoist;
145 }
146 }
147 return ProcessResult::kSkipBlock;
148 }
149
150 template <typename NodeT>
151 ProcessResult Process(NodeT* node, const ProcessingState& state) {
152 // Ensure we are not hoisting over checks.
153 if (node->properties().can_eager_deopt()) {
154 loop_effects = nullptr;
155 return ProcessResult::kSkipBlock;
156 }
157 return ProcessResult::kContinue;
158 }
159
160 void PostProcessGraph(Graph* graph) {}
161
162 Zone* zone;
163 BasicBlock* current_block;
164 const LoopEffects* loop_effects;
165 bool was_deoptimized;
166};

168template <typename NodeT>
169constexpr bool CanBeStoreToNonEscapedObject() {
170 return std::is_same_v<NodeT, StoreMap> ||
171 std::is_same_v<NodeT, StoreTaggedFieldWithWriteBarrier> ||
172 std::is_same_v<NodeT, StoreTaggedFieldNoWriteBarrier> ||
173 std::is_same_v<NodeT, StoreTrustedPointerFieldWithWriteBarrier> ||
174 std::is_same_v<NodeT, StoreFloat64>;

177class AnyUseMarkingProcessor {
178 public:
179 void PreProcessGraph(Graph* graph) {}
180 BlockProcessResult PreProcessBasicBlock(BasicBlock* block) {
181 return BlockProcessResult::kContinue;
182 }
183 void PostPhiProcessing() {}
184
185 template <typename NodeT>
186 ProcessResult Process(NodeT* node, const ProcessingState& state) {
187 if constexpr (IsValueNode(Node::opcode_of<NodeT>) &&
188 (!NodeT::kProperties.is_required_when_unused() ||
189 std::is_same_v<ArgumentsElements, NodeT>)) {
190 if (!node->is_used()) {
191 if (!node->unused_inputs_were_visited()) {
192 DropInputUses(node);
193 }
194 return ProcessResult::kRemove;
195 }
196 }
197
198 if constexpr (CanBeStoreToNonEscapedObject<NodeT>()) {
199 if (node->input(0).node()->template Is<InlinedAllocation>()) {
200 stores_to_allocations_.push_back(node);
201 }
202 }
203
204 return ProcessResult::kContinue;
205 }
206
207#ifdef DEBUG
208 ProcessResult Process(Dead* node, const ProcessingState& state) {
209 UNREACHABLE();
210 }
211#endif // DEBUG
212
213 void PostProcessGraph(Graph* graph) {
214 RunEscapeAnalysis(graph);
215 DropUseOfValueInStoresToCapturedAllocations();
216 }
217
218 private:
219 std::vector<Node*> stores_to_allocations_;
220
221 void EscapeAllocation(Graph* graph, InlinedAllocation* alloc,
222 Graph::SmallAllocationVector& deps) {
223 if (alloc->HasBeenAnalysed() && alloc->HasEscaped()) return;
224 alloc->SetEscaped();
225 for (auto dep : deps) {
226 EscapeAllocation(graph, dep,
227 graph->allocations_escape_map().find(dep)->second);
228 }
229 }
230
231 void VerifyEscapeAnalysis(Graph* graph) {
232#ifdef DEBUG
233 for (auto it : graph->allocations_escape_map()) {
234 auto alloc = it.first;
235 DCHECK(alloc->HasBeenAnalysed());
236 if (alloc->HasEscaped()) {
237 for (auto dep : it.second) {
238 DCHECK(dep->HasEscaped());
239 }
240 }
241 }
242#endif // DEBUG
243 }
244
245 void RunEscapeAnalysis(Graph* graph) {
246 for (auto it : graph->allocations_escape_map()) {
247 auto alloc = it.first;
248 if (alloc->HasBeenAnalysed()) continue;
249 // Check if all its uses are non escaping.
250 if (alloc->IsEscaping()) {
251 // Escape this allocation and all its dependencies.
252 EscapeAllocation(graph, alloc, it.second);
253 } else {
254 // Try to capture the allocation. This can still change if a escaped
255 // allocation has this value as one of its dependencies.
256 alloc->SetElided();
257 }
258 }
259 // Check that we've reached a fixpoint.
260 VerifyEscapeAnalysis(graph);
261 }
262
263 void DropUseOfValueInStoresToCapturedAllocations() {
264 for (Node* node : stores_to_allocations_) {
265 InlinedAllocation* alloc =
266 node->input(0).node()->Cast<InlinedAllocation>();
267 // Since we don't analyze if allocations will escape until a fixpoint,
268 // this could drop an use of an allocation and turn it non-escaping.
269 if (alloc->HasBeenElided()) {
270 // Skip first input.
271 for (int i = 1; i < node->input_count(); i++) {
272 DropInputUses(node->input(i));
273 }
274 }
275 }
276 }
277
278 void DropInputUses(Input& input) {
279 ValueNode* input_node = input.node();
280 if (input_node->properties().is_required_when_unused() &&
281 !input_node->Is<ArgumentsElements>())
282 return;
283 input_node->remove_use();
284 if (!input_node->is_used() && !input_node->unused_inputs_were_visited()) {
285 DropInputUses(input_node);
286 }
287 }
288
289 void DropInputUses(ValueNode* node) {
290 for (Input& input : *node) {
291 DropInputUses(input);
292 }
293 DCHECK(!node->properties().can_eager_deopt());
294 DCHECK(!node->properties().can_lazy_deopt());
295 node->mark_unused_inputs_visited();
296 }
297};

299class DeadNodeSweepingProcessor {
300 public:
301 explicit DeadNodeSweepingProcessor(MaglevCompilationInfo* compilation_info) {
302 if (V8_UNLIKELY(compilation_info->has_graph_labeller())) {
303 labeller_ = compilation_info->graph_labeller();
304 }
305 }
306
307 void PreProcessGraph(Graph* graph) {}
308 void PostProcessGraph(Graph* graph) {}
309 BlockProcessResult PreProcessBasicBlock(BasicBlock* block) {
310 return BlockProcessResult::kContinue;
311 }
312 void PostPhiProcessing() {}
313
314 ProcessResult Process(AllocationBlock* node, const ProcessingState& state) {
315 // Note: this need to be done before ValueLocationConstraintProcessor, since
316 // it access the allocation offsets.
317 int size = 0;
318 for (auto alloc : node->allocation_list()) {
319 if (alloc->HasEscaped()) {
320 alloc->set_offset(size);
321 size += alloc->size();
322 }
323 }
324 // ... and update its size.
325 node->set_size(size);
326 // If size is zero, then none of the inlined allocations have escaped, we
327 // can remove the allocation block.
328 if (size == 0) return ProcessResult::kRemove;
329 return ProcessResult::kContinue;
330 }
331
332 ProcessResult Process(InlinedAllocation* node, const ProcessingState& state) {
333 // Remove inlined allocation that became non-escaping.
334 if (!node->HasEscaped()) {
335 if (v8_flags.trace_maglev_escape_analysis) {
336 std::cout << "* Removing allocation node "
337 << PrintNodeLabel(labeller_, node) << std::endl;
338 }
339 return ProcessResult::kRemove;
340 }
341 return ProcessResult::kContinue;
342 }
343
344 template <typename NodeT>
345 ProcessResult Process(NodeT* node, const ProcessingState& state) {
346 if constexpr (IsValueNode(Node::opcode_of<NodeT>) &&
347 (!NodeT::kProperties.is_required_when_unused() ||
348 std::is_same_v<ArgumentsElements, NodeT>)) {
349 if (!node->is_used()) {
350 return ProcessResult::kRemove;
351 }
352 return ProcessResult::kContinue;
353 }
354
355 if constexpr (CanBeStoreToNonEscapedObject<NodeT>()) {
356 if (InlinedAllocation* object =
357 node->input(0).node()->template TryCast<InlinedAllocation>()) {
358 if (!object->HasEscaped()) {
359 if (v8_flags.trace_maglev_escape_analysis) {
360 std::cout << "* Removing store node "
361 << PrintNodeLabel(labeller_, node) << " to allocation "
362 << PrintNodeLabel(labeller_, object) << std::endl;
363 }
364 return ProcessResult::kRemove;
365 }
366 }
367 }
368 return ProcessResult::kContinue;
369 }
370
371 private:
372 MaglevGraphLabeller* labeller_ = nullptr;
373};
