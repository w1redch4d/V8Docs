40template <YoungGenerationMarkingVisitationMode marking_mode>
41YoungGenerationMarkingVisitor<marking_mode>::~YoungGenerationMarkingVisitor() {
42 PublishWorklists();
43
44 // Flush memory chunk live bytes. Atomics are used for incrementing the live
45 // bytes counter of the page, so there is no need to defer flushing to the
46 // main thread.
47 for (auto& pair : live_bytes_data_) {
48 if (pair.first) {
49 pair.first->IncrementLiveBytesAtomically(pair.second);
50 }
51 }

54template <YoungGenerationMarkingVisitationMode marking_mode>
55void YoungGenerationMarkingVisitor<marking_mode>::VisitCppHeapPointer(
56 Tagged<HeapObject> host, CppHeapPointerSlot slot) {
57 if (!marking_worklists_local_.cpp_marking_state()) return;
58
59 // The table is not reclaimed in the young generation, so we only need to mark
60 // through to the C++ pointer.
61
62 if (auto cpp_heap_pointer = slot.try_load(isolate_, kAnyCppHeapPointer)) {
63 marking_worklists_local_.cpp_marking_state()->MarkAndPush(
64 reinterpret_cast<void*>(cpp_heap_pointer));
65 }
66}

88template <YoungGenerationMarkingVisitationMode marking_mode>
89size_t YoungGenerationMarkingVisitor<marking_mode>::VisitEphemeronHashTable(
90 Tagged<Map> map, Tagged<EphemeronHashTable> table, MaybeObjectSize) {
91 // Register table with Minor MC, so it can take care of the weak keys later.
92 // This allows to only iterate the tables' values, which are treated as strong
93 // independently of whether the key is live.
94 ephemeron_table_list_local_.Push(table);
95 for (InternalIndex i : table->IterateEntries()) {
96 ObjectSlot value_slot =
97 table->RawFieldOfElementAt(EphemeronHashTable::EntryToValueIndex(i));
98 VisitPointer(table, value_slot);
99 }
100 return EphemeronHashTable::BodyDescriptor::SizeOf(map, table);
101}

104template <YoungGenerationMarkingVisitationMode marking_mode>
105void YoungGenerationMarkingVisitor<marking_mode>::VisitExternalPointer(
106 Tagged<HeapObject> host, ExternalPointerSlot slot) {
107 // With sticky mark-bits the host object was already marked (old).
108 DCHECK_IMPLIES(!v8_flags.sticky_mark_bits,
109 HeapLayout::InYoungGeneration(host));
110 DCHECK(!slot.tag_range().IsEmpty());
111 DCHECK(!IsSharedExternalPointerType(slot.tag_range()));
112
113 // TODO(chromium:337580006): Remove when pointer compression always uses
114 // EPT.
115 if (!slot.HasExternalPointerHandle()) return;
116
117 ExternalPointerHandle handle = slot.Relaxed_LoadHandle();
118 if (handle != kNullExternalPointerHandle) {
119 ExternalPointerTable& table = isolate_->external_pointer_table();
120 auto* space = isolate_->heap()->young_external_pointer_space();
121 table.Mark(space, handle, slot.address());
122 }
123
124 // Add to the remset whether the handle is null or not, as the slot could be
125 // set to a non-null value before the marking pause.
126 // TODO(342905179): Avoid adding null handle locations to the remset, and
127 // instead make external pointer writes invoke a marking barrier.
128 auto slot_chunk = MutablePageMetadata::FromHeapObject(host);
129 RememberedSet<SURVIVOR_TO_EXTERNAL_POINTER>::template Insert<
130 AccessMode::ATOMIC>(slot_chunk, slot_chunk->Offset(slot.address()));
131}
132#endif // V8_COMPRESS_POINTERS

135template <typename TSlot>
136void YoungGenerationMarkingVisitor<marking_mode>::VisitPointersImpl(
137 Tagged<HeapObject> host, TSlot start, TSlot end) {
138 for (TSlot slot = start; slot < end; ++slot) {
139 if constexpr (marking_mode ==

141 VisitObjectViaSlot<ObjectVisitationMode::kPushToWorklist,
142 SlotTreatmentMode::kReadOnly>(slot);
143 } else {
144 VisitObjectViaSlot<ObjectVisitationMode::kPushToWorklist,
145 SlotTreatmentMode::kReadWrite>(slot);
146 }
147 }

165template <YoungGenerationMarkingVisitationMode marking_mode>
166template <typename YoungGenerationMarkingVisitor<
167 marking_mode>::ObjectVisitationMode visitation_mode,
168 typename YoungGenerationMarkingVisitor<
169 marking_mode>::SlotTreatmentMode slot_treatment_mode,
170 typename TSlot>
171V8_INLINE bool YoungGenerationMarkingVisitor<marking_mode>::VisitObjectViaSlot(
172 TSlot slot) {
173 const std::optional<Tagged<Object>> optional_object =
174 this->GetObjectFilterReadOnlyAndSmiFast(slot);
175 if (!optional_object) {
176 return false;
177 }
178 typename TSlot::TObject target = *optional_object;
179#ifdef V8_ENABLE_DIRECT_HANDLE
180 if (target.ptr() == kTaggedNullAddress) return false;
181#endif
182 Tagged<HeapObject> heap_object;
183 // Treat weak references as strong.
184 if (!target.GetHeapObject(&heap_object)) {
185 return false;
186 }
187
188#ifdef THREAD_SANITIZER
189 MemoryChunk::FromHeapObject(heap_object)->SynchronizedLoad();
190#endif // THREAD_SANITIZER
191
192 if (!HeapLayout::InYoungGeneration(heap_object)) {
193 return false;
194 }
195
196#ifdef V8_MINORMS_STRING_SHORTCUTTING
197 if (slot_treatment_mode == SlotTreatmentMode::kReadWrite &&
198 !ShortCutStrings(reinterpret_cast<HeapObjectSlot&>(slot), &heap_object)) {
199 return false;
200 }
201#endif // V8_MINORMS_STRING_SHORTCUTTING
202
203 if (!TryMark(heap_object)) return true;
204
205 // Maps won't change in the atomic pause, so the map can be read without
206 // atomics.
207 if constexpr (visitation_mode == ObjectVisitationMode::kVisitDirectly) {
208 Tagged<Map> map = heap_object->map(isolate_);
209 const size_t visited_size = Base::Visit(map, heap_object);
210 if (visited_size) {
211 IncrementLiveBytesCached(
212 MutablePageMetadata::cast(
213 MemoryChunkMetadata::FromHeapObject(heap_object)),
214 ALIGN_TO_ALLOCATION_ALIGNMENT(visited_size));
215 }
216 return true;
217 }
218 // Default case: Visit via worklist.
219 marking_worklists_local_.Push(heap_object);
220
221 return true;
222}

225template <YoungGenerationMarkingVisitationMode marking_mode>
226V8_INLINE bool YoungGenerationMarkingVisitor<marking_mode>::ShortCutStrings(
227 HeapObjectSlot slot, Tagged<HeapObject>* heap_object) {
228 DCHECK_EQ(YoungGenerationMarkingVisitationMode::kParallel, marking_mode);
229 if (shortcut_strings_) {
230 DCHECK(V8_STATIC_ROOTS_BOOL);
231#if V8_STATIC_ROOTS_BOOL
232 ObjectSlot map_slot = (*heap_object)->map_slot();
233 Address map_address = map_slot.load_map().ptr();
234 if (map_address == StaticReadOnlyRoot::kThinOneByteStringMap ||
235 map_address == StaticReadOnlyRoot::kThinTwoByteStringMap) {
236 DCHECK_EQ((*heap_object)
237 ->map(ObjectVisitorWithCageBases::cage_base())
238 ->visitor_id(),
239 VisitorId::kVisitThinString);
240 *heap_object = Cast<ThinString>(*heap_object)->actual();
241 // ThinStrings always refer to internalized strings, which are always
242 // in old space.
243 DCHECK(!Heap::InYoungGeneration(*heap_object));
244 slot.StoreHeapObject(*heap_object);
245 return false;
246 } else if (map_address == StaticReadOnlyRoot::kConsOneByteStringMap ||
247 map_address == StaticReadOnlyRoot::kConsTwoByteStringMap) {
248 // Not all ConsString are short cut candidates.
249 const VisitorId visitor_id =
250 (*heap_object)
251 ->map(ObjectVisitorWithCageBases::cage_base())
252 ->visitor_id();
253 if (visitor_id == VisitorId::kVisitShortcutCandidate) {
254 Tagged<ConsString> string = Cast<ConsString>(*heap_object);
255 if (static_cast<Tagged_t>(string->second().ptr()) ==
256 StaticReadOnlyRoot::kempty_string) {
257 *heap_object = string->first();
258 slot.StoreHeapObject(*heap_object);
259 if (!Heap::InYoungGeneration(*heap_object)) {
260 return false;
261 }
262 }
263 }
264 }
265#endif // V8_STATIC_ROOTS_BOOL
266 }
267 return true;
268}
269#endif // V8_MINORMS_STRING_SHORTCUTTING

271template <YoungGenerationMarkingVisitationMode marking_mode>
272V8_INLINE void
273YoungGenerationMarkingVisitor<marking_mode>::IncrementLiveBytesCached(
274 MutablePageMetadata* chunk, intptr_t by) {
275 DCHECK_IMPLIES(V8_COMPRESS_POINTERS_8GB_BOOL,
276 IsAligned(by, kObjectAlignment8GbHeap));
277 const size_t hash = base::hash<MutablePageMetadata*>()(chunk) & kEntriesMask;
278 auto& entry = live_bytes_data_[hash];
279 if (entry.first && entry.first != chunk) {
280 entry.first->IncrementLiveBytesAtomically(entry.second);
281 entry.first = chunk;
282 entry.second = 0;
283 } else {
284 entry.first = chunk;
285 }
286 entry.second += by;
287}
