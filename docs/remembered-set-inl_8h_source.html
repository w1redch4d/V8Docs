16template <typename Callback>
17SlotCallbackResult UpdateTypedSlotHelper::UpdateTypedSlot(
18 WritableJitAllocation& jit_allocation, Heap* heap, SlotType slot_type,
19 Address addr, Callback callback) {
20 switch (slot_type) {
21 case SlotType::kCodeEntry: {
22 WritableRelocInfo rinfo(jit_allocation, addr, RelocInfo::CODE_TARGET);
23 return UpdateCodeTarget(&rinfo, callback);
24 }
25 case SlotType::kConstPoolCodeEntry: {
26 return UpdateCodeEntry(addr, callback);
27 }
28 case SlotType::kEmbeddedObjectCompressed: {
29 WritableRelocInfo rinfo(jit_allocation, addr,
30 RelocInfo::COMPRESSED_EMBEDDED_OBJECT);
31 return UpdateEmbeddedPointer(heap, &rinfo, callback);
32 }
33 case SlotType::kEmbeddedObjectFull: {
34 WritableRelocInfo rinfo(jit_allocation, addr,
35 RelocInfo::FULL_EMBEDDED_OBJECT);
36 return UpdateEmbeddedPointer(heap, &rinfo, callback);
37 }
38 case SlotType::kConstPoolEmbeddedObjectCompressed: {
39 Tagged<HeapObject> old_target = Cast<HeapObject>(
40 Tagged<Object>(V8HeapCompressionScheme::DecompressTagged(
41 heap->isolate(), base::Memory<Tagged_t>(addr))));
42 Tagged<HeapObject> new_target = old_target;
43 SlotCallbackResult result = callback(FullMaybeObjectSlot(&new_target));
44 DCHECK(!HasWeakHeapObjectTag(new_target));
45 if (new_target != old_target) {
46 jit_allocation.WriteValue<Tagged_t>(
47 addr, V8HeapCompressionScheme::CompressObject(new_target.ptr()));
48 }
49 return result;
50 }
51 case SlotType::kConstPoolEmbeddedObjectFull: {
52 Tagged<HeapObject> old_target =
53 Cast<HeapObject>(Tagged<Object>(base::Memory<Address>(addr)));
54 Tagged<HeapObject> new_target = old_target;
55 SlotCallbackResult result = callback(FullMaybeObjectSlot(&new_target));
56 if (new_target != old_target) {
57 jit_allocation.WriteValue(addr, new_target.ptr());
58 }
59 return result;
60 }
61 case SlotType::kCleared:
62 break;
63 }
64 UNREACHABLE();
65}

67Tagged<HeapObject> UpdateTypedSlotHelper::GetTargetObject(Heap* heap,
68 SlotType slot_type,
69 Address addr) {
70 switch (slot_type) {
71 case SlotType::kCodeEntry: {
72 RelocInfo rinfo(addr, RelocInfo::CODE_TARGET);
73 return InstructionStream::FromTargetAddress(rinfo.target_address());
74 }
75 case SlotType::kConstPoolCodeEntry: {
76 return InstructionStream::FromEntryAddress(addr);
77 }
78 case SlotType::kEmbeddedObjectCompressed: {
79 RelocInfo rinfo(addr, RelocInfo::COMPRESSED_EMBEDDED_OBJECT);
80 return rinfo.target_object(heap->isolate());
81 }
82 case SlotType::kEmbeddedObjectFull: {
83 RelocInfo rinfo(addr, RelocInfo::FULL_EMBEDDED_OBJECT);
84 return rinfo.target_object(heap->isolate());
85 }
86 case SlotType::kConstPoolEmbeddedObjectCompressed: {
87 Address full = V8HeapCompressionScheme::DecompressTagged(
88 heap->isolate(), base::Memory<Tagged_t>(addr));
89 return Cast<HeapObject>(Tagged<Object>(full));
90 }
91 case SlotType::kConstPoolEmbeddedObjectFull: {
92 FullHeapObjectSlot slot(addr);
93 return (*slot).GetHeapObjectAssumeStrong(heap->isolate());
94 }
95 case SlotType::kCleared:
96 break;
97 }
98 UNREACHABLE();
99}
