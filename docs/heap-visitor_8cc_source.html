15// We don't record weak slots during marking or scavenges. Instead we do it
16// once when we complete mark-compact cycle. Note that write barrier has no
17// effect if we are already in the middle of compacting mark-sweep cycle and we
18// have to record slots manually.
19static bool MustRecordSlots(Heap* heap) {
20 return heap->gc_state() == Heap::MARK_COMPACT &&
21 heap->mark_compact_collector()->is_compacting();
22}

27template <class T>
28Tagged<Object> VisitWeakList(Heap* heap, Tagged<Object> list,
29 WeakObjectRetainer* retainer) {
30 Tagged<HeapObject> undefined = ReadOnlyRoots(heap).undefined_value();
31 Tagged<Object> head = undefined;
32 Tagged<T> tail;
33 bool record_slots = MustRecordSlots(heap);
34
35 while (list != undefined) {
36 // Check whether to keep the candidate in the list.
37 Tagged<T> candidate = Cast<T>(list);
38
39 Tagged<Object> retained = retainer->RetainAs(list);
40
41 // Move to the next element before the WeakNext is cleared.
42 list = WeakListVisitor<T>::WeakNext(candidate);
43
44 if (retained != Tagged<Object>()) {
45 if (head == undefined) {
46 // First element in the list.
47 head = retained;
48 } else {
49 // Subsequent elements in the list.
50 DCHECK(!tail.is_null());
51 WeakListVisitor<T>::SetWeakNext(tail, Cast<HeapObject>(retained));
52 if (record_slots) {
53 Tagged<HeapObject> slot_holder =
54 WeakListVisitor<T>::WeakNextHolder(tail);
55 int slot_offset = WeakListVisitor<T>::WeakNextOffset();
56 ObjectSlot slot = slot_holder->RawField(slot_offset);
57 MarkCompactCollector::RecordSlot(slot_holder, slot,
58 Cast<HeapObject>(retained));
59 }
60 }
61 // Retained object is new tail.
62 DCHECK(!IsUndefined(retained, heap->isolate()));
63 candidate = Cast<T>(retained);
64 tail = candidate;
65
66 // tail is a live object, visit it.
67 WeakListVisitor<T>::VisitLiveObject(heap, tail, retainer);
68
69 } else {
70 WeakListVisitor<T>::VisitPhantomObject(heap, candidate);
71 }
72 }
73
74 // Terminate the list if there is one or more elements.
75 if (!tail.is_null()) WeakListVisitor<T>::SetWeakNext(tail, undefined);
76 return head;
77}

79template <class T>
80static void ClearWeakList(Heap* heap, Tagged<Object> list) {
81 Tagged<Object> undefined = ReadOnlyRoots(heap).undefined_value();
82 while (list != undefined) {
83 Tagged<T> candidate = Cast<T>(list);
84 list = WeakListVisitor<T>::WeakNext(candidate);
85 WeakListVisitor<T>::SetWeakNext(candidate, undefined);
86 }
87}

107 static void VisitLiveObject(Heap* heap, Tagged<Context> context,
108 WeakObjectRetainer* retainer) {
109 if (heap->gc_state() == Heap::MARK_COMPACT) {
110 // Record the slots of the weak entries in the native context.
111 for (int idx = Context::FIRST_WEAK_SLOT;
112 idx < Context::NATIVE_CONTEXT_SLOTS; ++idx) {
113 ObjectSlot slot = context->RawField(Context::OffsetOfElementAt(idx));
114 MarkCompactCollector::RecordSlot(context, slot,
115 Cast<HeapObject>(*slot));
116 }
117 }
118 }

120 template <class T>
121 static void DoWeakList(Heap* heap, Tagged<Context> context,
122 WeakObjectRetainer* retainer, int index) {
123 // Visit the weak list, removing dead intermediate elements.
124 Tagged<Object> list_head =
125 VisitWeakList<T>(heap, context->get(index), retainer);
126
127 // Update the list head.
128 context->set(index, list_head, UPDATE_WRITE_BARRIER);
129
130 if (MustRecordSlots(heap)) {
131 // Record the updated slot if necessary.
132 ObjectSlot head_slot = context->RawField(FixedArray::SizeFor(index));
133 heap->mark_compact_collector()->RecordSlot(context, head_slot,
134 Cast<HeapObject>(list_head));
135 }
136 }
137
138 static void VisitPhantomObject(Heap* heap, Tagged<Context> context) {}
139};
