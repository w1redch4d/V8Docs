168void WriteBarrier::MarkingSlow(Tagged<HeapObject> host,
169 JSDispatchHandle handle) {
170#ifdef V8_ENABLE_LEAPTIERING
171 MarkingBarrier* marking_barrier = CurrentMarkingBarrier(host);
172
173 // The JSDispatchTable is only marked during major GC so we can skip the
174 // barrier if we're only doing a minor GC.
175 // This is mostly an optimization, but it does help avoid scenarios where a
176 // minor GC marking barrier marks a table entry as alive but not the Code
177 // object contained in it (because it's not a young-gen object).
178 if (marking_barrier->is_minor()) return;
179
180 // Mark both the table entry and its content.
181 JSDispatchTable* jdt = IsolateGroup::current()->js_dispatch_table();
182 static_assert(JSDispatchTable::kWriteBarrierSetsEntryMarkBit);
183 jdt->Mark(handle);
184 marking_barrier->MarkValue(host, jdt->GetCode(handle));
185
186 // We don't need to record a slot here because the entries in the
187 // JSDispatchTable are not compacted and because the pointers stored in the
188 // table entries are updated after compacting GC.
189 static_assert(!JSDispatchTable::kSupportsCompaction);
190#else
191 UNREACHABLE();
192#endif
193}

286// static
287bool WriteBarrier::PageFlagsAreConsistent(Tagged<HeapObject> object) {
288 MemoryChunkMetadata* metadata = MemoryChunkMetadata::FromHeapObject(object);
289 MemoryChunk* chunk = MemoryChunk::FromHeapObject(object);
290
291 // Slim chunk flags consistency.
292 CHECK_EQ(chunk->IsFlagSet(MemoryChunk::INCREMENTAL_MARKING),
293 chunk->IsMarking());
294
295 if (!v8_flags.sticky_mark_bits) {
296 AllocationSpace identity = metadata->owner()->identity();
297
298 // Generation consistency.
299 CHECK_EQ(identity == NEW_SPACE || identity == NEW_LO_SPACE,
300 chunk->InYoungGeneration());
301 }
302
303 // Marking consistency.
304 if (metadata->IsWritable()) {
305 // RO_SPACE can be shared between heaps, so we can't use RO_SPACE objects to
306 // find a heap. The exception is when the ReadOnlySpace is writeable, during
307 // bootstrapping, so explicitly allow this case.
308 Heap* heap = Heap::FromWritableHeapObject(object);
309 if (chunk->InWritableSharedSpace()) {
310 // The marking bit is not set for chunks in shared spaces during MinorMS
311 // concurrent marking.
312 CHECK_EQ(chunk->IsMarking(),
313 heap->incremental_marking()->IsMajorMarking());
314 } else {
315 CHECK_EQ(chunk->IsMarking(), heap->incremental_marking()->IsMarking());
316 }
317 } else {
318 // Non-writable RO_SPACE must never have marking flag set.
319 CHECK(!chunk->IsMarking());
320 }
321 return true;
322}

392enum RangeWriteBarrierMode {
393 kDoGenerationalOrShared = 1 << 0,
394 kDoMarking = 1 << 1,
395 kDoEvacuationSlotRecording = 1 << 2,
396};
397
398template <int kModeMask, typename TSlot>
399void ForRangeImpl(Heap* heap, MemoryChunk* source_chunk,
400 Tagged<HeapObject> object, TSlot start_slot, TSlot end_slot) {
401 // At least one of generational or marking write barrier should be requested.
402 static_assert(kModeMask & (kDoGenerationalOrShared | kDoMarking));
403 // kDoEvacuationSlotRecording implies kDoMarking.
404 static_assert(!(kModeMask & kDoEvacuationSlotRecording) ||
405 (kModeMask & kDoMarking));
406
407 MarkingBarrier* marking_barrier = nullptr;
408 static constexpr Tagged_t kPageMask =
409 ~static_cast<Tagged_t>(PageMetadata::kPageSize - 1);
410 Tagged_t cached_uninteresting_page =
411 static_cast<Tagged_t>(heap->read_only_space()->FirstPageAddress()) &
412 kPageMask;
413
414 if (kModeMask & kDoMarking) {
415 marking_barrier = WriteBarrier::CurrentMarkingBarrier(object);
416 }
417
418 MarkCompactCollector* collector = heap->mark_compact_collector();
419 MutablePageMetadata* source_page_metadata =
420 MutablePageMetadata::cast(source_chunk->Metadata());
421
422 for (TSlot slot = start_slot; slot < end_slot; ++slot) {
423 // If we *only* need the generational or shared WB, we can skip objects
424 // residing on uninteresting pages.
425 Tagged_t compressed_page;
426 if (kModeMask == kDoGenerationalOrShared) {
427 Tagged_t tagged_value = *slot.location();
428 if (HAS_SMI_TAG(tagged_value)) continue;
429 compressed_page = tagged_value & kPageMask;
430 if (compressed_page == cached_uninteresting_page) {
431#if DEBUG
432 typename TSlot::TObject value = *slot;
433 Tagged<HeapObject> value_heap_object;
434 if (value.GetHeapObject(&value_heap_object)) {
435 CHECK(!HeapLayout::InYoungGeneration(value_heap_object));
436 CHECK(!HeapLayout::InWritableSharedSpace(value_heap_object));
437 }
438#endif // DEBUG
439 continue;
440 }
441 // Fall through to decompressing the pointer and fetching its actual
442 // page header flags.
443 }
444 typename TSlot::TObject value = *slot;
445 Tagged<HeapObject> value_heap_object;
446 if (!value.GetHeapObject(&value_heap_object)) continue;
447
448 if (kModeMask & kDoGenerationalOrShared) {
449 if (HeapLayout::InYoungGeneration(value_heap_object)) {
450 RememberedSet<OLD_TO_NEW>::Insert<AccessMode::NON_ATOMIC>(
451 source_page_metadata, source_chunk->Offset(slot.address()));
452 } else if (HeapLayout::InWritableSharedSpace(value_heap_object)) {
453 RememberedSet<OLD_TO_SHARED>::Insert<AccessMode::ATOMIC>(
454 source_page_metadata, source_chunk->Offset(slot.address()));
455 } else if (kModeMask == kDoGenerationalOrShared) {
456 cached_uninteresting_page = compressed_page;
457 }
458 }
459
460 if (kModeMask & kDoMarking) {
461 marking_barrier->MarkValue(object, value_heap_object);
462 if (kModeMask & kDoEvacuationSlotRecording) {
463 collector->RecordSlot(source_chunk, HeapObjectSlot(slot),
464 value_heap_object);
465 }
466 }
467 }
468}

472// Instantiate `WriteBarrier::WriteBarrierForRange()` for `ObjectSlot` and
473// `MaybeObjectSlot`.
474template void WriteBarrier::ForRange<ObjectSlot>(Heap* heap,
475 Tagged<HeapObject> object,
476 ObjectSlot start_slot,
477 ObjectSlot end_slot);
478template void WriteBarrier::ForRange<MaybeObjectSlot>(
479 Heap* heap, Tagged<HeapObject> object, MaybeObjectSlot start_slot,
480 MaybeObjectSlot end_slot);
481
482template <typename TSlot>
483// static
484void WriteBarrier::ForRange(Heap* heap, Tagged<HeapObject> object,
485 TSlot start_slot, TSlot end_slot) {
486 if (v8_flags.disable_write_barriers) return;
487 MemoryChunk* source_chunk = MemoryChunk::FromHeapObject(object);
488 base::Flags<RangeWriteBarrierMode> mode;
489
490 if (!HeapLayout::InYoungGeneration(object) &&
491 !source_chunk->InWritableSharedSpace()) {
492 mode |= kDoGenerationalOrShared;
493 }
494
495 if (heap->incremental_marking()->IsMarking()) {
496 mode |= kDoMarking;
497 if (!source_chunk->ShouldSkipEvacuationSlotRecording()) {
498 mode |= kDoEvacuationSlotRecording;
499 }
500 }
501
502 switch (mode) {
503 // Nothing to be done.
504 case 0:
505 return;
506 // Generational only.
507 case kDoGenerationalOrShared:
508 return ForRangeImpl<kDoGenerationalOrShared>(heap, source_chunk, object,
509 start_slot, end_slot);
510 // Marking, no evacuation slot recording.
511 case kDoMarking:
512 return ForRangeImpl<kDoMarking>(heap, source_chunk, object, start_slot,
513 end_slot);
514 // Marking with evacuation slot recording.
515 case kDoMarking | kDoEvacuationSlotRecording:
516 return ForRangeImpl<kDoMarking | kDoEvacuationSlotRecording>(
517 heap, source_chunk, object, start_slot, end_slot);
518 // Generational and marking, no evacuation slot recording.
519 case kDoGenerationalOrShared | kDoMarking:
520 return ForRangeImpl<kDoGenerationalOrShared | kDoMarking>(
521 heap, source_chunk, object, start_slot, end_slot);
522 // Generational and marking with evacuation slot recording.
523 case kDoGenerationalOrShared | kDoMarking | kDoEvacuationSlotRecording:
524 return ForRangeImpl<kDoGenerationalOrShared | kDoMarking |
525 kDoEvacuationSlotRecording>(
526 heap, source_chunk, object, start_slot, end_slot);
527 default:
528 UNREACHABLE();
529 }

534// static
535bool WriteBarrier::VerifyDispatchHandleMarkingState(Tagged<HeapObject> host,
536 JSDispatchHandle handle,
537 WriteBarrierMode mode) {
538#ifdef V8_ENABLE_LEAPTIERING
539 JSDispatchTable* jdt = IsolateGroup::current()->js_dispatch_table();
540 if (mode == SKIP_WRITE_BARRIER &&
541 WriteBarrier::IsRequired(host, jdt->GetCode(handle))) {
542 return false;
543 }
544
545 if (CurrentMarkingBarrier(host)->is_not_major()) return true;
546
547 // Ensure we don't have a black -> white -> black edge. This could happen when
548 // skipping a write barrier while concurrently the dispatch entry is marked
549 // from another JSFunction.
550 if (ReadOnlyHeap::Contains(host) ||
551 (IsMarking(host) && mode != SKIP_WRITE_BARRIER) ||
552 !CurrentMarkingBarrier(host)->IsMarked(host)) {
553 return true;
554 }
555 if (jdt->IsMarked(handle)) {
556 return true;
557 }
558 Tagged<Code> value = jdt->GetCode(handle);
559 if (ReadOnlyHeap::Contains(value)) {
560 return true;
561 }
562 return !CurrentMarkingBarrier(host)->IsMarked(value);
563#else
564 return true;
565#endif // V8_ENABLE_LEAPTIERING
566}
