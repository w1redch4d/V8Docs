40// static
41Tagged<InstructionStream> InstructionStream::Initialize(
42 Tagged<HeapObject> self, Tagged<Map> map, uint32_t body_size,
43 int constant_pool_offset, Tagged<TrustedByteArray> reloc_info) {
44 {
45 WritableJitAllocation writable_allocation =
46 ThreadIsolation::RegisterInstructionStreamAllocation(
47 self.address(), InstructionStream::SizeFor(body_size));
48 CHECK_EQ(InstructionStream::SizeFor(body_size), writable_allocation.size());
49
50 writable_allocation.WriteHeaderSlot<Map, kMapOffset>(map, kRelaxedStore);
51
52 writable_allocation.WriteHeaderSlot<uint32_t, kBodySizeOffset>(body_size);
53
54 if constexpr (V8_EMBEDDED_CONSTANT_POOL_BOOL) {
55 writable_allocation.WriteHeaderSlot<int, kConstantPoolOffsetOffset>(
56 kHeaderSize + constant_pool_offset);
57 }
58
59 // During the Code initialization process, InstructionStream::code is
60 // briefly unset (the Code object has not been allocated yet). In this state
61 // it is only visible through heap iteration.
62 writable_allocation.WriteHeaderSlot<Smi, kCodeOffset>(Smi::zero(),
63 kReleaseStore);
64
65 DCHECK(!HeapLayout::InYoungGeneration(reloc_info));
66 writable_allocation.WriteProtectedPointerHeaderSlot<TrustedByteArray,
67 kRelocationInfoOffset>(
68 reloc_info, kRelaxedStore);
69
70 // Clear header padding
71 writable_allocation.ClearBytes(kUnalignedSize,
72 kHeaderSize - kUnalignedSize);
73 // Clear trailing padding.
74 writable_allocation.ClearBytes(kHeaderSize + body_size,
75 TrailingPaddingSizeFor(body_size));
76 }
77
78 Tagged<InstructionStream> istream = Cast<InstructionStream>(self);
79
80 // We want to keep the code minimal that runs with write access to a JIT
81 // allocation, so trigger the write barriers after the WritableJitAllocation
82 // went out of scope.
83 SLOW_DCHECK(!WriteBarrier::IsRequired(istream, map));
84 CONDITIONAL_PROTECTED_POINTER_WRITE_BARRIER(*istream, kRelocationInfoOffset,
85 reloc_info, UPDATE_WRITE_BARRIER);
86
87 return istream;
88}

90// Copy from compilation artifacts stored in CodeDesc to the target on-heap
91// objects.
92//
93// Note this is quite convoluted for historical reasons. The CodeDesc buffer
94// contains instructions, a part of inline metadata, and the relocation info.
95// Additionally, the unwinding_info is stored in a separate buffer
96// `desc.unwinding_info`. In this method, we copy all these parts into the
97// final on-heap representation.
98//
99// The off-heap representation:
100//
101// CodeDesc.buffer:
102//
103// +-------------------
104// | instructions
105// +-------------------
106// | inline metadata
107// | .. safepoint table
108// | .. handler table
109// | .. constant pool
110// | .. code comments
111// +-------------------
112// | reloc info
113// +-------------------
114//
115// CodeDesc.unwinding_info: .. the unwinding info.
116//
117// This is transformed into the on-heap representation, where
118// InstructionStream contains all instructions and inline metadata, and a
119// pointer to the relocation info byte array.
120void InstructionStream::Finalize(Tagged<Code> code,
121 Tagged<TrustedByteArray> reloc_info,
122 CodeDesc desc, Heap* heap) {
123 DisallowGarbageCollection no_gc;
124 std::optional<WriteBarrierPromise> promise;
125
126 // Copy the relocation info first before we unlock the Jit allocation.
127 // TODO(sroettger): reloc info should live in protected memory.
128 DCHECK_EQ(reloc_info->length(), desc.reloc_size);
129 CopyBytes(reloc_info->begin(), desc.buffer + desc.reloc_offset,
130 static_cast<size_t>(desc.reloc_size));
131
132 {
133 WritableJitAllocation writable_allocation =
134 ThreadIsolation::LookupJitAllocation(
135 address(), InstructionStream::SizeFor(body_size()),
136 ThreadIsolation::JitAllocationType::kInstructionStream, true);
137
138 // Copy code and inline metadata.
139 static_assert(InstructionStream::kOnHeapBodyIsContiguous);
140 writable_allocation.CopyCode(kHeaderSize, desc.buffer,
141 static_cast<size_t>(desc.instr_size));
142 writable_allocation.CopyData(kHeaderSize + desc.instr_size,
143 desc.unwinding_info,
144 static_cast<size_t>(desc.unwinding_info_size));
145 DCHECK_EQ(desc.body_size(), desc.instr_size + desc.unwinding_info_size);
146 DCHECK_EQ(code->body_size(),
147 code->instruction_size() + code->metadata_size());
148
149 promise.emplace(RelocateFromDesc(writable_allocation, heap, desc,
150 code->constant_pool(), no_gc));
151
152 // Publish the code pointer after the istream has been fully initialized.
153 writable_allocation.WriteProtectedPointerHeaderSlot<Code, kCodeOffset>(
154 code, kReleaseStore);
155 }
156
157 // Trigger the write barriers after we dropped the JIT write permissions.
158 RelocateFromDescWriteBarriers(heap, desc, code->constant_pool(), *promise,
159 no_gc);
160 CONDITIONAL_PROTECTED_POINTER_WRITE_BARRIER(*this, kCodeOffset, code,
161 UPDATE_WRITE_BARRIER);
162
163 code->FlushICache();
164}

175Tagged<Object> InstructionStream::raw_code(AcquireLoadTag tag) const {
176 Tagged<Object> value = RawProtectedPointerField(kCodeOffset).Acquire_Load();
177 DCHECK(!HeapLayout::InYoungGeneration(value));
178 DCHECK(IsSmi(value) || HeapLayout::InTrustedSpace(Cast<HeapObject>(value)));
179 return value;
180}

186bool InstructionStream::TryGetCode(Tagged<Code>* code_out,

189 if (maybe_code == Smi::zero()) return false;
190 *code_out = Cast<Code>(maybe_code);

231// static
232Tagged<InstructionStream> InstructionStream::FromTargetAddress(
233 Address address) {
234 {
235 // TODO(jgruber,v8:6666): Support embedded builtins here. We'd need to pass
236 // in the current isolate.
237 Address start =
238 reinterpret_cast<Address>(Isolate::CurrentEmbeddedBlobCode());
239 Address end = start + Isolate::CurrentEmbeddedBlobCodeSize();
240 CHECK(address < start || address >= end);
241 }
242
243 Tagged<HeapObject> code =
244 HeapObject::FromAddress(address - InstructionStream::kHeaderSize);
245 // Unchecked cast because we can't rely on the map currently not being a
246 // forwarding pointer.
247 return UncheckedCast<InstructionStream>(code);
248}
