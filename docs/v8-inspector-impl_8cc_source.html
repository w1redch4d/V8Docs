147std::unique_ptr<V8InspectorSession> V8InspectorImpl::connect(
148 int contextGroupId, V8Inspector::Channel* channel, StringView state,
149 ClientTrustLevel client_trust_level, SessionPauseState pause_state) {
150 int sessionId = ++m_lastSessionId;
151 std::shared_ptr<V8DebuggerBarrier> debuggerBarrier;
152 if (pause_state == kWaitingForDebugger) {
153 auto it = m_debuggerBarriers.find(contextGroupId);
154 if (it != m_debuggerBarriers.end()) {
155 // Note this will be empty in case a pre-existent barrier is already
156 // released. This is by design, as a released throttle is no longer
157 // efficient.
158 debuggerBarrier = it->second.lock();
159 } else {
160 debuggerBarrier =
161 std::make_shared<V8DebuggerBarrier>(m_client, contextGroupId);
162 m_debuggerBarriers.insert(it, {contextGroupId, debuggerBarrier});
163 }
164 }
165 std::unique_ptr<V8InspectorSessionImpl> session =
166 V8InspectorSessionImpl::create(this, contextGroupId, sessionId, channel,
167 state, client_trust_level,
168 std::move(debuggerBarrier));
169 m_sessions[contextGroupId][sessionId] = session.get();
170 return std::move(session);
171}

212void V8InspectorImpl::contextCreated(const V8ContextInfo& info) {
213 int contextId = ++m_lastContextId;
214 auto* context = new InspectedContext(this, info, contextId);
215 m_contextIdToGroupIdMap[contextId] = info.contextGroupId;
216
217 DCHECK(m_uniqueIdToContextId.find(context->uniqueId().pair()) ==
218 m_uniqueIdToContextId.end());
219 m_uniqueIdToContextId.insert(
220 std::make_pair(context->uniqueId().pair(), contextId));
221
222 auto contextIt = m_contexts.find(info.contextGroupId);
223 if (contextIt == m_contexts.end())
224 contextIt = m_contexts
225 .insert(std::make_pair(
226 info.contextGroupId,
227 std::unique_ptr<ContextByIdMap>(new ContextByIdMap())))
228 .first;
229 const auto& contextById = contextIt->second;
230
231 DCHECK(contextById->find(contextId) == contextById->cend());
232 (*contextById)[contextId].reset(context);
233 forEachSession(
234 info.contextGroupId, [&context](V8InspectorSessionImpl* session) {
235 session->runtimeAgent()->addBindings(context);
236 session->runtimeAgent()->reportExecutionContextCreated(context);
237 });
238}

246void V8InspectorImpl::contextCollected(int groupId, int contextId) {
247 m_contextIdToGroupIdMap.erase(contextId);
248
249 auto storageIt = m_consoleStorageMap.find(groupId);
250 if (storageIt != m_consoleStorageMap.end())
251 storageIt->second->contextDestroyed(contextId);
252
253 InspectedContext* inspectedContext = getContext(groupId, contextId);
254 if (!inspectedContext) return;
255
256 forEachSession(groupId, [&inspectedContext](V8InspectorSessionImpl* session) {
257 session->runtimeAgent()->reportExecutionContextDestroyed(inspectedContext);
258 });
259 discardInspectedContext(groupId, contextId);
260}

262void V8InspectorImpl::resetContextGroup(int contextGroupId) {
263 m_consoleStorageMap.erase(contextGroupId);
264 m_muteExceptionsMap.erase(contextGroupId);
265 auto contextsIt = m_contexts.find(contextGroupId);
266 // Context might have been removed already by discardContextScript()
267 if (contextsIt != m_contexts.end()) {
268 for (const auto& map_entry : *contextsIt->second)
269 m_uniqueIdToContextId.erase(map_entry.second->uniqueId().pair());
270 m_contexts.erase(contextsIt);
271 }
272 forEachSession(contextGroupId,
273 [](V8InspectorSessionImpl* session) { session->reset(); });
274}

398void V8InspectorImpl::forEachContext(
399 int contextGroupId,
400 const std::function<void(InspectedContext*)>& callback) {
401 auto it = m_contexts.find(contextGroupId);
402 if (it == m_contexts.end()) return;
403 std::vector<int> ids;
404 ids.reserve(it->second->size());
405 for (auto& contextIt : *(it->second)) ids.push_back(contextIt.first);
406
407 // Retrieve by ids each time since |callback| may destroy some contexts.
408 for (auto& contextId : ids) {
409 it = m_contexts.find(contextGroupId);
410 if (it == m_contexts.end()) continue;
411 auto contextIt = it->second->find(contextId);
412 if (contextIt != it->second->end()) callback(contextIt->second.get());
413 }
414}

416void V8InspectorImpl::forEachSession(
417 int contextGroupId,
418 const std::function<void(V8InspectorSessionImpl*)>& callback) {
419 auto it = m_sessions.find(contextGroupId);
420 if (it == m_sessions.end()) return;
421 std::vector<int> ids;
422 ids.reserve(it->second.size());
423 for (auto& sessionIt : it->second) ids.push_back(sessionIt.first);
424
425 // Retrieve by ids each time since |callback| may destroy some contexts.
426 for (auto& sessionId : ids) {
427 it = m_sessions.find(contextGroupId);
428 if (it == m_sessions.end()) continue;
429 auto sessionIt = it->second.find(sessionId);
430 if (sessionIt != it->second.end()) callback(sessionIt->second);
431 }
432}

461class V8InspectorImpl::EvaluateScope::TerminateTask : public v8::Task {
462 public:
463 TerminateTask(v8::Isolate* isolate, std::shared_ptr<CancelToken> token)
464 : m_isolate(isolate), m_token(std::move(token)) {}
465
466 void Run() override {
467 // CancelToken contains m_canceled bool which may be changed from main

490bool V8InspectorImpl::associateExceptionData(v8::Local<v8::Context>,
491 v8::Local<v8::Value> exception,
492 v8::Local<v8::Name> key,
493 v8::Local<v8::Value> value) {
494 if (!exception->IsObject()) {
495 return false;
496 }
497 v8::Local<v8::Context> context;
498 if (!exceptionMetaDataContext().ToLocal(&context)) return false;
499 v8::TryCatch tryCatch(m_isolate);
500 v8::Context::Scope contextScope(context);
501 v8::HandleScope handles(m_isolate);
502 if (m_exceptionMetaData.IsEmpty())
503 m_exceptionMetaData.Reset(m_isolate,
504 v8::debug::EphemeronTable::New(m_isolate));
505
506 v8::Local<v8::debug::EphemeronTable> map = m_exceptionMetaData.Get(m_isolate);
507 v8::MaybeLocal<v8::Value> entry = map->Get(m_isolate, exception);
508 v8::Local<v8::Object> object;
509 if (entry.IsEmpty() || !entry.ToLocalChecked()->IsObject()) {
510 object =
511 v8::Object::New(m_isolate, v8::Null(m_isolate), nullptr, nullptr, 0);
512 m_exceptionMetaData.Reset(m_isolate,
513 map->Set(m_isolate, exception, object));
514 } else {
515 object = entry.ToLocalChecked().As<v8::Object>();
516 }
517 CHECK(object->IsObject());
518 v8::Maybe<bool> result = object->CreateDataProperty(context, key, value);
519 return result.FromMaybe(false);
520}

522v8::MaybeLocal<v8::Object> V8InspectorImpl::getAssociatedExceptionData(
523 v8::Local<v8::Value> exception) {
524 if (!exception->IsObject()) {
525 return v8::MaybeLocal<v8::Object>();
526 }
527 v8::EscapableHandleScope scope(m_isolate);
528 v8::Local<v8::Context> context;
529 if (m_exceptionMetaData.IsEmpty() ||
530 !exceptionMetaDataContext().ToLocal(&context)) {
531 return v8::MaybeLocal<v8::Object>();
532 }
533 v8::Local<v8::debug::EphemeronTable> map = m_exceptionMetaData.Get(m_isolate);
534 auto entry = map->Get(m_isolate, exception);
535 v8::Local<v8::Value> object;
536 if (!entry.ToLocal(&object) || !object->IsObject())
537 return v8::MaybeLocal<v8::Object>();
538 return scope.Escape(object.As<v8::Object>());
539}
