213class V8_NODISCARD SharedMutexGuardIfOffThread<LocalIsolate, kIsShared> final {
214 public:
215 SharedMutexGuardIfOffThread(base::SharedMutex* mutex, LocalIsolate* isolate) {
216 DCHECK_NOT_NULL(mutex);
217 DCHECK_NOT_NULL(isolate);
218 if (!isolate->is_main_thread()) mutex_guard_.emplace(mutex);
219 }
220
221 SharedMutexGuardIfOffThread(const SharedMutexGuardIfOffThread&) = delete;
222 SharedMutexGuardIfOffThread& operator=(const SharedMutexGuardIfOffThread&) =
223 delete;
224
225 private:
226 std::optional<base::SharedMutexGuard<kIsShared>> mutex_guard_;
227};
