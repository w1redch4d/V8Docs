38// SemiSpace in young generation
39//
40// A SemiSpace is a contiguous chunk of memory holding page-like memory chunks.
41// The mark-compact collector uses the memory of the first page in the from
42// space as a marking stack when tracing live objects.
43class SemiSpace final : public Space {
44 public:
45 using iterator = PageIterator;

49
50 SemiSpace(Heap* heap, SemiSpaceId semispace, size_t initial_capacity,
51 size_t maximum_capacity)
52 : Space(heap, NEW_SPACE, nullptr),
53 maximum_capacity_(RoundDown<PageMetadata::kPageSize>(maximum_capacity)),
54 minimum_capacity_(RoundDown<PageMetadata::kPageSize>(initial_capacity)),
55 target_capacity_(minimum_capacity_),
56 id_(semispace) {
57 DCHECK_GE(maximum_capacity, static_cast<size_t>(PageMetadata::kPageSize));
58 }
59 V8_EXPORT_PRIVATE ~SemiSpace();
60
61 inline bool Contains(Tagged<HeapObject> o) const;
62 inline bool Contains(Tagged<Object> o) const;
63 template <typename T>
64 inline bool Contains(Tagged<T> o) const;
65 inline bool ContainsSlow(Address a) const;
66
67 bool Commit();
68 void Uncommit();
69 bool IsCommitted() const { return !memory_chunk_list_.Empty(); }
70
71 // Grow the semispace to the new capacity. The new capacity requested must
72 // be larger than the current capacity and less than the maximum capacity.
73 bool GrowTo(size_t new_capacity);
74
75 // Shrinks the semispace to the new capacity. The new capacity requested
76 // must be more than the amount of used memory in the semispace and less
77 // than the current capacity.

80 bool EnsureCurrentCapacity();
81
82 // Returns the start address of the first page of the space.
83 Address space_start() const {
84 DCHECK_NE(memory_chunk_list_.front(), nullptr);

88 PageMetadata* current_page() { return current_page_; }
89
90 // Returns the start address of the current page of the space.
91 Address page_low() const { return current_page_->area_start(); }
92
93 // Returns one past the end address of the current page of the space.
94 Address page_high() const { return current_page_->area_end(); }
95
96 bool AdvancePage() {
97 PageMetadata* next_page = current_page_->next_page();
98 // We cannot expand if we reached the target capacity. Note
99 // that we need to account for the next page already for this check as we
100 // could potentially fill the whole page after advancing.
101 if (next_page == nullptr || ((current_capacity_ == target_capacity_) &&
102 !allow_to_grow_beyond_capacity_)) {
103 return false;
104 }
105 current_page_ = next_page;
106 base::AsAtomicWord::Relaxed_Store(
107 &current_capacity_, current_capacity_ + PageMetadata::kPageSize);
108 return true;
109 }
110
111 // Resets the space to using the first page.
112 void Reset();
113
114 void RemovePage(PageMetadata* page);
115 void MovePageToTheEnd(PageMetadata* page);

118

122
123 // Returns the current capacity of the semispace.
124 size_t current_capacity() const { return current_capacity_; }
125 // Returns the current capacity of the semispace using an atomic load.
126 size_t current_capacity_safe() const {

128 }
129
130 // Returns the target capacity of the semispace.

132
133 // Returns the maximum capacity of the semispace.

208 bool EnsureCapacity(size_t capacity);

211 // beyond that size.
212 const size_t maximum_capacity_ = 0;

214 const size_t minimum_capacity_ = 0;
